44  package sun.text.resources;
45  
46  import java.util.ListResourceBundle;
47  
48  public class CollationData_hu extends ListResourceBundle {
49  
50      protected final Object[][] getContents() {
51          return new Object[][] {
52              { "Rule",
53                  /* for hu, default sorting except for the following: */
54                  /* add cs "ligature" between c and d. */
55                  /* add d<stroke> between d and e. */
56                  /* add gy "ligature" between g and h. */
57                  /* add ly "ligature" between l and l<stroke>. */
58                  /* add l<stroke> between l and m. */
59                  /* add sz "ligature" between s and t. */
60                  /* add zs "ligature" between z and z<abovedot> */
61                  /* add z<abovedot> after z.       */
62                  "& C < cs , cS , Cs , CS " // cs ligatures
63                  + "& D < \u0111, \u0110 "    // tal : african d < d-stroke
64                  + "& G < gy, Gy, gY, GY "    // gy ligatures
65                  + "& L < ly, Ly, lY, LY "    // l < ly
66                  + "& O < o\u0308 , O\u0308 " // O < o-umlaut
67                  + "< o\u030b , O\u030b "     // o-double-accute
68                  + "& S < sz , sZ , Sz , SZ " // s < sz ligature
69                  + "& U < u\u0308 , U\u0308 " // u < u-umlaut
70                  + "< u\u030b , U\u030b "     // u-double-accute
71                  + "& Z < zs , zS , Zs , ZS " // stop-stroke < zs ligature
72              }
73          };
74      }
75  }
