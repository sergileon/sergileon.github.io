18  package backtype.storm.metric;
19  
20  import backtype.storm.Config;
21  import backtype.storm.metric.api.IMetricsConsumer;
22  import backtype.storm.task.IBolt;
23  import backtype.storm.task.IErrorReporter;
24  import backtype.storm.task.OutputCollector;
25  import backtype.storm.task.TopologyContext;
26  import backtype.storm.tuple.Tuple;
27  import java.util.Collection;
28  import java.util.Map;
29  
30  public class MetricsConsumerBolt implements IBolt {
31      IMetricsConsumer _metricsConsumer;
32      String _consumerClassName;
33      OutputCollector _collector;
34      Object _registrationArgument;
35  
36      public MetricsConsumerBolt(String consumerClassName, Object registrationArgument) {
37          _consumerClassName = consumerClassName;
38          _registrationArgument = registrationArgument;
39      }
40  
41      @Override
42      public void prepare(Map stormConf, TopologyContext context, OutputCollector collector) {
43          try {
44              _metricsConsumer = (IMetricsConsumer)Class.forName(_consumerClassName).newInstance();
45          } catch (Exception e) {
46              throw new RuntimeException("Could not instantiate a class listed in config under section " +
47                  Config.TOPOLOGY_METRICS_CONSUMER_REGISTER + " with fully qualified name " + _consumerClassName, e);
48          }
49          _metricsConsumer.prepare(stormConf, _registrationArgument, context, (IErrorReporter)collector);
50          _collector = collector;
51      }
52      
53      @Override
54      public void execute(Tuple input) {
55          _metricsConsumer.handleDataPoints((IMetricsConsumer.TaskInfo)input.getValue(0), (Collection)input.getValue(1));
56          _collector.ack(input);
57      }
58  
59      @Override
60      public void cleanup() {
61          _metricsConsumer.cleanup();
62      }
63      
64  }
