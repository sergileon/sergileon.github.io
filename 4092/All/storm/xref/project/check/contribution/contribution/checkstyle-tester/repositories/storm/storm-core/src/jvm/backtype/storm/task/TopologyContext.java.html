42  /**
43   * A TopologyContext is given to bolts and spouts in their "prepare" and "open"
44   * methods, respectively. This object provides information about the component's
45   * place within the topology, such as task ids, inputs and outputs, etc.

47   * <p>The TopologyContext is also used to declare ISubscribedState objects to
48   * synchronize state with StateSpouts this object is subscribed to.</p>
49   */

163     /**
164      * Gets the index of this task id in getComponentTasks(getThisComponentId()).
165      * An example use case for this method is determining which task
166      * accesses which resource in a distributed resource to ensure an even distribution.
167      */
168     public int getThisTaskIndex() {
169         List<Integer> tasks = new ArrayList<Integer>(getComponentTasks(getThisComponentId()));
170         Collections.sort(tasks);
171         for(int i=0; i<tasks.size(); i++) {
172             if(tasks.get(i) == getThisTaskId()) {
173                 return i;
174             }
175         }
176         throw new RuntimeException("Fatal: could not find this task id in this component");
177     }

232     /*
233      * Register a IMetric instance. 
234      * Storm will then call getValueAndReset on the metric every timeBucketSizeInSecs
235      * and the returned value is sent to all metrics consumers.
236      * You must call this during IBolt::prepare or ISpout::open.
237      * @return The IMetric argument unchanged.
238      */
239     public <T extends IMetric> T registerMetric(String name, T metric, int timeBucketSizeInSecs) {
240         if((Boolean)_openOrPrepareWasCalled.deref() == true) {
241             throw new RuntimeException("TopologyContext.registerMetric can only be called from within overridden " + 
242                                        "IBolt::prepare() or ISpout::open() method.");
243         }
244 
245         if (metric == null) {
246             throw new IllegalArgumentException("Cannot register a null metric");
247         }
248 
249         if (timeBucketSizeInSecs <= 0) {
250             throw new IllegalArgumentException("TopologyContext.registerMetric can only be called with timeBucketSizeInSecs " +
251                                                "greater than or equal to 1 second.");
252         }
253         
254         if (getRegisteredMetricByName(name) != null) {
255             throw new RuntimeException("The same metric name `" + name + "` was registered twice." );
256         }
257 
258         Map m1 = _registeredMetrics;
259         if(!m1.containsKey(timeBucketSizeInSecs)) {
260             m1.put(timeBucketSizeInSecs, new HashMap());
261         }
262 
263         Map m2 = (Map)m1.get(timeBucketSizeInSecs);
264         if(!m2.containsKey(_taskId)) {
265             m2.put(_taskId, new HashMap());
266         }
267 
268         Map m3 = (Map)m2.get(_taskId);
269         if(m3.containsKey(name)) {
270             throw new RuntimeException("The same metric name `" + name + "` was registered twice." );
271         } else {
272             m3.put(name, metric);
273         }
274 
275         return metric;
276     }

278     /**
279      * Get component's metric from registered metrics by name.
280      * Notice: Normally, one component can only register one metric name once.
281      *         But now registerMetric has a bug(https://issues.apache.org/jira/browse/STORM-254) 
282      *         cause the same metric name can register twice.
283      *         So we just return the first metric we meet.
284      */
285     public IMetric getRegisteredMetricByName(String name) {
286         IMetric metric = null;
287 
288         for (Map<Integer, Map<String, IMetric>> taskIdToNameToMetric: _registeredMetrics.values()) {
289             Map<String, IMetric> nameToMetric = taskIdToNameToMetric.get(_taskId);
290             if (nameToMetric != null) {
291                 metric = nameToMetric.get(name);
292                 if (metric != null) {
293                     //we just return the first metric we meet
294                     break;  
295                 }
296             }
297         } 
298         
299         return metric;
300     }
