51  /**
52   * A Netty client for sending task messages to a remote destination (Netty server).
53   *
54   * Implementation details:
55   *
56   * - Sending messages, i.e. writing to the channel, is performed asynchronously.
57   * - Messages are sent in batches to optimize for network throughput at the expense of network latency.  The message
58   *   batch size is configurable.
59   * - Connecting and reconnecting are performed asynchronously.
60   *     - Note: The current implementation drops any messages that are being enqueued for sending if the connection to
61   *       the remote destination is currently unavailable.
62   */
63  public class Client extends ConnectionWithStatus implements IStatefulObject {
64      private static final long PENDING_MESSAGES_FLUSH_TIMEOUT_MS = 600000L;
65      private static final long PENDING_MESSAGES_FLUSH_INTERVAL_MS = 1000L;
66  
67      private static final Logger LOG = LoggerFactory.getLogger(Client.class);
68      private static final String PREFIX = "Netty-Client-";
69      private static final long NO_DELAY_MS = 0L;
70  
71      private final StormBoundedExponentialBackoffRetry retryPolicy;
72      private final ClientBootstrap bootstrap;
73      private final InetSocketAddress dstAddress;
74      protected final String dstAddressPrefixedName;
75  
76      /**
77       * The channel used for all write operations from this client to the remote destination.
78       */
79      private final AtomicReference<Channel> channelRef = new AtomicReference<Channel>();
80  
81  
82      /**
83       * Maximum number of reconnection attempts we will perform after a disconnect before giving up.
84       */
85      private final int maxReconnectionAttempts;
86  
87      /**
88       * Total number of connection attempts.
89       */
90      private final AtomicInteger totalConnectionAttempts = new AtomicInteger(0);
91  
92      /**
93       * Number of connection attempts since the last disconnect.
94       */
95      private final AtomicInteger connectionAttempts = new AtomicInteger(0);
96  
97      /**
98       * Number of messages successfully sent to the remote destination.
99       */
100     private final AtomicInteger messagesSent = new AtomicInteger(0);
101 
102     /**
103      * Number of messages that could not be sent to the remote destination.
104      */
105     private final AtomicInteger messagesLost = new AtomicInteger(0);
106 
107     /**
108      * Number of messages buffered in memory.
109      */
110     private final AtomicLong pendingMessages = new AtomicLong(0);

113     /**
114      * This flag is set to true if and only if a client instance is being closed.
115      */
116     private volatile boolean closing = false;
117 
118     private final HashedWheelTimer scheduler;
119 
120     private final MessageBuffer batcher;
121 
122     private final Object writeLock = new Object();

124     @SuppressWarnings("rawtypes")
125     Client(Map stormConf, ChannelFactory factory, HashedWheelTimer scheduler, String host, int port) {
126         closing = false;
127         this.scheduler = scheduler;
128         int bufferSize = Utils.getInt(stormConf.get(Config.STORM_MESSAGING_NETTY_BUFFER_SIZE));
129         LOG.info("creating Netty Client, connecting to {}:{}, bufferSize: {}", host, port, bufferSize);
130         int messageBatchSize = Utils.getInt(stormConf.get(Config.STORM_NETTY_MESSAGE_BATCH_SIZE), 262144);
131 
132         maxReconnectionAttempts = Utils.getInt(stormConf.get(Config.STORM_MESSAGING_NETTY_MAX_RETRIES));
133         int minWaitMs = Utils.getInt(stormConf.get(Config.STORM_MESSAGING_NETTY_MIN_SLEEP_MS));
134         int maxWaitMs = Utils.getInt(stormConf.get(Config.STORM_MESSAGING_NETTY_MAX_SLEEP_MS));
135         retryPolicy = new StormBoundedExponentialBackoffRetry(minWaitMs, maxWaitMs, maxReconnectionAttempts);
136 
137         // Initiate connection to remote destination
138         bootstrap = createClientBootstrap(factory, bufferSize);
139         dstAddress = new InetSocketAddress(host, port);
140         dstAddressPrefixedName = prefixedName(dstAddress);
141         scheduleConnect(NO_DELAY_MS);
142         batcher = new MessageBuffer(messageBatchSize);
143     }
144 
145     private ClientBootstrap createClientBootstrap(ChannelFactory factory, int bufferSize) {
146         ClientBootstrap bootstrap = new ClientBootstrap(factory);
147         bootstrap.setOption("tcpNoDelay", true);
148         bootstrap.setOption("sendBufferSize", bufferSize);
149         bootstrap.setOption("keepAlive", true);
150         bootstrap.setPipelineFactory(new StormClientPipelineFactory(this));
151         return bootstrap;
152     }
153 
154     private String prefixedName(InetSocketAddress dstAddress) {
155         if (null != dstAddress) {
156             return PREFIX + dstAddress.toString();
157         }
158         return "";
159     }

161     /**
162      * We will retry connection with exponential back-off policy
163      */
164     private void scheduleConnect(long delayMs) {
165         scheduler.newTimeout(new Connect(dstAddress), delayMs, TimeUnit.MILLISECONDS);
166     }
167 
168     private boolean reconnectingAllowed() {
169         return !closing && connectionAttempts.get() <= (maxReconnectionAttempts + 1);
170     }
171 
172     private boolean connectionEstablished(Channel channel) {
173         // Because we are using TCP (which is a connection-oriented transport unlike UDP), a connection is only fully
174         // established iff the channel is connected.  That is, a TCP-based channel must be in the CONNECTED state before
175         // anything can be read or written to the channel.
176         //
177         // See:
178         // - http://netty.io/3.9/api/org/jboss/netty/channel/ChannelEvent.html
179         // - http://stackoverflow.com/questions/13356622/what-are-the-netty-channel-state-transitions
180         return channel != null && channel.isConnected();
181     }
182 
183     /**
184      * Note:  Storm will check via this method whether a worker can be activated safely during the initial startup of a
185      * topology.  The worker will only be activated once all of the its connections are ready.
186      */
187     @Override
188     public Status status() {
189         if (closing) {
190             return Status.Closed;
191         } else if (!connectionEstablished(channelRef.get())) {
192             return Status.Connecting;
193         } else {
194             return Status.Ready;
195         }
196     }

208     @Override
209     public void send(int taskId, byte[] payload) {
210         TaskMessage msg = new TaskMessage(taskId, payload);
211         List<TaskMessage> wrapper = new ArrayList<TaskMessage>(1);
212         wrapper.add(msg);
213         send(wrapper.iterator());
214     }

216     /**
217      * Enqueue task messages to be sent to the remote destination (cf. `host` and `port`).
218      */
219     @Override
220     public void send(Iterator<TaskMessage> msgs) {
221         if (closing) {
222             int numMessages = iteratorSize(msgs);
223             LOG.error("discarding {} messages because the Netty client to {} is being closed", numMessages,
224                     dstAddressPrefixedName);
225             return;
226         }
227 
228         if (!hasMessages(msgs)) {
229             return;
230         }
231 
232         Channel channel = getConnectedChannel();
233         if (channel == null) {
234             /*
235              * Connection is unavailable. We will drop pending messages and let at-least-once message replay kick in.
236              *
237              * Another option would be to buffer the messages in memory.  But this option has the risk of causing OOM errors,
238              * especially for topologies that disable message acking because we don't know whether the connection recovery will
239              * succeed  or not, and how long the recovery will take.
240              */
241             dropMessages(msgs);
242             return;
243         }
244 
245         synchronized (writeLock) {
246             while (msgs.hasNext()) {
247                 TaskMessage message = msgs.next();
248                 MessageBatch full = batcher.add(message);
249                 if(full != null){
250                     flushMessages(channel, full);
251                 }
252             }
253         }
254 
255         if(channel.isWritable()){
256             synchronized (writeLock) {
257                 // Netty's internal buffer is not full and we still have message left in the buffer.
258                 // We should write the unfilled MessageBatch immediately to reduce latency
259                 MessageBatch batch = batcher.drain();
260                 if(batch != null) {
261                     flushMessages(channel, batch);
262                 }
263             }
264         } else {
265             // Channel's buffer is full, meaning that we have time to wait other messages to arrive, and create a bigger
266             // batch. This yields better throughput.
267             // We can rely on `notifyInterestChanged` to push these messages as soon as there is spece in Netty's buffer
268             // because we know `Channel.isWritable` was false after the messages were already in the buffer.
269         }
270     }
271 
272     private Channel getConnectedChannel() {
273         Channel channel = channelRef.get();
274         if (connectionEstablished(channel)) {
275             return channel;
276         } else {
277             // Closing the channel and reconnecting should be done before handling the messages.
278             boolean reconnectScheduled = closeChannelAndReconnect(channel);
279             if (reconnectScheduled) {
280                 // Log the connection error only once
281                 LOG.error("connection to {} is unavailable", dstAddressPrefixedName);
282             }
283             return null;
284         }
285     }
286 
287     private boolean hasMessages(Iterator<TaskMessage> msgs) {
288         return msgs != null && msgs.hasNext();
289     }
290 
291 
292     private void dropMessages(Iterator<TaskMessage> msgs) {
293         // We consume the iterator by traversing and thus "emptying" it.
294         int msgCount = iteratorSize(msgs);
295         messagesLost.getAndAdd(msgCount);
296     }
297 
298     private int iteratorSize(Iterator<TaskMessage> msgs) {
299         int size = 0;
300         if (msgs != null) {
301             while (msgs.hasNext()) {
302                 size++;
303                 msgs.next();
304             }
305         }
306         return size;
307     }
308 
309     /**
310      * Asynchronously writes the message batch to the channel.
311      *
312      * If the write operation fails, then we will close the channel and trigger a reconnect.
313      */
314     private void flushMessages(Channel channel, final MessageBatch batch) {
315         if (null == batch || batch.isEmpty()) {
316             return;
317         }
318 
319         final int numMessages = batch.size();
320         LOG.debug("writing {} messages to channel {}", batch.size(), channel.toString());
321         pendingMessages.addAndGet(numMessages);
322 
323         ChannelFuture future = channel.write(batch);
324         future.addListener(new ChannelFutureListener() {
325             public void operationComplete(ChannelFuture future) throws Exception {
326                 pendingMessages.addAndGet(0 - numMessages);
327                 if (future.isSuccess()) {
328                     LOG.debug("sent {} messages to {}", numMessages, dstAddressPrefixedName);
329                     messagesSent.getAndAdd(batch.size());
330                 } else {
331                     LOG.error("failed to send {} messages to {}: {}", numMessages, dstAddressPrefixedName,
332                             future.getCause());
333                     closeChannelAndReconnect(future.getChannel());
334                     messagesLost.getAndAdd(numMessages);
335                 }
336             }
337 
338         });
339     }

341     /**
342      * Schedule a reconnect if we closed a non-null channel, and acquired the right to
343      * provide a replacement by successfully setting a null to the channel field
344      * @param channel
345      * @return if the call scheduled a re-connect task
346      */
347     private boolean closeChannelAndReconnect(Channel channel) {
348         if (channel != null) {
349             channel.close();
350             if (channelRef.compareAndSet(channel, null)) {
351                 scheduleConnect(NO_DELAY_MS);
352                 return true;
353             }
354         }
355         return false;
356     }
357 
358     /**
359      * Gracefully close this client.
360      */
361     @Override
362     public void close() {
363         if (!closing) {
364             LOG.info("closing Netty Client {}", dstAddressPrefixedName);
365             // Set closing to true to prevent any further reconnection attempts.
366             closing = true;
367             waitForPendingMessagesToBeSent();
368             closeChannel();
369         }
370     }
371 
372     private void waitForPendingMessagesToBeSent() {
373         LOG.info("waiting up to {} ms to send {} pending messages to {}",
374                 PENDING_MESSAGES_FLUSH_TIMEOUT_MS, pendingMessages.get(), dstAddressPrefixedName);
375         long totalPendingMsgs = pendingMessages.get();
376         long startMs = System.currentTimeMillis();
377         while (pendingMessages.get() != 0) {
378             try {
379                 long deltaMs = System.currentTimeMillis() - startMs;
380                 if (deltaMs > PENDING_MESSAGES_FLUSH_TIMEOUT_MS) {
381                     LOG.error("failed to send all pending messages to {} within timeout, {} of {} messages were not " +
382                             "sent", dstAddressPrefixedName, pendingMessages.get(), totalPendingMsgs);
383                     break;
384                 }
385                 Thread.sleep(PENDING_MESSAGES_FLUSH_INTERVAL_MS);
386             }
387             catch (InterruptedException e) {
388                 break;
389             }
390         }
391 
392     }
393 
394 
395     private void closeChannel() {
396         Channel channel = channelRef.get();
397         if (channel != null) {
398             channel.close();
399             LOG.debug("channel to {} closed", dstAddressPrefixedName);
400         }
401     }

403     @Override
404     public Object getState() {
405         LOG.info("Getting metrics for client connection to {}", dstAddressPrefixedName);
406         HashMap<String, Object> ret = new HashMap<String, Object>();
407         ret.put("reconnects", totalConnectionAttempts.getAndSet(0));
408         ret.put("sent", messagesSent.getAndSet(0));
409         ret.put("pending", pendingMessages.get());
410         ret.put("lostOnSend", messagesLost.getAndSet(0));
411         ret.put("dest", dstAddress.toString());
412         String src = srcAddressName();
413         if (src != null) {
414             ret.put("src", src);
415         }
416         return ret;
417     }
418 
419     private String srcAddressName() {
420         String name = null;
421         Channel channel = channelRef.get();
422         if (channel != null) {
423             SocketAddress address = channel.getLocalAddress();
424             if (address != null) {
425                 name = address.toString();
426             }
427         }
428         return name;
429     }

436     /**
437      * Called by Netty thread on change in channel interest
438      * @param channel
439      */
440     public void notifyInterestChanged(Channel channel) {
441         if(channel.isWritable()){
442             synchronized (writeLock) {
443                 // Channel is writable again, write if there are any messages pending
444                 MessageBatch pending = batcher.drain();
445                 flushMessages(channel, pending);
446             }
447         }
448     }

450     /**
451      * Asynchronously establishes a Netty connection to the remote address
452      * This task runs on a single thread shared among all clients, and thus
453      * should not perform operations that block.
454      */
455     private class Connect implements TimerTask {
456 
457         private final InetSocketAddress address;
458 
459         public Connect(InetSocketAddress address) {
460             this.address = address;
461         }
462 
463         private void reschedule(Throwable t) {
464             String baseMsg = String.format("connection attempt %s to %s failed", connectionAttempts,
465                     dstAddressPrefixedName);
466             String failureMsg = (t == null) ? baseMsg : baseMsg + ": " + t.toString();
467             LOG.error(failureMsg);
468             long nextDelayMs = retryPolicy.getSleepTimeMs(connectionAttempts.get(), 0);
469             scheduleConnect(nextDelayMs);
470         }
471 
472 
473         @Override
474         public void run(Timeout timeout) throws Exception {
475             if (reconnectingAllowed()) {
476                 final int connectionAttempt = connectionAttempts.getAndIncrement();
477                 totalConnectionAttempts.getAndIncrement();
478 
479                 LOG.debug("connecting to {} [attempt {}]", address.toString(), connectionAttempt);
480                 ChannelFuture future = bootstrap.connect(address);
481                 future.addListener(new ChannelFutureListener() {
482                     @Override
483                     public void operationComplete(ChannelFuture future) throws Exception {
484                         // This call returns immediately
485                         Channel newChannel = future.getChannel();
486 
487                         if (future.isSuccess() && connectionEstablished(newChannel)) {
488                             boolean setChannel = channelRef.compareAndSet(null, newChannel);
489                             checkState(setChannel);
490                             LOG.debug("successfully connected to {}, {} [attempt {}]", address.toString(), newChannel.toString(),
491                                     connectionAttempt);
492                             if (messagesLost.get() > 0) {
493                                 LOG.warn("Re-connection to {} was successful but {} messages has been lost so far", address.toString(), messagesLost.get());
494                             }
495                         } else {
496                             Throwable cause = future.getCause();
497                             reschedule(cause);
498                             if (newChannel != null) {
499                                 newChannel.close();
500                             }
501                         }
502                     }
503                 });
504             } else {
505                 close();
506                 throw new RuntimeException("Giving up to scheduleConnect to " + dstAddressPrefixedName + " after " +
507                         connectionAttempts + " failed attempts. " + messagesLost.get() + " messages were lost");
508 
509             }
510         }
511     }
