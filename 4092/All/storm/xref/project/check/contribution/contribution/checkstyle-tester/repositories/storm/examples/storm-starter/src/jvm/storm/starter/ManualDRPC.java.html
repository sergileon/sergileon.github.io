18  package storm.starter;
19  
20  import backtype.storm.Config;
21  import backtype.storm.LocalCluster;
22  import backtype.storm.LocalDRPC;
23  import backtype.storm.drpc.DRPCSpout;
24  import backtype.storm.drpc.ReturnResults;
25  import backtype.storm.topology.BasicOutputCollector;
26  import backtype.storm.topology.OutputFieldsDeclarer;
27  import backtype.storm.topology.TopologyBuilder;
28  import backtype.storm.topology.base.BaseBasicBolt;
29  import backtype.storm.tuple.Fields;
30  import backtype.storm.tuple.Tuple;
31  import backtype.storm.tuple.Values;
32  
33  
34  public class ManualDRPC {
35    public static class ExclamationBolt extends BaseBasicBolt {
36  
37      @Override
38      public void declareOutputFields(OutputFieldsDeclarer declarer) {
39        declarer.declare(new Fields("result", "return-info"));
40      }
41  
42      @Override
43      public void execute(Tuple tuple, BasicOutputCollector collector) {
44        String arg = tuple.getString(0);
45        Object retInfo = tuple.getValue(1);
46        collector.emit(new Values(arg + "!!!", retInfo));
47      }
48  
49    }
50  
51    public static void main(String[] args) {
52      TopologyBuilder builder = new TopologyBuilder();
53      LocalDRPC drpc = new LocalDRPC();
54  
55      DRPCSpout spout = new DRPCSpout("exclamation", drpc);
56      builder.setSpout("drpc", spout);
57      builder.setBolt("exclaim", new ExclamationBolt(), 3).shuffleGrouping("drpc");
58      builder.setBolt("return", new ReturnResults(), 3).shuffleGrouping("exclaim");
59  
60      LocalCluster cluster = new LocalCluster();
61      Config conf = new Config();
62      cluster.submitTopology("exclaim", conf, builder.createTopology());
63  
64      System.out.println(drpc.execute("exclamation", "aaa"));
65      System.out.println(drpc.execute("exclamation", "bbb"));
66  
67    }
68  }
