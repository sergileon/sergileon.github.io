142     public class CoordinatedOutputCollector implements IOutputCollector {
143         IOutputCollector _delegate;
144         
145         TrackedBatch _currBatch = null;;
146         
147         public void setCurrBatch(TrackedBatch batch) {
148             _currBatch = batch;
149         }

151         public CoordinatedOutputCollector(IOutputCollector delegate) {
152             _delegate = delegate;
153         }
154 
155         public List<Integer> emit(String stream, Collection<Tuple> anchors, List<Object> tuple) {
156             List<Integer> tasks = _delegate.emit(stream, anchors, tuple);
157             updateTaskCounts(tasks);
158             return tasks;
159         }
160 
161         public void emitDirect(int task, String stream, Collection<Tuple> anchors, List<Object> tuple) {
162             updateTaskCounts(Arrays.asList(task));
163             _delegate.emitDirect(task, stream, anchors, tuple);
164         }
165 
166         public void ack(Tuple tuple) {
167             throw new IllegalStateException("Method should never be called");
168         }
169 
170         public void fail(Tuple tuple) {
171             throw new IllegalStateException("Method should never be called");
172         }
173         
174         public void reportError(Throwable error) {
175             _delegate.reportError(error);
176         }

179         private void updateTaskCounts(List<Integer> tasks) {
180             if(_currBatch!=null) {
181                 Map<Integer, Integer> taskEmittedTuples = _currBatch.taskEmittedTuples;
182                 for(Integer task: tasks) {
183                     int newCount = Utils.get(taskEmittedTuples, task, 0) + 1;
184                     taskEmittedTuples.put(task, newCount);
185                 }
186             }
187         }
188     }

195     @Override
196     public void prepare(Map conf, TopologyContext context, OutputCollector collector) {        
197         _messageTimeoutMs = context.maxTopologyMessageTimeout() * 1000L;
198         _lastRotate = System.currentTimeMillis();
199         _batches = new RotatingMap(2);
200         _context = context;
201         _collector = collector;
202         _coordCollector = new CoordinatedOutputCollector(collector);
203         _coordOutputCollector = new BatchOutputCollectorImpl(new OutputCollector(_coordCollector));
204                 
205         _coordConditions = (Map) context.getExecutorData("__coordConditions");
206         if(_coordConditions==null) {
207             _coordConditions = new HashMap();
208             for(String batchGroup: _coordSpecs.keySet()) {
209                 CoordSpec spec = _coordSpecs.get(batchGroup);
210                 CoordCondition cond = new CoordCondition();
211                 cond.commitStream = spec.commitStream;
212                 cond.expectedTaskReports = 0;
213                 for(String comp: spec.coords.keySet()) {
214                     CoordType ct = spec.coords.get(comp);
215                     if(ct.equals(CoordType.single())) {
216                         cond.expectedTaskReports+=1;
217                     } else {
218                         cond.expectedTaskReports+=context.getComponentTasks(comp).size();
219                     }
220                 }
221                 cond.targetTasks = new HashSet<Integer>();
222                 for(String component: Utils.get(context.getThisTargets(),
223                                         COORD_STREAM(batchGroup),
224                                         new HashMap<String, Grouping>()).keySet()) {
225                     cond.targetTasks.addAll(context.getComponentTasks(component));
226                 }
227                 _coordConditions.put(batchGroup, cond);
228             }
229             context.setExecutorData("_coordConditions", _coordConditions);
230         }
231         _bolt.prepare(conf, context, _coordOutputCollector);
232     }

234     private void failBatch(TrackedBatch tracked, FailedException e) {
235         if(e!=null && e instanceof ReportedFailedException) {
236             _collector.reportError(e);
237         }
238         tracked.failed = true;
239         if(tracked.delayedAck!=null) {
240             _collector.fail(tracked.delayedAck);
241             tracked.delayedAck = null;
242         }
243     }
244     
245     private void failBatch(TrackedBatch tracked) {
246         failBatch(tracked, null);
247     }
248 
249     private boolean finishBatch(TrackedBatch tracked, Tuple finishTuple) {
250         boolean success = true;
251         try {
252             _bolt.finishBatch(tracked.info);
253             String stream = COORD_STREAM(tracked.info.batchGroup);
254             for(Integer task: tracked.condition.targetTasks) {
255                 _collector.emitDirect(task, stream, finishTuple, new Values(tracked.info.batchId, Utils.get(tracked.taskEmittedTuples, task, 0)));
256             }
257             if(tracked.delayedAck!=null) {
258                 _collector.ack(tracked.delayedAck);
259                 tracked.delayedAck = null;
260             }
261         } catch(FailedException e) {
262             failBatch(tracked, e);
263             success = false;
264         }
265         _batches.remove(tracked.info.batchId.getId());
266         return success;
267     }
268     
269     private void checkFinish(TrackedBatch tracked, Tuple tuple, TupleType type) {
270         if(tracked.failed) {
271             failBatch(tracked);
272             _collector.fail(tuple);
273             return;
274         }
275         CoordCondition cond = tracked.condition;
276         boolean delayed = tracked.delayedAck==null &&
277                               (cond.commitStream!=null && type==TupleType.COMMIT
278                                || cond.commitStream==null);
279         if(delayed) {
280             tracked.delayedAck = tuple;
281         }
282         boolean failed = false;
283         if(tracked.receivedCommit && tracked.reportedTasks == cond.expectedTaskReports) {
284             if(tracked.receivedTuples == tracked.expectedTupleCount) {
285                 finishBatch(tracked, tuple);                
286             } else {
287                 //TODO: add logging that not all tuples were received
288                 failBatch(tracked);
289                 _collector.fail(tuple);
290                 failed = true;
291             }
292         }
293         
294         if(!delayed && !failed) {
295             _collector.ack(tuple);
296         }
297         
298     }

300     @Override
301     public void execute(Tuple tuple) {
302         if(tuple.getSourceStreamId().equals(Constants.SYSTEM_TICK_STREAM_ID)) {
303             long now = System.currentTimeMillis();
304             if(now - _lastRotate > _messageTimeoutMs) {
305                 _batches.rotate();
306                 _lastRotate = now;
307             }
308             return;
309         }
310         String batchGroup = _batchGroupIds.get(tuple.getSourceGlobalStreamid());
311         if(batchGroup==null) {
312             // this is so we can do things like have simple DRPC that doesn't need to use batch processing
313             _coordCollector.setCurrBatch(null);
314             _bolt.execute(null, tuple);
315             _collector.ack(tuple);
316             return;
317         }
318         IBatchID id = (IBatchID) tuple.getValue(0);
319         //get transaction id
320         //if it already exissts and attempt id is greater than the attempt there
321         
322         
323         TrackedBatch tracked = (TrackedBatch) _batches.get(id.getId());
324 //        if(_batches.size() > 10 && _context.getThisTaskIndex() == 0) {
325 //            System.out.println("Received in " + _context.getThisComponentId() + " " + _context.getThisTaskIndex()
326 //                    + " (" + _batches.size() + ")" +
327 //                    "\ntuple: " + tuple +
328 //                    "\nwith tracked " + tracked +
329 //                    "\nwith id " + id + 
330 //                    "\nwith group " + batchGroup
331 //                    + "\n");
332 //            
333 //        }
334         //System.out.println("Num tracked: " + _batches.size() + " " + _context.getThisComponentId() + " " + _context.getThisTaskIndex());
335         
336         // this code here ensures that only one attempt is ever tracked for a batch, so when
337         // failures happen you don't get an explosion in memory usage in the tasks
338         if(tracked!=null) {
339             if(id.getAttemptId() > tracked.attemptId) {
340                 _batches.remove(id.getId());
341                 tracked = null;
342             } else if(id.getAttemptId() < tracked.attemptId) {
343                 // no reason to try to execute a previous attempt than we've already seen
344                 return;
345             }
346         }
347         
348         if(tracked==null) {
349             tracked = new TrackedBatch(new BatchInfo(batchGroup, id, _bolt.initBatchState(batchGroup, id)), _coordConditions.get(batchGroup), id.getAttemptId());
350             _batches.put(id.getId(), tracked);
351         }
352         _coordCollector.setCurrBatch(tracked);
353         
354         //System.out.println("TRACKED: " + tracked + " " + tuple);
355         
356         TupleType t = getTupleType(tuple, tracked);
357         if(t==TupleType.COMMIT) {
358             tracked.receivedCommit = true;
359             checkFinish(tracked, tuple, t);
360         } else if(t==TupleType.COORD) {
361             int count = tuple.getInteger(1);
362             tracked.reportedTasks++;
363             tracked.expectedTupleCount+=count;
364             checkFinish(tracked, tuple, t);
365         } else {
366             tracked.receivedTuples++;
367             boolean success = true;
368             try {
369                 _bolt.execute(tracked.info, tuple);
370                 if(tracked.condition.expectedTaskReports==0) {
371                     success = finishBatch(tracked, tuple);
372                 }
373             } catch(FailedException e) {
374                 failBatch(tracked, e);
375             }
376             if(success) {
377                 _collector.ack(tuple);                   
378             } else {
379                 _collector.fail(tuple);
380             }
381         }
382         _coordCollector.setCurrBatch(null);
383     }

407     private TupleType getTupleType(Tuple tuple, TrackedBatch batch) {
408         CoordCondition cond = batch.condition;
409         if(cond.commitStream!=null
410                 && tuple.getSourceGlobalStreamid().equals(cond.commitStream)) {
411             return TupleType.COMMIT;
412         } else if(cond.expectedTaskReports > 0
413                 && tuple.getSourceStreamId().startsWith(COORD_STREAM_PREFIX)) {
414             return TupleType.COORD;
415         } else {
416             return TupleType.REGULAR;
417         }
418     }
