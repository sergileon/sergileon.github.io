18  package storm.trident.state.map;
19  
20  import backtype.storm.tuple.Fields;
21  import backtype.storm.tuple.Values;
22  import java.util.ArrayList;
23  import java.util.List;
24  import java.util.Map;
25  import storm.trident.operation.CombinerAggregator;
26  import storm.trident.operation.TridentCollector;
27  import storm.trident.operation.TridentOperationContext;
28  import storm.trident.state.CombinerValueUpdater;
29  import storm.trident.state.StateUpdater;
30  import storm.trident.state.ValueUpdater;
31  import storm.trident.tuple.ComboList;
32  import storm.trident.tuple.TridentTuple;
33  import storm.trident.tuple.TridentTupleView.ProjectionFactory;
34  
35  public class MapCombinerAggStateUpdater implements StateUpdater<MapState> {
36      CombinerAggregator _agg;
37      Fields _groupFields;
38      Fields _inputFields;
39      ProjectionFactory _groupFactory;
40      ProjectionFactory _inputFactory;
41      ComboList.Factory _factory;
42      
43      
44      public MapCombinerAggStateUpdater(CombinerAggregator agg, Fields groupFields, Fields inputFields) {
45          _agg = agg;
46          _groupFields = groupFields;
47          _inputFields = inputFields;
48          if(inputFields.size()!=1) {
49              throw new IllegalArgumentException("Combiner aggs only take a single field as input. Got this instead: " + inputFields.toString());
50          }
51          _factory = new ComboList.Factory(groupFields.size(), inputFields.size());
52      }
53      
54  
55      @Override
56      public void updateState(MapState map, List<TridentTuple> tuples, TridentCollector collector) {
57          List<List<Object>> groups = new ArrayList<List<Object>>(tuples.size());
58          List<ValueUpdater> updaters = new ArrayList<ValueUpdater>(tuples.size());
59                  
60          for(TridentTuple t: tuples) {
61              groups.add(_groupFactory.create(t));
62              updaters.add(new CombinerValueUpdater(_agg,_inputFactory.create(t).getValue(0)));
63          }
64          List<Object> newVals = map.multiUpdate(groups, updaters);
65         
66          for(int i=0; i<tuples.size(); i++) {
67              List<Object> key = groups.get(i);
68              Object result = newVals.get(i);            
69              collector.emit(_factory.create(new List[] {key, new Values(result) }));
70          }
71      }
72  
73      @Override
74      public void prepare(Map conf, TridentOperationContext context) {
75          _groupFactory = context.makeProjectionFactory(_groupFields);
76          _inputFactory = context.makeProjectionFactory(_inputFields);
77      }
78  
79      @Override
80      public void cleanup() {
81      }
82      
83  }
