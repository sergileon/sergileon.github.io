42  /**
43   * A bolt that shells out to another process to process tuples. ShellBolt
44   * communicates with that process over stdio using a special protocol. An ~100
45   * line library is required to implement that protocol, and adapter libraries
46   * currently exist for Ruby and Python.
47   *
48   * <p>To run a ShellBolt on a cluster, the scripts that are shelled out to must be
49   * in the resources directory within the jar submitted to the master.
50   * During development/testing on a local machine, that resources directory just
51   * needs to be on the classpath.</p>
52   *
53   * <p>When creating topologies using the Java API, subclass this bolt and implement
54   * the IRichBolt interface to create components for the topology that use other languages. For example:
55   * </p>
56   *
57   * <pre>
58   * public class MyBolt extends ShellBolt implements IRichBolt {
59   *      public MyBolt() {
60   *          super("python", "mybolt.py");
61   *      }
62   *
63   *      public void declareOutputFields(OutputFieldsDeclarer declarer) {
64   *          declarer.declare(new Fields("field1", "field2"));
65   *      }
66   * }
67   * </pre>
68   */
69  public class ShellBolt implements IBolt {
70      public static final String HEARTBEAT_STREAM_ID = "__heartbeat";
71      public static Logger LOG = LoggerFactory.getLogger(ShellBolt.class);
72      Process _subprocess;
73      OutputCollector _collector;
74      Map<String, Tuple> _inputs = new ConcurrentHashMap<String, Tuple>();
75  
76      private String[] _command;
77      private ShellProcess _process;
78      private volatile boolean _running = true;
79      private volatile Throwable _exception;
80      private LinkedBlockingQueue _pendingWrites = new LinkedBlockingQueue();
81      private Random _rand;
82  
83      private Thread _readerThread;
84      private Thread _writerThread;
85      
86      private TopologyContext _context;
87  
88      private int workerTimeoutMills;
89      private ScheduledExecutorService heartBeatExecutorService;
90      private AtomicLong lastHeartbeatTimestamp = new AtomicLong();
91      private AtomicBoolean sendHeartbeatFlag = new AtomicBoolean(false);
92  
93      public ShellBolt(ShellComponent component) {
94          this(component.get_execution_command(), component.get_script());
95      }
96  
97      public ShellBolt(String... command) {
98          _command = command;
99      }
100 
101     public void prepare(Map stormConf, TopologyContext context,
102                         final OutputCollector collector) {
103         Object maxPending = stormConf.get(Config.TOPOLOGY_SHELLBOLT_MAX_PENDING);
104         if (maxPending != null) {
105            this._pendingWrites = new LinkedBlockingQueue(((Number)maxPending).intValue());
106         }
107         _rand = new Random();
108         _collector = collector;
109 
110         _context = context;
111 
112         workerTimeoutMills = 1000 * RT.intCast(stormConf.get(Config.SUPERVISOR_WORKER_TIMEOUT_SECS));
113 
114         _process = new ShellProcess(_command);
115 
116         //subprocesses must send their pid first thing
117         Number subpid = _process.launch(stormConf, context);
118         LOG.info("Launched subprocess with pid " + subpid);
119 
120         // reader
121         _readerThread = new Thread(new BoltReaderRunnable());
122         _readerThread.start();
123 
124         _writerThread = new Thread(new BoltWriterRunnable());
125         _writerThread.start();
126 
127         heartBeatExecutorService = MoreExecutors.getExitingScheduledExecutorService(new ScheduledThreadPoolExecutor(1));
128         heartBeatExecutorService.scheduleAtFixedRate(new BoltHeartbeatTimerTask(this), 1, 1, TimeUnit.SECONDS);
129 
130         LOG.info("Start checking heartbeat...");
131         setHeartbeat();
132     }
133 
134     public void execute(Tuple input) {
135         if (_exception != null) {
136             throw new RuntimeException(_exception);
137         }
138 
139         //just need an id
140         String genId = Long.toString(_rand.nextLong());
141         _inputs.put(genId, input);
142         try {
143             BoltMsg boltMsg = createBoltMessage(input, genId);
144 
145             _pendingWrites.put(boltMsg);
146         } catch(InterruptedException e) {
147             String processInfo = _process.getProcessInfoString() + _process.getProcessTerminationInfoString();
148             throw new RuntimeException("Error during multilang processing " + processInfo, e);
149         }
150     }
151 
152     private BoltMsg createBoltMessage(Tuple input, String genId) {
153         BoltMsg boltMsg = new BoltMsg();
154         boltMsg.setId(genId);
155         boltMsg.setComp(input.getSourceComponent());
156         boltMsg.setStream(input.getSourceStreamId());
157         boltMsg.setTask(input.getSourceTask());
158         boltMsg.setTuple(input.getValues());
159         return boltMsg;
160     }
161 
162     public void cleanup() {
163         _running = false;
164         heartBeatExecutorService.shutdownNow();
165         _writerThread.interrupt();
166         _readerThread.interrupt();
167         _process.destroy();
168         _inputs.clear();
169     }
170 
171     private void handleAck(Object id) {
172         Tuple acked = _inputs.remove(id);
173         if(acked==null) {
174             throw new RuntimeException("Acked a non-existent or already acked/failed id: " + id);
175         }
176         _collector.ack(acked);
177     }
178 
179     private void handleFail(Object id) {
180         Tuple failed = _inputs.remove(id);
181         if(failed==null) {
182             throw new RuntimeException("Failed a non-existent or already acked/failed id: " + id);
183         }
184         _collector.fail(failed);
185     }
186 
187     private void handleError(String msg) {
188         _collector.reportError(new Exception("Shell Process Exception: " + msg));
189     }
190 
191     private void handleEmit(ShellMsg shellMsg) throws InterruptedException {
192         List<Tuple> anchors = new ArrayList<Tuple>();
193         List<String> recvAnchors = shellMsg.getAnchors();
194         if (recvAnchors != null) {
195             for (String anchor : recvAnchors) {
196                 Tuple t = _inputs.get(anchor);
197                 if (t == null) {
198                     throw new RuntimeException("Anchored onto " + anchor + " after ack/fail");
199                 }
200                 anchors.add(t);
201             }
202         }
203 
204         if(shellMsg.getTask() == 0) {
205             List<Integer> outtasks = _collector.emit(shellMsg.getStream(), anchors, shellMsg.getTuple());
206             if (shellMsg.areTaskIdsNeeded()) {
207                 _pendingWrites.put(outtasks);
208             }
209         } else {
210             _collector.emitDirect((int) shellMsg.getTask(),
211                     shellMsg.getStream(), anchors, shellMsg.getTuple());
212         }
213     }
214 
215     private void handleLog(ShellMsg shellMsg) {
216         String msg = shellMsg.getMsg();
217         msg = "ShellLog " + _process.getProcessInfoString() + " " + msg;
218         ShellMsg.ShellLogLevel logLevel = shellMsg.getLogLevel();
219 
220         switch (logLevel) {
221             case TRACE:
222                 LOG.trace(msg);
223                 break;
224             case DEBUG:
225                 LOG.debug(msg);
226                 break;
227             case INFO:
228                 LOG.info(msg);
229                 break;
230             case WARN:
231                 LOG.warn(msg);
232                 break;
233             case ERROR:
234                 LOG.error(msg);
235                 _collector.reportError(new ReportedFailedException(msg));
236                 break;
237             default:
238                 LOG.info(msg);
239                 break;
240         }
241     }
242 
243     private void handleMetrics(ShellMsg shellMsg) {
244         //get metric name
245         String name = shellMsg.getMetricName();
246         if (name.isEmpty()) {
247             throw new RuntimeException("Receive Metrics name is empty");
248         }
249         
250         //get metric by name
251         IMetric iMetric = _context.getRegisteredMetricByName(name);
252         if (iMetric == null) {
253             throw new RuntimeException("Could not find metric by name["+name+"] ");
254         }
255         if ( !(iMetric instanceof IShellMetric)) {
256             throw new RuntimeException("Metric["+name+"] is not IShellMetric, can not call by RPC");
257         }
258         IShellMetric iShellMetric = (IShellMetric)iMetric;
259         
260         //call updateMetricFromRPC with params
261         Object paramsObj = shellMsg.getMetricParams();
262         try {
263             iShellMetric.updateMetricFromRPC(paramsObj);
264         } catch (RuntimeException re) {
265             throw re;
266         } catch (Exception e) {
267             throw new RuntimeException(e);
268         }       
269     }
270 
271     private void setHeartbeat() {
272         lastHeartbeatTimestamp.set(System.currentTimeMillis());
273     }
274 
275     private long getLastHeartbeat() {
276         return lastHeartbeatTimestamp.get();
277     }
278 
279     private void die(Throwable exception) {
280         String processInfo = _process.getProcessInfoString() + _process.getProcessTerminationInfoString();
281         _exception = new RuntimeException(processInfo, exception);
282         LOG.error("Halting process: ShellBolt died.", exception);
283         _collector.reportError(exception);
284         if (_running || (exception instanceof Error)) { //don't exit if not running, unless it is an Error
285             System.exit(11);
286         }
287     }

289     private class BoltHeartbeatTimerTask extends TimerTask {
290         private ShellBolt bolt;
291 
292         public BoltHeartbeatTimerTask(ShellBolt bolt) {
293             this.bolt = bolt;
294         }
295 
296         @Override
297         public void run() {
298             long currentTimeMillis = System.currentTimeMillis();
299             long lastHeartbeat = getLastHeartbeat();
300 
301             LOG.debug("BOLT - current time : {}, last heartbeat : {}, worker timeout (ms) : {}",
302                     currentTimeMillis, lastHeartbeat, workerTimeoutMills);
303 
304             if (currentTimeMillis - lastHeartbeat > workerTimeoutMills) {
305                 bolt.die(new RuntimeException("subprocess heartbeat timeout"));
306             }
307 
308             sendHeartbeatFlag.compareAndSet(false, true);
309         }
310 
311 
312     }
313 
314     private class BoltReaderRunnable implements Runnable {
315         public void run() {
316             while (_running) {
317                 try {
318                     ShellMsg shellMsg = _process.readShellMsg();
319 
320                     String command = shellMsg.getCommand();
321                     if (command == null) {
322                         throw new IllegalArgumentException("Command not found in bolt message: " + shellMsg);
323                     }
324                     if (command.equals("sync")) {
325                         setHeartbeat();
326                     } else if(command.equals("ack")) {
327                         handleAck(shellMsg.getId());
328                     } else if (command.equals("fail")) {
329                         handleFail(shellMsg.getId());
330                     } else if (command.equals("error")) {
331                         handleError(shellMsg.getMsg());
332                     } else if (command.equals("log")) {
333                         handleLog(shellMsg);
334                     } else if (command.equals("emit")) {
335                         handleEmit(shellMsg);
336                     } else if (command.equals("metrics")) {
337                         handleMetrics(shellMsg);
338                     }
339                 } catch (InterruptedException e) {
340                 } catch (Throwable t) {
341                     die(t);
342                 }
343             }
344         }
345     }
346 
347     private class BoltWriterRunnable implements Runnable {
348         public void run() {
349             while (_running) {
350                 try {
351                     if (sendHeartbeatFlag.get()) {
352                         LOG.debug("BOLT - sending heartbeat request to subprocess");
353 
354                         String genId = Long.toString(_rand.nextLong());
355                         _process.writeBoltMsg(createHeartbeatBoltMessage(genId));
356                         sendHeartbeatFlag.compareAndSet(true, false);
357                     }
358 
359                     Object write = _pendingWrites.poll(1, SECONDS);
360                     if (write instanceof BoltMsg) {
361                         _process.writeBoltMsg((BoltMsg) write);
362                     } else if (write instanceof List<?>) {
363                         _process.writeTaskIds((List<Integer>)write);
364                     } else if (write != null) {
365                         throw new RuntimeException("Unknown class type to write: " + write.getClass().getName());
366                     }
367                 } catch (InterruptedException e) {
368                 } catch (Throwable t) {
369                     die(t);
370                 }
371             }
372         }
373 
374         private BoltMsg createHeartbeatBoltMessage(String genId) {
375             BoltMsg msg = new BoltMsg();
376             msg.setId(genId);
377             msg.setTask(Constants.SYSTEM_TASK_ID);
378             msg.setStream(HEARTBEAT_STREAM_ID);
379             msg.setTuple(new ArrayList<Object>());
380             return msg;
381         }
382     }
