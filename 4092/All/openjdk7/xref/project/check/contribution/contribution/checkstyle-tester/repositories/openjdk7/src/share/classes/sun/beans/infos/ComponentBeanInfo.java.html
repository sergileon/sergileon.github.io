26  package sun.beans.infos;
27  
28  import java.beans.*;
29  
30  /**
31   * BeanInfo descriptor for a standard AWT component.
32   */
33  
34  public class ComponentBeanInfo extends SimpleBeanInfo {
35      private static final Class beanClass = java.awt.Component.class;
36  
37      public PropertyDescriptor[] getPropertyDescriptors() {
38          try {
39              PropertyDescriptor
40                        name = new PropertyDescriptor("name",       beanClass),
41                  background = new PropertyDescriptor("background", beanClass),
42                  foreground = new PropertyDescriptor("foreground", beanClass),
43                        font = new PropertyDescriptor("font",       beanClass),
44                     enabled = new PropertyDescriptor("enabled",    beanClass),
45                     visible = new PropertyDescriptor("visible",    beanClass),
46                   focusable = new PropertyDescriptor("focusable",  beanClass);
47  
48              enabled.setExpert(true);
49              visible.setHidden(true);
50  
51              background.setBound(true);
52              foreground.setBound(true);
53              font.setBound(true);
54              focusable.setBound(true);
55  
56              PropertyDescriptor[] rv = {name, background, foreground, font, enabled, visible, focusable };
57              return rv;
58          } catch (IntrospectionException e) {
59              throw new Error(e.toString());
60          }
61      }
62  }
