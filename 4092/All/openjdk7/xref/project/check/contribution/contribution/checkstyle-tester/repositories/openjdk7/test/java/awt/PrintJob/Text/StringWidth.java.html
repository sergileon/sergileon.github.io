24  import java.awt.*;
25  import java.util.Properties;
26  import sun.awt.*;
27  
28  public class StringWidth extends Frame {
29  
30    public StringWidth() {
31      Font plain = new Font("Dialog", Font.PLAIN, 10);
32      Font bold = new Font("Dialog", Font.BOLD, 10);
33      Properties props = new Properties();
34      int x, y;
35  
36      // we must have visible Frame context for PrintDialog in Solaris
37      setSize(400, 300);
38      setVisible(true);
39  
40      PrintJob pj = getToolkit().getPrintJob(this, "", props);
41      if (pj == null) {
42          return;
43      }
44      Graphics  pg = pj.getGraphics();
45  
46  
47      String test = "Hello World!";
48  
49      FontMetrics plainFm = pg.getFontMetrics(plain);
50      FontMetrics boldFm = pg.getFontMetrics(bold);
51      Dimension size = pj.getPageDimension();
52  
53      // now right justify on the printed page
54      int center = size.width/2;
55      y = 150;
56      x = center - plainFm.stringWidth(test);
57      pg.setFont(plain);
58      pg.drawString(test, x-1, y);
59  
60      pg.dispose();
61      pj.end();
62      setVisible(false);
63    }
64  
65    public static void main(String[] args) {
66      StringWidth sw = new StringWidth();
67      sw.dispose();
68    }
69  
70  }
