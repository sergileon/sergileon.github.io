5   import java.util.List;

12  
13  public class MergeTest {
14  
15      @Test public void
16      merges_streams_to_list_with_unit_merger_and_combiner() {
17          Stream<String> streamA = Stream.of("A", "B", "C");
18          Stream<String> streamB = Stream.of("apple", "banana", "carrot", "date");
19          Stream<String> streamC = Stream.of("fritter", "split", "cake", "roll", "pastry");
20  
21          Stream<List<String>> merged = StreamUtils.mergeToList(
22                  streamA, streamB, streamC);
23  
24          assertThat(merged.collect(toList()), contains(
25                  asList("A", "apple", "fritter"),
26                  asList("B", "banana", "split"),
27                  asList("C", "carrot", "cake"),
28                  asList("date", "roll"),
29                  asList("pastry")));
30      }
