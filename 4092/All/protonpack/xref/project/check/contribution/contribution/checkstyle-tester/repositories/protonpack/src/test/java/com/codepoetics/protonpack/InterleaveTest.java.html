1   package com.codepoetics.protonpack;
2   
3   import com.codepoetics.protonpack.selectors.Selectors;
4   import org.junit.Test;
5   
6   import java.util.Comparator;
7   import java.util.List;
8   import java.util.function.Function;
9   import java.util.stream.Collectors;
10  import java.util.stream.Stream;
11  
12  import static org.hamcrest.MatcherAssert.assertThat;
13  import static org.hamcrest.Matchers.contains;
14  
15  public class InterleaveTest {
16  
17      @Test public void
18      round_robin_interleaving() {
19          Stream<String> streamA = Stream.of("Peter", "Paul", "Mary");
20          Stream<String> streamB = Stream.of("A", "B", "C", "D", "E");
21          Stream<String> streamC = Stream.of("foo", "bar", "baz", "xyzzy");
22  
23          Stream<String> interleaved = StreamUtils.interleave(Selectors.roundRobin(), streamA, streamB, streamC);
24  
25          assertThat(interleaved.collect(Collectors.toList()), contains(
26                  "Peter", "A", "foo",
27                  "Paul", "B", "bar",
28                  "Mary", "C", "baz",
29                  "D", "xyzzy",
30                  "E"));
31      }
32  
33      @Test public void
34      sorted_interleaving() {
35          Stream<String> streamA = Stream.of("Peter", "B", "xyzzy");
36          Stream<String> streamB = Stream.of("A", "Paul", "C", "baz", "E");
37          Stream<String> streamC = Stream.of("foo", "bar", "D", "Mary");
38  
39          Stream<String> interleaved = StreamUtils.interleave(Selectors.takeMin(),
40                  streamA.sorted(), streamB.sorted(), streamC.sorted());
41  
42          List<String> collected = interleaved.collect(Collectors.toList());
43  
44          assertThat(collected, contains("A", "B", "C", "D", "E", "Mary", "Paul", "Peter", "bar", "baz", "foo", "xyzzy"));
45      }
46  
47      @Test public void
48      prioritised_interleaving() {
49          Stream<String> streamA = Stream.of("1 A1", "1 A2", "2 A3");
50          Stream<String> streamB = Stream.of("2 B1", "1 B2", "1 B3");
51  
52          Function<String, Integer> priority = s -> Integer.valueOf(s.substring(0, 1));
53          Stream<String> interleaved = StreamUtils.interleave(Selectors.takeMax(Comparator.comparing(priority)),
54                  streamA, streamB);
55  
56          List<String> collected = interleaved.collect(Collectors.toList());
57  
58          assertThat(collected, contains("2 B1", "1 A1", "1 B2", "1 A2", "2 A3", "1 B3"));
59      }
60  
61  }
