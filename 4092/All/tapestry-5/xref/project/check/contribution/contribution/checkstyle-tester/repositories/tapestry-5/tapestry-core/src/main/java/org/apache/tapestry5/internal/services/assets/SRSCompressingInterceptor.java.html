15  package org.apache.tapestry5.internal.services.assets;
16  
17  import org.apache.tapestry5.ioc.Resource;
18  import org.apache.tapestry5.services.assets.*;
19  
20  import java.io.IOException;
21  
22  public class SRSCompressingInterceptor extends DelegatingSRS
23  {
24      private final int compressionCutoff;
25  
26      private final AssetChecksumGenerator checksumGenerator;
27  
28      public SRSCompressingInterceptor(StreamableResourceSource delegate, int compressionCutoff, AssetChecksumGenerator checksumGenerator)
29      {
30          super(delegate);
31          this.compressionCutoff = compressionCutoff;
32          this.checksumGenerator = checksumGenerator;
33      }
34  
35      public StreamableResource getStreamableResource(Resource baseResource, StreamableResourceProcessing processing, ResourceDependencies dependencies)
36              throws IOException
37      {
38          StreamableResource streamable = delegate.getStreamableResource(baseResource, processing, dependencies);
39  
40          return processing == StreamableResourceProcessing.COMPRESSION_ENABLED ? compress(streamable) : streamable;
41      }
42  
43      private StreamableResource compress(StreamableResource uncompressed) throws IOException
44      {
45          if (uncompressed.getCompression() != CompressionStatus.COMPRESSABLE)
46          {
47              return uncompressed;
48          }
49  
50          int size = uncompressed.getSize();
51  
52          // Because of GZIP overhead, streams below a certain point actually get larger when compressed so
53          // we don't even try.
54  
55          if (size < compressionCutoff)
56          {
57              return uncompressed;
58          }
59  
60          return new CompressedStreamableResource(uncompressed, checksumGenerator);
61      }
62  }
