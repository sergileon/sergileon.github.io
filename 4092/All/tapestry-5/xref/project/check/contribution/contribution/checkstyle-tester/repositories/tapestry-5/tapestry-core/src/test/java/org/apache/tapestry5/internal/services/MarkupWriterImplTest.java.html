76      @Test
77      public void preamble_content() throws Exception
78      {
79          MarkupWriter w = new MarkupWriterImpl(new XMLMarkupModel());
80  
81          w.comment(" preamble start ");
82          w.write("preamble text");
83          w.cdata("CDATA content");
84          w.writeRaw("&nbsp;");
85          w.element("root");
86          w.end();
87          // You really shouldn't have any text after the close tag of the document, so it
88          // gets moved to the top, to the "preamble", before the first element.
89          w.comment(" content after root element in preamble ");
90  
91          assertEquals(w.getDocument().toString(), readFile("preamble_content.txt"));
92      }

147     @Test
148     public void element_nesting()
149     {
150         MarkupWriter w = new MarkupWriterImpl();
151 
152         Element root = w.element("root");
153 
154         w.attributes("foo", "bar");
155 
156         w.write("before child");
157 
158         assertNotSame(w.element("nested"), root);
159 
160         w.write("inner text");
161 
162         assertSame(w.end(), root);
163 
164         w.write("after child");
165 
166         root.attribute("gnip", "gnop");
167 
168         assertEquals(w.toString(),
169                 "<root gnip=\"gnop\" foo=\"bar\">before child<nested>inner text</nested>after child</root>");

292     @Test
293     public void namespaced_elements_and_attributes()
294     {
295         MarkupWriter w = new MarkupWriterImpl(new XMLMarkupModel());
296 
297         Element root = w.elementNS("fredns", "root");
298 
299         assertSame(root.defineNamespace("fredns", "fred"), root);
300 
301         root.defineNamespace("barneyns", "barney");
302 
303         assertSame(w.attributeNS("fredns", "foo", "bar"), root);
304 
305         Element child = w.elementNS("barneyns", "child");
306 
307         assertSame(child.getContainer(), root);
308 
309         w.end(); // child
310         w.end(); // root
311 
312         assertEquals(
313                 w.toString(),
314                 "<?xml version=\"1.0\"?>\n<fred:root fred:foo=\"bar\" xmlns:barney=\"barneyns\" xmlns:fred=\"fredns\"><barney:child/></fred:root>");
315     }

331     @Test
332     public void listeners()
333     {
334         MarkupWriter w = new MarkupWriterImpl(new XMLMarkupModel());
335 
336         MarkupWriterListener l = new MarkupWriterListener()
337         {
338             public void elementDidStart(Element element)
339             {
340                 element.text("[Start: " + element.getName() + "]");
341             }
342 
343             public void elementDidEnd(Element element)
344             {
345                 element.text("[End: " + element.getName() + "]");
346             }
347         };
348 
349         w.element("root");
350         w.element("no-listener");
351 
352         w.write("before listener");
353 
354         w.addListener(l);
355 
356         w.element("listener");
357         w.write("before n-w-l");
358         w.element("nested-with-listener");
359         w.write("n-w-l text");
360         w.end();
361         w.write("after n-w-l");
362         w.end();
363 
364         w.removeListener(l);
365 
366         w.write("after listener");
367 
368         w.end();
369         w.end();
370 
371         // Because we are invoking Element.text(), the text added by the listener is appended to the body of the
372         // element,
373         // which is correct but may not be what you'd expect.
374 
375         assertEquals(
376                 w.toString(),
377                 "<?xml version=\"1.0\"?>\n"
378                         + "<root><no-listener>before listener<listener>[Start: listener]before n-w-l<nested-with-listener>[Start: nested-with-listener]n-w-l text[End: nested-with-listener]</nested-with-listener>after n-w-l[End: listener]</listener>after listener</no-listener></root>");
379     }
