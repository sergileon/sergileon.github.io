13  package org.apache.tapestry5.services.assets;
14  
15  import org.apache.tapestry5.ContentType;
16  import org.apache.tapestry5.ioc.Resource;
17  import org.apache.tapestry5.ioc.annotations.IncompatibleChange;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  import java.io.OutputStream;
22  
23  /**
24   * An object, derived from a {@link Resource}, that can be streamed (ultimately, to a client web browser). In addition,
25   * a StreamableResource may be transformed (by the stack of interceptors around {@link StreamableResourceSource}: this
26   * includes transforming a Resource (example: CoffeeScript to JavaScript compilation), as well as aggregation
27   * and compression.
28   *
29   * @since 5.3
30   */
31  public interface StreamableResource
32  {
33      /**
34       * Describes the underlying {@link Resource} (or resources} for this streamable resource; expressly used
35       * as part of the object's {@code toString()}.
36       */
37      String getDescription();
38  
39      /**
40       * Indicates if the content is compressed, or compressable.
41       */
42      CompressionStatus getCompression();
43  
44      /**
45       * Returns the resource's content type.
46       */
47      @IncompatibleChange(release = "5.4", details = "Changed from type String to ContentType")
48      ContentType getContentType();
49  
50      /**
51       * The size, in bytes, of the underlying bytestream.
52       */
53      int getSize();
54  
55      /**
56       * Streams the resource's content to the provided stream. The caller is responsible for flushing or closing
57       * the output stream.
58       */
59      void streamTo(OutputStream os) throws IOException;
60  
61      /**
62       * Opens the content of the resource as an input stream; the caller is responsible for closing the stream
63       * after reading it.
64       *
65       * @return stream of the contents of the resource
66       * @throws IOException
67       */
68      InputStream openStream() throws IOException;
69  
70      /**
71       * Returns the time the resource was last modified, with accuracy to one second (so as to match
72       * the HTTP request/response date headers).
73       */
74      long getLastModified();
75  
76      /**
77       * Compute and return the checksum of the content for this asset; the checksum should be computed
78       * based on the uncompressed content.
79       *
80       * @return checksum for uncompressed content
81       * @see AssetChecksumGenerator#generateChecksum(StreamableResource)
82       * @since 5.4
83       */
84      String getChecksum() throws IOException;
85  
86  
87      /**
88       * Returns a new StreamableResource that includes the provided customizer. Customizers are invoked
89       * in the order they are added.
90       *
91       * @since 5.4
92       */
93      StreamableResource addResponseCustomizer(ResponseCustomizer customizer);
94  
95      /**
96       * Returns the customizer, if any, for this resource.  This may represent an aggregate customizer.
97       *
98       * @since 5.4
99       */
100     ResponseCustomizer getResponseCustomizer();
101 
102     /**
103      * Returns a new StreamableResource instance with the new content type.
104      *
105      * @param newContentType
106      * @since 5.4
107      */
108     StreamableResource withContentType(ContentType newContentType);
109 }
