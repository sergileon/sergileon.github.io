15  package org.apache.tapestry5.func;
16  
17  import java.util.Map;
18  
19  /**
20   * The result of the {@link Flow#zipWith(Flow)} method (or created from a Map via {@link F#zippedFlow(Map)}), a Flow of
21   * combined {@link Tuple} values (that can be deconstructed, eventually, using {@link #unzip()}).
22   * 
23   * @param <A>
24   * @param <B>
25   * @since 5.3
26   */
27  public interface ZippedFlow<A, B> extends FlowOperations<Tuple<A, B>, ZippedFlow<A, B>>
28  {
29      /**
30       * Mapping for zipped flows; a mapper is used to map tuples of this zipped flow into new tuples
31       * with a new type, forming the resulting zipped flow. This is a lazy operation.
32       */
33      <X, Y> ZippedFlow<X, Y> mapTuples(Mapper<Tuple<A, B>, Tuple<X, Y>> mapper);
34  
35      /**
36       * A ZippedFlow is a Flow of Tuples; this inverts that, splitting each Tuple into
37       * a Flow of elements, then assembling the result as a Tuple of two values. This is a lazy
38       * operation.
39       * 
40       * @return two flows of unzipped Tuples
41       */
42      Tuple<Flow<A>, Flow<B>> unzip();
43  
44      /**
45       * Returns a flow of the first values of the tuples of the zipped flow. This is a lazy
46       * operation.
47       */
48      Flow<A> firsts();
49  
50      /**
51       * Returns a flow of the second values of the tuples of the zipped flow. This is a lazy
52       * operation.
53       */
54      Flow<B> seconds();
55  
56      /**
57       * Filters the tuples in the zipped flow by applying a predicate to the first value in each tuple.
58       * This is a lazy operation.
59       */
60      ZippedFlow<A, B> filterOnFirst(Predicate<? super A> predicate);
61  
62      /**
63       * Filters the tuples in the zipped flow by applying a predicate to the second value in each tuple. This
64       * is a lazy operations.
65       */
66      ZippedFlow<A, B> filterOnSecond(Predicate<? super B> predicate);
67  
68      /**
69       * Removes tuples from the zipped flow by applying a predicate to the first value in each tuple.
70       * This is a lazy operation.
71       */
72      ZippedFlow<A, B> removeOnFirst(Predicate<? super A> predicate);
73  
74      /**
75       * Removes tuples from the zipped flow by applying a predicate to the second value in each tuple. This
76       * is a lazy operations.
77       */
78      ZippedFlow<A, B> removeOnSecond(Predicate<? super B> predicate);
79  
80      /**
81       * Constructs a HashMap by converting the tuples of the zipped flow into keys (first tuple value) and values (second
82       * tuple value).
83       */
84      Map<A, B> toMap();
85  }
