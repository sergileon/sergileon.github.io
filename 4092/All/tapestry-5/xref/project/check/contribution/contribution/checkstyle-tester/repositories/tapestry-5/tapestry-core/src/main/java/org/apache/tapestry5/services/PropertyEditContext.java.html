15  package org.apache.tapestry5.services;
16  
17  import org.apache.tapestry5.Field;
18  import org.apache.tapestry5.FieldTranslator;
19  import org.apache.tapestry5.FieldValidator;
20  import org.apache.tapestry5.ioc.AnnotationProvider;
21  import org.apache.tapestry5.ioc.Messages;
22  
23  /**
24   * Defines a context for editing a property of a bean via {@link org.apache.tapestry5.corelib.components.BeanEditor}.
25   * This value is made available to blocks via the {@link org.apache.tapestry5.annotations.Environmental} annotation.
26   *
27   * @see org.apache.tapestry5.services.BeanBlockSource
28   */
29  public interface PropertyEditContext extends AnnotationProvider
30  {
31      /**
32       * Returns the current value of the property being edited (the context encapsulates the object containing the
33       * property).
34       */
35      Object getPropertyValue();
36  
37      /**
38       * Updates the value of the property being edited (the context encapsulates the object containing the property).
39       *
40       * @param value new value for the property
41       */
42      void setPropertyValue(Object value);
43  
44      /**
45       * Returns the user-presentable label, for use with the {@link org.apache.tapestry5.corelib.components.Label}
46       * component, or to be integrated into any validation error messages.
47       */
48      String getLabel();
49  
50      /**
51       * Returns the translator appropriate for the field (this is based on the property type).
52       *
53       * @param field
54       * @see org.apache.tapestry5.services.TranslatorSource
55       */
56      FieldTranslator getTranslator(Field field);
57  
58      /**
59       * Returns the FieldValidator for the field.
60       *
61       * @see org.apache.tapestry5.beaneditor.Validate
62       * @see org.apache.tapestry5.services.FieldValidatorDefaultSource
63       */
64      FieldValidator getValidator(Field field);
65  
66      /**
67       * Returns a string that identifies the property, usually the property name. This is used as the basis for the
68       * client-side client id.
69       */
70      String getPropertyId();
71  
72      /**
73       * Returns the type of the property being edited.
74       */
75      Class getPropertyType();
76  
77      /**
78       * Returns the message catalog for the container of the {@link org.apache.tapestry5.corelib.components.BeanEditForm},
79       * which is the correct place to look for strings used for labels, etc.
80       */
81      Messages getContainerMessages();
82  }
