13  package org.apache.tapestry5.jpa;
14  
15  import org.apache.tapestry5.grid.GridDataSource;
16  import org.apache.tapestry5.grid.SortConstraint;
17  
18  import javax.persistence.EntityManager;
19  import javax.persistence.TypedQuery;
20  import javax.persistence.criteria.CriteriaBuilder;
21  import javax.persistence.criteria.CriteriaQuery;
22  import javax.persistence.criteria.Path;
23  import javax.persistence.criteria.Root;
24  import java.util.List;
25  
26  /**
27   * A simple implementation of {@link org.apache.tapestry5.grid.GridDataSource} based on a
28   * {@linkplain javax.persistence.EntityManager} and a known
29   * entity class. This implementation does support multiple
30   * {@link org.apache.tapestry5.grid.SortConstraint sort
31   * constraints}.
32   *
33   * This class is <em>not</em> thread-safe; it maintains internal state.
34   *
35   * Typically, an instance of this object is created fresh as needed (that is, it is not stored
36   * between requests).
37   *
38   * @since 5.3
39   */
40  public class JpaGridDataSource<E> implements GridDataSource
41  {
42  
43      private final EntityManager entityManager;
44  
45      private final Class<E> entityType;
46  
47      private int startIndex;
48  
49      private List<E> preparedResults;
50  
51      public JpaGridDataSource(final EntityManager entityManager, final Class<E> entityType)
52      {
53          super();
54          this.entityManager = entityManager;
55          this.entityType = entityType;
56      }
57  
58      /**
59       * {@inheritDoc}
60       */
61      @Override
62      public int getAvailableRows()
63      {
64          final CriteriaBuilder builder = entityManager.getCriteriaBuilder();
65  
66          CriteriaQuery<Long> criteria = builder.createQuery(Long.class);
67  
68          final Root<E> root = criteria.from(entityType);
69  
70          criteria = criteria.select(builder.count(root));
71  
72          applyAdditionalConstraints(criteria, root, builder);
73  
74          return entityManager.createQuery(criteria).getSingleResult().intValue();
75      }
76  
77      /**
78       * {@inheritDoc}
79       */
80      @Override
81      public void prepare(final int startIndex, final int endIndex,
82              final List<SortConstraint> sortConstraints)
83      {
84          final CriteriaBuilder builder = entityManager.getCriteriaBuilder();
85  
86          final CriteriaQuery<E> criteria = builder.createQuery(entityType);
87  
88          final Root<E> root = criteria.from(entityType);
89  
90          applyAdditionalConstraints(criteria.select(root), root, builder);
91  
92          for (final SortConstraint constraint : sortConstraints)
93          {
94  
95              final String propertyName = constraint.getPropertyModel().getPropertyName();
96  
97              final Path<Object> propertyPath = root.get(propertyName);
98  
99              switch (constraint.getColumnSort())
100             {
101 
102                 case ASCENDING:
103 
104                     criteria.orderBy(builder.asc(propertyPath));
105                     break;
106 
107                 case DESCENDING:
108                     criteria.orderBy(builder.desc(propertyPath));
109                     break;
110 
111                 default:
112             }
113         }
114 
115         final TypedQuery<E> query = entityManager.createQuery(criteria);
116 
117         query.setFirstResult(startIndex);
118         query.setMaxResults(endIndex - startIndex + 1);
119 
120         this.startIndex = startIndex;
121 
122         preparedResults = query.getResultList();
123 
124     }
125 
126     protected void applyAdditionalConstraints(final CriteriaQuery<?> criteria, final Root<E> root,
127             final CriteriaBuilder builder)
128     {
129     }
130 
131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     public Object getRowValue(final int index)
136     {
137         return preparedResults.get(index - startIndex);
138     }
139 
140     /**
141      * {@inheritDoc}
142      */
143     @Override
144     public Class<E> getRowType()
145     {
146         return entityType;
147     }
148 
149 }
