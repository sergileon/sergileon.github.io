15  package org.apache.tapestry5.ioc.util;
16  
17  import org.apache.tapestry5.ioc.MessageFormatter;
18  import org.apache.tapestry5.ioc.Messages;
19  import org.apache.tapestry5.ioc.internal.util.CollectionFactory;
20  import org.apache.tapestry5.ioc.internal.util.MessageFormatterImpl;
21  
22  import java.util.Locale;
23  import java.util.Map;
24  
25  /**
26   * Abstract implementation of {@link Messages} that doesn't know where values come from (that information is supplied in
27   * a subclass, via the {@link #valueForKey(String)} method).
28   */
29  public abstract class AbstractMessages implements Messages
30  {
31      /**
32       * String key to MF instance.
33       */
34      private final Map<String, MessageFormatter> cache = CollectionFactory.newConcurrentMap();
35  
36      private final Locale locale;
37  
38      protected AbstractMessages(Locale locale)
39      {
40          this.locale = locale;
41      }
42  
43      /**
44       * Invoked to provide the value for a particular key. This may be invoked multiple times even for the same key. The
45       * implementation should <em>ignore the case of the key</em>.
46       *
47       * @param key the key to obtain a value for (case insensitive)
48       * @return the value for the key, or null if this instance can not provide the value
49       */
50      protected abstract String valueForKey(String key);
51  
52  
53      @Override
54      public boolean contains(String key)
55      {
56          return valueForKey(key) != null;
57      }
58  
59      @Override
60      public String get(String key)
61      {
62          if (contains(key)) return valueForKey(key);
63  
64          return String.format("[[missing key: %s]]", key);
65      }
66  
67      @Override
68      public MessageFormatter getFormatter(String key)
69      {
70          MessageFormatter result = cache.get(key);
71  
72          if (result == null)
73          {
74              result = buildMessageFormatter(key);
75              cache.put(key, result);
76          }
77  
78          return result;
79      }
80  
81      private MessageFormatter buildMessageFormatter(String key)
82      {
83          String format = get(key);
84  
85          return new MessageFormatterImpl(format, locale);
86      }
87  
88      @Override
89      public String format(String key, Object... args)
90      {
91          return getFormatter(key).format(args);
92      }
93  
94  }
