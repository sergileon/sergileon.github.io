15  package org.apache.tapestry5.internal.services;
16  
17  import org.apache.tapestry5.internal.InternalConstants;
18  import org.apache.tapestry5.json.JSONArray;
19  import org.apache.tapestry5.json.JSONObject;
20  import org.apache.tapestry5.services.javascript.InitializationPriority;
21  import org.apache.tapestry5.services.javascript.ModuleConfigurationCallback;
22  import org.apache.tapestry5.services.javascript.StylesheetLink;
23  
24  import java.util.List;
25  
26  public class PartialMarkupDocumentLinker implements DocumentLinker
27  {
28      private final JSONArray libraryURLs = new JSONArray();
29  
30      private final JSONArray stylesheets = new JSONArray();
31  
32      private final ModuleInitsManager initsManager = new ModuleInitsManager();
33  
34      public void addCoreLibrary(String libraryURL)
35      {
36          notImplemented("addCoreLibrary");
37      }
38  
39      public void addLibrary(String libraryURL)
40      {
41          libraryURLs.put(libraryURL);
42      }
43  
44      public void addStylesheetLink(StylesheetLink stylesheet)
45      {
46          JSONObject object = new JSONObject(
47  
48                  "href", stylesheet.getURL(),
49  
50                  "media", stylesheet.getOptions().media);
51  
52          stylesheets.put(object);
53      }
54  
55      private void notImplemented(String methodName)
56      {
57          throw new UnsupportedOperationException(String.format("DocumentLinker.%s() is not implemented for partial page renders.", methodName));
58      }
59  
60      public void addScript(InitializationPriority priority, String script)
61      {
62          notImplemented("addScript");
63      }
64      
65      public void addModuleConfigurationCallback(ModuleConfigurationCallback callback)
66      {
67          notImplemented("moduleConfigurationCallback");
68      }
69  
70      public void addInitialization(InitializationPriority priority, String moduleName, String functionName, JSONArray arguments)
71      {
72          initsManager.addInitialization(priority, moduleName, functionName, arguments);
73      }
74  
75      /**
76       * Commits changes, adding one or more keys to the reply.
77       *
78       * @param reply
79       *         JSON Object to be sent to client
80       */
81      public void commit(JSONObject reply)
82      {
83          if (libraryURLs.length() > 0)
84          {
85              reply.in(InternalConstants.PARTIAL_KEY).put("libraries", libraryURLs);
86          }
87  
88          if (stylesheets.length() > 0)
89          {
90              reply.in(InternalConstants.PARTIAL_KEY).put("stylesheets", stylesheets);
91          }
92  
93          List<?> inits = initsManager.getSortedInits();
94  
95          if (inits.size() > 0)
96          {
97              reply.in(InternalConstants.PARTIAL_KEY).put("inits", JSONArray.from(inits));
98          }
99      }
100 }
