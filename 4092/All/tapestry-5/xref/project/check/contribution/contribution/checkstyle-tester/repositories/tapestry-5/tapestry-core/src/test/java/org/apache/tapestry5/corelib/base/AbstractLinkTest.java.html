15  package org.apache.tapestry5.corelib.base;
16  
17  import org.apache.tapestry5.ComponentResources;
18  import org.apache.tapestry5.Link;
19  import org.apache.tapestry5.MarkupWriter;
20  import org.apache.tapestry5.dom.Element;
21  import org.apache.tapestry5.internal.services.MarkupWriterImpl;
22  import org.apache.tapestry5.internal.test.InternalBaseTestCase;
23  import org.testng.annotations.Test;
24  
25  public class AbstractLinkTest extends InternalBaseTestCase
26  {
27      private final static String LINK_URI = "/foo/bar.baz";
28  
29      private final AbstractLink linkFixture = new AbstractLink()
30      {
31      };
32  
33      @Test
34      public void no_anchor()
35      {
36          Link link = mockLink();
37          ComponentResources resources = mockComponentResources();
38          MarkupWriter writer = new MarkupWriterImpl();
39  
40          train_toURI(link, LINK_URI);
41          //note that we aren't trying to test the parameters code here, so we only worry about the single, simplest branch.
42          //The parameters code is fully tested in CoreBehaviorsTests.
43          expect(resources.isBound("parameters")).andReturn(false);
44  
45          resources.renderInformalParameters(writer);
46  
47          replay();
48  
49          linkFixture.inject(null, resources);
50  
51          linkFixture.writeLink(writer, link);
52  
53          verify();
54  
55          Element e = writer.getElement();
56  
57          writer.write("link text");
58          writer.end();
59  
60          assertEquals(writer.toString(), "<a href=\"/foo/bar.baz\">link text</a>");
61      }
62  
63      @Test
64      public void with_anchor()
65      {
66          Link link = mockLink();
67          ComponentResources resources = mockComponentResources();
68          MarkupWriter writer = new MarkupWriterImpl();
69  
70          train_toURI(link, LINK_URI);
71  
72          //see note in no_anchor.
73          expect(resources.isBound("parameters")).andReturn(false);
74  
75          resources.renderInformalParameters(writer);
76  
77          replay();
78  
79          linkFixture.inject("wilma", resources);
80  
81          linkFixture.writeLink(writer, link);
82  
83          verify();
84  
85          writer.write("link text");
86          writer.end();
87  
88          assertEquals(writer.toString(), "<a href=\"/foo/bar.baz#wilma\">link text</a>");
89      }
90  }
