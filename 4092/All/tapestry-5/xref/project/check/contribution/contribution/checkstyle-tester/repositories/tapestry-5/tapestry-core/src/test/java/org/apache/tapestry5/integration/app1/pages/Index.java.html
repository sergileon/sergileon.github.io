37      public static class Item implements Comparable<Item>
38      {
39          public final String pageName;
40          public final String label;
41          public final String description;
42          public final Object[] context;
43  
44          public Item(String pageName, String label, String description, Object... context)
45          {
46              this.pageName = pageName;
47              this.label = label;
48              this.description = description;
49              this.context = context;

52          public int compareTo(Item o)
53          {
54              return label.compareTo(o.label);
55          }
