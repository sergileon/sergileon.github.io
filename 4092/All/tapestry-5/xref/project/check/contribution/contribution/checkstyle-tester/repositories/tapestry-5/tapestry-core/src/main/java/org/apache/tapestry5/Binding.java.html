13  package org.apache.tapestry5;
14  
15  import org.apache.tapestry5.ioc.AnnotationProvider;
16  
17  /**
18   * A binding is a connection between a component and its container (another component), that allows the embedded
19   * component to gain access to <em>resources</em> defined by the container. Resources can represent any kind of value
20   * that can be obtained from the parent component, but is often a JavaBean property that can be read and updated.
21   * Different implementations of Binding are used to access different kinds of resources of the container.
22   *
23   * A binding ultimately must provide access to the underlying annotations. In most cases, there are no annotations, but
24   * bindings that ultimate invoke methods or read and update fields must provide access to those annotations.
25   */
26  public interface Binding extends AnnotationProvider
27  {
28      /**
29       * Reads the current value of the property (or other resource). When reading properties of objects that are
30       * primitive types, this will return an instance of the wrapper type. In some cases, a binding is read only and this
31       * method will throw a runtime exception.
32       */
33      Object get();
34  
35      /**
36       * Updates the current value. Most types of bindings are read-only, and this method will throw a runtime exception.
37       * It is the caller's responsibility to ensure that the value passed in is of the appropriate type.
38       *
39       * @param value
40       */
41      void set(Object value);
42  
43      /**
44       * Returns true if the value of the binding does not ever change. Components will often cache such values
45       * aggressively.
46       */
47      boolean isInvariant();
48  
49      /**
50       * Returns the type of the binding, either the type of resource exposed by the binding, or the type of the property
51       * bound.
52       */
53      Class getBindingType();
54  }
