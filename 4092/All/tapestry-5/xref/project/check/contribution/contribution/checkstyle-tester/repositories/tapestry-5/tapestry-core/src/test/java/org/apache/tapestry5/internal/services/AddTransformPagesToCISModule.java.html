1   package org.apache.tapestry5.internal.services;
2   
3   import org.apache.tapestry5.ioc.Configuration;
4   import org.apache.tapestry5.ioc.MappedConfiguration;
5   import org.apache.tapestry5.ioc.annotations.Contribute;
6   import org.apache.tapestry5.services.ComponentClassResolver;
7   import org.apache.tapestry5.services.LibraryMapping;
8   import org.apache.tapestry5.services.transform.ControlledPackageType;
9   
10  public class AddTransformPagesToCISModule
11  {
12      @Contribute(ComponentInstantiatorSource.class)
13      public static void setupPackageForReload(MappedConfiguration<String, ControlledPackageType> configuration)
14      {
15          configuration.add("org.apache.tapestry5.internal.transform.pages", ControlledPackageType.COMPONENT);
16      }
17  
18      /**
19       * There used to be a t5internal module used for some tests and such, but that was refactored out in 5.4. Some
20       * tests rely on it though.
21       */
22      @Contribute(ComponentClassResolver.class)
23      public static void setupT5InternalLibrary(Configuration<LibraryMapping> configuration)
24      {
25          configuration.add(new LibraryMapping("t5internal", "org.apache.tapestry5.internal.t5internal"));
26      }
27  
28  }
