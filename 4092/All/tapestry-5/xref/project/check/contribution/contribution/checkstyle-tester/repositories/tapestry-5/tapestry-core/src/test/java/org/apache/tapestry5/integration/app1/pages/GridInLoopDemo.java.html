1   package org.apache.tapestry5.integration.app1.pages;
2   
3   import org.apache.tapestry5.beaneditor.BeanModel;
4   import org.apache.tapestry5.integration.app1.data.Track;
5   import org.apache.tapestry5.ioc.Messages;
6   import org.apache.tapestry5.ioc.annotations.Inject;
7   import org.apache.tapestry5.services.BeanModelSource;
8   
9   import java.util.Collection;
10  
11  public class GridInLoopDemo extends InplaceGridInLoopDemo {
12      @Inject
13      private BeanModelSource beanModelSource;
14  
15      @Inject
16      private Messages messages;
17  
18      public BeanModel<Track> getModel() {
19          BeanModel<Track> model = beanModelSource.createDisplayModel(Track.class,messages);
20          //For every loop iteration after the first remove one more column
21          int toRemove = Math.min(index, model.getPropertyNames().size()) - 1;
22          for(int i = 0;i<toRemove;i++) {
23              model.exclude(model.getPropertyNames().get(0));
24          }
25          //Disable sorting for all columns except rating.The reason for that is because all
26          //grids share the same GridPaginationModel but not the same BeanModel.This might result to
27          //a sort of a column that is included in ones grid model but not in another.The only column
28          //that is included in all models is the rating column
29          for(String propertyName:(Collection<String>)model.getPropertyNames()) {
30              if(!propertyName.equals("rating")) {
31                  model.get(propertyName).sortable(false);
32              }
33          }
34          return model;
35      }
