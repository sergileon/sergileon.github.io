15  package org.apache.tapestry5.internal.transform;
16  
17  import org.apache.tapestry5.ioc.Invokable;
18  import org.apache.tapestry5.ioc.OperationTracker;
19  import org.apache.tapestry5.runtime.ComponentEvent;
20  
21  /**
22   * Used  to encapsulate the list of {@link EventHandlerMethodParameterProvider}s for a particular
23   * method of a particular component, providing {@link OperationTracker} behavior as parameter values
24   * are obtained/computed/coerced.
25   *
26   * @since 5.3
27   */
28  public class EventHandlerMethodParameterSource
29  {
30      private static final class ParameterExtractor implements Invokable<Object> {
31          private final EventHandlerMethodParameterProvider[] providers;
32          private final int index;
33          private final ComponentEvent event;
34  
35          private ParameterExtractor(EventHandlerMethodParameterProvider[] providers, int index, ComponentEvent event) {
36              this.providers = providers;
37              this.index = index;
38              this.event = event;
39          }
40  
41          public Object invoke()
42          {
43              return providers[index].valueForEventHandlerMethodParameter(event);
44          }
45      }
46  
47      private final String methodIdentifier;
48  
49      private final OperationTracker operationTracker;
50  
51      private final EventHandlerMethodParameterProvider[] providers;
52  
53      public EventHandlerMethodParameterSource(String methodIdentifier, OperationTracker operationTracker, EventHandlerMethodParameterProvider[] providers)
54      {
55  
56          this.methodIdentifier = methodIdentifier;
57          this.operationTracker = operationTracker;
58          this.providers = providers;
59      }
60  
61      public Object get(final ComponentEvent event, final int index)
62      {
63          // Hopefully this will not be too much overhead; it's really nice to be able to track what parameter
64          // caused a failure.
65  
66          return operationTracker.invoke("Obtaining value for parameter #" + (index + 1) + " of "+ methodIdentifier,
67                  new ParameterExtractor(providers, index, event));
68      }
69  }
