18  package org.apache.tapestry5.jcache.internal;
19  
20  // import org.jsr107.tck.testutil.AbstractTestExcluder;
21  import java.util.Iterator;
22  import java.util.ServiceLoader;
23  
24  import javax.cache.annotation.BeanProvider;
25  
26  import org.junit.Rule;
27  import org.junit.rules.MethodRule;
28  
29  /**
30   * Base class that ALL annotation/interceptor tests MUST extend from
31   * 
32   * @author Eric Dalquist
33   * @version $Revision$
34   */
35  @SuppressWarnings("deprecation")
36  public class AbstractInterceptionTest
37  {
38      private static final BeanProvider beanProvider;
39  
40      static
41      {
42          BeanProvider localBeanProvider = null;
43          try
44          {
45              final ServiceLoader<BeanProvider> serviceLoader = ServiceLoader
46                      .load(BeanProvider.class);
47              final Iterator<BeanProvider> it = serviceLoader.iterator();
48              localBeanProvider = it.hasNext() ? it.next() : null;
49          }
50          catch (Throwable t)
51          {
52              // ignore
53              System.err
54                      .println("Failed to load BeanProvider SPI impl, annotation tests will be ignored");
55              t.printStackTrace(System.err);
56          }
57  
58          beanProvider = localBeanProvider;
59      }
60  
61      /**
62       * Rule used to exclude tests that do not implement Annotations
63       */
64      @Rule
65      public final MethodRule rule = new AbstractTestExcluder()
66      {
67          @Override
68          protected boolean isExcluded(String methodName)
69          {
70              // Exclude all tests if annotations are not supported or no beanProvider has been set
71              return beanProvider == null;
72          }
73      };
74  
75      /**
76       * Loads a specified bean by type, used to retrieve an annotated bean to test from the
77       * underlying implementation
78       * 
79       * @param beanClass
80       *            The type to load
81       * @return The loaded bean
82       */
83      protected final <T> T getBeanByType(Class<T> beanClass)
84      {
85          if (beanProvider == null) { throw new IllegalStateException(
86                  "No tests should be run if beanProvider is null"); }
87  
88          return beanProvider.getBeanByType(beanClass);
89      }
90  }
