14  package org.apache.tapestry5.jcache.internal;
15  
16  import org.junit.rules.MethodRule;
17  import org.junit.runners.model.FrameworkMethod;
18  import org.junit.runners.model.Statement;
19  
20  import java.util.logging.Logger;
21  
22  /**
23   * For the TCK we need to have an exclude list of bad tests so that disabling tests
24   * can be done without changing code.
25   *
26   * This class creates a rule for the class provided
27   *
28   * The exclude list is created by {@link ExcludeList} by creating a file in the root of your
29   * classpath called "ExcludeList". There is an example in the testRI module for testing the RI.
30   * 
31   * @author Yannis Cosmadopoulos
32   * @since 1.0
33   */
34  @SuppressWarnings("deprecation")
35  public abstract class AbstractTestExcluder implements MethodRule
36  {
37      private final Logger logger = Logger.getLogger(getClass().getName());
38  
39      /**
40       * {@inheritDoc}
41       */
42      public Statement apply(Statement statement, FrameworkMethod frameworkMethod, Object o)
43      {
44          final String methodName = frameworkMethod.getName();
45          final String className = frameworkMethod.getMethod().getDeclaringClass().getName();
46          if (isExcluded(methodName))
47          {
48              return new ExcludedStatement(className, methodName, logger);
49          }
50          else
51          {
52              return statement;
53          }
54      }
55  
56      protected abstract boolean isExcluded(String methodName);
57  
58      protected Logger getLogger()
59      {
60          return logger;
61      }
62  
63      /**
64       * Statement for excluded methods
65       */
66      private static final class ExcludedStatement extends Statement
67      {
68          private final String methodName;
69          private final String className;
70          private final Logger logger;
71  
72          private ExcludedStatement(String className, String methodName, Logger logger)
73          {
74              this.className = className;
75              this.methodName = methodName;
76              this.logger = logger;
77          }
78  
79          @Override
80          public void evaluate() throws Throwable
81          {
82              logger.info("===== EXCLUDING TEST '" + className + "'\t'" + methodName + "'.");
83          }
84      }
85  }
