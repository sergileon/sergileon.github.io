13  package org.apache.tapestry5.ioc.services;
14  
15  import org.apache.tapestry5.ioc.AnnotationProvider;
16  import org.apache.tapestry5.ioc.ObjectLocator;
17  import org.apache.tapestry5.ioc.ObjectProvider;
18  import org.apache.tapestry5.ioc.annotations.UsesOrderedConfiguration;
19  
20  /**
21   * A service that acts as a chain-of-command over a number of {@link org.apache.tapestry5.ioc.ObjectProvider}, but
22   * allows for the case where no object may be provided.
23   *
24   * This service is itself a key part of Tapestry's general injection mechanism; it is used when instantiating a service
25   * implementation instance, invoking module methods (service builder, decorator, or contribution methods), when
26   * {@linkplain ObjectLocator#autobuild(Class) autobuilding} objects of any type.
27   *
28   * As of Tapestry 5.3, the MasterObjectProvider allows injection of {@link org.apache.tapestry5.ioc.OperationTracker} as a
29   * special case (not based on a contributed ObjectProvider).
30   */
31  @UsesOrderedConfiguration(ObjectProvider.class)
32  public interface MasterObjectProvider
33  {
34      /**
35       * Provides an object based on an expression. The process of providing objects occurs within a particular
36       * <em>context</em>, which will typically be a service builder method, service contributor method, or service
37       * decorator method. The locator parameter provides access to the services visible <em>to that context</em>.
38       *
39       * When the value is required and no {@link ObjectProvider} provided a non-null value, then {@link
40       * ObjectLocator#getService(Class, Class[])}  is invoked (with no marker annotations),
41       * to provide a uniquely matching service, or throw a failure exception
42       * if no <em>single</em> service can be found.
43       *
44       * @param objectType         the expected object type
45       * @param annotationProvider provides access to annotations (typically, the field or parameter to which an
46       *                           injection-related annotation is attached); annotations on the field or parameter may
47       *                           also be used when resolving the desired object
48       * @param locator            locator for the <em>context</em> in which the provider is being used
49       * @param required           if true (normal case) a value must be provided; if false then it is allowed for no
50       *                           ObjectProvider to provide a value, and this method may return null to indicate the
51       *                           failure
52       * @param <T>
53       * @return the requested object, or null if this object provider can not supply an object
54       * @throws RuntimeException if the expression can not be evaluated, or the type of object identified is not
55       *                          assignable to the type specified by the objectType parameter
56       */
57      <T> T provide(Class<T> objectType, AnnotationProvider annotationProvider, ObjectLocator locator, boolean required);
58  }
