8   /**
9    * Provides the "t5internal" library, that provides common text utilities needed by some of the other
10   * projects' integration tests. To normalize in-IDE development with command-line development, this module
11   * is not set up to auto load via a manifest attribute, instead it is referenced via {@link org.apache.tapestry5.ioc.annotations.ImportModule}.
12   *
13   * @since 5.4
14   */
15  public class InternalTestModule
16  {
17      @Contribute(ComponentClassResolver.class)
18      public static void provideT5InternalLibrary(Configuration<LibraryMapping> configuration)
19      {
20          configuration.add(new LibraryMapping("t5internal", "org.apache.tapestry5.internal.t5internal"));
21      }

23  }
