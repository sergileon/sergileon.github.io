60      @Test
61      public void add_new_parameter()
62      {
63          Resource r = mockResource();
64          Logger logger = mockLogger();
65  
66          replay();
67  
68          MutableComponentModel model = new MutableComponentModelImpl(CLASS_NAME, logger, r, null, false, null);
69  
70          assertTrue(model.getParameterNames().isEmpty());
71  
72          String parameterName = "value";
73  
74          model.addParameter(parameterName, true, true, BindingConstants.PROP);
75  
76          ParameterModel pm = model.getParameterModel(parameterName);
77  
78          assertEquals(pm.getName(), parameterName);
79          assertEquals(true, pm.isRequired());
80          assertEquals(pm.getDefaultBindingPrefix(), BindingConstants.PROP);
81  
82          assertEquals(model.getDeclaredParameterNames(), Arrays.asList(parameterName));
83  
84          // Verify that the binding prefix is actually stored:
85  
86          model.addParameter("fred", true, true, "flint");
87  
88          // Checks that parameter names are case insensitive
89  
90          assertEquals(model.getParameterModel("Fred").getDefaultBindingPrefix(), "flint");
91  
92          verify();
93      }

215     @Test
216     public void add_embedded()
217     {
218         Resource r = mockResource();
219         Logger logger = mockLogger();
220         Location l = mockLocation();
221 
222         replay();
223 
224         MutableComponentModel model = new MutableComponentModelImpl(CLASS_NAME, logger, r, null, false, null);
225 
226         assertTrue(model.getEmbeddedComponentIds().isEmpty());
227 
228         MutableEmbeddedComponentModel fred = model.addEmbeddedComponent("fred", "Fred", COMPONENT_CLASS_NAME, false, l);
229 
230         assertEquals(fred.getId(), "fred");
231         assertEquals(fred.getComponentType(), "Fred");
232         assertFalse(fred.getInheritInformalParameters());
233         assertSame(fred.getLocation(), l);
234 
235         MutableEmbeddedComponentModel barney = model.addEmbeddedComponent("barney", "Barney", COMPONENT_CLASS_NAME,
236                 false, null);
237 
238         assertEquals(model.getEmbeddedComponentIds(), Arrays.asList("barney", "fred"));
239 
240         assertSame(model.getEmbeddedComponentModel("fred"), fred);
241         assertSame(model.getEmbeddedComponentModel("barney"), barney);
242 
243         // Access by id is case insensitive
244 
245         assertSame(model.getEmbeddedComponentModel("FRED"), fred);
246         assertSame(model.getEmbeddedComponentModel("BARNEY"), barney);
247 
248         assertEquals(fred.toString(),
249                 "EmbeddedComponentModel[id=fred type=Fred class=org.example.components.Fred inheritInformals=false]");
250 
251         verify();
252     }
