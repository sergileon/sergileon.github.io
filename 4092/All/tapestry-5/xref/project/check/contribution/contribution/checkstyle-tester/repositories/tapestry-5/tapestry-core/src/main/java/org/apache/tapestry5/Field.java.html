13  package org.apache.tapestry5;
14  
15  /**
16   * Defines a field within a form.  Fields have a <a href="http://www.w3.org/TR/html4/interact/forms.html#control-name">control
17   * name</a> that is used when rendering and, later, when the form is submitted, to identify the query parameter.
18   *
19   * Timing is important, as components may render multiple times, due to looping and other factors. Generally, a
20   * component's {@link #getControlName()} will only be accurate after it has rendered.  In some cases, when generating
21   * JavaScript for example, it is necessary to {@linkplain org.apache.tapestry5.services.Heartbeat#defer(Runnable) wait
22   * until the end of the current Heartbeat} to ensure that all components have had their chance to render.
23   */
24  public interface Field extends ClientElement
25  {
26      /**
27       * Returns the value used as the name attribute of the rendered element. This value will be unique within an
28       * enclosing form, even if the same component renders multiple times.
29       *
30       * @see org.apache.tapestry5.services.FormSupport#allocateControlName(String)
31       */
32      String getControlName();
33  
34      /**
35       * Returns a user presentable (localized) label for the field, which may be used inside &lt;label&gt; elements on
36       * the client, and inside client or server-side validation error messages.
37       *
38       * @return the label
39       * @see org.apache.tapestry5.corelib.components.Label
40       */
41      String getLabel();
42  
43      /**
44       * Returns true if the field is disabled; A disabled field will render a disabled attribute so that it is
45       * non-responsive on the client (at least, until its disabled status is changed on the client using JavaScript). A
46       * disabled field will ignore any value passed up in a form submit request. Care must be taken if the disabled
47       * status of a field can change between the time the field is rendered and the time the enclosing form is
48       * submitted.
49       */
50      boolean isDisabled();
51  
52      /**
53       * Returns true if this field required (as per {@link org.apache.tapestry5.FieldValidator#isRequired()}).
54       *
55       * @return true if a non-blank value is required for the field
56       */
57      boolean isRequired();
58  }
