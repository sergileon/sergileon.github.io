15  package org.apache.tapestry5.internal.services;
16  
17  import org.apache.tapestry5.ioc.services.PropertyAdapter;
18  import org.apache.tapestry5.ioc.util.StrategyRegistry;
19  import org.apache.tapestry5.services.DataTypeAnalyzer;
20  import org.apache.tapestry5.services.InvalidationListener;
21  
22  import java.util.Map;
23  
24  /**
25   * The default data type analyzer, which is based entirely on the type of the property (and not on annotations or naming
26   * conventions). This is based on a configuration of property type class to string provided as an IoC service
27   * configuration.
28   */
29  public class DefaultDataTypeAnalyzer implements DataTypeAnalyzer, Runnable
30  {
31      private final StrategyRegistry<String> registry;
32  
33      public DefaultDataTypeAnalyzer(Map<Class, String> configuration)
34      {
35          registry = StrategyRegistry.newInstance(String.class, configuration);
36      }
37  
38      /**
39       * Clears the registry on an invalidation event (this is because the registry caches results, and the keys are
40       * classes that may be component classes from the invalidated component class loader).
41       */
42      public void run()
43      {
44          registry.clearCache();
45      }
46  
47      public String identifyDataType(PropertyAdapter adapter)
48      {
49          Class propertyType = adapter.getType();
50  
51          String dataType = registry.get(propertyType);
52  
53          // To avoid "no strategy" exceptions, we expect a contribution of Object.class to the empty
54          // string. We convert that back to a null.
55  
56          if (dataType.equals(""))
57              return null;
58  
59          return dataType;
60      }
61  }
