14  package org.example.testapp.pages;
15  
16  import java.util.ArrayList;
17  import java.util.Collection;
18  import java.util.Date;
19  
20  import javax.validation.constraints.NotNull;
21  import javax.validation.constraints.Pattern;
22  import javax.validation.constraints.Size;
23  
24  import org.apache.tapestry5.annotations.Import;
25  import org.apache.tapestry5.annotations.Persist;
26  import org.apache.tapestry5.annotations.Property;
27  import org.apache.tapestry5.beaneditor.Validate;
28  import org.apache.tapestry5.internal.services.StringValueEncoder;
29  import org.example.testapp.services.Foo;
30  
31  @Import(stack = "core")
32  public class FormClientValidationDemo
33  {
34  	@NotNull(groups=Foo.class)
35  	@Pattern(regexp="[a-zA-Z]+")
36  	@Validate("minlength=5")
37  	@Property
38  	@Persist
39  	private String userName;
40  	
41  	@NotNull
42  	@Property
43  	@Persist
44  	private String password;
45  	
46  	@NotNull
47  	@Size(min=2, max=3)
48  	@Property
49  	@Persist
50  	private Collection<String> languages;
51  	
52  	@NotNull
53  	@Property
54  	@Persist
55  	private String color; 
56  	
57  	@NotNull
58  	@Property
59  	@Persist
60  	private Date date; 
61  
62      void onPrepare() {
63          if (languages == null) {
64              languages = new ArrayList<String>();
65          }
66      }
67  
68  	public StringValueEncoder getStringValueEncoder()
69  	{
70  		return new StringValueEncoder();
71  	}
72  
73  }
