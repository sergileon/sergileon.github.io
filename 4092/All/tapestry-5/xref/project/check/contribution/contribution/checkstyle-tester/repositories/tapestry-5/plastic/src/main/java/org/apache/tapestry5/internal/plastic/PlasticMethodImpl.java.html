39      public PlasticMethodImpl(PlasticClassImpl plasticClass, MethodNode node)
40      {

114     @Override
115     public MethodHandle getHandle()
116     {
117         plasticClass.check();
118 
119         if (handle == null)
120         {
121             methodIndex = plasticClass.nextMethodIndex++;
122             handle = new MethodHandleImpl(plasticClass.className, description.toString(), methodIndex);
123             plasticClass.shimMethods.add(this);
124         }
125 
126         return handle;
127     }

129     @Override
130     public PlasticMethod changeImplementation(InstructionBuilderCallback callback)
131     {
132         plasticClass.check();
133 
134         // If the method is currently abstract, clear that flag.
135         if (Modifier.isAbstract(node.access))
136         {
137             node.access = node.access & ~org.apache.tapestry5.internal.plastic.asm.Opcodes.ACC_ABSTRACT;
138             description = description.withModifiers(node.access);
139         }
140 
141         node.instructions.clear();
142 
143         plasticClass.newBuilder(description, node).doCallback(callback);
144 
145         // With the implementation changed, it is necessary to intercept field reads/writes.
146         // The node may not already have been in the fieldTransformMethods Set if it was
147         // an introduced method.
148 
149         plasticClass.fieldTransformMethods.add(node);
150 
151         return this;
152     }
153 
154     @Override
155     public PlasticMethod addAdvice(MethodAdvice advice)
156     {
157         plasticClass.check();
158 
159         assert advice != null;
160 
161         if (adviceManager == null)
162         {
163             adviceManager = new MethodAdviceManager(plasticClass, description, node);
164             plasticClass.advisedMethods.add(this);
165         }
166 
167         adviceManager.add(advice);
168 
169         return this;
170     }

172     @Override
173     public PlasticMethod delegateTo(final PlasticField field)
174     {
175         plasticClass.check();
176 
177         assert field != null;
178         assert field.getPlasticClass() == plasticClass;
179 
180         // TODO: Better handling error case where delegating to a primitive or object array.
181 
182         // TODO: Is there a easy way to ensure that the type has the necessary method? I don't
183         // like that errors along those lines may be deferred until execution time.
184 
185         changeImplementation(new InstructionBuilderCallback()
186         {
187             @Override
188             public void doBuild(InstructionBuilder builder)
189             {
190                 // Load the field
191 
192                 builder.loadThis().getField(field);
193                 builder.loadArguments();
194 
195                 invokeDelegateAndReturnResult(builder, field.getTypeName());
196             }
197         });
198 
199         return this;
200     }
201 
202     @Override
203     public PlasticMethod delegateTo(PlasticMethod delegateProvidingMethod)
204     {
205         plasticClass.check();
206 
207         assert delegateProvidingMethod != null;
208         assert delegateProvidingMethod.getPlasticClass() == plasticClass;
209 
210         // TODO: ensure not primitive/array type
211         final MethodDescription providerDescriptor = delegateProvidingMethod.getDescription();
212         final String delegateType = providerDescriptor.returnType;
213 
214         if (delegateType.equals("void") || providerDescriptor.argumentTypes.length > 0)
215             throw new IllegalArgumentException(
216                     String.format(
217                             "Method %s is not usable as a delegate provider; it must be a void method that takes no arguments.",
218                             delegateProvidingMethod));
219 
220         changeImplementation(new InstructionBuilderCallback()
221         {
222             @Override
223             public void doBuild(InstructionBuilder builder)
224             {
225                 // Load the field
226 
227                 builder.loadThis();
228 
229                 if (Modifier.isPrivate(providerDescriptor.modifiers))
230                 {
231                     builder.invokeSpecial(plasticClass.className, providerDescriptor);
232                 } else
233                 {
234                     builder.invokeVirtual(plasticClass.className, delegateType, providerDescriptor.methodName);
235                 }
236 
237                 builder.loadArguments();
238 
239                 invokeDelegateAndReturnResult(builder, delegateType);
240             }
241         });
242 
243         return this;
244     }
245 
246     @Override
247     public List<MethodParameter> getParameters()
248     {
249         if (parameters == null)
250         {
251             parameters = PlasticInternalUtils.newList();
252 
253             for (int i = 0; i < description.argumentTypes.length; i++)
254             {
255 
256                 parameters.add(new MethodParameterImpl(plasticClass,
257                         PlasticClassImpl.safeArrayDeref(node.visibleParameterAnnotations, i),
258                         PlasticClassImpl.safeArrayDeref(description.argumentTypes, i), i));
259             }
260         }
261 
262         return parameters;
263     }

270     private boolean isPrivate()
271     {
272         return Modifier.isPrivate(node.access);
273     }

275     /**
276      * If a MethodHandle has been requested and the method itself is private, then create a non-private
277      * access method. Returns the access method name (which is different from the method name itself only
278      * for private methods, where an access method is created).
279      */
280     private String setupMethodHandleAccess()
281     {
282         if (isPrivate())
283             return createAccessMethod();
284         else
285             return node.name;
286     }
287 
288     private String createAccessMethod()
289     {
290         String name = String.format("%s$access%s", node.name, PlasticUtils.nextUID());
291 
292         // Kind of awkward that exceptions are specified as String[] when what we have handy is List<String>
293         MethodNode mn = new MethodNode(Opcodes.ACC_SYNTHETIC | Opcodes.ACC_FINAL, name, node.desc, node.signature, null);
294         // But it is safe enough for the two nodes to share
295         mn.exceptions = node.exceptions;
296 
297         InstructionBuilder builder = plasticClass.newBuilder(mn);
298 
299         builder.loadThis();
300         builder.loadArguments();
301         builder.invokeSpecial(plasticClass.className, description);
302         builder.returnResult();
303 
304         plasticClass.addMethod(mn);
305 
306         return name;
307     }

314     /**
315      * The properly cast target instance will be on the stack.
316      */
317     void extendShimInvoke(SwitchBlock block)
318     {
319         final String accessMethodName = setupMethodHandleAccess();
320 
321         block.addCase(methodIndex, false, new InstructionBuilderCallback()
322         {
323             @Override
324             public void doBuild(InstructionBuilder builder)
325             {
326                 builder.startTryCatch(new TryCatchCallback()
327                 {
328                     @Override
329                     public void doBlock(TryCatchBlock block)
330                     {
331                         block.addTry(new InstructionBuilderCallback()
332                         {
333                             @Override
334                             public void doBuild(InstructionBuilder builder)
335                             {
336                                 // The third argument is an Object array; get each
337                                 for (int i = 0; i < description.argumentTypes.length; i++)
338                                 {
339                                     String argumentType = description.argumentTypes[i];
340 
341                                     builder.loadArgument(2);
342                                     builder.loadArrayElement(i, Object.class.getName());
343                                     builder.castOrUnbox(argumentType);
344                                 }
345 
346                                 builder.invokeVirtual(plasticClass.className, description.returnType, accessMethodName,
347                                         description.argumentTypes);
348 
349                                 // TODO: hate see "void" just there.
350 
351                                 if (description.returnType.equals("void"))
352                                     builder.loadNull();
353                                 else
354                                     builder.boxPrimitive(description.returnType);
355 
356                                 builder.newInstance(SuccessMethodInvocationResult.class).dupe(1).swap();
357                                 builder.invokeConstructor(SuccessMethodInvocationResult.class, Object.class);
358                                 builder.returnResult();
359                             }
360                         });
361 
362                         for (String exceptionType : description.checkedExceptionTypes)
363                         {
364                             block.addCatch(exceptionType, new InstructionBuilderCallback()
365                             {
366                                 @Override
367                                 public void doBuild(InstructionBuilder builder)
368                                 {
369                                     builder.newInstance(FailureMethodInvocationResult.class).dupe(1).swap();
370                                     builder.invokeConstructor(FailureMethodInvocationResult.class, Throwable.class);
371                                     builder.returnResult();
372                                 }
373                             });
374                         }
375                     }
376                 });
377             }
378         });
379     }

381     private void invokeDelegateAndReturnResult(InstructionBuilder builder, String delegateType)
382     {
383         // The trick is that you have to be careful to use the right opcode based on the field type
384         // (interface vs. ordinary object).
385 
386         final TypeCategory typeCategory = plasticClass.pool.getTypeCategory(delegateType);
387 
388         if (typeCategory == TypeCategory.INTERFACE)
389             builder.invokeInterface(delegateType, description.returnType, description.methodName,
390                     description.argumentTypes);
391         else
392             builder.invokeVirtual(delegateType, description.returnType, description.methodName,
393                     description.argumentTypes);
394 
395         builder.returnResult();
396     }
