13  package org.apache.tapestry5.plastic;
14  
15  /**
16   * Represents a field of a class being {@linkplain PlasticClass transformed}.
17   *
18   * No methods of this object should be invoked after the class transformation is completed.
19   */
20  public interface PlasticField extends AnnotationAccess
21  {
22      /** Returns the class containing this field. */
23      PlasticClass getPlasticClass();
24  
25      /**
26       * Returns a handle that can be used to directly access a private field of a
27       * transformed class instance.
28       */
29      FieldHandle getHandle();
30  
31      /**
32       * Returns the name of the field.
33       */
34      String getName();
35  
36      /**
37       * Returns the fully qualified class name for the field's type or (for a primitive type)
38       * the primitive type name ("int", "char", etc.). For array types, the returned name includes
39       * a "[]" suffix.
40       */
41      String getTypeName();
42  
43      /**
44       * Claims the field, used to indicate that the field is "processed". A field may only
45       * be claimed once. Claiming a field is intended as a mechanism to detect or prevent
46       * conflicts between different isolated transformations of the field. The tag value used does not matter, and is
47       * typically either an annotation (that drove the transformation) or the instance of {@link PlasticClassTransformer}
48       * that performed the transformation. That tag value is only used when generating the error message for the case
49       * where a field is claimed for than once.
50       * 
51       * @throws RuntimeException
52       *             if the field is claimed a second time
53       * @throws AssertionError
54       *             if tag is null
55       * @see PlasticClass#getUnclaimedFields()
56       * @return the field for further manipulation
57       */
58      PlasticField claim(Object tag);
59  
60      /**
61       * Returns true if the field has already been {@linkplain #claim(Object) claimed}.
62       * 
63       * @see PlasticClass#getUnclaimedFields()
64       */
65      boolean isClaimed();
66  
67      /**
68       * Converts the field to be read-only, and provide the indicated value. The field's value will be
69       * set inside the class' constructor.
70       * 
71       * @param value
72       *            to inject, which must be type compatible with the field (possibly, a wrapper type if the field is
73       *            a primitive value). The value may not be null.
74       * @return the field for further manipulation
75       * @throws IllegalStateException
76       *             if the field already has an injection, or the field has a conduit
77       */
78      PlasticField inject(Object value);
79  
80      /**
81       * Converts the field to be read-only, and provide the value, which is computed
82       * indirectly inside the class' constructor.
83       * 
84       * @param computedValue
85       *            provides the actual value to be injected, and must return a value type compatible
86       *            with the field (possibly a wrapper type if the field is a primitive value). The computedValue may not
87       *            be null.
88       * @return the field for further manipulation
89       * @throws IllegalStateException
90       *             if the field already has an injection, or the field has a conduit
91       */
92      PlasticField injectComputed(ComputedValue<?> computedValue);
93  
94      /**
95       * As with {@link #inject(Object)}, but the value is extracted from the {@link InstanceContext}.
96       * 
97       * @return this field for further manipulation
98       */
99      PlasticField injectFromInstanceContext();
100 
101     /**
102      * Intercepts all access to the field, replacing such access with calls on the conduit. Even access via
103      * the FieldHandle will instead delegate to the conduit. Once a conduit is provided, it is not possible
104      * to inject a value into the field.
105      *
106      * Normally, once a conduit is in place, the field will never be actually read or written. This is problematic for
107      * debugging, so {@link TransformationOption#FIELD_WRITEBEHIND} is useful when operating in a non-production mode.
108      *
109      * @return the field for further manipulation
110      * @throws IllegalStateException
111      *             if the field already has an injection or a conduit
112      * @return this field for further manipulation
113      */
114     <F> PlasticField setConduit(FieldConduit<F> conduit);
115 
116     /**
117      * Sets the conduit for the field to a value computed when the class is instantiated
118      * 
119      * @param computedConduit
120      *            object that will compute the actual conduit to be used
121      * @return this field for further manipulation
122      */
123    <F>  PlasticField setComputedConduit(ComputedValue<FieldConduit<F>> computedConduit);
124 
125     /**
126      * Creates access to the field, using the default property name derived from the name of the field.
127      * The default property name is the same as the name of the field, but with any leading or trailing underscore
128      * characters removed (a common convention among some programmers). Also, strips leading "m_" from the field name
129      * (another common convention).
130      * 
131      * @param accessType
132      *            which methods to create
133      * @return the field for further manipulation
134      * @throws IllegalArgumentException
135      *             if an accessor method to be created already exists (possibly inherited from a base class)
136      */
137     PlasticField createAccessors(PropertyAccessType accessType);
138 
139     /**
140      * Creates accessors, possibly replacing existing methods (or overriding methods from a super class).
141      * The method names consist of the property name, with its first character converted to upper-case, prefixed
142      * with "get" or "set". The accessor methods must not already exist.
143      * 
144      * @param accessType
145      *            which methods to create
146      * @param propertyName
147      *            the name of the property (from which the names of the methods are generated)
148      * @return the field for further manipulation
149      * @throws IllegalArgumentException
150      *             if an accessor method to be created already exists (possibly inherited from a base class)
151      */
152     PlasticField createAccessors(PropertyAccessType accessType, String propertyName);
153 
154     /** Returns the field's fully qualified generic type, or null if not defined. */
155     String getGenericSignature();
156 
157     /** Returns the modifiers on the field. */
158     int getModifiers();
159 }
