15  package org.apache.tapestry5.internal.plastic;
16  
17  import java.util.HashMap;
18  import java.util.Map;
19  
20  import org.apache.tapestry5.internal.plastic.asm.Opcodes;
21  
22  /**
23   * Collects together information needed to write code that involves primitive types, including
24   * moving between wrapper types and primitive values, or extracting a primitive value from
25   * the {@link StaticContext}.
26   */
27  @SuppressWarnings("rawtypes")
28  public enum PrimitiveType implements Opcodes
29  {
30      VOID("void", "V", void.class, Void.class, null, null, ILOAD, ISTORE, RETURN),
31  
32      BOOLEAN("boolean", "Z", boolean.class, Boolean.class, "booleanValue", "getBoolean", ILOAD, ISTORE, IRETURN),
33  
34      CHAR("char", "C", char.class, Character.class, "charValue", "getChar", ILOAD, ISTORE, IRETURN),
35  
36      BYTE("byte", "B", byte.class, Byte.class, "byteValue", "getByte", ILOAD, ISTORE, IRETURN),
37  
38      SHORT("short", "S", short.class, Short.class, "shortValue", "getShort", ILOAD, ISTORE, IRETURN),
39  
40      INT("int", "I", int.class, Integer.class, "intValue", "getInt", ILOAD, ISTORE, IRETURN),
41  
42      FLOAT("float", "F", float.class, Float.class, "floatValue", "getFloat", FLOAD, FSTORE, FRETURN),
43  
44      LONG("long", "J", long.class, Long.class, "longValue", "getLong", LLOAD, LSTORE, LRETURN),
45  
46      DOUBLE("double", "D", double.class, Double.class, "doubleValue", "getDouble", DLOAD, DSTORE, DRETURN);
47  
48      /**
49       * @param name
50       *            the Java source name for the type
51       * @param descriptor
52       *            Java descriptor for the type ('Z', 'I', etc.)
53       * @param primitiveType
54       *            TODO
55       * @param wrapperType
56       *            wrapper type, e.g., java.lang.Integer
57       * @param toValueMethodName
58       *            name of method of wrapper class to extract primitive value
59       * @param getFromStaticContextMethodName
60       *            name of method of {@link StaticContext} used to extract primitive context value
61       * @param loadOpcode
62       *            Correct opcode for loading an argument or local variable onto the stack (ILOAD, LLOAD, FLOAD or
63       *            DLOAD)
64       * @param storeOpcode
65       *            matching opcode for storing a value to a local variable (ISTORE, LSTORE, FSTORE or DSTORE)
66       * @param returnOpcode
67       *            Correct opcode for returning the top value on the stack (IRETURN, LRETURN, FRETURN
68       *            or DRETURN)
69       */
70      private PrimitiveType(String name, String descriptor, Class primitiveType, Class wrapperType,
71              String toValueMethodName, String getFromStaticContextMethodName, int loadOpcode, int storeOpcode,
72              int returnOpcode)
73      {
74          this.name = name;
75          this.descriptor = descriptor;
76          this.primitiveType = primitiveType;
77          this.wrapperType = wrapperType;
78          this.wrapperInternalName = wrapperType == null ? null : PlasticInternalUtils.toInternalName(wrapperType
79                  .getName());
80          this.toValueMethodName = toValueMethodName;
81          this.getFromStaticContextMethodName = getFromStaticContextMethodName;
82          this.loadOpcode = loadOpcode;
83          this.storeOpcode = storeOpcode;
84          this.returnOpcode = returnOpcode;
85  
86          this.valueOfMethodDescriptor = String.format("(%s)L%s;", descriptor, wrapperInternalName);
87          this.toValueMethodDescriptor = "()" + descriptor;
88          this.getFromStaticContextMethodDescriptor = "(I)" + descriptor;
89      }
90  
91      public final String name, descriptor, wrapperInternalName, valueOfMethodDescriptor, toValueMethodName,
92              getFromStaticContextMethodName, toValueMethodDescriptor, getFromStaticContextMethodDescriptor;
93  
94      public final Class primitiveType, wrapperType;
95  
96      public final int loadOpcode, storeOpcode, returnOpcode;
97  
98      private static final Map<String, PrimitiveType> BY_NAME = new HashMap<String, PrimitiveType>();
99      private static final Map<String, PrimitiveType> BY_DESC = new HashMap<String, PrimitiveType>();
100     private static final Map<Class, PrimitiveType> BY_PRIMITIVE_TYPE = new HashMap<Class, PrimitiveType>();
101 
102     static
103     {
104         for (PrimitiveType type : values())
105         {
106             BY_NAME.put(type.name, type);
107             BY_DESC.put(type.descriptor, type);
108             BY_PRIMITIVE_TYPE.put(type.primitiveType, type);
109         }
110     }
111 
112     public boolean isWide()
113     {
114         return this == LONG || this == DOUBLE;
115     }
116 
117     /**
118      * Returns the primitive type matching the given type name or null for a non-primitive type (an array type,
119      * or an class name).
120      * 
121      * @param name
122      *            possible primitive name
123      * @return the type or null
124      */
125     public static PrimitiveType getByName(String name)
126     {
127         return BY_NAME.get(name);
128     }
129 
130     public static PrimitiveType getByPrimitiveType(Class primitiveType)
131     {
132         return BY_PRIMITIVE_TYPE.get(primitiveType);
133     }
