15  package org.apache.tapestry5.internal.hibernate;
16  
17  import org.apache.tapestry5.hibernate.HibernateSessionManager;
18  import org.apache.tapestry5.hibernate.annotations.CommitAfter;
19  import org.apache.tapestry5.model.MutableComponentModel;
20  import org.apache.tapestry5.plastic.MethodAdvice;
21  import org.apache.tapestry5.plastic.MethodInvocation;
22  import org.apache.tapestry5.plastic.PlasticClass;
23  import org.apache.tapestry5.plastic.PlasticMethod;
24  import org.apache.tapestry5.services.transform.ComponentClassTransformWorker2;
25  import org.apache.tapestry5.services.transform.TransformationSupport;
26  
27  /**
28   * Searches for methods that have the {@link org.apache.tapestry5.hibernate.annotations.CommitAfter} annotation and adds
29   * logic around the method to commit or abort the transaction. The commit/abort logic is the same as for the
30   * {@link org.apache.tapestry5.hibernate.HibernateTransactionDecorator} service.
31   */
32  public class CommitAfterWorker implements ComponentClassTransformWorker2
33  {
34      private final HibernateSessionManager manager;
35  
36      private final MethodAdvice advice = new MethodAdvice()
37      {
38          private void abort()
39          {
40              try
41              {
42                  manager.abort();
43              } catch (Exception e)
44              {
45                  // Ignore.
46              }
47          }
48  
49          @Override
50          public void advise(MethodInvocation invocation)
51          {
52              try
53              {
54                  invocation.proceed();
55  
56                  // Success or checked exception:
57  
58                  manager.commit();
59              } catch (RuntimeException ex)
60              {
61                  abort();
62  
63                  throw ex;
64              }
65          }
66      };
67  
68      public CommitAfterWorker(HibernateSessionManager manager)
69      {
70          this.manager = manager;
71      }
72  
73      @Override
74      public void transform(PlasticClass plasticClass, TransformationSupport support, MutableComponentModel model)
75      {
76          for (PlasticMethod method : plasticClass.getMethodsWithAnnotation(CommitAfter.class))
77          {
78              method.addAdvice(advice);
79          }
80      }
81  }
