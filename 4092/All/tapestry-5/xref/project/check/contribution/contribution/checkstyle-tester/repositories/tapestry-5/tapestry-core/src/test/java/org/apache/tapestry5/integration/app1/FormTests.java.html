252     @Test
253     public void datefield_select_newmonth_samedate()
254     {
255         openLinks("DateField Demo", "Reset Page State", "english");
256 
257         //start with a known date...
258         type("asteroidImpact", "05/28/2035");
259 
260         click("css=.x-impact .btn");
261 
262         waitForCSSSelectedElementToAppear("div.datePicker");
263         assertEquals(getText("css=td.selected"), "28");
264 
265         //move to the next month.
266         click("css=button.nextButton");
267 
268         //first, make sure that NOTHING shows as selected! The selected date is still 5/28/46
269 
270         waitForSelectedToBeRemoved();
271 
272 
273         //make sure it's still selected if we navigate back...
274         click("css=button.previousButton");
275         waitForCSSSelectedElementToAppear("td.selected");
276 
277         click("css=button.nextButton");
278 
279         waitForSelectedToBeRemoved();
280 
281         click("xpath=//td[text()='28']");
282         String pickerGoneSelector = "css=div.datePicker";
283         waitForInvisible(pickerGoneSelector);
284 
285         assertFieldValue("asteroidImpact", "6/28/2035");
286 
287         //a few other behaviors to check on as a side-effect of implementing the fix for 1409:
288         //1) If today is selected and it's the current month, pressing the "Today" button should close the popup
289         //2) If today is selected and we're on some other month, pressing the "Today" button should just take us
290         //   back to the today.
291         //3) If today is not selected, pressing the "Today" button should set the date and close the popup.
292         //4) Pressing the "None" button should always close the popup and result in no date.
293 
294         //#3
295         click("css=.x-impact .btn");
296 
297         waitForCSSSelectedElementToAppear("div.datePicker");
298         click("css=div.datePicker .footerTable button");
299         waitForInvisible(pickerGoneSelector);
300 
301         String value = getValue("asteroidImpact");
302         assertFieldValue("asteroidImpact", new SimpleDateFormat("M/d/yyyy").format(new Date()));
303 
304         //#2...
305         click("css=.x-impact .btn");
306 
307         waitForCSSSelectedElementToAppear("div.datePicker");
308         click("css=button.nextButton");
309 
310         waitForSelectedToBeRemoved();
311 
312         click("css=div.datePicker .footerTable button");
313         waitForCSSSelectedElementToAppear("td.selected");
314 
315         //#1
316         click("css=div.datePicker .footerTable button");
317         waitForInvisible(pickerGoneSelector);
318         assertFieldValue("asteroidImpact", value);
319 
320         //#4...
321 
322         click("css=.x-impact .btn");
323 
324         waitForCSSSelectedElementToAppear("div.datePicker");
325 
326         String noneButton = "//button[text()='None']";
327 
328         click(noneButton);
329 
330         waitForInvisible(pickerGoneSelector);
331 
332         assertFieldValue("asteroidImpact", "");
333 
334         click(noneButton);
335 
336         waitForCSSSelectedElementToAppear("div.datePicker");
337         assertFalse(isElementPresent("css=td.selected"));
338 
339         click(noneButton);
340 
341         waitForInvisible(pickerGoneSelector);
342         assertFieldValue("asteroidImpact", "");
343     }
