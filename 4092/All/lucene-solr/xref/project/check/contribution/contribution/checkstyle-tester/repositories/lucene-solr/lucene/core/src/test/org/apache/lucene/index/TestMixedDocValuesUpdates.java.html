45    public void testManyReopensAndFields() throws Exception {
46      Directory dir = newDirectory();
47      final Random random = random();
48      IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random));
49      LogMergePolicy lmp = newLogMergePolicy();
50      lmp.setMergeFactor(3); // merge often
51      conf.setMergePolicy(lmp);
52      IndexWriter writer = new IndexWriter(dir, conf);
53  
54      final boolean isNRT = random.nextBoolean();
55      DirectoryReader reader;
56      if (isNRT) {
57        reader = DirectoryReader.open(writer, true);
58      } else {
59        writer.commit();
60        reader = DirectoryReader.open(dir);
61      }
62      
63      final int numFields = random.nextInt(4) + 3; // 3-7
64      final int numNDVFields = random.nextInt(numFields/2) + 1; // 1-3
65      final long[] fieldValues = new long[numFields];
66      for (int i = 0; i < fieldValues.length; i++) {
67        fieldValues[i] = 1;
68      }
69      
70      int numRounds = atLeast(15);
71      int docID = 0;
72      for (int i = 0; i < numRounds; i++) {
73        int numDocs = atLeast(5);
74        // System.out.println("TEST: round=" + i + ", numDocs=" + numDocs);
75        for (int j = 0; j < numDocs; j++) {
76          Document doc = new Document();
77          doc.add(new StringField("id", "doc-" + docID, Store.NO));
78          doc.add(new StringField("key", "all", Store.NO)); // update key
79          // add all fields with their current value
80          for (int f = 0; f < fieldValues.length; f++) {
81            if (f < numNDVFields) {
82              doc.add(new NumericDocValuesField("f" + f, fieldValues[f]));
83            } else {
84              doc.add(new BinaryDocValuesField("f" + f, TestBinaryDocValuesUpdates.toBytes(fieldValues[f])));
85            }
86          }
87          writer.addDocument(doc);
88          ++docID;
89        }
90        
91        int fieldIdx = random.nextInt(fieldValues.length);
92        String updateField = "f" + fieldIdx;
93        if (fieldIdx < numNDVFields) {
94          writer.updateNumericDocValue(new Term("key", "all"), updateField, ++fieldValues[fieldIdx]);
95        } else {
96          writer.updateBinaryDocValue(new Term("key", "all"), updateField, TestBinaryDocValuesUpdates.toBytes(++fieldValues[fieldIdx]));
97        }
98        //System.out.println("TEST: updated field '" + updateField + "' to value " + fieldValues[fieldIdx]);
99  
100       if (random.nextDouble() < 0.2) {
101         int deleteDoc = random.nextInt(docID); // might also delete an already deleted document, ok!
102         writer.deleteDocuments(new Term("id", "doc-" + deleteDoc));
103 //        System.out.println("[" + Thread.currentThread().getName() + "]: deleted document: doc-" + deleteDoc);
104       }
105       
106       // verify reader
107       if (!isNRT) {
108         writer.commit();
109       }
110       
111 //      System.out.println("[" + Thread.currentThread().getName() + "]: reopen reader: " + reader);
112       DirectoryReader newReader = DirectoryReader.openIfChanged(reader);
113       assertNotNull(newReader);
114       reader.close();
115       reader = newReader;
116 //      System.out.println("[" + Thread.currentThread().getName() + "]: reopened reader: " + reader);
117       assertTrue(reader.numDocs() > 0); // we delete at most one document per round
118       for (LeafReaderContext context : reader.leaves()) {
119         LeafReader r = context.reader();
120 //        System.out.println(((SegmentReader) r).getSegmentName());
121         Bits liveDocs = r.getLiveDocs();
122         for (int field = 0; field < fieldValues.length; field++) {
123           String f = "f" + field;
124           BinaryDocValues bdv = r.getBinaryDocValues(f);
125           NumericDocValues ndv = r.getNumericDocValues(f);
126           Bits docsWithField = r.getDocsWithField(f);
127           if (field < numNDVFields) {
128             assertNotNull(ndv);
129             assertNull(bdv);
130           } else {
131             assertNull(ndv);
132             assertNotNull(bdv);
133           }
134           int maxDoc = r.maxDoc();
135           for (int doc = 0; doc < maxDoc; doc++) {
136             if (liveDocs == null || liveDocs.get(doc)) {
137 //              System.out.println("doc=" + (doc + context.docBase) + " f='" + f + "' vslue=" + getValue(bdv, doc, scratch));
138               assertTrue(docsWithField.get(doc));
139               if (field < numNDVFields) {
140                 assertEquals("invalid numeric value for doc=" + doc + ", field=" + f + ", reader=" + r, fieldValues[field], ndv.get(doc));
141               } else {
142                 assertEquals("invalid binary value for doc=" + doc + ", field=" + f + ", reader=" + r, fieldValues[field], TestBinaryDocValuesUpdates.getValue(bdv, doc));
143               }
144             }
145           }
146         }
147       }
148 //      System.out.println();
149     }
150     
151     writer.close();
152     IOUtils.close(reader, dir);
153   }

155   public void testStressMultiThreading() throws Exception {
156     final Directory dir = newDirectory();
157     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
158     final IndexWriter writer = new IndexWriter(dir, conf);
159     
160     // create index
161     final int numFields = TestUtil.nextInt(random(), 2, 4);
162     final int numThreads = TestUtil.nextInt(random(), 3, 6);
163     final int numDocs = atLeast(2000);
164     for (int i = 0; i < numDocs; i++) {
165       Document doc = new Document();
166       doc.add(new StringField("id", "doc" + i, Store.NO));
167       double group = random().nextDouble();
168       String g;
169       if (group < 0.1) g = "g0";
170       else if (group < 0.5) g = "g1";
171       else if (group < 0.8) g = "g2";
172       else g = "g3";
173       doc.add(new StringField("updKey", g, Store.NO));
174       for (int j = 0; j < numFields; j++) {
175         long value = random().nextInt();
176         doc.add(new BinaryDocValuesField("f" + j, TestBinaryDocValuesUpdates.toBytes(value)));
177         doc.add(new NumericDocValuesField("cf" + j, value * 2)); // control, always updated to f * 2
178       }
179       writer.addDocument(doc);
180     }
181     
182     final CountDownLatch done = new CountDownLatch(numThreads);
183     final AtomicInteger numUpdates = new AtomicInteger(atLeast(100));
184     
185     // same thread updates a field as well as reopens
186     Thread[] threads = new Thread[numThreads];
187     for (int i = 0; i < threads.length; i++) {
188       threads[i] = new Thread("UpdateThread-" + i) {
189         @Override
190         public void run() {
191           DirectoryReader reader = null;
192           boolean success = false;
193           try {
194             Random random = random();
195             while (numUpdates.getAndDecrement() > 0) {
196               double group = random.nextDouble();
197               Term t;
198               if (group < 0.1) t = new Term("updKey", "g0");
199               else if (group < 0.5) t = new Term("updKey", "g1");
200               else if (group < 0.8) t = new Term("updKey", "g2");
201               else t = new Term("updKey", "g3");
202 //              System.out.println("[" + Thread.currentThread().getName() + "] numUpdates=" + numUpdates + " updateTerm=" + t);
203               int field = random().nextInt(numFields);
204               final String f = "f" + field;
205               final String cf = "cf" + field;
206               long updValue = random.nextInt();
207 //              System.err.println("[" + Thread.currentThread().getName() + "] t=" + t + ", f=" + f + ", updValue=" + updValue);
208               writer.updateDocValues(t, new BinaryDocValuesField(f, TestBinaryDocValuesUpdates.toBytes(updValue)),
209                   new NumericDocValuesField(cf, updValue*2));
210               
211               if (random.nextDouble() < 0.2) {
212                 // delete a random document
213                 int doc = random.nextInt(numDocs);
214 //                System.out.println("[" + Thread.currentThread().getName() + "] deleteDoc=doc" + doc);
215                 writer.deleteDocuments(new Term("id", "doc" + doc));
216               }
217   
218               if (random.nextDouble() < 0.05) { // commit every 20 updates on average
219 //                  System.out.println("[" + Thread.currentThread().getName() + "] commit");
220                 writer.commit();
221               }
222               
223               if (random.nextDouble() < 0.1) { // reopen NRT reader (apply updates), on average once every 10 updates
224                 if (reader == null) {
225 //                  System.out.println("[" + Thread.currentThread().getName() + "] open NRT");
226                   reader = DirectoryReader.open(writer, true);
227                 } else {
228 //                  System.out.println("[" + Thread.currentThread().getName() + "] reopen NRT");
229                   DirectoryReader r2 = DirectoryReader.openIfChanged(reader, writer, true);
230                   if (r2 != null) {
231                     reader.close();
232                     reader = r2;
233                   }
234                 }
235               }
236             }
237 //            System.out.println("[" + Thread.currentThread().getName() + "] DONE");
238             success = true;
239           } catch (IOException e) {
240             throw new RuntimeException(e);
241           } finally {
242             if (reader != null) {
243               try {
244                 reader.close();
245               } catch (IOException e) {
246                 if (success) { // suppress this exception only if there was another exception
247                   throw new RuntimeException(e);
248                 }
249               }
250             }
251             done.countDown();
252           }
253         }
254       };
255     }
256     
257     for (Thread t : threads) t.start();
258     done.await();
259     writer.close();
260     
261     DirectoryReader reader = DirectoryReader.open(dir);
262     BytesRef scratch = new BytesRef();
263     for (LeafReaderContext context : reader.leaves()) {
264       LeafReader r = context.reader();
265       for (int i = 0; i < numFields; i++) {
266         BinaryDocValues bdv = r.getBinaryDocValues("f" + i);
267         NumericDocValues control = r.getNumericDocValues("cf" + i);
268         Bits docsWithBdv = r.getDocsWithField("f" + i);
269         Bits docsWithControl = r.getDocsWithField("cf" + i);
270         Bits liveDocs = r.getLiveDocs();
271         for (int j = 0; j < r.maxDoc(); j++) {
272           if (liveDocs == null || liveDocs.get(j)) {
273             assertTrue(docsWithBdv.get(j));
274             assertTrue(docsWithControl.get(j));
275             long ctrlValue = control.get(j);
276             long bdvValue = TestBinaryDocValuesUpdates.getValue(bdv, j) * 2;
277 //              if (ctrlValue != bdvValue) {
278 //                System.out.println("seg=" + r + ", f=f" + i + ", doc=" + j + ", group=" + r.document(j).get("updKey") + ", ctrlValue=" + ctrlValue + ", bdvBytes=" + scratch);
279 //              }
280             assertEquals(ctrlValue, bdvValue);
281           }
282         }
283       }
284     }
285     reader.close();
286     
287     dir.close();
288   }

290   public void testUpdateDifferentDocsInDifferentGens() throws Exception {
291     // update same document multiple times across generations
292     Directory dir = newDirectory();
293     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
294     conf.setMaxBufferedDocs(4);
295     IndexWriter writer = new IndexWriter(dir, conf);
296     final int numDocs = atLeast(10);
297     for (int i = 0; i < numDocs; i++) {
298       Document doc = new Document();
299       doc.add(new StringField("id", "doc" + i, Store.NO));
300       long value = random().nextInt();
301       doc.add(new BinaryDocValuesField("f", TestBinaryDocValuesUpdates.toBytes(value)));
302       doc.add(new NumericDocValuesField("cf", value * 2));
303       writer.addDocument(doc);
304     }
305     
306     int numGens = atLeast(5);
307     for (int i = 0; i < numGens; i++) {
308       int doc = random().nextInt(numDocs);
309       Term t = new Term("id", "doc" + doc);
310       long value = random().nextLong();
311       writer.updateDocValues(t, new BinaryDocValuesField("f", TestBinaryDocValuesUpdates.toBytes(value)),
312           new NumericDocValuesField("cf", value*2));
313       DirectoryReader reader = DirectoryReader.open(writer, true);
314       for (LeafReaderContext context : reader.leaves()) {
315         LeafReader r = context.reader();
316         BinaryDocValues fbdv = r.getBinaryDocValues("f");
317         NumericDocValues cfndv = r.getNumericDocValues("cf");
318         for (int j = 0; j < r.maxDoc(); j++) {
319           assertEquals(cfndv.get(j), TestBinaryDocValuesUpdates.getValue(fbdv, j) * 2);
320         }
321       }
322       reader.close();
323     }
324     writer.close();
325     dir.close();
326   }

328   @Nightly
329   public void testTonsOfUpdates() throws Exception {
330     // LUCENE-5248: make sure that when there are many updates, we don't use too much RAM
331     Directory dir = newDirectory();
332     final Random random = random();
333     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random));
334     conf.setRAMBufferSizeMB(IndexWriterConfig.DEFAULT_RAM_BUFFER_SIZE_MB);
335     conf.setMaxBufferedDocs(IndexWriterConfig.DISABLE_AUTO_FLUSH); // don't flush by doc
336     IndexWriter writer = new IndexWriter(dir, conf);
337     
338     // test data: lots of documents (few 10Ks) and lots of update terms (few hundreds)
339     final int numDocs = atLeast(20000);
340     final int numBinaryFields = atLeast(5);
341     final int numTerms = TestUtil.nextInt(random, 10, 100); // terms should affect many docs
342     Set<String> updateTerms = new HashSet<>();
343     while (updateTerms.size() < numTerms) {
344       updateTerms.add(TestUtil.randomSimpleString(random));
345     }
346 
347 //    System.out.println("numDocs=" + numDocs + " numBinaryFields=" + numBinaryFields + " numTerms=" + numTerms);
348     
349     // build a large index with many BDV fields and update terms
350     for (int i = 0; i < numDocs; i++) {
351       Document doc = new Document();
352       int numUpdateTerms = TestUtil.nextInt(random, 1, numTerms / 10);
353       for (int j = 0; j < numUpdateTerms; j++) {
354         doc.add(new StringField("upd", RandomPicks.randomFrom(random, updateTerms), Store.NO));
355       }
356       for (int j = 0; j < numBinaryFields; j++) {
357         long val = random.nextInt();
358         doc.add(new BinaryDocValuesField("f" + j, TestBinaryDocValuesUpdates.toBytes(val)));
359         doc.add(new NumericDocValuesField("cf" + j, val * 2));
360       }
361       writer.addDocument(doc);
362     }
363     
364     writer.commit(); // commit so there's something to apply to
365     
366     // set to flush every 2048 bytes (approximately every 12 updates), so we get
367     // many flushes during binary updates
368     writer.getConfig().setRAMBufferSizeMB(2048.0 / 1024 / 1024);
369     final int numUpdates = atLeast(100);
370 //    System.out.println("numUpdates=" + numUpdates);
371     for (int i = 0; i < numUpdates; i++) {
372       int field = random.nextInt(numBinaryFields);
373       Term updateTerm = new Term("upd", RandomPicks.randomFrom(random, updateTerms));
374       long value = random.nextInt();
375       writer.updateDocValues(updateTerm, new BinaryDocValuesField("f"+field, TestBinaryDocValuesUpdates.toBytes(value)),
376           new NumericDocValuesField("cf"+field, value*2));
377     }
378 
379     writer.close();
380     
381     DirectoryReader reader = DirectoryReader.open(dir);
382     for (LeafReaderContext context : reader.leaves()) {
383       for (int i = 0; i < numBinaryFields; i++) {
384         LeafReader r = context.reader();
385         BinaryDocValues f = r.getBinaryDocValues("f" + i);
386         NumericDocValues cf = r.getNumericDocValues("cf" + i);
387         for (int j = 0; j < r.maxDoc(); j++) {
388           assertEquals("reader=" + r + ", field=f" + i + ", doc=" + j, cf.get(j), TestBinaryDocValuesUpdates.getValue(f, j) * 2);
389         }
390       }
391     }
392     reader.close();
393     
394     dir.close();
395   }
