20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.apache.zookeeper.ZooDefs;
24  import org.apache.zookeeper.data.ACL;
25  import org.apache.zookeeper.data.Id;
26  
27  /**
28   * ZkACLProvider that gives all permissions for the user specified in System
29   * property "solr.authorization.superuser" (default: "solr") when using sasl,
30   * and gives read permissions for anyone else.  Designed for a setup where
31   * configurations have already been set up and will not be modified, or
32   * where configuration changes are controlled via Solr APIs.
33   */
34  public class SaslZkACLProvider extends DefaultZkACLProvider {
35  
36    private static String superUser = System.getProperty("solr.authorization.superuser", "solr");
37  
38    @Override
39    protected List<ACL> createGlobalACLsToAdd() {
40      List<ACL> result = new ArrayList<ACL>();
41      result.add(new ACL(ZooDefs.Perms.ALL, new Id("sasl", superUser)));
42      result.add(new ACL(ZooDefs.Perms.READ, ZooDefs.Ids.ANYONE_ID_UNSAFE));
43  
44      if (result.isEmpty()) {
45        result = super.createGlobalACLsToAdd();
46      }
47      return result;
48    }
49  }
