30   * A {@link FilterLeafReader} that can be used to apply
31   * additional checks for tests.

33  public class AssertingLeafReader extends FilterLeafReader {

35    private static void assertThread(String object, Thread creationThread) {
36      if (creationThread != Thread.currentThread()) {
37        throw new AssertionError(object + " are only supposed to be consumed in "
38            + "the thread in which they have been acquired. But was acquired in "
39            + creationThread + " and consumed in " + Thread.currentThread() + ".");
40      }
41    }

43    public AssertingLeafReader(LeafReader in) {
44      super(in);
45      // check some basic reader sanity
46      assert in.maxDoc() >= 0;
47      assert in.numDocs() <= in.maxDoc();
48      assert in.numDeletedDocs() + in.numDocs() == in.maxDoc();
49      assert !in.hasDeletions() || in.numDeletedDocs() > 0 && in.numDocs() < in.maxDoc();
50  
51      addCoreClosedListener(new CoreClosedListener() {
52        @Override
53        public void onClose(Object ownerCoreCacheKey) throws IOException {
54          final Object expectedKey = getCoreCacheKey();
55          assert expectedKey == ownerCoreCacheKey
56              : "Core closed listener called on a different key " + expectedKey + " <> " + ownerCoreCacheKey;
57        }
58      });
59    }

132   static final VirtualMethod<TermsEnum> SEEK_EXACT = new VirtualMethod<>(TermsEnum.class, "seekExact", BytesRef.class);

134   static class AssertingTermsEnum extends FilterTermsEnum {
135     private final Thread creationThread = Thread.currentThread();
136     private enum State {INITIAL, POSITIONED, UNPOSITIONED};
137     private State state = State.INITIAL;
138     private final boolean delegateOverridesSeekExact;
139 
140     public AssertingTermsEnum(TermsEnum in) {
141       super(in);
142       delegateOverridesSeekExact = SEEK_EXACT.isOverriddenAsOf(in.getClass());
143     }
144 
145     @Override
146     public PostingsEnum postings(PostingsEnum reuse, int flags) throws IOException {
147       assertThread("Terms enums", creationThread);
148       assert state == State.POSITIONED: "docs(...) called on unpositioned TermsEnum";
149 
150       // reuse if the codec reused
151       final PostingsEnum actualReuse;
152       if (reuse instanceof AssertingPostingsEnum) {
153         actualReuse = ((AssertingPostingsEnum) reuse).in;
154       } else {
155         actualReuse = null;
156       }
157       PostingsEnum docs = super.postings(actualReuse, flags);
158       if (docs == null) {
159         assert PostingsEnum.featureRequested(flags, DocsAndPositionsEnum.OLD_NULL_SEMANTICS);
160         return null;
161       }
162       if (docs == actualReuse) {
163         // codec reused, reset asserting state
164         ((AssertingPostingsEnum)reuse).reset();
165         return reuse;
166       } else {
167         return new AssertingPostingsEnum(docs);
168       }
169     }
170 
171     // TODO: we should separately track if we are 'at the end' ?
172     // someone should not call next() after it returns null!!!!
173     @Override
174     public BytesRef next() throws IOException {
175       assertThread("Terms enums", creationThread);
176       assert state == State.INITIAL || state == State.POSITIONED: "next() called on unpositioned TermsEnum";
177       BytesRef result = super.next();
178       if (result == null) {
179         state = State.UNPOSITIONED;
180       } else {
181         assert result.isValid();
182         state = State.POSITIONED;
183       }
184       return result;
185     }
186 
187     @Override
188     public long ord() throws IOException {
189       assertThread("Terms enums", creationThread);
190       assert state == State.POSITIONED : "ord() called on unpositioned TermsEnum";
191       return super.ord();
192     }
193 
194     @Override
195     public int docFreq() throws IOException {
196       assertThread("Terms enums", creationThread);
197       assert state == State.POSITIONED : "docFreq() called on unpositioned TermsEnum";
198       return super.docFreq();
199     }
200 
201     @Override
202     public long totalTermFreq() throws IOException {
203       assertThread("Terms enums", creationThread);
204       assert state == State.POSITIONED : "totalTermFreq() called on unpositioned TermsEnum";
205       return super.totalTermFreq();
206     }
207 
208     @Override
209     public BytesRef term() throws IOException {
210       assertThread("Terms enums", creationThread);
211       assert state == State.POSITIONED : "term() called on unpositioned TermsEnum";
212       BytesRef ret = super.term();
213       assert ret == null || ret.isValid();
214       return ret;
215     }
216 
217     @Override
218     public void seekExact(long ord) throws IOException {
219       assertThread("Terms enums", creationThread);
220       super.seekExact(ord);
221       state = State.POSITIONED;
222     }
223 
224     @Override
225     public SeekStatus seekCeil(BytesRef term) throws IOException {
226       assertThread("Terms enums", creationThread);
227       assert term.isValid();
228       SeekStatus result = super.seekCeil(term);
229       if (result == SeekStatus.END) {
230         state = State.UNPOSITIONED;
231       } else {
232         state = State.POSITIONED;
233       }
234       return result;
235     }
236 
237     @Override
238     public boolean seekExact(BytesRef text) throws IOException {
239       assertThread("Terms enums", creationThread);
240       assert text.isValid();
241       boolean result;
242       if (delegateOverridesSeekExact) {
243         result = in.seekExact(text);
244       } else {
245         result = super.seekExact(text);
246       }
247       if (result) {
248         state = State.POSITIONED;
249       } else {
250         state = State.UNPOSITIONED;
251       }
252       return result;
253     }
254 
255     @Override
256     public TermState termState() throws IOException {
257       assertThread("Terms enums", creationThread);
258       assert state == State.POSITIONED : "termState() called on unpositioned TermsEnum";
259       return in.termState();
260     }
261 
262     @Override
263     public void seekExact(BytesRef term, TermState state) throws IOException {
264       assertThread("Terms enums", creationThread);
265       assert term.isValid();
266       in.seekExact(term, state);
267       this.state = State.POSITIONED;
268     }
269 
270     @Override
271     public String toString() {
272       return "AssertingTermsEnum(" + in + ")";
273     }
274     
275     void reset() {
276       state = State.INITIAL;
277     }
278   }

280   static enum DocsEnumState { START, ITERATING, FINISHED };
281 
282   /** Wraps a docsenum with additional checks */
283   public static class AssertingPostingsEnum extends FilterPostingsEnum {
284     private final Thread creationThread = Thread.currentThread();
285     private DocsEnumState state = DocsEnumState.START;
286     int positionCount = 0;
287     int positionMax = 0;
288     private int doc;
289 
290     public AssertingPostingsEnum(PostingsEnum in) {
291       super(in);
292       this.doc = in.docID();
293     }
294 
295     @Override
296     public int nextDoc() throws IOException {
297       assertThread("Docs enums", creationThread);
298       assert state != DocsEnumState.FINISHED : "nextDoc() called after NO_MORE_DOCS";
299       int nextDoc = super.nextDoc();
300       assert nextDoc > doc : "backwards nextDoc from " + doc + " to " + nextDoc + " " + in;
301       if (nextDoc == DocIdSetIterator.NO_MORE_DOCS) {
302         state = DocsEnumState.FINISHED;
303         positionMax = 0;
304       } else {
305         state = DocsEnumState.ITERATING;
306         positionMax = super.freq();
307       }
308       positionCount = 0;
309       assert super.docID() == nextDoc;
310       return doc = nextDoc;
311     }
312 
313     @Override
314     public int advance(int target) throws IOException {
315       assertThread("Docs enums", creationThread);
316       assert state != DocsEnumState.FINISHED : "advance() called after NO_MORE_DOCS";
317       assert target > doc : "target must be > docID(), got " + target + " <= " + doc;
318       int advanced = super.advance(target);
319       assert advanced >= target : "backwards advance from: " + target + " to: " + advanced;
320       if (advanced == DocIdSetIterator.NO_MORE_DOCS) {
321         state = DocsEnumState.FINISHED;
322         positionMax = 0;
323       } else {
324         state = DocsEnumState.ITERATING;
325         positionMax = super.freq();
326       }
327       positionCount = 0;
328       assert super.docID() == advanced;
329       return doc = advanced;
330     }
331 
332     @Override
333     public int docID() {
334       assertThread("Docs enums", creationThread);
335       assert doc == super.docID() : " invalid docID() in " + in.getClass() + " " + super.docID() + " instead of " + doc;
336       return doc;
337     }
338 
339     @Override
340     public int freq() throws IOException {
341       assertThread("Docs enums", creationThread);
342       assert state != DocsEnumState.START : "freq() called before nextDoc()/advance()";
343       assert state != DocsEnumState.FINISHED : "freq() called after NO_MORE_DOCS";
344       int freq = super.freq();
345       assert freq > 0;
346       return freq;
347     }
348 
349     @Override
350     public int nextPosition() throws IOException {
351       assert state != DocsEnumState.START : "nextPosition() called before nextDoc()/advance()";
352       assert state != DocsEnumState.FINISHED : "nextPosition() called after NO_MORE_DOCS";
353       assert positionCount < positionMax : "nextPosition() called more than freq() times!";
354       int position = super.nextPosition();
355       assert position >= 0 || position == -1 : "invalid position: " + position;
356       positionCount++;
357       return position;
358     }
359 
360     @Override
361     public int startOffset() throws IOException {
362       assert state != DocsEnumState.START : "startOffset() called before nextDoc()/advance()";
363       assert state != DocsEnumState.FINISHED : "startOffset() called after NO_MORE_DOCS";
364       assert positionCount > 0 : "startOffset() called before nextPosition()!";
365       return super.startOffset();
366     }
367 
368     @Override
369     public int endOffset() throws IOException {
370       assert state != DocsEnumState.START : "endOffset() called before nextDoc()/advance()";
371       assert state != DocsEnumState.FINISHED : "endOffset() called after NO_MORE_DOCS";
372       assert positionCount > 0 : "endOffset() called before nextPosition()!";
373       return super.endOffset();
374     }
375 
376     @Override
377     public BytesRef getPayload() throws IOException {
378       assert state != DocsEnumState.START : "getPayload() called before nextDoc()/advance()";
379       assert state != DocsEnumState.FINISHED : "getPayload() called after NO_MORE_DOCS";
380       assert positionCount > 0 : "getPayload() called before nextPosition()!";
381       BytesRef payload = super.getPayload();
382       assert payload == null || payload.length > 0 : "getPayload() returned payload with invalid length!";
383       return payload;
384     }
385     
386     void reset() {
387       state = DocsEnumState.START;
388       doc = in.docID();
389       positionCount = positionMax = 0;
390     }
391   }

761   /** Wraps a Bits but with additional asserts */
762   public static class AssertingBits implements Bits {
763     private final Thread creationThread = Thread.currentThread();
764     final Bits in;
765     
766     public AssertingBits(Bits in) {
767       this.in = in;
768     }
769     
770     @Override
771     public boolean get(int index) {
772       assertThread("Bits", creationThread);
773       assert index >= 0 && index < length();
774       return in.get(index);
775     }
776 
777     @Override
778     public int length() {
779       assertThread("Bits", creationThread);
780       return in.length();
781     }

786     Bits liveDocs = super.getLiveDocs();
787     if (liveDocs != null) {
788       assert maxDoc() == liveDocs.length();
789       liveDocs = new AssertingBits(liveDocs);
790     } else {
791       assert maxDoc() == numDocs();
792       assert !hasDeletions();
