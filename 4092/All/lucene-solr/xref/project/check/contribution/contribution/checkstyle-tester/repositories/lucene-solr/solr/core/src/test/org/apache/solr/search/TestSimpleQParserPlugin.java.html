38      assertU(adoc("id", "44", "text0", "t0 t1 t1", "text1", "t6 t7 t8", "text-keyword0", "kw3 kw4 kw5"));
39      assertU(adoc("id", "45", "text0", "t0 t0 t1", "text1", "t9 t10 t11", "text-keyword0", "kw6 kw7 kw8"));
40      assertU(adoc("id", "46", "text0", "t1 t1 t1", "text1", "t12 t13 t14", "text-keyword0", "kw9 kw10 kw11"));
41      assertU(adoc("id", "47", "text0", "and", "text1", "+", "text-keyword0", "+"));
42      assertU(adoc("id", "48", "text0", "not", "text1", "-", "text-keyword0", "-"));
43      assertU(adoc("id", "49", "text0", "or", "text1", "|", "text-keyword0", "|"));
44      assertU(adoc("id", "50", "text0", "prefix", "text1", "t*", "text-keyword0", "kw*"));
45      assertU(adoc("id", "51", "text0", "phrase", "text1", "\"", "text-keyword0", "\""));
46      assertU(adoc("id", "52", "text0", "open", "text1", "(", "text-keyword0", "("));
47      assertU(adoc("id", "53", "text0", "close", "text1", ")", "text-keyword0", ")"));
48      assertU(adoc("id", "54", "text0", "escape", "text1", "\\", "text-keyword0", "\\"));
49      assertU(adoc("id", "55", "text0", "whitespace", "text1", "whitespace", "text-keyword0", " "));
50      assertU(adoc("id", "56", "text0", "whitespace", "text1", "whitespace", "text-keyword0", "\n"));
51      assertU(adoc("id", "57", "text0", "foobar", "text1", "foo bar", "text-keyword0", "fb"));
52      assertU(adoc("id", "58", "text-query0", "HELLO"));
53      assertU(adoc("id", "59", "text-query0", "hello"));
54      assertU(commit());
55    }

57    @Test
58    public void testQueryFields() throws Exception {
59      assertJQ(req("defType", "simple", "qf", "text0^2 text1 text-keyword0", "q", "t3"), "/response/numFound==1");
60      assertJQ(req("defType", "simple", "qf", "text0^3 text1^4 text-keyword0^0.55", "q", "t0"), "/response/numFound==4");
61      assertJQ(req("defType", "simple", "qf", "text-keyword0^9.2", "q", "\"kw9 kw10 kw11\""), "/response/numFound==1");
62      assertJQ(req("defType", "simple", "qf", "text-keyword0", "q", "kw9 kw10 kw11"), "/response/numFound==0");
63      assertJQ(req("defType", "simple", "qf", "text1 text-keyword0", "q", "kw9"), "/response/numFound==0");
64      assertJQ(req("defType", "simple", "qf", "text0", "q", "t2"), "/response/numFound==1");
65      assertJQ(req("defType", "simple", "qf", "text0^1.1 text1^0.9", "q", "t2 t9 t12"), "/response/numFound==4");
66    }

75      assertJQ(req("defType", "simple", "df", "text-keyword0", "q", "kw9 kw10 kw11"), "/response/numFound==0");
