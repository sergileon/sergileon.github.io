63    public void testAllInfoPresent() throws IOException {
64  
65      File testSrcRoot = new File(SolrTestCaseJ4.TEST_HOME());
66      FileUtils.copyFile(new File(testSrcRoot, "solr-50-all.xml"), new File(solrHome, "solr.xml"));
67  
68      NodeConfig cfg = SolrXmlConfig.fromSolrHome(loader, solrHome.toPath());
69      CloudConfig ccfg = cfg.getCloudConfig();
70      UpdateShardHandlerConfig ucfg = cfg.getUpdateShardHandlerConfig();
71      
72      assertEquals("core admin handler class", "testAdminHandler", cfg.getCoreAdminHandlerClass());
73      assertEquals("collection handler class", "testCollectionsHandler", cfg.getCollectionsHandlerClass());
74      assertEquals("info handler class", "testInfoHandler", cfg.getInfoHandlerClass());
75      assertEquals("config set handler class", "testConfigSetsHandler", cfg.getConfigSetsHandlerClass());
76      assertEquals("core load threads", 11, cfg.getCoreLoadThreadCount());
77      assertThat("core root dir", cfg.getCoreRootDirectory().toString(), containsString("testCoreRootDirectory"));
78      assertEquals("distrib conn timeout", 22, cfg.getDistributedConnectionTimeout());
79      assertEquals("distrib conn timeout", 22, cfg.getUpdateShardHandlerConfig().getDistributedConnectionTimeout());
80      assertEquals("distrib socket timeout", 33, cfg.getDistributedSocketTimeout());
81      assertEquals("distrib socket timeout", 33, cfg.getUpdateShardHandlerConfig().getDistributedSocketTimeout());
82      assertEquals("max update conn", 3, cfg.getMaxUpdateConnections());
83      assertEquals("max update conn", 3, cfg.getUpdateShardHandlerConfig().getMaxUpdateConnections());
84      assertEquals("max update conn/host", 37, cfg.getMaxUpdateConnectionsPerHost());
85      assertEquals("max update conn/host", 37, cfg.getUpdateShardHandlerConfig().getMaxUpdateConnectionsPerHost());
86      assertEquals("distrib conn timeout", 22, ucfg.getDistributedConnectionTimeout());
87      assertEquals("distrib socket timeout", 33, ucfg.getDistributedSocketTimeout());
88      assertEquals("max update conn", 3, ucfg.getMaxUpdateConnections());
89      assertEquals("max update conn/host", 37, ucfg.getMaxUpdateConnectionsPerHost());
90      assertEquals("host", "testHost", ccfg.getHost());
91      assertEquals("zk host context", "testHostContext", ccfg.getSolrHostContext());
92      assertEquals("solr host port", 44, ccfg.getSolrHostPort());
93      assertEquals("leader vote wait", 55, ccfg.getLeaderVoteWait());
94      assertEquals("logging class", "testLoggingClass", cfg.getLogWatcherConfig().getLoggingClass());
95      assertEquals("log watcher", true, cfg.getLogWatcherConfig().isEnabled());
96      assertEquals("log watcher size", 88, cfg.getLogWatcherConfig().getWatcherSize());
97      assertEquals("log watcher thresh", "99", cfg.getLogWatcherConfig().getWatcherThreshold());
98      assertEquals("manage path", "testManagementPath", cfg.getManagementPath());
99      assertEquals("shardLib", "testSharedLib", cfg.getSharedLibDirectory());
100     assertEquals("schema cache", true, cfg.hasSchemaCache());
101     assertEquals("trans cache size", 66, cfg.getTransientCacheSize());
102     assertEquals("zk client timeout", 77, ccfg.getZkClientTimeout());
103     assertEquals("zk host", "testZkHost", ccfg.getZkHost());
104     assertEquals("zk ACL provider", "DefaultZkACLProvider", ccfg.getZkACLProviderClass());
105     assertEquals("zk credentials provider", "DefaultZkCredentialsProvider", ccfg.getZkCredentialsProviderClass());
106   }

108   // Test  a few property substitutions that happen to be in solr-50-all.xml.
109   public void testPropertySub() throws IOException {
110 
111     System.setProperty("coreRootDirectory", "myCoreRoot" + File.separator);
112     System.setProperty("hostPort", "8888");
113     System.setProperty("shareSchema", "false");
114     System.setProperty("socketTimeout", "220");
115     System.setProperty("connTimeout", "200");
116 
117     File testSrcRoot = new File(SolrTestCaseJ4.TEST_HOME());
118     FileUtils.copyFile(new File(testSrcRoot, "solr-50-all.xml"), new File(solrHome, "solr.xml"));
119 
120     NodeConfig cfg = SolrXmlConfig.fromSolrHome(loader, solrHome.toPath());
121     assertThat(cfg.getCoreRootDirectory().toString(), containsString("myCoreRoot"));
122     assertEquals("solr host port", 8888, cfg.getCloudConfig().getSolrHostPort());
123     assertEquals("schema cache", false, cfg.hasSchemaCache());
124   }

126   public void testExplicitNullGivesDefaults() throws IOException {
127     String solrXml = "<solr>" +
128         "<solrcloud>" +
129         "<str name=\"host\">host</str>" +
130         "<int name=\"hostPort\">8983</int>" +
131         "<str name=\"hostContext\">solr</str>" +
132         "<null name=\"leaderVoteWait\"/>" +
133         "</solrcloud></solr>";
134 
135     NodeConfig cfg = SolrXmlConfig.fromString(loader, solrXml);
136     assertEquals("leaderVoteWait", 180000, cfg.getCloudConfig().getLeaderVoteWait());
137   }
