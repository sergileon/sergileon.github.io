96    // this should only be used from SolrCoreState
97    public RecoveryStrategy(CoreContainer cc, CoreDescriptor cd, RecoveryListener recoveryListener) {
98      this.cc = cc;
99      this.coreName = cd.getName();
100     this.recoveryListener = recoveryListener;
101     setName("RecoveryThread-"+this.coreName);
102     zkController = cc.getZkController();
103     zkStateReader = zkController.getZkStateReader();
104     baseUrl = zkController.getBaseUrl();
105     coreZkNodeName = cd.getCloudDescriptor().getCoreNodeName();
106   }

108   public void setRecoveringAfterStartup(boolean recoveringAfterStartup) {
109     this.recoveringAfterStartup = recoveringAfterStartup;
110   }

112   // make sure any threads stop retrying
113   @Override
114   public void close() {
115     close = true;
116     try {
117       prevSendPreRecoveryHttpUriRequest.abort();
118     } catch (NullPointerException e) {
119       // okay
120     }
121     log.warn("Stopping recovery for core={} coreNodeName={}", coreName, coreZkNodeName);
122   }

125   private void recoveryFailed(final SolrCore core,
126       final ZkController zkController, final String baseUrl,
127       final String shardZkNodeName, final CoreDescriptor cd) throws KeeperException, InterruptedException {
128     SolrException.log(log, "Recovery failed - I give up.");
129     try {
130       zkController.publish(cd, Replica.State.RECOVERY_FAILED);
131     } finally {
132       close();
133       recoveryListener.failed();
134     }
135   }

137   private void replicate(String nodeName, SolrCore core, ZkNodeProps leaderprops)
138       throws SolrServerException, IOException {
139 
140     ZkCoreNodeProps leaderCNodeProps = new ZkCoreNodeProps(leaderprops);
141     String leaderUrl = leaderCNodeProps.getCoreUrl();
142     
143     log.info("Attempting to replicate from " + leaderUrl + ".");
144     
145     // send commit
146     commitOnLeader(leaderUrl);
147     
148     // use rep handler directly, so we can do this sync rather than async
149     SolrRequestHandler handler = core.getRequestHandler(REPLICATION_HANDLER);
150     ReplicationHandler replicationHandler = (ReplicationHandler) handler;
151     
152     if (replicationHandler == null) {
153       throw new SolrException(ErrorCode.SERVICE_UNAVAILABLE,
154           "Skipping recovery, no " + REPLICATION_HANDLER + " handler found");
155     }
156     
157     ModifiableSolrParams solrParams = new ModifiableSolrParams();
158     solrParams.set(ReplicationHandler.MASTER_URL, leaderUrl);
159     
160     if (isClosed()) return; // we check closed on return
161     boolean success = replicationHandler.doFetch(solrParams, false);
162     
163     if (!success) {
164       throw new SolrException(ErrorCode.SERVER_ERROR, "Replication for recovery failed.");
165     }
166     
167     // solrcloud_debug
168     if (log.isDebugEnabled()) {
169       try {
170         RefCounted<SolrIndexSearcher> searchHolder = core
171             .getNewestSearcher(false);
172         SolrIndexSearcher searcher = searchHolder.get();
173         Directory dir = core.getDirectoryFactory().get(core.getIndexDir(), DirContext.META_DATA, null);
174         try {
175           log.debug(core.getCoreDescriptor().getCoreContainer()
176               .getZkController().getNodeName()
177               + " replicated "
178               + searcher.search(new MatchAllDocsQuery(), 1).totalHits
179               + " from "
180               + leaderUrl
181               + " gen:"
182               + core.getDeletionPolicy().getLatestCommit() != null ? "null" : core.getDeletionPolicy().getLatestCommit().getGeneration()
183               + " data:" + core.getDataDir()
184               + " index:" + core.getIndexDir()
185               + " newIndex:" + core.getNewIndexDir()
186               + " files:" + Arrays.asList(dir.listAll()));
187         } finally {
188           core.getDirectoryFactory().release(dir);
189           searchHolder.decref();
190         }
191       } catch (Exception e) {
192         log.debug("Error in solrcloud_debug block", e);
193       }
194     }
195 
196   }

198   private void commitOnLeader(String leaderUrl) throws SolrServerException,
199       IOException {
200     try (HttpSolrClient client = new HttpSolrClient(leaderUrl)) {
201       client.setConnectionTimeout(30000);
202       UpdateRequest ureq = new UpdateRequest();
203       ureq.setParams(new ModifiableSolrParams());
204       ureq.getParams().set(DistributedUpdateProcessor.COMMIT_END_POINT, true);
205       ureq.getParams().set(UpdateParams.OPEN_SEARCHER, false);
206       ureq.setAction(AbstractUpdateRequest.ACTION.COMMIT, false, true).process(
207           client);
208     }
209   }

211   @Override
212   public void run() {
213 
214     // set request info for logging
215     try (SolrCore core = cc.getCore(coreName)) {
216 
217       if (core == null) {
218         SolrException.log(log, "SolrCore not found - cannot recover:" + coreName);
219         return;
220       }
221       MDCLoggingContext.setCore(core);
222 
223       log.info("Starting recovery process. recoveringAfterStartup=" + recoveringAfterStartup);
224 
225       try {
226         doRecovery(core);
227       } catch (InterruptedException e) {
228         Thread.currentThread().interrupt();
229         SolrException.log(log, "", e);
230         throw new ZooKeeperException(SolrException.ErrorCode.SERVER_ERROR, "", e);
231       } catch (Exception e) {
232         log.error("", e);
233         throw new ZooKeeperException(SolrException.ErrorCode.SERVER_ERROR, "", e);
234       }
235     } finally {
236       MDCLoggingContext.clear();
237     }
238   }

240   // TODO: perhaps make this grab a new core each time through the loop to handle core reloads?
241   public void doRecovery(SolrCore core) throws KeeperException, InterruptedException {
242     boolean replayed = false;
243     boolean successfulRecovery = false;
244 
245     UpdateLog ulog;
246     ulog = core.getUpdateHandler().getUpdateLog();
247     if (ulog == null) {
248       SolrException.log(log, "No UpdateLog found - cannot recover.");
249       recoveryFailed(core, zkController, baseUrl, coreZkNodeName,
250           core.getCoreDescriptor());
251       return;
252     }
253 
254     boolean firstTime = true;
255 
256     List<Long> recentVersions;
257     UpdateLog.RecentUpdates recentUpdates = null;
258     try {
259       recentUpdates = ulog.getRecentUpdates();
260       recentVersions = recentUpdates.getVersions(ulog.getNumRecordsToKeep());
261     } catch (Exception e) {
262       SolrException.log(log, "Corrupt tlog - ignoring.", e);
263       recentVersions = new ArrayList<>(0);
264     } finally {
265       if (recentUpdates != null) {
266         recentUpdates.close();
267       }
268     }
269 
270     List<Long> startingVersions = ulog.getStartingVersions();
271 
272     if (startingVersions != null && recoveringAfterStartup) {
273       try {
274         int oldIdx = 0; // index of the start of the old list in the current
275                         // list
276         long firstStartingVersion = startingVersions.size() > 0 ? startingVersions
277             .get(0) : 0;
278         
279         for (; oldIdx < recentVersions.size(); oldIdx++) {
280           if (recentVersions.get(oldIdx) == firstStartingVersion) break;
281         }
282         
283         if (oldIdx > 0) {
284           log.info("####### Found new versions added after startup: num="
285               + oldIdx);
286           log.info("###### currentVersions=" + recentVersions);
287         }
288         
289         log.info("###### startupVersions=" + startingVersions);
290       } catch (Exception e) {
291         SolrException.log(log, "Error getting recent versions.", e);
292         recentVersions = new ArrayList<>(0);
293       }
294     }
295 
296     if (recoveringAfterStartup) {
297       // if we're recovering after startup (i.e. we have been down), then we need to know what the last versions were
298       // when we went down.  We may have received updates since then.
299       recentVersions = startingVersions;
300       try {
301         if ((ulog.getStartingOperation() & UpdateLog.FLAG_GAP) != 0) {
302           // last operation at the time of startup had the GAP flag set...
303           // this means we were previously doing a full index replication
304           // that probably didn't complete and buffering updates in the
305           // meantime.
306           log.info("Looks like a previous replication recovery did not complete - skipping peer sync.");
307           firstTime = false; // skip peersync
308         }
309       } catch (Exception e) {
310         SolrException.log(log, "Error trying to get ulog starting operation.", e);
311         firstTime = false; // skip peersync
312       }
313     }
314 
315     Future<RecoveryInfo> replayFuture = null;
316     while (!successfulRecovery && !isInterrupted() && !isClosed()) { // don't use interruption or it will close channels though
317       try {
318         CloudDescriptor cloudDesc = core.getCoreDescriptor()
319             .getCloudDescriptor();
320         ZkNodeProps leaderprops = zkStateReader.getLeaderRetry(
321             cloudDesc.getCollectionName(), cloudDesc.getShardId());
322       
323         final String leaderBaseUrl = leaderprops.getStr(ZkStateReader.BASE_URL_PROP);
324         final String leaderCoreName = leaderprops.getStr(ZkStateReader.CORE_NAME_PROP);
325 
326         String leaderUrl = ZkCoreNodeProps.getCoreUrl(leaderBaseUrl, leaderCoreName);
327 
328         String ourUrl = ZkCoreNodeProps.getCoreUrl(baseUrl, coreName);
329 
330         boolean isLeader = leaderUrl.equals(ourUrl);
331         if (isLeader && !cloudDesc.isLeader()) {
332           throw new SolrException(ErrorCode.SERVER_ERROR, "Cloud state still says we are leader.");
333         }
334         if (cloudDesc.isLeader()) {
335           // we are now the leader - no one else must have been suitable
336           log.warn("We have not yet recovered - but we are now the leader!");
337           log.info("Finished recovery process.");
338           zkController.publish(core.getCoreDescriptor(), Replica.State.ACTIVE);
339           return;
340         }
341         
342         log.info("Begin buffering updates. core=" + coreName);
343         ulog.bufferUpdates();
344         replayed = false;
345         
346         log.info("Publishing state of core " + core.getName() + " as recovering, leader is " + leaderUrl + " and I am "
347             + ourUrl);
348         zkController.publish(core.getCoreDescriptor(), Replica.State.RECOVERING);
349         
350         
351         final Slice slice = zkStateReader.getClusterState().getSlice(cloudDesc.getCollectionName(), cloudDesc.getShardId());
352 
353         try {
354           prevSendPreRecoveryHttpUriRequest.abort();
355         } catch (NullPointerException e) {
356           // okay
357         }
358         
359         if (isClosed()) {
360           log.info("Recovery was cancelled");
361           break;
362         }
363 
364         sendPrepRecoveryCmd(leaderBaseUrl, leaderCoreName, slice);
365         
366         if (isClosed()) {
367           log.info("Recovery was cancelled");
368           break;
369         }
370         
371         // we wait a bit so that any updates on the leader
372         // that started before they saw recovering state 
373         // are sure to have finished (see SOLR-7141 for
374         // discussion around current value)
375         try {
376           Thread.sleep(WAIT_FOR_UPDATES_WITH_STALE_STATE_PAUSE);
377         } catch (InterruptedException e) {
378           Thread.currentThread().interrupt();
379         }
380 
381         // first thing we just try to sync
382         if (firstTime) {
383           firstTime = false; // only try sync the first time through the loop
384           log.info("Attempting to PeerSync from " + leaderUrl + " - recoveringAfterStartup="+recoveringAfterStartup);
385           // System.out.println("Attempting to PeerSync from " + leaderUrl
386           // + " i am:" + zkController.getNodeName());
387           PeerSync peerSync = new PeerSync(core,
388               Collections.singletonList(leaderUrl), ulog.getNumRecordsToKeep(), false, false);
389           peerSync.setStartingVersions(recentVersions);
390           boolean syncSuccess = peerSync.sync();
391           if (syncSuccess) {
392             SolrQueryRequest req = new LocalSolrQueryRequest(core,
393                 new ModifiableSolrParams());
394             // force open a new searcher
395             core.getUpdateHandler().commit(new CommitUpdateCommand(req, false));
396             log.info("PeerSync stage of recovery was successful.");
397 
398             // solrcloud_debug
399             if (log.isDebugEnabled()) {
400               try {
401                 RefCounted<SolrIndexSearcher> searchHolder = core
402                     .getNewestSearcher(false);
403                 SolrIndexSearcher searcher = searchHolder.get();
404                 try {
405                   log.debug(core.getCoreDescriptor()
406                       .getCoreContainer().getZkController().getNodeName()
407                       + " synched "
408                       + searcher.search(new MatchAllDocsQuery(), 1).totalHits);
409                 } finally {
410                   searchHolder.decref();
411                 }
412               } catch (Exception e) {
413                 log.debug("Error in solrcloud_debug block", e);
414               }
415             }
416             log.info("Replaying updates buffered during PeerSync.");
417             replay(core);
418             replayed = true;
419             
420             // sync success
421             successfulRecovery = true;
422             return;
423           }
424 
425           log.info("PeerSync Recovery was not successful - trying replication.");
426         }
427 
428         if (isClosed()) {
429           log.info("Recovery was cancelled");
430           break;
431         }
432         
433         log.info("Starting Replication Recovery.");
434 
435         try {
436 
437           replicate(zkController.getNodeName(), core, leaderprops);
438 
439           if (isClosed()) {
440             log.info("Recovery was cancelled");
441             break;
442           }
443 
444           replayFuture = replay(core);
445           replayed = true;
446           
447           if (isClosed()) {
448             log.info("Recovery was cancelled");
449             break;
450           }
451 
452           log.info("Replication Recovery was successful.");
453           successfulRecovery = true;
454         } catch (InterruptedException e) {
455           Thread.currentThread().interrupt();
456           log.warn("Recovery was interrupted", e);
457           close = true;
458         } catch (Exception e) {
459           SolrException.log(log, "Error while trying to recover", e);
460         }
461 
462       } catch (Exception e) {
463         SolrException.log(log, "Error while trying to recover. core=" + coreName, e);
464       } finally {
465         if (!replayed) {
466           try {
467             ulog.dropBufferedUpdates();
468           } catch (Exception e) {
469             SolrException.log(log, "", e);
470           }
471         }
472         if (successfulRecovery) {
473           log.info("Registering as Active after recovery.");
474           try {
475             zkController.publish(core.getCoreDescriptor(), Replica.State.ACTIVE);
476           } catch (Exception e) {
477             log.error("Could not publish as ACTIVE after succesful recovery", e);
478             successfulRecovery = false;
479           }
480           
481           if (successfulRecovery) {
482             close = true;
483             recoveryListener.recovered();
484           }
485         }
486       }
487 
488       if (!successfulRecovery) {
489         // lets pause for a moment and we need to try again...
490         // TODO: we don't want to retry for some problems?
491         // Or do a fall off retry...
492         try {
493 
494           if (isClosed()) {
495             break;
496           }
497           
498           log.error("Recovery failed - trying again... (" + retries + ")");
499           
500           retries++;
501           if (retries >= MAX_RETRIES) {
502             SolrException.log(log, "Recovery failed - max retries exceeded (" + retries + ").");
503             try {
504               recoveryFailed(core, zkController, baseUrl, coreZkNodeName, core.getCoreDescriptor());
505             } catch (Exception e) {
506               SolrException.log(log, "Could not publish that recovery failed", e);
507             }
508             break;
509           }
510         } catch (Exception e) {
511           SolrException.log(log, "", e);
512         }
513 
514         try {
515           // start at 1 sec and work up to a min
516           double loopCount = Math.min(Math.pow(2, retries), 60);
517           log.info("Wait {} seconds before trying to recover again ({})", loopCount, retries);
518           for (int i = 0; i < loopCount; i++) {
519             if (isClosed()) break; // check if someone closed us
520             Thread.sleep(STARTING_RECOVERY_DELAY);
521           }
522         } catch (InterruptedException e) {
523           Thread.currentThread().interrupt();
524           log.warn("Recovery was interrupted.", e);
525           close = true;
526         }
527       }
528 
529     }
530 
531     // if replay was skipped (possibly to due pulling a full index from the leader),
532     // then we still need to update version bucket seeds after recovery
533     if (successfulRecovery && replayFuture == null) {
534       log.info("Updating version bucket highest from index after successful recovery.");
535       core.seedVersionBuckets();
536     }
537 
538     log.info("Finished recovery process.");
539 
540     
541   }

543   private Future<RecoveryInfo> replay(SolrCore core)
544       throws InterruptedException, ExecutionException {
545     Future<RecoveryInfo> future = core.getUpdateHandler().getUpdateLog().applyBufferedUpdates();
546     if (future == null) {
547       // no replay needed\
548       log.info("No replay needed.");
549     } else {
550       log.info("Replaying buffered documents.");
551       // wait for replay
552       RecoveryInfo report = future.get();
553       if (report.failed) {
554         SolrException.log(log, "Replay failed");
555         throw new SolrException(ErrorCode.SERVER_ERROR, "Replay failed");
556       }
557     }
558     
559     // solrcloud_debug
560     if (log.isDebugEnabled()) {
561       try {
562         RefCounted<SolrIndexSearcher> searchHolder = core
563             .getNewestSearcher(false);
564         SolrIndexSearcher searcher = searchHolder.get();
565         try {
566           log.debug(core.getCoreDescriptor().getCoreContainer()
567               .getZkController().getNodeName()
568               + " replayed "
569               + searcher.search(new MatchAllDocsQuery(), 1).totalHits);
570         } finally {
571           searchHolder.decref();
572         }
573       } catch (Exception e) {
574         log.debug("Error in solrcloud_debug block", e);
575       }
576     }
577     
578     return future;
579   }

586   private void sendPrepRecoveryCmd(String leaderBaseUrl, String leaderCoreName, Slice slice)
587       throws SolrServerException, IOException, InterruptedException, ExecutionException {
588 
589     try (HttpSolrClient client = new HttpSolrClient(leaderBaseUrl)) {
590       client.setConnectionTimeout(30000);
591       WaitForState prepCmd = new WaitForState();
592       prepCmd.setCoreName(leaderCoreName);
593       prepCmd.setNodeName(zkController.getNodeName());
594       prepCmd.setCoreNodeName(coreZkNodeName);
595       prepCmd.setState(Replica.State.RECOVERING);
596       prepCmd.setCheckLive(true);
597       prepCmd.setOnlyIfLeader(true);
598       final Slice.State state = slice.getState();
599       if (state != Slice.State.CONSTRUCTION && state != Slice.State.RECOVERY) {
600         prepCmd.setOnlyIfLeaderActive(true);
601       }
602       HttpUriRequestResponse mrr = client.httpUriRequest(prepCmd);
603       prevSendPreRecoveryHttpUriRequest = mrr.httpUriRequest;
604       
605       log.info("Sending prep recovery command to {}; {}", leaderBaseUrl, prepCmd.toString());
606       
607       mrr.future.get();
608     }
609   }
