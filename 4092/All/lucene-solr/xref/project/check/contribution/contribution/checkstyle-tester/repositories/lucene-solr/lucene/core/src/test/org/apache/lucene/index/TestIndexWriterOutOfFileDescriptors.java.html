20  import java.io.IOException;
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  import org.apache.lucene.analysis.MockAnalyzer;
25  import org.apache.lucene.store.Directory;
26  import org.apache.lucene.store.IOContext;
27  import org.apache.lucene.store.MockDirectoryWrapper;
28  import org.apache.lucene.util.LineFileDocs;
29  import org.apache.lucene.util.LuceneTestCase;
30  import org.apache.lucene.util.PrintStreamInfoStream;
31  import org.apache.lucene.util.TestUtil;
32  
33  public class TestIndexWriterOutOfFileDescriptors extends LuceneTestCase {
34    public void test() throws Exception {
35      MockDirectoryWrapper dir = newMockFSDirectory(createTempDir("TestIndexWriterOutOfFileDescriptors"));
36      dir.setPreventDoubleWrite(false);
37      double rate = random().nextDouble()*0.01;
38      //System.out.println("rate=" + rate);
39      dir.setRandomIOExceptionRateOnOpen(rate);
40      int iters = atLeast(20);
41      LineFileDocs docs = new LineFileDocs(random());
42      DirectoryReader r = null;
43      DirectoryReader r2 = null;
44      boolean any = false;
45      MockDirectoryWrapper dirCopy = null;
46      int lastNumDocs = 0;
47      for(int iter=0;iter<iters;iter++) {
48  
49        IndexWriter w = null;
50        if (VERBOSE) {
51          System.out.println("TEST: iter=" + iter);
52        }
53        try {
54          MockAnalyzer analyzer = new MockAnalyzer(random());
55          analyzer.setMaxTokenLength(TestUtil.nextInt(random(), 1, IndexWriter.MAX_TERM_LENGTH));
56          IndexWriterConfig iwc = newIndexWriterConfig(analyzer);
57  
58          if (VERBOSE) {
59            // Do this ourselves instead of relying on LTC so
60            // we see incrementing messageID:
61            iwc.setInfoStream(new PrintStreamInfoStream(System.out));
62          }
63          MergeScheduler ms = iwc.getMergeScheduler();
64          if (ms instanceof ConcurrentMergeScheduler) {
65            ((ConcurrentMergeScheduler) ms).setSuppressExceptions();
66          }
67          w = new IndexWriter(dir, iwc);
68          if (r != null && random().nextInt(5) == 3) {
69            if (random().nextBoolean()) {
70              if (VERBOSE) {
71                System.out.println("TEST: addIndexes LR[]");
72              }
73              TestUtil.addIndexesSlowly(w, r);
74            } else {
75              if (VERBOSE) {
76                System.out.println("TEST: addIndexes Directory[]");
77              }
78              w.addIndexes(new Directory[] {dirCopy});
79            }
80          } else {
81            if (VERBOSE) {
82              System.out.println("TEST: addDocument");
83            }
84            w.addDocument(docs.nextDoc());
85          }
86          dir.setRandomIOExceptionRateOnOpen(0.0);
87          if (ms instanceof ConcurrentMergeScheduler) {
88            ((ConcurrentMergeScheduler) ms).sync();
89          }
90          // If exc hit CMS then writer will be tragically closed:
91          if (w.getTragicException() == null) {
92            w.close();
93          }
94          w = null;
95  
96          // NOTE: This is O(N^2)!  Only enable for temporary debugging:
97          //dir.setRandomIOExceptionRateOnOpen(0.0);
98          //_TestUtil.checkIndex(dir);
99          //dir.setRandomIOExceptionRateOnOpen(rate);
100 
101         // Verify numDocs only increases, to catch IndexWriter
102         // accidentally deleting the index:
103         dir.setRandomIOExceptionRateOnOpen(0.0);
104         assertTrue(DirectoryReader.indexExists(dir));
105         if (r2 == null) {
106           r2 = DirectoryReader.open(dir);
107         } else {
108           DirectoryReader r3 = DirectoryReader.openIfChanged(r2);
109           if (r3 != null) {
110             r2.close();
111             r2 = r3;
112           }
113         }
114         assertTrue("before=" + lastNumDocs + " after=" + r2.numDocs(), r2.numDocs() >= lastNumDocs);
115         lastNumDocs = r2.numDocs();
116         //System.out.println("numDocs=" + lastNumDocs);
117         dir.setRandomIOExceptionRateOnOpen(rate);
118 
119         any = true;
120         if (VERBOSE) {
121           System.out.println("TEST: iter=" + iter + ": success");
122         }
123       } catch (AssertionError | IOException ioe) {
124         if (VERBOSE) {
125           System.out.println("TEST: iter=" + iter + ": exception");
126           ioe.printStackTrace();
127         }
128         if (w != null) {
129           // NOTE: leave random IO exceptions enabled here,
130           // to verify that rollback does not try to write
131           // anything:
132           w.rollback();
133         }
134       }
135 
136       if (any && r == null && random().nextBoolean()) {
137         // Make a copy of a non-empty index so we can use
138         // it to addIndexes later:
139         dir.setRandomIOExceptionRateOnOpen(0.0);
140         r = DirectoryReader.open(dir);
141         dirCopy = newMockFSDirectory(createTempDir("TestIndexWriterOutOfFileDescriptors.copy"));
142         Set<String> files = new HashSet<>();
143         for (String file : dir.listAll()) {
144           if (file.startsWith(IndexFileNames.SEGMENTS) || IndexFileNames.CODEC_FILE_PATTERN.matcher(file).matches()) {
145             dirCopy.copyFrom(dir, file, file, IOContext.DEFAULT);
146             files.add(file);
147           }
148         }
149         dirCopy.sync(files);
150         // Have IW kiss the dir so we remove any leftover
151         // files ... we can easily have leftover files at
152         // the time we take a copy because we are holding
153         // open a reader:
154         new IndexWriter(dirCopy, newIndexWriterConfig(new MockAnalyzer(random()))).close();
155         dirCopy.setRandomIOExceptionRate(rate);
156         dir.setRandomIOExceptionRateOnOpen(rate);
157       }
158     }
159 
160     if (r2 != null) {
161       r2.close();
162     }
163     if (r != null) {
164       r.close();
165       dirCopy.close();
166     }
167     dir.close();
168   }
169 }
