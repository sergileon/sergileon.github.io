77  /**
78   * Helper base class for distributed search test cases
79   *
80   * By default, all tests in sub-classes will be executed with
81   * 1, 2, ... DEFAULT_MAX_SHARD_COUNT number of shards set up repeatedly.
82   *
83   * In general, it's preferable to annotate the tests in sub-classes with a
84   * {@literal @}ShardsFixed(num = N) or a {@literal @}ShardsRepeat(min = M, max = N)
85   * to indicate whether the test should be called once, with a fixed number of shards,
86   * or called repeatedly for number of shards = M to N.
87   *
88   * In some cases though, if the number of shards has to be fixed, but the number
89   * itself is dynamic, or if it has to be set as a default for all sub-classes
90   * of a sub-class, there's a fixShardCount(N) available, which is identical to
91   * {@literal @}ShardsFixed(num = N) for all tests without annotations in that class
92   * hierarchy. Ideally this function should be retired in favour of better annotations..
93   *
94   * @since solr 1.5
95   */
96  public abstract class BaseDistributedSearchTestCase extends SolrTestCaseJ4 {

109   /**
110    * Set's the value of the "hostContext" system property to a random path 
111    * like string (which may or may not contain sub-paths).  This is used 
112    * in the default constructor for this test to help ensure no code paths have
113    * hardcoded assumptions about the servlet context used to run solr.
114    * <p>
115    * Test configs may use the <code>${hostContext}</code> variable to access 
116    * this system property.
117    * </p>
118    * @see #BaseDistributedSearchTestCase()
119    * @see #clearHostContext
120    */
121   @BeforeClass
122   public static void initHostContext() {
123     // Can't use randomRealisticUnicodeString because unescaped unicode is 
124     // not allowed in URL paths
125     // Can't use URLEncoder.encode(randomRealisticUnicodeString) because
126     // Jetty freaks out and returns 404's when the context uses escapes
127 
128     StringBuilder hostContext = new StringBuilder("/");
129     if (random().nextBoolean()) {
130       // half the time we use the root context, the other half...
131 
132       // Remember: randomSimpleString might be the empty string
133       hostContext.append(TestUtil.randomSimpleString(random(), 2));
134       if (random().nextBoolean()) {
135         hostContext.append("_");
136       }
137       hostContext.append(TestUtil.randomSimpleString(random(), 3));
138       if ( ! "/".equals(hostContext.toString())) {
139         // if our random string is empty, this might add a trailing slash, 
140         // but our code should be ok with that
141         hostContext.append("/").append(TestUtil.randomSimpleString(random(), 2));
142       } else {
143         // we got 'lucky' and still just have the root context,
144         // NOOP: don't try to add a subdir to nothing (ie "//" is bad)
145       }
146     }
147     // paranoia, we *really* don't want to ever get "//" in a path...
148     final String hc = hostContext.toString().replaceAll("\\/+","/");
149 
150     log.info("Setting hostContext system property: " + hc);
151     System.setProperty("hostContext", hc);
152   }

163   private static String getHostContextSuitableForServletContext() {
164     String ctx = System.getProperty("hostContext","/solr");
165     if ("".equals(ctx)) ctx = "/solr";
166     if (ctx.endsWith("/")) ctx = ctx.substring(0,ctx.length()-1);;
167     if (!ctx.startsWith("/")) ctx = "/" + ctx;
168     return ctx;

324   protected void createServers(int numShards) throws Exception {
325 
326     System.setProperty("configSetBaseDir", getSolrHome());
327 
328     controlJetty = createControlJetty();
329     controlClient = createNewSolrClient(controlJetty.getLocalPort());
330 
331     shardsArr = new String[numShards];
332     StringBuilder sb = new StringBuilder();
333     for (int i = 0; i < numShards; i++) {
334       if (sb.length() > 0) sb.append(',');
335       final String shardname = "shard" + i;
336       Path jettyHome = testDir.toPath().resolve(shardname);
337       File jettyHomeFile = jettyHome.toFile();
338       seedSolrHome(jettyHomeFile);
339       seedCoreRootDirWithDefaultTestCore(jettyHome.resolve("cores"));
340       JettySolrRunner j = createJetty(jettyHomeFile, null, null, getSolrConfigFile(), getSchemaFile());
341       jettys.add(j);
342       clients.add(createNewSolrClient(j.getLocalPort()));
343       String shardStr = buildUrl(j.getLocalPort()) + "/" + DEFAULT_TEST_CORENAME;
344       shardsArr[i] = shardStr;
345       sb.append(shardStr);
346     }
347 
348     shards = sb.toString();
349   }

356   protected String getShardsString() {
357     if (deadServers == null) return shards;
358     
359     StringBuilder sb = new StringBuilder();
360     for (String shard : shardsArr) {
361       if (sb.length() > 0) sb.append(',');
362       int nDeadServers = r.nextInt(deadServers.length+1);
363       if (nDeadServers > 0) {
364         List<String> replicas = new ArrayList<>(Arrays.asList(deadServers));
365         Collections.shuffle(replicas, r);
366         replicas.add(r.nextInt(nDeadServers+1), shard);
367         for (int i=0; i<nDeadServers+1; i++) {
368           if (i!=0) sb.append('|');
369           sb.append(replicas.get(i));
370         }
371       } else {
372         sb.append(shard);
373       }
374     }
375 
376     return sb.toString();
377   }

401

403     if (solrConfigOverride != null)

405     if (schemaOverride != null)
406       props.setProperty("schema", schemaOverride);
407     if (shardList != null)
408       props.setProperty("shards", shardList);
409     if (dataDir != null) {
410       props.setProperty("solr.data.dir", dataDir);
411     }
412     if (explicitCoreNodeName) {

416 
417     JettySolrRunner jetty = new JettySolrRunner(solrHome.getAbsolutePath(), props, JettyConfig.builder()
418         .stopAtShutdown(true)
419         .setContext(context)
420         .withFilters(getExtraRequestFilters())
421         .withServlets(getExtraServlets())
422         .withSSLConfig(sslConfig)
423         .build());
424 
425     jetty.start();
426     
427     return jetty;
428   }

595   /**
596    * Returns the QueryResponse from {@link #queryServer}  
597    */
598   protected QueryResponse query(boolean setDistribParams, SolrParams p) throws Exception {
599     
600     final ModifiableSolrParams params = new ModifiableSolrParams(p);
601 
602     // TODO: look into why passing true causes fails
603     params.set("distrib", "false");
604     final QueryResponse controlRsp = controlClient.query(params);
605     validateControlData(controlRsp);
606 
607     params.remove("distrib");
608     if (setDistribParams) setDistributedParams(params);
609 
610     QueryResponse rsp = queryServer(params);
611 
612     compareResponses(rsp, controlRsp);
613 
614     if (stress > 0) {
615       log.info("starting stress...");
616       Thread[] threads = new Thread[nThreads];
617       for (int i = 0; i < threads.length; i++) {
618         threads[i] = new Thread() {
619           @Override
620           public void run() {
621             for (int j = 0; j < stress; j++) {
622               int which = r.nextInt(clients.size());
623               SolrClient client = clients.get(which);
624               try {
625                 QueryResponse rsp = client.query(new ModifiableSolrParams(params));
626                 if (verifyStress) {
627                   compareResponses(rsp, controlRsp);
628                 }
629               } catch (SolrServerException | IOException e) {
630                 throw new RuntimeException(e);
631               }
632             }
633           }
634         };
635         threads[i].start();
636       }
637 
638       for (Thread thread : threads) {
639         thread.join();
640       }
641     }
642     return rsp;
643   }

648   public QueryResponse queryAndCompare(SolrParams params, Iterable<SolrClient> clients) throws SolrServerException, IOException {
649     QueryResponse first = null;
650     for (SolrClient client : clients) {

652       if (first == null) {
653         first = rsp;
654       } else {
655         compareResponses(first, rsp);

672   public static String compare(NamedList a, NamedList b, int flags, Map<String, Integer> handle) {
673 //    System.out.println("resp a:" + a);
674 //    System.out.println("resp b:" + b);
675     boolean ordered = (flags & UNORDERED) == 0;
676 
677     if (!ordered) {
678       Map mapA = new HashMap(a.size());
679       for (int i=0; i<a.size(); i++) {
680         Object prev = mapA.put(a.getName(i), a.getVal(i));
681       }
682 
683       Map mapB = new HashMap(b.size());
684       for (int i=0; i<b.size(); i++) {
685         Object prev = mapB.put(b.getName(i), b.getVal(i));
686       }
687 
688       return compare(mapA, mapB, flags, handle);
689     }
690 
691     int posa = 0, posb = 0;
692     int aSkipped = 0, bSkipped = 0;
693 
694     for (; ;) {
695       if (posa >= a.size() && posb >= b.size()) {
696         break;
697       }
698 
699       String namea = null, nameb = null;
700       Object vala = null, valb = null;
701 
702       int flagsa = 0, flagsb = 0;
703       while (posa < a.size()) {
704         namea = a.getName(posa);
705         vala = a.getVal(posa);
706         posa++;
707         flagsa = flags(handle, namea);
708         if ((flagsa & SKIP) != 0) {
709           namea = null; vala = null;
710           aSkipped++;
711           continue;
712         }
713         break;
714       }
715 
716       while (posb < b.size()) {
717         nameb = b.getName(posb);
718         valb = b.getVal(posb);
719         posb++;
720         flagsb = flags(handle, nameb);
721         if ((flagsb & SKIP) != 0) {
722           nameb = null; valb = null;
723           bSkipped++;
724           continue;
725         }
726         if (eq(namea, nameb)) {
727           break;
728         }
729         return "." + namea + "!=" + nameb + " (unordered or missing)";
730         // if unordered, continue until we find the right field.
731       }
732 
733       // ok, namea and nameb should be equal here already.
734       if ((flagsa & SKIPVAL) != 0) continue;  // keys matching is enough
735 
736       String cmp = compare(vala, valb, flagsa, handle);
737       if (cmp != null) return "." + namea + cmp;
738     }
739 
740 
741     if (a.size() - aSkipped != b.size() - bSkipped) {
742       return ".size()==" + a.size() + "," + b.size() + " skipped=" + aSkipped + "," + bSkipped;
743     }
744 
745     return null;
746   }

748   public static String compare1(Map a, Map b, int flags, Map<String, Integer> handle) {
749     String cmp;
750 
751     for (Object keya : a.keySet()) {
752       Object vala = a.get(keya);
753       int flagsa = flags(handle, keya);
754       if ((flagsa & SKIP) != 0) continue;
755       if (!b.containsKey(keya)) {
756         return "[" + keya + "]==null";
757       }
758       if ((flagsa & SKIPVAL) != 0) continue;
759       Object valb = b.get(keya);
760       cmp = compare(vala, valb, flagsa, handle);
761       if (cmp != null) return "[" + keya + "]" + cmp;
762     }
763     return null;
764   }

777   public static String compare(SolrDocumentList a, SolrDocumentList b, int flags, Map<String, Integer> handle) {
778     boolean ordered = (flags & UNORDERED) == 0;
779 
780     String cmp;
781     int f = flags(handle, "maxScore");
782     if ((f & SKIPVAL) == 0) {
783       cmp = compare(a.getMaxScore(), b.getMaxScore(), 0, handle);
784       if (cmp != null) return ".maxScore" + cmp;
785     } else {
786       if (b.getMaxScore() != null) {
787         if (a.getMaxScore() == null) {
788           return ".maxScore missing";
789         }
790       }
791     }
792 
793     cmp = compare(a.getNumFound(), b.getNumFound(), 0, handle);
794     if (cmp != null) return ".numFound" + cmp;
795 
796     cmp = compare(a.getStart(), b.getStart(), 0, handle);
797     if (cmp != null) return ".start" + cmp;
798 
799     cmp = compare(a.size(), b.size(), 0, handle);
800     if (cmp != null) return ".size()" + cmp;
801 
802     // only for completely ordered results (ties might be in a different order)
803     if (ordered) {
804       for (int i = 0; i < a.size(); i++) {
805         cmp = compare(a.get(i), b.get(i), 0, handle);
806         if (cmp != null) return "[" + i + "]" + cmp;
807       }
808       return null;
809     }
810 
811     // unordered case
812     for (int i = 0; i < a.size(); i++) {
813       SolrDocument doc = a.get(i);
814       Object key = doc.getFirstValue("id");
815       SolrDocument docb = null;
816       if (key == null) {
817         // no id field to correlate... must compare ordered
818         docb = b.get(i);
819       } else {
820         for (int j = 0; j < b.size(); j++) {
821           docb = b.get(j);
822           if (key.equals(docb.getFirstValue("id"))) break;
823         }
824       }
825       // if (docb == null) return "[id="+key+"]";
826       cmp = compare(doc, docb, 0, handle);
827       if (cmp != null) return "[id=" + key + "]" + cmp;
828     }
829     return null;
830   }

843   public static String compare(Object a, Object b, int flags, Map<String, Integer> handle) {
844     if (a == b) return null;
845     if (a == null || b == null) return ":" + a + "!=" + b;
846 
847     if (a instanceof NamedList && b instanceof NamedList) {
848       return compare((NamedList) a, (NamedList) b, flags, handle);
849     }
850 
851     if (a instanceof SolrDocumentList && b instanceof SolrDocumentList) {
852       return compare((SolrDocumentList) a, (SolrDocumentList) b, flags, handle);
853     }
854 
855     if (a instanceof SolrDocument && b instanceof SolrDocument) {
856       return compare((SolrDocument) a, (SolrDocument) b, flags, handle);
857     }
858 
859     if (a instanceof Map && b instanceof Map) {
860       return compare((Map) a, (Map) b, flags, handle);
861     }
862 
863     if (a instanceof Object[] && b instanceof Object[]) {
864       return compare((Object[]) a, (Object[]) b, flags, handle);
865     }
866 
867     if (a instanceof byte[] && b instanceof byte[]) {
868       if (!Arrays.equals((byte[]) a, (byte[]) b)) {
869         return ":" + a + "!=" + b;
870       }
871       return null;
872     }
873 
874     if (a instanceof List && b instanceof List) {
875       return compare(((List) a).toArray(), ((List) b).toArray(), flags, handle);
876 
877     }
878 
879     if (!(a.equals(b))) {
880       return ":" + a + "!=" + b;
881     }
882 
883     return null;
884   }

929     private abstract class ShardsStatement extends Statement {
930       abstract protected void callStatement() throws Throwable;
931 
932       @Override
933       public void evaluate() throws Throwable {
934         distribSetUp();
935         if (! distribSetUpCalled) {
936           Assert.fail("One of the overrides of distribSetUp does not propagate the call.");
937         }
938         try {
939           callStatement();
940         } finally {
941           distribTearDown();
942           if (! distribTearDownCalled) {
943             Assert.fail("One of the overrides of distribTearDown does not propagate the call.");
944           }
945         }
946       }
947     }

949     private class ShardsFixedStatement extends ShardsStatement {
950 
951       private final int numShards;
952       private final Statement statement;
953 
954       private ShardsFixedStatement(int numShards, Statement statement) {
955         this.numShards = numShards;
956         this.statement = statement;
957       }
958 
959       @Override
960       public void callStatement() throws Throwable {
961         fixShardCount(numShards);
962         createServers(numShards);
963         RandVal.uniqueValues = new HashSet(); //reset random values
964         statement.evaluate();
965         destroyServers();
966       }
967     }

969     private class ShardsRepeatStatement extends ShardsStatement {
970 
971       private final int min;
972       private final int max;
973       private final Statement statement;
974 
975       private ShardsRepeatStatement(int min, int max, Statement statement) {
976         this.min = min;
977         this.max = max;
978         this.statement = statement;
979       }
980 
981       @Override
982       public void callStatement() throws Throwable {
983         for (shardCount = min; shardCount <= max; shardCount++) {
984           createServers(shardCount);
985           RandVal.uniqueValues = new HashSet(); //reset random values
986           statement.evaluate();
987           destroyServers();
988         }
989       }
990     }
991 
992     @Override
993     public Statement apply(Statement statement, Description description) {
994       ShardsFixed fixed = description.getAnnotation(ShardsFixed.class);
995       ShardsRepeat repeat = description.getAnnotation(ShardsRepeat.class);
996       if (fixed != null && repeat != null) {
997         throw new RuntimeException("ShardsFixed and ShardsRepeat annotations can't coexist");
998       }
999       else if (fixed != null) {
1000         return new ShardsFixedStatement(fixed.num(), statement);
1001       }
1002       else if (repeat != null) {
1003         return new ShardsRepeatStatement(repeat.min(), repeat.max(), statement);
1004       }
1005       else {
1006         return (isShardCountFixed ? new ShardsFixedStatement(shardCount, statement) :
1007           new ShardsRepeatStatement(1, DEFAULT_MAX_SHARD_COUNT, statement));
1008       }
1009     }
