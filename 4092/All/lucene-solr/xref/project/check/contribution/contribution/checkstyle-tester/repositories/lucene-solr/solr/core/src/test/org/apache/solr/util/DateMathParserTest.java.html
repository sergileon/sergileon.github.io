40    /**
41     * A formatter for specifying every last nuance of a Date for easy
42     * refernece in assertion statements
43     */
44    private DateFormat fmt;
45    /**
46     * A parser for reading in explicit dates that are convinient to type
47     * in a test
48     */
49    private DateFormat parser;

51    public DateMathParserTest() {
52      super();
53      fmt = new SimpleDateFormat
54        ("G yyyyy MM ww WW DD dd F E aa HH hh mm ss SSS z Z",Locale.ROOT);
55      fmt.setTimeZone(UTC);
56  
57      parser = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss.SSS",Locale.ROOT);
58      parser.setTimeZone(UTC);
59    }

61    /** MACRO: Round: parses s, rounds with u, fmts */
62    protected String r(String s, String u) throws Exception {
63      Date d = parser.parse(s);
64      Calendar c = Calendar.getInstance(UTC, Locale.ROOT);
65      c.setTime(d);
66      DateMathParser.round(c, u);
67      return fmt.format(c.getTime());
68    }
69    
70    /** MACRO: Add: parses s, adds v u, fmts */
71    protected String a(String s, int v, String u) throws Exception {
72      Date d = parser.parse(s);
73      Calendar c = Calendar.getInstance(UTC, Locale.ROOT);
74      c.setTime(d);
75      DateMathParser.add(c, v, u);
76      return fmt.format(c.getTime());
77    }
78  
79    /** MACRO: Expected: parses s, fmts */
80    protected String e(String s) throws Exception {
81      return fmt.format(parser.parse(s));
82    }
83  
84    protected void assertRound(String e, String i, String u) throws Exception {
85      String ee = e(e);
86      String rr = r(i,u);
87      assertEquals(ee + " != " + rr + " round:" + i + ":" + u, ee, rr);
88    }
89    protected void assertAdd(String e, String i, int v, String u)
90      throws Exception {
91      
92      String ee = e(e);
93      String aa = a(i,v,u);
94      assertEquals(ee + " != " + aa + " add:" + i + "+" + v + ":" + u, ee, aa);
95    }
96  
97    protected void assertMath(String e, DateMathParser p, String i)
98      throws Exception {
99      
100     String ee = e(e);
101     String aa = fmt.format(p.parseMath(i));
102     assertEquals(ee + " != " + aa + " math:" +
103                  parser.format(p.getNow()) + ":" + i, ee, aa);
104   }

106   public void testCalendarUnitsConsistency() throws Exception {
107     String input = "2001-07-04T12:08:56.235";
108     for (String u : DateMathParser.CALENDAR_UNITS.keySet()) {
109       try {
110         r(input, u);
111       } catch (IllegalStateException e) {
112         assertNotNull("no logic for rounding: " + u, e);
113       }
114       try {
115         a(input, 1, u);
116       } catch (IllegalStateException e) {
117         assertNotNull("no logic for rounding: " + u, e);
118       }
119     }
120   }

160 
161     DateMathParser p = new DateMathParser(UTC, Locale.ROOT);
162     p.setNow(parser.parse("2001-07-04T12:08:56.235"));
163 
164     String e = fmt.format(p.parseMath(""));

176   public void testParseMath() throws Exception {
177 
178     DateMathParser p = new DateMathParser(UTC, Locale.ROOT);
179     p.setNow(parser.parse("2001-07-04T12:08:56.235"));
180 
181     // No-Op
182     assertMath("2001-07-04T12:08:56.235", p, "");
183     
184     // simple round
185     assertMath("2001-07-04T12:08:56.000", p, "/SECOND");
186     assertMath("2001-07-04T12:08:00.000", p, "/MINUTE");
187     assertMath("2001-07-04T12:00:00.000", p, "/HOUR");
188     assertMath("2001-07-04T00:00:00.000", p, "/DAY");
189     assertMath("2001-07-01T00:00:00.000", p, "/MONTH");
190     assertMath("2001-01-01T00:00:00.000", p, "/YEAR");
191 
192     // simple addition
193     assertMath("2001-07-04T12:08:56.236", p, "+1MILLISECOND");
194     assertMath("2001-07-04T12:08:57.235", p, "+1SECOND");
195     assertMath("2001-07-04T12:09:56.235", p, "+1MINUTE");
196     assertMath("2001-07-04T13:08:56.235", p, "+1HOUR");
197     assertMath("2001-07-05T12:08:56.235", p, "+1DAY");
198     assertMath("2001-08-04T12:08:56.235", p, "+1MONTH");
199     assertMath("2002-07-04T12:08:56.235", p, "+1YEAR");
200 
201     // simple subtraction
202     assertMath("2001-07-04T12:08:56.234", p, "-1MILLISECOND");
203     assertMath("2001-07-04T12:08:55.235", p, "-1SECOND");
204     assertMath("2001-07-04T12:07:56.235", p, "-1MINUTE");
205     assertMath("2001-07-04T11:08:56.235", p, "-1HOUR");
206     assertMath("2001-07-03T12:08:56.235", p, "-1DAY");
207     assertMath("2001-06-04T12:08:56.235", p, "-1MONTH");
208     assertMath("2000-07-04T12:08:56.235", p, "-1YEAR");
209 
210     // simple '+/-'
211     assertMath("2001-07-04T12:08:56.235", p, "+1MILLISECOND-1MILLISECOND");
212     assertMath("2001-07-04T12:08:56.235", p, "+1SECOND-1SECOND");
213     assertMath("2001-07-04T12:08:56.235", p, "+1MINUTE-1MINUTE");
214     assertMath("2001-07-04T12:08:56.235", p, "+1HOUR-1HOUR");
215     assertMath("2001-07-04T12:08:56.235", p, "+1DAY-1DAY");
216     assertMath("2001-07-04T12:08:56.235", p, "+1MONTH-1MONTH");
217     assertMath("2001-07-04T12:08:56.235", p, "+1YEAR-1YEAR");
218 
219     // simple '-/+'
220     assertMath("2001-07-04T12:08:56.235", p, "-1MILLISECOND+1MILLISECOND");
221     assertMath("2001-07-04T12:08:56.235", p, "-1SECOND+1SECOND");
222     assertMath("2001-07-04T12:08:56.235", p, "-1MINUTE+1MINUTE");
223     assertMath("2001-07-04T12:08:56.235", p, "-1HOUR+1HOUR");
224     assertMath("2001-07-04T12:08:56.235", p, "-1DAY+1DAY");
225     assertMath("2001-07-04T12:08:56.235", p, "-1MONTH+1MONTH");
226     assertMath("2001-07-04T12:08:56.235", p, "-1YEAR+1YEAR");
227 
228     // more complex stuff
229     assertMath("2000-07-04T12:08:56.236", p, "+1MILLISECOND-1YEAR");
230     assertMath("2000-07-04T12:08:57.235", p, "+1SECOND-1YEAR");
231     assertMath("2000-07-04T12:09:56.235", p, "+1MINUTE-1YEAR");
232     assertMath("2000-07-04T13:08:56.235", p, "+1HOUR-1YEAR");
233     assertMath("2000-07-05T12:08:56.235", p, "+1DAY-1YEAR");
234     assertMath("2000-08-04T12:08:56.235", p, "+1MONTH-1YEAR");
235     assertMath("2000-07-04T12:08:56.236", p, "-1YEAR+1MILLISECOND");
236     assertMath("2000-07-04T12:08:57.235", p, "-1YEAR+1SECOND");
237     assertMath("2000-07-04T12:09:56.235", p, "-1YEAR+1MINUTE");
238     assertMath("2000-07-04T13:08:56.235", p, "-1YEAR+1HOUR");
239     assertMath("2000-07-05T12:08:56.235", p, "-1YEAR+1DAY");
240     assertMath("2000-08-04T12:08:56.235", p, "-1YEAR+1MONTH");
241     assertMath("2000-07-01T00:00:00.000", p, "-1YEAR+1MILLISECOND/MONTH");
242     assertMath("2000-07-04T00:00:00.000", p, "-1YEAR+1SECOND/DAY");
243     assertMath("2000-07-04T00:00:00.000", p, "-1YEAR+1MINUTE/DAY");
244     assertMath("2000-07-04T13:00:00.000", p, "-1YEAR+1HOUR/HOUR");
245     assertMath("2000-07-05T12:08:56.000", p, "-1YEAR+1DAY/SECOND");
246     assertMath("2000-08-04T12:08:56.000", p, "-1YEAR+1MONTH/SECOND");
247 
248     // "tricky" cases
249     p.setNow(parser.parse("2006-01-31T17:09:59.999"));
250     assertMath("2006-02-28T17:09:59.999", p, "+1MONTH");
251     assertMath("2008-02-29T17:09:59.999", p, "+25MONTH");
252     assertMath("2006-02-01T00:00:00.000", p, "/MONTH+35DAYS/MONTH");
253     assertMath("2006-01-31T17:10:00.000", p, "+3MILLIS/MINUTE");
254 
255     
256   }

258   public void testParseMathTz() throws Exception {
259 
260     final String PLUS_TZS = "America/Los_Angeles";
261     final String NEG_TZS = "Europe/Paris";
262     
263     assumeTrue("Test requires JVM to know about about TZ: " + PLUS_TZS,
264                TimeZoneUtils.KNOWN_TIMEZONE_IDS.contains(PLUS_TZS)); 
265     assumeTrue("Test requires JVM to know about about TZ: " + NEG_TZS,
266                TimeZoneUtils.KNOWN_TIMEZONE_IDS.contains(NEG_TZS)); 
267 
268     // US, Positive Offset with DST
269 
270     TimeZone tz = TimeZone.getTimeZone(PLUS_TZS);
271     DateMathParser p = new DateMathParser(tz, Locale.ROOT);
272 
273     p.setNow(parser.parse("2001-07-04T12:08:56.235"));
274 
275     // No-Op
276     assertMath("2001-07-04T12:08:56.235", p, "");
277     
278     assertMath("2001-07-04T12:08:56.000", p, "/SECOND");
279     assertMath("2001-07-04T12:08:00.000", p, "/MINUTE");
280     assertMath("2001-07-04T12:00:00.000", p, "/HOUR");
281     assertMath("2001-07-04T07:00:00.000", p, "/DAY");
282     assertMath("2001-07-01T07:00:00.000", p, "/MONTH");
283     // no DST in jan
284     assertMath("2001-01-01T08:00:00.000", p, "/YEAR");
285     // no DST in nov 2001
286     assertMath("2001-11-04T08:00:00.000", p, "+4MONTH/DAY");
287     // yes DST in nov 2010
288     assertMath("2010-11-04T07:00:00.000", p, "+9YEAR+4MONTH/DAY");
289 
290     // France, Negative Offset with DST
291 
292     tz = TimeZone.getTimeZone(NEG_TZS);
293     p = new DateMathParser(tz, Locale.ROOT);
294     p.setNow(parser.parse("2001-07-04T12:08:56.235"));
295 
296     assertMath("2001-07-04T12:08:56.000", p, "/SECOND");
297     assertMath("2001-07-04T12:08:00.000", p, "/MINUTE");
298     assertMath("2001-07-04T12:00:00.000", p, "/HOUR");
299     assertMath("2001-07-03T22:00:00.000", p, "/DAY");
300     assertMath("2001-06-30T22:00:00.000", p, "/MONTH");
301     // no DST in dec
302     assertMath("2000-12-31T23:00:00.000", p, "/YEAR");
303     // no DST in nov
304     assertMath("2001-11-03T23:00:00.000", p, "+4MONTH/DAY");
305 
306   }

308   public void testParseMathExceptions() throws Exception {
309     
310     DateMathParser p = new DateMathParser(UTC, Locale.ROOT);
311     p.setNow(parser.parse("2001-07-04T12:08:56.235"));
312     
313     Map<String,Integer> badCommands = new HashMap<>();
314     badCommands.put("/", 1);
315     badCommands.put("+", 1);
316     badCommands.put("-", 1);
317     badCommands.put("/BOB", 1);
318     badCommands.put("+SECOND", 1);
319     badCommands.put("-2MILLI/", 4);
320     badCommands.put(" +BOB", 0);
321     badCommands.put("+2SECONDS ", 3);
322     badCommands.put("/4", 1);
323     badCommands.put("?SECONDS", 0);
324 
325     for (String command : badCommands.keySet()) {
326       try {
327         Date out = p.parseMath(command);
328         fail("Didn't generate SyntaxError for: " + command);
329       } catch (ParseException e) {
330         assertEquals("Wrong pos for: " + command + " => " + e.getMessage(),
331                      badCommands.get(command).intValue(), e.getErrorOffset());
332 
333       }
334     }
335     
336   }
