49  /**
50   * A FilterReader that exposes <i>indexed</i> values as if they also had
51   * docvalues.
52   * <p>
53   * This is accomplished by "inverting the inverted index" or "uninversion".
54   * <p>
55   * The uninversion process happens lazily: upon the first request for the 
56   * field's docvalues (e.g. via {@link org.apache.lucene.index.LeafReader#getNumericDocValues(String)} 
57   * or similar), it will create the docvalues on-the-fly if needed and cache it,
58   * based on the core cache key of the wrapped LeafReader.
59   */
60  public class UninvertingReader extends FilterLeafReader {
61    
62    /**
63     * Specifies the type of uninversion to apply for the field. 
64     */
65    public static enum Type {
66      /** 
67       * Single-valued Integer, (e.g. indexed with {@link IntField})
68       * <p>
69       * Fields with this type act as if they were indexed with
70       * {@link NumericDocValuesField}.
71       */
72      INTEGER,
73      /** 
74       * Single-valued Long, (e.g. indexed with {@link LongField}) 
75       * <p>
76       * Fields with this type act as if they were indexed with
77       * {@link NumericDocValuesField}.
78       */
79      LONG,
80      /** 
81       * Single-valued Float, (e.g. indexed with {@link FloatField}) 
82       * <p>
83       * Fields with this type act as if they were indexed with
84       * {@link NumericDocValuesField}.
85       */
86      FLOAT,
87      /** 
88       * Single-valued Double, (e.g. indexed with {@link DoubleField}) 
89       * <p>
90       * Fields with this type act as if they were indexed with
91       * {@link NumericDocValuesField}.
92       */
93      DOUBLE,
94      /** 
95       * Single-valued Binary, (e.g. indexed with {@link StringField}) 
96       * <p>
97       * Fields with this type act as if they were indexed with
98       * {@link BinaryDocValuesField}.
99       */
100     BINARY,
101     /** 
102      * Single-valued Binary, (e.g. indexed with {@link StringField}) 
103      * <p>
104      * Fields with this type act as if they were indexed with
105      * {@link SortedDocValuesField}.
106      */
107     SORTED,
108     /** 
109      * Multi-valued Binary, (e.g. indexed with {@link StringField}) 
110      * <p>
111      * Fields with this type act as if they were indexed with
112      * {@link SortedSetDocValuesField}.
113      */
114     SORTED_SET_BINARY,
115     /** 
116      * Multi-valued Integer, (e.g. indexed with {@link IntField}) 
117      * <p>
118      * Fields with this type act as if they were indexed with
119      * {@link SortedSetDocValuesField}.
120      */
121     SORTED_SET_INTEGER,
122     /** 
123      * Multi-valued Float, (e.g. indexed with {@link FloatField}) 
124      * <p>
125      * Fields with this type act as if they were indexed with
126      * {@link SortedSetDocValuesField}.
127      */
128     SORTED_SET_FLOAT,
129     /** 
130      * Multi-valued Long, (e.g. indexed with {@link LongField}) 
131      * <p>
132      * Fields with this type act as if they were indexed with
133      * {@link SortedSetDocValuesField}.
134      */
135     SORTED_SET_LONG,
136     /** 
137      * Multi-valued Double, (e.g. indexed with {@link DoubleField}) 
138      * <p>
139      * Fields with this type act as if they were indexed with
140      * {@link SortedSetDocValuesField}.
141      */
142     SORTED_SET_DOUBLE
143   }
144   
145   /**
146    * Wraps a provided DirectoryReader. Note that for convenience, the returned reader
147    * can be used normally (e.g. passed to {@link DirectoryReader#openIfChanged(DirectoryReader)})
148    * and so on. 
149    */
150   public static DirectoryReader wrap(DirectoryReader in, final Map<String,Type> mapping) throws IOException {
151     return new UninvertingDirectoryReader(in, mapping);
152   }
153   
154   static class UninvertingDirectoryReader extends FilterDirectoryReader {
155     final Map<String,Type> mapping;
156     
157     public UninvertingDirectoryReader(DirectoryReader in, final Map<String,Type> mapping) throws IOException {
158       super(in, new FilterDirectoryReader.SubReaderWrapper() {
159         @Override
160         public LeafReader wrap(LeafReader reader) {
161           return new UninvertingReader(reader, mapping);
162         }
163       });
164       this.mapping = mapping;
165     }
166 
167     @Override
168     protected DirectoryReader doWrapDirectoryReader(DirectoryReader in) throws IOException {
169       return new UninvertingDirectoryReader(in, mapping);
170     }
171   }
172   
173   final Map<String,Type> mapping;
174   final FieldInfos fieldInfos;
175   
176   /** 
177    * Create a new UninvertingReader with the specified mapping 
178    * <p>
179    * Expert: This should almost never be used. Use {@link #wrap(DirectoryReader, Map)}
180    * instead.
181    *  
182    * @lucene.internal
183    */
184   public UninvertingReader(LeafReader in, Map<String,Type> mapping) {
185     super(in);
186     this.mapping = mapping;
187     ArrayList<FieldInfo> filteredInfos = new ArrayList<>();
188     for (FieldInfo fi : in.getFieldInfos()) {
189       DocValuesType type = fi.getDocValuesType();
190       if (fi.getIndexOptions() != IndexOptions.NONE && fi.getDocValuesType() == DocValuesType.NONE) {
191         Type t = mapping.get(fi.name);
192         if (t != null) {
193           switch(t) {
194             case INTEGER:
195             case LONG:
196             case FLOAT:
197             case DOUBLE:
198               type = DocValuesType.NUMERIC;
199               break;
200             case BINARY:
201               type = DocValuesType.BINARY;
202               break;
203             case SORTED:
204               type = DocValuesType.SORTED;
205               break;
206             case SORTED_SET_BINARY:
207             case SORTED_SET_INTEGER:
208             case SORTED_SET_FLOAT:
209             case SORTED_SET_LONG:
210             case SORTED_SET_DOUBLE:
211               type = DocValuesType.SORTED_SET;
212               break;
213             default:
214               throw new AssertionError();
215           }
216         }
217       }
218       filteredInfos.add(new FieldInfo(fi.name, fi.number, fi.hasVectors(), fi.omitsNorms(),
219                                       fi.hasPayloads(), fi.getIndexOptions(), type, -1, Collections.<String,String>emptyMap()));
220     }
221     fieldInfos = new FieldInfos(filteredInfos.toArray(new FieldInfo[filteredInfos.size()]));
222   }
223 
224   @Override
225   public FieldInfos getFieldInfos() {
226     return fieldInfos;
227   }
228 
229   @Override
230   public NumericDocValues getNumericDocValues(String field) throws IOException {
231     Type v = getType(field);
232     if (v != null) {
233       switch (v) {
234         case INTEGER: return FieldCache.DEFAULT.getNumerics(in, field, FieldCache.NUMERIC_UTILS_INT_PARSER, true);
235         case FLOAT: return FieldCache.DEFAULT.getNumerics(in, field, FieldCache.NUMERIC_UTILS_FLOAT_PARSER, true);
236         case LONG: return FieldCache.DEFAULT.getNumerics(in, field, FieldCache.NUMERIC_UTILS_LONG_PARSER, true);
237         case DOUBLE: return FieldCache.DEFAULT.getNumerics(in, field, FieldCache.NUMERIC_UTILS_DOUBLE_PARSER, true);
238       }
239     }
240     return super.getNumericDocValues(field);
241   }
242 
243   @Override
244   public BinaryDocValues getBinaryDocValues(String field) throws IOException {
245     Type v = getType(field);
246     if (v == Type.BINARY) {
247       return FieldCache.DEFAULT.getTerms(in, field, true);
248     } else {
249       return in.getBinaryDocValues(field);
250     }
251   }
252 
253   @Override
254   public SortedDocValues getSortedDocValues(String field) throws IOException {
255     Type v = getType(field);
256     if (v == Type.SORTED) {
257       return FieldCache.DEFAULT.getTermsIndex(in, field);
258     } else {
259       return in.getSortedDocValues(field);
260     }
261   }
262   
263   @Override
264   public SortedSetDocValues getSortedSetDocValues(String field) throws IOException {
265     Type v = getType(field);
266     if (v != null) {
267       switch (v) {
268         case SORTED_SET_INTEGER:
269         case SORTED_SET_FLOAT: 
270           return FieldCache.DEFAULT.getDocTermOrds(in, field, FieldCache.INT32_TERM_PREFIX);
271         case SORTED_SET_LONG:
272         case SORTED_SET_DOUBLE:
273           return FieldCache.DEFAULT.getDocTermOrds(in, field, FieldCache.INT64_TERM_PREFIX);
274         case SORTED_SET_BINARY:
275           return FieldCache.DEFAULT.getDocTermOrds(in, field, null);
276       }
277     }
278     return in.getSortedSetDocValues(field);
279   }
280 
281   @Override
282   public Bits getDocsWithField(String field) throws IOException {
283     if (getType(field) != null) {
284       return FieldCache.DEFAULT.getDocsWithField(in, field);
285     } else {
286       return in.getDocsWithField(field);
287     }
288   }
289   
290   /** 
291    * Returns the field's uninversion type, or null 
292    * if the field doesn't exist or doesn't have a mapping.
293    */
294   private Type getType(String field) {
295     FieldInfo info = fieldInfos.fieldInfo(field);
296     if (info == null || info.getDocValuesType() == DocValuesType.NONE) {
297       return null;
298     }
299     return mapping.get(field);
300   }
301 
302   @Override
303   public Object getCoreCacheKey() {
304     return in.getCoreCacheKey();
305   }
306 
307   @Override
308   public Object getCombinedCoreAndDeletesKey() {
309     return in.getCombinedCoreAndDeletesKey();
310   }
311 
312   @Override
313   public String toString() {
314     return "Uninverting(" + in.toString() + ")";
315   }

317   /** 
318    * Return information about the backing cache
319    * @lucene.internal 
320    */
321   public static String[] getUninvertedStats() {
322     CacheEntry[] entries = FieldCache.DEFAULT.getCacheEntries();
323     String[] info = new String[entries.length];
324     for (int i = 0; i < entries.length; i++) {
325       info[i] = entries[i].toString();
326     }
327     return info;
328   }
