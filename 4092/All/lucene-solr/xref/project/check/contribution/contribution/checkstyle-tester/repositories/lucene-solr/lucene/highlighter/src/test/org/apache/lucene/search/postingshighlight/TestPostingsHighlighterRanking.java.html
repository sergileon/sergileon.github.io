49    /** 
50     * indexes a bunch of gibberish, and then highlights top(n).
51     * asserts that top(n) highlights is a subset of top(n+1) up to some max N
52     */
53    // TODO: this only tests single-valued fields. we should also index multiple values per field!
54    public void testRanking() throws Exception {
55      // number of documents: we will check each one
56      final int numDocs = atLeast(100);
57      // number of top-N snippets, we will check 1 .. N
58      final int maxTopN = 5;
59      // maximum number of elements to put in a sentence.
60      final int maxSentenceLength = 10;
61      // maximum number of sentences in a document
62      final int maxNumSentences = 20;
63      
64      Directory dir = newDirectory();
65      RandomIndexWriter iw = new RandomIndexWriter(random(), dir, new MockAnalyzer(random(), MockTokenizer.SIMPLE, true));
66      Document document = new Document();
67      Field id = new StringField("id", "", Field.Store.NO);
68      FieldType offsetsType = new FieldType(TextField.TYPE_STORED);
69      offsetsType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
70      Field body = new Field("body", "", offsetsType);
71      document.add(id);
72      document.add(body);
73      
74      for (int i = 0; i < numDocs; i++) {
75        StringBuilder bodyText = new StringBuilder();
76        int numSentences = TestUtil.nextInt(random(), 1, maxNumSentences);
77        for (int j = 0; j < numSentences; j++) {
78          bodyText.append(newSentence(random(), maxSentenceLength));
79        }
80        body.setStringValue(bodyText.toString());
81        id.setStringValue(Integer.toString(i));
82        iw.addDocument(document);
83      }
84      
85      IndexReader ir = iw.getReader();
86      IndexSearcher searcher = newSearcher(ir);
87      for (int i = 0; i < numDocs; i++) {
88        checkDocument(searcher, i, maxTopN);
89      }
90      iw.close();
91      ir.close();
92      dir.close();
93    }
94    
95    private void checkDocument(IndexSearcher is, int doc, int maxTopN) throws IOException {
96      for (int ch = 'a'; ch <= 'z'; ch++) {
97        Term term = new Term("body", "" + (char)ch);
98        // check a simple term query
99        checkQuery(is, new TermQuery(term), doc, maxTopN);
100       // check a boolean query
101       BooleanQuery.Builder bq = new BooleanQuery.Builder();
102       bq.add(new TermQuery(term), BooleanClause.Occur.SHOULD);
103       Term nextTerm = new Term("body", "" + (char)(ch+1));
104       bq.add(new TermQuery(nextTerm), BooleanClause.Occur.SHOULD);
105       checkQuery(is, bq.build(), doc, maxTopN);
106     }
107   }
108   
109   private void checkQuery(IndexSearcher is, Query query, int doc, int maxTopN) throws IOException {
110     for (int n = 1; n < maxTopN; n++) {
111       final FakePassageFormatter f1 = new FakePassageFormatter();
112       PostingsHighlighter p1 = new PostingsHighlighter(Integer.MAX_VALUE-1) {
113           @Override
114           protected PassageFormatter getFormatter(String field) {
115             assertEquals("body", field);
116             return f1;
117           }
118         };
119 
120       final FakePassageFormatter f2 = new FakePassageFormatter();
121       PostingsHighlighter p2 = new PostingsHighlighter(Integer.MAX_VALUE-1) {
122           @Override
123           protected PassageFormatter getFormatter(String field) {
124             assertEquals("body", field);
125             return f2;
126           }
127         };
128 
129       BooleanQuery.Builder bq = new BooleanQuery.Builder();
130       bq.add(query, BooleanClause.Occur.MUST);
131       bq.add(new TermQuery(new Term("id", Integer.toString(doc))), BooleanClause.Occur.MUST);
132       TopDocs td = is.search(bq.build(), 1);
133       p1.highlight("body", bq.build(), is, td, n);
134       p2.highlight("body", bq.build(), is, td, n+1);
135       assertTrue(f2.seen.containsAll(f1.seen));
136     }
137   }

139   /** 
140    * returns a new random sentence, up to maxSentenceLength "words" in length.
141    * each word is a single character (a-z). The first one is capitalized.
142    */
143   private String newSentence(Random r, int maxSentenceLength) {
144     StringBuilder sb = new StringBuilder();
145     int numElements = TestUtil.nextInt(r, 1, maxSentenceLength);
146     for (int i = 0; i < numElements; i++) {
147       if (sb.length() > 0) {
148         sb.append(' ');
149         sb.append((char) TestUtil.nextInt(r, 'a', 'z'));
150       } else {
151         // capitalize the first word to help breakiterator
152         sb.append((char) TestUtil.nextInt(r, 'A', 'Z'));
153       }
154     }
155     sb.append(". "); // finalize sentence
156     return sb.toString();
157   }

159   /** 
160    * a fake formatter that doesn't actually format passages.
161    * instead it just collects them for asserts!
162    */
163   static class FakePassageFormatter extends PassageFormatter {
164     HashSet<Pair> seen = new HashSet<>();
165     
166     @Override
167     public String format(Passage passages[], String content) {
168       for (Passage p : passages) {
169         // verify some basics about the passage
170         assertTrue(p.getScore() >= 0);
171         assertTrue(p.getNumMatches() > 0);
172         assertTrue(p.getStartOffset() >= 0);
173         assertTrue(p.getStartOffset() <= content.length());
174         assertTrue(p.getEndOffset() >= p.getStartOffset());
175         assertTrue(p.getEndOffset() <= content.length());
176         // we use a very simple analyzer. so we can assert the matches are correct
177         int lastMatchStart = -1;
178         for (int i = 0; i < p.getNumMatches(); i++) {
179           BytesRef term = p.getMatchTerms()[i];
180           int matchStart = p.getMatchStarts()[i];
181           assertTrue(matchStart >= 0);
182           // must at least start within the passage
183           assertTrue(matchStart < p.getEndOffset());
184           int matchEnd = p.getMatchEnds()[i];
185           assertTrue(matchEnd >= 0);
186           // always moving forward
187           assertTrue(matchStart >= lastMatchStart);
188           lastMatchStart = matchStart;
189           // single character terms
190           assertEquals(matchStart+1, matchEnd);
191           // and the offsets must be correct...
192           assertEquals(1, term.length);
193           assertEquals((char)term.bytes[term.offset], Character.toLowerCase(content.charAt(matchStart)));
194         }
195         // record just the start/end offset for simplicity
196         seen.add(new Pair(p.getStartOffset(), p.getEndOffset()));
197       }
198       return "bogus!!!!!!";
199     }
200   }

202   static class Pair {
203     final int start;
204     final int end;
205     
206     Pair(int start, int end) {
207       this.start = start;
208       this.end = end;
209     }
210 
211     @Override
212     public int hashCode() {
213       final int prime = 31;
214       int result = 1;
215       result = prime * result + end;
216       result = prime * result + start;
217       return result;
218     }
219 
220     @Override
221     public boolean equals(Object obj) {
222       if (this == obj) {
223         return true;
224       }
225       if (obj == null) {
226         return false;
227       }
228       if (getClass() != obj.getClass()) {
229         return false;
230       }
231       Pair other = (Pair) obj;
232       if (end != other.end) {
233         return false;
234       }
235       if (start != other.start) {
236         return false;
237       }
238       return true;
239     }
240 
241     @Override
242     public String toString() {
243       return "Pair [start=" + start + ", end=" + end + "]";
244     }
245   }

247   /** sets b=0 to disable passage length normalization */
248   public void testCustomB() throws Exception {
249     Directory dir = newDirectory();
250     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random(), MockTokenizer.SIMPLE, true));
251     iwc.setMergePolicy(newLogMergePolicy());
252     RandomIndexWriter iw = new RandomIndexWriter(random(), dir, iwc);
253     
254     FieldType offsetsType = new FieldType(TextField.TYPE_STORED);
255     offsetsType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
256     Field body = new Field("body", "", offsetsType);
257     Document doc = new Document();
258     doc.add(body);
259     
260     body.setStringValue("This is a test.  This test is a better test but the sentence is excruiatingly long, " + 
261                         "you have no idea how painful it was for me to type this long sentence into my IDE.");
262     iw.addDocument(doc);
263     
264     IndexReader ir = iw.getReader();
265     iw.close();
266     
267     IndexSearcher searcher = newSearcher(ir);
268     PostingsHighlighter highlighter = new PostingsHighlighter(10000) {
269         @Override
270         protected PassageScorer getScorer(String field) {
271           return new PassageScorer(1.2f, 0, 87);
272         }
273       };
274     Query query = new TermQuery(new Term("body", "test"));
275     TopDocs topDocs = searcher.search(query, 10, Sort.INDEXORDER);
276     assertEquals(1, topDocs.totalHits);
277     String snippets[] = highlighter.highlight("body", query, searcher, topDocs, 1);
278     assertEquals(1, snippets.length);
279     assertTrue(snippets[0].startsWith("This <b>test</b> is a better <b>test</b>"));
280     
281     ir.close();
282     dir.close();
283   }

285   /** sets k1=0 for simple coordinate-level match (# of query terms present) */
286   public void testCustomK1() throws Exception {
287     Directory dir = newDirectory();
288     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random(), MockTokenizer.SIMPLE, true));
289     iwc.setMergePolicy(newLogMergePolicy());
290     RandomIndexWriter iw = new RandomIndexWriter(random(), dir, iwc);
291     
292     FieldType offsetsType = new FieldType(TextField.TYPE_STORED);
293     offsetsType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
294     Field body = new Field("body", "", offsetsType);
295     Document doc = new Document();
296     doc.add(body);
297     
298     body.setStringValue("This has only foo foo. " + 
299                         "On the other hand this sentence contains both foo and bar. " + 
300                         "This has only bar bar bar bar bar bar bar bar bar bar bar bar.");
301     iw.addDocument(doc);
302     
303     IndexReader ir = iw.getReader();
304     iw.close();
305     
306     IndexSearcher searcher = newSearcher(ir);
307     PostingsHighlighter highlighter = new PostingsHighlighter(10000) {
308         @Override
309         protected PassageScorer getScorer(String field) {
310           return new PassageScorer(0, 0.75f, 87);
311         }
312       };
313     BooleanQuery.Builder query = new BooleanQuery.Builder();
314     query.add(new TermQuery(new Term("body", "foo")), BooleanClause.Occur.SHOULD);
315     query.add(new TermQuery(new Term("body", "bar")), BooleanClause.Occur.SHOULD);
316     TopDocs topDocs = searcher.search(query.build(), 10, Sort.INDEXORDER);
317     assertEquals(1, topDocs.totalHits);
318     String snippets[] = highlighter.highlight("body", query.build(), searcher, topDocs, 1);
319     assertEquals(1, snippets.length);
320     assertTrue(snippets[0].startsWith("On the other hand"));
321     
322     ir.close();
323     dir.close();
324   }
