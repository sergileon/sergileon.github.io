124   boolean forceElevation = false;
125   // For each IndexReader, keep a query->elevation map
126   // When the configuration is loaded from the data directory.
127   // The key is null if loaded from the config directory, and
128   // is never re-loaded.
129   final Map<IndexReader, Map<String, ElevationObj>> elevationCache =
130       new WeakHashMap<>();

132   class ElevationObj {
133     final String text;
134     final String analyzed;
135     final TermQuery [] exclude;//just keep the term query, b/c we will not always explicitly exclude the item based on markExcludes query time param
136     final BooleanQuery include;
137     final Map<BytesRef, Integer> priority;
138     final Set<String> ids;
139     final Set<String> excludeIds;
140 
141     ElevationObj(String qstr, List<String> elevate, List<String> exclude) throws IOException {
142       this.text = qstr;
143       this.analyzed = getAnalyzedQuery(this.text);
144       this.ids = new HashSet<>();
145       this.excludeIds = new HashSet<>();
146 
147       BooleanQuery.Builder include = new BooleanQuery.Builder();
148       this.priority = new HashMap<>();
149       int max = elevate.size() + 5;
150       for (String id : elevate) {
151         id = idSchemaFT.readableToIndexed(id);
152         ids.add(id);
153         TermQuery tq = new TermQuery(new Term(idField, id));
154         include.add(tq, BooleanClause.Occur.SHOULD);
155         this.priority.put(new BytesRef(id), max--);
156       }
157       this.include = include.build();
158 
159       if (exclude == null || exclude.isEmpty()) {
160         this.exclude = null;
161       } else {
162         this.exclude = new TermQuery[exclude.size()];
163         for (int i = 0; i < exclude.size(); i++) {
164           String id = idSchemaFT.readableToIndexed(exclude.get(i));
165           excludeIds.add(id);
166           this.exclude[i] = new TermQuery(new Term(idField, id));
167         }
168       }
169     }
170   }

177   @Override
178   public void inform(SolrCore core) {
179     IndexSchema schema = core.getLatestSchema();
180     String a = initArgs.get(FIELD_TYPE);
181     if (a != null) {
182       FieldType ft = schema.getFieldTypes().get(a);
183       if (ft == null) {
184         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
185             "Unknown FieldType: '" + a + "' used in QueryElevationComponent");
186       }
187       analyzer = ft.getQueryAnalyzer();
188     }
189 
190     SchemaField sf = schema.getUniqueKeyField();
191     if( sf == null) {
192       throw new SolrException( SolrException.ErrorCode.SERVER_ERROR, 
193           "QueryElevationComponent requires the schema to have a uniqueKeyField." );
194     }
195     idSchemaFT = sf.getType();
196     idField = sf.getName();
197     //register the EditorialMarkerFactory
198     String excludeName = initArgs.get(QueryElevationParams.EXCLUDE_MARKER_FIELD_NAME, "excluded");
199     if (excludeName == null || excludeName.equals("") == true){
200       excludeName = "excluded";
201     }
202     ExcludedMarkerFactory excludedMarkerFactory = new ExcludedMarkerFactory();
203     core.addTransformerFactory(excludeName, excludedMarkerFactory);
204     ElevatedMarkerFactory elevatedMarkerFactory = new ElevatedMarkerFactory();
205     String markerName = initArgs.get(QueryElevationParams.EDITORIAL_MARKER_FIELD_NAME, "elevated");
206     if (markerName == null || markerName.equals("") == true) {
207       markerName = "elevated";
208     }
209     core.addTransformerFactory(markerName, elevatedMarkerFactory);
210     forceElevation = initArgs.getBool(QueryElevationParams.FORCE_ELEVATION, forceElevation);
211     try {
212       synchronized (elevationCache) {
213         elevationCache.clear();
214         String f = initArgs.get(CONFIG_FILE);
215         if (f == null) {
216           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
217               "QueryElevationComponent must specify argument: '" + CONFIG_FILE
218                   + "' -- path to elevate.xml");
219         }
220         boolean exists = false;
221 
222         // check if using ZooKeeper
223         ZkController zkController = core.getCoreDescriptor().getCoreContainer().getZkController();
224         if (zkController != null) {
225           // TODO : shouldn't have to keep reading the config name when it has been read before
226           exists = zkController.configFileExists(zkController.getZkStateReader().readConfigName(core.getCoreDescriptor().getCloudDescriptor().getCollectionName()), f);
227         } else {
228           File fC = new File(core.getResourceLoader().getConfigDir(), f);
229           File fD = new File(core.getDataDir(), f);
230           if (fC.exists() == fD.exists()) {
231             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
232                 "QueryElevationComponent missing config file: '" + f + "\n"
233                     + "either: " + fC.getAbsolutePath() + " or " + fD.getAbsolutePath() + " must exist, but not both.");
234           }
235           if (fC.exists()) {
236             exists = true;
237             log.info("Loading QueryElevation from: " + fC.getAbsolutePath());
238             Config cfg = new Config(core.getResourceLoader(), f);
239             elevationCache.put(null, loadElevationMap(cfg));
240           }
241         }
242         //in other words, we think this is in the data dir, not the conf dir
243         if (!exists) {
244           // preload the first data
245           RefCounted<SolrIndexSearcher> searchHolder = null;
246           try {
247             searchHolder = core.getNewestSearcher(false);
248             IndexReader reader = searchHolder.get().getIndexReader();
249             getElevationMap(reader, core);
250           } finally {
251             if (searchHolder != null) searchHolder.decref();
252           }
253         }
254       }
255     } catch (Exception ex) {
256       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
257           "Error initializing QueryElevationComponent.", ex);
258     }
259   }

261   //get the elevation map from the data dir
262   Map<String, ElevationObj> getElevationMap(IndexReader reader, SolrCore core) throws Exception {
263     synchronized (elevationCache) {
264       Map<String, ElevationObj> map = elevationCache.get(null);
265       if (map != null) return map;
266 
267       map = elevationCache.get(reader);
268       if (map == null) {
269         String f = initArgs.get(CONFIG_FILE);
270         if (f == null) {
271           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
272               "QueryElevationComponent must specify argument: " + CONFIG_FILE);
273         }
274         log.info("Loading QueryElevation from data dir: " + f);
275         
276         Config cfg;
277         
278         ZkController zkController = core.getCoreDescriptor().getCoreContainer().getZkController();
279         if (zkController != null) {
280           cfg = new Config(core.getResourceLoader(), f, null, null);
281         } else {
282           InputStream is = VersionedFile.getLatestFile(core.getDataDir(), f);
283           cfg = new Config(core.getResourceLoader(), f, new InputSource(is), null);
284         }
285   
286         map = loadElevationMap(cfg);
287         elevationCache.put(reader, map);
288       }
289       return map;
290     }
291   }

293   //load up the elevation map
294   private Map<String, ElevationObj> loadElevationMap(Config cfg) throws IOException {
295     XPath xpath = XPathFactory.newInstance().newXPath();
296     Map<String, ElevationObj> map = new HashMap<>();
297     NodeList nodes = (NodeList) cfg.evaluate("elevate/query", XPathConstants.NODESET);
298     for (int i = 0; i < nodes.getLength(); i++) {
299       Node node = nodes.item(i);
300       String qstr = DOMUtil.getAttr(node, "text", "missing query 'text'");
301 
302       NodeList children = null;
303       try {
304         children = (NodeList) xpath.evaluate("doc", node, XPathConstants.NODESET);
305       } catch (XPathExpressionException e) {
306         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
307             "query requires '<doc .../>' child");
308       }
309 
310       ArrayList<String> include = new ArrayList<>();
311       ArrayList<String> exclude = new ArrayList<>();
312       for (int j = 0; j < children.getLength(); j++) {
313         Node child = children.item(j);
314         String id = DOMUtil.getAttr(child, "id", "missing 'id'");
315         String e = DOMUtil.getAttr(child, EXCLUDE, null);
316         if (e != null) {
317           if (Boolean.valueOf(e)) {
318             exclude.add(id);
319             continue;
320           }
321         }
322         include.add(id);
323       }
324 
325       ElevationObj elev = new ElevationObj(qstr, include, exclude);
326       if (map.containsKey(elev.analyzed)) {
327         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
328             "Boosting query defined twice for query: '" + elev.text + "' (" + elev.analyzed + "')");
329       }
330       map.put(elev.analyzed, elev);
331     }
332     return map;
333   }

340   void setTopQueryResults(IndexReader reader, String query, String[] ids, String[] ex) throws IOException {
341     if (ids == null) {
342       ids = new String[0];
343     }
344     if (ex == null) {
345       ex = new String[0];
346     }
347 
348     Map<String, ElevationObj> elev = elevationCache.get(reader);
349     if (elev == null) {
350       elev = new HashMap<>();
351       elevationCache.put(reader, elev);
352     }
353     ElevationObj obj = new ElevationObj(query, Arrays.asList(ids), Arrays.asList(ex));
354     elev.put(obj.analyzed, obj);
355   }

357   String getAnalyzedQuery(String query) throws IOException {
358     if (analyzer == null) {
359       return query;
360     }
361     StringBuilder norm = new StringBuilder();
362     try (TokenStream tokens = analyzer.tokenStream("", query)) {
363       tokens.reset();
364 
365       CharTermAttribute termAtt = tokens.addAttribute(CharTermAttribute.class);
366       while (tokens.incrementToken()) {
367         norm.append(termAtt.buffer(), 0, termAtt.length());
368       }
369       tokens.end();
370       return norm.toString();
371     }
372   }

378   @Override
379   public void prepare(ResponseBuilder rb) throws IOException {
380     SolrQueryRequest req = rb.req;
381     SolrParams params = req.getParams();
382     // A runtime param can skip 
383     if (!params.getBool(QueryElevationParams.ENABLE, true)) {
384       return;
385     }
386 
387     boolean exclusive = params.getBool(QueryElevationParams.EXCLUSIVE, false);
388     // A runtime parameter can alter the config value for forceElevation
389     boolean force = params.getBool(QueryElevationParams.FORCE_ELEVATION, forceElevation);
390     boolean markExcludes = params.getBool(QueryElevationParams.MARK_EXCLUDES, false);
391     String boostStr = params.get(QueryElevationParams.IDS);
392     String exStr = params.get(QueryElevationParams.EXCLUDE);
393 
394     Query query = rb.getQuery();
395     SolrParams localParams = rb.getQparser().getLocalParams();
396     String qstr = localParams == null ? rb.getQueryString() : localParams.get(QueryParsing.V);
397     if (query == null || qstr == null) {
398       return;
399     }
400 
401     ElevationObj booster = null;
402     try {
403       if(boostStr != null || exStr != null) {
404         List<String> boosts = (boostStr != null) ? StrUtils.splitSmart(boostStr,",", true) : new ArrayList<String>(0);
405         List<String> excludes = (exStr != null) ? StrUtils.splitSmart(exStr, ",", true) : new ArrayList<String>(0);
406         booster = new ElevationObj(qstr, boosts, excludes);
407       } else {
408         IndexReader reader = req.getSearcher().getIndexReader();
409         qstr = getAnalyzedQuery(qstr);
410         booster = getElevationMap(reader, req.getCore()).get(qstr);
411       }
412     } catch (Exception ex) {
413       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
414           "Error loading elevation", ex);
415     }
416 
417     if (booster != null) {
418       rb.req.getContext().put(BOOSTED, booster.ids);
419       rb.req.getContext().put(BOOSTED_PRIORITY, booster.priority);
420 
421       // Change the query to insert forced documents
422       if (exclusive == true) {
423         //we only want these results
424         rb.setQuery(new BoostQuery(booster.include, 0f));
425       } else {
426         BooleanQuery.Builder newq = new BooleanQuery.Builder();
427         newq.setDisableCoord(true);
428         newq.add(query, BooleanClause.Occur.SHOULD);
429         newq.add(new BoostQuery(booster.include, 0f), BooleanClause.Occur.SHOULD);
430         if (booster.exclude != null) {
431           if (markExcludes == false) {
432             for (TermQuery tq : booster.exclude) {
433               newq.add(new BooleanClause(tq, BooleanClause.Occur.MUST_NOT));
434             }
435           } else {
436             //we are only going to mark items as excluded, not actually exclude them.  This works
437             //with the EditorialMarkerFactory
438             rb.req.getContext().put(EXCLUDED, booster.excludeIds);
439           }
440         }
441         rb.setQuery(newq.build());
442       }
443 
444       ElevationComparatorSource comparator = new ElevationComparatorSource(booster);
445       // if the sort is 'score desc' use a custom sorting method to 
446       // insert documents in their proper place 
447       SortSpec sortSpec = rb.getSortSpec();
448       if (sortSpec.getSort() == null) {
449         sortSpec.setSortAndFields(new Sort(new SortField[]{
450               new SortField("_elevate_", comparator, true),
451               new SortField(null, SortField.Type.SCORE, false)
452             }),
453           Arrays.asList(new SchemaField[2]));
454       } else {
455         // Check if the sort is based on score
456         SortSpec modSortSpec = this.modifySortSpec(sortSpec, force, comparator);
457         if (null != modSortSpec) {
458           rb.setSortSpec(modSortSpec);
459         }
460       }
461 
462       // alter the sorting in the grouping specification if there is one
463       GroupingSpecification groupingSpec = rb.getGroupingSpec();
464       if(groupingSpec != null) {
465         SortField[] groupSort = groupingSpec.getGroupSort().getSort();
466         Sort modGroupSort = this.modifySort(groupSort, force, comparator);
467         if(modGroupSort != null) {
468           groupingSpec.setGroupSort(modGroupSort);
469         }
470         SortField[] withinGroupSort = groupingSpec.getSortWithinGroup().getSort();
471         Sort modWithinGroupSort = this.modifySort(withinGroupSort, force, comparator);
472         if(modWithinGroupSort != null) {
473           groupingSpec.setSortWithinGroup(modWithinGroupSort);
474         }
475       }
476     }
477 
478     // Add debugging information
479     if (rb.isDebug()) {
480       List<String> match = null;
481       if (booster != null) {
482         // Extract the elevated terms into a list
483         match = new ArrayList<>(booster.priority.size());
484         for (Object o : booster.include.clauses()) {
485           TermQuery tq = (TermQuery) ((BooleanClause) o).getQuery();
486           match.add(tq.getTerm().text());
487         }
488       }
489 
490       SimpleOrderedMap<Object> dbg = new SimpleOrderedMap<>();
491       dbg.add("q", qstr);
492       dbg.add("match", match);
493       if (rb.isDebugQuery()) {
494         rb.addDebugInfo("queryBoosting", dbg);
495       }
496     }
497   }

499   private Sort modifySort(SortField[] current, boolean force, ElevationComparatorSource comparator) {
500     SortSpec tmp = new SortSpec(new Sort(current), Arrays.asList(new SchemaField[current.length]));
501     tmp = modifySortSpec(tmp, force, comparator);
502     return null == tmp ? null : tmp.getSort();
503   }
504 
505   private SortSpec modifySortSpec(SortSpec current, boolean force, ElevationComparatorSource comparator) {
506     boolean modify = false;
507     SortField[] currentSorts = current.getSort().getSort();
508     List<SchemaField> currentFields = current.getSchemaFields();
509 
510     ArrayList<SortField> sorts = new ArrayList<>(currentSorts.length + 1);
511     List<SchemaField> fields = new ArrayList<>(currentFields.size() + 1);
512 
513     // Perhaps force it to always sort by score
514     if (force && currentSorts[0].getType() != SortField.Type.SCORE) {
515       sorts.add(new SortField("_elevate_", comparator, true));
516       fields.add(null);
517       modify = true;
518     }
519     for (int i = 0; i < currentSorts.length; i++) {
520       SortField sf = currentSorts[i];
521       if (sf.getType() == SortField.Type.SCORE) {
522         sorts.add(new SortField("_elevate_", comparator, !sf.getReverse()));
523         fields.add(null);
524         modify = true;
525       }
526       sorts.add(sf);
527       fields.add(currentFields.get(i));
528     }
529     if (modify) {
530       SortSpec newSpec = new SortSpec(new Sort(sorts.toArray(new SortField[sorts.size()])),
531                                       fields);
532       newSpec.setOffset(current.getOffset());
533       newSpec.setCount(current.getCount());
534       return newSpec;
535     }
536     return null;
537   }

540   public static IntIntHashMap getBoostDocs(SolrIndexSearcher indexSearcher, Map<BytesRef, Integer>boosted, Map context) throws IOException {
541 
542     IntIntHashMap boostDocs = null;
543 
544     if(boosted != null) {
545 
546       //First see if it's already in the request context. Could have been put there
547       //by another caller.
548       if(context != null) {
549         boostDocs = (IntIntHashMap) context.get(BOOSTED_DOCIDS);
550       }
551 
552       if(boostDocs != null) {
553         return boostDocs;
554       }
555       //Not in the context yet so load it.
556 
557       SchemaField idField = indexSearcher.getSchema().getUniqueKeyField();
558       String fieldName = idField.getName();
559       HashSet<BytesRef> localBoosts = new HashSet<>(boosted.size()*2);
560       Iterator<BytesRef> boostedIt = boosted.keySet().iterator();
561       while(boostedIt.hasNext()) {
562         localBoosts.add(boostedIt.next());
563       }
564 
565       boostDocs = new IntIntHashMap(boosted.size());
566 
567       List<LeafReaderContext>leaves = indexSearcher.getTopReaderContext().leaves();
568       PostingsEnum postingsEnum = null;
569       for(LeafReaderContext leaf : leaves) {
570         LeafReader reader = leaf.reader();
571         int docBase = leaf.docBase;
572         Bits liveDocs = reader.getLiveDocs();
573         Terms terms = reader.terms(fieldName);
574         TermsEnum termsEnum = terms.iterator();
575         Iterator<BytesRef> it = localBoosts.iterator();
576         while(it.hasNext()) {
577           BytesRef ref = it.next();
578           if(termsEnum.seekExact(ref)) {
579             postingsEnum = termsEnum.postings(postingsEnum);
580             int doc = postingsEnum.nextDoc();
581             while (doc != PostingsEnum.NO_MORE_DOCS && liveDocs != null && liveDocs.get(doc) == false) {
582               doc = postingsEnum.nextDoc();
583             }
584             if(doc != PostingsEnum.NO_MORE_DOCS) {
585               //Found the document.
586               int p = boosted.get(ref);
587               boostDocs.put(doc+docBase, p);
588               it.remove();
589             }
590           }
591         }
592       }
593     }
594 
595     if(context != null) {
596       context.put(BOOSTED_DOCIDS, boostDocs);
597     }
598 
599     return boostDocs;
600   }

626   class ElevationComparatorSource extends FieldComparatorSource {
627   private QueryElevationComponent.ElevationObj elevations;
628   private SentinelIntSet ordSet; //the key half of the map
629   private BytesRef[] termValues;//the value half of the map
630 
631   public ElevationComparatorSource(final QueryElevationComponent.ElevationObj elevations) {
632     this.elevations = elevations;
633     int size = elevations.ids.size();
634     ordSet = new SentinelIntSet(size, -1);
635     termValues = new BytesRef[ordSet.keys.length];
636   }
637 
638   @Override
639   public FieldComparator<Integer> newComparator(String fieldname, final int numHits, int sortPos, boolean reversed) throws IOException {
640     return new SimpleFieldComparator<Integer>() {
641       private final int[] values = new int[numHits];
642       private int bottomVal;
643       private int topVal;
644       private PostingsEnum postingsEnum;
645       Set<String> seen = new HashSet<>(elevations.ids.size());
646 
647       @Override
648       public int compare(int slot1, int slot2) {
649         return values[slot1] - values[slot2];  // values will be small enough that there is no overflow concern
650       }
651 
652       @Override
653       public void setBottom(int slot) {
654         bottomVal = values[slot];
655       }
656 
657       @Override
658       public void setTopValue(Integer value) {
659         topVal = value.intValue();
660       }
661 
662       private int docVal(int doc) {
663         if (ordSet.size() > 0) {
664           int slot = ordSet.find(doc);
665           if (slot >= 0) {
666             BytesRef id = termValues[slot];
667             Integer prio = elevations.priority.get(id);
668             return prio == null ? 0 : prio.intValue();
669           }
670         }
671         return 0;
672       }
673 
674       @Override
675       public int compareBottom(int doc) {
676         return bottomVal - docVal(doc);
677       }
678 
679       @Override
680       public void copy(int slot, int doc) {
681         values[slot] = docVal(doc);
682       }
683 
684       @Override
685       protected void doSetNextReader(LeafReaderContext context) throws IOException {
686         //convert the ids to Lucene doc ids, the ordSet and termValues needs to be the same size as the number of elevation docs we have
687         ordSet.clear();
688         Fields fields = context.reader().fields();
689         if (fields == null) return;
690         Terms terms = fields.terms(idField);
691         if (terms == null) return;
692         TermsEnum termsEnum = terms.iterator();
693         BytesRefBuilder term = new BytesRefBuilder();
694         Bits liveDocs = context.reader().getLiveDocs();
695 
696         for (String id : elevations.ids) {
697           term.copyChars(id);
698           if (seen.contains(id) == false  && termsEnum.seekExact(term.get())) {
699             postingsEnum = termsEnum.postings(postingsEnum, PostingsEnum.NONE);
700             int docId = postingsEnum.nextDoc();
701             while (docId != DocIdSetIterator.NO_MORE_DOCS && liveDocs != null && liveDocs.get(docId) == false) {
702               docId = postingsEnum.nextDoc();
703             }
704             if (docId == DocIdSetIterator.NO_MORE_DOCS ) continue;  // must have been deleted
705             termValues[ordSet.put(docId)] = term.toBytesRef();
706             seen.add(id);
707             assert postingsEnum.nextDoc() == DocIdSetIterator.NO_MORE_DOCS;
708           }
709         }
710       }
711 
712       @Override
713       public Integer value(int slot) {
714         return values[slot];
715       }
716 
717       @Override
718       public int compareTop(int doc) {
719         final int docValue = docVal(doc);
720         return topVal - docValue;  // values will be small enough that there is no overflow concern
721       }
722     };
723   }
