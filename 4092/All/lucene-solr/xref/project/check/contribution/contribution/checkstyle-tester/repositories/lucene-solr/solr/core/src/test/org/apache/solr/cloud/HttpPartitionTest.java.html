188   protected void testMinRf() throws Exception {
189     // create a collection that has 1 shard and 3 replicas
190     String testCollectionName = "collMinRf_1x3";
191     createCollection(testCollectionName, 1, 3, 1);
192     cloudClient.setDefaultCollection(testCollectionName);
193 
194     sendDoc(1, 2);
195 
196     List<Replica> notLeaders =
197         ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 3, maxWaitSecsToSeeAllActive);
198     assertTrue("Expected 2 non-leader replicas for collection " + testCollectionName
199             + " but found " + notLeaders.size() + "; clusterState: "
200             + printClusterStateInfo(testCollectionName),
201         notLeaders.size() == 2);
202 
203     assertDocsExistInAllReplicas(notLeaders, testCollectionName, 1, 1);
204 
205     // Now introduce a network partition between the leader and 1 replica, so a minRf of 2 is still achieved
206     SocketProxy proxy0 = getProxyForReplica(notLeaders.get(0));
207 
208     proxy0.close();
209 
210     // indexing during a partition
211     int achievedRf = sendDoc(2, 2);
212     assertEquals("Unexpected achieved replication factor", 2, achievedRf);
213 
214     Thread.sleep(sleepMsBeforeHealPartition);
215 
216     // Verify that the partitioned replica is DOWN
217     ZkStateReader zkr = cloudClient.getZkStateReader();
218     zkr.updateClusterState(); // force the state to be fresh
219     ClusterState cs = zkr.getClusterState();
220     Collection<Slice> slices = cs.getActiveSlices(testCollectionName);
221     Slice slice = slices.iterator().next();
222     Replica partitionedReplica = slice.getReplica(notLeaders.get(0).getName());
223     assertEquals("The partitioned replica did not get marked down",
224         Replica.State.DOWN.toString(), partitionedReplica.getStr(ZkStateReader.STATE_PROP));
225 
226     proxy0.reopen();
227 
228     notLeaders =
229         ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 3, maxWaitSecsToSeeAllActive);
230 
231     // Since minRf is achieved, we expect recovery, so we expect seeing 2 documents
232     assertDocsExistInAllReplicas(notLeaders, testCollectionName, 1, 2);
233 
234     // Now introduce a network partition between the leader and both of its replicas, so a minRf of 2 is NOT achieved
235     proxy0 = getProxyForReplica(notLeaders.get(0));
236     proxy0.close();
237     SocketProxy proxy1 = getProxyForReplica(notLeaders.get(1));
238     proxy1.close();
239 
240     achievedRf = sendDoc(3, 2);
241     assertEquals("Unexpected achieved replication factor", 1, achievedRf);
242 
243     Thread.sleep(sleepMsBeforeHealPartition);
244 
245     // Verify that the partitioned replicas are NOT DOWN since minRf wasn't achieved
246     ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 3, 1);
247 
248     proxy0.reopen();
249     proxy1.reopen();
250 
251     notLeaders =
252         ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 3, maxWaitSecsToSeeAllActive);
253 
254     // Check that doc 3 is on the leader but not on the notLeaders
255     Replica leader = cloudClient.getZkStateReader().getLeaderRetry(testCollectionName, "shard1", 10000);
256     HttpSolrClient leaderSolr = getHttpSolrClient(leader, testCollectionName);
257     assertDocExists(leaderSolr, testCollectionName, "3");
258 
259     for (Replica notLeader : notLeaders) {
260       HttpSolrClient notLeaderSolr = getHttpSolrClient(notLeader, testCollectionName);
261       assertDocNotExists(notLeaderSolr, testCollectionName, "3");
262     }
263 
264     // Retry sending doc 3
265     achievedRf = sendDoc(3, 2);
266     assertEquals("Unexpected achieved replication factor", 3, achievedRf);
267 
268     // Now doc 3 should be on all replicas
269     assertDocsExistInAllReplicas(notLeaders, testCollectionName, 1, 3);
270   }

272   protected void testRf2() throws Exception {
273     // create a collection that has 1 shard but 2 replicas
274     String testCollectionName = "c8n_1x2";
275     createCollectionRetry(testCollectionName, 1, 2, 1);
276     cloudClient.setDefaultCollection(testCollectionName);
277     
278     sendDoc(1);
279     
280     Replica notLeader = 
281         ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 2, maxWaitSecsToSeeAllActive).get(0);
282     
283     // ok, now introduce a network partition between the leader and the replica
284     SocketProxy proxy = getProxyForReplica(notLeader);
285     
286     proxy.close();
287     
288     // indexing during a partition
289     sendDoc(2);
290     
291     // Have the partition last at least 1 sec
292     // While this gives the impression that recovery is timing related, this is
293     // really only
294     // to give time for the state to be written to ZK before the test completes.
295     // In other words,
296     // without a brief pause, the test finishes so quickly that it doesn't give
297     // time for the recovery process to kick-in
298     Thread.sleep(sleepMsBeforeHealPartition);
299     
300     proxy.reopen();
301     
302     List<Replica> notLeaders = 
303         ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 2, maxWaitSecsToSeeAllActive);
304     
305     sendDoc(3);
306     
307     // sent 3 docs in so far, verify they are on the leader and replica
308     assertDocsExistInAllReplicas(notLeaders, testCollectionName, 1, 3);
309 
310     // Get the max version from the replica core to make sure it gets updated after recovery (see SOLR-7625)
311     JettySolrRunner replicaJetty = getJettyOnPort(getReplicaPort(notLeader));
312     CoreContainer coreContainer = replicaJetty.getCoreContainer();
313     ZkCoreNodeProps replicaCoreNodeProps = new ZkCoreNodeProps(notLeader);
314     String coreName = replicaCoreNodeProps.getCoreName();
315     Long maxVersionBefore = null;
316     try (SolrCore core = coreContainer.getCore(coreName)) {
317       assertNotNull("Core '"+coreName+"' not found for replica: "+notLeader.getName(), core);
318       UpdateLog ulog = core.getUpdateHandler().getUpdateLog();
319       maxVersionBefore = ulog.getCurrentMaxVersion();
320     }
321     assertNotNull("max version bucket seed not set for core " + coreName, maxVersionBefore);
322     log.info("Looked up max version bucket seed "+maxVersionBefore+" for core "+coreName);
323 
324     // now up the stakes and do more docs
325     int numDocs = 1000;
326     boolean hasPartition = false;
327     for (int d = 0; d < numDocs; d++) {
328       // create / restore partition every 100 docs
329       if (d % 100 == 0) {
330         if (hasPartition) {
331           proxy.reopen();
332           hasPartition = false;
333         } else {
334           if (d >= 100) {
335             proxy.close();
336             hasPartition = true;
337             Thread.sleep(sleepMsBeforeHealPartition);
338           }
339         }
340       }
341       sendDoc(d + 4); // 4 is offset as we've already indexed 1-3
342     }
343     
344     // restore connectivity if lost
345     if (hasPartition) {
346       proxy.reopen();
347     }
348     
349     notLeaders = ensureAllReplicasAreActive(testCollectionName, "shard1", 1, 2, maxWaitSecsToSeeAllActive);
350 
351     try (SolrCore core = coreContainer.getCore(coreName)) {
352       assertNotNull("Core '" + coreName + "' not found for replica: " + notLeader.getName(), core);
353       Long currentMaxVersion = core.getUpdateHandler().getUpdateLog().getCurrentMaxVersion();
354       log.info("After recovery, looked up NEW max version bucket seed " + currentMaxVersion +
355           " for core " + coreName + ", was: " + maxVersionBefore);
356       assertTrue("max version bucket seed not updated after recovery!", currentMaxVersion > maxVersionBefore);
357     }
358 
359     // verify all docs received
360     assertDocsExistInAllReplicas(notLeaders, testCollectionName, 1, numDocs + 3);
361 
362     log.info("testRf2 succeeded ... deleting the "+testCollectionName+" collection");
363 
364     // try to clean up
365     try {
366       new CollectionAdminRequest.Delete()
367               .setCollectionName(testCollectionName).process(cloudClient);
368     } catch (Exception e) {
369       // don't fail the test
370       log.warn("Could not delete collection {} after test completed", testCollectionName);
371     }
372   }

431   private void createCollectionRetry(String testCollectionName, int numShards, int replicationFactor, int maxShardsPerNode)
432       throws SolrServerException, IOException {
433     CollectionAdminResponse resp = createCollection(testCollectionName, numShards, replicationFactor, maxShardsPerNode);
434     if (resp.getResponse().get("failure") != null) {
435       CollectionAdminRequest.Delete req = new CollectionAdminRequest.Delete();
436       req.setCollectionName(testCollectionName);
437       req.process(cloudClient);
438       
439       resp = createCollection(testCollectionName, numShards, replicationFactor, maxShardsPerNode);
440       
441       if (resp.getResponse().get("failure") != null) {
442         fail("Could not create " + testCollectionName);
443       }
444     }
445   }

538   protected List<Replica> getActiveOrRecoveringReplicas(String testCollectionName, String shardId) throws Exception {    
539     Map<String,Replica> activeReplicas = new HashMap<String,Replica>();    
540     ZkStateReader zkr = cloudClient.getZkStateReader();
541     ClusterState cs = zkr.getClusterState();
542     assertNotNull(cs);
543     for (Slice shard : cs.getActiveSlices(testCollectionName)) {
544       if (shard.getName().equals(shardId)) {
545         for (Replica replica : shard.getReplicas()) {
546           final Replica.State state = replica.getState();
547           if (state == Replica.State.ACTIVE || state == Replica.State.RECOVERING) {
548             activeReplicas.put(replica.getName(), replica);
549           }
550         }
551       }
552     }        
553     List<Replica> replicas = new ArrayList<Replica>();
554     replicas.addAll(activeReplicas.values());
555     return replicas;
556   }

558   protected void assertDocsExistInAllReplicas(List<Replica> notLeaders,
559       String testCollectionName, int firstDocId, int lastDocId)
560       throws Exception {
561     Replica leader =
562         cloudClient.getZkStateReader().getLeaderRetry(testCollectionName, "shard1", 10000);
563     HttpSolrClient leaderSolr = getHttpSolrClient(leader, testCollectionName);
564     List<HttpSolrClient> replicas =
565         new ArrayList<HttpSolrClient>(notLeaders.size());
566 
567     for (Replica r : notLeaders) {
568       replicas.add(getHttpSolrClient(r, testCollectionName));
569     }
570     try {
571       for (int d = firstDocId; d <= lastDocId; d++) {
572         String docId = String.valueOf(d);
573         assertDocExists(leaderSolr, testCollectionName, docId);
574         for (HttpSolrClient replicaSolr : replicas) {
575           assertDocExists(replicaSolr, testCollectionName, docId);
576         }
577       }
578     } finally {
579       if (leaderSolr != null) {
580         leaderSolr.close();
581       }
582       for (HttpSolrClient replicaSolr : replicas) {
583         replicaSolr.close();
584       }
585     }
586   }

598   protected int sendDoc(int docId, Integer minRf) throws Exception {
599     SolrInputDocument doc = new SolrInputDocument();
600     doc.addField(id, String.valueOf(docId));
601     doc.addField("a_t", "hello" + docId);

603     UpdateRequest up = new UpdateRequest();
604     if (minRf != null) {
605       up.setParam(UpdateRequest.MIN_REPFACT, String.valueOf(minRf));
606     }
607     up.add(doc);
608 
609     return cloudClient.getMinAchievedReplicationFactor(cloudClient.getDefaultCollection(), cloudClient.request(up));
610   }

644   protected void waitToSeeReplicasActive(String testCollectionName, String shardId, Set<String> replicasToCheck, int maxWaitSecs) throws Exception {
645     final RTimer timer = new RTimer();
646 
647     ZkStateReader zkr = cloudClient.getZkStateReader();
648     zkr.updateClusterState(); // force the state to be fresh
649 
650     ClusterState cs = zkr.getClusterState();
651     Collection<Slice> slices = cs.getActiveSlices(testCollectionName);
652     boolean allReplicasUp = false;
653     long waitMs = 0L;
654     long maxWaitMs = maxWaitSecs * 1000L;
655     while (waitMs < maxWaitMs && !allReplicasUp) {
656       // refresh state every 2 secs
657       if (waitMs % 2000 == 0)
658         cloudClient.getZkStateReader().updateClusterState();
659 
660       cs = cloudClient.getZkStateReader().getClusterState();
661       assertNotNull(cs);
662       Slice shard = cs.getSlice(testCollectionName, shardId);
663       assertNotNull("No Slice for "+shardId, shard);
664       allReplicasUp = true; // assume true
665 
666       // wait to see all replicas are "active"
667       for (Replica replica : shard.getReplicas()) {
668         if (!replicasToCheck.contains(replica.getName()))
669           continue;
670 
671         final Replica.State state = replica.getState();
672         if (state != Replica.State.ACTIVE) {
673           log.info("Replica " + replica.getName() + " is currently " + state);
674           allReplicasUp = false;
675         }
676       }
677 
678       if (!allReplicasUp) {
679         try {
680           Thread.sleep(1000L);
681         } catch (Exception ignoreMe) {}
682         waitMs += 1000L;
683       }
684     } // end while
685 
686     if (!allReplicasUp)
687       fail("Didn't see replicas "+ replicasToCheck +
688           " come up within " + maxWaitMs + " ms! ClusterState: " + printClusterStateInfo(testCollectionName));
689 
690     log.info("Took {} ms to see replicas [{}] become active.", timer.getTime(), replicasToCheck);
691   }
