66  /**
67   * {@link TaxonomyWriter} which uses a {@link Directory} to store the taxonomy
68   * information on disk, and keeps an additional in-memory cache of some or all
69   * categories.
70   * <p>
71   * In addition to the permanently-stored information in the {@link Directory},
72   * efficiency dictates that we also keep an in-memory cache of <B>recently
73   * seen</B> or <B>all</B> categories, so that we do not need to go back to disk
74   * for every category addition to see which ordinal this category already has,
75   * if any. A {@link TaxonomyWriterCache} object determines the specific caching
76   * algorithm used.
77   * <p>
78   * This class offers some hooks for extending classes to control the
79   * {@link IndexWriter} instance that is used. See {@link #openIndexWriter}.
80   * 
81   * @lucene.experimental
82   */
83  public class DirectoryTaxonomyWriter implements TaxonomyWriter {
84    
85    /**
86     * Property name of user commit data that contains the index epoch. The epoch
87     * changes whenever the taxonomy is recreated (i.e. opened with
88     * {@link OpenMode#CREATE}.
89     * <p>
90     * Applications should not use this property in their commit data because it
91     * will be overridden by this taxonomy writer.
92     */
93    public static final String INDEX_EPOCH = "index.epoch";

115   /**
116    * We call the cache "complete" if we know that every category in our
117    * taxonomy is in the cache. When the cache is <B>not</B> complete, and
118    * we can't find a category in the cache, we still need to look for it
119    * in the on-disk index; Therefore when the cache is not complete, we
120    * need to open a "reader" to the taxonomy index.
121    * The cache becomes incomplete if it was never filled with the existing
122    * categories, or if a put() to the cache ever returned true (meaning
123    * that some of the cached data was cleared).
124    */
125   private volatile boolean cacheIsComplete;
126   private volatile boolean isClosed = false;
127   private volatile TaxonomyIndexArrays taxoArrays;
128   private volatile int nextID;

133     return infos.getUserData();

136   /**
137    * Construct a Taxonomy writer.
138    * 
139    * @param directory
140    *    The {@link Directory} in which to store the taxonomy. Note that
141    *    the taxonomy is written directly to that directory (not to a
142    *    subdirectory of it).
143    * @param openMode
144    *    Specifies how to open a taxonomy for writing: <code>APPEND</code>
145    *    means open an existing index for append (failing if the index does
146    *    not yet exist). <code>CREATE</code> means create a new index (first
147    *    deleting the old one if it already existed).
148    *    <code>APPEND_OR_CREATE</code> appends to an existing index if there
149    *    is one, otherwise it creates a new index.
150    * @param cache
151    *    A {@link TaxonomyWriterCache} implementation which determines
152    *    the in-memory caching policy. See for example
153    *    {@link LruTaxonomyWriterCache} and {@link Cl2oTaxonomyWriterCache}.
154    *    If null or missing, {@link #defaultTaxonomyWriterCache()} is used.
155    * @throws CorruptIndexException
156    *     if the taxonomy is corrupted.
157    * @throws LockObtainFailedException
158    *     if the taxonomy is locked by another writer.
159    * @throws IOException
160    *     if another error occurred.
161    */
162   public DirectoryTaxonomyWriter(Directory directory, OpenMode openMode,
163       TaxonomyWriterCache cache) throws IOException {
164 
165     dir = directory;
166     IndexWriterConfig config = createIndexWriterConfig(openMode);
167     indexWriter = openIndexWriter(dir, config);
168 
169     // verify (to some extent) that merge policy in effect would preserve category docids 
170     assert !(indexWriter.getConfig().getMergePolicy() instanceof TieredMergePolicy) : 
171       "for preserving category docids, merging none-adjacent segments is not allowed";
172     
173     // after we opened the writer, and the index is locked, it's safe to check
174     // the commit data and read the index epoch
175     openMode = config.getOpenMode();
176     if (!DirectoryReader.indexExists(directory)) {
177       indexEpoch = 1;
178     } else {
179       String epochStr = null;
180       Map<String, String> commitData = readCommitData(directory);
181       if (commitData != null) {
182         epochStr = commitData.get(INDEX_EPOCH);
183       }
184       // no commit data, or no epoch in it means an old taxonomy, so set its epoch to 1, for lack
185       // of a better value.
186       indexEpoch = epochStr == null ? 1 : Long.parseLong(epochStr, 16);
187     }
188     
189     if (openMode == OpenMode.CREATE) {
190       ++indexEpoch;
191     }
192     
193     FieldType ft = new FieldType(TextField.TYPE_NOT_STORED);
194     ft.setOmitNorms(true);
195     parentStreamField = new Field(Consts.FIELD_PAYLOADS, parentStream, ft);
196     fullPathField = new StringField(Consts.FULL, "", Field.Store.YES);
197 
198     nextID = indexWriter.maxDoc();
199 
200     if (cache == null) {
201       cache = defaultTaxonomyWriterCache();
202     }
203     this.cache = cache;
204 
205     if (nextID == 0) {
206       cacheIsComplete = true;
207       // Make sure that the taxonomy always contain the root category
208       // with category id 0.
209       addCategory(new FacetLabel());
210     } else {
211       // There are some categories on the disk, which we have not yet
212       // read into the cache, and therefore the cache is incomplete.
213       // We choose not to read all the categories into the cache now,
214       // to avoid terrible performance when a taxonomy index is opened
215       // to add just a single category. We will do it later, after we
216       // notice a few cache misses.
217       cacheIsComplete = false;
218     }
219   }

243   /**
244    * Create the {@link IndexWriterConfig} that would be used for opening the internal index writer.
245    * <br>Extensions can configure the {@link IndexWriter} as they see fit,
246    * including setting a {@link org.apache.lucene.index.MergeScheduler merge-scheduler}, or
247    * {@link org.apache.lucene.index.IndexDeletionPolicy deletion-policy}, different RAM size
248    * etc.<br>
249    * <br><b>NOTE:</b> internal docids of the configured index must not be altered.
250    * For that, categories are never deleted from the taxonomy index.
251    * In addition, merge policy in effect must not merge none adjacent segments.
252    * 
253    * @see #openIndexWriter(Directory, IndexWriterConfig)
254    * 
255    * @param openMode see {@link OpenMode}
256    */
257   protected IndexWriterConfig createIndexWriterConfig(OpenMode openMode) {
258     // TODO: should we use a more optimized Codec?
259     // The taxonomy has a unique structure, where each term is associated with one document
260 
261     // Make sure we use a MergePolicy which always merges adjacent segments and thus
262     // keeps the doc IDs ordered as well (this is crucial for the taxonomy index).
263     return new IndexWriterConfig(null).setOpenMode(openMode).setMergePolicy(
264         new LogByteSizeMergePolicy());
265   }

267   /** Opens a {@link ReaderManager} from the internal {@link IndexWriter}. */
268   private void initReaderManager() throws IOException {
269     if (!initializedReaderManager) {
270       synchronized (this) {
271         // verify that the taxo-writer hasn't been closed on us.
272         ensureOpen();
273         if (!initializedReaderManager) {
274           readerManager = new ReaderManager(indexWriter, false);
275           shouldRefreshReaderManager = false;
276           initializedReaderManager = true;
277         }
278       }
279     }
280   }

328    * A hook for extending classes to close additional resources that were used.
329    * The default implementation closes the {@link IndexReader} as well as the
330    * {@link TaxonomyWriterCache} instances that were used. <br>
331    * <b>NOTE:</b> if you override this method, you should include a

334   protected synchronized void closeResources() throws IOException {
335     if (initializedReaderManager) {
336       readerManager.close();
337       readerManager = null;
338       initializedReaderManager = false;
339     }
340     if (cache != null) {
341       cache.close();
342     }

345   /**
346    * Look up the given category in the cache and/or the on-disk storage,
347    * returning the category's ordinal, or a negative number in case the
348    * category does not yet exist in the taxonomy.
349    */
350   protected synchronized int findCategory(FacetLabel categoryPath) throws IOException {
351     // If we can find the category in the cache, or we know the cache is
352     // complete, we can return the response directly from it
353     int res = cache.get(categoryPath);
354     if (res >= 0 || cacheIsComplete) {
355       return res;
356     }
357 
358     cacheMisses.incrementAndGet();
359     // After a few cache misses, it makes sense to read all the categories
360     // from disk and into the cache. The reason not to do this on the first
361     // cache miss (or even when opening the writer) is that it will
362     // significantly slow down the case when a taxonomy is opened just to
363     // add one category. The idea only spending a long time on reading
364     // after enough time was spent on cache misses is known as an "online
365     // algorithm".
366     perhapsFillCache();
367     res = cache.get(categoryPath);
368     if (res >= 0 || cacheIsComplete) {
369       // if after filling the cache from the info on disk, the category is in it
370       // or the cache is complete, return whatever cache.get returned.
371       return res;
372     }
373 
374     // if we get here, it means the category is not in the cache, and it is not
375     // complete, and therefore we must look for the category on disk.
376     
377     // We need to get an answer from the on-disk index.
378     initReaderManager();
379 
380     int doc = -1;
381     DirectoryReader reader = readerManager.acquire();
382     try {
383       final BytesRef catTerm = new BytesRef(FacetsConfig.pathToString(categoryPath.components, categoryPath.length));
384       PostingsEnum docs = null; // reuse
385       for (LeafReaderContext ctx : reader.leaves()) {
386         Terms terms = ctx.reader().terms(Consts.FULL);
387         if (terms != null) {
388           // TODO: share per-segment TermsEnum here!
389           TermsEnum termsEnum = terms.iterator();
390           if (termsEnum.seekExact(catTerm)) {
391             // liveDocs=null because the taxonomy has no deletes
392             docs = termsEnum.postings(docs, 0 /* freqs not required */);
393             // if the term was found, we know it has exactly one document.
394             doc = docs.nextDoc() + ctx.docBase;
395             break;
396           }
397         }
398       }
399     } finally {
400       readerManager.release(reader);
401     }
402     if (doc > 0) {
403       addToCache(categoryPath, doc);
404     }
405     return doc;
406   }

408   @Override
409   public int addCategory(FacetLabel categoryPath) throws IOException {
410     ensureOpen();
411     // check the cache outside the synchronized block. this results in better
412     // concurrency when categories are there.
413     int res = cache.get(categoryPath);
414     if (res < 0) {
415       // the category is not in the cache - following code cannot be executed in parallel.
416       synchronized (this) {
417         res = findCategory(categoryPath);
418         if (res < 0) {
419           // This is a new category, and we need to insert it into the index
420           // (and the cache). Actually, we might also need to add some of
421           // the category's ancestors before we can add the category itself
422           // (while keeping the invariant that a parent is always added to
423           // the taxonomy before its child). internalAddCategory() does all
424           // this recursively
425           res = internalAddCategory(categoryPath);
426         }
427       }
428     }
429     return res;
430   }
431 
432   /**
433    * Add a new category into the index (and the cache), and return its new
434    * ordinal.
435    * <p>
436    * Actually, we might also need to add some of the category's ancestors
437    * before we can add the category itself (while keeping the invariant that a
438    * parent is always added to the taxonomy before its child). We do this by
439    * recursion.
440    */
441   private int internalAddCategory(FacetLabel cp) throws IOException {
442     // Find our parent's ordinal (recursively adding the parent category
443     // to the taxonomy if it's not already there). Then add the parent
444     // ordinal as payloads (rather than a stored field; payloads can be
445     // more efficiently read into memory in bulk by LuceneTaxonomyReader)
446     int parent;
447     if (cp.length > 1) {
448       FacetLabel parentPath = cp.subpath(cp.length - 1);
449       parent = findCategory(parentPath);
450       if (parent < 0) {
451         parent = internalAddCategory(parentPath);
452       }
453     } else if (cp.length == 1) {
454       parent = TaxonomyReader.ROOT_ORDINAL;
455     } else {
456       parent = TaxonomyReader.INVALID_ORDINAL;
457     }
458     int id = addCategoryDocument(cp, parent);
459 
460     return id;
461   }

473   /**
474    * Note that the methods calling addCategoryDocument() are synchornized, so
475    * this method is effectively synchronized as well.
476    */
477   private int addCategoryDocument(FacetLabel categoryPath, int parent) throws IOException {
478     // Before Lucene 2.9, position increments >=0 were supported, so we
479     // added 1 to parent to allow the parent -1 (the parent of the root).
480     // Unfortunately, starting with Lucene 2.9, after LUCENE-1542, this is
481     // no longer enough, since 0 is not encoded consistently either (see
482     // comment in SinglePositionTokenStream). But because we must be
483     // backward-compatible with existing indexes, we can't just fix what
484     // we write here (e.g., to write parent+2), and need to do a workaround
485     // in the reader (which knows that anyway only category 0 has a parent
486     // -1).    
487     parentStream.set(Math.max(parent + 1, 1));
488     Document d = new Document();
489     d.add(parentStreamField);
490 
491     fullPathField.setStringValue(FacetsConfig.pathToString(categoryPath.components, categoryPath.length));
492     d.add(fullPathField);
493 
494     // Note that we do no pass an Analyzer here because the fields that are
495     // added to the Document are untokenized or contains their own TokenStream.
496     // Therefore the IndexWriter's Analyzer has no effect.
497     indexWriter.addDocument(d);
498     int id = nextID++;
499 
500     // added a category document, mark that ReaderManager is not up-to-date
501     shouldRefreshReaderManager = true;
502     
503     // also add to the parent array
504     taxoArrays = getTaxoArrays().add(id, parent);
505 
506     // NOTE: this line must be executed last, or else the cache gets updated
507     // before the parents array (LUCENE-4596)
508     addToCache(categoryPath, id);
509 
510     return id;
511   }

513   private static class SinglePositionTokenStream extends TokenStream {
514     private CharTermAttribute termAtt;
515     private PositionIncrementAttribute posIncrAtt;
516     private boolean returned;
517     private int val;
518     private final String word;
519 
520     public SinglePositionTokenStream(String word) {
521       termAtt = addAttribute(CharTermAttribute.class);
522       posIncrAtt = addAttribute(PositionIncrementAttribute.class);
523       this.word = word;
524       returned = true;
525     }
526 
527     /**
528      * Set the value we want to keep, as the position increment.
529      * Note that when TermPositions.nextPosition() is later used to
530      * retrieve this value, val-1 will be returned, not val.
531      * <P>
532      * IMPORTANT NOTE: Before Lucene 2.9, val&gt;=0 were safe (for val==0,
533      * the retrieved position would be -1). But starting with Lucene 2.9,
534      * this unfortunately changed, and only val&gt;0 are safe. val=0 can
535      * still be used, but don't count on the value you retrieve later
536      * (it could be 0 or -1, depending on circumstances or versions).
537      * This change is described in Lucene's JIRA: LUCENE-1542. 
538      */
539     public void set(int val) {
540       this.val = val;
541       returned = false;
542     }
543 
544     @Override
545     public boolean incrementToken() throws IOException {
546       if (returned) {
547         return false;
548       }
549       clearAttributes();
550       posIncrAtt.setPositionIncrement(val);
551       termAtt.setEmpty();
552       termAtt.append(word);
553       returned = true;
554       return true;
555     }
556   }
557 
558   private void addToCache(FacetLabel categoryPath, int id) throws IOException {
559     if (cache.put(categoryPath, id)) {
560       // If cache.put() returned true, it means the cache was limited in
561       // size, became full, and parts of it had to be evicted. It is
562       // possible that a relatively-new category that isn't yet visible
563       // to our 'reader' was evicted, and therefore we must now refresh 
564       // the reader.
565       refreshReaderManager();
566       cacheIsComplete = false;
567     }
568   }
569 
570   private synchronized void refreshReaderManager() throws IOException {
571     // this method is synchronized since it cannot happen concurrently with
572     // addCategoryDocument -- when this method returns, we must know that the
573     // reader manager's state is current. also, it sets shouldRefresh to false, 
574     // and this cannot overlap with addCatDoc too.
575     // NOTE: since this method is sync'ed, it can call maybeRefresh, instead of
576     // maybeRefreshBlocking. If ever this is changed, make sure to change the
577     // call too.
578     if (shouldRefreshReaderManager && initializedReaderManager) {
579       readerManager.maybeRefresh();
580       shouldRefreshReaderManager = false;
581     }
582   }

584   @Override
585   public synchronized void commit() throws IOException {
586     ensureOpen();
587     // LUCENE-4972: if we always call setCommitData, we create empty commits
588     String epochStr = indexWriter.getCommitData().get(INDEX_EPOCH);
589     if (epochStr == null || Long.parseLong(epochStr, 16) != indexEpoch) {
590       indexWriter.setCommitData(combinedCommitData(indexWriter.getCommitData()));
591     }
592     indexWriter.commit();
593   }
594 
595   /** Combine original user data with the taxonomy epoch. */
596   private Map<String,String> combinedCommitData(Map<String,String> commitData) {
597     Map<String,String> m = new HashMap<>();
598     if (commitData != null) {
599       m.putAll(commitData);
600     }
601     m.put(INDEX_EPOCH, Long.toString(indexEpoch, 16));
602     return m;
603   }

608   }

636   /**
637    * Set the number of cache misses before an attempt is made to read the entire
638    * taxonomy into the in-memory cache.
639    * <p>
640    * This taxonomy writer holds an in-memory cache of recently seen categories
641    * to speed up operation. On each cache-miss, the on-disk index needs to be
642    * consulted. When an existing taxonomy is opened, a lot of slow disk reads
643    * like that are needed until the cache is filled, so it is more efficient to
644    * read the entire taxonomy into memory at once. We do this complete read
645    * after a certain number (defined by this method) of cache misses.
646    * <p>
647    * If the number is set to {@code 0}, the entire taxonomy is read into the
648    * cache on first use, without fetching individual categories first.
649    * <p>
650    * NOTE: it is assumed that this method is called immediately after the
651    * taxonomy writer has been created.
652    */
653   public void setCacheMissesUntilFill(int i) {
654     ensureOpen();
655     cacheMissesUntilFill = i;
656   }
657   
658   // we need to guarantee that if several threads call this concurrently, only
659   // one executes it, and after it returns, the cache is updated and is either
660   // complete or not.
661   private synchronized void perhapsFillCache() throws IOException {
662     if (cacheMisses.get() < cacheMissesUntilFill) {
663       return;
664     }
665     
666     if (!shouldFillCache) {
667       // we already filled the cache once, there's no need to re-fill it
668       return;
669     }
670     shouldFillCache = false;
671     
672     initReaderManager();
673 
674     boolean aborted = false;
675     DirectoryReader reader = readerManager.acquire();
676     try {
677       PostingsEnum postingsEnum = null;
678       for (LeafReaderContext ctx : reader.leaves()) {
679         Terms terms = ctx.reader().terms(Consts.FULL);
680         if (terms != null) { // cannot really happen, but be on the safe side
681           // TODO: share per-segment TermsEnum here!
682           TermsEnum termsEnum = terms.iterator();
683           while (termsEnum.next() != null) {
684             if (!cache.isFull()) {
685               BytesRef t = termsEnum.term();
686               // Since we guarantee uniqueness of categories, each term has exactly
687               // one document. Also, since we do not allow removing categories (and
688               // hence documents), there are no deletions in the index. Therefore, it
689               // is sufficient to call next(), and then doc(), exactly once with no
690               // 'validation' checks.
691               FacetLabel cp = new FacetLabel(FacetsConfig.stringToPath(t.utf8ToString()));
692               postingsEnum = termsEnum.postings(postingsEnum, PostingsEnum.NONE);
693               boolean res = cache.put(cp, postingsEnum.nextDoc() + ctx.docBase);
694               assert !res : "entries should not have been evicted from the cache";
695             } else {
696               // the cache is full and the next put() will evict entries from it, therefore abort the iteration.
697               aborted = true;
698               break;
699             }
700           }
701         }
702         if (aborted) {
703           break;
704         }
705       }
706     } finally {
707       readerManager.release(reader);
708     }
709 
710     cacheIsComplete = !aborted;
711     if (cacheIsComplete) {
712       synchronized (this) {
713         // everything is in the cache, so no need to keep readerManager open.
714         // this block is executed in a sync block so that it works well with
715         // initReaderManager called in parallel.
716         readerManager.close();
717         readerManager = null;
718         initializedReaderManager = false;
719       }
720     }
721   }

723   private TaxonomyIndexArrays getTaxoArrays() throws IOException {
724     if (taxoArrays == null) {
725       synchronized (this) {
726         if (taxoArrays == null) {
727           initReaderManager();
728           DirectoryReader reader = readerManager.acquire();
729           try {
730             // according to Java Concurrency, this might perform better on some
731             // JVMs, since the object initialization doesn't happen on the
732             // volatile member.
733             TaxonomyIndexArrays tmpArrays = new TaxonomyIndexArrays(reader);
734             taxoArrays = tmpArrays;
735           } finally {
736             readerManager.release(reader);
737           }
738         }
739       }
740     }
741     return taxoArrays;
742   }

744   @Override
745   public int getParent(int ordinal) throws IOException {
746     ensureOpen();
747     // Note: the following if() just enforces that a user can never ask
748     // for the parent of a nonexistant category - even if the parent array
749     // was allocated bigger than it really needs to be.
750     if (ordinal >= nextID) {
751       throw new ArrayIndexOutOfBoundsException("requested ordinal is bigger than the largest ordinal in the taxonomy");
752     }
753     
754     int[] parents = getTaxoArrays().parents();
755     assert ordinal < parents.length : "requested ordinal (" + ordinal + "); parents.length (" + parents.length + ") !";
756     return parents[ordinal];
757   }

759   /**
760    * Takes the categories from the given taxonomy directory, and adds the
761    * missing ones to this taxonomy. Additionally, it fills the given
762    * {@link OrdinalMap} with a mapping from the original ordinal to the new
763    * ordinal.
764    */
765   public void addTaxonomy(Directory taxoDir, OrdinalMap map) throws IOException {
766     ensureOpen();
767     DirectoryReader r = DirectoryReader.open(taxoDir);
768     try {
769       final int size = r.numDocs();
770       final OrdinalMap ordinalMap = map;
771       ordinalMap.setSize(size);
772       int base = 0;
773       PostingsEnum docs = null;
774       for (final LeafReaderContext ctx : r.leaves()) {
775         final LeafReader ar = ctx.reader();
776         final Terms terms = ar.terms(Consts.FULL);
777         // TODO: share per-segment TermsEnum here!
778         TermsEnum te = terms.iterator();
779         while (te.next() != null) {
780           FacetLabel cp = new FacetLabel(FacetsConfig.stringToPath(te.term().utf8ToString()));
781           final int ordinal = addCategory(cp);
782           docs = te.postings(docs, PostingsEnum.NONE);
783           ordinalMap.addMapping(docs.nextDoc() + base, ordinal);
784         }
785         base += ar.maxDoc(); // no deletions, so we're ok
786       }
787       ordinalMap.addDone();
788     } finally {
789       r.close();
790     }
791   }

793   /**
794    * Mapping from old ordinal to new ordinals, used when merging indexes 
795    * wit separate taxonomies.
796    * <p> 
797    * addToTaxonomies() merges one or more taxonomies into the given taxonomy
798    * (this). An OrdinalMap is filled for each of the added taxonomies,
799    * containing the new ordinal (in the merged taxonomy) of each of the
800    * categories in the old taxonomy.
801    * <P>  
802    * There exist two implementations of OrdinalMap: MemoryOrdinalMap and
803    * DiskOrdinalMap. As their names suggest, the former keeps the map in
804    * memory and the latter in a temporary disk file. Because these maps will
805    * later be needed one by one (to remap the counting lists), not all at the
806    * same time, it is recommended to put the first taxonomy's map in memory,
807    * and all the rest on disk (later to be automatically read into memory one
808    * by one, when needed).
809    */
810   public static interface OrdinalMap {
811     /**
812      * Set the size of the map. This MUST be called before addMapping().
813      * It is assumed (but not verified) that addMapping() will then be
814      * called exactly 'size' times, with different origOrdinals between 0
815      * and size-1.  
816      */
817     public void setSize(int size) throws IOException;
818 
819     /** Record a mapping. */
820     public void addMapping(int origOrdinal, int newOrdinal) throws IOException;
821 
822     /**
823      * Call addDone() to say that all addMapping() have been done.
824      * In some implementations this might free some resources. 
825      */
826     public void addDone() throws IOException;
827 
828     /**
829      * Return the map from the taxonomy's original (consecutive) ordinals
830      * to the new taxonomy's ordinals. If the map has to be read from disk
831      * and ordered appropriately, it is done when getMap() is called.
832      * getMap() should only be called once, and only when the map is actually
833      * needed. Calling it will also free all resources that the map might
834      * be holding (such as temporary disk space), other than the returned int[].
835      */
836     public int[] getMap() throws IOException;
837   }
838 
839   /**
840    * {@link OrdinalMap} maintained in memory
841    */
842   public static final class MemoryOrdinalMap implements OrdinalMap {
843     int[] map;
844 
845     /** Sole constructor. */
846     public MemoryOrdinalMap() {
847     }
848 
849     @Override
850     public void setSize(int taxonomySize) {
851       map = new int[taxonomySize];
852     }
853     @Override
854     public void addMapping(int origOrdinal, int newOrdinal) {
855       map[origOrdinal] = newOrdinal;
856     }
857     @Override
858     public void addDone() { /* nothing to do */ }
859     @Override
860     public int[] getMap() {
861       return map;
862     }
863   }
864 
865   /**
866    * {@link OrdinalMap} maintained on file system
867    */
868   public static final class DiskOrdinalMap implements OrdinalMap {
869     Path tmpfile;
870     DataOutputStream out;
871 
872     /** Sole constructor. */
873     public DiskOrdinalMap(Path tmpfile) throws IOException {
874       this.tmpfile = tmpfile;
875       out = new DataOutputStream(new BufferedOutputStream(
876           Files.newOutputStream(tmpfile)));
877     }
878 
879     @Override
880     public void addMapping(int origOrdinal, int newOrdinal) throws IOException {
881       out.writeInt(origOrdinal);
882       out.writeInt(newOrdinal);
883     }
884 
885     @Override
886     public void setSize(int taxonomySize) throws IOException {
887       out.writeInt(taxonomySize);
888     }
889 
890     @Override
891     public void addDone() throws IOException {
892       if (out!=null) {
893         out.close();
894         out = null;
895       }
896     }
897 
898     int[] map = null;
899 
900     @Override
901     public int[] getMap() throws IOException {
902       if (map!=null) {
903         return map;
904       }
905       addDone(); // in case this wasn't previously called
906       DataInputStream in = new DataInputStream(new BufferedInputStream(
907           Files.newInputStream(tmpfile)));
908       map = new int[in.readInt()];
909       // NOTE: The current code assumes here that the map is complete,
910       // i.e., every ordinal gets one and exactly one value. Otherwise,
911       // we may run into an EOF here, or vice versa, not read everything.
912       for (int i=0; i<map.length; i++) {
913         int origordinal = in.readInt();
914         int newordinal = in.readInt();
915         map[origordinal] = newordinal;
916       }
917       in.close();
918 
919       // Delete the temporary file, which is no longer needed.
920       Files.delete(tmpfile);
921 
922       return map;
923     }
924   }

938   /**
939    * Replaces the current taxonomy with the given one. This method should
940    * generally be called in conjunction with
941    * {@link IndexWriter#addIndexes(Directory...)} to replace both the taxonomy
942    * as well as the search index content.
943    */
944   public synchronized void replaceTaxonomy(Directory taxoDir) throws IOException {
945     // replace the taxonomy by doing IW optimized operations
946     indexWriter.deleteAll();
947     indexWriter.addIndexes(taxoDir);
948     shouldRefreshReaderManager = true;
949     initReaderManager(); // ensure that it's initialized
950     refreshReaderManager();
951     nextID = indexWriter.maxDoc();
952     taxoArrays = null; // must nullify so that it's re-computed next time it's needed
953     
954     // need to clear the cache, so that addCategory won't accidentally return
955     // old categories that are in the cache.
956     cache.clear();
957     cacheIsComplete = false;
958     shouldFillCache = true;
959     cacheMisses.set(0);
960     
961     // update indexEpoch as a taxonomy replace is just like it has be recreated
962     ++indexEpoch;
963   }
