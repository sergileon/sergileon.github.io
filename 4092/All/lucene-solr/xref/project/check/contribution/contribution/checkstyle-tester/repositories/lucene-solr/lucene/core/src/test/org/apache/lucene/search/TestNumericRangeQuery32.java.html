146   /** test for both constant score and boolean query, the other tests only use the constant score mode */
147   private void testRange(int precisionStep) throws Exception {
148     String field="field"+precisionStep;
149     int count=3000;
150     int lower=(distance*3/2)+startOffset, upper=lower + count*distance + (distance/3);
151     NumericRangeQuery<Integer> q = NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, true, true);
152     for (byte i=0; i<2; i++) {
153       TopDocs topDocs;
154       String type;
155       switch (i) {
156         case 0:
157           type = " (constant score filter rewrite)";
158           q.setRewriteMethod(MultiTermQuery.CONSTANT_SCORE_REWRITE);
159           topDocs = searcher.search(q, noDocs, Sort.INDEXORDER);
160           break;
161         case 1:
162           type = " (constant score boolean rewrite)";
163           q.setRewriteMethod(MultiTermQuery.CONSTANT_SCORE_BOOLEAN_REWRITE);
164           topDocs = searcher.search(q, noDocs, Sort.INDEXORDER);
165           break;
166         default:
167           return;
168       }
169       ScoreDoc[] sd = topDocs.scoreDocs;
170       assertNotNull(sd);
171       assertEquals("Score doc count"+type, count, sd.length );
172       Document doc=searcher.doc(sd[0].doc);
173       assertEquals("First doc"+type, 2*distance+startOffset, doc.getField(field).numericValue().intValue());
174       doc=searcher.doc(sd[sd.length-1].doc);
175       assertEquals("Last doc"+type, (1+count)*distance+startOffset, doc.getField(field).numericValue().intValue());
176     }
177   }

354   private void testRandomTrieAndClassicRangeQuery(int precisionStep) throws Exception {
355     String field="field"+precisionStep;
356     int totalTermCountT=0,totalTermCountC=0,termCountT,termCountC;
357     int num = TestUtil.nextInt(random(), 10, 20);
358     for (int i = 0; i < num; i++) {
359       int lower=(int)(random().nextDouble()*noDocs*distance)+startOffset;
360       int upper=(int)(random().nextDouble()*noDocs*distance)+startOffset;
361       if (lower>upper) {
362         int a=lower; lower=upper; upper=a;
363       }
364       final BytesRef lowerBytes, upperBytes;
365       BytesRefBuilder b = new BytesRefBuilder();
366       NumericUtils.intToPrefixCodedBytes(lower, 0, b);
367       lowerBytes = b.toBytesRef();
368       NumericUtils.intToPrefixCodedBytes(upper, 0, b);
369       upperBytes = b.toBytesRef();
370 
371       // test inclusive range
372       NumericRangeQuery<Integer> tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, true, true);
373       TermRangeQuery cq=new TermRangeQuery(field, lowerBytes, upperBytes, true, true);
374       TopDocs tTopDocs = searcher.search(tq, 1);
375       TopDocs cTopDocs = searcher.search(cq, 1);
376       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
377       totalTermCountT += termCountT = countTerms(tq);
378       totalTermCountC += termCountC = countTerms(cq);
379       checkTermCounts(precisionStep, termCountT, termCountC);
380       // test exclusive range
381       tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, false, false);
382       cq=new TermRangeQuery(field, lowerBytes, upperBytes, false, false);
383       tTopDocs = searcher.search(tq, 1);
384       cTopDocs = searcher.search(cq, 1);
385       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
386       totalTermCountT += termCountT = countTerms(tq);
387       totalTermCountC += termCountC = countTerms(cq);
388       checkTermCounts(precisionStep, termCountT, termCountC);
389       // test left exclusive range
390       tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, false, true);
391       cq=new TermRangeQuery(field, lowerBytes, upperBytes, false, true);
392       tTopDocs = searcher.search(tq, 1);
393       cTopDocs = searcher.search(cq, 1);
394       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
395       totalTermCountT += termCountT = countTerms(tq);
396       totalTermCountC += termCountC = countTerms(cq);
397       checkTermCounts(precisionStep, termCountT, termCountC);
398       // test right exclusive range
399       tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, true, false);
400       cq=new TermRangeQuery(field, lowerBytes, upperBytes, true, false);
401       tTopDocs = searcher.search(tq, 1);
402       cTopDocs = searcher.search(cq, 1);
403       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
404       totalTermCountT += termCountT = countTerms(tq);
405       totalTermCountC += termCountC = countTerms(cq);
406       checkTermCounts(precisionStep, termCountT, termCountC);
407     }
408     
409     checkTermCounts(precisionStep, totalTermCountT, totalTermCountC);
410     if (VERBOSE && precisionStep != Integer.MAX_VALUE) {
411       System.out.println("Average number of terms during random search on '" + field + "':");
412       System.out.println(" Numeric query: " + (((double)totalTermCountT)/(num * 4)));
413       System.out.println(" Classical query: " + (((double)totalTermCountC)/(num * 4)));
414     }
415   }

432   private int countTerms(MultiTermQuery q) throws Exception {
433     final Terms terms = MultiFields.getTerms(reader, q.getField());
434     if (terms == null)
435       return 0;
436     final TermsEnum termEnum = q.getTermsEnum(terms);
437     assertNotNull(termEnum);
438     int count = 0;
439     BytesRef cur, last = null;
440     while ((cur = termEnum.next()) != null) {
441       count++;
442       if (last != null) {
443         assertTrue(last.compareTo(cur) < 0);
444       }
445       last = BytesRef.deepCopyOf(cur);
446     } 
447     // LUCENE-3314: the results after next() already returned null are undefined,
448     // assertNull(termEnum.next());
449     return count;
450   }

452   private void checkTermCounts(int precisionStep, int termCountT, int termCountC) {
453     if (precisionStep == Integer.MAX_VALUE) {
454       assertEquals("Number of terms should be equal for unlimited precStep", termCountC, termCountT);
455     } else {
456       assertTrue("Number of terms for NRQ should be <= compared to classical TRQ", termCountT <= termCountC);
457     }
458   }

480   private void testRangeSplit(int precisionStep) throws Exception {
481     String field="ascfield"+precisionStep;
482     // 10 random tests
483     int num = TestUtil.nextInt(random(), 10, 20);
484     for (int  i =0;  i< num; i++) {
485       int lower=(int)(random().nextDouble()*noDocs - noDocs/2);
486       int upper=(int)(random().nextDouble()*noDocs - noDocs/2);
487       if (lower>upper) {
488         int a=lower; lower=upper; upper=a;
489       }
490       // test inclusive range
491       Query tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, true, true);
492       TopDocs tTopDocs = searcher.search(tq, 1);
493       assertEquals("Returned count of range query must be equal to inclusive range length", upper-lower+1, tTopDocs.totalHits );
494       // test exclusive range
495       tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, false, false);
496       tTopDocs = searcher.search(tq, 1);
497       assertEquals("Returned count of range query must be equal to exclusive range length", Math.max(upper-lower-1, 0), tTopDocs.totalHits );
498       // test left exclusive range
499       tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, false, true);
500       tTopDocs = searcher.search(tq, 1);
501       assertEquals("Returned count of range query must be equal to half exclusive range length", upper-lower, tTopDocs.totalHits );
502       // test right exclusive range
503       tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, true, false);
504       tTopDocs = searcher.search(tq, 1);
505       assertEquals("Returned count of range query must be equal to half exclusive range length", upper-lower, tTopDocs.totalHits );
506     }
507   }
