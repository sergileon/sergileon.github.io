168   int index(SolrClient s, Object... fields) throws Exception {
169     SolrInputDocument doc = new SolrInputDocument();
170     for (int i = 0; i < fields.length; i += 2) {
171       doc.addField((String) (fields[i]), fields[i + 1]);
172     }
173     return s.add(doc).getStatus();

177     ModifiableSolrParams params = new ModifiableSolrParams();

180     params.add("sort","id desc");

182     QueryResponse qres = s.query(params);

186   /** will sleep up to 30 seconds, looking for expectedDocCount */
187   private NamedList rQuery(int expectedDocCount, String query, SolrClient client) throws Exception {
188     int timeSlept = 0;
189     NamedList res = query(query, client);
190     while (expectedDocCount != numFound(res)
191            && timeSlept < 30000) {
192       log.info("Waiting for " + expectedDocCount + " docs");
193       timeSlept += 100;
194       Thread.sleep(100);
195       res = query(query, client);
196     }
197     log.info("Waited for {}ms and found {} docs", timeSlept, numFound(res));
198     return res;
199   }

348   /**
349    * Verify that empty commits and/or commits with openSearcher=false
350    * on the master do not cause subsequent replication problems on the slave 
351    */
352   public void testEmptyCommits() throws Exception {
353     clearIndexWithReplication();
354     
355     // add a doc to master and commit
356     index(masterClient, "id", "1", "name", "empty1");
357     emptyUpdate(masterClient, "commit", "true");
358     // force replication
359     pullFromMasterToSlave();
360     // verify doc is on slave
361     rQuery(1, "name:empty1", slaveClient);
362     assertVersions(masterClient, slaveClient);
363 
364     // do a completely empty commit on master and force replication
365     emptyUpdate(masterClient, "commit", "true");
366     pullFromMasterToSlave();
367 
368     // add another doc and verify slave gets it
369     index(masterClient, "id", "2", "name", "empty2");
370     emptyUpdate(masterClient, "commit", "true");
371     // force replication
372     pullFromMasterToSlave();
373 
374     rQuery(1, "name:empty2", slaveClient);
375     assertVersions(masterClient, slaveClient);
376 
377     // add a third doc but don't open a new searcher on master
378     index(masterClient, "id", "3", "name", "empty3");
379     emptyUpdate(masterClient, "commit", "true", "openSearcher", "false");
380     pullFromMasterToSlave();
381     
382     // verify slave can search the doc, but master doesn't
383     rQuery(0, "name:empty3", masterClient);
384     rQuery(1, "name:empty3", slaveClient);
385 
386     // final doc with hard commit, slave and master both showing all docs
387     index(masterClient, "id", "4", "name", "empty4");
388     emptyUpdate(masterClient, "commit", "true");
389     pullFromMasterToSlave();
390 
391     String q = "name:(empty1 empty2 empty3 empty4)";
392     rQuery(4, q, masterClient);
393     rQuery(4, q, slaveClient);
394     assertVersions(masterClient, slaveClient);
395 
396   }

926     params.set("qt", "/replication");
927     params.set("_trace", "assertVersions");
928     params.set("command", "indexversion");
929     QueryRequest req = new QueryRequest(params);
930     NamedList<Object> resp = client1.request(req);
931     
932     Long version = (Long) resp.get("indexversion");
933     assertEquals(maxVersionClient1, version);
934     
935     // check vs /replication?command=indexversion call

941   private Long getVersion(SolrClient client) throws Exception {
942     NamedList<Object> details;
943     ArrayList<NamedList<Object>> commits;
944     details = getDetails(client);
945     commits = (ArrayList<NamedList<Object>>) details.get("commits");
946     Long maxVersionSlave= 0L;
947     for(NamedList<Object> commit : commits) {
948       Long version = (Long) commit.get("indexVersion");
949       maxVersionSlave = Math.max(version, maxVersionSlave);

951     return maxVersionSlave;

1260     //Start master with the new solrconfig
1261     master.copyConfigFile(CONF_DIR + "solrconfig-master-throttled.xml", "solrconfig.xml");
1262     useFactory(null);
1263     masterJetty = createJetty(master);
1264     masterClient.close();
1265     masterClient = createNewSolrClient(masterJetty.getLocalPort());
1266 
1267     //index docs
1268     final int totalDocs = TestUtil.nextInt(random(), 50, 100);
1269     for (int i = 0; i < totalDocs; i++)
1270       index(masterClient, "id", i, "name", TestUtil.randomSimpleString(random(), 1000 , 5000));
1271 
1272     masterClient.commit();
1273 
1274     //Check Index Size
1275     String dataDir = master.getDataDir();
1276     masterClient.close();
1277     masterJetty.stop();
1278 
1279     Directory dir = FSDirectory.open(Paths.get(dataDir).resolve("index"));
1280     String[] files = dir.listAll();
1281     long totalBytes = 0;
1282     for(String file : files) {
1283       totalBytes += dir.fileLength(file);
1284     }
1285 
1286     float approximateTimeInSeconds = Math.round( totalBytes/1024/1024/0.1 ); // maxWriteMBPerSec=0.1 in solrconfig
1287 
1288     //Start again and replicate the data
1289     useFactory(null);
1290     masterJetty = createJetty(master);
1291     masterClient = createNewSolrClient(masterJetty.getLocalPort());
1292 
1293     //start slave
1294     slave.setTestPort(masterJetty.getLocalPort());
1295     slave.copyConfigFile(CONF_DIR + "solrconfig-slave1.xml", "solrconfig.xml");
1296     slaveJetty = createJetty(slave);
1297     slaveClient.close();
1298     slaveClient = createNewSolrClient(slaveJetty.getLocalPort());
1299 
1300     long startTime = System.nanoTime();
1301 
1302     pullFromMasterToSlave();
1303 
1304     //Add a few more docs in the master. Just to make sure that we are replicating the correct index point
1305     //These extra docs should not get replicated
1306     new Thread(new AddExtraDocs(masterClient, totalDocs)).start();
1307 
1308     //Wait and make sure that it actually replicated correctly.
1309     NamedList slaveQueryRsp = rQuery(totalDocs, "*:*", slaveClient);
1310     SolrDocumentList slaveQueryResult = (SolrDocumentList) slaveQueryRsp.get("response");
1311     assertEquals(totalDocs, slaveQueryResult.getNumFound());
1312 
1313     long timeTaken = System.nanoTime() - startTime;
1314 
1315     long timeTakenInSeconds = TimeUnit.SECONDS.convert(timeTaken, TimeUnit.NANOSECONDS);
1316 
1317     //Let's make sure it took more than approximateTimeInSeconds to make sure that it was throttled
1318     log.info("approximateTimeInSeconds = " + approximateTimeInSeconds + " timeTakenInSeconds = " + timeTakenInSeconds);
1319     assertTrue(timeTakenInSeconds - approximateTimeInSeconds > 0);
1320   }

1350    * character copy of file using UTF-8. If port is non-null, will be substituted any time "TEST_PORT" is found.
1351    */
1352   private static void copyFile(File src, File dst, Integer port, boolean internalCompression) throws IOException {
1353     BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(src), StandardCharsets.UTF_8));
1354     Writer out = new OutputStreamWriter(new FileOutputStream(dst), StandardCharsets.UTF_8);
1355 
1356     for (String line = in.readLine(); null != line; line = in.readLine()) {
1357 
1358       if (null != port)
1359         line = line.replace("TEST_PORT", port.toString());
1360       
1361       line = line.replace("COMPRESSION", internalCompression?"internal":"false");
1362 
1363       out.write(line);
1364     }
1365     in.close();
1366     out.close();
1367   }

1377   /**
1378    * Polls the SolrCore stats using the specified client until the "startTime" 
1379    * time for collection is after the specified "min".  Will loop for 
1380    * at most "timeout" milliseconds before throwing an assertion failure.
1381    * 
1382    * @param client The SolrClient to poll
1383    * @param timeout the max milliseconds to continue polling for
1384    * @param min the startTime value must exceed this value before the method will return, if null this method will return the first startTime value encountered.
1385    * @return the startTime value of collection
1386    */
1387   private Date watchCoreStartAt(SolrClient client, final long timeout,
1388                                 final Date min) throws InterruptedException, IOException, SolrServerException {
1389     final long sleepInterval = 200;
1390     long timeSlept = 0;
1391 
1392     try (HttpSolrClient adminClient = adminClient(client)) {
1393       SolrParams p = params("action", "status", "core", "collection1");
1394       while (timeSlept < timeout) {
1395         QueryRequest req = new QueryRequest(p);
1396         req.setPath("/admin/cores");
1397         try {
1398           NamedList data = adminClient.request(req);
1399           for (String k : new String[]{"status", "collection1"}) {
1400             Object o = data.get(k);
1401             assertNotNull("core status rsp missing key: " + k, o);
1402             data = (NamedList) o;
1403           }
1404           Date startTime = (Date) data.get("startTime");
1405           assertNotNull("core has null startTime", startTime);
1406           if (null == min || startTime.after(min)) {
1407             return startTime;
1408           }
1409         } catch (SolrException e) {
1410           // workarround for SOLR-4668
1411           if (500 != e.code()) {
1412             throw e;
1413           } // else server possibly from the core reload in progress...
1414         }
1415 
1416         timeSlept += sleepInterval;
1417         Thread.sleep(sleepInterval);
1418       }
1419       fail("timed out waiting for collection1 startAt time to exceed: " + min);
1420       return min; // compilation neccessity
1421     }
1422   }
