105   // used by IndexWriterConfig
106   LiveIndexWriterConfig(Analyzer analyzer) {
107     this.analyzer = analyzer;
108     ramBufferSizeMB = IndexWriterConfig.DEFAULT_RAM_BUFFER_SIZE_MB;
109     maxBufferedDocs = IndexWriterConfig.DEFAULT_MAX_BUFFERED_DOCS;
110     maxBufferedDeleteTerms = IndexWriterConfig.DEFAULT_MAX_BUFFERED_DELETE_TERMS;
111     mergedSegmentWarmer = null;
112     delPolicy = new KeepOnlyLastCommitDeletionPolicy();
113     commit = null;
114     useCompoundFile = IndexWriterConfig.DEFAULT_USE_COMPOUND_FILE_SYSTEM;
115     openMode = OpenMode.CREATE_OR_APPEND;
116     similarity = IndexSearcher.getDefaultSimilarity();
117     mergeScheduler = new ConcurrentMergeScheduler();
118     writeLockTimeout = IndexWriterConfig.WRITE_LOCK_TIMEOUT;
119     indexingChain = DocumentsWriterPerThread.defaultIndexingChain;
120     codec = Codec.getDefault();
121     if (codec == null) {
122       throw new NullPointerException();
123     }
124     infoStream = InfoStream.getDefault();
125     mergePolicy = new TieredMergePolicy();
126     flushPolicy = new FlushByRamOrCountsPolicy();
127     readerPooling = IndexWriterConfig.DEFAULT_READER_POOLING;
128     indexerThreadPool = new DocumentsWriterPerThreadPool();
129     perThreadHardLimitMB = IndexWriterConfig.DEFAULT_RAM_PER_THREAD_HARD_LIMIT_MB;
130   }

137   /**
138    * Determines the maximum number of delete-by-term operations that will be
139    * buffered before both the buffered in-memory delete terms and queries are
140    * applied and flushed.
141    * <p>
142    * Disabled by default (writer flushes by RAM usage).
143    * <p>
144    * NOTE: This setting won't trigger a segment flush.
145    * 
146    * <p>
147    * Takes effect immediately, but only the next time a document is added,
148    * updated or deleted. Also, if you only delete-by-query, this setting has no
149    * effect, i.e. delete queries are buffered until the next segment is flushed.
150    * 
151    * @throws IllegalArgumentException
152    *           if maxBufferedDeleteTerms is enabled but smaller than 1
153    * 
154    * @see #setRAMBufferSizeMB
155    */
156   public LiveIndexWriterConfig setMaxBufferedDeleteTerms(int maxBufferedDeleteTerms) {
157     if (maxBufferedDeleteTerms != IndexWriterConfig.DISABLE_AUTO_FLUSH && maxBufferedDeleteTerms < 1) {
158       throw new IllegalArgumentException("maxBufferedDeleteTerms must at least be 1 when enabled");
159     }
160     this.maxBufferedDeleteTerms = maxBufferedDeleteTerms;
161     return this;
162   }

171     return maxBufferedDeleteTerms;

174   /**
175    * Determines the amount of RAM that may be used for buffering added documents
176    * and deletions before they are flushed to the Directory. Generally for
177    * faster indexing performance it's best to flush by RAM usage instead of
178    * document count and use as large a RAM buffer as you can.
179    * <p>
180    * When this is set, the writer will flush whenever buffered documents and
181    * deletions use this much RAM. Pass in
182    * {@link IndexWriterConfig#DISABLE_AUTO_FLUSH} to prevent triggering a flush
183    * due to RAM usage. Note that if flushing by document count is also enabled,
184    * then the flush will be triggered by whichever comes first.
185    * <p>
186    * The maximum RAM limit is inherently determined by the JVMs available
187    * memory. Yet, an {@link IndexWriter} session can consume a significantly
188    * larger amount of memory than the given RAM limit since this limit is just
189    * an indicator when to flush memory resident documents to the Directory.
190    * Flushes are likely happen concurrently while other threads adding documents
191    * to the writer. For application stability the available memory in the JVM
192    * should be significantly larger than the RAM buffer used for indexing.
193    * <p>
194    * <b>NOTE</b>: the account of RAM usage for pending deletions is only
195    * approximate. Specifically, if you delete by Query, Lucene currently has no
196    * way to measure the RAM usage of individual Queries so the accounting will
197    * under-estimate and you should compensate by either calling commit()
198    * periodically yourself, or by using {@link #setMaxBufferedDeleteTerms(int)}
199    * to flush and apply buffered deletes by count instead of RAM usage (for each
200    * buffered delete Query a constant number of bytes is used to estimate RAM
201    * usage). Note that enabling {@link #setMaxBufferedDeleteTerms(int)} will not
202    * trigger any segment flushes.
203    * <p>
204    * <b>NOTE</b>: It's not guaranteed that all memory resident documents are
205    * flushed once this limit is exceeded. Depending on the configured
206    * {@link FlushPolicy} only a subset of the buffered documents are flushed and
207    * therefore only parts of the RAM buffer is released.
208    * <p>
209    * 
210    * The default value is {@link IndexWriterConfig#DEFAULT_RAM_BUFFER_SIZE_MB}.
211    * 
212    * <p>
213    * Takes effect immediately, but only the next time a document is added,
214    * updated or deleted.
215    * 
216    * @see IndexWriterConfig#setRAMPerThreadHardLimitMB(int)
217    * 
218    * @throws IllegalArgumentException
219    *           if ramBufferSize is enabled but non-positive, or it disables
220    *           ramBufferSize when maxBufferedDocs is already disabled
221    */
222   public synchronized LiveIndexWriterConfig setRAMBufferSizeMB(double ramBufferSizeMB) {
223     if (ramBufferSizeMB != IndexWriterConfig.DISABLE_AUTO_FLUSH && ramBufferSizeMB <= 0.0) {
224       throw new IllegalArgumentException("ramBufferSize should be > 0.0 MB when enabled");
225     }
226     if (ramBufferSizeMB == IndexWriterConfig.DISABLE_AUTO_FLUSH
227         && maxBufferedDocs == IndexWriterConfig.DISABLE_AUTO_FLUSH) {
228       throw new IllegalArgumentException("at least one of ramBufferSize and maxBufferedDocs must be enabled");
229     }
230     this.ramBufferSizeMB = ramBufferSizeMB;
231     return this;
232   }
233 
234   /** Returns the value set by {@link #setRAMBufferSizeMB(double)} if enabled. */
235   public double getRAMBufferSizeMB() {
236     return ramBufferSizeMB;
237   }

239   /**
240    * Determines the minimal number of documents required before the buffered
241    * in-memory documents are flushed as a new Segment. Large values generally
242    * give faster indexing.
243    * 
244    * <p>
245    * When this is set, the writer will flush every maxBufferedDocs added
246    * documents. Pass in {@link IndexWriterConfig#DISABLE_AUTO_FLUSH} to prevent
247    * triggering a flush due to number of buffered documents. Note that if
248    * flushing by RAM usage is also enabled, then the flush will be triggered by
249    * whichever comes first.
250    * 
251    * <p>
252    * Disabled by default (writer flushes by RAM usage).
253    * 
254    * <p>
255    * Takes effect immediately, but only the next time a document is added,
256    * updated or deleted.
257    * 
258    * @see #setRAMBufferSizeMB(double)
259    * @throws IllegalArgumentException
260    *           if maxBufferedDocs is enabled but smaller than 2, or it disables
261    *           maxBufferedDocs when ramBufferSize is already disabled
262    */
263   public synchronized LiveIndexWriterConfig setMaxBufferedDocs(int maxBufferedDocs) {
264     if (maxBufferedDocs != IndexWriterConfig.DISABLE_AUTO_FLUSH && maxBufferedDocs < 2) {
265       throw new IllegalArgumentException("maxBufferedDocs must at least be 2 when enabled");
266     }
267     if (maxBufferedDocs == IndexWriterConfig.DISABLE_AUTO_FLUSH
268         && ramBufferSizeMB == IndexWriterConfig.DISABLE_AUTO_FLUSH) {
269       throw new IllegalArgumentException("at least one of ramBufferSize and maxBufferedDocs must be enabled");
270     }
271     this.maxBufferedDocs = maxBufferedDocs;
272     return this;
273   }

286    * Expert: {@link MergePolicy} is invoked whenever there are changes to the
287    * segments in the index. Its role is to select which merges to do, if any,
288    * and return a {@link MergePolicy.MergeSpecification} describing the merges.
289    * It also selects merges to do for forceMerge.
290    * 
291    * <p>

293    * merges already registered by the previous {@link MergePolicy} are not
294    * affected.
295    */

297     if (mergePolicy == null) {
298       throw new IllegalArgumentException("mergePolicy must not be null");
299     }
300     this.mergePolicy = mergePolicy;
301     return this;
302   }

468   @Override
469   public String toString() {
470     StringBuilder sb = new StringBuilder();
471     sb.append("analyzer=").append(analyzer == null ? "null" : analyzer.getClass().getName()).append("\n");
472     sb.append("ramBufferSizeMB=").append(getRAMBufferSizeMB()).append("\n");
473     sb.append("maxBufferedDocs=").append(getMaxBufferedDocs()).append("\n");
474     sb.append("maxBufferedDeleteTerms=").append(getMaxBufferedDeleteTerms()).append("\n");
475     sb.append("mergedSegmentWarmer=").append(getMergedSegmentWarmer()).append("\n");
476     sb.append("delPolicy=").append(getIndexDeletionPolicy().getClass().getName()).append("\n");
477     IndexCommit commit = getIndexCommit();
478     sb.append("commit=").append(commit == null ? "null" : commit).append("\n");
479     sb.append("openMode=").append(getOpenMode()).append("\n");
480     sb.append("similarity=").append(getSimilarity().getClass().getName()).append("\n");
481     sb.append("mergeScheduler=").append(getMergeScheduler()).append("\n");
482     sb.append("default WRITE_LOCK_TIMEOUT=").append(IndexWriterConfig.WRITE_LOCK_TIMEOUT).append("\n");
483     sb.append("writeLockTimeout=").append(getWriteLockTimeout()).append("\n");
484     sb.append("codec=").append(getCodec()).append("\n");
485     sb.append("infoStream=").append(getInfoStream().getClass().getName()).append("\n");
486     sb.append("mergePolicy=").append(getMergePolicy()).append("\n");
487     sb.append("indexerThreadPool=").append(getIndexerThreadPool()).append("\n");
488     sb.append("readerPooling=").append(getReaderPooling()).append("\n");
489     sb.append("perThreadHardLimitMB=").append(getRAMPerThreadHardLimitMB()).append("\n");
490     sb.append("useCompoundFile=").append(getUseCompoundFile()).append("\n");
491     sb.append("commitOnClose=").append(getCommitOnClose()).append("\n");
492     return sb.toString();
493   }
