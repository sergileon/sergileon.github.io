52    static final int BLOCK_SIZE = 16384;
53    
54    // address terms in blocks of 16 terms
55    static final int INTERVAL_SHIFT = 4;
56    static final int INTERVAL_COUNT = 1 << INTERVAL_SHIFT;
57    static final int INTERVAL_MASK = INTERVAL_COUNT - 1;
58    
59    // build reverse index from every 1024th term
60    static final int REVERSE_INTERVAL_SHIFT = 10;
61    static final int REVERSE_INTERVAL_COUNT = 1 << REVERSE_INTERVAL_SHIFT;
62    static final int REVERSE_INTERVAL_MASK = REVERSE_INTERVAL_COUNT - 1;
63    
64    // for conversion from reverse index to block
65    static final int BLOCK_INTERVAL_SHIFT = REVERSE_INTERVAL_SHIFT - INTERVAL_SHIFT;
66    static final int BLOCK_INTERVAL_COUNT = 1 << BLOCK_INTERVAL_SHIFT;
67    static final int BLOCK_INTERVAL_MASK = BLOCK_INTERVAL_COUNT - 1;
68  
69    /** Compressed using packed blocks of ints. */
70    public static final int DELTA_COMPRESSED = 0;
71    /** Compressed by computing the GCD. */
72    public static final int GCD_COMPRESSED = 1;
73    /** Compressed by giving IDs to unique values. */
74    public static final int TABLE_COMPRESSED = 2;
75    /** Compressed with monotonically increasing values */
76    public static final int MONOTONIC_COMPRESSED = 3;
77    
78    /** Uncompressed binary, written directly (fixed length). */
79    public static final int BINARY_FIXED_UNCOMPRESSED = 0;
80    /** Uncompressed binary, written directly (variable length). */
81    public static final int BINARY_VARIABLE_UNCOMPRESSED = 1;
82    /** Compressed binary with shared prefixes */
83    public static final int BINARY_PREFIX_COMPRESSED = 2;
84  
85    /** Standard storage for sorted set values with 1 level of indirection:
86     *  docId -> address -> ord. */
87    public static final int SORTED_WITH_ADDRESSES = 0;
88    /** Single-valued sorted set values, encoded as sorted values, so no level
89     *  of indirection: docId -> ord. */
90    public static final int SORTED_SINGLE_VALUED = 1;

92    IndexOutput data, meta;
93    final int maxDoc;

95    /** expert: Creates a new writer */
96    public Lucene410DocValuesConsumer(SegmentWriteState state, String dataCodec, String dataExtension, String metaCodec, String metaExtension) throws IOException {
97      boolean success = false;
98      try {
99        String dataName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, dataExtension);
100       data = state.directory.createOutput(dataName, state.context);
101       CodecUtil.writeHeader(data, dataCodec, Lucene410DocValuesFormat.VERSION_CURRENT);
102       String metaName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, metaExtension);
103       meta = state.directory.createOutput(metaName, state.context);
104       CodecUtil.writeHeader(meta, metaCodec, Lucene410DocValuesFormat.VERSION_CURRENT);
105       maxDoc = state.segmentInfo.maxDoc();
106       success = true;
107     } finally {
108       if (!success) {
109         IOUtils.closeWhileHandlingException(this);
110       }
111     }
112   }

120   void addNumericField(FieldInfo field, Iterable<Number> values, boolean optimizeStorage) throws IOException {
121     long count = 0;
122     long minValue = Long.MAX_VALUE;
123     long maxValue = Long.MIN_VALUE;
124     long gcd = 0;
125     boolean missing = false;
126     // TODO: more efficient?
127     HashSet<Long> uniqueValues = null;
128     if (optimizeStorage) {
129       uniqueValues = new HashSet<>();
130 
131       for (Number nv : values) {
132         final long v;
133         if (nv == null) {
134           v = 0;
135           missing = true;
136         } else {
137           v = nv.longValue();
138         }
139 
140         if (gcd != 1) {
141           if (v < Long.MIN_VALUE / 2 || v > Long.MAX_VALUE / 2) {
142             // in that case v - minValue might overflow and make the GCD computation return
143             // wrong results. Since these extreme values are unlikely, we just discard
144             // GCD computation for them
145             gcd = 1;
146           } else if (count != 0) { // minValue needs to be set first
147             gcd = MathUtil.gcd(gcd, v - minValue);
148           }
149         }
150 
151         minValue = Math.min(minValue, v);
152         maxValue = Math.max(maxValue, v);
153 
154         if (uniqueValues != null) {
155           if (uniqueValues.add(v)) {
156             if (uniqueValues.size() > 256) {
157               uniqueValues = null;
158             }
159           }
160         }
161 
162         ++count;
163       }
164     } else {
165       for (Number nv : values) {
166         long v = nv.longValue();
167         minValue = Math.min(minValue, v);
168         maxValue = Math.max(maxValue, v);
169         ++count;
170       }
171     }
172     
173     final long delta = maxValue - minValue;
174     final int deltaBitsRequired = DirectWriter.unsignedBitsRequired(delta);
175     final int tableBitsRequired = uniqueValues == null
176         ? Integer.MAX_VALUE
177         : DirectWriter.bitsRequired(uniqueValues.size() - 1);
178 
179     final int format;
180     if (uniqueValues != null && tableBitsRequired < deltaBitsRequired) {
181       format = TABLE_COMPRESSED;
182     } else if (gcd != 0 && gcd != 1) {
183       final long gcdDelta = (maxValue - minValue) / gcd;
184       final long gcdBitsRequired = DirectWriter.unsignedBitsRequired(gcdDelta);
185       format = gcdBitsRequired < deltaBitsRequired ? GCD_COMPRESSED : DELTA_COMPRESSED;
186     } else {
187       format = DELTA_COMPRESSED;
188     }
189     meta.writeVInt(field.number);
190     meta.writeByte(Lucene410DocValuesFormat.NUMERIC);
191     meta.writeVInt(format);
192     if (missing) {
193       meta.writeLong(data.getFilePointer());
194       writeMissingBitset(values);
195     } else {
196       meta.writeLong(-1L);
197     }
198     meta.writeLong(data.getFilePointer());
199     meta.writeVLong(count);
200 
201     switch (format) {
202       case GCD_COMPRESSED:
203         meta.writeLong(minValue);
204         meta.writeLong(gcd);
205         final long maxDelta = (maxValue - minValue) / gcd;
206         final int bits = DirectWriter.unsignedBitsRequired(maxDelta);
207         meta.writeVInt(bits);
208         final DirectWriter quotientWriter = DirectWriter.getInstance(data, count, bits);
209         for (Number nv : values) {
210           long value = nv == null ? 0 : nv.longValue();
211           quotientWriter.add((value - minValue) / gcd);
212         }
213         quotientWriter.finish();
214         break;
215       case DELTA_COMPRESSED:
216         final long minDelta = delta < 0 ? 0 : minValue;
217         meta.writeLong(minDelta);
218         meta.writeVInt(deltaBitsRequired);
219         final DirectWriter writer = DirectWriter.getInstance(data, count, deltaBitsRequired);
220         for (Number nv : values) {
221           long v = nv == null ? 0 : nv.longValue();
222           writer.add(v - minDelta);
223         }
224         writer.finish();
225         break;
226       case TABLE_COMPRESSED:
227         final Long[] decode = uniqueValues.toArray(new Long[uniqueValues.size()]);
228         Arrays.sort(decode);
229         final HashMap<Long,Integer> encode = new HashMap<>();
230         meta.writeVInt(decode.length);
231         for (int i = 0; i < decode.length; i++) {
232           meta.writeLong(decode[i]);
233           encode.put(decode[i], i);
234         }
235         meta.writeVInt(tableBitsRequired);
236         final DirectWriter ordsWriter = DirectWriter.getInstance(data, count, tableBitsRequired);
237         for (Number nv : values) {
238           ordsWriter.add(encode.get(nv == null ? 0 : nv.longValue()));
239         }
240         ordsWriter.finish();
241         break;
242       default:
243         throw new AssertionError();
244     }
245     meta.writeLong(data.getFilePointer());
246   }

248   // TODO: in some cases representing missing with minValue-1 wouldn't take up additional space and so on,
249   // but this is very simple, and algorithms only check this for values of 0 anyway (doesnt slow down normal decode)
250   void writeMissingBitset(Iterable<?> values) throws IOException {
251     byte bits = 0;
252     int count = 0;
253     for (Object v : values) {
254       if (count == 8) {
255         data.writeByte(bits);
256         count = 0;
257         bits = 0;
258       }
259       if (v != null) {
260         bits |= 1 << (count & 7);
261       }
262       count++;
263     }
264     if (count > 0) {
265       data.writeByte(bits);
266     }
267   }

269   @Override
270   public void addBinaryField(FieldInfo field, Iterable<BytesRef> values) throws IOException {
271     checkCanWrite(field);
272     // write the byte[] data
273     meta.writeVInt(field.number);
274     meta.writeByte(Lucene410DocValuesFormat.BINARY);
275     int minLength = Integer.MAX_VALUE;
276     int maxLength = Integer.MIN_VALUE;
277     final long startFP = data.getFilePointer();
278     long count = 0;
279     boolean missing = false;
280     for(BytesRef v : values) {
281       final int length;
282       if (v == null) {
283         length = 0;
284         missing = true;
285       } else {
286         length = v.length;
287       }
288       minLength = Math.min(minLength, length);
289       maxLength = Math.max(maxLength, length);
290       if (v != null) {
291         data.writeBytes(v.bytes, v.offset, v.length);
292       }
293       count++;
294     }
295     meta.writeVInt(minLength == maxLength ? BINARY_FIXED_UNCOMPRESSED : BINARY_VARIABLE_UNCOMPRESSED);
296     if (missing) {
297       meta.writeLong(data.getFilePointer());
298       writeMissingBitset(values);
299     } else {
300       meta.writeLong(-1L);
301     }
302     meta.writeVInt(minLength);
303     meta.writeVInt(maxLength);
304     meta.writeVLong(count);
305     meta.writeLong(startFP);
306     
307     // if minLength == maxLength, its a fixed-length byte[], we are done (the addresses are implicit)
308     // otherwise, we need to record the length fields...
309     if (minLength != maxLength) {
310       meta.writeLong(data.getFilePointer());
311       meta.writeVInt(PackedInts.VERSION_CURRENT);
312       meta.writeVInt(BLOCK_SIZE);
313 
314       final MonotonicBlockPackedWriter writer = new MonotonicBlockPackedWriter(data, BLOCK_SIZE);
315       long addr = 0;
316       writer.add(addr);
317       for (BytesRef v : values) {
318         if (v != null) {
319           addr += v.length;
320         }
321         writer.add(addr);
322       }
323       writer.finish();
324     }
325   }

327   /** expert: writes a value dictionary for a sorted/sortedset field */
328   private void addTermsDict(FieldInfo field, final Iterable<BytesRef> values) throws IOException {
329     // first check if its a "fixed-length" terms dict
330     int minLength = Integer.MAX_VALUE;
331     int maxLength = Integer.MIN_VALUE;
332     long numValues = 0;
333     for (BytesRef v : values) {
334       minLength = Math.min(minLength, v.length);
335       maxLength = Math.max(maxLength, v.length);
336       numValues++;
337     }
338     if (minLength == maxLength) {
339       // no index needed: direct addressing by mult
340       addBinaryField(field, values);
341     } else if (numValues < REVERSE_INTERVAL_COUNT) {
342       // low cardinality: waste a few KB of ram, but can't really use fancy index etc
343       addBinaryField(field, values);
344     } else {
345       assert numValues > 0; // we don't have to handle the empty case
346       // header
347       meta.writeVInt(field.number);
348       meta.writeByte(Lucene410DocValuesFormat.BINARY);
349       meta.writeVInt(BINARY_PREFIX_COMPRESSED);
350       meta.writeLong(-1L);
351       // now write the bytes: sharing prefixes within a block
352       final long startFP = data.getFilePointer();
353       // currently, we have to store the delta from expected for every 1/nth term
354       // we could avoid this, but its not much and less overall RAM than the previous approach!
355       RAMOutputStream addressBuffer = new RAMOutputStream();
356       MonotonicBlockPackedWriter termAddresses = new MonotonicBlockPackedWriter(addressBuffer, BLOCK_SIZE);
357       // buffers up 16 terms
358       RAMOutputStream bytesBuffer = new RAMOutputStream();
359       // buffers up block header
360       RAMOutputStream headerBuffer = new RAMOutputStream();
361       BytesRefBuilder lastTerm = new BytesRefBuilder();
362       lastTerm.grow(maxLength);
363       long count = 0;
364       int suffixDeltas[] = new int[INTERVAL_COUNT];
365       for (BytesRef v : values) {
366         int termPosition = (int) (count & INTERVAL_MASK);
367         if (termPosition == 0) {
368           termAddresses.add(data.getFilePointer() - startFP);
369           // abs-encode first term
370           headerBuffer.writeVInt(v.length);
371           headerBuffer.writeBytes(v.bytes, v.offset, v.length);
372           lastTerm.copyBytes(v);
373         } else {
374           // prefix-code: we only share at most 255 characters, to encode the length as a single
375           // byte and have random access. Larger terms just get less compression.
376           int sharedPrefix = Math.min(255, StringHelper.bytesDifference(lastTerm.get(), v));
377           bytesBuffer.writeByte((byte) sharedPrefix);
378           bytesBuffer.writeBytes(v.bytes, v.offset + sharedPrefix, v.length - sharedPrefix);
379           // we can encode one smaller, because terms are unique.
380           suffixDeltas[termPosition] = v.length - sharedPrefix - 1;
381         }
382         
383         count++;
384         // flush block
385         if ((count & INTERVAL_MASK) == 0) {
386           flushTermsDictBlock(headerBuffer, bytesBuffer, suffixDeltas);
387         }
388       }
389       // flush trailing crap
390       int leftover = (int) (count & INTERVAL_MASK);
391       if (leftover > 0) {
392         Arrays.fill(suffixDeltas, leftover, suffixDeltas.length, 0);
393         flushTermsDictBlock(headerBuffer, bytesBuffer, suffixDeltas);
394       }
395       final long indexStartFP = data.getFilePointer();
396       // write addresses of indexed terms
397       termAddresses.finish();
398       addressBuffer.writeTo(data);
399       addressBuffer = null;
400       termAddresses = null;
401       meta.writeVInt(minLength);
402       meta.writeVInt(maxLength);
403       meta.writeVLong(count);
404       meta.writeLong(startFP);
405       meta.writeLong(indexStartFP);
406       meta.writeVInt(PackedInts.VERSION_CURRENT);
407       meta.writeVInt(BLOCK_SIZE);
408       addReverseTermIndex(field, values, maxLength);
409     }
410   }

412   // writes term dictionary "block"
413   // first term is absolute encoded as vint length + bytes.
414   // lengths of subsequent N terms are encoded as either N bytes or N shorts.
415   // in the double-byte case, the first byte is indicated with -1.
416   // subsequent terms are encoded as byte suffixLength + bytes.
417   private void flushTermsDictBlock(RAMOutputStream headerBuffer, RAMOutputStream bytesBuffer, int suffixDeltas[]) throws IOException {
418     boolean twoByte = false;
419     for (int i = 1; i < suffixDeltas.length; i++) {
420       if (suffixDeltas[i] > 254) {
421         twoByte = true;
422       }
423     }
424     if (twoByte) {
425       headerBuffer.writeByte((byte)255);
426       for (int i = 1; i < suffixDeltas.length; i++) {
427         headerBuffer.writeShort((short) suffixDeltas[i]);
428       }
429     } else {
430       for (int i = 1; i < suffixDeltas.length; i++) {
431         headerBuffer.writeByte((byte) suffixDeltas[i]);
432       }
433     }
434     headerBuffer.writeTo(data);
435     headerBuffer.reset();
436     bytesBuffer.writeTo(data);
437     bytesBuffer.reset();
438   }

440   // writes reverse term index: used for binary searching a term into a range of 64 blocks
441   // for every 64 blocks (1024 terms) we store a term, trimming any suffix unnecessary for comparison
442   // terms are written as a contiguous byte[], but never spanning 2^15 byte boundaries.
443   private void addReverseTermIndex(FieldInfo field, final Iterable<BytesRef> values, int maxLength) throws IOException {
444     long count = 0;
445     BytesRefBuilder priorTerm = new BytesRefBuilder();
446     priorTerm.grow(maxLength);
447     BytesRef indexTerm = new BytesRef();
448     long startFP = data.getFilePointer();
449     PagedBytes pagedBytes = new PagedBytes(15);
450     MonotonicBlockPackedWriter addresses = new MonotonicBlockPackedWriter(data, BLOCK_SIZE);
451     
452     for (BytesRef b : values) {
453       int termPosition = (int) (count & REVERSE_INTERVAL_MASK);
454       if (termPosition == 0) {
455         int len = StringHelper.sortKeyLength(priorTerm.get(), b);
456         indexTerm.bytes = b.bytes;
457         indexTerm.offset = b.offset;
458         indexTerm.length = len;
459         addresses.add(pagedBytes.copyUsingLengthPrefix(indexTerm));
460       } else if (termPosition == REVERSE_INTERVAL_MASK) {
461         priorTerm.copyBytes(b);
462       }
463       count++;
464     }
465     addresses.finish();
466     long numBytes = pagedBytes.getPointer();
467     pagedBytes.freeze(true);
468     PagedBytesDataInput in = pagedBytes.getDataInput();
469     meta.writeLong(startFP);
470     data.writeVLong(numBytes);
471     data.copyBytes(in, numBytes);
472   }

480     addNumericField(field, docToOrd, false);

483   @Override
484   public void addSortedNumericField(FieldInfo field, final Iterable<Number> docToValueCount, final Iterable<Number> values) throws IOException {
485     checkCanWrite(field);
486     meta.writeVInt(field.number);
487     meta.writeByte(Lucene410DocValuesFormat.SORTED_NUMERIC);
488     if (isSingleValued(docToValueCount)) {
489       meta.writeVInt(SORTED_SINGLE_VALUED);
490       // The field is single-valued, we can encode it as NUMERIC
491       addNumericField(field, singletonView(docToValueCount, values, null));
492     } else {
493       meta.writeVInt(SORTED_WITH_ADDRESSES);
494       // write the stream of values as a numeric field
495       addNumericField(field, values, true);
496       // write the doc -> ord count as a absolute index to the stream
497       addAddresses(field, docToValueCount);
498     }
499   }

501   @Override
502   public void addSortedSetField(FieldInfo field, Iterable<BytesRef> values, final Iterable<Number> docToOrdCount, final Iterable<Number> ords) throws IOException {
503     checkCanWrite(field);
504     meta.writeVInt(field.number);
505     meta.writeByte(Lucene410DocValuesFormat.SORTED_SET);
506 
507     if (isSingleValued(docToOrdCount)) {
508       meta.writeVInt(SORTED_SINGLE_VALUED);
509       // The field is single-valued, we can encode it as SORTED
510       addSortedField(field, values, singletonView(docToOrdCount, ords, -1L));
511     } else {
512       meta.writeVInt(SORTED_WITH_ADDRESSES);
513 
514       // write the ord -> byte[] as a binary field
515       addTermsDict(field, values);
516 
517       // write the stream of ords as a numeric field
518       // NOTE: we could return an iterator that delta-encodes these within a doc
519       addNumericField(field, ords, false);
520 
521       // write the doc -> ord count as a absolute index to the stream
522       addAddresses(field, docToOrdCount);
523     }
524   }

526   // writes addressing information as MONOTONIC_COMPRESSED integer
527   private void addAddresses(FieldInfo field, Iterable<Number> values) throws IOException {
528     meta.writeVInt(field.number);
529     meta.writeByte(Lucene410DocValuesFormat.NUMERIC);
530     meta.writeVInt(MONOTONIC_COMPRESSED);
531     meta.writeLong(-1L);
532     meta.writeLong(data.getFilePointer());
533     meta.writeVLong(maxDoc);
534     meta.writeVInt(PackedInts.VERSION_CURRENT);
535     meta.writeVInt(BLOCK_SIZE);
536 
537     final MonotonicBlockPackedWriter writer = new MonotonicBlockPackedWriter(data, BLOCK_SIZE);
538     long addr = 0;
539     writer.add(addr);
540     for (Number v : values) {
541       addr += v.longValue();
542       writer.add(addr);
543     }
544     writer.finish();
545     meta.writeLong(data.getFilePointer());
546   }

548   @Override
549   public void close() throws IOException {
550     boolean success = false;
551     try {
552       if (meta != null) {
553         meta.writeVInt(-1); // write EOF marker
554         CodecUtil.writeFooter(meta); // write checksum
555       }
556       if (data != null) {
557         CodecUtil.writeFooter(data); // write checksum
558       }
559       success = true;
560     } finally {
561       if (success) {
562         IOUtils.close(data, meta);
563       } else {
564         IOUtils.closeWhileHandlingException(data, meta);
565       }
566       meta = data = null;
567     }
568   }

570   void checkCanWrite(FieldInfo field) {
571     if ((field.getDocValuesType() == DocValuesType.NUMERIC || 
572         field.getDocValuesType() == DocValuesType.BINARY) && 
573         field.getDocValuesGen() != -1) {
574       // ok
575     } else {
576       throw new UnsupportedOperationException("this codec can only be used for reading");
577     }
578   }
