109   private static enum ShowStyle {
110     ALL,
111     DOC,
112     SCHEMA,
113     INDEX;
114 
115     public static ShowStyle get(String v) {
116       if(v==null) return null;
117       if("schema".equalsIgnoreCase(v)) return SCHEMA;
118       if("index".equalsIgnoreCase(v))  return INDEX;
119       if("doc".equalsIgnoreCase(v))    return DOC;
120       if("all".equalsIgnoreCase(v))    return ALL;
121       throw new SolrException(ErrorCode.BAD_REQUEST, "Unknown Show Style: "+v);
122     }
123   };

126   @Override
127   public void handleRequestBody(SolrQueryRequest req, SolrQueryResponse rsp) throws Exception
128   {
129     IndexSchema schema = req.getSchema();
130     SolrIndexSearcher searcher = req.getSearcher();
131     DirectoryReader reader = searcher.getIndexReader();
132     SolrParams params = req.getParams();
133     ShowStyle style = ShowStyle.get(params.get("show"));
134 
135     // If no doc is given, show all fields and top terms
136 
137     rsp.add("index", getIndexInfo(reader));
138 
139     if(ShowStyle.INDEX==style) {
140       return; // that's all we need
141     }
142 
143 
144     Integer docId = params.getInt( DOC_ID );
145     if( docId == null && params.get( ID ) != null ) {
146       // Look for something with a given solr ID
147       SchemaField uniqueKey = schema.getUniqueKeyField();
148       String v = uniqueKey.getType().toInternal( params.get(ID) );
149       Term t = new Term( uniqueKey.getName(), v );
150       docId = searcher.getFirstMatch( t );
151       if( docId < 0 ) {
152         throw new SolrException( SolrException.ErrorCode.NOT_FOUND, "Can't find document: "+params.get( ID ) );
153       }
154     }
155 
156     // Read the document from the index
157     if( docId != null ) {
158       if( style != null && style != ShowStyle.DOC ) {
159         throw new SolrException(ErrorCode.BAD_REQUEST, "missing doc param for doc style");
160       }
161       Document doc = null;
162       try {
163         doc = reader.document( docId );
164       }
165       catch( Exception ex ) {}
166       if( doc == null ) {
167         throw new SolrException( SolrException.ErrorCode.NOT_FOUND, "Can't find document: "+docId );
168       }
169 
170       SimpleOrderedMap<Object> info = getDocumentFieldsInfo( doc, docId, reader, schema );
171 
172       SimpleOrderedMap<Object> docinfo = new SimpleOrderedMap<>();
173       docinfo.add( "docId", docId );
174       docinfo.add( "lucene", info );
175       docinfo.add( "solr", doc );
176       rsp.add( "doc", docinfo );
177     }
178     else if ( ShowStyle.SCHEMA == style ) {
179       rsp.add( "schema", getSchemaInfo( req.getSchema() ) );
180     }
181     else {
182       rsp.add( "fields", getIndexedFieldsInfo(req) ) ;
183     }
184 
185     // Add some generally helpful information
186     NamedList<Object> info = new SimpleOrderedMap<>();
187     info.add( "key", getFieldFlagsKey() );
188     info.add( "NOTE", "Document Frequency (df) is not updated when a document is marked for deletion.  df values include deleted documents." );
189     rsp.add( "info", info );
190     rsp.setHttpCaching(false);
191   }

195   /**
196    * @return a string representing a IndexableField's flags.  
197    */
198   private static String getFieldFlags( IndexableField f )
199   {
200     IndexOptions opts = (f == null) ? null : f.fieldType().indexOptions();
201 
202     StringBuilder flags = new StringBuilder();
203 
204     flags.append( (f != null && f.fieldType().indexOptions() != IndexOptions.NONE)                     ? FieldFlag.INDEXED.getAbbreviation() : '-' );
205     flags.append( (f != null && f.fieldType().tokenized())                   ? FieldFlag.TOKENIZED.getAbbreviation() : '-' );
206     flags.append( (f != null && f.fieldType().stored())                      ? FieldFlag.STORED.getAbbreviation() : '-' );
207     flags.append( (f != null && f.fieldType().docValuesType() != DocValuesType.NONE)        ? FieldFlag.DOC_VALUES.getAbbreviation() : "-" );
208     flags.append( (false)                                          ? FieldFlag.MULTI_VALUED.getAbbreviation() : '-' ); // SchemaField Specific
209     flags.append( (f != null && f.fieldType().storeTermVectors())            ? FieldFlag.TERM_VECTOR_STORED.getAbbreviation() : '-' );
210     flags.append( (f != null && f.fieldType().storeTermVectorOffsets())   ? FieldFlag.TERM_VECTOR_OFFSET.getAbbreviation() : '-' );
211     flags.append( (f != null && f.fieldType().storeTermVectorPositions()) ? FieldFlag.TERM_VECTOR_POSITION.getAbbreviation() : '-' );
212     flags.append( (f != null && f.fieldType().storeTermVectorPayloads())   ? FieldFlag.TERM_VECTOR_PAYLOADS.getAbbreviation() : '-' );
213     flags.append( (f != null && f.fieldType().omitNorms())                  ? FieldFlag.OMIT_NORMS.getAbbreviation() : '-' );
214 
215     flags.append( (f != null && DOCS == opts ) ?
216         FieldFlag.OMIT_TF.getAbbreviation() : '-' );
217 
218     flags.append((f != null && DOCS_AND_FREQS == opts) ?
219         FieldFlag.OMIT_POSITIONS.getAbbreviation() : '-');
220     
221     flags.append((f != null && DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS == opts) ?
222         FieldFlag.STORE_OFFSETS_WITH_POSITIONS.getAbbreviation() : '-');
223 
224     flags.append( (f != null && f.getClass().getSimpleName().equals("LazyField")) ? FieldFlag.LAZY.getAbbreviation() : '-' );
225     flags.append( (f != null && f.binaryValue()!=null)                      ? FieldFlag.BINARY.getAbbreviation() : '-' );
226     flags.append( (false)                                          ? FieldFlag.SORT_MISSING_FIRST.getAbbreviation() : '-' ); // SchemaField Specific
227     flags.append( (false)                                          ? FieldFlag.SORT_MISSING_LAST.getAbbreviation() : '-' ); // SchemaField Specific
228     return flags.toString();
229   }

231   /**
232    * @return a string representing a SchemaField's flags.  
233    */
234   private static String getFieldFlags( SchemaField f )
235   {
236     FieldType t = (f==null) ? null : f.getType();
237 
238     // see: http://www.nabble.com/schema-field-properties-tf3437753.html#a9585549
239     boolean lazy = false; // "lazy" is purely a property of reading fields
240     boolean binary = false; // Currently not possible
241 
242     StringBuilder flags = new StringBuilder();
243     flags.append( (f != null && f.indexed())             ? FieldFlag.INDEXED.getAbbreviation() : '-' );
244     flags.append( (t != null && t.isTokenized())         ? FieldFlag.TOKENIZED.getAbbreviation() : '-' );
245     flags.append( (f != null && f.stored())              ? FieldFlag.STORED.getAbbreviation() : '-' );
246     flags.append( (f != null && f.hasDocValues())        ? FieldFlag.DOC_VALUES.getAbbreviation() : "-" );
247     flags.append( (f != null && f.multiValued())         ? FieldFlag.MULTI_VALUED.getAbbreviation() : '-' );
248     flags.append( (f != null && f.storeTermVector() )    ? FieldFlag.TERM_VECTOR_STORED.getAbbreviation() : '-' );
249     flags.append( (f != null && f.storeTermOffsets() )   ? FieldFlag.TERM_VECTOR_OFFSET.getAbbreviation() : '-' );
250     flags.append( (f != null && f.storeTermPositions() ) ? FieldFlag.TERM_VECTOR_POSITION.getAbbreviation() : '-' );
251     flags.append( (f != null && f.storeTermPayloads() )  ? FieldFlag.TERM_VECTOR_PAYLOADS.getAbbreviation() : '-' );
252     flags.append( (f != null && f.omitNorms())           ? FieldFlag.OMIT_NORMS.getAbbreviation() : '-' );
253     flags.append( (f != null &&
254         f.omitTermFreqAndPositions() )        ? FieldFlag.OMIT_TF.getAbbreviation() : '-' );
255     flags.append( (f != null && f.omitPositions() )      ? FieldFlag.OMIT_POSITIONS.getAbbreviation() : '-' );
256     flags.append( (f != null && f.storeOffsetsWithPositions() )      ? FieldFlag.STORE_OFFSETS_WITH_POSITIONS.getAbbreviation() : '-' );
257     flags.append( (lazy)                                 ? FieldFlag.LAZY.getAbbreviation() : '-' );
258     flags.append( (binary)                               ? FieldFlag.BINARY.getAbbreviation() : '-' );
259     flags.append( (f != null && f.sortMissingFirst() )   ? FieldFlag.SORT_MISSING_FIRST.getAbbreviation() : '-' );
260     flags.append( (f != null && f.sortMissingLast() )    ? FieldFlag.SORT_MISSING_LAST.getAbbreviation() : '-' );
261     return flags.toString();
262   }

275   private static SimpleOrderedMap<Object> getDocumentFieldsInfo( Document doc, int docId, IndexReader reader,
276                                                                  IndexSchema schema ) throws IOException
277   {
278     final CharsRefBuilder spare = new CharsRefBuilder();
279     SimpleOrderedMap<Object> finfo = new SimpleOrderedMap<>();
280     for( Object o : doc.getFields() ) {
281       Field field = (Field)o;
282       SimpleOrderedMap<Object> f = new SimpleOrderedMap<>();
283 
284       SchemaField sfield = schema.getFieldOrNull( field.name() );
285       FieldType ftype = (sfield==null)?null:sfield.getType();
286 
287       f.add( "type", (ftype==null)?null:ftype.getTypeName() );
288       f.add( "schema", getFieldFlags( sfield ) );
289       f.add( "flags", getFieldFlags( field ) );
290 
291       Term t = new Term(field.name(), ftype!=null ? ftype.storedToIndexed(field) : field.stringValue());
292 
293       f.add( "value", (ftype==null)?null:ftype.toExternal( field ) );
294 
295       // TODO: this really should be "stored"
296       f.add( "internal", field.stringValue() );  // may be a binary number
297 
298       BytesRef bytes = field.binaryValue();
299       if (bytes != null) {
300         f.add( "binary", Base64.byteArrayToBase64(bytes.bytes, bytes.offset, bytes.length));
301       }
302       f.add( "boost", field.boost() );
303       f.add( "docFreq", t.text()==null ? 0 : reader.docFreq( t ) ); // this can be 0 for non-indexed fields
304 
305       // If we have a term vector, return that
306       if( field.fieldType().storeTermVectors() ) {
307         try {
308           Terms v = reader.getTermVector( docId, field.name() );
309           if( v != null ) {
310             SimpleOrderedMap<Integer> tfv = new SimpleOrderedMap<>();
311             final TermsEnum termsEnum = v.iterator();
312             BytesRef text;
313             while((text = termsEnum.next()) != null) {
314               final int freq = (int) termsEnum.totalTermFreq();
315               spare.copyUTF8Bytes(text);
316               tfv.add(spare.toString(), freq);
317             }
318             f.add( "termVector", tfv );
319           }
320         }
321         catch( Exception ex ) {
322           log.warn( "error writing term vector", ex );
323         }
324       }
325 
326       finfo.add( field.name(), f );
327     }
328     return finfo;
329   }

331   private static SimpleOrderedMap<Object> getIndexedFieldsInfo(SolrQueryRequest req)
332       throws Exception {
333 
334     SolrIndexSearcher searcher = req.getSearcher();
335     SolrParams params = req.getParams();
336 
337     Set<String> fields = null;
338     String fl = params.get(CommonParams.FL);
339     if (fl != null) {
340       fields = new TreeSet<>(Arrays.asList(fl.split( "[,\\s]+" )));
341     }
342 
343     LeafReader reader = searcher.getLeafReader();
344     IndexSchema schema = searcher.getSchema();
345 
346     // Don't be tempted to put this in the loop below, the whole point here is to alphabetize the fields!
347     Set<String> fieldNames = new TreeSet<>();
348     for(FieldInfo fieldInfo : reader.getFieldInfos()) {
349       fieldNames.add(fieldInfo.name);
350     }
351 
352     // Walk the term enum and keep a priority queue for each map in our set
353     SimpleOrderedMap<Object> finfo = new SimpleOrderedMap<>();
354 
355     for (String fieldName : fieldNames) {
356       if (fields != null && ! fields.contains(fieldName) && ! fields.contains("*")) {
357         continue; //we're not interested in this field Still an issue here
358       }
359 
360       SimpleOrderedMap<Object> fieldMap = new SimpleOrderedMap<>();
361 
362       SchemaField sfield = schema.getFieldOrNull( fieldName );
363       FieldType ftype = (sfield==null)?null:sfield.getType();
364 
365       fieldMap.add( "type", (ftype==null)?null:ftype.getTypeName() );
366       fieldMap.add("schema", getFieldFlags(sfield));
367       if (sfield != null && schema.isDynamicField(sfield.getName()) && schema.getDynamicPattern(sfield.getName()) != null) {
368         fieldMap.add("dynamicBase", schema.getDynamicPattern(sfield.getName()));
369       }
370       Terms terms = reader.fields().terms(fieldName);
371       if (terms == null) { // Not indexed, so we need to report what we can (it made it through the fl param if specified)
372         finfo.add( fieldName, fieldMap );
373         continue;
374       }
375 
376       if(sfield != null && sfield.indexed() ) {
377         if (params.getBool(INCLUDE_INDEX_FIELD_FLAGS,true)) {
378           Document doc = getFirstLiveDoc(terms, reader);
379           if (doc != null) {
380             // Found a document with this field
381             try {
382               IndexableField fld = doc.getField(fieldName);
383               if (fld != null) {
384                 fieldMap.add("index", getFieldFlags(fld));
385               } else {
386                 // it is a non-stored field...
387                 fieldMap.add("index", "(unstored field)");
388               }
389             } catch (Exception ex) {
390               log.warn("error reading field: " + fieldName);
391             }
392           }
393         }
394 
395         fieldMap.add("docs", terms.getDocCount());
396       }
397       if (fields != null && (fields.contains(fieldName) || fields.contains("*"))) {
398         getDetailedFieldInfo(req, fieldName, fieldMap);
399       }
400       // Add the field
401       finfo.add( fieldName, fieldMap );
402     }
403     return finfo;
404   }

406   // Just get a document with the term in it, the first one will do!
407   // Is there a better way to do this? Shouldn't actually be very costly
408   // to do it this way.
409   private static Document getFirstLiveDoc(Terms terms, LeafReader reader) throws IOException {
410     PostingsEnum postingsEnum = null;
411     TermsEnum termsEnum = terms.iterator();
412     BytesRef text;
413     // Deal with the chance that the first bunch of terms are in deleted documents. Is there a better way?
414     for (int idx = 0; idx < 1000 && postingsEnum == null; ++idx) {
415       text = termsEnum.next();
416       if (text == null) { // Ran off the end of the terms enum without finding any live docs with that field in them.
417         return null;
418       }
419       postingsEnum = termsEnum.postings(postingsEnum, PostingsEnum.NONE);
420       final Bits liveDocs = reader.getLiveDocs();
421       if (postingsEnum.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
422         if (liveDocs != null && liveDocs.get(postingsEnum.docID())) {
423           continue;
424         }
425         return reader.document(postingsEnum.docID());
426       }
427     }
428     return null;
429   }

431   /**
432    * Return info from the index
433    */
434   private static SimpleOrderedMap<Object> getSchemaInfo( IndexSchema schema ) {
435     Map<String, List<String>> typeusemap = new TreeMap<>();
436     Map<String, Object> fields = new TreeMap<>();
437     SchemaField uniqueField = schema.getUniqueKeyField();
438     for( SchemaField f : schema.getFields().values() ) {
439       populateFieldInfo(schema, typeusemap, fields, uniqueField, f);
440     }
441 
442     Map<String, Object> dynamicFields = new TreeMap<>();
443     for (SchemaField f : schema.getDynamicFieldPrototypes()) {
444       populateFieldInfo(schema, typeusemap, dynamicFields, uniqueField, f);
445     }
446     SimpleOrderedMap<Object> types = new SimpleOrderedMap<>();
447     Map<String, FieldType> sortedTypes = new TreeMap<>(schema.getFieldTypes());
448     for( FieldType ft : sortedTypes.values() ) {
449       SimpleOrderedMap<Object> field = new SimpleOrderedMap<>();
450       field.add("fields", typeusemap.get( ft.getTypeName() ) );
451       field.add("tokenized", ft.isTokenized() );
452       field.add("className", ft.getClass().getName());
453       field.add("indexAnalyzer", getAnalyzerInfo(ft.getIndexAnalyzer()));
454       field.add("queryAnalyzer", getAnalyzerInfo(ft.getQueryAnalyzer()));
455       field.add("similarity", getSimilarityInfo(ft.getSimilarity()));
456       types.add( ft.getTypeName(), field );
457     }
458 
459     // Must go through this to maintain binary compatbility. Putting a TreeMap into a resp leads to casting errors
460     SimpleOrderedMap<Object> finfo = new SimpleOrderedMap<>();
461 
462     SimpleOrderedMap<Object> fieldsSimple = new SimpleOrderedMap<>();
463     for (Map.Entry<String, Object> ent : fields.entrySet()) {
464       fieldsSimple.add(ent.getKey(), ent.getValue());
465     }
466     finfo.add("fields", fieldsSimple);
467 
468     SimpleOrderedMap<Object> dynamicSimple = new SimpleOrderedMap<>();
469     for (Map.Entry<String, Object> ent : dynamicFields.entrySet()) {
470       dynamicSimple.add(ent.getKey(), ent.getValue());
471     }
472     finfo.add("dynamicFields", dynamicSimple);
473 
474     finfo.add("uniqueKeyField",
475         null == uniqueField ? null : uniqueField.getName());
476     finfo.add("defaultSearchField", schema.getDefaultSearchFieldName());
477     finfo.add("types", types);
478     return finfo;
479   }

490   private static SimpleOrderedMap<Object> getAnalyzerInfo(Analyzer analyzer) {
491     SimpleOrderedMap<Object> aninfo = new SimpleOrderedMap<>();
492     aninfo.add("className", analyzer.getClass().getName());
493     if (analyzer instanceof TokenizerChain) {
494 
495       TokenizerChain tchain = (TokenizerChain)analyzer;
496 
497       CharFilterFactory[] cfiltfacs = tchain.getCharFilterFactories();
498       if (0 < cfiltfacs.length) {
499         SimpleOrderedMap<Map<String, Object>> cfilters = new SimpleOrderedMap<>();
500         for (CharFilterFactory cfiltfac : cfiltfacs) {
501           Map<String, Object> tok = new HashMap<>();
502           String className = cfiltfac.getClass().getName();
503           tok.put("className", className);
504           tok.put("args", cfiltfac.getOriginalArgs());
505           cfilters.add(className.substring(className.lastIndexOf('.')+1), tok);
506         }
507         aninfo.add("charFilters", cfilters);
508       }
509 
510       SimpleOrderedMap<Object> tokenizer = new SimpleOrderedMap<>();
511       TokenizerFactory tfac = tchain.getTokenizerFactory();
512       tokenizer.add("className", tfac.getClass().getName());
513       tokenizer.add("args", tfac.getOriginalArgs());
514       aninfo.add("tokenizer", tokenizer);
515 
516       TokenFilterFactory[] filtfacs = tchain.getTokenFilterFactories();
517       if (0 < filtfacs.length) {
518         SimpleOrderedMap<Map<String, Object>> filters = new SimpleOrderedMap<>();
519         for (TokenFilterFactory filtfac : filtfacs) {
520           Map<String, Object> tok = new HashMap<>();
521           String className = filtfac.getClass().getName();
522           tok.put("className", className);
523           tok.put("args", filtfac.getOriginalArgs());
524           filters.add(className.substring(className.lastIndexOf('.')+1), tok);
525         }
526         aninfo.add("filters", filters);
527       }
528     }
529     return aninfo;
530   }

532   private static void populateFieldInfo(IndexSchema schema,
533                                         Map<String, List<String>> typeusemap, Map<String, Object> fields,
534                                         SchemaField uniqueField, SchemaField f) {
535     FieldType ft = f.getType();
536     SimpleOrderedMap<Object> field = new SimpleOrderedMap<>();
537     field.add( "type", ft.getTypeName() );
538     field.add( "flags", getFieldFlags(f) );
539     if( f.isRequired() ) {
540       field.add( "required", f.isRequired() );
541     }
542     if( f.getDefaultValue() != null ) {
543       field.add( "default", f.getDefaultValue() );
544     }
545     if (f == uniqueField){
546       field.add("uniqueKey", true);
547     }
548     if (ft.getIndexAnalyzer().getPositionIncrementGap(f.getName()) != 0) {
549       field.add("positionIncrementGap", ft.getIndexAnalyzer().getPositionIncrementGap(f.getName()));
550     }
551     field.add("copyDests", toListOfStringDests(schema.getCopyFieldsList(f.getName())));
552     field.add("copySources", schema.getCopySources(f.getName()));
553 
554 
555     fields.put( f.getName(), field );
556 
557     List<String> v = typeusemap.get( ft.getTypeName() );
558     if( v == null ) {
559       v = new ArrayList<>();
560     }
561     v.add( f.getName() );
562     typeusemap.put( ft.getTypeName(), v );
563   }

575   public static SimpleOrderedMap<Object> getIndexInfo(DirectoryReader reader) throws IOException {
576     Directory dir = reader.directory();
577     SimpleOrderedMap<Object> indexInfo = new SimpleOrderedMap<>();

579     indexInfo.add("numDocs", reader.numDocs());
580     indexInfo.add("maxDoc", reader.maxDoc());
581     indexInfo.add("deletedDocs", reader.maxDoc() - reader.numDocs());
582     indexInfo.add("indexHeapUsageBytes", getIndexHeapUsed(reader));
583 
584     indexInfo.add("version", reader.getVersion());  // TODO? Is this different then: IndexReader.getCurrentVersion( dir )?
585     indexInfo.add("segmentCount", reader.leaves().size());
586     indexInfo.add("current", reader.isCurrent() );
587     indexInfo.add("hasDeletions", reader.hasDeletions() );
588     indexInfo.add("directory", dir );
589     indexInfo.add("userData", reader.getIndexCommit().getUserData());
590     String s = reader.getIndexCommit().getUserData().get(SolrIndexWriter.COMMIT_TIME_MSEC_KEY);
591     if (s != null) {
592       indexInfo.add("lastModified", new Date(Long.parseLong(s)));
593     }
594     return indexInfo;

597   /** Returns the sum of RAM bytes used by each segment */
598   private static long getIndexHeapUsed(DirectoryReader reader) {
599     long indexHeapRamBytesUsed = 0;
600     for(LeafReaderContext leafReaderContext : reader.leaves()) {
601       LeafReader leafReader = leafReaderContext.reader();
602       if (leafReader instanceof SegmentReader) {
603         indexHeapRamBytesUsed += ((SegmentReader) leafReader).ramBytesUsed();
604       } else {
605         // Not supported for any reader that is not a SegmentReader
606         return -1;
607       }
608     }
609     return indexHeapRamBytesUsed;
610   }

612   // Get terribly detailed information about a particular field. This is a very expensive call, use it with caution
613   // especially on large indexes!
614   @SuppressWarnings("unchecked")
615   private static void getDetailedFieldInfo(SolrQueryRequest req, String field, SimpleOrderedMap<Object> fieldMap)
616       throws IOException {
617 
618     SolrParams params = req.getParams();
619     final int numTerms = params.getInt( NUMTERMS, DEFAULT_COUNT );
620 
621     TopTermQueue tiq = new TopTermQueue(numTerms + 1);  // Something to collect the top N terms in.
622 
623     final CharsRefBuilder spare = new CharsRefBuilder();
624 
625     Terms terms = MultiFields.getTerms(req.getSearcher().getIndexReader(), field);
626     if (terms == null) {  // field does not exist
627       return;
628     }
629     TermsEnum termsEnum = terms.iterator();
630     BytesRef text;
631     int[] buckets = new int[HIST_ARRAY_SIZE];
632     while ((text = termsEnum.next()) != null) {
633       ++tiq.distinctTerms;
634       int freq = termsEnum.docFreq();  // This calculation seems odd, but it gives the same results as it used to.
635       int slot = 32 - Integer.numberOfLeadingZeros(Math.max(0, freq - 1));
636       buckets[slot] = buckets[slot] + 1;
637       if (numTerms > 0 && freq > tiq.minFreq) {
638         spare.copyUTF8Bytes(text);
639         String t = spare.toString();
640 
641         tiq.add(new TopTermQueue.TermInfo(new Term(field, t), termsEnum.docFreq()));
642         if (tiq.size() > numTerms) { // if tiq full
643           tiq.pop(); // remove lowest in tiq
644           tiq.minFreq = tiq.getTopTermInfo().docFreq;
645         }
646       }
647     }
648     tiq.histogram.add(buckets);
649     fieldMap.add("distinct", tiq.distinctTerms);
650 
651     // Include top terms
652     fieldMap.add("topTerms", tiq.toNamedList(req.getSearcher().getSchema()));
653 
654     // Add a histogram
655     fieldMap.add("histogram", tiq.histogram.toNamedList());
656   }

690   static class TermHistogram
691   {
692     int _maxBucket = -1;
693     int _buckets[] = new int[HIST_ARRAY_SIZE];
694     public void add(int[] buckets) {
695       for (int idx = 0; idx < buckets.length; ++idx) {
696         if (buckets[idx] != 0) _maxBucket = idx;
697       }
698       for (int idx = 0; idx <= _maxBucket; ++idx) {
699         _buckets[idx] = buckets[idx];
700       }
701     }
702     // TODO? should this be a list or a map?
703     public NamedList<Integer> toNamedList()
704     {
705       NamedList<Integer> nl = new NamedList<>();
706       for( int bucket = 0; bucket <= _maxBucket; bucket++ ) {
707         nl.add( ""+ (1 << bucket), _buckets[bucket] );
708       }
709       return nl;
710     }
711   }
712   /**
713    * Private internal class that counts up frequent terms
714    */
715   private static class TopTermQueue extends PriorityQueue
716   {
717     static class TermInfo {
718       TermInfo(Term t, int df) {
719         term = t;
720         docFreq = df;
721       }
722       int docFreq;
723       Term term;
724     }
725 
726     public int minFreq = 0;
727     public int distinctTerms = 0;
728     public TermHistogram histogram;
729 
730 
731     TopTermQueue(int size) {
732       super(size);
733       histogram = new TermHistogram();
734     }
735 
736     @Override
737     protected final boolean lessThan(Object a, Object b) {
738       TermInfo termInfoA = (TermInfo)a;
739       TermInfo termInfoB = (TermInfo)b;
740       return termInfoA.docFreq < termInfoB.docFreq;
741     }
742 
743     /**
744      * This is a destructive call... the queue is empty at the end
745      */
746     public NamedList<Integer> toNamedList( IndexSchema schema )
747     {
748       // reverse the list..
749       List<TermInfo> aslist = new LinkedList<>();
750       while( size() > 0 ) {
751         aslist.add( 0, (TermInfo)pop() );
752       }
753 
754       NamedList<Integer> list = new NamedList<>();
755       for (TermInfo i : aslist) {
756         String txt = i.term.text();
757         SchemaField ft = schema.getFieldOrNull( i.term.field() );
758         if( ft != null ) {
759           txt = ft.getType().indexedToReadable( txt );
760         }
761         list.add( txt, i.docFreq );
762       }
763       return list;
764     }
765     public TermInfo getTopTermInfo() {
766       return (TermInfo)top();
767     }
768   }
