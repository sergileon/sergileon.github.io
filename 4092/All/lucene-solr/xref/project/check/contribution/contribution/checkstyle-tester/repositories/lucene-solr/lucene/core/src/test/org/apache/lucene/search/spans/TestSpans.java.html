282   public void testSpanScorerZeroSloppyFreq() throws Exception {
283     IndexReaderContext topReaderContext = searcher.getTopReaderContext();
284     List<LeafReaderContext> leaves = topReaderContext.leaves();
285     int subIndex = ReaderUtil.subIndex(11, leaves);
286     for (int i = 0, c = leaves.size(); i < c; i++) {
287       final LeafReaderContext ctx = leaves.get(i);
288      
289       final Similarity sim = new DefaultSimilarity() {
290         @Override
291         public float sloppyFreq(int distance) {
292           return 0.0f;
293         }
294       };
295   
296       final Similarity oldSim = searcher.getSimilarity(true);
297       Scorer spanScorer;
298       try {
299         searcher.setSimilarity(sim);
300         SpanQuery snq = spanNearOrderedQuery(field, 1, "t1", "t2");
301         spanScorer = searcher.createNormalizedWeight(snq, true).scorer(ctx);
302       } finally {
303         searcher.setSimilarity(oldSim);
304       }
305       if (i == subIndex) {
306         assertTrue("first doc", spanScorer.nextDoc() != DocIdSetIterator.NO_MORE_DOCS);
307         assertEquals("first doc number", spanScorer.docID() + ctx.docBase, 11);
308         float score = spanScorer.score();
309         assertTrue("first doc score should be zero, " + score, score == 0.0f);
310       } else {
311         assertTrue("no second doc", spanScorer == null || spanScorer.nextDoc() == DocIdSetIterator.NO_MORE_DOCS);
312       }
313     }
314   }

349   public void testNPESpanQuery() throws Throwable {
350     final Directory dir = newDirectory();
351     final IndexWriter writer = new IndexWriter(dir, new IndexWriterConfig(new MockAnalyzer(random())));
352 
353     // Add documents
354     addDoc(writer, "1", "the big dogs went running to the market");
355     addDoc(writer, "2", "the cat chased the mouse, then the cat ate the mouse quickly");
356     
357     // Commit
358     writer.close();
359 
360     // Get searcher
361     final IndexReader reader = DirectoryReader.open(dir);
362     final IndexSearcher searcher = newSearcher(reader);
363 
364     // Control (make sure docs indexed)
365     assertEquals(2, hitCount(searcher, "the"));
366     assertEquals(1, hitCount(searcher, "cat"));
367     assertEquals(1, hitCount(searcher, "dogs"));
368     assertEquals(0, hitCount(searcher, "rabbit"));
369 
370     // This throws exception (it shouldn't)
371     assertEquals(1,
372                  searcher.search(createSpan(0, true,                                 
373                                             new SpanQuery[] {createSpan(4, false, "chased", "cat"),
374                                                              createSpan("ate")}), 10).totalHits);
375     reader.close();
376     dir.close();
377   }

382      assertEquals("SpanNotIncludeExcludeSame2", 0, spanCount("s2", "s2", 10, 10), 0);
383      
384      //focus on behind
385      assertEquals("SpanNotS2NotS1_6_0", 1, spanCount("s2", "s1", 6, 0));
386      assertEquals("SpanNotS2NotS1_5_0", 2, spanCount("s2", "s1", 5, 0));
387      assertEquals("SpanNotS2NotS1_3_0", 3, spanCount("s2", "s1", 3, 0));
388      assertEquals("SpanNotS2NotS1_2_0", 4, spanCount("s2", "s1", 2, 0));
389      assertEquals("SpanNotS2NotS1_0_0", 4, spanCount("s2", "s1", 0, 0));
390      
391      //focus on both
392      assertEquals("SpanNotS2NotS1_3_1", 2, spanCount("s2", "s1", 3, 1));
393      assertEquals("SpanNotS2NotS1_2_1", 3, spanCount("s2", "s1", 2, 1));
394      assertEquals("SpanNotS2NotS1_1_1", 3, spanCount("s2", "s1", 1, 1));
395      assertEquals("SpanNotS2NotS1_10_10", 0, spanCount("s2", "s1", 10, 10));
396      
397      //focus on ahead
398      assertEquals("SpanNotS1NotS2_10_10", 0, spanCount("s1", "s2", 10, 10));  
399      assertEquals("SpanNotS1NotS2_0_1", 3, spanCount("s1", "s2", 0, 1));  
400      assertEquals("SpanNotS1NotS2_0_2", 3, spanCount("s1", "s2", 0, 2));

407 
408      //include doesn't exist
409      assertEquals("SpanNotS3NotS1_8_8", 0, spanCount("s3", "s1", 8, 8));

413   private int spanCount(String include, String exclude, int pre, int post) throws IOException{
414      SpanQuery iq = spanTermQuery(field, include);
415      SpanQuery eq = spanTermQuery(field, exclude);
416      SpanQuery snq = spanNotQuery(iq, eq, pre, post);
417      Spans spans = MultiSpansWrapper.wrap(searcher.getIndexReader(), snq);
418 
419      int i = 0;
420      if (spans != null) {
421        while (spans.nextDoc() != Spans.NO_MORE_DOCS){
422          while (spans.nextStartPosition() != Spans.NO_MORE_POSITIONS) {
423            i++;
424          }
425        }
426      }
427      return i;
428   }
