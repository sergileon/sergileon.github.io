96    /**
97     * Constructs a SolrIndexConfig which parses the Lucene related config params in solrconfig.xml
98     * @param solrConfig the overall SolrConfig object
99     * @param prefix the XPath prefix for which section to parse (mandatory)
100    * @param def a SolrIndexConfig instance to pick default values from (optional)
101    */
102   @SuppressWarnings("deprecation")
103   public SolrIndexConfig(SolrConfig solrConfig, String prefix, SolrIndexConfig def)  {
104     if (prefix == null) {
105       prefix = "indexConfig";
106       log.debug("Defaulting to prefix \""+prefix+"\" for index configuration");
107     }
108     
109     if (def == null) {
110       def = new SolrIndexConfig(solrConfig);
111     }
112 
113     // sanity check: this will throw an error for us if there is more then one
114     // config section
115     Object unused = solrConfig.getNode(prefix, false);
116 
117     luceneVersion = solrConfig.luceneMatchVersion;
118 
119     // Assert that end-of-life parameters or syntax is not in our config.
120     // Warn for luceneMatchVersion's before LUCENE_3_6, fail fast above
121     assertWarnOrFail("The <mergeScheduler>myclass</mergeScheduler> syntax is no longer supported in solrconfig.xml. Please use syntax <mergeScheduler class=\"myclass\"/> instead.",
122         !((solrConfig.getNode(prefix + "/mergeScheduler", false) != null) && (solrConfig.get(prefix + "/mergeScheduler/@class", null) == null)),
123         true);
124     assertWarnOrFail("The <mergePolicy>myclass</mergePolicy> syntax is no longer supported in solrconfig.xml. Please use syntax <mergePolicy class=\"myclass\"/> instead.",
125         !((solrConfig.getNode(prefix + "/mergePolicy", false) != null) && (solrConfig.get(prefix + "/mergePolicy/@class", null) == null)),
126         true);
127     assertWarnOrFail("The <luceneAutoCommit>true|false</luceneAutoCommit> parameter is no longer valid in solrconfig.xml.",
128         solrConfig.get(prefix + "/luceneAutoCommit", null) == null,
129         true);
130 
131     defaultMergePolicyClassName = def.defaultMergePolicyClassName;
132     effectiveUseCompoundFileSetting = solrConfig.getBool(prefix+"/useCompoundFile", def.getUseCompoundFile());
133     maxBufferedDocs=solrConfig.getInt(prefix+"/maxBufferedDocs",def.maxBufferedDocs);
134     maxMergeDocs=solrConfig.getInt(prefix+"/maxMergeDocs",def.maxMergeDocs);
135     mergeFactor=solrConfig.getInt(prefix+"/mergeFactor",def.mergeFactor);
136     ramBufferSizeMB = solrConfig.getDouble(prefix+"/ramBufferSizeMB", def.ramBufferSizeMB);
137 
138     writeLockTimeout=solrConfig.getInt(prefix+"/writeLockTimeout", def.writeLockTimeout);
139     lockType=solrConfig.get(prefix+"/lockType", def.lockType);
140 
141     mergeSchedulerInfo = getPluginInfo(prefix + "/mergeScheduler", solrConfig, def.mergeSchedulerInfo);
142     mergePolicyInfo = getPluginInfo(prefix + "/mergePolicy", solrConfig, def.mergePolicyInfo);
143     
144     String val = solrConfig.get(prefix + "/termIndexInterval", null);
145     if (val != null) {
146       throw new IllegalArgumentException("Illegal parameter 'termIndexInterval'");
147     }
148 
149     boolean infoStreamEnabled = solrConfig.getBool(prefix + "/infoStream", false);
150     if(infoStreamEnabled) {
151       String infoStreamFile = solrConfig.get(prefix + "/infoStream/@file", null);
152       if (infoStreamFile == null) {
153         log.info("IndexWriter infoStream solr logging is enabled");
154         infoStream = new LoggingInfoStream();
155       } else {
156         throw new IllegalArgumentException("Remove @file from <infoStream> to output messages to solr's logfile");
157       }
158     }
159     mergedSegmentWarmerInfo = getPluginInfo(prefix + "/mergedSegmentWarmer", solrConfig, def.mergedSegmentWarmerInfo);
160 
161     assertWarnOrFail("Begining with Solr 5.0, <checkIntegrityAtMerge> option is no longer supported and should be removed from solrconfig.xml (these integrity checks are now automatic)",
162         (null == solrConfig.getNode(prefix + "/checkIntegrityAtMerge", false)),
163         false);
164   }
165   @Override
166   public Map<String, Object> toMap() {
167     Map<String, Object> m = Utils.makeMap("useCompoundFile", effectiveUseCompoundFileSetting,
168         "maxBufferedDocs", maxBufferedDocs,
169         "maxMergeDocs", maxMergeDocs,
170         "mergeFactor", mergeFactor,
171         "ramBufferSizeMB", ramBufferSizeMB,
172         "writeLockTimeout", writeLockTimeout,
173         "lockType", lockType,
174         "infoStreamEnabled", infoStream != InfoStream.NO_OUTPUT);
175     if(mergeSchedulerInfo != null) m.put("mergeScheduler",mergeSchedulerInfo.toMap());
176     if(mergePolicyInfo != null) m.put("mergePolicy",mergePolicyInfo.toMap());
177     if(mergedSegmentWarmerInfo != null) m.put("mergedSegmentWarmer",mergedSegmentWarmerInfo.toMap());
178     return m;
179   }
180 
181   private PluginInfo getPluginInfo(String path, SolrConfig solrConfig, PluginInfo def)  {
182     List<PluginInfo> l = solrConfig.readPluginInfos(path, false, true);
183     return l.isEmpty() ? def : l.get(0);
184   }

186   private static class DelayedSchemaAnalyzer extends DelegatingAnalyzerWrapper {
187     private final SolrCore core;
188 
189     public DelayedSchemaAnalyzer(SolrCore core) {
190       super(PER_FIELD_REUSE_STRATEGY);
191       this.core = core;
192     }
193 
194     @Override
195     protected Analyzer getWrappedAnalyzer(String fieldName) {
196       return core.getLatestSchema().getIndexAnalyzer();
197     }
198   }
199 
200   public IndexWriterConfig toIndexWriterConfig(SolrCore core) {
201     IndexSchema schema = core.getLatestSchema();
202     IndexWriterConfig iwc = new IndexWriterConfig(new DelayedSchemaAnalyzer(core));
203     if (maxBufferedDocs != -1)
204       iwc.setMaxBufferedDocs(maxBufferedDocs);
205 
206     if (ramBufferSizeMB != -1)
207       iwc.setRAMBufferSizeMB(ramBufferSizeMB);
208 
209     if (writeLockTimeout != -1)
210       iwc.setWriteLockTimeout(writeLockTimeout);
211 
212     iwc.setSimilarity(schema.getSimilarity());
213     iwc.setMergePolicy(buildMergePolicy(schema));
214     iwc.setMergeScheduler(buildMergeScheduler(schema));
215     iwc.setInfoStream(infoStream);
216 
217     // do this after buildMergePolicy since the backcompat logic 
218     // there may modify the effective useCompoundFile
219     iwc.setUseCompoundFile(getUseCompoundFile());
220 
221     if (mergedSegmentWarmerInfo != null) {
222       // TODO: add infostream -> normal logging system (there is an issue somewhere)
223       IndexReaderWarmer warmer = schema.getResourceLoader().newInstance(mergedSegmentWarmerInfo.className, 
224                                                                         IndexReaderWarmer.class,
225                                                                         null,
226                                                                         new Class[] { InfoStream.class },
227                                                                         new Object[] { iwc.getInfoStream() });
228       iwc.setMergedSegmentWarmer(warmer);
229     }
230 
231     return iwc;
232   }

234   /**
235    * Builds a MergePolicy, may also modify the value returned by
236    * getUseCompoundFile() for use by the IndexWriterConfig if 
237    * "useCompoundFile" is specified as an init arg for 
238    * an out of the box MergePolicy that no longer supports it
239    *
240    * @see #fixUseCFMergePolicyInitArg
241    * @see #getUseCompoundFile
242    */
243   private MergePolicy buildMergePolicy(IndexSchema schema) {
244     String mpClassName = mergePolicyInfo == null ? defaultMergePolicyClassName : mergePolicyInfo.className;
245 
246     MergePolicy policy = schema.getResourceLoader().newInstance(mpClassName, MergePolicy.class);
247 
248     if (policy instanceof LogMergePolicy) {
249       LogMergePolicy logMergePolicy = (LogMergePolicy) policy;
250       fixUseCFMergePolicyInitArg(LogMergePolicy.class);
251 
252       if (maxMergeDocs != -1)
253         logMergePolicy.setMaxMergeDocs(maxMergeDocs);
254 
255       if (mergeFactor != -1)
256         logMergePolicy.setMergeFactor(mergeFactor);
257 
258     } else if (policy instanceof TieredMergePolicy) {
259       TieredMergePolicy tieredMergePolicy = (TieredMergePolicy) policy;
260       fixUseCFMergePolicyInitArg(TieredMergePolicy.class);
261       
262       if (mergeFactor != -1) {
263         tieredMergePolicy.setMaxMergeAtOnce(mergeFactor);
264         tieredMergePolicy.setSegmentsPerTier(mergeFactor);
265       }
266 
267 
268     } else if (mergeFactor != -1) {
269       log.warn("Use of <mergeFactor> cannot be configured if merge policy is not an instance of LogMergePolicy or TieredMergePolicy. The configured policy's defaults will be used.");
270     }
271 
272     if (mergePolicyInfo != null)
273       SolrPluginUtils.invokeSetters(policy, mergePolicyInfo.initArgs);
274 
275     return policy;
276   }
277 
278   private MergeScheduler buildMergeScheduler(IndexSchema schema) {
279     String msClassName = mergeSchedulerInfo == null ? SolrIndexConfig.DEFAULT_MERGE_SCHEDULER_CLASSNAME : mergeSchedulerInfo.className;
280     MergeScheduler scheduler = schema.getResourceLoader().newInstance(msClassName, MergeScheduler.class);
281 
282     if (mergeSchedulerInfo != null) {
283       // LUCENE-5080: these two setters are removed, so we have to invoke setMaxMergesAndThreads
284       // if someone has them configured.
285       if (scheduler instanceof ConcurrentMergeScheduler) {
286         NamedList args = mergeSchedulerInfo.initArgs.clone();
287         Integer maxMergeCount = (Integer) args.remove("maxMergeCount");
288         if (maxMergeCount == null) {
289           maxMergeCount = ((ConcurrentMergeScheduler) scheduler).getMaxMergeCount();
290         }
291         Integer maxThreadCount = (Integer) args.remove("maxThreadCount");
292         if (maxThreadCount == null) {
293           maxThreadCount = ((ConcurrentMergeScheduler) scheduler).getMaxThreadCount();
294         }
295         ((ConcurrentMergeScheduler)scheduler).setMaxMergesAndThreads(maxMergeCount, maxThreadCount);
296         SolrPluginUtils.invokeSetters(scheduler, args);
297       } else {
298         SolrPluginUtils.invokeSetters(scheduler, mergeSchedulerInfo.initArgs);
299       }
300     }
301 
302     return scheduler;
303   }

305   public boolean getUseCompoundFile() {
306     return effectiveUseCompoundFileSetting;

309   /**
310    * Lucene 4.4 removed the setUseCompoundFile(boolean) method from the two 
311    * conrete MergePolicies provided with Lucene/Solr and added it to the 
312    * IndexWriterConfig.  
313    * In the event that users have a value explicitly configured for this 
314    * setter in their MergePolicy init args, we remove it from the MergePolicy 
315    * init args, update the 'effective' useCompoundFile setting used by the 
316    * IndexWriterConfig, and warn about discontinuing to use this init arg.
317    * 
318    * @see #getUseCompoundFile
319    */
320   private void fixUseCFMergePolicyInitArg(Class c) {
321 
322     if (null == mergePolicyInfo || null == mergePolicyInfo.initArgs) return;
323 
324     Object useCFSArg = mergePolicyInfo.initArgs.remove("useCompoundFile");
325     if (null != useCFSArg) {
326       log.warn("Ignoring 'useCompoundFile' specified as an init arg for the <mergePolicy> since it is no directly longer supported by " + c.getSimpleName());
327       if (useCFSArg instanceof Boolean) {
328         boolean cfs = ((Boolean)useCFSArg).booleanValue();
329         log.warn("Please update your config to specify <useCompoundFile>"+cfs+"</useCompoundFile> directly in your <indexConfig> settings.");
330         effectiveUseCompoundFileSetting = cfs;
331       } else {
332         log.error("MergePolicy's 'useCompoundFile' init arg is not a boolean, can not apply back compat logic to apply to the IndexWriterConfig: " + useCFSArg.toString());
333       }
334     }
