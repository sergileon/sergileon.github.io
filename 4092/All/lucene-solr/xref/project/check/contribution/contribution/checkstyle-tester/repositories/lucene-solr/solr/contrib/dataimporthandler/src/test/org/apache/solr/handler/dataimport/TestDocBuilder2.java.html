207   @Test
208   @SuppressWarnings("unchecked")
209   public void testDeleteDocs() throws Exception {
210     List rows = new ArrayList();
211     rows.add(createMap("id", "1", "desc", "one"));
212     rows.add(createMap("id", "2", "desc", "two"));
213     rows.add(createMap("id", "3", "desc", "two", DocBuilder.DELETE_DOC_BY_ID, "2"));
214     MockDataSource.setIterator("select * from x", rows.iterator());
215 
216     runFullImport(dataConfigForSkipTransform);
217 
218     assertQ(req("id:1"), "//*[@numFound='1']");
219     assertQ(req("id:2"), "//*[@numFound='0']");
220     assertQ(req("id:3"), "//*[@numFound='1']");
221 
222     assertTrue("Update request processor processDelete was not called", TestUpdateRequestProcessor.processDeleteCalled);
223     assertTrue("Update request processor finish was not called", TestUpdateRequestProcessor.finishCalled);
224     
225     MockDataSource.clearCache();
226     rows = new ArrayList();
227     rows.add(createMap("id", "1", "desc", "one"));
228     rows.add(createMap("id", "2", "desc", "one"));
229     rows.add(createMap("id", "3", "desc", "two", DocBuilder.DELETE_DOC_BY_QUERY, "desc:one"));
230     MockDataSource.setIterator("select * from x", rows.iterator());
231 
232     runFullImport(dataConfigForSkipTransform);
233 
234     assertQ(req("id:1"), "//*[@numFound='0']");
235     assertQ(req("id:2"), "//*[@numFound='0']");
236     assertQ(req("id:3"), "//*[@numFound='1']");
237     
238     assertTrue("Update request processor processDelete was not called", TestUpdateRequestProcessor.processDeleteCalled);
239     assertTrue("Update request processor finish was not called", TestUpdateRequestProcessor.finishCalled);
240     
241     MockDataSource.clearCache();
242     rows = new ArrayList();
243     rows.add(createMap(DocBuilder.DELETE_DOC_BY_ID, "3"));
244     MockDataSource.setIterator("select * from x", rows.iterator());
245     runFullImport(dataConfigForSkipTransform, createMap("clean","false"));
246     assertQ(req("id:3"), "//*[@numFound='0']");
247     
248     assertTrue("Update request processor processDelete was not called", TestUpdateRequestProcessor.processDeleteCalled);
249     assertTrue("Update request processor finish was not called", TestUpdateRequestProcessor.finishCalled);
250     
251   }
