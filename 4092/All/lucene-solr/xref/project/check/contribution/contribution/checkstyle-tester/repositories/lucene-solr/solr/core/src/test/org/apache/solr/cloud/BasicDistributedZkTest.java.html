396   private void testShardParamVariations() throws Exception {
397     SolrQuery query = new SolrQuery("*:*");
398     Map<String,Long> shardCounts = new HashMap<>();
399 
400     for (String shard : shardToJetty.keySet()) {
401       // every client should give the same numDocs for this shard
402       // shffle the clients in a diff order for each shard
403       List<SolrClient> solrclients = new ArrayList<>(this.clients);
404       Collections.shuffle(solrclients, random());
405       for (SolrClient client : solrclients) {
406         query.set("shards", shard);
407         long numDocs = client.query(query).getResults().getNumFound();
408         assertTrue("numDocs < 0 for shard "+shard+" via "+client,
409                    0 <= numDocs);
410         if (!shardCounts.containsKey(shard)) {
411           shardCounts.put(shard, numDocs);
412         }
413         assertEquals("inconsitent numDocs for shard "+shard+" via "+client,
414                      shardCounts.get(shard).longValue(), numDocs);
415         
416         List<CloudJettyRunner> replicaJetties 
417           = new ArrayList<>(shardToJetty.get(shard));
418         Collections.shuffle(replicaJetties, random());
419 
420         // each replica should also give the same numDocs
421         ArrayList<String> replicaAlts = new ArrayList<>(replicaJetties.size() * 2);
422         for (CloudJettyRunner replicaJetty : shardToJetty.get(shard)) {
423           String replica = replicaJetty.url;
424           query.set("shards", replica);
425 
426           // replicas already shuffled, use this in the alternative check below
427           if (0 == random().nextInt(3) || replicaAlts.size() < 2) {
428             replicaAlts.add(replica);
429           }
430 
431           numDocs = client.query(query).getResults().getNumFound();
432           assertTrue("numDocs < 0 for replica "+replica+" via "+client,
433                      0 <= numDocs);
434           assertEquals("inconsitent numDocs for shard "+shard+
435                        " in replica "+replica+" via "+client,
436                        shardCounts.get(shard).longValue(), numDocs);
437         }
438 
439         // any combination of replica alternatives should give same numDocs
440         String replicas = StringUtils.join(replicaAlts.toArray(), "|");
441         query.set("shards", replicas);
442         numDocs = client.query(query).getResults().getNumFound();
443         assertTrue("numDocs < 0 for replicas "+replicas+" via "+client,
444                    0 <= numDocs);
445           assertEquals("inconsitent numDocs for replicas "+replicas+
446                        " via "+client,
447                        shardCounts.get(shard).longValue(), numDocs);
448       }
449     }
450 
451     // sums of multiple shards should add up regardless of how we 
452     // query those shards or which client we use
453     long randomShardCountsExpected = 0;
454     ArrayList<String> randomShards = new ArrayList<>(shardCounts.size());
455     for (Map.Entry<String,Long> shardData : shardCounts.entrySet()) {
456       if (random().nextBoolean() || randomShards.size() < 2) {
457         String shard = shardData.getKey();
458         randomShardCountsExpected += shardData.getValue();
459         if (random().nextBoolean()) {
460           // use shard id
461           randomShards.add(shard);
462         } else {
463           // use some set explicit replicas
464           ArrayList<String> replicas = new ArrayList<>(7);
465           for (CloudJettyRunner replicaJetty : shardToJetty.get(shard)) {
466             if (0 == random().nextInt(3) || 0 == replicas.size()) {
467               replicas.add(replicaJetty.url);
468             }
469           }
470           Collections.shuffle(replicas, random());
471           randomShards.add(StringUtils.join(replicas, "|"));
472         }
473       }
474     }
475     String randShards = StringUtils.join(randomShards, ",");
476     query.set("shards", randShards);
477     for (SolrClient client : this.clients) {
478       assertEquals("numDocs for "+randShards+" via "+client,
479                    randomShardCountsExpected, 
480                    client.query(query).getResults().getNumFound());
481     }
482 
483     // total num docs must match sum of every shard's numDocs
484     query = new SolrQuery("*:*");
485     long totalShardNumDocs = 0;
486     for (Long c : shardCounts.values()) {
487       totalShardNumDocs += c;
488     }
489     for (SolrClient client : clients) {
