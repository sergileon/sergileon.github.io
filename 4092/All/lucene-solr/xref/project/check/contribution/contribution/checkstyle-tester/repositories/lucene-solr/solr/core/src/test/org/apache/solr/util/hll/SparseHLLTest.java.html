109     /**
110      * Smoke test for {@link HLL#cardinality()} and the proper use of the small
111      * range correction.
112      */
113     @Test
114     public void smallRangeSmokeTest() {
115         final int log2m = 11;
116         final int m = (1 << log2m);
117         final int regwidth = 5;
118 
119         // only one register set
120         {
121             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, 256/*sparseThreshold, arbitrary*/, HLLType.SPARSE);
122             hll.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, 0, 1));
123 
124             final long cardinality = hll.cardinality();
125 
126             // Trivially true that small correction conditions hold: one register
127             // set implies zeroes exist, and estimator trivially smaller than 5m/2.
128             // Small range correction: m * log(m/V)
129             final long expected = (long)Math.ceil(m * Math.log((double)m / (m - 1)/*# of zeroes*/));
130             assertEquals(cardinality, expected);
131         }
132 
133         // all but one register set
134         {
135             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, 256/*sparseThreshold, arbitrary*/, HLLType.SPARSE);
136             for(int i=0; i<(m - 1); i++) {
137                 hll.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, i, 1));
138             }
139 
140             // Trivially true that small correction conditions hold: all but
141             // one register set implies a zero exists, and estimator trivially
142             // smaller than 5m/2 since it's alpha / ((m-1)/2)
143             final long cardinality = hll.cardinality();
144 
145             // Small range correction: m * log(m/V)
146             final long expected = (long)Math.ceil(m * Math.log((double)m / 1/*# of zeroes*/));
147             assertEquals(cardinality, expected);
148         }
149     }

151     /**
152      * Smoke test for {@link HLL#cardinality()} and the proper use of the
153      * uncorrected estimator.
154      */
155     @Test
156     public void normalRangeSmokeTest() {
157         final int log2m = 11;
158         final int m = (1 << log2m);
159         final int regwidth = 5;
160         // regwidth = 5, so hash space is
161         // log2m + (2^5 - 1 - 1), so L = log2m + 30
162         final int l = log2m + 30;
163 
164         // all registers at 'medium' value
165         {
166             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, m/*sparseThreshold*/, HLLType.SPARSE);
167 
168             final int registerValue = 7/*chosen to ensure neither correction kicks in*/;
169             for(int i=0; i<m; i++) {
170                 hll.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, i, registerValue));
171             }
172 
173             final long cardinality = hll.cardinality();
174 
175             // Simplified estimator when all registers take same value: alpha / (m/2^val)
176             final double estimator = HLLUtil.alphaMSquared(m)/((double)m/Math.pow(2, registerValue));
177 
178             // Assert conditions for uncorrected range
179             assertTrue(estimator <= Math.pow(2,l)/30);
180             assertTrue(estimator > (5 * m /(double)2));
181 
182             final long expected = (long)Math.ceil(estimator);
183             assertEquals(cardinality, expected);
184         }
185     }

187     /**
188      * Smoke test for {@link HLL#cardinality()} and the proper use of the large
189      * range correction.
190      */
191     @Test
192     public void largeRangeSmokeTest() {
193         final int log2m = 11;
194         final int m = (1 << log2m);
195         final int regwidth = 5;
196         // regwidth = 5, so hash space is
197         // log2m + (2^5 - 1 - 1), so L = log2m + 30
198         final int l = log2m + 30;
199 
200         // all registers at large value
201         {
202             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, m/*sparseThreshold*/, HLLType.SPARSE);
203 
204             final int registerValue = 31/*chosen to ensure large correction kicks in*/;
205             for(int i=0; i<m; i++) {
206                 hll.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, i, registerValue));
207             }
208 
209             final long cardinality = hll.cardinality();
210 
211 
212             // Simplified estimator when all registers take same value: alpha / (m/2^val)
213             final double estimator = HLLUtil.alphaMSquared(m)/((double)m/Math.pow(2, registerValue));
214 
215             // Assert conditions for large range
216             assertTrue(estimator > Math.pow(2, l)/30);
217 
218             // Large range correction: -2^32 * log(1 - E/2^32)
219             final long expected = (long)Math.ceil(-1.0 * Math.pow(2, l) * Math.log(1.0 - estimator/Math.pow(2, l)));
220             assertEquals(cardinality, expected);
221         }
222     }

224     /**
225      * Tests {@link HLL#union(HLL)}.
226      */
227     @Test
228     public void unionTest() {
229         final int log2m = 11/*arbitrary*/;
230         final int sparseThreshold = 256/*arbitrary*/;
231 
232         { // two empty multisets should union to an empty set
233             final HLL hllA = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
234             final HLL hllB = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
235 
236             hllA.union(hllB);
237 
238             assertEquals(hllA.getType(), HLLType.SPARSE/*unchanged*/);
239             assertEquals(hllA.cardinality(), 0L);
240         }
241         { // two disjoint multisets should union properly
242             final HLL hllA = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
243             hllA.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, 1, 1));
244             final HLL hllB = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
245             hllB.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, 2, 1));
246 
247 
248             hllA.union(hllB);
249 
250             assertEquals(hllA.getType(), HLLType.SPARSE/*unchanged*/);
251             assertEquals(hllA.cardinality(), 3L/*precomputed*/);
252             assertRegisterPresent(hllA, 1, (byte)1);
253             assertRegisterPresent(hllA, 2, (byte)1);
254         }
255         { // two exactly overlapping multisets should union properly
256             final HLL hllA = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
257             hllA.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, 1, 10));
258             final HLL hllB = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
259             hllB.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, 1, 13));
260 
261             hllA.union(hllB);
262 
263             assertEquals(hllA.getType(), HLLType.SPARSE/*unchanged*/);
264             assertEquals(hllA.cardinality(), 2L/*precomputed*/);
265             assertOneRegisterSet(hllA, 1, (byte)13/*max(10,13)*/);
266         }
267         { // overlapping multisets should union properly
268             final HLL hllA = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
269             final HLL hllB = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
270             // register index = 3
271             final long rawValueA = ProbabilisticTestUtil.constructHLLValue(log2m, 3, 11);
272 
273             // register index = 4
274             final long rawValueB = ProbabilisticTestUtil.constructHLLValue(log2m, 4, 13);
275             final long rawValueBPrime = ProbabilisticTestUtil.constructHLLValue(log2m, 4, 21);
276 
277             // register index = 5
278             final long rawValueC = ProbabilisticTestUtil.constructHLLValue(log2m, 5, 14);
279 
280             hllA.addRaw(rawValueA);
281             hllA.addRaw(rawValueB);
282 
283             hllB.addRaw(rawValueBPrime);
284             hllB.addRaw(rawValueC);
285 
286             hllA.union(hllB);
287             // union should have three registers set, with partition B set to the
288             // max of the two registers
289             assertRegisterPresent(hllA, 3, (byte)11);
290             assertRegisterPresent(hllA, 4, (byte)21/*max(21,13)*/);
291             assertRegisterPresent(hllA, 5, (byte)14);
292         }
293         { // too-large unions should promote
294             final HLL hllA = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
295             final HLL hllB = new HLL(log2m, 5/*regwidth*/, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
296 
297             // fill up sets to maxCapacity
298             for(int i=0; i<sparseThreshold; i++) {
299                 hllA.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, i, 1));
300                 hllB.addRaw(ProbabilisticTestUtil.constructHLLValue(log2m, (i + sparseThreshold)/*non-overlapping*/, 1));
301             }
302 
303             hllA.union(hllB);
304 
305             assertEquals(hllA.getType(), HLLType.FULL);
306         }
307     }

320     /**
321      * Tests {@link HLL#toBytes(ISchemaVersion)} and
322      * {@link HLL#fromBytes(byte[])}.
323      */
324     @Test
325     public void toFromBytesTest() {
326         final int log2m = 11/*arbitrary*/;
327         final int regwidth = 5/*arbitrary*/;
328         final int sparseThreshold = 256/*arbitrary*/;
329         final int shortWordLength = 16/*log2m + regwidth = 11 + 5*/;
330 
331         final ISchemaVersion schemaVersion = SerializationUtil.DEFAULT_SCHEMA_VERSION;
332         final HLLType type = HLLType.SPARSE;
333         final int padding = schemaVersion.paddingBytes(type);
334 
335         {// Should work on an empty element
336             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
337             final byte[] bytes = hll.toBytes(schemaVersion);
338 
339             // output should just be padding since no registers are used
340             assertEquals(bytes.length, padding);
341 
342             final HLL inHLL = HLL.fromBytes(bytes);
343 
344             // assert register values correct
345             assertElementsEqual(hll, inHLL);
346         }
347         {// Should work on a partially filled element
348             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
349 
350             for(int i=0; i<3; i++) {
351                 final long rawValue = ProbabilisticTestUtil.constructHLLValue(log2m, i, (i+9));
352                 hll.addRaw(rawValue);
353             }
354 
355             final byte[] bytes = hll.toBytes(schemaVersion);
356 
357             assertEquals(bytes.length, padding + ProbabilisticTestUtil.getRequiredBytes(shortWordLength, 3/*registerCount*/));
358 
359             final HLL inHLL = HLL.fromBytes(bytes);
360 
361             // assert register values correct
362             assertElementsEqual(hll, inHLL);
363         }
364         {// Should work on a full set
365             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
366 
367             for(int i=0; i<sparseThreshold; i++) {
368                 final long rawValue = ProbabilisticTestUtil.constructHLLValue(log2m, i, (i % 9) + 1);
369                 hll.addRaw(rawValue);
370             }
371 
372             final byte[] bytes = hll.toBytes(schemaVersion);
373 
374             // 'short words' should be 12 bits + 5 bits = 17 bits long
375             assertEquals(bytes.length, padding + ProbabilisticTestUtil.getRequiredBytes(shortWordLength, sparseThreshold));
376 
377             final HLL inHLL = HLL.fromBytes(bytes);
378 
379             // assert register values correct
380             assertElementsEqual(hll, inHLL);
381         }
382     }

384     /**
385      * Smoke tests the multisets by adding random values.
386      */
387     @Test
388     public void randomValuesTest() {
389         final int log2m = 11/*arbitrary*/;
390         final int regwidth = 5/*arbitrary*/;
391         final int sparseThreshold = 256/*arbitrary*/;
392 
393         for(int run=0; run<100; run++) {
394             final HLL hll = new HLL(log2m, regwidth, 128/*explicitThreshold, arbitrary, unused*/, sparseThreshold, HLLType.SPARSE);
395 
396             final IntByteHashMap map = new IntByteHashMap();
397 
398             for(int i=0; i<sparseThreshold; i++) {
399                 final long rawValue = RandomizedTest.randomLong();
400 
401                 final short registerIndex = ProbabilisticTestUtil.getRegisterIndex(rawValue, log2m);
402                 final byte registerValue = ProbabilisticTestUtil.getRegisterValue(rawValue, log2m);
403                 if(map.get(registerIndex) < registerValue) {
404                     map.put(registerIndex, registerValue);
405                 }
406 
407                 hll.addRaw(rawValue);
408             }
409 
410             for (IntByteCursor c : map) {
411                 final byte expectedRegisterValue = map.get(c.key);
412                 assertRegisterPresent(hll, c.key, expectedRegisterValue);
413             }
414         }
415     }

419     /**
420      * Asserts that the register at the specified index is set to the specified
421      * value.
422      */
423     private static void assertRegisterPresent(final HLL hll,
424                                               final int registerIndex,
425                                               final int registerValue) {
426         final IntByteHashMap sparseProbabilisticStorage = hll.sparseProbabilisticStorage;
427         assertEquals(sparseProbabilisticStorage.get(registerIndex), registerValue);
428     }
429 
430     /**
431      * Asserts that only the specified register is set and has the specified value.
432      */
433     private static void assertOneRegisterSet(final HLL hll,
434                                              final int registerIndex,
435                                              final byte registerValue) {
436         final IntByteHashMap sparseProbabilisticStorage = hll.sparseProbabilisticStorage;
437         assertEquals(sparseProbabilisticStorage.size(), 1);
438         assertEquals(sparseProbabilisticStorage.get(registerIndex), registerValue);
439     }
440 
441     /**
442      * Asserts that all registers in the two {@link HLL} instances are identical.
443      */
444     private static void assertElementsEqual(final HLL hllA, final HLL hllB) {
445         final IntByteHashMap sparseProbabilisticStorageA = hllA.sparseProbabilisticStorage;
446         final IntByteHashMap sparseProbabilisticStorageB = hllB.sparseProbabilisticStorage;
447         assertEquals(sparseProbabilisticStorageA.size(), sparseProbabilisticStorageB.size());
448         for (IntByteCursor c : sparseProbabilisticStorageA) {
449             assertEquals(sparseProbabilisticStorageA.get(c.key), 
450                          sparseProbabilisticStorageB.get(c.key));
451         }
452     }
