48  /**
49   * Clones the values found in any matching <code>source</code> field into 
50   * a configured <code>dest</code> field.
51   * <p>
52   * The <code>source</code> field(s) can be configured as either:
53   * </p>
54   * <ul>
55   *  <li>One or more <code>&lt;str&gt;</code></li>
56   *  <li>An <code>&lt;arr&gt;</code> of <code>&lt;str&gt;</code></li>
57   *  <li>A <code>&lt;lst&gt;</code> containing {@link FieldMutatingUpdateProcessorFactory FieldMutatingUpdateProcessorFactory style selector arguments}</li>
58   * </ul>
59   *
60   * <p> The <code>dest</code> field can be a single <code>&lt;str&gt;</code> 
61   * containing the literal name of a destination field, or it may be a <code>&lt;lst&gt;</code> specifying a 
62   * regex <code>pattern</code> and a <code>replacement</code> string. If the pattern + replacement option 
63   * is used the pattern will be matched against all fields matched by the source selector, and the replacement 
64   * string (including any capture groups specified from the pattern) will be evaluated a using 
65   * {@link Matcher#replaceAll(String)} to generate the literal name of the destination field.
66   * </p>
67   *
68   * <p>If the resolved <code>dest</code> field already exists in the document, then the 
69   * values from the <code>source</code> fields will be added to it.  The 
70   * "boost" value associated with the <code>dest</code> will not be changed, 
71   * and any boost specified on the <code>source</code> fields will be ignored.  
72   * (If the <code>dest</code> field did not exist prior to this processor, the 
73   * newly created <code>dest</code> field will have the default boost of 1.0)
74   * </p>
75   * <p>
76   * In the example below:
77   * </p>
78   * <ul>
79   *   <li>The <code>category</code> field will be cloned into the <code>category_s</code> field</li>
80   *   <li>Both the <code>authors</code> and <code>editors</code> fields will be cloned into the 
81   *       <code>contributors</code> field
82   *   </li>
83   *   <li>Any field with a name ending in <code>_price</code> -- except for 
84   *       <code>list_price</code> -- will be cloned into the <code>all_prices</code>
85   *   </li>
86   *   <li>Any field name beginning with feat and ending in s (i.e. feats or features) 
87   *       will be cloned into a field prefixed with key_ and not ending in s. (i.e. key_feat or key_feature)
88   *   </li>
89   * </ul>
90   *
91   * <!-- see solrconfig-update-processors-chains.xml and 
92   *      CloneFieldUpdateProcessorFactoryTest.testCloneFieldExample for where this is tested -->
93   * <pre class="prettyprint">
94   *   &lt;updateRequestProcessorChain name="multiple-clones"&gt;
95   *     &lt;processor class="solr.CloneFieldUpdateProcessorFactory"&gt;
96   *       &lt;str name="source"&gt;category&lt;/str&gt;
97   *       &lt;str name="dest"&gt;category_s&lt;/str&gt;
98   *     &lt;/processor&gt;
99   *     &lt;processor class="solr.CloneFieldUpdateProcessorFactory"&gt;
100  *       &lt;arr name="source"&gt;
101  *         &lt;str&gt;authors&lt;/str&gt;
102  *         &lt;str&gt;editors&lt;/str&gt;
103  *       &lt;/arr&gt;
104  *       &lt;str name="dest"&gt;contributors&lt;/str&gt;
105  *     &lt;/processor&gt;
106  *     &lt;processor class="solr.CloneFieldUpdateProcessorFactory"&gt;
107  *       &lt;lst name="source"&gt;
108  *         &lt;str name="fieldRegex"&gt;.*_price$&lt;/str&gt;
109  *         &lt;lst name="exclude"&gt;
110  *           &lt;str name="fieldName"&gt;list_price&lt;/str&gt;
111  *         &lt;/lst&gt;
112  *       &lt;/lst&gt;
113  *       &lt;str name="dest"&gt;all_prices&lt;/str&gt;
114  *     &lt;/processor&gt;
115  *     &lt;processor class="solr.processor.CloneFieldUpdateProcessorFactory"&gt;
116  *       &lt;lst name="source"&gt;
117  *         &lt;str name="fieldRegex"&gt;^feat(.*)s$&lt;/str&gt;
118  *       &lt;/lst&gt;
119  *       &lt;lst name="dest"&gt;
120  *         &lt;str name="pattern"&gt;^feat(.*)s$&lt;/str&gt;
121  *         &lt;str name="replacement"&gt;key_feat$1&lt;/str&gt;
122  *       &lt;/str&gt;
123  *     &lt;/processor&gt;
124  *   &lt;/updateRequestProcessorChain&gt;
125  * </pre>
126  *
127  * <p>
128  * In common case situations where you wish to use a single regular expression as both a 
129  * <code>fieldRegex</code> selector and a destination <code>pattern</code>, a "short hand" syntax 
130  * is support for convinience: The <code>pattern</code> and <code>replacement</code> may be specified 
131  * at the top level, omitting <code>source</code> and <code>dest</code> declarations completely, and 
132  * the <code>pattern</code> will be used to construct an equivilent <code>source</code> selector internally.
133  * </p>
134  * <p>
135  * For example, both of the following configurations are equivilent:
136  * </p>
137  * <pre class="prettyprint">
138  * &lt;!-- full syntax --&gt;
139  * &lt;processor class="solr.processor.CloneFieldUpdateProcessorFactory"&gt;
140  *   &lt;lst name="source"&gt;
141  *     &lt;str name="fieldRegex"^gt;$feat(.*)s$&lt;/str&gt;
142  *   &lt;/lst&gt;
143  *   &lt;lst name="dest"&gt;
144  *     &lt;str name="pattern"&gt;^feat(.*)s$&lt;/str&gt;
145  *     &lt;str name="replacement"&gt;key_feat$1&lt;/str&gt;
146  *   &lt;/str&gt;
147  * &lt;/processor&gt;
148  * 
149  * &lt;!-- syntactic sugar syntax --&gt;
150  * &lt;processor class="solr.processor.CloneFieldUpdateProcessorFactory"&gt;
151  *   &lt;str name="pattern"&gt;^feat(.*)s$&lt;/str&gt;
152  *   &lt;str name="replacement"&gt;key_feat$1&lt;/str&gt;
153  * &lt;/processor&gt;
154  * </pre>
155  *
156  * <p>
157  * When cloning multiple fields (or a single multivalued field) into a single valued field, one of the 
158  * {@link FieldValueSubsetUpdateProcessorFactory} implementations configured after the 
159  * <code>CloneFieldUpdateProcessorFactory</code> can be useful to reduce the list of values down to a 
160  * single value.
161  * </p>
162  * 
163  * @see FieldValueSubsetUpdateProcessorFactory
164  */
165 public class CloneFieldUpdateProcessorFactory 
166   extends UpdateRequestProcessorFactory implements SolrCoreAware {
167 
168   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
169   
170   public static final String SOURCE_PARAM = "source";
171   public static final String DEST_PARAM = "dest";
172   public static final String PATTERN_PARAM = "pattern";
173   public static final String REPLACEMENT_PARAM = "replacement";
174 
175   private SelectorParams srcInclusions = new SelectorParams();
176   private Collection<SelectorParams> srcExclusions 
177     = new ArrayList<>();
178 
179   private FieldNameSelector srcSelector = null;
180 
181   /** 
182    * If pattern is null, this this is a literal field name.  If pattern is non-null then this
183    * is a replacement string that may contain meta-characters (ie: capture group identifiers)
184    * @see #pattern
185    */
186   private String dest = null;
187   /** @see #dest */
188   private Pattern pattern = null;
189 
190   protected final FieldNameSelector getSourceSelector() {
191     if (null != srcSelector) return srcSelector;
192 
193     throw new SolrException(SERVER_ERROR, "selector was never initialized, "+
194                             " inform(SolrCore) never called???");
195   }

197   @SuppressWarnings("unchecked")
198   @Override
199   public void init(NamedList args) {
200 
201     // high level (loose) check for which type of config we have.
202     // 
203     // individual init methods do more strict syntax checking
204     if (0 <= args.indexOf(SOURCE_PARAM, 0) && 0 <= args.indexOf(DEST_PARAM, 0) ) {
205       initSourceSelectorSyntax(args);
206     } else if (0 <= args.indexOf(PATTERN_PARAM, 0) && 0 <= args.indexOf(REPLACEMENT_PARAM, 0)) {
207       initSimpleRegexReplacement(args);
208     } else {
209       throw new SolrException(SERVER_ERROR, "A combination of either '" + SOURCE_PARAM + "' + '"+
210                               DEST_PARAM + "', or '" + REPLACEMENT_PARAM + "' + '" +
211                               PATTERN_PARAM + "' init params are mandatory");
212     }
213     
214     if (0 < args.size()) {
215       throw new SolrException(SERVER_ERROR,
216           "Unexpected init param(s): '" +
217               args.getName(0) + "'");
218     }
219 
220     super.init(args);
221   }
222 
223   /**
224    * init helper method that should only be called when we know for certain that both the 
225    * "source" and "dest" init params do <em>not</em> exist.
226    */
227   @SuppressWarnings("unchecked")
228   private void initSimpleRegexReplacement(NamedList args) {
229     // The syntactic sugar for the case where there is only one regex pattern for source and the same pattern
230     // is used for the destination pattern...
231     //
232     //  pattern != null && replacement != null
233     //    
234     // ...as top level elements, with no other config options specified
235     
236     // if we got here we know we had pattern and replacement, now check for the other two  so that we can give a better
237     // message than "unexpected"
238     if (0 <= args.indexOf(SOURCE_PARAM, 0) || 0 <= args.indexOf(DEST_PARAM, 0) ) {
239       throw new SolrException(SERVER_ERROR,"Short hand syntax must not be mixed with full syntax. Found " + 
240           PATTERN_PARAM + " and " + REPLACEMENT_PARAM + " but also found " + SOURCE_PARAM + " or " + DEST_PARAM);
241     }
242     
243     assert args.indexOf(SOURCE_PARAM, 0) < 0;
244     
245     Object patt = args.remove(PATTERN_PARAM);
246     Object replacement = args.remove(REPLACEMENT_PARAM);
247 
248     if (null == patt || null == replacement) {
249       throw new SolrException(SERVER_ERROR, "Init params '" + PATTERN_PARAM + "' and '" +
250                               REPLACEMENT_PARAM + "' are both mandatory if '" + SOURCE_PARAM + "' and '"+
251                               DEST_PARAM + "' are not both specified");
252     }
253 
254     if (0 != args.size()) {
255       throw new SolrException(SERVER_ERROR, "Init params '" + REPLACEMENT_PARAM + "' and '" +
256                               PATTERN_PARAM + "' must be children of '" + DEST_PARAM +
257                               "' to be combined with other options.");
258     }
259     
260     if (!(replacement instanceof String)) {
261       throw new SolrException(SERVER_ERROR, "Init param '" + REPLACEMENT_PARAM + "' must be a string (i.e. <str>)");
262     }
263     if (!(patt instanceof String)) {
264       throw new SolrException(SERVER_ERROR, "Init param '" + PATTERN_PARAM + "' must be a string (i.e. <str>)");
265     }
266     
267     dest = replacement.toString();
268     try {
269       this.pattern = Pattern.compile(patt.toString());
270     } catch (PatternSyntaxException pe) {
271       throw new SolrException(SERVER_ERROR, "Init param " + PATTERN_PARAM +
272                               " is not a valid regex pattern: " + patt, pe);
273       
274     }
275     srcInclusions = new SelectorParams();
276     srcInclusions.fieldRegex = Collections.singletonList(this.pattern);
277   }
278  
279   /**
280    * init helper method that should only be called when we know for certain that both the 
281    * "source" and "dest" init params <em>do</em> exist.
282    */
283   @SuppressWarnings("unchecked")
284   private void initSourceSelectorSyntax(NamedList args) {
285     // Full and complete syntax where source and dest are mandatory.
286     //
287     // source may be a single string or a selector.
288     // dest may be a single string or list containing pattern and replacement
289     //
290     //   source != null && dest != null
291 
292     // if we got here we know we had source and dest, now check for the other two so that we can give a better
293     // message than "unexpected"
294     if (0 <= args.indexOf(PATTERN_PARAM, 0) || 0 <= args.indexOf(REPLACEMENT_PARAM, 0) ) {
295       throw new SolrException(SERVER_ERROR,"Short hand syntax must not be mixed with full syntax. Found " +
296           SOURCE_PARAM + " and " + DEST_PARAM + " but also found " + PATTERN_PARAM + " or " + REPLACEMENT_PARAM);
297     }
298 
299     Object d = args.remove(DEST_PARAM);
300     assert null != d;
301     
302     List<Object> sources = args.getAll(SOURCE_PARAM);
303     assert null != sources;
304 
305     if (1 == sources.size()) {
306       if (sources.get(0) instanceof NamedList) {
307         // nested set of selector options
308         NamedList selectorConfig = (NamedList) args.remove(SOURCE_PARAM);
309 
310         srcInclusions = parseSelectorParams(selectorConfig);
311 
312         List<Object> excList = selectorConfig.getAll("exclude");
313 
314         for (Object excObj : excList) {
315           if (null == excObj) {
316             throw new SolrException(SERVER_ERROR, "Init param '" + SOURCE_PARAM +
317                                     "' child 'exclude' can not be null");
318           }
319           if (!(excObj instanceof NamedList)) {
320             throw new SolrException(SERVER_ERROR, "Init param '" + SOURCE_PARAM +
321                                     "' child 'exclude' must be <lst/>");
322           }
323           NamedList exc = (NamedList) excObj;
324           srcExclusions.add(parseSelectorParams(exc));
325           if (0 < exc.size()) {
326             throw new SolrException(SERVER_ERROR, "Init param '" + SOURCE_PARAM +
327                                     "' has unexpected 'exclude' sub-param(s): '"
328                                     + selectorConfig.getName(0) + "'");
329           }
330           // call once per instance
331           selectorConfig.remove("exclude");
332         }
333 
334         if (0 < selectorConfig.size()) {
335           throw new SolrException(SERVER_ERROR, "Init param '" + SOURCE_PARAM +
336                                   "' contains unexpected child param(s): '" +
337                                   selectorConfig.getName(0) + "'");
338         }
339         // consume from the named list so it doesn't interfere with subsequent processing
340         sources.remove(0);
341       }
342     }
343     if (1 <= sources.size()) {
344       // source better be one or more strings
345       srcInclusions.fieldName = new HashSet<>(args.removeConfigArgs("source"));
346     }
347     if (srcInclusions == null) {
348       throw new SolrException(SERVER_ERROR, "Init params do not specify anything to clone, please supply either "
349       + SOURCE_PARAM + " and " + DEST_PARAM + " or " + PATTERN_PARAM + " and " + REPLACEMENT_PARAM + ". See javadocs" +
350           "for CloneFieldUpdateProcessorFactory for further details.");
351     }
352     
353     if (d instanceof NamedList) {
354       NamedList destList = (NamedList) d;
355 
356       Object patt = destList.remove(PATTERN_PARAM);
357       Object replacement = destList.remove(REPLACEMENT_PARAM);
358       
359       if (null == patt || null == replacement) {
360         throw new SolrException(SERVER_ERROR, "Init param '" + DEST_PARAM + "' children '" +
361                                 PATTERN_PARAM + "' and '" + REPLACEMENT_PARAM +
362                                 "' are both mandatoryand can not be null");
363       }
364       if (! (patt instanceof String && replacement instanceof String)) {
365         throw new SolrException(SERVER_ERROR, "Init param '" + DEST_PARAM + "' children '" +
366                                 PATTERN_PARAM + "' and '" + REPLACEMENT_PARAM +
367                                 "' must both be strings (i.e. <str>)");
368       }
369       if (0 != destList.size()) {
370         throw new SolrException(SERVER_ERROR, "Init param '" + DEST_PARAM + "' has unexpected children: '"
371                                 + destList.getName(0) + "'");
372       }
373       
374       try {
375         this.pattern = Pattern.compile(patt.toString());
376       } catch (PatternSyntaxException pe) {
377         throw new SolrException(SERVER_ERROR, "Init param '" + DEST_PARAM + "' child '" + PATTERN_PARAM +
378                                 " is not a valid regex pattern: " + patt, pe);
379       }
380       dest = replacement.toString();
381         
382     } else if (d instanceof String) {
383       dest = d.toString();
384     } else {
385       throw new SolrException(SERVER_ERROR, "Init param '" + DEST_PARAM + "' must either be a string " +
386                               "(i.e. <str>) or a list (i.e. <lst>) containing '" +
387                               PATTERN_PARAM + "' and '" + REPLACEMENT_PARAM);
388     }
389 
390   }
391 
392   @Override
393   public void inform(final SolrCore core) {
394     
395     srcSelector = 
396       FieldMutatingUpdateProcessor.createFieldNameSelector
397           (core.getResourceLoader(), core, srcInclusions, FieldMutatingUpdateProcessor.SELECT_NO_FIELDS);
398 
399     for (SelectorParams exc : srcExclusions) {
400       srcSelector = FieldMutatingUpdateProcessor.wrap
401         (srcSelector,
402          FieldMutatingUpdateProcessor.createFieldNameSelector
403              (core.getResourceLoader(), core, exc, FieldMutatingUpdateProcessor.SELECT_NO_FIELDS));
404     }
405   }
406 
407   @Override
408   public final UpdateRequestProcessor getInstance(SolrQueryRequest req,
409                                                   SolrQueryResponse rsp,
410                                                   UpdateRequestProcessor next) {
411     final FieldNameSelector srcSelector = getSourceSelector();
412     return new UpdateRequestProcessor(next) {
413       @Override
414       public void processAdd(AddUpdateCommand cmd) throws IOException {
415 
416         final SolrInputDocument doc = cmd.getSolrInputDocument();
417 
418         // destination may be regex replace string, which can cause multiple output fields.
419         Map<String,SolrInputField> destMap = new HashMap<>();
420 
421         // preserve initial values and boost (if any)
422         for (final String fname : doc.getFieldNames()) {
423           if (! srcSelector.shouldMutate(fname)) continue;
424           
425           Collection<Object> srcFieldValues = doc.getFieldValues(fname);
426           if(srcFieldValues == null || srcFieldValues.isEmpty()) continue;
427           
428           String resolvedDest = dest;
429 
430           if (pattern != null) {
431             Matcher matcher = pattern.matcher(fname);
432             if (matcher.find()) {
433               resolvedDest = matcher.replaceAll(dest);
434             } else {
435               log.debug("CloneFieldUpdateProcessor.srcSelector.shouldMutate(\"{}\") returned true, " +
436                   "but replacement pattern did not match, field skipped.", fname);
437               continue;
438             }
439           }
440           SolrInputField destField;
441           if (doc.containsKey(resolvedDest)) {
442             destField = doc.getField(resolvedDest);
443           } else {
444             SolrInputField targetField = destMap.get(resolvedDest);
445             if (targetField == null) {
446               destField = new SolrInputField(resolvedDest);
447             } else {
448               destField = targetField;
449             }
450           }
451 
452           for (Object val : srcFieldValues) {
453             // preserve existing dest boost (multiplicitive), ignore src boost
454             destField.addValue(val, 1.0f);
455           }
456           // put it in map to avoid concurrent modification...
457           destMap.put(resolvedDest, destField);
458         }
459 
460         for (String dest : destMap.keySet()) {
461           doc.put(dest, destMap.get(dest));
462         }
463         super.processAdd(cmd);
464       }
465     };
466   }
467 
468   /** macro */
469   private static SelectorParams parseSelectorParams(NamedList args) {
470     return FieldMutatingUpdateProcessorFactory.parseSelectorParams(args);
471   }
