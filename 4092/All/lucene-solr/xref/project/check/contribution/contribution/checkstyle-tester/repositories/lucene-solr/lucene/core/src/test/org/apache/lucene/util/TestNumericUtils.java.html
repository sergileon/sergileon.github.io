27    public void testLongConversionAndOrdering() throws Exception {
28      // generate a series of encoded longs, each numerical one bigger than the one before
29      BytesRefBuilder last = new BytesRefBuilder();
30      BytesRefBuilder act = new BytesRefBuilder();
31      for (long l=-100000L; l<100000L; l++) {
32        NumericUtils.longToPrefixCodedBytes(l, 0, act);
33        if (last!=null) {
34          // test if smaller
35          assertTrue("actual bigger than last (BytesRef)", last.get().compareTo(act.get()) < 0 );
36          assertTrue("actual bigger than last (as String)", last.get().utf8ToString().compareTo(act.get().utf8ToString()) < 0 );
37        }
38        // test is back and forward conversion works
39        assertEquals("forward and back conversion should generate same long", l, NumericUtils.prefixCodedToLong(act.get()));
40        // next step
41        last.copyBytes(act);
42      }
43    }

63    public void testLongSpecialValues() throws Exception {
64      long[] vals=new long[]{
65        Long.MIN_VALUE, Long.MIN_VALUE+1, Long.MIN_VALUE+2, -5003400000000L,
66        -4000L, -3000L, -2000L, -1000L, -1L, 0L, 1L, 10L, 300L, 50006789999999999L, Long.MAX_VALUE-2, Long.MAX_VALUE-1, Long.MAX_VALUE
67      };
68      BytesRefBuilder[] prefixVals = new BytesRefBuilder[vals.length];
69      
70      for (int i=0; i<vals.length; i++) {
71        prefixVals[i] = new BytesRefBuilder();
72        NumericUtils.longToPrefixCodedBytes(vals[i], 0, prefixVals[i]);
73        
74        // check forward and back conversion
75        assertEquals( "forward and back conversion should generate same long", vals[i], NumericUtils.prefixCodedToLong(prefixVals[i].get()) );
76  
77        // test if decoding values as int fails correctly
78        try {
79          NumericUtils.prefixCodedToInt(prefixVals[i].get());
80          fail("decoding a prefix coded long value as int should fail");
81        } catch (NumberFormatException e) {
82          // worked
83        }
84      }
85      
86      // check sort order (prefixVals should be ascending)
87      for (int i=1; i<prefixVals.length; i++) {
88        assertTrue( "check sort order", prefixVals[i-1].get().compareTo(prefixVals[i].get()) < 0 );
89      }
90          
91      // check the prefix encoding, lower precision should have the difference to original value equal to the lower removed bits
92      final BytesRefBuilder ref = new BytesRefBuilder();
93      for (int i=0; i<vals.length; i++) {
94        for (int j=0; j<64; j++) {
95          NumericUtils.longToPrefixCodedBytes(vals[i], j, ref);
96          long prefixVal=NumericUtils.prefixCodedToLong(ref.get());
97          long mask=(1L << j) - 1L;
98          assertEquals( "difference between prefix val and original value for "+vals[i]+" with shift="+j, vals[i] & mask, vals[i]-prefixVal );
99        }
100     }
101   }

219   /** Note: The neededBounds Iterable must be unsigned (easier understanding what's happening) */
220   private void assertLongRangeSplit(final long lower, final long upper, int precisionStep,
221     final boolean useBitSet, final Iterable<Long> expectedBounds, final Iterable<Integer> expectedShifts
222   ) {
223     // Cannot use FixedBitSet since the range could be long:
224     final LongBitSet bits=useBitSet ? new LongBitSet(upper-lower+1) : null;
225     final Iterator<Long> neededBounds = (expectedBounds == null) ? null : expectedBounds.iterator();
226     final Iterator<Integer> neededShifts = (expectedShifts == null) ? null : expectedShifts.iterator();
227 
228     NumericUtils.splitLongRange(new NumericUtils.LongRangeBuilder() {
229       @Override
230       public void addRange(long min, long max, int shift) {
231         assertTrue("min, max should be inside bounds", min>=lower && min<=upper && max>=lower && max<=upper);
232         if (useBitSet) for (long l=min; l<=max; l++) {
233           assertFalse("ranges should not overlap", bits.getAndSet(l-lower) );
234           // extra exit condition to prevent overflow on MAX_VALUE
235           if (l == max) break;
236         }
237         if (neededBounds == null || neededShifts == null)
238           return;
239         // make unsigned longs for easier display and understanding
240         min ^= 0x8000000000000000L;
241         max ^= 0x8000000000000000L;
242         //System.out.println("0x"+Long.toHexString(min>>>shift)+"L,0x"+Long.toHexString(max>>>shift)+"L)/*shift="+shift+"*/,");
243         assertEquals( "shift", neededShifts.next().intValue(), shift);
244         assertEquals( "inner min bound", neededBounds.next().longValue(), min>>>shift);
245         assertEquals( "inner max bound", neededBounds.next().longValue(), max>>>shift);
246       }
247     }, precisionStep, lower, upper);
248     
249     if (useBitSet) {
250       // after flipping all bits in the range, the cardinality should be zero
251       bits.flip(0,upper-lower+1);
252       assertEquals("The sub-range concenated should match the whole range", 0, bits.cardinality());
253     }
254   }

354   private void executeOneRandomSplit(final Random random) throws Exception {
355     long lower = randomLong(random);
356     long len = random.nextInt(16384*1024); // not too large bitsets, else OOME!
357     while (lower + len < lower) { // overflow
358       lower >>= 1;
359     }
360     assertLongRangeSplit(lower, lower + len, random.nextInt(64) + 1, true, null, null);
361   }

363   private long randomLong(final Random random) {
364     long val;
365     switch(random.nextInt(4)) {
366       case 0:
367         val = 1L << (random.nextInt(63)); //  patterns like 0x000000100000 (-1 yields patterns like 0x0000fff)
368         break;
369       case 1:
370         val = -1L << (random.nextInt(63)); // patterns like 0xfffff00000
371         break;
372       default:
373         val = random.nextLong();
374     }
375 
376     val += random.nextInt(5)-2;
377 
378     if (random.nextBoolean()) {
379       if (random.nextBoolean()) val += random.nextInt(100)-50;
380       if (random.nextBoolean()) val = ~val;
381       if (random.nextBoolean()) val = val<<1;
382       if (random.nextBoolean()) val = val>>>1;
383     }
384 
385     return val;
386   }
