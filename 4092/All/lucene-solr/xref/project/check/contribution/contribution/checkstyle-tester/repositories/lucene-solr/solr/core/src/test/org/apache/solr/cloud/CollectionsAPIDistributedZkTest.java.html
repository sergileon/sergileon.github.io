243   private void deleteCollectionWithDownNodes() throws Exception {
244     String baseUrl = getBaseUrl((HttpSolrClient) clients.get(0));
245     // now try to remove a collection when a couple of its nodes are down
246     if (secondConfigSet) {
247       try (SolrClient client = createNewSolrClient("", baseUrl)) {
248         createCollection(null, "halfdeletedcollection2", 3, 3, 6, client, null, "conf2");
249       }
250     } else {
251       try (SolrClient client = createNewSolrClient("", baseUrl)) {
252         createCollection(null, "halfdeletedcollection2", 3, 3, 6, client, null);
253       }
254     }
255     
256     waitForRecoveriesToFinish("halfdeletedcollection2", false);
257     
258     // stop a couple nodes
259     ChaosMonkey.stop(jettys.get(0));
260     ChaosMonkey.stop(jettys.get(1));
261     
262     // wait for leaders to settle out
263     for (int i = 1; i < 4; i++) {
264       cloudClient.getZkStateReader().getLeaderRetry("halfdeletedcollection2", "shard" + i, 30000);
265     }
266     
267     baseUrl = getBaseUrl((HttpSolrClient) clients.get(2));
268     
269     // remove a collection
270     ModifiableSolrParams params = new ModifiableSolrParams();
271     params.set("action", CollectionAction.DELETE.toString());
272     params.set("name", "halfdeletedcollection2");
273     QueryRequest request = new QueryRequest(params);
274     request.setPath("/admin/collections");
275     
276     makeRequest(baseUrl, request);
277 
278     TimeOut timeout = new TimeOut(10, TimeUnit.SECONDS);
279     while (cloudClient.getZkStateReader().getClusterState().hasCollection("halfdeletedcollection2")) {
280       if (timeout.hasTimedOut()) {
281         throw new AssertionError("Timeout waiting to see removed collection leave clusterstate");
282       }
283       
284       Thread.sleep(200);
285       cloudClient.getZkStateReader().updateClusterState();
286     }
287 
288     assertFalse("Still found collection that should be gone", cloudClient.getZkStateReader().getClusterState().hasCollection("halfdeletedcollection2"));
289 
290   }

568   private void testCollectionsAPI() throws Exception {
569 
570     boolean disableLegacy = random().nextBoolean();
571     CloudSolrClient client1 = null;
572 
573     if (disableLegacy) {
574       log.info("legacyCloud=false");
575       client1 = createCloudClient(null);
576       setClusterProp(client1, ZkStateReader.LEGACY_CLOUD, "false");
577     }
578 
579     // TODO: fragile - because we dont pass collection.confName, it will only
580     // find a default if a conf set with a name matching the collection name is found, or 
581     // if there is only one conf set. That and the fact that other tests run first in this
582     // env make this pretty fragile
583     
584     // create new collections rapid fire
585     Map<String,List<Integer>> collectionInfos = new HashMap<>();
586     int cnt = random().nextInt(TEST_NIGHTLY ? 6 : 1) + 1;
587     
588     for (int i = 0; i < cnt; i++) {
589       int numShards = TestUtil.nextInt(random(), 0, getShardCount()) + 1;
590       int replicationFactor = TestUtil.nextInt(random(), 0, 3) + 1;
591       int maxShardsPerNode = (((numShards * replicationFactor) / getCommonCloudSolrClient()
592           .getZkStateReader().getClusterState().getLiveNodes().size())) + 1;
593 
594       
595       CloudSolrClient client = null;
596       try {
597         if (i == 0) {
598           // Test if we can create a collection through CloudSolrServer where
599           // you havnt set default-collection
600           // This is nice because you want to be able to create you first
601           // collection using CloudSolrServer, and in such case there is
602           // nothing reasonable to set as default-collection
603           client = createCloudClient(null);
604         } else if (i == 1) {
605           // Test if we can create a collection through CloudSolrServer where
606           // you have set default-collection to a non-existing collection
607           // This is nice because you want to be able to create you first

1004         }
1005       }
1006     }
1007     
1008     throw new RuntimeException("Could not find a live node for collection:" + collection);
