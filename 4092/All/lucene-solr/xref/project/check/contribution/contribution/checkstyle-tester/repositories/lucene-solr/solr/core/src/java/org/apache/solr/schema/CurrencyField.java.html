104   @Override
105   protected void init(IndexSchema schema, Map<String, String> args) {
106     super.init(schema, args);
107     if (this.isMultiValued()) { 
108       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, 
109                               "CurrencyField types can not be multiValued: " + 
110                               this.typeName);
111     }
112     this.schema = schema;
113     this.exchangeRateProviderClass = args.get(PARAM_RATE_PROVIDER_CLASS);
114     this.defaultCurrency = args.get(PARAM_DEFAULT_CURRENCY);
115 
116     if (this.defaultCurrency == null) {
117       this.defaultCurrency = DEFAULT_DEFAULT_CURRENCY;
118     }
119     
120     if (this.exchangeRateProviderClass == null) {
121       this.exchangeRateProviderClass = DEFAULT_RATE_PROVIDER_CLASS;
122     }
123 
124     if (null == getCurrency(this.defaultCurrency)) {
125       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Default currency code is not supported by this JVM: " + this.defaultCurrency);
126     }
127 
128     String precisionStepString = args.get(PARAM_PRECISION_STEP);
129     if (precisionStepString == null) {
130       precisionStepString = DEFAULT_PRECISION_STEP;
131     }
132 
133     // Initialize field type for amount
134     fieldTypeAmountRaw = new TrieLongField();
135     fieldTypeAmountRaw.setTypeName("amount_raw_type_tlong");
136     Map<String,String> map = new HashMap<>(1);
137     map.put("precisionStep", precisionStepString);
138     fieldTypeAmountRaw.init(schema, map);
139     
140     // Initialize field type for currency string
141     fieldTypeCurrency = new StrField();
142     fieldTypeCurrency.setTypeName("currency_type_string");
143     fieldTypeCurrency.init(schema, new HashMap<String,String>());
144     
145     args.remove(PARAM_RATE_PROVIDER_CLASS);
146     args.remove(PARAM_DEFAULT_CURRENCY);
147     args.remove(PARAM_PRECISION_STEP);
148 
149     try {
150       Class<? extends ExchangeRateProvider> c = schema.getResourceLoader().findClass(exchangeRateProviderClass, ExchangeRateProvider.class);
151       provider = c.newInstance();
152       provider.init(args);
153     } catch (Exception e) {
154       throw new SolrException(ErrorCode.BAD_REQUEST, "Error instantiating exchange rate provider "+exchangeRateProviderClass+": " + e.getMessage(), e);
155     }
156   }

173   @Override
174   public List<IndexableField> createFields(SchemaField field, Object externalVal, float boost) {
175     CurrencyValue value = CurrencyValue.parse(externalVal.toString(), defaultCurrency);
176 
177     List<IndexableField> f = new ArrayList<>();
178     SchemaField amountField = getAmountField(field);
179     f.add(amountField.createField(String.valueOf(value.getAmount()), amountField.indexed() && !amountField.omitNorms() ? boost : 1F));
180     SchemaField currencyField = getCurrencyField(field);
181     f.add(currencyField.createField(value.getCurrencyCode(), currencyField.indexed() && !currencyField.omitNorms() ? boost : 1F));
182 
183     if (field.stored()) {
184       org.apache.lucene.document.FieldType customType = new org.apache.lucene.document.FieldType();
185       assert !customType.omitNorms();
186       customType.setStored(true);
187       String storedValue = externalVal.toString().trim();
188       if (storedValue.indexOf(",") < 0) {
189         storedValue += "," + defaultCurrency;
190       }
191       f.add(createField(field.getName(), storedValue, customType, 1F));
192     }
193 
194     return f;
195   }

205   private void createDynamicCurrencyField(String suffix, FieldType type) {
206     String name = "*" + POLY_FIELD_SEPARATOR + suffix;
207     Map<String, String> props = new HashMap<>();
208     props.put("indexed", "true");
209     props.put("stored", "false");
210     props.put("multiValued", "false");
211     props.put("omitNorms", "true");
212     int p = SchemaField.calcProps(name, type, props);
213     schema.registerDynamicFields(SchemaField.create(name, type, p, null));
214   }

253   /**
254    * <p>
255    * Returns a ValueSource over this field in which the numeric value for 
256    * each document represents the indexed value as converted to the default 
257    * currency for the field, normalized to its most granular form based 
258    * on the default fractional digits.
259    * </p>
260    * <p>
261    * For example: If the default Currency specified for a field is 
262    * <code>USD</code>, then the values returned by this value source would 
263    * represent the equivilent number of "cents" (ie: value in dollars * 100) 
264    * after converting each document's native currency to USD -- because the 
265    * default fractional digits for <code>USD</code> is "<code>2</code>".  
266    * So for a document whose indexed value was currently equivilent to 
267    * "<code>5.43,USD</code>" using the the exchange provider for this field, 
268    * this ValueSource would return a value of "<code>543</code>"
269    * </p>
270    *
271    * @see #PARAM_DEFAULT_CURRENCY
272    * @see #DEFAULT_DEFAULT_CURRENCY
273    * @see Currency#getDefaultFractionDigits
274    * @see #getConvertedValueSource
275    */
276   public RawCurrencyValueSource getValueSource(SchemaField field, 
277                                                QParser parser) {
278     field.checkFieldCacheSource();
279     return new RawCurrencyValueSource(field, defaultCurrency, parser);
280   }

283    * <p>
284    * Returns a ValueSource over this field in which the numeric value for 
285    * each document represents the value from the underlying 
286    * <code>RawCurrencyValueSource</code> as converted to the specified target 
287    * Currency.
288    * </p>
289    * <p>
290    * For example: If the <code>targetCurrencyCode</code> param is set to
291    * <code>USD</code>, then the values returned by this value source would 
292    * represent the equivilent number of dollars after converting each 
293    * document's raw value to <code>USD</code>.  So for a document whose 
294    * indexed value was currently equivilent to "<code>5.43,USD</code>" 
295    * using the the exchange provider for this field, this ValueSource would 
296    * return a value of "<code>5.43</code>"
297    * </p>
298    *
299    * @param targetCurrencyCode The target currency for the resulting value source, if null the defaultCurrency for this field type will be used
300    * @param source the raw ValueSource to wrap
301    * @see #PARAM_DEFAULT_CURRENCY
302    * @see #DEFAULT_DEFAULT_CURRENCY

314   @Override
315   public Query getRangeQuery(QParser parser, SchemaField field, String part1, String part2, final boolean minInclusive, final boolean maxInclusive) {
316       final CurrencyValue p1 = CurrencyValue.parse(part1, defaultCurrency);
317       final CurrencyValue p2 = CurrencyValue.parse(part2, defaultCurrency);
318 
319       if (p1 != null && p2 != null && !p1.getCurrencyCode().equals(p2.getCurrencyCode())) {
320           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
321                   "Cannot parse range query " + part1 + " to " + part2 +
322                           ": range queries only supported when upper and lower bound have same currency.");
323       }
324 
325       return getRangeQuery(parser, field, p1, p2, minInclusive, maxInclusive);
326   }

328   public Query getRangeQuery(QParser parser, SchemaField field, final CurrencyValue p1, final CurrencyValue p2, final boolean minInclusive, final boolean maxInclusive) {
329     String currencyCode = (p1 != null) ? p1.getCurrencyCode() :
330                           (p2 != null) ? p2.getCurrencyCode() : defaultCurrency;
331 
332     // ValueSourceRangeFilter doesn't check exists(), so we have to
333     final Filter docsWithValues = new QueryWrapperFilter(new FieldValueQuery(getAmountField(field).getName()));
334     final Filter vsRangeFilter = new ValueSourceRangeFilter
335       (new RawCurrencyValueSource(field, currencyCode, parser),
336        p1 == null ? null : p1.getAmount() + "", 
337        p2 == null ? null : p2.getAmount() + "",
338        minInclusive, maxInclusive);
339     final BooleanQuery.Builder docsInRange = new BooleanQuery.Builder();
340     docsInRange.add(docsWithValues, Occur.FILTER);
341     docsInRange.add(vsRangeFilter, Occur.FILTER);
342 
343     return new SolrConstantScoreQuery(new QueryWrapperFilter(docsInRange.build()));
344   }

366   /**
367    * <p>
368    * A value source whose values represent the "normal" values
369    * in the specified target currency.
370    * </p>
371    * @see RawCurrencyValueSource
372    */
373   class ConvertedCurrencyValueSource extends ValueSource {
374     private final Currency targetCurrency;
375     private final RawCurrencyValueSource source;
376     private final double rate;
377     public ConvertedCurrencyValueSource(String targetCurrencyCode, 
378                                         RawCurrencyValueSource source) {
379       this.source = source;
380       this.targetCurrency = getCurrency(targetCurrencyCode);
381       if (null == targetCurrency) {
382         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Currency code not supported by this JVM: " + targetCurrencyCode);
383       }
384       // the target digits & currency of our source, 
385       // become the source digits & currency of ourselves
386       this.rate = provider.getExchangeRate
387         (source.getTargetCurrency().getCurrencyCode(), 
388          targetCurrency.getCurrencyCode());
389     }
390 
391     @Override
392     public FunctionValues getValues(Map context, LeafReaderContext reader) 
393       throws IOException {
394       final FunctionValues amounts = source.getValues(context, reader);
395       // the target digits & currency of our source, 
396       // become the source digits & currency of ourselves
397       final String sourceCurrencyCode = source.getTargetCurrency().getCurrencyCode();
398       final int sourceFractionDigits = source.getTargetCurrency().getDefaultFractionDigits();
399       final double divisor = Math.pow(10D, targetCurrency.getDefaultFractionDigits());
400       return new FunctionValues() {
401         @Override
402         public boolean exists(int doc) {
403           return amounts.exists(doc);
404         }
405         @Override
406         public long longVal(int doc) {
407           return (long) doubleVal(doc);
408         }
409         @Override
410         public int intVal(int doc) {
411           return (int) doubleVal(doc);
412         }
413 
414         @Override
415         public double doubleVal(int doc) {
416           return CurrencyValue.convertAmount(rate, sourceCurrencyCode, amounts.longVal(doc), targetCurrency.getCurrencyCode()) / divisor;
417         }
418 
419         @Override
420         public float floatVal(int doc) {
421           return CurrencyValue.convertAmount(rate, sourceCurrencyCode, amounts.longVal(doc), targetCurrency.getCurrencyCode()) / ((float)divisor);
422         }
423 
424         @Override
425         public String strVal(int doc) {
426           return Double.toString(doubleVal(doc));
427         }
428 
429         @Override
430         public String toString(int doc) {
431           return name() + '(' + strVal(doc) + ')';
432         }
433       };
434     }
435     public String name() {
436       return "currency";
437     }
438 
439     @Override
440     public String description() {
441       return name() + "(" + source.getField().getName() + "," + targetCurrency.getCurrencyCode()+")";
442     }
443 
444     @Override
445     public boolean equals(Object o) {
446       if (this == o) return true;
447       if (o == null || getClass() != o.getClass()) return false;
448 
449       ConvertedCurrencyValueSource that = (ConvertedCurrencyValueSource) o;
450 
451       return !(source != null ? !source.equals(that.source) : that.source != null) &&
452         (rate == that.rate) && 
453         !(targetCurrency != null ? !targetCurrency.equals(that.targetCurrency) : that.targetCurrency != null);
454 
455     }
456 
457     @Override
458     public int hashCode() {
459       int result = targetCurrency != null ? targetCurrency.hashCode() : 0;
460       result = 31 * result + (source != null ? source.hashCode() : 0);
461       result = 31 * (int) Double.doubleToLongBits(rate);
462       return result;
463     }
464   }

466   /**
467    * <p>
468    * A value source whose values represent the "raw" (ie: normalized using 
469    * the number of default fractional digits) values in the specified 
470    * target currency).
471    * </p>
472    * <p>
473    * For example: if the specified target currency is "<code>USD</code>" 
474    * then the numeric values are the number of pennies in the value 
475    * (ie: <code>$n * 100</code>) since the number of defalt fractional 
476    * digits for <code>USD</code> is "<code>2</code>")
477    * </p>
478    * @see ConvertedCurrencyValueSource
479    */
480   class RawCurrencyValueSource extends ValueSource {
481     private static final long serialVersionUID = 1L;
482     private final Currency targetCurrency;
483     private ValueSource currencyValues;
484     private ValueSource amountValues;
485     private final SchemaField sf;
486 
487     public RawCurrencyValueSource(SchemaField sfield, String targetCurrencyCode, QParser parser) {
488       this.sf = sfield;
489       this.targetCurrency = getCurrency(targetCurrencyCode);
490       if (null == targetCurrency) {
491         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Currency code not supported by this JVM: " + targetCurrencyCode);
492       }
493 
494       SchemaField amountField = schema.getField(sf.getName() + POLY_FIELD_SEPARATOR + FIELD_SUFFIX_AMOUNT_RAW);
495       SchemaField currencyField = schema.getField(sf.getName() + POLY_FIELD_SEPARATOR + FIELD_SUFFIX_CURRENCY);
496 
497       currencyValues = currencyField.getType().getValueSource(currencyField, parser);
498       amountValues = amountField.getType().getValueSource(amountField, parser);
499     }
500     
501     public SchemaField getField() { return sf; }
502     public Currency getTargetCurrency() { return targetCurrency; }
503 
504     @Override
505     public FunctionValues getValues(Map context, LeafReaderContext reader) throws IOException {
506       final FunctionValues amounts = amountValues.getValues(context, reader);
507       final FunctionValues currencies = currencyValues.getValues(context, reader);
508 
509       return new FunctionValues() {
510         private final int MAX_CURRENCIES_TO_CACHE = 256;
511         private final int[] fractionDigitCache = new int[MAX_CURRENCIES_TO_CACHE];
512         private final String[] currencyOrdToCurrencyCache = new String[MAX_CURRENCIES_TO_CACHE];
513         private final double[] exchangeRateCache = new double[MAX_CURRENCIES_TO_CACHE];
514         private int targetFractionDigits = -1;
515         private int targetCurrencyOrd = -1;
516         private boolean initializedCache;
517 
518         private String getDocCurrencyCode(int doc, int currencyOrd) {
519           if (currencyOrd < MAX_CURRENCIES_TO_CACHE) {
520             String currency = currencyOrdToCurrencyCache[currencyOrd];
521 
522             if (currency == null) {
523               currencyOrdToCurrencyCache[currencyOrd] = currency = currencies.strVal(doc);
524             }
525             
526             if (currency == null) {
527               currency = defaultCurrency;
528             }
529 
530             if (targetCurrencyOrd == -1 && 
531                 currency.equals(targetCurrency.getCurrencyCode() )) {
532               targetCurrencyOrd = currencyOrd;
533             }
534 
535             return currency;
536           } else {
537             return currencies.strVal(doc);
538           }
539         }
540         /** throws a (Server Error) SolrException if the code is not valid */
541         private Currency getDocCurrency(int doc, int currencyOrd) {
542           String code = getDocCurrencyCode(doc, currencyOrd);
543           Currency c = getCurrency(code);
544           if (null == c) {
545             throw new SolrException
546               (SolrException.ErrorCode.SERVER_ERROR, 
547                "Currency code of document is not supported by this JVM: "+code);
548           }
549           return c;
550         }
551 
552         @Override
553         public boolean exists(int doc) {
554           return amounts.exists(doc);
555         }
556         
557         @Override
558         public long longVal(int doc) {
559           long amount = amounts.longVal(doc);
560           // bail fast using whatever ammounts defaults to if no value
561           // (if we don't do this early, currencyOrd may be < 0, 
562           // causing index bounds exception
563           if ( ! exists(doc) ) {
564             return amount;
565           }
566 
567           if (!initializedCache) {
568             for (int i = 0; i < fractionDigitCache.length; i++) {
569               fractionDigitCache[i] = -1;
570             }
571 
572             initializedCache = true;
573           }
574 
575           int currencyOrd = currencies.ordVal(doc);
576 
577           if (currencyOrd == targetCurrencyOrd) {
578             return amount;
579           }
580 
581           double exchangeRate;
582           int sourceFractionDigits;
583 
584           if (targetFractionDigits == -1) {
585             targetFractionDigits = targetCurrency.getDefaultFractionDigits();
586           }
587 
588           if (currencyOrd < MAX_CURRENCIES_TO_CACHE) {
589             exchangeRate = exchangeRateCache[currencyOrd];
590 
591             if (exchangeRate <= 0.0) {
592               String sourceCurrencyCode = getDocCurrencyCode(doc, currencyOrd);
593               exchangeRate = exchangeRateCache[currencyOrd] = provider.getExchangeRate(sourceCurrencyCode, targetCurrency.getCurrencyCode());
594             }
595 
596             sourceFractionDigits = fractionDigitCache[currencyOrd];
597 
598             if (sourceFractionDigits == -1) {
599               sourceFractionDigits = fractionDigitCache[currencyOrd] = getDocCurrency(doc, currencyOrd).getDefaultFractionDigits();
600             }
601           } else {
602             Currency source = getDocCurrency(doc, currencyOrd);
603             exchangeRate = provider.getExchangeRate(source.getCurrencyCode(), targetCurrency.getCurrencyCode());
604             sourceFractionDigits = source.getDefaultFractionDigits();
605           }
606 
607           return CurrencyValue.convertAmount(exchangeRate, sourceFractionDigits, amount, targetFractionDigits);
608         }

612           return (int) longVal(doc);

618         }

630         @Override
631         public String toString(int doc) {
632           return name() + '(' + amounts.toString(doc) + ',' + currencies.toString(doc) + ')';

670 /**
671  * Configuration for currency. Provides currency exchange rates.
672  */
673 class FileExchangeRateProvider implements ExchangeRateProvider {
674   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
675   protected static final String PARAM_CURRENCY_CONFIG       = "currencyConfig";
676 
677   // Exchange rate map, maps Currency Code -> Currency Code -> Rate
678   private Map<String, Map<String, Double>> rates = new HashMap<>();
679 
680   private String currencyConfigFile;
681   private ResourceLoader loader;
682 
683   /**
684    * Returns the currently known exchange rate between two currencies. If a direct rate has been loaded,
685    * it is used. Otherwise, if a rate is known to convert the target currency to the source, the inverse
686    * exchange rate is computed.
687    *
688    * @param sourceCurrencyCode The source currency being converted from.
689    * @param targetCurrencyCode The target currency being converted to.
690    * @return The exchange rate.
691    * @throws SolrException if the requested currency pair cannot be found
692    */
693   @Override
694   public double getExchangeRate(String sourceCurrencyCode, String targetCurrencyCode) {
695     if (sourceCurrencyCode == null || targetCurrencyCode == null) {
696       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Cannot get exchange rate; currency was null.");
697     }
698     
699     if (sourceCurrencyCode.equals(targetCurrencyCode)) {
700       return 1.0;
701     }
702 
703     Double directRate = lookupRate(sourceCurrencyCode, targetCurrencyCode);
704 
705     if (directRate != null) {
706       return directRate;
707     }
708 
709     Double symmetricRate = lookupRate(targetCurrencyCode, sourceCurrencyCode);
710 
711     if (symmetricRate != null) {
712       return 1.0 / symmetricRate;
713     }
714 
715     throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "No available conversion rate between " + sourceCurrencyCode + " to " + targetCurrencyCode);
716   }
717 
718   /**
719    * Looks up the current known rate, if any, between the source and target currencies.
720    *
721    * @param sourceCurrencyCode The source currency being converted from.
722    * @param targetCurrencyCode The target currency being converted to.
723    * @return The exchange rate, or null if no rate has been registered.
724    */
725   private Double lookupRate(String sourceCurrencyCode, String targetCurrencyCode) {
726     Map<String, Double> rhs = rates.get(sourceCurrencyCode);
727 
728     if (rhs != null) {
729       return rhs.get(targetCurrencyCode);
730     }
731 
732     return null;
733   }
734 
735   /**
736    * Registers the specified exchange rate.
737    *
738    * @param ratesMap           The map to add rate to
739    * @param sourceCurrencyCode The source currency.
740    * @param targetCurrencyCode The target currency.
741    * @param rate               The known exchange rate.
742    */
743   private void addRate(Map<String, Map<String, Double>> ratesMap, String sourceCurrencyCode, String targetCurrencyCode, double rate) {
744     Map<String, Double> rhs = ratesMap.get(sourceCurrencyCode);
745 
746     if (rhs == null) {
747       rhs = new HashMap<>();
748       ratesMap.put(sourceCurrencyCode, rhs);
749     }
750 
751     rhs.put(targetCurrencyCode, rate);
752   }
753 
754   @Override
755   public boolean equals(Object o) {
756     if (this == o) return true;
757     if (o == null || getClass() != o.getClass()) return false;
758 
759     FileExchangeRateProvider that = (FileExchangeRateProvider) o;
760 
761     return !(rates != null ? !rates.equals(that.rates) : that.rates != null);
762   }
763 
764   @Override
765   public int hashCode() {
766     return rates != null ? rates.hashCode() : 0;
767   }
768 
769   @Override
770   public String toString() {
771     return "["+this.getClass().getName()+" : " + rates.size() + " rates.]";
772   }

774   @Override
775   public Set<String> listAvailableCurrencies() {
776     Set<String> currencies = new HashSet<>();
777     for(String from : rates.keySet()) {
778       currencies.add(from);
779       for(String to : rates.get(from).keySet()) {
780         currencies.add(to);
781       }
782     }
783     return currencies;
784   }

786   @Override
787   public boolean reload() throws SolrException {
788     InputStream is = null;
789     Map<String, Map<String, Double>> tmpRates = new HashMap<>();
790     try {
791       log.info("Reloading exchange rates from file "+this.currencyConfigFile);
792 
793       is = loader.openResource(currencyConfigFile);
794       javax.xml.parsers.DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
795       try {
796         dbf.setXIncludeAware(true);
797         dbf.setNamespaceAware(true);
798       } catch (UnsupportedOperationException e) {
799         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "XML parser doesn't support XInclude option", e);
800       }
801       
802       try {
803         Document doc = dbf.newDocumentBuilder().parse(is);
804         XPathFactory xpathFactory = XPathFactory.newInstance();
805         XPath xpath = xpathFactory.newXPath();
806         
807         // Parse exchange rates.
808         NodeList nodes = (NodeList) xpath.evaluate("/currencyConfig/rates/rate", doc, XPathConstants.NODESET);
809         
810         for (int i = 0; i < nodes.getLength(); i++) {
811           Node rateNode = nodes.item(i);
812           NamedNodeMap attributes = rateNode.getAttributes();
813           Node from = attributes.getNamedItem("from");
814           Node to = attributes.getNamedItem("to");
815           Node rate = attributes.getNamedItem("rate");
816           
817           if (from == null || to == null || rate == null) {
818             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Exchange rate missing attributes (required: from, to, rate) " + rateNode);
819           }
820           
821           String fromCurrency = from.getNodeValue();
822           String toCurrency = to.getNodeValue();
823           Double exchangeRate;
824           
825           if (null == CurrencyField.getCurrency(fromCurrency)) {
826             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Specified 'from' currency not supported in this JVM: " + fromCurrency);
827           }
828           if (null == CurrencyField.getCurrency(toCurrency)) {
829             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Specified 'to' currency not supported in this JVM: " + toCurrency);
830           }
831           
832           try {
833             exchangeRate = Double.parseDouble(rate.getNodeValue());
834           } catch (NumberFormatException e) {
835             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Could not parse exchange rate: " + rateNode, e);
836           }
837           
838           addRate(tmpRates, fromCurrency, toCurrency, exchangeRate);
839         }
840       } catch (SAXException | XPathExpressionException | ParserConfigurationException | IOException e) {
841         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Error parsing currency config.", e);
842       }
843     } catch (IOException e) {
844       throw new SolrException(ErrorCode.SERVER_ERROR, "Error while opening Currency configuration file "+currencyConfigFile, e);
845     } finally {
846       try {
847         if (is != null) {
848           is.close();
849         }
850       } catch (IOException e) {
851         e.printStackTrace();
852       }
853     }
854     // Atomically swap in the new rates map, if it loaded successfully
855     this.rates = tmpRates;
856     return true;
857   }
858 
859   @Override
860   public void init(Map<String,String> params) throws SolrException {
861     this.currencyConfigFile = params.get(PARAM_CURRENCY_CONFIG);
862     if(currencyConfigFile == null) {
863       throw new SolrException(ErrorCode.NOT_FOUND, "Missing required configuration "+PARAM_CURRENCY_CONFIG);
864     }
865     
866     // Removing config params custom to us
867     params.remove(PARAM_CURRENCY_CONFIG);
868   }
869 
870   @Override
871   public void inform(ResourceLoader loader) throws SolrException {
872     if(loader == null) {
873       throw new SolrException(ErrorCode.SERVER_ERROR, "Needs ResourceLoader in order to load config file");
874     }
875     this.loader = loader;
876     reload();
877   }
878 }

880 /**
881  * Represents a Currency field value, which includes a long amount and ISO currency code.
882  */
883 class CurrencyValue {
884   private long amount;
885   private String currencyCode;
886 
887   /**
888    * Constructs a new currency value.
889    *
890    * @param amount       The amount.
891    * @param currencyCode The currency code.
892    */
893   public CurrencyValue(long amount, String currencyCode) {
894     this.amount = amount;
895     this.currencyCode = currencyCode;
896   }
897 
898   /**
899    * Constructs a new currency value by parsing the specific input.
900    * <p/>
901    * Currency values are expected to be in the format &lt;amount&gt;,&lt;currency code&gt;,
902    * for example, "500,USD" would represent 5 U.S. Dollars.
903    * <p/>
904    * If no currency code is specified, the default is assumed.
905    *
906    * @param externalVal The value to parse.
907    * @param defaultCurrency The default currency.
908    * @return The parsed CurrencyValue.
909    */
910   public static CurrencyValue parse(String externalVal, String defaultCurrency) {
911     if (externalVal == null) {
912       return null;
913     }
914     String amount = externalVal;
915     String code = defaultCurrency;
916 
917     if (externalVal.contains(",")) {
918       String[] amountAndCode = externalVal.split(",");
919       amount = amountAndCode[0];
920       code = amountAndCode[1];
921     }
922 
923     if (amount.equals("*")) {
924       return null;
925     }
926     
927     Currency currency = CurrencyField.getCurrency(code);
928 
929     if (currency == null) {
930       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Currency code not supported by this JVM: " + code);
931     }
932 
933     try {
934       double value = Double.parseDouble(amount);
935       long currencyValue = Math.round(value * Math.pow(10.0, currency.getDefaultFractionDigits()));
936 
937       return new CurrencyValue(currencyValue, code);
938     } catch (NumberFormatException e) {
939       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, e);
940     }
941   }
942 
943   /**
944    * The amount of the CurrencyValue.
945    *
946    * @return The amount.
947    */
948   public long getAmount() {
949     return amount;
950   }
951 
952   /**
953    * The ISO currency code of the CurrencyValue.
954    *
955    * @return The currency code.
956    */
957   public String getCurrencyCode() {
958     return currencyCode;
959   }
960 
961   /**
962    * Performs a currency conversion & unit conversion.
963    *
964    * @param exchangeRates      Exchange rates to apply.
965    * @param sourceCurrencyCode The source currency code.
966    * @param sourceAmount       The source amount.
967    * @param targetCurrencyCode The target currency code.
968    * @return The converted indexable units after the exchange rate and currency fraction digits are applied.
969    */
970   public static long convertAmount(ExchangeRateProvider exchangeRates, String sourceCurrencyCode, long sourceAmount, String targetCurrencyCode) {
971     double exchangeRate = exchangeRates.getExchangeRate(sourceCurrencyCode, targetCurrencyCode);
972     return convertAmount(exchangeRate, sourceCurrencyCode, sourceAmount, targetCurrencyCode);
973   }
974 
975   /**
976    * Performs a currency conversion & unit conversion.
977    *
978    * @param exchangeRate         Exchange rate to apply.
979    * @param sourceFractionDigits The fraction digits of the source.
980    * @param sourceAmount         The source amount.
981    * @param targetFractionDigits The fraction digits of the target.
982    * @return The converted indexable units after the exchange rate and currency fraction digits are applied.
983    */
984   public static long convertAmount(final double exchangeRate, final int sourceFractionDigits, final long sourceAmount, final int targetFractionDigits) {
985     int digitDelta = targetFractionDigits - sourceFractionDigits;
986     double value = ((double) sourceAmount * exchangeRate);
987 
988     if (digitDelta != 0) {
989       if (digitDelta < 0) {
990         for (int i = 0; i < -digitDelta; i++) {
991           value *= 0.1;
992         }
993       } else {
994         for (int i = 0; i < digitDelta; i++) {
995           value *= 10.0;
996         }
997       }
998     }
999 
1000     return (long) value;
1001   }
1002 
1003   /**
1004    * Performs a currency conversion & unit conversion.
1005    *
1006    * @param exchangeRate       Exchange rate to apply.
1007    * @param sourceCurrencyCode The source currency code.
1008    * @param sourceAmount       The source amount.
1009    * @param targetCurrencyCode The target currency code.
1010    * @return The converted indexable units after the exchange rate and currency fraction digits are applied.
1011    */
1012   public static long convertAmount(double exchangeRate, String sourceCurrencyCode, long sourceAmount, String targetCurrencyCode) {
1013     if (targetCurrencyCode.equals(sourceCurrencyCode)) {
1014       return sourceAmount;
1015     }
1016 
1017     int sourceFractionDigits = Currency.getInstance(sourceCurrencyCode).getDefaultFractionDigits();
1018     Currency targetCurrency = Currency.getInstance(targetCurrencyCode);
1019     int targetFractionDigits = targetCurrency.getDefaultFractionDigits();
1020     return convertAmount(exchangeRate, sourceFractionDigits, sourceAmount, targetFractionDigits);
1021   }
1022 
1023   /**
1024    * Returns a new CurrencyValue that is the conversion of this CurrencyValue to the specified currency.
1025    *
1026    * @param exchangeRates      The exchange rate provider.
1027    * @param targetCurrencyCode The target currency code to convert this CurrencyValue to.
1028    * @return The converted CurrencyValue.
1029    */
1030   public CurrencyValue convertTo(ExchangeRateProvider exchangeRates, String targetCurrencyCode) {
1031     return new CurrencyValue(convertAmount(exchangeRates, this.getCurrencyCode(), this.getAmount(), targetCurrencyCode), targetCurrencyCode);
1032   }
1033 
1034   @Override
1035   public String toString() {
1036     return String.valueOf(amount) + "," + currencyCode;
1037   }
