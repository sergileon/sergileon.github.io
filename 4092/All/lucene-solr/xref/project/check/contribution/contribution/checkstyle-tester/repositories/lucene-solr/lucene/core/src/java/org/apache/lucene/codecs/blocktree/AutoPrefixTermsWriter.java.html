40  /** Used in the first pass when writing a segment to locate
41   *  "appropriate" auto-prefix terms to pre-compile into the index.
42   *  This visits every term in the index to find prefixes that
43   *  match {@code >= min} and {@code <= max} number of terms. */
44  
45  class AutoPrefixTermsWriter {
46

52    /** Describes a range of term-space to match, either a simple prefix
53     *  (foo*) or a floor-block range of a prefix (e.g. foo[a-m]*,
54     *  foo[n-z]*) when there are too many terms starting with foo*. */
55    public static final class PrefixTerm implements Comparable<PrefixTerm> {
56      /** Common prefix */
57      public final byte[] prefix;
58  
59      /** If this is -2, this is a normal prefix (foo *), else it's the minimum lead byte of the suffix (e.g. 'd' in foo[d-m]*). */
60      public final int floorLeadStart;
61  
62      /** The lead byte (inclusive) of the suffix for the term range we match (e.g. 'm' in foo[d-m*]); this is ignored when
63       *  floorLeadStart is -2. */
64      public final int floorLeadEnd;
65  
66      public final BytesRef term;
67  
68      /** Sole constructor. */
69      public PrefixTerm(byte[] prefix, int floorLeadStart, int floorLeadEnd) {
70        this.prefix = prefix;
71        this.floorLeadStart = floorLeadStart;
72        this.floorLeadEnd = floorLeadEnd;
73        this.term = toBytesRef(prefix, floorLeadStart);
74  
75        assert floorLeadEnd >= floorLeadStart;
76        assert floorLeadEnd >= 0;
77        assert floorLeadStart == -2 || floorLeadStart >= 0;
78  
79        // We should never create empty-string prefix term:
80        assert prefix.length > 0 || floorLeadStart != -2 || floorLeadEnd != 0xff;
81      }
82  
83      @Override
84      public String toString() {
85        String s = brToString(new BytesRef(prefix));
86        if (floorLeadStart == -2) {
87          s += "[-" + Integer.toHexString(floorLeadEnd) + "]";
88        } else {
89          s += "[" + Integer.toHexString(floorLeadStart) + "-" + Integer.toHexString(floorLeadEnd) + "]";
90        }
91        return s;
92      }
93  
94      @Override
95      public int compareTo(PrefixTerm other) {
96        int cmp = term.compareTo(other.term);
97        if (cmp == 0) {
98          if (prefix.length != other.prefix.length) {
99            return prefix.length - other.prefix.length;
100         }
101 
102         // On tie, sort the bigger floorLeadEnd, earlier, since it
103         // spans more terms, so during intersect, we want to encounter this one
104         // first so we can use it if the automaton accepts the larger range:
105         cmp = other.floorLeadEnd - floorLeadEnd;
106       }
107 
108       return cmp;
109     }
110 
111     /** Returns the leading term for this prefix term, e.g. "foo" (for
112      *  the foo* prefix) or "foom" (for the foo[m-z]* case). */
113     private static BytesRef toBytesRef(byte[] prefix, int floorLeadStart) {
114       BytesRef br;
115       if (floorLeadStart != -2) {
116         assert floorLeadStart >= 0;
117         br = new BytesRef(prefix.length+1);
118       } else {
119         br = new BytesRef(prefix.length);
120       }
121       System.arraycopy(prefix, 0, br.bytes, 0, prefix.length);
122       br.length = prefix.length;
123       if (floorLeadStart != -2) {
124         assert floorLeadStart >= 0;
125         br.bytes[br.length++] = (byte) floorLeadStart;
126       }
127 
128       return br;
129     }
130 
131     public int compareTo(BytesRef term) {
132       return this.term.compareTo(term);
133     }
134 
135     public TermsEnum getTermsEnum(TermsEnum in) {
136 
137       final BytesRef prefixRef = new BytesRef(prefix);
138 
139       return new FilteredTermsEnum(in) {
140           {
141             setInitialSeekTerm(term);
142           }
143 
144           @Override
145           protected AcceptStatus accept(BytesRef term) {
146             if (StringHelper.startsWith(term, prefixRef) &&
147                 (floorLeadEnd == -1 || term.length == prefixRef.length || (term.bytes[term.offset + prefixRef.length] & 0xff) <= floorLeadEnd)) {
148               return AcceptStatus.YES;
149             } else {
150               return AcceptStatus.END;
151             }
152           }
153         };
154     }
155   }

157   // for debugging
158   static String brToString(BytesRef b) {
159     try {
160       return b.utf8ToString() + " " + b;
161     } catch (Throwable t) {
162       // If BytesRef isn't actually UTF8, or it's eg a
163       // prefix of UTF8 that ends mid-unicode-char, we
164       // fallback to hex:
165       return b.toString();
166     }
167   }

169   final List<PrefixTerm> prefixes = new ArrayList<>();
170   private final int minItemsInPrefix;
171   private final int maxItemsInPrefix;
172 
173   // Records index into pending where the current prefix at that
174   // length "started"; for example, if current term starts with 't',
175   // startsByPrefix[0] is the index into pending for the first
176   // term/sub-block starting with 't'.  We use this to figure out when
177   // to write a new block:
178   private final BytesRefBuilder lastTerm = new BytesRefBuilder();
179   private int[] prefixStarts = new int[8];
180   private List<Object> pending = new ArrayList<>();
181 
182   //private final String segment;
183 
184   public AutoPrefixTermsWriter(Terms terms, int minItemsInPrefix, int maxItemsInPrefix) throws IOException {
185     this.minItemsInPrefix = minItemsInPrefix;
186     this.maxItemsInPrefix = maxItemsInPrefix;
187     //this.segment = segment;
188 
189     TermsEnum termsEnum = terms.iterator();
190     while (true) {
191       BytesRef term = termsEnum.next();
192       if (term == null) {
193         break;
194       }
195       //if (DEBUG) System.out.println("pushTerm: " + brToString(term));
196       pushTerm(term);
197     }
198 
199     if (pending.size() > 1) {
200       pushTerm(BlockTreeTermsWriter.EMPTY_BYTES_REF);
201 
202       // Also maybe save floor prefixes in root block; this can be a biggish perf gain for large ranges:
203       /*
204       System.out.println("root block pending.size=" + pending.size());
205       for(Object o : pending) {
206         System.out.println("  " + o);
207       }
208       */
209       while (pending.size() >= minItemsInPrefix) {
210         savePrefixes(0, pending.size());
211       }
212     }
213 
214     // Even though we visited terms in already-sorted order, the prefixes
215     // can be slightly unsorted, e.g. aaaaa will be before aaa, so we
216     // must sort here so our caller can do merge sort into actual terms
217     // when writing.  Probably we should use CollectionUtil.timSort here?
218     Collections.sort(prefixes);
219   }
220 
221   /** Pushes the new term to the top of the stack, and writes new blocks. */
222   private void pushTerm(BytesRef text) throws IOException {
223     int limit = Math.min(lastTerm.length(), text.length);
224     //if (DEBUG) System.out.println("\nterm: " + text.utf8ToString());
225 
226     // Find common prefix between last term and current term:
227     int pos = 0;
228     while (pos < limit && lastTerm.byteAt(pos) == text.bytes[text.offset+pos]) {
229       pos++;
230     }
231 
232     //if (DEBUG) System.out.println("  shared=" + pos + "  lastTerm.length=" + lastTerm.length());
233 
234     // Close the "abandoned" suffix now:
235     for(int i=lastTerm.length()-1;i>=pos;i--) {
236 
237       // How many items on top of the stack share the current suffix
238       // we are closing:
239       int prefixTopSize = pending.size() - prefixStarts[i];
240 
241       while (prefixTopSize >= minItemsInPrefix) {       
242         //if (DEBUG) System.out.println("  pop: i=" + i + " prefixTopSize=" + prefixTopSize + " minItemsInBlock=" + minItemsInPrefix);
243         savePrefixes(i+1, prefixTopSize);
244         //prefixStarts[i] -= prefixTopSize;
245         //if (DEBUG) System.out.println("    after savePrefixes: " + (pending.size() - prefixStarts[i]) + " pending.size()=" + pending.size() + " start=" + prefixStarts[i]);
246 
247         // For large floor blocks, it's possible we should now re-run on the new prefix terms we just created:
248         prefixTopSize = pending.size() - prefixStarts[i];
249       }
250     }
251 
252     if (prefixStarts.length < text.length) {
253       prefixStarts = ArrayUtil.grow(prefixStarts, text.length);
254     }
255 
256     // Init new tail:
257     for(int i=pos;i<text.length;i++) {
258       prefixStarts[i] = pending.size();
259     }
260 
261     lastTerm.copyBytes(text);
262 
263     // Only append the first (optional) empty string, no the fake last one used to close all prefixes:
264     if (text.length > 0 || pending.isEmpty()) {
265       byte[] termBytes = new byte[text.length];
266       System.arraycopy(text.bytes, text.offset, termBytes, 0, text.length);
267       pending.add(termBytes);
268     }
269   }

271   void savePrefixes(int prefixLength, int count) throws IOException {
272 
273     assert count > 0;
274 
275     /*
276     if (DEBUG2) {
277       BytesRef br = new BytesRef(lastTerm.bytes());
278       br.length = prefixLength;
279       //System.out.println("  savePrefixes: seg=" + segment + " " + brToString(br) + " count=" + count + " pending.size()=" + pending.size());
280       System.out.println("  savePrefixes: " + brToString(br) + " count=" + count + " pending.size()=" + pending.size());
281     }
282     */
283 
284     int lastSuffixLeadLabel = -2;
285 
286     int start = pending.size()-count;
287     assert start >=0;
288 
289     // Special case empty-string suffix case: we are being asked to build prefix terms for all aaa* terms, but 
290     // the exact term aaa is here, and we must skip it (it is handled "higher", under the aa* terms):
291     Object o = pending.get(start);
292     boolean skippedEmptyStringSuffix = false;
293     if (o instanceof byte[]) {
294       if (((byte[]) o).length == prefixLength) {
295         start++;
296         count--;
297         //if (DEBUG) System.out.println("  skip empty-string term suffix");
298         skippedEmptyStringSuffix = true;
299       }
300     } else {
301       PrefixTerm prefix = (PrefixTerm) o;
302       if (prefix.term.bytes.length == prefixLength) {
303         start++;
304         count--;
305         //if (DEBUG) System.out.println("  skip empty-string PT suffix");
306         skippedEmptyStringSuffix = true;
307       }
308     }
309 
310     int end = pending.size();
311     int nextBlockStart = start;
312     int nextFloorLeadLabel = -1;
313     int prefixCount = 0;
314 
315     PrefixTerm lastPTEntry = null;
316 
317     for (int i=start; i<end; i++) {
318 
319       byte[] termBytes;
320       o = pending.get(i);
321       PrefixTerm ptEntry;
322       if (o instanceof byte[]) {
323         ptEntry = null;
324         termBytes = (byte[]) o;
325       } else {
326         ptEntry = (PrefixTerm) o;
327         termBytes = ptEntry.term.bytes;
328         if (ptEntry.prefix.length != prefixLength) {
329           assert ptEntry.prefix.length > prefixLength;
330           ptEntry = null;
331         }
332       }
333 
334       //if (DEBUG) System.out.println("    check term=" + brToString(new BytesRef(termBytes)) + " o=" + o);
335 
336       // We handled the empty-string suffix case up front:
337       assert termBytes.length > prefixLength;
338 
339       int suffixLeadLabel = termBytes[prefixLength] & 0xff;
340 
341       //if (DEBUG) System.out.println("  i=" + i + " o=" + o + " suffixLeadLabel=" + Integer.toHexString(suffixLeadLabel) + " pendingCount=" + (i - nextBlockStart) + " min=" + minItemsInPrefix);
342 
343       if (suffixLeadLabel != lastSuffixLeadLabel) {
344         // This is a boundary, a chance to make an auto-prefix term if we want:
345 
346         // When we are "recursing" (generating auto-prefix terms on a block of
347         // floor'd auto-prefix terms), this assert is non-trivial because it
348         // ensures the floorLeadEnd of the previous terms is in fact less
349         // than the lead start of the current entry:
350         assert suffixLeadLabel > lastSuffixLeadLabel: "suffixLeadLabel=" + suffixLeadLabel + " vs lastSuffixLeadLabel=" + lastSuffixLeadLabel;
351 
352         int itemsInBlock = i - nextBlockStart;
353 
354         if (itemsInBlock >= minItemsInPrefix && end-nextBlockStart > maxItemsInPrefix) {
355           // The count is too large for one block, so we must break it into "floor" blocks, where we record
356           // the leading label of the suffix of the first term in each floor block, so at search time we can
357           // jump to the right floor block.  We just use a naive greedy segmenter here: make a new floor
358           // block as soon as we have at least minItemsInBlock.  This is not always best: it often produces
359           // a too-small block as the final block:
360 
361           // If the last entry was another prefix term of the same length, then it represents a range of terms, so we must use its ending
362           // prefix label as our ending label:
363           if (lastPTEntry != null) {
364             //if (DEBUG) System.out.println("  use last");
365             lastSuffixLeadLabel = lastPTEntry.floorLeadEnd;
366           }
367           savePrefix(prefixLength, nextFloorLeadLabel, lastSuffixLeadLabel);
368 
369           prefixCount++;
370           nextFloorLeadLabel = suffixLeadLabel;
371           nextBlockStart = i;
372         }
373 
374         if (nextFloorLeadLabel == -1) {
375           nextFloorLeadLabel = suffixLeadLabel;
376           //if (DEBUG) System.out.println("set first lead label=" + nextFloorLeadLabel);
377         }
378 
379         lastSuffixLeadLabel = suffixLeadLabel;
380       }
381 
382       lastPTEntry = ptEntry;
383     }
384 
385     // Write last block, if any:
386     if (nextBlockStart < end) {
387       //System.out.println("  lastPTEntry=" + lastPTEntry + " lastSuffixLeadLabel=" + lastSuffixLeadLabel);
388       if (lastPTEntry != null) {
389         lastSuffixLeadLabel = lastPTEntry.floorLeadEnd;
390       }
391       assert lastSuffixLeadLabel >= nextFloorLeadLabel: "lastSuffixLeadLabel=" + lastSuffixLeadLabel + " nextFloorLeadLabel=" + nextFloorLeadLabel;
392       if (prefixCount == 0) {
393         if (prefixLength > 0) {
394           savePrefix(prefixLength, -2, 0xff);
395           prefixCount++;
396           
397           // If we skipped empty string suffix, e.g. term aaa for prefix aaa*, since we
398           // are now writing the full aaa* prefix term, we include it here:
399           if (skippedEmptyStringSuffix) {
400             count++;
401           }
402         } else {
403           // Don't add a prefix term for all terms in the index!
404         }
405       } else {
406         if (lastSuffixLeadLabel == -2) {
407           // Special case when closing the empty string root block:
408           lastSuffixLeadLabel = 0xff;
409         }
410         savePrefix(prefixLength, nextFloorLeadLabel, lastSuffixLeadLabel);
411         prefixCount++;
412       }
413     }
414 
415     // Remove slice from the top of the pending stack, that we just wrote:
416 
417     pending.subList(pending.size()-count, pending.size()).clear();
418 
419     // Append prefix terms for each prefix, since these count like real terms that also need to be "rolled up":
420     for(int i=0;i<prefixCount;i++) {
421       PrefixTerm pt = prefixes.get(prefixes.size()-(prefixCount-i));
422       pending.add(pt);
423     }
424   }
425 
426   private void savePrefix(int prefixLength, int floorLeadStart, int floorLeadEnd) {
427     byte[] prefix = new byte[prefixLength];
428     System.arraycopy(lastTerm.bytes(), 0, prefix, 0, prefixLength);
429     assert floorLeadStart != -1;
430     assert floorLeadEnd != -1;
431 
432     PrefixTerm pt = new PrefixTerm(prefix, floorLeadStart, floorLeadEnd); 
433     //if (DEBUG2) System.out.println("    savePrefix: seg=" + segment + " " + pt + " count=" + count);
434     //if (DEBUG) System.out.println("    savePrefix: " + pt);
435 
436     prefixes.add(pt);
437   }
