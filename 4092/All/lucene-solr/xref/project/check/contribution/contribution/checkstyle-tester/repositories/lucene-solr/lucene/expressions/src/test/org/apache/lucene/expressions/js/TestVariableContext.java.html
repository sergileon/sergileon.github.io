20  import org.apache.lucene.util.LuceneTestCase;
21  
22  import static org.apache.lucene.expressions.js.VariableContext.Type.MEMBER;
23  import static org.apache.lucene.expressions.js.VariableContext.Type.STR_INDEX;
24  import static org.apache.lucene.expressions.js.VariableContext.Type.INT_INDEX;
25  import static org.apache.lucene.expressions.js.VariableContext.Type.METHOD;
26  
27  public class TestVariableContext extends LuceneTestCase {
28  
29    public void testSimpleVar() {
30      VariableContext[] x = VariableContext.parse("foo");
31      assertEquals(1, x.length);
32      assertEquals(x[0].type, MEMBER);
33      assertEquals(x[0].text, "foo");
34    }
35  
36    public void testEmptyString() {
37      VariableContext[] x = VariableContext.parse("foo['']");
38      assertEquals(2, x.length);
39      assertEquals(x[1].type, STR_INDEX);
40      assertEquals(x[1].text, "");
41    }
42  
43    public void testUnescapeString() {
44      VariableContext[] x = VariableContext.parse("foo['\\'\\\\']");
45      assertEquals(2, x.length);
46      assertEquals(x[1].type, STR_INDEX);
47      assertEquals(x[1].text, "'\\");
48    }
49  
50    public void testMember() {
51      VariableContext[] x = VariableContext.parse("foo.bar");
52      assertEquals(2, x.length);
53      assertEquals(x[1].type, MEMBER);
54      assertEquals(x[1].text, "bar");
55    }
56  
57    public void testMemberFollowedByMember() {
58      VariableContext[] x = VariableContext.parse("foo.bar.baz");
59      assertEquals(3, x.length);
60      assertEquals(x[2].type, MEMBER);
61      assertEquals(x[2].text, "baz");
62    }
63  
64    public void testMemberFollowedByIntArray() {
65      VariableContext[] x = VariableContext.parse("foo.bar[1]");
66      assertEquals(3, x.length);
67      assertEquals(x[2].type, INT_INDEX);
68      assertEquals(x[2].integer, 1);
69    }
70  
71    public void testMethodWithMember() {
72      VariableContext[] x = VariableContext.parse("m.m()");
73      assertEquals(2, x.length);
74      assertEquals(x[1].type, METHOD);
75      assertEquals(x[1].text, "m");
76    }
77  
78    public void testMethodWithStrIndex() {
79      VariableContext[] x = VariableContext.parse("member['blah'].getMethod()");
80      assertEquals(3, x.length);
81      assertEquals(x[2].type, METHOD);
82      assertEquals(x[2].text, "getMethod");
83    }
84  
85    public void testMethodWithNumericalIndex() {
86      VariableContext[] x = VariableContext.parse("member[0].getMethod()");
87      assertEquals(3, x.length);
88      assertEquals(x[2].type, METHOD);
89      assertEquals(x[2].text, "getMethod");
90    }
91  }
