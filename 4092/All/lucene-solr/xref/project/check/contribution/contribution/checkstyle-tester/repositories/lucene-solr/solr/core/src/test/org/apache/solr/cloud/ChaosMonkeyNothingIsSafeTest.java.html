118   @Test
119   public void test() throws Exception {
120     boolean testSuccessful = false;
121     try {
122       handle.clear();
123       handle.put("timestamp", SKIPVAL);
124       ZkStateReader zkStateReader = cloudClient.getZkStateReader();
125       // make sure we have leaders for each shard
126       for (int j = 1; j < sliceCount; j++) {
127         zkStateReader.getLeaderRetry(DEFAULT_COLLECTION, "shard" + j, 10000);
128       }      // make sure we again have leaders for each shard
129       
130       waitForRecoveriesToFinish(false);
131       
132       // we cannot do delete by query
133       // as it's not supported for recovery
134       del("*:*");
135       
136       List<StoppableThread> threads = new ArrayList<>();
137       List<StoppableIndexingThread> indexTreads = new ArrayList<>();
138       int threadCount = TEST_NIGHTLY ? 3 : 1;
139       int i = 0;
140       for (i = 0; i < threadCount; i++) {
141         StoppableIndexingThread indexThread = new StoppableIndexingThread(controlClient, cloudClient, Integer.toString(i), true);
142         threads.add(indexThread);
143         indexTreads.add(indexThread);
144         indexThread.start();
145       }
146       
147       threadCount = 1;
148       i = 0;
149       for (i = 0; i < threadCount; i++) {
150         StoppableSearchThread searchThread = new StoppableSearchThread(cloudClient);
151         threads.add(searchThread);
152         searchThread.start();
153       }
154       
155       // TODO: we only do this sometimes so that we can sometimes compare against control,
156       // it's currently hard to know what requests failed when using ConcurrentSolrUpdateServer
157       boolean runFullThrottle = random().nextBoolean();
158       if (runFullThrottle) {
159         FullThrottleStoppableIndexingThread ftIndexThread = new FullThrottleStoppableIndexingThread(
160             clients, "ft1", true);
161         threads.add(ftIndexThread);
162         ftIndexThread.start();
163       }
164       
165       chaosMonkey.startTheMonkey(true, 10000);
166       try {
167         long runLength;
168         if (RUN_LENGTH != -1) {
169           runLength = RUN_LENGTH;
170         } else {
171           int[] runTimes;
172           if (TEST_NIGHTLY) {
173             runTimes = new int[] {5000, 6000, 10000, 15000, 25000, 30000,
174                 30000, 45000, 90000, 120000};
175           } else {
176             runTimes = new int[] {5000, 7000, 15000};
177           }
178           runLength = runTimes[random().nextInt(runTimes.length - 1)];
179         }
180         
181         Thread.sleep(runLength);
182       } finally {
183         chaosMonkey.stopTheMonkey();
184       }
185 
186       // ideally this should go into chaosMonkey
187       restartZk(1000 * (5 + random().nextInt(4)));
188 
189       for (StoppableThread indexThread : threads) {
190         indexThread.safeStop();
191       }
192       
193       // start any downed jetties to be sure we still will end up with a leader per shard...
194       
195       // wait for stop...
196       for (StoppableThread indexThread : threads) {
197         indexThread.join();
198       }
199       
200       // try and wait for any replications and what not to finish...
201       
202       Thread.sleep(2000);
203       
204       // wait until there are no recoveries...
205       waitForThingsToLevelOut(Integer.MAX_VALUE);//Math.round((runLength / 1000.0f / 3.0f)));
206       
207       // make sure we again have leaders for each shard
208       for (int j = 1; j < sliceCount; j++) {
209         zkStateReader.getLeaderRetry(DEFAULT_COLLECTION, "shard" + j, 30000);
210       }
211       
212       commit();
213       
214       // TODO: assert we didnt kill everyone
215       
216       zkStateReader.updateClusterState();
217       assertTrue(zkStateReader.getClusterState().getLiveNodes().size() > 0);
218       
219       
220       // we expect full throttle fails, but cloud client should not easily fail
221       for (StoppableThread indexThread : threads) {
222         if (indexThread instanceof StoppableIndexingThread && !(indexThread instanceof FullThrottleStoppableIndexingThread)) {
223           int failCount = ((StoppableIndexingThread) indexThread).getFailCount();
224           assertFalse("There were too many update fails (" + failCount + " > " + FAIL_TOLERANCE
225               + ") - we expect it can happen, but shouldn't easily", failCount > FAIL_TOLERANCE);
226         }
227       }
228       
229       
230       Set<String> addFails = getAddFails(indexTreads);
231       Set<String> deleteFails = getDeleteFails(indexTreads);
232       // full throttle thread can
233       // have request fails 
234       checkShardConsistency(!runFullThrottle, true, addFails, deleteFails);
235       
236       long ctrlDocs = controlClient.query(new SolrQuery("*:*")).getResults()
237       .getNumFound(); 
238       
239       // ensure we have added more than 0 docs
240       long cloudClientDocs = cloudClient.query(new SolrQuery("*:*"))
241           .getResults().getNumFound();
242       
243       assertTrue("Found " + ctrlDocs + " control docs", cloudClientDocs > 0);
244       
245       if (VERBOSE) System.out.println("control docs:"
246           + controlClient.query(new SolrQuery("*:*")).getResults()
247               .getNumFound() + "\n\n");
248       
249       // try and make a collection to make sure the overseer has survived the expiration and session loss
250 
251       // sometimes we restart zookeeper as well
252       if (random().nextBoolean()) {
253         restartZk(1000 * (5 + random().nextInt(4)));
254       }
255 
256       try (CloudSolrClient client = createCloudClient("collection1")) {
257           createCollection(null, "testcollection",
258               1, 1, 1, client, null, "conf1");
259 
260       }
261       List<Integer> numShardsNumReplicas = new ArrayList<>(2);
262       numShardsNumReplicas.add(1);
263       numShardsNumReplicas.add(1);
264       checkForCollection("testcollection", numShardsNumReplicas, null);
265       
266       testSuccessful = true;
267     } finally {
268       if (!testSuccessful) {
269         printLayout();
270       }
271     }

274   private Set<String> getAddFails(List<StoppableIndexingThread> threads) {
275     Set<String> addFails = new HashSet<String>();
276     for (StoppableIndexingThread thread : threads)   {
277       addFails.addAll(thread.getAddFails());
278     }
279     return addFails;
280   }
281   
282   private Set<String> getDeleteFails(List<StoppableIndexingThread> threads) {
283     Set<String> deleteFails = new HashSet<String>();
284     for (StoppableIndexingThread thread : threads)   {
285       deleteFails.addAll(thread.getDeleteFails());
286     }
287     return deleteFails;
288   }

290   class FullThrottleStoppableIndexingThread extends StoppableIndexingThread {
291     private CloseableHttpClient httpClient = HttpClientUtil.createClient(null);
292     private volatile boolean stop = false;
293     int clientIndex = 0;
294     private ConcurrentUpdateSolrClient cusc;
295     private List<SolrClient> clients;
296     private AtomicInteger fails = new AtomicInteger();
297     
298     public FullThrottleStoppableIndexingThread(List<SolrClient> clients,
299                                                String id, boolean doDeletes) {
300       super(controlClient, cloudClient, id, doDeletes);
301       setName("FullThrottleStopableIndexingThread");
302       setDaemon(true);
303       this.clients = clients;
304       HttpClientUtil.setConnectionTimeout(httpClient, clientConnectionTimeout);
305       HttpClientUtil.setSoTimeout(httpClient, clientSoTimeout);
306       cusc = new ConcurrentUpdateSolrClient(
307           ((HttpSolrClient) clients.get(0)).getBaseURL(), httpClient, 8,
308           2) {
309         @Override
310         public void handleError(Throwable ex) {
311           log.warn("cusc error", ex);
312         }
313       };
314     }
315     
316     @Override
317     public void run() {
318       int i = 0;
319       int numDeletes = 0;
320       int numAdds = 0;
321 
322       while (true && !stop) {
323         String id = this.id + "-" + i;
324         ++i;
325         
326         if (doDeletes && random().nextBoolean() && deletes.size() > 0) {
327           String delete = deletes.remove(0);
328           try {
329             numDeletes++;
330             cusc.deleteById(delete);
331           } catch (Exception e) {
332             changeUrlOnError(e);
333             fails.incrementAndGet();
334           }
335         }
336         
337         try {
338           numAdds++;
339           if (numAdds > (TEST_NIGHTLY ? 4002 : 197))
340             continue;
341           SolrInputDocument doc = getDoc(
342               "id",
343               id,
344               i1,
345               50,
346               t1,
347               "Saxon heptarchies that used to rip around so in old times and raise Cain.  My, you ought to seen old Henry the Eight when he was in bloom.  He WAS a blossom.  He used to marry a new wife every day, and chop off her head next morning.  And he would do it just as indifferent as if ");
348           cusc.add(doc);
349         } catch (Exception e) {
350           changeUrlOnError(e);
351           fails.incrementAndGet();
352         }
353         
354         if (doDeletes && random().nextBoolean()) {
355           deletes.add(id);
356         }
357         
358       }
359 
360       log.info("FT added docs:" + numAdds + " with " + fails + " fails" + " deletes:" + numDeletes);
361     }
362 
363     private void changeUrlOnError(Exception e) {
364       if (e instanceof ConnectException) {
365         clientIndex++;
366         if (clientIndex > clients.size() - 1) {
367           clientIndex = 0;
368         }
369         cusc.shutdownNow();
370         cusc = new ConcurrentUpdateSolrClient(
371             ((HttpSolrClient) clients.get(clientIndex)).getBaseURL(),
372             httpClient, 30, 3) {
373           @Override
374           public void handleError(Throwable ex) {
375             log.warn("cusc error", ex);
376           }
377         };
378       }
379     }
380     
381     @Override
382     public void safeStop() {
383       stop = true;
384       cusc.blockUntilFinished();
385       cusc.shutdownNow();
386       IOUtils.closeQuietly(httpClient);
387     }
388 
389     @Override
390     public int getFailCount() {
391       return fails.get();
392     }
393     
394     @Override
395     public Set<String> getAddFails() {
396       throw new UnsupportedOperationException();
397     }
398     
399     @Override
400     public Set<String> getDeleteFails() {
401       throw new UnsupportedOperationException();
402     }
403     
404   };
