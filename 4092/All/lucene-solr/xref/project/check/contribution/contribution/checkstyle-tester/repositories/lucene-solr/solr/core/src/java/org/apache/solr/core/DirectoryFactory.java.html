39  /**
40   * Provides access to a Directory implementation. You must release every
41   * Directory that you get.
42   */
43  public abstract class DirectoryFactory implements NamedListInitializedPlugin,
44      Closeable {

46    // Estimate 10M docs, 100GB size, to avoid caching by NRTCachingDirectory
47    // Stayed away from upper bounds of the int/long in case any other code tried to aggregate these numbers.
48    // A large estimate should currently have no other side effects.
49    public static final IOContext IOCONTEXT_NO_CACHE = new IOContext(new FlushInfo(10*1000*1000, 100L*1000*1000*1000));

51    protected static final String INDEX_W_TIMESTAMP_REGEX = "index\\.[0-9]{17}"; // see SnapShooter.DATE_FMT

135    * @param path to remove
136    * @throws IOException If there is a low-level I/O error.

140   /**
141    * Override for more efficient moves.
142    * 
143    * Intended for use with replication - use
144    * carefully - some Directory wrappers will
145    * cache files for example.
146    * 
147    * @throws IOException If there is a low-level I/O error.
148    */
149   public void move(Directory fromDir, Directory toDir, String fileName, IOContext ioContext) throws IOException {
150     toDir.copyFrom(fromDir, fileName, fileName, ioContext);
151     fromDir.deleteFile(fileName);
152   }

212   public static long sizeOfDirectory(Directory directory) throws IOException {
213     final String[] files = directory.listAll();
214     long size = 0;
215     
216     for (final String file : files) {
217       size += sizeOf(directory, file);
218       if (size < 0) {
219         break;
220       }
221     }
222     
223     return size;
224   }
225   
226   public static long sizeOf(Directory directory, String file) throws IOException {
227     try {
228       return directory.fileLength(file);
229     } catch (IOException e) {
230       // could be a race, file no longer exists, access denied, is a directory, etc.
231       return 0;
232     }
233   }
234   
235   /**
236    * Delete the files in the Directory
237    */
238   public static boolean empty(Directory dir) {
239     boolean isSuccess = true;
240     String contents[];
241     try {
242       contents = dir.listAll();
243       if (contents != null) {
244         for (String file : contents) {
245           dir.deleteFile(file);
246         }
247       }
248     } catch (IOException e) {
249       SolrException.log(log, "Error deleting files from Directory", e);
250       isSuccess = false;
251     }
252     return isSuccess;
253   }

267   public String getDataHome(CoreDescriptor cd) throws IOException {
268     // by default, we go off the instance directory
269     String instanceDir = new File(cd.getInstanceDir()).getAbsolutePath();
270     return normalize(SolrResourceLoader.normalizeDir(instanceDir) + cd.getDataDir());

280   public void cleanupOldIndexDirectories(final String dataDirPath, final String currentIndexDirPath) {
281     File dataDir = new File(dataDirPath);
282     if (!dataDir.isDirectory()) {
283       log.warn("{} does not point to a valid data directory; skipping clean-up of old index directories.", dataDirPath);
284       return;
285     }
286 
287     final File currentIndexDir = new File(currentIndexDirPath);
288     File[] oldIndexDirs = dataDir.listFiles(new FileFilter() {
289       @Override
290       public boolean accept(File file) {
291         String fileName = file.getName();
292         return file.isDirectory() &&
293                !file.equals(currentIndexDir) &&
294                (fileName.equals("index") || fileName.matches(INDEX_W_TIMESTAMP_REGEX));
295       }
296     });
297 
298     if (oldIndexDirs == null || oldIndexDirs.length == 0)
299       return; // nothing to do (no log message needed)
300 
301     log.info("Found {} old index directories to clean-up under {}", oldIndexDirs.length, dataDirPath);
302     for (File dir : oldIndexDirs) {
303 
304       String dirToRmPath = dir.getAbsolutePath();
305       try {
306         if (deleteOldIndexDirectory(dirToRmPath)) {
307           log.info("Deleted old index directory: {}", dirToRmPath);
308         } else {
309           log.warn("Delete old index directory {} failed.", dirToRmPath);
310         }
311       } catch (IOException ioExc) {
312         log.error("Failed to delete old directory {} due to: {}", dir.getAbsolutePath(), ioExc.toString());
313       }
314     }
315   }
316 
317   // Extension point to allow sub-classes to infuse additional code when deleting old index directories
318   protected boolean deleteOldIndexDirectory(String oldDirPath) throws IOException {
319     File dirToRm = new File(oldDirPath);
320     FileUtils.deleteDirectory(dirToRm);
321     return !dirToRm.isDirectory();
322   }
