121   /** expert: instantiates a new reader */
122   Lucene410DocValuesProducer(SegmentReadState state, String dataCodec, String dataExtension, String metaCodec, String metaExtension) throws IOException {
123     String metaName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, metaExtension);
124     this.maxDoc = state.segmentInfo.maxDoc();
125     merging = false;
126     
127     int version = -1;
128     int numFields = -1;
129     
130     // read in the entries from the metadata file.
131     try (ChecksumIndexInput in = state.directory.openChecksumInput(metaName, state.context)) {
132       Throwable priorE = null;
133       try {
134         version = CodecUtil.checkHeader(in, metaCodec, 
135                                         Lucene410DocValuesFormat.VERSION_START,
136                                         Lucene410DocValuesFormat.VERSION_CURRENT);
137         numFields = readFields(in, state.fieldInfos);
138       } catch (Throwable exception) {
139         priorE = exception;
140       } finally {
141         CodecUtil.checkFooter(in, priorE);
142       }
143     }
144     
145     this.numFields = numFields;
146     String dataName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, dataExtension);
147     this.data = state.directory.openInput(dataName, state.context);
148     boolean success = false;
149     try {
150       final int version2 = CodecUtil.checkHeader(data, dataCodec, 
151                                                  Lucene410DocValuesFormat.VERSION_START,
152                                                  Lucene410DocValuesFormat.VERSION_CURRENT);
153       if (version != version2) {
154         throw new CorruptIndexException("Format versions mismatch: meta=" + version + ", data=" + version2, data);
155       }
156       
157       // NOTE: data file is too costly to verify checksum against all the bytes on open,
158       // but for now we at least verify proper structure of the checksum footer: which looks
159       // for FOOTER_MAGIC + algorithmID. This is cheap and can detect some forms of corruption
160       // such as file truncation.
161       CodecUtil.retrieveChecksum(data);
162 
163       success = true;
164     } finally {
165       if (!success) {
166         IOUtils.closeWhileHandlingException(this.data);
167       }
168     }
169     
170     ramBytesUsed = new AtomicLong(RamUsageEstimator.shallowSizeOfInstance(getClass()));
171   }

173   private void readSortedField(FieldInfo info, IndexInput meta) throws IOException {
174     // sorted = binary + numeric
175     if (meta.readVInt() != info.number) {
176       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
177     }
178     if (meta.readByte() != Lucene410DocValuesFormat.BINARY) {
179       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
180     }
181     BinaryEntry b = readBinaryEntry(meta);
182     binaries.put(info.name, b);
183     
184     if (meta.readVInt() != info.number) {
185       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
186     }
187     if (meta.readByte() != Lucene410DocValuesFormat.NUMERIC) {
188       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
189     }
190     NumericEntry n = readNumericEntry(meta);
191     ords.put(info.name, n);
192   }

194   private void readSortedSetFieldWithAddresses(FieldInfo info, IndexInput meta) throws IOException {
195     // sortedset = binary + numeric (addresses) + ordIndex
196     if (meta.readVInt() != info.number) {
197       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
198     }
199     if (meta.readByte() != Lucene410DocValuesFormat.BINARY) {
200       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
201     }
202     BinaryEntry b = readBinaryEntry(meta);
203     binaries.put(info.name, b);
204 
205     if (meta.readVInt() != info.number) {
206       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
207     }
208     if (meta.readByte() != Lucene410DocValuesFormat.NUMERIC) {
209       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
210     }
211     NumericEntry n1 = readNumericEntry(meta);
212     ords.put(info.name, n1);
213 
214     if (meta.readVInt() != info.number) {
215       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
216     }
217     if (meta.readByte() != Lucene410DocValuesFormat.NUMERIC) {
218       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
219     }
220     NumericEntry n2 = readNumericEntry(meta);
221     ordIndexes.put(info.name, n2);
222   }

224   private int readFields(IndexInput meta, FieldInfos infos) throws IOException {
225     int numFields = 0;
226     int fieldNumber = meta.readVInt();
227     while (fieldNumber != -1) {
228       numFields++;
229       FieldInfo info = infos.fieldInfo(fieldNumber);
230       if (info == null) {
231         // trickier to validate more: because we use multiple entries for "composite" types like sortedset, etc.
232         throw new CorruptIndexException("Invalid field number: " + fieldNumber, meta);
233       }
234       byte type = meta.readByte();
235       if (type == Lucene410DocValuesFormat.NUMERIC) {
236         numerics.put(info.name, readNumericEntry(meta));
237       } else if (type == Lucene410DocValuesFormat.BINARY) {
238         BinaryEntry b = readBinaryEntry(meta);
239         binaries.put(info.name, b);
240       } else if (type == Lucene410DocValuesFormat.SORTED) {
241         readSortedField(info, meta);
242       } else if (type == Lucene410DocValuesFormat.SORTED_SET) {
243         SortedSetEntry ss = readSortedSetEntry(meta);
244         sortedSets.put(info.name, ss);
245         if (ss.format == SORTED_WITH_ADDRESSES) {
246           readSortedSetFieldWithAddresses(info, meta);
247         } else if (ss.format == SORTED_SINGLE_VALUED) {
248           if (meta.readVInt() != fieldNumber) {
249             throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
250           }
251           if (meta.readByte() != Lucene410DocValuesFormat.SORTED) {
252             throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
253           }
254           readSortedField(info, meta);
255         } else {
256           throw new AssertionError();
257         }
258       } else if (type == Lucene410DocValuesFormat.SORTED_NUMERIC) {
259         SortedSetEntry ss = readSortedSetEntry(meta);
260         sortedNumerics.put(info.name, ss);
261         if (meta.readVInt() != fieldNumber) {
262           throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
263         }
264         if (meta.readByte() != Lucene410DocValuesFormat.NUMERIC) {
265           throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
266         }
267         numerics.put(info.name, readNumericEntry(meta));
268         if (ss.format == SORTED_WITH_ADDRESSES) {
269           if (meta.readVInt() != fieldNumber) {
270             throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
271           }
272           if (meta.readByte() != Lucene410DocValuesFormat.NUMERIC) {
273             throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
274           }
275           NumericEntry ordIndex = readNumericEntry(meta);
276           ordIndexes.put(info.name, ordIndex);
277         } else if (ss.format != SORTED_SINGLE_VALUED) {
278           throw new AssertionError();
279         }
280       } else {
281         throw new CorruptIndexException("invalid type: " + type, meta);
282       }
283       fieldNumber = meta.readVInt();
284     }
285     return numFields;
286   }

288   static NumericEntry readNumericEntry(IndexInput meta) throws IOException {
289     NumericEntry entry = new NumericEntry();
290     entry.format = meta.readVInt();
291     entry.missingOffset = meta.readLong();
292     entry.offset = meta.readLong();
293     entry.count = meta.readVLong();
294     switch(entry.format) {
295       case GCD_COMPRESSED:
296         entry.minValue = meta.readLong();
297         entry.gcd = meta.readLong();
298         entry.bitsPerValue = meta.readVInt();
299         break;
300       case TABLE_COMPRESSED:
301         final int uniqueValues = meta.readVInt();
302         if (uniqueValues > 256) {
303           throw new CorruptIndexException("TABLE_COMPRESSED cannot have more than 256 distinct values, got=" + uniqueValues, meta);
304         }
305         entry.table = new long[uniqueValues];
306         for (int i = 0; i < uniqueValues; ++i) {
307           entry.table[i] = meta.readLong();
308         }
309         entry.bitsPerValue = meta.readVInt();
310         break;
311       case DELTA_COMPRESSED:
312         entry.minValue = meta.readLong();
313         entry.bitsPerValue = meta.readVInt();
314         break;
315       case MONOTONIC_COMPRESSED:
316         entry.packedIntsVersion = meta.readVInt();
317         entry.blockSize = meta.readVInt();
318         break;
319       default:
320         throw new CorruptIndexException("Unknown format: " + entry.format + ", input=", meta);
321     }
322     entry.endOffset = meta.readLong();
323     return entry;
324   }

326   static BinaryEntry readBinaryEntry(IndexInput meta) throws IOException {
327     BinaryEntry entry = new BinaryEntry();
328     entry.format = meta.readVInt();
329     entry.missingOffset = meta.readLong();
330     entry.minLength = meta.readVInt();
331     entry.maxLength = meta.readVInt();
332     entry.count = meta.readVLong();
333     entry.offset = meta.readLong();
334     switch(entry.format) {
335       case BINARY_FIXED_UNCOMPRESSED:
336         break;
337       case BINARY_PREFIX_COMPRESSED:
338         entry.addressesOffset = meta.readLong();
339         entry.packedIntsVersion = meta.readVInt();
340         entry.blockSize = meta.readVInt();
341         entry.reverseIndexOffset = meta.readLong();
342         break;
343       case BINARY_VARIABLE_UNCOMPRESSED:
344         entry.addressesOffset = meta.readLong();
345         entry.packedIntsVersion = meta.readVInt();
346         entry.blockSize = meta.readVInt();
347         break;
348       default:
349         throw new CorruptIndexException("Unknown format: " + entry.format, meta);
350     }
351     return entry;
352   }

356     entry.format = meta.readVInt();
357     if (entry.format != SORTED_SINGLE_VALUED && entry.format != SORTED_WITH_ADDRESSES) {

393   LongValues getNumeric(NumericEntry entry) throws IOException {
394     RandomAccessInput slice = this.data.randomAccessSlice(entry.offset, entry.endOffset - entry.offset);
395     switch (entry.format) {
396       case DELTA_COMPRESSED:
397         final long delta = entry.minValue;
398         final LongValues values = DirectReader.getInstance(slice, entry.bitsPerValue);
399         return new LongValues() {
400           @Override
401           public long get(long id) {
402             return delta + values.get(id);
403           }
404         };
405       case GCD_COMPRESSED:
406         final long min = entry.minValue;
407         final long mult = entry.gcd;
408         final LongValues quotientReader = DirectReader.getInstance(slice, entry.bitsPerValue);
409         return new LongValues() {
410           @Override
411           public long get(long id) {
412             return min + mult * quotientReader.get(id);
413           }
414         };
415       case TABLE_COMPRESSED:
416         final long table[] = entry.table;
417         final LongValues ords = DirectReader.getInstance(slice, entry.bitsPerValue);
418         return new LongValues() {
419           @Override
420           public long get(long id) {
421             return table[(int) ords.get(id)];
422           }
423         };
424       default:
425         throw new AssertionError();
426     }
427   }

444   private BinaryDocValues getFixedBinary(FieldInfo field, final BinaryEntry bytes) throws IOException {
445     final IndexInput data = this.data.slice("fixed-binary", bytes.offset, bytes.count * bytes.maxLength);
446 
447     final BytesRef term = new BytesRef(bytes.maxLength);
448     final byte[] buffer = term.bytes;
449     final int length = term.length = bytes.maxLength;
450     
451     return new LongBinaryDocValues() {
452       @Override
453       public BytesRef get(long id) {
454         try {
455           data.seek(id * length);
456           data.readBytes(buffer, 0, buffer.length);
457           return term;
458         } catch (IOException e) {
459           throw new RuntimeException(e);
460         }
461       }
462     };
463   }

465   /** returns an address instance for variable-length binary values. */
466   private synchronized MonotonicBlockPackedReader getAddressInstance(FieldInfo field, BinaryEntry bytes) throws IOException {
467     MonotonicBlockPackedReader addresses = addressInstances.get(field.name);
468     if (addresses == null) {
469       data.seek(bytes.addressesOffset);
470       addresses = MonotonicBlockPackedReader.of(data, bytes.packedIntsVersion, bytes.blockSize, bytes.count+1, false);
471       if (!merging) {
472         addressInstances.put(field.name, addresses);
473         ramBytesUsed.addAndGet(addresses.ramBytesUsed() + RamUsageEstimator.NUM_BYTES_INT);
474       }
475     }
476     return addresses;
477   }

479   private BinaryDocValues getVariableBinary(FieldInfo field, final BinaryEntry bytes) throws IOException {
480     final MonotonicBlockPackedReader addresses = getAddressInstance(field, bytes);
481 
482     final IndexInput data = this.data.slice("var-binary", bytes.offset, bytes.addressesOffset - bytes.offset);
483     final BytesRef term = new BytesRef(Math.max(0, bytes.maxLength));
484     final byte buffer[] = term.bytes;
485     
486     return new LongBinaryDocValues() {      
487       @Override
488       public BytesRef get(long id) {
489         long startAddress = addresses.get(id);
490         long endAddress = addresses.get(id+1);
491         int length = (int) (endAddress - startAddress);
492         try {
493           data.seek(startAddress);
494           data.readBytes(buffer, 0, length);
495           term.length = length;
496           return term;
497         } catch (IOException e) {
498           throw new RuntimeException(e);
499         }
500       }
501     };
502   }

504   /** returns an address instance for prefix-compressed binary values. */
505   private synchronized MonotonicBlockPackedReader getIntervalInstance(FieldInfo field, BinaryEntry bytes) throws IOException {
506     MonotonicBlockPackedReader addresses = addressInstances.get(field.name);
507     if (addresses == null) {
508       data.seek(bytes.addressesOffset);
509       final long size = (bytes.count + INTERVAL_MASK) >>> INTERVAL_SHIFT;
510       addresses = MonotonicBlockPackedReader.of(data, bytes.packedIntsVersion, bytes.blockSize, size, false);
511       if (!merging) {
512         addressInstances.put(field.name, addresses);
513         ramBytesUsed.addAndGet(addresses.ramBytesUsed() + RamUsageEstimator.NUM_BYTES_INT);
514       }
515     }
516     return addresses;
517   }

519   /** returns a reverse lookup instance for prefix-compressed binary values. */
520   private synchronized ReverseTermsIndex getReverseIndexInstance(FieldInfo field, BinaryEntry bytes) throws IOException {
521     ReverseTermsIndex index = reverseIndexInstances.get(field.name);
522     if (index == null) {
523       index = new ReverseTermsIndex();
524       data.seek(bytes.reverseIndexOffset);
525       long size = (bytes.count + REVERSE_INTERVAL_MASK) >>> REVERSE_INTERVAL_SHIFT;
526       index.termAddresses = MonotonicBlockPackedReader.of(data, bytes.packedIntsVersion, bytes.blockSize, size, false);
527       long dataSize = data.readVLong();
528       PagedBytes pagedBytes = new PagedBytes(15);
529       pagedBytes.copy(data, dataSize);
530       index.terms = pagedBytes.freeze(true);
531       if (!merging) {
532         reverseIndexInstances.put(field.name, index);
533         ramBytesUsed.addAndGet(index.ramBytesUsed());
534       }
535     }
536     return index;
537   }

539   private BinaryDocValues getCompressedBinary(FieldInfo field, final BinaryEntry bytes) throws IOException {
540     final MonotonicBlockPackedReader addresses = getIntervalInstance(field, bytes);
541     final ReverseTermsIndex index = getReverseIndexInstance(field, bytes);
542     assert addresses.size() > 0; // we don't have to handle empty case
543     IndexInput slice = data.slice("terms", bytes.offset, bytes.addressesOffset - bytes.offset);
544     return new CompressedBinaryDocValues(bytes, addresses, index, slice);
545   }

570       @Override

719   private Bits getMissingBits(final long offset) throws IOException {
720     if (offset == -1) {
721       return new Bits.MatchAllBits(maxDoc);
722     } else {
723       int length = (int) ((maxDoc + 7L) >>> 3);
724       final RandomAccessInput in = data.randomAccessSlice(offset, length);
725       return new Bits() {
726         @Override
727         public boolean get(int index) {
728           try {
729             return (in.readByte(index >> 3) & (1 << (index & 7))) != 0;
730           } catch (IOException e) {
731             throw new RuntimeException(e);
732           }
733         }
734 
735         @Override
736         public int length() {
737           return maxDoc;
738         }
739       };
740     }
741   }

748       case SORTED_NUMERIC:

863   static final class CompressedBinaryDocValues extends LongBinaryDocValues {    
864     final long numValues;

874     public CompressedBinaryDocValues(BinaryEntry bytes, MonotonicBlockPackedReader addresses, ReverseTermsIndex index, IndexInput data) throws IOException {
875       this.maxTermLength = bytes.maxLength;
876       this.numValues = bytes.count;
877       this.addresses = addresses;
878       this.numIndexValues = addresses.size();
879       this.data = data;

883       this.termsEnum = getTermsEnum(data);

886     @Override
887     public BytesRef get(long id) {
888       try {
889         termsEnum.seekExact(id);

891       } catch (IOException e) {

893       }

895     
896     long lookupTerm(BytesRef key) {
897       try {
898         switch (termsEnum.seekCeil(key)) {
899           case FOUND: return termsEnum.ord();
900           case NOT_FOUND: return -termsEnum.ord()-1;
901           default: return -numValues-1;
902         }

904         throw new RuntimeException(bogus);

916     private CompressedBinaryTermsEnum getTermsEnum(IndexInput input) throws IOException {
917       return new CompressedBinaryTermsEnum(input);
918     }
919     
920     class CompressedBinaryTermsEnum extends TermsEnum {
921       private long currentOrd = -1;
922       // offset to the start of the current block 
923       private long currentBlockStart;
924       private final IndexInput input;
925       // delta from currentBlockStart to start of each term
926       private final int offsets[] = new int[INTERVAL_COUNT];
927       private final byte buffer[] = new byte[2*INTERVAL_COUNT-1];
928       
929       private final BytesRef term = new BytesRef(maxTermLength);
930       private final BytesRef firstTerm = new BytesRef(maxTermLength);
931       private final BytesRef scratch = new BytesRef();
932       
933       CompressedBinaryTermsEnum(IndexInput input) throws IOException {
934         this.input = input;
935         input.seek(0);
936       }
937       
938       private void readHeader() throws IOException {
939         firstTerm.length = input.readVInt();
940         input.readBytes(firstTerm.bytes, 0, firstTerm.length);
941         input.readBytes(buffer, 0, INTERVAL_COUNT-1);
942         if (buffer[0] == -1) {
943           readShortAddresses();
944         } else {
945           readByteAddresses();
946         }
947         currentBlockStart = input.getFilePointer();
948       }
949       
950       // read single byte addresses: each is delta - 2
951       // (shared prefix byte and length > 0 are both implicit)
952       private void readByteAddresses() throws IOException {
953         int addr = 0;
954         for (int i = 1; i < offsets.length; i++) {
955           addr += 2 + (buffer[i-1] & 0xFF);
956           offsets[i] = addr;
957         }
958       }
959       
960       // read double byte addresses: each is delta - 2
961       // (shared prefix byte and length > 0 are both implicit)
962       private void readShortAddresses() throws IOException {
963         input.readBytes(buffer, INTERVAL_COUNT-1, INTERVAL_COUNT);
964         int addr = 0;
965         for (int i = 1; i < offsets.length; i++) {
966           int x = i<<1;
967           addr += 2 + ((buffer[x-1] << 8) | (buffer[x] & 0xFF));
968           offsets[i] = addr;
969         }
970       }
971       
972       // set term to the first term
973       private void readFirstTerm() throws IOException {
974         term.length = firstTerm.length;
975         System.arraycopy(firstTerm.bytes, firstTerm.offset, term.bytes, 0, term.length);
976       }
977       
978       // read term at offset, delta encoded from first term
979       private void readTerm(int offset) throws IOException {
980         int start = input.readByte() & 0xFF;
981         System.arraycopy(firstTerm.bytes, firstTerm.offset, term.bytes, 0, start);
982         int suffix = offsets[offset] - offsets[offset-1] - 1;
983         input.readBytes(term.bytes, start, suffix);
984         term.length = start + suffix;
985       }
986       
987       @Override
988       public BytesRef next() throws IOException {
989         currentOrd++;
990         if (currentOrd >= numValues) {
991           return null;
992         } else { 
993           int offset = (int) (currentOrd & INTERVAL_MASK);
994           if (offset == 0) {
995             // switch to next block
996             readHeader();
997             readFirstTerm();
998           } else {
999             readTerm(offset);
1000           }
1001           return term;
1002         }
1003       }
1004       
1005       // binary search reverse index to find smaller 
1006       // range of blocks to search
1007       long binarySearchIndex(BytesRef text) throws IOException {
1008         long low = 0;
1009         long high = numReverseIndexValues - 1;
1010         while (low <= high) {
1011           long mid = (low + high) >>> 1;
1012           reverseTerms.fill(scratch, reverseAddresses.get(mid));
1013           int cmp = scratch.compareTo(text);
1014           
1015           if (cmp < 0) {
1016             low = mid + 1;
1017           } else if (cmp > 0) {
1018             high = mid - 1;
1019           } else {
1020             return mid;
1021           }
1022         }
1023         return high;
1024       }
1025       
1026       // binary search against first term in block range 
1027       // to find term's block
1028       long binarySearchBlock(BytesRef text, long low, long high) throws IOException {       
1029         while (low <= high) {
1030           long mid = (low + high) >>> 1;
1031           input.seek(addresses.get(mid));
1032           term.length = input.readVInt();
1033           input.readBytes(term.bytes, 0, term.length);
1034           int cmp = term.compareTo(text);
1035           
1036           if (cmp < 0) {
1037             low = mid + 1;
1038           } else if (cmp > 0) {
1039             high = mid - 1;
1040           } else {
1041             return mid;
1042           }
1043         }
1044         return high;
1045       }
1046       
1047       @Override
1048       public SeekStatus seekCeil(BytesRef text) throws IOException {
1049         // locate block: narrow to block range with index, then search blocks
1050         final long block;
1051         long indexPos = binarySearchIndex(text);
1052         if (indexPos < 0) {
1053           block = 0;
1054         } else {
1055           long low = indexPos << BLOCK_INTERVAL_SHIFT;
1056           long high = Math.min(numIndexValues - 1, low + BLOCK_INTERVAL_MASK);
1057           block = Math.max(low, binarySearchBlock(text, low, high));
1058         }
1059         
1060         // position before block, then scan to term.
1061         input.seek(addresses.get(block));
1062         currentOrd = (block << INTERVAL_SHIFT) - 1;
1063         
1064         while (next() != null) {
1065           int cmp = term.compareTo(text);
1066           if (cmp == 0) {
1067             return SeekStatus.FOUND;
1068           } else if (cmp > 0) {
1069             return SeekStatus.NOT_FOUND;
1070           }
1071         }
1072         return SeekStatus.END;
1073       }
1074       
1075       @Override
1076       public void seekExact(long ord) throws IOException {
1077         long block = ord >>> INTERVAL_SHIFT;
1078         if (block != currentOrd >>> INTERVAL_SHIFT) {
1079           // switch to different block
1080           input.seek(addresses.get(block));
1081           readHeader();
1082         }
1083         
1084         currentOrd = ord;
1085         
1086         int offset = (int) (ord & INTERVAL_MASK);
1087         if (offset == 0) {
1088           readFirstTerm();
1089         } else {
1090           input.seek(currentBlockStart + offsets[offset-1]);
1091           readTerm(offset);
1092         }
1093       }
