57    private static class WildCardSupportMap extends HashMap<String, List<Permission>> {
58      final Set<String> wildcardPrefixes = new HashSet<>();
59  
60      @Override
61      public List<Permission> put(String key, List<Permission> value) {
62        if (key != null && key.endsWith("/*")) {
63          key = key.substring(0, key.length() - 2);
64          wildcardPrefixes.add(key);
65        }
66        return super.put(key, value);
67      }
68  
69      @Override
70      public List<Permission> get(Object key) {
71        List<Permission> result = super.get(key);
72        if (key == null || result != null) return result;
73        if (!wildcardPrefixes.isEmpty()) {
74          for (String s : wildcardPrefixes) {
75            if (key.toString().startsWith(s)) {
76              List<Permission> l = super.get(s);
77              if (l != null) {
78                result = result == null ? new ArrayList<Permission>() : new ArrayList<Permission>(result);
79                result.addAll(l);
80              }
81            }
82          }
83        }
84        return result;
85      }
86    }

88    @Override
89    public AuthorizationResponse authorize(AuthorizationContext context) {
90      List<AuthorizationContext.CollectionRequest> collectionRequests = context.getCollectionRequests();
91      if (context.getRequestType() == AuthorizationContext.RequestType.ADMIN) {
92        MatchStatus flag = checkCollPerm(mapping.get(""), context);
93        return flag.rsp;
94      }
95  
96      for (AuthorizationContext.CollectionRequest collreq : collectionRequests) {
97        //check permissions for each collection
98        MatchStatus flag = checkCollPerm(mapping.get(collreq.collectionName), context);
99        if (flag != MatchStatus.NO_PERMISSIONS_FOUND) return flag.rsp;
100     }
101     //check global permissions.
102     MatchStatus flag = checkCollPerm(mapping.get(null), context);
103     return flag.rsp;
104   }
105 
106   private MatchStatus checkCollPerm(Map<String, List<Permission>> pathVsPerms,
107                                     AuthorizationContext context) {
108     if (pathVsPerms == null) return MatchStatus.NO_PERMISSIONS_FOUND;
109 
110     String path = context.getResource();
111     MatchStatus flag = checkPathPerm(pathVsPerms.get(path), context);
112     if (flag != MatchStatus.NO_PERMISSIONS_FOUND) return flag;
113     return checkPathPerm(pathVsPerms.get(null), context);
114   }
115 
116   private MatchStatus checkPathPerm(List<Permission> permissions, AuthorizationContext context) {
117     if (permissions == null || permissions.isEmpty()) return MatchStatus.NO_PERMISSIONS_FOUND;
118     Principal principal = context.getUserPrincipal();
119     loopPermissions:
120     for (int i = 0; i < permissions.size(); i++) {
121       Permission permission = permissions.get(i);
122       if (permission.method != null && !permission.method.contains(context.getHttpMethod())) {
123         //this permissions HTTP method does not match this rule. try other rules
124         continue;
125       }
126       if(permission.predicate != null){
127         if(!permission.predicate.test(context)) continue ;
128       }
129 
130       if (permission.params != null) {
131         for (Map.Entry<String, Object> e : permission.params.entrySet()) {
132           String paramVal = context.getParams().get(e.getKey());
133           Object val = e.getValue();
134           if (val instanceof List) {
135             if (!((List) val).contains(paramVal)) continue loopPermissions;
136           } else if (!Objects.equals(val, paramVal)) continue loopPermissions;
137         }
138       }
139 
140       if (permission.role == null) {
141         //no role is assigned permission.That means everybody is allowed to access
142         return MatchStatus.PERMITTED;
143       }
144       if (principal == null) {
145         log.info("request has come without principal. failed permission {} ",permission);
146         //this resource needs a principal but the request has come without
147         //any credential.
148         return MatchStatus.USER_REQUIRED;
149       }
150 
151       for (String role : permission.role) {
152         Set<String> userRoles = usersVsRoles.get(principal.getName());
153         if (userRoles != null && userRoles.contains(role)) return MatchStatus.PERMITTED;
154       }
155       log.info("This resource is configured to have a permission {}, The principal {} does not have the right role ", permission, principal);
156       return MatchStatus.FORBIDDEN;
157     }
158     log.debug("No permissions configured for the resource {} . So allowed to access", context.getResource());
159     return MatchStatus.NO_PERMISSIONS_FOUND;
160   }

162   @Override
163   public void init(Map<String, Object> initInfo) {
164     mapping.put(null, new WildCardSupportMap());
165     Map<String, Object> map = getMapValue(initInfo, "user-role");
166     for (Object o : map.entrySet()) {
167       Map.Entry e = (Map.Entry) o;
168       String roleName = (String) e.getKey();
169       usersVsRoles.put(roleName, readValueAsSet(map, roleName));
170     }
171     List<Map> perms = getListValue(initInfo, "permissions");
172     for (Map o : perms) {
173       Permission p;
174       try {
175         p = Permission.load(o);
176       } catch (Exception exp) {
177         log.error("Invalid permission ", exp);
178         continue;
179       }
180       permissions.add(p);
181       add2Mapping(p);
182     }
183   }
184 
185   //this is to do optimized lookup of permissions for a given collection/path
186   private void add2Mapping(Permission permission) {
187     for (String c : permission.collections) {
188       WildCardSupportMap m = mapping.get(c);
189       if (m == null) mapping.put(c, m = new WildCardSupportMap());
190       for (String path : permission.path) {
191         List<Permission> perms = m.get(path);
192         if (perms == null) m.put(path, perms = new ArrayList<>());
193         perms.add(permission);
194       }
195     }
196   }

198   /**
199    * read a key value as a set. if the value is a single string ,
200    * return a singleton set
201    *
202    * @param m   the map from which to lookup
203    * @param key the key with which to do lookup
204    */
205   static Set<String> readValueAsSet(Map m, String key) {
206     Set<String> result = new HashSet<>();
207     Object val = m.get(key);
208     if (val == null) {
209       if("collection".equals(key)){
210         //for collection collection: null means a core admin/ collection admin request
211         // otherwise it means a request where collection name is ignored
212         return m.containsKey(key) ?  Collections.singleton("") : Collections.<String>singleton(null);
213       }
214       return null;
215     }
216     if (val instanceof Collection) {
217       Collection list = (Collection) val;
218       for (Object o : list) result.add(String.valueOf(o));
219     } else if (val instanceof String) {
220       result.add((String) val);
221     } else {
222       throw new RuntimeException("Bad value for : " + key);
223     }
224     return result.isEmpty() ? null : Collections.unmodifiableSet(result);
225   }

231     String name;
232     Set<String> path, role, collections, method;
233     Map<String, Object> params;

239     static Permission load(Map m) {
240       Permission p = new Permission();
241       String name = (String) m.get(NAME);
242       if (!m.containsKey("role")) throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "role not specified");
243       p.role = readValueAsSet(m, "role");
244       if (well_known_permissions.containsKey(name)) {
245         HashSet<String> disAllowed = new HashSet<>(knownKeys);
246         disAllowed.remove("role");//these are the only
247         disAllowed.remove(NAME);//allowed keys for well-known permissions
248         for (String s : disAllowed) {
249           if (m.containsKey(s))
250             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, s + " is not a valid key for the permission : " + name);
251         }
252         p.predicate = (Predicate<AuthorizationContext>) ((Map) well_known_permissions.get(name)).get(Predicate.class.getName());
253         m = well_known_permissions.get(name);
254       }
255       p.name = name;
256       p.path = readSetSmart(name, m, "path");
257       p.collections = readSetSmart(name, m, "collection");
258       p.method = readSetSmart(name, m, "method");
259       p.params = (Map<String, Object>) m.get("params");
260       return p;
261     }
262 
263     static final Set<String> knownKeys = ImmutableSet.of("collection", "role", "params", "path", "method", NAME);
264   }

279   /**
280    * This checks for the defaults available other rules for the keys
281    */
282   private static Set<String> readSetSmart(String permissionName, Map m, String key) {
283     Set<String> set = readValueAsSet(m, key);
284     if (set == null && well_known_permissions.containsKey(permissionName)) {
285       set = readValueAsSet((Map) well_known_permissions.get(permissionName), key);
286     }
287     if ("method".equals(key)) {
288       if (set != null) {
289         for (String s : set) if (!HTTP_METHODS.contains(s)) return null;
290       }
291       return set;
292     }
293     return set == null ? singleton((String)null) : set;
294   }

296   @Override
297   public Map<String, Object> edit(Map<String, Object> latestConf, List<CommandOperation> commands) {
298     for (CommandOperation op : commands) {
299       OPERATION operation = null;
300       for (OPERATION o : OPERATION.values()) {
301         if (o.name.equals(op.name)) {
302           operation = o;
303           break;
304         }
305       }
306       if (operation == null) {
307         op.unknownOperation();
308         return null;
309       }
310       latestConf = operation.edit(latestConf, op);
311       if (latestConf == null) return null;
312 
313     }
314     return latestConf;
315   }

317   enum OPERATION {
318     SET_USER_ROLE("set-user-role") {
319       @Override
320       public Map<String, Object> edit(Map<String, Object> latestConf, CommandOperation op) {
321         Map<String, Object> roleMap = getMapValue(latestConf, "user-role");
322         Map<String, Object> map = op.getDataMap();
323         if (op.hasError()) return null;
324         for (Map.Entry<String, Object> e : map.entrySet()) {
325           if (e.getValue() == null) {
326             roleMap.remove(e.getKey());
327             continue;
328           }
329           if (e.getValue() instanceof String || e.getValue() instanceof List) {
330             roleMap.put(e.getKey(), e.getValue());
331           } else {
332             op.addError("Unexpected value ");
333             return null;
334           }
335         }
336         return latestConf;
337       }
338     },
339     SET_PERMISSION("set-permission") {
340       @Override
341       public Map<String, Object> edit(Map<String, Object> latestConf, CommandOperation op) {
342         String name = op.getStr(NAME);
343         Map<String, Object> dataMap = op.getDataMap();
344         if (op.hasError()) return null;
345         dataMap = getDeepCopy(dataMap, 3);
346         String before = (String) dataMap.remove("before");
347         for (String key : dataMap.keySet()) {
348           if (!Permission.knownKeys.contains(key)) op.addError("Unknown key, " + key);
349         }
350         try {
351           Permission.load(dataMap);
352         } catch (Exception e) {
353           op.addError(e.getMessage());
354           return null;
355         }
356         List<Map> permissions = getListValue(latestConf, "permissions");
357         List<Map> permissionsCopy = new ArrayList<>();
358         boolean added = false;
359         for (Map e : permissions) {
360           Object n = e.get(NAME);
361           if (n.equals(before) || n.equals(name)) {
362             added = true;
363             permissionsCopy.add(dataMap);
364           }
365           if (!n.equals(name)) permissionsCopy.add(e);
366         }
367         if (!added && before != null) {
368           op.addError("Invalid 'before' :" + before);
369           return null;
370         }
371         if (!added) permissionsCopy.add(dataMap);
372         latestConf.put("permissions", permissionsCopy);
373         return latestConf;
374       }
375     },
376     UPDATE_PERMISSION("update-permission") {
377       @Override
378       public Map<String, Object> edit(Map<String, Object> latestConf, CommandOperation op) {
379         String name = op.getStr(NAME);
380         if (op.hasError()) return null;
381         for (Map permission : (List<Map>) getListValue(latestConf, "permissions")) {
382           if (name.equals(permission.get(NAME))) {
383             LinkedHashMap copy = new LinkedHashMap<>(permission);
384             copy.putAll(op.getDataMap());
385             op.setCommandData(copy);
386             return SET_PERMISSION.edit(latestConf, op);
387           }
388         }
389         op.addError("No such permission " + name);
390         return null;
391       }
392     },
393     DELETE_PERMISSION("delete-permission") {
394       @Override
395       public Map<String, Object> edit(Map<String, Object> latestConf, CommandOperation op) {
396         List<String> names = op.getStrs("");
397         if (names == null || names.isEmpty()) {
398           op.addError("Invalid command");
399           return null;
400         }
401         names = new ArrayList<>(names);
402         List<Map> copy = new ArrayList<>();
403         List<Map> p = getListValue(latestConf, "permissions");
404         for (Map map : p) {
405           Object n = map.get(NAME);
406           if (names.contains(n)) {
407             names.remove(n);
408             continue;
409           } else {
410             copy.add(map);
411           }
412         }
413         if (!names.isEmpty()) {
414           op.addError("Unknown permission name(s) " + names);
415           return null;
416         }
417         latestConf.put("permissions", copy);
418         return latestConf;
419       }
420     };

476   private static Predicate<AuthorizationContext> getPredicate(final boolean isEdit) {
477     return new Predicate<AuthorizationContext>() {
478       @Override
479       public boolean test(AuthorizationContext context) {
480         String action = context.getParams().get("action");
481         if (action == null) return false;
482         CollectionParams.CollectionAction collectionAction = CollectionParams.CollectionAction.get(action);
483         if (collectionAction == null) return false;
484         return isEdit ? collectionAction.isWrite : !collectionAction.isWrite;
485       }
486     };
487   }
