37  /**
38   * Splits words into subwords and performs optional transformations on subword
39   * groups. Words are split into subwords with the following rules:
40   * <ul>
41   * <li>split on intra-word delimiters (by default, all non alpha-numeric
42   * characters): <code>"Wi-Fi"</code> &#8594; <code>"Wi", "Fi"</code></li>
43   * <li>split on case transitions: <code>"PowerShot"</code> &#8594;
44   * <code>"Power", "Shot"</code></li>
45   * <li>split on letter-number transitions: <code>"SD500"</code> &#8594;
46   * <code>"SD", "500"</code></li>
47   * <li>leading and trailing intra-word delimiters on each subword are ignored:
48   * <code>"//hello---there, 'dude'"</code> &#8594;
49   * <code>"hello", "there", "dude"</code></li>
50   * <li>trailing "'s" are removed for each subword: <code>"O'Neil's"</code>
51   * &#8594; <code>"O", "Neil"</code>
52   * <ul>
53   * <li>Note: this step isn't performed in a separate filter because of possible
54   * subword combinations.</li>
55   * </ul>
56   * </li>
57   * </ul>
58   * 
59   * The <b>combinations</b> parameter affects how subwords are combined:
60   * <ul>
61   * <li>combinations="0" causes no subword combinations: <code>"PowerShot"</code>
62   * &#8594; <code>0:"Power", 1:"Shot"</code> (0 and 1 are the token positions)</li>
63   * <li>combinations="1" means that in addition to the subwords, maximum runs of
64   * non-numeric subwords are catenated and produced at the same position of the
65   * last subword in the run:
66   * <ul>
67   * <li><code>"PowerShot"</code> &#8594;
68   * <code>0:"Power", 1:"Shot" 1:"PowerShot"</code></li>
69   * <li><code>"A's+B's&amp;C's"</code> &gt; <code>0:"A", 1:"B", 2:"C", 2:"ABC"</code>
70   * </li>
71   * <li><code>"Super-Duper-XL500-42-AutoCoder!"</code> &#8594;
72   * <code>0:"Super", 1:"Duper", 2:"XL", 2:"SuperDuperXL", 3:"500" 4:"42", 5:"Auto", 6:"Coder", 6:"AutoCoder"</code>
73   * </li>
74   * </ul>
75   * </li>
76   * </ul>
77   * One use for {@link WordDelimiterFilter} is to help match words with different
78   * subword delimiters. For example, if the source text contained "wi-fi" one may
79   * want "wifi" "WiFi" "wi-fi" "wi+fi" queries to all match. One way of doing so
80   * is to specify combinations="1" in the analyzer used for indexing, and
81   * combinations="0" (the default) in the analyzer used for querying. Given that
82   * the current {@link StandardTokenizer} immediately removes many intra-word
83   * delimiters, it is recommended that this filter be used after a tokenizer that
84   * does not do this (such as {@link WhitespaceTokenizer}).
85   */
86  public final class WordDelimiterFilter extends TokenFilter {

98     * Causes parts of words to be generated:
99     * <p>
100    * "PowerShot" =&gt; "Power" "Shot"
101    */
102   public static final int GENERATE_WORD_PARTS = 1;

104   /**
105    * Causes number subwords to be generated:
106    * <p>
107    * "500-42" =&gt; "500" "42"
108    */
109   public static final int GENERATE_NUMBER_PARTS = 2;

111   /**
112    * Causes maximum runs of word parts to be catenated:
113    * <p>
114    * "wi-fi" =&gt; "wifi"
115    */
116   public static final int CATENATE_WORDS = 4;
117 
118   /**
119    * Causes maximum runs of word parts to be catenated:
120    * <p>
121    * "wi-fi" =&gt; "wifi"
122    */
123   public static final int CATENATE_NUMBERS = 8;
124 
125   /**
126    * Causes all subword parts to be catenated:
127    * <p>
128    * "wi-fi-4000" =&gt; "wifi4000"

130   public static final int CATENATE_ALL = 16;

135    * "500-42" =&gt; "500" "42" "500-42"

137   public static final int PRESERVE_ORIGINAL = 32;

154    * "O'Neil's" =&gt; "O", "Neil"

156   public static final int STEM_ENGLISH_POSSESSIVE = 256;

182   // used for accumulating position increment gaps
183   private int accumPosInc = 0;

185   private char savedBuffer[] = new char[1024];
186   private int savedStartOffset;

188   private String savedType;

190   // if length by start + end offsets doesn't match the term text then assume

195   private boolean hasOutputToken = false;

200   /**
201    * Creates a new WordDelimiterFilter
202    *
203    * @param in TokenStream to be filtered
204    * @param charTypeTable table containing character types
205    * @param configurationFlags Flags configuring the filter
206    * @param protWords If not null is the set of tokens to protect from being delimited

208   public WordDelimiterFilter(TokenStream in, byte[] charTypeTable, int configurationFlags, CharArraySet protWords) {
209     super(in);
210     this.flags = configurationFlags;
211     this.protWords = protWords;
212     this.iterator = new WordDelimiterIterator(
213         charTypeTable, has(SPLIT_ON_CASE_CHANGE), has(SPLIT_ON_NUMERICS), has(STEM_ENGLISH_POSSESSIVE));
214   }

228   @Override
229   public boolean incrementToken() throws IOException {
230     while (true) {
231       if (!hasSavedState) {
232         // process a new input word
233         if (!input.incrementToken()) {
234           return false;
235         }
236 
237         int termLength = termAttribute.length();
238         char[] termBuffer = termAttribute.buffer();
239         
240         accumPosInc += posIncAttribute.getPositionIncrement();
241 
242         iterator.setText(termBuffer, termLength);
243         iterator.next();
244 
245         // word of no delimiters, or protected word: just return it
246         if ((iterator.current == 0 && iterator.end == termLength) ||
247             (protWords != null && protWords.contains(termBuffer, 0, termLength))) {
248           posIncAttribute.setPositionIncrement(accumPosInc);
249           accumPosInc = 0;
250           first = false;
251           return true;
252         }
253         
254         // word of simply delimiters
255         if (iterator.end == WordDelimiterIterator.DONE && !has(PRESERVE_ORIGINAL)) {
256           // if the posInc is 1, simply ignore it in the accumulation
257           // TODO: proper hole adjustment (FilteringTokenFilter-like) instead of this previous logic!
258           if (posIncAttribute.getPositionIncrement() == 1 && !first) {
259             accumPosInc--;
260           }
261           continue;
262         }
263 
264         saveState();
265 
266         hasOutputToken = false;
267         hasOutputFollowingOriginal = !has(PRESERVE_ORIGINAL);
268         lastConcatCount = 0;
269         
270         if (has(PRESERVE_ORIGINAL)) {
271           posIncAttribute.setPositionIncrement(accumPosInc);
272           accumPosInc = 0;
273           first = false;
274           return true;
275         }
276       }
277       
278       // at the end of the string, output any concatenations
279       if (iterator.end == WordDelimiterIterator.DONE) {
280         if (!concat.isEmpty()) {
281           if (flushConcatenation(concat)) {
282             buffer();
283             continue;
284           }
285         }
286         
287         if (!concatAll.isEmpty()) {
288           // only if we haven't output this same combo above!
289           if (concatAll.subwordCount > lastConcatCount) {
290             concatAll.writeAndClear();
291             buffer();
292             continue;
293           }
294           concatAll.clear();
295         }
296         
297         if (bufferedPos < bufferedLen) {
298           if (bufferedPos == 0) {
299             sorter.sort(0, bufferedLen);
300           }
301           clearAttributes();
302           restoreState(buffered[bufferedPos++]);
303           if (first && posIncAttribute.getPositionIncrement() == 0) {
304             // can easily happen with strange combinations (e.g. not outputting numbers, but concat-all)
305             posIncAttribute.setPositionIncrement(1);
306           }
307           first = false;
308           return true;
309         }
310         
311         // no saved concatenations, on to the next input word
312         bufferedPos = bufferedLen = 0;
313         hasSavedState = false;
314         continue;
315       }
316       
317       // word surrounded by delimiters: always output
318       if (iterator.isSingleWord()) {
319         generatePart(true);
320         iterator.next();
321         first = false;
322         return true;
323       }
324       
325       int wordType = iterator.type();
326       
327       // do we already have queued up incompatible concatenations?
328       if (!concat.isEmpty() && (concat.type & wordType) == 0) {
329         if (flushConcatenation(concat)) {
330           hasOutputToken = false;
331           buffer();
332           continue;
333         }
334         hasOutputToken = false;
335       }
336       
337       // add subwords depending upon options
338       if (shouldConcatenate(wordType)) {
339         if (concat.isEmpty()) {
340           concat.type = wordType;
341         }
342         concatenate(concat);
343       }
344       
345       // add all subwords (catenateAll)
346       if (has(CATENATE_ALL)) {
347         concatenate(concatAll);
348       }
349       
350       // if we should output the word or number part
351       if (shouldGenerateParts(wordType)) {
352         generatePart(false);
353         buffer();
354       }
355         
356       iterator.next();
357     }
358   }
359 
360   @Override
361   public void reset() throws IOException {
362     super.reset();
363     hasSavedState = false;
364     concat.clear();
365     concatAll.clear();
366     accumPosInc = bufferedPos = bufferedLen = 0;
367     first = true;
368   }

373   private AttributeSource.State buffered[] = new AttributeSource.State[8];
374   private int startOff[] = new int[8];
375   private int posInc[] = new int[8];
376   private int bufferedLen = 0;
377   private int bufferedPos = 0;
378   private boolean first;
379   
380   private class OffsetSorter extends InPlaceMergeSorter {
381     @Override
382     protected int compare(int i, int j) {
383       int cmp = Integer.compare(startOff[i], startOff[j]);
384       if (cmp == 0) {
385         cmp = Integer.compare(posInc[j], posInc[i]);
386       }
387       return cmp;
388     }
389 
390     @Override
391     protected void swap(int i, int j) {
392       AttributeSource.State tmp = buffered[i];
393       buffered[i] = buffered[j];
394       buffered[j] = tmp;
395       
396       int tmp2 = startOff[i];
397       startOff[i] = startOff[j];
398       startOff[j] = tmp2;
399       
400       tmp2 = posInc[i];
401       posInc[i] = posInc[j];
402       posInc[j] = tmp2;
403     }
404   }
405   
406   final OffsetSorter sorter = new OffsetSorter();
407   
408   private void buffer() {
409     if (bufferedLen == buffered.length) {
410       int newSize = ArrayUtil.oversize(bufferedLen+1, 8);
411       buffered = Arrays.copyOf(buffered, newSize);
412       startOff = Arrays.copyOf(startOff, newSize);
413       posInc = Arrays.copyOf(posInc, newSize);
414     }
415     startOff[bufferedLen] = offsetAttribute.startOffset();
416     posInc[bufferedLen] = posIncAttribute.getPositionIncrement();
417     buffered[bufferedLen] = captureState();
418     bufferedLen++;
419   }
420   
421   /**
422    * Saves the existing attribute states
423    */
424   private void saveState() {
425     // otherwise, we have delimiters, save state
426     savedStartOffset = offsetAttribute.startOffset();
427     savedEndOffset = offsetAttribute.endOffset();
428     // if length by start + end offsets doesn't match the term text then assume this is a synonym and don't adjust the offsets.
429     hasIllegalOffsets = (savedEndOffset - savedStartOffset != termAttribute.length());
430     savedType = typeAttribute.type();
431 
432     if (savedBuffer.length < termAttribute.length()) {
433       savedBuffer = new char[ArrayUtil.oversize(termAttribute.length(), RamUsageEstimator.NUM_BYTES_CHAR)];
434     }
435 
436     System.arraycopy(termAttribute.buffer(), 0, savedBuffer, 0, termAttribute.length());
437     iterator.text = savedBuffer;
438 
439     hasSavedState = true;
440   }

442   /**
443    * Flushes the given WordDelimiterConcatenation by either writing its concat and then clearing, or just clearing.
444    *
445    * @param concatenation WordDelimiterConcatenation that will be flushed
446    * @return {@code true} if the concatenation was written before it was cleared, {@code false} otherwise
447    */
448   private boolean flushConcatenation(WordDelimiterConcatenation concatenation) {
449     lastConcatCount = concatenation.subwordCount;
450     if (concatenation.subwordCount != 1 || !shouldGenerateParts(concatenation.type)) {
451       concatenation.writeAndClear();
452       return true;
453     }
454     concatenation.clear();
455     return false;
456   }
457 
458   /**
459    * Determines whether to concatenate a word or number if the current word is the given type
460    *
461    * @param wordType Type of the current word used to determine if it should be concatenated
462    * @return {@code true} if concatenation should occur, {@code false} otherwise
463    */
464   private boolean shouldConcatenate(int wordType) {
465     return (has(CATENATE_WORDS) && isAlpha(wordType)) || (has(CATENATE_NUMBERS) && isDigit(wordType));
466   }
467 
468   /**
469    * Determines whether a word/number part should be generated for a word of the given type
470    *
471    * @param wordType Type of the word used to determine if a word/number part should be generated
472    * @return {@code true} if a word/number part should be generated, {@code false} otherwise
473    */
474   private boolean shouldGenerateParts(int wordType) {
475     return (has(GENERATE_WORD_PARTS) && isAlpha(wordType)) || (has(GENERATE_NUMBER_PARTS) && isDigit(wordType));
476   }

478   /**
479    * Concatenates the saved buffer to the given WordDelimiterConcatenation
480    *
481    * @param concatenation WordDelimiterConcatenation to concatenate the buffer to
482    */
483   private void concatenate(WordDelimiterConcatenation concatenation) {
484     if (concatenation.isEmpty()) {
485       concatenation.startOffset = savedStartOffset + iterator.current;
486     }
487     concatenation.append(savedBuffer, iterator.current, iterator.end - iterator.current);
488     concatenation.endOffset = savedStartOffset + iterator.end;
489   }
490 
491   /**
492    * Generates a word/number part, updating the appropriate attributes
493    *
494    * @param isSingleWord {@code true} if the generation is occurring from a single word, {@code false} otherwise
495    */
496   private void generatePart(boolean isSingleWord) {
497     clearAttributes();
498     termAttribute.copyBuffer(savedBuffer, iterator.current, iterator.end - iterator.current);
499 
500     int startOffset = savedStartOffset + iterator.current;
501     int endOffset = savedStartOffset + iterator.end;
502     
503     if (hasIllegalOffsets) {
504       // historically this filter did this regardless for 'isSingleWord', 
505       // but we must do a sanity check:
506       if (isSingleWord && startOffset <= savedEndOffset) {
507         offsetAttribute.setOffset(startOffset, savedEndOffset);
508       } else {
509         offsetAttribute.setOffset(savedStartOffset, savedEndOffset);
510       }
511     } else {
512       offsetAttribute.setOffset(startOffset, endOffset);
513     }
514     posIncAttribute.setPositionIncrement(position(false));
515     typeAttribute.setType(savedType);
516   }
517 
518   /**
519    * Get the position increment gap for a subword or concatenation
520    *
521    * @param inject true if this token wants to be injected
522    * @return position increment gap
523    */
524   private int position(boolean inject) {
525     int posInc = accumPosInc;
526 
527     if (hasOutputToken) {
528       accumPosInc = 0;
529       return inject ? 0 : Math.max(1, posInc);
530     }
531 
532     hasOutputToken = true;
533     
534     if (!hasOutputFollowingOriginal) {
535       // the first token following the original is 0 regardless
536       hasOutputFollowingOriginal = true;
537       return 0;
538     }
539     // clear the accumulated position increment
540     accumPosInc = 0;
541     return Math.max(1, posInc);
542   }

596   /**
597    * A WDF concatenated 'run'
598    */
599   final class WordDelimiterConcatenation {
600     final StringBuilder buffer = new StringBuilder();
601     int startOffset;
602     int endOffset;
603     int type;
604     int subwordCount;
605 
606     /**
607      * Appends the given text of the given length, to the concetenation at the given offset
608      *
609      * @param text Text to append
610      * @param offset Offset in the concetenation to add the text
611      * @param length Length of the text to append
612      */
613     void append(char text[], int offset, int length) {
614       buffer.append(text, offset, length);
615       subwordCount++;
616     }
617 
618     /**
619      * Writes the concatenation to the attributes
620      */
621     void write() {
622       clearAttributes();
623       if (termAttribute.length() < buffer.length()) {
624         termAttribute.resizeBuffer(buffer.length());
625       }
626       char termbuffer[] = termAttribute.buffer();
627       
628       buffer.getChars(0, buffer.length(), termbuffer, 0);
629       termAttribute.setLength(buffer.length());
630         
631       if (hasIllegalOffsets) {
632         offsetAttribute.setOffset(savedStartOffset, savedEndOffset);
633       }
634       else {
635         offsetAttribute.setOffset(startOffset, endOffset);
636       }
637       posIncAttribute.setPositionIncrement(position(true));
638       typeAttribute.setType(savedType);
639       accumPosInc = 0;
640     }
641 
642     /**
643      * Determines if the concatenation is empty
644      *
645      * @return {@code true} if the concatenation is empty, {@code false} otherwise
646      */
647     boolean isEmpty() {
648       return buffer.length() == 0;
649     }
650 
651     /**
652      * Clears the concatenation and resets its state
653      */
654     void clear() {
655       buffer.setLength(0);
656       startOffset = endOffset = type = subwordCount = 0;
657     }
658 
659     /**
660      * Convenience method for the common scenario of having to write the concetenation and then clearing its state
661      */
662     void writeAndClear() {
663       write();
664       clear();
665     }
666   }
