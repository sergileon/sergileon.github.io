136   private void doTestReadRecreatedTaxonomy(Random random, boolean closeReader) throws Exception {
137     Directory dir = null;
138     TaxonomyWriter tw = null;
139     TaxonomyReader tr = null;
140     
141     // prepare a few categories
142     int  n = 10;
143     FacetLabel[] cp = new FacetLabel[n];
144     for (int i=0; i<n; i++) {
145       cp[i] = new FacetLabel("a", Integer.toString(i));
146     }
147     
148     try {
149       dir = newDirectory();
150       
151       tw = new DirectoryTaxonomyWriter(dir);
152       tw.addCategory(new FacetLabel("a"));
153       tw.close();
154       
155       tr = new DirectoryTaxonomyReader(dir);
156       int baseNumCategories = tr.getSize();
157       
158       for (int i=0; i<n; i++) {
159         int k = random.nextInt(n);
160         tw = new DirectoryTaxonomyWriter(dir, OpenMode.CREATE);
161         for (int j = 0; j <= k; j++) {
162           tw.addCategory(cp[j]);
163         }
164         tw.close();
165         if (closeReader) {
166           tr.close();
167           tr = new DirectoryTaxonomyReader(dir);
168         } else {
169           TaxonomyReader newtr = TaxonomyReader.openIfChanged(tr);
170           assertNotNull(newtr);
171           tr.close();
172           tr = newtr;
173         }
174         assertEquals("Wrong #categories in taxonomy (i="+i+", k="+k+")", baseNumCategories + 1 + k, tr.getSize());
175       }
176     } finally {
177       IOUtils.close(tr, tw, dir);
178     }
179   }

208   @Test
209   public void testOpenIfChangedManySegments() throws Exception {
210     // test openIfChanged() when the taxonomy contains many segments
211     Directory dir = newDirectory();
212     
213     DirectoryTaxonomyWriter writer = new DirectoryTaxonomyWriter(dir) {
214       @Override
215       protected IndexWriterConfig createIndexWriterConfig(OpenMode openMode) {
216         IndexWriterConfig conf = super.createIndexWriterConfig(openMode);
217         LogMergePolicy lmp = (LogMergePolicy) conf.getMergePolicy();
218         lmp.setMergeFactor(2);
219         return conf;
220       }
221     };
222     TaxonomyReader reader = new DirectoryTaxonomyReader(writer);
223     
224     int numRounds = random().nextInt(10) + 10;
225     int numCategories = 1; // one for root
226     for (int i = 0; i < numRounds; i++) {
227       int numCats = random().nextInt(4) + 1;
228       for (int j = 0; j < numCats; j++) {
229         writer.addCategory(new FacetLabel(Integer.toString(i), Integer.toString(j)));
230       }
231       numCategories += numCats + 1 /* one for round-parent */;
232       TaxonomyReader newtr = TaxonomyReader.openIfChanged(reader);
233       assertNotNull(newtr);
234       reader.close();
235       reader = newtr;
236       
237       // assert categories
238       assertEquals(numCategories, reader.getSize());
239       int roundOrdinal = reader.getOrdinal(new FacetLabel(Integer.toString(i)));
240       int[] parents = reader.getParallelTaxonomyArrays().parents();
241       assertEquals(0, parents[roundOrdinal]); // round's parent is root
242       for (int j = 0; j < numCats; j++) {
243         int ord = reader.getOrdinal(new FacetLabel(Integer.toString(i), Integer.toString(j)));
244         assertEquals(roundOrdinal, parents[ord]); // round's parent is root
245       }
246     }
247     
248     reader.close();
249     writer.close();
250     dir.close();
251   }

253   @Test
254   public void testOpenIfChangedMergedSegment() throws Exception {
255     // test openIfChanged() when all index segments were merged - used to be
256     // a bug in ParentArray, caught by testOpenIfChangedManySegments - only
257     // this test is not random
258     Directory dir = newDirectory();
259     
260     // hold onto IW to forceMerge
261     // note how we don't close it, since DTW will close it.
262     final IndexWriter iw = new IndexWriter(dir,
263         new IndexWriterConfig(new MockAnalyzer(random()))
264             .setMergePolicy(new LogByteSizeMergePolicy()));
265     DirectoryTaxonomyWriter writer = new DirectoryTaxonomyWriter(dir) {
266       @Override
267       protected IndexWriter openIndexWriter(Directory directory,
268           IndexWriterConfig config) throws IOException {
269         return iw;
270       }
271     };
272     
273     TaxonomyReader reader = new DirectoryTaxonomyReader(writer);
274     assertEquals(1, reader.getSize());
275     assertEquals(1, reader.getParallelTaxonomyArrays().parents().length);
276 
277     // add category and call forceMerge -- this should flush IW and merge segments down to 1
278     // in ParentArray.initFromReader, this used to fail assuming there are no parents.
279     writer.addCategory(new FacetLabel("1"));
280     iw.forceMerge(1);
281     
282     // now calling openIfChanged should trip on the bug
283     TaxonomyReader newtr = TaxonomyReader.openIfChanged(reader);
284     assertNotNull(newtr);
285     reader.close();
286     reader = newtr;
287     assertEquals(2, reader.getSize());
288     assertEquals(2, reader.getParallelTaxonomyArrays().parents().length);
289     
290     reader.close();
291     writer.close();
292     dir.close();

340   @Test
341   public void testOpenIfChangedReuseAfterRecreate() throws Exception {
342     // tests that if the taxonomy is recreated, no data is reused from the previous taxonomy
343     Directory dir = newDirectory();
344     DirectoryTaxonomyWriter writer = new DirectoryTaxonomyWriter(dir);
345     FacetLabel cp_a = new FacetLabel("a");
346     writer.addCategory(cp_a);
347     writer.close();
348     
349     DirectoryTaxonomyReader r1 = new DirectoryTaxonomyReader(dir);
350     // fill r1's caches
351     assertEquals(1, r1.getOrdinal(cp_a));
352     assertEquals(cp_a, r1.getPath(1));
353     
354     // now recreate, add a different category
355     writer = new DirectoryTaxonomyWriter(dir, OpenMode.CREATE);
356     FacetLabel cp_b = new FacetLabel("b");
357     writer.addCategory(cp_b);
358     writer.close();
359     
360     DirectoryTaxonomyReader r2 = TaxonomyReader.openIfChanged(r1);
361     assertNotNull(r2);
362     
363     // fill r2's caches
364     assertEquals(1, r2.getOrdinal(cp_b));
365     assertEquals(cp_b, r2.getPath(1));
366     
367     // check that r1 doesn't see cp_b
368     assertEquals(TaxonomyReader.INVALID_ORDINAL, r1.getOrdinal(cp_b));
369     assertEquals(cp_a, r1.getPath(1));
370 
371     // check that r2 doesn't see cp_a
372     assertEquals(TaxonomyReader.INVALID_ORDINAL, r2.getOrdinal(cp_a));
373     assertEquals(cp_b, r2.getPath(1));
374 
375     r2.close();
376     r1.close();
377     dir.close();
378   }

380   @Test
381   public void testOpenIfChangedReuse() throws Exception {
382     // test the reuse of data from the old DTR instance
383     for (boolean nrt : new boolean[] {false, true}) {
384       Directory dir = newDirectory();
385       DirectoryTaxonomyWriter writer = new DirectoryTaxonomyWriter(dir);
386       
387       FacetLabel cp_a = new FacetLabel("a");
388       writer.addCategory(cp_a);
389       if (!nrt) writer.commit();
390       
391       DirectoryTaxonomyReader r1 = nrt ? new DirectoryTaxonomyReader(writer) : new DirectoryTaxonomyReader(dir);
392       // fill r1's caches
393       assertEquals(1, r1.getOrdinal(cp_a));
394       assertEquals(cp_a, r1.getPath(1));
395       
396       FacetLabel cp_b = new FacetLabel("b");
397       writer.addCategory(cp_b);
398       if (!nrt) writer.commit();
399       
400       DirectoryTaxonomyReader r2 = TaxonomyReader.openIfChanged(r1);
401       assertNotNull(r2);
402       
403       // add r2's categories to the caches
404       assertEquals(2, r2.getOrdinal(cp_b));
405       assertEquals(cp_b, r2.getPath(2));
406       
407       // check that r1 doesn't see cp_b
408       assertEquals(TaxonomyReader.INVALID_ORDINAL, r1.getOrdinal(cp_b));
409       assertNull(r1.getPath(2));
410       
411       r1.close();
412       r2.close();
413       writer.close();
414       dir.close();
415     }
416   }

469   @Test
470   public void testGetChildren() throws Exception {
471     Directory dir = newDirectory();
472     DirectoryTaxonomyWriter taxoWriter = new DirectoryTaxonomyWriter(dir);
473     int numCategories = atLeast(10);
474     int numA = 0, numB = 0;
475     Random random = random();
476     // add the two categories for which we'll also add children (so asserts are simpler)
477     taxoWriter.addCategory(new FacetLabel("a"));
478     taxoWriter.addCategory(new FacetLabel("b"));
479     for (int i = 0; i < numCategories; i++) {
480       if (random.nextBoolean()) {
481         taxoWriter.addCategory(new FacetLabel("a", Integer.toString(i)));
482         ++numA;
483       } else {
484         taxoWriter.addCategory(new FacetLabel("b", Integer.toString(i)));
485         ++numB;
486       }
487     }
488     // add category with no children
489     taxoWriter.addCategory(new FacetLabel("c"));
490     taxoWriter.close();
491     
492     DirectoryTaxonomyReader taxoReader = new DirectoryTaxonomyReader(dir);
493 
494     // non existing category
495     ChildrenIterator it = taxoReader.getChildren(taxoReader.getOrdinal(new FacetLabel("invalid")));
496     assertEquals(TaxonomyReader.INVALID_ORDINAL, it.next());
497 
498     // a category with no children
499     it = taxoReader.getChildren(taxoReader.getOrdinal(new FacetLabel("c")));
500     assertEquals(TaxonomyReader.INVALID_ORDINAL, it.next());
501 
502     // arbitrary negative ordinal
503     it = taxoReader.getChildren(-2);
504     assertEquals(TaxonomyReader.INVALID_ORDINAL, it.next());
505 
506     // root's children
507     Set<String> roots = new HashSet<>(Arrays.asList("a", "b", "c"));
508     it = taxoReader.getChildren(TaxonomyReader.ROOT_ORDINAL);
509     while (!roots.isEmpty()) {
510       FacetLabel root = taxoReader.getPath(it.next());
511       assertEquals(1, root.length);
512       assertTrue(roots.remove(root.components[0]));
513     }
514     assertEquals(TaxonomyReader.INVALID_ORDINAL, it.next());
515     
516     for (int i = 0; i < 2; i++) {
517       FacetLabel cp = i == 0 ? new FacetLabel("a") : new FacetLabel("b");
518       int ordinal = taxoReader.getOrdinal(cp);
519       it = taxoReader.getChildren(ordinal);
520       int numChildren = 0;
521       int child;
522       while ((child = it.next()) != TaxonomyReader.INVALID_ORDINAL) {
523         FacetLabel path = taxoReader.getPath(child);
524         assertEquals(2, path.length);
525         assertEquals(path.components[0], i == 0 ? "a" : "b");
526         ++numChildren;
527       }
528       int expected = i == 0 ? numA : numB;
529       assertEquals("invalid num children", expected, numChildren);
530     }
531     taxoReader.close();
532     
533     dir.close();
534   }
