514   private long testIndexQueryDeleteHierarchical(long docId) throws Exception {
515     //index
516     int topDocsNum = atLeast(10);
517     int childsNum = 5+random().nextInt(5);
518     for (int i = 0; i < topDocsNum; ++i) {
519       UpdateRequest uReq = new UpdateRequest();
520       SolrInputDocument topDocument = new SolrInputDocument();
521       topDocument.addField("id", docId++);
522       topDocument.addField("type_s", "parent");
523       topDocument.addField(i + "parent_f1_s", "v1");
524       topDocument.addField(i + "parent_f2_s", "v2");
525       
526       
527       for (int index = 0; index < childsNum; ++index) {
528         docId = addChildren("child", topDocument, index, false, docId);
529       }
530       
531       uReq.add(topDocument);
532       uReq.process(cloudClient);
533       uReq.process(controlClient);
534     }
535     
536     commit();
537     checkShardConsistency();
538     assertDocCounts(VERBOSE);
539     
540     //query
541     // parents
542     SolrQuery query = new SolrQuery("type_s:parent");
543     QueryResponse results = cloudClient.query(query);
544     assertEquals(topDocsNum, results.getResults().getNumFound());
545     
546     //childs 
547     query = new SolrQuery("type_s:child");
548     results = cloudClient.query(query);
549     assertEquals(topDocsNum * childsNum, results.getResults().getNumFound());
550     
551     //grandchilds
552     query = new SolrQuery("type_s:grand");
553     results = cloudClient.query(query);
554     //each topDoc has t childs where each child has x = 0 + 2 + 4 + ..(t-1)*2 grands
555     //x = 2 * (1 + 2 + 3 +.. (t-1)) => arithmetic summ of t-1 
556     //x = 2 * ((t-1) * t / 2) = t * (t - 1)
557     assertEquals(topDocsNum * childsNum * (childsNum - 1), results.getResults().getNumFound());
558     
559     //delete
560     del("*:*");
561     commit();
562     
563     return docId;
564   }

566   private long addChildren(String prefix, SolrInputDocument topDocument, int childIndex, boolean lastLevel, long docId) {
567     SolrInputDocument childDocument = new SolrInputDocument();
568     childDocument.addField("id", docId++);
569     childDocument.addField("type_s", prefix);
570     for (int index = 0; index < childIndex; ++index) {
571       childDocument.addField(childIndex + prefix + index + "_s", childIndex + "value"+ index);
572     }   
573   
574     if (!lastLevel) {
575       for (int i = 0; i < childIndex * 2; ++i) {
576         docId = addChildren("grand", childDocument, i, true, docId);
577       }
578     }
579     topDocument.addChildDocument(childDocument);
580     return docId;
581   }

603   private long testIndexingBatchPerRequestWithHttpSolrClient(long docId) throws Exception {
604     
605     // remove collection
606     ModifiableSolrParams params = new ModifiableSolrParams();
607     params.set("action", CollectionAction.DELETE.toString());
608     params.set("name", "collection1");
609     QueryRequest request = new QueryRequest(params);
610     request.setPath("/admin/collections");
611     
612   
613     cloudClient.request(request);
614     
615     controlClient.deleteByQuery("*:*");
616     controlClient.commit();
617     
618     // somtimes we use an oversharded collection
619     createCollection(null, "collection2", 7, 3, 100000, cloudClient, null, "conf1");
620     cloudClient.setDefaultCollection("collection2");
621     waitForRecoveriesToFinish("collection2", false);
622     
623     class IndexThread extends Thread {
624       Integer name;
625       
626       public IndexThread(Integer name) {
627         this.name = name;
628       }
629       
630       @Override
631       public void run() {
632         int rnds = random().nextInt(TEST_NIGHTLY ? 25 : 3) + 1;
633         for (int i = 0; i < rnds; i++) {
634           UpdateRequest uReq;
635           uReq = new UpdateRequest();
636           int cnt = random().nextInt(TEST_NIGHTLY ? 3313 : 350) + 1;
637           for (int j = 0; j <cnt; j++) {
638             addDoc("thread" + name + "_" + i + "_" + j, uReq);
639           }
640           
641           try {
642             uReq.process(cloudClient);
643             uReq.process(controlClient);
644           } catch (SolrServerException | IOException e) {
645             throw new RuntimeException(e);
646           }
647 
648 
649         }
650       }
651     };
