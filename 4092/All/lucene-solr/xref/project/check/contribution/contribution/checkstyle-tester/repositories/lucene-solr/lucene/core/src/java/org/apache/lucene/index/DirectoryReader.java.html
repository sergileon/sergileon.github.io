30  /** DirectoryReader is an implementation of {@link CompositeReader}
31   that can read indexes in a {@link Directory}. 
32  
33   <p>DirectoryReader instances are usually constructed with a call to
34   one of the static <code>open()</code> methods, e.g. {@link
35   #open(Directory)}.
36  
37   <p> For efficiency, in this API documents are often referred to via
38   <i>document numbers</i>, non-negative integers which each name a unique
39   document in the index.  These document numbers are ephemeral -- they may change
40   as documents are added to and deleted from an index.  Clients should thus not
41   rely on a given document having the same number between sessions.
42  
43   <p>
44   <a name="thread-safety"></a><p><b>NOTE</b>: {@link
45   IndexReader} instances are completely thread
46   safe, meaning multiple threads can call any of its methods,
47   concurrently.  If your application requires external
48   synchronization, you should <b>not</b> synchronize on the
49   <code>IndexReader</code> instance; use your own
50   (non-Lucene) objects instead.
51  */
52  public abstract class DirectoryReader extends BaseCompositeReader<LeafReader> {
53  
54    /** The index directory. */
55    protected final Directory directory;
56    
57    /** Returns a IndexReader reading the index in the given
58     *  Directory
59     * @param directory the index directory
60     * @throws IOException if there is a low-level IO error
61     */
62    public static DirectoryReader open(final Directory directory) throws IOException {
63      return StandardDirectoryReader.open(directory, null);
64    }
65    
66    /**
67     * Open a near real time IndexReader from the {@link org.apache.lucene.index.IndexWriter}.
68     *
69     * @param writer The IndexWriter to open from
70     * @param applyAllDeletes If true, all buffered deletes will
71     * be applied (made visible) in the returned reader.  If
72     * false, the deletes are not applied but remain buffered
73     * (in IndexWriter) so that they will be applied in the
74     * future.  Applying deletes can be costly, so if your app
75     * can tolerate deleted documents being returned you might
76     * gain some performance by passing false.
77     * @return The new IndexReader
78     * @throws CorruptIndexException if the index is corrupt
79     * @throws IOException if there is a low-level IO error
80     *
81     * @see #openIfChanged(DirectoryReader,IndexWriter,boolean)
82     *
83     * @lucene.experimental
84     */
85    public static DirectoryReader open(final IndexWriter writer, boolean applyAllDeletes) throws IOException {
86      return writer.getReader(applyAllDeletes);
87    }
88  
89    /** Expert: returns an IndexReader reading the index in the given
90     *  {@link IndexCommit}.
91     * @param commit the commit point to open
92     * @throws IOException if there is a low-level IO error
93     */
94    public static DirectoryReader open(final IndexCommit commit) throws IOException {
95      return StandardDirectoryReader.open(commit.getDirectory(), commit);
96    }

98    /**
99     * If the index has changed since the provided reader was
100    * opened, open and return a new reader; else, return
101    * null.  The new reader, if not null, will be the same
102    * type of reader as the previous one, ie an NRT reader
103    * will open a new NRT reader, a MultiReader will open a
104    * new MultiReader,  etc.
105    *
106    * <p>This method is typically far less costly than opening a
107    * fully new <code>DirectoryReader</code> as it shares
108    * resources (for example sub-readers) with the provided
109    * <code>DirectoryReader</code>, when possible.
110    *
111    * <p>The provided reader is not closed (you are responsible
112    * for doing so); if a new reader is returned you also
113    * must eventually close it.  Be sure to never close a
114    * reader while other threads are still using it; see
115    * {@link SearcherManager} to simplify managing this.
116    *
117    * @throws CorruptIndexException if the index is corrupt
118    * @throws IOException if there is a low-level IO error
119    * @return null if there are no changes; else, a new
120    * DirectoryReader instance which you must eventually close
121    */  
122   public static DirectoryReader openIfChanged(DirectoryReader oldReader) throws IOException {
123     final DirectoryReader newReader = oldReader.doOpenIfChanged();
124     assert newReader != oldReader;
125     return newReader;
126   }
127 
128   /**
129    * If the IndexCommit differs from what the
130    * provided reader is searching, open and return a new
131    * reader; else, return null.
132    *
133    * @see #openIfChanged(DirectoryReader)
134    */
135   public static DirectoryReader openIfChanged(DirectoryReader oldReader, IndexCommit commit) throws IOException {
136     final DirectoryReader newReader = oldReader.doOpenIfChanged(commit);
137     assert newReader != oldReader;
138     return newReader;
139   }

141   /**
142    * Expert: If there changes (committed or not) in the
143    * {@link IndexWriter} versus what the provided reader is
144    * searching, then open and return a new
145    * IndexReader searching both committed and uncommitted
146    * changes from the writer; else, return null (though, the
147    * current implementation never returns null).
148    *
149    * <p>This provides "near real-time" searching, in that
150    * changes made during an {@link IndexWriter} session can be
151    * quickly made available for searching without closing
152    * the writer nor calling {@link IndexWriter#commit}.
153    *
154    * <p>It's <i>near</i> real-time because there is no hard
155    * guarantee on how quickly you can get a new reader after
156    * making changes with IndexWriter.  You'll have to
157    * experiment in your situation to determine if it's
158    * fast enough.  As this is a new and experimental
159    * feature, please report back on your findings so we can
160    * learn, improve and iterate.</p>
161    *
162    * <p>The very first time this method is called, this
163    * writer instance will make every effort to pool the
164    * readers that it opens for doing merges, applying
165    * deletes, etc.  This means additional resources (RAM,
166    * file descriptors, CPU time) will be consumed.</p>
167    *
168    * <p>For lower latency on reopening a reader, you should
169    * call {@link IndexWriterConfig#setMergedSegmentWarmer} to
170    * pre-warm a newly merged segment before it's committed
171    * to the index.  This is important for minimizing
172    * index-to-search delay after a large merge.  </p>
173    *
174    * <p>If an addIndexes* call is running in another thread,
175    * then this reader will only search those segments from
176    * the foreign index that have been successfully copied
177    * over, so far.</p>
178    *
179    * <p><b>NOTE</b>: Once the writer is closed, any
180    * outstanding readers may continue to be used.  However,
181    * if you attempt to reopen any of those readers, you'll
182    * hit an {@link org.apache.lucene.store.AlreadyClosedException}.</p>
183    *
184    * @return DirectoryReader that covers entire index plus all
185    * changes made so far by this IndexWriter instance, or
186    * null if there are no new changes
187    *
188    * @param writer The IndexWriter to open from
189    *
190    * @param applyAllDeletes If true, all buffered deletes will
191    * be applied (made visible) in the returned reader.  If
192    * false, the deletes are not applied but remain buffered
193    * (in IndexWriter) so that they will be applied in the
194    * future.  Applying deletes can be costly, so if your app
195    * can tolerate deleted documents being returned you might
196    * gain some performance by passing false.
197    *
198    * @throws IOException if there is a low-level IO error
199    *
200    * @lucene.experimental
201    */
202   public static DirectoryReader openIfChanged(DirectoryReader oldReader, IndexWriter writer, boolean applyAllDeletes) throws IOException {
203     final DirectoryReader newReader = oldReader.doOpenIfChanged(writer, applyAllDeletes);
204     assert newReader != oldReader;
205     return newReader;
206   }

208   /** Returns all commit points that exist in the Directory.
209    *  Normally, because the default is {@link
210    *  KeepOnlyLastCommitDeletionPolicy}, there would be only
211    *  one commit point.  But if you're using a custom {@link
212    *  IndexDeletionPolicy} then there could be many commits.
213    *  Once you have a given commit, you can open a reader on
214    *  it by calling {@link DirectoryReader#open(IndexCommit)}
215    *  There must be at least one commit in
216    *  the Directory, else this method throws {@link
217    *  IndexNotFoundException}.  Note that if a commit is in
218    *  progress while this method is running, that commit
219    *  may or may not be returned.
220    *  
221    *  @return a sorted list of {@link IndexCommit}s, from oldest 
222    *  to latest. */
223   public static List<IndexCommit> listCommits(Directory dir) throws IOException {
224     final String[] files = dir.listAll();
225 
226     List<IndexCommit> commits = new ArrayList<>();
227 
228     SegmentInfos latest = SegmentInfos.readLatestCommit(dir);
229     final long currentGen = latest.getGeneration();
230 
231     commits.add(new StandardDirectoryReader.ReaderCommit(null, latest, dir));
232 
233     for(int i=0;i<files.length;i++) {
234 
235       final String fileName = files[i];
236 
237       if (fileName.startsWith(IndexFileNames.SEGMENTS) &&
238           !fileName.equals(IndexFileNames.OLD_SEGMENTS_GEN) &&
239           SegmentInfos.generationFromSegmentsFileName(fileName) < currentGen) {
240 
241         SegmentInfos sis = null;
242         try {
243           // IOException allowed to throw there, in case
244           // segments_N is corrupt
245           sis = SegmentInfos.readCommit(dir, fileName);
246         } catch (FileNotFoundException | NoSuchFileException fnfe) {
247           // LUCENE-948: on NFS (and maybe others), if
248           // you have writers switching back and forth
249           // between machines, it's very likely that the
250           // dir listing will be stale and will claim a
251           // file segments_X exists when in fact it
252           // doesn't.  So, we catch this and handle it
253           // as if the file does not exist
254         }
255 
256         if (sis != null) {
257           commits.add(new StandardDirectoryReader.ReaderCommit(null, sis, dir));
258         }
259       }
260     }
261 
262     // Ensure that the commit points are sorted in ascending order.
263     Collections.sort(commits);
264 
265     return commits;
266   }

268   /**
269    * Returns <code>true</code> if an index likely exists at
270    * the specified directory.  Note that if a corrupt index
271    * exists, or if an index in the process of committing 
272    * @param  directory the directory to check for an index
273    * @return <code>true</code> if an index exists; <code>false</code> otherwise
274    */
275   public static boolean indexExists(Directory directory) throws IOException {
276     // LUCENE-2812, LUCENE-2727, LUCENE-4738: this logic will
277     // return true in cases that should arguably be false,
278     // such as only IW.prepareCommit has been called, or a
279     // corrupt first commit, but it's too deadly to make
280     // this logic "smarter" and risk accidentally returning
281     // false due to various cases like file description
282     // exhaustion, access denied, etc., because in that
283     // case IndexWriter may delete the entire index.  It's
284     // safer to err towards "index exists" than try to be
285     // smart about detecting not-yet-fully-committed or
286     // corrupt indices.  This means that IndexWriter will
287     // throw an exception on such indices and the app must
288     // resolve the situation manually:
289     String[] files = directory.listAll();
290 
291     String prefix = IndexFileNames.SEGMENTS + "_";
292     for(String file : files) {
293       if (file.startsWith(prefix)) {
294         return true;
295       }
296     }
297     return false;
298   }

300   /**
301    * Expert: Constructs a {@code DirectoryReader} on the given subReaders.
302    * @param segmentReaders the wrapped atomic index segment readers. This array is
303    * returned by {@link #getSequentialSubReaders} and used to resolve the correct
304    * subreader for docID-based methods. <b>Please note:</b> This array is <b>not</b>
305    * cloned and not protected for modification outside of this reader.
306    * Subclasses of {@code DirectoryReader} should take care to not allow
307    * modification of this internal array, e.g. {@link #doOpenIfChanged()}.
308    */
309   protected DirectoryReader(Directory directory, LeafReader[] segmentReaders) throws IOException {
310     super(segmentReaders);
311     this.directory = directory;
312   }

314   /** Returns the directory this index resides in. */
315   public final Directory directory() {
316     // Don't ensureOpen here -- in certain cases, when a
317     // cloned/reopened reader needs to commit, it may call
318     // this method on the closed original reader
319     return directory;
320   }

323    * If this reader does not support reopen, return {@code null}, so
324    * client code is happy. This should be consistent with {@link #isCurrent}
325    * (should always return {@code true}) if reopen is not supported.
326    * @throws IOException if there is a low-level IO error
327    * @return null if there are no changes; else, a new
328    * DirectoryReader instance.
329    */
330   protected abstract DirectoryReader doOpenIfChanged() throws IOException;

333    * If this reader does not support reopen from a specific {@link IndexCommit},
334    * throw {@link UnsupportedOperationException}.

336    * @return null if there are no changes; else, a new
337    * DirectoryReader instance.
338    */
339   protected abstract DirectoryReader doOpenIfChanged(final IndexCommit commit) throws IOException;

342    * If this reader does not support reopen from {@link IndexWriter},

351    * Version number when this IndexReader was opened.
352    *
353    * <p>This method
354    * returns the version recorded in the commit that the
355    * reader opened.  This version is advanced every time
356    * a change is made with {@link IndexWriter}.</p>
357    */
358   public abstract long getVersion();
359 
360   /**
361    * Check whether any new changes have occurred to the
362    * index since this reader was opened.
363    *
364    * <p>If this reader was created by calling {@link #open},  
365    * then this method checks if any further commits 
366    * (see {@link IndexWriter#commit}) have occurred in the 
367    * directory.</p>
368    *
369    * <p>If instead this reader is a near real-time reader
370    * (ie, obtained by a call to {@link
371    * DirectoryReader#open(IndexWriter,boolean)}, or by calling {@link #openIfChanged}
372    * on a near real-time reader), then this method checks if
373    * either a new commit has occurred, or any new
374    * uncommitted changes have taken place via the writer.
375    * Note that even if the writer has only performed
376    * merging, this method will still return false.</p>
377    *
378    * <p>In any event, if this returns false, you should call
379    * {@link #openIfChanged} to get a new reader that sees the
380    * changes.</p>
381    *
382    * @throws IOException           if there is a low-level IO error
383    */
384   public abstract boolean isCurrent() throws IOException;

390   public abstract IndexCommit getIndexCommit() throws IOException;
