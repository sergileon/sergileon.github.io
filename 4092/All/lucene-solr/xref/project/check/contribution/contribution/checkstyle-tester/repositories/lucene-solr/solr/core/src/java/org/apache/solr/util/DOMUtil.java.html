43    public static Map<String,String> toMap(NamedNodeMap attrs) {
44      return toMapExcept(attrs);
45    }

47    public static Map<String,String> toMapExcept(NamedNodeMap attrs, String... exclusions) {
48      Map<String,String> args = new HashMap<>();
49      outer: for (int j=0; j<attrs.getLength(); j++) {
50        Node attr = attrs.item(j);
51  
52        // automatically exclude things in the xml namespace, ie: xml:base
53        if (XML_RESERVED_PREFIX.equals(attr.getPrefix())) continue outer;
54  
55        String attrName = attr.getNodeName();
56        for (String ex : exclusions)
57          if (ex.equals(attrName)) continue outer;
58        String val = attr.getNodeValue();
59        args.put(attrName, val);
60      }
61      return args;
62    }

64    public static Node getChild(Node node, String name) {
65      if (!node.hasChildNodes()) return null;
66      NodeList lst = node.getChildNodes();
67      if (lst == null) return null;
68      for (int i=0; i<lst.getLength(); i++) {
69        Node child = lst.item(i);
70        if (name.equals(child.getNodeName())) return child;
71      }
72      return null;
73    }

75    public static String getAttr(NamedNodeMap attrs, String name) {
76      return getAttr(attrs,name,null);
77    }

79    public static String getAttr(Node nd, String name) {
80      return getAttr(nd.getAttributes(), name);
81    }

83    public static String getAttr(NamedNodeMap attrs, String name, String missing_err) {
84      Node attr = attrs==null? null : attrs.getNamedItem(name);
85      if (attr==null) {
86        if (missing_err==null) return null;
87        throw new RuntimeException(missing_err + ": missing mandatory attribute '" + name + "'");
88      }
89      String val = attr.getNodeValue();
90      return val;
91    }
92  
93    public static String getAttr(Node node, String name, String missing_err) {
94      return getAttr(node.getAttributes(), name, missing_err);
95    }

110   public static NamedList<Object> nodesToNamedList(NodeList nlst) {
111     NamedList<Object> clst = new NamedList<>();
112     for (int i=0; i<nlst.getLength(); i++) {
113       addToNamedList(nlst.item(i), clst, null);
114     }
115     return clst;
116   }

118   public static List nodesToList(NodeList nlst) {
119     List lst = new ArrayList();
120     for (int i=0; i<nlst.getLength(); i++) {
121       addToNamedList(nlst.item(i), null, lst);
122     }
123     return lst;
124   }

126   /**
127    * Examines a Node from the DOM representation of a NamedList and adds the
128    * contents of that node to both the specified NamedList and List passed
129    * as arguments.
130    *
131    * @param nd The Node whose type will be used to determine how to parse the
132    *           text content.  If there is a 'name' attribute it will be used
133    *           when adding to the NamedList
134    * @param nlst A NamedList to add the item to with name if application.
135    *             If this param is null it will be ignored.
136    * @param arr A List to add the item to.
137    *             If this param is null it will be ignored.
138    */
139   @SuppressWarnings("unchecked")
140   public static void addToNamedList(Node nd, NamedList nlst, List arr) {
141     // Nodes often include whitespace, etc... so just return if this
142     // is not an Element.
143     if (nd.getNodeType() != Node.ELEMENT_NODE) return;
144 
145     final String type = nd.getNodeName();
146 
147     final String name = getAttr(nd, NAME);
148 
149     Object val=null;
150 
151     if ("lst".equals(type)) {
152       val = childNodesToNamedList(nd);
153     } else if ("arr".equals(type)) {
154       val = childNodesToList(nd);
155     } else {
156       final String textValue = getText(nd);
157       try {
158         if ("str".equals(type)) {
159           val = textValue;
160         } else if ("int".equals(type)) {
161           val = Integer.valueOf(textValue);
162         } else if ("long".equals(type)) {
163           val = Long.valueOf(textValue);
164         } else if ("float".equals(type)) {
165           val = Float.valueOf(textValue);
166         } else if ("double".equals(type)) {
167           val = Double.valueOf(textValue);
168         } else if ("bool".equals(type)) {
169           val = StrUtils.parseBool(textValue);
170         }
171         // :NOTE: Unexpected Node names are ignored
172         // :TODO: should we generate an error here?
173       } catch (NumberFormatException nfe) {
174         throw new SolrException
175           (SolrException.ErrorCode.SERVER_ERROR,
176            "Value " + (null != name ? ("of '" +name+ "' ") : "") +
177            "can not be parsed as '" +type+ "': \"" + textValue + "\"",
178            nfe);
179       }
180     }
181 
182     if (nlst != null) nlst.add(name,val);
183     if (arr != null) arr.add(val);
184   }

186   /**
187    * Drop in replacement for Node.getTextContent().
188    *
189    * <p>
190    * This method is provided to support the same functionality as
191    * Node.getTextContent() but in a way that is DOM Level 2 compatible.
192    * </p>
193    *
194    * @see <a href="http://www.w3.org/TR/DOM-Level-3-Core/core.html#Node3-textContent">DOM Object Model Core</a>
195    */
196   public static String getText(Node nd) {
197 
198     short type = nd.getNodeType();
199 
200     // for most node types, we can defer to the recursive helper method,
201     // but when asked for the text of these types, we must return null
202     // (Not the empty string)
203     switch (type) {
204 
205     case Node.DOCUMENT_NODE: /* fall through */
206     case Node.DOCUMENT_TYPE_NODE: /* fall through */
207     case Node.NOTATION_NODE: /* fall through */
208       return null;
209     }
210 
211     StringBuilder sb = new StringBuilder();
212     getText(nd, sb);
213     return sb.toString();
214   }

216   /** @see #getText(Node) */
217   private static void getText(Node nd, StringBuilder buf) {
218 
219     short type = nd.getNodeType();
220 
221     switch (type) {
222 
223     case Node.ELEMENT_NODE: /* fall through */
224     case Node.ENTITY_NODE: /* fall through */
225     case Node.ENTITY_REFERENCE_NODE: /* fall through */
226     case Node.DOCUMENT_FRAGMENT_NODE:
227       NodeList childs = nd.getChildNodes();
228       for (int i = 0; i < childs.getLength(); i++) {
229         Node child = childs.item(i);
230         short childType = child.getNodeType();
231         if (childType != Node.COMMENT_NODE &&
232             childType != Node.PROCESSING_INSTRUCTION_NODE) {
233           getText(child, buf);
234         }
235       }
236       break;
237 
238     case Node.ATTRIBUTE_NODE: /* fall through */
239       /* Putting Attribute nodes in this section does not exactly
240          match the definition of how textContent should behave
241          according to the DOM Level-3 Core documentation - which
242          specifies that the Attr's children should have their
243          textContent concated (Attr's can have a single child which
244          is either Text node or an EntityReference).  In practice,
245          DOM implementations do not seem to use child nodes of
246          Attributes, storing the "text" directly as the nodeValue.
247          Fortunately, the DOM Spec indicates that when Attr.nodeValue
248          is read, it should return the nodeValue from the child Node,
249          so this approach should work both for strict implementations,
250          and implementations actually encountered.
251       */
252     case Node.TEXT_NODE: /* fall through */
253     case Node.CDATA_SECTION_NODE: /* fall through */
254     case Node.COMMENT_NODE: /* fall through */
255     case Node.PROCESSING_INSTRUCTION_NODE: /* fall through */
256       buf.append(nd.getNodeValue());
257       break;
258 
259     case Node.DOCUMENT_NODE: /* fall through */
260     case Node.DOCUMENT_TYPE_NODE: /* fall through */
261     case Node.NOTATION_NODE: /* fall through */
262     default:
263       /* :NOOP: */
264 
265     }
266   }

269    * Replaces ${system.property[:default value]} references in all attributes
270    * and text nodes of supplied node.  If the system property is not defined and no
271    * default value is provided, a runtime exception is thrown.
272    *
273    * @param node DOM node to walk for substitutions
274    */
275   public static void substituteSystemProperties(Node node) {
276     substituteProperties(node, null);
277   }

279   /**
280    * Replaces ${property[:default value]} references in all attributes
281    * and text nodes of supplied node.  If the property is not defined neither in the
282    * given Properties instance nor in System.getProperty and no
283    * default value is provided, a runtime exception is thrown.
284    *
285    * @param node DOM node to walk for substitutions
286    * @param properties the Properties instance from which a value can be looked up
287    */
288   public static void substituteProperties(Node node, Properties properties) {
289     // loop through child nodes
290     Node child;
291     Node next = node.getFirstChild();
292     while ((child = next) != null) {
293 
294       // set next before we change anything
295       next = child.getNextSibling();
296 
297       // handle child by node type
298       if (child.getNodeType() == Node.TEXT_NODE) {
299         child.setNodeValue(PropertiesUtil.substituteProperty(child.getNodeValue(), properties));
300       } else if (child.getNodeType() == Node.ELEMENT_NODE) {
301         // handle child elements with recursive call
302         NamedNodeMap attributes = child.getAttributes();
303         for (int i = 0; i < attributes.getLength(); i++) {
304           Node attribute = attributes.item(i);
305           attribute.setNodeValue(PropertiesUtil.substituteProperty(attribute.getNodeValue(), properties));
306         }
307         substituteProperties(child, properties);
308       }
309     }
310   }

312   public static String substituteProperty(String value, Properties coreProperties) {
313     if (value == null || value.indexOf('$') == -1) {
314       return value;
315     }
316 
317     List<String> fragments = new ArrayList<>();
318     List<String> propertyRefs = new ArrayList<>();
319     parsePropertyString(value, fragments, propertyRefs);
320 
321     StringBuilder sb = new StringBuilder();
322     Iterator<String> i = fragments.iterator();
323     Iterator<String> j = propertyRefs.iterator();
324 
325     while (i.hasNext()) {
326       String fragment = i.next();
327       if (fragment == null) {
328         String propertyName = j.next();
329         String defaultValue = null;
330         int colon_index = propertyName.indexOf(':');
331         if (colon_index > -1) {
332           defaultValue = propertyName.substring(colon_index + 1);
333           propertyName = propertyName.substring(0,colon_index);
334         }
335         if (coreProperties != null) {
336           fragment = coreProperties.getProperty(propertyName);
337         }
338         if (fragment == null) {
339           fragment = System.getProperty(propertyName, defaultValue);
340         }
341         if (fragment == null) {
342           throw new SolrException( SolrException.ErrorCode.SERVER_ERROR, "No system property or default value specified for " + propertyName + " value:" + value);
343         }
344       }
345       sb.append(fragment);
346     }
347     return sb.toString();
348   }

350   /*
351    * This method borrowed from Ant's PropertyHelper.parsePropertyStringDefault:
352    *   http://svn.apache.org/repos/asf/ant/core/trunk/src/main/org/apache/tools/ant/PropertyHelper.java
353    */
354   private static void parsePropertyString(String value, List<String> fragments, List<String> propertyRefs) {
355       int prev = 0;
356       int pos;
357       //search for the next instance of $ from the 'prev' position
358       while ((pos = value.indexOf("$", prev)) >= 0) {
359 
360           //if there was any text before this, add it as a fragment
361           //TODO, this check could be modified to go if pos>prev;
362           //seems like this current version could stick empty strings
363           //into the list
364           if (pos > 0) {
365               fragments.add(value.substring(prev, pos));
366           }
367           //if we are at the end of the string, we tack on a $
368           //then move past it
369           if (pos == (value.length() - 1)) {
370               fragments.add("$");
371               prev = pos + 1;
372           } else if (value.charAt(pos + 1) != '{') {
373               //peek ahead to see if the next char is a property or not
374               //not a property: insert the char as a literal
375               /*
376               fragments.addElement(value.substring(pos + 1, pos + 2));
377               prev = pos + 2;
378               */
379               if (value.charAt(pos + 1) == '$') {
380                   //backwards compatibility two $ map to one mode
381                   fragments.add("$");
382                   prev = pos + 2;
383               } else {
384                   //new behaviour: $X maps to $X for all values of X!='$'
385                   fragments.add(value.substring(pos, pos + 2));
386                   prev = pos + 2;
387               }
388 
389           } else {
390               //property found, extract its name or bail on a typo
391               int endName = value.indexOf('}', pos);
392               if (endName < 0) {
393                 throw new RuntimeException("Syntax error in property: " + value);
394               }
395               String propertyName = value.substring(pos + 2, endName);
396               fragments.add(null);
397               propertyRefs.add(propertyName);
398               prev = endName + 1;
399           }
400       }
401       //no more $ signs found
402       //if there is any tail to the string, append it
403       if (prev < value.length()) {
404           fragments.add(value.substring(prev));
405       }
406   }
