41    public void testPercentageAutowarm() throws IOException {
42      FastLRUCache<Object, Object> fastCache = new FastLRUCache<>();
43      Map<String, String> params = new HashMap<>();
44      params.put("size", "100");
45      params.put("initialSize", "10");
46      params.put("autowarmCount", "100%");
47      CacheRegenerator cr = new NoOpRegenerator();
48      Object o = fastCache.init(params, null, cr);
49      fastCache.setState(SolrCache.State.LIVE);
50      for (int i = 0; i < 101; i++) {
51        fastCache.put(i + 1, "" + (i + 1));
52      }
53      assertEquals("25", fastCache.get(25));
54      assertEquals(null, fastCache.get(110));
55      NamedList<Serializable> nl = fastCache.getStatistics();
56      assertEquals(2L, nl.get("lookups"));
57      assertEquals(1L, nl.get("hits"));
58      assertEquals(101L, nl.get("inserts"));
59      assertEquals(null, fastCache.get(1));  // first item put in should be the first out
60      FastLRUCache<Object, Object> fastCacheNew = new FastLRUCache<>();
61      fastCacheNew.init(params, o, cr);
62      fastCacheNew.warm(null, fastCache);
63      fastCacheNew.setState(SolrCache.State.LIVE);
64      fastCache.close();
65      fastCacheNew.put(103, "103");
66      assertEquals("90", fastCacheNew.get(90));
67      assertEquals("50", fastCacheNew.get(50));
68      nl = fastCacheNew.getStatistics();
69      assertEquals(2L, nl.get("lookups"));
70      assertEquals(2L, nl.get("hits"));
71      assertEquals(1L, nl.get("inserts"));
72      assertEquals(0L, nl.get("evictions"));
73      assertEquals(5L, nl.get("cumulative_lookups"));
74      assertEquals(3L, nl.get("cumulative_hits"));
75      assertEquals(102L, nl.get("cumulative_inserts"));
76      fastCacheNew.close();
77    }

79    public void testPercentageAutowarmMultiple() throws IOException {
80      doTestPercentageAutowarm(100, 50, new int[]{51, 55, 60, 70, 80, 99, 100}, new int[]{1, 2, 3, 5, 10, 20, 30, 40, 50});
81      doTestPercentageAutowarm(100, 25, new int[]{76, 80, 99, 100}, new int[]{1, 2, 3, 5, 10, 20, 30, 40, 50, 51, 55, 60, 70});
82      doTestPercentageAutowarm(1000, 10, new int[]{901, 930, 950, 999, 1000}, new int[]{1, 5, 100, 200, 300, 400, 800, 899, 900});
83      doTestPercentageAutowarm(100, 200, new int[]{1, 10, 25, 51, 55, 60, 70, 80, 99, 100}, new int[]{200, 300});
84      doTestPercentageAutowarm(100, 0, new int[]{}, new int[]{1, 10, 25, 51, 55, 60, 70, 80, 99, 100, 200, 300});
85    }

87    private void doTestPercentageAutowarm(int limit, int percentage, int[] hits, int[]misses) {
88      FastLRUCache<Object, Object> fastCache = new FastLRUCache<>();
89      Map<String, String> params = new HashMap<>();
90      params.put("size", String.valueOf(limit));
91      params.put("initialSize", "10");
92      params.put("autowarmCount", percentage + "%");
93      CacheRegenerator cr = new NoOpRegenerator();
94      Object o = fastCache.init(params, null, cr);
95      fastCache.setState(SolrCache.State.LIVE);
96      for (int i = 1; i <= limit; i++) {
97        fastCache.put(i, "" + i);//adds numbers from 1 to 100
98      }
99  
100     FastLRUCache<Object, Object> fastCacheNew = new FastLRUCache<>();
101     fastCacheNew.init(params, o, cr);
102     fastCacheNew.warm(null, fastCache);
103     fastCacheNew.setState(SolrCache.State.LIVE);
104     fastCache.close();
105       
106     for(int hit:hits) {
107       assertEquals("The value " + hit + " should be on new cache", String.valueOf(hit), fastCacheNew.get(hit));
108     }
109       
110     for(int miss:misses) {
111       assertEquals("The value " + miss + " should NOT be on new cache", null, fastCacheNew.get(miss));
112     }
113     NamedList<Serializable> nl = fastCacheNew.getStatistics();
114     assertEquals(Long.valueOf(hits.length + misses.length), nl.get("lookups"));
115     assertEquals(Long.valueOf(hits.length), nl.get("hits"));
116     fastCacheNew.close();
117   }

221     ConcurrentLRUCache<Integer, String> cache = new ConcurrentLRUCache<>(100, 90);
222     for (int i = 0; i < 50; i++) {
223       cache.put(i + 1, "" + (i + 1));
224     }
225     cache.get(1);
226     cache.get(3);
227     Map<Integer, String> m = cache.getOldestAccessedItems(5);

230     assertNotNull(m.get(6));
231     assertNotNull(m.get(5));
232     assertNotNull(m.get(4));
233     assertNotNull(m.get(2));
234

237 
238     //test this too
239     m = cache.getLatestAccessedItems(0);
240     assertTrue(m.isEmpty());
241 
242     cache.destroy();
243   }

257   void doPerfTest(int iter, int cacheSize, int maxKey) {
258     final RTimer timer = new RTimer();
259 
260     int lowerWaterMark = cacheSize;
261     int upperWaterMark = (int)(lowerWaterMark * 1.1);
262 
263     Random r = random();
264     ConcurrentLRUCache cache = new ConcurrentLRUCache(upperWaterMark, lowerWaterMark, (upperWaterMark+lowerWaterMark)/2, upperWaterMark, false, false, null);
265     boolean getSize=false;
266     int minSize=0,maxSize=0;
267     for (int i=0; i<iter; i++) {
268       cache.put(r.nextInt(maxKey),"TheValue");
269       int sz = cache.size();
270       if (!getSize && sz >= cacheSize) {
271         getSize = true;
272         minSize = sz;
273       } else {
274         if (sz < minSize) minSize=sz;
275         else if (sz > maxSize) maxSize=sz;
276       }
277     }
278     cache.destroy();
279 
280     System.out.println("time=" + timer.getTime() + ", minSize="+minSize+",maxSize="+maxSize);
281   }

293   // returns number of puts
294   int useCache(SolrCache sc, int numGets, int maxKey, int seed) {
295     int ret = 0;
296     Random r = new Random(seed);
297 
298     // use like a cache... gets and a put if not found
299     for (int i=0; i<numGets; i++) {
300       Integer k = r.nextInt(maxKey);
301       Integer v = (Integer)sc.get(k);
302       if (v == null) {
303         sc.put(k, k);
304         ret++;
305       }
306     }
307 
308     return ret;
309   }

311   void fillCache(SolrCache sc, int cacheSize, int maxKey) {
312     for (int i=0; i<cacheSize; i++) {
313       Integer kv = random().nextInt(maxKey);
314       sc.put(kv,kv);
315     }
316   }

319   void cachePerfTest(final SolrCache sc, final int nThreads, final int numGets, int cacheSize, final int maxKey) {
320     Map l = new HashMap();
321     l.put("size", ""+cacheSize);
322     l.put("initialSize", ""+cacheSize);
323 
324     Object o = sc.init(l, null, null);
325     sc.setState(SolrCache.State.LIVE);
326 
327     fillCache(sc, cacheSize, maxKey);
328 
329     final RTimer timer = new RTimer();
330 
331     Thread[] threads = new Thread[nThreads];
332     final AtomicInteger puts = new AtomicInteger(0);
333     for (int i=0; i<threads.length; i++) {
334       final int seed=random().nextInt();
335       threads[i] = new Thread() {
336           @Override
337           public void run() {
338             int ret = useCache(sc, numGets/nThreads, maxKey, seed);
339             puts.addAndGet(ret);
340           }
341         };
342     }
343 
344     for (Thread thread : threads) {
345       try {
346         thread.start();
347       } catch (Exception e) {
348         e.printStackTrace();
349       }
350     }
351 
352     for (Thread thread : threads) {
353       try {
354         thread.join();
355       } catch (Exception e) {
356         e.printStackTrace();
357       }
358     }
359 
360     System.out.println("time=" + timer.getTime() + " impl=" +sc.getClass().getSimpleName()
361                        +" nThreads= " + nThreads + " size="+cacheSize+" maxKey="+maxKey+" gets="+numGets
362                        +" hitRatio="+(1-(((double)puts.get())/numGets)));
363   }
