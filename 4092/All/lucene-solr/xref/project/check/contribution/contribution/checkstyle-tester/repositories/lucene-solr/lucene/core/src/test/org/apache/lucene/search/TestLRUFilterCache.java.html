85    public void testConcurrency() throws Throwable {
86      final LRUFilterCache filterCache = new LRUFilterCache(1 + random().nextInt(20), 1 + random().nextInt(10000));
87      Directory dir = newDirectory();
88      final RandomIndexWriter w = new RandomIndexWriter(random(), dir);
89      final SearcherManager mgr = new SearcherManager(w.w, random().nextBoolean(), new SearcherFactory() {
90        @Override
91        public IndexSearcher newSearcher(IndexReader reader, IndexReader previousReader) throws IOException {
92          IndexSearcher searcher = new IndexSearcher(reader);
93          // disable built-in caching
94          searcher.setQueryCache(null);
95          return searcher;
96        }
97      });
98      final AtomicBoolean indexing = new AtomicBoolean(true);
99      final AtomicReference<Throwable> error = new AtomicReference<>();
100     final int numDocs = atLeast(10000);
101     Thread[] threads = new Thread[3];
102     threads[0] = new Thread() {
103       public void run() {
104         Document doc = new Document();
105         StringField f = new StringField("color", "", Store.NO);
106         doc.add(f);
107         for (int i = 0; indexing.get() && i < numDocs; ++i) {
108           f.setStringValue(RandomPicks.randomFrom(random(), new String[] {"blue", "red", "yellow"}));
109           try {
110             w.addDocument(doc);
111             if ((i & 63) == 0) {
112               mgr.maybeRefresh();
113               if (rarely()) {
114                 filterCache.clear();
115               }
116               if (rarely()) {
117                 final String color = RandomPicks.randomFrom(random(), new String[] {"blue", "red", "yellow"});
118                 w.deleteDocuments(new Term("color", color));
119               }
120             }
121           } catch (Throwable t) {
122             error.compareAndSet(null, t);
123             break;
124           }
125         }
126         indexing.set(false);
127       }
128     };
129     for (int i = 1; i < threads.length; ++i) {
130       threads[i] = new Thread() {
131         @Override
132         public void run() {
133           while (indexing.get()) {
134             try {
135               final IndexSearcher searcher = mgr.acquire();
136               try {
137                 final String value = RandomPicks.randomFrom(random(), new String[] {"blue", "red", "yellow", "green"});
138                 final Filter f = new QueryWrapperFilter(new TermQuery(new Term("color", value)));
139                 final Filter cached = filterCache.doCache(f, MAYBE_CACHE_POLICY);
140                 TotalHitCountCollector collector = new TotalHitCountCollector();
141                 searcher.search(new ConstantScoreQuery(cached), collector);
142                 TotalHitCountCollector collector2 = new TotalHitCountCollector();
143                 searcher.search(new ConstantScoreQuery(f), collector2);
144                 assertEquals(collector.getTotalHits(), collector2.getTotalHits());
145               } finally {
146                 mgr.release(searcher);
147               }
148             } catch (Throwable t) {
149               error.compareAndSet(null, t);
150             }
151           }
152         }
153       };
154     }
155 
156     for (Thread thread : threads) {
157       thread.start();
158     }
159 
160     for (Thread thread : threads) {
161       thread.join();
162     }
163 
164     if (error.get() != null) {
165       throw error.get();
166     }
167     filterCache.assertConsistent();
168     mgr.close();
169     w.close();
170     dir.close();
171     filterCache.assertConsistent();
172   }

224     dir.close();

295   // This test makes sure that by making the same assumptions as LRUFilterCache, RAMUsageTester
296   // computes the same memory usage.
297   public void testRamBytesUsedAgreesWithRamUsageTester() throws IOException {
298     final LRUFilterCache filterCache = new LRUFilterCache(1 + random().nextInt(5), 1 + random().nextInt(10000));
299     // an accumulator that only sums up memory usage of referenced filters and doc id sets
300     final RamUsageTester.Accumulator acc = new RamUsageTester.Accumulator() {
301       @Override
302       public long accumulateObject(Object o, long shallowSize, Map<Field,Object> fieldValues, Collection<Object> queue) {
303         if (o instanceof DocIdSet) {
304           return ((DocIdSet) o).ramBytesUsed();
305         }
306         if (o instanceof Filter) {
307           return filterCache.ramBytesUsed((Filter) o);
308         }
309         if (o.getClass().getSimpleName().equals("SegmentCoreReaders")) {
310           // do not take core cache keys into account
311           return 0;
312         }
313         if (o instanceof Map) {
314           Map<?,?> map = (Map<?,?>) o;
315           queue.addAll(map.keySet());
316           queue.addAll(map.values());
317           final long sizePerEntry = o instanceof LinkedHashMap
318               ? LRUFilterCache.LINKED_HASHTABLE_RAM_BYTES_PER_ENTRY
319               : LRUFilterCache.HASHTABLE_RAM_BYTES_PER_ENTRY;
320           return sizePerEntry * map.size();
321         }
322         // follow links to other objects, but ignore their memory usage
323         super.accumulateObject(o, shallowSize, fieldValues, queue);
324         return  0;
325       }
326       @Override
327       public long accumulateArray(Object array, long shallowSize, List<Object> values, Collection<Object> queue) {
328         // follow links to other objects, but ignore their memory usage
329         super.accumulateArray(array, shallowSize, values, queue);
330         return 0;
331       }
332     };
333 
334     Directory dir = newDirectory();
335     // serial merges so that segments do not get closed while we are measuring ram usage
336     // with RamUsageTester
337     IndexWriterConfig iwc = newIndexWriterConfig().setMergeScheduler(new SerialMergeScheduler());
338     final RandomIndexWriter w = new RandomIndexWriter(random(), dir, iwc);
339 
340     final List<String> colors = Arrays.asList("blue", "red", "green", "yellow");
341 
342     Document doc = new Document();
343     StringField f = new StringField("color", "", Store.NO);
344     doc.add(f);
345     final int iters = atLeast(5);
346     for (int iter = 0; iter < iters; ++iter) {
347       final int numDocs = atLeast(10);
348       for (int i = 0; i < numDocs; ++i) {
349         f.setStringValue(RandomPicks.randomFrom(random(), colors));
350         w.addDocument(doc);
351       }
352       try (final DirectoryReader reader = w.getReader()) {
353         final IndexSearcher searcher = new IndexSearcher(reader);
354         searcher.setQueryCache(null);
355         for (int i = 0; i < 3; ++i) {
356           final Filter filter = new QueryWrapperFilter(new TermQuery(new Term("color", RandomPicks.randomFrom(random(), colors))));
357           searcher.search(new ConstantScoreQuery(filterCache.doCache(filter, MAYBE_CACHE_POLICY)), 1);
358         }
359       }
360       filterCache.assertConsistent();
361       assertEquals(RamUsageTester.sizeOf(filterCache, acc), filterCache.ramBytesUsed());
362     }
363 
364     w.close();
365     dir.close();
366   }

398   // that require very little memory. In that case most of the memory is taken
399   // by the cache itself, not cache entries, and we want to make sure that
400   // memory usage is not grossly underestimated.

402     final LRUFilterCache filterCache = new LRUFilterCache(1000000, 10000000);
403 
404     final RamUsageTester.Accumulator acc = new RamUsageTester.Accumulator() {
405       @Override
406       public long accumulateObject(Object o, long shallowSize, Map<Field,Object> fieldValues, Collection<Object> queue) {

412         }

414           // do not follow references to core cache keys
415           return 0;
416         }
417         return super.accumulateObject(o, shallowSize, fieldValues, queue);
418       }
419     };
420 
421     Directory dir = newDirectory();
422     final RandomIndexWriter w = new RandomIndexWriter(random(), dir);
423     Document doc = new Document();
424     final int numDocs = atLeast(100);

449   public void testOnUse() throws IOException {
450     final LRUFilterCache filterCache = new LRUFilterCache(1 + random().nextInt(5), 1 + random().nextInt(1000));
451 
452     Directory dir = newDirectory();
453     final RandomIndexWriter w = new RandomIndexWriter(random(), dir);
454 
455     Document doc = new Document();
456     StringField f = new StringField("color", "", Store.NO);
457     doc.add(f);
458     final int numDocs = atLeast(10);
459     for (int i = 0; i < numDocs; ++i) {
460       f.setStringValue(RandomPicks.randomFrom(random(), Arrays.asList("red", "blue", "green", "yellow")));
461       w.addDocument(doc);

463         w.getReader().close();

467     final IndexSearcher searcher = new IndexSearcher(reader);

471     final Map<Filter, Integer> expectedCounts = new HashMap<>();
472

474 
475       @Override
476       public boolean shouldCache(Filter filter, LeafReaderContext context, DocIdSet set) throws IOException {
477         return random().nextBoolean();
478       }
479 
480       @Override
481       public void onUse(Filter filter) {

483       }

490       cachedFilters[i] = filterCache.doCache(filters[i], countingPolicy);
491     }
492 
493     for (int i = 0; i < 20; ++i) {
494       final int idx = random().nextInt(filters.length);
495       searcher.search(new ConstantScoreQuery(cachedFilters[idx]), 1);
496       actualCounts.put(filters[idx], 1 + (actualCounts.containsKey(filters[idx]) ? actualCounts.get(filters[idx]) : 0));
497     }
498 
499     assertEquals(actualCounts, expectedCounts);
500 
501     reader.close();
502     w.close();
503     dir.close();

506   public void testStats() throws IOException {
507     final LRUFilterCache filterCache = new LRUFilterCache(1, 10000000);
508 
509     Directory dir = newDirectory();
510     final RandomIndexWriter w = new RandomIndexWriter(random(), dir);
511 
512     final List<String> colors = Arrays.asList("blue", "red", "green", "yellow");
513 
514     Document doc = new Document();
515     StringField f = new StringField("color", "", Store.NO);
516     doc.add(f);
517     for (int i = 0; i < 10; ++i) {
518       f.setStringValue(RandomPicks.randomFrom(random(), colors));
519       w.addDocument(doc);
520       if (random().nextBoolean()) {
521         w.getReader().close();
522       }
523     }
524 
525     final DirectoryReader reader = w.getReader();
526     final int segmentCount = reader.leaves().size();
527     final IndexSearcher searcher = new IndexSearcher(reader);
528     searcher.setQueryCache(null);
529     final Filter filter = new QueryWrapperFilter(new TermQuery(new Term("color", "red")));
530     final Filter filter2 = new QueryWrapperFilter(new TermQuery(new Term("color", "blue")));
531 
532     // first pass, lookups without caching that all miss
533     Filter cached = filterCache.doCache(filter, NEVER_CACHE);
534     for (int i = 0; i < 10; ++i) {
535       searcher.search(new ConstantScoreQuery(cached), 1);
536     }
537     assertEquals(10 * segmentCount, filterCache.getTotalCount());
538     assertEquals(0, filterCache.getHitCount());
539     assertEquals(10 * segmentCount, filterCache.getMissCount());
540     assertEquals(0, filterCache.getCacheCount());
541     assertEquals(0, filterCache.getEvictionCount());
542     assertEquals(0, filterCache.getCacheSize());
543 
544     // second pass, lookups + caching, only the first one is a miss
545     cached = filterCache.doCache(filter, FilterCachingPolicy.ALWAYS_CACHE);
546     for (int i = 0; i < 10; ++i) {
547       searcher.search(new ConstantScoreQuery(cached), 1);
548     }
549     assertEquals(20 * segmentCount, filterCache.getTotalCount());
550     assertEquals(9 * segmentCount, filterCache.getHitCount());
551     assertEquals(11 * segmentCount, filterCache.getMissCount());
552     assertEquals(1 * segmentCount, filterCache.getCacheCount());
553     assertEquals(0, filterCache.getEvictionCount());
554     assertEquals(1 * segmentCount, filterCache.getCacheSize());
555 
556     // third pass lookups without caching, we only have hits
557     cached = filterCache.doCache(filter, NEVER_CACHE);
558     for (int i = 0; i < 10; ++i) {
559       searcher.search(new ConstantScoreQuery(cached), 1);
560     }
561     assertEquals(30 * segmentCount, filterCache.getTotalCount());
562     assertEquals(19 * segmentCount, filterCache.getHitCount());
563     assertEquals(11 * segmentCount, filterCache.getMissCount());
564     assertEquals(1 * segmentCount, filterCache.getCacheCount());
565     assertEquals(0, filterCache.getEvictionCount());
566     assertEquals(1 * segmentCount, filterCache.getCacheSize());
567 
568     // fourth pass with a different filter which will trigger evictions since the size is 1
569     cached = filterCache.doCache(filter2, FilterCachingPolicy.ALWAYS_CACHE);
570     for (int i = 0; i < 10; ++i) {
571       searcher.search(new ConstantScoreQuery(cached), 1);
572     }
573     assertEquals(40 * segmentCount, filterCache.getTotalCount());
574     assertEquals(28 * segmentCount, filterCache.getHitCount());
575     assertEquals(12 * segmentCount, filterCache.getMissCount());
576     assertEquals(2 * segmentCount, filterCache.getCacheCount());
577     assertEquals(1 * segmentCount, filterCache.getEvictionCount());
578     assertEquals(1 * segmentCount, filterCache.getCacheSize());
579 
580     // now close, causing evictions due to the closing of segment cores
581     reader.close();
582     w.close();
583     assertEquals(40 * segmentCount, filterCache.getTotalCount());
584     assertEquals(28 * segmentCount, filterCache.getHitCount());
585     assertEquals(12 * segmentCount, filterCache.getMissCount());
586     assertEquals(2 * segmentCount, filterCache.getCacheCount());
587     assertEquals(2 * segmentCount, filterCache.getEvictionCount());
588     assertEquals(0, filterCache.getCacheSize());
589 
590     dir.close();
591   }

612     }

614     final DirectoryReader reader1 = w1.getReader();
615     final int segmentCount1 = reader1.leaves().size();
616     final IndexSearcher searcher1 = new IndexSearcher(reader1);
617     searcher1.setQueryCache(null);
618 
619     final DirectoryReader reader2 = w2.getReader();
620     final int segmentCount2 = reader2.leaves().size();
621     final IndexSearcher searcher2 = new IndexSearcher(reader2);
622     searcher2.setQueryCache(null);
623 
624     final Map<Object, Integer> indexId = new HashMap<>();
625     for (LeafReaderContext ctx : reader1.leaves()) {
626       indexId.put(ctx.reader().getCoreCacheKey(), 1);
627     }
628     for (LeafReaderContext ctx : reader2.leaves()) {
629       indexId.put(ctx.reader().getCoreCacheKey(), 2);
630     }
631 
632     final AtomicLong hitCount1 = new AtomicLong();
633     final AtomicLong hitCount2 = new AtomicLong();
634     final AtomicLong missCount1 = new AtomicLong();
635     final AtomicLong missCount2 = new AtomicLong();
636 
637     final AtomicLong ramBytesUsage = new AtomicLong();
638     final AtomicLong cacheSize = new AtomicLong();
639 
640     final LRUFilterCache filterCache = new LRUFilterCache(2, 10000000) {
641       @Override
642       protected void onHit(Object readerCoreKey, Filter filter) {
643         super.onHit(readerCoreKey, filter);
644         switch(indexId.get(readerCoreKey).intValue()) {
645           case 1:
646             hitCount1.incrementAndGet();
647             break;
648           case 2:
649             hitCount2.incrementAndGet();
650             break;
651           default:
652             throw new AssertionError();
653         }
654       }
655 
656       @Override
657       protected void onMiss(Object readerCoreKey, Filter filter) {
658         super.onMiss(readerCoreKey, filter);
659         switch(indexId.get(readerCoreKey).intValue()) {
660           case 1:
661             missCount1.incrementAndGet();
662             break;
663           case 2:
664             missCount2.incrementAndGet();
665             break;
666           default:
667             throw new AssertionError();
668         }
669       }
670 
671       @Override
672       protected void onFilterCache(Filter filter, long ramBytesUsed) {
673         super.onFilterCache(filter, ramBytesUsed);
674         ramBytesUsage.addAndGet(ramBytesUsed);
675       }
676 
677       @Override
678       protected void onFilterEviction(Filter filter, long ramBytesUsed) {
679         super.onFilterEviction(filter, ramBytesUsed);
680         ramBytesUsage.addAndGet(-ramBytesUsed);
681       }
682 
683       @Override
684       protected void onDocIdSetCache(Object readerCoreKey, long ramBytesUsed) {
685         super.onDocIdSetCache(readerCoreKey, ramBytesUsed);
686         ramBytesUsage.addAndGet(ramBytesUsed);
687         cacheSize.incrementAndGet();
688       }
689 
690       @Override
691       protected void onDocIdSetEviction(Object readerCoreKey, int numEntries, long sumRamBytesUsed) {
692         super.onDocIdSetEviction(readerCoreKey, numEntries, sumRamBytesUsed);
693         ramBytesUsage.addAndGet(-sumRamBytesUsed);
694         cacheSize.addAndGet(-numEntries);
695       }
696 
697       @Override
698       protected void onClear() {
699         super.onClear();
700         ramBytesUsage.set(0);
701         cacheSize.set(0);
702       }
703     };

705     final Filter filter = new QueryWrapperFilter(new TermQuery(new Term("color", "red")));
706     final Filter filter2 = new QueryWrapperFilter(new TermQuery(new Term("color", "blue")));
707     final Filter filter3 = new QueryWrapperFilter(new TermQuery(new Term("color", "green")));
708 
709     // search on searcher1
710     Filter cached = filterCache.doCache(filter, FilterCachingPolicy.ALWAYS_CACHE);
711     for (int i = 0; i < 10; ++i) {
712       searcher1.search(new ConstantScoreQuery(cached), 1);
713     }
714     assertEquals(9 * segmentCount1, hitCount1.longValue());
715     assertEquals(0, hitCount2.longValue());
716     assertEquals(segmentCount1, missCount1.longValue());
717     assertEquals(0, missCount2.longValue());
718 
719     // then on searcher2
720     cached = filterCache.doCache(filter2, FilterCachingPolicy.ALWAYS_CACHE);
721     for (int i = 0; i < 20; ++i) {
722       searcher2.search(new ConstantScoreQuery(cached), 1);
723     }
724     assertEquals(9 * segmentCount1, hitCount1.longValue());
725     assertEquals(19 * segmentCount2, hitCount2.longValue());
726     assertEquals(segmentCount1, missCount1.longValue());
727     assertEquals(segmentCount2, missCount2.longValue());
728 
729     // now on searcher1 again to trigger evictions
730     cached = filterCache.doCache(filter3, FilterCachingPolicy.ALWAYS_CACHE);
731     for (int i = 0; i < 30; ++i) {
732       searcher1.search(new ConstantScoreQuery(cached), 1);
733     }
734     assertEquals(segmentCount1, filterCache.getEvictionCount());
735     assertEquals(38 * segmentCount1, hitCount1.longValue());
736     assertEquals(19 * segmentCount2, hitCount2.longValue());
737     assertEquals(2 * segmentCount1, missCount1.longValue());
738     assertEquals(segmentCount2, missCount2.longValue());
739 
740     // check that the recomputed stats are the same as those reported by the cache
741     assertEquals(filterCache.ramBytesUsed(), (segmentCount1 + segmentCount2) * LRUFilterCache.HASHTABLE_RAM_BYTES_PER_ENTRY + ramBytesUsage.longValue());
742     assertEquals(filterCache.getCacheSize(), cacheSize.longValue());
743 
744     reader1.close();
745     reader2.close();
746     w1.close();
747     w2.close();
748 
749     assertEquals(filterCache.ramBytesUsed(), ramBytesUsage.longValue());
750     assertEquals(0, cacheSize.longValue());
751 
752     filterCache.clear();
753     assertEquals(0, ramBytesUsage.longValue());
754     assertEquals(0, cacheSize.longValue());

756     dir1.close();
757     dir2.close();
