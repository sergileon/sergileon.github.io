40  // NOTE: outputs should be per-field, since
41  // longsSize is fixed for each field
42  class FSTTermOutputs extends Outputs<FSTTermOutputs.TermData> {
43    private final static TermData NO_OUTPUT = new TermData();
44    //private static boolean TEST = false;
45    private final boolean hasPos;
46    private final int longsSize;
47  
48    /** 
49     * Represents the metadata for one term.
50     * On an FST, only long[] part is 'shared' and pushed towards root.
51     * byte[] and term stats will be kept on deeper arcs.
52     */
53    static class TermData implements Accountable {
54      private static final long BASE_RAM_BYTES_USED = RamUsageEstimator.shallowSizeOfInstance(TermData.class);
55      long[] longs;
56      byte[] bytes;
57      int docFreq;
58      long totalTermFreq;
59      TermData() {
60        this.longs = null;
61        this.bytes = null;
62        this.docFreq = 0;
63        this.totalTermFreq = -1;
64      }
65      TermData(long[] longs, byte[] bytes, int docFreq, long totalTermFreq) {
66        this.longs = longs;
67        this.bytes = bytes;
68        this.docFreq = docFreq;
69        this.totalTermFreq = totalTermFreq;
70      }
71  
72      @Override
73      public long ramBytesUsed() {
74        long ramBytesUsed = BASE_RAM_BYTES_USED;
75        if (longs != null) {
76          ramBytesUsed += RamUsageEstimator.sizeOf(longs);
77        }
78        if (bytes != null) {
79          ramBytesUsed += RamUsageEstimator.sizeOf(bytes);
80        }
81        return ramBytesUsed;
82      }
83  
84      @Override
85      public Collection<Accountable> getChildResources() {
86        return Collections.emptyList();
87      }
88      
89      // NOTE: actually, FST nodes are seldom 
90      // identical when outputs on their arcs 
91      // aren't NO_OUTPUTs.
92      @Override
93      public int hashCode() {
94        int hash = 0;
95        if (longs != null) {
96          final int end = longs.length;
97          for (int i = 0; i < end; i++) {
98            hash -= longs[i];
99          }
100       }
101       if (bytes != null) {
102         hash = -hash;
103         final int end = bytes.length;
104         for (int i = 0; i < end; i++) {
105           hash += bytes[i];
106         }
107       }
108       hash += docFreq + totalTermFreq;
109       return hash;
110     }
111 
112     @Override
113     public String toString() {
114       return "FSTTermOutputs$TermData longs=" + Arrays.toString(longs) + " bytes=" + Arrays.toString(bytes) + " docFreq=" + docFreq + " totalTermFreq=" + totalTermFreq;
115     }
116 
117     @Override
118     public boolean equals(Object other_) {
119       if (other_ == this) {
120         return true;
121       } else if (!(other_ instanceof FSTTermOutputs.TermData)) {
122         return false;
123       }
124       TermData other = (TermData) other_;
125       return statsEqual(this, other) && 
126              longsEqual(this, other) && 
127              bytesEqual(this, other);
128 
129     }
130   }

132   protected FSTTermOutputs(FieldInfo fieldInfo, int longsSize) {
133     this.hasPos = fieldInfo.getIndexOptions() != IndexOptions.DOCS;
134     this.longsSize = longsSize;
135   }

137   @Override
138   public long ramBytesUsed(TermData output) {
139     return output.ramBytesUsed();
140   }

142   @Override
143   //
144   // The return value will be the smaller one, when these two are 
145   // 'comparable', i.e. 
146   // 1. every value in t1 is not larger than in t2, or
147   // 2. every value in t1 is not smaller than t2.
148   //
149   public TermData common(TermData t1, TermData t2) {
150     //if (TEST) System.out.print("common("+t1+", "+t2+") = ");
151     if (t1 == NO_OUTPUT || t2 == NO_OUTPUT) {
152       //if (TEST) System.out.println("ret:"+NO_OUTPUT);
153       return NO_OUTPUT;
154     }
155     assert t1.longs.length == t2.longs.length;
156 
157     long[] min = t1.longs, max = t2.longs;
158     int pos = 0;
159     TermData ret;
160 
161     while (pos < longsSize && min[pos] == max[pos]) {
162       pos++;
163     }
164     if (pos < longsSize) {  // unequal long[]
165       if (min[pos] > max[pos]) {
166         min = t2.longs;
167         max = t1.longs;
168       }
169       // check whether strictly smaller
170       while (pos < longsSize && min[pos] <= max[pos]) {
171         pos++;
172       }
173       if (pos < longsSize || allZero(min)) {  // not comparable or all-zero
174         ret = NO_OUTPUT;
175       } else {
176         ret = new TermData(min, null, 0, -1);
177       }
178     } else {  // equal long[]
179       if (statsEqual(t1, t2) && bytesEqual(t1, t2)) {
180         ret = t1;
181       } else if (allZero(min)) {
182         ret = NO_OUTPUT;
183       } else {
184         ret = new TermData(min, null, 0, -1);
185       }
186     }
187     //if (TEST) System.out.println("ret:"+ret);
188     return ret;
189   }
190 
191   @Override
192   public TermData subtract(TermData t1, TermData t2) {
193     //if (TEST) System.out.print("subtract("+t1+", "+t2+") = ");
194     if (t2 == NO_OUTPUT) {
195       //if (TEST) System.out.println("ret:"+t1);
196       return t1;
197     }
198     assert t1.longs.length == t2.longs.length;
199 
200     int pos = 0;
201     long diff = 0;
202     long[] share = new long[longsSize];
203 
204     while (pos < longsSize) {
205       share[pos] = t1.longs[pos] - t2.longs[pos];
206       diff += share[pos];
207       pos++;
208     }
209 
210     TermData ret;
211     if (diff == 0 && statsEqual(t1, t2) && bytesEqual(t1, t2)) {
212       ret = NO_OUTPUT;
213     } else {
214       ret = new TermData(share, t1.bytes, t1.docFreq, t1.totalTermFreq);
215     }
216     //if (TEST) System.out.println("ret:"+ret);
217     return ret;
218   }

220   // TODO: if we refactor a 'addSelf(TermData other)',
221   // we can gain about 5~7% for fuzzy queries, however this also 
222   // means we are putting too much stress on FST Outputs decoding?
223   @Override
224   public TermData add(TermData t1, TermData t2) {
225     //if (TEST) System.out.print("add("+t1+", "+t2+") = ");
226     if (t1 == NO_OUTPUT) {
227       //if (TEST) System.out.println("ret:"+t2);
228       return t2;
229     } else if (t2 == NO_OUTPUT) {
230       //if (TEST) System.out.println("ret:"+t1);
231       return t1;
232     }
233     assert t1.longs.length == t2.longs.length;
234 
235     int pos = 0;
236     long[] accum = new long[longsSize];
237 
238     while (pos < longsSize) {
239       accum[pos] = t1.longs[pos] + t2.longs[pos];
240       pos++;
241     }
242 
243     TermData ret;
244     if (t2.bytes != null || t2.docFreq > 0) {
245       ret = new TermData(accum, t2.bytes, t2.docFreq, t2.totalTermFreq);
246     } else {
247       ret = new TermData(accum, t1.bytes, t1.docFreq, t1.totalTermFreq);
248     }
249     //if (TEST) System.out.println("ret:"+ret);
250     return ret;
251   }

253   @Override
254   public void write(TermData data, DataOutput out) throws IOException {
255     assert hasPos || data.totalTermFreq == -1;
256     int bit0 = allZero(data.longs) ? 0 : 1;
257     int bit1 = ((data.bytes == null || data.bytes.length == 0) ? 0 : 1) << 1;
258     int bit2 = ((data.docFreq == 0)  ? 0 : 1) << 2;
259     int bits = bit0 | bit1 | bit2;
260     if (bit1 > 0) {  // determine extra length
261       if (data.bytes.length < 32) {
262         bits |= (data.bytes.length << 3);
263         out.writeByte((byte)bits);
264       } else {
265         out.writeByte((byte)bits);
266         out.writeVInt(data.bytes.length);
267       }
268     } else {
269       out.writeByte((byte)bits);
270     }
271     if (bit0 > 0) {  // not all-zero case
272       for (int pos = 0; pos < longsSize; pos++) {
273         out.writeVLong(data.longs[pos]);
274       }
275     }
276     if (bit1 > 0) {  // bytes exists
277       out.writeBytes(data.bytes, 0, data.bytes.length);
278     }
279     if (bit2 > 0) {  // stats exist
280       if (hasPos) {
281         if (data.docFreq == data.totalTermFreq) {
282           out.writeVInt((data.docFreq << 1) | 1);
283         } else {
284           out.writeVInt((data.docFreq << 1));
285           out.writeVLong(data.totalTermFreq - data.docFreq);
286         }
287       } else {
288         out.writeVInt(data.docFreq);
289       }
290     }
291   }

293   @Override
294   public TermData read(DataInput in) throws IOException {
295     long[] longs = new long[longsSize];
296     byte[] bytes = null;
297     int docFreq = 0;
298     long totalTermFreq = -1;
299     int bits = in.readByte() & 0xff;
300     int bit0 = bits & 1;
301     int bit1 = bits & 2;
302     int bit2 = bits & 4;
303     int bytesSize = (bits >>> 3);
304     if (bit1 > 0 && bytesSize == 0) {  // determine extra length
305       bytesSize = in.readVInt();
306     }
307     if (bit0 > 0) {  // not all-zero case
308       for (int pos = 0; pos < longsSize; pos++) {
309         longs[pos] = in.readVLong();
310       }
311     }
312     if (bit1 > 0) {  // bytes exists
313       bytes = new byte[bytesSize];
314       in.readBytes(bytes, 0, bytesSize);
315     }
316     if (bit2 > 0) {  // stats exist
317       int code = in.readVInt();
318       if (hasPos) {
319         totalTermFreq = docFreq = code >>> 1;
320         if ((code & 1) == 0) {
321           totalTermFreq += in.readVLong();
322         }
323       } else {
324         docFreq = code;
325       }
326     }
327     return new TermData(longs, bytes, docFreq, totalTermFreq);
328   }

331   @Override
332   public void skipOutput(DataInput in) throws IOException {
333     int bits = in.readByte() & 0xff;
334     int bit0 = bits & 1;
335     int bit1 = bits & 2;
336     int bit2 = bits & 4;
337     int bytesSize = (bits >>> 3);
338     if (bit1 > 0 && bytesSize == 0) {  // determine extra length
339       bytesSize = in.readVInt();
340     }
341     if (bit0 > 0) {  // not all-zero case
342       for (int pos = 0; pos < longsSize; pos++) {
343         in.readVLong();
344       }
345     }
346     if (bit1 > 0) {  // bytes exists
347       in.skipBytes(bytesSize);
348     }
349     if (bit2 > 0) {  // stats exist
350       int code = in.readVInt();
351       if (hasPos && (code & 1) == 0) {
352         in.readVLong();
353       }
354     }
355   }

367   static boolean statsEqual(final TermData t1, final TermData t2) {
368     return t1.docFreq == t2.docFreq && t1.totalTermFreq == t2.totalTermFreq;
369   }
370   static boolean bytesEqual(final TermData t1, final TermData t2) {
371     if (t1.bytes == null && t2.bytes == null) {
372       return true;
373     }
374     return t1.bytes != null && t2.bytes != null && Arrays.equals(t1.bytes, t2.bytes);
375   }
376   static boolean longsEqual(final TermData t1, final TermData t2) {
377     if (t1.longs == null && t2.longs == null) {
378       return true;
379     }
380     return t1.longs != null && t2.longs != null && Arrays.equals(t1.longs, t2.longs);
381   }
382   static boolean allZero(final long[] l) {
383     for (int i = 0; i < l.length; i++) {
384       if (l[i] != 0) {
385         return false;
386       }
387     }
388     return true;
389   }
