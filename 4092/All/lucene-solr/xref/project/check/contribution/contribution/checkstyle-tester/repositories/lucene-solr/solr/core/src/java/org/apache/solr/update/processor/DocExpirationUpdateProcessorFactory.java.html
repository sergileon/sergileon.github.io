62  /**
63   * <p>
64   * Update Processor Factory for managing automatic "expiration" of documents.
65   * </p>
66   * 
67   * <p>
68   * The <code>DocExpirationUpdateProcessorFactory</code> provides two features related 
69   * to the "expiration" of documents which can be used individually, or in combination:
70   * </p>
71   * <ol>
72   *  <li>Computing expiration field values for documents from a "time to live" (TTL)</li>
73   *  <li>Periodically delete documents from the index based on an expiration field</li>
74   * </ol>
75   * 
76   * <p>
77   * Documents with expiration field values computed from a TTL can be be excluded from 
78   * searchers using simple date based filters relative to <code>NOW</code>, or completely 
79   * removed from the index using the periodic delete function of this factory.  Alternatively, 
80   * the periodic delete function of this factory can be used to remove any document with an 
81   * expiration value - even if that expiration was explicitly set with-out leveraging the TTL 
82   * feature of this factory.
83   * </p>
84   *
85   * <p>
86   * The following configuration options are supported:
87   * </p>
88   *
89   * <ul>
90   *  <li><code>expirationFieldName</code> - The name of the expiration field to use 
91   *      in any operations (mandatory).
92   *  </li>
93   *  <li><code>ttlFieldName</code> - Name of a field this process should look 
94   *      for in each document processed, defaulting to <code>_ttl_</code>.  
95   *      If the specified field name exists in a document, the document field value 
96   *      will be parsed as a {@linkplain DateMathParser Date Math Expression} relative to 
97   *      <code>NOW</code> and the result will be added to the document using the 
98   *      <code>expirationFieldName</code>.  Use <code>&lt;null name="ttlFieldName"/&gt;</code>
99   *      to disable this feature.
100  *  </li>
101  *  <li><code>ttlParamName</code> - Name of an update request param this process should 
102  *      look for in each request when processing document additions, defaulting to  
103  *      <code>_ttl_</code>. If the the specified param name exists in an update request, 
104  *      the param value will be parsed as a {@linkplain DateMathParser Date Math Expression}
105  *      relative to <code>NOW</code> and the result will be used as a default for any 
106  *      document included in that request that does not already have a value in the 
107  *      field specified by <code>ttlFieldName</code>.  Use 
108  *      <code>&lt;null name="ttlParamName"/&gt;</code> to disable this feature.
109  *  </li>
110  *  <li><code>autoDeletePeriodSeconds</code> - Optional numeric value indicating how 
111  *      often this factory should trigger a delete to remove documents.  If this option is 
112  *      used, and specifies a non-negative numeric value, a background thread will be 
113  *      created that will execute recurring <code>deleteByQuery</code> commands using the 
114  *      specified period.  The delete query will remove all documents with an 
115  *      <code>expirationFieldName</code> up to <code>NOW</code>.
116  *  </li>
117  *  <li><code>autoDeleteChainName</code> - Optional name of an 
118  *      <code>updateRequestProcessorChain</code> to use when executing automatic deletes.  
119  *      If not specified, or <code>&lt;null/&gt;</code>, the default 
120  *      <code>updateRequestProcessorChain</code> for this collection is used.  
121  *      This option is ignored unless <code>autoDeletePeriodSeconds</code> is configured 
122  *      and is non-negative.
123  *  </li>
124  * </ul>
125  *
126  * <p>
127  * For example: The configuration below will cause any document with a field named 
128  * <code>_ttl_</code> to have a Date field named <code>_expire_at_</code> computed 
129  * for it when added -- but no automatic deletion will happen.
130  * </p>
131  *
132  * <pre class="prettyprint">
133  * &lt;processor class="solr.processor.DocExpirationUpdateProcessorFactory"&gt;
134  *   &lt;str name="expirationFieldName"&gt;_expire_at_&lt;/str&gt;
135  * &lt;/processor&gt;</pre>
136  *
137  * <p>
138  * Alternatively, in this configuration deletes will occur automatically against the 
139  * <code>_expire_at_</code> field every 5 minutes - but this processor will not 
140  * automatically populate the <code>_expire_at_</code> using any sort of TTL expression.  
141  * Only documents that were added with an explicit <code>_expire_at_</code> field value 
142  * will ever be deleted.
143  * </p>
144  *
145  * <pre class="prettyprint">
146  * &lt;processor class="solr.processor.DocExpirationUpdateProcessorFactory"&gt;
147  *   &lt;null name="ttlFieldName"/&gt;
148  *   &lt;null name="ttlParamName"/&gt;
149  *   &lt;int name="autoDeletePeriodSeconds"&gt;300&lt;/int&gt;
150  *   &lt;str name="expirationFieldName"&gt;_expire_at_&lt;/str&gt;
151  * &lt;/processor&gt;</pre>
152  *
153  * <p>
154  * This last example shows the combination of both features using a custom 
155  * <code>ttlFieldName</code>: Documents with a <code>my_ttl</code> field will 
156  * have an <code>_expire_at_</code> field computed, and deletes will be triggered 
157  * every 5 minutes to remove documents whose 
158  * <code>_expire_at_</code> field value is in the past.
159  * </p>
160  *
161  * <pre class="prettyprint">
162  * &lt;processor class="solr.processor.DocExpirationUpdateProcessorFactory"&gt;
163  *   &lt;int name="autoDeletePeriodSeconds"&gt;300&lt;/int&gt;
164  *   &lt;str name="ttlFieldName"&gt;my_ttl&lt;/str&gt;
165  *   &lt;null name="ttlParamName"/&gt;
166  *   &lt;str name="expirationFieldName"&gt;_expire_at_&lt;/str&gt;
167  * &lt;/processor&gt;</pre> 
168  */
169 public final class DocExpirationUpdateProcessorFactory 
170   extends UpdateRequestProcessorFactory 
171   implements SolrCoreAware {
172 
173   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
174 
175   private static final String DEF_TTL_KEY = "_ttl_";
176   private static final String EXP_FIELD_NAME_CONF = "expirationFieldName";
177   private static final String TTL_FIELD_NAME_CONF = "ttlFieldName";
178   private static final String TTL_PARAM_NAME_CONF = "ttlParamName";
179   private static final String DEL_CHAIN_NAME_CONF = "autoDeleteChainName";
180   private static final String DEL_PERIOD_SEC_CONF = "autoDeletePeriodSeconds";
181   
182   private SolrCore core;
183   private ScheduledThreadPoolExecutor executor;
184 
185   private String expireField = null;
186   private String ttlField = null;
187   private String ttlParam = null;
188 
189   private String deleteChainName = null;
190   private long deletePeriodSeconds = -1L;

192   private SolrException confErr(final String msg) {
193     return confErr(msg, null);
194   }
195   private SolrException confErr(final String msg, SolrException root) {
196     return new SolrException(SERVER_ERROR, this.getClass().getSimpleName()+": "+msg, root);
197   }
198   private String removeArgStr(final NamedList args, final String arg, final String def,
199                               final String errMsg) {
200 
201     if (args.indexOf(arg,0) < 0) return def;
202 
203     Object tmp = args.remove(arg);
204     if (null == tmp) return null;
205 
206     if (tmp instanceof String) return tmp.toString();
207 
208     throw confErr(arg + " " + errMsg);
209   }

211   @SuppressWarnings("unchecked")
212   @Override
213   public void init(NamedList args) {
214 
215     deleteChainName = removeArgStr(args, DEL_CHAIN_NAME_CONF, null,
216                                    "must be a <str> or <null/> for default chain");
217 
218     ttlField = removeArgStr(args, TTL_FIELD_NAME_CONF, DEF_TTL_KEY, 
219                             "must be a <str> or <null/> to disable");
220     ttlParam = removeArgStr(args, TTL_PARAM_NAME_CONF, DEF_TTL_KEY, 
221                             "must be a <str> or <null/> to disable");
222 
223     expireField = removeArgStr(args, EXP_FIELD_NAME_CONF, null, "must be a <str>");
224     if (null == expireField) {
225       throw confErr(EXP_FIELD_NAME_CONF + " must be configured");
226     }
227 
228     Object tmp = args.remove(DEL_PERIOD_SEC_CONF);
229     if (null != tmp) {
230       if (! (tmp instanceof Number)) {
231         throw confErr(DEL_PERIOD_SEC_CONF + " must be an <int> or <long>");
232       }
233       deletePeriodSeconds = ((Number)tmp).longValue();
234     }
235     
236     super.init(args);
237   }

239   @Override
240   public void inform(SolrCore core) {
241     this.core = core;
242     
243     if (null == core.getLatestSchema().getFieldTypeNoEx(expireField)) {
244       // TODO: check for managed schema and auto-add as a date field?
245       throw confErr(EXP_FIELD_NAME_CONF + " does not exist in schema: " + expireField);
246     }
247 
248     if (0 < deletePeriodSeconds) {
249       // validate that we have a chain we can work with
250       try {
251         Object ignored = core.getUpdateProcessingChain(deleteChainName);
252       } catch (SolrException e) {
253         throw confErr(DEL_CHAIN_NAME_CONF + " does not exist: " + deleteChainName, e);
254       }
255       // schedule recuring deletion
256       initDeleteExpiredDocsScheduler(core);
257     }
258   }
259 
260   private void initDeleteExpiredDocsScheduler(SolrCore core) {
261     executor = new ScheduledThreadPoolExecutor
262       (1, new DefaultSolrThreadFactory("autoExpireDocs"),
263        new RejectedExecutionHandler() {
264         public void rejectedExecution(Runnable r, ThreadPoolExecutor e) {
265           log.warn("Skipping execution of '{}' using '{}'", r, e);
266         }
267       });
268 
269     core.addCloseHook(new CloseHook() {
270       public void postClose(SolrCore core) {
271         // update handler is gone, terminate anything that's left.
272 
273         if (executor.isTerminating()) {
274           log.info("Waiting for close of DocExpiration Executor");
275           ExecutorUtil.shutdownAndAwaitTermination(executor);
276         }
277       }
278       public void preClose(SolrCore core) {
279         log.info("Triggering Graceful close of DocExpiration Executor");
280         executor.shutdown();
281       }
282     });
283 
284     executor.setExecuteExistingDelayedTasksAfterShutdownPolicy(false);
285     executor.setContinueExistingPeriodicTasksAfterShutdownPolicy(false);
286     // we don't want this firing right away, since the core may not be ready
287     final long initialDelay = deletePeriodSeconds;
288     // TODO: should we make initialDelay configurable
289     // TODO: should we make initialDelay some fraction of the period?
290     executor.scheduleAtFixedRate(new DeleteExpiredDocsRunnable(this), 
291                                  deletePeriodSeconds,
292                                  deletePeriodSeconds,
293                                  TimeUnit.SECONDS);
294 
295   }
296 
297   @Override
298   public UpdateRequestProcessor getInstance(SolrQueryRequest req, 
299                                             SolrQueryResponse rsp, 
300                                             UpdateRequestProcessor next ) {
301 
302     String defaultTtl = (null == ttlParam) ? null : req.getParams().get(ttlParam);
303 
304     if (null == ttlField && null == defaultTtl) {
305       // nothing to do, shortcircut ourselves out of the chain.
306       return next;
307     } else {
308       return new TTLUpdateProcessor(defaultTtl, expireField, ttlField, next);
309     }
310   }
311 
312   private static final class TTLUpdateProcessor extends UpdateRequestProcessor {
313 
314     final String defaultTtl;
315     final String expireField;
316     final String ttlField;
317     public TTLUpdateProcessor(final String defaultTtl,
318                               final String expireField,
319                               final String ttlField,
320                               final UpdateRequestProcessor next) {
321       super(next);
322       this.defaultTtl = defaultTtl;
323       this.expireField = expireField;
324       this.ttlField = ttlField;
325     }
326 
327     @Override
328     public void processAdd(AddUpdateCommand cmd) throws IOException {
329       final SolrInputDocument doc = cmd.getSolrInputDocument();
330 
331       final String math = doc.containsKey(ttlField) 
332         ? doc.getFieldValue(ttlField).toString() : defaultTtl;
333 
334       if (null != math) {
335         try {
336           final DateMathParser dmp = new DateMathParser();
337           // TODO: should we try to accept things like "1DAY" as well as "+1DAY" ?
338           // How? 
339           // 'startsWith("+")' is a bad idea because it would cause porblems with
340           // things like "/DAY+1YEAR"
341           // Maybe catch ParseException and rety with "+" prepended?
342           doc.addField(expireField, dmp.parseMath(math));
343         } catch (ParseException pe) {
344           throw new SolrException(BAD_REQUEST, "Can't parse ttl as date math: " + math, pe);
345         }
346       }
347 
348       super.processAdd(cmd);
349     }
350   }

352   /**
353    * <p>
354    * Runnable that uses the the <code>deleteChainName</code> configured for 
355    * this factory to execute a delete by query (using the configured 
356    * <code>expireField</code>) followed by a soft commit to re-open searchers (if needed)
357    * </p>
358    * <p>
359    * This logic is all wrapped up in a new SolrRequestInfo context with 
360    * some logging to help make it obvious this background activity is happening.
361    * </p>
362    * <p>
363    * In cloud mode, this runner only triggers deletes if 
364    * {@link #iAmInChargeOfPeriodicDeletes} is true.
365    * (logging is minimal in this situation)
366    * </p>
367    *
368    * @see #iAmInChargeOfPeriodicDeletes
369    */
370   private static final class DeleteExpiredDocsRunnable implements Runnable {
371     final DocExpirationUpdateProcessorFactory factory;
372     final SolrCore core;
373     final String deleteChainName;
374     final String expireField;
375     public DeleteExpiredDocsRunnable(final DocExpirationUpdateProcessorFactory factory) {
376       this.factory = factory;
377       this.core = factory.core; 
378       this.deleteChainName = factory.deleteChainName;
379       this.expireField = factory.expireField;
380    }
381 
382     public void run() {
383       // setup the request context early so the logging (including any from 
384       // shouldWeDoPeriodicDelete() ) includes the core context info
385       final SolrQueryRequest req = new LocalSolrQueryRequest
386         (factory.core, Collections.<String,String[]>emptyMap());
387       try {
388         final SolrQueryResponse rsp = new SolrQueryResponse();
389         SolrRequestInfo.setRequestInfo(new SolrRequestInfo(req, rsp));
390         try {
391           
392           if (! factory.iAmInChargeOfPeriodicDeletes() ) {
393             // No-Op
394             return;
395           }
396           log.info("Begining periodic deletion of expired docs");
397 
398           UpdateRequestProcessorChain chain = core.getUpdateProcessingChain(deleteChainName);
399           UpdateRequestProcessor proc = chain.createProcessor(req, rsp);
400           if (null == proc) {
401             log.warn("No active processors, skipping automatic deletion " + 
402                      "of expired docs using chain: {}", deleteChainName);
403             return;
404           }
405           try {
406             DeleteUpdateCommand del = new DeleteUpdateCommand(req);
407             del.setQuery("{!cache=false}" + expireField + ":[* TO " +
408                          DateFormatUtil.formatExternal(SolrRequestInfo.getRequestInfo().getNOW())
409                          + "]");
410             proc.processDelete(del);
411             
412             // TODO: should this be more configurable? 
413             // TODO: in particular: should hard commit be optional?
414             CommitUpdateCommand commit = new CommitUpdateCommand(req, false);
415             commit.softCommit = true;
416             commit.openSearcher = true;
417             proc.processCommit(commit);
418             
419           } finally {
420             proc.finish();
421           }
422 
423           log.info("Finished periodic deletion of expired docs");
424         } catch (IOException ioe) {
425           log.error("IOException in periodic deletion of expired docs: " +
426                     ioe.getMessage(), ioe);
427           // DO NOT RETHROW: ScheduledExecutor will supress subsequent executions
428         } catch (RuntimeException re) {
429           log.error("Runtime error in periodic deletion of expired docs: " + 
430                     re.getMessage(), re);
431           // DO NOT RETHROW: ScheduledExecutor will supress subsequent executions
432         } finally {
433           SolrRequestInfo.clearRequestInfo();
434         }
435       } finally {
436         req.close();
437       }
438     }
439   }

441   /**
442    * <p>
443    * Helper method that returns true if the Runnable managed by this factory 
444    * should be responseible of doing periodica deletes.
445    * </p>
446    * <p>
447    * In simple standalone instalations this method always returns true, 
448    * but in cloud mode it will be true if and only if we are currently the leader 
449    * of the (active) slice with the first name (lexigraphically).
450    * </p>
451    * <p>
452    * If this method returns false, it may have also logged a message letting the user 
453    * know why we aren't attempting period deletion (but it will attempt to not log 
454    * this excessively)
455    * </p>
456    */
457   private boolean iAmInChargeOfPeriodicDeletes() {
458     ZkController zk = core.getCoreDescriptor().getCoreContainer().getZkController();
459 
460     if (null == zk) return true;
461     
462     // This is a lot simpler then doing our own "leader" election across all replicas 
463     // of all shards since:
464     //   a) we already have a per shard leader
465     //   b) shard names must be unique
466     //   c) ClusterState is already being "watched" by ZkController, no additional zk hits
467     //   d) there might be multiple instances of this factory (in multiple chains) per 
468     //      collection, so picking an ephemeral node name for our election would be tricky
469 
470     CloudDescriptor desc = core.getCoreDescriptor().getCloudDescriptor();
471     String col = desc.getCollectionName();
472 
473     List<Slice> slices = new ArrayList<Slice>(zk.getClusterState().getActiveSlices(col));
474     Collections.sort(slices, COMPARE_SLICES_BY_NAME);
475     if (slices.isEmpty()) {
476       log.error("Collection {} has no active Slices?", col);
477       return false;
478     }
479     Replica firstSliceLeader = slices.get(0).getLeader();
480     if (null == firstSliceLeader) {
481       log.warn("Slice in charge of periodic deletes for {} does not currently have a leader",
482                col);
483       return false;
484     }
485     String leaderInCharge = firstSliceLeader.getName();
486     String myCoreNodeName = desc.getCoreNodeName();
487     
488     boolean inChargeOfDeletesRightNow = leaderInCharge.equals(myCoreNodeName);
489 
490     if (previouslyInChargeOfDeletes && ! inChargeOfDeletesRightNow) {
491       // don't spam the logs constantly, just log when we know that we're not the guy
492       // (the first time -- or anytime we were, but no longer are)
493       log.info("Not currently in charge of periodic deletes for this collection, " + 
494                "will not trigger delete or log again until this changes");
495     }
496 
497     previouslyInChargeOfDeletes = inChargeOfDeletesRightNow;
498     return inChargeOfDeletesRightNow;
499   }
500 
501   /** @see #iAmInChargeOfPeriodicDeletes */
502   private volatile boolean previouslyInChargeOfDeletes = true;
503 
504   private static final Comparator<Slice> COMPARE_SLICES_BY_NAME = new Comparator<Slice>() {
505     public int compare(Slice a, Slice b) {
506       return a.getName().compareTo(b.getName());
507     }
508   };
509 
510 }
