144   private void buildClients() throws Exception {
145 
146     jetty.start();
147     url = buildUrl(jetty.getLocalPort(), "/solr/");
148 
149     // Mostly to keep annoying logging messages from being sent out all the time.
150 
151     for (int idx = 0; idx < indexingThreads; ++idx) {
152       HttpSolrClient client = new HttpSolrClient(url);
153       client.setDefaultMaxConnectionsPerHost(25);
154       client.setConnectionTimeout(30000);
155       client.setSoTimeout(60000);
156       indexingClients.add(client);
157     }
158     for (int idx = 0; idx < queryThreads; ++idx) {
159       HttpSolrClient client = new HttpSolrClient(url);
160       client.setDefaultMaxConnectionsPerHost(25);
161       client.setConnectionTimeout(30000);
162       client.setSoTimeout(30000);
163       queryingClients.add(client);
164     }

168   // Unless things go _really_ well, stop after you have the directories set up.
169   private void doStress(int secondsToRun) throws Exception {
170     makeCores(solrHomeDirectory);
171 
172     //MUST start the server after the cores are made.
173     buildClients();
174 
175     try {
176 
177       log.info("Starting indexing and querying");
178 
179       int secondsRun = 0;
180       int secondsRemaining = secondsToRun;
181       do {
182 
183         int cycleSeconds = Math.min(resetInterval, secondsRemaining);
184         log.info(String.format(Locale.ROOT, "\n\n\n\n\nStarting a %,d second cycle, seconds left: %,d. Seconds run so far: %,d.",
185             cycleSeconds, secondsRemaining, secondsRun));
186 
187         Indexer idxer = new Indexer(this, url, indexingClients, indexingThreads, cycleSeconds, random());
188 
189         Queries queries = new Queries(this, url, queryingClients, queryThreads, random());
190 
191         idxer.waitOnThreads();
192 
193         queries.waitOnThreads();
194 
195         secondsRemaining = Math.max(secondsRemaining - resetInterval, 0);
196 
197         checkResults(queryingClients.get(0), queries, idxer);
198 
199         secondsRun += cycleSeconds;
200 
201         if (secondsRemaining > 0) {
202           deleteAllDocuments(queryingClients.get(0), queries);
203         }
204       } while (secondsRemaining > 0);
205 
206       assertTrue("We didn't index any documents, somethings really messed up", cumulativeDocs > 0);
207     } catch (Exception e) {
208       e.printStackTrace();
209       fail("Caught unexpected exception");
210     }
211   }

213   private void makeCores(File home) throws Exception {
214     File testSrcRoot = new File(SolrTestCaseJ4.TEST_HOME());
215     String srcSolrXml = "solr-stress-new.xml";
216 
217     FileUtils.copyFile(new File(testSrcRoot, srcSolrXml), new File(home, "solr.xml"));
218 
219     // create directories in groups of 100 until you have enough.
220     for (int idx = 0; idx < numCores; ++idx) {
221       String coreName = String.format(Locale.ROOT, "%05d_core", idx);
222       makeCore(new File(home, coreName), testSrcRoot);
223       coreCounts.put(coreName, 0L);
224       coreNames.add(coreName);
225     }
226   }

228   private void makeCore(File coreDir, File testSrcRoot) throws IOException {
229     File conf = new File(coreDir, "conf");
230 
231     if (!conf.mkdirs()) log.warn("mkdirs returned false in makeCore... ignoring");
232 
233     File testConf = new File(testSrcRoot, "collection1/conf");

238     FileUtils.copyFile(new File(testConf, "solrconfig.snippet.randomindexconfig.xml"),

240 
241     FileUtils.copyFile(new File(testSrcRoot, "conf/core.properties"), new File(coreDir, "core.properties"));
242 
243   }

246   void deleteAllDocuments(HttpSolrClient client, Queries queries) {
247     log.info("Deleting data from last cycle, this may take a few minutes.");
248 
249     for (String core : coreNames) {
250       try {
251         client.setBaseURL(url + core);
252         client.deleteByQuery("*:*");
253         client.optimize(true, true); // should be close to a no-op.
254       } catch (Exception e) {
255         e.printStackTrace();
256       }
257     }
258 
259     // We're testing, after all. Let's be really sure things are as we expect.
260     log.info("Insuring all cores empty");
261     long foundDocs = 0;
262     for (String core : coreNames) {
263       try {
264         long found = queries.getCount(client, core);
265         assertEquals("Cores should be empty", found, 0L);
266         foundDocs += found;
267       } catch (Exception e) {
268         e.printStackTrace();
269       }
270     }
271 
272     if (foundDocs > 0) {
273       log.warn("Found docs after purging done, this is bad.");
274     }
275     // Reset counters for another go-round
276     coreCounts.clear();
277     for (String core : coreNames) {
278       coreCounts.put(core, 0L);
279     }
280   }

282   private void checkResults(HttpSolrClient client, Queries queries, Indexer idxer) throws InterruptedException {
283     log.info("Checking if indexes have all the documents they should...");
284     long totalDocsFound = 0;
285     for (Map.Entry<String, Long> ent : coreCounts.entrySet()) {
286       client.setBaseURL(url + ent.getKey());
287       for (int idx = 0; idx < 3; ++idx) {
288         try {
289           client.commit(true, true);
290           break; // retry loop
291         } catch (Exception e) {
292           log.warn("Exception when committing core " + ent.getKey() + " " + e.getMessage());
293           Thread.sleep(100L);
294         }
295       }
296       long numFound = queries.getCount(client, ent.getKey());
297       totalDocsFound += numFound;
298       assertEquals(String.format(Locale.ROOT, "Core %s bad!", ent.getKey()), (long) ent.getValue(), numFound);
299     }
300 
301     log.info(String.format(Locale.ROOT, "\n\nDocs indexed (cumulative, all cycles): %,d, total docs: %,d: Cycle stats: updates: %,d: qtimes: %,d",
302         Indexer.idUnique.get(), totalDocsFound, idxer.getAccumUpdates(), idxer.getAccumQtimes()));
303 
304     cumulativeDocs += totalDocsFound;
305   }

307   String getRandomCore(Random random) {
308     return coreNames.get(Math.abs(random.nextInt()) % coreNames.size());
309   }

311   void incrementCoreCount(String core) {
312     synchronized (locker) {

318 class Indexer {
319 
320   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
321 
322   static AtomicInteger idUnique = new AtomicInteger(0);
323 
324   static AtomicInteger errors = new AtomicInteger(0);
325 
326   static AtomicInteger docsThisCycle = new AtomicInteger(0);
327 
328   static AtomicLong qTimesAccum = new AtomicLong(0);
329 
330   static AtomicInteger updateCounts = new AtomicInteger(0);
331 
332   static volatile int lastCount;
333 
334   static volatile TimeOut stopTimeout;
335   private static volatile TimeOut nextTimeout;
336 
337   ArrayList<OneIndexer> _threads = new ArrayList<>();
338 
339   public Indexer(OpenCloseCoreStressTest OCCST, String url, List<HttpSolrClient> clients, int numThreads, int secondsToRun, Random random) {
340     stopTimeout = new TimeOut(secondsToRun, TimeUnit.SECONDS);
341     nextTimeout = new TimeOut(60, TimeUnit.SECONDS);
342     docsThisCycle.set(0);
343     qTimesAccum.set(0);
344     updateCounts.set(0);
345     for (int idx = 0; idx < numThreads; ++idx) {
346       OneIndexer one = new OneIndexer(OCCST, url, clients.get(idx), random.nextLong());
347       _threads.add(one);
348       one.start();
349     }
350   }
351 
352   public void waitOnThreads() {
353     for (Thread thread : _threads) {
354       try {
355         thread.join();
356       } catch (InterruptedException e) {
357         e.printStackTrace();
358       }
359     }
360   }
361 
362   public long getAccumQtimes() {
363     return qTimesAccum.get();
364   }
365 
366   public int getAccumUpdates() {
367     return updateCounts.get();
368   }
369 
370   synchronized static void progress(int myId, String core) {
371     if (nextTimeout.hasTimedOut()) {
372       log.info(String.format(Locale.ROOT, " s indexed: [run %,8d] [cycle %,8d] [last minute %,8d] Last core updated: %s. Seconds left in cycle %,4d",
373           myId, docsThisCycle.get(), myId - lastCount, core, stopTimeout.timeLeft(TimeUnit.SECONDS)));
374       lastCount = myId;
375       nextTimeout = new TimeOut(60, TimeUnit.SECONDS);
376     }
377   }
378 
379 }

381 class OneIndexer extends Thread {
382   private final OpenCloseCoreStressTest OCCST;
383   private final HttpSolrClient client;
384   private final String baseUrl;
385   private final Random random;
386 
387   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
388 
389   OneIndexer(OpenCloseCoreStressTest OCCST, String url, HttpSolrClient client, long seed) {
390     this.OCCST = OCCST;
391     this.client = client;
392     this.baseUrl = url;
393     this.random = new Random(seed);
394   }
395 
396   @Override
397   public void run() {
398     log.info(String.format(Locale.ROOT, "Starting indexing thread: " + getId()));
399 
400     while (! Indexer.stopTimeout.hasTimedOut()) {
401       int myId = Indexer.idUnique.incrementAndGet();
402       Indexer.docsThisCycle.incrementAndGet();
403       String core = OCCST.getRandomCore(random);
404       OCCST.incrementCoreCount(core);
405       Indexer.progress(myId, core);
406       for (int idx = 0; idx < 3; ++idx) {
407         SolrInputDocument doc = new SolrInputDocument();
408         doc.addField("id", "id" + Integer.toString(myId));
409         doc.addField("text", "text " + Integer.toString(myId));
410         UpdateRequest update = new UpdateRequest();
411         update.add(doc);
412 
413         try {
414           client.setBaseURL(baseUrl + core);
415           UpdateResponse response = client.add(doc, OpenCloseCoreStressTest.COMMIT_WITHIN);
416           if (response.getStatus() != 0) {
417             log.warn("Failed to index a document to core " + core + " with status " + response.getStatus());
418           } else {
419             Indexer.qTimesAccum.addAndGet(response.getQTime());
420             Indexer.updateCounts.incrementAndGet();
421             break; // retry loop.
422           }
423           Thread.sleep(100L); // Let's not go crazy here.
424         } catch (Exception e) {
425           if (e instanceof InterruptedException) return;
426           Indexer.errors.incrementAndGet();
427           if (idx == 2) {
428             log.warn("Could not reach server while indexing for three tries, quitting " + e.getMessage());
429           } else {
430             log.info("Indexing thread " + Thread.currentThread().getId() + " swallowed one exception " + e.getMessage());
431             try {
432               Thread.sleep(500);
433             } catch (InterruptedException tex) {
434               return;
435             }
436           }
437         }
438       }
439     }
440     log.info("Leaving indexing thread " + getId());
441   }
442 }

444 class Queries {
445   static AtomicBoolean _keepon = new AtomicBoolean(true);
446 
447   List<Thread> _threads = new ArrayList<>();
448   static AtomicInteger _errors = new AtomicInteger(0);
449   String baseUrl;
450 
451   public Queries(OpenCloseCoreStressTest OCCST, String url, List<HttpSolrClient> clients, int numThreads, Random random) {
452     baseUrl = url;
453     for (int idx = 0; idx < numThreads; ++idx) {
454       Thread one = new OneQuery(OCCST, url, clients.get(idx), random.nextLong());
455       _threads.add(one);
456       one.start();
457     }
458 
459   }
460 
461   public void waitOnThreads() {
462     Queries._keepon.set(false);
463     for (Thread thread : _threads) {
464       try {
465         thread.join();
466       } catch (InterruptedException e) {
467         e.printStackTrace();
468       }
469     }
470   }
471 
472   public long getCount(HttpSolrClient client, String core) {
473     ModifiableSolrParams params = new ModifiableSolrParams();
474     params.set("qt", "/select");
475     params.set("q", "*:*");
476     long numFound = 0;
477     client.setBaseURL(baseUrl + core);
478     try {
479       QueryResponse response = client.query(params);
480       numFound = response.getResults().getNumFound();
481     } catch (Exception e) {
482       e.printStackTrace();
483     }
484     return numFound;
485   }
486 }

488 class OneQuery extends Thread {
489   OpenCloseCoreStressTest OCCST;
490   private final HttpSolrClient client;
491   private final String baseUrl;
492   private final Random random;
493 
494   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
495 
496   OneQuery(OpenCloseCoreStressTest OCCST, String url, HttpSolrClient client, long seed) {
497     this.OCCST = OCCST;
498     this.client = client;
499     this.baseUrl = url;
500     this.random = new Random(seed);
501   }
502 
503   @Override
504   public void run() {
505     log.info(String.format(Locale.ROOT, "Starting query thread: " + getId()));
506     while (Queries._keepon.get()) {
507       String core = OCCST.getRandomCore(random);
508       for (int idx = 0; idx < 3; ++idx) {
509         ModifiableSolrParams params = new ModifiableSolrParams();
510         params.set("qt", "/select");
511         params.set("q", "*:*");
512 
513         try {
514           // sleep between 250ms and 10000 ms
515           Thread.sleep(100L); // Let's not go crazy here.
516           client.setBaseURL(baseUrl + core);
517           QueryResponse response = client.query(params);
518 
519           if (response.getStatus() != 0) {
520             log.warn("Failed to query core " + core + " with status " + response.getStatus());
521           }
522             // Perhaps collect some stats here in future.
523           break; // retry loop
524         } catch (Exception e) {
525           if (e instanceof InterruptedException) return;
526           Queries._errors.incrementAndGet();
527           if (idx == 2) {
528             log.warn("Could not reach server while indexing for three tries, quitting " + e.getMessage());
529           } else {
530             log.info("Querying thread: " + Thread.currentThread().getId() + " swallowed exception: " + e.getMessage());
531             try {
532               Thread.sleep(500L);
533             } catch (InterruptedException tex) {
534               return;
535             }
536           }
537         }
538       }
539     }
540     log.info(String.format(Locale.ROOT, "Leaving query thread: " + getId()));
541   }
