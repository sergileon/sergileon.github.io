71    @Override
72    public void process(ResponseBuilder rb) throws IOException {
73  
74      SolrParams params = rb.req.getParams();
75      ReturnFields returnFields = new SolrReturnFields( rb.req );
76  
77      int flags = 0;
78      if (returnFields.wantsScore()) {
79        flags |= SolrIndexSearcher.GET_SCORES;
80      }
81  
82      rb.setFieldFlags(flags);
83  
84      if (params.getBool(MoreLikeThisParams.MLT, false)) {
85        log.debug("Starting MoreLikeThis.Process.  isShard: "
86            + params.getBool(ShardParams.IS_SHARD));
87        SolrIndexSearcher searcher = rb.req.getSearcher();
88  
89        if (params.getBool(ShardParams.IS_SHARD, false)) {
90          if (params.get(MoreLikeThisComponent.DIST_DOC_ID) == null) {
91            if (rb.getResults().docList.size() == 0) {
92              // return empty response
93              rb.rsp.add("moreLikeThis", new NamedList<DocList>());
94              return;
95            }
96            
97            MoreLikeThisHandler.MoreLikeThisHelper mlt = new MoreLikeThisHandler.MoreLikeThisHelper(
98                params, searcher);
99  
100           NamedList<BooleanQuery> bQuery = mlt.getMoreLikeTheseQuery(rb
101               .getResults().docList);
102           
103           NamedList<String> temp = new NamedList<>();
104           Iterator<Entry<String,BooleanQuery>> idToQueryIt = bQuery.iterator();
105 
106           
107           while (idToQueryIt.hasNext()) {
108             Entry<String,BooleanQuery> idToQuery = idToQueryIt.next();
109             String s = idToQuery.getValue().toString();
110 
111             log.debug("MLT Query:" + s);
112             temp.add(idToQuery.getKey(), idToQuery.getValue().toString());
113           }
114 
115           rb.rsp.add("moreLikeThis", temp);
116         } else {
117           NamedList<DocList> sim = getMoreLikeThese(rb, rb.req.getSearcher(),
118               rb.getResults().docList, flags);
119           rb.rsp.add("moreLikeThis", sim);
120         }
121       } else {
122         // non distrib case
123         NamedList<DocList> sim = getMoreLikeThese(rb, rb.req.getSearcher(), rb.getResults().docList,
124             flags);
125         rb.rsp.add("moreLikeThis", sim);
126       }
127     }
128   }

130   @Override
131   public void handleResponses(ResponseBuilder rb, ShardRequest sreq) {
132     if ((sreq.purpose & ShardRequest.PURPOSE_GET_TOP_IDS) != 0
133         && rb.req.getParams().getBool(COMPONENT_NAME, false)) {
134       log.debug("ShardRequest.response.size: " + sreq.responses.size());
135       for (ShardResponse r : sreq.responses) {
136         if (r.getException() != null) {
137           // This should only happen in case of using shards.tolerant=true. Omit this ShardResponse
138           continue;
139         }
140         NamedList<?> moreLikeThisReponse = (NamedList<?>) r.getSolrResponse()
141             .getResponse().get("moreLikeThis");
142         log.debug("ShardRequest.response.shard: " + r.getShard());
143         if (moreLikeThisReponse != null) {
144           for (Entry<String,?> entry : moreLikeThisReponse) {
145             log.debug("id: \"" + entry.getKey() + "\" Query: \""
146                 + entry.getValue() + "\"");
147             ShardRequest s = buildShardQuery(rb, (String) entry.getValue(),
148                 entry.getKey());
149             rb.addRequest(this, s);
150           }
151         }
152       }
153     }
154     
155     if ((sreq.purpose & ShardRequest.PURPOSE_GET_MLT_RESULTS) != 0) {
156       for (ShardResponse r : sreq.responses) {
157         log.debug("MLT Query returned: "
158             + r.getSolrResponse().getResponse().toString());
159       }
160     }
161   }

163   @Override
164   public void finishStage(ResponseBuilder rb) {
165     
166     // Handling Responses in finishStage, because solrResponse will put
167     // moreLikeThis xml
168     // segment ahead of result/response.
169     if (rb.stage == ResponseBuilder.STAGE_GET_FIELDS
170         && rb.req.getParams().getBool(COMPONENT_NAME, false)) {
171       Map<Object,SolrDocumentList> tempResults = new LinkedHashMap<>();
172       
173       int mltcount = rb.req.getParams().getInt(MoreLikeThisParams.DOC_COUNT, 5);
174       String keyName = rb.req.getSchema().getUniqueKeyField().getName();
175       
176       for (ShardRequest sreq : rb.finished) {
177         if ((sreq.purpose & ShardRequest.PURPOSE_GET_MLT_RESULTS) != 0) {
178           for (ShardResponse r : sreq.responses) {
179             log.debug("ShardRequest.response.shard: " + r.getShard());
180             String key = r.getShardRequest().params
181                 .get(MoreLikeThisComponent.DIST_DOC_ID);
182             SolrDocumentList shardDocList =  (SolrDocumentList) r.getSolrResponse().getResponse().get("response");
183 
184             if (shardDocList == null) {
185               continue;
186             }
187  
188             log.info("MLT: results added for key: " + key + " documents: "
189                 + shardDocList.toString());
190 //            if (log.isDebugEnabled()) {
191 //              for (SolrDocument doc : shardDocList) {
192 //                doc.addField("shard", "=" + r.getShard());
193 //              }
194 //            }
195             SolrDocumentList mergedDocList = tempResults.get(key);
196  
197             if (mergedDocList == null) {
198               mergedDocList = new SolrDocumentList();
199               mergedDocList.addAll(shardDocList);
200               mergedDocList.setNumFound(shardDocList.getNumFound());
201               mergedDocList.setStart(shardDocList.getStart());
202               mergedDocList.setMaxScore(shardDocList.getMaxScore());
203             } else {
204               mergedDocList = mergeSolrDocumentList(mergedDocList,
205                   shardDocList, mltcount, keyName);
206             }
207             log.debug("Adding docs for key: " + key);
208             tempResults.put(key, mergedDocList);
209           }
210         }
211       }
212 
213       NamedList<SolrDocumentList> list = buildMoreLikeThisNamed(tempResults,
214           rb.resultIds);
215      
216       rb.rsp.add("moreLikeThis", list);
217       
218     }
219     super.finishStage(rb);
220   }

222   /**
223    * Returns NamedList based on the order of
224    * resultIds.shardDoc.positionInResponse
225    */
226   NamedList<SolrDocumentList> buildMoreLikeThisNamed(
227       Map<Object,SolrDocumentList> allMlt, Map<Object,ShardDoc> resultIds) {
228     NamedList<SolrDocumentList> result = new NamedList<>();
229     TreeMap<Integer,Object> sortingMap = new TreeMap<>();
230     for (Entry<Object,ShardDoc> next : resultIds.entrySet()) {
231       sortingMap.put(next.getValue().positionInResponse, next.getKey());
232     }
233     for (Object key : sortingMap.values()) {
234       SolrDocumentList sdl = allMlt.get(key);
235       if (sdl == null) {
236         sdl = new SolrDocumentList();
237         sdl.setNumFound(0);
238         sdl.setStart(0);
239       }
240       result.add(key.toString(), sdl);
241     }
242     return result;
243   }

245   public SolrDocumentList mergeSolrDocumentList(SolrDocumentList one,
246       SolrDocumentList two, int maxSize, String idField) {
247 
248     List<SolrDocument> l = new ArrayList<>();
249     
250     // De-dup records sets. Shouldn't happen if indexed correctly.
251     Map<String,SolrDocument> map = new HashMap<>();
252     for (SolrDocument doc : one) {
253       Object id = doc.getFieldValue(idField);
254       assert id != null : doc.toString();
255       map.put(id.toString(), doc);
256     }
257     for (SolrDocument doc : two) {
258       map.put(doc.getFieldValue(idField).toString(), doc);
259     }
260     
261     l = new ArrayList<>(map.values());
262     
263     // Comparator to sort docs based on score. null scores/docs are set to 0.
264     
265     // hmm...we are ordering by scores that are not really comparable...
266     Comparator<SolrDocument> c = new Comparator<SolrDocument>() {
267       public int compare(SolrDocument o1, SolrDocument o2) {
268         Float f1 = getFloat(o1);
269         Float f2 = getFloat(o2);
270         return f2.compareTo(f1);
271       }
272       
273       private Float getFloat(SolrDocument doc) {
274         Float f = 0f;
275         if (doc != null) {
276           Object o = doc.getFieldValue("score");
277           if (o != null && o instanceof Float) {
278             f = (Float) o;
279           }
280         }
281         return f;
282       }
283     };
284     
285     Collections.sort(l, c);
286     
287     // Truncate list to maxSize
288     if (l.size() > maxSize) {
289       l = l.subList(0, maxSize);
290     }
291     
292     // Create SolrDocumentList Attributes from originals
293     SolrDocumentList result = new SolrDocumentList();
294     result.addAll(l);
295     result.setMaxScore(Math.max(one.getMaxScore(), two.getMaxScore()));
296     result.setNumFound(one.getNumFound() + two.getNumFound());
297     result.setStart(Math.min(one.getStart(), two.getStart()));
298 
299     return result;
300   }

302   ShardRequest buildShardQuery(ResponseBuilder rb, String q, String key) {
303     ShardRequest s = new ShardRequest();
304     s.params = new ModifiableSolrParams(rb.req.getParams());
305     s.purpose |= ShardRequest.PURPOSE_GET_MLT_RESULTS;
306     // Maybe unnecessary, but safe.
307     s.purpose |= ShardRequest.PURPOSE_PRIVATE;
308     
309     s.params.remove(ShardParams.SHARDS);
310     // s.params.remove(MoreLikeThisComponent.COMPONENT_NAME);
311     
312     // needed to correlate results
313     s.params.set(MoreLikeThisComponent.DIST_DOC_ID, key);
314     s.params.set(CommonParams.START, 0);
315     int mltcount = s.params.getInt(MoreLikeThisParams.DOC_COUNT, 20); // overrequest
316     s.params.set(CommonParams.ROWS, mltcount);
317     
318     // adding score to rank moreLikeThis
319     s.params.remove(CommonParams.FL);
320     
321     // Should probably add something like this:
322     // String fl = s.params.get(MoreLikeThisParams.RETURN_FL, "*");
323     // if(fl != null){
324     // s.params.set(CommonParams.FL, fl + ",score");
325     // }
326     String id = rb.req.getSchema().getUniqueKeyField()
327     .getName();
328     s.params.set(CommonParams.FL, "score," + id);
329     s.params.set("sort", "score desc");
330     // MLT Query is submitted as normal query to shards.
331     s.params.set(CommonParams.Q, q);
332     
333     return s;
334   }

336   ShardRequest buildMLTQuery(ResponseBuilder rb, String q) {
337     ShardRequest s = new ShardRequest();
338     s.params = new ModifiableSolrParams();
339     
340 
341     s.params.set(CommonParams.START, 0);
342 
343     String id = rb.req.getSchema().getUniqueKeyField().getName();
344 
345     s.params.set(CommonParams.FL, "score," + id);
346     // MLT Query is submitted as normal query to shards.
347     s.params.set(CommonParams.Q, q);
348     
349     s.shards = ShardRequest.ALL_SHARDS;
350     return s;
351   }

353   NamedList<DocList> getMoreLikeThese(ResponseBuilder rb,
354       SolrIndexSearcher searcher, DocList docs, int flags) throws IOException {
355     SolrParams p = rb.req.getParams();
356     IndexSchema schema = searcher.getSchema();
357     MoreLikeThisHandler.MoreLikeThisHelper mltHelper = new MoreLikeThisHandler.MoreLikeThisHelper(
358         p, searcher);
359     NamedList<DocList> mlt = new SimpleOrderedMap<>();
360     DocIterator iterator = docs.iterator();
361     
362     SimpleOrderedMap<Object> dbg = null;
363     if (rb.isDebug()) {
364       dbg = new SimpleOrderedMap<>();
365     }
366     
367     while (iterator.hasNext()) {
368       int id = iterator.nextDoc();
369       int rows = p.getInt(MoreLikeThisParams.DOC_COUNT, 5);
370       DocListAndSet sim = mltHelper.getMoreLikeThis(id, 0, rows, null, null,
371           flags);
372       String name = schema.printableUniqueKey(searcher.doc(id));
373       mlt.add(name, sim.docList);
374       
375       if (dbg != null) {
376         SimpleOrderedMap<Object> docDbg = new SimpleOrderedMap<>();
377         docDbg.add("rawMLTQuery", mltHelper.getRawMLTQuery().toString());
378         docDbg
379             .add("boostedMLTQuery", mltHelper.getBoostedMLTQuery().toString());
380         docDbg.add("realMLTQuery", mltHelper.getRealMLTQuery().toString());
381         SimpleOrderedMap<Object> explains = new SimpleOrderedMap<>();
382         DocIterator mltIte = sim.docList.iterator();
383         while (mltIte.hasNext()) {
384           int mltid = mltIte.nextDoc();
385           String key = schema.printableUniqueKey(searcher.doc(mltid));
386           explains.add(key,
387               searcher.explain(mltHelper.getRealMLTQuery(), mltid));
388         }
389         docDbg.add("explain", explains);
390         dbg.add(name, docDbg);
391       }
392     }
393     
394     // add debug information
395     if (dbg != null) {
396       rb.addDebugInfo("moreLikeThis", dbg);
397     }
398     return mlt;
399   }
