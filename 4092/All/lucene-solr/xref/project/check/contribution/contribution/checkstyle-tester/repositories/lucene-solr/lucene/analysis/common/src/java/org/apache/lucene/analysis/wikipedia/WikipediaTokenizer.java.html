94     * Only output tokens

96    public static final int TOKENS_ONLY = 0;

98     * Only output untokenized tokens, which are tokens that would normally be split into several tokens

154     init(tokenOutput, untokenizedTypes);
155   }

157   private void init(int tokenOutput, Set<String> untokenizedTypes) {
158     // TODO: cutover to enum
159     if (tokenOutput != TOKENS_ONLY &&
160         tokenOutput != UNTOKENIZED_ONLY &&
161         tokenOutput != BOTH) {
162       throw new IllegalArgumentException("tokenOutput must be TOKENS_ONLY, UNTOKENIZED_ONLY or BOTH");
163     }
164     this.tokenOutput = tokenOutput;
165     this.untokenizedTypes = untokenizedTypes;    
166   }

168   /*
169   * (non-Javadoc)
170   *
171   * @see org.apache.lucene.analysis.TokenStream#next()
172   */
173   @Override
174   public final boolean incrementToken() throws IOException {
175     if (tokens != null && tokens.hasNext()){
176       AttributeSource.State state = tokens.next();
177       restoreState(state);
178       return true;
179     }
180     clearAttributes();
181     int tokenType = scanner.getNextToken();
182 
183     if (tokenType == WikipediaTokenizerImpl.YYEOF) {
184       return false;
185     }
186     String type = WikipediaTokenizerImpl.TOKEN_TYPES[tokenType];
187     if (tokenOutput == TOKENS_ONLY || untokenizedTypes.contains(type) == false){
188       setupToken();
189     } else if (tokenOutput == UNTOKENIZED_ONLY && untokenizedTypes.contains(type) == true){
190       collapseTokens(tokenType);
191 
192     }
193     else if (tokenOutput == BOTH){
194       //collapse into a single token, add it to tokens AND output the individual tokens
195       //output the untokenized Token first
196       collapseAndSaveTokens(tokenType, type);
197     }
198     int posinc = scanner.getPositionIncrement();
199     if (first && posinc == 0) {
200       posinc = 1; // don't emit posinc=0 for the first token!
201     }
202     posIncrAtt.setPositionIncrement(posinc);
203     typeAtt.setType(type);
204     first = false;
205     return true;
206   }
207 
208   private void collapseAndSaveTokens(int tokenType, String type) throws IOException {
209     //collapse
210     StringBuilder buffer = new StringBuilder(32);
211     int numAdded = scanner.setText(buffer);
212     //TODO: how to know how much whitespace to add
213     int theStart = scanner.yychar();
214     int lastPos = theStart + numAdded;
215     int tmpTokType;
216     int numSeen = 0;
217     List<AttributeSource.State> tmp = new ArrayList<>();
218     setupSavedToken(0, type);
219     tmp.add(captureState());
220     //while we can get a token and that token is the same type and we have not transitioned to a new wiki-item of the same type
221     while ((tmpTokType = scanner.getNextToken()) != WikipediaTokenizerImpl.YYEOF && tmpTokType == tokenType && scanner.getNumWikiTokensSeen() > numSeen){
222       int currPos = scanner.yychar();
223       //append whitespace
224       for (int i = 0; i < (currPos - lastPos); i++){
225         buffer.append(' ');
226       }
227       numAdded = scanner.setText(buffer);
228       setupSavedToken(scanner.getPositionIncrement(), type);
229       tmp.add(captureState());
230       numSeen++;
231       lastPos = currPos + numAdded;
232     }
233     //trim the buffer
234     // TODO: this is inefficient
235     String s = buffer.toString().trim();
236     termAtt.setEmpty().append(s);
237     offsetAtt.setOffset(correctOffset(theStart), correctOffset(theStart + s.length()));
238     flagsAtt.setFlags(UNTOKENIZED_TOKEN_FLAG);
239     //The way the loop is written, we will have proceeded to the next token.  We need to pushback the scanner to lastPos
240     if (tmpTokType != WikipediaTokenizerImpl.YYEOF){
241       scanner.yypushback(scanner.yylength());
242     }
243     tokens = tmp.iterator();
244   }
245 
246   private void setupSavedToken(int positionInc, String type){
247     setupToken();
248     posIncrAtt.setPositionIncrement(positionInc);
249     typeAtt.setType(type);
250   }

252   private void collapseTokens(int tokenType) throws IOException {
253     //collapse
254     StringBuilder buffer = new StringBuilder(32);
255     int numAdded = scanner.setText(buffer);
256     //TODO: how to know how much whitespace to add
257     int theStart = scanner.yychar();
258     int lastPos = theStart + numAdded;
259     int tmpTokType;
260     int numSeen = 0;
261     //while we can get a token and that token is the same type and we have not transitioned to a new wiki-item of the same type
262     while ((tmpTokType = scanner.getNextToken()) != WikipediaTokenizerImpl.YYEOF && tmpTokType == tokenType && scanner.getNumWikiTokensSeen() > numSeen){
263       int currPos = scanner.yychar();
264       //append whitespace
265       for (int i = 0; i < (currPos - lastPos); i++){
266         buffer.append(' ');
267       }
268       numAdded = scanner.setText(buffer);
269       numSeen++;
270       lastPos = currPos + numAdded;
271     }
272     //trim the buffer
273     // TODO: this is inefficient
274     String s = buffer.toString().trim();
275     termAtt.setEmpty().append(s);
276     offsetAtt.setOffset(correctOffset(theStart), correctOffset(theStart + s.length()));
277     flagsAtt.setFlags(UNTOKENIZED_TOKEN_FLAG);
278     //The way the loop is written, we will have proceeded to the next token.  We need to pushback the scanner to lastPos
279     if (tmpTokType != WikipediaTokenizerImpl.YYEOF){
280       scanner.yypushback(scanner.yylength());
281     } else {
282       tokens = null;
283     }
284   }
285 
286   private void setupToken() {
287     scanner.getText(termAtt);
288     final int start = scanner.yychar();
289     offsetAtt.setOffset(correctOffset(start), correctOffset(start + termAtt.length()));
290   }

292   @Override
293   public void close() throws IOException {
294     super.close();
295     scanner.yyreset(input);
296   }
297 
298   /*
299   * (non-Javadoc)
300   *
301   * @see org.apache.lucene.analysis.TokenStream#reset()
302   */
303   @Override
304   public void reset() throws IOException {
305     super.reset();
306     scanner.yyreset(input);
307     tokens = null;
308     scanner.reset();
309     first = true;
310   }
311 
312   @Override
313   public void end() throws IOException {
314     super.end();
315     // set final offset
316     final int finalOffset = correctOffset(scanner.yychar() + scanner.yylength());
317     this.offsetAtt.setOffset(finalOffset, finalOffset);
318   }
