39    public void testMapHashes() throws Exception {
40      DocRouter hp = DocRouter.DEFAULT;
41      List<Range> ranges;
42  
43      // make sure the partitioner uses the "natural" boundaries and doesn't suffer from an off-by-one
44      ranges = hp.partitionRange(2, hp.fullRange());
45      assertEquals(Integer.MIN_VALUE, ranges.get(0).min);
46      assertEquals(0x80000000, ranges.get(0).min);
47      assertEquals(0xffffffff, ranges.get(0).max);
48      assertEquals(0x00000000, ranges.get(1).min);
49      assertEquals(0x7fffffff, ranges.get(1).max);
50  
51      ranges = hp.partitionRange(2, new DocRouter.Range(0, 0x7fffffff));
52      assertEquals(0x00000000, ranges.get(0).min);
53      assertEquals(0x3fffffff, ranges.get(0).max);
54      assertEquals(0x40000000, ranges.get(1).min);
55      assertEquals(0x7fffffff, ranges.get(1).max);
56  
57      int defaultLowerBits = 0x0000ffff;
58  
59      for (int i = 1; i <= 30000; i++) {
60        // start skipping at higher numbers
61        if (i > 100) i+=13;
62        else if (i > 1000) i+=31;
63        else if (i > 5000) i+=101;
64  
65        long rangeSize = 0x0000000100000000L / i;
66  
67        ranges = hp.partitionRange(i, hp.fullRange());
68        assertEquals(i, ranges.size());
69        assertTrue("First range does not start before " + Integer.MIN_VALUE
70            + " it is:" + ranges.get(0).min,
71            ranges.get(0).min <= Integer.MIN_VALUE);
72        assertTrue("Last range does not end after " + Integer.MAX_VALUE
73            + " it is:" + ranges.get(ranges.size() - 1).max,
74            ranges.get(ranges.size() - 1).max >= Integer.MAX_VALUE);
75  
76        for (Range range : ranges) {
77          String s = range.toString();
78          Range newRange = hp.fromString(s);
79          assertEquals(range, newRange);
80        }
81  
82        // ensure that ranges are contiguous and that size deviations are not too large.
83        int lastEnd = Integer.MIN_VALUE - 1;
84        for (Range range : ranges) {
85          int currStart = range.min;
86          int currEnd = range.max;
87          assertEquals(lastEnd+1, currStart);
88  
89          if (ranges.size() < 4000) {
90            // ranges should be rounded to avoid crossing hash domains
91            assertEquals(defaultLowerBits, currEnd & defaultLowerBits);
92  
93            // given our rounding condition that domains should be less than 1/16 of the step size,
94            // this means that any sizing deviations should also be less than 1/16th of the idealized range size.
95            // boolean round = rangeStep >= (1<<bits)*16;
96  
97            long currRangeSize = (long)currEnd - (long)currStart;
98            long error = Math.abs(rangeSize - currRangeSize);
99            assertTrue( error < rangeSize/16);
100         }
101 
102 
103         // String s = range.toString();
104         // Range newRange = hp.fromString(s);
105         // assertEquals(range, newRange);
106         lastEnd = currEnd;
107       }
108 
109     }
110   }

112   public int hash(String id) {
113     // our hashing is defined to be murmurhash3 on the UTF-8 bytes of the key.
114     return Hash.murmurhash3_x86_32(id, 0, id.length(), 0);
115   }

124   public void doNormalIdHashing(DocCollection coll) throws Exception {
125     assertEquals(4, coll.getSlices().size());
126 
127     doId(coll, "b", "shard1");
128     doId(coll, "c", "shard2");
129     doId(coll, "d", "shard3");
130     doId(coll, "e", "shard4");
131   }

133   public void doId(DocCollection coll, String id, String expectedShard) {
134     doIndex(coll, id, expectedShard);
135     doQuery(coll, id, expectedShard);
136   }

138   public void doIndex(DocCollection coll, String id, String expectedShard) {
139     DocRouter router = coll.getRouter();
140     Slice target = router.getTargetSlice(id, null, null, null, coll);
141     assertEquals(expectedShard, target.getName());
142   }

144   public void doQuery(DocCollection coll, String id, String expectedShards) {
145     DocRouter router = coll.getRouter();
146     Collection<Slice> slices = router.getSearchSlices(id, null, coll);
147 
148     List<String> expectedShardStr = StrUtils.splitSmart(expectedShards, ",", true);
149 
150     HashSet<String> expectedSet = new HashSet<>(expectedShardStr);
151     HashSet<String> obtainedSet = new HashSet<>();
152     for (Slice slice : slices) {
153       obtainedSet.add(slice.getName());
154     }
155 
156     assertEquals(slices.size(), obtainedSet.size());  // make sure no repeated slices
157     assertEquals(expectedSet, obtainedSet);
158   }

160   public void testCompositeHashCodes() throws Exception {
161     DocRouter router = DocRouter.getDocRouter(CompositeIdRouter.NAME);
162     assertTrue(router instanceof CompositeIdRouter);
163     router = DocRouter.DEFAULT;
164     assertTrue(router instanceof CompositeIdRouter);
165 
166     DocCollection coll = createCollection(4, router);
167     doNormalIdHashing(coll);
168 
169     // ensure that the shard hashed to is only dependent on the first part of the compound key
170     doId(coll, "b!foo", "shard1");
171     doId(coll, "c!bar", "shard2");
172     doId(coll, "d!baz", "shard3");
173     doId(coll, "e!qux", "shard4");
174 
175     // syntax to specify bits.
176     // Anything over 2 bits should give the same results as above (since only top 2 bits
177     // affect our 4 slice collection).
178     doId(coll, "b/2!foo", "shard1");
179     doId(coll, "c/2!bar", "shard2");
180     doId(coll, "d/2!baz", "shard3");
181     doId(coll, "e/2!qux", "shard4");
182 
183     doId(coll, "b/32!foo", "shard1");
184     doId(coll, "c/32!bar", "shard2");
185     doId(coll, "d/32!baz", "shard3");
186     doId(coll, "e/32!qux", "shard4");
187 
188     // no bits allocated to the first part (kind of odd why anyone would do that though)
189     doIndex(coll, "foo/0!b", "shard1");
190     doIndex(coll, "foo/0!c", "shard2");
191     doIndex(coll, "foo/0!d", "shard3");
192     doIndex(coll, "foo/0!e", "shard4");
193 
194     // means cover whole range on the query side
195     doQuery(coll, "foo/0!", "shard1,shard2,shard3,shard4");
196 
197     doQuery(coll, "b/1!", "shard1,shard2");   // top bit of hash(b)==1, so shard1 and shard2
198     doQuery(coll, "d/1!", "shard3,shard4");   // top bit of hash(b)==0, so shard3 and shard4
199   }

201   /** Make sure CompositeIdRouter doesn't throw exceptions for non-conforming IDs */
202   public void testNonConformingCompositeIds() throws Exception {
203     DocRouter router = DocRouter.getDocRouter(CompositeIdRouter.NAME);
204     DocCollection coll = createCollection(4, router);
205     String[] ids = { "A!B!C!D", "!!!!!!", "A!!!!B", "A!!B!!C", "A/59!B", "A/8/!B/19/", 
206                      "A!B/-5", "!/130!", "!!A/1000", "A//8!B///10!C////" };
207     for (int i = 0 ; i < ids.length ; ++i) {
208       try {
209         Slice targetSlice = coll.getRouter().getTargetSlice(ids[i], null, null, null, coll);
210         assertNotNull(targetSlice);
211       } catch (Exception e) {
212         throw new Exception("Exception routing id '" + ids[i] + "'", e);
213       }
214     }
215   }

217   /** Make sure CompositeIdRouter can route random IDs without throwing exceptions */
218   public void testRandomCompositeIds() throws Exception {
219     DocRouter router = DocRouter.getDocRouter(CompositeIdRouter.NAME);
220     DocCollection coll = createCollection(TestUtil.nextInt(random(), 1, 10), router);
221     StringBuilder idBuilder = new StringBuilder();
222     for (int i = 0 ; i < 10000 ; ++i) {
223       idBuilder.setLength(0);
224       int numParts = TestUtil.nextInt(random(), 1, 30);
225       for (int part = 0; part < numParts; ++part) {
226         switch (random().nextInt(5)) {
227           case 0: idBuilder.append('!'); break;
228           case 1: idBuilder.append('/'); break;
229           case 2: idBuilder.append(TestUtil.nextInt(random(),-100, 1000)); break;
230           default: {
231             int length = TestUtil.nextInt(random(), 1, 10);
232             char[] str = new char[length];
233             TestUtil.randomFixedLengthUnicodeString(random(), str, 0, length);
234             idBuilder.append(str);
235             break;
236           } 
237         }
238       }
239       String id = idBuilder.toString();
240       try {
241         Slice targetSlice = router.getTargetSlice(id, null, null, null, coll);
242         assertNotNull(targetSlice);
243       } catch (Exception e) {
244         throw new Exception("Exception routing id '" + id + "'", e);
245       }
246     }
247   }

271   DocCollection createCollection(int nSlices, DocRouter router) {
272     List<Range> ranges = router.partitionRange(nSlices, router.fullRange());
273 
274     Map<String,Slice> slices = new HashMap<>();
275     for (int i=0; i<ranges.size(); i++) {
276       Range range = ranges.get(i);
277       Slice slice = new Slice("shard"+(i+1), null, map("range",range));
278       slices.put(slice.getName(), slice);
279     }
280 
281     DocCollection coll = new DocCollection("collection1", slices, null, router);
282     return coll;
283   }
