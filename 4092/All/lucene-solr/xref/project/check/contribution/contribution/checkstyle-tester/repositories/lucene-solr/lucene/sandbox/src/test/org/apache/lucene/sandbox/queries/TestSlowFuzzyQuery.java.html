44    public void testFuzziness() throws Exception {
45      //every test with SlowFuzzyQuery.defaultMinSimilarity
46      //is exercising the Automaton, not the brute force linear method
47      
48      Directory directory = newDirectory();
49      RandomIndexWriter writer = new RandomIndexWriter(random(), directory);
50      addDoc("aaaaa", writer);
51      addDoc("aaaab", writer);
52      addDoc("aaabb", writer);
53      addDoc("aabbb", writer);
54      addDoc("abbbb", writer);
55      addDoc("bbbbb", writer);
56      addDoc("ddddd", writer);
57  
58      IndexReader reader = writer.getReader();
59      IndexSearcher searcher = newSearcher(reader);
60      writer.close();
61  
62      SlowFuzzyQuery query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 0);   
63      ScoreDoc[] hits = searcher.search(query, 1000).scoreDocs;
64      assertEquals(3, hits.length);
65      
66      // same with prefix
67      query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 1);   
68      hits = searcher.search(query, 1000).scoreDocs;
69      assertEquals(3, hits.length);
70      query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 2);   
71      hits = searcher.search(query, 1000).scoreDocs;
72      assertEquals(3, hits.length);
73      query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 3);   
74      hits = searcher.search(query, 1000).scoreDocs;
75      assertEquals(3, hits.length);
76      query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 4);   
77      hits = searcher.search(query, 1000).scoreDocs;
78      assertEquals(2, hits.length);
79      query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 5);   
80      hits = searcher.search(query, 1000).scoreDocs;
81      assertEquals(1, hits.length);
82      query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 6);   
83      hits = searcher.search(query, 1000).scoreDocs;
84      assertEquals(1, hits.length);
85      
86      // test scoring
87      query = new SlowFuzzyQuery(new Term("field", "bbbbb"), SlowFuzzyQuery.defaultMinSimilarity, 0);   
88      hits = searcher.search(query, 1000).scoreDocs;
89      assertEquals("3 documents should match", 3, hits.length);
90      List<String> order = Arrays.asList("bbbbb","abbbb","aabbb");
91      for (int i = 0; i < hits.length; i++) {
92        final String term = searcher.doc(hits[i].doc).get("field");
93        //System.out.println(hits[i].score);
94        assertEquals(order.get(i), term);
95      }
96  
97      // test pq size by supplying maxExpansions=2
98      // This query would normally return 3 documents, because 3 terms match (see above):
99      query = new SlowFuzzyQuery(new Term("field", "bbbbb"), SlowFuzzyQuery.defaultMinSimilarity, 0, 2); 
100     hits = searcher.search(query, 1000).scoreDocs;
101     assertEquals("only 2 documents should match", 2, hits.length);
102     order = Arrays.asList("bbbbb","abbbb");
103     for (int i = 0; i < hits.length; i++) {
104       final String term = searcher.doc(hits[i].doc).get("field");
105       //System.out.println(hits[i].score);
106       assertEquals(order.get(i), term);
107     }
108 
109     // not similar enough:
110     query = new SlowFuzzyQuery(new Term("field", "xxxxx"), SlowFuzzyQuery.defaultMinSimilarity, 0);
111     hits = searcher.search(query, 1000).scoreDocs;
112     assertEquals(0, hits.length);
113     query = new SlowFuzzyQuery(new Term("field", "aaccc"), SlowFuzzyQuery.defaultMinSimilarity, 0);   // edit distance to "aaaaa" = 3
114     hits = searcher.search(query, 1000).scoreDocs;
115     assertEquals(0, hits.length);
116 
117     // query identical to a word in the index:
118     query = new SlowFuzzyQuery(new Term("field", "aaaaa"), SlowFuzzyQuery.defaultMinSimilarity, 0);   
119     hits = searcher.search(query, 1000).scoreDocs;
120     assertEquals(3, hits.length);
121     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaa"));
122     // default allows for up to two edits:
123     assertEquals(searcher.doc(hits[1].doc).get("field"), ("aaaab"));
124     assertEquals(searcher.doc(hits[2].doc).get("field"), ("aaabb"));
125 
126     // query similar to a word in the index:
127     query = new SlowFuzzyQuery(new Term("field", "aaaac"), SlowFuzzyQuery.defaultMinSimilarity, 0);   
128     hits = searcher.search(query, 1000).scoreDocs;
129     assertEquals(3, hits.length);
130     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaa"));
131     assertEquals(searcher.doc(hits[1].doc).get("field"), ("aaaab"));
132     assertEquals(searcher.doc(hits[2].doc).get("field"), ("aaabb"));
133     
134     // now with prefix
135     query = new SlowFuzzyQuery(new Term("field", "aaaac"), SlowFuzzyQuery.defaultMinSimilarity, 1);   
136     hits = searcher.search(query, 1000).scoreDocs;
137     assertEquals(3, hits.length);
138     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaa"));
139     assertEquals(searcher.doc(hits[1].doc).get("field"), ("aaaab"));
140     assertEquals(searcher.doc(hits[2].doc).get("field"), ("aaabb"));
141     query = new SlowFuzzyQuery(new Term("field", "aaaac"), SlowFuzzyQuery.defaultMinSimilarity, 2);   
142     hits = searcher.search(query, 1000).scoreDocs;
143     assertEquals(3, hits.length);
144     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaa"));
145     assertEquals(searcher.doc(hits[1].doc).get("field"), ("aaaab"));
146     assertEquals(searcher.doc(hits[2].doc).get("field"), ("aaabb"));
147     query = new SlowFuzzyQuery(new Term("field", "aaaac"), SlowFuzzyQuery.defaultMinSimilarity, 3);   
148     hits = searcher.search(query, 1000).scoreDocs;
149     assertEquals(3, hits.length);
150     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaa"));
151     assertEquals(searcher.doc(hits[1].doc).get("field"), ("aaaab"));
152     assertEquals(searcher.doc(hits[2].doc).get("field"), ("aaabb"));
153     query = new SlowFuzzyQuery(new Term("field", "aaaac"), SlowFuzzyQuery.defaultMinSimilarity, 4);   
154     hits = searcher.search(query, 1000).scoreDocs;
155     assertEquals(2, hits.length);
156     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaa"));
157     assertEquals(searcher.doc(hits[1].doc).get("field"), ("aaaab"));
158     query = new SlowFuzzyQuery(new Term("field", "aaaac"), SlowFuzzyQuery.defaultMinSimilarity, 5);   
159     hits = searcher.search(query, 1000).scoreDocs;
160     assertEquals(0, hits.length);
161     
162 
163     query = new SlowFuzzyQuery(new Term("field", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 0);   
164     hits = searcher.search(query, 1000).scoreDocs;
165     assertEquals(1, hits.length);
166     assertEquals(searcher.doc(hits[0].doc).get("field"), ("ddddd"));
167     
168     // now with prefix
169     query = new SlowFuzzyQuery(new Term("field", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 1);   
170     hits = searcher.search(query, 1000).scoreDocs;
171     assertEquals(1, hits.length);
172     assertEquals(searcher.doc(hits[0].doc).get("field"), ("ddddd"));
173     query = new SlowFuzzyQuery(new Term("field", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 2);   
174     hits = searcher.search(query, 1000).scoreDocs;
175     assertEquals(1, hits.length);
176     assertEquals(searcher.doc(hits[0].doc).get("field"), ("ddddd"));
177     query = new SlowFuzzyQuery(new Term("field", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 3);   
178     hits = searcher.search(query, 1000).scoreDocs;
179     assertEquals(1, hits.length);
180     assertEquals(searcher.doc(hits[0].doc).get("field"), ("ddddd"));
181     query = new SlowFuzzyQuery(new Term("field", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 4);   
182     hits = searcher.search(query, 1000).scoreDocs;
183     assertEquals(1, hits.length);
184     assertEquals(searcher.doc(hits[0].doc).get("field"), ("ddddd"));
185     query = new SlowFuzzyQuery(new Term("field", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 5);   
186     hits = searcher.search(query, 1000).scoreDocs;
187     assertEquals(0, hits.length);
188     
189 
190     // different field = no match:
191     query = new SlowFuzzyQuery(new Term("anotherfield", "ddddX"), SlowFuzzyQuery.defaultMinSimilarity, 0);   
192     hits = searcher.search(query, 1000).scoreDocs;
193     assertEquals(0, hits.length);
194 
195     reader.close();
196     directory.close();
197   }

223   public void testFuzzinessLong() throws Exception {
224     Directory directory = newDirectory();
225     RandomIndexWriter writer = new RandomIndexWriter(random(), directory);
226     addDoc("aaaaaaa", writer);
227     addDoc("segment", writer);
228 
229     IndexReader reader = writer.getReader();
230     IndexSearcher searcher = newSearcher(reader);
231     writer.close();
232 
233     SlowFuzzyQuery query;
234     // not similar enough:
235     query = new SlowFuzzyQuery(new Term("field", "xxxxx"), 0.5f, 0);   
236     ScoreDoc[] hits = searcher.search(query, 1000).scoreDocs;
237     assertEquals(0, hits.length);
238     // edit distance to "aaaaaaa" = 3, this matches because the string is longer than
239     // in testDefaultFuzziness so a bigger difference is allowed:
240     query = new SlowFuzzyQuery(new Term("field", "aaaaccc"), 0.5f, 0);   
241     hits = searcher.search(query, 1000).scoreDocs;
242     assertEquals(1, hits.length);
243     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaaaa"));
244     
245     // now with prefix
246     query = new SlowFuzzyQuery(new Term("field", "aaaaccc"), 0.5f, 1);   
247     hits = searcher.search(query, 1000).scoreDocs;
248     assertEquals(1, hits.length);
249     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaaaa"));
250     query = new SlowFuzzyQuery(new Term("field", "aaaaccc"), 0.5f, 4);   
251     hits = searcher.search(query, 1000).scoreDocs;
252     assertEquals(1, hits.length);
253     assertEquals(searcher.doc(hits[0].doc).get("field"), ("aaaaaaa"));
254     query = new SlowFuzzyQuery(new Term("field", "aaaaccc"), 0.5f, 5);   
255     hits = searcher.search(query, 1000).scoreDocs;
256     assertEquals(0, hits.length);
257 
258     // no match, more than half of the characters is wrong:
259     query = new SlowFuzzyQuery(new Term("field", "aaacccc"), 0.5f, 0);   
260     hits = searcher.search(query, 1000).scoreDocs;
261     assertEquals(0, hits.length);
262     
263     // now with prefix
264     query = new SlowFuzzyQuery(new Term("field", "aaacccc"), 0.5f, 2);   
265     hits = searcher.search(query, 1000).scoreDocs;
266     assertEquals(0, hits.length);
267 
268     // "student" and "stellent" are indeed similar to "segment" by default:
269     query = new SlowFuzzyQuery(new Term("field", "student"), 0.5f, 0);   
270     hits = searcher.search(query, 1000).scoreDocs;
271     assertEquals(1, hits.length);
272     query = new SlowFuzzyQuery(new Term("field", "stellent"), 0.5f, 0);   
273     hits = searcher.search(query, 1000).scoreDocs;
274     assertEquals(1, hits.length);
275     
276     // now with prefix
277     query = new SlowFuzzyQuery(new Term("field", "student"), 0.5f, 1);   
278     hits = searcher.search(query, 1000).scoreDocs;
279     assertEquals(1, hits.length);
280     query = new SlowFuzzyQuery(new Term("field", "stellent"), 0.5f, 1);   
281     hits = searcher.search(query, 1000).scoreDocs;
282     assertEquals(1, hits.length);
283     query = new SlowFuzzyQuery(new Term("field", "student"), 0.5f, 2);   
284     hits = searcher.search(query, 1000).scoreDocs;
285     assertEquals(0, hits.length);
286     query = new SlowFuzzyQuery(new Term("field", "stellent"), 0.5f, 2);   
287     hits = searcher.search(query, 1000).scoreDocs;
288     assertEquals(0, hits.length);
289     
290     // "student" doesn't match anymore thanks to increased minimum similarity:
291     query = new SlowFuzzyQuery(new Term("field", "student"), 0.6f, 0);   
292     hits = searcher.search(query, 1000).scoreDocs;
293     assertEquals(0, hits.length);
294 
295     try {
296       query = new SlowFuzzyQuery(new Term("field", "student"), 1.1f);
297       fail("Expected IllegalArgumentException");
298     } catch (IllegalArgumentException e) {
299       // expecting exception
300     }
301     try {
302       query = new SlowFuzzyQuery(new Term("field", "student"), -0.1f);
303       fail("Expected IllegalArgumentException");
304     } catch (IllegalArgumentException e) {
305       // expecting exception
306     }
307 
308     reader.close();
309     directory.close();
310   }

312   /** 
313    * MultiTermQuery provides (via attribute) information about which values
314    * must be competitive to enter the priority queue. 
315    * 
316    * SlowFuzzyQuery optimizes itself around this information, if the attribute
317    * is not implemented correctly, there will be problems!
318    */
319   public void testTieBreaker() throws Exception {
320     Directory directory = newDirectory();
321     RandomIndexWriter writer = new RandomIndexWriter(random(), directory);
322     addDoc("a123456", writer);
323     addDoc("c123456", writer);
324     addDoc("d123456", writer);
325     addDoc("e123456", writer);
326     
327     Directory directory2 = newDirectory();
328     RandomIndexWriter writer2 = new RandomIndexWriter(random(), directory2);
329     addDoc("a123456", writer2);
330     addDoc("b123456", writer2);
331     addDoc("b123456", writer2);
332     addDoc("b123456", writer2);
333     addDoc("c123456", writer2);
334     addDoc("f123456", writer2);
335     
336     IndexReader ir1 = writer.getReader();
337     IndexReader ir2 = writer2.getReader();
338     
339     MultiReader mr = new MultiReader(ir1, ir2);
340     IndexSearcher searcher = newSearcher(mr);
341     SlowFuzzyQuery fq = new SlowFuzzyQuery(new Term("field", "z123456"), 1f, 0, 2);
342     TopDocs docs = searcher.search(fq, 2);
343     assertEquals(5, docs.totalHits); // 5 docs, from the a and b's
344     mr.close();
345     ir1.close();
346     ir2.close();
347     writer.close();
348     writer2.close();
349     directory.close();
350     directory2.close(); 
351   }

353   public void testTokenLengthOpt() throws IOException {
354     Directory directory = newDirectory();
355     RandomIndexWriter writer = new RandomIndexWriter(random(), directory);
356     addDoc("12345678911", writer);
357     addDoc("segment", writer);
358 
359     IndexReader reader = writer.getReader();
360     IndexSearcher searcher = newSearcher(reader);
361     writer.close();
362 
363     Query query;
364     // term not over 10 chars, so optimization shortcuts
365     query = new SlowFuzzyQuery(new Term("field", "1234569"), 0.9f);
366     ScoreDoc[] hits = searcher.search(query, 1000).scoreDocs;
367     assertEquals(0, hits.length);
368 
369     // 10 chars, so no optimization
370     query = new SlowFuzzyQuery(new Term("field", "1234567891"), 0.9f);
371     hits = searcher.search(query, 1000).scoreDocs;
372     assertEquals(0, hits.length);
373     
374     // over 10 chars, so no optimization
375     query = new SlowFuzzyQuery(new Term("field", "12345678911"), 0.9f);
376     hits = searcher.search(query, 1000).scoreDocs;
377     assertEquals(1, hits.length);
378 
379     // over 10 chars, no match
380     query = new SlowFuzzyQuery(new Term("field", "sdfsdfsdfsdf"), 0.9f);
381     hits = searcher.search(query, 1000).scoreDocs;
382     assertEquals(0, hits.length);
383     
384     reader.close();
385     directory.close();
386   }

388   /** Test the TopTermsBoostOnlyBooleanQueryRewrite rewrite method. */
389   public void testBoostOnlyRewrite() throws Exception {
390     Directory directory = newDirectory();
391     RandomIndexWriter writer = new RandomIndexWriter(random(), directory);
392     addDoc("Lucene", writer);
393     addDoc("Lucene", writer);
394     addDoc("Lucenne", writer);
395 
396     IndexReader reader = writer.getReader();
397     IndexSearcher searcher = newSearcher(reader);
398     writer.close();
399     
400     SlowFuzzyQuery query = new SlowFuzzyQuery(new Term("field", "lucene"));
401     query.setRewriteMethod(new MultiTermQuery.TopTermsBoostOnlyBooleanQueryRewrite(50));
402     ScoreDoc[] hits = searcher.search(query, 1000).scoreDocs;
403     assertEquals(3, hits.length);
404     // normally, 'Lucenne' would be the first result as IDF will skew the score.
405     assertEquals("Lucene", reader.document(hits[0].doc).get("field"));
406     assertEquals("Lucene", reader.document(hits[1].doc).get("field"));
407     assertEquals("Lucenne", reader.document(hits[2].doc).get("field"));
408     reader.close();
409     directory.close();
410   }

448   public void testDistanceAsEditsSearching() throws Exception {
449     Directory index = newDirectory();
450     RandomIndexWriter w = new RandomIndexWriter(random(), index);
451     addDoc("foobar", w);
452     addDoc("test", w);
453     addDoc("working", w);
454     IndexReader reader = w.getReader();
455     IndexSearcher searcher = newSearcher(reader);
456     w.close();
457     
458     SlowFuzzyQuery q = new SlowFuzzyQuery(new Term("field", "fouba"), 2);
459     ScoreDoc[] hits = searcher.search(q, 10).scoreDocs;
460     assertEquals(1, hits.length);
461     assertEquals("foobar", searcher.doc(hits[0].doc).get("field"));
462     
463     q = new SlowFuzzyQuery(new Term("field", "foubara"), 2);
464     hits = searcher.search(q, 10).scoreDocs;
465     assertEquals(1, hits.length);
466     assertEquals("foobar", searcher.doc(hits[0].doc).get("field"));
467     
468     q = new SlowFuzzyQuery(new Term("field", "t"), 3);
469     hits = searcher.search(q, 10).scoreDocs;
470     assertEquals(1, hits.length);
471     assertEquals("test", searcher.doc(hits[0].doc).get("field"));
472     
473     q = new SlowFuzzyQuery(new Term("field", "a"), 4f, 0, 50);
474     hits = searcher.search(q, 10).scoreDocs;
475     assertEquals(1, hits.length);
476     assertEquals("test", searcher.doc(hits[0].doc).get("field"));
477     
478     q = new SlowFuzzyQuery(new Term("field", "a"), 6f, 0, 50);
479     hits = searcher.search(q, 10).scoreDocs;
480     assertEquals(2, hits.length);
481     assertEquals("test", searcher.doc(hits[0].doc).get("field"));
482     assertEquals("foobar", searcher.doc(hits[1].doc).get("field"));
483     
484     reader.close();
485     index.close();
486   }
