36    public void testNormalization() throws IOException {
37      String input = "ʰ㌰゙5℃№㈱㌘，バッファーの正規化のテスト．㋐㋑㋒㋓㋔ｶｷｸｹｺｻﾞｼﾞｽﾞｾﾞｿﾞg̈각/각நிเกषिchkʷक्षि";
38      Normalizer2 normalizer = Normalizer2.getInstance(null, "nfkc_cf", Normalizer2.Mode.COMPOSE);
39      String expectedOutput = normalizer.normalize(input);
40  
41      CharFilter reader = new ICUNormalizer2CharFilter(new StringReader(input), normalizer);
42      char[] tempBuff = new char[10];
43      StringBuilder output = new StringBuilder();
44      while (true) {
45        int length = reader.read(tempBuff);
46        if (length == -1) {
47          break;
48        }
49        output.append(tempBuff, 0, length);
50        assertEquals(output.toString(), normalizer.normalize(input.substring(0, reader.correctOffset(output.length()))));
51      }
52  
53      assertEquals(expectedOutput, output.toString());
54    }

56    public void testTokenStream() throws IOException {
57      // '℃', '№', '㈱', '㌘', 'ｻ'+'<<', 'ｿ'+'<<', '㌰'+'<<'
58      String input = "℃ № ㈱ ㌘ ｻﾞ ｿﾞ ㌰ﾞ";
59  
60      CharFilter reader = new ICUNormalizer2CharFilter(new StringReader(input),
61        Normalizer2.getInstance(null, "nfkc", Normalizer2.Mode.COMPOSE));

63      Tokenizer tokenStream = new MockTokenizer(MockTokenizer.WHITESPACE, false);
64      tokenStream.setReader(reader);
65  
66      assertTokenStreamContents(tokenStream,
67        new String[] {"°C", "No", "(株)", "グラム", "ザ", "ゾ", "ピゴ"},
68        new int[] {0, 2, 4, 6, 8, 11, 14},
69        new int[] {1, 3, 5, 7, 10, 13, 16},
70        input.length());
71    }

108   public void doTestMode(final Normalizer2 normalizer, int maxLength, int iterations, final int bufferSize) throws IOException {
109     Analyzer a = new Analyzer() {
110       @Override
111       protected TokenStreamComponents createComponents(String fieldName) {
112         return new TokenStreamComponents(new MockTokenizer(MockTokenizer.KEYWORD, false));
113       }
114 
115       @Override
116       protected Reader initReader(String fieldName, Reader reader) {
117         return new ICUNormalizer2CharFilter(reader, normalizer, bufferSize);
118       }
119     };
120 
121     for (int i = 0; i < iterations; i++) {
122       String input = TestUtil.randomUnicodeString(random(), maxLength);
123       if (input.length() == 0) {
124         continue;
125       }
126       String normalized = normalizer.normalize(input);
127       if (normalized.length() == 0) {
128         continue; // MockTokenizer doesnt tokenize empty string...
129       }
130       checkOneTerm(a, input, normalized);
131     }
132     a.close();
133   }
