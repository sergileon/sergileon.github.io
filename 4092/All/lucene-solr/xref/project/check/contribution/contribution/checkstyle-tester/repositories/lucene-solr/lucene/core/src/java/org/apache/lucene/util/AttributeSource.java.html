32  /**
33   * An AttributeSource contains a list of different {@link AttributeImpl}s,
34   * and methods to add and get them. There can only be a single instance
35   * of an attribute in the same AttributeSource instance. This is ensured
36   * by passing in the actual type of the Attribute (Class&lt;Attribute&gt;) to 
37   * the {@link #addAttribute(Class)}, which then checks if an instance of
38   * that type is already present. If yes, it returns the instance, otherwise
39   * it creates a new instance and returns it.
40   */
41  public class AttributeSource {
42    
43    /**
44     * This class holds the state of an AttributeSource.
45     * @see #captureState
46     * @see #restoreState
47     */
48    public static final class State implements Cloneable {
49      AttributeImpl attribute;
50      State next;
51      
52      @Override
53      public State clone() {
54        State clone = new State();
55        clone.attribute = attribute.clone();
56        
57        if (next != null) {
58          clone.next = next.clone();
59        }
60        
61        return clone;
62      }
63    }
64      
65    // These two maps must always be in sync!!!
66    // So they are private, final and read-only from the outside (read-only iterators)
67    private final Map<Class<? extends Attribute>, AttributeImpl> attributes;
68    private final Map<Class<? extends AttributeImpl>, AttributeImpl> attributeImpls;
69    private final State[] currentState;
70  
71    private final AttributeFactory factory;
72    
73    /**
74     * An AttributeSource using the default attribute factory {@link AttributeFactory#DEFAULT_ATTRIBUTE_FACTORY}.
75     */
76    public AttributeSource() {
77      this(AttributeFactory.DEFAULT_ATTRIBUTE_FACTORY);
78    }

80    /**
81     * An AttributeSource that uses the same attributes as the supplied one.
82     */
83    public AttributeSource(AttributeSource input) {
84      Objects.requireNonNull(input, "input AttributeSource must not be null");
85      this.attributes = input.attributes;
86      this.attributeImpls = input.attributeImpls;
87      this.currentState = input.currentState;
88      this.factory = input.factory;
89    }

91    /**
92     * An AttributeSource using the supplied {@link AttributeFactory} for creating new {@link Attribute} instances.
93     */
94    public AttributeSource(AttributeFactory factory) {
95      this.attributes = new LinkedHashMap<>();
96      this.attributeImpls = new LinkedHashMap<>();
97      this.currentState = new State[1];
98      this.factory = Objects.requireNonNull(factory, "AttributeFactory must not be null");
99    }

102    * returns the used AttributeFactory.
103    */
104   public final AttributeFactory getAttributeFactory() {
105     return this.factory;

108   /** Returns a new iterator that iterates the attribute classes
109    * in the same order they were added in.
110    */
111   public final Iterator<Class<? extends Attribute>> getAttributeClassesIterator() {
112     return Collections.unmodifiableSet(attributes.keySet()).iterator();
113   }
114   
115   /** Returns a new iterator that iterates all unique Attribute implementations.
116    * This iterator may contain less entries that {@link #getAttributeClassesIterator},
117    * if one instance implements more than one Attribute interface.
118    */
119   public final Iterator<AttributeImpl> getAttributeImplsIterator() {
120     final State initState = getCurrentState();
121     if (initState != null) {
122       return new Iterator<AttributeImpl>() {
123         private State state = initState;
124       
125         @Override
126         public void remove() {
127           throw new UnsupportedOperationException();
128         }
129         
130         @Override
131         public AttributeImpl next() {
132           if (state == null)
133             throw new NoSuchElementException();
134           final AttributeImpl att = state.attribute;
135           state = state.next;
136           return att;
137         }
138         
139         @Override
140         public boolean hasNext() {
141           return state != null;
142         }
143       };
144     } else {
145       return Collections.<AttributeImpl>emptySet().iterator();
146     }
147   }

149   /** a cache that stores all interfaces for known implementation classes for performance (slow reflection) */
150   private static final ClassValue<Class<? extends Attribute>[]> implInterfaces = new ClassValue<Class<? extends Attribute>[]>() {
151     @Override
152     protected Class<? extends Attribute>[] computeValue(Class<?> clazz) {
153       final Set<Class<? extends Attribute>> intfSet = new LinkedHashSet<>();
154       // find all interfaces that this attribute instance implements
155       // and that extend the Attribute interface
156       do {
157         for (Class<?> curInterface : clazz.getInterfaces()) {
158           if (curInterface != Attribute.class && Attribute.class.isAssignableFrom(curInterface)) {
159             intfSet.add(curInterface.asSubclass(Attribute.class));
160           }
161         }
162         clazz = clazz.getSuperclass();
163       } while (clazz != null);
164       @SuppressWarnings({"unchecked", "rawtypes"}) final Class<? extends Attribute>[] a =
165           intfSet.toArray(new Class[intfSet.size()]);
166       return a;
167     }
168   };
169   
170   static Class<? extends Attribute>[] getAttributeInterfaces(final Class<? extends AttributeImpl> clazz) {
171     return implInterfaces.get(clazz);
172   }
173   
174   /** <b>Expert:</b> Adds a custom AttributeImpl instance with one or more Attribute interfaces.
175    * <p><b>NOTE:</b> It is not guaranteed, that <code>att</code> is added to
176    * the <code>AttributeSource</code>, because the provided attributes may already exist.
177    * You should always retrieve the wanted attributes using {@link #getAttribute} after adding
178    * with this method and cast to your class.
179    * The recommended way to use custom implementations is using an {@link AttributeFactory}.
180    * </p>
181    */
182   public final void addAttributeImpl(final AttributeImpl att) {
183     final Class<? extends AttributeImpl> clazz = att.getClass();
184     if (attributeImpls.containsKey(clazz)) return;
185     
186     // add all interfaces of this AttributeImpl to the maps
187     for (final Class<? extends Attribute> curInterface : getAttributeInterfaces(clazz)) {
188       // Attribute is a superclass of this interface
189       if (!attributes.containsKey(curInterface)) {
190         // invalidate state to force recomputation in captureState()
191         this.currentState[0] = null;
192         attributes.put(curInterface, att);
193         attributeImpls.put(clazz, att);
194       }
195     }
196   }

198   /**
199    * The caller must pass in a Class&lt;? extends Attribute&gt; value.
200    * This method first checks if an instance of that class is 
201    * already in this AttributeSource and returns it. Otherwise a
202    * new instance is created, added to this AttributeSource and returned. 
203    */
204   public final <T extends Attribute> T addAttribute(Class<T> attClass) {
205     AttributeImpl attImpl = attributes.get(attClass);
206     if (attImpl == null) {
207       if (!(attClass.isInterface() && Attribute.class.isAssignableFrom(attClass))) {
208         throw new IllegalArgumentException(
209           "addAttribute() only accepts an interface that extends Attribute, but " +
210           attClass.getName() + " does not fulfil this contract."
211         );
212       }
213       addAttributeImpl(attImpl = this.factory.createAttributeInstance(attClass));
214     }
215     return attClass.cast(attImpl);
216   }
217   
218   /** Returns true, iff this AttributeSource has any attributes */
219   public final boolean hasAttributes() {
220     return !this.attributes.isEmpty();
221   }
222 
223   /**
224    * The caller must pass in a Class&lt;? extends Attribute&gt; value. 
225    * Returns true, iff this AttributeSource contains the passed-in Attribute.
226    */
227   public final boolean hasAttribute(Class<? extends Attribute> attClass) {
228     return this.attributes.containsKey(attClass);
229   }
230 
231   /**
232    * Returns the instance of the passed in Attribute contained in this AttributeSource
233    * <p>
234    * The caller must pass in a Class&lt;? extends Attribute&gt; value. 
235    * 
236    * @return instance of the passed in Attribute, or {@code null} if this AttributeSource 
237    *         does not contain the Attribute. It is recommended to always use 
238    *         {@link #addAttribute} even in consumers  of TokenStreams, because you cannot 
239    *         know if a specific TokenStream really uses a specific Attribute. 
240    *         {@link #addAttribute} will automatically make the attribute available. 
241    *         If you want to only use the attribute, if it is available (to optimize
242    *         consuming), use {@link #hasAttribute}.
243    */
244   public final <T extends Attribute> T getAttribute(Class<T> attClass) {
245     return attClass.cast(attributes.get(attClass));
246   }

248   private State getCurrentState() {
249     State s  = currentState[0];
250     if (s != null || !hasAttributes()) {
251       return s;
252     }
253     State c = s = currentState[0] = new State();
254     final Iterator<AttributeImpl> it = attributeImpls.values().iterator();
255     c.attribute = it.next();
256     while (it.hasNext()) {
257       c.next = new State();
258       c = c.next;
259       c.attribute = it.next();
260     }
261     return s;
262   }

264   /**
265    * Resets all Attributes in this AttributeSource by calling
266    * {@link AttributeImpl#clear()} on each Attribute implementation.
267    */
268   public final void clearAttributes() {
269     for (State state = getCurrentState(); state != null; state = state.next) {
270       state.attribute.clear();
271     }
272   }
273   
274   /**
275    * Captures the state of all Attributes. The return value can be passed to
276    * {@link #restoreState} to restore the state of this or another AttributeSource.
277    */
278   public final State captureState() {
279     final State state = this.getCurrentState();
280     return (state == null) ? null : state.clone();
281   }

283   /**
284    * Restores this state by copying the values of all attribute implementations
285    * that this state contains into the attributes implementations of the targetStream.
286    * The targetStream must contain a corresponding instance for each argument
287    * contained in this state (e.g. it is not possible to restore the state of
288    * an AttributeSource containing a TermAttribute into a AttributeSource using
289    * a Token instance as implementation).
290    * <p>
291    * Note that this method does not affect attributes of the targetStream
292    * that are not contained in this state. In other words, if for example
293    * the targetStream contains an OffsetAttribute, but this state doesn't, then
294    * the value of the OffsetAttribute remains unchanged. It might be desirable to
295    * reset its value to the default, in which case the caller should first
296    * call {@link TokenStream#clearAttributes()} on the targetStream.   
297    */
298   public final void restoreState(State state) {
299     if (state == null)  return;
300     
301     do {
302       AttributeImpl targetImpl = attributeImpls.get(state.attribute.getClass());
303       if (targetImpl == null) {
304         throw new IllegalArgumentException("State contains AttributeImpl of type " +
305           state.attribute.getClass().getName() + " that is not in in this AttributeSource");
306       }
307       state.attribute.copyTo(targetImpl);
308       state = state.next;
309     } while (state != null);
310   }

312   @Override
313   public int hashCode() {
314     int code = 0;
315     for (State state = getCurrentState(); state != null; state = state.next) {
316       code = code * 31 + state.attribute.hashCode();
317     }
318     return code;
319   }

321   @Override
322   public boolean equals(Object obj) {
323     if (obj == this) {
324       return true;
325     }
326 
327     if (obj instanceof AttributeSource) {
328       AttributeSource other = (AttributeSource) obj;  
329     
330       if (hasAttributes()) {
331         if (!other.hasAttributes()) {
332           return false;
333         }
334         
335         if (this.attributeImpls.size() != other.attributeImpls.size()) {
336           return false;
337         }
338   
339         // it is only equal if all attribute impls are the same in the same order
340         State thisState = this.getCurrentState();
341         State otherState = other.getCurrentState();
342         while (thisState != null && otherState != null) {
343           if (otherState.attribute.getClass() != thisState.attribute.getClass() || !otherState.attribute.equals(thisState.attribute)) {
344             return false;
345           }
346           thisState = thisState.next;
347           otherState = otherState.next;
348         }
349         return true;
350       } else {
351         return !other.hasAttributes();
352       }
353     } else
354       return false;
355   }

357   /**
358    * This method returns the current attribute values as a string in the following format
359    * by calling the {@link #reflectWith(AttributeReflector)} method:
360    * 
361    * <ul>
362    * <li><em>iff {@code prependAttClass=true}:</em> {@code "AttributeClass#key=value,AttributeClass#key=value"}
363    * <li><em>iff {@code prependAttClass=false}:</em> {@code "key=value,key=value"}
364    * </ul>
365    *
366    * @see #reflectWith(AttributeReflector)
367    */
368   public final String reflectAsString(final boolean prependAttClass) {
369     final StringBuilder buffer = new StringBuilder();
370     reflectWith(new AttributeReflector() {
371       @Override
372       public void reflect(Class<? extends Attribute> attClass, String key, Object value) {
373         if (buffer.length() > 0) {
374           buffer.append(',');
375         }
376         if (prependAttClass) {
377           buffer.append(attClass.getName()).append('#');
378         }
379         buffer.append(key).append('=').append((value == null) ? "null" : value);
380       }
381     });
382     return buffer.toString();
383   }

385   /**
386    * This method is for introspection of attributes, it should simply
387    * add the key/values this AttributeSource holds to the given {@link AttributeReflector}.
388    *
389    * <p>This method iterates over all Attribute implementations and calls the
390    * corresponding {@link AttributeImpl#reflectWith} method.</p>
391    *
392    * @see AttributeImpl#reflectWith
393    */
394   public final void reflectWith(AttributeReflector reflector) {
395     for (State state = getCurrentState(); state != null; state = state.next) {
396       state.attribute.reflectWith(reflector);
397     }
398   }

400   /**
401    * Performs a clone of all {@link AttributeImpl} instances returned in a new
402    * {@code AttributeSource} instance. This method can be used to e.g. create another TokenStream
403    * with exactly the same attributes (using {@link #AttributeSource(AttributeSource)}).
404    * You can also use it as a (non-performant) replacement for {@link #captureState}, if you need to look
405    * into / modify the captured state.
406    */
407   public final AttributeSource cloneAttributes() {
408     final AttributeSource clone = new AttributeSource(this.factory);
409     
410     if (hasAttributes()) {
411       // first clone the impls
412       for (State state = getCurrentState(); state != null; state = state.next) {
413         clone.attributeImpls.put(state.attribute.getClass(), state.attribute.clone());
414       }
415       
416       // now the interfaces
417       for (Entry<Class<? extends Attribute>, AttributeImpl> entry : this.attributes.entrySet()) {
418         clone.attributes.put(entry.getKey(), clone.attributeImpls.get(entry.getValue().getClass()));
419       }
420     }
421     
422     return clone;
423   }
424   
425   /**
426    * Copies the contents of this {@code AttributeSource} to the given target {@code AttributeSource}.
427    * The given instance has to provide all {@link Attribute}s this instance contains. 
428    * The actual attribute implementations must be identical in both {@code AttributeSource} instances;
429    * ideally both AttributeSource instances should use the same {@link AttributeFactory}.
430    * You can use this method as a replacement for {@link #restoreState}, if you use
431    * {@link #cloneAttributes} instead of {@link #captureState}.
432    */
433   public final void copyTo(AttributeSource target) {
434     for (State state = getCurrentState(); state != null; state = state.next) {
435       final AttributeImpl targetImpl = target.attributeImpls.get(state.attribute.getClass());
436       if (targetImpl == null) {
437         throw new IllegalArgumentException("This AttributeSource contains AttributeImpl of type " +
438           state.attribute.getClass().getName() + " that is not in the target");
439       }
440       state.attribute.copyTo(targetImpl);
441     }
442   }

444   /**
445    * Returns a string consisting of the class's simple name, the hex representation of the identity hash code,
446    * and the current reflection of all attributes.
447    * @see #reflectAsString(boolean)
448    */
449   @Override
450   public String toString() {
451     return getClass().getSimpleName() + '@' + Integer.toHexString(System.identityHashCode(this)) + " " + reflectAsString(false);
452   }
