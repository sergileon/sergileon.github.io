36  /**
37   * <code>BitDocSet</code> represents an unordered set of Lucene Document Ids
38   * using a BitSet.  A set bit represents inclusion in the set for that document.
39   *
40   * @since solr 0.9
41   */
42  public class BitDocSet extends DocSetBase {
43    private static final long BASE_RAM_BYTES_USED = RamUsageEstimator.shallowSizeOfInstance(BitDocSet.class)
44        + RamUsageEstimator.shallowSizeOfInstance(FixedBitSet.class)
45        + RamUsageEstimator.NUM_BYTES_ARRAY_HEADER;  // for the array object inside the FixedBitSet. long[] array won't change alignment, so no need to calculate it.
46  
47    final FixedBitSet bits;
48    int size;    // number of docs in the set (cached for perf)
49  
50    public BitDocSet() {
51      bits = new FixedBitSet(64);
52    }
53  
54    /**
55     * Construct a BitDocSet. The capacity of the {@link FixedBitSet} should be at
56     * least maxDoc()
57     */
58    public BitDocSet(FixedBitSet bits) {
59      this.bits = bits;
60      size=-1;
61    }
62  
63    /**
64     * Construct a BitDocSet, and provides the number of set bits. The capacity of
65     * the {@link FixedBitSet} should be at least maxDoc()
66     */
67    public BitDocSet(FixedBitSet bits, int size) {
68      this.bits = bits;
69      this.size = size;
70    }

101   @Override
102   public DocIterator iterator() {
103     return new DocIterator() {
104       private final BitSetIterator iter = new BitSetIterator(bits, 0L); // cost is not useful here
105       private int pos = iter.nextDoc();
106       @Override
107       public boolean hasNext() {
108         return pos != DocIdSetIterator.NO_MORE_DOCS;
109       }
110 
111       @Override
112       public Integer next() {
113         return nextDoc();
114       }
115 
116       @Override
117       public void remove() {
118         bits.clear(pos);
119       }
120 
121       @Override
122       public int nextDoc() {
123         int old=pos;
124         pos=iter.nextDoc();
125         return old;
126       }
127 
128       @Override
129       public float score() {
130         return 0.0f;
131       }
132     };
133   }

135   /**
136    * @return the <b>internal</b> {@link FixedBitSet} that should <b>not</b> be modified.
137    */
138   @Override
139   public FixedBitSet getBits() {
140     return bits;
141   }
142 
143   @Override
144   public void add(int doc) {
145     bits.set(doc);
146     size=-1;  // invalidate size
147   }
148 
149   @Override
150   public void addUnique(int doc) {
151     bits.set(doc);
152     size=-1;  // invalidate size
153   }
154 
155   @Override
156   public int size() {
157     if (size!=-1) return size;
158     return size = bits.cardinality();
159   }
160 
161   /**
162    * The number of set bits - size - is cached.  If the bitset is changed externally,
163    * this method should be used to invalidate the previously cached size.
164    */
165   public void invalidateSize() {
166     size=-1;
167   }
168 
169   /**
170    * Returns true of the doc exists in the set. Should only be called when doc &lt;
171    * {@link FixedBitSet#length()}.
172    */
173   @Override
174   public boolean exists(int doc) {
175     return bits.get(doc);
176   }
177 
178   @Override
179   public int intersectionSize(DocSet other) {
180     if (other instanceof BitDocSet) {
181       return (int) FixedBitSet.intersectionCount(this.bits, ((BitDocSet) other).bits);
182     } else {
183       // they had better not call us back!
184       return other.intersectionSize(this);
185     }
186   }
187 
188   @Override
189   public boolean intersects(DocSet other) {
190     if (other instanceof BitDocSet) {
191       return bits.intersects(((BitDocSet)other).bits);
192     } else {
193       // they had better not call us back!
194       return other.intersects(this);
195     }
196   }
197 
198   @Override
199   public int unionSize(DocSet other) {
200     if (other instanceof BitDocSet) {
201       // if we don't know our current size, this is faster than
202       // size + other.size - intersection_size
203       return (int) FixedBitSet.unionCount(this.bits, ((BitDocSet) other).bits);
204     } else {
205       // they had better not call us back!
206       return other.unionSize(this);
207     }
208   }
209 
210   @Override
211   public int andNotSize(DocSet other) {
212     if (other instanceof BitDocSet) {
213       // if we don't know our current size, this is faster than
214       // size - intersection_size
215       return (int) FixedBitSet.andNotCount(this.bits, ((BitDocSet)other).bits);
216     } else {
217       return super.andNotSize(other);
218     }
219   }
220 
221   @Override
222   public void addAllTo(DocSet target) {
223     if (target instanceof BitDocSet) {
224       ((BitDocSet) target).bits.or(bits);
225     } else {
226       super.addAllTo(target);
227     }
228   }
229 
230   @Override
231   public DocSet andNot(DocSet other) {
232     FixedBitSet newbits = bits.clone();
233     if (other instanceof BitDocSet) {
234       newbits.andNot(((BitDocSet) other).bits);
235     } else {
236       DocIterator iter = other.iterator();
237       while (iter.hasNext()) {
238         int doc = iter.nextDoc();
239         if (doc < newbits.length()) {
240           newbits.clear(doc);
241         }
242       }
243     }
244     return new BitDocSet(newbits);
245   }
246   
247   @Override
248   public DocSet union(DocSet other) {
249     FixedBitSet newbits = bits.clone();
250     if (other instanceof BitDocSet) {
251       BitDocSet otherDocSet = (BitDocSet) other;
252       newbits = FixedBitSet.ensureCapacity(newbits, otherDocSet.bits.length());
253       newbits.or(otherDocSet.bits);
254     } else {
255       DocIterator iter = other.iterator();
256       while (iter.hasNext()) {
257         int doc = iter.nextDoc();
258         newbits = FixedBitSet.ensureCapacity(newbits, doc);
259         newbits.set(doc);
260       }
261     }
262     return new BitDocSet(newbits);
263   }
264   
265   @Override
266   protected BitDocSet clone() {
267     return new BitDocSet(bits.clone(), size);
268   }

270   @Override
271   public Filter getTopFilter() {
272     final FixedBitSet bs = bits;
273     // TODO: if cardinality isn't cached, do a quick measure of sparseness
274     // and return null from bits() if too sparse.
275 
276     return new Filter() {
277       @Override
278       public DocIdSet getDocIdSet(final LeafReaderContext context, final Bits acceptDocs) {
279         LeafReader reader = context.reader();
280         // all Solr DocSets that are used as filters only include live docs
281         final Bits acceptDocs2 = acceptDocs == null ? null : (reader.getLiveDocs() == acceptDocs ? null : acceptDocs);
282 
283         if (context.isTopLevel) {
284           return BitsFilteredDocIdSet.wrap(new BitDocIdSet(bs), acceptDocs);
285         }
286 
287         final int base = context.docBase;
288         final int maxDoc = reader.maxDoc();
289         final int max = base + maxDoc;   // one past the max doc in this segment.
290 
291         return BitsFilteredDocIdSet.wrap(new DocIdSet() {
292           @Override
293           public DocIdSetIterator iterator() {
294             return new DocIdSetIterator() {
295               int pos = base - 1;
296               int adjustedDoc = -1;
297 
298               @Override
299               public int docID() {
300                 return adjustedDoc;
301               }
302 
303               @Override
304               public int nextDoc() {
305                 if (pos >= bs.length() - 1) {
306                   return adjustedDoc = NO_MORE_DOCS;
307                 } else {
308                   pos = bs.nextSetBit(pos + 1);
309                   return adjustedDoc = pos < max ? pos - base : NO_MORE_DOCS;
310                 }
311               }
312 
313               @Override
314               public int advance(int target) {
315                 if (target == NO_MORE_DOCS) return adjustedDoc = NO_MORE_DOCS;
316                 int adjusted = target + base;
317                 if (adjusted >= bs.length()) {
318                   return adjustedDoc = NO_MORE_DOCS;
319                 } else {
320                   pos = bs.nextSetBit(adjusted);
321                   return adjustedDoc = pos < max ? pos - base : NO_MORE_DOCS;
322                 }
323               }
324 
325               @Override
326               public long cost() {
327                 // we don't want to actually compute cardinality, but
328                 // if it's already been computed, we use it (pro-rated for the segment)
329                 if (size != -1) {
330                   return (long)(size * ((FixedBitSet.bits2words(maxDoc)<<6) / (float)bs.length()));
331                 } else {
332                   return maxDoc;
333                 }
334               }
335             };
336           }
337 
338           @Override
339           public boolean isCacheable() {
340             return true;
341           }
342 
343           @Override
344           public long ramBytesUsed() {
345             return bs.ramBytesUsed();
346           }
347 
348           @Override
349           public Bits bits() {
350             return new Bits() {
351               @Override
352               public boolean get(int index) {
353                 return bs.get(index + base);
354               }
355 
356               @Override
357               public int length() {
358                 return maxDoc;
359               }
360             };
361           }
362 
363         }, acceptDocs2);
364       }
365       @Override
366       public String toString(String field) {
367         return "BitSetDocTopFilter";
368       }
369     };
370   }
371 
372   @Override
373   public long ramBytesUsed() {
374     return BASE_RAM_BYTES_USED + ((long)bits.getBits().length << 3);
375   }
376 
377   @Override
378   public Collection<Accountable> getChildResources() {
379     return Collections.emptyList();
380   }
