166     try {

170       if (loader instanceof ZkSolrResourceLoader) {
171         resource = name;
172       } else {

180    * Creates a configuration instance from a resource loader, a configuration name and a stream.
181    * If the stream is null, the resource loader will open the configuration stream.
182    * If the stream is not null, no attempt to load the resource will occur (the name is not used).
183    *
184    * @param loader the resource loader
185    * @param name   the configuration name
186    * @param is     the configuration stream
187    */
188   public SolrConfig(SolrResourceLoader loader, String name, InputSource is)
189       throws ParserConfigurationException, IOException, SAXException {
190     super(loader, name, is, "/config/");
191     getOverlay();//just in case it is not initialized
192     getRequestParams();
193     initLibs();
194     luceneMatchVersion = getLuceneVersion("luceneMatchVersion");
195     String indexConfigPrefix;
196 
197     // Old indexDefaults and mainIndex sections are deprecated and fails fast for luceneMatchVersion=>LUCENE_4_0_0.
198     // For older solrconfig.xml's we allow the old sections, but never mixed with the new <indexConfig>
199     boolean hasDeprecatedIndexConfig = (getNode("indexDefaults", false) != null) || (getNode("mainIndex", false) != null);
200     boolean hasNewIndexConfig = getNode("indexConfig", false) != null;
201     if (hasDeprecatedIndexConfig) {
202       if(luceneMatchVersion.onOrAfter(Version.LUCENE_4_0_0_ALPHA)) {
203         throw new SolrException(ErrorCode.FORBIDDEN, "<indexDefaults> and <mainIndex> configuration sections are discontinued. Use <indexConfig> instead.");
204       } else {
205         // Still allow the old sections for older LuceneMatchVersion's
206         if(hasNewIndexConfig) {
207           throw new SolrException(ErrorCode.FORBIDDEN, "Cannot specify both <indexDefaults>, <mainIndex> and <indexConfig> at the same time. Please use <indexConfig> only.");
208         }
209         log.warn("<indexDefaults> and <mainIndex> configuration sections are deprecated and will fail for luceneMatchVersion=LUCENE_4_0_0 and later. Please use <indexConfig> instead.");
210         defaultIndexConfig = new SolrIndexConfig(this, "indexDefaults", null);
211         mainIndexConfig = new SolrIndexConfig(this, "mainIndex", defaultIndexConfig);
212         indexConfigPrefix = "mainIndex";
213       }
214     } else {
215       defaultIndexConfig = mainIndexConfig = null;
216       indexConfigPrefix = "indexConfig";
217     }
218     assertWarnOrFail("The <nrtMode> config has been discontinued and NRT mode is always used by Solr." +
219             " This config will be removed in future versions.", getNode(indexConfigPrefix + "/nrtMode", false) == null,
220         false
221     );
222     assertWarnOrFail("Solr no longer supports forceful unlocking via the 'unlockOnStartup' option.  "+
223                      "This is no longer neccessary for the default lockType except in situations where "+
224                      "it would be dangerous and should not be done.  For other lockTypes and/or "+
225                      "directoryFactory options it may also be dangerous and users must resolve "+
226                      "problematic locks manually.",
227                      null == getNode(indexConfigPrefix + "/unlockOnStartup", false),
228                      false // warn in 5x
229                      );
230                      
231     // Parse indexConfig section, using mainIndex as backup in case old config is used
232     indexConfig = new SolrIndexConfig(this, "indexConfig", mainIndexConfig);
233 
234     booleanQueryMaxClauseCount = getInt("query/maxBooleanClauses", BooleanQuery.getMaxClauseCount());
235     log.info("Using Lucene MatchVersion: " + luceneMatchVersion);
236 
237     // Warn about deprecated / discontinued parameters
238     // boolToFilterOptimizer has had no effect since 3.1
239     if (get("query/boolTofilterOptimizer", null) != null)
240       log.warn("solrconfig.xml: <boolTofilterOptimizer> is currently not implemented and has no effect.");
241     if (get("query/HashDocSet", null) != null)
242       log.warn("solrconfig.xml: <HashDocSet> is deprecated and no longer recommended used.");
243 
244 // TODO: Old code - in case somebody wants to re-enable. Also see SolrIndexSearcher#search()
245 //    filtOptEnabled = getBool("query/boolTofilterOptimizer/@enabled", false);
246 //    filtOptCacheSize = getInt("query/boolTofilterOptimizer/@cacheSize",32);
247 //    filtOptThreshold = getFloat("query/boolTofilterOptimizer/@threshold",.05f);
248 
249     useFilterForSortedQuery = getBool("query/useFilterForSortedQuery", false);
250     queryResultWindowSize = Math.max(1, getInt("query/queryResultWindowSize", 1));
251     queryResultMaxDocsCached = getInt("query/queryResultMaxDocsCached", Integer.MAX_VALUE);
252     enableLazyFieldLoading = getBool("query/enableLazyFieldLoading", false);
253 
254 
255     filterCacheConfig = CacheConfig.getConfig(this, "query/filterCache");
256     queryResultCacheConfig = CacheConfig.getConfig(this, "query/queryResultCache");
257     documentCacheConfig = CacheConfig.getConfig(this, "query/documentCache");
258     CacheConfig conf = CacheConfig.getConfig(this, "query/fieldValueCache");
259     if (conf == null) {
260       Map<String, String> args = new HashMap<>();
261       args.put(NAME, "fieldValueCache");
262       args.put("size", "10000");
263       args.put("initialSize", "10");
264       args.put("showItems", "-1");
265       conf = new CacheConfig(FastLRUCache.class, args, null);
266     }
267     fieldValueCacheConfig = conf;
268     useColdSearcher = getBool("query/useColdSearcher", false);
269     dataDir = get("dataDir", null);
270     if (dataDir != null && dataDir.length() == 0) dataDir = null;
271 
272     userCacheConfigs = CacheConfig.getMultipleConfigs(this, "query/cache");
273 
274     org.apache.solr.search.SolrIndexSearcher.initRegenerators(this);
275 
276     hashSetInverseLoadFactor = 1.0f / getFloat("//HashDocSet/@loadFactor", 0.75f);
277     hashDocSetMaxSize = getInt("//HashDocSet/@maxSize", 3000);
278 
279     httpCachingConfig = new HttpCachingConfig(this);
280 
281     Node jmx = getNode("jmx", false);
282     if (jmx != null) {
283       jmxConfig = new JmxConfiguration(true,
284           get("jmx/@agentId", null),
285           get("jmx/@serviceUrl", null),
286           get("jmx/@rootName", null));
287 
288     } else {
289       jmxConfig = new JmxConfiguration(false, null, null, null);
290     }
291     maxWarmingSearchers = getInt("query/maxWarmingSearchers", Integer.MAX_VALUE);
292     slowQueryThresholdMillis = getInt("query/slowQueryThresholdMillis", -1);
293     for (SolrPluginInfo plugin : plugins) loadPluginInfo(plugin);
294     updateHandlerInfo = loadUpdatehandlerInfo();
295 
296     multipartUploadLimitKB = getInt(
297         "requestDispatcher/requestParsers/@multipartUploadLimitInKB", 2048);
298 
299     formUploadLimitKB = getInt(
300         "requestDispatcher/requestParsers/@formdataUploadLimitInKB", 2048);
301 
302     enableRemoteStreams = getBool(
303         "requestDispatcher/requestParsers/@enableRemoteStreaming", false);
304 
305     // Let this filter take care of /select?xxx format
306     handleSelect = getBool(
307         "requestDispatcher/@handleSelect", true);
308 
309     addHttpRequestToContext = getBool(
310         "requestDispatcher/requestParsers/@addHttpRequestToContext", false);
311 
312     List<PluginInfo> argsInfos = getPluginInfos(InitParams.class.getName());
313     if (argsInfos != null) {
314       Map<String, InitParams> argsMap = new HashMap<>();
315       for (PluginInfo p : argsInfos) {
316         InitParams args = new InitParams(p);
317         argsMap.put(args.name == null ? String.valueOf(args.hashCode()) : args.name, args);
318       }
319       this.initParams = Collections.unmodifiableMap(argsMap);
320 
321     }
322 
323     solrRequestParsers = new SolrRequestParsers(this);
324     log.info("Loaded SolrConfig: " + name);
325   }

389   public static ConfigOverlay getConfigOverlay(SolrResourceLoader loader) {
390     InputStream in = null;
391     InputStreamReader isr = null;
392     try {
393       try {
394         in = loader.openResource(ConfigOverlay.RESOURCE_NAME);
395       } catch (IOException e) {
396         // TODO: we should be explicitly looking for file not found exceptions
397         // and logging if it's not the expected IOException
398         // hopefully no problem, assume no overlay.json file
399         return new ConfigOverlay(Collections.EMPTY_MAP, -1);
400       }
401       
402       int version = 0; // will be always 0 for file based resourceLoader
403       if (in instanceof ZkSolrResourceLoader.ZkByteArrayInputStream) {
404         version = ((ZkSolrResourceLoader.ZkByteArrayInputStream) in).getStat().getVersion();
405         log.info("config overlay loaded . version : {} ", version);
406       }
407       isr = new InputStreamReader(in, StandardCharsets.UTF_8);
408       Map m = (Map) ObjectBuilder.getVal(new JSONParser(isr));
409       return new ConfigOverlay(m, version);
410     } catch (Exception e) {
411       throw new SolrException(ErrorCode.SERVER_ERROR, "Error reading config overlay", e);
412     } finally {
413       IOUtils.closeQuietly(isr);
414       IOUtils.closeQuietly(in);
415     }
416   }

435   private void loadPluginInfo(SolrPluginInfo pluginInfo) {
436     boolean requireName = pluginInfo.options.contains(REQUIRE_NAME);
437     boolean requireClass = pluginInfo.options.contains(REQUIRE_CLASS);
438 
439     List<PluginInfo> result = readPluginInfos(pluginInfo.tag, requireName, requireClass);
440 
441     if (1 < result.size() && !pluginInfo.options.contains(MULTI_OK)) {
442       throw new SolrException
443           (SolrException.ErrorCode.SERVER_ERROR,
444               "Found " + result.size() + " configuration sections when at most "
445                   + "1 is allowed matching expression: " + pluginInfo.getCleanTag());
446     }
447     if (!result.isEmpty()) pluginStore.put(pluginInfo.clazz.getName(), result);
448   }

450   public List<PluginInfo> readPluginInfos(String tag, boolean requireName, boolean requireClass) {
451     ArrayList<PluginInfo> result = new ArrayList<>();
452     NodeList nodes = (NodeList) evaluate(tag, XPathConstants.NODESET);
453     for (int i = 0; i < nodes.getLength(); i++) {
454       PluginInfo pluginInfo = new PluginInfo(nodes.item(i), "[solrconfig.xml] " + tag, requireName, requireClass);
455       if (pluginInfo.isEnabled()) result.add(pluginInfo);
456     }
457     return result;
458   }

511   public static class JmxConfiguration implements MapSerializable {
512     public boolean enabled = false;
513     public String agentId;
514     public String serviceUrl;
515     public String rootName;
516 
517     public JmxConfiguration(boolean enabled,
518                             String agentId,
519                             String serviceUrl,
520                             String rootName) {
521       this.enabled = enabled;
522       this.agentId = agentId;
523       this.serviceUrl = serviceUrl;
524       this.rootName = rootName;
525 
526       if (agentId != null && serviceUrl != null) {
527         throw new SolrException
528             (SolrException.ErrorCode.SERVER_ERROR,
529                 "Incorrect JMX Configuration in solrconfig.xml, " +
530                     "both agentId and serviceUrl cannot be specified at the same time");
531       }
532 
533     }
534 
535     @Override
536     public Map<String, Object> toMap() {
537       LinkedHashMap map = new LinkedHashMap();
538       map.put("agentId", agentId);
539       map.put("serviceUrl", serviceUrl);
540       map.put("rootName", rootName);
541       return map;
542     }

589     private HttpCachingConfig(SolrConfig conf) {
590 
591       never304 = conf.getBool(CACHE_PRE + "@never304", false);
592 
593       etagSeed = conf.get(CACHE_PRE + "@etagSeed", "Solr");
594 
595 
596       lastModFrom = LastModFrom.parse(conf.get(CACHE_PRE + "@lastModFrom",
597           "openTime"));
598 
599       cacheControlHeader = conf.get(CACHE_PRE + "cacheControl", null);
600 
601       Long tmp = null; // maxAge
602       if (null != cacheControlHeader) {
603         try {
604           final Matcher ttlMatcher = MAX_AGE.matcher(cacheControlHeader);
605           final String ttlStr = ttlMatcher.find() ? ttlMatcher.group(1) : null;
606           tmp = (null != ttlStr && !"".equals(ttlStr))
607               ? Long.valueOf(ttlStr)
608               : null;
609         } catch (Exception e) {
610           log.warn("Ignoring exception while attempting to " +
611               "extract max-age from cacheControl config: " +
612               cacheControlHeader, e);
613         }
614       }
615       maxAge = tmp;
616 
617     }

700   /**
701    * SolrConfig keeps a repository of plugins by the type. The known interfaces are the types.
702    *
703    * @param type The key is FQN of the plugin class there are a few  known types : SolrFormatter, SolrFragmenter
704    *             SolrRequestHandler,QParserPlugin, QueryResponseWriter,ValueSourceParser,
705    *             SearchComponent, QueryConverter, SolrEventListener, DirectoryFactory,
706    *             IndexDeletionPolicy, IndexReaderFactory, {@link TransformerFactory}
707    */
708   public List<PluginInfo> getPluginInfos(String type) {
709     List<PluginInfo> result = pluginStore.get(type);
710     SolrPluginInfo info = classVsSolrPluginInfo.get(type);
711     if (info != null &&
712         (info.options.contains(REQUIRE_NAME) || info.options.contains(REQUIRE_NAME_IN_OVERLAY))) {
713       Map<String, Map> infos = overlay.getNamedPlugins(info.getCleanTag());
714       if (!infos.isEmpty()) {
715         LinkedHashMap<String, PluginInfo> map = new LinkedHashMap<>();
716         if (result != null) for (PluginInfo pluginInfo : result) {
717           //just create a UUID for the time being so that map key is not null
718           String name = pluginInfo.name == null ?
719               UUID.randomUUID().toString().toLowerCase(Locale.ROOT) :
720               pluginInfo.name;
721           map.put(name, pluginInfo);
722         }
723         for (Map.Entry<String, Map> e : infos.entrySet()) {
724           map.put(e.getKey(), new PluginInfo(info.getCleanTag(), e.getValue()));
725         }
726         result = new ArrayList<>(map.values());
727       }
728     }
729     return result == null ? Collections.<PluginInfo>emptyList() : result;
730   }

745   private void initLibs() {
746     NodeList nodes = (NodeList) evaluate("lib", XPathConstants.NODESET);
747     if (nodes == null || nodes.getLength() == 0) return;
748 
749     log.info("Adding specified lib dirs to ClassLoader");
750     SolrResourceLoader loader = getResourceLoader();
751     List<URL> urls = new ArrayList<>();
752 
753     for (int i = 0; i < nodes.getLength(); i++) {
754       Node node = nodes.item(i);
755       String baseDir = DOMUtil.getAttr(node, "dir");
756       String path = DOMUtil.getAttr(node, PATH);
757       if (null != baseDir) {
758         // :TODO: add support for a simpler 'glob' mutually exclusive of regex
759         Path dir = loader.getInstancePath().resolve(baseDir);
760         String regex = DOMUtil.getAttr(node, "regex");
761         try {
762           if (regex == null)
763             urls.addAll(SolrResourceLoader.getURLs(dir));
764           else
765             urls.addAll(SolrResourceLoader.getFilteredURLs(dir, regex));
766         } catch (IOException e) {
767           log.warn("Couldn't add files from {} filtered by {} to classpath: {}", dir, regex, e.getMessage());
768         }
769       } else if (null != path) {
770         final Path dir = loader.getInstancePath().resolve(path);
771         try {
772           urls.add(dir.toUri().toURL());
773         } catch (MalformedURLException e) {
774           log.warn("Couldn't add file {} to classpath: {}", dir, e.getMessage());
775         }
776       } else {
777         throw new RuntimeException("lib: missing mandatory attributes: 'dir' or 'path'");
778       }
779     }
780 
781     if (urls.size() > 0) {
782       loader.addToClassLoader(urls);
783       loader.reloadLuceneSPI();
784     }
785   }

839   @Override
840   public Map<String, Object> toMap() {
841     LinkedHashMap result = new LinkedHashMap();
842     if (getZnodeVersion() > -1) result.put(ZNODEVER, getZnodeVersion());
843     result.put("luceneMatchVersion", luceneMatchVersion);
844     result.put("updateHandler", getUpdateHandlerInfo().toMap());
845     Map m = new LinkedHashMap();
846     result.put("query", m);
847     m.put("useFilterForSortedQuery", useFilterForSortedQuery);
848     m.put("queryResultWindowSize", queryResultWindowSize);
849     m.put("queryResultMaxDocsCached", queryResultMaxDocsCached);
850     m.put("enableLazyFieldLoading", enableLazyFieldLoading);
851     m.put("maxBooleanClauses", booleanQueryMaxClauseCount);
852     if (jmxConfig != null) result.put("jmx", jmxConfig.toMap());
853     for (SolrPluginInfo plugin : plugins) {
854       List<PluginInfo> infos = getPluginInfos(plugin.clazz.getName());
855       if (infos == null || infos.isEmpty()) continue;
856       String tag = plugin.getCleanTag();
857       tag = tag.replace("/", "");
858       if (plugin.options.contains(PluginOpts.REQUIRE_NAME)) {
859         LinkedHashMap items = new LinkedHashMap();
860         for (PluginInfo info : infos) items.put(info.name, info.toMap());
861         for (Map.Entry e : overlay.getNamedPlugins(plugin.tag).entrySet()) items.put(e.getKey(), e.getValue());
862         result.put(tag, items);
863       } else {
864         if (plugin.options.contains(MULTI_OK)) {
865           ArrayList<Map> l = new ArrayList<>();
866           for (PluginInfo info : infos) l.add(info.toMap());
867           result.put(tag, l);
868         } else {
869           result.put(tag, infos.get(0).toMap());
870         }
871 
872       }
873 
874     }
875 
876 
877     addCacheConfig(m, filterCacheConfig, queryResultCacheConfig, documentCacheConfig, fieldValueCacheConfig);
878     if (jmxConfig != null) result.put("jmx", jmxConfig.toMap());
879     m = new LinkedHashMap();
880     result.put("requestDispatcher", m);
881     m.put("handleSelect", handleSelect);
882     if (httpCachingConfig != null) m.put("httpCaching", httpCachingConfig.toMap());
883     m.put("requestParsers", makeMap("multipartUploadLimitKB", multipartUploadLimitKB,
884         "formUploadLimitKB", formUploadLimitKB,
885         "addHttpRequestToContext", addHttpRequestToContext));
886     if (indexConfig != null) result.put("indexConfig", indexConfig.toMap());
887 
888     //TODO there is more to add
889 
890     return result;
891   }
