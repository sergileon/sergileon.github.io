46    public static UserDictionary readDict() {
47      InputStream is = TestJapaneseTokenizer.class.getResourceAsStream("userdict.txt");
48      if (is == null) {
49        throw new RuntimeException("Cannot find userdict.txt in test classpath!");
50      }
51      try {
52        try {
53          Reader reader = new InputStreamReader(is, StandardCharsets.UTF_8);
54          return UserDictionary.open(reader);
55        } finally {
56          is.close();
57        }
58      } catch (IOException ioe) {
59        throw new RuntimeException(ioe);
60      }
61    }

76        @Override
77        protected TokenStreamComponents createComponents(String fieldName) {
78          Tokenizer tokenizer = new JapaneseTokenizer(newAttributeFactory(), readDict(), false, Mode.NORMAL);
79          return new TokenStreamComponents(tokenizer, tokenizer);
80        }
81      };
82      analyzerNoPunct = new Analyzer() {
83        @Override
84        protected TokenStreamComponents createComponents(String fieldName) {

384   private void assertReadings(String input, String... readings) throws IOException {
385     try (TokenStream ts = analyzer.tokenStream("ignored", input)) {
386       ReadingAttribute readingAtt = ts.addAttribute(ReadingAttribute.class);
387       ts.reset();
388       for(String reading : readings) {
389         assertTrue(ts.incrementToken());
390         assertEquals(reading, readingAtt.getReading());
391       }
392       assertFalse(ts.incrementToken());
393       ts.end();
394     }
395   }
