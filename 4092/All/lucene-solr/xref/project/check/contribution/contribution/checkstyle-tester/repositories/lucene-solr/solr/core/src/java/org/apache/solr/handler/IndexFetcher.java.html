178   public IndexFetcher(final NamedList initArgs, final ReplicationHandler handler, final SolrCore sc) {
179     solrCore = sc;
180     String masterUrl = (String) initArgs.get(MASTER_URL);
181     if (masterUrl == null)
182       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
183               "'masterUrl' is required for a slave");
184     if (masterUrl.endsWith("/replication")) {
185       masterUrl = masterUrl.substring(0, masterUrl.length()-12);
186       LOG.warn("'masterUrl' must be specified without the /replication suffix");
187     }
188     this.masterUrl = masterUrl;
189     
190     this.replicationHandler = handler;
191     String compress = (String) initArgs.get(COMPRESSION);
192     useInternalCompression = INTERNAL.equals(compress);
193     useExternalCompression = EXTERNAL.equals(compress);
194     String connTimeout = (String) initArgs.get(HttpClientUtil.PROP_CONNECTION_TIMEOUT);
195     String readTimeout = (String) initArgs.get(HttpClientUtil.PROP_SO_TIMEOUT);
196     String httpBasicAuthUser = (String) initArgs.get(HttpClientUtil.PROP_BASIC_AUTH_USER);
197     String httpBasicAuthPassword = (String) initArgs.get(HttpClientUtil.PROP_BASIC_AUTH_PASS);
198     myHttpClient = createHttpClient(solrCore, connTimeout, readTimeout, httpBasicAuthUser, httpBasicAuthPassword, useExternalCompression);
199   }

227     ModifiableSolrParams params = new ModifiableSolrParams();
228     params.set(COMMAND,  CMD_GET_FILE_LIST);
229     params.set(GENERATION, String.valueOf(gen));
230     params.set(CommonParams.WT, JAVABIN);
231     params.set(CommonParams.QT, "/replication");
232     QueryRequest req = new QueryRequest(params);
233 
234     // TODO modify to use shardhandler
235     try (HttpSolrClient client = new HttpSolrClient(masterUrl, myHttpClient)) {
236       client.setSoTimeout(60000);
237       client.setConnectionTimeout(15000);
238       NamedList response = client.request(req);
239 
240       List<Map<String, Object>> files = (List<Map<String,Object>>) response.get(CMD_GET_FILE_LIST);
241       if (files != null)
242         filesToDownload = Collections.synchronizedList(files);
243       else {
244         filesToDownload = Collections.emptyList();
245         LOG.error("No files to download for index generation: "+ gen);
246       }
247 
248       files = (List<Map<String,Object>>) response.get(CONF_FILES);
249       if (files != null)
250         confFilesToDownload = Collections.synchronizedList(files);
251 
252     } catch (SolrServerException e) {
253       throw new IOException(e);
254     }

261   /**
262    * This command downloads all the necessary files from master to install a index commit point. Only changed files are
263    * downloaded. It also downloads the conf files (if they are modified).
264    *
265    * @param forceReplication force a replication in all cases 
266    * @param forceCoreReload force a core reload in all cases
267    * @return true on success, false if slave is already in sync
268    * @throws IOException if an exception occurs
269    */
270   boolean fetchLatestIndex(boolean forceReplication, boolean forceCoreReload) throws IOException, InterruptedException {
271     
272     boolean cleanupDone = false;
273     boolean successfulInstall = false;
274     markReplicationStart();
275     Directory tmpIndexDir = null;
276     String tmpIndex;
277     Directory indexDir = null;
278     String indexDirPath;
279     boolean deleteTmpIdxDir = true;
280     
281     if (!solrCore.getSolrCoreState().getLastReplicateIndexSuccess()) {
282       // if the last replication was not a success, we force a full replication
283       // when we are a bit more confident we may want to try a partial replication
284       // if the error is connection related or something, but we have to be careful
285       forceReplication = true;
286     }
287     
288     try {
289       //get the current 'replicateable' index version in the master
290       NamedList response;
291       try {
292         response = getLatestVersion();
293       } catch (Exception e) {
294         LOG.error("Master at: " + masterUrl + " is not available. Index fetch failed. Exception: " + e.getMessage());
295         return false;
296       }
297       long latestVersion = (Long) response.get(CMD_INDEX_VERSION);
298       long latestGeneration = (Long) response.get(GENERATION);
299 
300       // TODO: make sure that getLatestCommit only returns commit points for the main index (i.e. no side-car indexes)
301       IndexCommit commit = solrCore.getDeletionPolicy().getLatestCommit();
302       if (commit == null) {
303         // Presumably the IndexWriter hasn't been opened yet, and hence the deletion policy hasn't been updated with commit points
304         RefCounted<SolrIndexSearcher> searcherRefCounted = null;
305         try {
306           searcherRefCounted = solrCore.getNewestSearcher(false);
307           if (searcherRefCounted == null) {
308             LOG.warn("No open searcher found - fetch aborted");
309             return false;
310           }
311           commit = searcherRefCounted.get().getIndexReader().getIndexCommit();
312         } finally {
313           if (searcherRefCounted != null)
314             searcherRefCounted.decref();
315         }
316       }
317 
318       if (latestVersion == 0L) {
319         if (forceReplication && commit.getGeneration() != 0) {
320           // since we won't get the files for an empty index,
321           // we just clear ours and commit
322           RefCounted<IndexWriter> iw = solrCore.getUpdateHandler().getSolrCoreState().getIndexWriter(solrCore);
323           try {
324             iw.get().deleteAll();
325           } finally {
326             iw.decref();
327           }
328           SolrQueryRequest req = new LocalSolrQueryRequest(solrCore, new ModifiableSolrParams());
329           solrCore.getUpdateHandler().commit(new CommitUpdateCommand(req, false));
330         }
331         
332         //there is nothing to be replicated
333         successfulInstall = true;
334         return true;
335       }
336       
337       // TODO: Should we be comparing timestamps (across machines) here?
338       if (!forceReplication && IndexDeletionPolicyWrapper.getCommitTimestamp(commit) == latestVersion) {
339         //master and slave are already in sync just return
340         LOG.info("Slave in sync with master.");
341         successfulInstall = true;
342         return true;
343       }
344       LOG.info("Master's generation: " + latestGeneration);
345       LOG.info("Slave's generation: " + commit.getGeneration());
346       LOG.info("Starting replication process");
347       // get the list of files first
348       fetchFileList(latestGeneration);
349       // this can happen if the commit point is deleted before we fetch the file list.
350       if (filesToDownload.isEmpty()) {
351         return false;
352       }
353       LOG.info("Number of files in latest index in master: " + filesToDownload.size());
354 
355       // Create the sync service
356       fsyncService = ExecutorUtil.newMDCAwareSingleThreadExecutor(new DefaultSolrThreadFactory("fsyncService"));
357       // use a synchronized list because the list is read by other threads (to show details)
358       filesDownloaded = Collections.synchronizedList(new ArrayList<Map<String, Object>>());
359       // if the generation of master is older than that of the slave , it means they are not compatible to be copied
360       // then a new index directory to be created and all the files need to be copied
361       boolean isFullCopyNeeded = IndexDeletionPolicyWrapper
362           .getCommitTimestamp(commit) >= latestVersion
363           || commit.getGeneration() >= latestGeneration || forceReplication;
364 
365       String tmpIdxDirName = "index." + new SimpleDateFormat(SnapShooter.DATE_FMT, Locale.ROOT).format(new Date());
366       tmpIndex = solrCore.getDataDir() + tmpIdxDirName;
367 
368       tmpIndexDir = solrCore.getDirectoryFactory().get(tmpIndex, DirContext.DEFAULT, solrCore.getSolrConfig().indexConfig.lockType);
369       
370       // cindex dir...
371       indexDirPath = solrCore.getIndexDir();
372       indexDir = solrCore.getDirectoryFactory().get(indexDirPath, DirContext.DEFAULT, solrCore.getSolrConfig().indexConfig.lockType);
373 
374       try {
375 
376         SegmentInfos infos = SegmentInfos.readCommit(indexDir, commit.getSegmentsFileName());
377         Version oldestVersion = IndexFetcher.checkOldestVersion(infos);
378 
379         //We will compare all the index files from the master vs the index files on disk to see if there is a mismatch
380         //in the metadata. If there is a mismatch for the same index file then we download the entire index again.
381         if (!isFullCopyNeeded && isIndexStale(indexDir, oldestVersion)) {
382           isFullCopyNeeded = true;
383         }
384         
385         if (!isFullCopyNeeded) {
386           // a searcher might be using some flushed but not committed segments
387           // because of soft commits (which open a searcher on IW's data)
388           // so we need to close the existing searcher on the last commit
389           // and wait until we are able to clean up all unused lucene files
390           if (solrCore.getCoreDescriptor().getCoreContainer().isZooKeeperAware()) {
391             solrCore.closeSearcher();
392           }
393 
394           // rollback and reopen index writer and wait until all unused files
395           // are successfully deleted
396           solrCore.getUpdateHandler().newIndexWriter(true);
397           RefCounted<IndexWriter> writer = solrCore.getUpdateHandler().getSolrCoreState().getIndexWriter(null);
398           try {
399             IndexWriter indexWriter = writer.get();
400             int c = 0;
401             indexWriter.deleteUnusedFiles();
402             while (hasUnusedFiles(indexDir, commit)) {
403               indexWriter.deleteUnusedFiles();
404               LOG.info("Sleeping for 1000ms to wait for unused lucene index files to be delete-able");
405               Thread.sleep(1000);
406               c++;
407               if (c >= 30)  {
408                 LOG.warn("IndexFetcher unable to cleanup unused lucene index files so we must do a full copy instead");
409                 isFullCopyNeeded = true;
410                 break;
411               }
412             }
413             if (c > 0)  {
414               LOG.info("IndexFetcher slept for " + (c * 1000) + "ms for unused lucene index files to be delete-able");
415             }
416           } finally {
417             writer.decref();
418           }
419           solrCore.getUpdateHandler().getSolrCoreState().closeIndexWriter(solrCore, true);
420         }
421         boolean reloadCore = false;
422         
423         try {
424           LOG.info("Starting download (fullCopy={}) to {}", isFullCopyNeeded, tmpIndexDir);
425           successfulInstall = false;
426           
427           final long bytesDownloaded = downloadIndexFiles(isFullCopyNeeded, indexDir, oldestVersion, tmpIndexDir, latestGeneration);
428           final long timeTakenSeconds = getReplicationTimeElapsed();
429           final Long bytesDownloadedPerSecond = (timeTakenSeconds != 0 ? new Long(bytesDownloaded/timeTakenSeconds) : null);
430           LOG.info("Total time taken for download (fullCopy={},bytesDownloaded={}) : {} secs ({} bytes/sec) to {}",
431               isFullCopyNeeded, bytesDownloaded, timeTakenSeconds, bytesDownloadedPerSecond, tmpIndexDir);
432 
433           Collection<Map<String,Object>> modifiedConfFiles = getModifiedConfFiles(confFilesToDownload);
434           if (!modifiedConfFiles.isEmpty()) {
435             reloadCore = true;
436             downloadConfFiles(confFilesToDownload, latestGeneration);
437             if (isFullCopyNeeded) {
438               successfulInstall = IndexFetcher.modifyIndexProps(solrCore, tmpIdxDirName);
439               deleteTmpIdxDir = false;
440             } else {
441               successfulInstall = moveIndexFiles(tmpIndexDir, indexDir);
442             }
443             if (successfulInstall) {
444               if (isFullCopyNeeded) {
445                 // let the system know we are changing dir's and the old one
446                 // may be closed
447                 if (indexDir != null) {
448                   LOG.info("removing old index directory " + indexDir);
449                   solrCore.getDirectoryFactory().doneWithDirectory(indexDir);
450                   solrCore.getDirectoryFactory().remove(indexDir);
451                 }
452               }
453               
454               LOG.info("Configuration files are modified, core will be reloaded");
455               logReplicationTimeAndConfFiles(modifiedConfFiles,
456                   successfulInstall);// write to a file time of replication and
457                                      // conf files.
458             }
459           } else {
460             terminateAndWaitFsyncService();
461             if (isFullCopyNeeded) {
462               successfulInstall = IndexFetcher.modifyIndexProps(solrCore, tmpIdxDirName);
463               deleteTmpIdxDir = false;
464             } else {
465               successfulInstall = moveIndexFiles(tmpIndexDir, indexDir);
466             }
467             if (successfulInstall) {
468               logReplicationTimeAndConfFiles(modifiedConfFiles,
469                   successfulInstall);
470             }
471           }
472         } finally {
473           if (!isFullCopyNeeded) {
474             solrCore.getUpdateHandler().getSolrCoreState().openIndexWriter(solrCore);
475           }
476         }
477         
478         // we must reload the core after we open the IW back up
479        if (successfulInstall && (reloadCore || forceCoreReload)) {
480           LOG.info("Reloading SolrCore {}", solrCore.getName());
481           reloadCore();
482         }
483 
484         if (successfulInstall) {
485           if (isFullCopyNeeded) {
486             // let the system know we are changing dir's and the old one
487             // may be closed
488             if (indexDir != null) {
489               LOG.info("removing old index directory " + indexDir);
490               solrCore.getDirectoryFactory().doneWithDirectory(indexDir);
491               solrCore.getDirectoryFactory().remove(indexDir);
492             }
493           }
494           if (isFullCopyNeeded) {
495             solrCore.getUpdateHandler().newIndexWriter(isFullCopyNeeded);
496           }
497           
498           openNewSearcherAndUpdateCommitPoint();
499         }
500         
501         if (!isFullCopyNeeded && !forceReplication && !successfulInstall) {
502           cleanup(solrCore, tmpIndexDir, indexDir, deleteTmpIdxDir, successfulInstall);
503           cleanupDone = true;
504           // we try with a full copy of the index
505           LOG.warn(
506               "Replication attempt was not successful - trying a full index replication reloadCore={}",
507               reloadCore);
508           successfulInstall = fetchLatestIndex(true, reloadCore);
509         }
510         
511         markReplicationStop();
512         return successfulInstall;
513       } catch (ReplicationHandlerException e) {
514         LOG.error("User aborted Replication");
515         return false;
516       } catch (SolrException e) {
517         throw e;
518       } catch (InterruptedException e) {
519         throw new InterruptedException("Index fetch interrupted");
520       } catch (Exception e) {
521         throw new SolrException(ErrorCode.SERVER_ERROR, "Index fetch failed : ", e);
522       }
523     } finally {
524       if (!cleanupDone) {
525         cleanup(solrCore, tmpIndexDir, indexDir, deleteTmpIdxDir, successfulInstall);
526       }
527     }
528   }

530   private void cleanup(final SolrCore core, Directory tmpIndexDir,
531       Directory indexDir, boolean deleteTmpIdxDir, boolean successfulInstall) throws IOException {
532     try {
533       if (!successfulInstall) {
534         try {
535           logReplicationTimeAndConfFiles(null, successfulInstall);
536         } catch (Exception e) {
537           LOG.error("caught", e);
538         }
539       }
540 
541       core.getUpdateHandler().getSolrCoreState().setLastReplicateIndexSuccess(successfulInstall);
542 
543       filesToDownload = filesDownloaded = confFilesDownloaded = confFilesToDownload = null;
544       markReplicationStop();
545       dirFileFetcher = null;
546       localFileFetcher = null;
547       if (fsyncService != null && !fsyncService.isShutdown()) fsyncService.shutdown();
548       fsyncService = null;
549       stop = false;
550       fsyncException = null;
551     } finally {
552       if (deleteTmpIdxDir && tmpIndexDir != null) {
553         try {
554           core.getDirectoryFactory().doneWithDirectory(tmpIndexDir);
555           core.getDirectoryFactory().remove(tmpIndexDir);
556         } catch (IOException e) {
557           SolrException.log(LOG, "Error removing directory " + tmpIndexDir, e);
558         }
559       }
560 
561       if (tmpIndexDir != null) {
562         core.getDirectoryFactory().release(tmpIndexDir);
563       }
564 
565       if (indexDir != null) {
566         core.getDirectoryFactory().release(indexDir);
567       }
568     }
569   }

571   private boolean hasUnusedFiles(Directory indexDir, IndexCommit commit) throws IOException {
572     String segmentsFileName = commit.getSegmentsFileName();
573     SegmentInfos infos = SegmentInfos.readCommit(indexDir, segmentsFileName);
574     Set<String> currentFiles = new HashSet<>(infos.files(true));
575     String[] allFiles = indexDir.listAll();
576     for (String file : allFiles) {
577       if (!file.equals(segmentsFileName) && !currentFiles.contains(file) && !file.endsWith(".lock")) {
578         LOG.info("Found unused file: " + file);
579         return true;
580       }
581     }
582     return false;
583   }

841     boolean equal = false;
842     boolean checkSummed = false;

845   protected static CompareResult compareFile(Directory indexDir, Version version, String filename, Long backupIndexFileLen, Long backupIndexFileChecksum) {
846     CompareResult compareResult = new CompareResult();
847     try {
848       try (final IndexInput indexInput = indexDir.openInput(filename, IOContext.READONCE)) {
849         long indexFileLen = indexInput.length();
850         long indexFileChecksum = 0;
851 
852         if (backupIndexFileChecksum != null && version.onOrAfter(Version.LUCENE_4_8_0)) {
853           try {
854             indexFileChecksum = CodecUtil.retrieveChecksum(indexInput);
855             compareResult.checkSummed = true;
856           } catch (Exception e) {
857             LOG.warn("Could not retrieve checksum from file.", e);
858           }
859         }
860         
861         if (!compareResult.checkSummed) {
862           // we don't have checksums to compare
863           
864           if (indexFileLen == backupIndexFileLen) {
865             compareResult.equal = true;
866             return compareResult;
867           } else {
868             LOG.warn(
869                 "File {} did not match. expected length is {} and actual length is {}", filename, backupIndexFileLen, indexFileLen);
870             compareResult.equal = false;
871             return compareResult;
872           }
873         }
874         
875         // we have checksums to compare
876         
877         if (indexFileLen == backupIndexFileLen && indexFileChecksum == backupIndexFileChecksum) {
878           compareResult.equal = true;
879           return compareResult;
880         } else {
881           LOG.warn("File {} did not match. expected checksum is {} and actual is checksum {}. " +
882               "expected length is {} and actual length is {}", filename, backupIndexFileChecksum, indexFileChecksum,
883               backupIndexFileLen, indexFileLen);
884           compareResult.equal = false;
885           return compareResult;
886         }
887       }
888     } catch (NoSuchFileException | FileNotFoundException e) {
889       compareResult.equal = false;
890       return compareResult;
891     } catch (IOException e) {
892       LOG.error("Could not read file " + filename + ". Downloading it again", e);
893       compareResult.equal = false;
894       return compareResult;
895     }

913    * they are not compatible (stale).
914    *
915    * @return true if the index stale and we need to download a fresh copy, false otherwise.
916    * @throws IOException  if low level io error
917    */
918   private boolean isIndexStale(Directory dir, Version indexVersion) throws IOException {
919     for (Map<String, Object> file : filesToDownload) {
920       String filename = (String) file.get(NAME);
921       Long length = (Long) file.get(SIZE);
922       Long checksum = (Long) file.get(CHECKSUM);
923       if (slowFileExists(dir, filename)) {
924         if (checksum != null) {
925           if (!(compareFile(dir, indexVersion, filename, length, checksum).equal)) {
926             // file exists and size or checksum is different, therefore we must download it again
927             return true;
928           }
929         } else {
930           if (length != dir.fileLength(filename)) {
931             LOG.warn("File {} did not match. expected length is {} and actual length is {}",
932                 filename, length, dir.fileLength(filename));
933             return true;
934           }
935         }
936       }
937     }
938     return false;
939   }

1116   /**
1117    * The local conf files are compared with the conf files in the master. If they are same (by checksum) do not copy.
1118    *
1119    * @param confFilesToDownload The list of files obtained from master
1120    *
1121    * @return a list of configuration files which have changed on the master and need to be downloaded.
1122    */
1123   private Collection<Map<String, Object>> getModifiedConfFiles(List<Map<String, Object>> confFilesToDownload) {
1124     if (confFilesToDownload == null || confFilesToDownload.isEmpty())
1125       return Collections.EMPTY_LIST;
1126     //build a map with alias/name as the key
1127     Map<String, Map<String, Object>> nameVsFile = new HashMap<>();
1128     NamedList names = new NamedList();
1129     for (Map<String, Object> map : confFilesToDownload) {
1130       //if alias is present that is the name the file may have in the slave
1131       String name = (String) (map.get(ALIAS) == null ? map.get(NAME) : map.get(ALIAS));
1132       nameVsFile.put(name, map);
1133       names.add(name, null);
1134     }
1135     //get the details of the local conf files with the same alias/name
1136     List<Map<String, Object>> localFilesInfo = replicationHandler.getConfFileInfoFromCache(names, confFileInfoCache);
1137     //compare their size/checksum to see if
1138     for (Map<String, Object> fileInfo : localFilesInfo) {
1139       String name = (String) fileInfo.get(NAME);
1140       Map<String, Object> m = nameVsFile.get(name);
1141       if (m == null) continue; // the file is not even present locally (so must be downloaded)
1142       if (m.get(CHECKSUM).equals(fileInfo.get(CHECKSUM))) {
1143         nameVsFile.remove(name); //checksums are same so the file need not be downloaded
1144       }
1145     }
1146     return nameVsFile.isEmpty() ? Collections.EMPTY_LIST : nameVsFile.values();
1147   }

1291      * The main method which downloads file
1292      */
1293     public void fetchFile() throws Exception {
1294       try {
1295         while (true) {
1296           final FastInputStream is = getStream();
1297           int result;
1298           try {
1299             //fetch packets one by one in a single request
1300             result = fetchPackets(is);
1301             if (result == 0 || result == NO_CONTENT) {
1302 
1303               return;
1304             }
1305             //if there is an error continue. But continue from the point where it got broken
1306           } finally {
1307             IOUtils.closeQuietly(is);
1308           }
1309         }
1310       } finally {
1311         cleanup();
1312         //if cleanup succeeds . The file is downloaded fully. do an fsync
1313         fsyncService.submit(new Runnable(){
1314           @Override
1315           public void run() {
1316             try {
1317               file.sync();
1318             } catch (IOException e) {
1319               fsyncException = e;

1321           }
1322         });

1326     private int fetchPackets(FastInputStream fis) throws Exception {
1327       byte[] intbytes = new byte[4];
1328       byte[] longbytes = new byte[8];
1329       try {
1330         while (true) {
1331           if (stop) {
1332             stop = false;
1333             aborted = true;
1334             throw new ReplicationHandlerException("User aborted replication");
1335           }
1336           long checkSumServer = -1;
1337           fis.readFully(intbytes);
1338           //read the size of the packet
1339           int packetSize = readInt(intbytes);
1340           if (packetSize <= 0) {
1341             LOG.warn("No content received for file: {}", fileName);
1342             return NO_CONTENT;
1343           }
1344           if (buf.length < packetSize)
1345             buf = new byte[packetSize];
1346           if (checksum != null) {
1347             //read the checksum
1348             fis.readFully(longbytes);
1349             checkSumServer = readLong(longbytes);
1350           }
1351           //then read the packet of bytes
1352           fis.readFully(buf, 0, packetSize);
1353           //compare the checksum as sent from the master
1354           if (includeChecksum) {
1355             checksum.reset();
1356             checksum.update(buf, 0, packetSize);
1357             long checkSumClient = checksum.getValue();
1358             if (checkSumClient != checkSumServer) {
1359               LOG.error("Checksum not matched between client and server for file: {}", fileName);
1360               //if checksum is wrong it is a problem return for retry
1361               return 1;
1362             }
1363           }
1364           //if everything is fine, write down the packet to the file
1365           file.write(buf, packetSize);
1366           bytesDownloaded += packetSize;
1367           LOG.debug("Fetched and wrote {} bytes of file: {}", bytesDownloaded, fileName);
1368           if (bytesDownloaded >= size)
1369             return 0;
1370           //errorCount is always set to zero after a successful packet
1371           errorCount = 0;
1372         }
1373       } catch (ReplicationHandlerException e) {
1374         throw e;
1375       } catch (Exception e) {
1376         LOG.warn("Error in fetching file: {} (downloaded {} of {} bytes)",
1377             fileName, bytesDownloaded, size, e);
1378         //for any failure, increment the error count
1379         errorCount++;
1380         //if it fails for the same packet for MAX_RETRIES fail and come out
1381         if (errorCount > MAX_RETRIES) {
1382           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
1383               "Failed to fetch file: " + fileName +
1384                   " (downloaded " + bytesDownloaded + " of " + size + " bytes" +
1385                   ", error count: " + errorCount + " > " + MAX_RETRIES + ")", e);
1386         }
1387         return ERR;
1388       }
1389     }

1391     /**
1392      * The webcontainer flushes the data only after it fills the buffer size. So, all data has to be read as readFully()
1393      * other wise it fails. So read everything as bytes and then extract an integer out of it
1394      */
1395     private int readInt(byte[] b) {
1396       return (((b[0] & 0xff) << 24) | ((b[1] & 0xff) << 16)
1397           | ((b[2] & 0xff) << 8) | (b[3] & 0xff));
1398 
1399     }
1400 
1401     /**
1402      * Same as above but to read longs from a byte array
1403      */
1404     private long readLong(byte[] b) {
1405       return (((long) (b[0] & 0xff)) << 56) | (((long) (b[1] & 0xff)) << 48)
1406           | (((long) (b[2] & 0xff)) << 40) | (((long) (b[3] & 0xff)) << 32)
1407           | (((long) (b[4] & 0xff)) << 24) | ((b[5] & 0xff) << 16)
1408           | ((b[6] & 0xff) << 8) | ((b[7] & 0xff));
1409 
1410     }

1412     /**
1413      * cleanup everything
1414      */
1415     private void cleanup() {
1416       try {
1417         file.close();
1418       } catch (Exception e) {/* no-op */
1419         LOG.error("Error closing file: {}", this.saveAs, e);
1420       }
1421       if (bytesDownloaded != size) {
1422         //if the download is not complete then
1423         //delete the file being downloaded
1424         try {
1425           file.delete();
1426         } catch (Exception e) {
1427           LOG.error("Error deleting file: {}", this.saveAs, e);
1428         }
1429         //if the failure is due to a user abort it is returned normally else an exception is thrown
1430         if (!aborted)
1431           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
1432               "Unable to download " + fileName + " completely. Downloaded "
1433                   + bytesDownloaded + "!=" + size);
1434       }

1437     /**
1438      * Open a new stream using HttpClient
1439      */
1440     private FastInputStream getStream() throws IOException {
1441 
1442       ModifiableSolrParams params = new ModifiableSolrParams();
1443 
1444 //    //the method is command=filecontent
1445       params.set(COMMAND, CMD_GET_FILE);
1446       params.set(GENERATION, Long.toString(indexGen));
1447       params.set(CommonParams.QT, "/replication");
1448       //add the version to download. This is used to reserve the download
1449       if (isConf) {
1450         //set cf instead of file for config file
1451         params.set(CONF_FILE_SHORT, fileName);
1452       } else {
1453         params.set(FILE, fileName);
1454       }
1455       if (useInternalCompression) {
1456         params.set(COMPRESSION, "true");
1457       }
1458       //use checksum
1459       if (this.includeChecksum) {
1460         params.set(CHECKSUM, true);
1461       }
1462       //wt=filestream this is a custom protocol
1463       params.set(CommonParams.WT, FILE_STREAM);
1464       // This happen if there is a failure there is a retry. the offset=<sizedownloaded> ensures that
1465       // the server starts from the offset
1466       if (bytesDownloaded > 0) {
1467         params.set(OFFSET, Long.toString(bytesDownloaded));
1468       }
1469 
1470 
1471       NamedList response;
1472       InputStream is = null;
1473 
1474       // TODO use shardhandler
1475       try (HttpSolrClient client = new HttpSolrClient(masterUrl, myHttpClient, null)) {
1476         client.setSoTimeout(60000);
1477         client.setConnectionTimeout(15000);
1478         QueryRequest req = new QueryRequest(params);
1479         response = client.request(req);
1480         is = (InputStream) response.get("stream");
1481         if(useInternalCompression) {
1482           is = new InflaterInputStream(is);
1483         }
1484         return new FastInputStream(is);
1485       } catch (Exception e) {
1486         //close stream on error
1487         IOUtils.closeQuietly(is);
1488         throw new IOException("Could not download file '" + fileName + "'", e);
1489       }
1490     }
