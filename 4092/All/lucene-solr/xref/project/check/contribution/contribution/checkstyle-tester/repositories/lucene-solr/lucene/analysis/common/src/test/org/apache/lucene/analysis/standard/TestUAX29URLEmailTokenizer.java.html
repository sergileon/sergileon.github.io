42    // LUCENE-5440: extremely slow tokenization of text matching email <local-part> (before the '@')
43    @Slow
44    public void testLongEMAILatomText() throws Exception {
45      // EMAILatomText = [A-Za-z0-9!#$%&'*+-/=?\^_`{|}~]
46      char[] emailAtomChars
47          = "!#$%&'*+,-./0123456789=?ABCDEFGHIJKLMNOPQRSTUVWXYZ^_`abcdefghijklmnopqrstuvwxyz{|}~".toCharArray();
48      StringBuilder builder = new StringBuilder();
49      int numChars = TestUtil.nextInt(random(), 100 * 1024, 3 * 1024 * 1024);
50      for (int i = 0 ; i < numChars ; ++i) {
51        builder.append(emailAtomChars[random().nextInt(emailAtomChars.length)]);
52      }
53      int tokenCount = 0;
54      UAX29URLEmailTokenizer ts = new UAX29URLEmailTokenizer();
55      String text = builder.toString();
56      ts.setReader(new StringReader(text));
57      ts.reset();
58      while (ts.incrementToken()) {
59        tokenCount++;
60      }
61      ts.end();
62      ts.close();
63      assertTrue(tokenCount > 0);
64  
65      tokenCount = 0;
66      int newBufferSize = TestUtil.nextInt(random(), 200, 8192);
67      ts.setMaxTokenLength(newBufferSize);
68      ts.setReader(new StringReader(text));
69      ts.reset();
70      while (ts.incrementToken()) {
71        tokenCount++;
72      }
73      ts.end();
74      ts.close();
75      assertTrue(tokenCount > 0);
76    }

127   /** Passes through tokens with type "<URL>" and blocks all other types. */
128   private class URLFilter extends TokenFilter {
129     private final TypeAttribute typeAtt = addAttribute(TypeAttribute.class);
130     public URLFilter(TokenStream in) {
131       super(in);
132     }
133     @Override
134     public final boolean incrementToken() throws java.io.IOException {
135       boolean isTokenAvailable = false;
136       while (input.incrementToken()) {
137         if (typeAtt.type() == UAX29URLEmailTokenizer.TOKEN_TYPES[UAX29URLEmailTokenizer.URL]) {
138           isTokenAvailable = true;
139           break;
140         }
141       }
142       return isTokenAvailable;
143     }
144   }

146   /** Passes through tokens with type "<EMAIL>" and blocks all other types. */
147   private class EmailFilter extends TokenFilter {
148     private final TypeAttribute typeAtt = addAttribute(TypeAttribute.class);
149     public EmailFilter(TokenStream in) {
150       super(in);
151     }
152     @Override
153     public final boolean incrementToken() throws java.io.IOException {
154       boolean isTokenAvailable = false;
155       while (input.incrementToken()) {
156         if (typeAtt.type() == UAX29URLEmailTokenizer.TOKEN_TYPES[UAX29URLEmailTokenizer.EMAIL]) {
157           isTokenAvailable = true;
158           break;
159         }
160       }
161       return isTokenAvailable;
162     }

311     Reader reader = null;
312     String luceneResourcesWikiPage;
313     try {
314       reader = new InputStreamReader(getClass().getResourceAsStream
315         ("LuceneResourcesWikiPage.html"), StandardCharsets.UTF_8);
316       StringBuilder builder = new StringBuilder();
317       char[] buffer = new char[1024];
318       int numCharsRead;
319       while (-1 != (numCharsRead = reader.read(buffer))) {
320         builder.append(buffer, 0, numCharsRead);
321       }
322       luceneResourcesWikiPage = builder.toString(); 
323     } finally {
324       if (null != reader) {
325         reader.close();
326       }
327     }
328     assertTrue(null != luceneResourcesWikiPage

330     BufferedReader bufferedReader = null;

334       bufferedReader = new BufferedReader(new InputStreamReader
335         (getClass().getResourceAsStream("LuceneResourcesWikiPageURLs.txt"), StandardCharsets.UTF_8));
336       String line;
337       while (null != (line = bufferedReader.readLine())) {
338         line = line.trim();
339         if (line.length() > 0) {
340           urlList.add(line);
341         }
342       }
343       urls = urlList.toArray(new String[urlList.size()]);
344     } finally {
345       if (null != bufferedReader) {
346         bufferedReader.close();
347       }
348     }
349     assertTrue(null != urls && urls.length > 0);
350     BaseTokenStreamTestCase.assertAnalyzesTo
351       (urlAnalyzer, luceneResourcesWikiPage, urls);

499   /**
500    * Multiple consecutive chars in \p{Word_Break = MidLetter},
501    * \p{Word_Break = MidNumLet}, and/or \p{Word_Break = MidNum}
502    * should trigger a token split.
503    */
504   public void testMid() throws Exception {
505     // ':' is in \p{WB:MidLetter}, which should trigger a split unless there is a Letter char on both sides
506     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:B", new String[] { "A:B" });
507     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A::B", new String[] { "A", "B" });
508 
509     // '.' is in \p{WB:MidNumLet}, which should trigger a split unless there is a Letter or Numeric char on both sides
510     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.2", new String[] { "1.2" });
511     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.B", new String[] { "A.B" });
512     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1..2", new String[] { "1", "2" });
513     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A..B", new String[] { "A", "B" });
514 
515     // ',' is in \p{WB:MidNum}, which should trigger a split unless there is a Numeric char on both sides
516     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,2", new String[] { "1,2" });
517     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,,2", new String[] { "1", "2" });
518 
519     // Mixed consecutive \p{WB:MidLetter} and \p{WB:MidNumLet} should trigger a split
520     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.:B", new String[] { "A", "B" });
521     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:.B", new String[] { "A", "B" });
522 
523     // Mixed consecutive \p{WB:MidNum} and \p{WB:MidNumLet} should trigger a split
524     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,.2", new String[] { "1", "2" });
525     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.,2", new String[] { "1", "2" });
526 
527     // '_' is in \p{WB:ExtendNumLet}
528 
529     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:B_A:B", new String[] { "A:B_A:B" });
530     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:B_A::B", new String[] { "A:B_A", "B" });
531 
532     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.2_1.2", new String[] { "1.2_1.2" });
533     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.B_A.B", new String[] { "A.B_A.B" });
534     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.2_1..2", new String[] { "1.2_1", "2" });
535     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.B_A..B", new String[] { "A.B_A", "B" });
536 
537     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,2_1,2", new String[] { "1,2_1,2" });
538     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,2_1,,2", new String[] { "1,2_1", "2" });
539 
540     BaseTokenStreamTestCase.assertAnalyzesTo(a, "C_A.:B", new String[] { "C_A", "B" });
541     BaseTokenStreamTestCase.assertAnalyzesTo(a, "C_A:.B", new String[] { "C_A", "B" });
542 
543     BaseTokenStreamTestCase.assertAnalyzesTo(a, "3_1,.2", new String[] { "3_1", "2" });
544     BaseTokenStreamTestCase.assertAnalyzesTo(a, "3_1.,2", new String[] { "3_1", "2" });
545   }
