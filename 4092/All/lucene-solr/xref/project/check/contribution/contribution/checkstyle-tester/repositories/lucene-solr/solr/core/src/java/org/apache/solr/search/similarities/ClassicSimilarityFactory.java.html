20  import org.apache.lucene.search.similarities.ClassicSimilarity;
21  import org.apache.lucene.search.similarities.Similarity;
22  import org.apache.lucene.search.similarities.TFIDFSimilarity; // javadoc
23  import org.apache.solr.common.params.SolrParams;
24  import org.apache.solr.schema.SimilarityFactory;
25  
26  /**
27   * Factory for {@link ClassicSimilarity}
28   * <p>
29   * ClassicSimilarity is Lucene's original scoring implementation, based
30   * upon the Vector Space Model.
31   * <p>
32   * Optional settings:
33   * <ul>
34   *   <li>discountOverlaps (bool): Sets
35   *       {@link ClassicSimilarity#setDiscountOverlaps(boolean)}</li>
36   * </ul>
37   * @see TFIDFSimilarity
38   * @lucene.experimental
39   */
40  public class ClassicSimilarityFactory extends SimilarityFactory {
41  
42    /** Init param name for specifying the value to use in 
43     * {@link ClassicSimilarity#setDiscountOverlaps(boolean)} 
44     */
45    public static final String DISCOUNT_OVERLAPS = "discountOverlaps";
46  
47    /** 
48     * Controls the value of {@link ClassicSimilarity#setDiscountOverlaps(boolean)} 
49     * on newly constructed instances of {@link ClassicSimilarity}
50     */
51    protected boolean discountOverlaps = true;
52  
53    @Override
54    public void init(SolrParams params) {
55      super.init(params);
56      discountOverlaps = params.getBool(DISCOUNT_OVERLAPS, true);
57    }
58  
59    @Override
60    public Similarity getSimilarity() {
61      ClassicSimilarity sim = new ClassicSimilarity();
62      sim.setDiscountOverlaps(discountOverlaps);
63      return sim;
64    }
65  }
