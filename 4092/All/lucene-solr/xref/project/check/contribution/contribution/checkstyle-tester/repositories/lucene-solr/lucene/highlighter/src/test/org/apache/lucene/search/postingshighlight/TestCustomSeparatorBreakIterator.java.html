20  import com.carrotsearch.randomizedtesting.generators.RandomPicks;
21  import org.apache.lucene.util.LuceneTestCase;
22  
23  import java.text.BreakIterator;
24  import java.util.Locale;
25  
26  import static org.apache.lucene.search.postingshighlight.TestWholeBreakIterator.assertSameBreaks;
27  import static org.hamcrest.CoreMatchers.equalTo;
28  
29  public class TestCustomSeparatorBreakIterator extends LuceneTestCase {
30  
31    private static final Character[] SEPARATORS = new Character[]{' ', '\u0000', 8233};
32  
33    public void testBreakOnCustomSeparator() throws Exception {
34      Character separator = randomSeparator();
35      BreakIterator bi = new CustomSeparatorBreakIterator(separator);
36      String source = "this" + separator + "is" + separator + "the" + separator + "first" + separator + "sentence";
37      bi.setText(source);
38      assertThat(bi.current(), equalTo(0));
39      assertThat(bi.first(), equalTo(0));
40      assertThat(source.substring(bi.current(), bi.next()), equalTo("this" + separator));
41      assertThat(source.substring(bi.current(), bi.next()), equalTo("is" + separator));
42      assertThat(source.substring(bi.current(), bi.next()), equalTo("the" + separator));
43      assertThat(source.substring(bi.current(), bi.next()), equalTo("first" + separator));
44      assertThat(source.substring(bi.current(), bi.next()), equalTo("sentence"));
45      assertThat(bi.next(), equalTo(BreakIterator.DONE));
46  
47      assertThat(bi.last(), equalTo(source.length()));
48      int current = bi.current();
49      assertThat(source.substring(bi.previous(), current), equalTo("sentence"));
50      current = bi.current();
51      assertThat(source.substring(bi.previous(), current), equalTo("first" + separator));
52      current = bi.current();
53      assertThat(source.substring(bi.previous(), current), equalTo("the" + separator));
54      current = bi.current();
55      assertThat(source.substring(bi.previous(), current), equalTo("is" + separator));
56      current = bi.current();
57      assertThat(source.substring(bi.previous(), current), equalTo("this" + separator));
58      assertThat(bi.previous(), equalTo(BreakIterator.DONE));
59      assertThat(bi.current(), equalTo(0));
60  
61      assertThat(source.substring(0, bi.following(9)), equalTo("this" + separator + "is" + separator + "the" + separator));
62  
63      assertThat(source.substring(0, bi.preceding(9)), equalTo("this" + separator + "is" + separator));
64  
65      assertThat(bi.first(), equalTo(0));
66      assertThat(source.substring(0, bi.next(3)), equalTo("this" + separator + "is" + separator + "the" + separator));
67    }
68  
69    public void testSingleSentences() throws Exception {
70      BreakIterator expected = BreakIterator.getSentenceInstance(Locale.ROOT);
71      BreakIterator actual = new CustomSeparatorBreakIterator(randomSeparator());
72      assertSameBreaks("a", expected, actual);
73      assertSameBreaks("ab", expected, actual);
74      assertSameBreaks("abc", expected, actual);
75      assertSameBreaks("", expected, actual);
76    }
77  
78    public void testSliceEnd() throws Exception {
79      BreakIterator expected = BreakIterator.getSentenceInstance(Locale.ROOT);
80      BreakIterator actual = new CustomSeparatorBreakIterator(randomSeparator());
81      assertSameBreaks("a000", 0, 1, expected, actual);
82      assertSameBreaks("ab000", 0, 1, expected, actual);
83      assertSameBreaks("abc000", 0, 1, expected, actual);
84      assertSameBreaks("000", 0, 0, expected, actual);
85    }
86  
87    public void testSliceStart() throws Exception {
88      BreakIterator expected = BreakIterator.getSentenceInstance(Locale.ROOT);
89      BreakIterator actual = new CustomSeparatorBreakIterator(randomSeparator());
90      assertSameBreaks("000a", 3, 1, expected, actual);
91      assertSameBreaks("000ab", 3, 2, expected, actual);
92      assertSameBreaks("000abc", 3, 3, expected, actual);
93      assertSameBreaks("000", 3, 0, expected, actual);
94    }
95  
96    public void testSliceMiddle() throws Exception {
97      BreakIterator expected = BreakIterator.getSentenceInstance(Locale.ROOT);
98      BreakIterator actual = new CustomSeparatorBreakIterator(randomSeparator());
99      assertSameBreaks("000a000", 3, 1, expected, actual);
100     assertSameBreaks("000ab000", 3, 2, expected, actual);
101     assertSameBreaks("000abc000", 3, 3, expected, actual);
102     assertSameBreaks("000000", 3, 0, expected, actual);
103   }
104 
105   /** the current position must be ignored, initial position is always first() */
106   public void testFirstPosition() throws Exception {
107     BreakIterator expected = BreakIterator.getSentenceInstance(Locale.ROOT);
108     BreakIterator actual = new CustomSeparatorBreakIterator(randomSeparator());
109     assertSameBreaks("000ab000", 3, 2, 4, expected, actual);
110   }
111 
112   private static char randomSeparator() {
113     return RandomPicks.randomFrom(random(), SEPARATORS);
114   }
115 }
