137   private void testOverwriteOption() throws Exception, SolrServerException,
138       IOException {
139     String collectionName = "overwriteCollection";
140     createCollection(collectionName, controlClientCloud, 1, 1);
141     waitForRecoveriesToFinish(collectionName, false);
142     try (CloudSolrClient cloudClient = createCloudClient(collectionName)) {
143       SolrInputDocument doc1 = new SolrInputDocument();
144       doc1.addField(id, "0");
145       doc1.addField("a_t", "hello1");
146       SolrInputDocument doc2 = new SolrInputDocument();
147       doc2.addField(id, "0");
148       doc2.addField("a_t", "hello2");
149       
150       UpdateRequest request = new UpdateRequest();
151       request.add(doc1);
152       request.add(doc2);
153       request.setAction(AbstractUpdateRequest.ACTION.COMMIT, false, false);
154       NamedList<Object> response = cloudClient.request(request);
155       QueryResponse resp = cloudClient.query(new SolrQuery("*:*"));
156       
157       assertEquals("There should be one document because overwrite=true", 1, resp.getResults().getNumFound());
158       
159       doc1 = new SolrInputDocument();
160       doc1.addField(id, "1");
161       doc1.addField("a_t", "hello1");
162       doc2 = new SolrInputDocument();
163       doc2.addField(id, "1");
164       doc2.addField("a_t", "hello2");
165       
166       request = new UpdateRequest();
167       // overwrite=false
168       request.add(doc1, false);
169       request.add(doc2, false);
170       request.setAction(AbstractUpdateRequest.ACTION.COMMIT, false, false);
171       response = cloudClient.request(request);
172       
173       resp = cloudClient.query(new SolrQuery("*:*"));
174 
175       assertEquals("There should be 3 documents because there should be two id=1 docs due to overwrite=false", 3, resp.getResults().getNumFound());
176     }

364     controlClient.commit();
365     cloudClient.commit();
366     cloudClient.close();

431   private void queryWithPreferLocalShards(CloudSolrClient cloudClient,
432                                           boolean preferLocalShards,
433                                           String collectionName)
434       throws Exception
435   {
436     SolrQuery qRequest = new SolrQuery();
437     qRequest.setQuery("*:*");
438 
439     ModifiableSolrParams qParams = new ModifiableSolrParams();
440     qParams.add("preferLocalShards", Boolean.toString(preferLocalShards));
441     qParams.add(ShardParams.SHARDS_INFO, "true");
442     qRequest.add(qParams);
443 
444     // CloudSolrClient sends the request to some node.
445     // And since all the nodes are hosting cores from all shards, the
446     // distributed query formed by this node will select cores from the
447     // local shards only
448     QueryResponse qResponse = cloudClient.query (qRequest);
449 
450     Object shardsInfo = qResponse.getResponse().get(ShardParams.SHARDS_INFO);
451     assertNotNull("Unable to obtain "+ShardParams.SHARDS_INFO, shardsInfo);
452 
453     // Iterate over shards-info and check what cores responded
454     SimpleOrderedMap<?> shardsInfoMap = (SimpleOrderedMap<?>)shardsInfo;
455     Iterator<Map.Entry<String, ?>> itr = shardsInfoMap.asMap(100).entrySet().iterator();
456     List<String> shardAddresses = new ArrayList<String>();
457     while (itr.hasNext()) {
458       Map.Entry<String, ?> e = itr.next();
459       assertTrue("Did not find map-type value in "+ShardParams.SHARDS_INFO, e.getValue() instanceof Map);
460       String shardAddress = (String)((Map)e.getValue()).get("shardAddress");
461       assertNotNull(ShardParams.SHARDS_INFO+" did not return 'shardAddress' parameter", shardAddress);
462       shardAddresses.add(shardAddress);
463     }
464     log.info("Shards giving the response: " + Arrays.toString(shardAddresses.toArray()));
465 
466     // Make sure the distributed queries were directed to a single node only
467     if (preferLocalShards) {
468       Set<Integer> ports = new HashSet<Integer>();
469       for (String shardAddr: shardAddresses) {
470         URL url = new URL (shardAddr);
471         ports.add(url.getPort());
472       }
473 
474       // This assertion would hold true as long as every shard has a core on each node
475       assertTrue ("Response was not received from shards on a single node",
476           shardAddresses.size() > 1 && ports.size()==1);
477     }
478   }

480   private Long getNumRequests(String baseUrl, String collectionName) throws
481       SolrServerException, IOException {
482 
483     NamedList<Object> resp;
484     try (HttpSolrClient client = new HttpSolrClient(baseUrl + "/"+ collectionName)) {
485       client.setConnectionTimeout(15000);
486       client.setSoTimeout(60000);
487       ModifiableSolrParams params = new ModifiableSolrParams();
488       params.set("qt", "/admin/mbeans");
489       params.set("stats", "true");
490       params.set("key", "standard");
491       params.set("cat", "QUERYHANDLER");
492       // use generic request to avoid extra processing of queries
493       QueryRequest req = new QueryRequest(params);
494       resp = client.request(req);
495     }
496     return (Long) resp.findRecursive("solr-mbeans", "QUERYHANDLER",
497         "standard", "stats", "requests");
498   }
