51  /**
52   * This manages the lifecycle of a set of plugin of the same type .
53   */
54  public class PluginBag<T> implements AutoCloseable {
55    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
56  
57    private final Map<String, PluginHolder<T>> registry;
58    private final Map<String, PluginHolder<T>> immutableRegistry;
59    private String def;
60    private final Class klass;
61    private SolrCore core;
62    private final SolrConfig.SolrPluginInfo meta;
63  
64    /**
65     * Pass needThreadSafety=true if plugins can be added and removed concurrently with lookups.
66     */
67    public PluginBag(Class<T> klass, SolrCore core, boolean needThreadSafety) {
68      this.core = core;
69      this.klass = klass;
70      // TODO: since reads will dominate writes, we could also think about creating a new instance of a map each time it changes.
71      // Not sure how much benefit this would have over ConcurrentHashMap though
72      // We could also perhaps make this constructor into a factory method to return different implementations depending on thread safety needs.
73      this.registry = needThreadSafety ? new ConcurrentHashMap<String, PluginHolder<T>>() : new HashMap<String, PluginHolder<T>>();
74      this.immutableRegistry = Collections.unmodifiableMap(registry);
75      meta = SolrConfig.classVsSolrPluginInfo.get(klass.getName());
76      if (meta == null) {
77        throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Unknown Plugin : " + klass.getName());
78      }
79    }
80  
81    /**
82     * Constructs a non-threadsafe plugin registry
83     */
84    public PluginBag(Class<T> klass, SolrCore core) {
85      this(klass, core, false);
86    }

88    static void initInstance(Object inst, PluginInfo info) {
89      if (inst instanceof PluginInfoInitialized) {
90        ((PluginInfoInitialized) inst).init(info);
91      } else if (inst instanceof NamedListInitializedPlugin) {
92        ((NamedListInitializedPlugin) inst).init(info.initArgs);
93      } else if (inst instanceof SolrRequestHandler) {
94        ((SolrRequestHandler) inst).init(info.initArgs);
95      }
96      if (inst instanceof SearchComponent) {
97        ((SearchComponent) inst).setName(info.name);
98      }
99      if (inst instanceof RequestHandlerBase) {
100       ((RequestHandlerBase) inst).setPluginInfo(info);
101     }
102 
103   }

105   /**
106    * Check if any of the mentioned names are missing. If yes, return the Set of missing names
107    */
108   public Set<String> checkContains(Collection<String> names) {
109     if (names == null || names.isEmpty()) return Collections.EMPTY_SET;
110     HashSet<String> result = new HashSet<>();
111     for (String s : names) if (!this.registry.containsKey(s)) result.add(s);
112     return result;
113   }
114 
115   PluginHolder<T> createPlugin(PluginInfo info) {
116     if ("true".equals(String.valueOf(info.attributes.get("runtimeLib")))) {
117       log.info(" {} : '{}'  created with runtimeLib=true ", meta.getCleanTag(), info.name);
118       return new LazyPluginHolder<>(meta, info, core, core.getMemClassLoader());
119     } else if ("lazy".equals(info.attributes.get("startup")) && meta.options.contains(SolrConfig.PluginOpts.LAZY)) {
120       log.info("{} : '{}' created with startup=lazy ", meta.getCleanTag(), info.name);
121       return new LazyPluginHolder<T>(meta, info, core, core.getResourceLoader());
122     } else {
123       T inst = core.createInstance(info.className, (Class<T>) meta.clazz, meta.getCleanTag(), null, core.getResourceLoader());
124       initInstance(inst, info);
125       return new PluginHolder<>(info, inst);
126     }
127   }

129   /** make a plugin available in an alternate name. This is an internal API and not for public use
130    * @param src key in which the plugin is already registered
131    * @param target the new key in which the plugin should be aliased to. If target exists already, the alias fails
132    * @return flag if the operation is successful or not
133    */
134   boolean alias(String src, String target) {
135     if (src == null) return false;
136     PluginHolder<T> a = registry.get(src);
137     if (a == null) return false;
138     PluginHolder<T> b = registry.get(target);
139     if (b != null) return false;
140     registry.put(target, a);
141     return true;
142   }

144   /**
145    * Get a plugin by name. If the plugin is not already instantiated, it is
146    * done here
147    */
148   public T get(String name) {
149     PluginHolder<T> result = registry.get(name);
150     return result == null ? null : result.get();
151   }

153   /**
154    * Fetches a plugin by name , or the default
155    *
156    * @param name       name using which it is registered
157    * @param useDefault Return the default , if a plugin by that name does not exist
158    */
159   public T get(String name, boolean useDefault) {
160     T result = get(name);
161     if (useDefault && result == null) return get(def);
162     return result;
163   }

165   public Set<String> keySet() {
166     return immutableRegistry.keySet();
167   }

169   /**
170    * register a plugin by a name
171    */
172   public T put(String name, T plugin) {
173     if (plugin == null) return null;
174     PluginHolder<T> old = put(name, new PluginHolder<T>(null, plugin));
175     return old == null ? null : old.get();
176   }
177 
178 
179   PluginHolder<T> put(String name, PluginHolder<T> plugin) {
180     PluginHolder<T> old = registry.put(name, plugin);
181     if (plugin.pluginInfo != null && plugin.pluginInfo.isDefault()) {
182       setDefault(name);
183     }
184     if (plugin.isLoaded()) registerMBean(plugin.get(), core, name);
185     return old;
186   }
187 
188   void setDefault(String def) {
189     if (!registry.containsKey(def)) return;
190     if (this.def != null) log.warn("Multiple defaults for : " + meta.getCleanTag());
191     this.def = def;
192   }

194   public Map<String, PluginHolder<T>> getRegistry() {
195     return immutableRegistry;
196   }
197 
198   public boolean contains(String name) {
199     return registry.containsKey(name);
200   }
201 
202   String getDefault() {
203     return def;
204   }

206   T remove(String name) {
207     PluginHolder<T> removed = registry.remove(name);
208     return removed == null ? null : removed.get();
209   }

211   void init(Map<String, T> defaults, SolrCore solrCore) {
212     init(defaults, solrCore, solrCore.getSolrConfig().getPluginInfos(klass.getName()));
213   }

215   /**
216    * Initializes the plugins after reading the meta data from {@link org.apache.solr.core.SolrConfig}.
217    *
218    * @param defaults These will be registered if not explicitly specified
219    */
220   void init(Map<String, T> defaults, SolrCore solrCore, List<PluginInfo> infos) {
221     core = solrCore;
222     for (PluginInfo info : infos) {
223       PluginHolder<T> o = createPlugin(info);
224       String name = info.name;
225       if (meta.clazz.equals(SolrRequestHandler.class)) name = RequestHandlers.normalize(info.name);
226       PluginHolder<T> old = put(name, o);
227       if (old != null) log.warn("Multiple entries of {} with name {}", meta.getCleanTag(), name);
228     }
229     for (Map.Entry<String, T> e : defaults.entrySet()) {
230       if (!contains(e.getKey())) {
231         put(e.getKey(), new PluginHolder<T>(null, e.getValue()));
232       }
233     }
234   }

236   /**
237    * To check if a plugin by a specified name is already loaded
238    */
239   public boolean isLoaded(String name) {
240     PluginHolder<T> result = registry.get(name);
241     if (result == null) return false;
242     return result.isLoaded();
243   }

245   private static void registerMBean(Object inst, SolrCore core, String pluginKey) {
246     if (core == null) return;
247     if (inst instanceof SolrInfoMBean) {
248       SolrInfoMBean mBean = (SolrInfoMBean) inst;
249       String name = (inst instanceof SolrRequestHandler) ? pluginKey : mBean.getName();
250       core.registerInfoBean(name, mBean);
251     }
252   }

255   /**
256    * Close this registry. This will in turn call a close on all the contained plugins
257    */
258   @Override
259   public void close() {
260     for (Map.Entry<String, PluginHolder<T>> e : registry.entrySet()) {
261       try {
262         e.getValue().close();
263       } catch (Exception exp) {
264         log.error("Error closing plugin " + e.getKey() + " of type : " + meta.getCleanTag(), exp);
265       }
266     }
267   }

269   /**
270    * An indirect reference to a plugin. It just wraps a plugin instance.
271    * subclasses may choose to lazily load the plugin
272    */
273   public static class PluginHolder<T> implements AutoCloseable {
274     private T inst;
275     protected final PluginInfo pluginInfo;
276 
277     public PluginHolder(PluginInfo info) {
278       this.pluginInfo = info;
279     }
280 
281     public PluginHolder(PluginInfo info, T inst) {
282       this.inst = inst;
283       this.pluginInfo = info;
284     }
285 
286     public T get() {
287       return inst;
288     }
289 
290     public boolean isLoaded() {
291       return inst != null;
292     }
293 
294     @Override
295     public void close() throws Exception {
296       // TODO: there may be a race here.  One thread can be creating a plugin
297       // and another thread can come along and close everything (missing the plugin
298       // that is in the state of being created and will probably never have close() called on it).
299       // can close() be called concurrently with other methods?
300       if (isLoaded()) {
301         T myInst = get();
302         if (myInst != null && myInst instanceof AutoCloseable) ((AutoCloseable) myInst).close();
303       }
304     }
305 
306     public String getClassName() {
307       if (isLoaded()) return inst.getClass().getName();
308       if (pluginInfo != null) return pluginInfo.className;
309       return null;
310     }
311   }

313   /**
314    * A class that loads plugins Lazily. When the get() method is invoked
315    * the Plugin is initialized and returned.
316    */
317   public static class LazyPluginHolder<T> extends PluginHolder<T> {
318     private volatile T lazyInst;
319     private final SolrConfig.SolrPluginInfo pluginMeta;
320     protected SolrException solrException;
321     private final SolrCore core;
322     protected ResourceLoader resourceLoader;
323 
324 
325     LazyPluginHolder(SolrConfig.SolrPluginInfo pluginMeta, PluginInfo pluginInfo, SolrCore core, ResourceLoader loader) {
326       super(pluginInfo);
327       this.pluginMeta = pluginMeta;
328       this.core = core;
329       this.resourceLoader = loader;
330       if (loader instanceof MemClassLoader) {
331         if (!"true".equals(System.getProperty("enable.runtime.lib"))) {
332           String s = "runtime library loading is not enabled, start Solr with -Denable.runtime.lib=true";
333           log.warn(s);
334           solrException = new SolrException(SolrException.ErrorCode.SERVER_ERROR, s);
335         }
336       }
337     }
338 
339     @Override
340     public boolean isLoaded() {
341       return lazyInst != null;
342     }
343 
344     @Override
345     public T get() {
346       if (lazyInst != null) return lazyInst;
347       if (solrException != null) {
348         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,"Unrecoverable error", solrException);
349       }
350       if (createInst()) {
351         // check if we created the instance to avoid registering it again
352         registerMBean(lazyInst, core, pluginInfo.name);
353       }
354       return lazyInst;
355     }
356 
357     private synchronized boolean createInst() {
358       if (lazyInst != null) return false;
359       log.info("Going to create a new {} with {} ", pluginMeta.getCleanTag(), pluginInfo.toString());
360       if (resourceLoader instanceof MemClassLoader) {
361         MemClassLoader loader = (MemClassLoader) resourceLoader;
362         loader.loadJars();
363       }
364       Class<T> clazz = (Class<T>) pluginMeta.clazz;
365       T localInst = core.createInstance(pluginInfo.className, clazz, pluginMeta.getCleanTag(), null, resourceLoader);
366       initInstance(localInst, pluginInfo);
367       if (localInst instanceof SolrCoreAware) {
368         SolrResourceLoader.assertAwareCompatibility(SolrCoreAware.class, localInst);
369         ((SolrCoreAware) localInst).inform(core);
370       }
371       if (localInst instanceof ResourceLoaderAware) {
372         SolrResourceLoader.assertAwareCompatibility(ResourceLoaderAware.class, localInst);
373         try {
374           ((ResourceLoaderAware) localInst).inform(core.getResourceLoader());
375         } catch (IOException e) {
376           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "error initializing component", e);
377         }
378       }
379       lazyInst = localInst;  // only assign the volatile until after the plugin is completely ready to use
380       return true;
381     }
382 
383 
384   }

386   /**
387    * This represents a Runtime Jar. A jar requires two details , name and version
388    */
389   public static class RuntimeLib implements PluginInfoInitialized, AutoCloseable {
390     private String name, version, sig;
391     private JarRepository.JarContentRef jarContent;
392     private final CoreContainer coreContainer;
393     private boolean verified = false;
394 
395     @Override
396     public void init(PluginInfo info) {
397       name = info.attributes.get(NAME);
398       Object v = info.attributes.get("version");
399       if (name == null || v == null) {
400         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "runtimeLib must have name and version");
401       }
402       version = String.valueOf(v);
403       sig = info.attributes.get("sig");
404     }
405 
406     public RuntimeLib(SolrCore core) {
407       coreContainer = core.getCoreDescriptor().getCoreContainer();
408     }
409 
410 
411     void loadJar() {
412       if (jarContent != null) return;
413       synchronized (this) {
414         if (jarContent != null) return;
415         jarContent = coreContainer.getJarRepository().getJarIncRef(name + "/" + version);
416       }
417     }
418 
419     public String getName() {
420       return name;
421     }
422 
423     public String getVersion() {
424       return version;
425     }
426 
427     public String getSig() {
428       return sig;
429 
430     }
431 
432     public ByteBuffer getFileContent(String entryName) throws IOException {
433       if (jarContent == null)
434         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "jar not available: " + name + "/" + version);
435       return jarContent.jar.getFileContent(entryName);
436 
437     }
438 
439     @Override
440     public void close() throws Exception {
441       if (jarContent != null) coreContainer.getJarRepository().decrementJarRefCount(jarContent);
442     }
443 
444     public static List<RuntimeLib> getLibObjects(SolrCore core, List<PluginInfo> libs) {
445       List<RuntimeLib> l = new ArrayList<>(libs.size());
446       for (PluginInfo lib : libs) {
447         RuntimeLib rtl = new RuntimeLib(core);
448         rtl.init(lib);
449         l.add(rtl);
450       }
451       return l;
452     }
453 
454     public void verify() throws Exception {
455       if (verified) return;
456       if (jarContent == null) {
457         log.error("Calling verify before loading the jar");
458         return;
459       }
460 
461       if (!coreContainer.isZooKeeperAware())
462         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Signing jar is possible only in cloud");
463       Map<String, byte[]> keys = CloudUtil.getTrustedKeys(coreContainer.getZkController().getZkClient(), "exe");
464       if (keys.isEmpty()) {
465         if (sig == null) {
466           verified = true;
467           log.info("A run time lib {} is loaded  without verification ", name);
468           return;
469         } else {
470           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "No public keys are available in ZK to verify signature for runtime lib  " + name);
471         }
472       } else if (sig == null) {
473         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, StrUtils.formatString("runtimelib {0} should be signed with one of the keys in ZK /keys/exe ", name));
474       }
475 
476       try {
477         String matchedKey = jarContent.jar.checkSignature(sig, new CryptoKeys(keys));
478         if (matchedKey == null)
479           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "No key matched signature for jar : " + name + " version: " + version);
480         log.info("Jar {} signed with {} successfully verified", name, matchedKey);
481       } catch (Exception e) {
482         if (e instanceof SolrException) throw e;
483         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Error verifying key ", e);
484       }
485     }
486   }
