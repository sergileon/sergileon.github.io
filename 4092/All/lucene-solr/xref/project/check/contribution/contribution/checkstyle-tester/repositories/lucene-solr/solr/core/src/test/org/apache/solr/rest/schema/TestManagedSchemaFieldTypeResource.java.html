292   /**
293    * Helper function to check fieldType settings against a set of expected values.
294    */
295   protected void checkFieldTypeProps(Map<String,Object> expected, int expectedChildCount) {
296     checkFieldTypeProps(expected, expectedChildCount, null);
297   }
298   
299   protected void checkFieldTypeProps(Map<String,Object> expected, int expectedChildCount, List<String> addlTests) {
300     String fieldTypeName = (String)expected.get("name");
301     
302     List<String> tests = new ArrayList<>();
303     tests.add("count(/response/lst[@name='fieldType']) = 1");
304     tests.add("count(/response/lst[@name='fieldType']/*) = "+expectedChildCount);
305     tests.add("count(/response/lst[@name='fieldType']/arr[@name='fields']/*) = 0");
306     tests.add("count(/response/lst[@name='fieldType']/arr[@name='dynamicFields']/*) = 0");
307     for (Map.Entry<String,Object> next : expected.entrySet()) {
308       Object val = next.getValue();
309       String pathType = null;
310       if (val instanceof Boolean)
311         pathType = "bool";
312       else if (val instanceof String)
313         pathType = "str";
314       else
315         fail("Unexpected value type "+val.getClass().getName());
316       // NOTE: it seems like the fieldtypes endpoint only returns strings or booleans
317       
318       String xpath = 
319           "/response/lst[@name='fieldType']/"+pathType+"[@name='"+next.getKey()+"']";
320       tests.add(xpath+" = '"+val+"'");
321     }
322     
323     if (addlTests != null)
324       tests.addAll(addlTests);
325     
326     assertQ("/schema/fieldtypes/"+fieldTypeName+"?indent=on&wt=xml&showDefaults=true",
327         tests.toArray(new String[0]));
328   }
