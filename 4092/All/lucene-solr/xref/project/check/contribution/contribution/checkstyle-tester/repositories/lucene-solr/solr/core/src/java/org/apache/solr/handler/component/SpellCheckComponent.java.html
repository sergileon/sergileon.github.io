115   @Override
116   @SuppressWarnings("unchecked")
117   public void prepare(ResponseBuilder rb) throws IOException {
118 
119     SolrParams params = rb.req.getParams();
120     if (!params.getBool(COMPONENT_NAME, false)) {
121       return;
122     }
123     SolrSpellChecker spellChecker = getSpellChecker(params);
124     if (params.getBool(SPELLCHECK_BUILD, false)) {
125       spellChecker.build(rb.req.getCore(), rb.req.getSearcher());
126       rb.rsp.add("command", "build");
127     } else if (params.getBool(SPELLCHECK_RELOAD, false)) {
128       spellChecker.reload(rb.req.getCore(), rb.req.getSearcher());
129       rb.rsp.add("command", "reload");
130     }
131   }

133   @Override
134   @SuppressWarnings("unchecked")
135   public void process(ResponseBuilder rb) throws IOException {
136     SolrParams params = rb.req.getParams();
137     if (!params.getBool(COMPONENT_NAME, false) || spellCheckers.isEmpty()) {
138       return;
139     }
140     boolean shardRequest = "true".equals(params.get(ShardParams.IS_SHARD));
141     String q = params.get(SPELLCHECK_Q);
142     SolrSpellChecker spellChecker = getSpellChecker(params);
143     Collection<Token> tokens = null;
144 
145     if (q != null) {
146       //we have a spell check param, tokenize it with the query analyzer applicable for this spellchecker
147       tokens = getTokens(q, spellChecker.getQueryAnalyzer());
148     } else {
149       q = rb.getQueryString();
150       if (q == null) {
151         q = params.get(CommonParams.Q);
152       }
153       tokens = queryConverter.convert(q);
154     }
155     if (tokens != null && tokens.isEmpty() == false) {
156       if (spellChecker != null) {
157         int count = params.getInt(SPELLCHECK_COUNT, 1);
158         boolean onlyMorePopular = params.getBool(SPELLCHECK_ONLY_MORE_POPULAR, DEFAULT_ONLY_MORE_POPULAR);
159         boolean extendedResults = params.getBool(SPELLCHECK_EXTENDED_RESULTS, false);
160         boolean collate = params.getBool(SPELLCHECK_COLLATE, false);
161         float accuracy = params.getFloat(SPELLCHECK_ACCURACY, Float.MIN_VALUE);
162         int alternativeTermCount = params.getInt(SpellingParams.SPELLCHECK_ALTERNATIVE_TERM_COUNT, 0);
163         Integer maxResultsForSuggest = params.getInt(SpellingParams.SPELLCHECK_MAX_RESULTS_FOR_SUGGEST);
164         ModifiableSolrParams customParams = new ModifiableSolrParams();
165         for (String checkerName : getDictionaryNames(params)) {
166           customParams.add(getCustomParams(checkerName, params));
167         }
168 
169         Integer hitsInteger = (Integer) rb.rsp.getToLog().get("hits");
170         long hits = 0;
171         if (hitsInteger == null) {
172           hits = rb.getNumberDocumentsFound();
173         } else {
174           hits = hitsInteger.longValue();
175         }
176         SpellingResult spellingResult = null;
177         if (maxResultsForSuggest == null || hits <= maxResultsForSuggest) {
178           SuggestMode suggestMode = SuggestMode.SUGGEST_WHEN_NOT_IN_INDEX;
179           if (onlyMorePopular) {
180             suggestMode = SuggestMode.SUGGEST_MORE_POPULAR;
181           } else if (alternativeTermCount > 0) {
182             suggestMode = SuggestMode.SUGGEST_ALWAYS;
183           }
184 
185           IndexReader reader = rb.req.getSearcher().getIndexReader();
186           SpellingOptions options = new SpellingOptions(tokens, reader, count,
187               alternativeTermCount, suggestMode, extendedResults, accuracy,
188               customParams);
189           spellingResult = spellChecker.getSuggestions(options);
190         } else {
191           spellingResult = new SpellingResult();
192         }
193         boolean isCorrectlySpelled = hits > (maxResultsForSuggest==null ? 0 : maxResultsForSuggest);
194 
195         NamedList response = new SimpleOrderedMap();
196         NamedList suggestions = toNamedList(shardRequest, spellingResult, q, extendedResults);
197         response.add("suggestions", suggestions);
198 
199         if (extendedResults) {
200           response.add("correctlySpelled", isCorrectlySpelled);
201         }
202         if (collate) {
203           addCollationsToResponse(params, spellingResult, rb, q, response, spellChecker.isSuggestionsMayOverlap());
204         }
205         if (shardRequest) {
206           addOriginalTermsToResponse(response, tokens);
207         }
208 
209         rb.rsp.add("spellcheck", response);
210 
211       } else {
212         throw new SolrException(SolrException.ErrorCode.NOT_FOUND,
213             "Specified dictionaries do not exist: " + getDictionaryNameAsSingleString(getDictionaryNames(params)));
214       }
215     }
216   }

218   @SuppressWarnings("unchecked")
219   protected void addCollationsToResponse(SolrParams params, SpellingResult spellingResult, ResponseBuilder rb, String q,
220       NamedList response, boolean suggestionsMayOverlap) {
221     int maxCollations = params.getInt(SPELLCHECK_MAX_COLLATIONS, 1);
222     int maxCollationTries = params.getInt(SPELLCHECK_MAX_COLLATION_TRIES, 0);
223     int maxCollationEvaluations = params.getInt(SPELLCHECK_MAX_COLLATION_EVALUATIONS, 10000);
224     boolean collationExtendedResults = params.getBool(SPELLCHECK_COLLATE_EXTENDED_RESULTS, false);
225     int maxCollationCollectDocs = params.getInt(SPELLCHECK_COLLATE_MAX_COLLECT_DOCS, 0);
226     // If not reporting hits counts, don't bother collecting more than 1 document per try.
227     if (!collationExtendedResults) {
228       maxCollationCollectDocs = 1;
229     }
230     boolean shard = params.getBool(ShardParams.IS_SHARD, false);
231     SpellCheckCollator collator = new SpellCheckCollator()
232         .setMaxCollations(maxCollations)
233         .setMaxCollationTries(maxCollationTries)
234         .setMaxCollationEvaluations(maxCollationEvaluations)
235         .setSuggestionsMayOverlap(suggestionsMayOverlap)
236         .setDocCollectionLimit(maxCollationCollectDocs)
237     ;
238     List<SpellCheckCollation> collations = collator.collate(spellingResult, q, rb);
239     //by sorting here we guarantee a non-distributed request returns all 
240     //results in the same order as a distributed request would,
241     //even in cases when the internal rank is the same.
242     Collections.sort(collations);
243 
244     NamedList collationList = new NamedList();
245     for (SpellCheckCollation collation : collations) {
246       if (collationExtendedResults) {
247         NamedList extendedResult = new SimpleOrderedMap();
248         extendedResult.add("collationQuery", collation.getCollationQuery());
249         extendedResult.add("hits", collation.getHits());
250         extendedResult.add("misspellingsAndCorrections", collation.getMisspellingsAndCorrections());
251         if(maxCollationTries>0 && shard)
252         {
253           extendedResult.add("collationInternalRank", collation.getInternalRank());
254         }
255         collationList.add("collation", extendedResult);
256       } else {
257         collationList.add("collation", collation.getCollationQuery());
258         if (maxCollationTries>0 && shard) {
259           collationList.add("collationInternalRank", collation.getInternalRank());
260         }
261       }
262     }
263     response.add("collations", collationList);
264   }

266   private void addOriginalTermsToResponse(NamedList response, Collection<Token> originalTerms) {
267     List<String> originalTermStr = new ArrayList<String>();
268     for(Token t : originalTerms) {
269       originalTermStr.add(t.toString());
270     }
271     response.add("originalTerms", originalTermStr);
272   }

274   /**
275    * For every param that is of the form "spellcheck.[dictionary name].XXXX=YYYY, add
276    * XXXX=YYYY as a param to the custom param list
277    * @param params The original SolrParams
278    * @return The new Params
279    */
280   protected SolrParams getCustomParams(String dictionary, SolrParams params) {
281     ModifiableSolrParams result = new ModifiableSolrParams();
282     Iterator<String> iter = params.getParameterNamesIterator();
283     String prefix = SpellingParams.SPELLCHECK_PREFIX + "." + dictionary + ".";
284     while (iter.hasNext()){
285       String nxt = iter.next();
286       if (nxt.startsWith(prefix)){
287         result.add(nxt.substring(prefix.length()), params.getParams(nxt));
288       }
289     }
290     return result;
291   }

294   @Override
295   public void modifyRequest(ResponseBuilder rb, SearchComponent who, ShardRequest sreq) {
296     SolrParams params = rb.req.getParams();
297     if (!params.getBool(COMPONENT_NAME, false)) return;
298     int purpose = rb.grouping() ? ShardRequest.PURPOSE_GET_TOP_GROUPS : ShardRequest.PURPOSE_GET_TOP_IDS;
299     if ((sreq.purpose & purpose) != 0) {
300       // fetch at least 5 suggestions from each shard
301       int count = sreq.params.getInt(SPELLCHECK_COUNT, 1);
302       if (count < 5)  count = 5;
303       sreq.params.set(SPELLCHECK_COUNT, count);
304       sreq.params.set("spellcheck", "true");
305     } else  {
306       sreq.params.set("spellcheck", "false");
307     }
308   }

310   @Override
311   @SuppressWarnings({"unchecked", "deprecation"})
312   public void finishStage(ResponseBuilder rb) {
313     SolrParams params = rb.req.getParams();
314     if (!params.getBool(COMPONENT_NAME, false) || rb.stage != ResponseBuilder.STAGE_GET_FIELDS)
315       return;
316 
317     boolean extendedResults = params.getBool(SPELLCHECK_EXTENDED_RESULTS, false);
318     boolean collate = params.getBool(SPELLCHECK_COLLATE, false);
319     boolean collationExtendedResults = params.getBool(SPELLCHECK_COLLATE_EXTENDED_RESULTS, false);
320     int maxCollationTries = params.getInt(SPELLCHECK_MAX_COLLATION_TRIES, 0);
321     int maxCollations = params.getInt(SPELLCHECK_MAX_COLLATIONS, 1);
322     Integer maxResultsForSuggest = params.getInt(SpellingParams.SPELLCHECK_MAX_RESULTS_FOR_SUGGEST);
323     int count = rb.req.getParams().getInt(SPELLCHECK_COUNT, 1);
324     int numSug = Math.max(count, AbstractLuceneSpellChecker.DEFAULT_SUGGESTION_COUNT);
325 
326     String origQuery = params.get(SPELLCHECK_Q);
327     if (origQuery == null) {
328       origQuery = rb.getQueryString();
329       if (origQuery == null) {
330         origQuery = params.get(CommonParams.Q);
331       }
332     }
333 
334     long hits = rb.grouping() ? rb.totalHitCount : rb.getNumberDocumentsFound();
335     boolean isCorrectlySpelled = hits > (maxResultsForSuggest==null ? 0 : maxResultsForSuggest);
336 
337     SpellCheckMergeData mergeData = new SpellCheckMergeData();
338     if (maxResultsForSuggest==null || !isCorrectlySpelled) {
339       for (ShardRequest sreq : rb.finished) {
340         for (ShardResponse srsp : sreq.responses) {
341           NamedList nl = null;
342           try {
343             nl = (NamedList) srsp.getSolrResponse().getResponse().get("spellcheck");
344           } catch (Exception e) {
345             if (rb.req.getParams().getBool(ShardParams.SHARDS_TOLERANT, false)) {
346               continue; // looks like a shard did not return anything
347             }
348             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
349                 "Unable to read spelling info for shard: " + srsp.getShard(), e);
350           }
351           LOG.info(srsp.getShard() + " " + nl);
352           if (nl != null) {
353             mergeData.totalNumberShardResponses++;
354             collectShardSuggestions(nl, mergeData);
355             collectShardCollations(mergeData, nl, maxCollationTries);
356           }
357         }
358       }
359     }
360 
361     // all shard responses have been collected
362     // create token and get top suggestions
363     SolrSpellChecker checker = getSpellChecker(rb.req.getParams());
364     SpellingResult result = checker.mergeSuggestions(mergeData, numSug, count, extendedResults);
365 
366     NamedList response = new SimpleOrderedMap();
367 
368     NamedList suggestions = toNamedList(false, result, origQuery, extendedResults);
369     response.add("suggestions", suggestions);
370 
371     if (extendedResults) {
372       response.add("correctlySpelled", isCorrectlySpelled);
373     }
374 
375     if (collate) {
376       SpellCheckCollation[] sortedCollations = mergeData.collations.values()
377           .toArray(new SpellCheckCollation[mergeData.collations.size()]);
378       Arrays.sort(sortedCollations);
379 
380       NamedList collations = new NamedList();
381       int i = 0;
382       while (i < maxCollations && i < sortedCollations.length) {
383         SpellCheckCollation collation = sortedCollations[i];
384         i++;
385         if (collationExtendedResults) {
386           SimpleOrderedMap extendedResult = new SimpleOrderedMap();
387           extendedResult.add("collationQuery", collation.getCollationQuery());
388           extendedResult.add("hits", collation.getHits());
389           extendedResult.add("misspellingsAndCorrections", collation
390               .getMisspellingsAndCorrections());
391           collations.add("collation", extendedResult);
392         } else {
393           collations.add("collation", collation.getCollationQuery());
394         }
395       }
396 
397       response.add("collations", collations);
398     }
399 
400     rb.rsp.add("spellcheck", response);
401   }

403   @SuppressWarnings("unchecked")
404   private void collectShardSuggestions(NamedList nl, SpellCheckMergeData mergeData) {
405     SpellCheckResponse spellCheckResp = new SpellCheckResponse(nl);
406     Iterable<Object> originalTermStrings = (Iterable<Object>) nl.get("originalTerms");
407     if(originalTermStrings!=null) {
408       mergeData.originalTerms = new HashSet<>();
409       for (Object originalTermObj : originalTermStrings) {
410         mergeData.originalTerms.add(originalTermObj.toString());
411       }
412     }
413     for (SpellCheckResponse.Suggestion suggestion : spellCheckResp.getSuggestions()) {
414       mergeData.origVsSuggestion.put(suggestion.getToken(), suggestion);
415       HashSet<String> suggested = mergeData.origVsSuggested.get(suggestion.getToken());
416       if (suggested == null) {
417         suggested = new HashSet<>();
418         mergeData.origVsSuggested.put(suggestion.getToken(), suggested);
419       }
420 
421       // sum up original frequency          
422       int origFreq = 0;
423       Integer o = mergeData.origVsFreq.get(suggestion.getToken());
424       if (o != null)  origFreq += o;
425       origFreq += suggestion.getOriginalFrequency();
426       mergeData.origVsFreq.put(suggestion.getToken(), origFreq);
427 
428       //# shards reporting
429       Integer origShards = mergeData.origVsShards.get(suggestion.getToken());
430       if(origShards==null) {
431         mergeData.origVsShards.put(suggestion.getToken(), 1);
432       } else {
433         mergeData.origVsShards.put(suggestion.getToken(), ++origShards);
434       }
435 
436       // find best suggestions
437       for (int i = 0; i < suggestion.getNumFound(); i++) {
438         String alternative = suggestion.getAlternatives().get(i);
439         suggested.add(alternative);
440         SuggestWord sug = mergeData.suggestedVsWord.get(alternative);
441         if (sug == null)  {
442           sug = new SuggestWord();
443           mergeData.suggestedVsWord.put(alternative, sug);
444         }
445         sug.string = alternative;
446         // alternative frequency is present only for extendedResults=true
447         if (suggestion.getAlternativeFrequencies() != null
448             && suggestion.getAlternativeFrequencies().size() > 0) {
449           Integer freq = suggestion.getAlternativeFrequencies().get(i);
450           if (freq != null) sug.freq += freq;
451         }
452       }
453     }
454   }

456   @SuppressWarnings("unchecked")
457   private void collectShardCollations(SpellCheckMergeData mergeData, NamedList spellCheckResponse, int maxCollationTries) {
458     Map<String, SpellCheckCollation> collations = mergeData.collations;
459     NamedList collationHolder = (NamedList) spellCheckResponse.get("collations");
460     if(collationHolder != null) {
461       List<Object> collationList = collationHolder.getAll("collation");
462       List<Object> collationRankList = collationHolder.getAll("collationInternalRank");
463       int i=0;
464       if(collationList != null) {
465         for(Object o : collationList)
466         {
467           if(o instanceof String)
468           {
469             SpellCheckCollation coll = new SpellCheckCollation();
470             coll.setCollationQuery((String) o);
471             if(collationRankList!= null && collationRankList.size()>0)
472             {
473               coll.setInternalRank((Integer) collationRankList.get(i));
474               i++;
475             }
476             SpellCheckCollation priorColl = collations.get(coll.getCollationQuery());
477             if(priorColl != null)
478             {
479               coll.setInternalRank(Math.max(coll.getInternalRank(),priorColl.getInternalRank()));
480             }
481             collations.put(coll.getCollationQuery(), coll);
482           } else
483           {
484             NamedList expandedCollation = (NamedList) o;
485             SpellCheckCollation coll = new SpellCheckCollation();
486             coll.setCollationQuery((String) expandedCollation.get("collationQuery"));
487             coll.setHits((Integer) expandedCollation.get("hits"));
488             if(maxCollationTries>0)
489             {
490               coll.setInternalRank((Integer) expandedCollation.get("collationInternalRank"));
491             }
492             coll.setMisspellingsAndCorrections((NamedList) expandedCollation.get("misspellingsAndCorrections"));
493             SpellCheckCollation priorColl = collations.get(coll.getCollationQuery());
494             if(priorColl != null)
495             {
496               coll.setHits(coll.getHits() + priorColl.getHits());
497               coll.setInternalRank(Math.max(coll.getInternalRank(),priorColl.getInternalRank()));
498             }
499             collations.put(coll.getCollationQuery(), coll);
500           }
501         }
502       }
503     }
504   }

506   private Collection<Token> getTokens(String q, Analyzer analyzer) throws IOException {
507     Collection<Token> result = new ArrayList<>();
508     assert analyzer != null;
509     try (TokenStream ts = analyzer.tokenStream("", q)) {
510       ts.reset();
511       // TODO: support custom attributes
512       CharTermAttribute termAtt = ts.addAttribute(CharTermAttribute.class);
513       OffsetAttribute offsetAtt = ts.addAttribute(OffsetAttribute.class);
514       TypeAttribute typeAtt = ts.addAttribute(TypeAttribute.class);
515       FlagsAttribute flagsAtt = ts.addAttribute(FlagsAttribute.class);
516       PayloadAttribute payloadAtt = ts.addAttribute(PayloadAttribute.class);
517       PositionIncrementAttribute posIncAtt = ts.addAttribute(PositionIncrementAttribute.class);
518 
519       while (ts.incrementToken()){
520         Token token = new Token();
521         token.copyBuffer(termAtt.buffer(), 0, termAtt.length());
522         token.setOffset(offsetAtt.startOffset(), offsetAtt.endOffset());
523         token.setType(typeAtt.type());
524         token.setFlags(flagsAtt.getFlags());
525         token.setPayload(payloadAtt.getPayload());
526         token.setPositionIncrement(posIncAtt.getPositionIncrement());
527         result.add(token);
528       }
529       ts.end();
530       return result;
531     }
532   }

534   protected SolrSpellChecker getSpellChecker(SolrParams params) {
535     String[] dictName = getDictionaryNames(params);
536     if (dictName.length == 1) {
537       return spellCheckers.get(dictName[0]);
538     } else {
539       String singleStr = getDictionaryNameAsSingleString(dictName);
540       SolrSpellChecker ssc = spellCheckers.get(singleStr);
541       if (ssc == null) {
542         ConjunctionSolrSpellChecker cssc = new ConjunctionSolrSpellChecker();
543         for (String dn : dictName) {
544           cssc.addChecker(spellCheckers.get(dn));
545         }
546         ssc = cssc;
547       }
548       return ssc;
549     }
550   }

552   private String getDictionaryNameAsSingleString(String[] dictName) {
553     StringBuilder sb = new StringBuilder();
554     for (String dn : dictName) {
555       if (sb.length() > 0) {
556         sb.append(" ");
557       }
558       sb.append(dn);
559     }
560     return sb.toString();
561   }

563   private String[] getDictionaryNames(SolrParams params) {
564     String[] dictName = params.getParams(SPELLCHECK_DICT);
565     if (dictName == null) {
566       return new String[] {SolrSpellChecker.DEFAULT_DICTIONARY_NAME};
567     }
568     return dictName;
569   }

578   protected NamedList toNamedList(boolean shardRequest,
579       SpellingResult spellingResult, String origQuery, boolean extendedResults) {
580     NamedList result = new NamedList();
581     Map<Token,LinkedHashMap<String,Integer>> suggestions = spellingResult
582         .getSuggestions();
583     boolean hasFreqInfo = spellingResult.hasTokenFrequencyInfo();
584     boolean hasSuggestions = false;
585     boolean hasZeroFrequencyToken = false;
586     for (Map.Entry<Token,LinkedHashMap<String,Integer>> entry : suggestions
587         .entrySet()) {
588       Token inputToken = entry.getKey();
589       String tokenString = new String(inputToken.buffer(), 0, inputToken
590           .length());
591       Map<String,Integer> theSuggestions = new LinkedHashMap<>(
592           entry.getValue());
593       Iterator<String> sugIter = theSuggestions.keySet().iterator();
594       while (sugIter.hasNext()) {
595         String sug = sugIter.next();
596         if (sug.equals(tokenString)) {
597           sugIter.remove();
598         }
599       }
600       if (theSuggestions.size() > 0) {
601         hasSuggestions = true;
602       }
603       if (theSuggestions != null && (theSuggestions.size() > 0 || shardRequest)) {
604         SimpleOrderedMap suggestionList = new SimpleOrderedMap();
605         suggestionList.add("numFound", theSuggestions.size());
606         suggestionList.add("startOffset", inputToken.startOffset());
607         suggestionList.add("endOffset", inputToken.endOffset());
608 
609         // Logical structure of normal (non-extended) results:
610         // "suggestion":["alt1","alt2"]
611         //
612         // Logical structure of the extended results:
613         // "suggestion":[
614         // {"word":"alt1","freq":7},
615         // {"word":"alt2","freq":4}
616         // ]
617         if (extendedResults && hasFreqInfo) {
618           suggestionList.add("origFreq", spellingResult
619               .getTokenFrequency(inputToken));
620 
621           ArrayList<SimpleOrderedMap> sugs = new ArrayList<>();
622           suggestionList.add("suggestion", sugs);
623           for (Map.Entry<String,Integer> suggEntry : theSuggestions.entrySet()) {
624             SimpleOrderedMap sugEntry = new SimpleOrderedMap();
625             sugEntry.add("word", suggEntry.getKey());
626             sugEntry.add("freq", suggEntry.getValue());
627             sugs.add(sugEntry);
628           }
629         } else {
630           suggestionList.add("suggestion", theSuggestions.keySet());
631         }
632 
633         if (hasFreqInfo) {
634           Integer tokenFrequency = spellingResult.getTokenFrequency(inputToken);
635           if (tokenFrequency==null || tokenFrequency == 0) {
636             hasZeroFrequencyToken = true;
637           }
638         }
639         result.add(tokenString, suggestionList);
640       }
641     }
642     return result;
643   }

645   @Override
646   public void inform(SolrCore core) {
647     if (initParams != null) {
648       LOG.info("Initializing spell checkers");
649       boolean hasDefault = false;
650       for (int i = 0; i < initParams.size(); i++) {
651         if (initParams.getName(i).equals("spellchecker")) {
652           Object cfg = initParams.getVal(i);
653           if (cfg instanceof NamedList) {
654             addSpellChecker(core, hasDefault, (NamedList) cfg);
655           } else if (cfg instanceof Map) {
656             addSpellChecker(core, hasDefault, new NamedList((Map) cfg));
657           } else if (cfg instanceof List) {
658             for (Object o : (List) cfg) {
659               if (o instanceof Map) {
660                 addSpellChecker(core, hasDefault, new NamedList((Map) o));
661               }
662             }
663           }
664         }
665       }
666 
667       Map<String, QueryConverter> queryConverters = new HashMap<>();
668       core.initPlugins(queryConverters,QueryConverter.class);
669 
670       //ensure that there is at least one query converter defined
671       if (queryConverters.size() == 0) {
672         LOG.info("No queryConverter defined, using default converter");
673         queryConverters.put("queryConverter", new SpellingQueryConverter());
674       }
675 
676       //there should only be one
677       if (queryConverters.size() == 1) {
678         queryConverter = queryConverters.values().iterator().next();
679         IndexSchema schema = core.getLatestSchema();
680         String fieldTypeName = (String) initParams.get("queryAnalyzerFieldType");
681         FieldType fieldType = schema.getFieldTypes().get(fieldTypeName);
682         Analyzer analyzer = fieldType == null ? new WhitespaceAnalyzer()
683                 : fieldType.getQueryAnalyzer();
684         //TODO: There's got to be a better way!  Where's Spring when you need it?
685         queryConverter.setAnalyzer(analyzer);
686       }
687     }
688   }

690   private boolean addSpellChecker(SolrCore core, boolean hasDefault, NamedList spellchecker) {
691     String className = (String) spellchecker.get("classname");
692     if (className == null) className = (String) spellchecker.get("class");
693     // TODO: this is a little bit sneaky: warn if class isnt supplied
694     // so that it's mandatory in a future release?
695     if (className == null)
696       className = IndexBasedSpellChecker.class.getName();
697     SolrResourceLoader loader = core.getResourceLoader();
698     SolrSpellChecker checker = loader.newInstance(className, SolrSpellChecker.class);
699     if (checker != null) {
700       String dictionary = checker.init(spellchecker, core);
701       if (dictionary != null) {
702         boolean isDefault = dictionary.equals(SolrSpellChecker.DEFAULT_DICTIONARY_NAME);
703         if (isDefault && !hasDefault) {
704           hasDefault = true;
705         } else if (isDefault && hasDefault) {
706           throw new RuntimeException("More than one dictionary is missing name.");
707         }
708         spellCheckers.put(dictionary, checker);
709       } else {
710         if (!hasDefault) {
711           spellCheckers.put(SolrSpellChecker.DEFAULT_DICTIONARY_NAME, checker);
712           hasDefault = true;
713         } else {
714           throw new RuntimeException("More than one dictionary is missing name.");
715         }
716       }
717       // Register event listeners for this SpellChecker
718       core.registerFirstSearcherListener(new SpellCheckerListener(core, checker, false, false));
719       boolean buildOnCommit = Boolean.parseBoolean((String) spellchecker.get("buildOnCommit"));
720       boolean buildOnOptimize = Boolean.parseBoolean((String) spellchecker.get("buildOnOptimize"));
721       if (buildOnCommit || buildOnOptimize) {
722         LOG.info("Registering newSearcher listener for spellchecker: " + checker.getDictionaryName());
723         core.registerNewSearcherListener(new SpellCheckerListener(core, checker, buildOnCommit, buildOnOptimize));
724       }
725     } else {
726       throw new RuntimeException("Can't load spell checker: " + className);
727     }
728     return hasDefault;
729   }

731   private static class SpellCheckerListener implements SolrEventListener {
732     private final SolrCore core;
733     private final SolrSpellChecker checker;
734     private final boolean buildOnCommit;
735     private final boolean buildOnOptimize;
736 
737     public SpellCheckerListener(SolrCore core, SolrSpellChecker checker, boolean buildOnCommit, boolean buildOnOptimize) {
738       this.core = core;
739       this.checker = checker;
740       this.buildOnCommit = buildOnCommit;
741       this.buildOnOptimize = buildOnOptimize;
742     }
743 
744     @Override
745     public void init(NamedList args) {
746     }
747 
748     @Override
749     public void newSearcher(SolrIndexSearcher newSearcher,
750                             SolrIndexSearcher currentSearcher) {
751       if (currentSearcher == null) {
752         // firstSearcher event
753         try {
754           LOG.info("Loading spell index for spellchecker: "
755                   + checker.getDictionaryName());
756           checker.reload(core, newSearcher);
757         } catch (IOException e) {
758           LOG.error( "Exception in reloading spell check index for spellchecker: " + checker.getDictionaryName(), e);
759         }
760       } else {
761         // newSearcher event
762         if (buildOnCommit)  {
763           buildSpellIndex(newSearcher);
764         } else if (buildOnOptimize) {
765           if (newSearcher.getIndexReader().leaves().size() == 1)  {
766             buildSpellIndex(newSearcher);
767           } else  {
768             LOG.info("Index is not optimized therefore skipping building spell check index for: " + checker.getDictionaryName());
769           }
770         }
771       }
772 
773     }
774 
775     private void buildSpellIndex(SolrIndexSearcher newSearcher) {
776       try {
777         LOG.info("Building spell index for spell checker: " + checker.getDictionaryName());
778         checker.build(core, newSearcher);
779       } catch (Exception e) {
780         LOG.error(
781                 "Exception in building spell check index for spellchecker: " + checker.getDictionaryName(), e);
782       }
783     }

787     }

791     }
