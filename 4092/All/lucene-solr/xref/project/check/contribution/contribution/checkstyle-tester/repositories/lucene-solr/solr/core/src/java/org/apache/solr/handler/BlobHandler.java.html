67  public class BlobHandler extends RequestHandlerBase implements PluginInfoInitialized {
68    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

70    private static final long DEFAULT_MAX_SIZE = 5 * 1024 * 1024; // 5MB
71    private long maxSize = DEFAULT_MAX_SIZE;
72  
73    @Override
74    public void handleRequestBody(final SolrQueryRequest req, SolrQueryResponse rsp) throws Exception {
75      String httpMethod = (String) req.getContext().get("httpMethod");
76      String path = (String) req.getContext().get("path");
77      SolrConfigHandler.setWt(req, JSON);
78  
79      List<String> pieces = StrUtils.splitSmart(path, '/');
80      String blobName = null;
81      if (pieces.size() >= 3) blobName = pieces.get(2);
82  
83      if ("POST".equals(httpMethod)) {
84        if (blobName == null || blobName.isEmpty()) {
85          rsp.add("error", "Name not found");
86          return;
87        }
88        String err = SolrConfigHandler.validateName(blobName);
89        if (err != null) {
90          log.warn("no blob name");
91          rsp.add("error", err);
92          return;
93        }
94        if (req.getContentStreams() == null) {
95          log.warn("no content stream");
96          rsp.add("error", "No stream");
97          return;
98        }
99  
100 
101       for (ContentStream stream : req.getContentStreams()) {
102         ByteBuffer payload = SimplePostTool.inputStreamToByteArray(stream.getStream(), maxSize);
103         MessageDigest m = MessageDigest.getInstance("MD5");
104         m.update(payload.array(), payload.position(), payload.limit());
105         String md5 = new BigInteger(1, m.digest()).toString(16);
106 
107         TopDocs duplicate = req.getSearcher().search(new TermQuery(new Term("md5", md5)), 1);
108         if (duplicate.totalHits > 0) {
109           rsp.add("error", "duplicate entry");
110           forward(req, null,
111               new MapSolrParams((Map) makeMap(
112                   "q", "md5:" + md5,
113                   "fl", "id,size,version,timestamp,blobName")),
114               rsp);
115           log.warn("duplicate entry for blob :" + blobName);
116           return;
117         }
118 
119         TopFieldDocs docs = req.getSearcher().search(new TermQuery(new Term("blobName", blobName)),
120             1, new Sort(new SortField("version", SortField.Type.LONG, true)));
121 
122         long version = 0;
123         if (docs.totalHits > 0) {
124           Document doc = req.getSearcher().doc(docs.scoreDocs[0].doc);
125           Number n = doc.getField("version").numericValue();
126           version = n.longValue();
127         }
128         version++;
129         String id = blobName + "/" + version;
130         Map<String, Object> doc = makeMap(
131             "id", id,
132             "md5", md5,
133             "blobName", blobName,
134             "version", version,
135             "timestamp", new Date(),
136             "size", payload.limit(),
137             "blob", payload);
138         verifyWithRealtimeGet(blobName, version, req, doc);
139         log.info(StrUtils.formatString("inserting new blob {0} ,size {1}, md5 {2}", doc.get("id"), String.valueOf(payload.limit()), md5));
140         indexMap(req, rsp, doc);
141         log.info(" Successfully Added and committed a blob with id {} and size {} ", id, payload.limit());
142 
143         break;
144       }
145 
146     } else {
147       int version = -1;
148       if (pieces.size() > 3) {
149         try {
150           version = Integer.parseInt(pieces.get(3));
151         } catch (NumberFormatException e) {
152           rsp.add("error", "Invalid version" + pieces.get(3));
153           return;
154         }
155 
156       }
157       if (ReplicationHandler.FILE_STREAM.equals(req.getParams().get(CommonParams.WT))) {
158         if (blobName == null) {
159           throw new SolrException(SolrException.ErrorCode.NOT_FOUND, "Please send the request in the format /blob/<blobName>/<version>");
160         } else {
161           String q = "blobName:{0}";
162           if (version != -1) q = "id:{0}/{1}";
163           QParser qparser = QParser.getParser(StrUtils.formatString(q, blobName, version), "lucene", req);
164           final TopDocs docs = req.getSearcher().search(qparser.parse(), 1, new Sort(new SortField("version", SortField.Type.LONG, true)));
165           if (docs.totalHits > 0) {
166             rsp.add(ReplicationHandler.FILE_STREAM, new SolrCore.RawWriter() {
167 
168               @Override
169               public void write(OutputStream os) throws IOException {
170                 Document doc = req.getSearcher().doc(docs.scoreDocs[0].doc);
171                 IndexableField sf = doc.getField("blob");
172                 FieldType fieldType = req.getSchema().getField("blob").getType();
173                 ByteBuffer buf = (ByteBuffer) fieldType.toObject(sf);
174                 if (buf == null) {
175                   //should never happen unless a user wrote this document directly
176                   throw new SolrException(SolrException.ErrorCode.NOT_FOUND, "Invalid document . No field called blob");
177                 } else {
178                   os.write(buf.array(), 0, buf.limit());
179                 }
180               }
181             });
182 
183           } else {
184             throw new SolrException(SolrException.ErrorCode.NOT_FOUND,
185                 StrUtils.formatString("Invalid combination of blobName {0} and version {1}", blobName, version));
186           }
187 
188         }
189       } else {
190         String q = "*:*";
191         if (blobName != null) {
192           q = "blobName:{0}";
193           if (version != -1) {
194             q = "id:{0}/{1}";
195           }
196         }
197 
198         forward(req, null,
199             new MapSolrParams((Map) makeMap(
200                 "q", StrUtils.formatString(q, blobName, version),
201                 "fl", "id,size,version,timestamp,blobName,md5",
202                 "sort", "version desc"))
203             , rsp);
204       }
205     }
206   }

208   private void verifyWithRealtimeGet(String blobName, long version, SolrQueryRequest req, Map<String, Object> doc) {
209     for (; ; ) {
210       SolrQueryResponse response = new SolrQueryResponse();
211       String id = blobName + "/" + version;
212       forward(req, "/get", new MapSolrParams(singletonMap("id", id)), response);
213       if (response.getValues().get("doc") == null) {
214         //ensure that the version does not exist
215         return;
216       } else {
217         log.info("id {} already exists trying next ", id);
218         version++;
219         doc.put("version", version);
220         id = blobName + "/" + version;
221         doc.put("id", id);
222       }
223     }
224 
225   }

227   public static void indexMap(SolrQueryRequest req, SolrQueryResponse rsp, Map<String, Object> doc) throws IOException {
228     SolrInputDocument solrDoc = new SolrInputDocument();
229     for (Map.Entry<String, Object> e : doc.entrySet()) solrDoc.addField(e.getKey(), e.getValue());
230     UpdateRequestProcessorChain processorChain = req.getCore().getUpdateProcessorChain(req.getParams());
231     UpdateRequestProcessor processor = processorChain.createProcessor(req, rsp);
232     AddUpdateCommand cmd = new AddUpdateCommand(req);
233     cmd.solrDoc = solrDoc;
234     log.info("Adding doc: " + doc);
235     processor.processAdd(cmd);
236     log.info("committing doc: " + doc);
237     processor.processCommit(new CommitUpdateCommand(req, false));
238     processor.finish();
239   }

243     if (StrUtils.splitSmart(subPath, '/').size() > 4) return null;
244     return this;

294   @Override
295   public void init(PluginInfo info) {
296     super.init(info.initArgs);
297     if (info.initArgs != null) {
298       NamedList invariants = (NamedList) info.initArgs.get(PluginInfo.INVARIANTS);
299       if (invariants != null) {
300         Object o = invariants.get("maxSize");
301         if (o != null) {
302           maxSize = Long.parseLong(String.valueOf(o));
303           maxSize = maxSize * 1024 * 1024;
304         }
305       }
306 
307     }
308   }

310   // This does not work for the general case of forwarding requests.  It probably currently
311   // works OK for real-time get (which is all that BlobHandler uses it for).
312   private static void forward(SolrQueryRequest req, String handler ,SolrParams params, SolrQueryResponse rsp){
313     try(LocalSolrQueryRequest r = new LocalSolrQueryRequest(req.getCore(), params)) {
314       req.getCore().getRequestHandler(handler).handleRequest(r, rsp);
315     }
