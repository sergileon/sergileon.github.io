45  public abstract class CollectionAdminRequest <Q extends CollectionAdminRequest<Q>> extends SolrRequest<CollectionAdminResponse> {
46  
47    protected CollectionAction action = null;
48  
49    private static String PROPERTY_PREFIX = "property.";
50  
51    protected CollectionAdminRequest setAction(CollectionAction action) {
52      this.action = action;
53      return this;
54    }
55  
56    public CollectionAdminRequest() {
57      super(METHOD.GET, "/admin/collections");
58    }
59  
60    public CollectionAdminRequest(String path) {
61      super(METHOD.GET, path);
62    }
63  
64    protected abstract Q getThis();
65  
66    @Override
67    public SolrParams getParams() {
68      if (action == null) {
69        throw new RuntimeException( "no action specified!" );
70      }
71      ModifiableSolrParams params = new ModifiableSolrParams();
72      params.set(CoreAdminParams.ACTION, action.toString());
73      return params;
74    }
75  
76    @Override
77    public Collection<ContentStream> getContentStreams() throws IOException {
78      return null;
79    }
80  
81    @Override
82    protected CollectionAdminResponse createResponse(SolrClient client) {
83      return new CollectionAdminResponse();
84    }
85    
86    protected void addProperties(ModifiableSolrParams params, Properties props) {
87      Iterator<Map.Entry<Object, Object>> iter = props.entrySet().iterator();
88      while(iter.hasNext()) {
89        Map.Entry<Object, Object> prop = iter.next();
90        String key = (String) prop.getKey();
91        String value = (String) prop.getValue();
92        params.set(PROPERTY_PREFIX + key, value);
93      }
94    }

273       }

275         addProperties(params, properties);
276       }
277       if (stateFormat != null) {
278         params.set(DocCollection.STATE_FORMAT, stateFormat);
279       }
280       if(rule != null) params.set("rule", rule);
281       if(snitch != null) params.set("snitch", snitch);
