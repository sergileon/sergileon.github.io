45  public class TestStressLucene extends TestRTGBase {
46  
47    // The purpose of this test is to roughly model how solr uses lucene
48    DirectoryReader reader;
49    @Test
50    public void testStressLuceneNRT() throws Exception {
51      final int commitPercent = 5 + random().nextInt(20);
52      final int softCommitPercent = 30+random().nextInt(75); // what percent of the commits are soft
53      final int deletePercent = 4+random().nextInt(25);
54      final int deleteByQueryPercent = 1+random().nextInt(5);
55      final int ndocs = 5 + (random().nextBoolean() ? random().nextInt(25) : random().nextInt(200));
56      int nWriteThreads = 5 + random().nextInt(25);
57  
58      final int maxConcurrentCommits = nWriteThreads;   // number of committers at a time... it should be <= maxWarmingSearchers
59  
60      final AtomicLong operations = new AtomicLong(100000);  // number of query operations to perform in total
61      int nReadThreads = 5 + random().nextInt(25);
62      final boolean tombstones = random().nextBoolean();
63      final boolean syncCommits = random().nextBoolean();
64  
65      verbose("commitPercent=", commitPercent);
66      verbose("softCommitPercent=",softCommitPercent);
67      verbose("deletePercent=",deletePercent);
68      verbose("deleteByQueryPercent=", deleteByQueryPercent);
69      verbose("ndocs=", ndocs);
70      verbose("nWriteThreads=", nWriteThreads);
71      verbose("nReadThreads=", nReadThreads);
72      verbose("maxConcurrentCommits=", maxConcurrentCommits);
73      verbose("operations=", operations);
74      verbose("tombstones=", tombstones);
75      verbose("syncCommits=", syncCommits);
76  
77      initModel(ndocs);
78  
79      final AtomicInteger numCommitting = new AtomicInteger();
80  
81      List<Thread> threads = new ArrayList<>();
82  
83  
84      final FieldType idFt = new FieldType();
85      idFt.setStored(true);
86      idFt.setOmitNorms(true);
87      idFt.setTokenized(false);
88      idFt.setIndexOptions(IndexOptions.DOCS);
89  
90      final FieldType ft2 = new FieldType();
91      ft2.setStored(true);
92  
93  
94      // model how solr does locking - only allow one thread to do a hard commit at once, and only one thread to do a soft commit, but
95      // a hard commit in progress does not stop a soft commit.
96      final Lock hardCommitLock = syncCommits ? new ReentrantLock() : null;
97      final Lock reopenLock = syncCommits ? new ReentrantLock() : null;
98  
99  
100     // RAMDirectory dir = new RAMDirectory();
101     // final IndexWriter writer = new IndexWriter(dir, new IndexWriterConfig(new WhitespaceAnalyzer()));
102 
103     Directory dir = newDirectory();
104 
105     final RandomIndexWriter writer = new RandomIndexWriter(random(), dir, newIndexWriterConfig(new MockAnalyzer(random())));
106     writer.setDoRandomForceMergeAssert(false);
107 
108     // writer.commit();
109     // reader = IndexReader.open(dir);
110     // make this reader an NRT reader from the start to avoid the first non-writer openIfChanged
111     // to only opening at the last commit point.
112     reader = DirectoryReader.open(writer.w, true);
113 
114     for (int i=0; i<nWriteThreads; i++) {
115       Thread thread = new Thread("WRITER"+i) {
116         Random rand = new Random(random().nextInt());
117 
118         @Override
119         public void run() {
120           try {
121             while (operations.get() > 0) {
122               int oper = rand.nextInt(100);
123 
124               if (oper < commitPercent) {
125                 if (numCommitting.incrementAndGet() <= maxConcurrentCommits) {
126                   Map<Integer,DocInfo> newCommittedModel;
127                   long version;
128                   DirectoryReader oldReader;
129 
130                   boolean softCommit = rand.nextInt(100) < softCommitPercent;
131 
132                   if (!softCommit) {
133                     // only allow one hard commit to proceed at once
134                     if (hardCommitLock != null) hardCommitLock.lock();
135                     verbose("hardCommit start");
136 
137                     writer.commit();
138                   }
139 
140                   if (reopenLock != null) reopenLock.lock();
141 
142                   synchronized(globalLock) {
143                     newCommittedModel = new HashMap<>(model);  // take a snapshot
144                     version = snapshotCount++;
145                     oldReader = reader;
146                     oldReader.incRef();  // increment the reference since we will use this for reopening
147                   }
148 
149                   if (!softCommit) {
150                     // must commit after taking a snapshot of the model
151                     // writer.commit();
152                   }
153 
154                   verbose("reopen start using", oldReader);
155 
156                   DirectoryReader newReader;
157                   if (softCommit) {
158                     newReader = DirectoryReader.openIfChanged(oldReader, writer.w, true);
159                   } else {
160                     // will only open to last commit
161                     newReader = DirectoryReader.openIfChanged(oldReader);
162                   }
163 
164 
165                   if (newReader == null) {
166                     oldReader.incRef();
167                     newReader = oldReader;
168                   }
169                   oldReader.decRef();
170 
171                   verbose("reopen result", newReader);
172 
173                   synchronized(globalLock) {
174                     assert newReader.getRefCount() > 0;
175                     assert reader.getRefCount() > 0;
176 
177                     // install the new reader if it's newest (and check the current version since another reader may have already been installed)
178                     if (newReader.getVersion() > reader.getVersion()) {
179                       reader.decRef();
180                       reader = newReader;
181 
182                       // install this snapshot only if it's newer than the current one
183                       if (version >= committedModelClock) {
184                         committedModel = newCommittedModel;
185                         committedModelClock = version;
186                       }
187 
188                     } else {
189                       // close if unused
190                       newReader.decRef();
191                     }
192 
193                   }
194 
195                   if (reopenLock != null) reopenLock.unlock();
196 
197                   if (!softCommit) {
198                     if (hardCommitLock != null) hardCommitLock.unlock();
199                   }
200 
201                 }
202                 numCommitting.decrementAndGet();
203                 continue;
204               }
205 
206 
207               int id = rand.nextInt(ndocs);
208               Object sync = syncArr[id];
209 
210               // set the lastId before we actually change it sometimes to try and
211               // uncover more race conditions between writing and reading
212               boolean before = rand.nextBoolean();
213               if (before) {
214                 lastId = id;
215               }
216 
217               // We can't concurrently update the same document and retain our invariants of increasing values
218               // since we can't guarantee what order the updates will be executed.
219               synchronized (sync) {
220                 DocInfo info = model.get(id);
221                 long val = info.val;
222                 long nextVal = Math.abs(val)+1;
223 
224                 if (oper < commitPercent + deletePercent) {
225                   // add tombstone first
226                   if (tombstones) {
227                     Document d = new Document();
228                     d.add(new Field("id","-"+Integer.toString(id), idFt));
229                     d.add(new Field(field, Long.toString(nextVal), ft2));
230                     verbose("adding tombstone for id",id,"val=",nextVal);
231                     writer.updateDocument(new Term("id", "-"+Integer.toString(id)), d);
232                   }
233 
234                   verbose("deleting id",id,"val=",nextVal);
235                   writer.deleteDocuments(new Term("id",Integer.toString(id)));
236                   model.put(id, new DocInfo(0,-nextVal));
237                   verbose("deleting id",id,"val=",nextVal,"DONE");
238 
239                 } else if (oper < commitPercent + deletePercent + deleteByQueryPercent) {
240                   //assertU("<delete><query>id:" + id + "</query></delete>");
241 
242                   // add tombstone first
243                   if (tombstones) {
244                     Document d = new Document();
245                     d.add(new Field("id","-"+Integer.toString(id), idFt));
246                     d.add(new Field(field, Long.toString(nextVal), ft2));
247                     verbose("adding tombstone for id",id,"val=",nextVal);
248                     writer.updateDocument(new Term("id", "-"+Integer.toString(id)), d);
249                   }
250 
251                   verbose("deleteByQuery",id,"val=",nextVal);
252                   writer.deleteDocuments(new TermQuery(new Term("id", Integer.toString(id))));
253                   model.put(id, new DocInfo(0,-nextVal));
254                   verbose("deleteByQuery",id,"val=",nextVal,"DONE");
255                 } else {
256                   // model.put(id, nextVal);   // uncomment this and this test should fail.
257 
258                   // assertU(adoc("id",Integer.toString(id), field, Long.toString(nextVal)));
259                   Document d = new Document();
260                   d.add(new Field("id",Integer.toString(id), idFt));
261                   d.add(new Field(field, Long.toString(nextVal), ft2));
262                   verbose("adding id",id,"val=",nextVal);
263                   writer.updateDocument(new Term("id", Integer.toString(id)), d);
264                   if (tombstones) {
265                     // remove tombstone after new addition (this should be optional?)
266                     verbose("deleting tombstone for id",id);
267                     writer.deleteDocuments(new Term("id","-"+Integer.toString(id)));
268                     verbose("deleting tombstone for id",id,"DONE");
269                   }
270 
271                   model.put(id, new DocInfo(0,nextVal));
272                   verbose("adding id",id,"val=",nextVal,"DONE");
273                 }
274               }
275 
276               if (!before) {
277                 lastId = id;
278               }
279             }
280           } catch (Exception  ex) {
281             throw new RuntimeException(ex);
282           }
283         }
284       };
285 
286       threads.add(thread);
287     }
288 
289 
290     for (int i=0; i<nReadThreads; i++) {
291       Thread thread = new Thread("READER"+i) {
292         Random rand = new Random(random().nextInt());
293 
294         @Override
295         public void run() {
296           try {
297             while (operations.decrementAndGet() >= 0) {
298               // bias toward a recently changed doc
299               int id = rand.nextInt(100) < 25 ? lastId : rand.nextInt(ndocs);
300 
301               // when indexing, we update the index, then the model
302               // so when querying, we should first check the model, and then the index
303 
304               DocInfo info;
305               synchronized(globalLock) {
306                 info = committedModel.get(id);
307               }
308               long val = info.val;
309 
310               IndexReader r;
311               synchronized(globalLock) {
312                 r = reader;
313                 r.incRef();
314               }
315 
316               int docid = getFirstMatch(r, new Term("id",Integer.toString(id)));
317 
318               if (docid < 0 && tombstones) {
319                 // if we couldn't find the doc, look for its tombstone
320                 docid = getFirstMatch(r, new Term("id","-"+Integer.toString(id)));
321                 if (docid < 0) {
322                   if (val == -1L) {
323                     // expected... no doc was added yet
324                     r.decRef();
325                     continue;
326                   }
327                   verbose("ERROR: Couldn't find a doc  or tombstone for id", id, "using reader",r,"expected value",val);
328                   fail("No documents or tombstones found for id " + id + ", expected at least " + val);
329                 }
330               }
331 
332               if (docid < 0 && !tombstones) {
333                 // nothing to do - we can't tell anything from a deleted doc without tombstones
334               } else {
335                 if (docid < 0) {
336                   verbose("ERROR: Couldn't find a doc for id", id, "using reader",r);
337                 }
338                 assertTrue(docid >= 0);   // we should have found the document, or its tombstone
339                 Document doc = r.document(docid);
340                 long foundVal = Long.parseLong(doc.get(field));
341                 if (foundVal < Math.abs(val)) {
342                   verbose("ERROR: id",id,"model_val=",val," foundVal=",foundVal,"reader=",reader);
343                 }
344                 assertTrue(foundVal >= Math.abs(val));
345               }
346 
347               r.decRef();
348             }
349           } catch (Throwable e) {
350             operations.set(-1L);
351             throw new RuntimeException(e);
352           }
353         }
354       };
355 
356       threads.add(thread);
357     }
358 
359 
360     for (Thread thread : threads) {
361       thread.start();
362     }
363 
364     for (Thread thread : threads) {
365       thread.join();
366     }
367 
368     writer.close();
369     reader.close();
370     dir.close();
371   }
