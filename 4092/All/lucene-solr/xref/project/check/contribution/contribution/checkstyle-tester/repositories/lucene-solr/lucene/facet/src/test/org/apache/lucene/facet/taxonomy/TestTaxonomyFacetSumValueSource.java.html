68    public void testBasic() throws Exception {
69  
70      Directory dir = newDirectory();
71      Directory taxoDir = newDirectory();
72  
73      // Writes facet ords to a separate directory from the
74      // main index:
75      DirectoryTaxonomyWriter taxoWriter = new DirectoryTaxonomyWriter(taxoDir, IndexWriterConfig.OpenMode.CREATE);
76  
77      RandomIndexWriter writer = new RandomIndexWriter(random(), dir);
78      FacetsConfig config = new FacetsConfig();
79  
80      // Reused across documents, to add the necessary facet
81      // fields:
82      Document doc = new Document();
83      doc.add(new NumericDocValuesField("num", 10));
84      doc.add(new FacetField("Author", "Bob"));
85      writer.addDocument(config.build(taxoWriter, doc));
86  
87      doc = new Document();
88      doc.add(new NumericDocValuesField("num", 20));
89      doc.add(new FacetField("Author", "Lisa"));
90      writer.addDocument(config.build(taxoWriter, doc));
91  
92      doc = new Document();
93      doc.add(new NumericDocValuesField("num", 30));
94      doc.add(new FacetField("Author", "Lisa"));
95      writer.addDocument(config.build(taxoWriter, doc));
96  
97      doc = new Document();
98      doc.add(new NumericDocValuesField("num", 40));
99      doc.add(new FacetField("Author", "Susan"));
100     writer.addDocument(config.build(taxoWriter, doc));
101 
102     doc = new Document();
103     doc.add(new NumericDocValuesField("num", 45));
104     doc.add(new FacetField("Author", "Frank"));
105     writer.addDocument(config.build(taxoWriter, doc));
106 
107     // NRT open
108     IndexSearcher searcher = newSearcher(writer.getReader());
109     writer.close();
110 
111     // NRT open
112     TaxonomyReader taxoReader = new DirectoryTaxonomyReader(taxoWriter);
113     taxoWriter.close();
114 
115     // Aggregate the facet counts:
116     FacetsCollector c = new FacetsCollector();
117 
118     // MatchAllDocsQuery is for "browsing" (counts facets
119     // for all non-deleted docs in the index); normally
120     // you'd use a "normal" query and one of the
121     // Facets.search utility methods:
122     searcher.search(new MatchAllDocsQuery(), c);
123 
124     TaxonomyFacetSumValueSource facets = new TaxonomyFacetSumValueSource(taxoReader, new FacetsConfig(), c, new IntFieldSource("num"));
125 
126     // Retrieve & verify results:
127     assertEquals("dim=Author path=[] value=145.0 childCount=4\n  Lisa (50.0)\n  Frank (45.0)\n  Susan (40.0)\n  Bob (10.0)\n", facets.getTopChildren(10, "Author").toString());
128 
129     taxoReader.close();
130     searcher.getIndexReader().close();
131     dir.close();
132     taxoDir.close();
133   }

250   public void testSumScoreAggregator() throws Exception {
251     Directory indexDir = newDirectory();
252     Directory taxoDir = newDirectory();
253 
254     DirectoryTaxonomyWriter taxoWriter = new DirectoryTaxonomyWriter(taxoDir);
255     IndexWriter iw = new IndexWriter(indexDir, newIndexWriterConfig(new MockAnalyzer(random())));
256 
257     FacetsConfig config = new FacetsConfig();
258 
259     for(int i = atLeast(30); i > 0; --i) {
260       Document doc = new Document();
261       if (random().nextBoolean()) { // don't match all documents
262         doc.add(new StringField("f", "v", Field.Store.NO));
263       }
264       doc.add(new FacetField("dim", "a"));
265       iw.addDocument(config.build(taxoWriter, doc));
266     }
267     
268     DirectoryReader r = DirectoryReader.open(iw, true);
269     DirectoryTaxonomyReader taxoReader = new DirectoryTaxonomyReader(taxoWriter);
270     
271     FacetsCollector fc = new FacetsCollector(true);
272     BoostQuery csq = new BoostQuery(new ConstantScoreQuery(new MatchAllDocsQuery()), 2f);
273     
274     TopDocs td = FacetsCollector.search(newSearcher(r), csq, 10, fc);
275 
276     Facets facets = new TaxonomyFacetSumValueSource(taxoReader, config, fc, new TaxonomyFacetSumValueSource.ScoreValueSource());
277     
278     int expected = (int) (td.getMaxScore() * td.totalHits);
279     assertEquals(expected, facets.getSpecificValue("dim", "a").intValue());
280 
281     iw.close();
282     IOUtils.close(taxoWriter, taxoReader, taxoDir, r, indexDir);
283   }

311   public void testWithScore() throws Exception {
312     Directory indexDir = newDirectory();
313     Directory taxoDir = newDirectory();
314 
315     DirectoryTaxonomyWriter taxoWriter = new DirectoryTaxonomyWriter(taxoDir);
316     IndexWriter iw = new IndexWriter(indexDir, newIndexWriterConfig(new MockAnalyzer(random())));
317 
318     FacetsConfig config = new FacetsConfig();
319     for (int i = 0; i < 4; i++) {
320       Document doc = new Document();
321       doc.add(new NumericDocValuesField("price", (i+1)));
322       doc.add(new FacetField("a", Integer.toString(i % 2)));
323       iw.addDocument(config.build(taxoWriter, doc));
324     }
325     
326     DirectoryReader r = DirectoryReader.open(iw, true);
327     DirectoryTaxonomyReader taxoReader = new DirectoryTaxonomyReader(taxoWriter);
328 
329     ValueSource valueSource = new ValueSource() {
330       @Override
331       public FunctionValues getValues(@SuppressWarnings("rawtypes") Map context, LeafReaderContext readerContext) throws IOException {
332         final Scorer scorer = (Scorer) context.get("scorer");
333         assert scorer != null;
334         return new DoubleDocValues(this) {
335           @Override
336           public double doubleVal(int document) {
337             try {
338               return scorer.score();
339             } catch (IOException exception) {
340               throw new RuntimeException(exception);
341             }
342           }
343         };
344       }
345 
346       @Override public boolean equals(Object o) { return o == this; }
347       @Override public int hashCode() { return System.identityHashCode(this); }
348       @Override public String description() { return "score()"; }
349     };
350     
351     FacetsCollector fc = new FacetsCollector(true);
352     // score documents by their 'price' field - makes asserting the correct counts for the categories easier
353     Query q = new FunctionQuery(new LongFieldSource("price"));
354     FacetsCollector.search(newSearcher(r), q, 10, fc);
355     Facets facets = new TaxonomyFacetSumValueSource(taxoReader, config, fc, valueSource);
356     
357     assertEquals("dim=a path=[] value=10.0 childCount=2\n  1 (6.0)\n  0 (4.0)\n", facets.getTopChildren(10, "a").toString());
358 
359     iw.close();
360     IOUtils.close(taxoWriter, taxoReader, taxoDir, r, indexDir);
361   }

426   public void testRandom() throws Exception {
427     String[] tokens = getRandomTokens(10);
428     Directory indexDir = newDirectory();
429     Directory taxoDir = newDirectory();
430 
431     RandomIndexWriter w = new RandomIndexWriter(random(), indexDir);
432     DirectoryTaxonomyWriter tw = new DirectoryTaxonomyWriter(taxoDir);
433     FacetsConfig config = new FacetsConfig();
434     int numDocs = atLeast(1000);
435     int numDims = TestUtil.nextInt(random(), 1, 7);
436     List<TestDoc> testDocs = getRandomDocs(tokens, numDocs, numDims);
437     for(TestDoc testDoc : testDocs) {
438       Document doc = new Document();
439       doc.add(newStringField("content", testDoc.content, Field.Store.NO));
440       testDoc.value = random().nextFloat();
441       doc.add(new FloatDocValuesField("value", testDoc.value));
442       for(int j=0;j<numDims;j++) {
443         if (testDoc.dims[j] != null) {
444           doc.add(new FacetField("dim" + j, testDoc.dims[j]));
445         }
446       }
447       w.addDocument(config.build(tw, doc));
448     }
449 
450     // NRT open
451     IndexSearcher searcher = newSearcher(w.getReader());
452     
453     // NRT open
454     TaxonomyReader tr = new DirectoryTaxonomyReader(tw);
455 
456     ValueSource values = new FloatFieldSource("value");
457 
458     int iters = atLeast(100);
459     for(int iter=0;iter<iters;iter++) {
460       String searchToken = tokens[random().nextInt(tokens.length)];
461       if (VERBOSE) {
462         System.out.println("\nTEST: iter content=" + searchToken);
463       }
464       FacetsCollector fc = new FacetsCollector();
465       FacetsCollector.search(searcher, new TermQuery(new Term("content", searchToken)), 10, fc);
466       Facets facets = new TaxonomyFacetSumValueSource(tr, config, fc, values);
467 
468       // Slow, yet hopefully bug-free, faceting:
469       @SuppressWarnings({"rawtypes","unchecked"}) Map<String,Float>[] expectedValues = new HashMap[numDims];
470       for(int i=0;i<numDims;i++) {
471         expectedValues[i] = new HashMap<>();
472       }
473 
474       for(TestDoc doc : testDocs) {
475         if (doc.content.equals(searchToken)) {
476           for(int j=0;j<numDims;j++) {
477             if (doc.dims[j] != null) {
478               Float v = expectedValues[j].get(doc.dims[j]);
479               if (v == null) {
480                 expectedValues[j].put(doc.dims[j], doc.value);
481               } else {
482                 expectedValues[j].put(doc.dims[j], v.floatValue() + doc.value);
483               }
484             }
485           }
486         }
487       }
488 
489       List<FacetResult> expected = new ArrayList<>();
490       for(int i=0;i<numDims;i++) {
491         List<LabelAndValue> labelValues = new ArrayList<>();
492         double totValue = 0;
493         for(Map.Entry<String,Float> ent : expectedValues[i].entrySet()) {
494           labelValues.add(new LabelAndValue(ent.getKey(), ent.getValue()));
495           totValue += ent.getValue();
496         }
497         sortLabelValues(labelValues);
498         if (totValue > 0) {
499           expected.add(new FacetResult("dim" + i, new String[0], totValue, labelValues.toArray(new LabelAndValue[labelValues.size()]), labelValues.size()));
500         }
501       }
502 
503       // Sort by highest value, tie break by value:
504       sortFacetResults(expected);
505 
506       List<FacetResult> actual = facets.getAllDims(10);
507 
508       // Messy: fixup ties
509       sortTies(actual);
510 
511       if (VERBOSE) {
512         System.out.println("expected=\n" + expected.toString());
513         System.out.println("actual=\n" + actual.toString());
514       }
515 
516       assertFloatValuesEquals(expected, actual);
517     }
518 
519     w.close();
520     IOUtils.close(tw, searcher.getIndexReader(), tr, indexDir, taxoDir);
521   }
