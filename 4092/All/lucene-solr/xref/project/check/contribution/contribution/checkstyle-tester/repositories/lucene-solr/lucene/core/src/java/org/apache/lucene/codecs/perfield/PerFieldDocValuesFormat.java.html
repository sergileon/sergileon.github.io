49  /**
50   * Enables per field docvalues support.
51   * <p>
52   * Note, when extending this class, the name ({@link #getName}) is 
53   * written into the index. In order for the field to be read, the
54   * name must resolve to your implementation via {@link #forName(String)}.
55   * This method uses Java's 
56   * {@link ServiceLoader Service Provider Interface} to resolve format names.
57   * <p>
58   * Files written by each docvalues format have an additional suffix containing the 
59   * format name. For example, in a per-field configuration instead of <tt>_1.dat</tt> 
60   * filenames would look like <tt>_1_Lucene40_0.dat</tt>.
61   * @see ServiceLoader
62   * @lucene.experimental
63   */
64  
65  public abstract class PerFieldDocValuesFormat extends DocValuesFormat {
66    /** Name of this {@link PostingsFormat}. */
67    public static final String PER_FIELD_NAME = "PerFieldDV40";
68  
69    /** {@link FieldInfo} attribute name used to store the
70     *  format name for each field. */
71    public static final String PER_FIELD_FORMAT_KEY = PerFieldDocValuesFormat.class.getSimpleName() + ".format";
72  
73    /** {@link FieldInfo} attribute name used to store the
74     *  segment suffix name for each field. */
75    public static final String PER_FIELD_SUFFIX_KEY = PerFieldDocValuesFormat.class.getSimpleName() + ".suffix";
76  
77    
78    /** Sole constructor. */
79    public PerFieldDocValuesFormat() {
80      super(PER_FIELD_NAME);
81    }
82  
83    @Override
84    public final DocValuesConsumer fieldsConsumer(SegmentWriteState state) throws IOException {
85      return new FieldsWriter(state);
86    }
87  
88    static class ConsumerAndSuffix implements Closeable {
89      DocValuesConsumer consumer;
90      int suffix;
91      
92      @Override
93      public void close() throws IOException {
94        consumer.close();
95      }
96    }

98    private class FieldsWriter extends DocValuesConsumer {
99  
100     private final Map<DocValuesFormat,ConsumerAndSuffix> formats = new HashMap<>();
101     private final Map<String,Integer> suffixes = new HashMap<>();

134     private DocValuesConsumer getInstance(FieldInfo field) throws IOException {
135       DocValuesFormat format = null;
136       if (field.getDocValuesGen() != -1) {
137         final String formatName = field.getAttribute(PER_FIELD_FORMAT_KEY);
138         // this means the field never existed in that segment, yet is applied updates
139         if (formatName != null) {
140           format = DocValuesFormat.forName(formatName);
141         }
142       }
143       if (format == null) {
144         format = getDocValuesFormatForField(field.name);
145       }
146       if (format == null) {
147         throw new IllegalStateException("invalid null DocValuesFormat for field=\"" + field.name + "\"");
148       }
149       final String formatName = format.getName();
150       
151       String previousValue = field.putAttribute(PER_FIELD_FORMAT_KEY, formatName);
152       if (field.getDocValuesGen() == -1 && previousValue != null) {
153         throw new IllegalStateException("found existing value for " + PER_FIELD_FORMAT_KEY + 
154                                         ", field=" + field.name + ", old=" + previousValue + ", new=" + formatName);
155       }
156       
157       Integer suffix = null;
158       
159       ConsumerAndSuffix consumer = formats.get(format);
160       if (consumer == null) {
161         // First time we are seeing this format; create a new instance
162 
163         if (field.getDocValuesGen() != -1) {
164           final String suffixAtt = field.getAttribute(PER_FIELD_SUFFIX_KEY);
165           // even when dvGen is != -1, it can still be a new field, that never
166           // existed in the segment, and therefore doesn't have the recorded
167           // attributes yet.
168           if (suffixAtt != null) {
169             suffix = Integer.valueOf(suffixAtt);
170           }
171         }
172         
173         if (suffix == null) {
174           // bump the suffix
175           suffix = suffixes.get(formatName);
176           if (suffix == null) {
177             suffix = 0;
178           } else {
179             suffix = suffix + 1;
180           }
181         }
182         suffixes.put(formatName, suffix);
183         
184         final String segmentSuffix = getFullSegmentSuffix(segmentWriteState.segmentSuffix,
185                                                           getSuffix(formatName, Integer.toString(suffix)));
186         consumer = new ConsumerAndSuffix();
187         consumer.consumer = format.fieldsConsumer(new SegmentWriteState(segmentWriteState, segmentSuffix));
188         consumer.suffix = suffix;
189         formats.put(format, consumer);
190       } else {
191         // we've already seen this format, so just grab its suffix
192         assert suffixes.containsKey(formatName);
193         suffix = consumer.suffix;
194       }
195       
196       previousValue = field.putAttribute(PER_FIELD_SUFFIX_KEY, Integer.toString(suffix));
197       if (field.getDocValuesGen() == -1 && previousValue != null) {
198         throw new IllegalStateException("found existing value for " + PER_FIELD_SUFFIX_KEY + 
199                                         ", field=" + field.name + ", old=" + previousValue + ", new=" + suffix);
200       }
201 
202       // TODO: we should only provide the "slice" of FIS
203       // that this DVF actually sees ...
204       return consumer.consumer;
205     }
206 
207     @Override
208     public void close() throws IOException {
209       // Close all subs
210       IOUtils.close(formats.values());
211     }
212   }
213   
214   static String getSuffix(String formatName, String suffix) {
215     return formatName + "_" + suffix;
216   }
217 
218   static String getFullSegmentSuffix(String outerSegmentSuffix, String segmentSuffix) {
219     if (outerSegmentSuffix.length() == 0) {
220       return segmentSuffix;
221     } else {
222       return outerSegmentSuffix + "_" + segmentSuffix;
223     }
224   }

226   private class FieldsReader extends DocValuesProducer {
227 
228     private final Map<String,DocValuesProducer> fields = new TreeMap<>();
229     private final Map<String,DocValuesProducer> formats = new HashMap<>();
230     
231     // clone for merge
232     FieldsReader(FieldsReader other) throws IOException {
233       Map<DocValuesProducer,DocValuesProducer> oldToNew = new IdentityHashMap<>();
234       // First clone all formats
235       for(Map.Entry<String,DocValuesProducer> ent : other.formats.entrySet()) {
236         DocValuesProducer values = ent.getValue().getMergeInstance();
237         formats.put(ent.getKey(), values);
238         oldToNew.put(ent.getValue(), values);
239       }
240 
241       // Then rebuild fields:
242       for(Map.Entry<String,DocValuesProducer> ent : other.fields.entrySet()) {
243         DocValuesProducer producer = oldToNew.get(ent.getValue());
244         assert producer != null;
245         fields.put(ent.getKey(), producer);
246       }
247     }
248 
249     public FieldsReader(final SegmentReadState readState) throws IOException {
250 
251       // Init each unique format:
252       boolean success = false;
253       try {
254         // Read field name -> format name
255         for (FieldInfo fi : readState.fieldInfos) {
256           if (fi.getDocValuesType() != DocValuesType.NONE) {
257             final String fieldName = fi.name;
258             final String formatName = fi.getAttribute(PER_FIELD_FORMAT_KEY);
259             if (formatName != null) {
260               // null formatName means the field is in fieldInfos, but has no docvalues!
261               final String suffix = fi.getAttribute(PER_FIELD_SUFFIX_KEY);
262               if (suffix == null) {
263                 throw new IllegalStateException("missing attribute: " + PER_FIELD_SUFFIX_KEY + " for field: " + fieldName);
264               }
265               DocValuesFormat format = DocValuesFormat.forName(formatName);
266               String segmentSuffix = getFullSegmentSuffix(readState.segmentSuffix, getSuffix(formatName, suffix));
267               if (!formats.containsKey(segmentSuffix)) {
268                 formats.put(segmentSuffix, format.fieldsProducer(new SegmentReadState(readState, segmentSuffix)));
269               }
270               fields.put(fieldName, formats.get(segmentSuffix));
271             }
272           }
273         }
274         success = true;
275       } finally {
276         if (!success) {
277           IOUtils.closeWhileHandlingException(formats.values());
278         }
279       }
280     }

323     @Override
324     public long ramBytesUsed() {
325       long size = 0;
326       for (Map.Entry<String,DocValuesProducer> entry : formats.entrySet()) {
327         size += (entry.getKey().length() * RamUsageEstimator.NUM_BYTES_CHAR) + 
328             entry.getValue().ramBytesUsed();
329       }
330       return size;
331     }

333     @Override
334     public Collection<Accountable> getChildResources() {
335       return Accountables.namedAccountables("format", formats);
336     }

338     @Override
339     public void checkIntegrity() throws IOException {
340       for (DocValuesProducer format : formats.values()) {
341         format.checkIntegrity();
342       }
343     }

345     @Override
346     public DocValuesProducer getMergeInstance() throws IOException {
347       return new FieldsReader(this);
348     }
349 
350     @Override
351     public String toString() {
352       return "PerFieldDocValues(formats=" + formats.size() + ")";
353     }
354   }
