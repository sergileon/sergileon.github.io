161   @Test
162   public void testModifyRequestTrack() {
163     DebugComponent component = new DebugComponent();
164     List<SearchComponent> components = new ArrayList<>(1);
165     components.add(component);
166     for(int i = 0; i < 10; i++) {
167       SolrQueryRequest req = req("q", "test query", "distrib", "true", CommonParams.REQUEST_ID, "123456-my_rid");
168       SolrQueryResponse resp = new SolrQueryResponse();
169       ResponseBuilder rb = new ResponseBuilder(req, resp, components);
170       ShardRequest sreq = new ShardRequest();
171       sreq.params = new ModifiableSolrParams();
172       sreq.purpose = ShardRequest.PURPOSE_GET_FIELDS;
173       sreq.purpose |= ShardRequest.PURPOSE_GET_DEBUG;
174       //expecting the same results with debugQuery=true or debug=track
175       if(random().nextBoolean()) {
176         rb.setDebug(true);
177       } else {
178         rb.setDebug(false);
179         rb.setDebugTrack(true);
180         //should not depend on other debug options
181         rb.setDebugQuery(random().nextBoolean());
182         rb.setDebugTimings(random().nextBoolean());
183         rb.setDebugResults(random().nextBoolean());
184       }
185       component.modifyRequest(rb, null, sreq);
186       //if the request has debugQuery=true or debug=track, the sreq should get debug=track always
187       assertTrue(Arrays.asList(sreq.params.getParams(CommonParams.DEBUG)).contains(CommonParams.TRACK));
188       //the purpose must be added as readable param to be included in the shard logs
189       assertEquals("GET_FIELDS,GET_DEBUG", sreq.params.get(CommonParams.REQUEST_PURPOSE));
190       //the rid must be added to be included in the shard logs
191       assertEquals("123456-my_rid", sreq.params.get(CommonParams.REQUEST_ID));
192     }

196   @Test
197   public void testPrepare() throws IOException {
198     DebugComponent component = new DebugComponent();
199     List<SearchComponent> components = new ArrayList<>(1);
200     components.add(component);
201     SolrQueryRequest req;
202     ResponseBuilder rb;
203     for(int i = 0; i < 10; i++) {
204       req = req("q", "test query", "distrib", "true");
205       rb = new ResponseBuilder(req, new SolrQueryResponse(), components);
206       rb.isDistrib = true;
207       //expecting the same results with debugQuery=true or debug=track
208       if(random().nextBoolean()) {
209         rb.setDebug(true);
210       } else {
211         rb.setDebug(false);
212         rb.setDebugTrack(true);
213         //should not depend on other debug options
214         rb.setDebugQuery(random().nextBoolean());
215         rb.setDebugTimings(random().nextBoolean());
216         rb.setDebugResults(random().nextBoolean());
217       }
218       component.prepare(rb);
219       ensureRidPresent(rb, null);
220     }
221    
222     req = req("q", "test query", "distrib", "true", CommonParams.REQUEST_ID, "123");
223     rb = new ResponseBuilder(req, new SolrQueryResponse(), components);
224     rb.isDistrib = true;
225     rb.setDebug(true);
226     component.prepare(rb);
227     ensureRidPresent(rb, "123");
228   }

230   @SuppressWarnings("unchecked")
231   private void ensureRidPresent(ResponseBuilder rb, String expectedRid) {
232     SolrQueryRequest req = rb.req;
233     SolrQueryResponse resp = rb.rsp;
234     //a generated request ID should be added to the request
235     String rid = req.getParams().get(CommonParams.REQUEST_ID);
236     if(expectedRid == null) {
237       assertTrue(rid + " Doesn't match expected pattern.", Pattern.matches(".*-collection1-[0-9]*-[0-9]+", rid));
238     } else {
239       assertEquals("Expecting " + expectedRid + " but found " + rid, expectedRid, rid);
240     }
241     //The request ID is added to the debug/track section
242     assertEquals(rid, ((NamedList<Object>)rb.getDebugInfo().get("track")).get(CommonParams.REQUEST_ID));
243     //RID must be added to the toLog, so that it's included in the main request log
244     assertEquals(rid, resp.getToLog().get(CommonParams.REQUEST_ID));
245   }
