89    @Test
90    public void testArgsParserTypicalUse() {
91      String[] args = new String[] { 
92          "--input-list", "file:///tmp",
93          "--morphline-file", MORPHLINE_FILE,
94          "--morphline-id", "morphline_xyz",
95          "--output-dir", "file:/tmp/foo",
96          "--solr-home-dir", MINIMR_INSTANCE_DIR.getPath(),
97          "--mappers", "10", 
98          "--reducers", "9", 
99          "--fanout", "8", 
100         "--max-segments", "7", 
101         "--shards", "1",
102         "--verbose", 
103         "file:///home",
104         "file:///dev",
105         };
106     Integer res = parser.parseArgs(args, conf, opts);
107     assertNull(res != null ? res.toString() : "", res);
108     assertEquals(Collections.singletonList(new Path("file:///tmp")), opts.inputLists);
109     assertEquals(new Path("file:/tmp/foo"), opts.outputDir);
110     assertEquals(new File(MINIMR_INSTANCE_DIR.getPath()), opts.solrHomeDir);
111     assertEquals(10, opts.mappers);
112     assertEquals(9, opts.reducers);
113     assertEquals(8, opts.fanout);
114     assertEquals(7, opts.maxSegments);
115     assertEquals(new Integer(1), opts.shards);
116     assertEquals(null, opts.fairSchedulerPool);
117     assertTrue(opts.isVerbose);
118     assertEquals(Arrays.asList(new Path("file:///home"), new Path("file:///dev")), opts.inputFiles);
119     assertEquals(RetainMostRecentUpdateConflictResolver.class.getName(), opts.updateConflictResolver);
120     assertEquals(MORPHLINE_FILE, opts.morphlineFile.getPath());
121     assertEquals("morphline_xyz", opts.morphlineId);
122     assertEmptySystemErrAndEmptySystemOut();
123   }

459   private void assertArgumentParserException(String[] args) throws UnsupportedEncodingException {
460     assertEquals("should have returned fail code", new Integer(1), parser.parseArgs(args, conf, opts));
461     assertEquals("no sys out expected:" + new String(bout.toByteArray(), StandardCharsets.UTF_8), 0, bout.toByteArray().length);
462     String usageText;
463     usageText = new String(berr.toByteArray(), StandardCharsets.UTF_8);
464 
465     assertTrue("should start with usage msg \"usage: hadoop \":" + usageText, usageText.startsWith("usage: hadoop "));
466   }
