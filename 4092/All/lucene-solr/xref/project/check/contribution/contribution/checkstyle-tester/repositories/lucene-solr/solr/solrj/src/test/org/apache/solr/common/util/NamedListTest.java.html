20  import java.util.ArrayList;

23  import org.apache.lucene.util.LuceneTestCase;

26  public class NamedListTest extends LuceneTestCase {
27    public void testRemove() {
28      NamedList<String> nl = new NamedList<>();
29      nl.add("key1", "value1");
30      nl.add("key2", "value2");
31      assertEquals(2, nl.size());
32      String value = null;
33      value = nl.remove(0);
34      assertEquals("value1", value);
35      assertEquals(1, nl.size());
36      value = nl.remove("key2");
37      assertEquals("value2", value);
38      assertEquals(0, nl.size());
39    }

41    public void testRemoveAll() {
42      NamedList<String> nl = new NamedList<>();
43      nl.add("key1", "value1-1");
44      nl.add("key2", "value2-1");
45      nl.add("key1", "value1-2");
46      nl.add("key2", "value2-2");
47      nl.add("key1", "value1-3");
48      nl.add("key2", "value2-3");
49      nl.add("key1", "value1-4");
50      nl.add("key2", "value2-4");
51      nl.add("key1", "value1-5");
52      nl.add("key2", "value2-5");
53      nl.add("key1", "value1-6");
54      assertEquals(11, nl.size());
55      List<String> values = null;
56      values = nl.removeAll("key1");
57      assertEquals("value1-1", values.get(0));
58      assertEquals("value1-3", values.get(2));
59      assertEquals(6, values.size());
60      assertEquals(5, nl.size());
61      values = nl.removeAll("key2");
62      assertEquals(5, values.size());
63      assertEquals(0, nl.size());
64    }

66    public void testRemoveArgs() {
67      NamedList<Object> nl = new NamedList<>();
68      nl.add("key1", "value1-1");
69      nl.add("key2", "value2-1");
70      nl.add("key1", "value1-2");
71      nl.add("key2", "value2-2");
72      nl.add("key1", "value1-3");
73      nl.add("key2", "value2-3");
74      nl.add("key1", "value1-4");
75      nl.add("key2", "value2-4");
76      nl.add("key1", "value1-5");
77      nl.add("key2", "value2-5");
78      nl.add("key1", "value1-6");
79      nl.add("key2", 0);
80      nl.add("key2", "value2-7");
81      assertEquals(13, nl.size());
82      List<String> values = (ArrayList<String>) nl.removeConfigArgs("key1");
83      assertEquals("value1-1", values.get(0));
84      assertEquals("value1-3", values.get(2));
85      assertEquals(6, values.size());
86      assertEquals(7, nl.size());
87      try {
88        values = (ArrayList<String>) nl.removeConfigArgs("key2");
89        fail();
90      }
91      catch(SolrException e) {
92        // Expected exception.
93        assertTrue(true);
94      }
95      // nl should be unmodified when removeArgs throws an exception.
96      assertEquals(7, nl.size());
97    }

99    public void testRecursive() {
100     // key1
101     // key2
102     // - key2a
103     // - key2b
104     // --- key2b1
105     // --- key2b2
106     // - key2c
107     // - k2int1
108     // key3
109     // - key3a
110     // --- key3a1
111     // --- key3a2
112     // --- key3a3
113     // - key3b
114     // - key3c
115     
116     // this is a varied NL structure.
117     NamedList<String> nl2b = new NamedList<>();
118     nl2b.add("key2b1", "value2b1");
119     nl2b.add("key2b2", "value2b2");
120     NamedList<String> nl3a = new NamedList<>();
121     nl3a.add("key3a1", "value3a1");
122     nl3a.add("key3a2", "value3a2");
123     nl3a.add("key3a3", "value3a3");
124     NamedList<Object> nl2 = new NamedList<>();
125     nl2.add("key2a", "value2a");
126     nl2.add("key2b", nl2b);
127     nl2.add("k2int1", (int) 5);
128     NamedList<Object> nl3 = new NamedList<>();
129     nl3.add("key3a", nl3a);
130     nl3.add("key3b", "value3b");
131     nl3.add("key3c", "value3c");
132     nl3.add("key3c", "value3c2");
133     NamedList<Object> nl = new NamedList<>();
134     nl.add("key1", "value1");
135     nl.add("key2", nl2);
136     nl.add("key3", nl3);
137     
138     // Simple three-level checks.
139     String test1 = (String) nl.findRecursive("key2", "key2b", "key2b2");
140     assertEquals("value2b2", test1);
141     String test2 = (String) nl.findRecursive("key3", "key3a", "key3a3");
142     assertEquals("value3a3", test2);
143     // Two-level check.
144     String test3 = (String) nl.findRecursive("key3", "key3c");
145     assertEquals("value3c", test3);
146     // Checking that invalid values return null.
147     String test4 = (String) nl.findRecursive("key3", "key3c", "invalid");
148     assertEquals(null, test4);
149     String test5 = (String) nl.findRecursive("key3", "invalid", "invalid");
150     assertEquals(null, test5);
151     String test6 = (String) nl.findRecursive("invalid", "key3c");
152     assertEquals(null, test6);
153     // Verify that retrieved NamedList objects have the right type.
154     Object test7 = nl.findRecursive("key2", "key2b");
155     assertTrue(test7 instanceof NamedList);
156     // Integer check.
157     int test8 = (Integer) nl.findRecursive("key2", "k2int1");
158     assertEquals(5, test8);
159     // Check that a single argument works the same as get(String).
160     String test9 = (String) nl.findRecursive("key1");
161     assertEquals("value1", test9);
162     // enl == explicit nested list
163     //
164     // key1
165     // - key1a
166     // - key1b
167     // key2 (null list)
168     NamedList<NamedList<String>> enl = new NamedList<>();
169     NamedList<String> enlkey1 = new NamedList<>();
170     NamedList<String> enlkey2 = null;
171     enlkey1.add("key1a", "value1a");
172     enlkey1.add("key1b", "value1b");
173     enl.add("key1", enlkey1);
174     enl.add("key2", enlkey2);
175     
176     // Tests that are very similar to the test above, just repeated
177     // on the explicitly nested object type.
178     String enltest1 = (String) enl.findRecursive("key1", "key1a");
179     assertEquals("value1a", enltest1);
180     String enltest2 = (String) enl.findRecursive("key1", "key1b");
181     assertEquals("value1b", enltest2);
182     // Verify that when a null value is stored, the standard get method
183     // says it is null, then check the recursive method.
184     Object enltest3 = enl.get("key2");
185     assertNull(enltest3);
186     Object enltest4 = enl.findRecursive("key2");
187     assertNull(enltest4);
188   }
