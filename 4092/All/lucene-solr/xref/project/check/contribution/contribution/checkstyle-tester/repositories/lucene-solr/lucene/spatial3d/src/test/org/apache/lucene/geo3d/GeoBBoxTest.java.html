33    @Test
34    public void testBBoxDegenerate() {
35      GeoBBox box;
36      GeoConvexPolygon cp;
37      int relationship;
38      List<GeoPoint> points = new ArrayList<GeoPoint>();
39      points.add(new GeoPoint(PlanetModel.SPHERE, 24 * DEGREES_TO_RADIANS, -30 * DEGREES_TO_RADIANS));
40      points.add(new GeoPoint(PlanetModel.SPHERE, -11 * DEGREES_TO_RADIANS, 101 * DEGREES_TO_RADIANS));
41      points.add(new GeoPoint(PlanetModel.SPHERE, -49 * DEGREES_TO_RADIANS, -176 * DEGREES_TO_RADIANS));
42      GeoMembershipShape shape = GeoPolygonFactory.makeGeoPolygon(PlanetModel.SPHERE, points, 0);
43      box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, -64 * DEGREES_TO_RADIANS, -64 * DEGREES_TO_RADIANS, -180 * DEGREES_TO_RADIANS, 180 * DEGREES_TO_RADIANS);
44      relationship = box.getRelationship(shape);
45      assertEquals(GeoArea.CONTAINS, relationship);
46      box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, -61.85 * DEGREES_TO_RADIANS, -67.5 * DEGREES_TO_RADIANS, -180 * DEGREES_TO_RADIANS, -168.75 * DEGREES_TO_RADIANS);
47      //System.out.println("Shape = " + shape + " Rect = " + box);
48      relationship = box.getRelationship(shape);
49      assertEquals(GeoArea.CONTAINS, relationship);
50    }

52    @Test
53    public void testBBoxPointWithin() {
54      GeoBBox box;
55      GeoPoint gp;
56  
57      // Standard normal Rect box, not crossing dateline
58      box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, 0.0, -Math.PI * 0.25, -1.0, 1.0);
59      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, 0.0);
60      assertTrue(box.isWithin(gp));
61      gp = new GeoPoint(PlanetModel.SPHERE, 0.1, 0.0);
62      assertFalse(box.isWithin(gp));
63      gp = new GeoPoint(PlanetModel.SPHERE, -Math.PI * 0.5, 0.0);
64      assertFalse(box.isWithin(gp));
65      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, 1.1);
66      assertFalse(box.isWithin(gp));
67      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -1.1);
68      assertFalse(box.isWithin(gp));
69      assertEquals(0.1,box.computeOutsideDistance(DistanceStyle.ARC,gp),1e-2);
70      assertEquals(0.1,box.computeOutsideDistance(DistanceStyle.NORMAL,gp),1e-2);
71      assertEquals(0.1,box.computeOutsideDistance(DistanceStyle.NORMAL,gp),1e-2);
72  
73      // Standard normal Rect box, crossing dateline
74      box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, 0.0, -Math.PI * 0.25, Math.PI - 1.0, -Math.PI + 1.0);
75      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -Math.PI);
76      assertTrue(box.isWithin(gp));
77      gp = new GeoPoint(PlanetModel.SPHERE, 0.1, -Math.PI);
78      assertFalse(box.isWithin(gp));
79      gp = new GeoPoint(PlanetModel.SPHERE, -Math.PI * 0.5, -Math.PI);
80      assertFalse(box.isWithin(gp));
81      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -Math.PI + 1.1);
82      assertFalse(box.isWithin(gp));
83      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, (-Math.PI - 1.1) + Math.PI * 2.0);
84      assertFalse(box.isWithin(gp));
85  
86      // Latitude zone rectangle
87      box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, 0.0, -Math.PI * 0.25, -Math.PI, Math.PI);
88      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -Math.PI);
89      assertTrue(box.isWithin(gp));
90      gp = new GeoPoint(PlanetModel.SPHERE, 0.1, -Math.PI);
91      assertFalse(box.isWithin(gp));
92      gp = new GeoPoint(PlanetModel.SPHERE, -Math.PI * 0.5, -Math.PI);
93      assertFalse(box.isWithin(gp));
94      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -Math.PI + 1.1);
95      assertTrue(box.isWithin(gp));
96      gp = new GeoPoint(PlanetModel.SPHERE, -0.1, (-Math.PI - 1.1) + Math.PI * 2.0);
97      assertTrue(box.isWithin(gp));
98  
99      // World
100     box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, Math.PI * 0.5, -Math.PI * 0.5, -Math.PI, Math.PI);
101     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -Math.PI);
102     assertTrue(box.isWithin(gp));
103     gp = new GeoPoint(PlanetModel.SPHERE, 0.1, -Math.PI);
104     assertTrue(box.isWithin(gp));
105     gp = new GeoPoint(PlanetModel.SPHERE, -Math.PI * 0.5, -Math.PI);
106     assertTrue(box.isWithin(gp));
107     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -Math.PI + 1.1);
108     assertTrue(box.isWithin(gp));
109     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, (-Math.PI - 1.1) + Math.PI * 2.0);
110     assertTrue(box.isWithin(gp));
111 
112   }

114   @Test
115   public void testBBoxExpand() {
116     GeoBBox box;
117     GeoPoint gp;
118     // Standard normal Rect box, not crossing dateline
119     box = GeoBBoxFactory.makeGeoBBox(PlanetModel.SPHERE, 0.0, -Math.PI * 0.25, -1.0, 1.0);
120     box = box.expand(0.1);
121     gp = new GeoPoint(PlanetModel.SPHERE, 0.05, 0.0);
122     assertTrue(box.isWithin(gp));
123     gp = new GeoPoint(PlanetModel.SPHERE, 0.15, 0.0);
124     assertFalse(box.isWithin(gp));
125     gp = new GeoPoint(PlanetModel.SPHERE, -Math.PI * 0.25 - 0.05, 0.0);
126     assertTrue(box.isWithin(gp));
127     gp = new GeoPoint(PlanetModel.SPHERE, -Math.PI * 0.25 - 0.15, 0.0);
128     assertFalse(box.isWithin(gp));
129     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -1.05);
130     assertTrue(box.isWithin(gp));
131     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, -1.15);
132     assertFalse(box.isWithin(gp));
133     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, 1.05);
134     assertTrue(box.isWithin(gp));
135     gp = new GeoPoint(PlanetModel.SPHERE, -0.1, 1.15);
136     assertFalse(box.isWithin(gp));
137   }
