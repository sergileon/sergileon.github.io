159   class ExpiringCachedDocCollection {
160     final DocCollection cached;
161     long cachedAt;
162 
163     ExpiringCachedDocCollection(DocCollection cached) {
164       this.cached = cached;
165       this.cachedAt = System.nanoTime();
166     }
167 
168     boolean isExpired(long timeToLiveMs) {
169       return (System.nanoTime() - cachedAt)
170           > TimeUnit.NANOSECONDS.convert(timeToLiveMs, TimeUnit.MILLISECONDS);
171     }

174   /**
175    * Create a new client object that connects to Zookeeper and is always aware
176    * of the SolrCloud state. If there is a fully redundant Zookeeper quorum and
177    * SolrCloud has enough replicas for every shard in a collection, there is no
178    * single point of failure. Updates will be sent to shard leaders by default.
179    * 
180    * @param zkHost
181    *          The client endpoint of the zookeeper quorum containing the cloud
182    *          state. The full specification for this string is one or more comma
183    *          separated HOST:PORT values, followed by an optional chroot value
184    *          that starts with a forward slash. Using a chroot allows multiple
185    *          applications to coexist in one ensemble. For full details, see the
186    *          Zookeeper documentation. Some examples:
187    *          <p>
188    *          "host1:2181"
189    *          <p>
190    *          "host1:2181,host2:2181,host3:2181/mysolrchroot"
191    *          <p>
192    *          "zoo1.example.com:2181,zoo2.example.com:2181,zoo3.example.com:2181"
193    */
194   public CloudSolrClient(String zkHost) {
195       this.zkHost = zkHost;
196       this.clientIsInternal = true;
197       this.myClient = HttpClientUtil.createClient(null);
198       this.lbClient = new LBHttpSolrClient(myClient);
199       this.lbClient.setRequestWriter(new BinaryRequestWriter());
200       this.lbClient.setParser(new BinaryResponseParser());
201       this.updatesToLeaders = true;
202       shutdownLBHttpSolrServer = true;
203       lbClient.addQueryParams(STATE_VERSION);
204   }

263   /**
264    * Create a new client object using multiple string values in a Collection
265    * instead of a standard zkHost connection string. Note that this method will
266    * not be used if there is only one String argument - that will use
267    * {@link #CloudSolrClient(String)} instead.
268    *
269    * @param zkHosts
270    *          A Java Collection (List, Set, etc) of HOST:PORT strings, one for
271    *          each host in the zookeeper ensemble. Note that with certain
272    *          Collection types like HashSet, the order of hosts in the final
273    *          connect string may not be in the same order you added them.
274    * @param chroot
275    *          A chroot value for zookeeper, starting with a forward slash. If no
276    *          chroot is required, use null.
277    * @param httpClient
278    *          the {@link HttpClient} instance to be used for all requests. The provided httpClient should use a
279    *          multi-threaded connection manager.
280    * @throws IllegalArgumentException
281    *           if the chroot value does not start with a forward slash.
282    * @see #CloudSolrClient(String)
283    */
284   public CloudSolrClient(Collection<String> zkHosts, String chroot, HttpClient httpClient) {
285     StringBuilder zkBuilder = new StringBuilder();
286     int lastIndexValue = zkHosts.size() - 1;
287     int i = 0;
288     for (String zkHost : zkHosts) {
289       zkBuilder.append(zkHost);
290       if (i < lastIndexValue) {
291         zkBuilder.append(",");
292       }
293       i++;
294     }
295     if (chroot != null) {
296       if (chroot.startsWith("/")) {
297         zkBuilder.append(chroot);
298       } else {
299         throw new IllegalArgumentException(
300             "The chroot must start with a forward slash.");
301       }
302     }
303 
304     /* Log the constructed connection string and then initialize. */
305     log.info("Final constructed zkHost string: " + zkBuilder.toString());
306 
307     this.zkHost = zkBuilder.toString();
308     this.clientIsInternal = httpClient == null;
309     this.myClient = httpClient == null ? HttpClientUtil.createClient(null) : httpClient;
310     this.lbClient = new LBHttpSolrClient(myClient);
311     this.lbClient.setRequestWriter(new BinaryRequestWriter());
312     this.lbClient.setParser(new BinaryResponseParser());
313     this.updatesToLeaders = true;
314     shutdownLBHttpSolrServer = true;
315   }

456   /**
457    * Connect to the zookeeper ensemble.
458    * This is an optional method that may be used to force a connect before any other requests are sent.
459    *
460    */
461   public void connect() {
462     if (zkStateReader == null) {
463       synchronized (this) {
464         if (zkStateReader == null) {
465           ZkStateReader zk = null;
466           try {
467             zk = new ZkStateReader(zkHost, zkClientTimeout, zkConnectTimeout);
468             zk.createClusterStateWatchersAndUpdate();
469             zkStateReader = zk;
470           } catch (InterruptedException e) {
471             zk.close();
472             Thread.currentThread().interrupt();
473             throw new ZooKeeperException(SolrException.ErrorCode.SERVER_ERROR, "", e);
474           } catch (KeeperException e) {
475             zk.close();
476             throw new ZooKeeperException(SolrException.ErrorCode.SERVER_ERROR, "", e);
477           } catch (Exception e) {
478             if (zk != null) zk.close();
479             // do not wrap because clients may be relying on the underlying exception being thrown
480             throw e;
481           }
482         }
483       }
484     }
485   }

487   /**
488    * Connect to a cluster.  If the cluster is not ready, retry connection up to a given timeout.
489    * @param duration the timeout
490    * @param timeUnit the units of the timeout
491    * @throws TimeoutException if the cluster is not ready after the timeout
492    * @throws InterruptedException if the wait is interrupted
493    */
494   public void connect(long duration, TimeUnit timeUnit) throws TimeoutException, InterruptedException {
495     log.info("Waiting for {} {} for cluster at {} to be ready", duration, timeUnit, zkHost);
496     long timeout = System.nanoTime() + timeUnit.toNanos(duration);
497     while (System.nanoTime() < timeout) {
498       try {
499         connect();
500         log.info("Cluster at {} ready", zkHost);
501         return;
502       }
503       catch (RuntimeException e) {
504         // not ready yet, then...
505       }
506       TimeUnit.MILLISECONDS.sleep(250);
507     }
508     throw new TimeoutException("Timed out waiting for cluster");
509   }

542   private NamedList<Object> directUpdate(AbstractUpdateRequest request, String collection, ClusterState clusterState) throws SolrServerException {
543     UpdateRequest updateRequest = (UpdateRequest) request;
544     ModifiableSolrParams params = (ModifiableSolrParams) request.getParams();
545     ModifiableSolrParams routableParams = new ModifiableSolrParams();
546     ModifiableSolrParams nonRoutableParams = new ModifiableSolrParams();
547 
548     if(params != null) {
549       nonRoutableParams.add(params);
550       routableParams.add(params);
551       for(String param : NON_ROUTABLE_PARAMS) {
552         routableParams.remove(param);
553       }
554     }
555 
556     if (collection == null) {
557       throw new SolrServerException("No collection param specified on request and no default collection has been set.");
558     }
559 
560 
561     //Check to see if the collection is an alias.
562     Aliases aliases = zkStateReader.getAliases();
563     if(aliases != null) {
564       Map<String, String> collectionAliases = aliases.getCollectionAliasMap();
565       if(collectionAliases != null && collectionAliases.containsKey(collection)) {
566         collection = collectionAliases.get(collection);
567       }
568     }
569 
570     DocCollection col = getDocCollection(clusterState, collection,null);
571 
572     DocRouter router = col.getRouter();
573     
574     if (router instanceof ImplicitDocRouter) {
575       // short circuit as optimization
576       return null;
577     }
578 
579     //Create the URL map, which is keyed on slice name.
580     //The value is a list of URLs for each replica in the slice.
581     //The first value in the list is the leader for the slice.
582     Map<String,List<String>> urlMap = buildUrlMap(col);
583     if (urlMap == null) {
584       // we could not find a leader yet - use unoptimized general path
585       return null;
586     }
587 
588     NamedList<Throwable> exceptions = new NamedList<>();
589     NamedList<NamedList> shardResponses = new NamedList<>();
590 
591     Map<String, LBHttpSolrClient.Req> routes = updateRequest.getRoutes(router, col, urlMap, routableParams, this.idField);
592     if (routes == null) {
593       return null;
594     }
595 
596     long start = System.nanoTime();
597 
598     if (parallelUpdates) {
599       final Map<String, Future<NamedList<?>>> responseFutures = new HashMap<>(routes.size());
600       for (final Map.Entry<String, LBHttpSolrClient.Req> entry : routes.entrySet()) {
601         final String url = entry.getKey();
602         final LBHttpSolrClient.Req lbRequest = entry.getValue();
603         try {
604           MDC.put("CloudSolrClient.url", url);
605           responseFutures.put(url, threadPool.submit(new Callable<NamedList<?>>() {
606             @Override
607             public NamedList<?> call() throws Exception {
608               return lbClient.request(lbRequest).getResponse();
609             }
610           }));
611         } finally {
612           MDC.remove("CloudSolrClient.url");
613         }
614       }
615 
616       for (final Map.Entry<String, Future<NamedList<?>>> entry: responseFutures.entrySet()) {
617         final String url = entry.getKey();
618         final Future<NamedList<?>> responseFuture = entry.getValue();
619         try {
620           shardResponses.add(url, responseFuture.get());
621         } catch (InterruptedException e) {
622           Thread.currentThread().interrupt();
623           throw new RuntimeException(e);
624         } catch (ExecutionException e) {
625           exceptions.add(url, e.getCause());
626         }
627       }
628 
629       if (exceptions.size() > 0) {
630         Throwable firstException = exceptions.getVal(0);
631         if(firstException instanceof SolrException) {
632           SolrException e = (SolrException) firstException;
633           throw new RouteException(ErrorCode.getErrorCode(e.code()), exceptions, routes);
634         } else {
635           throw new RouteException(ErrorCode.SERVER_ERROR, exceptions, routes);
636         }
637       }
638     } else {
639       for (Map.Entry<String, LBHttpSolrClient.Req> entry : routes.entrySet()) {
640         String url = entry.getKey();
641         LBHttpSolrClient.Req lbRequest = entry.getValue();
642         try {
643           NamedList<Object> rsp = lbClient.request(lbRequest).getResponse();
644           shardResponses.add(url, rsp);
645         } catch (Exception e) {
646           if(e instanceof SolrException) {
647             throw (SolrException) e;
648           } else {
649             throw new SolrServerException(e);
650           }
651         }
652       }
653     }
654 
655     UpdateRequest nonRoutableRequest = null;
656     List<String> deleteQuery = updateRequest.getDeleteQuery();
657     if (deleteQuery != null && deleteQuery.size() > 0) {
658       UpdateRequest deleteQueryRequest = new UpdateRequest();
659       deleteQueryRequest.setDeleteQuery(deleteQuery);
660       nonRoutableRequest = deleteQueryRequest;
661     }
662     
663     Set<String> paramNames = nonRoutableParams.getParameterNames();
664     
665     Set<String> intersection = new HashSet<>(paramNames);
666     intersection.retainAll(NON_ROUTABLE_PARAMS);
667     
668     if (nonRoutableRequest != null || intersection.size() > 0) {
669       if (nonRoutableRequest == null) {
670         nonRoutableRequest = new UpdateRequest();
671       }
672       nonRoutableRequest.setParams(nonRoutableParams);
673       List<String> urlList = new ArrayList<>();
674       urlList.addAll(routes.keySet());
675       Collections.shuffle(urlList, rand);
676       LBHttpSolrClient.Req req = new LBHttpSolrClient.Req(nonRoutableRequest, urlList);
677       try {
678         LBHttpSolrClient.Rsp rsp = lbClient.request(req);
679         shardResponses.add(urlList.get(0), rsp.getResponse());
680       } catch (Exception e) {
681         throw new SolrException(ErrorCode.SERVER_ERROR, urlList.get(0), e);
682       }
683     }
684 
685     long end = System.nanoTime();
686 
687     RouteResponse rr = condenseResponse(shardResponses, (int) TimeUnit.MILLISECONDS.convert(end - start, TimeUnit.NANOSECONDS));
688     rr.setRouteResponses(shardResponses);
689     rr.setRoutes(routes);
690     return rr;
691   }

693   private Map<String,List<String>> buildUrlMap(DocCollection col) {
694     Map<String, List<String>> urlMap = new HashMap<>();
695     Collection<Slice> slices = col.getActiveSlices();
696     Iterator<Slice> sliceIterator = slices.iterator();
697     while (sliceIterator.hasNext()) {
698       Slice slice = sliceIterator.next();
699       String name = slice.getName();
700       List<String> urls = new ArrayList<>();
701       Replica leader = slice.getLeader();
702       if (leader == null) {
703         // take unoptimized general path - we cannot find a leader yet
704         return null;
705       }
706       ZkCoreNodeProps zkProps = new ZkCoreNodeProps(leader);
707       String url = zkProps.getCoreUrl();
708       urls.add(url);
709       Collection<Replica> replicas = slice.getReplicas();
710       Iterator<Replica> replicaIterator = replicas.iterator();
711       while (replicaIterator.hasNext()) {
712         Replica replica = replicaIterator.next();
713         if (!replica.getNodeName().equals(leader.getNodeName()) &&
714             !replica.getName().equals(leader.getName())) {
715           ZkCoreNodeProps zkProps1 = new ZkCoreNodeProps(replica);
716           String url1 = zkProps1.getCoreUrl();
717           urls.add(url1);
718         }
719       }
720       urlMap.put(name, urls);
721     }
722     return urlMap;

725   public RouteResponse condenseResponse(NamedList response, int timeMillis) {
726     RouteResponse condensed = new RouteResponse();
727     int status = 0;
728     Integer rf = null;
729     Integer minRf = null;
730     for(int i=0; i<response.size(); i++) {
731       NamedList shardResponse = (NamedList)response.getVal(i);
732       NamedList header = (NamedList)shardResponse.get("responseHeader");      
733       Integer shardStatus = (Integer)header.get("status");
734       int s = shardStatus.intValue();
735       if(s > 0) {
736           status = s;
737       }
738       Object rfObj = header.get(UpdateRequest.REPFACT);
739       if (rfObj != null && rfObj instanceof Integer) {
740         Integer routeRf = (Integer)rfObj;
741         if (rf == null || routeRf < rf)
742           rf = routeRf;
743       }
744       minRf = (Integer)header.get(UpdateRequest.MIN_REPFACT);
745     }
746 
747     NamedList cheader = new NamedList();
748     cheader.add("status", status);
749     cheader.add("QTime", timeMillis);
750     if (rf != null)
751       cheader.add(UpdateRequest.REPFACT, rf);
752     if (minRf != null)
753       cheader.add(UpdateRequest.MIN_REPFACT, minRf);
754     
755     condensed.add("responseHeader", cheader);
756     return condensed;
757   }

817    * As this class doesn't watch external collections on the client side,
818    * there's a chance that the request will fail due to cached stale state,
819    * which means the state must be refreshed from ZK and retried.
820    */
821   protected NamedList<Object> requestWithRetryOnStaleState(SolrRequest request, int retryCount, String collection)
822       throws SolrServerException, IOException {
823 
824     connect(); // important to call this before you start working with the ZkStateReader
825 
826     // build up a _stateVer_ param to pass to the server containing all of the
827     // external collection state versions involved in this request, which allows
828     // the server to notify us that our cached state for one or more of the external
829     // collections is stale and needs to be refreshed ... this code has no impact on internal collections
830     String stateVerParam = null;
831     List<DocCollection> requestedCollections = null;
832     boolean isAdmin = ADMIN_PATHS.contains(request.getPath());
833     if (collection != null &&  !isAdmin) { // don't do _stateVer_ checking for admin requests
834       Set<String> requestedCollectionNames = getCollectionNames(getZkStateReader().getClusterState(), collection);
835 
836       StringBuilder stateVerParamBuilder = null;
837       for (String requestedCollection : requestedCollectionNames) {
838         // track the version of state we're using on the client side using the _stateVer_ param
839         DocCollection coll = getDocCollection(getZkStateReader().getClusterState(), requestedCollection,null);
840         int collVer = coll.getZNodeVersion();
841         if (coll.getStateFormat()>1) {
842           if(requestedCollections == null) requestedCollections = new ArrayList<>(requestedCollectionNames.size());
843           requestedCollections.add(coll);
844 
845           if (stateVerParamBuilder == null) {
846             stateVerParamBuilder = new StringBuilder();
847           } else {
848             stateVerParamBuilder.append("|"); // hopefully pipe is not an allowed char in a collection name
849           }
850 
851           stateVerParamBuilder.append(coll.getName()).append(":").append(collVer);
852         }
853       }
854 
855       if (stateVerParamBuilder != null) {
856         stateVerParam = stateVerParamBuilder.toString();
857       }
858     }
859 
860     if (request.getParams() instanceof ModifiableSolrParams) {
861       ModifiableSolrParams params = (ModifiableSolrParams) request.getParams();
862       if (stateVerParam != null) {
863         params.set(STATE_VERSION, stateVerParam);
864       } else {
865         params.remove(STATE_VERSION);
866       }
867     } // else: ??? how to set this ???
868 
869     NamedList<Object> resp = null;
870     try {
871       resp = sendRequest(request, collection);
872       //to avoid an O(n) operation we always add STATE_VERSION to the last and try to read it from there
873       Object o = resp.get(STATE_VERSION, resp.size()-1);
874       if(o != null && o instanceof Map) {
875         //remove this because no one else needs this and tests would fail if they are comparing responses
876         resp.remove(resp.size()-1);
877         Map invalidStates = (Map) o;
878         for (Object invalidEntries : invalidStates.entrySet()) {
879           Map.Entry e = (Map.Entry) invalidEntries;
880           getDocCollection(getZkStateReader().getClusterState(),(String)e.getKey(), (Integer)e.getValue());
881         }
882 
883       }
884     } catch (Exception exc) {
885 
886       Throwable rootCause = SolrException.getRootCause(exc);
887       // don't do retry support for admin requests or if the request doesn't have a collection specified
888       if (collection == null || isAdmin) {
889         if (exc instanceof SolrServerException) {
890           throw (SolrServerException)exc;
891         } else if (exc instanceof IOException) {
892           throw (IOException)exc;
893         }else if (exc instanceof RuntimeException) {
894           throw (RuntimeException) exc;
895         }
896         else {
897           throw new SolrServerException(rootCause);
898         }
899       }
900 
901       int errorCode = (rootCause instanceof SolrException) ?
902           ((SolrException)rootCause).code() : SolrException.ErrorCode.UNKNOWN.code;
903 
904       log.error("Request to collection {} failed due to ("+errorCode+
905           ") {}, retry? "+retryCount, collection, rootCause.toString());
906 
907       boolean wasCommError =
908           (rootCause instanceof ConnectException ||
909               rootCause instanceof ConnectTimeoutException ||
910               rootCause instanceof NoHttpResponseException ||
911               rootCause instanceof SocketException);
912 
913       boolean stateWasStale = false;
914       if (retryCount < MAX_STALE_RETRIES  &&
915           requestedCollections != null    &&
916           !requestedCollections.isEmpty() &&
917           SolrException.ErrorCode.getErrorCode(errorCode) == SolrException.ErrorCode.INVALID_STATE)
918       {
919         // cached state for one or more external collections was stale
920         // re-issue request using updated state
921         stateWasStale = true;
922 
923         // just re-read state for all of them, which is a little heavy handed but hopefully a rare occurrence
924         for (DocCollection ext : requestedCollections) {
925           collectionStateCache.remove(ext.getName());
926         }
927       }
928 
929       // if we experienced a communication error, it's worth checking the state
930       // with ZK just to make sure the node we're trying to hit is still part of the collection
931       if (retryCount < MAX_STALE_RETRIES &&
932           !stateWasStale &&
933           requestedCollections != null &&
934           !requestedCollections.isEmpty() &&
935           wasCommError) {
936         for (DocCollection ext : requestedCollections) {
937           DocCollection latestStateFromZk = getDocCollection(zkStateReader.getClusterState(), ext.getName(),null);
938           if (latestStateFromZk.getZNodeVersion() != ext.getZNodeVersion()) {
939             // looks like we couldn't reach the server because the state was stale == retry
940             stateWasStale = true;
941             // we just pulled state from ZK, so update the cache so that the retry uses it
942             collectionStateCache.put(ext.getName(), new ExpiringCachedDocCollection(latestStateFromZk));
943           }
944         }
945       }
946 
947       if (requestedCollections != null) {
948         requestedCollections.clear(); // done with this
949       }
950 
951       // if the state was stale, then we retry the request once with new state pulled from Zk
952       if (stateWasStale) {
953         log.warn("Re-trying request to  collection(s) "+collection+" after stale state error from server.");
954         resp = requestWithRetryOnStaleState(request, retryCount+1, collection);
955       } else {
956         if(exc instanceof SolrException) {
957           throw exc;
958         } if (exc instanceof SolrServerException) {
959           throw (SolrServerException)exc;
960         } else if (exc instanceof IOException) {
961           throw (IOException)exc;
962         } else {
963           throw new SolrServerException(rootCause);
964         }
965       }
966     }
967 
968     return resp;
969   }

971   protected NamedList<Object> sendRequest(SolrRequest request, String collection)
972       throws SolrServerException, IOException {
973     connect();
974     
975     ClusterState clusterState = zkStateReader.getClusterState();
976     
977     boolean sendToLeaders = false;
978     List<String> replicas = null;
979     
980     if (request instanceof IsUpdateRequest) {
981       if (request instanceof UpdateRequest) {
982         NamedList<Object> response = directUpdate((AbstractUpdateRequest) request, collection, clusterState);
983         if (response != null) {
984           return response;
985         }
986       }
987       sendToLeaders = true;
988       replicas = new ArrayList<>();
989     }
990     
991     SolrParams reqParams = request.getParams();
992     if (reqParams == null) {
993       reqParams = new ModifiableSolrParams();
994     }
995     List<String> theUrlList = new ArrayList<>();
996     if (ADMIN_PATHS.contains(request.getPath())) {
997       Set<String> liveNodes = clusterState.getLiveNodes();
998       for (String liveNode : liveNodes) {
999         theUrlList.add(zkStateReader.getBaseUrlForNodeName(liveNode));
1000       }
1001     } else {
1002       
1003       if (collection == null) {
1004         throw new SolrServerException(
1005             "No collection param specified on request and no default collection has been set.");
1006       }
1007       
1008       Set<String> collectionNames = getCollectionNames(clusterState, collection);
1009       if (collectionNames.size() == 0) {
1010         throw new SolrException(ErrorCode.BAD_REQUEST,
1011             "Could not find collection: " + collection);
1012       }
1013 
1014       String shardKeys =  reqParams.get(ShardParams._ROUTE_);
1015 
1016       // TODO: not a big deal because of the caching, but we could avoid looking
1017       // at every shard
1018       // when getting leaders if we tweaked some things
1019       
1020       // Retrieve slices from the cloud state and, for each collection
1021       // specified,
1022       // add it to the Map of slices.
1023       Map<String,Slice> slices = new HashMap<>();
1024       for (String collectionName : collectionNames) {
1025         DocCollection col = getDocCollection(clusterState, collectionName, null);
1026         Collection<Slice> routeSlices = col.getRouter().getSearchSlices(shardKeys, reqParams , col);
1027         ClientUtils.addSlices(slices, collectionName, routeSlices, true);
1028       }
1029       Set<String> liveNodes = clusterState.getLiveNodes();
1030 
1031       List<String> leaderUrlList = null;
1032       List<String> urlList = null;
1033       List<String> replicasList = null;
1034       
1035       // build a map of unique nodes
1036       // TODO: allow filtering by group, role, etc
1037       Map<String,ZkNodeProps> nodes = new HashMap<>();
1038       List<String> urlList2 = new ArrayList<>();
1039       for (Slice slice : slices.values()) {
1040         for (ZkNodeProps nodeProps : slice.getReplicasMap().values()) {
1041           ZkCoreNodeProps coreNodeProps = new ZkCoreNodeProps(nodeProps);
1042           String node = coreNodeProps.getNodeName();
1043           if (!liveNodes.contains(coreNodeProps.getNodeName())
1044               || Replica.State.getState(coreNodeProps.getState()) != Replica.State.ACTIVE) continue;
1045           if (nodes.put(node, nodeProps) == null) {
1046             if (!sendToLeaders || coreNodeProps.isLeader()) {
1047               String url;
1048               if (reqParams.get(UpdateParams.COLLECTION) == null) {
1049                 url = ZkCoreNodeProps.getCoreUrl(nodeProps.getStr(ZkStateReader.BASE_URL_PROP), collection);
1050               } else {
1051                 url = coreNodeProps.getCoreUrl();
1052               }
1053               urlList2.add(url);
1054             } else {
1055               String url;
1056               if (reqParams.get(UpdateParams.COLLECTION) == null) {
1057                 url = ZkCoreNodeProps.getCoreUrl(nodeProps.getStr(ZkStateReader.BASE_URL_PROP), collection);
1058               } else {
1059                 url = coreNodeProps.getCoreUrl();
1060               }
1061               replicas.add(url);
1062             }
1063           }
1064         }
1065       }
1066       
1067       if (sendToLeaders) {
1068         leaderUrlList = urlList2;
1069         replicasList = replicas;
1070       } else {
1071         urlList = urlList2;
1072       }
1073       
1074       if (sendToLeaders) {
1075         theUrlList = new ArrayList<>(leaderUrlList.size());
1076         theUrlList.addAll(leaderUrlList);
1077       } else {
1078         theUrlList = new ArrayList<>(urlList.size());
1079         theUrlList.addAll(urlList);
1080       }
1081       if(theUrlList.isEmpty()) {
1082         for (String s : collectionNames) {
1083           if(s!=null) collectionStateCache.remove(s);
1084         }
1085         throw new SolrException(SolrException.ErrorCode.INVALID_STATE, "Could not find a healthy node to handle the request.");
1086       }
1087 
1088       Collections.shuffle(theUrlList, rand);
1089       if (sendToLeaders) {
1090         ArrayList<String> theReplicas = new ArrayList<>(
1091             replicasList.size());
1092         theReplicas.addAll(replicasList);
1093         Collections.shuffle(theReplicas, rand);
1094         theUrlList.addAll(theReplicas);
1095       }
1096       
1097     }
1098 
1099     LBHttpSolrClient.Req req = new LBHttpSolrClient.Req(request, theUrlList);
1100     LBHttpSolrClient.Rsp rsp = lbClient.request(req);
1101     return rsp.getResponse();
1102   }

1177   protected DocCollection getDocCollection(ClusterState clusterState, String collection, Integer expectedVersion) throws SolrException {
1178     if (collection == null) return null;
1179     DocCollection col = getFromCache(collection);
1180     if (col != null) {
1181       if (expectedVersion == null) return col;
1182       if (expectedVersion.intValue() == col.getZNodeVersion()) return col;
1183     }
1184 
1185     ClusterState.CollectionRef ref = clusterState.getCollectionRef(collection);
1186     if (ref == null) {
1187       //no such collection exists
1188       return null;
1189     }
1190     if (!ref.isLazilyLoaded()) {
1191       //it is readily available just return it
1192       return ref.get();
1193     }
1194     List locks = this.locks;
1195     final Object lock = locks.get(Math.abs(Hash.murmurhash3_x86_32(collection, 0, collection.length(), 0) % locks.size()));
1196     synchronized (lock) {
1197       //we have waited for sometime just check once again
1198       col = getFromCache(collection);
1199       if (col != null) {
1200         if (expectedVersion == null) return col;
1201         if (expectedVersion.intValue() == col.getZNodeVersion()) {
1202           return col;
1203         } else {
1204           collectionStateCache.remove(collection);
1205         }
1206       }
1207       col = ref.get();//this is a call to ZK
1208     }
1209     if (col == null) return null;
1210     if (col.getStateFormat() > 1) collectionStateCache.put(collection, new ExpiringCachedDocCollection(col));
1211     return col;
1212   }

1221    * Useful for determining the minimum achieved replication factor across
1222    * all shards involved in processing an update request, typically useful
1223    * for gauging the replication factor of a batch. 
1224    */
1225   @SuppressWarnings("rawtypes")
1226   public int getMinAchievedReplicationFactor(String collection, NamedList resp) {
1227     // it's probably already on the top-level header set by condense
1228     NamedList header = (NamedList)resp.get("responseHeader");
1229     Integer achRf = (Integer)header.get(UpdateRequest.REPFACT);
1230     if (achRf != null)
1231       return achRf.intValue();
1232 
1233     // not on the top-level header, walk the shard route tree
1234     Map<String,Integer> shardRf = getShardReplicationFactor(collection, resp);
1235     for (Integer rf : shardRf.values()) {
1236       if (achRf == null || rf < achRf) {
1237         achRf = rf;
1238       }
1239     }    
1240     return (achRf != null) ? achRf.intValue() : -1;

1244    * Walks the NamedList response after performing an update request looking for
1245    * the replication factor that was achieved in each shard involved in the request.
1246    * For single doc updates, there will be only one shard in the return value. 
1247    */
1248   @SuppressWarnings("rawtypes")
1249   public Map<String,Integer> getShardReplicationFactor(String collection, NamedList resp) {
1250     connect();
1251     
1252     Map<String,Integer> results = new HashMap<String,Integer>();
1253     if (resp instanceof CloudSolrClient.RouteResponse) {
1254       NamedList routes = ((CloudSolrClient.RouteResponse)resp).getRouteResponses();
1255       ClusterState clusterState = zkStateReader.getClusterState();     
1256       Map<String,String> leaders = new HashMap<String,String>();
1257       for (Slice slice : clusterState.getActiveSlices(collection)) {
1258         Replica leader = slice.getLeader();
1259         if (leader != null) {
1260           ZkCoreNodeProps zkProps = new ZkCoreNodeProps(leader);
1261           String leaderUrl = zkProps.getBaseUrl() + "/" + zkProps.getCoreName();
1262           leaders.put(leaderUrl, slice.getName());
1263           String altLeaderUrl = zkProps.getBaseUrl() + "/" + collection;
1264           leaders.put(altLeaderUrl, slice.getName());
1265         }
1266       }
1267       
1268       Iterator<Map.Entry<String,Object>> routeIter = routes.iterator();
1269       while (routeIter.hasNext()) {
1270         Map.Entry<String,Object> next = routeIter.next();
1271         String host = next.getKey();
1272         NamedList hostResp = (NamedList)next.getValue();
1273         Integer rf = (Integer)((NamedList)hostResp.get("responseHeader")).get(UpdateRequest.REPFACT);
1274         if (rf != null) {
1275           String shard = leaders.get(host);
1276           if (shard == null) {
1277             if (host.endsWith("/"))
1278               shard = leaders.get(host.substring(0,host.length()-1));
1279             if (shard == null) {
1280               shard = host;
1281             }
1282           }
1283           results.put(shard, rf);
1284         }
1285       }
1286     }    
1287     return results;
1288   }
