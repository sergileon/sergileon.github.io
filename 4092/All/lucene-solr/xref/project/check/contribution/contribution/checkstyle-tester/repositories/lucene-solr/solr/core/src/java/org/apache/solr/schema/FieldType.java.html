146   // Handle additional arguments...
147   protected void setArgs(IndexSchema schema, Map<String,String> args) {
148     // default to STORED, INDEXED, OMIT_TF_POSITIONS and MULTIVALUED depending on schema version
149     properties = (STORED | INDEXED);
150     float schemaVersion = schema.getVersion();
151     if (schemaVersion < 1.1f) properties |= MULTIVALUED;
152     if (schemaVersion > 1.1f) properties |= OMIT_TF_POSITIONS;
153     if (schemaVersion < 1.3) {
154       args.remove("compressThreshold");
155     }
156 
157     this.args = Collections.unmodifiableMap(args);
158     Map<String,String> initArgs = new HashMap<>(args);
159     initArgs.remove(CLASS_NAME); // consume the class arg 
160 
161     trueProperties = FieldProperties.parseProperties(initArgs,true,false);
162     falseProperties = FieldProperties.parseProperties(initArgs,false,false);
163 
164     properties &= ~falseProperties;
165     properties |= trueProperties;
166 
167     for (String prop : FieldProperties.propertyNames) initArgs.remove(prop);
168 
169     init(schema, initArgs);
170 
171     String positionInc = initArgs.get(POSITION_INCREMENT_GAP);
172     if (positionInc != null) {
173       Analyzer analyzer = getIndexAnalyzer();
174       if (analyzer instanceof SolrAnalyzer) {
175         ((SolrAnalyzer)analyzer).setPositionIncrementGap(Integer.parseInt(positionInc));
176       } else {
177         throw new RuntimeException("Can't set " + POSITION_INCREMENT_GAP + " on custom analyzer " + analyzer.getClass());
178       }
179       analyzer = getQueryAnalyzer();
180       if (analyzer instanceof SolrAnalyzer) {
181         ((SolrAnalyzer)analyzer).setPositionIncrementGap(Integer.parseInt(positionInc));
182       } else {
183         throw new RuntimeException("Can't set " + POSITION_INCREMENT_GAP + " on custom analyzer " + analyzer.getClass());
184       }
185       initArgs.remove(POSITION_INCREMENT_GAP);
186     }
187 
188     this.postingsFormat = initArgs.remove(POSTINGS_FORMAT);
189     this.docValuesFormat = initArgs.remove(DOC_VALUES_FORMAT);
190 
191     if (initArgs.size() > 0) {
192       throw new RuntimeException("schema fieldtype " + typeName
193               + "("+ this.getClass().getName() + ")"
194               + " invalid arguments:" + initArgs);
195     }
196   }

226   /**
227    * Used for adding a document when a field needs to be created from a
228    * type and a string.
229    *
230    * <p>
231    * By default, the indexed value is the same as the stored value
232    * (taken from toInternal()).   Having a different representation for
233    * external, internal, and indexed would present quite a few problems
234    * given the current Lucene architecture.  An analyzer for adding docs
235    * would need to translate internal-&gt;indexed while an analyzer for
236    * querying would need to translate external-&gt;indexed.
237    * </p>
238    * <p>
239    * The only other alternative to having internal==indexed would be to have
240    * internal==external.   In this case, toInternal should convert to
241    * the indexed representation, toExternal() should do nothing, and
242    * createField() should *not* call toInternal, but use the external
243    * value and set tokenized=true to get Lucene to convert to the
244    * internal(indexed) form.
245    * </p>
246    *
247    * :TODO: clean up and clarify this explanation.
248    *
249    * @see #toInternal
250    *
251    *
252    */
253   public IndexableField createField(SchemaField field, Object value, float boost) {
254     if (!field.indexed() && !field.stored()) {
255       if (log.isTraceEnabled())
256         log.trace("Ignoring unindexed/unstored field: " + field);
257       return null;
258     }
259     
260     String val;
261     try {
262       val = toInternal(value.toString());
263     } catch (RuntimeException e) {
264       throw new SolrException( SolrException.ErrorCode.SERVER_ERROR, "Error while creating field '" + field + "' from value '" + value + "'", e);
265     }
266     if (val==null) return null;
267 
268     org.apache.lucene.document.FieldType newType = new org.apache.lucene.document.FieldType();
269     newType.setTokenized(field.isTokenized());
270     newType.setStored(field.stored());
271     newType.setOmitNorms(field.omitNorms());
272     newType.setIndexOptions(field.indexed() ? getIndexOptions(field, val) : IndexOptions.NONE);
273     newType.setStoreTermVectors(field.storeTermVector());
274     newType.setStoreTermVectorOffsets(field.storeTermOffsets());
275     newType.setStoreTermVectorPositions(field.storeTermPositions());
276     newType.setStoreTermVectorPayloads(field.storeTermPayloads());
277 
278     return createField(field.getName(), val, newType, boost);
279   }

296   /**
297    * Given a {@link org.apache.solr.schema.SchemaField}, create one or more {@link org.apache.lucene.index.IndexableField} instances
298    * @param field the {@link org.apache.solr.schema.SchemaField}
299    * @param value The value to add to the field
300    * @param boost The boost to apply
301    * @return An array of {@link org.apache.lucene.index.IndexableField}
302    *
303    * @see #createField(SchemaField, Object, float)
304    * @see #isPolyField()
305    */
306   public List<IndexableField> createFields(SchemaField field, Object value, float boost) {
307     IndexableField f = createField( field, value, boost);
308     if (field.hasDocValues() && f.fieldType().docValuesType() == null) {
309       // field types that support doc values should either override createField
310       // to return a field with doc values or extend createFields if this can't
311       // be done in a single field instance (see StrField for example)
312       throw new UnsupportedOperationException("This field type does not support doc values: " + this);
313     }
314     return f==null ? Collections.<IndexableField>emptyList() : Collections.singletonList(f);
315   }

317   protected IndexOptions getIndexOptions(SchemaField field, String internalVal) {
318 
319     IndexOptions options = IndexOptions.DOCS_AND_FREQS_AND_POSITIONS;
320     if (field.omitTermFreqAndPositions()) {
321       options = IndexOptions.DOCS;
322     } else if (field.omitPositions()) {
323       options = IndexOptions.DOCS_AND_FREQS;
324     } else if (field.storeOffsetsWithPositions()) {
325       options = IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS;
326     }
327 
328     return options;
329   }

464   /**
465    * Default analyzer for types that only produce 1 verbatim token...
466    * A maximum size of chars to be read must be specified
467    */
468   protected final class DefaultAnalyzer extends SolrAnalyzer {
469     final int maxChars;
470 
471     DefaultAnalyzer(int maxChars) {
472       this.maxChars=maxChars;
473     }
474 
475     @Override
476     public TokenStreamComponents createComponents(String fieldName) {
477       Tokenizer ts = new Tokenizer() {
478         final char[] cbuf = new char[maxChars];
479         final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
480         final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
481         @Override
482         public boolean incrementToken() throws IOException {
483           clearAttributes();
484           int n = input.read(cbuf,0,maxChars);
485           if (n<=0) return false;
486           String s = toInternal(new String(cbuf,0,n));
487           termAtt.setEmpty().append(s);
488           offsetAtt.setOffset(correctOffset(0),correctOffset(n));
489           return true;
490         }
491       };
492 
493       return new TokenStreamComponents(ts);
494     }
495   }

697   /**
698    * Returns a Query instance for doing range searches on this field type. {@link org.apache.solr.search.SolrQueryParser}
699    * currently passes part1 and part2 as null if they are '*' respectively. minInclusive and maxInclusive are both true
700    * currently by SolrQueryParser but that may change in the future. Also, other QueryParser implementations may have
701    * different semantics.
702    * <p>
703    * Sub-classes should override this method to provide their own range query implementation. They should strive to
704    * handle nulls in part1 and/or part2 as well as unequal minInclusive and maxInclusive parameters gracefully.
705    *
706    * @param parser       the {@link org.apache.solr.search.QParser} calling the method
707    * @param field        the schema field
708    * @param part1        the lower boundary of the range, nulls are allowed.
709    * @param part2        the upper boundary of the range, nulls are allowed
710    * @param minInclusive whether the minimum of the range is inclusive or not
711    * @param maxInclusive whether the maximum of the range is inclusive or not
712    *  @return a Query instance to perform range search according to given parameters
713    *
714    */
715   public Query getRangeQuery(QParser parser, SchemaField field, String part1, String part2, boolean minInclusive, boolean maxInclusive) {
716     // TODO: change these all to use readableToIndexed/bytes instead (e.g. for unicode collation)
717     if (field.hasDocValues() && !field.indexed()) {
718       return DocValuesRangeQuery.newBytesRefRange(
719           field.getName(),
720           part1 == null ? null : new BytesRef(toInternal(part1)),
721           part2 == null ? null : new BytesRef(toInternal(part2)),
722           minInclusive, maxInclusive);
723     } else {
724       MultiTermQuery rangeQuery = TermRangeQuery.newStringRange(
725             field.getName(),
726             part1 == null ? null : toInternal(part1),
727             part2 == null ? null : toInternal(part2),
728             minInclusive, maxInclusive);
729       rangeQuery.setRewriteMethod(getRewriteMethod(parser, field));
730       return rangeQuery;
731     }
732   }

735    * Returns a Query instance for doing searches against a field.
736    * @param parser The {@link org.apache.solr.search.QParser} calling the method
737    * @param field The {@link org.apache.solr.schema.SchemaField} of the field to search
738    * @param externalVal The String representation of the value to search
739    * @return The {@link org.apache.lucene.search.Query} instance.  This implementation returns a {@link org.apache.lucene.search.TermQuery} but overriding queries may not
740    * 
741    */
742   public Query getFieldQuery(QParser parser, SchemaField field, String externalVal) {
743     BytesRefBuilder br = new BytesRefBuilder();
744     readableToIndexed(externalVal, br);
745     if (field.hasDocValues() && !field.indexed()) {
746       // match-only
747       return getRangeQuery(parser, field, externalVal, externalVal, true, true);
748     } else {
749       return new TermQuery(new Term(field.getName(), br));
750     }
751   }

809   /**
810    * Get a map of property name -&gt; value for this field type. 
811    * @param showDefaults if true, include default properties.
812    */
813   public SimpleOrderedMap<Object> getNamedPropertyValues(boolean showDefaults) {
814     SimpleOrderedMap<Object> namedPropertyValues = new SimpleOrderedMap<>();
815     namedPropertyValues.add(TYPE_NAME, getTypeName());
816     namedPropertyValues.add(CLASS_NAME, getClassArg());
817     if (showDefaults) {
818       Map<String,String> fieldTypeArgs = getNonFieldPropertyArgs();
819       if (null != fieldTypeArgs) {
820         for (String key : fieldTypeArgs.keySet()) {
821         if ( ! CLASS_NAME.equals(key) && ! TYPE_NAME.equals(key)) {
822             namedPropertyValues.add(key, fieldTypeArgs.get(key));
823           }
824         }
825       }
826       if (this instanceof TextField) {
827         namedPropertyValues.add(AUTO_GENERATE_PHRASE_QUERIES, ((TextField) this).getAutoGeneratePhraseQueries());
828       }
829       namedPropertyValues.add(getPropertyName(INDEXED), hasProperty(INDEXED));
830       namedPropertyValues.add(getPropertyName(STORED), hasProperty(STORED));
831       namedPropertyValues.add(getPropertyName(DOC_VALUES), hasProperty(DOC_VALUES));
832       namedPropertyValues.add(getPropertyName(STORE_TERMVECTORS), hasProperty(STORE_TERMVECTORS));
833       namedPropertyValues.add(getPropertyName(STORE_TERMPOSITIONS), hasProperty(STORE_TERMPOSITIONS));
834       namedPropertyValues.add(getPropertyName(STORE_TERMOFFSETS), hasProperty(STORE_TERMOFFSETS));
835       namedPropertyValues.add(getPropertyName(OMIT_NORMS), hasProperty(OMIT_NORMS));
836       namedPropertyValues.add(getPropertyName(OMIT_TF_POSITIONS), hasProperty(OMIT_TF_POSITIONS));
837       namedPropertyValues.add(getPropertyName(OMIT_POSITIONS), hasProperty(OMIT_POSITIONS));
838       namedPropertyValues.add(getPropertyName(STORE_OFFSETS), hasProperty(STORE_OFFSETS));
839       namedPropertyValues.add(getPropertyName(MULTIVALUED), hasProperty(MULTIVALUED));
840       if (hasProperty(SORT_MISSING_FIRST)) {
841         namedPropertyValues.add(getPropertyName(SORT_MISSING_FIRST), true);
842       } else if (hasProperty(SORT_MISSING_LAST)) {
843         namedPropertyValues.add(getPropertyName(SORT_MISSING_LAST), true);
844       }
845       namedPropertyValues.add(getPropertyName(TOKENIZED), isTokenized());
846       // The BINARY property is always false
847       // namedPropertyValues.add(getPropertyName(BINARY), hasProperty(BINARY));
848       if (null != getPostingsFormat()) {
849         namedPropertyValues.add(POSTINGS_FORMAT, getPostingsFormat());
850       }
851       if (null != getDocValuesFormat()) {
852         namedPropertyValues.add(DOC_VALUES_FORMAT, getDocValuesFormat());
853       }
854     } else { // Don't show defaults
855       Set<String> fieldProperties = new HashSet<>();
856       for (String propertyName : FieldProperties.propertyNames) {
857         fieldProperties.add(propertyName);
858       }
859 
860       for (String key : args.keySet()) {
861         if (fieldProperties.contains(key)) {
862           namedPropertyValues.add(key, StrUtils.parseBool(args.get(key)));
863         } else if (!CLASS_NAME.equals(key) && !TYPE_NAME.equals(key)) {
864           namedPropertyValues.add(key, args.get(key));
865         }
866       }
867     }
868 
869     if (null != getSimilarityFactory()) {
870       namedPropertyValues.add(SIMILARITY, getSimilarityFactory().getNamedPropertyValues());
871     }
872     
873     if (isExplicitAnalyzer()) {
874       String analyzerProperty = isExplicitQueryAnalyzer() ? INDEX_ANALYZER : ANALYZER;
875       namedPropertyValues.add(analyzerProperty, getAnalyzerProperties(getIndexAnalyzer()));
876     } 
877     if (isExplicitQueryAnalyzer()) {
878       String analyzerProperty = isExplicitAnalyzer() ? QUERY_ANALYZER : ANALYZER;
879       namedPropertyValues.add(analyzerProperty, getAnalyzerProperties(getQueryAnalyzer()));
880     }
881     if (this instanceof TextField) {
882       if (((TextField)this).isExplicitMultiTermAnalyzer()) {
883         namedPropertyValues.add(MULTI_TERM_ANALYZER, getAnalyzerProperties(((TextField) this).getMultiTermAnalyzer()));
884       }
885     }
886 
887     return namedPropertyValues;
888   }

899   /** 
900    * Returns a description of the given analyzer, by either reporting the Analyzer class
901    * name (and optionally luceneMatchVersion) if it's not a TokenizerChain, or if it is,
902    * querying each analysis factory for its name and args.
903    */
904   protected static SimpleOrderedMap<Object> getAnalyzerProperties(Analyzer analyzer) {
905     SimpleOrderedMap<Object> analyzerProps = new SimpleOrderedMap<>();
906 
907     if (analyzer instanceof TokenizerChain) {
908       Map<String,String> factoryArgs;
909       TokenizerChain tokenizerChain = (TokenizerChain)analyzer;
910       CharFilterFactory[] charFilterFactories = tokenizerChain.getCharFilterFactories();
911       if (0 < charFilterFactories.length) {
912         List<SimpleOrderedMap<Object>> charFilterProps = new ArrayList<>();
913         for (CharFilterFactory charFilterFactory : charFilterFactories) {
914           SimpleOrderedMap<Object> props = new SimpleOrderedMap<>();
915           props.add(CLASS_NAME, charFilterFactory.getClassArg());
916           factoryArgs = charFilterFactory.getOriginalArgs();
917           if (null != factoryArgs) {
918             for (String key : factoryArgs.keySet()) {
919               if ( ! CLASS_NAME.equals(key)) {
920                 if (LUCENE_MATCH_VERSION_PARAM.equals(key)) {
921                   if (charFilterFactory.isExplicitLuceneMatchVersion()) {
922                     props.add(key, factoryArgs.get(key));
923                   }
924                 } else {
925                    props.add(key, factoryArgs.get(key));
926                 }
927               }
928             }
929           }
930           charFilterProps.add(props);
931         }
932         analyzerProps.add(CHAR_FILTERS, charFilterProps);
933       }
934 
935       SimpleOrderedMap<Object> tokenizerProps = new SimpleOrderedMap<>();
936       TokenizerFactory tokenizerFactory = tokenizerChain.getTokenizerFactory();
937       tokenizerProps.add(CLASS_NAME, tokenizerFactory.getClassArg());
938       factoryArgs = tokenizerFactory.getOriginalArgs();
939       if (null != factoryArgs) {
940         for (String key : factoryArgs.keySet()) {
941           if ( ! CLASS_NAME.equals(key)) {
942             if (LUCENE_MATCH_VERSION_PARAM.equals(key)) {
943               if (tokenizerFactory.isExplicitLuceneMatchVersion()) {
944                 tokenizerProps.add(key, factoryArgs.get(key));
945               }
946             } else {
947               tokenizerProps.add(key, factoryArgs.get(key));
948             }
949           }
950         }
951       }
952       analyzerProps.add(TOKENIZER, tokenizerProps);
953 
954       TokenFilterFactory[] filterFactories = tokenizerChain.getTokenFilterFactories();
955       if (0 < filterFactories.length) {
956         List<SimpleOrderedMap<Object>> filterProps = new ArrayList<>();
957         for (TokenFilterFactory filterFactory : filterFactories) {
958           SimpleOrderedMap<Object> props = new SimpleOrderedMap<>();
959           props.add(CLASS_NAME, filterFactory.getClassArg());
960           factoryArgs = filterFactory.getOriginalArgs();
961           if (null != factoryArgs) {
962             for (String key : factoryArgs.keySet()) {
963               if ( ! CLASS_NAME.equals(key)) {
964                 if (LUCENE_MATCH_VERSION_PARAM.equals(key)) {
965                   if (filterFactory.isExplicitLuceneMatchVersion()) {
966                     props.add(key, factoryArgs.get(key));
967                   }
968                 } else {
969                   props.add(key, factoryArgs.get(key));
970                 }
971               }
972             }
973           }
974           filterProps.add(props);
975         }
976         analyzerProps.add(FILTERS, filterProps);
977       }
978     } else { // analyzer is not instanceof TokenizerChain
979       analyzerProps.add(CLASS_NAME, analyzer.getClass().getName());
980       if (analyzer.getVersion() != Version.LATEST) {
981         analyzerProps.add(LUCENE_MATCH_VERSION_PARAM, analyzer.getVersion().toString());
982       }
983     }
984     return analyzerProps;
985   }
