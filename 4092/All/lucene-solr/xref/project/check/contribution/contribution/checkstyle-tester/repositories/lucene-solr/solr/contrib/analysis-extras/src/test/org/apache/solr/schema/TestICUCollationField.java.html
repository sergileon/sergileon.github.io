39    @BeforeClass
40    public static void beforeClass() throws Exception {
41      String home = setupSolrHome();
42      initCore("solrconfig.xml","schema.xml", home);
43      // add some docs
44      assertU(adoc("id", "1", "text", "\u0633\u0627\u0628"));
45      assertU(adoc("id", "2", "text", "I WİLL USE TURKİSH CASING"));
46      assertU(adoc("id", "3", "text", "ı will use turkish casıng"));
47      assertU(adoc("id", "4", "text", "Töne"));
48      assertU(adoc("id", "5", "text", "I W\u0049\u0307LL USE TURKİSH CASING"));
49      assertU(adoc("id", "6", "text", "Ｔｅｓｔｉｎｇ"));
50      assertU(adoc("id", "7", "text", "Tone"));
51      assertU(adoc("id", "8", "text", "Testing"));
52      assertU(adoc("id", "9", "text", "testing"));
53      assertU(adoc("id", "10", "text", "toene"));
54      assertU(adoc("id", "11", "text", "Tzne"));
55      assertU(adoc("id", "12", "text", "\u0698\u0698"));
56      assertU(commit());
57    }

59    /**
60     * Ugly: but what to do? We want to test custom sort, which reads rules in as a resource.
61     * These are largish files, and jvm-specific (as our documentation says, you should always
62     * look out for jvm differences with collation).
63     * So it's preferable to create this file on-the-fly.
64     */
65    public static String setupSolrHome() throws Exception {
66      String tmpFile = createTempDir().toFile().getAbsolutePath();
67      // make data and conf dirs
68      new File(tmpFile  + "/collection1", "data").mkdirs();
69      File confDir = new File(tmpFile + "/collection1", "conf");
70      confDir.mkdirs();
71      
72      // copy over configuration files
73      FileUtils.copyFile(getFile("analysis-extras/solr/collection1/conf/solrconfig-icucollate.xml"), new File(confDir, "solrconfig.xml"));
74      FileUtils.copyFile(getFile("analysis-extras/solr/collection1/conf/schema-icucollate.xml"), new File(confDir, "schema.xml"));
75      
76      // generate custom collation rules (DIN 5007-2), saving to customrules.dat
77      RuleBasedCollator baseCollator = (RuleBasedCollator) Collator.getInstance(new ULocale("de", "DE"));
78  
79      String DIN5007_2_tailorings =
80        "& ae , a\u0308 & AE , A\u0308"+
81        "& oe , o\u0308 & OE , O\u0308"+
82        "& ue , u\u0308 & UE , u\u0308";
83  
84      RuleBasedCollator tailoredCollator = new RuleBasedCollator(baseCollator.getRules() + DIN5007_2_tailorings);
85      String tailoredRules = tailoredCollator.getRules();
86      FileOutputStream os = new FileOutputStream(new File(confDir, "customrules.dat"));
87      IOUtils.write(tailoredRules, os, "UTF-8");
88      os.close();
89  
90      return tmpFile;
91    }

93    /** 
94     * Test termquery with german DIN 5007-1 primary strength.
95     * In this case, ö is equivalent to o (but not oe) 
96     */
97    public void testBasicTermQuery() {
98      assertQ("Collated TQ: ",
99         req("fl", "id", "q", "sort_de:tone", "sort", "id asc" ),
100               "//*[@numFound='2']",
101               "//result/doc[1]/int[@name='id'][.=4]",
102               "//result/doc[2]/int[@name='id'][.=7]"
103     );
104   }
105   
106   /** 
107    * Test rangequery again with the DIN 5007-1 collator.
108    * We do a range query of tone .. tp, in binary order this
109    * would retrieve nothing due to case and accent differences.
110    */
111   public void testBasicRangeQuery() {
112     assertQ("Collated RangeQ: ",
113         req("fl", "id", "q", "sort_de:[tone TO tp]", "sort", "id asc" ),
114                "//*[@numFound='2']",
115                "//result/doc[1]/int[@name='id'][.=4]",
116                "//result/doc[2]/int[@name='id'][.=7]"
117      );
118   }
119   
120   /** 
121    * Test sort with a danish collator. ö is ordered after z
122    */
123   public void testBasicSort() {
124     assertQ("Collated Sort: ",
125         req("fl", "id", "q", "sort_da:[tz TO töz]", "sort", "sort_da asc" ),
126                "//*[@numFound='2']",
127                "//result/doc[1]/int[@name='id'][.=11]",
128                "//result/doc[2]/int[@name='id'][.=4]"
129      );
130   }
131   
132   /** 
133    * Test sort with an arabic collator. U+0633 is ordered after U+0698.
134    * With a binary collator, the range would also return nothing.
135    */
136   public void testArabicSort() {
137     assertQ("Collated Sort: ",
138         req("fl", "id", "q", "sort_ar:[\u0698 TO \u0633\u0633]", "sort", "sort_ar asc" ),
139                "//*[@numFound='2']",
140                "//result/doc[1]/int[@name='id'][.=12]",
141                "//result/doc[2]/int[@name='id'][.=1]"
142      );
143   }
144 
145   /** 
146    * Test rangequery again with an Arabic collator.
147    * Binary order would normally order U+0633 in this range.
148    */
149   public void testNegativeRangeQuery() {
150     assertQ("Collated RangeQ: ",
151         req("fl", "id", "q", "sort_ar:[\u062F TO \u0698]", "sort", "id asc" ),
152                "//*[@numFound='0']"
153      );
154   }
155   /**
156    * Test canonical decomposition with turkish primary strength. 
157    * With this sort order, İ is the uppercase form of i, and I is the uppercase form of ı.
158    * We index a decomposed form of İ.
159    */
160   public void testCanonicalDecomposition() {
161     assertQ("Collated TQ: ",
162         req("fl", "id", "q", "sort_tr_canon:\"I Will Use Turkish Casıng\"", "sort", "id asc" ),
163                "//*[@numFound='3']",
164                "//result/doc[1]/int[@name='id'][.=2]",
165                "//result/doc[2]/int[@name='id'][.=3]",
166                "//result/doc[3]/int[@name='id'][.=5]"
167      );
168   }

170   /** 
171    * Test termquery with custom collator (DIN 5007-2).
172    * In this case, ö is equivalent to oe (but not o) 
173    */
174   public void testCustomCollation() {
175     assertQ("Collated TQ: ",
176         req("fl", "id", "q", "sort_custom:toene", "sort", "id asc" ),
177                "//*[@numFound='2']",
178                "//result/doc[1]/int[@name='id'][.=4]",
179                "//result/doc[2]/int[@name='id'][.=10]"
180      );
