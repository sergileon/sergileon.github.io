43  // Used by IndexWriter to hold open SegmentReaders (for
44  // searching or merging), plus pending deletes and updates,
45  // for a given segment
46  class ReadersAndUpdates {
47    // Not final because we replace (clone) when we need to
48    // change it and it's been shared:
49    public final SegmentCommitInfo info;
50  
51    // Tracks how many consumers are using this instance:
52    private final AtomicInteger refCount = new AtomicInteger(1);
53  
54    private final IndexWriter writer;
55  
56    // Set once (null, and then maybe set, and never set again):
57    private SegmentReader reader;
58  
59    // Holds the current shared (readable and writable)
60    // liveDocs.  This is null when there are no deleted
61    // docs, and it's copy-on-write (cloned whenever we need
62    // to change it but it's been shared to an external NRT
63    // reader).
64    private Bits liveDocs;
65  
66    // How many further deletions we've done against
67    // liveDocs vs when we loaded it or last wrote it:
68    private int pendingDeleteCount;
69  
70    // True if the current liveDocs is referenced by an
71    // external NRT reader:
72    private boolean liveDocsShared;
73  
74    // Indicates whether this segment is currently being merged. While a segment
75    // is merging, all field updates are also registered in the
76    // mergingNumericUpdates map. Also, calls to writeFieldUpdates merge the 
77    // updates with mergingNumericUpdates.
78    // That way, when the segment is done merging, IndexWriter can apply the
79    // updates on the merged segment too.
80    private boolean isMerging = false;
81    
82    private final Map<String,DocValuesFieldUpdates> mergingDVUpdates = new HashMap<>();

124   public synchronized boolean verifyDocCounts() {
125     int count;
126     if (liveDocs != null) {
127       count = 0;
128       for(int docID=0;docID<info.info.maxDoc();docID++) {
129         if (liveDocs.get(docID)) {
130           count++;
131         }
132       }
133     } else {
134       count = info.info.maxDoc();
135     }
136 
137     assert info.info.maxDoc() - info.getDelCount() - pendingDeleteCount == count: "info.maxDoc=" + info.info.maxDoc() + " info.getDelCount()=" + info.getDelCount() + " pendingDeleteCount=" + pendingDeleteCount + " count=" + count;
138     return true;
139   }

141   /** Returns a {@link SegmentReader}. */
142   public SegmentReader getReader(IOContext context) throws IOException {
143     if (reader == null) {
144       // We steal returned ref:
145       reader = new SegmentReader(info, context);
146       if (liveDocs == null) {
147         liveDocs = reader.getLiveDocs();
148       }
149     }
150     
151     // Ref for caller
152     reader.incRef();
153     return reader;
154   }

161   public synchronized boolean delete(int docID) {
162     assert liveDocs != null;
163     assert Thread.holdsLock(writer);
164     assert docID >= 0 && docID < liveDocs.length() : "out of bounds: docid=" + docID + " liveDocsLength=" + liveDocs.length() + " seg=" + info.info.name + " maxDoc=" + info.info.maxDoc();
165     assert !liveDocsShared;
166     final boolean didDelete = liveDocs.get(docID);
167     if (didDelete) {
168       ((MutableBits) liveDocs).clear(docID);
169       pendingDeleteCount++;
170       //System.out.println("  new del seg=" + info + " docID=" + docID + " pendingDelCount=" + pendingDeleteCount + " totDelCount=" + (info.info.maxDoc()-liveDocs.count()));
171     }
172     return didDelete;
173   }

191   /**
192    * Returns a ref to a clone. NOTE: you should decRef() the reader when you're
193    * done (ie do not call close()).
194    */
195   public synchronized SegmentReader getReadOnlyClone(IOContext context) throws IOException {
196     if (reader == null) {
197       getReader(context).decRef();
198       assert reader != null;
199     }
200     // force new liveDocs in initWritableLiveDocs even if it's null
201     liveDocsShared = true;
202     if (liveDocs != null) {
203       return new SegmentReader(reader.getSegmentInfo(), reader, liveDocs, info.info.maxDoc() - info.getDelCount() - pendingDeleteCount);
204     } else {
205       // liveDocs == null and reader != null. That can only be if there are no deletes
206       assert reader.getLiveDocs() == null;
207       reader.incRef();
208       return reader;
209     }
210   }

212   public synchronized void initWritableLiveDocs() throws IOException {
213     assert Thread.holdsLock(writer);
214     assert info.info.maxDoc() > 0;
215     //System.out.println("initWritableLivedocs seg=" + info + " liveDocs=" + liveDocs + " shared=" + shared);
216     if (liveDocsShared) {
217       // Copy on write: this means we've cloned a
218       // SegmentReader sharing the current liveDocs
219       // instance; must now make a private clone so we can
220       // change it:
221       LiveDocsFormat liveDocsFormat = info.info.getCodec().liveDocsFormat();
222       if (liveDocs == null) {
223         //System.out.println("create BV seg=" + info);
224         liveDocs = liveDocsFormat.newLiveDocs(info.info.maxDoc());
225       } else {
226         liveDocs = liveDocsFormat.newLiveDocs(liveDocs);
227       }
228       liveDocsShared = false;
229     }
230   }

247   public synchronized void dropChanges() {
248     // Discard (don't save) changes when we are dropping
249     // the reader; this is used only on the sub-readers
250     // after a successful merge.  If deletes had
251     // accumulated on those sub-readers while the merge
252     // is running, by now we have carried forward those
253     // deletes onto the newly merged segment, so we can
254     // discard them on the sub-readers:
255     pendingDeleteCount = 0;
256     dropMergingUpdates();
257   }

259   // Commit live docs (writes new _X_N.del files) and field updates (writes new
260   // _X_N updates files) to the directory; returns true if it wrote any file
261   // and false if there were no new deletes or updates to write:
262   public synchronized boolean writeLiveDocs(Directory dir) throws IOException {
263     assert Thread.holdsLock(writer);
264     //System.out.println("rld.writeLiveDocs seg=" + info + " pendingDelCount=" + pendingDeleteCount + " numericUpdates=" + numericUpdates);
265     if (pendingDeleteCount == 0) {
266       return false;
267     }
268     
269     // We have new deletes
270     assert liveDocs.length() == info.info.maxDoc();
271     
272     // Do this so we can delete any created files on
273     // exception; this saves all codecs from having to do
274     // it:
275     TrackingDirectoryWrapper trackingDir = new TrackingDirectoryWrapper(dir);
276     
277     // We can write directly to the actual name (vs to a
278     // .tmp & renaming it) because the file is not live
279     // until segments file is written:
280     boolean success = false;
281     try {
282       Codec codec = info.info.getCodec();
283       codec.liveDocsFormat().writeLiveDocs((MutableBits)liveDocs, trackingDir, info, pendingDeleteCount, IOContext.DEFAULT);
284       success = true;
285     } finally {
286       if (!success) {
287         // Advance only the nextWriteDelGen so that a 2nd
288         // attempt to write will write to a new file
289         info.advanceNextWriteDelGen();
290         
291         // Delete any partially created file(s):
292         for (String fileName : trackingDir.getCreatedFiles()) {
293           IOUtils.deleteFilesIgnoringExceptions(dir, fileName);
294         }
295       }
296     }
297     
298     // If we hit an exc in the line above (eg disk full)
299     // then info's delGen remains pointing to the previous
300     // (successfully written) del docs:
301     info.advanceDelGen();
302     info.setDelCount(info.getDelCount() + pendingDeleteCount);
303     pendingDeleteCount = 0;
304     
305     return true;
306   }

308   @SuppressWarnings("synthetic-access")
309   private void handleNumericDVUpdates(FieldInfos infos, Map<String,NumericDocValuesFieldUpdates> updates,
310       Directory dir, DocValuesFormat dvFormat, final SegmentReader reader, Map<Integer,Set<String>> fieldFiles) throws IOException {
311     for (Entry<String,NumericDocValuesFieldUpdates> e : updates.entrySet()) {
312       final String field = e.getKey();
313       final NumericDocValuesFieldUpdates fieldUpdates = e.getValue();
314 
315       final long nextDocValuesGen = info.getNextDocValuesGen();
316       final String segmentSuffix = Long.toString(nextDocValuesGen, Character.MAX_RADIX);
317       final long estUpdatesSize = fieldUpdates.ramBytesPerDoc() * info.info.maxDoc();
318       final IOContext updatesContext = new IOContext(new FlushInfo(info.info.maxDoc(), estUpdatesSize));
319       final FieldInfo fieldInfo = infos.fieldInfo(field);
320       assert fieldInfo != null;
321       fieldInfo.setDocValuesGen(nextDocValuesGen);
322       final FieldInfos fieldInfos = new FieldInfos(new FieldInfo[] { fieldInfo });
323       // separately also track which files were created for this gen
324       final TrackingDirectoryWrapper trackingDir = new TrackingDirectoryWrapper(dir);
325       final SegmentWriteState state = new SegmentWriteState(null, trackingDir, info.info, fieldInfos, null, updatesContext, segmentSuffix);
326       try (final DocValuesConsumer fieldsConsumer = dvFormat.fieldsConsumer(state)) {
327         // write the numeric updates to a new gen'd docvalues file
328         fieldsConsumer.addNumericField(fieldInfo, new Iterable<Number>() {
329           final NumericDocValues currentValues = reader.getNumericDocValues(field);
330           final Bits docsWithField = reader.getDocsWithField(field);
331           final int maxDoc = reader.maxDoc();
332           final NumericDocValuesFieldUpdates.Iterator updatesIter = fieldUpdates.iterator();
333           @Override
334           public Iterator<Number> iterator() {
335             updatesIter.reset();
336             return new Iterator<Number>() {
337 
338               int curDoc = -1;
339               int updateDoc = updatesIter.nextDoc();
340               
341               @Override
342               public boolean hasNext() {
343                 return curDoc < maxDoc - 1;
344               }
345 
346               @Override
347               public Number next() {
348                 if (++curDoc >= maxDoc) {
349                   throw new NoSuchElementException("no more documents to return values for");
350                 }
351                 if (curDoc == updateDoc) { // this document has an updated value
352                   Long value = updatesIter.value(); // either null (unset value) or updated value
353                   updateDoc = updatesIter.nextDoc(); // prepare for next round
354                   return value;
355                 } else {
356                   // no update for this document
357                   assert curDoc < updateDoc;
358                   if (currentValues != null && docsWithField.get(curDoc)) {
359                     // only read the current value if the document had a value before
360                     return currentValues.get(curDoc);
361                   } else {
362                     return null;
363                   }
364                 }
365               }
366 
367               @Override
368               public void remove() {
369                 throw new UnsupportedOperationException("this iterator does not support removing elements");
370               }
371             };
372           }
373         });
374       }
375       info.advanceDocValuesGen();
376       assert !fieldFiles.containsKey(fieldInfo.number);
377       fieldFiles.put(fieldInfo.number, trackingDir.getCreatedFiles());
378     }
379   }

454   private Set<String> writeFieldInfosGen(FieldInfos fieldInfos, Directory dir, DocValuesFormat dvFormat, 
455       FieldInfosFormat infosFormat) throws IOException {
456     final long nextFieldInfosGen = info.getNextFieldInfosGen();
457     final String segmentSuffix = Long.toString(nextFieldInfosGen, Character.MAX_RADIX);
458     // we write approximately that many bytes (based on Lucene46DVF):
459     // HEADER + FOOTER: 40
460     // 90 bytes per-field (over estimating long name and attributes map)
461     final long estInfosSize = 40 + 90 * fieldInfos.size();
462     final IOContext infosContext = new IOContext(new FlushInfo(info.info.maxDoc(), estInfosSize));
463     // separately also track which files were created for this gen
464     final TrackingDirectoryWrapper trackingDir = new TrackingDirectoryWrapper(dir);
465     infosFormat.write(trackingDir, info.info, segmentSuffix, fieldInfos, infosContext);
466     info.advanceFieldInfosGen();
467     return trackingDir.getCreatedFiles();
468   }

470   // Writes field updates (new _X_N updates files) to the directory
471   public synchronized void writeFieldUpdates(Directory dir, DocValuesFieldUpdates.Container dvUpdates) throws IOException {
472     assert Thread.holdsLock(writer);
473     //System.out.println("rld.writeFieldUpdates: seg=" + info + " numericFieldUpdates=" + numericFieldUpdates);
474     
475     assert dvUpdates.any();
476     
477     // Do this so we can delete any created files on
478     // exception; this saves all codecs from having to do
479     // it:
480     TrackingDirectoryWrapper trackingDir = new TrackingDirectoryWrapper(dir);
481     
482     final Map<Integer,Set<String>> newDVFiles = new HashMap<>();
483     Set<String> fieldInfosFiles = null;
484     FieldInfos fieldInfos = null;
485     boolean success = false;
486     try {
487       final Codec codec = info.info.getCodec();
488 
489       // reader could be null e.g. for a just merged segment (from
490       // IndexWriter.commitMergedDeletes).
491       final SegmentReader reader = this.reader == null ? new SegmentReader(info, IOContext.READONCE) : this.reader;
492       try {
493         // clone FieldInfos so that we can update their dvGen separately from
494         // the reader's infos and write them to a new fieldInfos_gen file
495         FieldInfos.Builder builder = new FieldInfos.Builder(writer.globalFieldNumberMap);
496         // cannot use builder.add(reader.getFieldInfos()) because it does not
497         // clone FI.attributes as well FI.dvGen
498         for (FieldInfo fi : reader.getFieldInfos()) {
499           FieldInfo clone = builder.add(fi);
500           // copy the stuff FieldInfos.Builder doesn't copy
501           for (Entry<String,String> e : fi.attributes().entrySet()) {
502             clone.putAttribute(e.getKey(), e.getValue());
503           }
504           clone.setDocValuesGen(fi.getDocValuesGen());
505         }
506         // create new fields or update existing ones to have NumericDV type
507         for (String f : dvUpdates.numericDVUpdates.keySet()) {
508           FieldInfo fieldInfo = builder.getOrAdd(f);
509           fieldInfo.setDocValuesType(DocValuesType.NUMERIC);
510         }
511         // create new fields or update existing ones to have BinaryDV type
512         for (String f : dvUpdates.binaryDVUpdates.keySet()) {
513           FieldInfo fieldInfo = builder.getOrAdd(f);
514           fieldInfo.setDocValuesType(DocValuesType.BINARY);
515         }
516         
517         fieldInfos = builder.finish();
518         final DocValuesFormat docValuesFormat = codec.docValuesFormat();
519         
520 //          System.out.println("[" + Thread.currentThread().getName() + "] RLD.writeFieldUpdates: applying numeric updates; seg=" + info + " updates=" + numericFieldUpdates);
521         handleNumericDVUpdates(fieldInfos, dvUpdates.numericDVUpdates, trackingDir, docValuesFormat, reader, newDVFiles);
522         
523 //        System.out.println("[" + Thread.currentThread().getName() + "] RAU.writeFieldUpdates: applying binary updates; seg=" + info + " updates=" + dvUpdates.binaryDVUpdates);
524         handleBinaryDVUpdates(fieldInfos, dvUpdates.binaryDVUpdates, trackingDir, docValuesFormat, reader, newDVFiles);
525 
526 //        System.out.println("[" + Thread.currentThread().getName() + "] RAU.writeFieldUpdates: write fieldInfos; seg=" + info);
527         fieldInfosFiles = writeFieldInfosGen(fieldInfos, trackingDir, docValuesFormat, codec.fieldInfosFormat());
528       } finally {
529         if (reader != this.reader) {
530 //          System.out.println("[" + Thread.currentThread().getName() + "] RLD.writeLiveDocs: closeReader " + reader);
531           reader.close();
532         }
533       }
534     
535       success = true;
536     } finally {
537       if (!success) {
538         // Advance only the nextWriteFieldInfosGen and nextWriteDocValuesGen, so
539         // that a 2nd attempt to write will write to a new file
540         info.advanceNextWriteFieldInfosGen();
541         info.advanceNextWriteDocValuesGen();
542         
543         // Delete any partially created file(s):
544         for (String fileName : trackingDir.getCreatedFiles()) {
545           IOUtils.deleteFilesIgnoringExceptions(dir, fileName);
546         }
547       }
548     }
549     
550     // copy all the updates to mergingUpdates, so they can later be applied to the merged segment
551     if (isMerging) {
552       for (Entry<String,NumericDocValuesFieldUpdates> e : dvUpdates.numericDVUpdates.entrySet()) {
553         DocValuesFieldUpdates updates = mergingDVUpdates.get(e.getKey());
554         if (updates == null) {
555           mergingDVUpdates.put(e.getKey(), e.getValue());
556         } else {
557           updates.merge(e.getValue());
558         }
559       }
560       for (Entry<String,BinaryDocValuesFieldUpdates> e : dvUpdates.binaryDVUpdates.entrySet()) {
561         DocValuesFieldUpdates updates = mergingDVUpdates.get(e.getKey());
562         if (updates == null) {
563           mergingDVUpdates.put(e.getKey(), e.getValue());
564         } else {
565           updates.merge(e.getValue());
566         }
567       }
568     }
569     
570     // writing field updates succeeded
571     assert fieldInfosFiles != null;
572     info.setFieldInfosFiles(fieldInfosFiles);
573     
574     // update the doc-values updates files. the files map each field to its set
575     // of files, hence we copy from the existing map all fields w/ updates that
576     // were not updated in this session, and add new mappings for fields that
577     // were updated now.
578     assert !newDVFiles.isEmpty();
579     for (Entry<Integer,Set<String>> e : info.getDocValuesUpdatesFiles().entrySet()) {
580       if (!newDVFiles.containsKey(e.getKey())) {
581         newDVFiles.put(e.getKey(), e.getValue());
582       }
583     }
584     info.setDocValuesUpdatesFiles(newDVFiles);
585     
586     // wrote new files, should checkpoint()
587     writer.checkpoint();
588 
589     // if there is a reader open, reopen it to reflect the updates
590     if (reader != null) {
591       SegmentReader newReader = new SegmentReader(info, reader, liveDocs, info.info.maxDoc() - info.getDelCount() - pendingDeleteCount);
592       boolean reopened = false;
593       try {
594         reader.decRef();
595         reader = newReader;
596         reopened = true;
597       } finally {
598         if (!reopened) {
599           newReader.decRef();
600         }
601       }
602     }
603   }

605   /**
606    * Returns a reader for merge. This method applies field updates if there are
607    * any and marks that this segment is currently merging.
608    */
609   synchronized SegmentReader getReaderForMerge(IOContext context) throws IOException {
610     assert Thread.holdsLock(writer);
611     // must execute these two statements as atomic operation, otherwise we
612     // could lose updates if e.g. another thread calls writeFieldUpdates in
613     // between, or the updates are applied to the obtained reader, but then
614     // re-applied in IW.commitMergedDeletes (unnecessary work and potential
615     // bugs).
616     isMerging = true;
617     return getReader(context);
618   }
619   
620   /**
621    * Drops all merging updates. Called from IndexWriter after this segment
622    * finished merging (whether successfully or not).
623    */
624   public synchronized void dropMergingUpdates() {
625     mergingDVUpdates.clear();
626     isMerging = false;
627   }
628   
629   /** Returns updates that came in while this segment was merging. */
630   public synchronized Map<String,DocValuesFieldUpdates> getMergingFieldUpdates() {
631     return mergingDVUpdates;
632   }

634   @Override
635   public String toString() {
636     StringBuilder sb = new StringBuilder();
637     sb.append("ReadersAndLiveDocs(seg=").append(info);
638     sb.append(" pendingDeleteCount=").append(pendingDeleteCount);
639     sb.append(" liveDocsShared=").append(liveDocsShared);
640     return sb.toString();
641   }
