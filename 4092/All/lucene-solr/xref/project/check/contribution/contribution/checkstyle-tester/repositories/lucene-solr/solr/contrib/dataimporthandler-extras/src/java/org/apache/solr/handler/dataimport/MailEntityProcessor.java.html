65    public void init(Context context) {
66      super.init(context);
67      // set attributes using XXX getXXXFromContext(attribute, defualtValue);
68      // applies variable resolver and return default if value is not found or null
69      // REQUIRED : connection and folder info
70      user = getStringFromContext("user", null);
71      password = getStringFromContext("password", null);
72      host = getStringFromContext("host", null);
73      protocol = getStringFromContext("protocol", null);
74      folderNames = getStringFromContext("folders", null);
75      // validate
76      if (host == null || protocol == null || user == null || password == null
77          || folderNames == null) throw new DataImportHandlerException(
78          DataImportHandlerException.SEVERE,
79          "'user|password|protocol|host|folders' are required attributes");
80      
81      // OPTIONAL : have defaults and are optional
82      recurse = getBoolFromContext("recurse", true);
83      
84      exclude.clear();
85      String excludes = getStringFromContext("exclude", "");
86      if (excludes != null && !excludes.trim().equals("")) {
87        exclude = Arrays.asList(excludes.split(","));
88      }
89      
90      include.clear();
91      String includes = getStringFromContext("include", "");
92      if (includes != null && !includes.trim().equals("")) {
93        include = Arrays.asList(includes.split(","));
94      }
95      batchSize = getIntFromContext("batchSize", 20);
96      customFilter = getStringFromContext("customFilter", "");
97      if (filters != null) filters.clear();
98      folderIter = null;
99      msgIter = null;
100             
101     String lastIndexTime = null;
102     String command = 
103         String.valueOf(context.getRequestParameters().get("command"));
104     if (!DataImporter.FULL_IMPORT_CMD.equals(command))
105       throw new IllegalArgumentException(this.getClass().getSimpleName()+
106           " only supports "+DataImporter.FULL_IMPORT_CMD);
107     
108     // Read the last_index_time out of the dataimport.properties if available
109     String cname = getStringFromContext("name", "mailimporter");
110     String varName = ConfigNameConstants.IMPORTER_NS_SHORT + "." + cname + "."
111         + DocBuilder.LAST_INDEX_TIME;
112     Object varValue = context.getVariableResolver().resolve(varName);
113     LOG.info(varName+"="+varValue);    
114     
115     if (varValue != null && !"".equals(varValue) && 
116         !"".equals(getStringFromContext("fetchMailsSince", ""))) {
117 
118       // need to check if varValue is the epoch, which we'll take to mean the
119       // initial value, in which case means we should use fetchMailsSince instead
120       Date tmp = null;
121       try {
122         tmp = sinceDateParser.parse((String)varValue);
123         if (tmp.getTime() == 0) {
124           LOG.info("Ignoring initial value "+varValue+" for "+varName+
125               " in favor of fetchMailsSince config parameter");
126           tmp = null; // don't use this value
127         }
128       } catch (ParseException e) {
129         // probably ok to ignore this since we have other options below
130         // as we're just trying to figure out if the date is 0
131         LOG.warn("Failed to parse "+varValue+" from "+varName+" due to: "+e);
132       }    
133       
134       if (tmp == null) {
135         // favor fetchMailsSince in this case because the value from
136         // dataimport.properties is the default/init value
137         varValue = getStringFromContext("fetchMailsSince", "");
138         LOG.info("fetchMailsSince="+varValue);            
139       }
140     }
141     
142     if (varValue == null || "".equals(varValue)) {
143       varName = ConfigNameConstants.IMPORTER_NS_SHORT + "."
144           + DocBuilder.LAST_INDEX_TIME;
145       varValue = context.getVariableResolver().resolve(varName);
146       LOG.info(varName+"="+varValue);
147     }
148       
149     if (varValue != null && varValue instanceof String) {
150       lastIndexTime = (String)varValue;
151       if (lastIndexTime != null && lastIndexTime.length() == 0)
152         lastIndexTime = null;
153     }
154             
155     if (lastIndexTime == null) 
156       lastIndexTime = getStringFromContext("fetchMailsSince", "");
157 
158     LOG.info("Using lastIndexTime "+lastIndexTime+" for mail import");
159     
160     this.fetchMailsSince = null;
161     if (lastIndexTime != null && lastIndexTime.length() > 0) {
162       try {
163         fetchMailsSince = sinceDateParser.parse(lastIndexTime);
164         LOG.info("Parsed fetchMailsSince=" + lastIndexTime);
165       } catch (ParseException e) {
166         throw new DataImportHandlerException(DataImportHandlerException.SEVERE,
167             "Invalid value for fetchMailSince: " + lastIndexTime, e);
168       }
169     }
170         
171     fetchSize = getIntFromContext("fetchSize", 32 * 1024);
172     cTimeout = getIntFromContext("connectTimeout", 30 * 1000);
173     rTimeout = getIntFromContext("readTimeout", 60 * 1000);
174     
175     String tmp = context.getEntityAttribute("includeOtherUserFolders");
176     includeOtherUserFolders = (tmp != null && Boolean.valueOf(tmp.trim()));
177     tmp = context.getEntityAttribute("includeSharedFolders");
178     includeSharedFolders = (tmp != null && Boolean.valueOf(tmp.trim()));
179     
180     setProcessAttachmentConfig();
181     includeContent = getBoolFromContext("includeContent", true);
182           
183     logConfig();
184   }

186   private void setProcessAttachmentConfig() {
187     processAttachment = true;
188     String tbval = context.getEntityAttribute("processAttachments");
189     if (tbval == null) {
190       tbval = context.getEntityAttribute("processAttachement");
191       if (tbval != null) processAttachment = Boolean.valueOf(tbval);
192     } else processAttachment = Boolean.valueOf(tbval);
193   }

195   @Override
196   public Map<String,Object> nextRow() {
197     Message mail = null;
198     Map<String,Object> row = null;
199     do {
200       // try till there is a valid document or folders get exhausted.
201       // when mail == NULL, it means end of processing
202       mail = getNextMail();   
203       
204       if (mail != null)
205         row = getDocumentFromMail(mail);
206       
207       if (row != null && row.get("folder") == null) 
208         row.put("folder", mail.getFolder().getFullName());
209       
210     } while (row == null && mail != null);
211     return row;
212   }
213   
214   private Message getNextMail() {
215     if (!connected) {
216       if (!connectToMailBox()) return null;
217       connected = true;
218     }
219     if (folderIter == null) {
220       createFilters();
221       folderIter = new FolderIterator(mailbox);
222     }
223     // get next message from the folder
224     // if folder is exhausted get next folder
225     // loop till a valid mail or all folders exhausted.
226     while (msgIter == null || !msgIter.hasNext()) {
227       Folder next = folderIter.hasNext() ? folderIter.next() : null;
228       if (next == null) return null;
229       
230       msgIter = new MessageIterator(next, batchSize);
231     }
232     return msgIter.next();
233   }

235   private Map<String,Object> getDocumentFromMail(Message mail) {
236     Map<String,Object> row = new HashMap<>();
237     try {
238       addPartToDocument(mail, row, true);
239       return row;
240     } catch (Exception e) {
241       LOG.error("Failed to convert message [" + mail.toString()
242           + "] to document due to: " + e, e);
243       return null;
244     }
245   }

247   public void addPartToDocument(Part part, Map<String,Object> row, boolean outerMost) throws Exception {
248     if (part instanceof Message) {
249       addEnvelopeToDocument(part, row);
250     }
251     
252     String ct = part.getContentType().toLowerCase(Locale.ROOT);
253     ContentType ctype = new ContentType(ct);
254     if (part.isMimeType("multipart/*")) {
255       Object content = part.getContent();
256       if (content != null && content instanceof Multipart) {
257         Multipart mp = (Multipart) part.getContent();
258         int count = mp.getCount();
259         if (part.isMimeType("multipart/alternative")) count = 1;
260         for (int i = 0; i < count; i++)
261           addPartToDocument(mp.getBodyPart(i), row, false);
262       } else {
263         LOG.warn("Multipart content is a not an instance of Multipart! Content is: "
264             + (content != null ? content.getClass().getName() : "null")
265             + ". Typically, this is due to the Java Activation JAR being loaded by the wrong classloader.");
266       }
267     } else if (part.isMimeType("message/rfc822")) {
268       addPartToDocument((Part) part.getContent(), row, false);
269     } else {
270       String disp = part.getDisposition();
271       if (includeContent
272           && !(disp != null && disp.equalsIgnoreCase(Part.ATTACHMENT))) {
273         InputStream is = part.getInputStream();
274         Metadata contentTypeHint = new Metadata();
275         contentTypeHint.set(Metadata.CONTENT_TYPE, ctype.getBaseType()
276             .toLowerCase(Locale.ENGLISH));
277         String content = (new Tika()).parseToString(is, contentTypeHint);
278         if (row.get(CONTENT) == null) row.put(CONTENT, new ArrayList<String>());
279         List<String> contents = (List<String>) row.get(CONTENT);
280         contents.add(content.trim());
281         row.put(CONTENT, contents);
282       }
283       if (!processAttachment || disp == null
284           || !disp.equalsIgnoreCase(Part.ATTACHMENT)) return;
285       InputStream is = part.getInputStream();
286       String fileName = part.getFileName();
287       Metadata contentTypeHint = new Metadata();
288       contentTypeHint.set(Metadata.CONTENT_TYPE, ctype.getBaseType()
289           .toLowerCase(Locale.ENGLISH));
290       String content = (new Tika()).parseToString(is, contentTypeHint);
291       if (content == null || content.trim().length() == 0) return;
292       
293       if (row.get(ATTACHMENT) == null) row.put(ATTACHMENT,
294           new ArrayList<String>());
295       List<String> contents = (List<String>) row.get(ATTACHMENT);
296       contents.add(content.trim());
297       row.put(ATTACHMENT, contents);
298       if (row.get(ATTACHMENT_NAMES) == null) row.put(ATTACHMENT_NAMES,
299           new ArrayList<String>());
300       List<String> names = (List<String>) row.get(ATTACHMENT_NAMES);
301       names.add(fileName);
302       row.put(ATTACHMENT_NAMES, names);
303     }
304   }

306   private void addEnvelopeToDocument(Part part, Map<String,Object> row)
307       throws MessagingException {
308     MimeMessage mail = (MimeMessage) part;
309     Address[] adresses;
310     if ((adresses = mail.getFrom()) != null && adresses.length > 0) row.put(
311         FROM, adresses[0].toString());
312     
313     List<String> to = new ArrayList<>();
314     if ((adresses = mail.getRecipients(Message.RecipientType.TO)) != null) addAddressToList(
315         adresses, to);
316     if ((adresses = mail.getRecipients(Message.RecipientType.CC)) != null) addAddressToList(
317         adresses, to);
318     if ((adresses = mail.getRecipients(Message.RecipientType.BCC)) != null) addAddressToList(
319         adresses, to);
320     if (to.size() > 0) row.put(TO_CC_BCC, to);
321     
322     row.put(MESSAGE_ID, mail.getMessageID());
323     row.put(SUBJECT, mail.getSubject());
324     
325     Date d = mail.getSentDate();
326     if (d != null) {
327       row.put(SENT_DATE, d);
328     }
329     
330     List<String> flags = new ArrayList<>();
331     for (Flags.Flag flag : mail.getFlags().getSystemFlags()) {
332       if (flag == Flags.Flag.ANSWERED) flags.add(FLAG_ANSWERED);
333       else if (flag == Flags.Flag.DELETED) flags.add(FLAG_DELETED);
334       else if (flag == Flags.Flag.DRAFT) flags.add(FLAG_DRAFT);
335       else if (flag == Flags.Flag.FLAGGED) flags.add(FLAG_FLAGGED);
336       else if (flag == Flags.Flag.RECENT) flags.add(FLAG_RECENT);
337       else if (flag == Flags.Flag.SEEN) flags.add(FLAG_SEEN);
338     }
339     flags.addAll(Arrays.asList(mail.getFlags().getUserFlags()));
340     if (flags.size() == 0) flags.add(FLAG_NONE);
341     row.put(FLAGS, flags);
342     
343     String[] hdrs = mail.getHeader("X-Mailer");
344     if (hdrs != null) row.put(XMAILER, hdrs[0]);
345   }

347   private void addAddressToList(Address[] adresses, List<String> to)
348       throws AddressException {
349     for (Address address : adresses) {
350       to.add(address.toString());
351       InternetAddress ia = (InternetAddress) address;
352       if (ia.isGroup()) {
353         InternetAddress[] group = ia.getGroup(false);
354         for (InternetAddress member : group)
355           to.add(member.toString());
356       }
357     }
358   }

360   private boolean connectToMailBox() {
361     // this is needed to load the activation mail stuff correctly
362     // otherwise, the JavaMail multipart support doesn't get configured
363     // correctly, which leads to a class cast exception when processing
364     // multipart messages: IMAPInputStream cannot be cast to
365     // javax.mail.Multipart
366     Thread.currentThread().setContextClassLoader(getClass().getClassLoader());
367     
368     try {
369       Properties props = new Properties();
370       if (System.getProperty("mail.debug") != null) 
371         props.setProperty("mail.debug", System.getProperty("mail.debug"));
372       
373       if (("imap".equals(protocol) || "imaps".equals(protocol))
374           && "imap.gmail.com".equals(host)) {
375         LOG.info("Consider using 'gimaps' protocol instead of '" + protocol
376             + "' for enabling GMail specific extensions for " + host);
377       }
378       
379       props.setProperty("mail.store.protocol", protocol);
380       
381       String imapPropPrefix = protocol.startsWith("gimap") ? "gimap" : "imap";
382       props.setProperty("mail." + imapPropPrefix + ".fetchsize", "" + fetchSize);
383       props.setProperty("mail." + imapPropPrefix + ".timeout", "" + rTimeout);
384       props.setProperty("mail." + imapPropPrefix + ".connectiontimeout", "" + cTimeout);
385       
386       int port = -1;
387       int colonAt = host.indexOf(":");
388       if (colonAt != -1) {
389         port = Integer.parseInt(host.substring(colonAt + 1));
390         host = host.substring(0, colonAt);
391       }
392       
393       Session session = Session.getDefaultInstance(props, null);
394       mailbox = session.getStore(protocol);
395       if (port != -1) {
396         mailbox.connect(host, port, user, password);
397       } else {
398         mailbox.connect(host, user, password);
399       }
400       LOG.info("Connected to " + user + "'s mailbox on " + host);
401       
402       return true;
403     } catch (MessagingException e) {      
404       String errMsg = String.format(Locale.ENGLISH,
405           "Failed to connect to %s server %s as user %s due to: %s", protocol,
406           host, user, e.toString());
407       LOG.error(errMsg, e);
408       throw new DataImportHandlerException(DataImportHandlerException.SEVERE,
409           errMsg, e);
410     }
411   }

413   private void createFilters() {
414     if (fetchMailsSince != null) {
415       filters.add(new MailsSinceLastCheckFilter(fetchMailsSince));
416     }
417     if (customFilter != null && !customFilter.equals("")) {
418       try {
419         Class cf = Class.forName(customFilter);
420         Object obj = cf.newInstance();
421         if (obj instanceof CustomFilter) {
422           filters.add((CustomFilter) obj);
423         }
424       } catch (Exception e) {
425         throw new DataImportHandlerException(DataImportHandlerException.SEVERE,
426             "Custom filter could not be created", e);
427       }
428     }
429   }

478   class FolderIterator implements Iterator<Folder> {
479     private Store mailbox;
480     private List<String> topLevelFolders;
481     private List<Folder> folders = null;
482     private Folder lastFolder = null;
483     
484     public FolderIterator(Store mailBox) {
485       this.mailbox = mailBox;
486       folders = new ArrayList<>();
487       getTopLevelFolders(mailBox);
488       if (includeOtherUserFolders) getOtherUserFolders();
489       if (includeSharedFolders) getSharedFolders();
490     }
491     
492     public boolean hasNext() {
493       return !folders.isEmpty();
494     }
495     
496     public Folder next() {
497       try {
498         boolean hasMessages = false;
499         Folder next;
500         do {
501           if (lastFolder != null) {
502             lastFolder.close(false);
503             lastFolder = null;
504           }
505           if (folders.isEmpty()) {
506             mailbox.close();
507             return null;
508           }
509           next = folders.remove(0);
510           if (next != null) {
511             String fullName = next.getFullName();
512             if (!excludeFolder(fullName)) {
513               hasMessages = (next.getType() & Folder.HOLDS_MESSAGES) != 0;
514               next.open(Folder.READ_ONLY);
515               lastFolder = next;
516               LOG.info("Opened folder : " + fullName);
517             }
518             if (recurse && ((next.getType() & Folder.HOLDS_FOLDERS) != 0)) {
519               Folder[] children = next.list();
520               LOG.info("Added its children to list  : ");
521               for (int i = children.length - 1; i >= 0; i--) {
522                 folders.add(0, children[i]);
523                 LOG.info("child name : " + children[i].getFullName());
524               }
525               if (children.length == 0) LOG.info("NO children : ");
526             }
527           }
528         } while (!hasMessages);
529         return next;
530       } catch (Exception e) {
531         LOG.warn("Failed to read folders due to: "+e);
532         // throw new
533         // DataImportHandlerException(DataImportHandlerException.SEVERE,
534         // "Folder open failed", e);
535       }
536       return null;
537     }
538     
539     public void remove() {
540       throw new UnsupportedOperationException("It's read only mode...");
541     }
542     
543     private void getTopLevelFolders(Store mailBox) {
544       if (folderNames != null) topLevelFolders = Arrays.asList(folderNames
545           .split(","));
546       for (int i = 0; topLevelFolders != null && i < topLevelFolders.size(); i++) {
547         try {
548           folders.add(mailbox.getFolder(topLevelFolders.get(i)));
549         } catch (MessagingException e) {
550           // skip bad ones unless it's the last one and still no good folder
551           if (folders.size() == 0 && i == topLevelFolders.size() - 1) throw new DataImportHandlerException(
552               DataImportHandlerException.SEVERE, "Folder retreival failed");
553         }
554       }
555       if (topLevelFolders == null || topLevelFolders.size() == 0) {
556         try {
557           folders.add(mailBox.getDefaultFolder());
558         } catch (MessagingException e) {
559           throw new DataImportHandlerException(
560               DataImportHandlerException.SEVERE, "Folder retreival failed");
561         }
562       }
563     }
564     
565     private void getOtherUserFolders() {
566       try {
567         Folder[] ufldrs = mailbox.getUserNamespaces(null);
568         if (ufldrs != null) {
569           LOG.info("Found " + ufldrs.length + " user namespace folders");
570           for (Folder ufldr : ufldrs)
571             folders.add(ufldr);
572         }
573       } catch (MessagingException me) {
574         LOG.warn("Messaging exception retrieving user namespaces: "
575             + me.getMessage());
576       }
577     }
578     
579     private void getSharedFolders() {
580       try {
581         Folder[] sfldrs = mailbox.getSharedNamespaces();
582         if (sfldrs != null) {
583           LOG.info("Found " + sfldrs.length + " shared namespace folders");
584           for (Folder sfldr : sfldrs)
585             folders.add(sfldr);
586         }
587       } catch (MessagingException me) {
588         LOG.warn("Messaging exception retrieving shared namespaces: "
589             + me.getMessage());
590       }
591     }
592     
593     private boolean excludeFolder(String name) {
594       for (String s : exclude) {
595         if (name.matches(s)) return true;
596       }
597       for (String s : include) {
598         if (name.matches(s)) return false;
599       }
600       return include.size() > 0;
601     }
602   }

604   class MessageIterator extends SearchTerm implements Iterator<Message> {
605     private Folder folder;
606     private Message[] messagesInCurBatch = null;
607     private int current = 0;
608     private int currentBatch = 0;
609     private int batchSize = 0;
610     private int totalInFolder = 0;
611     private boolean doBatching = true;
612     
613     public MessageIterator(Folder folder, int batchSize) {
614       super();
615       
616       try {
617         this.folder = folder;
618         this.batchSize = batchSize;
619         SearchTerm st = getSearchTerm();
620         
621         LOG.info("SearchTerm=" + st);
622         
623         if (st != null || folder instanceof GmailFolder) {
624           doBatching = false;
625           // Searching can still take a while even though we're only pulling
626           // envelopes; unless you're using gmail server-side filter, which is
627           // fast
628           LOG.info("Searching folder " + folder.getName() + " for messages");
629           final RTimer searchTimer = new RTimer();
630 
631           // If using GMail, speed up the envelope processing by doing a
632           // server-side
633           // search for messages occurring on or after the fetch date (at
634           // midnight),
635           // which reduces the number of envelopes we need to pull from the
636           // server
637           // to apply the precise DateTerm filter; GMail server-side search has
638           // date
639           // granularity only but the local filters are also applied
640                     
641           if (folder instanceof GmailFolder && fetchMailsSince != null) {
642             String afterCrit = "after:" + afterFmt.format(fetchMailsSince);
643             LOG.info("Added server-side gmail filter: " + afterCrit);
644             Message[] afterMessages = folder.search(new GmailRawSearchTerm(
645                 afterCrit));
646             
647             LOG.info("GMail server-side filter found " + afterMessages.length
648                 + " messages received " + afterCrit + " in folder " + folder.getName());
649             
650             // now pass in the server-side filtered messages to the local filter
651             messagesInCurBatch = folder.search((st != null ? st : this), afterMessages);
652           } else {
653             messagesInCurBatch = folder.search(st);
654           }          
655           totalInFolder = messagesInCurBatch.length;
656           folder.fetch(messagesInCurBatch, fp);
657           current = 0;
658           LOG.info("Total messages : " + totalInFolder);
659           LOG.info("Search criteria applied. Batching disabled. Took {} (ms)", searchTimer.getTime());
660         } else {
661           totalInFolder = folder.getMessageCount();
662           LOG.info("Total messages : " + totalInFolder);
663           getNextBatch(batchSize, folder);
664         }
665       } catch (MessagingException e) {
666         throw new DataImportHandlerException(DataImportHandlerException.SEVERE,
667             "Message retreival failed", e);
668       }
669     }
670     
671     private void getNextBatch(int batchSize, Folder folder)
672         throws MessagingException {
673       // after each batch invalidate cache
674       if (messagesInCurBatch != null) {
675         for (Message m : messagesInCurBatch) {
676           if (m instanceof IMAPMessage) ((IMAPMessage) m).invalidateHeaders();
677         }
678       }
679       int lastMsg = (currentBatch + 1) * batchSize;
680       lastMsg = lastMsg > totalInFolder ? totalInFolder : lastMsg;
681       messagesInCurBatch = folder.getMessages(currentBatch * batchSize + 1,
682           lastMsg);
683       folder.fetch(messagesInCurBatch, fp);
684       current = 0;
685       currentBatch++;
686       LOG.info("Current Batch  : " + currentBatch);
687       LOG.info("Messages in this batch  : " + messagesInCurBatch.length);
688     }
689     
690     public boolean hasNext() {
691       boolean hasMore = current < messagesInCurBatch.length;
692       if (!hasMore && doBatching && currentBatch * batchSize < totalInFolder) {
693         // try next batch
694         try {
695           getNextBatch(batchSize, folder);
696           hasMore = current < messagesInCurBatch.length;
697         } catch (MessagingException e) {
698           throw new DataImportHandlerException(
699               DataImportHandlerException.SEVERE, "Message retreival failed", e);
700         }
701       }
702       return hasMore;
703     }
704     
705     public Message next() {
706       return hasNext() ? messagesInCurBatch[current++] : null;
707     }
708     
709     public void remove() {
710       throw new UnsupportedOperationException("It's read only mode...");
711     }

713     private SearchTerm getSearchTerm() {
714       if (filters.size() == 0) return null;
715       if (filters.size() == 1) return filters.get(0).getCustomSearch(folder);
716       SearchTerm last = filters.get(0).getCustomSearch(folder);
717       for (int i = 1; i < filters.size(); i++) {
718         CustomFilter filter = filters.get(i);
719         SearchTerm st = filter.getCustomSearch(folder);
720         if (st != null) {
721           last = new AndTerm(last, st);
722         }
723       }
724       return last;
725     }

728       return true;

732   class MailsSinceLastCheckFilter implements CustomFilter {
733     
734     private Date since;
735     
736     public MailsSinceLastCheckFilter(Date date) {
737       since = date;
738     }
739     
740     @SuppressWarnings("serial")
741     public SearchTerm getCustomSearch(final Folder folder) {
742       LOG.info("Building mail filter for messages in " + folder.getName()
743           + " that occur after " + sinceDateParser.format(since));
744       return new DateTerm(ComparisonTerm.GE, since) {
745         private int matched = 0;
746         private int seen = 0;
747         
748         @Override
749         public boolean match(Message msg) {
750           boolean isMatch = false;
751           ++seen;
752           try {
753             Date msgDate = msg.getReceivedDate();
754             if (msgDate == null) msgDate = msg.getSentDate();
755             
756             if (msgDate != null && msgDate.getTime() >= since.getTime()) {
757               ++matched;
758               isMatch = true;
759             } else {
760               String msgDateStr = (msgDate != null) ? sinceDateParser.format(msgDate) : "null";
761               String sinceDateStr = (since != null) ? sinceDateParser.format(since) : "null";
762               LOG.debug("Message " + msg.getSubject() + " was received at [" + msgDateStr
763                   + "], since filter is [" + sinceDateStr + "]");
764             }
765           } catch (MessagingException e) {
766             LOG.warn("Failed to process message due to: "+e, e);
767           }
768           
769           if (seen % 100 == 0) {
770             LOG.info("Matched " + matched + " of " + seen + " messages since: "
771                 + sinceDateParser.format(since));
772           }
773           
774           return isMatch;
775         }
776       };
777     }
