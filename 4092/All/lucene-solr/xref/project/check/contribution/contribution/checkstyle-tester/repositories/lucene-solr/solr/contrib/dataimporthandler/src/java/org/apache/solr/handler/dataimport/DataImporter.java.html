110   boolean maybeReloadConfiguration(RequestInfo params,
111       NamedList<?> defaultParams) throws IOException {
112   if (importLock.tryLock()) {
113       boolean success = false;
114       try {        
115         if (null != params.getRequest()) {
116           if (schema != params.getRequest().getSchema()) {
117             schema = params.getRequest().getSchema();
118           }
119         }
120         String dataConfigText = params.getDataConfig();
121         String dataconfigFile = params.getConfigFile();        
122         InputSource is = null;
123         if(dataConfigText!=null && dataConfigText.length()>0) {
124           is = new InputSource(new StringReader(dataConfigText));
125         } else if(dataconfigFile!=null) {
126           is = new InputSource(core.getResourceLoader().openResource(dataconfigFile));
127           is.setSystemId(SystemIdResolver.createSystemIdFromResourceName(dataconfigFile));
128           LOG.info("Loading DIH Configuration: " + dataconfigFile);
129         }
130         if(is!=null) {          
131           config = loadDataConfig(is);
132           success = true;
133         }      
134         
135         Map<String,Map<String,String>> dsProps = new HashMap<>();
136         if(defaultParams!=null) {
137           int position = 0;
138           while (position < defaultParams.size()) {
139             if (defaultParams.getName(position) == null) {
140               break;
141             }
142             String name = defaultParams.getName(position);            
143             if (name.equals("datasource")) {
144               success = true;
145               NamedList dsConfig = (NamedList) defaultParams.getVal(position);
146               LOG.info("Getting configuration for Global Datasource...");              
147               Map<String,String> props = new HashMap<>();
148               for (int i = 0; i < dsConfig.size(); i++) {
149                 props.put(dsConfig.getName(i), dsConfig.getVal(i).toString());
150               }
151               LOG.info("Adding properties to datasource: " + props);
152               dsProps.put((String) dsConfig.get("name"), props);
153             }
154             position++;
155           }
156         }
157         requestLevelDataSourceProps = Collections.unmodifiableMap(dsProps);
158       } catch(IOException ioe) {
159         throw ioe;
160       } finally {
161         importLock.unlock();
162       }
163       return success;
164     } else {
165       return false;
166     }
167   }

190   public DIHConfiguration loadDataConfig(InputSource configFile) {
191 
192     DIHConfiguration dihcfg = null;
193     try {
194       DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
195       
196       // only enable xinclude, if a a SolrCore and SystemId is present (makes no sense otherwise)
197       if (core != null && configFile.getSystemId() != null) {
198         try {
199           dbf.setXIncludeAware(true);
200           dbf.setNamespaceAware(true);
201         } catch( UnsupportedOperationException e ) {
202           LOG.warn( "XML parser doesn't support XInclude option" );
203         }
204       }
205       
206       DocumentBuilder builder = dbf.newDocumentBuilder();
207       if (core != null)
208         builder.setEntityResolver(new SystemIdResolver(core.getResourceLoader()));
209       builder.setErrorHandler(XMLLOG);
210       Document document;
211       try {
212         document = builder.parse(configFile);
213       } finally {
214         // some XML parsers are broken and don't close the byte stream (but they should according to spec)
215         IOUtils.closeQuietly(configFile.getByteStream());
216       }
217 
218       dihcfg = readFromXml(document);
219       LOG.info("Data Configuration loaded successfully");
220     } catch (Exception e) {
221       throw new DataImportHandlerException(SEVERE,
222               "Data Config problem: " + e.getMessage(), e);
223     }
224     for (Entity e : dihcfg.getEntities()) {
225       if (e.getAllAttributes().containsKey(SqlEntityProcessor.DELTA_QUERY)) {
226         isDeltaImportSupported = true;
227         break;
228       }
229     }
230     return dihcfg;
231   }

233   public DIHConfiguration readFromXml(Document xmlDocument) {
234     DIHConfiguration config;
235     List<Map<String, String >> functions = new ArrayList<>();
236     Script script = null;
237     Map<String, Map<String,String>> dataSources = new HashMap<>();
238     
239     NodeList dataConfigTags = xmlDocument.getElementsByTagName("dataConfig");
240     if(dataConfigTags == null || dataConfigTags.getLength() == 0) {
241       throw new DataImportHandlerException(SEVERE, "the root node '<dataConfig>' is missing");
242     }
243     Element e = (Element) dataConfigTags.item(0);
244     List<Element> documentTags = ConfigParseUtil.getChildNodes(e, "document");
245     if (documentTags.isEmpty()) {
246       throw new DataImportHandlerException(SEVERE, "DataImportHandler " +
247               "configuration file must have one <document> node.");
248     }
249 
250     List<Element> scriptTags = ConfigParseUtil.getChildNodes(e, ConfigNameConstants.SCRIPT);
251     if (!scriptTags.isEmpty()) {
252       script = new Script(scriptTags.get(0));
253     }
254 
255     // Add the provided evaluators
256     List<Element> functionTags = ConfigParseUtil.getChildNodes(e, ConfigNameConstants.FUNCTION);
257     if (!functionTags.isEmpty()) {
258       for (Element element : functionTags) {
259         String func = ConfigParseUtil.getStringAttribute(element, NAME, null);
260         String clz = ConfigParseUtil.getStringAttribute(element, ConfigNameConstants.CLASS, null);
261         if (func == null || clz == null){
262           throw new DataImportHandlerException(
263                   SEVERE,
264                   "<function> must have a 'name' and 'class' attributes");
265         } else {
266           functions.add(ConfigParseUtil.getAllAttributes(element));
267         }
268       }
269     }
270     List<Element> dataSourceTags = ConfigParseUtil.getChildNodes(e, ConfigNameConstants.DATA_SRC);
271     if (!dataSourceTags.isEmpty()) {
272       for (Element element : dataSourceTags) {
273         Map<String,String> p = new HashMap<>();
274         HashMap<String, String> attrs = ConfigParseUtil.getAllAttributes(element);
275         for (Map.Entry<String, String> entry : attrs.entrySet()) {
276           p.put(entry.getKey(), entry.getValue());
277         }
278         dataSources.put(p.get("name"), p);
279       }
280     }
281     if(dataSources.get(null) == null){
282       for (Map<String,String> properties : dataSources.values()) {
283         dataSources.put(null,properties);
284         break;        
285       } 
286     }
287     PropertyWriter pw = null;
288     List<Element> propertyWriterTags = ConfigParseUtil.getChildNodes(e, ConfigNameConstants.PROPERTY_WRITER);
289     if (propertyWriterTags.isEmpty()) {
290       boolean zookeeper = false;
291       if (this.core != null
292           && this.core.getCoreDescriptor().getCoreContainer()
293               .isZooKeeperAware()) {
294         zookeeper = true;
295       }
296       pw = new PropertyWriter(zookeeper ? "ZKPropertiesWriter"
297           : "SimplePropertiesWriter", Collections.<String,String> emptyMap());
298     } else if (propertyWriterTags.size() > 1) {
299       throw new DataImportHandlerException(SEVERE, "Only one "
300           + ConfigNameConstants.PROPERTY_WRITER + " can be configured.");
301     } else {
302       Element pwElement = propertyWriterTags.get(0);
303       String type = null;
304       Map<String,String> params = new HashMap<>();
305       for (Map.Entry<String,String> entry : ConfigParseUtil.getAllAttributes(
306           pwElement).entrySet()) {
307         if (TYPE.equals(entry.getKey())) {
308           type = entry.getValue();
309         } else {
310           params.put(entry.getKey(), entry.getValue());
311         }
312       }
313       if (type == null) {
314         throw new DataImportHandlerException(SEVERE, "The "
315             + ConfigNameConstants.PROPERTY_WRITER + " element must specify "
316             + TYPE);
317       }
318       pw = new PropertyWriter(type, params);
319     }
320     return new DIHConfiguration(documentTags.get(0), this, functions, script, dataSources, pw);
321   }

323   @SuppressWarnings("unchecked")
324   private DIHProperties createPropertyWriter() {
325     DIHProperties propWriter = null;
326     PropertyWriter configPw = config.getPropertyWriter();
327     try {
328       Class<DIHProperties> writerClass = DocBuilder.loadClass(configPw.getType(), this.core);
329       propWriter = writerClass.newInstance();
330       propWriter.init(this, configPw.getParameters());
331     } catch (Exception e) {
332       throw new DataImportHandlerException(DataImportHandlerException.SEVERE, "Unable to PropertyWriter implementation:" + configPw.getType(), e);
333     }
334     return propWriter;
335   }

357   public DataSource getDataSourceInstance(Entity key, String name, Context ctx) {
358     Map<String,String> p = requestLevelDataSourceProps.get(name);
359     if (p == null)
360       p = config.getDataSources().get(name);
361     if (p == null)
362       p = requestLevelDataSourceProps.get(null);// for default data source
363     if (p == null)
364       p = config.getDataSources().get(null);
365     if (p == null)  
366       throw new DataImportHandlerException(SEVERE,
367               "No dataSource :" + name + " available for entity :" + key.getName());
368     String type = p.get(TYPE);
369     DataSource dataSrc = null;
370     if (type == null) {
371       dataSrc = new JdbcDataSource();
372     } else {
373       try {
374         dataSrc = (DataSource) DocBuilder.loadClass(type, getCore()).newInstance();
375       } catch (Exception e) {
376         wrapAndThrow(SEVERE, e, "Invalid type for data source: " + type);
377       }
378     }
379     try {
380       Properties copyProps = new Properties();
381       copyProps.putAll(p);
382       Map<String, Object> map = ctx.getRequestParameters();
383       if (map.containsKey("rows")) {
384         int rows = Integer.parseInt((String) map.get("rows"));
385         if (map.containsKey("start")) {
386           rows += Integer.parseInt((String) map.get("start"));
387         }
388         copyProps.setProperty("maxRows", String.valueOf(rows));
389       }
390       dataSrc.init(ctx, copyProps);
391     } catch (Exception e) {
392       wrapAndThrow(SEVERE, e, "Failed to initialize DataSource: " + key.getDataSourceName());
393     }
394     return dataSrc;
395   }

409   public void doFullImport(DIHWriter writer, RequestInfo requestParams) {
410     LOG.info("Starting Full Import");
411     setStatus(Status.RUNNING_FULL_DUMP);
412     try {
413       DIHProperties dihPropWriter = createPropertyWriter();
414       setIndexStartTime(dihPropWriter.getCurrentTimestamp());
415       docBuilder = new DocBuilder(this, writer, dihPropWriter, requestParams);
416       checkWritablePersistFile(writer, dihPropWriter);
417       docBuilder.execute();
418       if (!requestParams.isDebug())
419         cumulativeStatistics.add(docBuilder.importStatistics);
420     } catch (Exception e) {
421       SolrException.log(LOG, "Full Import failed", e);
422       docBuilder.handleError("Full Import failed", e);
423     } finally {
424       setStatus(Status.IDLE);
425       DocBuilder.INSTANCE.set(null);
426     }
427 
428   }

430   private void checkWritablePersistFile(DIHWriter writer, DIHProperties dihPropWriter) {
431    if (isDeltaImportSupported && !dihPropWriter.isWritable()) {
432       throw new DataImportHandlerException(SEVERE,
433           "Properties is not writable. Delta imports are supported by data config but will not work.");
434     }
435   }

437   public void doDeltaImport(DIHWriter writer, RequestInfo requestParams) {
438     LOG.info("Starting Delta Import");
439     setStatus(Status.RUNNING_DELTA_DUMP);
440     try {
441       DIHProperties dihPropWriter = createPropertyWriter();
442       setIndexStartTime(dihPropWriter.getCurrentTimestamp());
443       docBuilder = new DocBuilder(this, writer, dihPropWriter, requestParams);
444       checkWritablePersistFile(writer, dihPropWriter);
445       docBuilder.execute();
446       if (!requestParams.isDebug())
447         cumulativeStatistics.add(docBuilder.importStatistics);
448     } catch (Exception e) {
449       LOG.error("Delta Import Failed", e);
450       docBuilder.handleError("Delta Import Failed", e);
451     } finally {
452       setStatus(Status.IDLE);
453       DocBuilder.INSTANCE.set(null);
454     }

458   public void runAsync(final RequestInfo reqParams, final DIHWriter sw) {
459     new Thread() {
460       @Override
461       public void run() {
462         runCmd(reqParams, sw);
463       }
464     }.start();
465   }
466 
467   void runCmd(RequestInfo reqParams, DIHWriter sw) {
468     String command = reqParams.getCommand();
469     if (command.equals(ABORT_CMD)) {
470       if (docBuilder != null) {
471         docBuilder.abort();
472       }
473       return;
474     }
475     if (!importLock.tryLock()){
476       LOG.warn("Import command failed . another import is running");      
477       return;
478     }
479     try {
480       if (FULL_IMPORT_CMD.equals(command) || IMPORT_CMD.equals(command)) {
481         doFullImport(sw, reqParams);
482       } else if (command.equals(DELTA_IMPORT_CMD)) {
483         doDeltaImport(sw, reqParams);
484       }
485     } finally {
486       importLock.unlock();
487     }
488   }

490   @SuppressWarnings("unchecked")
491   Map<String, String> getStatusMessages() {
492     //this map object is a Collections.synchronizedMap(new LinkedHashMap()). if we
493     // synchronize on the object it must be safe to iterate through the map
494     Map statusMessages = (Map) retrieve(STATUS_MSGS);
495     Map<String, String> result = new LinkedHashMap<>();
496     if (statusMessages != null) {
497       synchronized (statusMessages) {
498         for (Object o : statusMessages.entrySet()) {
499           Map.Entry e = (Map.Entry) o;
500           //the toString is taken because some of the Objects create the data lazily when toString() is called
501           result.put((String) e.getKey(), e.getValue().toString());
502         }
503       }
504     }
505     return result;
506 
507   }

522   /**
523    * used by tests.
524    */
525   Map<String, Evaluator> getEvaluators(List<Map<String,String>> fn) {
526     Map<String, Evaluator> evaluators = new HashMap<>();
527     evaluators.put(Evaluator.DATE_FORMAT_EVALUATOR, new DateFormatEvaluator());
528     evaluators.put(Evaluator.SQL_ESCAPE_EVALUATOR, new SqlEscapingEvaluator());
529     evaluators.put(Evaluator.URL_ENCODE_EVALUATOR, new UrlEvaluator());
530     evaluators.put(Evaluator.ESCAPE_SOLR_QUERY_CHARS, new SolrQueryEscapingEvaluator());
531     SolrCore core = docBuilder == null ? null : docBuilder.dataImporter.getCore();
532     for (Map<String, String> map : fn) {
533       try {
534         evaluators.put(map.get(NAME), (Evaluator) loadClass(map.get(CLASS), core).newInstance());
535       } catch (Exception e) {
536         wrapAndThrow(SEVERE, e, "Unable to instantiate evaluator: " + map.get(CLASS));
537       }
538     }
539     return evaluators;    
540   }
