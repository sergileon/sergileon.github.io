60    private static class FieldAndShardVersion {
61      private final long version;
62      private final int nodeID;
63      private final String field;
64  
65      public FieldAndShardVersion(int nodeID, long version, String field) {
66        this.nodeID = nodeID;
67        this.version = version;
68        this.field = field;
69      }
70  
71      @Override
72      public int hashCode() {
73        return (int) (version * nodeID + field.hashCode());
74      }
75  
76      @Override
77      public boolean equals(Object _other) {
78        if (!(_other instanceof FieldAndShardVersion)) {
79          return false;
80        }
81  
82        final FieldAndShardVersion other = (FieldAndShardVersion) _other;
83  
84        return field.equals(other.field) && version == other.version && nodeID == other.nodeID;
85      }
86  
87      @Override
88      public String toString() {
89        return "FieldAndShardVersion(field=" + field + " nodeID=" + nodeID + " version=" + version+ ")";
90      }

139       final CollectionStatistics stats = newSearcher.collectionStatistics(field);

141         // Don't put my own collection stats into the cache;
142         // we pull locally:
143         if (node.myNodeID != nodeID) {
144           node.collectionStatsCache.put(new FieldAndShardVersion(nodeID, version, field), stats);

146       }

168       } else {
169         assert searchAfter == null;  // not supported yet
170         return s.localSearch(q, numHits, sort);

172     } finally {
173       nodes[nodeID].release(s);
174     }

178   // term stats from remote node
179   Map<Term,TermStatistics> getNodeTermStats(Set<Term> terms, int nodeID, long version) throws IOException {
180     final NodeState node = nodes[nodeID];
181     final Map<Term,TermStatistics> stats = new HashMap<>();
182     final IndexSearcher s = node.searchers.acquire(version);
183     if (s == null) {
184       throw new SearcherExpiredException("node=" + nodeID + " version=" + version);
185     }
186     try {
187       for(Term term : terms) {
188         final TermContext termContext = TermContext.build(s.getIndexReader().getContext(), term);
189         stats.put(term, s.termStatistics(term, termContext));
190       }
191     } finally {
192       node.searchers.release(s);
193     }
194     return stats;
195   }

217     public class ShardIndexSearcher extends IndexSearcher {
218       // Version for the node searchers we search:
219       public final long[] nodeVersions;
220       public final int myNodeID;
221 
222       public ShardIndexSearcher(long[] nodeVersions, IndexReader localReader, int nodeID) {
223         super(localReader);
224         this.nodeVersions = nodeVersions;
225         myNodeID = nodeID;
226         assert myNodeID == NodeState.this.myNodeID: "myNodeID=" + nodeID + " NodeState.this.myNodeID=" + NodeState.this.myNodeID;
227       }
228 
229       @Override
230       public Query rewrite(Query original) throws IOException {
231         final IndexSearcher localSearcher = new IndexSearcher(getIndexReader());
232         final Weight weight = localSearcher.createNormalizedWeight(original, true);
233         final Set<Term> terms = new HashSet<>();
234         weight.extractTerms(terms);
235 
236         // Make a single request to remote nodes for term
237         // stats:
238         for(int nodeID=0;nodeID<nodeVersions.length;nodeID++) {
239           if (nodeID == myNodeID) {
240             continue;
241           }
242 
243           final Set<Term> missing = new HashSet<>();
244           for(Term term : terms) {
245             final TermAndShardVersion key = new TermAndShardVersion(nodeID, nodeVersions[nodeID], term);
246             if (!termStatsCache.containsKey(key)) {
247               missing.add(term);
248             }
249           }
250           if (missing.size() != 0) {
251             for(Map.Entry<Term,TermStatistics> ent : getNodeTermStats(missing, nodeID, nodeVersions[nodeID]).entrySet()) {
252               final TermAndShardVersion key = new TermAndShardVersion(nodeID, nodeVersions[nodeID], ent.getKey());
253               termStatsCache.put(key, ent.getValue());
254             }
255           }
256         }
257 
258         return weight.getQuery();
259       }
260 
261       @Override
262       public TermStatistics termStatistics(Term term, TermContext context) throws IOException {
263         assert term != null;
264         long docFreq = 0;
265         long totalTermFreq = 0;
266         for(int nodeID=0;nodeID<nodeVersions.length;nodeID++) {
267 
268           final TermStatistics subStats;
269           if (nodeID == myNodeID) {
270             subStats = super.termStatistics(term, context);
271           } else {
272             final TermAndShardVersion key = new TermAndShardVersion(nodeID, nodeVersions[nodeID], term);
273             subStats = termStatsCache.get(key);
274             // We pre-cached during rewrite so all terms
275             // better be here...
276             assert subStats != null;
277           }
278         
279           long nodeDocFreq = subStats.docFreq();
280           if (docFreq >= 0 && nodeDocFreq >= 0) {
281             docFreq += nodeDocFreq;
282           } else {
283             docFreq = -1;
284           }
285           
286           long nodeTotalTermFreq = subStats.totalTermFreq();
287           if (totalTermFreq >= 0 && nodeTotalTermFreq >= 0) {
288             totalTermFreq += nodeTotalTermFreq;
289           } else {
290             totalTermFreq = -1;
291           }
292         }
293 
294         return new TermStatistics(term.bytes(), docFreq, totalTermFreq);
295       }
296 
297       @Override
298       public CollectionStatistics collectionStatistics(String field) throws IOException {
299         // TODO: we could compute this on init and cache,
300         // since we are re-inited whenever any nodes have a
301         // new reader
302         long docCount = 0;
303         long sumTotalTermFreq = 0;
304         long sumDocFreq = 0;
305         long maxDoc = 0;
306 
307         for(int nodeID=0;nodeID<nodeVersions.length;nodeID++) {
308           final FieldAndShardVersion key = new FieldAndShardVersion(nodeID, nodeVersions[nodeID], field);
309           final CollectionStatistics nodeStats;
310           if (nodeID == myNodeID) {
311             nodeStats = super.collectionStatistics(field);
312           } else {
313             nodeStats = collectionStatsCache.get(key);
314           }
315           if (nodeStats == null) {
316             System.out.println("coll stats myNodeID=" + myNodeID + ": " + collectionStatsCache.keySet());
317           }
318           // Collection stats are pre-shared on reopen, so,
319           // we better not have a cache miss:
320           assert nodeStats != null: "myNodeID=" + myNodeID + " nodeID=" + nodeID + " version=" + nodeVersions[nodeID] + " field=" + field;
321           
322           long nodeDocCount = nodeStats.docCount();
323           if (docCount >= 0 && nodeDocCount >= 0) {
324             docCount += nodeDocCount;
325           } else {
326             docCount = -1;
327           }
328           
329           long nodeSumTotalTermFreq = nodeStats.sumTotalTermFreq();
330           if (sumTotalTermFreq >= 0 && nodeSumTotalTermFreq >= 0) {
331             sumTotalTermFreq += nodeSumTotalTermFreq;
332           } else {
333             sumTotalTermFreq = -1;
334           }
335           
336           long nodeSumDocFreq = nodeStats.sumDocFreq();
337           if (sumDocFreq >= 0 && nodeSumDocFreq >= 0) {
338             sumDocFreq += nodeSumDocFreq;
339           } else {
340             sumDocFreq = -1;
341           }
342           
343           assert nodeStats.maxDoc() >= 0;
344           maxDoc += nodeStats.maxDoc();
345         }
346 
347         return new CollectionStatistics(field, maxDoc, docCount, sumTotalTermFreq, sumDocFreq);
348       }

350       @Override
351       public TopDocs search(Query query, int numHits) throws IOException {
352         final TopDocs[] shardHits = new TopDocs[nodeVersions.length];
353         for(int nodeID=0;nodeID<nodeVersions.length;nodeID++) {
354           if (nodeID == myNodeID) {
355             // My node; run using local shard searcher we
356             // already aquired:
357             shardHits[nodeID] = localSearch(query, numHits);
358           } else {
359             shardHits[nodeID] = searchNode(nodeID, nodeVersions, query, null, numHits, null);
360           }
361         }
362 
363         // Merge:
364         return TopDocs.merge(numHits, shardHits);
365       }

367       public TopDocs localSearch(Query query, int numHits) throws IOException {
368         return super.search(query, numHits);
369       }

371       @Override
372       public TopDocs searchAfter(ScoreDoc after, Query query, int numHits) throws IOException {
373         if (after == null) {
374           return super.searchAfter(after, query, numHits);
375         }
376         final TopDocs[] shardHits = new TopDocs[nodeVersions.length];
377         // results are merged in that order: score, shardIndex, doc. therefore we set
378         // after to after.score and depending on the nodeID we set doc to either:
379         // - not collect any more documents with that score (only with worse score)
380         // - collect more documents with that score (and worse) following the last collected document
381         // - collect all documents with that score (and worse)
382         ScoreDoc shardAfter = new ScoreDoc(after.doc, after.score);
383         for (int nodeID = 0; nodeID < nodeVersions.length; nodeID++) {
384           if (nodeID < after.shardIndex) {
385             // all documents with after.score were already collected, so collect
386             // only documents with worse scores.
387             final NodeState.ShardIndexSearcher s = nodes[nodeID].acquire(nodeVersions);
388             try {
389               // Setting after.doc to reader.maxDoc-1 is a way to tell
390               // TopScoreDocCollector that no more docs with that score should
391               // be collected. note that in practice the shard which sends the
392               // request to a remote shard won't have reader.maxDoc at hand, so
393               // it will send some arbitrary value which will be fixed on the
394               // other end.
395               shardAfter.doc = s.getIndexReader().maxDoc() - 1;
396             } finally {
397               nodes[nodeID].release(s);
398             }
399           } else if (nodeID == after.shardIndex) {
400             // collect all documents following the last collected doc with
401             // after.score + documents with worse scores.  
402             shardAfter.doc = after.doc;
403           } else {
404             // all documents with after.score (and worse) should be collected
405             // because they didn't make it to top-N in the previous round.
406             shardAfter.doc = -1;
407           }
408           if (nodeID == myNodeID) {
409             // My node; run using local shard searcher we
410             // already aquired:
411             shardHits[nodeID] = localSearchAfter(shardAfter, query, numHits);
412           } else {
413             shardHits[nodeID] = searchNode(nodeID, nodeVersions, query, null, numHits, shardAfter);
414           }
415           //System.out.println("  node=" + nodeID + " totHits=" + shardHits[nodeID].totalHits);
416         }
417 
418         // Merge:
419         return TopDocs.merge(numHits, shardHits);
420       }
421 
422       public TopDocs localSearchAfter(ScoreDoc after, Query query, int numHits) throws IOException {
423         return super.searchAfter(after, query, numHits);
424       }

426       @Override
427       public TopFieldDocs search(Query query, int numHits, Sort sort) throws IOException {
428         assert sort != null;
429         final TopFieldDocs[] shardHits = new TopFieldDocs[nodeVersions.length];
430         for(int nodeID=0;nodeID<nodeVersions.length;nodeID++) {
431           if (nodeID == myNodeID) {
432             // My node; run using local shard searcher we
433             // already aquired:
434             shardHits[nodeID] = localSearch(query, numHits, sort);
435           } else {
436             shardHits[nodeID] = (TopFieldDocs) searchNode(nodeID, nodeVersions, query, sort, numHits, null);
437           }
438         }
439 
440         // Merge:
441         return TopDocs.merge(sort, numHits, shardHits);
442       }

452     public NodeState(Random random, int nodeID, int numNodes) throws IOException {
453       myNodeID = nodeID;
454       dir = newFSDirectory(createTempDir("ShardSearchingTestBase"));
455       // TODO: set warmer
456       MockAnalyzer analyzer = new MockAnalyzer(random());
457       analyzer.setMaxTokenLength(TestUtil.nextInt(random(), 1, IndexWriter.MAX_TERM_LENGTH));
458       IndexWriterConfig iwc = new IndexWriterConfig(analyzer);
459       iwc.setOpenMode(IndexWriterConfig.OpenMode.CREATE);
460       if (VERBOSE) {
461         iwc.setInfoStream(new PrintStreamInfoStream(System.out));
462       }
463       writer = new IndexWriter(dir, iwc);
464       mgr = new SearcherManager(writer, true, null);
465       searchers = new SearcherLifetimeManager();
466 
467       // Init w/ 0s... caller above will do initial
468       // "broadcast" by calling initSearcher:
469       currentNodeVersions = new long[numNodes];
470     }

472     public void initSearcher(long[] nodeVersions) throws IOException {
473       assert currentShardSearcher == null;
474       System.arraycopy(nodeVersions, 0, currentNodeVersions, 0, currentNodeVersions.length);
475       currentShardSearcher = new ShardIndexSearcher(currentNodeVersions.clone(),
476                                                     mgr.acquire().getIndexReader(),
477                                                     myNodeID);
478     }
479 
480     public void updateNodeVersion(int nodeID, long version) throws IOException {
481       currentNodeVersions[nodeID] = version;
482       if (currentShardSearcher != null) {
483         currentShardSearcher.getIndexReader().decRef();
484       }
485       currentShardSearcher = new ShardIndexSearcher(currentNodeVersions.clone(), 
486                                                     mgr.acquire().getIndexReader(),
487                                                     myNodeID);
488     }
489 
490     // Get the current (fresh) searcher for this node
491     public ShardIndexSearcher acquire() {
492       while(true) {
493         final ShardIndexSearcher s = currentShardSearcher;
494         // In theory the reader could get decRef'd to 0
495         // before we have a chance to incRef, ie if a reopen
496         // happens right after the above line, this thread
497         // gets stalled, and the old IR is closed.  So we
498         // must try/retry until incRef succeeds:
499         if (s.getIndexReader().tryIncRef()) {
500           return s;
501         }
502       }
503     }
504 
505     public void release(ShardIndexSearcher s) throws IOException {
506       s.getIndexReader().decRef();
507     }
508 
509     // Get and old searcher matching the specified versions:
510     public ShardIndexSearcher acquire(long[] nodeVersions) {
511       final IndexSearcher s = searchers.acquire(nodeVersions[myNodeID]);
512       if (s == null) {
513         throw new SearcherExpiredException("nodeID=" + myNodeID + " version=" + nodeVersions[myNodeID]);
514       }
515       return new ShardIndexSearcher(nodeVersions, s.getIndexReader(), myNodeID);
516     }
517 
518     // Reopen local reader
519     public void reopen() throws IOException {
520       final IndexSearcher before = mgr.acquire();
521       mgr.release(before);
522 
523       mgr.maybeRefresh();
524       final IndexSearcher after = mgr.acquire();
525       try {
526         if (after != before) {
527           // New searcher was opened
528           final long version = searchers.record(after);
529           searchers.prune(new SearcherLifetimeManager.PruneByAge(maxSearcherAgeSeconds));
530           broadcastNodeReopen(myNodeID, version, after);
531         }
532       } finally {
533         mgr.release(after);
534       }
535     }
536 
537     @Override
538     public void close() throws IOException {
539       if (currentShardSearcher != null) {
540         currentShardSearcher.getIndexReader().decRef();
541       }
542       searchers.close();
543       mgr.close();
544       writer.close();
545       dir.close();
546     }
547   }

549   // TODO: make this more realistic, ie, each node should
550   // have its own thread, so we have true node to node

552   private final class ChangeIndices extends Thread {
553     @Override
554     public void run() {
555       try {
556         final LineFileDocs docs = new LineFileDocs(random(), true);
557         int numDocs = 0;
558         while (System.nanoTime() < endTimeNanos) {
559           final int what = random().nextInt(3);
560           final NodeState node = nodes[random().nextInt(nodes.length)];
561           if (numDocs == 0 || what == 0) {
562             node.writer.addDocument(docs.nextDoc());
563             numDocs++;
564           } else if (what == 1) {
565             node.writer.updateDocument(new Term("docid", ""+random().nextInt(numDocs)),
566                                         docs.nextDoc());
567             numDocs++;
568           } else {
569             node.writer.deleteDocuments(new Term("docid", ""+random().nextInt(numDocs)));
570           }
571           // TODO: doc blocks too
572 
573           if (random().nextInt(17) == 12) {
574             node.writer.commit();
575           }
576 
577           if (random().nextInt(17) == 12) {
578             nodes[random().nextInt(nodes.length)].reopen();
579           }
580         }
581       } catch (Throwable t) {
582         System.out.println("FAILED:");
583         t.printStackTrace(System.out);
584         throw new RuntimeException(t);
585       }
586     }

589   protected NodeState[] nodes;
590   int maxSearcherAgeSeconds;
591   long endTimeNanos;
592   private Thread changeIndicesThread;
593 
594   protected void start(int numNodes, double runTimeSec, int maxSearcherAgeSeconds) throws IOException {
595 
596     endTimeNanos = System.nanoTime() + (long) (runTimeSec*1000000000);
597     this.maxSearcherAgeSeconds = maxSearcherAgeSeconds;
598 
599     nodes = new NodeState[numNodes];
600     for(int nodeID=0;nodeID<numNodes;nodeID++) {
601       nodes[nodeID] = new NodeState(random(), nodeID, numNodes);
602     }
603 
604     long[] nodeVersions = new long[nodes.length];
605     for(int nodeID=0;nodeID<numNodes;nodeID++) {
606       final IndexSearcher s = nodes[nodeID].mgr.acquire();
607       try {
608         nodeVersions[nodeID] = nodes[nodeID].searchers.record(s);
609       } finally {
610         nodes[nodeID].mgr.release(s);
611       }
612     }
613 
614     for(int nodeID=0;nodeID<numNodes;nodeID++) {
615       final IndexSearcher s = nodes[nodeID].mgr.acquire();
616       assert nodeVersions[nodeID] == nodes[nodeID].searchers.record(s);
617       assert s != null;
618       try {
619         broadcastNodeReopen(nodeID, nodeVersions[nodeID], s);
620       } finally {
621         nodes[nodeID].mgr.release(s);
622       }
623     }
624 
625     changeIndicesThread = new ChangeIndices();
626     changeIndicesThread.start();
627   }
628 
629   protected void finish() throws InterruptedException, IOException {
630     changeIndicesThread.join();
631     for(NodeState node : nodes) {
632       node.close();
633     }
