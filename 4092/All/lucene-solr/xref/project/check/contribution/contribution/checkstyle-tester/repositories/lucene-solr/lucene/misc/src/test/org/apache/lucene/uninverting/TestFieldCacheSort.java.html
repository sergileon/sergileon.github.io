67    /** Tests sorting on type string */
68    private void testString(SortField.Type sortType) throws IOException {
69      Directory dir = newDirectory();
70      RandomIndexWriter writer = new RandomIndexWriter(random(), dir);
71      Document doc = new Document();
72      doc.add(newStringField("value", "foo", Field.Store.YES));
73      writer.addDocument(doc);
74      doc = new Document();
75      doc.add(newStringField("value", "bar", Field.Store.YES));
76      writer.addDocument(doc);
77      Type type = sortType == SortField.Type.STRING ? Type.SORTED : Type.BINARY;
78      IndexReader ir = UninvertingReader.wrap(writer.getReader(), 
79                       Collections.singletonMap("value", type));
80      writer.close();
81      
82      IndexSearcher searcher = newSearcher(ir);
83      Sort sort = new Sort(new SortField("value", sortType));
84  
85      TopDocs td = searcher.search(new MatchAllDocsQuery(), 10, sort);
86      assertEquals(2, td.totalHits);
87      // 'bar' comes before 'foo'
88      assertEquals("bar", searcher.doc(td.scoreDocs[0].doc).get("value"));
89      assertEquals("foo", searcher.doc(td.scoreDocs[1].doc).get("value"));
90  
91      TestUtil.checkReader(ir);
92      ir.close();
93      dir.close();
94    }

389   /** Tests default sort (by score) */
390   public void testFieldScore() throws Exception {
391     Directory dir = newDirectory();
392     RandomIndexWriter writer = new RandomIndexWriter(random(), dir);
393     Document doc = new Document();
394     doc.add(newTextField("value", "foo bar bar bar bar", Field.Store.NO));
395     writer.addDocument(doc);
396     doc = new Document();
397     doc.add(newTextField("value", "foo foo foo foo foo", Field.Store.NO));
398     writer.addDocument(doc);
399     IndexReader ir = writer.getReader();
400     writer.close();
401     
402     IndexSearcher searcher = newSearcher(ir);
403     Sort sort = new Sort();
404 
405     TopDocs actual = searcher.search(new TermQuery(new Term("value", "foo")), 10, sort);
406     assertEquals(2, actual.totalHits);
407 
408     TopDocs expected = searcher.search(new TermQuery(new Term("value", "foo")), 10);
409     // the two topdocs should be the same
410     assertEquals(expected.totalHits, actual.totalHits);
411     for (int i = 0; i < actual.scoreDocs.length; i++) {
412       assertEquals(actual.scoreDocs[i].doc, expected.scoreDocs[i].doc);
413     }
414     TestUtil.checkReader(ir);
415     ir.close();
416     dir.close();
417   }
