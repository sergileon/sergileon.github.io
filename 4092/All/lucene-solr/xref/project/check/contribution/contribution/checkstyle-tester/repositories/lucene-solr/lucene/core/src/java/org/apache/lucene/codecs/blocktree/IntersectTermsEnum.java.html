38  /** This is used to implement efficient {@link Terms#intersect} for
39   *  block-tree.  Note that it cannot seek, except for the initial term on
40   *  init.  It just "nexts" through the intersection of the automaton and
41   *  the terms.  It does not use the terms index at all: on init, it
42   *  loads the root block, and scans its way to the initial term.
43   *  Likewise, in next it scans until it finds a term that matches the
44   *  current automaton transition.  If the index has auto-prefix terms
45   *  (only for DOCS_ONLY fields currently) it will visit these terms
46   *  when possible and then skip the real terms that auto-prefix term
47   *  matched. */
48

83    // TODO: in some cases we can filter by length?  eg
84    // regexp foo*bar must be at least length 6 bytes
85    public IntersectTermsEnum(FieldReader fr, Automaton automaton, RunAutomaton runAutomaton, BytesRef commonSuffix, BytesRef startTerm, int sinkState) throws IOException {
86      this.fr = fr;
87      this.sinkState = sinkState;
88  
89      assert automaton != null;
90      assert runAutomaton != null;
91  
92      this.runAutomaton = runAutomaton;
93      this.allowAutoPrefixTerms = sinkState != -1;
94      this.automaton = automaton;
95      this.commonSuffix = commonSuffix;
96  
97      in = fr.parent.termsIn.clone();
98      stack = new IntersectTermsEnumFrame[5];
99      for(int idx=0;idx<stack.length;idx++) {
100       stack[idx] = new IntersectTermsEnumFrame(this, idx);
101     }
102     for(int arcIdx=0;arcIdx<arcs.length;arcIdx++) {
103       arcs[arcIdx] = new FST.Arc<>();
104     }
105 
106     if (fr.index == null) {
107       fstReader = null;
108     } else {
109       fstReader = fr.index.getBytesReader();
110     }
111 
112     // TODO: if the automaton is "smallish" we really
113     // should use the terms index to seek at least to
114     // the initial term and likely to subsequent terms
115     // (or, maybe just fallback to ATE for such cases).
116     // Else the seek cost of loading the frames will be
117     // too costly.
118 
119     final FST.Arc<BytesRef> arc = fr.index.getFirstArc(arcs[0]);
120     // Empty string prefix must have an output in the index!
121     assert arc.isFinal();
122 
123     // Special pushFrame since it's the first one:
124     final IntersectTermsEnumFrame f = stack[0];
125     f.fp = f.fpOrig = fr.rootBlockFP;
126     f.prefix = 0;
127     f.setState(runAutomaton.getInitialState());
128     f.arc = arc;
129     f.outputPrefix = arc.output;
130     f.load(fr.rootCode);
131 
132     // for assert:
133     assert setSavedStartTerm(startTerm);
134 
135     currentFrame = f;
136     if (startTerm != null) {
137       seekToStartTerm(startTerm);
138     }
139     currentTransition = currentFrame.transition;
140   }

144     savedStartTerm = startTerm == null ? null : BytesRef.deepCopyOf(startTerm);
145     return true;

154   private IntersectTermsEnumFrame getFrame(int ord) throws IOException {
155     if (ord >= stack.length) {
156       final IntersectTermsEnumFrame[] next = new IntersectTermsEnumFrame[ArrayUtil.oversize(1+ord, RamUsageEstimator.NUM_BYTES_OBJECT_REF)];
157       System.arraycopy(stack, 0, next, 0, stack.length);
158       for(int stackOrd=stack.length;stackOrd<next.length;stackOrd++) {
159         next[stackOrd] = new IntersectTermsEnumFrame(this, stackOrd);
160       }
161       stack = next;
162     }
163     assert stack[ord].ord == ord;
164     return stack[ord];
165   }

167   private FST.Arc<BytesRef> getArc(int ord) {
168     if (ord >= arcs.length) {
169       @SuppressWarnings({"rawtypes","unchecked"}) final FST.Arc<BytesRef>[] next =
170       new FST.Arc[ArrayUtil.oversize(1+ord, RamUsageEstimator.NUM_BYTES_OBJECT_REF)];
171       System.arraycopy(arcs, 0, next, 0, arcs.length);
172       for(int arcOrd=arcs.length;arcOrd<next.length;arcOrd++) {
173         next[arcOrd] = new FST.Arc<>();
174       }
175       arcs = next;
176     }
177     return arcs[ord];
178   }

180   private IntersectTermsEnumFrame pushFrame(int state) throws IOException {
181     assert currentFrame != null;
182 
183     final IntersectTermsEnumFrame f = getFrame(currentFrame == null ? 0 : 1+currentFrame.ord);
184         
185     f.fp = f.fpOrig = currentFrame.lastSubFP;
186     f.prefix = currentFrame.prefix + currentFrame.suffix;
187     f.setState(state);
188 
189     // Walk the arc through the index -- we only
190     // "bother" with this so we can get the floor data
191     // from the index and skip floor blocks when
192     // possible:
193     FST.Arc<BytesRef> arc = currentFrame.arc;
194     int idx = currentFrame.prefix;
195     assert currentFrame.suffix > 0;
196     BytesRef output = currentFrame.outputPrefix;
197     while (idx < f.prefix) {
198       final int target = term.bytes[idx] & 0xff;
199       // TODO: we could be more efficient for the next()
200       // case by using current arc as starting point,
201       // passed to findTargetArc
202       arc = fr.index.findTargetArc(target, arc, getArc(1+idx), fstReader);
203       assert arc != null;
204       output = fstOutputs.add(output, arc.output);
205       idx++;
206     }
207 
208     f.arc = arc;
209     f.outputPrefix = output;
210     assert arc.isFinal();
211     f.load(fstOutputs.add(output, arc.nextFinalOutput));
212     return f;
213   }

238   private int getState() {
239     int state = currentFrame.state;
240     for(int idx=0;idx<currentFrame.suffix;idx++) {
241       state = runAutomaton.step(state,  currentFrame.suffixBytes[currentFrame.startBytePos+idx] & 0xff);
242       assert state != -1;
243     }
244     return state;
245   }

247   // NOTE: specialized to only doing the first-time
248   // seek, but we could generalize it to allow
249   // arbitrary seekExact/Ceil.  Note that this is a
250   // seekFloor!
251   private void seekToStartTerm(BytesRef target) throws IOException {
252     assert currentFrame.ord == 0;
253     if (term.length < target.length) {
254       term.bytes = ArrayUtil.grow(term.bytes, target.length);
255     }
256     FST.Arc<BytesRef> arc = arcs[0];
257     assert arc == currentFrame.arc;
258 
259     for(int idx=0;idx<=target.length;idx++) {
260 
261       while (true) {
262         final int savNextEnt = currentFrame.nextEnt;
263         final int savePos = currentFrame.suffixesReader.getPosition();
264         final int saveStartBytePos = currentFrame.startBytePos;
265         final int saveSuffix = currentFrame.suffix;
266         final long saveLastSubFP = currentFrame.lastSubFP;
267         final int saveTermBlockOrd = currentFrame.termState.termBlockOrd;
268         final boolean saveIsAutoPrefixTerm = currentFrame.isAutoPrefixTerm;
269 
270         final boolean isSubBlock = currentFrame.next();
271 
272         term.length = currentFrame.prefix + currentFrame.suffix;
273         if (term.bytes.length < term.length) {
274           term.bytes = ArrayUtil.grow(term.bytes, term.length);
275         }
276         System.arraycopy(currentFrame.suffixBytes, currentFrame.startBytePos, term.bytes, currentFrame.prefix, currentFrame.suffix);
277 
278         if (isSubBlock && StringHelper.startsWith(target, term)) {
279           // Recurse
280           currentFrame = pushFrame(getState());
281           break;
282         } else {
283           final int cmp = term.compareTo(target);
284           if (cmp < 0) {
285             if (currentFrame.nextEnt == currentFrame.entCount) {
286               if (!currentFrame.isLastInFloor) {
287                 // Advance to next floor block
288                 currentFrame.loadNextFloorBlock();
289                 continue;
290               } else {
291                 return;
292               }
293             }
294             continue;
295           } else if (cmp == 0) {
296             if (allowAutoPrefixTerms == false && currentFrame.isAutoPrefixTerm) {
297               continue;
298             }
299             return;
300           } else if (allowAutoPrefixTerms || currentFrame.isAutoPrefixTerm == false) {
301             // Fallback to prior entry: the semantics of
302             // this method is that the first call to
303             // next() will return the term after the
304             // requested term
305             currentFrame.nextEnt = savNextEnt;
306             currentFrame.lastSubFP = saveLastSubFP;
307             currentFrame.startBytePos = saveStartBytePos;
308             currentFrame.suffix = saveSuffix;
309             currentFrame.suffixesReader.setPosition(savePos);
310             currentFrame.termState.termBlockOrd = saveTermBlockOrd;
311             currentFrame.isAutoPrefixTerm = saveIsAutoPrefixTerm;
312             System.arraycopy(currentFrame.suffixBytes, currentFrame.startBytePos, term.bytes, currentFrame.prefix, currentFrame.suffix);
313             term.length = currentFrame.prefix + currentFrame.suffix;
314             // If the last entry was a block we don't
315             // need to bother recursing and pushing to
316             // the last term under it because the first
317             // next() will simply skip the frame anyway
318             return;
319           }
320         }
321       }
322     }
323 
324     assert false;
325   }

327   private boolean popPushNext() throws IOException {
328     // Pop finished frames
329     while (currentFrame.nextEnt == currentFrame.entCount) {
330       if (!currentFrame.isLastInFloor) {
331         // Advance to next floor block
332         currentFrame.loadNextFloorBlock();
333         break;
334       } else {
335         if (currentFrame.ord == 0) {
336           throw NoMoreTermsException.INSTANCE;
337         }
338         final long lastFP = currentFrame.fpOrig;
339         currentFrame = stack[currentFrame.ord-1];
340         currentTransition = currentFrame.transition;
341         assert currentFrame.lastSubFP == lastFP;
342       }
343     }
344 
345     return currentFrame.next();
346   }

348   private boolean skipPastLastAutoPrefixTerm() throws IOException {
349     assert currentFrame.isAutoPrefixTerm;
350     useAutoPrefixTerm = false;
351     currentFrame.termState.isRealTerm = true;
352 
353     // If we last returned an auto-prefix term, we must now skip all
354     // actual terms sharing that prefix.  At most, that skipping
355     // requires popping one frame, but it can also require simply
356     // scanning ahead within the current frame.  This scanning will
357     // skip sub-blocks that contain many terms, which is why the
358     // optimization "works":
359     int floorSuffixLeadEnd = currentFrame.floorSuffixLeadEnd;
360 
361     boolean isSubBlock;
362 
363     if (floorSuffixLeadEnd == -1) {
364       // An ordinary prefix, e.g. foo*
365       int prefix = currentFrame.prefix;
366       int suffix = currentFrame.suffix;
367       if (suffix == 0) {
368 
369         // Easy case: the prefix term's suffix is the empty string,
370         // meaning the prefix corresponds to all terms in the
371         // current block, so we just pop this entire block:
372         if (currentFrame.ord == 0) {
373           throw NoMoreTermsException.INSTANCE;
374         }
375         currentFrame = stack[currentFrame.ord-1];
376         currentTransition = currentFrame.transition;
377 
378         return popPushNext();
379 
380       } else {
381 
382         // Just next() until we hit an entry that doesn't share this
383         // prefix.  The first next should be a sub-block sharing the
384         // same prefix, because if there are enough terms matching a
385         // given prefix to warrant an auto-prefix term, then there
386         // must also be enough to make a sub-block (assuming
387         // minItemsInPrefix > minItemsInBlock):
388         scanPrefix:
389         while (true) {
390           if (currentFrame.nextEnt == currentFrame.entCount) {
391             if (currentFrame.isLastInFloor == false) {
392               currentFrame.loadNextFloorBlock();
393             } else if (currentFrame.ord == 0) {
394               throw NoMoreTermsException.INSTANCE;
395             } else {
396               // Pop frame, which also means we've moved beyond this
397               // auto-prefix term:
398               currentFrame = stack[currentFrame.ord-1];
399               currentTransition = currentFrame.transition;
400 
401               return popPushNext();
402             }
403           }
404           isSubBlock = currentFrame.next();
405           for(int i=0;i<suffix;i++) {
406             if (term.bytes[prefix+i] != currentFrame.suffixBytes[currentFrame.startBytePos+i]) {
407               break scanPrefix;
408             }
409           }
410         }
411       }
412     } else {
413       // Floor'd auto-prefix term; in this case we must skip all
414       // terms e.g. matching foo[a-m]*.  We are currently "on" fooa,
415       // which the automaton accepted (fooa* through foom*), and
416       // floorSuffixLeadEnd is m, so we must now scan to foon:
417       int prefix = currentFrame.prefix;
418       int suffix = currentFrame.suffix;
419 
420       if (currentFrame.floorSuffixLeadStart == -1) {
421         suffix++;
422       }
423 
424       if (suffix == 0) {
425 
426         // This means current frame is fooa*, so we have to first
427         // pop the current frame, then scan in parent frame:
428         if (currentFrame.ord == 0) {
429           throw NoMoreTermsException.INSTANCE;
430         }
431         currentFrame = stack[currentFrame.ord-1];
432         currentTransition = currentFrame.transition;
433 
434         // Current (parent) frame is now foo*, so now we just scan
435         // until the lead suffix byte is > floorSuffixLeadEnd
436         //assert currentFrame.prefix == prefix-1;
437         //prefix = currentFrame.prefix;
438 
439         // In case when we pop, and the parent block is not just prefix-1, e.g. in block 417* on
440         // its first term = floor prefix term 41[7-9], popping to block 4*:
441         prefix = currentFrame.prefix;
442 
443         suffix = term.length - currentFrame.prefix;
444       } else {
445         // No need to pop; just scan in currentFrame:
446       }
447 
448       // Now we scan until the lead suffix byte is > floorSuffixLeadEnd
449       scanFloor:
450       while (true) {
451         if (currentFrame.nextEnt == currentFrame.entCount) {
452           if (currentFrame.isLastInFloor == false) {
453             currentFrame.loadNextFloorBlock();
454           } else if (currentFrame.ord == 0) {
455             throw NoMoreTermsException.INSTANCE;
456           } else {
457             // Pop frame, which also means we've moved beyond this
458             // auto-prefix term:
459             currentFrame = stack[currentFrame.ord-1];
460             currentTransition = currentFrame.transition;
461 
462             return popPushNext();
463           }
464         }
465         isSubBlock = currentFrame.next();
466         for(int i=0;i<suffix-1;i++) {
467           if (term.bytes[prefix+i] != currentFrame.suffixBytes[currentFrame.startBytePos+i]) {
468             break scanFloor;
469           }
470         }
471         if (currentFrame.suffix >= suffix && (currentFrame.suffixBytes[currentFrame.startBytePos+suffix-1]&0xff) > floorSuffixLeadEnd) {
472           // Done scanning: we are now on the first term after all
473           // terms matched by this auto-prefix term
474           break;
475         }
476       }
477     }
478 
479     return isSubBlock;
480   }

482   // Only used internally when there are no more terms in next():
483   private static final class NoMoreTermsException extends RuntimeException {
484 
485     // Only used internally when there are no more terms in next():
486     public static final NoMoreTermsException INSTANCE = new NoMoreTermsException();
487 
488     private NoMoreTermsException() {
489     }

498   @Override
499   public BytesRef next() throws IOException {
500     try {
501       return _next();
502     } catch (NoMoreTermsException eoi) {
503       // Provoke NPE if we are (illegally!) called again:
504       currentFrame = null;
505       return null;
506     }
507   }

509   private BytesRef _next() throws IOException {
510 
511     boolean isSubBlock;
512 
513     if (useAutoPrefixTerm) {
514       // If the current term was an auto-prefix term, we have to skip past it:
515       isSubBlock = skipPastLastAutoPrefixTerm();
516       assert useAutoPrefixTerm == false;
517     } else {
518       isSubBlock = popPushNext();
519     }
520 
521     nextTerm:
522 
523     while (true) {
524       assert currentFrame.transition == currentTransition;
525 
526       int state;
527       int lastState;
528 
529       // NOTE: suffix == 0 can only happen on the first term in a block, when
530       // there is a term exactly matching a prefix in the index.  If we
531       // could somehow re-org the code so we only checked this case immediately
532       // after pushing a frame...
533       if (currentFrame.suffix != 0) {
534 
535         final byte[] suffixBytes = currentFrame.suffixBytes;
536 
537         // This is the first byte of the suffix of the term we are now on:
538         final int label = suffixBytes[currentFrame.startBytePos] & 0xff;
539 
540         if (label < currentTransition.min) {
541           // Common case: we are scanning terms in this block to "catch up" to
542           // current transition in the automaton:
543           int minTrans = currentTransition.min;
544           while (currentFrame.nextEnt < currentFrame.entCount) {
545             isSubBlock = currentFrame.next();
546             if ((suffixBytes[currentFrame.startBytePos] & 0xff) >= minTrans) {
547               continue nextTerm;
548             }
549           }
550 
551           // End of frame:
552           isSubBlock = popPushNext();
553           continue nextTerm;
554         }
555 
556         // Advance where we are in the automaton to match this label:
557 
558         while (label > currentTransition.max) {
559           if (currentFrame.transitionIndex >= currentFrame.transitionCount-1) {
560             // Pop this frame: no further matches are possible because
561             // we've moved beyond what the max transition will allow
562             if (currentFrame.ord == 0) {
563               // Provoke NPE if we are (illegally!) called again:
564               currentFrame = null;
565               return null;
566             }
567             currentFrame = stack[currentFrame.ord-1];
568             currentTransition = currentFrame.transition;
569             isSubBlock = popPushNext();
570             continue nextTerm;
571           }
572           currentFrame.transitionIndex++;
573           automaton.getNextTransition(currentTransition);
574 
575           if (label < currentTransition.min) {
576             int minTrans = currentTransition.min;
577             while (currentFrame.nextEnt < currentFrame.entCount) {
578               isSubBlock = currentFrame.next();
579               if ((suffixBytes[currentFrame.startBytePos] & 0xff) >= minTrans) {
580                 continue nextTerm;
581               }
582             }
583 
584             // End of frame:
585             isSubBlock = popPushNext();
586             continue nextTerm;
587           }
588         }
589 
590         if (commonSuffix != null && !isSubBlock) {
591           final int termLen = currentFrame.prefix + currentFrame.suffix;
592           if (termLen < commonSuffix.length) {
593             // No match
594             isSubBlock = popPushNext();
595             continue nextTerm;
596           }
597 
598           final byte[] commonSuffixBytes = commonSuffix.bytes;
599 
600           final int lenInPrefix = commonSuffix.length - currentFrame.suffix;
601           assert commonSuffix.offset == 0;
602           int suffixBytesPos;
603           int commonSuffixBytesPos = 0;
604 
605           if (lenInPrefix > 0) {
606             // A prefix of the common suffix overlaps with
607             // the suffix of the block prefix so we first
608             // test whether the prefix part matches:
609             final byte[] termBytes = term.bytes;
610             int termBytesPos = currentFrame.prefix - lenInPrefix;
611             assert termBytesPos >= 0;
612             final int termBytesPosEnd = currentFrame.prefix;
613             while (termBytesPos < termBytesPosEnd) {
614               if (termBytes[termBytesPos++] != commonSuffixBytes[commonSuffixBytesPos++]) {
615                 isSubBlock = popPushNext();
616                 continue nextTerm;
617               }
618             }
619             suffixBytesPos = currentFrame.startBytePos;
620           } else {
621             suffixBytesPos = currentFrame.startBytePos + currentFrame.suffix - commonSuffix.length;
622           }
623 
624           // Test overlapping suffix part:
625           final int commonSuffixBytesPosEnd = commonSuffix.length;
626           while (commonSuffixBytesPos < commonSuffixBytesPosEnd) {
627             if (suffixBytes[suffixBytesPos++] != commonSuffixBytes[commonSuffixBytesPos++]) {
628               isSubBlock = popPushNext();
629               continue nextTerm;
630             }
631           }
632         }
633 
634         // TODO: maybe we should do the same linear test
635         // that AutomatonTermsEnum does, so that if we
636         // reach a part of the automaton where .* is
637         // "temporarily" accepted, we just blindly .next()
638         // until the limit
639 
640         // See if the term suffix matches the automaton:
641 
642         // We know from above that the first byte in our suffix (label) matches
643         // the current transition, so we step from the 2nd byte
644         // in the suffix:
645         lastState = currentFrame.state;
646         state = currentTransition.dest;
647 
648         int end = currentFrame.startBytePos + currentFrame.suffix;
649         for (int idx=currentFrame.startBytePos+1;idx<end;idx++) {
650           lastState = state;
651           state = runAutomaton.step(state, suffixBytes[idx] & 0xff);
652           if (state == -1) {
653             // No match
654             isSubBlock = popPushNext();
655             continue nextTerm;
656           }
657         }
658       } else {
659         state = currentFrame.state;
660         lastState = currentFrame.lastState;
661       }
662 
663       if (isSubBlock) {
664         // Match!  Recurse:
665         copyTerm();
666         currentFrame = pushFrame(state);
667         currentTransition = currentFrame.transition;
668         currentFrame.lastState = lastState;
669       } else if (currentFrame.isAutoPrefixTerm) {
670         // We are on an auto-prefix term, meaning this term was compiled
671         // at indexing time, matching all terms sharing this prefix (or,
672         // a floor'd subset of them if that count was too high).  A
673         // prefix term represents a range of terms, so we now need to
674         // test whether, from the current state in the automaton, it
675         // accepts all terms in that range.  As long as it does, we can
676         // use this term and then later skip ahead past all terms in
677         // this range:
678         if (allowAutoPrefixTerms) {
679 
680           if (currentFrame.floorSuffixLeadEnd == -1) {
681             // Simple prefix case
682             useAutoPrefixTerm = state == sinkState;
683           } else {
684             if (currentFrame.floorSuffixLeadStart == -1) {
685               // Must also accept the empty string in this case
686               if (automaton.isAccept(state)) {
687                 useAutoPrefixTerm = acceptsSuffixRange(state, 0, currentFrame.floorSuffixLeadEnd);
688               }
689             } else {
690               useAutoPrefixTerm = acceptsSuffixRange(lastState, currentFrame.floorSuffixLeadStart, currentFrame.floorSuffixLeadEnd);
691             }
692           }
693 
694           if (useAutoPrefixTerm) {
695             // All suffixes of this auto-prefix term are accepted by the automaton, so we can use it:
696             copyTerm();
697             currentFrame.termState.isRealTerm = false;
698             return term;
699           } else {
700             // We move onto the next term
701           }
702         } else {
703           // We are not allowed to use auto-prefix terms, so we just skip it
704         }
705       } else if (runAutomaton.isAccept(state)) {
706         copyTerm();
707         assert savedStartTerm == null || term.compareTo(savedStartTerm) > 0: "saveStartTerm=" + savedStartTerm.utf8ToString() + " term=" + term.utf8ToString();
708         return term;
709       } else {
710         // This term is a prefix of a term accepted by the automaton, but is not itself acceptd
711       }
712 
713       isSubBlock = popPushNext();
714     }
715   }
716 
717   private final Transition scratchTransition = new Transition();
718 
719   /** Returns true if, from this state, the automaton accepts any suffix
720    *  starting with a label between start and end, inclusive.  We just
721    *  look for a transition, matching this range, to the sink state.  */
722   private boolean acceptsSuffixRange(int state, int start, int end) {
723 
724     int count = automaton.initTransition(state, scratchTransition);
725     for(int i=0;i<count;i++) {
726       automaton.getNextTransition(scratchTransition);
727       if (start >= scratchTransition.min && end <= scratchTransition.max && scratchTransition.dest == sinkState) {
728         return true;
729       }
730     }
731 
732     return false;
733   }

738     try {
739       return b.utf8ToString() + " " + b;

741       // If BytesRef isn't actually UTF8, or it's eg a
742       // prefix of UTF8 that ends mid-unicode-char, we
743       // fallback to hex:
744       return b.toString();
745     }

748   private void copyTerm() {
749     final int len = currentFrame.prefix + currentFrame.suffix;
750     if (term.bytes.length < len) {
751       term.bytes = ArrayUtil.grow(term.bytes, len);
752     }
753     System.arraycopy(currentFrame.suffixBytes, currentFrame.startBytePos, term.bytes, currentFrame.prefix, currentFrame.suffix);
754     term.length = len;
755   }
