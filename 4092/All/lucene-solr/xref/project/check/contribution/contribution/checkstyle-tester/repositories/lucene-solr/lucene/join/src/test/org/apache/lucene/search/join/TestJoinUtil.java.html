295   public void testRandomOrdinalsJoin() throws Exception {
296     IndexIterationContext context = createContext(512, false, true);
297     int searchIters = 10;
298     IndexSearcher indexSearcher = context.searcher;
299     for (int i = 0; i < searchIters; i++) {
300       if (VERBOSE) {
301         System.out.println("search iter=" + i);
302       }
303       int r = random().nextInt(context.randomUniqueValues.length);
304       boolean from = context.randomFrom[r];
305       String randomValue = context.randomUniqueValues[r];
306       BitSet expectedResult = createExpectedResult(randomValue, from, indexSearcher.getIndexReader(), context);
307 
308       final Query actualQuery = new TermQuery(new Term("value", randomValue));
309       if (VERBOSE) {
310         System.out.println("actualQuery=" + actualQuery);
311       }
312       final ScoreMode scoreMode = ScoreMode.values()[random().nextInt(ScoreMode.values().length)];
313       if (VERBOSE) {
314         System.out.println("scoreMode=" + scoreMode);
315       }
316 
317       final Query joinQuery;
318       if (from) {
319         BooleanQuery.Builder fromQuery = new BooleanQuery.Builder();
320         fromQuery.add(new TermQuery(new Term("type", "from")), BooleanClause.Occur.FILTER);
321         fromQuery.add(actualQuery, BooleanClause.Occur.MUST);
322         Query toQuery = new TermQuery(new Term("type", "to"));
323         joinQuery = JoinUtil.createJoinQuery("join_field", fromQuery.build(), toQuery, indexSearcher, scoreMode, context.ordinalMap);
324       } else {
325         BooleanQuery.Builder fromQuery = new BooleanQuery.Builder();
326         fromQuery.add(new TermQuery(new Term("type", "to")), BooleanClause.Occur.FILTER);
327         fromQuery.add(actualQuery, BooleanClause.Occur.MUST);
328         Query toQuery = new TermQuery(new Term("type", "from"));
329         joinQuery = JoinUtil.createJoinQuery("join_field", fromQuery.build(), toQuery, indexSearcher, scoreMode, context.ordinalMap);
330       }
331       if (VERBOSE) {
332         System.out.println("joinQuery=" + joinQuery);
333       }
334 
335       final BitSet actualResult = new FixedBitSet(indexSearcher.getIndexReader().maxDoc());
336       final TopScoreDocCollector topScoreDocCollector = TopScoreDocCollector.create(10);
337       indexSearcher.search(joinQuery, MultiCollector.wrap(new BitSetCollector(actualResult), topScoreDocCollector));
338       assertBitSet(expectedResult, actualResult, indexSearcher);
339       TopDocs expectedTopDocs = createExpectedTopDocs(randomValue, from, scoreMode, context);
340       TopDocs actualTopDocs = topScoreDocCollector.topDocs();
341       assertTopDocs(expectedTopDocs, actualTopDocs, scoreMode, indexSearcher, joinQuery);
342     }
343     context.close();
344   }

346   public void testMinMaxScore() throws Exception {
347     final String priceField = "price";
348     // FunctionQuery would be helpful, but join module doesn't depend on queries module.
349     Query priceQuery = new Query() {
350 
351       private final Query fieldQuery = new FieldValueQuery(priceField);
352 
353       @Override
354       public Weight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
355         final Weight fieldWeight = fieldQuery.createWeight(searcher, false);
356         return new Weight(this) {
357 
358           @Override
359           public void extractTerms(Set<Term> terms) {
360           }
361 
362           @Override
363           public Explanation explain(LeafReaderContext context, int doc) throws IOException {
364             return null;
365           }
366 
367           @Override
368           public float getValueForNormalization() throws IOException {
369             return 0;
370           }
371 
372           @Override
373           public void normalize(float norm, float topLevelBoost) {
374           }
375 
376           @Override
377           public Scorer scorer(LeafReaderContext context) throws IOException {
378             Scorer fieldScorer = fieldWeight.scorer(context);
379             if (fieldScorer == null) {
380               return null;
381             }
382             final NumericDocValues price = context.reader().getNumericDocValues(priceField);
383             return new FilterScorer(fieldScorer, this) {
384               @Override
385               public float score() throws IOException {
386                 return (float) price.get(in.docID());
387               }
388             };
389           }
390         };
391       }
392 
393       @Override
394       public String toString(String field) {
395         return fieldQuery.toString(field);
396       }
397     };
398 
399     Directory dir = newDirectory();
400     RandomIndexWriter iw = new RandomIndexWriter(
401         random(),
402         dir,
403         newIndexWriterConfig(new MockAnalyzer(random(), MockTokenizer.KEYWORD, false))
404     );
405 
406     Map<String, Float> lowestScoresPerParent = new HashMap<>();
407     Map<String, Float> highestScoresPerParent = new HashMap<>();
408     int numParents = RandomInts.randomIntBetween(random(), 16, 64);
409     for (int p = 0; p < numParents; p++) {
410       String parentId = Integer.toString(p);
411       Document parentDoc = new Document();
412       parentDoc.add(new StringField("id", parentId, Field.Store.YES));
413       parentDoc.add(new StringField("type", "to", Field.Store.NO));
414       parentDoc.add(new SortedDocValuesField("join_field", new BytesRef(parentId)));
415       iw.addDocument(parentDoc);
416       int numChildren = RandomInts.randomIntBetween(random(), 2, 16);
417       int lowest = Integer.MAX_VALUE;
418       int highest = Integer.MIN_VALUE;
419       for (int c = 0; c < numChildren; c++) {
420         String childId = Integer.toString(p + c);
421         Document childDoc = new Document();
422         childDoc.add(new StringField("id", childId, Field.Store.YES));
423         childDoc.add(new StringField("type", "from", Field.Store.NO));
424         childDoc.add(new SortedDocValuesField("join_field", new BytesRef(parentId)));
425         int price = random().nextInt(1000);
426         childDoc.add(new NumericDocValuesField(priceField, price));
427         iw.addDocument(childDoc);
428         lowest = Math.min(lowest, price);
429         highest = Math.max(highest, price);
430       }
431       lowestScoresPerParent.put(parentId, (float) lowest);
432       highestScoresPerParent.put(parentId, (float) highest);
433     }
434     iw.close();
435 
436 
437     IndexSearcher searcher = new IndexSearcher(DirectoryReader.open(dir));
438     SortedDocValues[] values = new SortedDocValues[searcher.getIndexReader().leaves().size()];
439     for (LeafReaderContext leadContext : searcher.getIndexReader().leaves()) {
440       values[leadContext.ord] = DocValues.getSorted(leadContext.reader(), "join_field");
441     }
442     MultiDocValues.OrdinalMap ordinalMap = MultiDocValues.OrdinalMap.build(
443         searcher.getIndexReader().getCoreCacheKey(), values, PackedInts.DEFAULT
444     );
445     BooleanQuery.Builder fromQuery = new BooleanQuery.Builder();
446     fromQuery.add(priceQuery, BooleanClause.Occur.MUST);
447     Query toQuery = new TermQuery(new Term("type", "to"));
448     Query joinQuery = JoinUtil.createJoinQuery("join_field", fromQuery.build(), toQuery, searcher, ScoreMode.Min, ordinalMap);
449     TopDocs topDocs = searcher.search(joinQuery, numParents);
450     assertEquals(numParents, topDocs.totalHits);
451     for (int i = 0; i < topDocs.scoreDocs.length; i++) {
452       ScoreDoc scoreDoc = topDocs.scoreDocs[i];
453       String id = SlowCompositeReaderWrapper.wrap(searcher.getIndexReader()).document(scoreDoc.doc).get("id");
454       assertEquals(lowestScoresPerParent.get(id), scoreDoc.score, 0f);
455     }
456 
457     joinQuery = JoinUtil.createJoinQuery("join_field", fromQuery.build(), toQuery, searcher, ScoreMode.Max, ordinalMap);
458     topDocs = searcher.search(joinQuery, numParents);
459     assertEquals(numParents, topDocs.totalHits);
460     for (int i = 0; i < topDocs.scoreDocs.length; i++) {
461       ScoreDoc scoreDoc = topDocs.scoreDocs[i];
462       String id = SlowCompositeReaderWrapper.wrap(searcher.getIndexReader()).document(scoreDoc.doc).get("id");
463       assertEquals(highestScoresPerParent.get(id), scoreDoc.score, 0f);
464     }
465 
466     searcher.getIndexReader().close();
467     dir.close();
468   }

470   public void testMinMaxDocs() throws Exception {
471     Directory dir = newDirectory();

480     int numParents = RandomInts.randomIntBetween(random(), 16, 64);
481     int[] childDocsPerParent = new int[numParents];
482     for (int p = 0; p < numParents; p++) {
483       String parentId = Integer.toString(p);
484       Document parentDoc = new Document();
485       parentDoc.add(new StringField("id", parentId, Field.Store.YES));
486       parentDoc.add(new StringField("type", "to", Field.Store.NO));
487       parentDoc.add(new SortedDocValuesField("join_field", new BytesRef(parentId)));
488       iw.addDocument(parentDoc);
489       int numChildren = RandomInts.randomIntBetween(random(), minChildDocsPerParent, maxChildDocsPerParent);
490       childDocsPerParent[p] = numChildren;
491       for (int c = 0; c < numChildren; c++) {
492         String childId = Integer.toString(p + c);
493         Document childDoc = new Document();
494         childDoc.add(new StringField("id", childId, Field.Store.YES));
495         childDoc.add(new StringField("type", "from", Field.Store.NO));
496         childDoc.add(new SortedDocValuesField("join_field", new BytesRef(parentId)));
497         iw.addDocument(childDoc);
498       }
499     }
500     iw.close();
501 
502     IndexSearcher searcher = new IndexSearcher(DirectoryReader.open(dir));
503     SortedDocValues[] values = new SortedDocValues[searcher.getIndexReader().leaves().size()];
504     for (LeafReaderContext leadContext : searcher.getIndexReader().leaves()) {
505       values[leadContext.ord] = DocValues.getSorted(leadContext.reader(), "join_field");
506     }
507     MultiDocValues.OrdinalMap ordinalMap = MultiDocValues.OrdinalMap.build(
508         searcher.getIndexReader().getCoreCacheKey(), values, PackedInts.DEFAULT
509     );
510     Query fromQuery = new TermQuery(new Term("type", "from"));
511     Query toQuery = new TermQuery(new Term("type", "to"));
512 
513     int iters = RandomInts.randomIntBetween(random(), 3, 9);
514     for (int i = 1; i <= iters; i++) {
515       final ScoreMode scoreMode = ScoreMode.values()[random().nextInt(ScoreMode.values().length)];
516       int min = RandomInts.randomIntBetween(random(), minChildDocsPerParent, maxChildDocsPerParent - 1);
517       int max = RandomInts.randomIntBetween(random(), min, maxChildDocsPerParent);
518       if (VERBOSE) {
519         System.out.println("iter=" + i);
520         System.out.println("scoreMode=" + scoreMode);
521         System.out.println("min=" + min);
522         System.out.println("max=" + max);
523       }
524       Query joinQuery = JoinUtil.createJoinQuery("join_field", fromQuery, toQuery, searcher, scoreMode, ordinalMap, min, max);
525       TotalHitCountCollector collector = new TotalHitCountCollector();
526       searcher.search(joinQuery, collector);

1012 
1013     if (random().nextBoolean()) {
1014       w.forceMerge(1);
1015     }
1016     w.close();
1017 
1018     // Pre-compute all possible hits for all unique random values. On top of this also compute all possible score for
1019     // any ScoreMode.
1020     DirectoryReader topLevelReader = DirectoryReader.open(dir);
1021     IndexSearcher searcher = newSearcher(topLevelReader);
1022     for (int i = 0; i < context.randomUniqueValues.length; i++) {
1023       String uniqueRandomValue = context.randomUniqueValues[i];
1024       final String fromField;
1025       final String toField;
1026       final Map<String, Map<Integer, JoinScore>> queryVals;
1027       if (context.randomFrom[i]) {
1028         fromField = "from";
1029         toField = "to";
1030         queryVals = context.fromHitsToJoinScore;
1031       } else {
1032         fromField = "to";
1033         toField = "from";
1034         queryVals = context.toHitsToJoinScore;
1035       }
1036       final Map<BytesRef, JoinScore> joinValueToJoinScores = new HashMap<>();
1037       if (multipleValuesPerDocument) {
1038         searcher.search(new TermQuery(new Term("value", uniqueRandomValue)), new SimpleCollector() {
1039 
1040           private Scorer scorer;
1041           private SortedSetDocValues docTermOrds;
1042 
1043           @Override
1044           public void collect(int doc) throws IOException {
1045             docTermOrds.setDocument(doc);
1046             long ord;
1047             while ((ord = docTermOrds.nextOrd()) != SortedSetDocValues.NO_MORE_ORDS) {
1048               final BytesRef joinValue = docTermOrds.lookupOrd(ord);
1049               JoinScore joinScore = joinValueToJoinScores.get(joinValue);
1050               if (joinScore == null) {
1051                 joinValueToJoinScores.put(BytesRef.deepCopyOf(joinValue), joinScore = new JoinScore());
1052               }
1053               joinScore.addScore(scorer.score());
1054             }
1055           }
1056 
1057           @Override
1058           protected void doSetNextReader(LeafReaderContext context) throws IOException {
1059             docTermOrds = DocValues.getSortedSet(context.reader(), fromField);
1060           }
1061 
1062           @Override
1063           public void setScorer(Scorer scorer) {
1064             this.scorer = scorer;
1065           }

1081             final BytesRef joinValue = terms.get(doc);

1083               return;
1084             }
1085

1087             if (joinScore == null) {

1089             }

1098             terms = DocValues.getBinary(context.reader(), fromField);

1115       if (multipleValuesPerDocument) {
1116         LeafReader slowCompositeReader = SlowCompositeReaderWrapper.wrap(topLevelReader);
1117         Terms terms = slowCompositeReader.terms(toField);
1118         if (terms != null) {
1119           PostingsEnum postingsEnum = null;
1120           SortedSet<BytesRef> joinValues = new TreeSet<>(BytesRef.getUTF8SortedAsUnicodeComparator());
1121           joinValues.addAll(joinValueToJoinScores.keySet());

1123             TermsEnum termsEnum = terms.iterator();
1124             if (termsEnum.seekExact(joinValue)) {
1125               postingsEnum = termsEnum.postings(postingsEnum, PostingsEnum.NONE);
1126               JoinScore joinScore = joinValueToJoinScores.get(joinValue);
1127 
1128               for (int doc = postingsEnum.nextDoc(); doc != DocIdSetIterator.NO_MORE_DOCS; doc = postingsEnum.nextDoc()) {
1129                 // First encountered join value determines the score.
1130                 // Something to keep in mind for many-to-many relations.
1131                 if (!docToJoinScore.containsKey(doc)) {
1132                   docToJoinScore.put(doc, joinScore);
1133                 }
1134               }
1135             }
1136           }
1137         }
1138       } else {
1139         searcher.search(new MatchAllDocsQuery(), new SimpleCollector() {

1146             final BytesRef joinValue = terms.get(doc);
1147             JoinScore joinScore = joinValueToJoinScores.get(joinValue);
1148             if (joinScore == null) {
1149               return;
1150             }
1151             docToJoinScore.put(docBase + doc, joinScore);
1152           }

1155           protected void doSetNextReader(LeafReaderContext context) throws IOException {
1156             terms = DocValues.getBinary(context.reader(), toField);
1157             docBase = context.docBase;
1158           }
1159 
1160           @Override
1161           public void setScorer(Scorer scorer) {
1162           }
1163 
1164           @Override
1165           public boolean needsScores() {
1166             return false;
1167           }
1168         });
1169       }
1170       queryVals.put(uniqueRandomValue, docToJoinScore);
1171     }
1172 
1173     if (globalOrdinalJoin) {

1175       for (LeafReaderContext leadContext : topLevelReader.leaves()) {
1176         values[leadContext.ord] = DocValues.getSorted(leadContext.reader(), "join_field");
1177       }
1178       context.ordinalMap = MultiDocValues.OrdinalMap.build(
1179           topLevelReader.getCoreCacheKey(), values, PackedInts.DEFAULT
1180       );
1181     }
1182

1184     context.dir = dir;

1189                                         final boolean from,
1190                                         final ScoreMode scoreMode,
1191                                         IndexIterationContext context) {
1192 
1193     Map<Integer, JoinScore> hitsToJoinScores;
1194     if (from) {
1195       hitsToJoinScores = context.fromHitsToJoinScore.get(queryValue);
1196     } else {
1197       hitsToJoinScores = context.toHitsToJoinScore.get(queryValue);
1198     }
1199     List<Map.Entry<Integer,JoinScore>> hits = new ArrayList<>(hitsToJoinScores.entrySet());
1200     Collections.sort(hits, new Comparator<Map.Entry<Integer, JoinScore>>() {
1201 
1202       @Override
1203       public int compare(Map.Entry<Integer, JoinScore> hit1, Map.Entry<Integer, JoinScore> hit2) {
1204         float score1 = hit1.getValue().score(scoreMode);
1205         float score2 = hit2.getValue().score(scoreMode);
1206 
1207         int cmp = Float.compare(score2, score1);
1208         if (cmp != 0) {
1209           return cmp;
1210         }
1211         return hit1.getKey() - hit2.getKey();
1212       }
1213 
1214     });
1215     ScoreDoc[] scoreDocs = new ScoreDoc[Math.min(10, hits.size())];
1216     for (int i = 0; i < scoreDocs.length; i++) {
1217       Map.Entry<Integer,JoinScore> hit = hits.get(i);
1218       scoreDocs[i] = new ScoreDoc(hit.getKey(), hit.getValue().score(scoreMode));
1219     }
1220     return new TopDocs(hits.size(), scoreDocs, hits.isEmpty() ? Float.NaN : hits.get(0).getValue().score(scoreMode));
1221   }

1297   private static class JoinScore {
1298 
1299     float minScore = Float.POSITIVE_INFINITY;
1300     float maxScore = Float.NEGATIVE_INFINITY;
1301     float total;
1302     int count;
1303 
1304     void addScore(float score) {
1305       if (score > maxScore) {
1306         maxScore = score;
1307       }
1308       if (score < minScore) {
1309         minScore = score;
1310       }
1311       total += score;
1312       count++;
1313     }
1314 
1315     float score(ScoreMode mode) {
1316       switch (mode) {
1317         case None:
1318           return 1f;
1319         case Total:
1320           return total;
1321         case Avg:
1322           return total / count;
1323         case Min:
1324           return minScore;
1325         case Max:
1326           return maxScore;
1327       }
1328       throw new IllegalArgumentException("Unsupported ScoreMode: " + mode);
1329     }
1330
