57      cleanup();

69    void makeFile(String contents) {
70      try {
71        Writer out = new OutputStreamWriter(new FileOutputStream(filename), StandardCharsets.UTF_8);
72        out.write(contents);
73        out.close();
74      } catch (Exception e) {
75        throw new RuntimeException(e);
76      }
77    }
78  
79    void cleanup() {
80      assertU(delQ("*:*"));
81      assertU(commit());
82    }
83  
84    void loadLocal(String... args) throws Exception {
85      LocalSolrQueryRequest req =  (LocalSolrQueryRequest)req(args);
86  
87      // TODO: stop using locally defined streams once stream.file and
88      // stream.body work everywhere
89      List<ContentStream> cs = new ArrayList<>(1);
90      ContentStreamBase f = new ContentStreamBase.FileStream(new File(filename));
91      f.setContentType("text/csv");
92      cs.add(f);
93      req.setContentStreams(cs);
94      h.query("/update",req);
95    }

97    @Test
98    public void testCSVLoad() throws Exception {
99      makeFile("id\n100\n101\n102");
100     loadLocal();
101     // check default commit of false
102     assertQ(req("id:[100 TO 110]"),"//*[@numFound='0']");
103     assertU(commit());
104     assertQ(req("id:[100 TO 110]"),"//*[@numFound='3']");
105   }

150   @Test
151   public void testCSV() throws Exception {
152     lrf.args.put(CommonParams.VERSION,"2.2");
153     
154     makeFile("id,str_s\n100,\"quoted\"\n101,\n102,\"\"\n103,");
155     loadLocal("commit","true");
156     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
157     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='quoted']");
158     assertQ(req("id:101"),"count(//str[@name='str_s'])=0");
159     // 102 is a quoted zero length field ,"", as opposed to ,,
160     // but we can't distinguish this case (and it's debateable
161     // if we should).  Does CSV have a way to specify missing
162     // from zero-length?
163     assertQ(req("id:102"),"count(//str[@name='str_s'])=0");
164     assertQ(req("id:103"),"count(//str[@name='str_s'])=0");
165 
166     // test overwrite by default
167     loadLocal("commit","true");
168     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
169 
170     // test explicitly adding header=true (the default)
171     loadLocal("commit","true","header","true");
172     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
173 
174     // test no overwrites
175     loadLocal("commit","true", "overwrite","false");
176     assertQ(req("id:[100 TO 110]"),"//*[@numFound='8']");
177 
178     // test overwrite
179     loadLocal("commit","true");
180     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
181 
182     // test global value mapping
183     loadLocal("commit","true", "map","quoted:QUOTED");
184     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
185     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='QUOTED']");
186     assertQ(req("id:101"),"count(//str[@name='str_s'])=0");
187     assertQ(req("id:102"),"count(//str[@name='str_s'])=0");
188     assertQ(req("id:103"),"count(//str[@name='str_s'])=0");
189 
190     // test value mapping to empty (remove)
191     loadLocal("commit","true", "map","quoted:");
192     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
193     assertQ(req("id:100"),"count(//str[@name='str_s'])=0");
194 
195     // test value mapping from empty
196     loadLocal("commit","true", "map",":EMPTY");
197     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
198     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='quoted']");
199     assertQ(req("id:101"),"//arr[@name='str_s']/str[.='EMPTY']");
200     assertQ(req("id:102"),"//arr[@name='str_s']/str[.='EMPTY']");
201     assertQ(req("id:103"),"//arr[@name='str_s']/str[.='EMPTY']");
202 
203     // test multiple map rules
204     loadLocal("commit","true", "map",":EMPTY", "map","quoted:QUOTED");
205     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
206     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='QUOTED']");
207     assertQ(req("id:101"),"//arr[@name='str_s']/str[.='EMPTY']");
208     assertQ(req("id:102"),"//arr[@name='str_s']/str[.='EMPTY']");
209     assertQ(req("id:103"),"//arr[@name='str_s']/str[.='EMPTY']");
210 
211     // test indexing empty fields
212     loadLocal("commit","true", "f.str_s.keepEmpty","true");
213     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
214     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='quoted']");
215     assertQ(req("id:101"),"//arr[@name='str_s']/str[.='']");
216     assertQ(req("id:102"),"//arr[@name='str_s']/str[.='']");
217     assertQ(req("id:103"),"//arr[@name='str_s']/str[.='']");
218 
219     // test overriding the name of fields
220     loadLocal("commit","true",
221              "fieldnames","id,my_s", "header","true",
222              "f.my_s.map",":EMPTY");
223     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
224     assertQ(req("id:100"),"//arr[@name='my_s']/str[.='quoted']");
225     assertQ(req("id:101"),"count(//arr[@name='str_s']/str)=0");
226     assertQ(req("id:102"),"count(//arr[@name='str_s']/str)=0");
227     assertQ(req("id:103"),"count(//arr[@name='str_s']/str)=0");
228     assertQ(req("id:101"),"//arr[@name='my_s']/str[.='EMPTY']");
229     assertQ(req("id:102"),"//arr[@name='my_s']/str[.='EMPTY']");
230     assertQ(req("id:103"),"//arr[@name='my_s']/str[.='EMPTY']");
231 
232     // test that header in file was skipped
233     assertQ(req("id:id"),"//*[@numFound='0']");
234 
235     // test skipping a field via the "skip" parameter
236     loadLocal("commit","true","keepEmpty","true","skip","str_s");
237     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
238     assertQ(req("id:[100 TO 110]"),"count(//str[@name='str_s']/str)=0");
239 
240     // test skipping a field by specifying an empty name
241     loadLocal("commit","true","keepEmpty","true","fieldnames","id,");
242     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
243     assertQ(req("id:[100 TO 110]"),"count(//str[@name='str_s']/str)=0");
244 
245     // test loading file as if it didn't have a header
246     loadLocal("commit","true",
247              "fieldnames","id,my_s", "header","false");
248     assertQ(req("id:id"),"//*[@numFound='1']");
249     assertQ(req("id:100"),"//arr[@name='my_s']/str[.='quoted']");
250 
251     // test skipLines
252     loadLocal("commit","true",
253              "fieldnames","id,my_s", "header","false", "skipLines","1");
254     assertQ(req("id:id"),"//*[@numFound='1']");
255     assertQ(req("id:100"),"//arr[@name='my_s']/str[.='quoted']");
256 
257 
258     // test multi-valued fields via field splitting w/ mapping of subvalues
259     makeFile("id,str_s\n"
260             +"100,\"quoted\"\n"
261             +"101,\"a,b,c\"\n"
262             +"102,\"a,,b\"\n"
263             +"103,\n");
264     loadLocal("commit","true",
265               "f.str_s.map",":EMPTY",
266               "f.str_s.split","true");
267     assertQ(req("id:[100 TO 110]"),"//*[@numFound='4']");
268     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='quoted']");
269     assertQ(req("id:101"),"//arr[@name='str_s']/str[1][.='a']");
270     assertQ(req("id:101"),"//arr[@name='str_s']/str[2][.='b']");
271     assertQ(req("id:101"),"//arr[@name='str_s']/str[3][.='c']");
272     assertQ(req("id:102"),"//arr[@name='str_s']/str[2][.='EMPTY']");
273     assertQ(req("id:103"),"//arr[@name='str_s']/str[.='EMPTY']");
274 
275 
276     // test alternate values for delimiters
277     makeFile("id|str_s\n"
278             +"100|^quoted^\n"
279             +"101|a;'b';c\n"
280             +"102|a;;b\n"
281             +"103|\n"
282             +"104|a\\\\b\n"  // no backslash escaping should be done by default
283     );
284 
285     loadLocal("commit","true",
286               "separator","|",
287               "encapsulator","^",
288               "f.str_s.map",":EMPTY",
289               "f.str_s.split","true",
290               "f.str_s.separator",";",
291               "f.str_s.encapsulator","'"
292     );
293     assertQ(req("id:[100 TO 110]"),"//*[@numFound='5']");
294     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='quoted']");
295     assertQ(req("id:101"),"//arr[@name='str_s']/str[1][.='a']");
296     assertQ(req("id:101"),"//arr[@name='str_s']/str[2][.='b']");
297     assertQ(req("id:101"),"//arr[@name='str_s']/str[3][.='c']");
298     assertQ(req("id:102"),"//arr[@name='str_s']/str[2][.='EMPTY']");
299     assertQ(req("id:103"),"//arr[@name='str_s']/str[.='EMPTY']");
300     assertQ(req("id:104"),"//arr[@name='str_s']/str[.='a\\\\b']");
301 
302     // test no escaping + double encapsulator escaping by default
303     makeFile("id,str_s\n"
304             +"100,\"quoted \"\" \\ string\"\n"
305             +"101,unquoted \"\" \\ string\n"     // double encap shouldn't be an escape outside encap
306             +"102,end quote \\\n"
307     );
308     loadLocal("commit","true"
309     );
310     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='quoted \" \\ string']");
311     assertQ(req("id:101"),"//arr[@name='str_s']/str[.='unquoted \"\" \\ string']");
312     assertQ(req("id:102"),"//arr[@name='str_s']/str[.='end quote \\']");
313 
314 
315     // setting an escape should disable encapsulator
316     makeFile("id,str_s\n"
317             +"100,\"quoted \"\" \\\" \\\\ string\"\n"  // quotes should be part of value
318             +"101,unquoted \"\" \\\" \\, \\\\ string\n"
319     );
320     loadLocal("commit","true"
321             ,"escape","\\"
322     );
323     assertQ(req("id:100"),"//arr[@name='str_s']/str[.='\"quoted \"\" \" \\ string\"']");
324     assertQ(req("id:101"),"//arr[@name='str_s']/str[.='unquoted \"\" \" , \\ string']");
325 
326   }
