121   @Test
122   public void testAttributes2LevelHetero() {
123     String xml="<root>\n"
124              + "<a>\n   <b a=\"x0\" b=\"y0\" />\n"
125              + "        <b a=\"x1\" b=\"y1\" />\n"
126              + "        <b a=\"x2\" b=\"y2\" />\n"
127              + "        </a>"
128              + "<x>\n   <b a=\"x4\" b=\"y4\" />\n"
129              + "        <b a=\"x5\" b=\"y5\" />\n"
130              + "        <b a=\"x6\" b=\"y6\" />\n"
131              + "        </x>"
132              + "</root>";
133     XPathRecordReader rr = new XPathRecordReader("/root/a | /root/x");
134     rr.addField("a", "/root/a/b/@a", false);
135     rr.addField("b", "/root/a/b/@b", false);
136     rr.addField("a", "/root/x/b/@a", false);
137     rr.addField("b", "/root/x/b/@b", false);
138 
139     final List<Map<String, Object>> a = new ArrayList<>();
140     final List<Map<String, Object>> x = new ArrayList<>();
141     rr.streamRecords(new StringReader(xml), new XPathRecordReader.Handler() {
142       @Override
143       public void handle(Map<String, Object> record, String xpath) {
144         if (record == null) return;
145         if (xpath.equals("/root/a")) a.add(record);
146         if (xpath.equals("/root/x")) x.add(record);
147       }
148     });
149 
150     assertEquals(1, a.size());
151     assertEquals(1, x.size());
152   }

222   public void testMixedContent() {
223     String xml = "<xhtml:p xmlns:xhtml=\"http://xhtml.com/\" >This text is \n" +
224             "  <xhtml:b>bold</xhtml:b> and this text is \n" +
225             "  <xhtml:u>underlined</xhtml:u>!\n" +
226             "</xhtml:p>";
227     XPathRecordReader rr = new XPathRecordReader("/p");
228     rr.addField("p", "/p", true);
229     rr.addField("b", "/p/b", true);
230     rr.addField("u", "/p/u", true);
231     List<Map<String, Object>> l = rr.getAllRecords(new StringReader(xml));
232     Map<String, Object> row = l.get(0);
233 
234     assertEquals("bold", ((List) row.get("b")).get(0));
235     assertEquals("underlined", ((List) row.get("u")).get(0));
236     String p = (String) ((List) row.get("p")).get(0);
237     assertTrue(p.contains("This text is"));
238     assertTrue(p.contains("and this text is"));
239     assertTrue(p.contains("!"));
240     // Should not contain content from child elements
241     assertFalse(p.contains("bold"));
242   }
