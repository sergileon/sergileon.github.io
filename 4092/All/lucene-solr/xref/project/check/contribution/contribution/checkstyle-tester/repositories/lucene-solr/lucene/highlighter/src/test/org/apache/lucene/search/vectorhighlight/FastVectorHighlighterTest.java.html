277   public void testBoostedPhraseHighlightTest() throws IOException {
278     Directory dir = newDirectory();
279     IndexWriter writer = new IndexWriter( dir, newIndexWriterConfig(new MockAnalyzer( random() ) ) );
280     Document doc = new Document();
281     FieldType type = new FieldType( TextField.TYPE_STORED  );
282     type.setStoreTermVectorOffsets( true );
283     type.setStoreTermVectorPositions( true );
284     type.setStoreTermVectors( true );
285     type.freeze();
286     StringBuilder text = new StringBuilder();
287     text.append("words words junk junk junk junk junk junk junk junk highlight junk junk junk junk together junk ");
288     for ( int i = 0; i<10; i++ ) {
289       text.append("junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk ");
290     }
291     text.append("highlight words together ");
292     for ( int i = 0; i<10; i++ ) {
293       text.append("junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk junk ");
294     }
295     doc.add( new Field( "text", text.toString().trim(), type ) );
296     writer.addDocument(doc);
297     FastVectorHighlighter highlighter = new FastVectorHighlighter();
298     IndexReader reader = DirectoryReader.open(writer, true);
299 
300     // This mimics what some query parsers do to <highlight words together>
301     BooleanQuery.Builder terms = new BooleanQuery.Builder();
302     terms.add( clause( "text", "highlight" ), Occur.MUST );
303     terms.add( clause( "text", "words" ), Occur.MUST );
304     terms.add( clause( "text", "together" ), Occur.MUST );
305     // This mimics what some query parsers do to <"highlight words together">
306     BooleanQuery.Builder phraseB = new BooleanQuery.Builder();
307     phraseB.add( clause( "text", "highlight", "words", "together" ), Occur.MUST );
308     Query phrase = phraseB.build();
309     phrase = new BoostQuery(phrase, 100f);
310     // Now combine those results in a boolean query which should pull the phrases to the front of the list of fragments 
311     BooleanQuery.Builder query = new BooleanQuery.Builder();
312     query.add( phrase, Occur.MUST );
313     query.add( phrase, Occur.SHOULD );
314     FieldQuery fieldQuery = new FieldQuery( query.build(), reader, true, false );
315     String fragment = highlighter.getBestFragment( fieldQuery, reader, 0, "text", 100 );
316     assertEquals( "junk junk junk junk junk junk junk junk <b>highlight words together</b> junk junk junk junk junk junk junk junk", fragment );
317 
318     reader.close();
319     writer.close();
320     dir.close();
321   }

323   public void testCommonTermsQueryHighlight() throws IOException {
324     Directory dir = newDirectory();
325     IndexWriter writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random(), MockTokenizer.SIMPLE, true, MockTokenFilter.ENGLISH_STOPSET)));
326     FieldType type = new FieldType(TextField.TYPE_STORED);
327     type.setStoreTermVectorOffsets(true);
328     type.setStoreTermVectorPositions(true);
329     type.setStoreTermVectors(true);
330     type.freeze();
331     String[] texts = {
332         "Hello this is a piece of text that is very long and contains too much preamble and the meat is really here which says kennedy has been shot",
333         "This piece of text refers to Kennedy at the beginning then has a longer piece of text that is very long in the middle and finally ends with another reference to Kennedy",
334         "JFK has been shot", "John Kennedy has been shot",
335         "This text has a typo in referring to Keneddy",
336         "wordx wordy wordz wordx wordy wordx worda wordb wordy wordc", "y z x y z a b", "lets is a the lets is a the lets is a the lets" };
337     for (int i = 0; i < texts.length; i++) {
338       Document doc = new Document();
339       Field field = new Field("field", texts[i], type);
340       doc.add(field);
341       writer.addDocument(doc);
342     }
343     CommonTermsQuery query = new CommonTermsQuery(Occur.MUST, Occur.SHOULD, 2);
344     query.add(new Term("field", "text"));
345     query.add(new Term("field", "long"));
346     query.add(new Term("field", "very"));
347    
348     FastVectorHighlighter highlighter = new FastVectorHighlighter();
349     IndexReader reader = DirectoryReader.open(writer, true);
350     IndexSearcher searcher = newSearcher(reader);
351     TopDocs hits = searcher.search(query, 10);
352     assertEquals(2, hits.totalHits);
353     FieldQuery fieldQuery  = highlighter.getFieldQuery(query, reader);
354     String[] bestFragments = highlighter.getBestFragments(fieldQuery, reader, hits.scoreDocs[0].doc, "field", 1000, 1);
355     assertEquals("This piece of <b>text</b> refers to Kennedy at the beginning then has a longer piece of <b>text</b> that is <b>very</b> <b>long</b> in the middle and finally ends with another reference to Kennedy", bestFragments[0]);
356 
357     fieldQuery  = highlighter.getFieldQuery(query, reader);
358     bestFragments = highlighter.getBestFragments(fieldQuery, reader, hits.scoreDocs[1].doc, "field", 1000, 1);
359     assertEquals("Hello this is a piece of <b>text</b> that is <b>very</b> <b>long</b> and contains too much preamble and the meat is really here which says kennedy has been shot", bestFragments[0]);
360 
361     reader.close();
362     writer.close();
363     dir.close();
364   }

366   public void testMatchedFields() throws IOException {
367     // Searching just on the stored field doesn't highlight a stopword
368     matchedFieldsTestCase( false, true, "a match", "a <b>match</b>",
369       clause( "field", "a" ), clause( "field", "match" ) );
370 
371     // Even if you add an unqueried matched field that would match it
372     matchedFieldsTestCase( "a match", "a <b>match</b>",
373       clause( "field", "a" ), clause( "field", "match" ) );
374 
375     // Nor if you query the field but don't add it as a matched field to the highlighter
376     matchedFieldsTestCase( false, false, "a match", "a <b>match</b>",
377       clause( "field_exact", "a" ), clause( "field", "match" ) );
378 
379     // But if you query the field and add it as a matched field to the highlighter then it is highlighted
380     matchedFieldsTestCase( "a match", "<b>a</b> <b>match</b>",
381       clause( "field_exact", "a" ), clause( "field", "match" ) );
382 
383     // It is also ok to match just the matched field but get highlighting from the stored field
384     matchedFieldsTestCase( "a match", "<b>a</b> <b>match</b>",
385       clause( "field_exact", "a" ), clause( "field_exact", "match" ) );
386 
387     // Boosted matched fields work too
388     matchedFieldsTestCase( "a match", "<b>a</b> <b>match</b>",
389       clause( "field_exact", 5, "a" ), clause( "field", "match" ) );
390 
391     // It is also ok if both the stored and the matched field match the term
392     matchedFieldsTestCase( "a match", "a <b>match</b>",
393       clause( "field_exact", "match" ), clause( "field", "match" ) );
394 
395     // And the highlighter respects the boosts on matched fields when sorting fragments
396     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
397       "junk junk <b>a cat</b> junk junk",
398       clause( "field", "cat" ), clause( "field_exact", 5, "a", "cat" ) );
399     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
400       "<b>cat</b> <b>cat</b> junk junk junk junk",
401       clause( "field", "cat" ), clause( "field_exact", "a", "cat" ) );
402 
403     // The same thing works across three fields as well
404     matchedFieldsTestCase( "cat cat CAT junk junk junk junk junk junk junk a cat junk junk",
405       "junk junk <b>a cat</b> junk junk",
406       clause( "field", "cat" ), clause( "field_exact", 200, "a", "cat" ), clause( "field_super_exact", 5, "CAT" ) );
407     matchedFieldsTestCase( "a cat cat junk junk junk junk junk junk junk a CAT junk junk",
408       "junk junk <b>a CAT</b> junk junk",
409       clause( "field", "cat" ), clause( "field_exact", 5, "a", "cat" ), clause( "field_super_exact", 200, "a", "CAT" ) );
410 
411     // And across fields with different tokenizers!
412     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
413       "junk junk <b>a cat</b> junk junk",
414       clause( "field_exact", 5, "a", "cat" ), clause( "field_characters", "c" ) );
415     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
416       "<b>c</b>at <b>c</b>at junk junk junk junk",
417       clause( "field_exact", "a", "cat" ), clause( "field_characters", "c" ) );
418     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
419       "ca<b>t</b> ca<b>t</b> junk junk junk junk",
420       clause( "field_exact", "a", "cat" ), clause( "field_characters", "t" ) );
421     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
422       "<b>cat</b> <b>cat</b> junk junk junk junk", // See how the phrases are joined?
423       clause( "field", "cat" ), clause( "field_characters", 5, "c" ) );
424     matchedFieldsTestCase( "cat cat junk junk junk junk junk junk junk a cat junk junk",
425       "junk junk <b>a cat</b> junk junk",
426       clause( "field", "cat" ), clause( "field_characters", 5, "a", " ", "c", "a", "t" ) );
427 
428     // Phrases and tokens inside one another are joined
429     matchedFieldsTestCase( "cats wow", "<b>cats w</b>ow",
430       clause( "field", "cats" ), clause( "field_tripples", "s w" ) );
431 
432     // Everything works pretty well even if you don't require a field match
433     matchedFieldsTestCase( true, false, "cat cat junk junk junk junk junk junk junk a cat junk junk",
434       "junk junk <b>a cat</b> junk junk",
435       clause( "field", "cat" ), clause( "field_characters", 10, "a", " ", "c", "a", "t" ) );
436 
437     // Even boosts keep themselves pretty much intact
438     matchedFieldsTestCase( true, false, "a cat cat junk junk junk junk junk junk junk a CAT junk junk",
439       "junk junk <b>a CAT</b> junk junk",
440       clause( "field", "cat" ), clause( "field_exact", 5, "a", "cat" ), clause( "field_super_exact", 200, "a", "CAT" ) );
441     matchedFieldsTestCase( true, false, "cat cat CAT junk junk junk junk junk junk junk a cat junk junk",
442       "junk junk <b>a cat</b> junk junk",
443       clause( "field", "cat" ), clause( "field_exact", 200, "a", "cat" ), clause( "field_super_exact", 5, "CAT" ) );
444 
445     // Except that all the matched field matches apply even if they aren't mentioned in the query
446     // which can make for some confusing scoring.  This isn't too big a deal, just something you
447     // need to think about when you don't force a field match.
448     matchedFieldsTestCase( true, false, "cat cat junk junk junk junk junk junk junk a cat junk junk",
449       "<b>cat</b> <b>cat</b> junk junk junk junk",
450       clause( "field", "cat" ), clause( "field_characters", 4, "a", " ", "c", "a", "t" ) );
451

648   private void matchedFieldsTestCase( boolean useMatchedFields, boolean fieldMatch, String fieldValue, String expected, Query... queryClauses ) throws IOException {
649     Document doc = new Document();
650     FieldType stored = new FieldType( TextField.TYPE_STORED );
651     stored.setStoreTermVectorOffsets( true );
652     stored.setStoreTermVectorPositions( true );
653     stored.setStoreTermVectors( true );
654     stored.freeze();
655     FieldType matched = new FieldType( TextField.TYPE_NOT_STORED );
656     matched.setStoreTermVectorOffsets( true );
657     matched.setStoreTermVectorPositions( true );
658     matched.setStoreTermVectors( true );
659     matched.freeze();
660     doc.add( new Field( "field", fieldValue, stored ) );               // Whitespace tokenized with English stop words
661     doc.add( new Field( "field_exact", fieldValue, matched ) );        // Whitespace tokenized without stop words
662     doc.add( new Field( "field_super_exact", fieldValue, matched ) );  // Whitespace tokenized without toLower
663     doc.add( new Field( "field_characters", fieldValue, matched ) );   // Each letter is a token
664     doc.add( new Field( "field_tripples", fieldValue, matched ) );     // Every three letters is a token
665     doc.add( new Field( "field_sliced", fieldValue.substring( 0,       // Sliced at 10 chars then analyzed just like field
666       Math.min( fieldValue.length() - 1 , 10 ) ), matched ) );
667     doc.add( new Field( "field_der_red", new CannedTokenStream(        // Hacky field containing "der" and "red" at pos = 0
668           token( "der", 1, 0, 3 ),
669           token( "red", 0, 0, 3 )
670         ), matched ) );
671 
672     final Map<String, Analyzer> fieldAnalyzers = new TreeMap<>();
673     fieldAnalyzers.put( "field", new MockAnalyzer( random(), MockTokenizer.WHITESPACE, true, MockTokenFilter.ENGLISH_STOPSET ) );
674     fieldAnalyzers.put( "field_exact", new MockAnalyzer( random() ) );
675     fieldAnalyzers.put( "field_super_exact", new MockAnalyzer( random(), MockTokenizer.WHITESPACE, false ) );
676     fieldAnalyzers.put( "field_characters", new MockAnalyzer( random(), new CharacterRunAutomaton( new RegExp(".").toAutomaton() ), true ) );
677     fieldAnalyzers.put( "field_tripples", new MockAnalyzer( random(), new CharacterRunAutomaton( new RegExp("...").toAutomaton() ), true ) );
678     fieldAnalyzers.put( "field_sliced", fieldAnalyzers.get( "field" ) );
679     fieldAnalyzers.put( "field_der_red", fieldAnalyzers.get( "field" ) );  // This is required even though we provide a token stream
680     Analyzer analyzer = new DelegatingAnalyzerWrapper(Analyzer.PER_FIELD_REUSE_STRATEGY) {
681       public Analyzer getWrappedAnalyzer(String fieldName) {
682         return fieldAnalyzers.get( fieldName );
683       }
684     };
685 
686     Directory dir = newDirectory();
687     IndexWriter writer = new IndexWriter( dir, newIndexWriterConfig(analyzer));
688     writer.addDocument( doc );
689 
690     FastVectorHighlighter highlighter = new FastVectorHighlighter();
691     FragListBuilder fragListBuilder = new SimpleFragListBuilder();
692     FragmentsBuilder fragmentsBuilder = new ScoreOrderFragmentsBuilder();
693     IndexReader reader = DirectoryReader.open( writer, true );
694     String[] preTags = new String[] { "<b>" };
695     String[] postTags = new String[] { "</b>" };
696     Encoder encoder = new DefaultEncoder();
697     int docId = 0;
698     BooleanQuery.Builder query = new BooleanQuery.Builder();
699     for ( Query clause : queryClauses ) {
700       query.add( clause, Occur.MUST );
701     }
702     FieldQuery fieldQuery = new FieldQuery( query.build(), reader, true, fieldMatch );
703     String[] bestFragments;
704     if ( useMatchedFields ) {
705       Set< String > matchedFields = new HashSet<>();
706       matchedFields.add( "field" );
707       matchedFields.add( "field_exact" );
708       matchedFields.add( "field_super_exact" );
709       matchedFields.add( "field_characters" );
710       matchedFields.add( "field_tripples" );
711       matchedFields.add( "field_sliced" );
712       matchedFields.add( "field_der_red" );
713       bestFragments = highlighter.getBestFragments( fieldQuery, reader, docId, "field", matchedFields, 25, 1,
714         fragListBuilder, fragmentsBuilder, preTags, postTags, encoder );
715     } else {
716       bestFragments = highlighter.getBestFragments( fieldQuery, reader, docId, "field", 25, 1,
717         fragListBuilder, fragmentsBuilder, preTags, postTags, encoder );
718     }
719     assertEquals( expected, bestFragments[ 0 ] );
720 
721     reader.close();
722     writer.close();
723     dir.close();
724   }

730   private Query clause( String field, float boost, String... terms ) {
731     Query q;
732     if ( terms.length == 1 ) {
733       q = new TermQuery( new Term( field, terms[ 0 ] ) );
734     } else {
735       q = new PhraseQuery(field, terms);
736     }
737     q = new BoostQuery( q, boost );
738     return q;
