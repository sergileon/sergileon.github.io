96      TermQuery[] queries = new TermQuery[] {
97          new TermQuery(new Term("foo", "bar")),

101       List<BooleanClause> clauses = new ArrayList<>();
102       final int numClauses = random().nextInt(20);
103       for (int i = 0; i < numClauses; ++i) {
104         Query query = RandomPicks.randomFrom(random(), queries);
105         if (random().nextBoolean()) {
106           query = new BoostQuery(query, random().nextFloat());
107         }
108         Occur occur = RandomPicks.randomFrom(random(), Occur.values());
109         clauses.add(new BooleanClause(query, occur));
110       }
111 
112       final boolean disableCoord = random().nextBoolean();
113       final int minShouldMatch = random().nextInt(5);
114       BooleanQuery.Builder bq1Builder = new BooleanQuery.Builder();
115       bq1Builder.setDisableCoord(disableCoord);
116       bq1Builder.setMinimumNumberShouldMatch(minShouldMatch);

118         bq1Builder.add(clause);
119       }
120       final BooleanQuery bq1 = bq1Builder.build();
121

123       BooleanQuery.Builder bq2Builder = new BooleanQuery.Builder();
124       bq2Builder.setDisableCoord(disableCoord);
125       bq2Builder.setMinimumNumberShouldMatch(minShouldMatch);

216   public void testNullOrSubScorer() throws Throwable {
217     Directory dir = newDirectory();
218     RandomIndexWriter w = new RandomIndexWriter(random(), dir);
219     Document doc = new Document();
220     doc.add(newTextField("field", "a b c d", Field.Store.NO));
221     w.addDocument(doc);
222 
223     IndexReader r = w.getReader();
224     IndexSearcher s = newSearcher(r);
225     // this test relies upon coord being the default implementation,
226     // otherwise scores are different!
227     s.setSimilarity(new DefaultSimilarity());
228 
229     BooleanQuery.Builder q = new BooleanQuery.Builder();
230     q.add(new TermQuery(new Term("field", "a")), BooleanClause.Occur.SHOULD);
231 
232     // LUCENE-2617: make sure that a term not in the index still contributes to the score via coord factor
233     float score = s.search(q.build(), 10).getMaxScore();
234     Query subQuery = new BoostQuery(new TermQuery(new Term("field", "not_in_index")), 0f);
235     q.add(subQuery, BooleanClause.Occur.SHOULD);
236     float score2 = s.search(q.build(), 10).getMaxScore();
237     assertEquals(score*.5F, score2, 1e-6);
238 
239     // LUCENE-2617: make sure that a clause not in the index still contributes to the score via coord factor
240     BooleanQuery.Builder qq = new BooleanQuery.Builder();
241     for (BooleanClause clause : q.build()) {
242       qq.add(clause);
243     }
244     PhraseQuery phrase = new PhraseQuery("field", "not_in_index", "another_not_in_index");
245     qq.add(new BoostQuery(phrase, 0f), BooleanClause.Occur.SHOULD);
246     score2 = s.search(qq.build(), 10).getMaxScore();
247     assertEquals(score*(1/3F), score2, 1e-6);
248 
249     // now test BooleanScorer2
250     subQuery = new BoostQuery(new TermQuery(new Term("field", "b")), 0f);
251     q.add(subQuery, BooleanClause.Occur.MUST);
252     score2 = s.search(q.build(), 10).getMaxScore();
253     assertEquals(score*(2/3F), score2, 1e-6);
254 
255     // PhraseQuery w/ no terms added returns a null scorer
256     PhraseQuery pq = new PhraseQuery("field", new String[0]);
257     q.add(pq, BooleanClause.Occur.SHOULD);
258     assertEquals(1, s.search(q.build(), 10).totalHits);
259 
260     // A required clause which returns null scorer should return null scorer to
261     // IndexSearcher.
262     q = new BooleanQuery.Builder();
263     pq = new PhraseQuery("field", new String[0]);
264     q.add(new TermQuery(new Term("field", "a")), BooleanClause.Occur.SHOULD);
265     q.add(pq, BooleanClause.Occur.MUST);
266     assertEquals(0, s.search(q.build(), 10).totalHits);
267 
268     DisjunctionMaxQuery dmq = new DisjunctionMaxQuery(
269         Arrays.asList(new TermQuery(new Term("field", "a")), pq),
270         1.0f);
271     assertEquals(1, s.search(dmq, 10).totalHits);
272 
273     r.close();
274     w.close();
275     dir.close();
276   }

320   public void testBS2DisjunctionNextVsAdvance() throws Exception {
321     final Directory d = newDirectory();
322     final RandomIndexWriter w = new RandomIndexWriter(random(), d);
323     final int numDocs = atLeast(300);
324     for(int docUpto=0;docUpto<numDocs;docUpto++) {
325       String contents = "a";
326       if (random().nextInt(20) <= 16) {
327         contents += " b";
328       }
329       if (random().nextInt(20) <= 8) {
330         contents += " c";
331       }
332       if (random().nextInt(20) <= 4) {
333         contents += " d";
334       }
335       if (random().nextInt(20) <= 2) {
336         contents += " e";
337       }
338       if (random().nextInt(20) <= 1) {
339         contents += " f";
340       }
341       Document doc = new Document();
342       doc.add(new TextField("field", contents, Field.Store.NO));
343       w.addDocument(doc);
344     }
345     w.forceMerge(1);
346     final IndexReader r = w.getReader();
347     final IndexSearcher s = newSearcher(r);
348     w.close();
349 
350     for(int iter=0;iter<10*RANDOM_MULTIPLIER;iter++) {
351       if (VERBOSE) {
352         System.out.println("iter=" + iter);
353       }
354       final List<String> terms = new ArrayList<>(Arrays.asList("a", "b", "c", "d", "e", "f"));
355       final int numTerms = TestUtil.nextInt(random(), 1, terms.size());
356       while(terms.size() > numTerms) {
357         terms.remove(random().nextInt(terms.size()));
358       }
359 
360       if (VERBOSE) {
361         System.out.println("  terms=" + terms);
362       }
363 
364       final BooleanQuery.Builder q = new BooleanQuery.Builder();
365       for(String term : terms) {
366         q.add(new BooleanClause(new TermQuery(new Term("field", term)), BooleanClause.Occur.SHOULD));
367       }
368 
369       Weight weight = s.createNormalizedWeight(q.build(), true);
370 
371       Scorer scorer = weight.scorer(s.leafContexts.get(0));
372 
373       // First pass: just use .nextDoc() to gather all hits
374       final List<ScoreDoc> hits = new ArrayList<>();
375       while(scorer.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
376         hits.add(new ScoreDoc(scorer.docID(), scorer.score()));
377       }
378 
379       if (VERBOSE) {
380         System.out.println("  " + hits.size() + " hits");
381       }
382 
383       // Now, randomly next/advance through the list and
384       // verify exact match:
385       for(int iter2=0;iter2<10;iter2++) {
386 
387         weight = s.createNormalizedWeight(q.build(), true);
388         scorer = weight.scorer(s.leafContexts.get(0));
389 
390         if (VERBOSE) {
391           System.out.println("  iter2=" + iter2);
392         }
393 
394         int upto = -1;
395         while(upto < hits.size()) {
396           final int nextUpto;
397           final int nextDoc;
398           final int left = hits.size() - upto;
399           if (left == 1 || random().nextBoolean()) {
400             // next
401             nextUpto = 1+upto;
402             nextDoc = scorer.nextDoc();
403           } else {
404             // advance
405             int inc = TestUtil.nextInt(random(), 1, left - 1);
406             nextUpto = inc + upto;
407             nextDoc = scorer.advance(hits.get(nextUpto).doc);
408           }
409 
410           if (nextUpto == hits.size()) {
411             assertEquals(DocIdSetIterator.NO_MORE_DOCS, nextDoc);
412           } else {
413             final ScoreDoc hit = hits.get(nextUpto);
414             assertEquals(hit.doc, nextDoc);
415             // Test for precise float equality:
416             assertTrue("doc " + hit.doc + " has wrong score: expected=" + hit.score + " actual=" + scorer.score(), hit.score == scorer.score());
417           }
418           upto = nextUpto;
419         }
420       }
421     }
422 
423     r.close();
424     d.close();
425   }

510   private static BitSet getMatches(IndexSearcher searcher, Query query) throws IOException {
511     final BitSet set = new BitSet();
512     searcher.search(query, new SimpleCollector() {
513       int docBase = 0;
514       @Override
515       public boolean needsScores() {
516         return random().nextBoolean();
517       }
518       @Override
519       protected void doSetNextReader(LeafReaderContext context)
520           throws IOException {
521         super.doSetNextReader(context);
522         docBase = context.docBase;
523       }
524       @Override
525       public void collect(int doc) throws IOException {
526         set.set(docBase + doc);
527       }
528     });
529     return set;
530   }

705     query2.add(new TermQuery(new Term("field", "b")), Occur.SHOULD);
706     final Weight weight = searcher.createNormalizedWeight(query2.build(), true);
707     final Scorer scorer = weight.scorer(reader.leaves().get(0));
708     assertEquals(0, scorer.nextDoc());

710     assertEquals(0f, scorer.score(), 0f);
