20  import org.apache.lucene.analysis.Analyzer;
21  import org.apache.lucene.analysis.AnalyzerWrapper;
22  import org.apache.lucene.analysis.standard.StandardAnalyzer;
23  
24  /**
25   * A ShingleAnalyzerWrapper wraps a {@link ShingleFilter} around another {@link Analyzer}.
26   * <p>
27   * A shingle is another name for a token based n-gram.
28   * </p>
29   */
30  public final class ShingleAnalyzerWrapper extends AnalyzerWrapper {
31  
32    private final Analyzer delegate;
33    private final int maxShingleSize;
34    private final int minShingleSize;
35    private final String tokenSeparator;
36    private final boolean outputUnigrams;
37    private final boolean outputUnigramsIfNoShingles;
38    private final String fillerToken;
39  
40    public ShingleAnalyzerWrapper(Analyzer defaultAnalyzer) {
41      this(defaultAnalyzer, ShingleFilter.DEFAULT_MAX_SHINGLE_SIZE);
42    }
43  
44    public ShingleAnalyzerWrapper(Analyzer defaultAnalyzer, int maxShingleSize) {
45      this(defaultAnalyzer, ShingleFilter.DEFAULT_MIN_SHINGLE_SIZE, maxShingleSize);
46    }
47  
48    public ShingleAnalyzerWrapper(Analyzer defaultAnalyzer, int minShingleSize, int maxShingleSize) {
49      this(defaultAnalyzer, minShingleSize, maxShingleSize, ShingleFilter.DEFAULT_TOKEN_SEPARATOR,
50           true, false, ShingleFilter.DEFAULT_FILLER_TOKEN);
51    }
52  
53    /**
54     * Creates a new ShingleAnalyzerWrapper
55     *
56     * @param delegate Analyzer whose TokenStream is to be filtered
57     * @param minShingleSize Min shingle (token ngram) size
58     * @param maxShingleSize Max shingle size
59     * @param tokenSeparator Used to separate input stream tokens in output shingles
60     * @param outputUnigrams Whether or not the filter shall pass the original
61     *        tokens to the output stream
62     * @param outputUnigramsIfNoShingles Overrides the behavior of outputUnigrams==false for those
63     *        times when no shingles are available (because there are fewer than
64     *        minShingleSize tokens in the input stream)?
65     *        Note that if outputUnigrams==true, then unigrams are always output,
66     *        regardless of whether any shingles are available.
67     * @param fillerToken filler token to use when positionIncrement is more than 1
68     */
69    public ShingleAnalyzerWrapper(
70        Analyzer delegate,
71        int minShingleSize,
72        int maxShingleSize,
73        String tokenSeparator,
74        boolean outputUnigrams,
75        boolean outputUnigramsIfNoShingles,
76        String fillerToken) {
77      super(delegate.getReuseStrategy());
78      this.delegate = delegate;
79  
80      if (maxShingleSize < 2) {
81        throw new IllegalArgumentException("Max shingle size must be >= 2");
82      }
83      this.maxShingleSize = maxShingleSize;
84  
85      if (minShingleSize < 2) {
86        throw new IllegalArgumentException("Min shingle size must be >= 2");
87      }
88      if (minShingleSize > maxShingleSize) {
89        throw new IllegalArgumentException
90          ("Min shingle size must be <= max shingle size");
91      }
92      this.minShingleSize = minShingleSize;
93  
94      this.tokenSeparator = (tokenSeparator == null ? "" : tokenSeparator);
95      this.outputUnigrams = outputUnigrams;
96      this.outputUnigramsIfNoShingles = outputUnigramsIfNoShingles;
97      this.fillerToken = fillerToken;
98    }
99  
100   /**
101    * Wraps {@link StandardAnalyzer}. 
102    */
103   public ShingleAnalyzerWrapper() {
104     this(ShingleFilter.DEFAULT_MIN_SHINGLE_SIZE, ShingleFilter.DEFAULT_MAX_SHINGLE_SIZE);
105   }
106 
107   /**
108    * Wraps {@link StandardAnalyzer}. 
109    */
110   public ShingleAnalyzerWrapper(int minShingleSize, int maxShingleSize) {
111     this(new StandardAnalyzer(), minShingleSize, maxShingleSize);
112   }
113 
114   /**
115    * The max shingle (token ngram) size
116    * 
117    * @return The max shingle (token ngram) size
118    */
119   public int getMaxShingleSize() {
120     return maxShingleSize;
121   }
122 
123   /**
124    * The min shingle (token ngram) size
125    * 
126    * @return The min shingle (token ngram) size
127    */
128   public int getMinShingleSize() {
129     return minShingleSize;
130   }
131 
132   public String getTokenSeparator() {
133     return tokenSeparator;
134   }
135   
136   public boolean isOutputUnigrams() {
137     return outputUnigrams;
138   }
139   
140   public boolean isOutputUnigramsIfNoShingles() {
141     return outputUnigramsIfNoShingles;
142   }
143 
144   public String getFillerToken() {
145     return fillerToken;
146   }
147 
148   @Override
149   public final Analyzer getWrappedAnalyzer(String fieldName) {
150     return delegate;
151   }
152 
153   @Override
154   protected TokenStreamComponents wrapComponents(String fieldName, TokenStreamComponents components) {
155     ShingleFilter filter = new ShingleFilter(components.getTokenStream(), minShingleSize, maxShingleSize);
156     filter.setMinShingleSize(minShingleSize);
157     filter.setMaxShingleSize(maxShingleSize);
158     filter.setTokenSeparator(tokenSeparator);
159     filter.setOutputUnigrams(outputUnigrams);
160     filter.setOutputUnigramsIfNoShingles(outputUnigramsIfNoShingles);
161     filter.setFillerToken(fillerToken);
162     return new TokenStreamComponents(components.getTokenizer(), filter);
163   }
164 }
