114   @Test
115   public void testVersions() throws Exception {
116     clearIndex();
117     assertU(commit());
118 
119     long version = addAndGetVersion(sdoc("id","1") , null);
120 
121     assertJQ(req("q","id:1")
122         ,"/response/numFound==0"
123     );
124 
125     // test version is there from rtg
126     assertJQ(req("qt","/get","id","1")
127         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
128     );
129 
130     // test version is there from the index
131     assertU(commit());
132     assertJQ(req("qt","/get","id","1")
133         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
134     );
135 
136     // simulate an update from the leader
137     version += 10;
138     updateJ(jsonAdd(sdoc("id","1", "_version_",Long.toString(version))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
139 
140     // test version is there from rtg
141     assertJQ(req("qt","/get","id","1")
142         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
143     );
144 
145     // simulate reordering: test that a version less than that does not take affect
146     updateJ(jsonAdd(sdoc("id","1", "_version_",Long.toString(version - 1))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
147 
148     // test that version hasn't changed
149     assertJQ(req("qt","/get","id","1")
150         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
151     );
152 
153     // simulate reordering: test that a delete w/ version less than that does not take affect
154     // TODO: also allow passing version on delete instead of on URL?
155     updateJ(jsonDelId("1"), params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_",Long.toString(version - 1)));
156 
157     // test that version hasn't changed
158     assertJQ(req("qt","/get","id","1")
159         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
160     );
161 
162     // make sure reordering detection also works after a commit
163     assertU(commit());
164 
165     // simulate reordering: test that a version less than that does not take affect
166     updateJ(jsonAdd(sdoc("id","1", "_version_",Long.toString(version - 1))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
167 
168     // test that version hasn't changed
169     assertJQ(req("qt","/get","id","1")
170         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
171     );
172 
173     // simulate reordering: test that a delete w/ version less than that does not take affect
174     updateJ(jsonDelId("1"), params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_",Long.toString(version - 1)));
175 
176     // test that version hasn't changed
177     assertJQ(req("qt","/get","id","1")
178         ,"=={'doc':{'id':'1','_version_':" + version + "}}"
179     );
180 
181     // now simulate a normal delete from the leader
182     version += 5;
183     updateJ(jsonDelId("1"), params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_",Long.toString(version)));
184 
185     // make sure a reordered add doesn't take affect.
186     updateJ(jsonAdd(sdoc("id","1", "_version_",Long.toString(version - 1))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
187 
188     // test that it's still deleted
189     assertJQ(req("qt","/get","id","1")
190         ,"=={'doc':null}"
191     );
192 
193     // test that we can remember the version of a delete after a commit
194     assertU(commit());
195 
196     // make sure a reordered add doesn't take affect.
197     long version2 = deleteByQueryAndGetVersion("id:2", null);
198 
199     // test that it's still deleted
200     assertJQ(req("qt","/get","id","1")
201         ,"=={'doc':null}"
202     );
203 
204     version = addAndGetVersion(sdoc("id","2"), null);
205     version2 = deleteByQueryAndGetVersion("id:2", null);
206     assertTrue(Math.abs(version2) > version );
207 
208     // test that it's deleted
209     assertJQ(req("qt","/get","id","2")
210         ,"=={'doc':null}");
211 
212 
213     version2 = Math.abs(version2) + 1000;
214     updateJ(jsonAdd(sdoc("id","3", "_version_",Long.toString(version2+100))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
215     updateJ(jsonAdd(sdoc("id","4", "_version_",Long.toString(version2+200))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
216 
217     // this should only affect id:3 so far
218     deleteByQueryAndGetVersion("id:(3 4 5 6)", params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_",Long.toString(-(version2+150))) );
219 
220     assertJQ(req("qt","/get","id","3"),"=={'doc':null}");
221     assertJQ(req("qt","/get","id","4", "fl","id"),"=={'doc':{'id':'4'}}");
222 
223     updateJ(jsonAdd(sdoc("id","5", "_version_",Long.toString(version2+201))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
224     updateJ(jsonAdd(sdoc("id","6", "_version_",Long.toString(version2+101))), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
225 
226    // the DBQ should also have caused id:6 to be removed
227     assertJQ(req("qt","/get","id","5", "fl","id"),"=={'doc':{'id':'5'}}");
228     assertJQ(req("qt","/get","id","6"),"=={'doc':null}");
229 
230     assertU(commit());
231

401   @Test
402   public void testStressGetRealtime() throws Exception {
403     clearIndex();
404     assertU(commit());
405 
406     // req().getCore().getUpdateHandler().getIndexWriterProvider().getIndexWriter(req().getCore()).setInfoStream(System.out);
407 
408     final int commitPercent = 5 + random().nextInt(20);
409     final int softCommitPercent = 30+random().nextInt(75); // what percent of the commits are soft
410     final int deletePercent = 4+random().nextInt(25);
411     final int deleteByQueryPercent = 1+random().nextInt(5);
412     final int optimisticPercent = 1+random().nextInt(50);    // percent change that an update uses optimistic locking
413     final int optimisticCorrectPercent = 25+random().nextInt(70);    // percent change that a version specified will be correct
414     final int ndocs = 5 + (random().nextBoolean() ? random().nextInt(25) : random().nextInt(200));
415     int nWriteThreads = 5 + random().nextInt(25);
416 
417     final int maxConcurrentCommits = nWriteThreads;   // number of committers at a time... it should be <= maxWarmingSearchers
418 
419         // query variables
420     final int percentRealtimeQuery = 60;
421     final AtomicLong operations = new AtomicLong(50000);  // number of query operations to perform in total
422     int nReadThreads = 5 + random().nextInt(25);
423 
424 
425     verbose("commitPercent=", commitPercent);
426     verbose("softCommitPercent=",softCommitPercent);
427     verbose("deletePercent=",deletePercent);
428     verbose("deleteByQueryPercent=", deleteByQueryPercent);
429     verbose("ndocs=", ndocs);
430     verbose("nWriteThreads=", nWriteThreads);
431     verbose("nReadThreads=", nReadThreads);
432     verbose("percentRealtimeQuery=", percentRealtimeQuery);
433     verbose("maxConcurrentCommits=", maxConcurrentCommits);
434     verbose("operations=", operations);
435 
436 
437     initModel(ndocs);
438 
439     final AtomicInteger numCommitting = new AtomicInteger();
440 
441     List<Thread> threads = new ArrayList<>();
442 
443     for (int i=0; i<nWriteThreads; i++) {
444       Thread thread = new Thread("WRITER"+i) {
445         Random rand = new Random(random().nextInt());
446 
447         @Override
448         public void run() {
449           try {
450           while (operations.get() > 0) {
451             int oper = rand.nextInt(100);
452 
453             if (oper < commitPercent) {
454               if (numCommitting.incrementAndGet() <= maxConcurrentCommits) {
455                 Map<Integer,DocInfo> newCommittedModel;
456                 long version;
457 
458                 synchronized(TestRealTimeGet.this) {
459                   newCommittedModel = new HashMap<>(model);  // take a snapshot
460                   version = snapshotCount++;
461                   verbose("took snapshot version=",version);
462                 }
463 
464                 if (rand.nextInt(100) < softCommitPercent) {
465                   verbose("softCommit start");
466                   assertU(TestHarness.commit("softCommit","true"));
467                   verbose("softCommit end");
468                 } else {
469                   verbose("hardCommit start");
470                   assertU(commit());
471                   verbose("hardCommit end");
472                 }
473 
474                 synchronized(TestRealTimeGet.this) {
475                   // install this model snapshot only if it's newer than the current one
476                   if (version >= committedModelClock) {
477                     if (VERBOSE) {
478                       verbose("installing new committedModel version="+committedModelClock);
479                     }
480                     committedModel = newCommittedModel;
481                     committedModelClock = version;
482                   }
483                 }
484               }
485               numCommitting.decrementAndGet();
486               continue;
487             }
488 
489 
490             int id = rand.nextInt(ndocs);
491             Object sync = syncArr[id];
492 
493             // set the lastId before we actually change it sometimes to try and
494             // uncover more race conditions between writing and reading
495             boolean before = rand.nextBoolean();
496             if (before) {
497               lastId = id;
498             }
499 
500             // We can't concurrently update the same document and retain our invariants of increasing values
501             // since we can't guarantee what order the updates will be executed.
502             // Even with versions, we can't remove the sync because increasing versions does not mean increasing vals.
503             synchronized (sync) {
504               DocInfo info = model.get(id);
505 
506               long val = info.val;
507               long nextVal = Math.abs(val)+1;
508 
509               if (oper < commitPercent + deletePercent) {
510                 boolean opt = rand.nextInt() < optimisticPercent;
511                 boolean correct = opt ? rand.nextInt() < optimisticCorrectPercent : false;
512                 long badVersion = correct ? 0 : badVersion(rand, info.version);
513 
514                 if (VERBOSE) {
515                   if (!opt) {
516                     verbose("deleting id",id,"val=",nextVal);
517                   } else {
518                     verbose("deleting id",id,"val=",nextVal, "existing_version=",info.version,  (correct ? "" : (" bad_version=" + badVersion)));
519                   }
520                 }
521 
522                 // assertU("<delete><id>" + id + "</id></delete>");
523                 Long version = null;
524 
525                 if (opt) {
526                   if (correct) {
527                     version = deleteAndGetVersion(Integer.toString(id), params("_version_", Long.toString(info.version)));
528                   } else {
529                     try {
530                       version = deleteAndGetVersion(Integer.toString(id), params("_version_", Long.toString(badVersion)));
531                       fail();
532                     } catch (SolrException se) {
533                       assertEquals(409, se.code());
534                     }
535                   }
536                 } else {
537                   version = deleteAndGetVersion(Integer.toString(id), null);
538                 }
539 
540                 if (version != null) {
541                   model.put(id, new DocInfo(version, -nextVal));
542                 }
543 
544                 if (VERBOSE) {
545                   verbose("deleting id", id, "val=",nextVal,"DONE");
546                 }
547               } else if (oper < commitPercent + deletePercent + deleteByQueryPercent) {
548                 if (VERBOSE) {
549                   verbose("deleteByQuery id ",id, "val=",nextVal);
550                 }
551 
552                 assertU("<delete><query>id:" + id + "</query></delete>");
553                 model.put(id, new DocInfo(-1L, -nextVal));
554                 if (VERBOSE) {
555                   verbose("deleteByQuery id",id, "val=",nextVal,"DONE");
556                 }
557               } else {
558                 boolean opt = rand.nextInt() < optimisticPercent;
559                 boolean correct = opt ? rand.nextInt() < optimisticCorrectPercent : false;
560                 long badVersion = correct ? 0 : badVersion(rand, info.version);
561 
562                 if (VERBOSE) {
563                   if (!opt) {
564                     verbose("adding id",id,"val=",nextVal);
565                   } else {
566                     verbose("adding id",id,"val=",nextVal, "existing_version=",info.version,  (correct ? "" : (" bad_version=" + badVersion)));
567                   }
568                 }
569 
570                 Long version = null;
571                 SolrInputDocument sd = sdoc("id", Integer.toString(id), field, Long.toString(nextVal));
572 
573                 if (opt) {
574                   if (correct) {
575                     version = addAndGetVersion(sd, params("_version_", Long.toString(info.version)));
576                   } else {
577                     try {
578                       version = addAndGetVersion(sd, params("_version_", Long.toString(badVersion)));
579                       fail();
580                     } catch (SolrException se) {
581                       assertEquals(409, se.code());
582                     }
583                   }
584                 } else {
585                   version = addAndGetVersion(sd, null);
586                 }
587 
588 
589                 if (version != null) {
590                   model.put(id, new DocInfo(version, nextVal));
591                 }
592 
593                 if (VERBOSE) {
594                   verbose("adding id", id, "val=", nextVal,"DONE");
595                 }
596 
597               }
598             }   // end sync
599 
600             if (!before) {
601               lastId = id;
602             }
603           }
604         } catch (Throwable e) {
605           operations.set(-1L);
606           throw new RuntimeException(e);
607         }
608         }
609       };
610 
611       threads.add(thread);
612     }
613 
614 
615     for (int i=0; i<nReadThreads; i++) {
616       Thread thread = new Thread("READER"+i) {
617         Random rand = new Random(random().nextInt());
618 
619         @Override
620         public void run() {
621           try {
622             while (operations.decrementAndGet() >= 0) {
623               // bias toward a recently changed doc
624               int id = rand.nextInt(100) < 25 ? lastId : rand.nextInt(ndocs);
625 
626               // when indexing, we update the index, then the model
627               // so when querying, we should first check the model, and then the index
628 
629               boolean realTime = rand.nextInt(100) < percentRealtimeQuery;
630               DocInfo info;
631 
632               if (realTime) {
633                 info = model.get(id);
634               } else {
635                 synchronized(TestRealTimeGet.this) {
636                   info = committedModel.get(id);
637                 }
638               }
639 
640               if (VERBOSE) {
641                 verbose("querying id", id);
642               }
643               SolrQueryRequest sreq;
644               if (realTime) {
645                 sreq = req("wt","json", "qt","/get", "ids",Integer.toString(id));
646               } else {
647                 sreq = req("wt","json", "q","id:"+Integer.toString(id), "omitHeader","true");
648               }
649 
650               String response = h.query(sreq);
651               Map rsp = (Map)ObjectBuilder.fromJSON(response);
652               List doclist = (List)(((Map)rsp.get("response")).get("docs"));
653               if (doclist.size() == 0) {
654                 // there's no info we can get back with a delete, so not much we can check without further synchronization
655               } else {
656                 assertEquals(1, doclist.size());
657                 long foundVal = (Long)(((Map)doclist.get(0)).get(field));
658                 long foundVer = (Long)(((Map)doclist.get(0)).get("_version_"));
659                 if (foundVal < Math.abs(info.val)
660                     || (foundVer == info.version && foundVal != info.val) ) {    // if the version matches, the val must
661                   verbose("ERROR, id=", id, "found=",response,"model",info);
662                   assertTrue(false);
663                 }
664               }
665             }
666           }
667         catch (Throwable e) {
668           operations.set(-1L);
669           throw new RuntimeException(e);
