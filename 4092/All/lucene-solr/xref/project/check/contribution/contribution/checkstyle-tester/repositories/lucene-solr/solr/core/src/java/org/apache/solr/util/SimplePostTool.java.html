158    * See usage() for valid command line usage
159    * @param args the params on the command line
160    */
161   public static void main(String[] args) {
162     info("SimplePostTool version " + VERSION_OF_THIS_TOOL);
163     if (0 < args.length && ("-help".equals(args[0]) || "--help".equals(args[0]) || "-h".equals(args[0]))) {
164       usage();
165     } else {
166       final SimplePostTool t = parseArgsAndInit(args);
167       t.execute();
168     }
169   }

171   /**
172    * After initialization, call execute to start the post job.
173    * This method delegates to the correct mode method.
174    */
175   public void execute() {
176     final RTimer timer = new RTimer();
177     if (DATA_MODE_FILES.equals(mode) && args.length > 0) {
178       doFilesMode();
179     } else if(DATA_MODE_ARGS.equals(mode) && args.length > 0) {
180       doArgsMode();
181     } else if(DATA_MODE_WEB.equals(mode) && args.length > 0) {
182       doWebMode();
183     } else if(DATA_MODE_STDIN.equals(mode)) {
184       doStdinMode();
185     } else {
186       usageShort();
187       return;
188     }
189     
190     if (commit)   commit();
191     if (optimize) optimize();
192     displayTiming((long) timer.getTime());
193   }

205   /**
206    * Parses incoming arguments and system params and initializes the tool
207    * @param args the incoming cmd line args
208    * @return an instance of SimplePostTool
209    */
210   protected static SimplePostTool parseArgsAndInit(String[] args) {
211     String urlStr = null;
212     try {
213       // Parse args
214       final String mode = System.getProperty("data", DEFAULT_DATA_MODE);
215       if (! DATA_MODES.contains(mode)) {
216         fatal("System Property 'data' is not valid for this tool: " + mode);
217       }
218       
219       String params = System.getProperty("params", "");
220 
221       String host = System.getProperty("host", DEFAULT_POST_HOST);
222       String port = System.getProperty("port", DEFAULT_POST_PORT);
223       String core = System.getProperty("c");
224       
225       urlStr = System.getProperty("url");
226       
227       if (urlStr == null && core == null) {
228         fatal("Specifying either url or core/collection is mandatory.\n" + USAGE_STRING_SHORT);
229       }
230       
231       if(urlStr == null) {
232         urlStr = String.format(Locale.ROOT, "http://%s:%s/solr/%s/update", host, port, core);
233       }
234       urlStr = SimplePostTool.appendParam(urlStr, params);
235       URL url = new URL(urlStr);
236       boolean auto = isOn(System.getProperty("auto", DEFAULT_AUTO));
237       String type = System.getProperty("type");
238       // Recursive
239       int recursive = 0;
240       String r = System.getProperty("recursive", DEFAULT_RECURSIVE);
241       try {
242         recursive = Integer.parseInt(r);
243       } catch(Exception e) {
244         if (isOn(r))
245           recursive = DATA_MODE_WEB.equals(mode)?1:999;
246       }
247       // Delay
248       int delay = DATA_MODE_WEB.equals(mode) ? DEFAULT_WEB_DELAY : 0;
249       try {
250         delay = Integer.parseInt(System.getProperty("delay", ""+delay));
251       } catch(Exception e) { }
252       OutputStream out = isOn(System.getProperty("out", DEFAULT_OUT)) ? System.out : null;
253       String fileTypes = System.getProperty("filetypes", DEFAULT_FILE_TYPES);
254       boolean commit = isOn(System.getProperty("commit",DEFAULT_COMMIT));
255       boolean optimize = isOn(System.getProperty("optimize",DEFAULT_OPTIMIZE));
256       
257       return new SimplePostTool(mode, url, auto, type, recursive, delay, fileTypes, out, commit, optimize, args);
258     } catch (MalformedURLException e) {
259       fatal("System Property 'url' is not a valid URL: " + urlStr);
260       return null;
261     }
262   }

302   private void doFilesMode() {
303     currentDepth = 0;
304     // Skip posting files if special param "-" given  
305     if (!args[0].equals("-")) {
306       info("Posting files to [base] url " + solrUrl + (!auto?" using content-type "+(type==null?DEFAULT_CONTENT_TYPE:type):"")+"...");
307       if(auto)
308         info("Entering auto mode. File endings considered are "+fileTypes);
309       if(recursive > 0)
310         info("Entering recursive mode, max depth="+recursive+", delay="+delay+"s"); 
311       int numFilesPosted = postFiles(args, 0, out, type);
312       info(numFilesPosted + " files indexed.");
313     }
314   }

317     info("POSTing args to " + solrUrl + "...");
318     for (String a : args) {

320     }
321   }

323   private int doWebMode() {
324     reset();
325     int numPagesPosted = 0;
326     try {
327       if(type != null) {
328         fatal("Specifying content-type with \"-Ddata=web\" is not supported");
329       }
330       if (args[0].equals("-")) {
331         // Skip posting url if special param "-" given  
332         return 0;
333       }
334       // Set Extracting handler as default
335       solrUrl = appendUrlPath(solrUrl, "/extract");
336       
337       info("Posting web pages to Solr url "+solrUrl);
338       auto=true;
339       info("Entering auto mode. Indexing pages with content-types corresponding to file endings "+fileTypes);
340       if(recursive > 0) {
341         if(recursive > MAX_WEB_DEPTH) {
342           recursive = MAX_WEB_DEPTH;
343           warn("Too large recursion depth for web mode, limiting to "+MAX_WEB_DEPTH+"...");
344         }
345         if(delay < DEFAULT_WEB_DELAY)
346           warn("Never crawl an external web site faster than every 10 seconds, your IP will probably be blocked");
347         info("Entering recursive mode, depth="+recursive+", delay="+delay+"s");
348       }
349       numPagesPosted = postWebPages(args, 0, out);
350       info(numPagesPosted + " web pages indexed.");
351     } catch(MalformedURLException e) {
352       fatal("Wrong URL trying to append /extract to "+solrUrl);
353     }
354     return numPagesPosted;
355   }

420   /** Post all filenames provided in args
421    * @param args array of file names
422    * @param startIndexInArgs offset to start
423    * @param out output stream to post data to
424    * @param type default content-type to use when posting (may be overridden in auto mode)
425    * @return number of files posted
426    * */
427   public int postFiles(String [] args,int startIndexInArgs, OutputStream out, String type) {
428     reset();
429     int filesPosted = 0;
430     for (int j = startIndexInArgs; j < args.length; j++) {
431       File srcFile = new File(args[j]);
432       if(srcFile.isDirectory() && srcFile.canRead()) {
433         filesPosted += postDirectory(srcFile, out, type);
434       } else if (srcFile.isFile() && srcFile.canRead()) {
435         filesPosted += postFiles(new File[] {srcFile}, out, type);
436       } else {
437         File parent = srcFile.getParentFile();
438         if(parent == null) parent = new File(".");
439         String fileGlob = srcFile.getName();
440         GlobFileFilter ff = new GlobFileFilter(fileGlob, false);
441         File[] files = parent.listFiles(ff);
442         if(files == null || files.length == 0) {
443           warn("No files or directories matching "+srcFile);
444           continue;          
445         }
446         filesPosted += postFiles(parent.listFiles(ff), out, type);
447       }
448     }
449     return filesPosted;
450   }

464         filesPosted += postDirectory(srcFile, out, type);
465       } else if (srcFile.isFile() && srcFile.canRead()) {
466         filesPosted += postFiles(new File[] {srcFile}, out, type);
467       } else {
468         File parent = srcFile.getParentFile();
469         if(parent == null) parent = new File(".");
470         String fileGlob = srcFile.getName();
471         GlobFileFilter ff = new GlobFileFilter(fileGlob, false);
472         File[] fileList = parent.listFiles(ff);
473         if(fileList == null || fileList.length == 0) {
474           warn("No files or directories matching "+srcFile);
475           continue;          
476         }
477         filesPosted += postFiles(fileList, out, type);
478       }
479     }
480     return filesPosted;
481   }

483   /**
484    * Posts a whole directory
485    * @return number of files posted total
486    */
487   private int postDirectory(File dir, OutputStream out, String type) {
488     if(dir.isHidden() && !dir.getName().equals("."))
489       return(0);
490     info("Indexing directory "+dir.getPath()+" ("+dir.listFiles(fileFilter).length+" files, depth="+currentDepth+")");
491     int posted = 0;
492     posted += postFiles(dir.listFiles(fileFilter), out, type);
493     if(recursive > currentDepth) {
494       for(File d : dir.listFiles()) {
495         if(d.isDirectory()) {
496           currentDepth++;
497           posted += postDirectory(d, out, type);
498           currentDepth--;
499         }
500       }
501     }
502     return posted;
503   }

505   /**
506    * Posts a list of file names
507    * @return number of files posted
508    */
509   int postFiles(File[] files, OutputStream out, String type) {
510     int filesPosted = 0;
511     for(File srcFile : files) {
512       try {
513         if(!srcFile.isFile() || srcFile.isHidden())
514           continue;
515         postFile(srcFile, out, type);
516         Thread.sleep(delay * 1000);
517         filesPosted++;
518       } catch (InterruptedException e) {
519         throw new RuntimeException(e);
520       }
521     }
522     return filesPosted;
523   }

525   /**
526    * This method takes as input a list of start URL strings for crawling,
527    * adds each one to a backlog and then starts crawling
528    * @param args the raw input args from main()
529    * @param startIndexInArgs offset for where to start
530    * @param out outputStream to write results to
531    * @return the number of web pages posted
532    */
533   public int postWebPages(String[] args, int startIndexInArgs, OutputStream out) {
534     reset();
535     LinkedHashSet<URL> s = new LinkedHashSet<>();
536     for (int j = startIndexInArgs; j < args.length; j++) {
537       try {
538         URL u = new URL(normalizeUrlEnding(args[j]));
539         s.add(u);
540       } catch(MalformedURLException e) {
541         warn("Skipping malformed input URL: "+args[j]);
542       }
543     }
544     // Add URLs to level 0 of the backlog and start recursive crawling
545     backlog.add(s);
546     return webCrawl(0, out);
547   }

549   /**
550    * Normalizes a URL string by removing anchor part and trailing slash
551    * @return the normalized URL string
552    */
553   protected static String normalizeUrlEnding(String link) {
554     if(link.indexOf("#") > -1)
555       link = link.substring(0,link.indexOf("#"));
556     if(link.endsWith("?"))
557       link = link.substring(0,link.length()-1);
558     if(link.endsWith("/"))
559       link = link.substring(0,link.length()-1);
560     return link;
561   }

563   /**
564    * A very simple crawler, pulling URLs to fetch from a backlog and then
565    * recurses N levels deep if recursive&gt;0. Links are parsed from HTML
566    * through first getting an XHTML version using SolrCell with extractOnly,
567    * and followed if they are local. The crawler pauses for a default delay
568    * of 10 seconds between each fetch, this can be configured in the delay
569    * variable. This is only meant for test purposes, as it does not respect
570    * robots or anything else fancy :)
571    * @param level which level to crawl
572    * @param out output stream to write to
573    * @return number of pages crawled on this level and below
574    */
575   protected int webCrawl(int level, OutputStream out) {
576     int numPages = 0;
577     LinkedHashSet<URL> stack = backlog.get(level);
578     int rawStackSize = stack.size();
579     stack.removeAll(visited);
580     int stackSize = stack.size();
581     LinkedHashSet<URL> subStack = new LinkedHashSet<>();
582     info("Entering crawl at level "+level+" ("+rawStackSize+" links total, "+stackSize+" new)");
583     for(URL u : stack) {
584       try {
585         visited.add(u);
586         PageFetcherResult result = pageFetcher.readPageFromUrl(u);
587         if(result.httpStatus == 200) {
588           u = (result.redirectUrl != null) ? result.redirectUrl : u;
589           URL postUrl = new URL(appendParam(solrUrl.toString(), 
590               "literal.id="+URLEncoder.encode(u.toString(),"UTF-8") +
591               "&literal.url="+URLEncoder.encode(u.toString(),"UTF-8")));
592           boolean success = postData(new ByteArrayInputStream(result.content.array(), result.content.arrayOffset(),result.content.limit() ), null, out, result.contentType, postUrl);
593           if (success) {
594             info("POSTed web resource "+u+" (depth: "+level+")");
595             Thread.sleep(delay * 1000);
596             numPages++;
597             // Pull links from HTML pages only
598             if(recursive > level && result.contentType.equals("text/html")) {
599               Set<URL> children = pageFetcher.getLinksFromWebPage(u, new ByteArrayInputStream(result.content.array(), result.content.arrayOffset(), result.content.limit()), result.contentType, postUrl);
600               subStack.addAll(children);
601             }
602           } else {
603             warn("An error occurred while posting "+u);
604           }
605         } else {
606           warn("The URL "+u+" returned a HTTP result status of "+result.httpStatus);
607         }
608       } catch (IOException e) {
609         warn("Caught exception when trying to open connection to "+u+": "+e.getMessage());
610       } catch (InterruptedException e) {
611         throw new RuntimeException(e);
612       }
613     }
614     if(!subStack.isEmpty()) {
615       backlog.add(subStack);
616       numPages += webCrawl(level+1, out);
617     }
618     return numPages;    
619   }

634    * @return the byte array
635    * @throws IOException If there is a low-level I/O error.

637   public static ByteBuffer inputStreamToByteArray(InputStream is, long maxSize) throws IOException {
638     BAOS bos =  new BAOS();
639     long sz = 0;
640     int next = is.read();
641     while (next > -1) {

651   /**
652    * Computes the full URL based on a base url and a possibly relative link found
653    * in the href param of an HTML anchor.
654    * @param baseUrl the base url from where the link was found
655    * @param link the absolute or relative link
656    * @return the string version of the full URL
657    */
658   protected String computeFullUrl(URL baseUrl, String link) {
659     if(link == null || link.length() == 0) {
660       return null;
661     }
662     if(!link.startsWith("http")) {
663       if(link.startsWith("/")) {
664         link = baseUrl.getProtocol() + "://" + baseUrl.getAuthority() + link;
665       } else {
666         if(link.contains(":")) {
667           return null; // Skip non-relative URLs
668         }
669         String path = baseUrl.getPath();
670         if(!path.endsWith("/")) {
671           int sep = path.lastIndexOf("/");
672           String file = path.substring(sep+1);
673           if(file.contains(".") || file.contains("?"))
674             path = path.substring(0,sep);
675         }
676         link = baseUrl.getProtocol() + "://" + baseUrl.getAuthority() + path + "/" + link;
677       }
678     }
679     link = normalizeUrlEnding(link);
680     String l = link.toLowerCase(Locale.ROOT);
681     // Simple brute force skip images
682     if(l.endsWith(".jpg") || l.endsWith(".jpeg") || l.endsWith(".png") || l.endsWith(".gif")) {
683       return null; // Skip images
684     }
685     return link;
686   }

745    * @param param the parameter(s) to append, separated by "&amp;"
746    * @return the string version of the resulting URL
747    */
748   public static String appendParam(String url, String param) {
749     String[] pa = param.split("&");

752       String[] kv = p.split("=");
753       if(kv.length == 2) {

757       }

766   public void postFile(File file, OutputStream output, String type) {
767     InputStream is = null;

769       URL url = solrUrl;

784             urlStr = appendParam(urlStr, "resource.name=" + URLEncoder.encode(file.getAbsolutePath(), "UTF-8"));
785           if(urlStr.indexOf("literal.id")==-1)
786             urlStr = appendParam(urlStr, "literal.id=" + URLEncoder.encode(file.getAbsolutePath(), "UTF-8"));
787           url = new URL(urlStr);
788         }
789       } else {
790         if(type == null) type = DEFAULT_CONTENT_TYPE;
791       }

793       is = new FileInputStream(file);
794       postData(is, (int)file.length(), output, type, url);
795     } catch (IOException e) {
796       e.printStackTrace();
797       warn("Can't open/read file: " + file);
798     } finally {
799       try {
800         if(is!=null) is.close();
801       } catch (IOException e) {
802         fatal("IOException while closing file: "+ e);
803       }
804     }

859   /**
860    * Reads data from the data stream and posts it to solr,
861    * writes to the response to output
862    * @return true if success
863    */
864   public boolean postData(InputStream data, Integer length, OutputStream output, String type, URL url) {
865     if(mockMode) return true;
866     boolean success = true;
867     if(type == null)
868       type = DEFAULT_CONTENT_TYPE;
869     HttpURLConnection urlc = null;
870     try {
871       try {
872         urlc = (HttpURLConnection) url.openConnection();
873         try {
874           urlc.setRequestMethod("POST");
875         } catch (ProtocolException e) {
876           fatal("Shouldn't happen: HttpURLConnection doesn't support POST??"+e);
877         }
878         urlc.setDoOutput(true);
879         urlc.setDoInput(true);
880         urlc.setUseCaches(false);
881         urlc.setAllowUserInteraction(false);
882         urlc.setRequestProperty("Content-type", type);
883         if (url.getUserInfo() != null) {
884           String encoding = DatatypeConverter.printBase64Binary(url.getUserInfo().getBytes(StandardCharsets.US_ASCII));
885           urlc.setRequestProperty("Authorization", "Basic " + encoding);
886         }
887         if (null != length) urlc.setFixedLengthStreamingMode(length);
888         urlc.connect();
889       } catch (IOException e) {
890         fatal("Connection error (is Solr running at " + solrUrl + " ?): " + e);
891         success = false;
892       }
893       
894       try (final OutputStream out = urlc.getOutputStream()) {
895         pipe(data, out);
896       } catch (IOException e) {
897         fatal("IOException while posting data: " + e);
898         success = false;
899       }
900       
901       try {
902         success &= checkResponseCode(urlc);
903         try (final InputStream in = urlc.getInputStream()) {
904           pipe(in, output);
905         }
906       } catch (IOException e) {
907         warn("IOException while reading response: " + e);
908         success = false;
909       }
910     } finally {
911       if (urlc!=null) urlc.disconnect();
912     }
913     return success;
914   }

916   private static boolean checkResponseCode(HttpURLConnection urlc) throws IOException {
917     if (urlc.getResponseCode() >= 400) {
918       warn("Solr returned an error #" + urlc.getResponseCode() + 
919             " (" + urlc.getResponseMessage() + ") for url: " + urlc.getURL());
920       Charset charset = StandardCharsets.ISO_8859_1;
921       final String contentType = urlc.getContentType();
922       // code cloned from ContentStreamBase, but post.jar should be standalone!
923       if (contentType != null) {
924         int idx = contentType.toLowerCase(Locale.ROOT).indexOf("charset=");
925         if (idx > 0) {
926           charset = Charset.forName(contentType.substring(idx + "charset=".length()).trim());
927         }
928       }
929       // Print the response returned by Solr
930       try (InputStream errStream = urlc.getErrorStream()) {
931         if (errStream != null) {
932           BufferedReader br = new BufferedReader(new InputStreamReader(errStream, charset));
933           final StringBuilder response = new StringBuilder("Response: ");
934           int ch;
935           while ((ch = br.read()) != -1) {
936             response.append((char) ch);
937           }
938           warn(response.toString().trim());
939         }
940       }
941       return false;
942     }
943     return true;
944   }

955   /**
956    * Pipes everything from the source to the dest.  If dest is null, 
957    * then everything is read from source and thrown away.
958    */
959   private static void pipe(InputStream source, OutputStream dest) throws IOException {
960     byte[] buf = new byte[1024];
961     int read = 0;
962     while ( (read = source.read(buf) ) >= 0) {
963       if (null != dest) dest.write(buf, 0, read);
964     }
965     if (null != dest) dest.flush();
966   }

1060   //
1061   // Simple crawler class which can fetch a page and check for robots.txt
1062   //
1063   class PageFetcher {
1064     Map<String, List<String>> robotsCache;
1065     final String DISALLOW = "Disallow:";
1066     
1067     public PageFetcher() {
1068       robotsCache = new HashMap<>();
1069     }
1070     
1071     public PageFetcherResult readPageFromUrl(URL u) {
1072       PageFetcherResult res = new PageFetcherResult();
1073       try {
1074         if (isDisallowedByRobots(u)) {
1075           warn("The URL "+u+" is disallowed by robots.txt and will not be crawled.");
1076           res.httpStatus = 403;
1077           visited.add(u);
1078           return res;
1079         }
1080         res.httpStatus = 404;
1081         HttpURLConnection conn = (HttpURLConnection) u.openConnection();
1082         conn.setRequestProperty("User-Agent", "SimplePostTool-crawler/"+VERSION_OF_THIS_TOOL+" (http://lucene.apache.org/solr/)");
1083         conn.setRequestProperty("Accept-Encoding", "gzip, deflate");
1084         conn.connect();
1085         res.httpStatus = conn.getResponseCode();
1086         if(!normalizeUrlEnding(conn.getURL().toString()).equals(normalizeUrlEnding(u.toString()))) {
1087           info("The URL "+u+" caused a redirect to "+conn.getURL());
1088           u = conn.getURL();
1089           res.redirectUrl = u;
1090           visited.add(u);
1091         }
1092         if(res.httpStatus == 200) {
1093           // Raw content type of form "text/html; encoding=utf-8"
1094           String rawContentType = conn.getContentType();
1095           String type = rawContentType.split(";")[0];
1096           if(typeSupported(type) || "*".equals(fileTypes)) {
1097             String encoding = conn.getContentEncoding();
1098             InputStream is;
1099             if (encoding != null && encoding.equalsIgnoreCase("gzip")) {
1100               is = new GZIPInputStream(conn.getInputStream());
1101             } else if (encoding != null && encoding.equalsIgnoreCase("deflate")) {
1102               is = new InflaterInputStream(conn.getInputStream(), new Inflater(true));
1103             } else {
1104               is = conn.getInputStream();
1105             }
1106             
1107             // Read into memory, so that we later can pull links from the page without re-fetching 
1108             res.content = inputStreamToByteArray(is);
1109             is.close();
1110           } else {
1111             warn("Skipping URL with unsupported type "+type);
1112             res.httpStatus = 415;
1113           }
1114         }
1115       } catch(IOException e) {
1116         warn("IOException when reading page from url "+u+": "+e.getMessage());
1117       }
1118       return res;
1119     }

1121     public boolean isDisallowedByRobots(URL url) {
1122       String host = url.getHost();
1123       String strRobot = url.getProtocol() + "://" + host + "/robots.txt";
1124       List<String> disallows = robotsCache.get(host);
1125       if(disallows == null) {
1126         disallows = new ArrayList<>();
1127         URL urlRobot;
1128         try { 
1129           urlRobot = new URL(strRobot);
1130           disallows = parseRobotsTxt(urlRobot.openStream());
1131         } catch (MalformedURLException e) {
1132           return true; // We cannot trust this robots URL, should not happen
1133         } catch (IOException e) {
1134           // There is no robots.txt, will cache an empty disallow list
1135         }
1136       }
1137       
1138       robotsCache.put(host, disallows);
1139 
1140       String strURL = url.getFile();
1141       for (String path : disallows) {
1142         if (path.equals("/") || strURL.indexOf(path) == 0)
1143           return true;
1144       }
1145       return false;
1146     }

1148     /**
1149      * Very simple robots.txt parser which obeys all Disallow lines regardless
1150      * of user agent or whether there are valid Allow: lines.
1151      * @param is Input stream of the robots.txt file
1152      * @return a list of disallow paths
1153      * @throws IOException if problems reading the stream
1154      */
1155     protected List<String> parseRobotsTxt(InputStream is) throws IOException {
1156       List<String> disallows = new ArrayList<>();
1157       BufferedReader r = new BufferedReader(new InputStreamReader(is, StandardCharsets.UTF_8));
1158       String l;
1159       while((l = r.readLine()) != null) {
1160         String[] arr = l.split("#");
1161         if(arr.length == 0) continue;
1162         l = arr[0].trim();
1163         if(l.startsWith(DISALLOW)) {
1164           l = l.substring(DISALLOW.length()).trim();
1165           if(l.length() == 0) continue;
1166           disallows.add(l);
1167         }
1168       }
1169       is.close();
1170       return disallows;
1171     }

1173     /**
1174      * Finds links on a web page, using /extract?extractOnly=true
1175      * @param u the URL of the web page
1176      * @param is the input stream of the page
1177      * @param type the content-type
1178      * @param postUrl the URL (typically /solr/extract) in order to pull out links
1179      * @return a set of URLs parsed from the page
1180      */
1181     protected Set<URL> getLinksFromWebPage(URL u, InputStream is, String type, URL postUrl) {
1182       Set<URL> l = new HashSet<>();
1183       URL url = null;
1184       try {
1185         ByteArrayOutputStream os = new ByteArrayOutputStream();
1186         URL extractUrl = new URL(appendParam(postUrl.toString(), "extractOnly=true"));
1187         boolean success = postData(is, null, os, type, extractUrl);
1188         if(success) {
1189           Document d = makeDom(os.toByteArray());
1190           String innerXml = getXP(d, "/response/str/text()[1]", false);
1191           d = makeDom(innerXml.getBytes(StandardCharsets.UTF_8));
1192           NodeList links = getNodesFromXP(d, "/html/body//a/@href");
1193           for(int i = 0; i < links.getLength(); i++) {
1194             String link = links.item(i).getTextContent();
1195             link = computeFullUrl(u, link);
1196             if(link == null)
1197               continue;
1198             url = new URL(link);
1199             if(url.getAuthority() == null || !url.getAuthority().equals(u.getAuthority()))
1200               continue;
1201             l.add(url);
1202           }
1203         }
1204       } catch (MalformedURLException e) {
1205         warn("Malformed URL "+url);
1206       } catch (IOException e) {
1207         warn("IOException opening URL "+url+": "+e.getMessage());
1208       } catch (Exception e) {
1209         throw new RuntimeException(e);
1210       }
1211       return l;
1212     }
