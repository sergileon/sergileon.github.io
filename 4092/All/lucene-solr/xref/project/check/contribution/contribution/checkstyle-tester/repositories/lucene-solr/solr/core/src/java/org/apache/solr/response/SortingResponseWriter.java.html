72    public void write(Writer writer, SolrQueryRequest req, SolrQueryResponse res) throws IOException {
73      Exception e1 = res.getException();
74      if(e1 != null) {
75        if(!(e1 instanceof IgnoreException)) {
76          e1.printStackTrace(new PrintWriter(writer));
77        }
78        return;
79      }
80      SolrRequestInfo info = SolrRequestInfo.getRequestInfo();
81      SortSpec sortSpec = info.getResponseBuilder().getSortSpec();
82  
83      if(sortSpec == null) {
84        throw new IOException(new SyntaxError("No sort criteria was provided."));
85      }
86  
87      SolrIndexSearcher searcher = req.getSearcher();
88      Sort sort = searcher.weightSort(sortSpec.getSort());
89  
90      if(sort == null) {
91        throw new IOException(new SyntaxError("No sort criteria was provided."));
92      }
93  
94      if(sort.needsScores()) {
95        throw new IOException(new SyntaxError("Scoring is not currently supported with xsort."));
96      }
97  
98      FixedBitSet[] sets = (FixedBitSet[])req.getContext().get("export");
99      Integer th = (Integer)req.getContext().get("totalHits");
100 
101     if(sets == null) {
102       throw new IOException(new SyntaxError("xport RankQuery is required for xsort: rq={!xport}"));
103     }
104 
105     int totalHits = th.intValue();
106     SolrParams params = req.getParams();
107     String fl = params.get("fl");
108 
109     if(fl == null) {
110       throw new IOException(new SyntaxError("export field list (fl) must be specified."));
111     }
112 
113     String[] fields = fl.split(",");
114 
115     for(int i=0;i<fields.length; i++) {
116       if(fl.trim().equals("score")) {
117         throw new IOException(new SyntaxError("Scoring is not currently supported with xsort."));
118       }
119     }
120 
121     FieldWriter[] fieldWriters = getFieldWriters(fields, req.getSearcher());
122     writer.write("{\"responseHeader\": {\"status\": 0}, \"response\":{\"numFound\":"+totalHits+", \"docs\":[");
123 
124     //Write the data.
125     List<LeafReaderContext> leaves = req.getSearcher().getTopReaderContext().leaves();
126     SortDoc sortDoc = getSortDoc(req.getSearcher(), sort.getSort());
127     int count = 0;
128     int queueSize = 30000;
129     SortQueue queue = new SortQueue(queueSize, sortDoc);
130     SortDoc[] outDocs = new SortDoc[queueSize];
131 
132     boolean commaNeeded = false;
133     while(count < totalHits) {
134       //long begin = System.nanoTime();
135       queue.reset();
136       SortDoc top = queue.top();
137       for(int i=0; i<leaves.size(); i++) {
138         sortDoc.setNextReader(leaves.get(i));
139         DocIdSetIterator it = new BitSetIterator(sets[i], 0); // cost is not useful here
140         int docId = -1;
141         while((docId = it.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
142           sortDoc.setValues(docId);
143           if(top.lessThan(sortDoc)) {
144             top.setValues(sortDoc);
145             top = queue.updateTop();
146           }
147         }
148       }
149 
150       int outDocsIndex = -1;
151 
152       for(int i=0; i<queueSize; i++) {
153         SortDoc s = queue.pop();
154         if(s.docId > -1) {
155           outDocs[++outDocsIndex] = s;
156         }
157       }
158 
159      //long end = System.nanoTime();
160 
161       count += (outDocsIndex+1);
162 
163       try {
164         for(int i=outDocsIndex; i>=0; --i) {
165           SortDoc s = outDocs[i];
166           if(commaNeeded){writer.write(',');}
167           writer.write('{');
168           writeDoc(s, leaves, fieldWriters, sets, writer);
169           writer.write('}');
170           commaNeeded = true;
171           s.reset();
172         }
173       } catch(Throwable e) {
174         Throwable ex = e;
175         while(ex != null) {
176           String m = ex.getMessage();
177           if(m != null && m.contains("Broken pipe")) {
178             throw new IgnoreException();
179           }
180           ex = ex.getCause();
181         }
182 
183         if(e instanceof IOException) {
184           throw ((IOException)e);
185         } else {
186           throw new IOException(e);
187         }
188       }
189     }
190 
191     //System.out.println("Sort Time 2:"+Long.toString(total/1000000));
192     writer.write("]}}");
193     writer.flush();
194   }

208   protected void writeDoc(SortDoc sortDoc,
209                           List<LeafReaderContext> leaves,
210                           FieldWriter[] fieldWriters,
211                           FixedBitSet[] sets,
212                           Writer out) throws IOException{
213 
214     int ord = sortDoc.ord;
215     FixedBitSet set = sets[ord];
216     set.clear(sortDoc.docId);
217     LeafReaderContext context = leaves.get(ord);
218     boolean needsComma = false;
219     for(FieldWriter fieldWriter : fieldWriters) {
220       if(needsComma) {
221         out.write(',');
222       }
223       fieldWriter.write(sortDoc.docId, context.reader(), out);
224       needsComma = true;
225     }
226   }

228   protected FieldWriter[] getFieldWriters(String[] fields, SolrIndexSearcher searcher) throws IOException {
229     IndexSchema schema = searcher.getSchema();
230     FieldWriter[] writers = new FieldWriter[fields.length];
231     for(int i=0; i<fields.length; i++) {
232       String field = fields[i];
233       SchemaField schemaField = null;
234 
235       try {
236         schemaField = schema.getField(field);
237       } catch (Exception e) {
238         throw new IOException(e);
239       }
240 
241       if(!schemaField.hasDocValues()) {
242         throw new IOException(field+" must have DocValues to use this feature.");
243       }
244 
245       boolean multiValued = schemaField.multiValued();
246       FieldType fieldType = schemaField.getType();
247       if(fieldType instanceof TrieIntField) {
248         if(multiValued) {
249           writers[i] = new MultiFieldWriter(field, fieldType, true);
250         } else {
251           writers[i] = new IntFieldWriter(field);
252         }
253       } else if (fieldType instanceof TrieLongField) {
254         if(multiValued) {
255           writers[i] = new MultiFieldWriter(field, fieldType, true);
256         } else {
257           writers[i] = new LongFieldWriter(field);
258         }
259       } else if (fieldType instanceof TrieFloatField) {
260         if(multiValued) {
261           writers[i] = new MultiFieldWriter(field, fieldType, true);
262         } else {
263           writers[i] = new FloatFieldWriter(field);
264         }
265       } else if(fieldType instanceof TrieDoubleField) {
266         if(multiValued) {
267           writers[i] = new MultiFieldWriter(field, fieldType, true);
268         } else {
269           writers[i] = new DoubleFieldWriter(field);
270         }
271       } else if(fieldType instanceof StrField) {
272         if(multiValued) {
273           writers[i] = new MultiFieldWriter(field, fieldType, false);
274         } else {
275           writers[i] = new StringFieldWriter(field, fieldType);
276         }
277       } else {
278         throw new IOException("Export fields must either be one of the following types: int,float,long,double,string");
279       }
280     }
281     return writers;
282   }

284   private SortDoc getSortDoc(SolrIndexSearcher searcher, SortField[] sortFields) throws IOException {
285     SortValue[] sortValues = new SortValue[sortFields.length];
286     IndexSchema schema = searcher.getSchema();
287     for(int i=0; i<sortFields.length; ++i) {
288       SortField sf = sortFields[i];
289       String field = sf.getField();
290       boolean reverse = sf.getReverse();
291       SchemaField schemaField = schema.getField(field);
292       FieldType ft = schemaField.getType();
293 
294       if(!schemaField.hasDocValues()) {
295         throw new IOException(field+" must have DocValues to use this feature.");
296       }
297 
298       if(ft instanceof TrieIntField) {
299         if(reverse) {
300           sortValues[i] = new IntValue(field, new IntDesc());
301         } else {
302           sortValues[i] = new IntValue(field, new IntAsc());
303         }
304       } else if(ft instanceof TrieFloatField) {
305         if(reverse) {
306           sortValues[i] = new FloatValue(field, new FloatDesc());
307         } else {
308           sortValues[i] = new FloatValue(field, new FloatAsc());
309         }
310       } else if(ft instanceof TrieDoubleField) {
311         if(reverse) {
312           sortValues[i] = new DoubleValue(field, new DoubleDesc());
313         } else {
314           sortValues[i] = new DoubleValue(field, new DoubleAsc());
315         }
316       } else if(ft instanceof TrieLongField) {
317         if(reverse) {
318           sortValues[i] = new LongValue(field, new LongDesc());
319         } else {
320           sortValues[i] = new LongValue(field, new LongAsc());
321         }
322       } else if(ft instanceof StrField) {
323         LeafReader reader = searcher.getLeafReader();
324         SortedDocValues vals =  reader.getSortedDocValues(field);
325         if(reverse) {
326           sortValues[i] = new StringValue(vals, field, new IntDesc());
327         } else {
328           sortValues[i] = new StringValue(vals, field, new IntAsc());
329         }
330       } else {
331         throw new IOException("Sort fields must be one of the following types: int,float,long,double,string");
332       }
333     }
334 
335     if(sortValues.length == 1) {
336       return new SingleValueSortDoc(sortValues[0]);
337     } else if(sortValues.length == 2) {
338       return new DoubleValueSortDoc(sortValues[0], sortValues[1]);
339     } else if(sortValues.length == 3) {
340       return new TripleValueSortDoc(sortValues[0], sortValues[1], sortValues[2]);
341     } else if(sortValues.length == 4) {
342       return new QuadValueSortDoc(sortValues[0], sortValues[1], sortValues[2], sortValues[3]);
343     } else {
344       throw new IOException("A max of 4 sorts can be specified");
345     }
346   }

348   class SortQueue extends PriorityQueue<SortDoc> {
349 
350     private SortDoc proto;
351     private Object[] cache;
352 
353     public SortQueue(int len, SortDoc proto) {
354       super(len);
355       this.proto = proto;
356     }
357 
358     protected boolean lessThan(SortDoc t1, SortDoc t2) {
359       return t1.lessThan(t2);
360     }
361 
362     private void populate() {
363       Object[] heap = getHeapArray();
364       cache = new SortDoc[heap.length];
365       for(int i=1; i<heap.length; i++) {
366         cache[i] = heap[i]  = proto.copy();
367       }
368       size = maxSize;
369     }
370 
371     private void reset() {
372       Object[] heap = getHeapArray();
373       if(cache != null) {
374         System.arraycopy(cache, 1, heap, 1, heap.length-1);
375         size = maxSize;
376       } else {
377         populate();
378       }
379     }
380   }

382   class SortDoc {
383 
384     protected int docId = -1;
385     protected int ord = -1;
386     protected int docBase = -1;
387 
388     private SortValue[] sortValues;
389 
390     public SortDoc() {
391 
392     }
393 
394     public void setNextReader(LeafReaderContext context) throws IOException {
395       this.ord = context.ord;
396       for(SortValue value : sortValues) {
397         value.setNextReader(context);
398       }
399     }
400 
401     public void reset() {
402       this.docId = -1;
403     }
404 
405     public void setValues(int docId) throws IOException {
406       this.docId = docId;
407       for(SortValue sortValue : sortValues) {
408         sortValue.setCurrentValue(docId);
409       }
410     }
411 
412     public void setValues(SortDoc sortDoc) throws IOException {
413       this.docId = sortDoc.docId;
414       this.ord = sortDoc.ord;
415       SortValue[] vals = sortDoc.sortValues;
416       for(int i=0; i<vals.length; i++) {
417         sortValues[i].setCurrentValue(vals[i]);
418       }
419     }
420 
421     public SortDoc(SortValue[] sortValues) {
422       this.sortValues = sortValues;
423     }
424 
425     public SortDoc copy() {
426       SortValue[] svs = new SortValue[sortValues.length];
427       for(int i=0; i<sortValues.length; i++) {
428         svs[i] = sortValues[i].copy();
429       }
430 
431       return new SortDoc(svs);
432     }
433 
434     public boolean lessThan(Object o) {
435       if(docId == -1) {
436         return true;
437       }
438 
439       SortDoc sd = (SortDoc)o;
440       SortValue[] sortValues1 = sd.sortValues;
441       for(int i=0; i<sortValues.length; i++) {
442         int comp = sortValues[i].compareTo(sortValues1[i]);
443         if(comp < 0) {
444           return true;
445         } if(comp > 0) {
446           return false;
447         }
448       }
449       return docId+docBase < sd.docId+sd.docBase;
450     }
451 
452     public String toString() {
453       return "";
454     }
455   }

741             }
742           }
743         }

758           }

761         }

1302   public abstract class PriorityQueue<T> {
1303     protected int size = 0;
1304     protected final int maxSize;
1305     private final T[] heap;
1306 
1307     public PriorityQueue(int maxSize) {
1308       this(maxSize, true);
1309     }
1310 
1311     public PriorityQueue(int maxSize, boolean prepopulate) {
1312       final int heapSize;
1313       if (0 == maxSize) {
1314         // We allocate 1 extra to avoid if statement in top()
1315         heapSize = 2;
1316       } else {
1317         if (maxSize > ArrayUtil.MAX_ARRAY_LENGTH) {
1318           // Don't wrap heapSize to -1, in this case, which
1319           // causes a confusing NegativeArraySizeException.
1320           // Note that very likely this will simply then hit
1321           // an OOME, but at least that's more indicative to
1322           // caller that this values is too big.  We don't +1
1323           // in this case, but it's very unlikely in practice
1324           // one will actually insert this many objects into
1325           // the PQ:
1326           // Throw exception to prevent confusing OOME:
1327           throw new IllegalArgumentException("maxSize must be <= " + ArrayUtil.MAX_ARRAY_LENGTH + "; got: " + maxSize);
1328         } else {
1329           // NOTE: we add +1 because all access to heap is
1330           // 1-based not 0-based.  heap[0] is unused.
1331           heapSize = maxSize + 1;
1332         }
1333       }
1334       // T is unbounded type, so this unchecked cast works always:
1335       @SuppressWarnings("unchecked") final T[] h = (T[]) new Object[heapSize];
1336       this.heap = h;
1337       this.maxSize = maxSize;
1338 
1339       if (prepopulate) {
1340         // If sentinel objects are supported, populate the queue with them
1341         T sentinel = getSentinelObject();
1342         if (sentinel != null) {
1343           heap[1] = sentinel;
1344           for (int i = 2; i < heap.length; i++) {
1345             heap[i] = getSentinelObject();
1346           }
1347           size = maxSize;
1348         }
1349       }
1350     }
1351 
1352     /** Determines the ordering of objects in this priority queue.  Subclasses
1353      *  must define this one method.
1354      *  @return <code>true</code> iff parameter <tt>a</tt> is less than parameter <tt>b</tt>.
1355      */
1356     protected abstract boolean lessThan(T a, T b);
1357 
1358 
1359     protected T getSentinelObject() {
1360       return null;
1361     }
1362 
1363     /**
1364      * Adds an Object to a PriorityQueue in log(size) time. If one tries to add
1365      * more objects than maxSize from initialize an
1366      *
1367      * @return the new 'top' element in the queue.
1368      */
1369     public final T add(T element) {
1370       size++;
1371       heap[size] = element;
1372       upHeap();
1373       return heap[1];
1374     }
1375 
1376     /**
1377      * Adds an Object to a PriorityQueue in log(size) time.
1378      * It returns the object (if any) that was
1379      * dropped off the heap because it was full. This can be
1380      * the given parameter (in case it is smaller than the
1381      * full heap's minimum, and couldn't be added), or another
1382      * object that was previously the smallest value in the
1383      * heap and now has been replaced by a larger one, or null
1384      * if the queue wasn't yet full with maxSize elements.
1385      */
1386     public T insertWithOverflow(T element) {
1387       if (size < maxSize) {
1388         add(element);
1389         return null;
1390       } else if (size > 0 && !lessThan(element, heap[1])) {
1391         T ret = heap[1];
1392         heap[1] = element;
1393         updateTop();
1394         return ret;
1395       } else {
1396         return element;
1397       }
1398     }
1399 
1400     /** Returns the least element of the PriorityQueue in constant time. */
1401     public final T top() {
1402       // We don't need to check size here: if maxSize is 0,
1403       // then heap is length 2 array with both entries null.
1404       // If size is 0 then heap[1] is already null.
1405       return heap[1];
1406     }
1407 
1408     /** Removes and returns the least element of the PriorityQueue in log(size)
1409      time. */
1410     public final T pop() {
1411       if (size > 0) {
1412         T result = heap[1];       // save first value
1413         heap[1] = heap[size];     // move last to first
1414         heap[size] = null;        // permit GC of objects
1415         size--;
1416         downHeap();               // adjust heap
1417         return result;
1418       } else {
1419         return null;
1420       }
1421     }
1422 
1423     /**
1424      * Should be called when the Object at top changes values. Still log(n) worst
1425      * case, but it's at least twice as fast to
1426      *
1427      * <pre class="prettyprint">
1428      * pq.top().change();
1429      * pq.updateTop();
1430      * </pre>
1431      *
1432      * instead of
1433      *
1434      * <pre class="prettyprint">
1435      * o = pq.pop();
1436      * o.change();
1437      * pq.push(o);
1438      * </pre>
1439      *
1440      * @return the new 'top' element.
1441      */
1442     public final T updateTop() {
1443       downHeap();
1444       return heap[1];
1445     }
1446 
1447     /** Returns the number of elements currently stored in the PriorityQueue. */
1448     public final int size() {
1449       return size;
1450     }
1451 
1452     /** Removes all entries from the PriorityQueue. */
1453     public final void clear() {
1454       for (int i = 0; i <= size; i++) {
1455         heap[i] = null;
1456       }
1457       size = 0;
1458     }
1459 
1460     private final void upHeap() {
1461       int i = size;
1462       T node = heap[i];          // save bottom node
1463       int j = i >>> 1;
1464       while (j > 0 && lessThan(node, heap[j])) {
1465         heap[i] = heap[j];       // shift parents down
1466         i = j;
1467         j = j >>> 1;
1468       }
1469       heap[i] = node;            // install saved node
1470     }
1471 
1472     private final void downHeap() {
1473       int i = 1;
1474       T node = heap[i];          // save top node
1475       int j = i << 1;            // find smaller child
1476       int k = j + 1;
1477       if (k <= size && lessThan(heap[k], heap[j])) {
1478         j = k;
1479       }
1480       while (j <= size && lessThan(heap[j], node)) {
1481         heap[i] = heap[j];       // shift up child
1482         i = j;
1483         j = i << 1;
1484         k = j + 1;
1485         if (k <= size && lessThan(heap[k], heap[j])) {
1486           j = k;
1487         }
1488       }
1489       heap[i] = node;            // install saved node
1490     }
1491 
1492     /** This method returns the internal heap array as Object[].
1493      * @lucene.internal
1494      */
1495     public final Object[] getHeapArray() {
1496       return (Object[]) heap;
1497     }
1498   }
