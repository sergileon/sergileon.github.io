20  /**
21   * 3D rectangle, bounded on six sides by X,Y,Z limits, degenerate in X.
22   *
23   * @lucene.internal
24   */
25  public class dXYZSolid extends BaseXYZSolid {
26  
27    /** X plane */
28    protected final Plane xPlane;
29    /** Min-Y plane */
30    protected final SidedPlane minYPlane;
31    /** Max-Y plane */
32    protected final SidedPlane maxYPlane;
33    /** Min-Z plane */
34    protected final SidedPlane minZPlane;
35    /** Max-Z plane */
36    protected final SidedPlane maxZPlane;
37    
38    /** These are the edge points of the shape, which are defined to be at least one point on
39     * each surface area boundary.  In the case of a solid, this includes points which represent
40     * the intersection of XYZ bounding planes and the planet, as well as points representing
41     * the intersection of single bounding planes with the planet itself.
42     */
43    protected final GeoPoint[] edgePoints;
44  
45    /** Notable points for XPlane */
46    protected final GeoPoint[] notableXPoints;
47  
48    /**
49     * Sole constructor
50     *
51     *@param planetModel is the planet model.
52     *@param X is the X value.
53     *@param minY is the minimum Y value.
54     *@param maxY is the maximum Y value.
55     *@param minZ is the minimum Z value.
56     *@param maxZ is the maximum Z value.
57     */
58    public dXYZSolid(final PlanetModel planetModel,
59      final double X,
60      final double minY,
61      final double maxY,
62      final double minZ,
63      final double maxZ) {
64      super(planetModel);
65      // Argument checking
66      if (maxY - minY < Vector.MINIMUM_RESOLUTION)
67        throw new IllegalArgumentException("Y values in wrong order or identical");
68      if (maxZ - minZ < Vector.MINIMUM_RESOLUTION)
69        throw new IllegalArgumentException("Z values in wrong order or identical");
70  
71      final double worldMinX = planetModel.getMinimumXValue();
72      final double worldMaxX = planetModel.getMaximumXValue();
73      
74      // Construct the planes
75      xPlane = new Plane(xUnitVector,-X);
76      minYPlane = new SidedPlane(0.0,maxY,0.0,yUnitVector,-minY);
77      maxYPlane = new SidedPlane(0.0,minY,0.0,yUnitVector,-maxY);
78      minZPlane = new SidedPlane(0.0,0.0,maxZ,zUnitVector,-minZ);
79      maxZPlane = new SidedPlane(0.0,0.0,minZ,zUnitVector,-maxZ);
80        
81      // We need at least one point on the planet surface for each manifestation of the shape.
82      // There can be up to 2 (on opposite sides of the world).  But we have to go through
83      // 4 combinations of adjacent planes in order to find out if any have 2 intersection solution.
84      // Typically, this requires 4 square root operations. 
85      final GeoPoint[] XminY = xPlane.findIntersections(planetModel,minYPlane,maxYPlane,minZPlane,maxZPlane);
86      final GeoPoint[] XmaxY = xPlane.findIntersections(planetModel,maxYPlane,minYPlane,minZPlane,maxZPlane);
87      final GeoPoint[] XminZ = xPlane.findIntersections(planetModel,minZPlane,maxZPlane,minYPlane,maxYPlane);
88      final GeoPoint[] XmaxZ = xPlane.findIntersections(planetModel,maxZPlane,minZPlane,minYPlane,maxYPlane);
89  
90      notableXPoints = glueTogether(XminY, XmaxY, XminZ, XmaxZ);
91  
92      // Now, compute the edge points.
93      // This is the trickiest part of setting up an XYZSolid.  We've computed intersections already, so
94      // we'll start there.  We know that at most there will be two disconnected shapes on the planet surface.
95      // But there's also a case where exactly one plane slices through the world, and none of the bounding plane
96      // intersections do.  Thus, if we don't find any of the edge intersection cases, we have to look for that last case.
97        
98      // We need to look at single-plane/world intersections.
99      // We detect these by looking at the world model and noting its x, y, and z bounds.
100     // For the single-dimension degenerate case, there's really only one plane that can possibly intersect the world.
101     // The cases we are looking for are when the four corner points for any given
102     // plane are all outside of the world, AND that plane intersects the world.
103     // There are four corner points all told; we must evaluate these WRT the planet surface.
104     final boolean XminYminZ = planetModel.pointOutside(X, minY, minZ);
105     final boolean XminYmaxZ = planetModel.pointOutside(X, minY, maxZ);
106     final boolean XmaxYminZ = planetModel.pointOutside(X, maxY, minZ);
107     final boolean XmaxYmaxZ = planetModel.pointOutside(X, maxY, maxZ);
108 
109     final GeoPoint[] xEdges;
110     if (X - worldMinX >= -Vector.MINIMUM_RESOLUTION && X - worldMaxX <= Vector.MINIMUM_RESOLUTION &&
111       minY < 0.0 && maxY > 0.0 && minZ < 0.0 && maxZ > 0.0 &&
112       XminYminZ && XminYmaxZ && XmaxYminZ && XmaxYmaxZ) {
113       // Find any point on the X plane that intersects the world
114       // First construct a perpendicular plane that will allow us to find a sample point.
115       // This plane is vertical and goes through the points (0,0,0) and (1,0,0)
116       // Then use it to compute a sample point.
117       final GeoPoint intPoint = xPlane.getSampleIntersectionPoint(planetModel, xVerticalPlane);
118       if (intPoint != null) {
119         xEdges = new GeoPoint[]{intPoint};
120       } else {
121         xEdges = EMPTY_POINTS;
122       }
123     } else {
124       xEdges = EMPTY_POINTS;
125     }
126 
127     this.edgePoints = glueTogether(XminY,XmaxY,XminZ,XmaxZ,xEdges);
128   }
129 
130   @Override
131   protected GeoPoint[] getEdgePoints() {
132     return edgePoints;
133   }
134   
135   @Override
136   public boolean isWithin(final double x, final double y, final double z) {
137     return xPlane.evaluateIsZero(x, y, z) &&
138       minYPlane.isWithin(x, y, z) &&
139       maxYPlane.isWithin(x, y, z) &&
140       minZPlane.isWithin(x, y, z) &&
141       maxZPlane.isWithin(x, y, z);
142   }
143 
144   @Override
145   public int getRelationship(final GeoShape path) {
146     //System.err.println(this+" getrelationship with "+path);
147     final int insideRectangle = isShapeInsideArea(path);
148     if (insideRectangle == SOME_INSIDE) {
149       //System.err.println(" some shape points inside area");
150       return OVERLAPS;
151     }
152 
153     // Figure out if the entire XYZArea is contained by the shape.
154     final int insideShape = isAreaInsideShape(path);
155     if (insideShape == SOME_INSIDE) {
156       //System.err.println(" some area points inside shape");
157       return OVERLAPS;
158     }
159 
160     if (insideRectangle == ALL_INSIDE && insideShape == ALL_INSIDE) {
161       //System.err.println(" inside of each other");
162       return OVERLAPS;
163     }
164 
165     // The entire locus of points in this shape is on a single plane, so we only need ot look for an intersection with that plane.
166     //System.err.println("xPlane = "+xPlane);
167     if (path.intersects(xPlane, notableXPoints, minYPlane, maxYPlane, minZPlane, maxZPlane)) {
168       //System.err.println(" edges intersect");
169       return OVERLAPS;
170     }
171 
172     if (insideRectangle == ALL_INSIDE) {
173       //System.err.println(" shape points inside area");
174       return WITHIN;
175     }
176 
177     if (insideShape == ALL_INSIDE) {
178       //System.err.println(" shape contains all area");
179       return CONTAINS;
180     }
181     //System.err.println(" disjoint");
182     return DISJOINT;
183   }
184 
185   @Override
186   public boolean equals(Object o) {
187     if (!(o instanceof dXYZSolid))
188       return false;
189     dXYZSolid other = (dXYZSolid) o;
190     if (!super.equals(other)) {
191       return false;
192     }
193     return other.xPlane.equals(xPlane) &&
194       other.minYPlane.equals(minYPlane) &&
195       other.maxYPlane.equals(maxYPlane) &&
196       other.minZPlane.equals(minZPlane) &&
197       other.maxZPlane.equals(maxZPlane);
198   }
199 
200   @Override
201   public int hashCode() {
202     int result = super.hashCode();
203     result = 31 * result  + xPlane.hashCode();
204     result = 31 * result  + minYPlane.hashCode();
205     result = 31 * result  + maxYPlane.hashCode();
206     result = 31 * result  + minZPlane.hashCode();
207     result = 31 * result  + maxZPlane.hashCode();
208     return result;
209   }
210 
211   @Override
212   public String toString() {
213     return "dXYZSolid: {planetmodel="+planetModel+", xplane="+xPlane+", minYplane="+minYPlane+", maxYplane="+maxYPlane+", minZplane="+minZPlane+", maxZplane="+maxZPlane+"}";
214   }
