749   SolrQueryResponse loadLocalFromHandler(String handler, String filename, 
750                                          String... args) throws Exception {
751                               
752     LocalSolrQueryRequest req = (LocalSolrQueryRequest) req(args);
753     try {
754       // TODO: stop using locally defined streams once stream.file and
755       // stream.body work everywhere
756       List<ContentStream> cs = new ArrayList<>();
757       cs.add(new ContentStreamBase.FileStream(getFile(filename)));
758       req.setContentStreams(cs);
759       return h.queryAndResponse(handler, req);
760     } finally {
761       req.close();
762     }
763   }

765   SolrQueryResponse loadLocal(String filename, String... args) throws Exception {
766     return loadLocalFromHandler("/update/extract", filename, args);
767   }
