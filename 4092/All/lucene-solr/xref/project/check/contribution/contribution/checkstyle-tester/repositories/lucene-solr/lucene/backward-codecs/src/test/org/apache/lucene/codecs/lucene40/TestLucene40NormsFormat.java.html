118    *                         ____\_\__\_\          
119    *                       /`   /`     o\          
120    *                       |___ |_______|
121    *

123   public void testReadUndeadNorms() throws Exception {
124     InputStream resource = TestLucene40NormsFormat.class.getResourceAsStream("index.40.undeadnorms.zip");
125     assertNotNull(resource);
126     Path path = createTempDir("undeadnorms");
127     TestUtil.unzip(resource, path);
128     Directory dir = FSDirectory.open(path);
129     IndexReader r = DirectoryReader.open(dir);
130     NumericDocValues undeadNorms = MultiDocValues.getNormValues(r, "content");
131     assertNotNull(undeadNorms);
132     assertEquals(2, r.maxDoc());
133     assertEquals(0, undeadNorms.get(0));
134     assertEquals(0, undeadNorms.get(1));
135     dir.close();
136     r.close();
137   }
