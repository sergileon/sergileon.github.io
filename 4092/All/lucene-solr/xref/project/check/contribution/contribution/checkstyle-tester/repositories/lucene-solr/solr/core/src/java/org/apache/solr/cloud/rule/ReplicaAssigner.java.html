67    public static class Position implements Comparable<Position> {
68      public final String shard;
69      public final int index;
70  
71      public Position(String shard, int replicaIdx) {
72        this.shard = shard;
73        this.index = replicaIdx;
74      }
75  
76      @Override
77      public int compareTo(Position that) {
78        //this is to ensure that we try one replica from each shard first instead of
79        // all replicas from same shard
80        return that.index > index ? -1 : that.index == index ? 0 : 1;
81      }
82  
83      @Override
84      public String toString() {
85        return shard + ":" + index;
86      }
87    }

90    /**
91     * @param shardVsReplicaCount shard names vs no:of replicas required for each of those shards
92     * @param snitches            snitches details
93     * @param shardVsNodes        The current state of the system. can be an empty map if no nodes
94     *                            are created in this collection till now
95     */
96    public ReplicaAssigner(List<Rule> rules,
97                           Map<String, Integer> shardVsReplicaCount,
98                           List snitches,
99                           Map<String, Map<String, Integer>> shardVsNodes,
100                          List<String> liveNodes,
101                          CoreContainer cc, ClusterState clusterState) {
102     this.rules = rules;
103     for (Rule rule : rules) tagNames.add(rule.tag.name);
104     this.shardVsReplicaCount = shardVsReplicaCount;
105     this.liveNodes = new ArrayList<>(liveNodes);
106     this.nodeVsTags = getTagsForNodes(cc, snitches);
107     this.shardVsNodes = getDeepCopy(shardVsNodes, 2);
108     validateTags(nodeVsTags);
109 
110     if (clusterState != null) {
111       for (String s : clusterState.getCollections()) {
112         DocCollection coll = clusterState.getCollection(s);
113         for (Slice slice : coll.getSlices()) {
114           for (Replica replica : slice.getReplicas()) {
115             AtomicInteger count = nodeVsCores.get(replica.getNodeName());
116             if (count == null) nodeVsCores.put(replica.getNodeName(), count = new AtomicInteger());
117             count.incrementAndGet();
118           }
119         }
120       }
121     }
122   }

130   /**
131    * For each shard return a new set of nodes where the replicas need to be created satisfying
132    * the specified rule
133    */
134   public Map<Position, String> getNodeMappings() {
135     Map<Position, String> result = getNodeMappings0();
136     if (result == null) {
137       String msg = "Could not identify nodes matching the rules " + rules;
138       if (!failedNodes.isEmpty()) {
139         Map<String, String> failedNodes = new HashMap<>();
140         for (Map.Entry<String, SnitchContext> e : this.failedNodes.entrySet()) {
141           failedNodes.put(e.getKey(), e.getValue().getErrMsg());
142         }
143         msg += " Some nodes where excluded from assigning replicas because tags could not be obtained from them " + failedNodes;
144       }
145       msg += "\n tag values" + Utils.toJSONString(getNodeVsTags());
146       if (!shardVsNodes.isEmpty()) {
147         msg += "\nInitial state for the coll : " + Utils.toJSONString(shardVsNodes);
148       }
149       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, msg);
150     }
151     return result;
152 
153   }

155   Map<Position, String> getNodeMappings0() {
156     List<String> shardNames = new ArrayList<>(shardVsReplicaCount.keySet());
157     int[] shardOrder = new int[shardNames.size()];
158     for (int i = 0; i < shardNames.size(); i++) shardOrder[i] = i;
159 
160     boolean hasFuzzyRules = false;
161     int nonWildCardShardRules = 0;
162     for (Rule r : rules) {
163       if (r.isFuzzy()) hasFuzzyRules = true;
164       if (!r.shard.isWildCard()) {
165         nonWildCardShardRules++;
166         //we will have to try all combinations
167         if (shardNames.size() > 10) {
168           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
169               "Max 10 shards allowed if there is a non wild card shard specified in rule");
170         }
171       }
172     }
173 
174     Map<Position, String> result = tryAllPermutations(shardNames, shardOrder, nonWildCardShardRules, false);
175     if (result == null && hasFuzzyRules) {
176       result = tryAllPermutations(shardNames, shardOrder, nonWildCardShardRules, true);
177     }
178     return result;
179   }

181   private Map<Position, String> tryAllPermutations(List<String> shardNames,
182                                                    int[] shardOrder,
183                                                    int nonWildCardShardRules,
184                                                    boolean fuzzyPhase) {
185 
186 
187     Iterator<int[]> shardPermutations = nonWildCardShardRules > 0 ?
188         permutations(shardNames.size()) :
189         singletonList(shardOrder).iterator();
190 
191     for (; shardPermutations.hasNext(); ) {
192       int[] p = shardPermutations.next();
193       for (int i = 0; i < p.length; i++) {
194         List<Position> positions = new ArrayList<>();
195         for (int pos : p) {
196           for (int j = 0; j < shardVsReplicaCount.get(shardNames.get(pos)); j++) {
197             positions.add(new Position(shardNames.get(pos), j));
198           }
199         }
200         Collections.sort(positions);
201         for (Iterator<int[]> it = permutations(rules.size()); it.hasNext(); ) {
202           int[] permutation = it.next();
203           Map<Position, String> result = tryAPermutationOfRules(permutation, positions, fuzzyPhase);
204           if (result != null) return result;
205         }
206       }
207     }
208 
209     return null;
210   }

213   private Map<Position, String> tryAPermutationOfRules(final int[] rulePermutation, List<Position> positions, boolean fuzzyPhase) {
214     final Map<String, Map<String, Object>> nodeVsTagsCopy = getDeepCopy(nodeVsTags, 2);
215     Map<Position, String> result = new LinkedHashMap<>();
216     int startPosition = 0;
217     final Map<String, Map<String, Integer>> copyOfCurrentState = getDeepCopy(shardVsNodes, 2);
218     List<String> sortedLiveNodes = new ArrayList<>(this.liveNodes);
219     Collections.sort(sortedLiveNodes, new Comparator<String>() {
220       @Override
221       public int compare(String n1, String n2) {
222         int result = 0;
223         for (int i = 0; i < rulePermutation.length; i++) {
224           Rule rule = rules.get(rulePermutation[i]);
225           int val = rule.compare(n1, n2, nodeVsTagsCopy, copyOfCurrentState);
226           if (val != 0) {//atleast one non-zero compare break now
227             result = val;
228             break;
229           }
230           if (result == 0) {//if all else is equal, prefer nodes with fewer cores
231             AtomicInteger n1Count = nodeVsCores.get(n1);
232             AtomicInteger n2Count = nodeVsCores.get(n2);
233             int a = n1Count == null ? 0 : n1Count.get();
234             int b = n2Count == null ? 0 : n2Count.get();
235             result = a > b ? 1 : a == b ? 0 : -1;
236           }
237 
238         }
239         return result;
240       }
241     });
242     forEachPosition:
243     for (Position position : positions) {
244       //trying to assign a node by verifying each rule in this rulePermutation
245       forEachNode:
246       for (int j = 0; j < sortedLiveNodes.size(); j++) {
247         String liveNode = sortedLiveNodes.get(startPosition % sortedLiveNodes.size());
248         startPosition++;
249         for (int i = 0; i < rulePermutation.length; i++) {
250           Rule rule = rules.get(rulePermutation[i]);
251           //trying to assign a replica into this node in this shard
252           Rule.MatchStatus status = rule.tryAssignNodeToShard(liveNode,
253               copyOfCurrentState, nodeVsTagsCopy, position.shard, fuzzyPhase ? FUZZY_ASSIGN : ASSIGN);
254           if (status == Rule.MatchStatus.CANNOT_ASSIGN_FAIL) {
255             continue forEachNode;//try another node for this position
256           }
257         }
258         //We have reached this far means this node can be applied to this position
259         //and all rules are fine. So let us change the currentState
260         result.put(position, liveNode);
261         Map<String, Integer> nodeNames = copyOfCurrentState.get(position.shard);
262         if (nodeNames == null) copyOfCurrentState.put(position.shard, nodeNames = new HashMap<>());
263         Integer n = nodeNames.get(liveNode);
264         n = n == null ? 1 : n + 1;
265         nodeNames.put(liveNode, n);
266         Number coreCount = (Number) nodeVsTagsCopy.get(liveNode).get(ImplicitSnitch.CORES);
267         if (coreCount != null) {
268           nodeVsTagsCopy.get(liveNode).put(ImplicitSnitch.CORES, coreCount.intValue() + 1);
269         }
270 
271         continue forEachPosition;
272       }
273       //if it reached here, we could not find a node for this position
274       return null;
275     }
276 
277     if (positions.size() > result.size()) {
278       return null;
279     }
280 
281     for (Map.Entry<Position, String> e : result.entrySet()) {
282       for (int i = 0; i < rulePermutation.length; i++) {
283         Rule rule = rules.get(rulePermutation[i]);
284         Rule.MatchStatus matchStatus = rule.tryAssignNodeToShard(e.getValue(),
285             copyOfCurrentState, nodeVsTagsCopy, e.getKey().shard, fuzzyPhase ? FUZZY_VERIFY : VERIFY);
286         if (matchStatus != NODE_CAN_BE_ASSIGNED) return null;
287       }
288     }
289     return result;
290   }

292   private void validateTags(Map<String, Map<String, Object>> nodeVsTags) {
293     List<String> errors = new ArrayList<>();
294     for (Rule rule : rules) {
295       for (Map.Entry<String, Map<String, Object>> e : nodeVsTags.entrySet()) {
296         if (e.getValue().get(rule.tag.name) == null) {
297           errors.add(formatString("The value for tag {0} is not available for node {1}", rule.tag.name, e.getKey()));
298         }
299       }
300     }
301     if (!errors.isEmpty()) {
302       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, StrUtils.join(errors, ','));
303     }
304   }

307   /**
308    * get all permutations for the int[] whose items are 0..level
309    */
310   public static Iterator<int[]> permutations(final int level) {
311     return new Iterator<int[]>() {
312       int i = 0;
313       int[] next;
314 
315       @Override
316       public boolean hasNext() {
317         AtomicReference<int[]> nthval = new AtomicReference<>();
318         permute(0, new int[level], new BitSet(level), nthval, i, new AtomicInteger());
319         i++;
320         next = nthval.get();
321         return next != null;
322       }
323 
324       @Override
325       public int[] next() {
326         return next;
327       }
328 
329       @Override
330       public void remove() { }
331     };
332 
333   }

336   private static void permute(int level, int[] permuted, BitSet used, AtomicReference<int[]> nthval,
337                               int requestedIdx, AtomicInteger seenSoFar) {
338     if (level == permuted.length) {
339       if (seenSoFar.get() == requestedIdx) nthval.set(permuted);
340       else seenSoFar.incrementAndGet();
341     } else {
342       for (int i = 0; i < permuted.length; i++) {
343         if (!used.get(i)) {
344           used.set(i);
345           permuted[level] = i;
346           permute(level + 1, permuted, used, nthval, requestedIdx, seenSoFar);
347           if (nthval.get() != null) break;
348           used.set(i, false);
349         }
350       }
351     }
352   }

357   static class SnitchInfoImpl extends SnitchContext.SnitchInfo {
358     final Snitch snitch;
359     final Set<String> myTags = new HashSet<>();
360     final Map<String, SnitchContext> nodeVsContext = new HashMap<>();
361     private final CoreContainer cc;
362 
363     SnitchInfoImpl(Map<String, Object> conf, Snitch snitch, CoreContainer cc) {
364       super(conf);
365       this.snitch = snitch;
366       this.cc = cc;
367     }
368 
369     @Override
370     public Set<String> getTagNames() {
371       return myTags;
372     }
373 
374     @Override
375     public CoreContainer getCoreContainer() {
376       return cc;
377     }
378   }

380   /**
381    * This method uses the snitches and get the tags for all the nodes
382    */
383   private Map<String, Map<String, Object>> getTagsForNodes(final CoreContainer cc, List snitchConf) {
384 
385     Map<Class, SnitchInfoImpl> snitches = getSnitchInfos(cc, snitchConf);
386     for (Class c : Snitch.WELL_KNOWN_SNITCHES) {
387       if (snitches.containsKey(c)) continue;// it is already specified explicitly , ignore
388       try {
389         snitches.put(c, new SnitchInfoImpl(Collections.EMPTY_MAP, (Snitch) c.newInstance(), cc));
390       } catch (Exception e) {
391         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Error instantiating Snitch " + c.getName());
392       }
393     }
394     for (String tagName : tagNames) {
395       //identify which snitch is going to provide values for a given tag
396       boolean foundProvider = false;
397       for (SnitchInfoImpl info : snitches.values()) {
398         if (info.snitch.isKnownTag(tagName)) {
399           foundProvider = true;
400           info.myTags.add(tagName);
401           break;
402         }
403       }
404       if (!foundProvider)
405         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Unknown tag in rules " + tagName);
406     }
407 
408 
409     for (String node : liveNodes) {
410       //now use the Snitch to get the tags
411       for (SnitchInfoImpl info : snitches.values()) {
412         if (!info.myTags.isEmpty()) {
413           SnitchContext context = new SnitchContext(info, node);
414           info.nodeVsContext.put(node, context);
415           try {
416             info.snitch.getTags(node, info.myTags, context);
417           } catch (Exception e) {
418             context.exception = e;
419           }
420         }
421       }
422     }
423 
424     Map<String, Map<String, Object>> result = new HashMap<>();
425     for (SnitchInfoImpl info : snitches.values()) {
426       for (Map.Entry<String, SnitchContext> e : info.nodeVsContext.entrySet()) {
427         SnitchContext context = e.getValue();
428         String node = e.getKey();
429         if (context.exception != null) {
430           failedNodes.put(node, context);
431           liveNodes.remove(node);
432           log.warn("Not all tags were obtained from node " + node);
433           context.exception = new SolrException(SolrException.ErrorCode.SERVER_ERROR,
434               "Not all tags were obtained from node " + node);
435         } else {
436           if (context.getTags().keySet().containsAll(context.snitchInfo.getTagNames())) {
437             Map<String, Object> tags = result.get(node);
438             if (tags == null) {
439               tags = new HashMap<>();
440               result.put(node, tags);
441             }
442             tags.putAll(context.getTags());
443           }
444         }
445       }
446     }
447 
448     if (liveNodes.isEmpty()) {
449       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Could not get all tags for any nodes");
450 
451     }
452     return result;
453 
454   }

461   static Map<Class, SnitchInfoImpl> getSnitchInfos(CoreContainer cc, List snitchConf) {
462     if (snitchConf == null) snitchConf = Collections.emptyList();
463     Map<Class, SnitchInfoImpl> snitches = new LinkedHashMap<>();
464     for (Object o : snitchConf) {
465       //instantiating explicitly specified snitches
466       String klas = null;
467       Map map = Collections.emptyMap();
468       if (o instanceof Map) {//it can be a Map
469         map = (Map) o;
470         klas = (String) map.get("class");
471         if (klas == null) {
472           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "snitch must have  a class attribute");
473         }
474       } else { //or just the snitch name
475         klas = o.toString();
476       }
477       try {
478         if (klas.indexOf('.') == -1) klas = Snitch.class.getPackage().getName() + "." + klas;
479         Snitch inst = cc == null ?
480             (Snitch) Snitch.class.getClassLoader().loadClass(klas).newInstance() :
481             cc.getResourceLoader().newInstance(klas, Snitch.class);
482         snitches.put(inst.getClass(), new SnitchInfoImpl(map, inst, cc));
483       } catch (Exception e) {
484         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, e);
485 
486       }
487 
488     }
489     return snitches;
490   }
