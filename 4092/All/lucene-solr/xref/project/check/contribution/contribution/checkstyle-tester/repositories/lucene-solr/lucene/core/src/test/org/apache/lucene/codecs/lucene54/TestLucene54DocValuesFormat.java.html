148   private void doTestSparseDocValuesVsStoredFields() throws Exception {
149     final long[] values = new long[TestUtil.nextInt(random(), 1, 500)];
150     for (int i = 0; i < values.length; ++i) {
151       values[i] = random().nextLong();
152     }
153 
154     Directory dir = newFSDirectory(createTempDir());
155     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
156     conf.setMergeScheduler(new SerialMergeScheduler());
157     RandomIndexWriter writer = new RandomIndexWriter(random(), dir, conf);
158 
159     // sparse compression is only enabled if less than 1% of docs have a value
160     final int avgGap = 100;
161 
162     final int numDocs = atLeast(200);
163     for (int i = random().nextInt(avgGap * 2); i >= 0; --i) {
164       writer.addDocument(new Document());
165     }
166     final int maxNumValuesPerDoc = random().nextBoolean() ? 1 : TestUtil.nextInt(random(), 2, 5);
167     for (int i = 0; i < numDocs; ++i) {
168       Document doc = new Document();
169 
170       // single-valued
171       long docValue = values[random().nextInt(values.length)];
172       doc.add(new NumericDocValuesField("numeric", docValue));
173       doc.add(new SortedDocValuesField("sorted", new BytesRef(Long.toString(docValue))));
174       doc.add(new BinaryDocValuesField("binary", new BytesRef(Long.toString(docValue))));
175       doc.add(new StoredField("value", docValue));
176 
177       // multi-valued
178       final int numValues = TestUtil.nextInt(random(), 1, maxNumValuesPerDoc);
179       for (int j = 0; j < numValues; ++j) {
180         docValue = values[random().nextInt(values.length)];
181         doc.add(new SortedNumericDocValuesField("sorted_numeric", docValue));
182         doc.add(new SortedSetDocValuesField("sorted_set", new BytesRef(Long.toString(docValue))));
183         doc.add(new StoredField("values", docValue));
184       }
185 
186       writer.addDocument(doc);
187 
188       // add a gap
189       for (int j = TestUtil.nextInt(random(), 0, avgGap * 2); j >= 0; --j) {
190         writer.addDocument(new Document());
191       }
192     }
193 
194     if (random().nextBoolean()) {
195       writer.forceMerge(1);
196     }
197 
198     final IndexReader indexReader = writer.getReader();
199     writer.close();
200 
201     for (LeafReaderContext context : indexReader.leaves()) {
202       final LeafReader reader = context.reader();
203       final NumericDocValues numeric = DocValues.getNumeric(reader, "numeric");
204       final Bits numericBits = DocValues.getDocsWithField(reader, "numeric");
205 
206       final SortedDocValues sorted = DocValues.getSorted(reader, "sorted");
207       final Bits sortedBits = DocValues.getDocsWithField(reader, "sorted");
208 
209       final BinaryDocValues binary = DocValues.getBinary(reader, "binary");
210       final Bits binaryBits = DocValues.getDocsWithField(reader, "binary");
211 
212       final SortedNumericDocValues sortedNumeric = DocValues.getSortedNumeric(reader, "sorted_numeric");
213       final Bits sortedNumericBits = DocValues.getDocsWithField(reader, "sorted_numeric");
214 
215       final SortedSetDocValues sortedSet = DocValues.getSortedSet(reader, "sorted_set");
216       final Bits sortedSetBits = DocValues.getDocsWithField(reader, "sorted_set");
217 
218       for (int i = 0; i < reader.maxDoc(); ++i) {
219         final Document doc = reader.document(i);
220         final IndexableField valueField = doc.getField("value");
221         final Long value = valueField == null ? null : valueField.numericValue().longValue();
222 
223         if (value == null) {
224           assertEquals(0, numeric.get(i));
225           assertEquals(-1, sorted.getOrd(i));
226           assertEquals(new BytesRef(), binary.get(i));
227 
228           assertFalse(numericBits.get(i));
229           assertFalse(sortedBits.get(i));
230           assertFalse(binaryBits.get(i));
231         } else {
232           assertEquals(value.longValue(), numeric.get(i));
233           assertTrue(sorted.getOrd(i) >= 0);
234           assertEquals(new BytesRef(Long.toString(value)), sorted.lookupOrd(sorted.getOrd(i)));
235           assertEquals(new BytesRef(Long.toString(value)), binary.get(i));
236 
237           assertTrue(numericBits.get(i));
238           assertTrue(sortedBits.get(i));
239           assertTrue(binaryBits.get(i));
240         }
241 
242         final IndexableField[] valuesFields = doc.getFields("values");
243         final Set<Long> valueSet = new HashSet<>();
244         for (IndexableField sf : valuesFields) {
245           valueSet.add(sf.numericValue().longValue());
246         }
247 
248         sortedNumeric.setDocument(i);
249         assertEquals(valuesFields.length, sortedNumeric.count());
250         for (int j = 0; j < sortedNumeric.count(); ++j) {
251           assertTrue(valueSet.contains(sortedNumeric.valueAt(j)));
252         }
253         sortedSet.setDocument(i);
254         int sortedSetCount = 0;
255         while (true) {
256           long ord = sortedSet.nextOrd();
257           if (ord == SortedSetDocValues.NO_MORE_ORDS) {
258             break;
259           }
260           assertTrue(valueSet.contains(Long.parseLong(sortedSet.lookupOrd(ord).utf8ToString())));
261           sortedSetCount++;
262         }
263         assertEquals(valueSet.size(), sortedSetCount);
264 
265         assertEquals(!valueSet.isEmpty(), sortedNumericBits.get(i));
266         assertEquals(!valueSet.isEmpty(), sortedSetBits.get(i));
267       }
268     }
269 
270     indexReader.close();
271     dir.close();
272   }

277   private void doTestTermsEnumRandom(int numDocs, int minLength, int maxLength) throws Exception {
278     Directory dir = newFSDirectory(createTempDir());
279     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
280     conf.setMergeScheduler(new SerialMergeScheduler());
281     // set to duel against a codec which has ordinals:
282     final PostingsFormat pf = TestUtil.getPostingsFormatWithOrds(random());
283     final DocValuesFormat dv = new Lucene54DocValuesFormat();
284     conf.setCodec(new AssertingCodec() {
285       @Override
286       public PostingsFormat getPostingsFormatForField(String field) {
287         return pf;
288       }
289 
290       @Override
291       public DocValuesFormat getDocValuesFormatForField(String field) {
292         return dv;
293       }
294     });
295     RandomIndexWriter writer = new RandomIndexWriter(random(), dir, conf);
296     
297     // index some docs
298     for (int i = 0; i < numDocs; i++) {
299       Document doc = new Document();
300       Field idField = new StringField("id", Integer.toString(i), Field.Store.NO);
301       doc.add(idField);
302       final int length = TestUtil.nextInt(random(), minLength, maxLength);
303       int numValues = random().nextInt(17);
304       // create a random list of strings
305       List<String> values = new ArrayList<>();
306       for (int v = 0; v < numValues; v++) {
307         values.add(TestUtil.randomSimpleString(random(), minLength, length));
308       }
309       
310       // add in any order to the indexed field
311       ArrayList<String> unordered = new ArrayList<>(values);
312       Collections.shuffle(unordered, random());
313       for (String v : values) {
314         doc.add(newStringField("indexed", v, Field.Store.NO));
315       }
316 
317       // add in any order to the dv field
318       ArrayList<String> unordered2 = new ArrayList<>(values);
319       Collections.shuffle(unordered2, random());
320       for (String v : unordered2) {
321         doc.add(new SortedSetDocValuesField("dv", new BytesRef(v)));
322       }
323 
324       writer.addDocument(doc);
325       if (random().nextInt(31) == 0) {
326         writer.commit();
327       }
328     }
329     
330     // delete some docs
331     int numDeletions = random().nextInt(numDocs/10);
332     for (int i = 0; i < numDeletions; i++) {
333       int id = random().nextInt(numDocs);
334       writer.deleteDocuments(new Term("id", Integer.toString(id)));
335     }
336     
337     // compare per-segment
338     DirectoryReader ir = writer.getReader();
339     for (LeafReaderContext context : ir.leaves()) {
340       LeafReader r = context.reader();
341       Terms terms = r.terms("indexed");
342       if (terms != null) {
343         SortedSetDocValues ssdv = r.getSortedSetDocValues("dv");
344         assertEquals(terms.size(), ssdv.getValueCount());
345         TermsEnum expected = terms.iterator();
346         TermsEnum actual = r.getSortedSetDocValues("dv").termsEnum();
347         assertEquals(terms.size(), expected, actual);
348 
349         doTestSortedSetEnumAdvanceIndependently(ssdv);
350       }
351     }
352     ir.close();
353     
354     writer.forceMerge(1);
355     
356     // now compare again after the merge
357     ir = writer.getReader();
358     LeafReader ar = getOnlySegmentReader(ir);
359     Terms terms = ar.terms("indexed");
360     if (terms != null) {
361       assertEquals(terms.size(), ar.getSortedSetDocValues("dv").getValueCount());
362       TermsEnum expected = terms.iterator();
363       TermsEnum actual = ar.getSortedSetDocValues("dv").termsEnum();
364       assertEquals(terms.size(), expected, actual);
365     }
366     ir.close();
367     
368     writer.close();
369     dir.close();
370   }

372   private void assertEquals(long numOrds, TermsEnum expected, TermsEnum actual) throws Exception {
373     BytesRef ref;
374     
375     // sequential next() through all terms
376     while ((ref = expected.next()) != null) {
377       assertEquals(ref, actual.next());
378       assertEquals(expected.ord(), actual.ord());
379       assertEquals(expected.term(), actual.term());
380     }
381     assertNull(actual.next());
382     
383     // sequential seekExact(ord) through all terms
384     for (long i = 0; i < numOrds; i++) {
385       expected.seekExact(i);
386       actual.seekExact(i);
387       assertEquals(expected.ord(), actual.ord());
388       assertEquals(expected.term(), actual.term());
389     }
390     
391     // sequential seekExact(BytesRef) through all terms
392     for (long i = 0; i < numOrds; i++) {
393       expected.seekExact(i);
394       assertTrue(actual.seekExact(expected.term()));
395       assertEquals(expected.ord(), actual.ord());
396       assertEquals(expected.term(), actual.term());
397     }
398     
399     // sequential seekCeil(BytesRef) through all terms
400     for (long i = 0; i < numOrds; i++) {
401       expected.seekExact(i);
402       assertEquals(SeekStatus.FOUND, actual.seekCeil(expected.term()));
403       assertEquals(expected.ord(), actual.ord());
404       assertEquals(expected.term(), actual.term());
405     }
406     
407     // random seekExact(ord)
408     for (long i = 0; i < numOrds; i++) {
409       long randomOrd = TestUtil.nextLong(random(), 0, numOrds - 1);
410       expected.seekExact(randomOrd);
411       actual.seekExact(randomOrd);
412       assertEquals(expected.ord(), actual.ord());
413       assertEquals(expected.term(), actual.term());
414     }
415     
416     // random seekExact(BytesRef)
417     for (long i = 0; i < numOrds; i++) {
418       long randomOrd = TestUtil.nextLong(random(), 0, numOrds - 1);
419       expected.seekExact(randomOrd);
420       actual.seekExact(expected.term());
421       assertEquals(expected.ord(), actual.ord());
422       assertEquals(expected.term(), actual.term());
423     }
424     
425     // random seekCeil(BytesRef)
426     for (long i = 0; i < numOrds; i++) {
427       BytesRef target = new BytesRef(TestUtil.randomUnicodeString(random()));
428       SeekStatus expectedStatus = expected.seekCeil(target);
429       assertEquals(expectedStatus, actual.seekCeil(target));
430       if (expectedStatus != SeekStatus.END) {
431         assertEquals(expected.ord(), actual.ord());
432         assertEquals(expected.term(), actual.term());
433       }
434     }
435   }

437   public void testSparseLongValues() {
438     final int iters = atLeast(5);
439     for (int iter = 0; iter < iters; ++iter) {
440       final int numDocs = TestUtil.nextInt(random(), 0, 100);
441       final long[] docIds = new long[numDocs];
442       final long[] values = new long[numDocs];
443       final long maxDoc;
444       if (numDocs == 0) {
445         maxDoc = 1 + random().nextInt(10);
446       } else {
447         docIds[0] = random().nextInt(10);
448         for (int i = 1; i < docIds.length; ++i) {
449           docIds[i] = docIds[i - 1] + 1 + random().nextInt(100);
450         }
451         maxDoc = docIds[numDocs - 1] + 1 + random().nextInt(10);
452       }
453       for (int i = 0; i < values.length; ++i) {
454         values[i] = random().nextLong();
455       }
456       final long missingValue = random().nextLong();
457       final LongValues docIdsValues = new LongValues() {
458         @Override
459         public long get(long index) {
460           return docIds[(int) index];
461         }
462       };
463       final LongValues valuesValues = new LongValues() {
464         @Override
465         public long get(long index) {
466           return values[(int) index];
467         }
468       };
469       final SparseBits liveBits = new SparseBits(maxDoc, numDocs, docIdsValues);
470       // random-access
471       for (int i = 0; i < 2000; ++i) {
472         final long docId = TestUtil.nextLong(random(), 0, maxDoc - 1);
473         final boolean exists = liveBits.get((int) docId);
474         assertEquals(Arrays.binarySearch(docIds, docId) >= 0, exists);
475       }
476       // sequential access
477       for (int docId = 0; docId < maxDoc; docId += random().nextInt(3)) {
478         final boolean exists = liveBits.get(docId);
479         assertEquals(Arrays.binarySearch(docIds, docId) >= 0, exists);
480       }
481 
482       final SparseLongValues sparseValues = new SparseLongValues(liveBits, valuesValues, missingValue);
483       // random-access
484       for (int i = 0; i < 2000; ++i) {
485         final long docId = TestUtil.nextLong(random(), 0, maxDoc - 1);
486         final int idx = Arrays.binarySearch(docIds, docId);
487         final long value = sparseValues.get(docId);
488         if (idx >= 0) {
489           assertEquals(values[idx], value);
490         } else {
491           assertEquals(missingValue, value);
492         }
493       }
494       // sequential access
495       for (int docId = 0; docId < maxDoc; docId += random().nextInt(3)) {
496         final int idx = Arrays.binarySearch(docIds, docId);
497         final long value = sparseValues.get(docId);
498         if (idx >= 0) {
499           assertEquals(values[idx], value);
500         } else {
501           assertEquals(missingValue, value);
502         }
503       }
504     }
505   }
