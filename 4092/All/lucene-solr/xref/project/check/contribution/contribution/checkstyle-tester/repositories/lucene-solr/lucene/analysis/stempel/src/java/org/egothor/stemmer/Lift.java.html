55  package org.egothor.stemmer;
56  
57  import java.util.ArrayList;
58  import java.util.Arrays;
59  import java.util.Iterator;
60  import java.util.List;
61  
62  /**
63   * The Lift class is a data structure that is a variation of a Patricia trie.
64   * <p>
65   * Lift's <i>raison d'etre</i> is to implement reduction of the trie via the
66   * Lift-Up method., which makes the data structure less liable to overstemming.
67   */
68  public class Lift extends Reduce {
69    boolean changeSkip;
70    
71    /**
72     * Constructor for the Lift object.
73     * 
74     * @param changeSkip when set to <tt>true</tt>, comparison of two Cells takes
75     *          a skip command into account
76     */
77    public Lift(boolean changeSkip) {
78      this.changeSkip = changeSkip;
79    }
80    
81    /**
82     * Optimize (eliminate rows with no content) the given Trie and return the
83     * reduced Trie.
84     * 
85     * @param orig the Trie to optimized
86     * @return the reduced Trie
87     */
88    @Override
89    public Trie optimize(Trie orig) {
90      List<CharSequence> cmds = orig.cmds;
91      List<Row> rows = new ArrayList<>();
92      List<Row> orows = orig.rows;
93      int remap[] = new int[orows.size()];
94      
95      for (int j = orows.size() - 1; j >= 0; j--) {
96        liftUp(orows.get(j), orows);
97      }
98      
99      Arrays.fill(remap, -1);
100     rows = removeGaps(orig.root, orows, new ArrayList<Row>(), remap);
101     
102     return new Trie(orig.forward, remap[orig.root], cmds, rows);
103   }
104   
105   /**
106    * Reduce the trie using Lift-Up reduction.
107    * <p>
108    * The Lift-Up reduction propagates all leaf-values (patch commands), where
109    * possible, to higher levels which are closer to the root of the trie.
110    * 
111    * @param in the Row to consider when optimizing
112    * @param nodes contains the patch commands
113    */
114   public void liftUp(Row in, List<Row> nodes) {
115     Iterator<Cell> i = in.cells.values().iterator();
116     for (; i.hasNext();) {
117       Cell c = i.next();
118       if (c.ref >= 0) {
119         Row to = nodes.get(c.ref);
120         int sum = to.uniformCmd(changeSkip);
121         if (sum >= 0) {
122           if (sum == c.cmd) {
123             if (changeSkip) {
124               if (c.skip != to.uniformSkip + 1) {
125                 continue;
126               }
127               c.skip = to.uniformSkip + 1;
128             } else {
129               c.skip = 0;
130             }
131             c.cnt += to.uniformCnt;
132             c.ref = -1;
133           } else if (c.cmd < 0) {
134             c.cnt = to.uniformCnt;
135             c.cmd = sum;
136             c.ref = -1;
137             if (changeSkip) {
138               c.skip = to.uniformSkip + 1;
139             } else {
140               c.skip = 0;
141             }
142           }
143         }
144       }
145     }
146   }
