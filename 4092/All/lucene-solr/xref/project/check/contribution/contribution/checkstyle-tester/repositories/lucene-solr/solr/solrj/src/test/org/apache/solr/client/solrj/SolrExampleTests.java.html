321   private String randomTestString(int maxLength) {
322     // we can't just use _TestUtil.randomUnicodeString() or we might get 0xfffe etc
323     // (considered invalid by XML)
324     
325     int size = random().nextInt(maxLength);
326     StringBuilder sb = new StringBuilder();
327     for (int i = 0; i < size; i++) {
328       switch(random().nextInt(4)) {
329         case 0: /* single byte */ 
330           sb.append('a'); 
331           break;
332         case 1: /* two bytes */
333           sb.append('\u0645');
334           break;
335         case 2: /* three bytes */
336           sb.append('\u092a');
337           break;
338         case 3: /* four bytes */
339           sb.appendCodePoint(0x29B05);
340       }
341     }
342     return sb.toString();
343   }

361         if (client instanceof HttpSolrClient) {
362           if (random.nextBoolean()) {
363             ((HttpSolrClient) client).setParser(new BinaryResponseParser());
364           } else {
365             ((HttpSolrClient) client).setParser(new XMLResponseParser());
366           }
367         }

370         
371         // Empty the database...
372         client.deleteByQuery("*:*");// delete everything!
373

375         for (int i = 0; i < numDocs; i++) {
376           // Now add something...
377           SolrInputDocument doc = new SolrInputDocument();
378           doc.addField("id", "" + i);
379           doc.addField("unicode_s", randomTestString(30));
380           docs.add(doc);
381         }
382         
383         client.add(docs);
384         client.commit();
385         
386         SolrQuery query = new SolrQuery();
387         query.setQuery("*:*");
388         query.setRows(numDocs);

390         QueryResponse rsp = client.query( query );
391         
392         for (int i = 0; i < numDocs; i++) {
393           String expected = (String) docs.get(i).getFieldValue("unicode_s");
394           String actual = (String) rsp.getResults().get(i).getFieldValue("unicode_s");

436       assertTrue(ex.getMessage().indexOf("??::?? ignore_exception")>0);  // The reason should get passed through

440       Assert.fail("should have thrown a SolrException! not: "+t);

2029   /** 
2030    * Depth first search of a SolrInputDocument looking for a decendent by id, 
2031    * returns null if it's not a decendent 
2032    */
2033   private SolrInputDocument findDecendent(SolrInputDocument parent, String childId) {
2034     if (childId.equals(parent.getFieldValue("id"))) {
2035       return parent;
2036     }
2037     if (! parent.hasChildDocuments() ) {
2038       return null;
2039     }
2040     for (SolrInputDocument kid : parent.getChildDocuments()) {
2041       SolrInputDocument result = findDecendent(kid, childId);
2042       if (null != result) {
2043         return result;
2044       }
2045     }
2046     return null;
2047   }
