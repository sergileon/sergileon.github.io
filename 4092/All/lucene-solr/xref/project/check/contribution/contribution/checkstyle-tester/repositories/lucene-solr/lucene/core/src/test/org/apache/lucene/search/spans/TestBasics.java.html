44  /**
45   * Tests basic search capabilities.
46   *
47   * <p>Uses a collection of 1000 documents, each the english rendition of their
48   * document number.  For example, the document numbered 333 has text "three
49   * hundred thirty three".
50   *
51   * <p>Tests are each a single query, and its hits are checked to ensure that
52   * all and only the correct documents are returned, thus providing end-to-end
53   * testing of the indexing and search code.
54   *
55   */
56  public class TestBasics extends LuceneTestCase {
57    private static IndexSearcher searcher;
58    private static IndexReader reader;
59    private static Directory directory;
60  
61    @BeforeClass
62    public static void beforeClass() throws Exception {
63      directory = newDirectory();
64      RandomIndexWriter writer = new RandomIndexWriter(random(), directory,
65          newIndexWriterConfig(new MockAnalyzer(random(), MockTokenizer.SIMPLE, true))
66              .setMaxBufferedDocs(TestUtil.nextInt(random(), 100, 1000)).setMergePolicy(newLogMergePolicy()));
67      //writer.infoStream = System.out;
68      for (int i = 0; i < 2000; i++) {
69        Document doc = new Document();
70        doc.add(newTextField("field", English.intToEnglish(i), Field.Store.YES));
71        writer.addDocument(doc);
72      }
73      reader = writer.getReader();
74      searcher = newSearcher(reader);
75      writer.close();
76    }
77  
78    @AfterClass
79    public static void afterClass() throws Exception {
80      reader.close();
81      directory.close();
82      searcher = null;
83      reader = null;
84      directory = null;
85    }
