39  /**
40   * Implements a {@link ContentSource} over the TREC collection.
41   * <p>
42   * Supports the following configuration parameters (on top of
43   * {@link ContentSource}):
44   * <ul>
45   * <li><b>work.dir</b> - specifies the working directory. Required if "docs.dir"
46   * denotes a relative path (<b>default=work</b>).
47   * <li><b>docs.dir</b> - specifies the directory where the TREC files reside.
48   * Can be set to a relative path if "work.dir" is also specified
49   * (<b>default=trec</b>).
50   * <li><b>trec.doc.parser</b> - specifies the {@link TrecDocParser} class to use for
51   * parsing the TREC documents content (<b>default=TrecGov2Parser</b>).
52   * <li><b>html.parser</b> - specifies the {@link HTMLParser} class to use for
53   * parsing the HTML parts of the TREC documents content (<b>default=DemoHTMLParser</b>).
54   * <li><b>content.source.encoding</b> - if not specified, ISO-8859-1 is used.
55   * <li><b>content.source.excludeIteration</b> - if true, do not append iteration number to docname
56   * </ul>
57   */
58  public class TrecContentSource extends ContentSource {

60    static final class DateFormatInfo {
61      DateFormat[] dfs;
62      ParsePosition pos;
63    }
64  
65    public static final String DOCNO = "<DOCNO>";
66    public static final String TERMINATING_DOCNO = "</DOCNO>";
67    public static final String DOC = "<DOC>";
68    public static final String TERMINATING_DOC = "</DOC>";
69  
70    /** separator between lines in the byffer */ 
71    public static final String NEW_LINE = System.getProperty("line.separator");
72  
73    private static final String DATE_FORMATS [] = {
74         "EEE, dd MMM yyyy kk:mm:ss z",   // Tue, 09 Dec 2003 22:39:08 GMT
75         "EEE MMM dd kk:mm:ss yyyy z",    // Tue Dec 09 16:45:08 2003 EST
76         "EEE, dd-MMM-':'y kk:mm:ss z",   // Tue, 09 Dec 2003 22:39:08 GMT
77         "EEE, dd-MMM-yyy kk:mm:ss z",    // Tue, 09 Dec 2003 22:39:08 GMT
78         "EEE MMM dd kk:mm:ss yyyy",      // Tue Dec 09 16:45:08 2003
79         "dd MMM yyyy",                   // 1 March 1994
80         "MMM dd, yyyy",                  // February 3, 1994
81         "yyMMdd",                        // 910513
82         "hhmm z.z.z. MMM dd, yyyy",       // 0901 u.t.c. April 28, 1994
83    };

85    private ThreadLocal<DateFormatInfo> dateFormats = new ThreadLocal<>();
86    private ThreadLocal<StringBuilder> trecDocBuffer = new ThreadLocal<>();
87    private Path dataDir = null;
88    private ArrayList<Path> inputFiles = new ArrayList<>();
89    private int nextFile = 0;
90    // Use to synchronize threads on reading from the TREC documents.
91    private Object lock = new Object();
92  
93    // Required for test
94    BufferedReader reader;
95    int iteration = 0;
96    HTMLParser htmlParser;
97    
98    private boolean excludeDocnameIteration;
99    private TrecDocParser trecDocParser = new TrecGov2Parser(); // default
100   ParsePathType currPathType; // not private for tests
101   
102   private DateFormatInfo getDateFormatInfo() {
103     DateFormatInfo dfi = dateFormats.get();
104     if (dfi == null) {
105       dfi = new DateFormatInfo();
106       dfi.dfs = new SimpleDateFormat[DATE_FORMATS.length];
107       for (int i = 0; i < dfi.dfs.length; i++) {
108         dfi.dfs[i] = new SimpleDateFormat(DATE_FORMATS[i], Locale.ENGLISH);
109         dfi.dfs[i].setLenient(true);
110       }
111       dfi.pos = new ParsePosition(0);
112       dateFormats.set(dfi);
113     }
114     return dfi;
115   }
116 
117   private StringBuilder getDocBuffer() {
118     StringBuilder sb = trecDocBuffer.get();
119     if (sb == null) {
120       sb = new StringBuilder();
121       trecDocBuffer.set(sb);
122     }
123     return sb;
124   }
125   
126   HTMLParser getHtmlParser() {
127     return htmlParser;
128   }

130   /**
131    * Read until a line starting with the specified <code>lineStart</code>.
132    * @param buf buffer for collecting the data if so specified/ 
133    * @param lineStart line start to look for, must not be null.
134    * @param collectMatchLine whether to collect the matching line into <code>buffer</code>.
135    * @param collectAll whether to collect all lines into <code>buffer</code>.
136    * @throws IOException If there is a low-level I/O error.
137    * @throws NoMoreDataException If the source is exhausted.
138    */
139    private void read(StringBuilder buf, String lineStart, 
140        boolean collectMatchLine, boolean collectAll) throws IOException, NoMoreDataException {
141     String sep = "";
142     while (true) {
143       String line = reader.readLine();
144 
145       if (line == null) {
146         openNextFile();
147         continue;
148       }
149 
150       line.length();
151 
152       if (lineStart!=null && line.startsWith(lineStart)) {
153         if (collectMatchLine) {
154           buf.append(sep).append(line);
155           sep = NEW_LINE;
156         }
157         return;
158       }
159 
160       if (collectAll) {
161         buf.append(sep).append(line);
162         sep = NEW_LINE;
163       }
164     }
165   }
166   
167   void openNextFile() throws NoMoreDataException, IOException {
168     close();
169     currPathType = null;
170     while (true) {
171       if (nextFile >= inputFiles.size()) { 
172         // exhausted files, start a new round, unless forever set to false.
173         if (!forever) {
174           throw new NoMoreDataException();
175         }
176         nextFile = 0;
177         iteration++;
178       }
179       Path f = inputFiles.get(nextFile++);
180       if (verbose) {
181         System.out.println("opening: " + f + " length: " + Files.size(f));
182       }
183       try {
184         InputStream inputStream = StreamUtils.inputStream(f); // support either gzip, bzip2, or regular text file, by extension  
185         reader = new BufferedReader(new InputStreamReader(inputStream, encoding), StreamUtils.BUFFER_SIZE);
186         currPathType = TrecDocParser.pathType(f);
187         return;
188       } catch (Exception e) {
189         if (verbose) {
190           System.out.println("Skipping 'bad' file " + f.toAbsolutePath()+" due to "+e.getMessage());
191           continue;
192         }
193         throw new NoMoreDataException();
194       }
195     }
196   }

198   public Date parseDate(String dateStr) {
199     dateStr = dateStr.trim();
200     DateFormatInfo dfi = getDateFormatInfo();
201     for (int i = 0; i < dfi.dfs.length; i++) {
202       DateFormat df = dfi.dfs[i];
203       dfi.pos.setIndex(0);
204       dfi.pos.setErrorIndex(-1);
205       Date d = df.parse(dateStr, dfi.pos);
206       if (d != null) {
207         // Parse succeeded.
208         return d;
209       }
210     }
211     // do not fail test just because a date could not be parsed
212     if (verbose) {
213       System.out.println("failed to parse date (assigning 'now') for: " + dateStr);
214     }
215     return null; 
216   }

218   @Override
219   public void close() throws IOException {
220     if (reader == null) {
221       return;
222     }
223 
224     try {
225       reader.close();
226     } catch (IOException e) {
227       if (verbose) {
228         System.out.println("failed to close reader !");
229         e.printStackTrace(System.out);
230       }
231     }
232     reader = null;
233   }

235   @Override
236   public DocData getNextDocData(DocData docData) throws NoMoreDataException, IOException {
237     String name = null;
238     StringBuilder docBuf = getDocBuffer();
239     ParsePathType parsedPathType;
240     
241     // protect reading from the TREC files by multiple threads. The rest of the
242     // method, i.e., parsing the content and returning the DocData can run unprotected.
243     synchronized (lock) {
244       if (reader == null) {
245         openNextFile();
246       }
247       
248       // 1. skip until doc start - required for all TREC formats
249       docBuf.setLength(0);
250       read(docBuf, DOC, false, false);
251       
252       // save parsedFile for passing trecDataParser after the sync block, in 
253       // case another thread will open another file in between.
254       parsedPathType = currPathType;
255       
256       // 2. name - required for all TREC formats
257       docBuf.setLength(0);
258       read(docBuf, DOCNO, true, false);
259       name = docBuf.substring(DOCNO.length(), docBuf.indexOf(TERMINATING_DOCNO,
260           DOCNO.length())).trim();
261       
262       if (!excludeDocnameIteration) {
263         name = name + "_" + iteration;
264       }
265 
266       // 3. read all until end of doc
267       docBuf.setLength(0);
268       read(docBuf, TERMINATING_DOC, false, true);
269     }
270       
271     // count char length of text to be parsed (may be larger than the resulted plain doc body text).
272     addBytes(docBuf.length()); 
273 
274     // This code segment relies on HtmlParser being thread safe. When we get 
275     // here, everything else is already private to that thread, so we're safe.
276     docData = trecDocParser.parse(docData, name, this, docBuf, parsedPathType);
277     addItem();
278 
279     return docData;
280   }
281 
282   @Override
283   public void resetInputs() throws IOException {
284     synchronized (lock) {
285       super.resetInputs();
286       close();
287       nextFile = 0;
288       iteration = 0;
289     }
290   }

292   @Override
293   public void setConfig(Config config) {
294     super.setConfig(config);
295     // dirs
296     Path workDir = Paths.get(config.get("work.dir", "work"));
297     String d = config.get("docs.dir", "trec");
298     dataDir = Paths.get(d);
299     if (!dataDir.isAbsolute()) {
300       dataDir = workDir.resolve(d);
301     }
302     // files
303     try {
304       collectFiles(dataDir, inputFiles);
305     } catch (IOException e) {
306       throw new RuntimeException(e);
307     }
308     if (inputFiles.size() == 0) {
309       throw new IllegalArgumentException("No files in dataDir: " + dataDir);
310     }
311     // trec doc parser
312     try {
313       String trecDocParserClassName = config.get("trec.doc.parser", "org.apache.lucene.benchmark.byTask.feeds.TrecGov2Parser");
314       trecDocParser = Class.forName(trecDocParserClassName).asSubclass(TrecDocParser.class).newInstance();
315     } catch (Exception e) {
316       // Should not get here. Throw runtime exception.
317       throw new RuntimeException(e);
318     }
319     // html parser
320     try {
321       String htmlParserClassName = config.get("html.parser",
322           "org.apache.lucene.benchmark.byTask.feeds.DemoHTMLParser");
323       htmlParser = Class.forName(htmlParserClassName).asSubclass(HTMLParser.class).newInstance();
324     } catch (Exception e) {
325       // Should not get here. Throw runtime exception.
326       throw new RuntimeException(e);
327     }
328     // encoding
329     if (encoding == null) {
330       encoding = StandardCharsets.ISO_8859_1.name();
331     }
332     // iteration exclusion in doc name 
333     excludeDocnameIteration = config.get("content.source.excludeIteration", false);
334   }
