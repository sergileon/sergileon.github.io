32  /**
33   * An abstract {@link Query} that matches documents
34   * containing a subset of terms provided by a {@link
35   * FilteredTermsEnum} enumeration.
36   *
37   * <p>This query cannot be used directly; you must subclass
38   * it and define {@link #getTermsEnum(Terms,AttributeSource)} to provide a {@link
39   * FilteredTermsEnum} that iterates through the terms to be
40   * matched.
41   *
42   * <p><b>NOTE</b>: if {@link #setRewriteMethod} is either
43   * {@link #CONSTANT_SCORE_BOOLEAN_REWRITE} or {@link
44   * #SCORING_BOOLEAN_REWRITE}, you may encounter a
45   * {@link BooleanQuery.TooManyClauses} exception during
46   * searching, which happens when the number of terms to be
47   * searched exceeds {@link
48   * BooleanQuery#getMaxClauseCount()}.  Setting {@link
49   * #setRewriteMethod} to {@link #CONSTANT_SCORE_REWRITE}
50   * prevents this.
51   *
52   * <p>The recommended rewrite method is {@link
53   * #CONSTANT_SCORE_REWRITE}: it doesn't spend CPU
54   * computing unhelpful scores, and is the most
55   * performant rewrite method given the query. If you
56   * need scoring (like {@link FuzzyQuery}, use
57   * {@link TopTermsScoringBooleanQueryRewrite} which uses
58   * a priority queue to only collect competitive terms
59   * and not hit this limitation.
60   *
61   * Note that org.apache.lucene.queryparser.classic.QueryParser produces
62   * MultiTermQueries using {@link #CONSTANT_SCORE_REWRITE}
63   * by default.
64   */
65  public abstract class MultiTermQuery extends Query {
66    protected final String field;
67    protected RewriteMethod rewriteMethod = CONSTANT_SCORE_REWRITE;
68  
69    /** Abstract class that defines how the query is rewritten. */
70    public static abstract class RewriteMethod {
71      public abstract Query rewrite(IndexReader reader, MultiTermQuery query) throws IOException;
72      /**
73       * Returns the {@link MultiTermQuery}s {@link TermsEnum}
74       * @see MultiTermQuery#getTermsEnum(Terms, AttributeSource)
75       */
76      protected TermsEnum getTermsEnum(MultiTermQuery query, Terms terms, AttributeSource atts) throws IOException {
77        return query.getTermsEnum(terms, atts); // allow RewriteMethod subclasses to pull a TermsEnum from the MTQ 
78      }
79    }
80  
81    /** A rewrite method that first creates a private Filter,
82     *  by visiting each term in sequence and marking all docs
83     *  for that term.  Matching documents are assigned a
84     *  constant score equal to the query's boost.
85     * 
86     *  <p> This method is faster than the BooleanQuery
87     *  rewrite methods when the number of matched terms or
88     *  matched documents is non-trivial. Also, it will never
89     *  hit an errant {@link BooleanQuery.TooManyClauses}
90     *  exception.
91     *
92     *  @see #setRewriteMethod */
93    public static final RewriteMethod CONSTANT_SCORE_REWRITE = new RewriteMethod() {
94      @Override
95      public Query rewrite(IndexReader reader, MultiTermQuery query) {
96        return new MultiTermQueryConstantScoreWrapper<>(query);
97      }
98    };
99  
100   /** Old name of {@link #CONSTANT_SCORE_REWRITE}
101    *  @deprecated old name of {@link #CONSTANT_SCORE_REWRITE} */
102   @Deprecated
103   public static final RewriteMethod CONSTANT_SCORE_FILTER_REWRITE = CONSTANT_SCORE_REWRITE;
104 
105   /** A rewrite method that first translates each term into
106    *  {@link BooleanClause.Occur#SHOULD} clause in a
107    *  BooleanQuery, and keeps the scores as computed by the
108    *  query.  Note that typically such scores are
109    *  meaningless to the user, and require non-trivial CPU
110    *  to compute, so it's almost always better to use {@link
111    *  #CONSTANT_SCORE_REWRITE} instead.
112    *
113    *  <p><b>NOTE</b>: This rewrite method will hit {@link
114    *  BooleanQuery.TooManyClauses} if the number of terms
115    *  exceeds {@link BooleanQuery#getMaxClauseCount}.
116    *
117    *  @see #setRewriteMethod */
118   public final static RewriteMethod SCORING_BOOLEAN_REWRITE = ScoringRewrite.SCORING_BOOLEAN_REWRITE;
119   
120   /** Old name of {@link #SCORING_BOOLEAN_REWRITE}
121    *  @deprecated old name of {@link #SCORING_BOOLEAN_REWRITE} */
122   @Deprecated
123   public final static RewriteMethod SCORING_BOOLEAN_QUERY_REWRITE = SCORING_BOOLEAN_REWRITE;
124   
125   /** Like {@link #SCORING_BOOLEAN_REWRITE} except
126    *  scores are not computed.  Instead, each matching
127    *  document receives a constant score equal to the
128    *  query's boost.
129    * 
130    *  <p><b>NOTE</b>: This rewrite method will hit {@link
131    *  BooleanQuery.TooManyClauses} if the number of terms
132    *  exceeds {@link BooleanQuery#getMaxClauseCount}.
133    *
134    *  @see #setRewriteMethod */
135   public final static RewriteMethod CONSTANT_SCORE_BOOLEAN_REWRITE = ScoringRewrite.CONSTANT_SCORE_BOOLEAN_REWRITE;
136 
137   /** Old name of {@link #CONSTANT_SCORE_BOOLEAN_REWRITE}
138    *  @deprecated old name of {@link #CONSTANT_SCORE_BOOLEAN_REWRITE} */
139   @Deprecated
140   public final static RewriteMethod CONSTANT_SCORE_BOOLEAN_QUERY_REWRITE = CONSTANT_SCORE_BOOLEAN_REWRITE;
141 
142   /**
143    * A rewrite method that first translates each term into
144    * {@link BooleanClause.Occur#SHOULD} clause in a BooleanQuery, and keeps the
145    * scores as computed by the query.
146    * 
147    * <p>
148    * This rewrite method only uses the top scoring terms so it will not overflow
149    * the boolean max clause count. It is the default rewrite method for
150    * {@link FuzzyQuery}.
151    * 
152    * @see #setRewriteMethod
153    */
154   public static final class TopTermsScoringBooleanQueryRewrite extends TopTermsRewrite<BooleanQuery.Builder> {
155 
156     /** 
157      * Create a TopTermsScoringBooleanQueryRewrite for 
158      * at most <code>size</code> terms.
159      * <p>
160      * NOTE: if {@link BooleanQuery#getMaxClauseCount} is smaller than 
161      * <code>size</code>, then it will be used instead. 
162      */
163     public TopTermsScoringBooleanQueryRewrite(int size) {
164       super(size);
165     }
166     
167     @Override
168     protected int getMaxSize() {
169       return BooleanQuery.getMaxClauseCount();
170     }
171     
172     @Override
173     protected BooleanQuery.Builder getTopLevelBuilder() {
174       BooleanQuery.Builder builder = new BooleanQuery.Builder();
175       builder.setDisableCoord(true);
176       return builder;
177     }
178 
179     @Override
180     protected Query build(BooleanQuery.Builder builder) {
181       return builder.build();
182     }
183     
184     @Override
185     protected void addClause(BooleanQuery.Builder topLevel, Term term, int docCount, float boost, TermContext states) {
186       final TermQuery tq = new TermQuery(term, states);
187       topLevel.add(new BoostQuery(tq, boost), BooleanClause.Occur.SHOULD);
188     }
189   }
190   
191   /**
192    * A rewrite method that first translates each term into
193    * {@link BooleanClause.Occur#SHOULD} clause in a BooleanQuery, but adjusts
194    * the frequencies used for scoring to be blended across the terms, otherwise
195    * the rarest term typically ranks highest (often not useful eg in the set of
196    * expanded terms in a FuzzyQuery).
197    * 
198    * <p>
199    * This rewrite method only uses the top scoring terms so it will not overflow
200    * the boolean max clause count.
201    * 
202    * @see #setRewriteMethod
203    */
204   public static final class TopTermsBlendedFreqScoringRewrite extends
205       TopTermsRewrite<BlendedTermQuery.Builder> {
206 
207     /**
208      * Create a TopTermsBlendedScoringBooleanQueryRewrite for at most
209      * <code>size</code> terms.
210      * <p>
211      * NOTE: if {@link BooleanQuery#getMaxClauseCount} is smaller than
212      * <code>size</code>, then it will be used instead.
213      */
214     public TopTermsBlendedFreqScoringRewrite(int size) {
215       super(size);
216     }
217 
218     @Override
219     protected int getMaxSize() {
220       return BooleanQuery.getMaxClauseCount();
221     }
222 
223     @Override
224     protected BlendedTermQuery.Builder getTopLevelBuilder() {
225       BlendedTermQuery.Builder builder = new BlendedTermQuery.Builder();
226       builder.setRewriteMethod(BlendedTermQuery.BOOLEAN_REWRITE);
227       return builder;
228     }
229 
230     @Override
231     protected Query build(BlendedTermQuery.Builder builder) {
232       return builder.build();
233     }
234 
235     @Override
236     protected void addClause(BlendedTermQuery.Builder topLevel, Term term, int docCount,
237         float boost, TermContext states) {
238       topLevel.add(term, boost, states);
239     }
240   }
241 
242   /**
243    * A rewrite method that first translates each term into
244    * {@link BooleanClause.Occur#SHOULD} clause in a BooleanQuery, but the scores
245    * are only computed as the boost.
246    * <p>
247    * This rewrite method only uses the top scoring terms so it will not overflow
248    * the boolean max clause count.
249    * 
250    * @see #setRewriteMethod
251    */
252   public static final class TopTermsBoostOnlyBooleanQueryRewrite extends TopTermsRewrite<BooleanQuery.Builder> {
253     
254     /** 
255      * Create a TopTermsBoostOnlyBooleanQueryRewrite for 
256      * at most <code>size</code> terms.
257      * <p>
258      * NOTE: if {@link BooleanQuery#getMaxClauseCount} is smaller than 
259      * <code>size</code>, then it will be used instead. 
260      */
261     public TopTermsBoostOnlyBooleanQueryRewrite(int size) {
262       super(size);
263     }
264     
265     @Override
266     protected int getMaxSize() {
267       return BooleanQuery.getMaxClauseCount();
268     }
269     
270     @Override
271     protected BooleanQuery.Builder getTopLevelBuilder() {
272       BooleanQuery.Builder builder = new BooleanQuery.Builder();
273       builder.setDisableCoord(true);
274       return builder;
275     }
276     
277     @Override
278     protected Query build(BooleanQuery.Builder builder) {
279       return builder.build();
280     }
281     
282     @Override
283     protected void addClause(BooleanQuery.Builder topLevel, Term term, int docFreq, float boost, TermContext states) {
284       final Query q = new ConstantScoreQuery(new TermQuery(term, states));
285       topLevel.add(new BoostQuery(q, boost), BooleanClause.Occur.SHOULD);
286     }
287   }
288 
289   /**
290    * Constructs a query matching terms that cannot be represented with a single
291    * Term.
292    */
293   public MultiTermQuery(final String field) {
294     this.field = Objects.requireNonNull(field, "field must not be null");
295   }
296 
297   /** Returns the field name for this query */
298   public final String getField() { return field; }
299 
300   /** Construct the enumeration to be used, expanding the
301    *  pattern term.  This method should only be called if
302    *  the field exists (ie, implementations can assume the
303    *  field does exist).  This method should not return null
304    *  (should instead return {@link TermsEnum#EMPTY} if no
305    *  terms match).  The TermsEnum must already be
306    *  positioned to the first matching term.
307    * The given {@link AttributeSource} is passed by the {@link RewriteMethod} to
308    * provide attributes, the rewrite method uses to inform about e.g. maximum competitive boosts.
309    * This is currently only used by {@link TopTermsRewrite}
310    */
311   protected abstract TermsEnum getTermsEnum(Terms terms, AttributeSource atts) throws IOException;
312 
313   /** Convenience method, if no attributes are needed:
314    * This simply passes empty attributes and is equal to:
315    * <code>getTermsEnum(terms, new AttributeSource())</code>
316    */
317   protected final TermsEnum getTermsEnum(Terms terms) throws IOException {
318     return getTermsEnum(terms, new AttributeSource());
319   }
320 
321   /**
322    * To rewrite to a simpler form, instead return a simpler
323    * enum from {@link #getTermsEnum(Terms, AttributeSource)}.  For example,
324    * to rewrite to a single term, return a {@link SingleTermsEnum}
325    */
326   @Override
327   public final Query rewrite(IndexReader reader) throws IOException {
328     if (getBoost() != 1f) {
329       return super.rewrite(reader);
330     }
331     return rewriteMethod.rewrite(reader, this);
332   }
333 
334   /**
335    * @see #setRewriteMethod
336    */
337   public RewriteMethod getRewriteMethod() {
338     return rewriteMethod;
339   }
340 
341   /**
342    * Sets the rewrite method to be used when executing the
343    * query.  You can use one of the four core methods, or
344    * implement your own subclass of {@link RewriteMethod}. */
345   public void setRewriteMethod(RewriteMethod method) {
346     rewriteMethod = method;
347   }
348 
349   @Override
350   public int hashCode() {
351     int h = super.hashCode();
352     h = 31 * h + rewriteMethod.hashCode();
353     h = 31 * h + Objects.hashCode(field);
354     return h;
355   }
356 
357   @Override
358   public boolean equals(Object obj) {
359     if (this == obj)
360       return true;
361     if (obj == null)
362       return false;
363     if (getClass() != obj.getClass())
364       return false;
365     MultiTermQuery other = (MultiTermQuery) obj;
366     if (!super.equals(obj))
367       return false;
368     if (!rewriteMethod.equals(other.rewriteMethod)) {
369       return false;
370     }
371     return (other.field == null ? field == null : other.field.equals(field));
372   }
