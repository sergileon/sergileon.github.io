73          if (fieldName.contains("payloadsFixed")) {
74            TokenFilter filter = new MockFixedLengthPayloadFilter(new Random(0), tokenizer, 1);
75            return new TokenStreamComponents(tokenizer, filter);
76          } else if (fieldName.contains("payloadsVariable")) {
77            TokenFilter filter = new MockVariableLengthPayloadFilter(new Random(0), tokenizer);
78            return new TokenStreamComponents(tokenizer, filter);

80            return new TokenStreamComponents(tokenizer);

89      Document doc = new Document();
90      FieldType docsOnlyType = new FieldType(TextField.TYPE_NOT_STORED);
91      // turn this on for a cross-check

99

101     // turn these on for a cross-check

103     positionsType.setStoreTermVectorPositions(true);
104     positionsType.setStoreTermVectorOffsets(true);
105     positionsType.setStoreTermVectorPayloads(true);
106     FieldType offsetsType = new FieldType(positionsType);
107     offsetsType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
108     Field field1 = new Field("field1docs", "", docsOnlyType);
109     Field field2 = new Field("field2freqs", "", docsAndFreqsType);

119     doc.add(field4);

165   public void assertTerms(Terms leftTerms, Terms rightTerms, boolean deep) throws Exception {
166     if (leftTerms == null || rightTerms == null) {
167       assertNull(leftTerms);
168       assertNull(rightTerms);
169       return;
170     }
171     assertTermsStatistics(leftTerms, rightTerms);
172     
173     // NOTE: we don't assert hasOffsets/hasPositions/hasPayloads because they are allowed to be different
174 
175     boolean bothHavePositions = leftTerms.hasPositions() && rightTerms.hasPositions();
176     TermsEnum leftTermsEnum = leftTerms.iterator();
177     TermsEnum rightTermsEnum = rightTerms.iterator();
178     assertTermsEnum(leftTermsEnum, rightTermsEnum, true, bothHavePositions);
179     
180     assertTermsSeeking(leftTerms, rightTerms);
181     
182     if (deep) {
183       int numIntersections = atLeast(3);
184       for (int i = 0; i < numIntersections; i++) {
185         String re = AutomatonTestUtil.randomRegexp(random());
186         CompiledAutomaton automaton = new CompiledAutomaton(new RegExp(re, RegExp.NONE).toAutomaton());
187         if (automaton.type == CompiledAutomaton.AUTOMATON_TYPE.NORMAL) {
188           // TODO: test start term too
189           TermsEnum leftIntersection = leftTerms.intersect(automaton, null);
190           TermsEnum rightIntersection = rightTerms.intersect(automaton, null);
191           assertTermsEnum(leftIntersection, rightIntersection, rarely(), bothHavePositions);
192         }
193       }
194     }
195   }

197   private void assertTermsSeeking(Terms leftTerms, Terms rightTerms) throws Exception {
198     TermsEnum leftEnum = null;
199     TermsEnum rightEnum = null;
200     
201     // just an upper bound
202     int numTests = atLeast(20);
203     Random random = random();
204     
205     // collect this number of terms from the left side
206     HashSet<BytesRef> tests = new HashSet<>();
207     int numPasses = 0;
208     while (numPasses < 10 && tests.size() < numTests) {
209       leftEnum = leftTerms.iterator();
210       BytesRef term = null;
211       while ((term = leftEnum.next()) != null) {
212         int code = random.nextInt(10);
213         if (code == 0) {
214           // the term
215           tests.add(BytesRef.deepCopyOf(term));
216         } else if (code == 1) {
217           // truncated subsequence of term
218           term = BytesRef.deepCopyOf(term);
219           if (term.length > 0) {
220             // truncate it
221             term.length = random.nextInt(term.length);
222           }
223         } else if (code == 2) {
224           // term, but ensure a non-zero offset
225           byte newbytes[] = new byte[term.length+5];
226           System.arraycopy(term.bytes, term.offset, newbytes, 5, term.length);
227           tests.add(new BytesRef(newbytes, 5, term.length));
228         }
229       }
230       numPasses++;
231     }
232     
233     ArrayList<BytesRef> shuffledTests = new ArrayList<>(tests);
234     Collections.shuffle(shuffledTests, random);
235     
236     for (BytesRef b : shuffledTests) {
237       leftEnum = leftTerms.iterator();
238       rightEnum = rightTerms.iterator();
239       
240       assertEquals(leftEnum.seekExact(b), rightEnum.seekExact(b));
241       assertEquals(leftEnum.seekExact(b), rightEnum.seekExact(b));
242       
243       SeekStatus leftStatus;
244       SeekStatus rightStatus;
245       
246       leftStatus = leftEnum.seekCeil(b);
247       rightStatus = rightEnum.seekCeil(b);
248       assertEquals(leftStatus, rightStatus);
249       if (leftStatus != SeekStatus.END) {
250         assertEquals(leftEnum.term(), rightEnum.term());
251       }
252       
253       leftStatus = leftEnum.seekCeil(b);
254       rightStatus = rightEnum.seekCeil(b);
255       assertEquals(leftStatus, rightStatus);
256       if (leftStatus != SeekStatus.END) {
257         assertEquals(leftEnum.term(), rightEnum.term());
258       }
259     }
260   }

276       assertEquals(leftTerms.size(), rightTerms.size());
277     }

280   /** 
281    * checks the terms enum sequentially
282    * if deep is false, it does a 'shallow' test that doesnt go down to the docsenums
283    */
284   public void assertTermsEnum(TermsEnum leftTermsEnum, TermsEnum rightTermsEnum, boolean deep, boolean hasPositions) throws Exception {
285     BytesRef term;
286     PostingsEnum leftPositions = null;
287     PostingsEnum rightPositions = null;
288     PostingsEnum leftDocs = null;
289     PostingsEnum rightDocs = null;
290     
291     while ((term = leftTermsEnum.next()) != null) {
292       assertEquals(term, rightTermsEnum.next());
293       assertTermStats(leftTermsEnum, rightTermsEnum);
294       if (deep) {
295         if (hasPositions) {
296           // with payloads + off
297           assertDocsAndPositionsEnum(leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.ALL),
298                                      rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.ALL));
299 
300           assertPositionsSkipping(leftTermsEnum.docFreq(),
301                                   leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.ALL),
302                                   rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.ALL));
303           // with payloads only
304           assertDocsAndPositionsEnum(leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.PAYLOADS),
305                                      rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.PAYLOADS));
306 
307           assertPositionsSkipping(leftTermsEnum.docFreq(),
308                                   leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.PAYLOADS),
309                                   rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.PAYLOADS));
310 
311           // with offsets only
312           assertDocsAndPositionsEnum(leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.OFFSETS),
313                                      rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.OFFSETS));
314 
315           assertPositionsSkipping(leftTermsEnum.docFreq(),
316                                   leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.OFFSETS),
317                                   rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.OFFSETS));
318 
319           // with positions only
320           assertDocsAndPositionsEnum(leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.POSITIONS),
321                                      rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.POSITIONS));
322 
323           assertPositionsSkipping(leftTermsEnum.docFreq(),
324                                   leftPositions = leftTermsEnum.postings(leftPositions, PostingsEnum.POSITIONS),
325                                   rightPositions = rightTermsEnum.postings(rightPositions, PostingsEnum.POSITIONS));
326         }
327         
328         // with freqs:
329         assertDocsEnum(leftDocs = leftTermsEnum.postings(leftDocs),
330             rightDocs = rightTermsEnum.postings(rightDocs));
331 
332         // w/o freqs:
333         assertDocsEnum(leftDocs = leftTermsEnum.postings(leftDocs, PostingsEnum.NONE),
334             rightDocs = rightTermsEnum.postings(rightDocs, PostingsEnum.NONE));
335 
336         // with freqs:
337         assertDocsSkipping(leftTermsEnum.docFreq(), 
338             leftDocs = leftTermsEnum.postings(leftDocs),
339             rightDocs = rightTermsEnum.postings(rightDocs));
340 
341         // w/o freqs:
342         assertDocsSkipping(leftTermsEnum.docFreq(), 
343             leftDocs = leftTermsEnum.postings(leftDocs, PostingsEnum.NONE),
344             rightDocs = rightTermsEnum.postings(rightDocs, PostingsEnum.NONE));
345       }
346     }
347     assertNull(rightTermsEnum.next());
348   }

360   /**
361    * checks docs + freqs + positions + payloads, sequentially
362    */
363   public void assertDocsAndPositionsEnum(PostingsEnum leftDocs, PostingsEnum rightDocs) throws Exception {
364     assertNotNull(leftDocs);
365     assertNotNull(rightDocs);
366     assertEquals(-1, leftDocs.docID());
367     assertEquals(-1, rightDocs.docID());
368     int docid;
369     while ((docid = leftDocs.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
370       assertEquals(docid, rightDocs.nextDoc());
371       int freq = leftDocs.freq();
372       assertEquals(freq, rightDocs.freq());
373       for (int i = 0; i < freq; i++) {
374         assertEquals(leftDocs.nextPosition(), rightDocs.nextPosition());
375         // we don't assert offsets/payloads, they are allowed to be different
376       }
377     }
378     assertEquals(DocIdSetIterator.NO_MORE_DOCS, rightDocs.nextDoc());
379   }

399   /**
400    * checks advancing docs
401    */
402   public void assertDocsSkipping(int docFreq, PostingsEnum leftDocs, PostingsEnum rightDocs) throws Exception {
403     if (leftDocs == null) {
404       assertNull(rightDocs);
405       return;
406     }
407     int docid = -1;
408     int averageGap = MAXDOC / (1+docFreq);
409     int skipInterval = 16;
410 
411     while (true) {
412       if (random().nextBoolean()) {
413         // nextDoc()
414         docid = leftDocs.nextDoc();
415         assertEquals(docid, rightDocs.nextDoc());
416       } else {
417         // advance()
418         int skip = docid + (int) Math.ceil(Math.abs(skipInterval + random().nextGaussian() * averageGap));
419         docid = leftDocs.advance(skip);
420         assertEquals(docid, rightDocs.advance(skip));
421       }
422       
423       if (docid == DocIdSetIterator.NO_MORE_DOCS) {
424         return;
425       }
426       // we don't assert freqs, they are allowed to be different
427     }
428   }

430   /**
431    * checks advancing docs + positions
432    */
433   public void assertPositionsSkipping(int docFreq, PostingsEnum leftDocs, PostingsEnum rightDocs) throws Exception {
434     if (leftDocs == null || rightDocs == null) {
435       assertNull(leftDocs);
436       assertNull(rightDocs);
437       return;
438     }
439     
440     int docid = -1;
441     int averageGap = MAXDOC / (1+docFreq);
442     int skipInterval = 16;
443 
444     while (true) {
445       if (random().nextBoolean()) {
446         // nextDoc()
447         docid = leftDocs.nextDoc();
448         assertEquals(docid, rightDocs.nextDoc());
449       } else {
450         // advance()
451         int skip = docid + (int) Math.ceil(Math.abs(skipInterval + random().nextGaussian() * averageGap));
452         docid = leftDocs.advance(skip);
453         assertEquals(docid, rightDocs.advance(skip));
454       }
455       
456       if (docid == DocIdSetIterator.NO_MORE_DOCS) {
457         return;
458       }
459       int freq = leftDocs.freq();
460       assertEquals(freq, rightDocs.freq());
461       for (int i = 0; i < freq; i++) {
462         assertEquals(leftDocs.nextPosition(), rightDocs.nextPosition());
463         // we don't compare the payloads, it's allowed that one is empty etc
464       }
465     }

468   private static class RandomBits implements Bits {
469     FixedBitSet bits;
470     
471     RandomBits(int maxDoc, double pctLive, Random random) {
472       bits = new FixedBitSet(maxDoc);
473       for (int i = 0; i < maxDoc; i++) {
474         if (random.nextDouble() <= pctLive) {        
475           bits.set(i);
476         }
477       }
478     }
479     
480     @Override
481     public boolean get(int index) {
482       return bits.get(index);
483     }
484 
485     @Override
486     public int length() {
487       return bits.length();
488     }
489   }
