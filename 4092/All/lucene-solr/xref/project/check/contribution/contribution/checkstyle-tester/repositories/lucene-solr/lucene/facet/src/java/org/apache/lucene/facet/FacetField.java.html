20  import java.util.Arrays;
21  
22  import org.apache.lucene.document.Document;
23  import org.apache.lucene.document.Field;
24  import org.apache.lucene.document.FieldType;
25  import org.apache.lucene.index.IndexOptions;
26  
27  /**
28   * Add an instance of this to your {@link Document} for every facet label.
29   * 
30   * <p>
31   * <b>NOTE:</b> you must call {@link FacetsConfig#build(Document)} before
32   * you add the document to IndexWriter.
33   */
34  public class FacetField extends Field {
35    static final FieldType TYPE = new FieldType();
36    static {
37      TYPE.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS);
38      TYPE.freeze();
39    }
40  
41    /** Dimension for this field. */
42    public final String dim;
43  
44    /** Path for this field. */
45    public final String[] path;
46  
47    /** Creates the this from {@code dim} and
48     *  {@code path}. */
49    public FacetField(String dim, String... path) {
50      super("dummy", TYPE);
51      verifyLabel(dim);
52      for(String label : path) {
53        verifyLabel(label);
54      }
55      this.dim = dim;
56      if (path.length == 0) {
57        throw new IllegalArgumentException("path must have at least one element");
58      }
59      this.path = path;
60    }
61  
62    @Override
63    public String toString() {
64      return "FacetField(dim=" + dim + " path=" + Arrays.toString(path) + ")";
65    }
66  
67    /** Verifies the label is not null or empty string.
68     * 
69     *  @lucene.internal */
70    public static void verifyLabel(String label) {
71      if (label == null || label.isEmpty()) {
72        throw new IllegalArgumentException("empty or null components not allowed; got: " + label);
73      }
74    }
75  }
