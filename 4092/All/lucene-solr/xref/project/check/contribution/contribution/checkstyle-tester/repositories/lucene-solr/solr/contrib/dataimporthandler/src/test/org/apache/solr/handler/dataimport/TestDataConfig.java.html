39  public class TestDataConfig extends AbstractDataImportHandlerTestCase {
40    @BeforeClass
41    public static void beforeClass() throws Exception {
42      initCore("dataimport-nodatasource-solrconfig.xml", "dataimport-schema.xml");
43    }
44  
45    @Test
46    @SuppressWarnings("unchecked")
47    public void testDataConfigWithDataSource() throws Exception {
48      List rows = new ArrayList();
49      rows.add(createMap("id", "1", "desc", "one"));
50      MockDataSource.setIterator("select * from x", rows.iterator());
51  
52      runFullImport(loadDataConfig("data-config-with-datasource.xml"));
53  
54      assertQ(req("id:1"), "//*[@numFound='1']");
55    }
56  
57    @Test
58    public void testBasic() throws Exception {
59      javax.xml.parsers.DocumentBuilder builder = DocumentBuilderFactory
60              .newInstance().newDocumentBuilder();
61      Document doc = builder.parse(new InputSource(new StringReader(xml)));
62      DataImporter di = new DataImporter();
63      DIHConfiguration dc = di.readFromXml(doc);
64      assertEquals("atrimlisting", dc.getEntities().get(0).getName());
65    }
66  
67    private static final String xml = "<dataConfig>\n"
68            + "\t<document name=\"autos\" >\n"
69            + "\t\t<entity name=\"atrimlisting\" pk=\"acode\"\n"
70            + "\t\t\tquery=\"select acode,make,model,year,msrp,category,image,izmo_image_url,price_range_low,price_range_high,invoice_range_low,invoice_range_high from atrimlisting\"\n"
71            + "\t\t\tdeltaQuery=\"select acode from atrimlisting where last_modified > '${indexer.last_index_time}'\">\n"
72            +
73  
74            "\t\t</entity>\n" +
75  
76            "\t</document>\n" + "</dataConfig>";
