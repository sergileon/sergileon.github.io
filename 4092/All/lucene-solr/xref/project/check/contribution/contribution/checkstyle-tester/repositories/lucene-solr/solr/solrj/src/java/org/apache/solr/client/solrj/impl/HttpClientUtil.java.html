112   /**
113    * Creates new http client by using the provided configuration.
114    * 
115    * @param params
116    *          http client configuration, if null a client with default
117    *          configuration (no additional configuration) is created. 
118    */
119   public static CloseableHttpClient createClient(final SolrParams params) {
120     final ModifiableSolrParams config = new ModifiableSolrParams(params);
121     if (logger.isDebugEnabled()) {
122       logger.debug("Creating new http client, config:" + config);
123     }
124     final DefaultHttpClient httpClient = HttpClientFactory.createHttpClient();
125     configureClient(httpClient, config);
126     return httpClient;
127   }
128   
129   /**
130    * Creates new http client by using the provided configuration.
131    * 
132    */
133   public static CloseableHttpClient createClient(final SolrParams params, ClientConnectionManager cm) {
134     final ModifiableSolrParams config = new ModifiableSolrParams(params);
135     if (logger.isDebugEnabled()) {
136       logger.debug("Creating new http client, config:" + config);
137     }
138     final DefaultHttpClient httpClient = HttpClientFactory.createHttpClient(cm);
139     configureClient(httpClient, config);
140     return httpClient;
141   }

143   /**
144    * Configures {@link DefaultHttpClient}, only sets parameters if they are
145    * present in config.
146    */
147   public static void configureClient(final DefaultHttpClient httpClient,
148       SolrParams config) {
149     configurer.configure(httpClient,  config);
150     synchronized(interceptors) {
151       for(HttpRequestInterceptor interceptor: interceptors) {
152         httpClient.addRequestInterceptor(interceptor);
153       }
154     }
155   }

157   public static void close(HttpClient httpClient) { 
158     if (httpClient instanceof CloseableHttpClient) {
159       org.apache.solr.common.util.IOUtils.closeQuietly((CloseableHttpClient) httpClient);
160     } else {
161       httpClient.getConnectionManager().shutdown();
162     }
163   }

173   /**
174    * Control HTTP payload compression.
175    * 
176    * @param allowCompression
177    *          true will enable compression (needs support from server), false
178    *          will disable compression.
179    */
180   public static void setAllowCompression(DefaultHttpClient httpClient,
181       boolean allowCompression) {
182     httpClient
183         .removeRequestInterceptorByClass(UseCompressionRequestInterceptor.class);
184     httpClient
185         .removeResponseInterceptorByClass(UseCompressionResponseInterceptor.class);
186     if (allowCompression) {
187       httpClient.addRequestInterceptor(new UseCompressionRequestInterceptor());
188       httpClient
189           .addResponseInterceptor(new UseCompressionResponseInterceptor());
190     }
191   }

193   /**
194    * Set http basic auth information. If basicAuthUser or basicAuthPass is null
195    * the basic auth configuration is cleared. Currently this is not preemtive
196    * authentication. So it is not currently possible to do a post request while
197    * using this setting.
198    */
199   public static void setBasicAuth(DefaultHttpClient httpClient,
200       String basicAuthUser, String basicAuthPass) {
201     if (basicAuthUser != null && basicAuthPass != null) {
202       httpClient.getCredentialsProvider().setCredentials(AuthScope.ANY,
203           new UsernamePasswordCredentials(basicAuthUser, basicAuthPass));
204     } else {
205       httpClient.getCredentialsProvider().clear();
206     }
207   }

209   /**
210    * Set max connections allowed per host. This call will only work when
211    * {@link ThreadSafeClientConnManager} or
212    * {@link PoolingClientConnectionManager} is used.
213    */
214   public static void setMaxConnectionsPerHost(HttpClient httpClient,
215       int max) {
216     // would have been nice if there was a common interface
217     if (httpClient.getConnectionManager() instanceof ThreadSafeClientConnManager) {
218       ThreadSafeClientConnManager mgr = (ThreadSafeClientConnManager)httpClient.getConnectionManager();
219       mgr.setDefaultMaxPerRoute(max);
220     } else if (httpClient.getConnectionManager() instanceof PoolingClientConnectionManager) {
221       PoolingClientConnectionManager mgr = (PoolingClientConnectionManager)httpClient.getConnectionManager();
222       mgr.setDefaultMaxPerRoute(max);
223     }
224   }

255   /**
256    * Control retry handler 
257    * @param useRetry when false the client will not try to retry failed requests.
258    */
259   public static void setUseRetry(final DefaultHttpClient httpClient,
260       boolean useRetry) {
261     if (!useRetry) {
262       httpClient.setHttpRequestRetryHandler(NO_RETRY);
263     } else {
264       // if the request is not fully sent, we retry
265       // streaming updates are not a problem, because they are not retryable
266       httpClient.setHttpRequestRetryHandler(new DefaultHttpRequestRetryHandler(){
267         @Override
268         protected boolean handleAsIdempotent(final HttpRequest request) {
269           return false; // we can't tell if a Solr request is idempotent
270         }
271       });
272     }
273   }

315   private static class UseCompressionRequestInterceptor implements
316       HttpRequestInterceptor {
317     
318     @Override
319     public void process(HttpRequest request, HttpContext context)
320         throws HttpException, IOException {
321       if (!request.containsHeader("Accept-Encoding")) {
322         request.addHeader("Accept-Encoding", "gzip, deflate");
323       }
324     }
325   }
326   
327   private static class UseCompressionResponseInterceptor implements
328       HttpResponseInterceptor {
329     
330     @Override
331     public void process(final HttpResponse response, final HttpContext context)
332         throws HttpException, IOException {
333       
334       HttpEntity entity = response.getEntity();
335       Header ceheader = entity.getContentEncoding();
336       if (ceheader != null) {
337         HeaderElement[] codecs = ceheader.getElements();
338         for (int i = 0; i < codecs.length; i++) {
339           if (codecs[i].getName().equalsIgnoreCase("gzip")) {
340             response
341                 .setEntity(new GzipDecompressingEntity(response.getEntity()));
342             return;
343           }
344           if (codecs[i].getName().equalsIgnoreCase("deflate")) {
345             response.setEntity(new DeflateDecompressingEntity(response
346                 .getEntity()));
347             return;
348           }
349         }
350       }
351     }
352   }
353   
354   private static class GzipDecompressingEntity extends HttpEntityWrapper {
355     public GzipDecompressingEntity(final HttpEntity entity) {
356       super(entity);
357     }
358     
359     @Override
360     public InputStream getContent() throws IOException, IllegalStateException {
361       return new GZIPInputStream(wrappedEntity.getContent());
362     }
363     
364     @Override
365     public long getContentLength() {
366       return -1;
367     }
368   }
369   
370   private static class DeflateDecompressingEntity extends
371       GzipDecompressingEntity {
372     public DeflateDecompressingEntity(final HttpEntity entity) {
373       super(entity);
374     }
375     
376     @Override
377     public InputStream getContent() throws IOException, IllegalStateException {
378       return new InflaterInputStream(wrappedEntity.getContent());
379     }
380   }

382   public static class HttpClientFactory {
383     private static Class<? extends DefaultHttpClient> defaultHttpClientClass = DefaultHttpClient.class;
384     private static Class<? extends SystemDefaultHttpClient> systemDefaultHttpClientClass = SystemDefaultHttpClient.class;
385 
386 
387     public static SystemDefaultHttpClient createHttpClient() {
388       Constructor<? extends SystemDefaultHttpClient> constructor;
389       try {
390         constructor = systemDefaultHttpClientClass.getDeclaredConstructor();
391         return constructor.newInstance();
392       } catch (NoSuchMethodException | SecurityException | InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
393         throw new SolrException(ErrorCode.SERVER_ERROR, "Unable to create HttpClient instance. ", e);
394       }
395     }
396 
397     public static DefaultHttpClient createHttpClient(ClientConnectionManager cm) {
398       Constructor<? extends DefaultHttpClient> constructor;
399       try {
400         constructor = defaultHttpClientClass.getDeclaredConstructor(new Class[]{ClientConnectionManager.class});
401         return constructor.newInstance(new Object[]{cm});
402       } catch (NoSuchMethodException | SecurityException | InstantiationException | IllegalAccessException | IllegalArgumentException | InvocationTargetException e) {
403         throw new SolrException(ErrorCode.SERVER_ERROR, "Unable to create HttpClient instance, registered class is: " + defaultHttpClientClass, e);
404       }
405     }
406 
407     public static void setHttpClientImpl(Class<? extends DefaultHttpClient> defaultHttpClient, Class<? extends SystemDefaultHttpClient> systemDefaultHttpClient) {
408       defaultHttpClientClass = defaultHttpClient;
409       systemDefaultHttpClientClass = systemDefaultHttpClient;
410     }
411   }
