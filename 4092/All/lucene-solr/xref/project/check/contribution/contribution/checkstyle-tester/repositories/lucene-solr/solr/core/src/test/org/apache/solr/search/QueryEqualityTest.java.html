34  /**
35   * Sanity checks that queries (generated by the QParser and ValueSourceParser 
36   * framework) are appropriately {@link Object#equals} and 
37   * {@link Object#hashCode()} equivalent.  If you are adding a new default 
38   * QParser or ValueSourceParser, you will most likely get a failure from
39   * {@link #testParserCoverage} until you add a new test method to this class.
40   *
41   * @see ValueSourceParser#standardValueSourceParsers
42   * @see QParserPlugin#standardPlugins
43   * @see QueryUtils
44   **/
45  public class QueryEqualityTest extends SolrTestCaseJ4 {
46

48    public static void beforeClass() throws Exception {
49      initCore("solrconfig.xml","schema15.xml");
50    }

52    /** @see #testParserCoverage */
53    @AfterClass
54    public static void afterClassParserCoverageTest() {
55  
56      if ( ! doAssertParserCoverage) return;
57      for (String name : QParserPlugin.standardPlugins.keySet()) {
58        assertTrue("testParserCoverage was run w/o any other method explicitly testing qparser: " + name, qParsersTested.contains(name));
59      }
60  
61      for (final String name : ValueSourceParser.standardValueSourceParsers.keySet()) {
62        assertTrue("testParserCoverage was run w/o any other method explicitly testing val parser: " + name, valParsersTested.contains(name));
63      }
64  
65    }

238     SolrQueryRequest req = req("myField","foo_s",
239                                "g_sort","foo_s1 asc, foo_i desc");
240 
241     try {
242       assertQueryEquals("collapse", req,
243           "{!collapse field=$myField}");
244 
245       assertQueryEquals("collapse", req,
246           "{!collapse field=$myField max=a}");
247 
248       assertQueryEquals("collapse", req,
249                         "{!collapse field=$myField min=a}",
250                         "{!collapse field=$myField min=a nullPolicy=ignore}");
251       
252       assertQueryEquals("collapse", req,
253                         "{!collapse field=$myField sort=$g_sort}",
254                         "{!collapse field=$myField sort='foo_s1 asc, foo_i desc'}",
255                         "{!collapse field=$myField sort=$g_sort nullPolicy=ignore}");
256 
257       assertQueryEquals("collapse", req,
258           "{!collapse field=$myField max=a nullPolicy=expand}");
259 
260       //Add boosted documents to the request context.
261       Map context = req.getContext();
262       Set boosted = new HashSet();
263       boosted.add("doc1");
264       boosted.add("doc2");
265       context.put("BOOSTED", boosted);
266 
267       assertQueryEquals("collapse", req,
268           "{!collapse field=$myField min=a}",
269           "{!collapse field=$myField min=a nullPolicy=ignore}");
270 
271 
272     } finally {
273       req.close();
274     }
275   }

349   private void checkQuerySpatial(final String type) throws Exception {
350     SolrQueryRequest req = req("myVar", "5",
351                                "d","109",
352                                "pt","10.312,-20.556",
353                                "sfield","store");
354     try {
355       assertQueryEquals(type, req, 
356                         "{!"+type+" d=109}",
357                         "{!"+type+" sfield=$sfield}",
358                         "{!"+type+" sfield=store d=109}",
359                         "{!"+type+" sfield=store d=$d pt=$pt}",
360                         "{!"+type+" sfield=store d=$d pt=10.312,-20.556}",
361                         "{!"+type+"}");
362       // diff SpatialQueryable FieldTypes matter for determining final query
363       assertQueryEquals(type, req, 
364                         "{!"+type+" sfield=point_hash}",
365                         "{!"+type+" sfield=point_hash d=109}",
366                         "{!"+type+" sfield=point_hash d=$d pt=$pt}",
367                         "{!"+type+" sfield=point_hash d=$d pt=10.312,-20.556}");
368       assertQueryEquals(type, req, 
369                         "{!"+type+" sfield=point}",
370                         "{!"+type+" sfield=point d=109}",
371                         "{!"+type+" sfield=point d=$d pt=$pt}",
372                         "{!"+type+" sfield=point d=$d pt=10.312,-20.556}");
373     } finally {
374       req.close();
375     }
376   }

925   /**
926    * NOTE: defType is not only used to pick the parser, but, if non-null it is 
927    * also to record the parser being tested for coverage sanity checking
928    *
929    * @see QueryUtils#check
930    * @see QueryUtils#checkEqual
931    * @see #testParserCoverage
932    */
933   protected void assertQueryEquals(final String defType,
934                                    final SolrQueryRequest req,
935                                    final String... inputs) throws Exception {
936 
937     if (null != defType) qParsersTested.add(defType);
938 
939     final Query[] queries = new Query[inputs.length];
940 
941     try {
942       SolrQueryResponse rsp = new SolrQueryResponse();
943       SolrRequestInfo.setRequestInfo(new SolrRequestInfo(req,rsp));
944       for (int i = 0; i < inputs.length; i++) {
945         queries[i] = (QParser.getParser(inputs[i], defType, req).getQuery());
946       }
947       for (int i = 0; i < queries.length; i++) {
948         QueryUtils.check(queries[i]);
949         // yes starting j=0 is redundent, we're making sure every query 
950         // is equal to itself, and that the quality checks work regardless 
951         // of which caller/callee is used.
952         for (int j = 0; j < queries.length; j++) {
953           QueryUtils.checkEqual(queries[i], queries[j]);
954         }
955       }
956     } finally {
957       SolrRequestInfo.clearRequestInfo();
958     }
959   }
