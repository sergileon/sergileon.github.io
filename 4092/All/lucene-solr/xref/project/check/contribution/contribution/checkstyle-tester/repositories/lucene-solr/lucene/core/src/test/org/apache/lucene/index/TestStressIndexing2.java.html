234   public static void indexSerial(Random random, Map<String,Document> docs, Directory dir) throws IOException {
235     IndexWriter w = new IndexWriter(dir, LuceneTestCase.newIndexWriterConfig(random, new MockAnalyzer(random)).setMergePolicy(newLogMergePolicy()));
236 
237     // index all docs in a single thread
238     Iterator<Document> iter = docs.values().iterator();
239     while (iter.hasNext()) {
240       Document d = iter.next();
241       ArrayList<IndexableField> fields = new ArrayList<>();
242       fields.addAll(d.getFields());
243       // put fields in same order each time
244       Collections.sort(fields, fieldNameComparator);
245       
246       Document d1 = new Document();
247       for (int i=0; i<fields.size(); i++) {
248         d1.add(fields.get(i));
249       }
250       w.addDocument(d1);
251       // System.out.println("indexing "+d1);
252     }
253     
254     w.close();
255   }

288   private static int nextNonDeletedDoc(PostingsEnum it, Bits liveDocs) throws IOException {
289     int doc = it.nextDoc();
290     while (doc != DocIdSetIterator.NO_MORE_DOCS && liveDocs != null && liveDocs.get(doc) == false) {
291       doc = it.nextDoc();
292     }
293     return doc;
294   }

706     private int addUTF8Token(int start) {
707       final int end = start + nextInt(20);
708       if (buffer.length < 1+end) {
709         char[] newBuffer = new char[(int) ((1+end)*1.25)];
710         System.arraycopy(buffer, 0, newBuffer, 0, buffer.length);
711         buffer = newBuffer;
712       }
713 
714       for(int i=start;i<end;i++) {
715         int t = nextInt(5);
716         if (0 == t && i < end-1) {
717           // Make a surrogate pair
718           // High surrogate
719           buffer[i++] = (char) nextInt(0xd800, 0xdc00);
720           // Low surrogate
721           buffer[i] = (char) nextInt(0xdc00, 0xe000);
722         } else if (t <= 1)
723           buffer[i] = (char) nextInt(0x80);
724         else if (2 == t)
725           buffer[i] = (char) nextInt(0x80, 0x800);
726         else if (3 == t)
727           buffer[i] = (char) nextInt(0x800, 0xd800);
728         else if (4 == t)
729           buffer[i] = (char) nextInt(0xe000, 0xffff);
730       }
731       buffer[end] = ' ';
732       return 1+end;
733     }

735     public String getString(int nTokens) {
736       nTokens = nTokens!=0 ? nTokens : r.nextInt(4)+1;
737 
738       // Half the time make a random UTF8 string
739       if (r.nextBoolean())
740         return getUTF8String(nTokens);
741 
742       // avoid StringBuffer because it adds extra synchronization.
743       char[] arr = new char[nTokens*2];
744       for (int i=0; i<nTokens; i++) {
745         arr[i*2] = (char)('A' + r.nextInt(10));
746         arr[i*2+1] = ' ';
747       }
748       return new String(arr);
749     }

751     public String getUTF8String(int nTokens) {
752       int upto = 0;
753       Arrays.fill(buffer, (char) 0);
754       for(int i=0;i<nTokens;i++)
755         upto = addUTF8Token(upto);
756       return new String(buffer, 0, upto);
757     }

763     public void indexDoc() throws IOException {
764       Document d = new Document();
765 
766       FieldType customType1 = new FieldType(TextField.TYPE_STORED);
767       customType1.setTokenized(false);
768       customType1.setOmitNorms(true);
769       
770       ArrayList<Field> fields = new ArrayList<>();
771       String idString = getIdString();
772       Field idField =  newField("id", idString, customType1);
773       fields.add(idField);
774 
775       Map<String,FieldType> tvTypes = new HashMap<>();
776 
777       int nFields = nextInt(maxFields);
778       for (int i=0; i<nFields; i++) {
779 
780         String fieldName = "f" + nextInt(100);
781         FieldType customType;
782 
783         // Use the same term vector settings if we already
784         // added this field to the doc:
785         FieldType oldTVType = tvTypes.get(fieldName);
786         if (oldTVType != null) {
787           customType = new FieldType(oldTVType);
788         } else {
789           customType = new FieldType();
790           switch (nextInt(4)) {
791           case 0:
792             break;
793           case 1:
794             customType.setStoreTermVectors(true);
795             break;
796           case 2:
797             customType.setStoreTermVectors(true);
798             customType.setStoreTermVectorPositions(true);
799             break;
800           case 3:
801             customType.setStoreTermVectors(true);
802             customType.setStoreTermVectorOffsets(true);
803             break;
804           }
805           FieldType newType = new FieldType(customType);
806           newType.freeze();
807           tvTypes.put(fieldName, newType);
808         }
809         
810         switch (nextInt(4)) {
811           case 0:
812             customType.setStored(true);
813             customType.setOmitNorms(true);
814             customType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS);
815             customType.freeze();
816             fields.add(newField(fieldName, getString(1), customType));
817             break;
818           case 1:
819             customType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS);
820             customType.setTokenized(true);
821             customType.freeze();
822             fields.add(newField(fieldName, getString(0), customType));
823             break;
824           case 2:
825             customType.setStored(true);
826             customType.setStoreTermVectors(false);
827             customType.setStoreTermVectorOffsets(false);
828             customType.setStoreTermVectorPositions(false);
829             customType.freeze();
830             fields.add(newField(fieldName, getString(0), customType));
831             break;
832           case 3:
833             customType.setStored(true);
834             customType.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS);
835             customType.setTokenized(true);
836             customType.freeze();
837             fields.add(newField(fieldName, getString(bigFieldSize), customType));
838             break;
839         }
840       }
841 
842       if (sameFieldOrder) {
843         Collections.sort(fields, fieldNameComparator);
844       } else {
845         // random placement of id field also
846         Collections.swap(fields,nextInt(fields.size()), 0);
847       }
848 
849       for (int i=0; i<fields.size(); i++) {
850         d.add(fields.get(i));
851       }
852       if (VERBOSE) {
853         System.out.println(Thread.currentThread().getName() + ": indexing id:" + idString);
854       }
855       w.updateDocument(new Term("id", idString), d);
856       //System.out.println(Thread.currentThread().getName() + ": indexing "+d);
857       docs.put(idString, d);
858     }
