49  /**
50   * Fuzzifies ALL terms provided as strings and then picks the best n differentiating terms.
51   * In effect this mixes the behaviour of FuzzyQuery and MoreLikeThis but with special consideration
52   * of fuzzy scoring factors.
53   * This generally produces good results for queries where users may provide details in a number of 
54   * fields and have no knowledge of boolean query syntax and also want a degree of fuzzy matching and
55   * a fast query.
56   * 
57   * For each source term the fuzzy variants are held in a BooleanQuery with no coord factor (because
58   * we are not looking for matches on multiple variants in any one doc). Additionally, a specialized
59   * TermQuery is used for variants and does not use that variant term's IDF because this would favour rarer 
60   * terms eg misspellings. Instead, all variants use the same IDF ranking (the one for the source query 
61   * term) and this is factored into the variant's boost. If the source query term does not exist in the
62   * index the average IDF of the variants is used.
63   */
64  public class FuzzyLikeThisQuery extends Query
65  {
66      // TODO: generalize this query (at least it should not reuse this static sim!
67      // a better way might be to convert this into multitermquery rewrite methods.
68      // the rewrite method can 'average' the TermContext's term statistics (docfreq,totalTermFreq) 
69      // provided to TermQuery, so that the general idea is agnostic to any scoring system...
70      static TFIDFSimilarity sim=new DefaultSimilarity();
71      ArrayList<FieldVals> fieldVals=new ArrayList<>();
72      Analyzer analyzer;

75      boolean ignoreTF=false;
76      private int maxNumTerms;

78      @Override
79      public int hashCode() {
80        final int prime = 31;
81        int result = super.hashCode();
82        result = prime * result + ((analyzer == null) ? 0 : analyzer.hashCode());
83        result = prime * result
84            + ((fieldVals == null) ? 0 : fieldVals.hashCode());
85        result = prime * result + (ignoreTF ? 1231 : 1237);
86        result = prime * result + maxNumTerms;
87        return result;
88      }

90      @Override
91      public boolean equals(Object obj) {
92        if (this == obj)
93          return true;
94        if (obj == null)
95          return false;
96        if (getClass() != obj.getClass())
97          return false;
98        if (!super.equals(obj)) {
99          return false;
100       }
101       FuzzyLikeThisQuery other = (FuzzyLikeThisQuery) obj;
102       if (analyzer == null) {
103         if (other.analyzer != null)
104           return false;
105       } else if (!analyzer.equals(other.analyzer))
106         return false;
107       if (fieldVals == null) {
108         if (other.fieldVals != null)
109           return false;
110       } else if (!fieldVals.equals(other.fieldVals))
111         return false;
112       if (ignoreTF != other.ignoreTF)
113         return false;
114       if (maxNumTerms != other.maxNumTerms)
115         return false;
116       return true;
117     }

200   private void addTerms(IndexReader reader, FieldVals f, ScoreTermQueue q) throws IOException {
201     if (f.queryString == null) return;
202     final Terms terms = MultiFields.getTerms(reader, f.fieldName);
203     if (terms == null) {
204       return;
205     }
206     try (TokenStream ts = analyzer.tokenStream(f.fieldName, f.queryString)) {
207       CharTermAttribute termAtt = ts.addAttribute(CharTermAttribute.class);
208 
209       int corpusNumDocs = reader.numDocs();
210       HashSet<String> processedTerms = new HashSet<>();
211       ts.reset();
212       while (ts.incrementToken()) {
213         String term = termAtt.toString();
214         if (!processedTerms.contains(term)) {
215           processedTerms.add(term);
216           ScoreTermQueue variantsQ = new ScoreTermQueue(MAX_VARIANTS_PER_TERM); //maxNum variants considered for any one term
217           float minScore = 0;
218           Term startTerm = new Term(f.fieldName, term);
219           AttributeSource atts = new AttributeSource();
220           MaxNonCompetitiveBoostAttribute maxBoostAtt =
221             atts.addAttribute(MaxNonCompetitiveBoostAttribute.class);
222           SlowFuzzyTermsEnum fe = new SlowFuzzyTermsEnum(terms, atts, startTerm, f.minSimilarity, f.prefixLength);
223           //store the df so all variants use same idf
224           int df = reader.docFreq(startTerm);
225           int numVariants = 0;
226           int totalVariantDocFreqs = 0;
227           BytesRef possibleMatch;
228           BoostAttribute boostAtt =
229             fe.attributes().addAttribute(BoostAttribute.class);
230           while ((possibleMatch = fe.next()) != null) {
231             numVariants++;
232             totalVariantDocFreqs += fe.docFreq();
233             float score = boostAtt.getBoost();
234             if (variantsQ.size() < MAX_VARIANTS_PER_TERM || score > minScore) {
235               ScoreTerm st = new ScoreTerm(new Term(startTerm.field(), BytesRef.deepCopyOf(possibleMatch)), score, startTerm);
236               variantsQ.insertWithOverflow(st);
237               minScore = variantsQ.top().score; // maintain minScore
238             }
239             maxBoostAtt.setMaxNonCompetitiveBoost(variantsQ.size() >= MAX_VARIANTS_PER_TERM ? minScore : Float.NEGATIVE_INFINITY);
240           }
241 
242           if (numVariants > 0) {
243             int avgDf = totalVariantDocFreqs / numVariants;
244             if (df == 0)//no direct match we can use as df for all variants
245             {
246               df = avgDf; //use avg df of all variants
247             }
248 
249             // take the top variants (scored by edit distance) and reset the score
250             // to include an IDF factor then add to the global queue for ranking
251             // overall top query terms
252             int size = variantsQ.size();
253             for (int i = 0; i < size; i++) {
254               ScoreTerm st = variantsQ.pop();
255               st.score = (st.score * st.score) * sim.idf(df, corpusNumDocs);
256               q.insertWithOverflow(st);
257             }
258           }
259         }
260       }
261       ts.end();
262     }
263   }

265   private Query newTermQuery(IndexReader reader, Term term) throws IOException {
266     if (ignoreTF) {
267       return new ConstantScoreQuery(new TermQuery(term));
268     } else {
269       // we build an artificial TermContext that will give an overall df and ttf
270       // equal to 1
271       TermContext context = new TermContext(reader.getContext());
272       for (LeafReaderContext leafContext : reader.leaves()) {
273         Terms terms = leafContext.reader().terms(term.field());
274         if (terms != null) {
275           TermsEnum termsEnum = terms.iterator();
276           if (termsEnum.seekExact(term.bytes())) {
277             int freq = 1 - context.docFreq(); // we want the total df and ttf to be 1
278             context.register(termsEnum.termState(), leafContext.ord, freq, freq);
279           }
280         }
281       }
282       return new TermQuery(term, context);
283     }

286   @Override
287     public Query rewrite(IndexReader reader) throws IOException
288     {
289         if (getBoost() != 1f) {
290           return super.rewrite(reader);
291         }
292         ScoreTermQueue q = new ScoreTermQueue(maxNumTerms);
293         //load up the list of possible terms
294         for (FieldVals f : fieldVals) {
295           addTerms(reader, f, q);
296         }
297         
298         BooleanQuery.Builder bq = new BooleanQuery.Builder();
299         
300         //create BooleanQueries to hold the variants for each token/field pair and ensure it
301         // has no coord factor
302         //Step 1: sort the termqueries by term/field
303         HashMap<Term,ArrayList<ScoreTerm>> variantQueries=new HashMap<>();
304         int size = q.size();
305         for(int i = 0; i < size; i++)
306         {
307           ScoreTerm st = q.pop();
308           ArrayList<ScoreTerm> l= variantQueries.get(st.fuzziedSourceTerm);
309           if(l==null)
310           {
311               l=new ArrayList<>();
312               variantQueries.put(st.fuzziedSourceTerm,l);
313           }
314           l.add(st);
315         }
316         //Step 2: Organize the sorted termqueries into zero-coord scoring boolean queries
317         for (Iterator<ArrayList<ScoreTerm>> iter = variantQueries.values().iterator(); iter.hasNext();)
318         {
319             ArrayList<ScoreTerm> variants = iter.next();
320             if(variants.size()==1)
321             {
322                 //optimize where only one selected variant
323                 ScoreTerm st= variants.get(0);
324                 Query tq = newTermQuery(reader, st.term);
325                 tq.setBoost(st.score); // set the boost to a mix of IDF and score
326                 bq.add(tq, BooleanClause.Occur.SHOULD); 
327             }
328             else
329             {
330                 BooleanQuery.Builder termVariants=new BooleanQuery.Builder();
331                 termVariants.setDisableCoord(true); //disable coord and IDF for these term variants
332                 for (Iterator<ScoreTerm> iterator2 = variants.iterator(); iterator2
333                         .hasNext();)
334                 {
335                     ScoreTerm st = iterator2.next();
336                     // found a match
337                     Query tq = newTermQuery(reader, st.term);
338                     tq.setBoost(st.score); // set the boost using the ScoreTerm's score
339                     termVariants.add(tq, BooleanClause.Occur.SHOULD);          // add to query                    
340                 }
341                 bq.add(termVariants.build(), BooleanClause.Occur.SHOULD);          // add to query
342             }
343         }
344         //TODO possible alternative step 3 - organize above booleans into a new layer of field-based
345         // booleans with a minimum-should-match of NumFields-1?
346         return bq.build();
347     }

349     //Holds info for a fuzzy term variant - initially score is set to edit distance (for ranking best
350     // term variants) then is reset with IDF for use in ranking against all other
351     // terms/fields
352     private static class ScoreTerm{
353         public Term term;
354         public float score;
355         Term fuzziedSourceTerm;
356         
357         public ScoreTerm(Term term, float score, Term fuzziedSourceTerm){
358           this.term = term;
359           this.score = score;
360           this.fuzziedSourceTerm=fuzziedSourceTerm;
361         }
362       }
363       
364       private static class ScoreTermQueue extends PriorityQueue<ScoreTerm> {        
365         public ScoreTermQueue(int size){
366           super(size);
367         }
368         
369         /* (non-Javadoc)
370          * @see org.apache.lucene.util.PriorityQueue#lessThan(java.lang.Object, java.lang.Object)
371          */
372         @Override
373         protected boolean lessThan(ScoreTerm termA, ScoreTerm termB) {
374           if (termA.score== termB.score)
375             return termA.term.compareTo(termB.term) > 0;
376           else
377             return termA.score < termB.score;
378         }
379         
380       }
