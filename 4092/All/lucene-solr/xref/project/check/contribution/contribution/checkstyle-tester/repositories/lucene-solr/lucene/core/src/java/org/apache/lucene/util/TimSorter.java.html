20  import java.util.Arrays;
21  
22  /**
23   * {@link Sorter} implementation based on the
24   * <a href="http://svn.python.org/projects/python/trunk/Objects/listsort.txt">TimSort</a>
25   * algorithm.
26   * <p>This implementation is especially good at sorting partially-sorted
27   * arrays and sorts small arrays with binary sort.
28   * <p><b>NOTE</b>:There are a few differences with the original implementation:<ul>
29   * <li><a name="maxTempSlots"></a>The extra amount of memory to perform merges is
30   * configurable. This allows small merges to be very fast while large merges
31   * will be performed in-place (slightly slower). You can make sure that the
32   * fast merge routine will always be used by having <code>maxTempSlots</code>
33   * equal to half of the length of the slice of data to sort.
34   * <li>Only the fast merge routine can gallop (the one that doesn't run
35   * in-place) and it only gallops on the longest slice.
36   * </ul>
37   * @lucene.internal
38   */
39  public abstract class TimSorter extends Sorter {
40  
41    static final int MINRUN = 32;
42    static final int THRESHOLD = 64;
43    static final int STACKSIZE = 49; // depends on MINRUN
44    static final int MIN_GALLOP = 7;
45  
46    final int maxTempSlots;
47    int minRun;
48    int to;
49    int stackSize;
50    int[] runEnds;
51  
52    /**
53     * Create a new {@link TimSorter}.
54     * @param maxTempSlots the <a href="#maxTempSlots">maximum amount of extra memory to run merges</a>
55     */
56    protected TimSorter(int maxTempSlots) {
57      super();
58      runEnds = new int[1 + STACKSIZE];
59      this.maxTempSlots = maxTempSlots;
60    }
61  
62    /** Minimum run length for an array of length <code>length</code>. */
63    static int minRun(int length) {
64      assert length >= MINRUN;
65      int n = length;
66      int r = 0;
67      while (n >= 64) {
68        r |= n & 1;
69        n >>>= 1;
70      }
71      final int minRun = n + r;
72      assert minRun >= MINRUN && minRun <= THRESHOLD;
73      return minRun;
74    }
75  
76    int runLen(int i) {
77      final int off = stackSize - i;
78      return runEnds[off] - runEnds[off - 1];
79    }
80  
81    int runBase(int i) {
82      return runEnds[stackSize - i - 1];
83    }
84  
85    int runEnd(int i) {
86      return runEnds[stackSize - i];
87    }
88  
89    void setRunEnd(int i, int runEnd) {
90      runEnds[stackSize - i] = runEnd;
91    }
92  
93    void pushRunLen(int len) {
94      runEnds[stackSize + 1] = runEnds[stackSize] + len;
95      ++stackSize;
96    }
97  
98    /** Compute the length of the next run, make the run sorted and return its
99     *  length. */
100   int nextRun() {
101     final int runBase = runEnd(0);
102     assert runBase < to;
103     if (runBase == to - 1) {
104       return 1;
105     }
106     int o = runBase + 2;
107     if (compare(runBase, runBase+1) > 0) {
108       // run must be strictly descending
109       while (o < to && compare(o - 1, o) > 0) {
110         ++o;
111       }
112       reverse(runBase, o);
113     } else {
114       // run must be non-descending
115       while (o < to && compare(o - 1, o) <= 0) {
116         ++o;
117       }
118     }
119     final int runHi = Math.max(o, Math.min(to, runBase + minRun));
120     binarySort(runBase, runHi, o);
121     return runHi - runBase;
122   }
123 
124   void ensureInvariants() {
125     while (stackSize > 1) {
126       final int runLen0 = runLen(0);
127       final int runLen1 = runLen(1);
128 
129       if (stackSize > 2) {
130         final int runLen2 = runLen(2);
131 
132         if (runLen2 <= runLen1 + runLen0) {
133           // merge the smaller of 0 and 2 with 1
134           if (runLen2 < runLen0) {
135             mergeAt(1);
136           } else {
137             mergeAt(0);
138           }
139           continue;
140         }
141       }
142 
143       if (runLen1 <= runLen0) {
144         mergeAt(0);
145         continue;
146       }
147 
148       break;
149     }
150   }
151 
152   void exhaustStack() {
153     while (stackSize > 1) {
154       mergeAt(0);
155     }
156   }
157 
158   void reset(int from, int to) {
159     stackSize = 0;
160     Arrays.fill(runEnds, 0);
161     runEnds[0] = from;
162     this.to = to;
163     final int length = to - from;
164     this.minRun = length <= THRESHOLD ? length : minRun(length);
165   }
166 
167   void mergeAt(int n) {
168     assert stackSize >= 2;
169     merge(runBase(n + 1), runBase(n), runEnd(n));
170     for (int j = n + 1; j > 0; --j) {
171       setRunEnd(j, runEnd(j-1));
172     }
173     --stackSize;
174   }
175 
176   void merge(int lo, int mid, int hi) {
177     if (compare(mid - 1, mid) <= 0) {
178       return;
179     }
180     lo = upper2(lo, mid, mid);
181     hi = lower2(mid, hi, mid - 1);
182 
183     if (hi - mid <= mid - lo && hi - mid <= maxTempSlots) {
184       mergeHi(lo, mid, hi);
185     } else if (mid - lo <= maxTempSlots) {
186       mergeLo(lo, mid, hi);
187     } else {
188       mergeInPlace(lo, mid, hi);
189     }
190   }
191 
192   @Override
193   public void sort(int from, int to) {
194     checkRange(from, to);
195     if (to - from <= 1) {
196       return;
197     }
198     reset(from, to);
199     do {
200       ensureInvariants();
201       pushRunLen(nextRun());
202     } while (runEnd(0) < to);
203     exhaustStack();
204     assert runEnd(0) == to;
205   }
206 
207   @Override
208   void doRotate(int lo, int mid, int hi) {
209     final int len1 = mid - lo;
210     final int len2 = hi - mid;
211     if (len1 == len2) {
212       while (mid < hi) {
213         swap(lo++, mid++);
214       }
215     } else if (len2 < len1 && len2 <= maxTempSlots) {
216       save(mid, len2);
217       for (int i = lo + len1 - 1, j = hi - 1; i >= lo; --i, --j) {
218         copy(i, j);
219       }
220       for (int i = 0, j = lo; i < len2; ++i, ++j) {
221         restore(i, j);
222       }
223     } else if (len1 <= maxTempSlots) {
224       save(lo, len1);
225       for (int i = mid, j = lo; i < hi; ++i, ++j) {
226         copy(i, j);
227       }
228       for (int i = 0, j = lo + len2; j < hi; ++i, ++j) {
229         restore(i, j);
230       }
231     } else {
232       reverse(lo, mid);
233       reverse(mid, hi);
234       reverse(lo, hi);
235     }
236   }
237 
238   void mergeLo(int lo, int mid, int hi) {
239     assert compare(lo, mid) > 0;
240     int len1 = mid - lo;
241     save(lo, len1);
242     copy(mid, lo);
243     int i = 0, j = mid + 1, dest = lo + 1;
244     outer: for (;;) {
245       for (int count = 0; count < MIN_GALLOP; ) {
246         if (i >= len1 || j >= hi) {
247           break outer;
248         } else if (compareSaved(i, j) <= 0) {
249           restore(i++, dest++);
250           count = 0;
251         } else {
252           copy(j++, dest++);
253           ++count;
254         }
255       }
256       // galloping...
257       int next = lowerSaved3(j, hi, i);
258       for (; j < next; ++dest) {
259         copy(j++, dest);
260       }
261       restore(i++, dest++);
262     }
263     for (; i < len1; ++dest) {
264       restore(i++, dest);
265     }
266     assert j == dest;
267   }
268 
269   void mergeHi(int lo, int mid, int hi) {
270     assert compare(mid - 1, hi - 1) > 0;
271     int len2 = hi - mid;
272     save(mid, len2);
273     copy(mid - 1, hi - 1);
274     int i = mid - 2, j = len2 - 1, dest = hi - 2;
275     outer: for (;;) {
276       for (int count = 0; count < MIN_GALLOP; ) {
277         if (i < lo || j < 0) {
278           break outer;
279         } else if (compareSaved(j, i) >= 0) {
280           restore(j--, dest--);
281           count = 0;
282         } else {
283           copy(i--, dest--);
284           ++count;
285         }
286       }
287       // galloping
288       int next = upperSaved3(lo, i + 1, j);
289       while (i >= next) {
290         copy(i--, dest--);
291       }
292       restore(j--, dest--);
293     }
294     for (; j >= 0; --dest) {
295       restore(j--, dest);
296     }
297     assert i == dest;
298   }
299 
300   int lowerSaved(int from, int to, int val) {
301     int len = to - from;
302     while (len > 0) {
303       final int half = len >>> 1;
304       final int mid = from + half;
305       if (compareSaved(val, mid) > 0) {
306         from = mid + 1;
307         len = len - half -1;
308       } else {
309         len = half;
310       }
311     }
312     return from;
313   }
314 
315   int upperSaved(int from, int to, int val) {
316     int len = to - from;
317     while (len > 0) {
318       final int half = len >>> 1;
319       final int mid = from + half;
320       if (compareSaved(val, mid) < 0) {
321         len = half;
322       } else {
323         from = mid + 1;
324         len = len - half -1;
325       }
326     }
327     return from;
328   }
329 
330   // faster than lowerSaved when val is at the beginning of [from:to[
331   int lowerSaved3(int from, int to, int val) {
332     int f = from, t = f + 1;
333     while (t < to) {
334       if (compareSaved(val, t) <= 0) {
335         return lowerSaved(f, t, val);
336       }
337       int delta = t - f;
338       f = t;
339       t += delta << 1;
340     }
341     return lowerSaved(f, to, val);
342   }
343 
344   //faster than upperSaved when val is at the end of [from:to[
345   int upperSaved3(int from, int to, int val) {
346     int f = to - 1, t = to;
347     while (f > from) {
348       if (compareSaved(val, f) >= 0) {
349         return upperSaved(f, t, val);
350       }
351       final int delta = t - f;
352       t = f;
353       f -= delta << 1;
354     }
355     return upperSaved(from, t, val);
356   }
357 
358   /** Copy data from slot <code>src</code> to slot <code>dest</code>. */
359   protected abstract void copy(int src, int dest);
360 
361   /** Save all elements between slots <code>i</code> and <code>i+len</code>
362    *  into the temporary storage. */
363   protected abstract void save(int i, int len);
364 
365   /** Restore element <code>j</code> from the temporary storage into slot <code>i</code>. */
366   protected abstract void restore(int i, int j);
367 
368   /** Compare element <code>i</code> from the temporary storage with element
369    *  <code>j</code> from the slice to sort, similarly to
370    *  {@link #compare(int, int)}. */
371   protected abstract int compareSaved(int i, int j);
