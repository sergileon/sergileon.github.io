47    private static class PreviousSearchState {
48      public final long searchTimeNanos;
49      public final long[] versions;
50      public final ScoreDoc searchAfterLocal;
51      public final ScoreDoc searchAfterShard;
52      public final Sort sort;
53      public final Query query;
54      public final int numHitsPaged;
55  
56      public PreviousSearchState(Query query, Sort sort, ScoreDoc searchAfterLocal, ScoreDoc searchAfterShard, long[] versions, int numHitsPaged) {
57        this.versions = versions.clone();
58        this.searchAfterLocal = searchAfterLocal;
59        this.searchAfterShard = searchAfterShard;
60        this.sort = sort;
61        this.query = query;
62        this.numHitsPaged = numHitsPaged;
63        searchTimeNanos = System.nanoTime();
64      }
65    }
66  
67    public void testSimple() throws Exception {
68      final int numNodes = TestUtil.nextInt(random(), 1, 10);
69  
70      final double runTimeSec = atLeast(3);
71  
72      final int minDocsToMakeTerms = TestUtil.nextInt(random(), 5, 20);
73  
74      final int maxSearcherAgeSeconds = TestUtil.nextInt(random(), 1, 3);
75  
76      if (VERBOSE) {
77        System.out.println("TEST: numNodes=" + numNodes + " runTimeSec=" + runTimeSec + " maxSearcherAgeSeconds=" + maxSearcherAgeSeconds);
78      }
79  
80      start(numNodes,
81            runTimeSec,
82            maxSearcherAgeSeconds
83            );
84  
85      final List<PreviousSearchState> priorSearches = new ArrayList<>();
86      List<BytesRef> terms = null;
87      while (System.nanoTime() < endTimeNanos) {
88  
89        final boolean doFollowon = priorSearches.size() > 0 && random().nextInt(7) == 1;
90  
91        // Pick a random node; we will run the query on this node:
92        final int myNodeID = random().nextInt(numNodes);
93  
94        final NodeState.ShardIndexSearcher localShardSearcher;
95  
96        final PreviousSearchState prevSearchState;
97  
98        if (doFollowon) {
99          // Pretend user issued a followon query:
100         prevSearchState = priorSearches.get(random().nextInt(priorSearches.size()));
101 
102         if (VERBOSE) {
103           System.out.println("\nTEST: follow-on query age=" + ((System.nanoTime() - prevSearchState.searchTimeNanos)/1000000000.0));
104         }
105 
106         try {
107           localShardSearcher = nodes[myNodeID].acquire(prevSearchState.versions);
108         } catch (SearcherExpiredException see) {
109           // Expected, sometimes; in a "real" app we would
110           // either forward this error to the user ("too
111           // much time has passed; please re-run your
112           // search") or sneakily just switch to newest
113           // searcher w/o telling them...
114           if (VERBOSE) {
115             System.out.println("  searcher expired during local shard searcher init: " + see);
116           }
117           priorSearches.remove(prevSearchState);
118           continue;
119         }
120       } else {
121         if (VERBOSE) {
122           System.out.println("\nTEST: fresh query");
123         }
124         // Do fresh query:
125         localShardSearcher = nodes[myNodeID].acquire();
126         prevSearchState = null;
127       }
128 
129       final IndexReader[] subs = new IndexReader[numNodes];
130 
131       PreviousSearchState searchState = null;
132 
133       try {
134 
135         // Mock: now make a single reader (MultiReader) from all node
136         // searchers.  In a real shard env you can't do this... we
137         // do it to confirm results from the shard searcher
138         // are correct:
139         int docCount = 0;
140         try {
141           for(int nodeID=0;nodeID<numNodes;nodeID++) {
142             final long subVersion = localShardSearcher.nodeVersions[nodeID];
143             final IndexSearcher sub = nodes[nodeID].searchers.acquire(subVersion);
144             if (sub == null) {
145               nodeID--;
146               while(nodeID >= 0) {
147                 subs[nodeID].decRef();
148                 subs[nodeID] = null;
149                 nodeID--;
150               }
151               throw new SearcherExpiredException("nodeID=" + nodeID + " version=" + subVersion);
152             }
153             subs[nodeID] = sub.getIndexReader();
154             docCount += subs[nodeID].maxDoc();
155           }
156         } catch (SearcherExpiredException see) {
157           // Expected
158           if (VERBOSE) {
159             System.out.println("  searcher expired during mock reader init: " + see);
160           }
161           continue;
162         }
163 
164         final IndexReader mockReader = new MultiReader(subs);
165         final IndexSearcher mockSearcher = new IndexSearcher(mockReader);
166 
167         Query query;
168         Sort sort;
169 
170         if (prevSearchState != null) {
171           query = prevSearchState.query;
172           sort = prevSearchState.sort;
173         } else {
174           if (terms == null && docCount > minDocsToMakeTerms) {
175             // TODO: try to "focus" on high freq terms sometimes too
176             // TODO: maybe also periodically reset the terms...?
177             final TermsEnum termsEnum = MultiFields.getTerms(mockReader, "body").iterator();
178             terms = new ArrayList<>();
179             while(termsEnum.next() != null) {
180               terms.add(BytesRef.deepCopyOf(termsEnum.term()));
181             }
182             if (VERBOSE) {
183               System.out.println("TEST: init terms: " + terms.size() + " terms");
184             }
185             if (terms.size() == 0) {
186               terms = null;
187             }
188           }
189 
190           if (VERBOSE) {
191             System.out.println("  maxDoc=" + mockReader.maxDoc());
192           }
193 
194           if (terms != null) {
195             if (random().nextBoolean()) {
196               query = new TermQuery(new Term("body", terms.get(random().nextInt(terms.size()))));
197             } else {
198               final String t = terms.get(random().nextInt(terms.size())).utf8ToString();
199               final String prefix;
200               if (t.length() <= 1) {
201                 prefix = t;
202               } else {
203                 prefix = t.substring(0, TestUtil.nextInt(random(), 1, 2));
204               }
205               query = new PrefixQuery(new Term("body", prefix));
206             }
207             
208             if (random().nextBoolean()) {
209               sort = null;
210             } else {
211               // TODO: sort by more than 1 field
212               final int what = random().nextInt(3);
213               if (what == 0) {
214                 sort = new Sort(SortField.FIELD_SCORE);
215               } else if (what == 1) {
216                 // TODO: this sort doesn't merge
217                 // correctly... it's tricky because you
218                 // could have > 2.1B docs across all shards: 
219                 //sort = new Sort(SortField.FIELD_DOC);
220                 sort = null;
221               } else if (what == 2) {
222                 sort = new Sort(new SortField[] {new SortField("docid_intDV", SortField.Type.INT, random().nextBoolean())});
223               } else {
224                 sort = new Sort(new SortField[] {new SortField("titleDV", SortField.Type.STRING, random().nextBoolean())});
225               }
226             }
227           } else {
228             query = null;
229             sort = null;
230           }
231         }
232 
233         if (query != null) {
234 
235           try {
236             searchState = assertSame(mockSearcher, localShardSearcher, query, sort, prevSearchState);
237           } catch (SearcherExpiredException see) {
238             // Expected; in a "real" app we would
239             // either forward this error to the user ("too
240             // much time has passed; please re-run your
241             // search") or sneakily just switch to newest
242             // searcher w/o telling them...
243             if (VERBOSE) {
244               System.out.println("  searcher expired during search: " + see);
245               see.printStackTrace(System.out);
246             }
247             // We can't do this in general: on a very slow
248             // computer it's possible the local searcher
249             // expires before we can finish our search:
250             // assert prevSearchState != null;
251             if (prevSearchState != null) {
252               priorSearches.remove(prevSearchState);
253             }
254           }
255         }
256       } finally {
257         nodes[myNodeID].release(localShardSearcher);
258         for(IndexReader sub : subs) {
259           if (sub != null) {
260             sub.decRef();
261           }
262         }
263       }
264 
265       if (searchState != null && searchState.searchAfterLocal != null && random().nextInt(5) == 3) {
266         priorSearches.add(searchState);
267         if (priorSearches.size() > 200) {
268           Collections.shuffle(priorSearches, random());
269           priorSearches.subList(100, priorSearches.size()).clear();
270         }
271       }
272     }
273 
274     finish();
275   }

277   private PreviousSearchState assertSame(IndexSearcher mockSearcher, NodeState.ShardIndexSearcher shardSearcher, Query q, Sort sort, PreviousSearchState state) throws IOException {
278 
279     int numHits = TestUtil.nextInt(random(), 1, 100);
280     if (state != null && state.searchAfterLocal == null) {
281       // In addition to what we last searched:
282       numHits += state.numHitsPaged;
283     }
284 
285     if (VERBOSE) {
286       System.out.println("TEST: query=" + q + " sort=" + sort + " numHits=" + numHits);
287       if (state != null) {
288         System.out.println("  prev: searchAfterLocal=" + state.searchAfterLocal + " searchAfterShard=" + state.searchAfterShard + " numHitsPaged=" + state.numHitsPaged);
289       }
290     }
291 
292     // Single (mock local) searcher:
293     final TopDocs hits;
294     if (sort == null) {
295       if (state != null && state.searchAfterLocal != null) {
296         hits = mockSearcher.searchAfter(state.searchAfterLocal, q, numHits);
297       } else {
298         hits = mockSearcher.search(q, numHits);
299       }
300     } else {
301       hits = mockSearcher.search(q, numHits, sort);
302     }
303 
304     // Shard searcher
305     final TopDocs shardHits;
306     if (sort == null) {
307       if (state != null && state.searchAfterShard != null) {
308         shardHits = shardSearcher.searchAfter(state.searchAfterShard, q, numHits);
309       } else {
310         shardHits = shardSearcher.search(q, numHits);
311       }
312     } else {
313       shardHits = shardSearcher.search(q, numHits, sort);
314     }
315 
316     final int numNodes = shardSearcher.nodeVersions.length;
317     int[] base = new int[numNodes];
318     final List<IndexReaderContext> subs = mockSearcher.getTopReaderContext().children();
319     assertEquals(numNodes, subs.size());
320 
321     for(int nodeID=0;nodeID<numNodes;nodeID++) {
322       base[nodeID] = subs.get(nodeID).docBaseInParent;
323     }
324 
325     if (VERBOSE) {
326       /*
327       for(int shardID=0;shardID<shardSearchers.length;shardID++) {
328         System.out.println("  shard=" + shardID + " maxDoc=" + shardSearchers[shardID].searcher.getIndexReader().maxDoc());
329       }
330       */
331       System.out.println("  single searcher: " + hits.totalHits + " totalHits maxScore=" + hits.getMaxScore());
332       for(int i=0;i<hits.scoreDocs.length;i++) {
333         final ScoreDoc sd = hits.scoreDocs[i];
334         System.out.println("    doc=" + sd.doc + " score=" + sd.score);
335       }
336       System.out.println("  shard searcher: " + shardHits.totalHits + " totalHits maxScore=" + shardHits.getMaxScore());
337       for(int i=0;i<shardHits.scoreDocs.length;i++) {
338         final ScoreDoc sd = shardHits.scoreDocs[i];
339         System.out.println("    doc=" + sd.doc + " (rebased: " + (sd.doc + base[sd.shardIndex]) + ") score=" + sd.score + " shard=" + sd.shardIndex);
340       }
341     }
342 
343     int numHitsPaged;
344     if (state != null && state.searchAfterLocal != null) {
345       numHitsPaged = hits.scoreDocs.length; 
346       if (state != null) {
347         numHitsPaged += state.numHitsPaged;
348       }
349     } else {
350       numHitsPaged = hits.scoreDocs.length;
351     }
352 
353     final boolean moreHits;
354 
355     final ScoreDoc bottomHit;
356     final ScoreDoc bottomHitShards;
357 
358     if (numHitsPaged < hits.totalHits) {
359       // More hits to page through
360       moreHits = true;
361       if (sort == null) {
362         bottomHit = hits.scoreDocs[hits.scoreDocs.length-1];
363         final ScoreDoc sd = shardHits.scoreDocs[shardHits.scoreDocs.length-1];
364         // Must copy because below we rebase:
365         bottomHitShards = new ScoreDoc(sd.doc, sd.score, sd.shardIndex);
366         if (VERBOSE) {
367           System.out.println("  save bottomHit=" + bottomHit);
368         }
369       } else {
370         bottomHit = null;
371         bottomHitShards = null;
372       }
373 
374     } else {
375       assertEquals(hits.totalHits, numHitsPaged);
376       bottomHit = null;
377       bottomHitShards = null;
378       moreHits = false;
379     }
380 
381     // Must rebase so assertEquals passes:
382     for(int hitID=0;hitID<shardHits.scoreDocs.length;hitID++) {
383       final ScoreDoc sd = shardHits.scoreDocs[hitID];
384       sd.doc += base[sd.shardIndex];
385     }
386 
387     TestUtil.assertEquals(hits, shardHits);
388 
389     if (moreHits) {
390       // Return a continuation:
391       return new PreviousSearchState(q, sort, bottomHit, bottomHitShards, shardSearcher.nodeVersions, numHitsPaged);
392     } else {
393       return null;
394     }
395   }
