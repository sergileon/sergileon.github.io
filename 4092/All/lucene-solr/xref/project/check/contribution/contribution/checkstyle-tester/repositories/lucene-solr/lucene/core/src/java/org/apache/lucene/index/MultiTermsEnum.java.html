29  /**
30   * Exposes {@link TermsEnum} API, merged from {@link TermsEnum} API of sub-segments.
31   * This does a merge sort, by term text, of the sub-readers.
32   *
33   * @lucene.experimental
34   */
35  public final class MultiTermsEnum extends TermsEnum {
36  
37    private static final Comparator<TermsEnumWithSlice> INDEX_COMPARATOR = new Comparator<TermsEnumWithSlice>() {
38      @Override
39      public int compare(TermsEnumWithSlice o1, TermsEnumWithSlice o2) {
40        return o1.index - o2.index;
41      }
42    };

44    private final TermMergeQueue queue;
45    private final TermsEnumWithSlice[] subs;        // all of our subs (one per sub-reader)
46    private final TermsEnumWithSlice[] currentSubs; // current subs that have at least one term for this field
47    private final TermsEnumWithSlice[] top;
48    private final MultiPostingsEnum.EnumWithSlice[] subDocs;
49  
50    private BytesRef lastSeek;
51    private boolean lastSeekExact;
52    private final BytesRefBuilder lastSeekScratch = new BytesRefBuilder();

54    private int numTop;
55    private int numSubs;
56    private BytesRef current;

58    static class TermsEnumIndex {
59      public final static TermsEnumIndex[] EMPTY_ARRAY = new TermsEnumIndex[0];
60      final int subIndex;

64        this.termsEnum = termsEnum;
65        this.subIndex = subIndex;

80    /** Sole constructor.
81     *  @param slices Which sub-reader slices we should
82     *  merge. */
83    public MultiTermsEnum(ReaderSlice[] slices) {
84      queue = new TermMergeQueue(slices.length);
85      top = new TermsEnumWithSlice[slices.length];
86      subs = new TermsEnumWithSlice[slices.length];
87      subDocs = new MultiPostingsEnum.EnumWithSlice[slices.length];
88      for(int i=0;i<slices.length;i++) {
89        subs[i] = new TermsEnumWithSlice(i, slices[i]);
90        subDocs[i] = new MultiPostingsEnum.EnumWithSlice();
91        subDocs[i].slice = slices[i];
92      }
93      currentSubs = new TermsEnumWithSlice[slices.length];
94    }

101   /** The terms array must be newly created TermsEnum, ie
102    *  {@link TermsEnum#next} has not yet been called. */
103   public TermsEnum reset(TermsEnumIndex[] termsEnumsIndex) throws IOException {
104     assert termsEnumsIndex.length <= top.length;
105     numSubs = 0;
106     numTop = 0;
107     queue.clear();
108     for(int i=0;i<termsEnumsIndex.length;i++) {
109 
110       final TermsEnumIndex termsEnumIndex = termsEnumsIndex[i];
111       assert termsEnumIndex != null;
112 
113       final BytesRef term = termsEnumIndex.termsEnum.next();
114       if (term != null) {
115         final TermsEnumWithSlice entry = subs[termsEnumIndex.subIndex];
116         entry.reset(termsEnumIndex.termsEnum, term);
117         queue.add(entry);
118         currentSubs[numSubs++] = entry;
119       } else {
120         // field has no terms
121       }
122     }
123 
124     if (queue.size() == 0) {
125       return TermsEnum.EMPTY;
126     } else {
127       return this;
128     }
129   }

131   @Override
132   public boolean seekExact(BytesRef term) throws IOException {
133     queue.clear();
134     numTop = 0;
135 
136     boolean seekOpt = false;
137     if (lastSeek != null && lastSeek.compareTo(term) <= 0) {
138       seekOpt = true;
139     }
140 
141     lastSeek = null;
142     lastSeekExact = true;
143 
144     for(int i=0;i<numSubs;i++) {
145       final boolean status;
146       // LUCENE-2130: if we had just seek'd already, prior
147       // to this seek, and the new seek term is after the
148       // previous one, don't try to re-seek this sub if its
149       // current term is already beyond this new seek term.
150       // Doing so is a waste because this sub will simply
151       // seek to the same spot.
152       if (seekOpt) {
153         final BytesRef curTerm = currentSubs[i].current;
154         if (curTerm != null) {
155           final int cmp = term.compareTo(curTerm);
156           if (cmp == 0) {
157             status = true;
158           } else if (cmp < 0) {
159             status = false;
160           } else {
161             status = currentSubs[i].terms.seekExact(term);
162           }
163         } else {
164           status = false;
165         }
166       } else {
167         status = currentSubs[i].terms.seekExact(term);
168       }
169 
170       if (status) {
171         top[numTop++] = currentSubs[i];
172         current = currentSubs[i].current = currentSubs[i].terms.term();
173         assert term.equals(currentSubs[i].current);
174       }
175     }
176 
177     // if at least one sub had exact match to the requested
178     // term then we found match
179     return numTop > 0;
180   }

182   @Override
183   public SeekStatus seekCeil(BytesRef term) throws IOException {
184     queue.clear();
185     numTop = 0;
186     lastSeekExact = false;
187 
188     boolean seekOpt = false;
189     if (lastSeek != null && lastSeek.compareTo(term) <= 0) {
190       seekOpt = true;
191     }
192 
193     lastSeekScratch.copyBytes(term);
194     lastSeek = lastSeekScratch.get();
195 
196     for(int i=0;i<numSubs;i++) {
197       final SeekStatus status;
198       // LUCENE-2130: if we had just seek'd already, prior
199       // to this seek, and the new seek term is after the
200       // previous one, don't try to re-seek this sub if its
201       // current term is already beyond this new seek term.
202       // Doing so is a waste because this sub will simply
203       // seek to the same spot.
204       if (seekOpt) {
205         final BytesRef curTerm = currentSubs[i].current;
206         if (curTerm != null) {
207           final int cmp = term.compareTo(curTerm);
208           if (cmp == 0) {
209             status = SeekStatus.FOUND;
210           } else if (cmp < 0) {
211             status = SeekStatus.NOT_FOUND;
212           } else {
213             status = currentSubs[i].terms.seekCeil(term);
214           }
215         } else {
216           status = SeekStatus.END;
217         }
218       } else {
219         status = currentSubs[i].terms.seekCeil(term);
220       }
221 
222       if (status == SeekStatus.FOUND) {
223         top[numTop++] = currentSubs[i];
224         current = currentSubs[i].current = currentSubs[i].terms.term();
225         queue.add(currentSubs[i]);
226       } else {
227         if (status == SeekStatus.NOT_FOUND) {
228           currentSubs[i].current = currentSubs[i].terms.term();
229           assert currentSubs[i].current != null;
230           queue.add(currentSubs[i]);
231         } else {
232           assert status == SeekStatus.END;
233           // enum exhausted
234           currentSubs[i].current = null;
235         }
236       }
237     }
238 
239     if (numTop > 0) {
240       // at least one sub had exact match to the requested term
241       return SeekStatus.FOUND;
242     } else if (queue.size() > 0) {
243       // no sub had exact match, but at least one sub found
244       // a term after the requested term -- advance to that
245       // next term:
246       pullTop();
247       return SeekStatus.NOT_FOUND;
248     } else {
249       return SeekStatus.END;
250     }
251   }

263   private void pullTop() {
264     // extract all subs from the queue that have the same
265     // top term
266     assert numTop == 0;
267     numTop = queue.fillTop(top);
268     current = top[0].current;
269   }
270 
271   private void pushTop() throws IOException {
272     // call next() on each top, and reorder queue
273     for (int i = 0; i < numTop; i++) {
274       TermsEnumWithSlice top = queue.top();
275       top.current = top.terms.next();
276       if (top.current == null) {
277         queue.pop();
278       } else {
279         queue.updateTop();
280       }
281     }
282     numTop = 0;
283   }
284 
285   @Override
286   public BytesRef next() throws IOException {
287     if (lastSeekExact) {
288       // Must seekCeil at this point, so those subs that
289       // didn't have the term can find the following term.
290       // NOTE: we could save some CPU by only seekCeil the
291       // subs that didn't match the last exact seek... but
292       // most impls short-circuit if you seekCeil to term
293       // they are already on.
294       final SeekStatus status = seekCeil(current);
295       assert status == SeekStatus.FOUND;
296       lastSeekExact = false;
297     }
298     lastSeek = null;
299 
300     // restore queue
301     pushTop();
302 
303     // gather equal top fields
304     if (queue.size() > 0) {
305       pullTop();
306     } else {
307       current = null;
308     }
309 
310     return current;
311   }

313   @Override
314   public int docFreq() throws IOException {
315     int sum = 0;
316     for(int i=0;i<numTop;i++) {
317       sum += top[i].terms.docFreq();
318     }
319     return sum;
320   }
321 
322   @Override
323   public long totalTermFreq() throws IOException {
324     long sum = 0;
325     for(int i=0;i<numTop;i++) {
326       final long v = top[i].terms.totalTermFreq();
327       if (v == -1) {
328         return v;
329       }
330       sum += v;
331     }
332     return sum;
333   }

335   @Override
336   public PostingsEnum postings(PostingsEnum reuse, int flags) throws IOException {
337     MultiPostingsEnum docsEnum;
338 
339     // Can only reuse if incoming enum is also a MultiDocsEnum
340     if (reuse != null && reuse instanceof MultiPostingsEnum) {
341       docsEnum = (MultiPostingsEnum) reuse;
342       // ... and was previously created w/ this MultiTermsEnum:
343       if (!docsEnum.canReuse(this)) {
344         docsEnum = new MultiPostingsEnum(this, subs.length);
345       }
346     } else {
347       docsEnum = new MultiPostingsEnum(this, subs.length);
348     }
349 
350     int upto = 0;
351 
352     ArrayUtil.timSort(top, 0, numTop, INDEX_COMPARATOR);
353 
354     for(int i=0;i<numTop;i++) {
355 
356       final TermsEnumWithSlice entry = top[i];
357 
358       assert entry.index < docsEnum.subPostingsEnums.length: entry.index + " vs " + docsEnum.subPostingsEnums.length + "; " + subs.length;
359       final PostingsEnum subPostingsEnum = entry.terms.postings(docsEnum.subPostingsEnums[entry.index], flags);
360       assert subPostingsEnum != null;
361       docsEnum.subPostingsEnums[entry.index] = subPostingsEnum;
362       subDocs[upto].postingsEnum = subPostingsEnum;
363       subDocs[upto].slice = entry.subSlice;
364       upto++;
365     }
366     
367     return docsEnum.reset(subDocs, upto);
368   }

370   final static class TermsEnumWithSlice {
371     private final ReaderSlice subSlice;
372     TermsEnum terms;
373     public BytesRef current;
374     final int index;
375 
376     public TermsEnumWithSlice(int index, ReaderSlice subSlice) {
377       this.subSlice = subSlice;
378       this.index = index;
379       assert subSlice.length >= 0: "length=" + subSlice.length;
380     }
381 
382     public void reset(TermsEnum terms, BytesRef term) {
383       this.terms = terms;
384       current = term;
385     }
386 
387     @Override
388     public String toString() {
389       return subSlice.toString()+":"+terms;
390     }
391   }
392 
393   private final static class TermMergeQueue extends PriorityQueue<TermsEnumWithSlice> {
394 
395     final int[] stack;
396 
397     TermMergeQueue(int size) {
398       super(size);
399       this.stack = new int[size];
400     }
401 
402     @Override
403     protected boolean lessThan(TermsEnumWithSlice termsA, TermsEnumWithSlice termsB) {
404       return termsA.current.compareTo(termsB.current) < 0;
405     }
406 
407     /** Add the {@link #top()} slice as well as all slices that are positionned
408      *  on the same term to {@code tops} and return how many of them there are. */
409     int fillTop(TermsEnumWithSlice[] tops) {
410       final int size = size();
411       if (size == 0) {
412         return 0;
413       }
414       tops[0] = top();
415       int numTop = 1;
416       stack[0] = 1;
417       int stackLen = 1;
418 
419       while (stackLen != 0) {
420         final int index = stack[--stackLen];
421         final int leftChild = index << 1;
422         for (int child = leftChild, end = Math.min(size, leftChild + 1); child <= end; ++child) {
423           TermsEnumWithSlice te = get(child);
424           if (te.current.equals(tops[0].current)) {
425             tops[numTop++] = te;
426             stack[stackLen++] = child;
427           }
428         }
429       }
430       return numTop;
431     }
432 
433     private TermsEnumWithSlice get(int i) {
434       return (TermsEnumWithSlice) getHeapArray()[i];
435     }
436   }
437 
438   @Override
439   public String toString() {
440     return "MultiTermsEnum(" + Arrays.toString(subs) + ")";
