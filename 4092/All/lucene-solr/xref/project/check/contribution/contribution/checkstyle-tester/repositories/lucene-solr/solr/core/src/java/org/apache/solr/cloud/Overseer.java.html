129     @Override
130     public void run() {
131 
132       LeaderStatus isLeader = amILeader();
133       while (isLeader == LeaderStatus.DONT_KNOW) {
134         log.debug("am_i_leader unclear {}", isLeader);
135         isLeader = amILeader();  // not a no, not a yes, try ask again
136       }
137 
138       log.info("Starting to work on the main queue");
139       try {
140         ZkStateWriter zkStateWriter = null;
141         ClusterState clusterState = null;
142         boolean refreshClusterState = true; // let's refresh in the first iteration
143         while (!this.isClosed) {
144           isLeader = amILeader();
145           if (LeaderStatus.NO == isLeader) {
146             break;
147           }
148           else if (LeaderStatus.YES != isLeader) {
149             log.debug("am_i_leader unclear {}", isLeader);
150             continue; // not a no, not a yes, try ask again
151           }
152 
153           if (refreshClusterState) {
154             try {
155               reader.updateClusterState();
156               clusterState = reader.getClusterState();
157               zkStateWriter = new ZkStateWriter(reader, stats);
158               refreshClusterState = false;
159 
160               // if there were any errors while processing
161               // the state queue, items would have been left in the
162               // work queue so let's process those first
163               byte[] data = workQueue.peek();
164               boolean hadWorkItems = data != null;
165               while (data != null)  {
166                 final ZkNodeProps message = ZkNodeProps.load(data);
167                 log.info("processMessage: workQueueSize: {}, message = {}", workQueue.getStats().getQueueLength(), message);
168                 // force flush to ZK after each message because there is no fallback if workQueue items
169                 // are removed from workQueue but fail to be written to ZK
170                 clusterState = processQueueItem(message, clusterState, zkStateWriter, false, null);
171                 workQueue.poll(); // poll-ing removes the element we got by peek-ing
172                 data = workQueue.peek();
173               }
174               // force flush at the end of the loop
175               if (hadWorkItems) {
176                 clusterState = zkStateWriter.writePendingUpdates();
177               }
178             } catch (KeeperException e) {
179               if (e.code() == KeeperException.Code.SESSIONEXPIRED) {
180                 log.warn("Solr cannot talk to ZK, exiting Overseer work queue loop", e);
181                 return;
182               }
183               log.error("Exception in Overseer work queue loop", e);
184             } catch (InterruptedException e) {
185               Thread.currentThread().interrupt();
186               return;
187             } catch (Exception e) {
188               log.error("Exception in Overseer work queue loop", e);
189             }
190           }
191 
192           byte[] head = null;
193           try {
194             head = stateUpdateQueue.peek(true);
195           } catch (KeeperException e) {
196             if (e.code() == KeeperException.Code.SESSIONEXPIRED) {
197               log.warn(
198                   "Solr cannot talk to ZK, exiting Overseer main queue loop", e);
199               return;
200             }
201             log.error("Exception in Overseer main queue loop", e);
202           } catch (InterruptedException e) {
203             Thread.currentThread().interrupt();
204             return;
205 
206           } catch (Exception e) {
207             log.error("Exception in Overseer main queue loop", e);
208           }
209           try {
210             while (head != null) {
211               final byte[] data = head;
212               final ZkNodeProps message = ZkNodeProps.load(data);
213               log.info("processMessage: queueSize: {}, message = {} current state version: {}", stateUpdateQueue.getStats().getQueueLength(), message, clusterState.getZkClusterStateVersion());
214               // we can batch here because workQueue is our fallback in case a ZK write failed
215               clusterState = processQueueItem(message, clusterState, zkStateWriter, true, new ZkStateWriter.ZkWriteCallback() {
216                 @Override
217                 public void onEnqueue() throws Exception {
218                   workQueue.offer(data);
219                 }
220 
221                 @Override
222                 public void onWrite() throws Exception {
223                   // remove everything from workQueue
224                   while (workQueue.poll() != null);
225                 }
226               });
227 
228               // it is safer to keep this poll here because an invalid message might never be queued
229               // and therefore we can't rely on the ZkWriteCallback to remove the item
230               stateUpdateQueue.poll();
231 
232               if (isClosed) break;
233               // if an event comes in the next 100ms batch it together
234               head = stateUpdateQueue.peek(100);
235             }
236             // we should force write all pending updates because the next iteration might sleep until there
237             // are more items in the main queue
238             clusterState = zkStateWriter.writePendingUpdates();
239             // clean work queue
240             while (workQueue.poll() != null);
241 
242           } catch (KeeperException.BadVersionException bve) {
243             log.warn("Bad version writing to ZK using compare-and-set, will force refresh cluster state", bve);
244             refreshClusterState = true;
245           } catch (KeeperException e) {
246             if (e.code() == KeeperException.Code.SESSIONEXPIRED) {
247               log.warn("Solr cannot talk to ZK, exiting Overseer main queue loop", e);
248               return;
249             }
250             log.error("Exception in Overseer main queue loop", e);
251             refreshClusterState = true; // force refresh state in case of all errors
252           } catch (InterruptedException e) {
253             Thread.currentThread().interrupt();
254             return;
255           } catch (Exception e) {
256             log.error("Exception in Overseer main queue loop", e);
257             refreshClusterState = true; // it might have been a bad version error
258           }
259         }
260       } finally {
261         log.info("Overseer Loop exiting : {}", LeaderElector.getNodeName(myId));
262         new Thread("OverseerExitThread"){
263           //do this in a separate thread because any wait is interrupted in this main thread
264           @Override
265           public void run() {
266             checkIfIamStillLeader();
267           }
268         }.start();
269       }
270     }

272     private ClusterState processQueueItem(ZkNodeProps message, ClusterState clusterState, ZkStateWriter zkStateWriter, boolean enableBatching, ZkStateWriter.ZkWriteCallback callback) throws Exception {
273       final String operation = message.getStr(QUEUE_OPERATION);
274       ZkWriteCommand zkWriteCommand = null;
275       final TimerContext timerContext = stats.time(operation);
276       try {
277         zkWriteCommand = processMessage(clusterState, message, operation);
278         stats.success(operation);
279       } catch (Exception e) {
280         // generally there is nothing we can do - in most cases, we have
281         // an issue that will fail again on retry or we cannot communicate with     a
282         // ZooKeeper in which case another Overseer should take over
283         // TODO: if ordering for the message is not important, we could
284         // track retries and put it back on the end of the queue
285         log.error("Overseer could not process the current clusterstate state update message, skipping the message.", e);
286         stats.error(operation);
287       } finally {
288         timerContext.stop();
289       }
290       if (zkWriteCommand != null) {
291         clusterState = zkStateWriter.enqueueUpdate(clusterState, zkWriteCommand, callback);
292         if (!enableBatching)  {
293           clusterState = zkStateWriter.writePendingUpdates();
294         }
295       }
296       return clusterState;
297     }

299     private void checkIfIamStillLeader() {
300       if (zkController != null && zkController.getCoreContainer().isShutDown()) return;//shutting down no need to go further
301       org.apache.zookeeper.data.Stat stat = new org.apache.zookeeper.data.Stat();
302       String path = OverseerElectionContext.OVERSEER_ELECT + "/leader";
303       byte[] data;
304       try {
305         data = zkClient.getData(path, null, stat, true);
306       } catch (Exception e) {
307         log.error("could not read the data" ,e);
308         return;
309       }
310       try {
311         Map m = (Map) Utils.fromJSON(data);
312         String id = (String) m.get("id");
313         if(overseerCollectionConfigSetProcessor.getId().equals(id)){
314           try {
315             log.info("I'm exiting , but I'm still the leader");
316             zkClient.delete(path,stat.getVersion(),true);
317           } catch (KeeperException.BadVersionException e) {
318             //no problem ignore it some other Overseer has already taken over
319           } catch (Exception e) {
320             log.error("Could not delete my leader node ", e);
321           }
322 
323         } else{
324           log.info("somebody else has already taken up the overseer position");
325         }
326       } finally {
327         //if I am not shutting down, Then I need to rejoin election
328         try {
329           if (zkController != null && !zkController.getCoreContainer().isShutDown()) {
330             zkController.rejoinOverseerElection(null, false);
331           }
332         } catch (Exception e) {
333           log.warn("Unable to rejoinElection ",e);
334         }
335       }
336     }

338     private ZkWriteCommand processMessage(ClusterState clusterState,
339         final ZkNodeProps message, final String operation) {
340       CollectionParams.CollectionAction collectionAction = CollectionParams.CollectionAction.get(operation);
341       if (collectionAction != null) {
342         switch (collectionAction) {
343           case CREATE:
344             return new ClusterStateMutator(getZkStateReader()).createCollection(clusterState, message);
345           case DELETE:
346             return new ClusterStateMutator(getZkStateReader()).deleteCollection(clusterState, message);
347           case CREATESHARD:
348             return new CollectionMutator(getZkStateReader()).createShard(clusterState, message);
349           case DELETESHARD:
350             return new CollectionMutator(getZkStateReader()).deleteShard(clusterState, message);
351           case ADDREPLICA:
352             return new SliceMutator(getZkStateReader()).addReplica(clusterState, message);
353           case ADDREPLICAPROP:
354             return new ReplicaMutator(getZkStateReader()).addReplicaProperty(clusterState, message);
355           case DELETEREPLICAPROP:
356             return new ReplicaMutator(getZkStateReader()).deleteReplicaProperty(clusterState, message);
357           case BALANCESHARDUNIQUE:
358             ExclusiveSliceProperty dProp = new ExclusiveSliceProperty(clusterState, message);
359             if (dProp.balanceProperty()) {
360               String collName = message.getStr(ZkStateReader.COLLECTION_PROP);
361               return new ZkWriteCommand(collName, dProp.getDocCollection());
362             }
363             break;
364           case MODIFYCOLLECTION:
365             CollectionsHandler.verifyRuleParams(zkController.getCoreContainer() ,message.getProperties());
366             return new CollectionMutator(reader).modifyCollection(clusterState,message);
367           case MIGRATESTATEFORMAT:
368             return new ClusterStateMutator(reader).migrateStateFormat(clusterState, message);
369           default:
370             throw new RuntimeException("unknown operation:" + operation
371                 + " contents:" + message.getProperties());
372         }
373       } else {
374         OverseerAction overseerAction = OverseerAction.get(operation);
375         if (overseerAction == null) {
376           throw new RuntimeException("unknown operation:" + operation + " contents:" + message.getProperties());
377         }
378         switch (overseerAction) {
379           case STATE:
380             return new ReplicaMutator(getZkStateReader()).setState(clusterState, message);
381           case LEADER:
382             return new SliceMutator(getZkStateReader()).setShardLeader(clusterState, message);
383           case DELETECORE:
384             return new SliceMutator(getZkStateReader()).removeReplica(clusterState, message);
385           case ADDROUTINGRULE:
386             return new SliceMutator(getZkStateReader()).addRoutingRule(clusterState, message);
387           case REMOVEROUTINGRULE:
388             return new SliceMutator(getZkStateReader()).removeRoutingRule(clusterState, message);
389           case UPDATESHARDSTATE:
390             return new SliceMutator(getZkStateReader()).updateShardState(clusterState, message);
391           case QUIT:
392             if (myId.equals(message.get("id"))) {
393               log.info("Quit command received {}", LeaderElector.getNodeName(myId));
394               overseerCollectionConfigSetProcessor.close();
395               close();
396             } else {
397               log.warn("Overseer received wrong QUIT message {}", message);
398             }
399             break;
400           default:
401             throw new RuntimeException("unknown operation:" + operation + " contents:" + message.getProperties());
402         }
403       }
404 
405       return ZkStateWriter.NO_OP;
406     }

408     private LeaderStatus amILeader() {
409       TimerContext timerContext = stats.time("am_i_leader");
410       boolean success = true;
411       try {
412         ZkNodeProps props = ZkNodeProps.load(zkClient.getData(
413             OverseerElectionContext.OVERSEER_ELECT + "/leader", null, null, true));
414         if (myId.equals(props.getStr("id"))) {
415           return LeaderStatus.YES;
416         }
417       } catch (KeeperException e) {
418         success = false;
419         if (e.code() == KeeperException.Code.CONNECTIONLOSS) {
420           log.error("", e);
421           return LeaderStatus.DONT_KNOW;
422         } else if (e.code() == KeeperException.Code.SESSIONEXPIRED) {
423           log.info("", e);
424         } else {
425           log.warn("", e);
426         }
427       } catch (InterruptedException e) {
428         success = false;
429         Thread.currentThread().interrupt();
430       } finally {
431         timerContext.stop();
432         if (success)  {
433           stats.success("am_i_leader");
434         } else  {
435           stats.error("am_i_leader");
436         }
437       }
438       log.info("According to ZK I (id=" + myId + ") am no longer a leader.");
439       return LeaderStatus.NO;
440     }

448   // Class to encapsulate processing replica properties that have at most one replica hosting a property per slice.
449   private class ExclusiveSliceProperty {
450     private ClusterState clusterState;
451     private final boolean onlyActiveNodes;
452     private final String property;
453     private final DocCollection collection;
454     private final String collectionName;
455 
456     // Key structure. For each node, list all replicas on it regardless of whether they have the property or not.
457     private final Map<String, List<SliceReplica>> nodesHostingReplicas = new HashMap<>();
458     // Key structure. For each node, a list of the replicas _currently_ hosting the property.
459     private final Map<String, List<SliceReplica>> nodesHostingProp = new HashMap<>();
460     Set<String> shardsNeedingHosts = new HashSet<String>();
461     Map<String, Slice> changedSlices = new HashMap<>(); // Work on copies rather than the underlying cluster state.
462 
463     private int origMaxPropPerNode = 0;
464     private int origModulo = 0;
465     private int tmpMaxPropPerNode = 0;
466     private int tmpModulo = 0;
467     Random rand = new Random();
468 
469     private int assigned = 0;
470 
471     ExclusiveSliceProperty(ClusterState clusterState, ZkNodeProps message) {
472       this.clusterState = clusterState;
473       String tmp = message.getStr(ZkStateReader.PROPERTY_PROP);
474       if (StringUtils.startsWith(tmp, OverseerCollectionMessageHandler.COLL_PROP_PREFIX) == false) {
475         tmp = OverseerCollectionMessageHandler.COLL_PROP_PREFIX + tmp;
476       }
477       this.property = tmp.toLowerCase(Locale.ROOT);
478       collectionName = message.getStr(ZkStateReader.COLLECTION_PROP);
479 
480       if (StringUtils.isBlank(collectionName) || StringUtils.isBlank(property)) {
481         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
482             "Overseer '" + message.getStr(QUEUE_OPERATION) + "'  requires both the '" + ZkStateReader.COLLECTION_PROP + "' and '" +
483                 ZkStateReader.PROPERTY_PROP + "' parameters. No action taken ");
484       }
485 
486       Boolean shardUnique = Boolean.parseBoolean(message.getStr(SHARD_UNIQUE));
487       if (shardUnique == false &&
488           SliceMutator.SLICE_UNIQUE_BOOLEAN_PROPERTIES.contains(this.property) == false) {
489         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Balancing properties amongst replicas in a slice requires that"
490             + " the property be a pre-defined property (e.g. 'preferredLeader') or that 'shardUnique' be set to 'true' " +
491             " Property: " + this.property + " shardUnique: " + Boolean.toString(shardUnique));
492       }
493 
494       collection = clusterState.getCollection(collectionName);
495       if (collection == null) {
496         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
497             "Could not find collection ' " + collectionName + "' for overseer operation '" +
498                 message.getStr(QUEUE_OPERATION) + "'. No action taken.");
499       }
500       onlyActiveNodes = Boolean.parseBoolean(message.getStr(ONLY_ACTIVE_NODES, "true"));
501     }

508     private boolean isActive(Replica replica) {
509       return replica.getState() == Replica.State.ACTIVE;
510     }
511 
512     // Collect a list of all the nodes that _can_ host the indicated property. Along the way, also collect any of
513     // the replicas on that node that _already_ host the property as well as any slices that do _not_ have the
514     // property hosted.
515     //
516     // Return true if anything node needs it's property reassigned. False if the property is already balanced for
517     // the collection.
518 
519     private boolean collectCurrentPropStats() {
520       int maxAssigned = 0;
521       // Get a list of potential replicas that can host the property _and_ their counts
522       // Move any obvious entries to a list of replicas to change the property on
523       Set<String> allHosts = new HashSet<>();
524       for (Slice slice : collection.getSlices()) {
525         boolean sliceHasProp = false;
526         for (Replica replica : slice.getReplicas()) {
527           if (onlyActiveNodes && isActive(replica) == false) {
528             if (StringUtils.isNotBlank(replica.getStr(property))) {
529               removeProp(slice, replica.getName()); // Note, we won't be committing this to ZK until later.
530             }
531             continue;
532           }
533           allHosts.add(replica.getNodeName());
534           String nodeName = replica.getNodeName();
535           if (StringUtils.isNotBlank(replica.getStr(property))) {
536             if (sliceHasProp) {
537               throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
538                   "'" + BALANCESHARDUNIQUE + "' should only be called for properties that have at most one member " +
539                       "in any slice with the property set. No action taken.");
540             }
541             if (nodesHostingProp.containsKey(nodeName) == false) {
542               nodesHostingProp.put(nodeName, new ArrayList<SliceReplica>());
543             }
544             nodesHostingProp.get(nodeName).add(new SliceReplica(slice, replica));
545             ++assigned;
546             maxAssigned = Math.max(maxAssigned, nodesHostingProp.get(nodeName).size());
547             sliceHasProp = true;
548           }
549           if (nodesHostingReplicas.containsKey(nodeName) == false) {
550             nodesHostingReplicas.put(nodeName, new ArrayList<SliceReplica>());
551           }
552           nodesHostingReplicas.get(nodeName).add(new SliceReplica(slice, replica));
553         }
554       }
555 
556       // If the total number of already-hosted properties assigned to nodes
557       // that have potential to host leaders is equal to the slice count _AND_ none of the current nodes has more than
558       // the max number of properties, there's nothing to do.
559       origMaxPropPerNode = collection.getSlices().size() / allHosts.size();
560 
561       // Some nodes can have one more of the proeprty if the numbers aren't exactly even.
562       origModulo = collection.getSlices().size() % allHosts.size();
563       if (origModulo > 0) {
564         origMaxPropPerNode++;  // have to have some nodes with 1 more property.
565       }
566 
567       // We can say for sure that we need to rebalance if we don't have as many assigned properties as slices.
568       if (assigned != collection.getSlices().size()) {
569         return true;
570       }
571 
572       // Make sure there are no more slices at the limit than the "leftovers"
573       // Let's say there's 7 slices and 3 nodes. We need to distribute the property as 3 on node1, 2 on node2 and 2 on node3
574       // (3, 2, 2) We need to be careful to not distribute them as 3, 3, 1. that's what this check is all about.
575       int counter = origModulo;
576       for (List<SliceReplica> list : nodesHostingProp.values()) {
577         if (list.size() == origMaxPropPerNode) --counter;
578       }
579       if (counter == 0) return false; // nodes with 1 extra leader are exactly the needed number
580 
581       return true;
582     }

584     private void removeSliceAlreadyHostedFromPossibles(String sliceName) {
585       for (Map.Entry<String, List<SliceReplica>> entReplica : nodesHostingReplicas.entrySet()) {
586 
587         ListIterator<SliceReplica> iter = entReplica.getValue().listIterator();
588         while (iter.hasNext()) {
589           SliceReplica sr = iter.next();
590           if (sr.slice.getName().equals(sliceName))
591             iter.remove();
592         }
593       }
594     }

596     private void balanceUnassignedReplicas() {
597       tmpMaxPropPerNode = origMaxPropPerNode; // A bit clumsy, but don't want to duplicate code.
598       tmpModulo = origModulo;
599 
600       // Get the nodeName and shardName for the node that has the least room for this
601 
602       while (shardsNeedingHosts.size() > 0) {
603         String nodeName = "";
604         int minSize = Integer.MAX_VALUE;
605         SliceReplica srToChange = null;
606         for (String slice : shardsNeedingHosts) {
607           for (Map.Entry<String, List<SliceReplica>> ent : nodesHostingReplicas.entrySet()) {
608             // A little tricky. If we don't set this to something below, then it means all possible places to
609             // put this property are full up, so just put it somewhere.
610             if (srToChange == null && ent.getValue().size() > 0) {
611               srToChange = ent.getValue().get(0);
612             }
613             ListIterator<SliceReplica> iter = ent.getValue().listIterator();
614             while (iter.hasNext()) {
615               SliceReplica sr = iter.next();
616               if (StringUtils.equals(slice, sr.slice.getName()) == false) {
617                 continue;
618               }
619               if (nodesHostingProp.containsKey(ent.getKey()) == false) {
620                 nodesHostingProp.put(ent.getKey(), new ArrayList<SliceReplica>());
621               }
622               if (minSize > nodesHostingReplicas.get(ent.getKey()).size() && nodesHostingProp.get(ent.getKey()).size() < tmpMaxPropPerNode) {
623                 minSize = nodesHostingReplicas.get(ent.getKey()).size();
624                 srToChange = sr;
625                 nodeName = ent.getKey();
626               }
627             }
628           }
629         }
630         // Now, you have a slice and node to put it on
631         shardsNeedingHosts.remove(srToChange.slice.getName());
632         if (nodesHostingProp.containsKey(nodeName) == false) {
633           nodesHostingProp.put(nodeName, new ArrayList<SliceReplica>());
634         }
635         nodesHostingProp.get(nodeName).add(srToChange);
636         adjustLimits(nodesHostingProp.get(nodeName));
637         removeSliceAlreadyHostedFromPossibles(srToChange.slice.getName());
638         addProp(srToChange.slice, srToChange.replica.getName());
639       }
640     }

642     // Adjust the min/max counts per allowed per node. Special handling here for dealing with the fact
643     // that no node should have more than 1 more replica with this property than any other.
644     private void adjustLimits(List<SliceReplica> changeList) {
645       if (changeList.size() == tmpMaxPropPerNode) {
646         if (tmpModulo < 0) return;
647 
648         --tmpModulo;
649         if (tmpModulo == 0) {
650           --tmpMaxPropPerNode;
651           --tmpModulo;  // Prevent dropping tmpMaxPropPerNode again.
652         }
653       }
654     }
655 
656     // Go through the list of presently-hosted proeprties and remove any that have too many replicas that host the property
657     private void removeOverallocatedReplicas() {
658       tmpMaxPropPerNode = origMaxPropPerNode; // A bit clumsy, but don't want to duplicate code.
659       tmpModulo = origModulo;
660 
661       for (Map.Entry<String, List<SliceReplica>> ent : nodesHostingProp.entrySet()) {
662         while (ent.getValue().size() > tmpMaxPropPerNode) { // remove delta nodes
663           ent.getValue().remove(rand.nextInt(ent.getValue().size()));
664         }
665         adjustLimits(ent.getValue());
666       }
667     }

669     private void removeProp(Slice origSlice, String replicaName) {
670       getReplicaFromChanged(origSlice, replicaName).getProperties().remove(property);
671     }

674       getReplicaFromChanged(origSlice, replicaName).getProperties().put(property, "true");

677     // Just a place to encapsulate the fact that we need to have new slices (copy) to update before we
678     // put this all in the cluster state.
679     private Replica getReplicaFromChanged(Slice origSlice, String replicaName) {
680       Slice newSlice = changedSlices.get(origSlice.getName());
681       Replica replica;
682       if (newSlice != null) {
683         replica = newSlice.getReplica(replicaName);
684       } else {
685         newSlice = new Slice(origSlice.getName(), origSlice.getReplicasCopy(), origSlice.shallowCopy());
686         changedSlices.put(origSlice.getName(), newSlice);
687         replica = newSlice.getReplica(replicaName);
688       }
689       if (replica == null) {
690         throw new SolrException(SolrException.ErrorCode.INVALID_STATE, "Should have been able to find replica '" +
691             replicaName + "' in slice '" + origSlice.getName() + "'. No action taken");
692       }
693       return replica;
694 
695     }
696     // Main entry point for carrying out the action. Returns "true" if we have actually moved properties around.
697 
698     private boolean balanceProperty() {
699       if (collectCurrentPropStats() == false) {
700         return false;
701       }
702 
703       // we have two lists based on nodeName
704       // 1> all the nodes that _could_ host a property for the slice
705       // 2> all the nodes that _currently_ host a property for the slice.
706 
707       // So, remove a replica from the nodes that have too many
708       removeOverallocatedReplicas();
709 
710       // prune replicas belonging to a slice that have the property currently assigned from the list of replicas
711       // that could host the property.
712       for (Map.Entry<String, List<SliceReplica>> entProp : nodesHostingProp.entrySet()) {
713         for (SliceReplica srHosting : entProp.getValue()) {
714           removeSliceAlreadyHostedFromPossibles(srHosting.slice.getName());
715         }
716       }
717 
718       // Assemble the list of slices that do not have any replica hosting the property:
719       for (Map.Entry<String, List<SliceReplica>> ent : nodesHostingReplicas.entrySet()) {
720         ListIterator<SliceReplica> iter = ent.getValue().listIterator();
721         while (iter.hasNext()) {
722           SliceReplica sr = iter.next();
723           shardsNeedingHosts.add(sr.slice.getName());
724         }
725       }
726 
727       // At this point, nodesHostingProp should contain _only_ lists of replicas that belong to slices that do _not_
728       // have any replica hosting the property. So let's assign them.
729 
730       balanceUnassignedReplicas();
731       for (Slice newSlice : changedSlices.values()) {
732         DocCollection docCollection = CollectionMutator.updateSlice(collectionName, clusterState.getCollection(collectionName), newSlice);
733         clusterState = ClusterStateMutator.newState(clusterState, collectionName, docCollection);
734       }
735       return true;
736     }

822     updaterThread = new OverseerThread(tg, new ClusterStateUpdater(reader, id, stats), "OverseerStateUpdate-" + id);
823     updaterThread.setDaemon(true);

825     ThreadGroup ccTg = new ThreadGroup("Overseer collection creation process.");

828     overseerCollectionConfigSetProcessor = new OverseerCollectionConfigSetProcessor(reader, id, shardHandler, adminPath, stats, Overseer.this, overseerPrioritizer);
829     ccThread = new OverseerThread(ccTg, overseerCollectionConfigSetProcessor, "OverseerCollectionConfigSetProcessor-" + id);

836     arfoThread.setDaemon(true);
837     
838     updaterThread.start();
839     ccThread.start();
840     arfoThread.start();
841   }

869   private void doClose() {
870     
871     if (updaterThread != null) {
872       IOUtils.closeQuietly(updaterThread);
873       updaterThread.interrupt();
874     }
875     if (ccThread != null) {
876       IOUtils.closeQuietly(ccThread);
877       ccThread.interrupt();
878     }
879     if (arfoThread != null) {
880       IOUtils.closeQuietly(arfoThread);
881       arfoThread.interrupt();
882     }
883     
884     updaterThread = null;
885     ccThread = null;
886     arfoThread = null;
887   }

979     private volatile int queueLength;

986       Stat stat = stats.get(operation.toLowerCase(Locale.ROOT));
987       return stat == null ? 0 : stat.success.get();
988     }

991       Stat stat = stats.get(operation.toLowerCase(Locale.ROOT));
992       return stat == null ? 0 : stat.errors.get();
993     }

995     public void success(String operation) {
996       String op = operation.toLowerCase(Locale.ROOT);
997       Stat stat = stats.get(op);
998       if (stat == null) {
999         stat = new Stat();
1000         stats.put(op, stat);
1001       }
1002       stat.success.incrementAndGet();
1003     }

1005     public void error(String operation) {
1006       String op = operation.toLowerCase(Locale.ROOT);
1007       Stat stat = stats.get(op);
1008       if (stat == null) {
1009         stat = new Stat();
1010         stats.put(op, stat);
1011       }
1012       stat.errors.incrementAndGet();
1013     }

1016       String op = operation.toLowerCase(Locale.ROOT);
1017       Stat stat = stats.get(op);
1018       if (stat == null) {
1019         stat = new Stat();
1020         stats.put(op, stat);
1021       }
1022       return stat.requestTime.time();
1023     }

1025     public void storeFailureDetails(String operation, ZkNodeProps request, SolrResponse resp) {
1026       String op = operation.toLowerCase(Locale.ROOT);
1027       Stat stat = stats.get(op);
1028       if (stat == null) {
1029         stat = new Stat();
1030         stats.put(op, stat);
1031       }
1032       LinkedList<FailedOp> failedOps = stat.failureDetails;
1033       synchronized (failedOps)  {
1034         if (failedOps.size() >= MAX_STORED_FAILURES)  {
1035           failedOps.removeFirst();
1036         }
1037         failedOps.addLast(new FailedOp(request, resp));
1038       }
1039     }
1040 
1041     public List<FailedOp> getFailureDetails(String operation) {
1042       Stat stat = stats.get(operation.toLowerCase(Locale.ROOT));
1043       if (stat == null || stat.failureDetails.isEmpty()) return null;
1044       LinkedList<FailedOp> failedOps = stat.failureDetails;
1045       synchronized (failedOps)  {
1046         ArrayList<FailedOp> ret = new ArrayList<>(failedOps);
1047         return ret;
1048       }
1049     }
