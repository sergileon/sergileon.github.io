42  public class TestStressNRT extends LuceneTestCase {
43    volatile DirectoryReader reader;
44  
45    final ConcurrentHashMap<Integer,Long> model = new ConcurrentHashMap<>();
46    Map<Integer,Long> committedModel = new HashMap<>();
47    long snapshotCount;
48    long committedModelClock;
49    volatile int lastId;
50    final String field = "val_l";
51    Object[] syncArr;
52  
53    private void initModel(int ndocs) {
54      snapshotCount = 0;
55      committedModelClock = 0;
56      lastId = 0;
57  
58      syncArr = new Object[ndocs];
59  
60      for (int i=0; i<ndocs; i++) {
61        model.put(i, -1L);
62        syncArr[i] = new Object();
63      }
64      committedModel.putAll(model);
65    }
66  
67    public void test() throws Exception {
68      // update variables
69      final int commitPercent = random().nextInt(20);
70      final int softCommitPercent = random().nextInt(100); // what percent of the commits are soft
71      final int deletePercent = random().nextInt(50);
72      final int deleteByQueryPercent = random().nextInt(25);
73      final int ndocs = atLeast(50);
74      final int nWriteThreads = TestUtil.nextInt(random(), 1, TEST_NIGHTLY ? 10 : 5);
75      final int maxConcurrentCommits = TestUtil.nextInt(random(), 1, TEST_NIGHTLY ? 10 : 5);   // number of committers at a time... needed if we want to avoid commit errors due to exceeding the max
76      
77      final boolean tombstones = random().nextBoolean();
78  
79      // query variables
80      final AtomicLong operations = new AtomicLong(atLeast(10000));  // number of query operations to perform in total
81  
82      final int nReadThreads = TestUtil.nextInt(random(), 1, TEST_NIGHTLY ? 10 : 5);
83      initModel(ndocs);
84  
85      final FieldType storedOnlyType = new FieldType();
86      storedOnlyType.setStored(true);
87  
88      if (VERBOSE) {
89        System.out.println("\n");
90        System.out.println("TEST: commitPercent=" + commitPercent);
91        System.out.println("TEST: softCommitPercent=" + softCommitPercent);
92        System.out.println("TEST: deletePercent=" + deletePercent);
93        System.out.println("TEST: deleteByQueryPercent=" + deleteByQueryPercent);
94        System.out.println("TEST: ndocs=" + ndocs);
95        System.out.println("TEST: nWriteThreads=" + nWriteThreads);
96        System.out.println("TEST: nReadThreads=" + nReadThreads);
97        System.out.println("TEST: maxConcurrentCommits=" + maxConcurrentCommits);
98        System.out.println("TEST: tombstones=" + tombstones);
99        System.out.println("TEST: operations=" + operations);
100       System.out.println("\n");
101     }
102 
103     final AtomicInteger numCommitting = new AtomicInteger();
104 
105     List<Thread> threads = new ArrayList<>();
106 
107     Directory dir = newDirectory();
108 
109     final RandomIndexWriter writer = new RandomIndexWriter(random(), dir, newIndexWriterConfig(new MockAnalyzer(random())));
110     writer.setDoRandomForceMergeAssert(false);
111     writer.commit();
112     reader = DirectoryReader.open(dir);
113 
114     for (int i=0; i<nWriteThreads; i++) {
115       Thread thread = new Thread("WRITER"+i) {
116         Random rand = new Random(random().nextInt());
117 
118         @Override
119         public void run() {
120           try {
121             while (operations.get() > 0) {
122               int oper = rand.nextInt(100);
123 
124               if (oper < commitPercent) {
125                 if (numCommitting.incrementAndGet() <= maxConcurrentCommits) {
126                   Map<Integer,Long> newCommittedModel;
127                   long version;
128                   DirectoryReader oldReader;
129 
130                   synchronized(TestStressNRT.this) {
131                     newCommittedModel = new HashMap<>(model);  // take a snapshot
132                     version = snapshotCount++;
133                     oldReader = reader;
134                     oldReader.incRef();  // increment the reference since we will use this for reopening
135                   }
136 
137                   DirectoryReader newReader;
138                   if (rand.nextInt(100) < softCommitPercent) {
139                     // assertU(h.commit("softCommit","true"));
140                     if (random().nextBoolean()) {
141                       if (VERBOSE) {
142                         System.out.println("TEST: " + Thread.currentThread().getName() + ": call writer.getReader");
143                       }
144                       newReader = writer.getReader(true);
145                     } else {
146                       if (VERBOSE) {
147                         System.out.println("TEST: " + Thread.currentThread().getName() + ": reopen reader=" + oldReader + " version=" + version);
148                       }
149                       newReader = DirectoryReader.openIfChanged(oldReader, writer.w, true);
150                     }
151                   } else {
152                     // assertU(commit());
153                     if (VERBOSE) {
154                       System.out.println("TEST: " + Thread.currentThread().getName() + ": commit+reopen reader=" + oldReader + " version=" + version);
155                     }
156                     writer.commit();
157                     if (VERBOSE) {
158                       System.out.println("TEST: " + Thread.currentThread().getName() + ": now reopen after commit");
159                     }
160                     newReader = DirectoryReader.openIfChanged(oldReader);
161                   }
162 
163                   // Code below assumes newReader comes w/
164                   // extra ref:
165                   if (newReader == null) {
166                     oldReader.incRef();
167                     newReader = oldReader;
168                   }
169 
170                   oldReader.decRef();
171 
172                   synchronized(TestStressNRT.this) {
173                     // install the new reader if it's newest (and check the current version since another reader may have already been installed)
174                     //System.out.println(Thread.currentThread().getName() + ": newVersion=" + newReader.getVersion());
175                     assert newReader.getRefCount() > 0;
176                     assert reader.getRefCount() > 0;
177                     if (newReader.getVersion() > reader.getVersion()) {
178                       if (VERBOSE) {
179                         System.out.println("TEST: " + Thread.currentThread().getName() + ": install new reader=" + newReader);
180                       }
181                       reader.decRef();
182                       reader = newReader;
183 
184                       // Silly: forces fieldInfos to be
185                       // loaded so we don't hit IOE on later
186                       // reader.toString
187                       newReader.toString();
188 
189                       // install this snapshot only if it's newer than the current one
190                       if (version >= committedModelClock) {
191                         if (VERBOSE) {
192                           System.out.println("TEST: " + Thread.currentThread().getName() + ": install new model version=" + version);
193                         }
194                         committedModel = newCommittedModel;
195                         committedModelClock = version;
196                       } else {
197                         if (VERBOSE) {
198                           System.out.println("TEST: " + Thread.currentThread().getName() + ": skip install new model version=" + version);
199                         }
200                       }
201                     } else {
202                       // if the same reader, don't decRef.
203                       if (VERBOSE) {
204                         System.out.println("TEST: " + Thread.currentThread().getName() + ": skip install new reader=" + newReader);
205                       }
206                       newReader.decRef();
207                     }
208                   }
209                 }
210                 numCommitting.decrementAndGet();
211               } else {
212 
213                 int id = rand.nextInt(ndocs);
214                 Object sync = syncArr[id];
215 
216                 // set the lastId before we actually change it sometimes to try and
217                 // uncover more race conditions between writing and reading
218                 boolean before = random().nextBoolean();
219                 if (before) {
220                   lastId = id;
221                 }
222 
223                 // We can't concurrently update the same document and retain our invariants of increasing values
224                 // since we can't guarantee what order the updates will be executed.
225                 synchronized (sync) {
226                   Long val = model.get(id);
227                   long nextVal = Math.abs(val)+1;
228 
229                   if (oper < commitPercent + deletePercent) {
230                     // assertU("<delete><id>" + id + "</id></delete>");
231 
232                     // add tombstone first
233                     if (tombstones) {
234                       Document d = new Document();
235                       d.add(newStringField("id", "-"+Integer.toString(id), Field.Store.YES));
236                       d.add(newField(field, Long.toString(nextVal), storedOnlyType));
237                       writer.updateDocument(new Term("id", "-"+Integer.toString(id)), d);
238                     }
239 
240                     if (VERBOSE) {
241                       System.out.println("TEST: " + Thread.currentThread().getName() + ": term delDocs id:" + id + " nextVal=" + nextVal);
242                     }
243                     writer.deleteDocuments(new Term("id",Integer.toString(id)));
244                     model.put(id, -nextVal);
245                   } else if (oper < commitPercent + deletePercent + deleteByQueryPercent) {
246                     //assertU("<delete><query>id:" + id + "</query></delete>");
247 
248                     // add tombstone first
249                     if (tombstones) {
250                       Document d = new Document();
251                       d.add(newStringField("id", "-"+Integer.toString(id), Field.Store.YES));
252                       d.add(newField(field, Long.toString(nextVal), storedOnlyType));
253                       writer.updateDocument(new Term("id", "-"+Integer.toString(id)), d);
254                     }
255 
256                     if (VERBOSE) {
257                       System.out.println("TEST: " + Thread.currentThread().getName() + ": query delDocs id:" + id + " nextVal=" + nextVal);
258                     }
259                     writer.deleteDocuments(new TermQuery(new Term("id", Integer.toString(id))));
260                     model.put(id, -nextVal);
261                   } else {
262                     // assertU(adoc("id",Integer.toString(id), field, Long.toString(nextVal)));
263                     Document d = new Document();
264                     d.add(newStringField("id", Integer.toString(id), Field.Store.YES));
265                     d.add(newField(field, Long.toString(nextVal), storedOnlyType));
266                     if (VERBOSE) {
267                       System.out.println("TEST: " + Thread.currentThread().getName() + ": u id:" + id + " val=" + nextVal);
268                     }
269                     writer.updateDocument(new Term("id", Integer.toString(id)), d);
270                     if (tombstones) {
271                       // remove tombstone after new addition (this should be optional?)
272                       writer.deleteDocuments(new Term("id","-"+Integer.toString(id)));
273                     }
274                     model.put(id, nextVal);
275                   }
276                 }
277 
278                 if (!before) {
279                   lastId = id;
280                 }
281               }
282             }
283           } catch (Throwable e) {
284             System.out.println(Thread.currentThread().getName() + ": FAILED: unexpected exception");
285             e.printStackTrace(System.out);
286             throw new RuntimeException(e);
287           }
288         }
289       };
290 
291       threads.add(thread);
292     }
293 
294     for (int i=0; i<nReadThreads; i++) {
295       Thread thread = new Thread("READER"+i) {
296         Random rand = new Random(random().nextInt());
297 
298         @Override
299         public void run() {
300           try {
301             IndexReader lastReader = null;
302             IndexSearcher lastSearcher = null;
303 
304             while (operations.decrementAndGet() >= 0) {
305               // bias toward a recently changed doc
306               int id = rand.nextInt(100) < 25 ? lastId : rand.nextInt(ndocs);
307 
308               // when indexing, we update the index, then the model
309               // so when querying, we should first check the model, and then the index
310 
311               long val;
312               DirectoryReader r;
313               synchronized(TestStressNRT.this) {
314                 val = committedModel.get(id);
315                 r = reader;
316                 r.incRef();
317               }
318 
319               if (VERBOSE) {
320                 System.out.println("TEST: " + Thread.currentThread().getName() + ": s id=" + id + " val=" + val + " r=" + r.getVersion());
321               }
322 
323               //  sreq = req("wt","json", "q","id:"+Integer.toString(id), "omitHeader","true");
324               IndexSearcher searcher;
325               if (r == lastReader) {
326                 // Just re-use lastSearcher, else
327                 // newSearcher may create too many thread
328                 // pools (ExecutorService):
329                 searcher = lastSearcher;
330               } else {
331                 searcher = newSearcher(r);
332                 lastReader = r;
333                 lastSearcher = searcher;
334               }
335               Query q = new TermQuery(new Term("id",Integer.toString(id)));
336               TopDocs results = searcher.search(q, 10);
337 
338               if (results.totalHits == 0 && tombstones) {
339                 // if we couldn't find the doc, look for its tombstone
340                 q = new TermQuery(new Term("id","-"+Integer.toString(id)));
341                 results = searcher.search(q, 1);
342                 if (results.totalHits == 0) {
343                   if (val == -1L) {
344                     // expected... no doc was added yet
345                     r.decRef();
346                     continue;
347                   }
348                   fail("No documents or tombstones found for id " + id + ", expected at least " + val + " reader=" + r);
349                 }
350               }
351 
352               if (results.totalHits == 0 && !tombstones) {
353                 // nothing to do - we can't tell anything from a deleted doc without tombstones
354               } else {
355                 // we should have found the document, or its tombstone
356                 if (results.totalHits != 1) {
357                   System.out.println("FAIL: hits id:" + id + " val=" + val);
358                   for(ScoreDoc sd : results.scoreDocs) {
359                     final Document doc = r.document(sd.doc);
360                     System.out.println("  docID=" + sd.doc + " id:" + doc.get("id") + " foundVal=" + doc.get(field));
361                   }
362                   fail("id=" + id + " reader=" + r + " totalHits=" + results.totalHits);
363                 }
364                 Document doc = searcher.doc(results.scoreDocs[0].doc);
365                 long foundVal = Long.parseLong(doc.get(field));
366                 if (foundVal < Math.abs(val)) {
367                   fail("foundVal=" + foundVal + " val=" + val + " id=" + id + " reader=" + r);
368                 }
369               }
370 
371               r.decRef();
372             }
373           } catch (Throwable e) {
374             operations.set(-1L);
375             System.out.println(Thread.currentThread().getName() + ": FAILED: unexpected exception");
376             e.printStackTrace(System.out);
377             throw new RuntimeException(e);
378           }
379         }
380       };
381 
382       threads.add(thread);
383     }
384 
385     for (Thread thread : threads) {
386       thread.start();
387     }
388 
389     for (Thread thread : threads) {
390       thread.join();
391     }
392 
393     writer.close();
394     if (VERBOSE) {
395       System.out.println("TEST: close reader=" + reader);
396     }
397     reader.close();
398     dir.close();
399   }
