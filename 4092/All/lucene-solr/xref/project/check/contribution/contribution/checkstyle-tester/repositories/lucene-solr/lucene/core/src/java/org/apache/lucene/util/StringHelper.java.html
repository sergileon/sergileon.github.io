34    /**
35     * Compares two {@link BytesRef}, element by element, and returns the
36     * number of elements common to both arrays (from the start of each).
37     *
38     * @param left The first {@link BytesRef} to compare
39     * @param right The second {@link BytesRef} to compare
40     * @return The number of common elements (from the start of each).
41     */
42    public static int bytesDifference(BytesRef left, BytesRef right) {
43      int len = left.length < right.length ? left.length : right.length;
44      final byte[] bytesLeft = left.bytes;
45      final int offLeft = left.offset;
46      byte[] bytesRight = right.bytes;
47      final int offRight = right.offset;
48      for (int i = 0; i < len; i++)
49        if (bytesLeft[i+offLeft] != bytesRight[i+offRight])
50          return i;
51      return len;
52    }

54    /** 
55     * Returns the length of {@code currentTerm} needed for use as a sort key.
56     * so that {@link BytesRef#compareTo(BytesRef)} still returns the same result.
57     * This method assumes currentTerm comes after priorTerm.
58     */
59    public static int sortKeyLength(final BytesRef priorTerm, final BytesRef currentTerm) {
60      final int currentTermOffset = currentTerm.offset;
61      final int priorTermOffset = priorTerm.offset;
62      final int limit = Math.min(priorTerm.length, currentTerm.length);
63      for (int i = 0; i < limit; i++) {
64        if (priorTerm.bytes[priorTermOffset+i] != currentTerm.bytes[currentTermOffset+i]) {
65          return i+1;
66        }
67      }
68      return Math.min(1+priorTerm.length, currentTerm.length);
69    }

74    public static boolean equals(String s1, String s2) {
75      if (s1 == null) {
76        return s2 == null;
77      } else {
78        return s1.equals(s2);
79      }

82    /**
83     * Returns <code>true</code> iff the ref starts with the given prefix.
84     * Otherwise <code>false</code>.
85     * 
86     * @param ref
87     *         the {@code byte[]} to test
88     * @param prefix
89     *         the expected prefix
90     * @return Returns <code>true</code> iff the ref starts with the given prefix.
91     *         Otherwise <code>false</code>.
92     */
93    public static boolean startsWith(byte[] ref, BytesRef prefix) {
94      if (ref.length < prefix.length) {
95        return false;
96      }
97  
98      for(int i=0;i<prefix.length;i++) {
99        if (ref[i] != prefix.bytes[prefix.offset+i]) {
100         return false;
101       }
102     }
103 
104     return true;
105   }

107   /**
108    * Returns <code>true</code> iff the ref starts with the given prefix.
109    * Otherwise <code>false</code>.
110    * 
111    * @param ref
112    *          the {@link BytesRef} to test
113    * @param prefix
114    *          the expected prefix
115    * @return Returns <code>true</code> iff the ref starts with the given prefix.
116    *         Otherwise <code>false</code>.
117    */
118   public static boolean startsWith(BytesRef ref, BytesRef prefix) {
119     return sliceEquals(ref, prefix, 0);
120   }
121 
122   /**
123    * Returns <code>true</code> iff the ref ends with the given suffix. Otherwise
124    * <code>false</code>.
125    * 
126    * @param ref
127    *          the {@link BytesRef} to test
128    * @param suffix
129    *          the expected suffix
130    * @return Returns <code>true</code> iff the ref ends with the given suffix.
131    *         Otherwise <code>false</code>.
132    */
133   public static boolean endsWith(BytesRef ref, BytesRef suffix) {
134     return sliceEquals(ref, suffix, ref.length - suffix.length);
135   }

137   private static boolean sliceEquals(BytesRef sliceToTest, BytesRef other, int pos) {
138     if (pos < 0 || sliceToTest.length - pos < other.length) {
139       return false;
140     }
141     int i = sliceToTest.offset + pos;
142     int j = other.offset;
143     final int k = other.offset + other.length;
144     
145     while (j < k) {
146       if (sliceToTest.bytes[i++] != other.bytes[j++]) {
147         return false;
148       }
149     }
150     
151     return true;
152   }

156   // Poached from Guava: set a different salt/seed
157   // for each JVM instance, to frustrate hash key collision
158   // denial of service attacks, and to catch any places that
159   // somehow rely on hash function/order across JVM
160   // instances:
161   public static final int GOOD_FAST_HASH_SEED;
162 
163   static {
164     String prop = System.getProperty("tests.seed");
165     if (prop != null) {
166       // So if there is a test failure that relied on hash
167       // order, we remain reproducible based on the test seed:
168       GOOD_FAST_HASH_SEED = prop.hashCode();
169     } else {
170       GOOD_FAST_HASH_SEED = (int) System.currentTimeMillis();
171     }
172   }

174   /** Returns the MurmurHash3_x86_32 hash.
175    * Original source/tests at https://github.com/yonik/java_util/
176    */
177   @SuppressWarnings("fallthrough")
178   public static int murmurhash3_x86_32(byte[] data, int offset, int len, int seed) {
179 
180     final int c1 = 0xcc9e2d51;
181     final int c2 = 0x1b873593;
182 
183     int h1 = seed;
184     int roundedEnd = offset + (len & 0xfffffffc);  // round down to 4 byte block
185 
186     for (int i=offset; i<roundedEnd; i+=4) {
187       // little endian load order
188       int k1 = (data[i] & 0xff) | ((data[i+1] & 0xff) << 8) | ((data[i+2] & 0xff) << 16) | (data[i+3] << 24);
189       k1 *= c1;
190       k1 = Integer.rotateLeft(k1, 15);
191       k1 *= c2;
192 
193       h1 ^= k1;
194       h1 = Integer.rotateLeft(h1, 13);
195       h1 = h1*5+0xe6546b64;
196     }
197 
198     // tail
199     int k1 = 0;
200 
201     switch(len & 0x03) {
202       case 3:
203         k1 = (data[roundedEnd + 2] & 0xff) << 16;
204         // fallthrough
205       case 2:
206         k1 |= (data[roundedEnd + 1] & 0xff) << 8;
207         // fallthrough
208       case 1:
209         k1 |= (data[roundedEnd] & 0xff);
210         k1 *= c1;
211         k1 = Integer.rotateLeft(k1, 15);
212         k1 *= c2;
213         h1 ^= k1;
214     }
215 
216     // finalization
217     h1 ^= len;
218 
219     // fmix(h1);
220     h1 ^= h1 >>> 16;
221     h1 *= 0x85ebca6b;
222     h1 ^= h1 >>> 13;
223     h1 *= 0xc2b2ae35;
224     h1 ^= h1 >>> 16;
225 
226     return h1;
227   }

229   public static int murmurhash3_x86_32(BytesRef bytes, int seed) {
230     return murmurhash3_x86_32(bytes.bytes, bytes.offset, bytes.length, seed);
231   }

233   // Holds 128 bit unsigned value:
234   private static BigInteger nextId;
235   private static final BigInteger mask128;
236   private static final Object idLock = new Object();
237 
238   static {
239     // 128 bit unsigned mask
240     byte[] maskBytes128 = new byte[16];
241     Arrays.fill(maskBytes128, (byte) 0xff);
242     mask128 = new BigInteger(1, maskBytes128);
243     
244     String prop = System.getProperty("tests.seed");
245 
246     // State for xorshift128:
247     long x0;
248     long x1;
249 
250     if (prop != null) {
251       // So if there is a test failure that somehow relied on this id,
252       // we remain reproducible based on the test seed:
253       if (prop.length() > 8) {
254         prop = prop.substring(prop.length()-8);
255       }
256       x0 = Long.parseLong(prop, 16);
257       x1 = x0;
258     } else {
259       // seed from /dev/urandom, if its available
260       try (DataInputStream is = new DataInputStream(Files.newInputStream(Paths.get("/dev/urandom")))) {
261         x0 = is.readLong();
262         x1 = is.readLong();
263       } catch (Exception unavailable) {
264         // may not be available on this platform
265         // fall back to lower quality randomness from 3 different sources:
266         x0 = System.nanoTime();
267         x1 = StringHelper.class.hashCode() << 32;
268         
269         StringBuilder sb = new StringBuilder();
270         // Properties can vary across JVM instances:
271         try {
272           Properties p = System.getProperties();
273           for (String s: p.stringPropertyNames()) {
274             sb.append(s);
275             sb.append(p.getProperty(s));
276           }
277           x1 |= sb.toString().hashCode();
278         } catch (SecurityException notallowed) {
279           // getting Properties requires wildcard read-write: may not be allowed
280           x1 |= StringBuffer.class.hashCode();
281         }
282       }
283     }
284 
285     // Use a few iterations of xorshift128 to scatter the seed
286     // in case multiple Lucene instances starting up "near" the same
287     // nanoTime, since we use ++ (mod 2^128) for full period cycle:
288     for(int i=0;i<10;i++) {
289       long s1 = x0;
290       long s0 = x1;
291       x0 = s0;
292       s1 ^= s1 << 23; // a
293       x1 = s1 ^ s0 ^ (s1 >>> 17) ^ (s0 >>> 26); // b, c
294     }
295     
296     // 64-bit unsigned mask
297     byte[] maskBytes64 = new byte[8];
298     Arrays.fill(maskBytes64, (byte) 0xff);
299     BigInteger mask64 = new BigInteger(1, maskBytes64);
300 
301     // First make unsigned versions of x0, x1:
302     BigInteger unsignedX0 = BigInteger.valueOf(x0).and(mask64);
303     BigInteger unsignedX1 = BigInteger.valueOf(x1).and(mask64);
304 
305     // Concatentate bits of x0 and x1, as unsigned 128 bit integer:
306     nextId = unsignedX0.shiftLeft(64).or(unsignedX1);
307   }

309   /** length in bytes of an ID */
310   public static final int ID_LENGTH = 16;
311 
312   /** Generates a non-cryptographic globally unique id. */
313   public static byte[] randomId() {
314 
315     // NOTE: we don't use Java's UUID.randomUUID() implementation here because:
316     //
317     //   * It's overkill for our usage: it tries to be cryptographically
318     //     secure, whereas for this use we don't care if someone can
319     //     guess the IDs.
320     //
321     //   * It uses SecureRandom, which on Linux can easily take a long time
322     //     (I saw ~ 10 seconds just running a Lucene test) when entropy
323     //     harvesting is falling behind.
324     //
325     //   * It loses a few (6) bits to version and variant and it's not clear
326     //     what impact that has on the period, whereas the simple ++ (mod 2^128)
327     //     we use here is guaranteed to have the full period.
328 
329     byte bits[];
330     synchronized(idLock) {
331       bits = nextId.toByteArray();
332       nextId = nextId.add(BigInteger.ONE).and(mask128);
333     }
334     
335     // toByteArray() always returns a sign bit, so it may require an extra byte (always zero)
336     if (bits.length > ID_LENGTH) {
337       assert bits.length == ID_LENGTH + 1;
338       assert bits[0] == 0;
339       return Arrays.copyOfRange(bits, 1, bits.length);
340     } else {
341       byte[] result = new byte[ID_LENGTH];
342       System.arraycopy(bits, 0, result, result.length - bits.length, bits.length);
343       return result;
344     }
345   }

347   /** 
348    * Helper method to render an ID as a string, for debugging
349    * <p>
350    * Returns the string {@code (null)} if the id is null.
351    * Otherwise, returns a string representation for debugging.
352    * Never throws an exception. The returned string may
353    * indicate if the id is definitely invalid.
354    */
355   public static String idToString(byte id[]) {
356     if (id == null) {
357       return "(null)";
358     } else {
359       StringBuilder sb = new StringBuilder();
360       sb.append(new BigInteger(1, id).toString(Character.MAX_RADIX));
361       if (id.length != ID_LENGTH) {
362         sb.append(" (INVALID FORMAT)");
363       }
364       return sb.toString();
365     }
366   }

368   /** Just converts each int in the incoming {@link IntsRef} to each byte
369    *  in the returned {@link BytesRef}, throwing {@code IllegalArgumentException}
370    *  if any int value is out of bounds for a byte. */
371   public static BytesRef intsRefToBytesRef(IntsRef ints) {
372     byte[] bytes = new byte[ints.length];
373     for(int i=0;i<ints.length;i++) {
374       int x = ints.ints[ints.offset+i];
375       if (x < 0 || x > 255) {
376         throw new IllegalArgumentException("int at pos=" + i + " with value=" + x + " is out-of-bounds for byte");
377       }
378       bytes[i] = (byte) x;
379     }
380 
381     return new BytesRef(bytes);
382   }
