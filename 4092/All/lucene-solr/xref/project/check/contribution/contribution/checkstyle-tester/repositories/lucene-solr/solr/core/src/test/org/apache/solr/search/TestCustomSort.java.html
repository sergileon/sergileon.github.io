20  import org.apache.solr.SolrTestCaseJ4;
21  import org.junit.BeforeClass;
22  
23  import java.nio.ByteBuffer;
24  
25  
26  /**
27   * Test SortField.CUSTOM sorts
28   */
29  public class TestCustomSort extends SolrTestCaseJ4 {
30  
31    @BeforeClass
32    public static void beforeClass() throws Exception {
33      initCore("solrconfig.xml", "schema-custom-field.xml");
34    }
35    
36    public void testSortableBinary() throws Exception {
37      clearIndex();
38      assertU(adoc(sdoc("id", "1", "text", "a", "payload", ByteBuffer.wrap(new byte[] { 0x12, 0x62, 0x15 }))));                    //  2
39      assertU(adoc(sdoc("id", "2", "text", "b", "payload", ByteBuffer.wrap(new byte[] { 0x25, 0x21, 0x16 }))));                    //  5
40      assertU(adoc(sdoc("id", "3", "text", "a", "payload", ByteBuffer.wrap(new byte[] { 0x35, 0x32, 0x58 }))));                    //  8
41      assertU(adoc(sdoc("id", "4", "text", "b", "payload", ByteBuffer.wrap(new byte[] { 0x25, 0x21, 0x15 }))));                    //  4
42      assertU(adoc(sdoc("id", "5", "text", "a", "payload", ByteBuffer.wrap(new byte[] { 0x35, 0x35, 0x10, 0x00 }))));              //  9
43      assertU(adoc(sdoc("id", "6", "text", "c", "payload", ByteBuffer.wrap(new byte[] { 0x1a, 0x2b, 0x3c, 0x00, 0x00, 0x03 }))));  //  3
44      assertU(adoc(sdoc("id", "7", "text", "c", "payload", ByteBuffer.wrap(new byte[] { 0x00, 0x3c, 0x73 }))));                    //  1
45      assertU(adoc(sdoc("id", "8", "text", "c", "payload", ByteBuffer.wrap(new byte[] { 0x59, 0x2d, 0x4d }))));                    // 11
46      assertU(adoc(sdoc("id", "9", "text", "a", "payload", ByteBuffer.wrap(new byte[] { 0x39, 0x79, 0x7a }))));                    // 10
47      assertU(adoc(sdoc("id", "10", "text", "b", "payload", ByteBuffer.wrap(new byte[] { 0x31, 0x39, 0x7c }))));                   //  6
48      assertU(adoc(sdoc("id", "11", "text", "d", "payload", ByteBuffer.wrap(new byte[] { (byte)0xff, (byte)0xaf, (byte)0x9c })))); // 13
49      assertU(adoc(sdoc("id", "12", "text", "d", "payload", ByteBuffer.wrap(new byte[] { 0x34, (byte)0xdd, 0x4d }))));             //  7
50      assertU(adoc(sdoc("id", "13", "text", "d", "payload", ByteBuffer.wrap(new byte[] { (byte)0x80, 0x11, 0x33 }))));             // 12
51      assertU(commit());
52      
53      assertQ(req("q", "*:*", "fl", "id", "sort", "payload asc", "rows", "20")
54          , "//result[@numFound='13']"                   // <result name="response" numFound="13" start="0">
55          , "//result/doc[int='7'  and position()=1]"    //   <doc><int name="id">7</int></doc>   00 3c 73
56          , "//result/doc[int='1'  and position()=2]"    //   <doc><int name="id">1</int></doc>   12 62 15
57          , "//result/doc[int='6'  and position()=3]"    //   <doc><int name="id">6</int></doc>   1a 2b 3c 00 00 03
58          , "//result/doc[int='4'  and position()=4]"    //   <doc><int name="id">4</int></doc>   25 21 15
59          , "//result/doc[int='2'  and position()=5]"    //   <doc><int name="id">2</int></doc>   25 21 16
60          , "//result/doc[int='10' and position()=6]"    //   <doc><int name="id">10</int></doc>  31 39 7c
61          , "//result/doc[int='12' and position()=7]"    //   <doc><int name="id">12</int></doc>  34 dd 4d
62          , "//result/doc[int='3'  and position()=8]"    //   <doc><int name="id">3</int></doc>   35 32 58
63          , "//result/doc[int='5'  and position()=9]"    //   <doc><int name="id">5</int></doc>   35 35 10 00
64          , "//result/doc[int='9'  and position()=10]"   //   <doc><int name="id">9</int></doc>   39 79 7a
65          , "//result/doc[int='8'  and position()=11]"   //   <doc><int name="id">8</int></doc>   59 2d 4d      
66          , "//result/doc[int='13' and position()=12]"   //   <doc><int name="id">13</int></doc>  80 11 33       
67          , "//result/doc[int='11' and position()=13]"); //   <doc><int name="id">11</int></doc>  ff af 9c
68      assertQ(req("q", "*:*", "fl", "id", "sort", "payload desc", "rows", "20")
69          , "//result[@numFound='13']"                   // <result name="response" numFound="13" start="0">
70          , "//result/doc[int='11' and position()=1]"    //   <doc><int name="id">11</int></doc>  ff af 9c            
71          , "//result/doc[int='13' and position()=2]"    //   <doc><int name="id">13</int></doc>  80 11 33                   
72          , "//result/doc[int='8'  and position()=3]"    //   <doc><int name="id">8</int></doc>   59 2d 4d                  
73          , "//result/doc[int='9'  and position()=4]"    //   <doc><int name="id">9</int></doc>   39 79 7a            
74          , "//result/doc[int='5'  and position()=5]"    //   <doc><int name="id">5</int></doc>   35 35 10 00         
75          , "//result/doc[int='3'  and position()=6]"    //   <doc><int name="id">3</int></doc>   35 32 58            
76          , "//result/doc[int='12' and position()=7]"    //   <doc><int name="id">12</int></doc>  34 dd 4d            
77          , "//result/doc[int='10' and position()=8]"    //   <doc><int name="id">10</int></doc>  31 39 7c            
78          , "//result/doc[int='2'  and position()=9]"    //   <doc><int name="id">2</int></doc>   25 21 16            
79          , "//result/doc[int='4'  and position()=10]"   //   <doc><int name="id">4</int></doc>   25 21 15            
80          , "//result/doc[int='6'  and position()=11]"   //   <doc><int name="id">6</int></doc>   1a 2b 3c 00 00 03   
81          , "//result/doc[int='1'  and position()=12]"   //   <doc><int name="id">1</int></doc>   12 62 15            
82          , "//result/doc[int='7'  and position()=13]"); //   <doc><int name="id">7</int></doc>   00 3c 73            
83      assertQ(req("q", "text:a", "fl", "id", "sort", "payload asc", "rows", "20")
84          , "//result[@numFound='4']"                    // <result name="response" numFound="4" start="0">
85          , "//result/doc[int='1'  and position()=1]"    //   <doc><int name="id">1</int></doc>   12 62 15    
86          , "//result/doc[int='3'  and position()=2]"    //   <doc><int name="id">3</int></doc>   35 32 58    
87          , "//result/doc[int='5'  and position()=3]"    //   <doc><int name="id">5</int></doc>   35 35 10 00 
88          , "//result/doc[int='9'  and position()=4]");  //   <doc><int name="id">9</int></doc>   39 79 7a    
89      assertQ(req("q", "text:a", "fl", "id", "sort", "payload desc", "rows", "20")
90          , "//result[@numFound='4']"                    // <result name="response" numFound="4" start="0">
91          , "//result/doc[int='9'  and position()=1]"    //   <doc><int name="id">9</int></doc>   39 79 7a    
92          , "//result/doc[int='5'  and position()=2]"    //   <doc><int name="id">5</int></doc>   35 35 10 00 
93          , "//result/doc[int='3'  and position()=3]"    //   <doc><int name="id">3</int></doc>   35 32 58    
94          , "//result/doc[int='1'  and position()=4]");  //   <doc><int name="id">1</int></doc>   12 62 15    
95      assertQ(req("q", "text:b", "fl", "id", "sort", "payload asc", "rows", "20")
96          , "//result[@numFound='3']"                    // <result name="response" numFound="3" start="0">
97          , "//result/doc[int='4'  and position()=1]"    //   <doc><int name="id">4</int></doc>   25 21 15
98          , "//result/doc[int='2'  and position()=2]"    //   <doc><int name="id">2</int></doc>   25 21 16
99          , "//result/doc[int='10' and position()=3]");  //   <doc><int name="id">10</int></doc>  31 39 7c
100     assertQ(req("q", "text:b", "fl", "id", "sort", "payload desc", "rows", "20")
101         , "//result[@numFound='3']"                    // <result name="response" numFound="3" start="0">
102         , "//result/doc[int='10' and position()=1]"    //   <doc><int name="id">10</int></doc>  31 39 7c
103         , "//result/doc[int='2'  and position()=2]"    //   <doc><int name="id">2</int></doc>   25 21 16
104         , "//result/doc[int='4'  and position()=3]");  //   <doc><int name="id">4</int></doc>   25 21 15
105     assertQ(req("q", "text:c", "fl", "id", "sort", "payload asc", "rows", "20")
106         , "//result[@numFound='3']"                   // <result name="response" numFound="3" start="0">
107         , "//result/doc[int='7'  and position()=1]"   //   <doc><int name="id">7</int></doc>    00 3c 73         
108         , "//result/doc[int='6'  and position()=2]"   //   <doc><int name="id">6</int></doc>    1a 2b 3c 00 00 03
109         , "//result/doc[int='8'  and position()=3]"); //   <doc><int name="id">8</int></doc>    59 2d 4d              
110     assertQ(req("q", "text:c", "fl", "id", "sort", "payload desc", "rows", "20")
111         , "//result[@numFound='3']"                   // <result name="response" numFound="3" start="0">
112         , "//result/doc[int='8'  and position()=1]"   //   <doc><int name="id">8</int></doc>    59 2d 4d              
113         , "//result/doc[int='6'  and position()=2]"   //   <doc><int name="id">6</int></doc>    1a 2b 3c 00 00 03
114         , "//result/doc[int='7'  and position()=3]"); //   <doc><int name="id">7</int></doc>    00 3c 73         
115     assertQ(req("q", "text:d", "fl", "id", "sort", "payload asc", "rows", "20")
116         , "//result[@numFound='3']"                   // <result name="response" numFound="3" start="0">
117         , "//result/doc[int='12' and position()=1]"   //   <doc><int name="id">12</int></doc>   34 dd 4d
118         , "//result/doc[int='13' and position()=2]"   //   <doc><int name="id">13</int></doc>   80 11 33      
119         , "//result/doc[int='11' and position()=3]"); //   <doc><int name="id">11</int></doc>   ff af 9c
120     assertQ(req("q", "text:d", "fl", "id", "sort", "payload desc", "rows", "20")
121         , "//result[@numFound='3']"                   // <result name="response" numFound="3" start="0">
122         , "//result/doc[int='11' and position()=1]"   //   <doc><int name="id">11</int></doc>   ff af 9c
123         , "//result/doc[int='13' and position()=2]"   //   <doc><int name="id">13</int></doc>   80 11 33      
124         , "//result/doc[int='12' and position()=3]"); //   <doc><int name="id">12</int></doc>   34 dd 4d
125   }
