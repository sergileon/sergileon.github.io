61    static final class NormsSimilarity extends Similarity {
62      
63      private final Similarity in;
64      
65      public NormsSimilarity(Similarity in) {
66        this.in = in;
67      }
68      
69      @Override
70      public long computeNorm(FieldInvertState state) {
71        if (state.getName().equals(NORMS_FIELD)) {
72          return Float.floatToIntBits(state.getBoost());
73        } else {
74          return in.computeNorm(state);
75        }
76      }
77      
78      @Override
79      public SimWeight computeWeight(CollectionStatistics collectionStats, TermStatistics... termStats) {
80        return in.computeWeight(collectionStats, termStats);
81      }
82      
83      @Override
84      public SimScorer simScorer(SimWeight weight, LeafReaderContext context) throws IOException {
85        return in.simScorer(weight, context);
86      }
87      
88    }

90    static final class PositionsTokenStream extends TokenStream {
91      
92      private final CharTermAttribute term;
93      private final PayloadAttribute payload;
94      private final OffsetAttribute offset;
95      
96      private int pos, off;
97      
98      public PositionsTokenStream() {
99        term = addAttribute(CharTermAttribute.class);
100       payload = addAttribute(PayloadAttribute.class);
101       offset = addAttribute(OffsetAttribute.class);
102     }
103     
104     @Override
105     public boolean incrementToken() throws IOException {
106       if (pos == 0) {
107         return false;
108       }
109       
110       clearAttributes();
111       term.append(DOC_POSITIONS_TERM);
112       payload.setPayload(new BytesRef(Integer.toString(pos)));
113       offset.setOffset(off, off);
114       --pos;
115       ++off;
116       return true;
117     }
118     
119     void setId(int id) {
120       pos = id / 10 + 1;
121       off = 0;
122     }
123   }

155   private static Document doc(final int id, PositionsTokenStream positions) {
156     final Document doc = new Document();
157     doc.add(new StringField(ID_FIELD, Integer.toString(id), Store.YES));
158     doc.add(new StringField(DOCS_ENUM_FIELD, DOCS_ENUM_TERM, Store.NO));
159     positions.setId(id);
160     doc.add(new Field(DOC_POSITIONS_FIELD, positions, POSITIONS_TYPE));
161     doc.add(new NumericDocValuesField(NUMERIC_DV_FIELD, id));
162     TextField norms = new TextField(NORMS_FIELD, Integer.toString(id), Store.NO);
163     norms.setBoost(Float.intBitsToFloat(id));
164     doc.add(norms);
165     doc.add(new BinaryDocValuesField(BINARY_DV_FIELD, new BytesRef(Integer.toString(id))));
166     doc.add(new SortedDocValuesField(SORTED_DV_FIELD, new BytesRef(Integer.toString(id))));
167     doc.add(new SortedSetDocValuesField(SORTED_SET_DV_FIELD, new BytesRef(Integer.toString(id))));
168     doc.add(new SortedSetDocValuesField(SORTED_SET_DV_FIELD, new BytesRef(Integer.toString(id + 1))));
169     doc.add(new SortedNumericDocValuesField(SORTED_NUMERIC_DV_FIELD, id));
170     doc.add(new SortedNumericDocValuesField(SORTED_NUMERIC_DV_FIELD, id + 1));
171     doc.add(new Field(TERM_VECTORS_FIELD, Integer.toString(id), TERM_VECTORS_TYPE));
172     return doc;
173   }

175   /** Creates an unsorted index; subclasses then sort this index and open sortedReader. */
176   private static void createIndex(Directory dir, int numDocs, Random random) throws IOException {
177     List<Integer> ids = new ArrayList<>();
178     for (int i = 0; i < numDocs; i++) {
179       ids.add(Integer.valueOf(i * 10));
180     }
181     // shuffle them for indexing
182     Collections.shuffle(ids, random);
183     if (VERBOSE) {
184       System.out.println("Shuffled IDs for indexing: " + Arrays.toString(ids.toArray()));
185     }
186     
187     PositionsTokenStream positions = new PositionsTokenStream();
188     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random));
189     conf.setMaxBufferedDocs(4); // create some segments
190     conf.setSimilarity(new NormsSimilarity(conf.getSimilarity())); // for testing norms field
191     RandomIndexWriter writer = new RandomIndexWriter(random, dir, conf);
192     writer.setDoRandomForceMerge(false);
193     for (int id : ids) {
194       writer.addDocument(doc(id, positions));
195     }
196     // delete some documents
197     writer.commit();
198     for (Integer id : ids) {
199       if (random.nextDouble() < 0.2) {
200         if (VERBOSE) {
201           System.out.println("delete doc_id " + id);
202         }
203         writer.deleteDocuments(new Term(ID_FIELD, id.toString()));
204       }
205     }
206     writer.close();
207   }

236   @Test
237   public void testDocsAndPositionsEnum() throws Exception {
238     TermsEnum termsEnum = sortedReader.terms(DOC_POSITIONS_FIELD).iterator();
239     assertEquals(SeekStatus.FOUND, termsEnum.seekCeil(new BytesRef(DOC_POSITIONS_TERM)));
240     PostingsEnum sortedPositions = termsEnum.postings(null, PostingsEnum.ALL);
241     int doc;
242     
243     // test nextDoc()
244     while ((doc = sortedPositions.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
245       int freq = sortedPositions.freq();
246       assertEquals("incorrect freq for doc=" + doc, sortedValues[doc].intValue() / 10 + 1, freq);
247       for (int i = 0; i < freq; i++) {
248         assertEquals("incorrect position for doc=" + doc, i, sortedPositions.nextPosition());
249         assertEquals("incorrect startOffset for doc=" + doc, i, sortedPositions.startOffset());
250         assertEquals("incorrect endOffset for doc=" + doc, i, sortedPositions.endOffset());
251         assertEquals("incorrect payload for doc=" + doc, freq - i, Integer.parseInt(sortedPositions.getPayload().utf8ToString()));
252       }
253     }
254     
255     // test advance()
256     final PostingsEnum reuse = sortedPositions;
257     sortedPositions = termsEnum.postings(reuse, PostingsEnum.ALL);
258     if (sortedPositions instanceof SortingDocsEnum) {
259       assertTrue(((SortingDocsEnum) sortedPositions).reused(reuse)); // make sure reuse worked
260     }
261     doc = 0;
262     while ((doc = sortedPositions.advance(doc + TestUtil.nextInt(random(), 1, 5))) != DocIdSetIterator.NO_MORE_DOCS) {
263       int freq = sortedPositions.freq();
264       assertEquals("incorrect freq for doc=" + doc, sortedValues[doc].intValue() / 10 + 1, freq);
265       for (int i = 0; i < freq; i++) {
266         assertEquals("incorrect position for doc=" + doc, i, sortedPositions.nextPosition());
267         assertEquals("incorrect startOffset for doc=" + doc, i, sortedPositions.startOffset());
268         assertEquals("incorrect endOffset for doc=" + doc, i, sortedPositions.endOffset());
269         assertEquals("incorrect payload for doc=" + doc, freq - i, Integer.parseInt(sortedPositions.getPayload().utf8ToString()));
270       }
271     }
272   }

274   Bits randomLiveDocs(int maxDoc) {
275     if (rarely()) {
276       if (random().nextBoolean()) {
277         return null;
278       } else {
279         return new Bits.MatchNoBits(maxDoc);
280       }
281     }
282     final FixedBitSet bits = new FixedBitSet(maxDoc);
283     final int bitsSet = TestUtil.nextInt(random(), 1, maxDoc - 1);
284     for (int i = 0; i < bitsSet; ++i) {
285       while (true) {
286         final int index = random().nextInt(maxDoc);
287         if (!bits.get(index)) {
288           bits.set(index);
289           break;
290         }
291       }
292     }
293     return bits;
294   }

298     TermsEnum termsEnum = sortedReader.terms(DOCS_ENUM_FIELD).iterator();
299     assertEquals(SeekStatus.FOUND, termsEnum.seekCeil(new BytesRef(DOCS_ENUM_TERM)));
300     PostingsEnum docs = termsEnum.postings(null);

304     while ((doc = docs.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
305       assertEquals("incorrect value; doc " + doc, sortedValues[doc].intValue(), Integer.parseInt(sortedReader.document(doc).get(ID_FIELD)));
306     }
307 
308     PostingsEnum reuse = docs;
309     docs = termsEnum.postings(reuse);
310     if (docs instanceof SortingDocsEnum) {
311       assertTrue(((SortingDocsEnum) docs).reused(reuse)); // make sure reuse worked
312     }
313     doc = -1;
314     prev = -1;
315     while ((doc = docs.advance(doc + 1)) != DocIdSetIterator.NO_MORE_DOCS) {
316       assertEquals("incorrect value; doc " + doc, sortedValues[doc].intValue(), Integer.parseInt(sortedReader.document(doc).get(ID_FIELD)));
317     }
