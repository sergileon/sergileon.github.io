29  import static org.junit.internal.matchers.StringContains.containsString;

31  /**
32   * Verify that remote (proxied) queries return proper error messages
33   */
34  @Slow
35  public class RemoteQueryErrorTest extends AbstractFullDistribZkTestBase {
36  
37    public RemoteQueryErrorTest() {
38      super();
39      sliceCount = 1;
40      fixShardCount(random().nextBoolean() ? 3 : 4);
41    }
42  
43    @Test
44    public void test() throws Exception {
45      handle.clear();
46      handle.put("timestamp", SKIPVAL);
47      
48      waitForThingsToLevelOut(15);
49  
50      del("*:*");
51      
52      createCollection("collection2", 2, 1, 10);
53      
54      List<Integer> numShardsNumReplicaList = new ArrayList<>(2);
55      numShardsNumReplicaList.add(2);
56      numShardsNumReplicaList.add(1);
57      checkForCollection("collection2", numShardsNumReplicaList, null);
58      waitForRecoveriesToFinish("collection2", true);
59  
60      for (SolrClient solrClient : clients) {
61        try {
62          SolrInputDocument emptyDoc = new SolrInputDocument();
63          solrClient.add(emptyDoc);
64          fail("Expected unique key exceptoin");
65        } catch (SolrException ex) {
66          assertThat(ex.getMessage(), containsString("Document is missing mandatory uniqueKey field: id"));
67        } catch(Exception ex) {
68          fail("Expected a SolrException to occur, instead received: " + ex.getClass());
69        } finally {
70          solrClient.close();
71        }
72      }
73    }
74  }
