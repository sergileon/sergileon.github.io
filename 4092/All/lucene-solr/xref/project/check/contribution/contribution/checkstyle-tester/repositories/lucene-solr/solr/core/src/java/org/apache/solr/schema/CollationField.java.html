51  /**
52   * Field for collated sort keys. 
53   * These can be used for locale-sensitive sort and range queries.
54   * <p>
55   * This field can be created in two ways: 
56   * <ul>
57   *  <li>Based upon a system collator associated with a Locale.
58   *  <li>Based upon a tailored ruleset.
59   * </ul>
60   * <p>
61   * Using a System collator:
62   * <ul>
63   *  <li>language: ISO-639 language code (mandatory)
64   *  <li>country: ISO-3166 country code (optional)
65   *  <li>variant: vendor or browser-specific code (optional)
66   *  <li>strength: 'primary','secondary','tertiary', or 'identical' (optional)
67   *  <li>decomposition: 'no','canonical', or 'full' (optional)
68   * </ul>
69   * <p>
70   * Using a Tailored ruleset:
71   * <ul>
72   *  <li>custom: UTF-8 text file containing rules supported by RuleBasedCollator (mandatory)
73   *  <li>strength: 'primary','secondary','tertiary', or 'identical' (optional)
74   *  <li>decomposition: 'no','canonical', or 'full' (optional)
75   * </ul>
76   * 
77   * @see Collator
78   * @see Locale
79   * @see RuleBasedCollator
80   * @since solr 4.0
81   */
82  public class CollationField extends FieldType {
83    private Analyzer analyzer;
84  
85    @Override
86    protected void init(IndexSchema schema, Map<String,String> args) {
87      properties |= TOKENIZED; // this ensures our analyzer gets hit
88      setup(schema.getResourceLoader(), args);
89      super.init(schema, args);
90    }
91    
92    /**
93     * Setup the field according to the provided parameters
94     */
95    private void setup(ResourceLoader loader, Map<String,String> args) {
96      String custom = args.remove("custom");
97      String language = args.remove("language");
98      String country = args.remove("country");
99      String variant = args.remove("variant");
100     String strength = args.remove("strength");
101     String decomposition = args.remove("decomposition");
102     
103     final Collator collator;
104 
105     if (custom == null && language == null)
106       throw new SolrException(ErrorCode.SERVER_ERROR, "Either custom or language is required.");
107     
108     if (custom != null && 
109         (language != null || country != null || variant != null))
110       throw new SolrException(ErrorCode.SERVER_ERROR, "Cannot specify both language and custom. "
111           + "To tailor rules for a built-in language, see the javadocs for RuleBasedCollator. "
112           + "Then save the entire customized ruleset to a file, and use with the custom parameter");
113     
114     if (language != null) { 
115       // create from a system collator, based on Locale.
116       collator = createFromLocale(language, country, variant);
117     } else { 
118       // create from a custom ruleset
119       collator = createFromRules(custom, loader);
120     }
121     
122     // set the strength flag, otherwise it will be the default.
123     if (strength != null) {
124       if (strength.equalsIgnoreCase("primary"))
125         collator.setStrength(Collator.PRIMARY);
126       else if (strength.equalsIgnoreCase("secondary"))
127         collator.setStrength(Collator.SECONDARY);
128       else if (strength.equalsIgnoreCase("tertiary"))
129         collator.setStrength(Collator.TERTIARY);
130       else if (strength.equalsIgnoreCase("identical"))
131         collator.setStrength(Collator.IDENTICAL);
132       else
133         throw new SolrException(ErrorCode.SERVER_ERROR, "Invalid strength: " + strength);
134     }
135     
136     // set the decomposition flag, otherwise it will be the default.
137     if (decomposition != null) {
138       if (decomposition.equalsIgnoreCase("no"))
139         collator.setDecomposition(Collator.NO_DECOMPOSITION);
140       else if (decomposition.equalsIgnoreCase("canonical"))
141         collator.setDecomposition(Collator.CANONICAL_DECOMPOSITION);
142       else if (decomposition.equalsIgnoreCase("full"))
143         collator.setDecomposition(Collator.FULL_DECOMPOSITION);
144       else
145         throw new SolrException(ErrorCode.SERVER_ERROR, "Invalid decomposition: " + decomposition);
146     }
147     analyzer = new CollationKeyAnalyzer(collator);
148   }
149   
150   /**
151    * Create a locale from language, with optional country and variant.
152    * Then return the appropriate collator for the locale.
153    */
154   private Collator createFromLocale(String language, String country, String variant) {
155     Locale locale;
156     
157     if (language != null && country == null && variant != null)
158       throw new SolrException(ErrorCode.SERVER_ERROR, 
159           "To specify variant, country is required");
160     else if (language != null && country != null && variant != null)
161       locale = new Locale(language, country, variant);
162     else if (language != null && country != null)
163       locale = new Locale(language, country);
164     else 
165       locale = new Locale(language);
166     
167     return Collator.getInstance(locale);
168   }
169   
170   /**
171    * Read custom rules from a file, and create a RuleBasedCollator
172    * The file cannot support comments, as # might be in the rules!
173    */
174   private Collator createFromRules(String fileName, ResourceLoader loader) {
175     InputStream input = null;
176     try {
177      input = loader.openResource(fileName);
178      String rules = IOUtils.toString(input, "UTF-8");
179      return new RuleBasedCollator(rules);
180     } catch (IOException | ParseException e) {
181       // io error or invalid rules
182       throw new RuntimeException(e);
183     } finally {
184       IOUtils.closeQuietly(input);
185     }
186   }

217   /**
218    * analyze the range with the analyzer, instead of the collator.
219    * because jdk collators might not be thread safe (when they are
220    * it's just that all methods are synced), this keeps things 
221    * simple (we already have a threadlocal clone in the reused TS)
222    */
223   private BytesRef getCollationKey(String field, String text) {     
224     try (TokenStream source = analyzer.tokenStream(field, text)) {
225       source.reset();    
226       TermToBytesRefAttribute termAtt = source.getAttribute(TermToBytesRefAttribute.class);
227 
228       // we control the analyzer here: most errors are impossible
229       if (!source.incrementToken())
230         throw new IllegalArgumentException("analyzer returned no terms for text: " + text);
231       BytesRef bytes = BytesRef.deepCopyOf(termAtt.getBytesRef());
232       assert !source.incrementToken();
233       source.end();
234       return bytes;
235     } catch (IOException e) {
236       throw new RuntimeException("Unable to analyze text: " + text, e);
237     }
238   }

240   @Override
241   public Query getRangeQuery(QParser parser, SchemaField field, String part1, String part2, boolean minInclusive, boolean maxInclusive) {
242     String f = field.getName();
243     BytesRef low = part1 == null ? null : getCollationKey(f, part1);
244     BytesRef high = part2 == null ? null : getCollationKey(f, part2);
245     if (!field.indexed() && field.hasDocValues()) {
246       return DocValuesRangeQuery.newBytesRefRange(
247           field.getName(), low, high, minInclusive, maxInclusive);
248     } else {
249       return new TermRangeQuery(field.getName(), low, high, minInclusive, maxInclusive);
250     }
251   }

258   @Override
259   public List<IndexableField> createFields(SchemaField field, Object value, float boost) {
260     if (field.hasDocValues()) {
261       List<IndexableField> fields = new ArrayList<>();
262       fields.add(createField(field, value, boost));
263       final BytesRef bytes = getCollationKey(field.getName(), value.toString());
264       if (field.multiValued()) {
265         fields.add(new SortedSetDocValuesField(field.getName(), bytes));
266       } else {
267         fields.add(new SortedDocValuesField(field.getName(), bytes));
268       }
269       return fields;
270     } else {
271       return Collections.singletonList(createField(field, value, boost));
272     }
273   }
