34  // See SOLR-6640
35  @SolrTestCaseJ4.SuppressSSL
36  public class RecoveryAfterSoftCommitTest extends AbstractFullDistribZkTestBase {
37    private static final int MAX_BUFFERED_DOCS = 2, ULOG_NUM_RECORDS_TO_KEEP = 2;
38  
39    public RecoveryAfterSoftCommitTest() {
40      sliceCount = 1;
41      fixShardCount(2);
42    }
43  
44    @BeforeClass
45    public static void beforeTests() {
46      System.setProperty("solr.tests.maxBufferedDocs", String.valueOf(MAX_BUFFERED_DOCS));
47      System.setProperty("solr.ulog.numRecordsToKeep", String.valueOf(ULOG_NUM_RECORDS_TO_KEEP));
48      // the default=7000ms artificially slows down recovery and is not needed for this test
49      System.setProperty("solr.cloud.wait-for-updates-with-stale-state-pause", "500");
50      // avoid creating too many files, see SOLR-7421
51      System.setProperty("useCompoundFile", "true");
52    }
53  
54    @AfterClass
55    public static void afterTest()  {
56      System.clearProperty("solr.tests.maxBufferedDocs");
57      System.clearProperty("solr.ulog.numRecordsToKeep");
58      System.clearProperty("solr.cloud.wait-for-updates-with-stale-state-pause");
59      System.clearProperty("useCompoundFile");
60    }
61  
62    /**
63     * Overrides the parent implementation to install a SocketProxy in-front of the Jetty server.
64     */
65    @Override
66    public JettySolrRunner createJetty(File solrHome, String dataDir,
67                                       String shardList, String solrConfigOverride, String schemaOverride)
68        throws Exception
69    {
70      return createProxiedJetty(solrHome, dataDir, shardList, solrConfigOverride, schemaOverride);
71    }
72  
73    @Test
74    public void test() throws Exception {
75      waitForRecoveriesToFinish(DEFAULT_COLLECTION, true);
76      // flush twice
77      int i = 0;
78      for (; i<MAX_BUFFERED_DOCS + 1; i++) {
79        SolrInputDocument document = new SolrInputDocument();
80        document.addField("id", String.valueOf(i));
81        document.addField("a_t", "text_" + i);
82        cloudClient.add(document);
83      }
84  
85      // soft-commit so searchers are open on un-committed but flushed segment files
86      AbstractUpdateRequest request = new UpdateRequest().setAction(AbstractUpdateRequest.ACTION.COMMIT, true, true, true);
87      cloudClient.request(request);
88  
89      Replica notLeader = ensureAllReplicasAreActive(DEFAULT_COLLECTION, "shard1", 1, 2, 30).get(0);
90      // ok, now introduce a network partition between the leader and the replica
91      SocketProxy proxy = getProxyForReplica(notLeader);
92  
93      proxy.close();
94  
95      // add more than ULOG_NUM_RECORDS_TO_KEEP docs so that peer sync cannot be used for recovery
96      int MAX_DOCS = 2 + MAX_BUFFERED_DOCS + ULOG_NUM_RECORDS_TO_KEEP;
97      for (; i < MAX_DOCS; i++) {
98        SolrInputDocument document = new SolrInputDocument();
99        document.addField("id", String.valueOf(i));
100       document.addField("a_t", "text_" + i);
101       cloudClient.add(document);
102     }
103 
104     // Have the partition last at least 1 sec
105     // While this gives the impression that recovery is timing related, this is
106     // really only
107     // to give time for the state to be written to ZK before the test completes.
108     // In other words,
109     // without a brief pause, the test finishes so quickly that it doesn't give
110     // time for the recovery process to kick-in
111     Thread.sleep(2000L);
112 
113     proxy.reopen();
114 
115     List<Replica> notLeaders =
116         ensureAllReplicasAreActive(DEFAULT_COLLECTION, "shard1", 1, 2, 30);
117   }
118 }
