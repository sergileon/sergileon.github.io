67    /***
68     * Tests that FORCELEADER can get an active leader after leader puts all replicas in LIR and itself goes down,
69     * hence resulting in a leaderless shard.
70     */
71    @Test
72    @Slow
73    public void testReplicasInLIRNoLeader() throws Exception {
74      handle.put("maxScore", SKIPVAL);
75      handle.put("timestamp", SKIPVAL);
76  
77      String testCollectionName = "forceleader_test_collection";
78      createCollection(testCollectionName, 1, 3, 1);
79      cloudClient.setDefaultCollection(testCollectionName);
80  
81      try {
82        List<Replica> notLeaders = ensureAllReplicasAreActive(testCollectionName, SHARD1, 1, 3, maxWaitSecsToSeeAllActive);
83        assertEquals("Expected 2 replicas for collection " + testCollectionName
84            + " but found " + notLeaders.size() + "; clusterState: "
85            + printClusterStateInfo(testCollectionName), 2, notLeaders.size());
86  
87        Replica leader = cloudClient.getZkStateReader().getLeaderRetry(testCollectionName, SHARD1);
88        JettySolrRunner notLeader0 = getJettyOnPort(getReplicaPort(notLeaders.get(0)));
89        ZkController zkController = notLeader0.getCoreContainer().getZkController();
90  
91        putNonLeadersIntoLIR(testCollectionName, SHARD1, zkController, leader, notLeaders);
92  
93        cloudClient.getZkStateReader().updateClusterState();
94        ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
95        int numActiveReplicas = getNumberOfActiveReplicas(clusterState, testCollectionName, SHARD1);
96        assertEquals("Expected only 0 active replica but found " + numActiveReplicas +
97            "; clusterState: " + printClusterStateInfo(), 0, numActiveReplicas);
98  
99        int numReplicasOnLiveNodes = 0;
100       for (Replica rep : clusterState.getSlice(testCollectionName, SHARD1).getReplicas()) {
101         if (clusterState.getLiveNodes().contains(rep.getNodeName())) {
102           numReplicasOnLiveNodes++;
103         }
104       }
105       assertEquals(2, numReplicasOnLiveNodes);
106       log.info("Before forcing leader: " + printClusterStateInfo());
107       // Assert there is no leader yet
108       assertNull("Expected no leader right now. State: " + clusterState.getSlice(testCollectionName, SHARD1),
109           clusterState.getSlice(testCollectionName, SHARD1).getLeader());
110 
111       assertSendDocFails(3);
112 
113       doForceLeader(cloudClient, testCollectionName, SHARD1);
114 
115       // By now we have an active leader. Wait for recoveries to begin
116       waitForRecoveriesToFinish(testCollectionName, cloudClient.getZkStateReader(), true);
117 
118       cloudClient.getZkStateReader().updateClusterState();
119       clusterState = cloudClient.getZkStateReader().getClusterState();
120       log.info("After forcing leader: " + clusterState.getSlice(testCollectionName, SHARD1));
121       // we have a leader
122       Replica newLeader = clusterState.getSlice(testCollectionName, SHARD1).getLeader();
123       assertNotNull(newLeader);
124       // leader is active
125       assertEquals(State.ACTIVE, newLeader.getState());
126 
127       numActiveReplicas = getNumberOfActiveReplicas(clusterState, testCollectionName, SHARD1);
128       assertEquals(2, numActiveReplicas);
129 
130       // Assert that indexing works again
131       log.info("Sending doc 4...");
132       sendDoc(4);
133       log.info("Committing...");
134       cloudClient.commit();
135       log.info("Doc 4 sent and commit issued");
136 
137       assertDocsExistInAllReplicas(notLeaders, testCollectionName, 1, 1);
138       assertDocsExistInAllReplicas(notLeaders, testCollectionName, 4, 4);
139 
140       // Docs 1 and 4 should be here. 2 was lost during the partition, 3 had failed to be indexed.
141       log.info("Checking doc counts...");
142       ModifiableSolrParams params = new ModifiableSolrParams();
143       params.add("q", "*:*");
144       assertEquals("Expected only 2 documents in the index", 2, cloudClient.query(params).getResults().getNumFound());
145 
146       bringBackOldLeaderAndSendDoc(testCollectionName, leader, notLeaders, 5);
147     } finally {
148       log.info("Cleaning up after the test.");
149       // try to clean up
150       try {
151         CollectionAdminRequest.Delete req = new CollectionAdminRequest.Delete();
152         req.setCollectionName(testCollectionName);
153         req.process(cloudClient);
154       } catch (Exception e) {
155         // don't fail the test
156         log.warn("Could not delete collection {} after test completed", testCollectionName);
157       }
158     }
159   }

162    * Test that FORCELEADER can set last published state of all down (live) replicas to active (so

165   @Slow
166   public void testLastPublishedStateIsActive() throws Exception {
167     handle.put("maxScore", SKIPVAL);
168     handle.put("timestamp", SKIPVAL);
169 
170     String testCollectionName = "forceleader_last_published";
171     createCollection(testCollectionName, 1, 3, 1);
172     cloudClient.setDefaultCollection(testCollectionName);
173     log.info("Collection created: " + testCollectionName);
174 
175     try {
176       List<Replica> notLeaders = ensureAllReplicasAreActive(testCollectionName, SHARD1, 1, 3, maxWaitSecsToSeeAllActive);
177       assertEquals("Expected 2 replicas for collection " + testCollectionName
178           + " but found " + notLeaders.size() + "; clusterState: "
179           + printClusterStateInfo(testCollectionName), 2, notLeaders.size());
180 
181       Replica leader = cloudClient.getZkStateReader().getLeaderRetry(testCollectionName, SHARD1);
182       JettySolrRunner notLeader0 = getJettyOnPort(getReplicaPort(notLeaders.get(0)));
183       ZkController zkController = notLeader0.getCoreContainer().getZkController();
184 
185       // Mark all replicas down
186       setReplicaState(testCollectionName, SHARD1, leader, State.DOWN);
187       for (Replica rep : notLeaders) {
188         setReplicaState(testCollectionName, SHARD1, rep, State.DOWN);
189       }
190 
191       zkController.getZkStateReader().updateClusterState();
192       // Assert all replicas are down and that there is no leader
193       assertEquals(0, getActiveOrRecoveringReplicas(testCollectionName, SHARD1).size());
194 
195       // Now force leader
196       doForceLeader(cloudClient, testCollectionName, SHARD1);
197 
198       // Assert that last published states of the two replicas are active now
199       for (Replica rep: notLeaders) {
200         assertEquals(Replica.State.ACTIVE, getLastPublishedState(testCollectionName, SHARD1, rep));
201       }
202     } finally {
203       log.info("Cleaning up after the test.");
204       // try to clean up
205       try {
206         CollectionAdminRequest.Delete req = new CollectionAdminRequest.Delete();
207         req.setCollectionName(testCollectionName);
208         req.process(cloudClient);
209       } catch (Exception e) {
210         // don't fail the test
211         log.warn("Could not delete collection {} after test completed", testCollectionName);
212       }
213     }
214   }

216   protected void unsetLeader(String collection, String slice) throws Exception {
217     DistributedQueue inQueue = Overseer.getInQueue(cloudClient.getZkStateReader().getZkClient());
218     ZkStateReader zkStateReader = cloudClient.getZkStateReader();
219 
220     ZkNodeProps m = new ZkNodeProps(Overseer.QUEUE_OPERATION, OverseerAction.LEADER.toLower(),
221         ZkStateReader.SHARD_ID_PROP, slice,
222         ZkStateReader.COLLECTION_PROP, collection);
223     inQueue.offer(Utils.toJSON(m));
224 
225     ClusterState clusterState = null;
226     boolean transition = false;
227     for (int counter = 10; counter > 0; counter--) {
228       zkStateReader.updateClusterState();
229       clusterState = zkStateReader.getClusterState();
230       Replica newLeader = clusterState.getSlice(collection, slice).getLeader();
231       if (newLeader == null) {
232         transition = true;
233         break;
234       }
235       Thread.sleep(1000);
236     }
237 
238     if (!transition) {
239       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Could not unset replica leader" +
240           ". Cluster state: " + printClusterStateInfo(collection));
241     }
242   }

244   protected void setReplicaState(String collection, String slice, Replica replica, Replica.State state) throws SolrServerException, IOException,
245       KeeperException, InterruptedException {
246     DistributedQueue inQueue = Overseer.getInQueue(cloudClient.getZkStateReader().getZkClient());
247     ZkStateReader zkStateReader = cloudClient.getZkStateReader();
248 
249     String baseUrl = zkStateReader.getBaseUrlForNodeName(replica.getNodeName());
250     ZkNodeProps m = new ZkNodeProps(Overseer.QUEUE_OPERATION, OverseerAction.STATE.toLower(),
251         ZkStateReader.BASE_URL_PROP, baseUrl,
252         ZkStateReader.NODE_NAME_PROP, replica.getNodeName(),
253         ZkStateReader.SHARD_ID_PROP, slice,
254         ZkStateReader.COLLECTION_PROP, collection,
255         ZkStateReader.CORE_NAME_PROP, replica.getStr(CORE_NAME_PROP),
256         ZkStateReader.CORE_NODE_NAME_PROP, replica.getName(),
257         ZkStateReader.STATE_PROP, state.toString());
258     inQueue.offer(Utils.toJSON(m));
259     boolean transition = false;
260 
261     Replica.State replicaState = null;
262     for (int counter = 10; counter > 0; counter--) {
263       zkStateReader.updateClusterState();
264       ClusterState clusterState = zkStateReader.getClusterState();
265       replicaState = clusterState.getSlice(collection, slice).getReplica(replica.getName()).getState();
266       if (replicaState == state) {
267         transition = true;
268         break;
269       }
270       Thread.sleep(1000);
271     }
272 
273     if (!transition) {
274       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Could not set replica [" + replica.getName() + "] as " + state +
275           ". Last known state of the replica: " + replicaState);
276     }
277   }

296   protected Replica.State getLastPublishedState(String collection, String slice, Replica replica) throws SolrServerException, IOException,
297   KeeperException, InterruptedException {
298     ZkStateReader zkStateReader = cloudClient.getZkStateReader();
299     String baseUrl = zkStateReader.getBaseUrlForNodeName(replica.getNodeName());
300 
301     ModifiableSolrParams params = new ModifiableSolrParams();
302     params.set(CoreAdminParams.ACTION, CoreAdminAction.STATUS.toString());
303     params.set(CoreAdminParams.CORE, replica.getStr("core"));
304 
305     SolrRequest<SimpleSolrResponse> req = new GenericSolrRequest(METHOD.GET, "/admin/cores", params);
306     NamedList resp = null;
307     try (HttpSolrClient hsc = new HttpSolrClient(baseUrl)) {
308        resp = hsc.request(req);
309     }
310 
311     String lastPublished = (((NamedList<NamedList<String>>)resp.get("status")).get(replica.getStr("core"))).get("lastPublished");
312     return Replica.State.getState(lastPublished);
313   }

326   void putNonLeadersIntoLIR(String collectionName, String shard, ZkController zkController, Replica leader, List<Replica> notLeaders) throws Exception {
327     SocketProxy[] nonLeaderProxies = new SocketProxy[notLeaders.size()];
328     for (int i = 0; i < notLeaders.size(); i++)
329       nonLeaderProxies[i] = getProxyForReplica(notLeaders.get(i));
330 
331     sendDoc(1);
332 
333     // ok, now introduce a network partition between the leader and both replicas
334     log.info("Closing proxies for the non-leader replicas...");
335     for (SocketProxy proxy : nonLeaderProxies)
336       proxy.close();
337 
338     // indexing during a partition
339     log.info("Sending a doc during the network partition...");
340     sendDoc(2);
341 
342     // Wait a little
343     Thread.sleep(2000);
344 
345     // Kill the leader
346     log.info("Killing leader for shard1 of " + collectionName + " on node " + leader.getNodeName() + "");
347     JettySolrRunner leaderJetty = getJettyOnPort(getReplicaPort(leader));
348     getProxyForReplica(leader).close();
349     leaderJetty.stop();
350 
351     // Wait for a steady state, till LIR flags have been set and the shard is leaderless
352     log.info("Sleep and periodically wake up to check for state...");
353     for (int i = 0; i < 20; i++) {
354       Thread.sleep(1000);
355       State lirStates[] = new State[notLeaders.size()];
356       for (int j = 0; j < notLeaders.size(); j++)
357         lirStates[j] = zkController.getLeaderInitiatedRecoveryState(collectionName, shard, notLeaders.get(j).getName());
358 
359       zkController.getZkStateReader().updateClusterState();
360       ClusterState clusterState = zkController.getZkStateReader().getClusterState();
361       boolean allDown = true;
362       for (State lirState : lirStates)
363         if (Replica.State.DOWN.equals(lirState) == false)
364           allDown = false;
365       if (allDown && clusterState.getSlice(collectionName, shard).getLeader() == null) {
366         break;
367       }
368       log.warn("Attempt " + i + ", waiting on for 1 sec to settle down in the steady state. State: " +
369           printClusterStateInfo(collectionName));
370       log.warn("LIR state: " + getLIRState(zkController, collectionName, shard));
371     }
372     log.info("Waking up...");
373 
374     // remove the network partition
375     log.info("Reopening the proxies for the non-leader replicas...");
376     for (SocketProxy proxy : nonLeaderProxies)
377       proxy.reopen();
378 
379     log.info("LIR state: " + getLIRState(zkController, collectionName, shard));
380 
381     State lirStates[] = new State[notLeaders.size()];
382     for (int j = 0; j < notLeaders.size(); j++)
383       lirStates[j] = zkController.getLeaderInitiatedRecoveryState(collectionName, shard, notLeaders.get(j).getName());
384     for (State lirState : lirStates)
385       assertTrue("Expected that the replicas would be in LIR state by now. LIR states: "+Arrays.toString(lirStates),
386           Replica.State.DOWN == lirState || Replica.State.RECOVERING == lirState);
387   }

389   protected void bringBackOldLeaderAndSendDoc(String collection, Replica leader, List<Replica> notLeaders, int docid) throws Exception {
390     // Bring back the leader which was stopped
391     log.info("Bringing back originally killed leader...");
392     JettySolrRunner leaderJetty = getJettyOnPort(getReplicaPort(leader));
393     leaderJetty.start();
394     waitForRecoveriesToFinish(collection, cloudClient.getZkStateReader(), true);
395     cloudClient.getZkStateReader().updateClusterState();
396     ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
397     log.info("After bringing back leader: " + clusterState.getSlice(collection, SHARD1));
398     int numActiveReplicas = getNumberOfActiveReplicas(clusterState, collection, SHARD1);
399     assertEquals(1+notLeaders.size(), numActiveReplicas);
400     log.info("Sending doc "+docid+"...");
401     sendDoc(docid);
402     log.info("Committing...");
403     cloudClient.commit();
404     log.info("Doc "+docid+" sent and commit issued");
405     assertDocsExistInAllReplicas(notLeaders, collection, docid, docid);
406     assertDocsExistInAllReplicas(Collections.singletonList(leader), collection, docid, docid);
407   }

409   protected String getLIRState(ZkController zkController, String collection, String shard) throws KeeperException, InterruptedException {
410     StringBuilder sb = new StringBuilder();
411     String path = zkController.getLeaderInitiatedRecoveryZnodePath(collection, shard);
412     if (path == null)
413       return null;
414     try {
415       zkController.getZkClient().printLayout(path, 4, sb);
416     } catch (NoNodeException ex) {
417       return null;
418     }
419     return sb.toString();
420   }

431   private void doForceLeader(SolrClient client, String collectionName, String shard) throws IOException, SolrServerException {
432       CollectionAdminRequest.ForceLeader forceLeader = new CollectionAdminRequest.ForceLeader();
433       forceLeader.setCollectionName(collectionName);
434       forceLeader.setShardName(shard);
435       client.request(forceLeader);
436   }

438   protected int getNumberOfActiveReplicas(ClusterState clusterState, String collection, String sliceId) {
439     int numActiveReplicas = 0;
440     // Assert all replicas are active
441     for (Replica rep : clusterState.getSlice(collection, sliceId).getReplicas()) {
442       if (rep.getState().equals(State.ACTIVE)) {
443         numActiveReplicas++;
444       }
445     }
446     return numActiveReplicas;
447   }
