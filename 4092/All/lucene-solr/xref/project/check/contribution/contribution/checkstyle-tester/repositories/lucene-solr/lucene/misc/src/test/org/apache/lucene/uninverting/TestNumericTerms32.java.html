44  public class TestNumericTerms32 extends LuceneTestCase {
45    // distance of entries
46    private static int distance;
47    // shift the starting of the values to the left, to also have negative values:
48    private static final int startOffset = - 1 << 15;
49    // number of docs to generate for testing
50    private static int noDocs;
51    
52    private static Directory directory = null;
53    private static IndexReader reader = null;
54    private static IndexSearcher searcher = null;
55    
56    @BeforeClass
57    public static void beforeClass() throws Exception {
58      noDocs = atLeast(4096);
59      distance = (1 << 30) / noDocs;
60      directory = newDirectory();
61      RandomIndexWriter writer = new RandomIndexWriter(random(), directory,
62          newIndexWriterConfig(new MockAnalyzer(random()))
63          .setMaxBufferedDocs(TestUtil.nextInt(random(), 100, 1000))
64          .setMergePolicy(newLogMergePolicy()));
65      
66      final FieldType storedInt = new FieldType(IntField.TYPE_NOT_STORED);
67      storedInt.setStored(true);
68      storedInt.freeze();
69  
70      final FieldType storedInt8 = new FieldType(storedInt);
71      storedInt8.setNumericPrecisionStep(8);
72  
73      final FieldType storedInt4 = new FieldType(storedInt);
74      storedInt4.setNumericPrecisionStep(4);
75  
76      final FieldType storedInt2 = new FieldType(storedInt);
77      storedInt2.setNumericPrecisionStep(2);
78  
79      IntField
80        field8 = new IntField("field8", 0, storedInt8),
81        field4 = new IntField("field4", 0, storedInt4),
82        field2 = new IntField("field2", 0, storedInt2);
83      
84      Document doc = new Document();
85      // add fields, that have a distance to test general functionality
86      doc.add(field8); doc.add(field4); doc.add(field2);
87      
88      // Add a series of noDocs docs with increasing int values
89      for (int l=0; l<noDocs; l++) {
90        int val=distance*l+startOffset;
91        field8.setIntValue(val);
92        field4.setIntValue(val);
93        field2.setIntValue(val);
94  
95        val=l-(noDocs/2);
96        writer.addDocument(doc);
97      }
98    
99      Map<String,Type> map = new HashMap<>();
100     map.put("field2", Type.INTEGER);
101     map.put("field4", Type.INTEGER);
102     map.put("field8", Type.INTEGER);
103     reader = UninvertingReader.wrap(writer.getReader(), map);
104     searcher=newSearcher(reader);
105     writer.close();
106   }
107   
108   @AfterClass
109   public static void afterClass() throws Exception {
110     searcher = null;
111     TestUtil.checkReader(reader);
112     reader.close();
113     reader = null;
114     directory.close();
115     directory = null;
116   }
117   
118   private void testSorting(int precisionStep) throws Exception {
119     String field="field"+precisionStep;
120     // 10 random tests, the index order is ascending,
121     // so using a reverse sort field should retun descending documents
122     int num = TestUtil.nextInt(random(), 10, 20);
123     for (int i = 0; i < num; i++) {
124       int lower=(int)(random().nextDouble()*noDocs*distance)+startOffset;
125       int upper=(int)(random().nextDouble()*noDocs*distance)+startOffset;
126       if (lower>upper) {
127         int a=lower; lower=upper; upper=a;
128       }
129       Query tq=NumericRangeQuery.newIntRange(field, precisionStep, lower, upper, true, true);
130       TopDocs topDocs = searcher.search(tq, noDocs, new Sort(new SortField(field, SortField.Type.INT, true)));
131       if (topDocs.totalHits==0) continue;
132       ScoreDoc[] sd = topDocs.scoreDocs;
133       assertNotNull(sd);
134       int last = searcher.doc(sd[0].doc).getField(field).numericValue().intValue();
135       for (int j=1; j<sd.length; j++) {
136         int act = searcher.doc(sd[j].doc).getField(field).numericValue().intValue();
137         assertTrue("Docs should be sorted backwards", last>act );
138         last=act;
139       }
140     }
141   }
142 
143   @Test
144   public void testSorting_8bit() throws Exception {
145     testSorting(8);
146   }
147   
148   @Test
149   public void testSorting_4bit() throws Exception {
150     testSorting(4);
151   }
152   
153   @Test
154   public void testSorting_2bit() throws Exception {
155     testSorting(2);
156   }
