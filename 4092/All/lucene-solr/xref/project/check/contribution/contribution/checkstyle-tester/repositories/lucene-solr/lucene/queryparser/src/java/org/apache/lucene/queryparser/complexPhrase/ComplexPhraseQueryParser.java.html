46  /**
47   * QueryParser which permits complex phrase query syntax eg "(john jon
48   * jonathan~) peters*".
49   * <p>
50   * Performs potentially multiple passes over Query text to parse any nested
51   * logic in PhraseQueries. - First pass takes any PhraseQuery content between
52   * quotes and stores for subsequent pass. All other query content is parsed as
53   * normal - Second pass parses any stored PhraseQuery content, checking all
54   * embedded clauses are referring to the same field and therefore can be
55   * rewritten as Span queries. All PhraseQuery clauses are expressed as
56   * ComplexPhraseQuery objects
57   * </p>
58   * <p>
59   * This could arguably be done in one pass using a new QueryParser but here I am
60   * working within the constraints of the existing parser as a base class. This
61   * currently simply feeds all phrase content through an analyzer to select
62   * phrase terms - any "special" syntax such as * ~ * etc are not given special
63   * status
64   * </p>
65   * 
66   */
67  public class ComplexPhraseQueryParser extends QueryParser {
68    private ArrayList<ComplexPhraseQuery> complexPhrases = null;
69  
70    private boolean isPass2ResolvingPhrases;
71  
72    private boolean inOrder = true;
73  
74    /**
75     * When <code>inOrder</code> is true, the search terms must
76     * exists in the documents as the same order as in query.
77     *
78     * @param inOrder parameter to choose between ordered or un-ordered proximity search
79     */
80    public void setInOrder(final boolean inOrder) {
81      this.inOrder = inOrder;
82    }
83  
84    private ComplexPhraseQuery currentPhraseQuery = null;
85  
86    public ComplexPhraseQueryParser(String f, Analyzer a) {
87      super(f, a);
88    }
89  
90    @Override
91    protected Query getFieldQuery(String field, String queryText, int slop) {
92      ComplexPhraseQuery cpq = new ComplexPhraseQuery(field, queryText, slop, inOrder);
93      complexPhrases.add(cpq); // add to list of phrases to be parsed once
94      // we
95      // are through with this pass
96      return cpq;
97    }

99    @Override
100   public Query parse(String query) throws ParseException {
101     if (isPass2ResolvingPhrases) {
102       MultiTermQuery.RewriteMethod oldMethod = getMultiTermRewriteMethod();
103       try {
104         // Temporarily force BooleanQuery rewrite so that Parser will
105         // generate visible
106         // collection of terms which we can convert into SpanQueries.
107         // ConstantScoreRewrite mode produces an
108         // opaque ConstantScoreQuery object which cannot be interrogated for
109         // terms in the same way a BooleanQuery can.
110         // QueryParser is not guaranteed threadsafe anyway so this temporary
111         // state change should not
112         // present an issue
113         setMultiTermRewriteMethod(MultiTermQuery.SCORING_BOOLEAN_REWRITE);
114         return super.parse(query);
115       } finally {
116         setMultiTermRewriteMethod(oldMethod);
117       }
118     }
119 
120     // First pass - parse the top-level query recording any PhraseQuerys
121     // which will need to be resolved
122     complexPhrases = new ArrayList<>();
123     Query q = super.parse(query);
124 
125     // Perform second pass, using this QueryParser to parse any nested
126     // PhraseQueries with different
127     // set of syntax restrictions (i.e. all fields must be same)
128     isPass2ResolvingPhrases = true;
129     try {
130       for (Iterator<ComplexPhraseQuery> iterator = complexPhrases.iterator(); iterator.hasNext();) {
131         currentPhraseQuery = iterator.next();
132         // in each phrase, now parse the contents between quotes as a
133         // separate parse operation
134         currentPhraseQuery.parsePhraseElements(this);
135       }
136     } finally {
137       isPass2ResolvingPhrases = false;
138     }
139     return q;
140   }
141 
142   // There is No "getTermQuery throws ParseException" method to override so
143   // unfortunately need
144   // to throw a runtime exception here if a term for another field is embedded
145   // in phrase query
146   @Override
147   protected Query newTermQuery(Term term) {
148     if (isPass2ResolvingPhrases) {
149       try {
150         checkPhraseClauseIsForSameField(term.field());
151       } catch (ParseException pe) {
152         throw new RuntimeException("Error parsing complex phrase", pe);
153       }
154     }
155     return super.newTermQuery(term);
156   }
157 
158   // Helper method used to report on any clauses that appear in query syntax
159   private void checkPhraseClauseIsForSameField(String field)
160       throws ParseException {
161     if (!field.equals(currentPhraseQuery.field)) {
162       throw new ParseException("Cannot have clause for field \"" + field
163           + "\" nested in phrase " + " for field \"" + currentPhraseQuery.field
164           + "\"");
165     }
166   }
167 
168   @Override
169   protected Query getWildcardQuery(String field, String termStr)
170       throws ParseException {
171     if (isPass2ResolvingPhrases) {
172       checkPhraseClauseIsForSameField(field);
173     }
174     return super.getWildcardQuery(field, termStr);
175   }
176 
177   @Override
178   protected Query getRangeQuery(String field, String part1, String part2,
179       boolean startInclusive, boolean endInclusive) throws ParseException {
180     if (isPass2ResolvingPhrases) {
181       checkPhraseClauseIsForSameField(field);
182     }
183     return super.getRangeQuery(field, part1, part2, startInclusive, endInclusive);
184   }
185 
186   @Override
187   protected Query newRangeQuery(String field, String part1, String part2,
188       boolean startInclusive, boolean endInclusive) {
189     if (isPass2ResolvingPhrases) {
190       // Must use old-style RangeQuery in order to produce a BooleanQuery
191       // that can be turned into SpanOr clause
192       TermRangeQuery rangeQuery = TermRangeQuery.newStringRange(field, part1, part2, startInclusive, endInclusive);
193       rangeQuery.setRewriteMethod(MultiTermQuery.SCORING_BOOLEAN_REWRITE);
194       return rangeQuery;
195     }
196     return super.newRangeQuery(field, part1, part2, startInclusive, endInclusive);
197   }
198 
199   @Override
200   protected Query getFuzzyQuery(String field, String termStr,
201       float minSimilarity) throws ParseException {
202     if (isPass2ResolvingPhrases) {
203       checkPhraseClauseIsForSameField(field);
204     }
205     return super.getFuzzyQuery(field, termStr, minSimilarity);
206   }

208   /*
209    * Used to handle the query content in between quotes and produced Span-based
210    * interpretations of the clauses.
211    */
212   static class ComplexPhraseQuery extends Query {
213 
214     final String field;
215 
216     final String phrasedQueryStringContents;
217 
218     final int slopFactor;
219 
220     private final boolean inOrder;
221 
222     private final Query[] contents = new Query[1];
223 
224     public ComplexPhraseQuery(String field, String phrasedQueryStringContents,
225         int slopFactor, boolean inOrder) {
226       super();
227       this.field = field;
228       this.phrasedQueryStringContents = phrasedQueryStringContents;
229       this.slopFactor = slopFactor;
230       this.inOrder = inOrder;
231     }
232 
233     // Called by ComplexPhraseQueryParser for each phrase after the main
234     // parse
235     // thread is through
236     protected void parsePhraseElements(ComplexPhraseQueryParser qp) throws ParseException {
237       // TODO ensure that field-sensitivity is preserved ie the query
238       // string below is parsed as
239       // field+":("+phrasedQueryStringContents+")"
240       // but this will need code in rewrite to unwrap the first layer of
241       // boolean query
242 
243       String oldDefaultParserField = qp.field;
244       try {
245         //temporarily set the QueryParser to be parsing the default field for this phrase e.g author:"fred* smith"
246         qp.field = this.field;
247         contents[0] = qp.parse(phrasedQueryStringContents);
248       }
249       finally {
250         qp.field = oldDefaultParserField;
251       }
252     }
253 
254     @Override
255     public Query rewrite(IndexReader reader) throws IOException {
256       if (getBoost() != 1f) {
257         return super.rewrite(reader);
258       }
259       final Query contents = this.contents[0];
260       // ArrayList spanClauses = new ArrayList();
261       if (contents instanceof TermQuery) {
262         return contents;
263       }
264       // Build a sequence of Span clauses arranged in a SpanNear - child
265       // clauses can be complex
266       // Booleans e.g. nots and ors etc
267       int numNegatives = 0;
268       if (!(contents instanceof BooleanQuery)) {
269         throw new IllegalArgumentException("Unknown query type \""
270             + contents.getClass().getName()
271             + "\" found in phrase query string \"" + phrasedQueryStringContents
272             + "\"");
273       }
274       BooleanQuery bq = (BooleanQuery) contents;
275       SpanQuery[] allSpanClauses = new SpanQuery[bq.clauses().size()];
276       // For all clauses e.g. one* two~
277       int i = 0;
278       for (BooleanClause clause : bq) {
279         // HashSet bclauseterms=new HashSet();
280         Query qc = clause.getQuery();
281         // Rewrite this clause e.g one* becomes (one OR onerous)
282         qc = new IndexSearcher(reader).rewrite(qc);
283         if (clause.getOccur().equals(BooleanClause.Occur.MUST_NOT)) {
284           numNegatives++;
285         }
286 
287         while (qc instanceof BoostQuery) {
288           qc = ((BoostQuery) qc).getQuery();
289         }
290 
291         if (qc instanceof BooleanQuery) {
292           ArrayList<SpanQuery> sc = new ArrayList<>();
293           addComplexPhraseClause(sc, (BooleanQuery) qc);
294           if (sc.size() > 0) {
295             allSpanClauses[i] = sc.get(0);
296           } else {
297             // Insert fake term e.g. phrase query was for "Fred Smithe*" and
298             // there were no "Smithe*" terms - need to
299             // prevent match on just "Fred".
300             allSpanClauses[i] = new SpanTermQuery(new Term(field,
301                 "Dummy clause because no terms found - must match nothing"));
302           }
303         } else {
304           if (qc instanceof TermQuery) {
305             TermQuery tq = (TermQuery) qc;
306             allSpanClauses[i] = new SpanTermQuery(tq.getTerm());
307           } else {
308             throw new IllegalArgumentException("Unknown query type \""
309                 + qc.getClass().getName()
310                 + "\" found in phrase query string \""
311                 + phrasedQueryStringContents + "\"");
312           }
313 
314         }
315         i += 1;
316       }
317       if (numNegatives == 0) {
318         // The simple case - no negative elements in phrase
319         return new SpanNearQuery(allSpanClauses, slopFactor, inOrder);
320       }
321       // Complex case - we have mixed positives and negatives in the
322       // sequence.
323       // Need to return a SpanNotQuery
324       ArrayList<SpanQuery> positiveClauses = new ArrayList<>();
325       i = 0;
326       for (BooleanClause clause : bq) {
327         if (!clause.getOccur().equals(BooleanClause.Occur.MUST_NOT)) {
328           positiveClauses.add(allSpanClauses[i]);
329         }
330         i += 1;
331       }
332 
333       SpanQuery[] includeClauses = positiveClauses
334           .toArray(new SpanQuery[positiveClauses.size()]);
335 
336       SpanQuery include = null;
337       if (includeClauses.length == 1) {
338         include = includeClauses[0]; // only one positive clause
339       } else {
340         // need to increase slop factor based on gaps introduced by
341         // negatives
342         include = new SpanNearQuery(includeClauses, slopFactor + numNegatives,
343             inOrder);
344       }
345       // Use sequence of positive and negative values as the exclude.
346       SpanNearQuery exclude = new SpanNearQuery(allSpanClauses, slopFactor,
347           inOrder);
348       SpanNotQuery snot = new SpanNotQuery(include, exclude);
349       return snot;
350     }
351 
352     private void addComplexPhraseClause(List<SpanQuery> spanClauses, BooleanQuery qc) {
353       ArrayList<SpanQuery> ors = new ArrayList<>();
354       ArrayList<SpanQuery> nots = new ArrayList<>();
355 
356       // For all clauses e.g. one* two~
357       for (BooleanClause clause : qc) {
358         Query childQuery = clause.getQuery();
359 
360         float boost = 1f;
361         while (childQuery instanceof BoostQuery) {
362           BoostQuery bq = (BoostQuery) childQuery;
363           boost *= bq.getBoost();
364           childQuery = bq.getQuery();
365         }
366 
367         // select the list to which we will add these options
368         ArrayList<SpanQuery> chosenList = ors;
369         if (clause.getOccur() == BooleanClause.Occur.MUST_NOT) {
370           chosenList = nots;
371         }
372 
373         if (childQuery instanceof TermQuery) {
374           TermQuery tq = (TermQuery) childQuery;
375           SpanQuery stq = new SpanTermQuery(tq.getTerm());
376           if (boost != 1f) {
377             stq = new SpanBoostQuery(stq, boost);
378           }
379           chosenList.add(stq);
380         } else if (childQuery instanceof BooleanQuery) {
381           BooleanQuery cbq = (BooleanQuery) childQuery;
382           addComplexPhraseClause(chosenList, cbq);
383         } else {
384           // TODO alternatively could call extract terms here?
385           throw new IllegalArgumentException("Unknown query type:"
386               + childQuery.getClass().getName());
387         }
388       }
389       if (ors.size() == 0) {
390         return;
391       }
392       SpanOrQuery soq = new SpanOrQuery(ors
393           .toArray(new SpanQuery[ors.size()]));
394       if (nots.size() == 0) {
395         spanClauses.add(soq);
396       } else {
397         SpanOrQuery snqs = new SpanOrQuery(nots
398             .toArray(new SpanQuery[nots.size()]));
399         SpanNotQuery snq = new SpanNotQuery(soq, snqs);
400         spanClauses.add(snq);
401       }
402     }

404     @Override
405     public String toString(String field) {
406       if (slopFactor == 0)
407         return "\"" + phrasedQueryStringContents + "\"";
408       else
409         return "\"" + phrasedQueryStringContents + "\"" + "~" + slopFactor;
410     }
411 
412     @Override
413     public int hashCode() {
414       final int prime = 31;
415       int result = super.hashCode();
416       result = prime * result + ((field == null) ? 0 : field.hashCode());
417       result = prime
418           * result
419           + ((phrasedQueryStringContents == null) ? 0
420               : phrasedQueryStringContents.hashCode());
421       result = prime * result + slopFactor;
422       result = prime * result + (inOrder ? 1 : 0);
423       return result;
424     }
425 
426     @Override
427     public boolean equals(Object obj) {
428       if (this == obj)
429         return true;
430       if (obj == null)
431         return false;
432       if (getClass() != obj.getClass())
433         return false;
434       if (!super.equals(obj)) {
435         return false;
436       }
437       ComplexPhraseQuery other = (ComplexPhraseQuery) obj;
438       if (field == null) {
439         if (other.field != null)
440           return false;
441       } else if (!field.equals(other.field))
442         return false;
443       if (phrasedQueryStringContents == null) {
444         if (other.phrasedQueryStringContents != null)
445           return false;
446       } else if (!phrasedQueryStringContents
447           .equals(other.phrasedQueryStringContents))
448         return false;
449       if (slopFactor != other.slopFactor)
450         return false;
451       return inOrder == other.inOrder;
452     }
