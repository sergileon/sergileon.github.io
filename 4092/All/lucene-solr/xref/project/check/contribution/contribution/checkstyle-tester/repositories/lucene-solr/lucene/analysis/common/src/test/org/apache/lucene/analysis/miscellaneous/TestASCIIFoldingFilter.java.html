34    /**
35     * Pop one input token's worth of tokens off the filter and verify that they are as expected.
36     */
37    void assertNextTerms(String expectedUnfolded, String expectedFolded, ASCIIFoldingFilter filter,
38        CharTermAttribute termAtt) throws Exception {
39      assertTrue(filter.incrementToken());
40      assertEquals(expectedFolded, termAtt.toString());
41      if (filter.isPreserveOriginal() && !expectedUnfolded.equals(expectedFolded)) {
42        assertTrue(filter.incrementToken());
43        assertEquals(expectedUnfolded, termAtt.toString());
44      }
45    }
