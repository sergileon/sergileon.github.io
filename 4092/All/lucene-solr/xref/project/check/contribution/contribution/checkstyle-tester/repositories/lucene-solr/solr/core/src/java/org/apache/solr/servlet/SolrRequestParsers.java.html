95    public SolrRequestParsers( SolrConfig globalConfig ) {
96      final int multipartUploadLimitKB, formUploadLimitKB;
97      if( globalConfig == null ) {
98        multipartUploadLimitKB = formUploadLimitKB = Integer.MAX_VALUE; 
99        enableRemoteStreams = true;
100       handleSelect = true;
101       addHttpRequestToContext = false;
102     } else {
103       multipartUploadLimitKB = globalConfig.getMultipartUploadLimitKB();
104       
105       formUploadLimitKB = globalConfig.getFormUploadLimitKB();
106       
107       enableRemoteStreams = globalConfig.isEnableRemoteStreams();
108   
109       // Let this filter take care of /select?xxx format
110       handleSelect = globalConfig.isHandleSelect();
111       
112       addHttpRequestToContext = globalConfig.isAddHttpRequestToContext();
113     }
114     init(multipartUploadLimitKB, formUploadLimitKB);
115   }

124   private void init( int multipartUploadLimitKB, int formUploadLimitKB) {       
125     MultipartRequestParser multi = new MultipartRequestParser( multipartUploadLimitKB );
126     RawRequestParser raw = new RawRequestParser();
127     FormDataRequestParser formdata = new FormDataRequestParser( formUploadLimitKB );
128     standard = new StandardRequestParser( multi, raw, formdata );
129     
130     // I don't see a need to have this publicly configured just yet
131     // adding it is trivial
132     parsers.put( MULTIPART, multi );
133     parsers.put( FORMDATA, formdata );
134     parsers.put( RAW, raw );
135     parsers.put( SIMPLE, new SimpleRequestParser() );
136     parsers.put( STANDARD, standard );
137     parsers.put( "", standard );
138   }

150   public SolrQueryRequest parse( SolrCore core, String path, HttpServletRequest req ) throws Exception
151   {
152     SolrRequestParser parser = standard;
153     
154     // TODO -- in the future, we could pick a different parser based on the request
155     
156     // Pick the parser from the request...
157     ArrayList<ContentStream> streams = new ArrayList<>(1);
158     SolrParams params = parser.parseParamsAndFillStreams( req, streams );
159     SolrQueryRequest sreq = buildRequestFrom(core, params, streams, getRequestTimer(req), req);
160 
161     // Handlers and login will want to know the path. If it contains a ':'
162     // the handler could use it for RESTful URLs
163     sreq.getContext().put(PATH, RequestHandlers.normalize(path));
164     sreq.getContext().put("httpMethod", req.getMethod());
165 
166     if(addHttpRequestToContext) {
167       sreq.getContext().put("httpRequest", req);
168     }
169     return sreq;
170   }

176   private SolrQueryRequest buildRequestFrom(SolrCore core, SolrParams params, Collection<ContentStream> streams,
177                                             RTimerTree requestTimer, final HttpServletRequest req) throws Exception {
178     // The content type will be applied to all streaming content
179     String contentType = params.get( CommonParams.STREAM_CONTENTTYPE );
180       
181     // Handle anything with a remoteURL
182     String[] strs = params.getParams( CommonParams.STREAM_URL );
183     if( strs != null ) {
184       if( !enableRemoteStreams ) {
185         throw new SolrException( ErrorCode.BAD_REQUEST, "Remote Streaming is disabled." );
186       }
187       for( final String url : strs ) {
188         ContentStreamBase stream = new ContentStreamBase.URLStream( new URL(url) );
189         if( contentType != null ) {
190           stream.setContentType( contentType );
191         }
192         streams.add( stream );
193       }
194     }
195     
196     // Handle streaming files
197     strs = params.getParams( CommonParams.STREAM_FILE );
198     if( strs != null ) {
199       if( !enableRemoteStreams ) {
200         throw new SolrException( ErrorCode.BAD_REQUEST, "Remote Streaming is disabled." );
201       }
202       for( final String file : strs ) {
203         ContentStreamBase stream = new ContentStreamBase.FileStream( new File(file) );
204         if( contentType != null ) {
205           stream.setContentType( contentType );
206         }
207         streams.add( stream );
208       }
209     }
210     
211     // Check for streams in the request parameters
212     strs = params.getParams( CommonParams.STREAM_BODY );
213     if( strs != null ) {
214       for( final String body : strs ) {
215         ContentStreamBase stream = new ContentStreamBase.StringStream( body );
216         if( contentType != null ) {
217           stream.setContentType( contentType );
218         }
219         streams.add( stream );
220       }
221     }
222 
223     SolrQueryRequestBase q = new SolrQueryRequestBase(core, params, requestTimer) {
224       @Override
225       public Principal getUserPrincipal() {
226         return req == null ? null : req.getUserPrincipal();
227       }
228     };
229     if( streams != null && streams.size() > 0 ) {
230       q.setContentStreams( streams );
231     }
232     return q;
233   }

244   /**
245    * Given a url-encoded query string (UTF-8), map it into the given map
246    * @param queryString as given from URL
247    * @param map place all parameters in this map
248    */
249   static void parseQueryString(final String queryString, final Map<String,String[]> map) {
250     if (queryString != null && queryString.length() > 0) {
251       try {
252         final int len = queryString.length();
253         // this input stream emulates to get the raw bytes from the URL as passed to servlet container, it disallows any byte > 127 and enforces to %-escape them:
254         final InputStream in = new InputStream() {
255           int pos = 0;
256           @Override
257           public int read() {
258             if (pos < len) {
259               final char ch = queryString.charAt(pos);
260               if (ch > 127) {
261                 throw new SolrException(ErrorCode.BAD_REQUEST, "URLDecoder: The query string contains a not-%-escaped byte > 127 at position " + pos);
262               }
263               pos++;
264               return ch;
265             } else {
266               return -1;
267             }
268           }
269         };
270         parseFormDataContent(in, Long.MAX_VALUE, StandardCharsets.UTF_8, map, true);
271       } catch (IOException ioe) {
272         throw new SolrException(ErrorCode.BAD_REQUEST, ioe);
273       }
274     }
275   }

277   /**
278    * Given a url-encoded form from POST content (as InputStream), map it into the given map.
279    * The given InputStream should be buffered!
280    * @param postContent to be parsed
281    * @param charset to be used to decode resulting bytes after %-decoding
282    * @param map place all parameters in this map
283    */
284   @SuppressWarnings({"fallthrough", "resource"})
285   static long parseFormDataContent(final InputStream postContent, final long maxLen, Charset charset, final Map<String,String[]> map, boolean supportCharsetParam) throws IOException {
286     CharsetDecoder charsetDecoder = supportCharsetParam ? null : getCharsetDecoder(charset);
287     final LinkedList<Object> buffer = supportCharsetParam ? new LinkedList<>() : null;
288     long len = 0L, keyPos = 0L, valuePos = 0L;
289     final ByteArrayOutputStream keyStream = new ByteArrayOutputStream(),
290       valueStream = new ByteArrayOutputStream();
291     ByteArrayOutputStream currentStream = keyStream;
292     for(;;) {
293       int b = postContent.read();
294       switch (b) {
295         case -1: // end of stream
296         case '&': // separator
297           if (keyStream.size() > 0) {
298             final byte[] keyBytes = keyStream.toByteArray(), valueBytes = valueStream.toByteArray();
299             if (Arrays.equals(keyBytes, INPUT_ENCODING_BYTES)) {
300               // we found a charset declaration in the raw bytes
301               if (charsetDecoder != null) {
302                 throw new SolrException(ErrorCode.BAD_REQUEST,
303                   supportCharsetParam ? (
304                     "Query string invalid: duplicate '"+
305                     INPUT_ENCODING_KEY + "' (input encoding) key."
306                   ) : (
307                     "Key '" + INPUT_ENCODING_KEY + "' (input encoding) cannot "+
308                     "be used in POSTed application/x-www-form-urlencoded form data. "+
309                     "To set the input encoding of POSTed form data, use the "+
310                     "'Content-Type' header and provide a charset!"
311                   )
312                 );
313               }
314               // decode the charset from raw bytes
315               charset = Charset.forName(decodeChars(valueBytes, keyPos, getCharsetDecoder(CHARSET_US_ASCII)));
316               charsetDecoder = getCharsetDecoder(charset);
317               // finally decode all buffered tokens
318               decodeBuffer(buffer, map, charsetDecoder);
319             } else if (charsetDecoder == null) {
320               // we have no charset decoder until now, buffer the keys / values for later processing:
321               buffer.add(keyBytes);
322               buffer.add(Long.valueOf(keyPos));
323               buffer.add(valueBytes);
324               buffer.add(Long.valueOf(valuePos));
325             } else {
326               // we already have a charsetDecoder, so we can directly decode without buffering:
327               final String key = decodeChars(keyBytes, keyPos, charsetDecoder),
328                   value = decodeChars(valueBytes, valuePos, charsetDecoder);
329               MultiMapSolrParams.addParam(key.trim(), value, map);
330             }
331           } else if (valueStream.size() > 0) {
332             throw new SolrException(ErrorCode.BAD_REQUEST, "application/x-www-form-urlencoded invalid: missing key");
333           }
334           keyStream.reset();
335           valueStream.reset();
336           keyPos = valuePos = len + 1;
337           currentStream = keyStream;
338           break;
339         case '+': // space replacement
340           currentStream.write(' ');
341           break;
342         case '%': // escape
343           final int upper = digit16(b = postContent.read());
344           len++;
345           final int lower = digit16(b = postContent.read());
346           len++;
347           currentStream.write(((upper << 4) + lower));
348           break;
349         case '=': // kv separator
350           if (currentStream == keyStream) {
351             valuePos = len + 1;
352             currentStream = valueStream;
353             break;
354           }
355           // fall-through
356         default:
357           currentStream.write(b);
358       }
359       if (b == -1) {
360         break;
361       }
362       len++;
363       if (len > maxLen) {
364         throw new SolrException(ErrorCode.BAD_REQUEST, "application/x-www-form-urlencoded content exceeds upload limit of " + (maxLen/1024L) + " KB");
365       }
366     }
367     // if we have not seen a charset declaration, decode the buffer now using the default one (UTF-8 or given via Content-Type):
368     if (buffer != null && !buffer.isEmpty()) {
369       assert charsetDecoder == null;
370       decodeBuffer(buffer, map, getCharsetDecoder(charset));
371     }
372     return len;
373   }
374   
375   private static CharsetDecoder getCharsetDecoder(Charset charset) {
376     return charset.newDecoder()
377       .onMalformedInput(CodingErrorAction.REPORT)
378       .onUnmappableCharacter(CodingErrorAction.REPORT);
379   }
380   
381   private static String decodeChars(byte[] bytes, long position, CharsetDecoder charsetDecoder) {
382     try {
383       return charsetDecoder.decode(ByteBuffer.wrap(bytes)).toString();
384     } catch (CharacterCodingException cce) {
385       throw new SolrException(ErrorCode.BAD_REQUEST,
386         "URLDecoder: Invalid character encoding detected after position " + position +
387         " of query string / form data (while parsing as " + charsetDecoder.charset().name() + ")"
388       );
389     }
390   }
391   
392   private static void decodeBuffer(final LinkedList<Object> input, final Map<String,String[]> map, CharsetDecoder charsetDecoder) {
393     for (final Iterator<Object> it = input.iterator(); it.hasNext(); ) {
394       final byte[] keyBytes = (byte[]) it.next();
395       it.remove();
396       final Long keyPos = (Long) it.next();
397       it.remove();
398       final byte[] valueBytes = (byte[]) it.next();
399       it.remove();
400       final Long valuePos = (Long) it.next();
401       it.remove();
402       MultiMapSolrParams.addParam(decodeChars(keyBytes, keyPos.longValue(), charsetDecoder).trim(),
403           decodeChars(valueBytes, valuePos.longValue(), charsetDecoder), map);
404     }
405   }

407   private static int digit16(int b) {
408     if (b == -1) {
409       throw new SolrException(ErrorCode.BAD_REQUEST, "URLDecoder: Incomplete trailing escape (%) pattern");
410     }
411     if (b >= '0' && b <= '9') {
412       return b - '0';
413     }
414     if (b >= 'A' && b <= 'F') {
415       return b - ('A' - 10);
416     }
417     if (b >= 'a' && b <= 'f') {
418       return b - ('a' - 10);
419     }
420     throw new SolrException(ErrorCode.BAD_REQUEST, "URLDecoder: Invalid digit (" + ((char) b) + ") in escape (%) pattern");
421   }

531   /**
532    * Extract Multipart streams
533    */
534   static class MultipartRequestParser implements SolrRequestParser
535   {
536     private final int uploadLimitKB;
537     
538     public MultipartRequestParser( int limit )
539     {
540       uploadLimitKB = limit;
541     }
542     
543     @Override
544     public SolrParams parseParamsAndFillStreams( 
545         final HttpServletRequest req, ArrayList<ContentStream> streams ) throws Exception
546     {
547       if( !ServletFileUpload.isMultipartContent(req) ) {
548         throw new SolrException( ErrorCode.BAD_REQUEST, "Not multipart content! "+req.getContentType() );
549       }
550       
551       MultiMapSolrParams params = parseQueryString( req.getQueryString() );
552       
553       // Create a factory for disk-based file items
554       DiskFileItemFactory factory = new DiskFileItemFactory();
555 
556       // Set factory constraints
557       // TODO - configure factory.setSizeThreshold(yourMaxMemorySize);
558       // TODO - configure factory.setRepository(yourTempDirectory);
559 
560       // Create a new file upload handler
561       ServletFileUpload upload = new ServletFileUpload(factory);
562       upload.setSizeMax( ((long) uploadLimitKB) * 1024L );
563 
564       // Parse the request
565       List items = upload.parseRequest(req);
566       Iterator iter = items.iterator();
567       while (iter.hasNext()) {
568           FileItem item = (FileItem) iter.next();
569 
570           // If it's a form field, put it in our parameter map
571           if (item.isFormField()) {
572             MultiMapSolrParams.addParam( 
573               item.getFieldName().trim(),
574               item.getString(), params.getMap() );
575           }
576           // Add the stream
577           else { 
578             streams.add( new FileItemContentStream( item ) );
579           }
580       }
581       return params;
582     }
583   }

586   /**
587    * Extract application/x-www-form-urlencoded form data for POST requests
588    */
589   static class FormDataRequestParser implements SolrRequestParser
590   {
591     private static final long WS_MASK=(1L<<' ')|(1L<<'\t')|(1L<<'\r')|(1L<<'\n')|(1L<<'#')|(1L<<'/')|(0x01); // set 1 bit so 0xA0 will be flagged as possible whitespace
592 
593     private final int uploadLimitKB;
594     
595     public FormDataRequestParser( int limit )
596     {
597       uploadLimitKB = limit;
598     }
599 
600     public SolrParams parseParamsAndFillStreams(HttpServletRequest req, ArrayList<ContentStream> streams, InputStream in) throws Exception {
601       final Map<String,String[]> map = new HashMap<>();
602 
603       // also add possible URL parameters and include into the map (parsed using UTF-8):
604       final String qs = req.getQueryString();
605       if (qs != null) {
606         parseQueryString(qs, map);
607       }
608 
609       // may be -1, so we check again later. But if it's already greater we can stop processing!
610       final long totalLength = req.getContentLength();
611       final long maxLength = ((long) uploadLimitKB) * 1024L;
612       if (totalLength > maxLength) {
613         throw new SolrException(ErrorCode.BAD_REQUEST, "application/x-www-form-urlencoded content length (" +
614             totalLength + " bytes) exceeds upload limit of " + uploadLimitKB + " KB");
615       }
616 
617       // get query String from request body, using the charset given in content-type:
618       final String cs = ContentStreamBase.getCharsetFromContentType(req.getContentType());
619       final Charset charset = (cs == null) ? StandardCharsets.UTF_8 : Charset.forName(cs);
620 
621       try {
622         in = FastInputStream.wrap( in == null ? req.getInputStream() : in);
623 
624         final long bytesRead = parseFormDataContent(in, maxLength, charset, map, false);
625         if (bytesRead == 0L && totalLength > 0L) {
626           throw getParameterIncompatibilityException();
627         }
628       } catch (IOException ioe) {
629         throw new SolrException(ErrorCode.BAD_REQUEST, ioe);
630       } catch (IllegalStateException ise) {
631         throw (SolrException) getParameterIncompatibilityException().initCause(ise);
632       } finally {
633         IOUtils.closeWhileHandlingException(in);
634       }
635 
636       return new MultiMapSolrParams(map);
637     }

641     public SolrParams parseParamsAndFillStreams(HttpServletRequest req, ArrayList<ContentStream> streams ) throws Exception {
642       if (!isFormData(req)) {
643         throw new SolrException(ErrorCode.BAD_REQUEST, "Not application/x-www-form-urlencoded content: " + req.getContentType());

645

647     }

649     public static  SolrException getParameterIncompatibilityException() {
650       return new SolrException(ErrorCode.SERVER_ERROR,
651         "Solr requires that request parameters sent using application/x-www-form-urlencoded " +
652         "content-type can be read through the request input stream. Unfortunately, the " +
653         "stream was empty / not available. This may be caused by another servlet filter calling " +
654         "ServletRequest.getParameter*() before SolrDispatchFilter, please remove it."
655       );
656     }

658     public boolean isFormData(HttpServletRequest req) {
659       String contentType = req.getContentType();
660       if (contentType != null) {
661         int idx = contentType.indexOf( ';' );
662         if( idx > 0 ) { // remove the charset definition "; charset=utf-8"
663           contentType = contentType.substring( 0, idx );
664         }
665         contentType = contentType.trim();
666         if( "application/x-www-form-urlencoded".equalsIgnoreCase(contentType)) {
667           return true;
668         }
669       }
670       return false;
671     }

675   /**
676    * The default Logic
677    */
678   static class StandardRequestParser implements SolrRequestParser
679   {
680     MultipartRequestParser multipart;
681     RawRequestParser raw;
682     FormDataRequestParser formdata;
683 
684     StandardRequestParser(MultipartRequestParser multi, RawRequestParser raw, FormDataRequestParser formdata) 
685     {
686       this.multipart = multi;
687       this.raw = raw;
688       this.formdata = formdata;
689     }
690 
691     @Override
692     public SolrParams parseParamsAndFillStreams(final HttpServletRequest req, ArrayList<ContentStream> streams ) throws Exception {
693       String contentType = req.getContentType();
694       String method = req.getMethod(); // No need to uppercase... HTTP verbs are case sensitive
695       String uri = req.getRequestURI();
696       boolean isPost = "POST".equals(method);
697 
698       // SOLR-6787 changed the behavior of a POST without content type.  Previously it would throw an exception,
699       // but now it will use the raw request parser.
700       /***
701        if (contentType == null && isPost) {
702          throw new SolrException(ErrorCode.UNSUPPORTED_MEDIA_TYPE, "Must specify a Content-Type header with POST requests");
703        }
704        ***/
705 
706 
707       // According to previous StandardRequestParser logic (this is a re-written version),
708       // POST was handled normally, but other methods (PUT/DELETE)
709       // were handled by restlet if the URI contained /schema or /config
710       // "handled by restlet" means that we don't attempt to handle any request body here.
711       if (!isPost) {
712         if (contentType == null) {
713           return parseQueryString(req.getQueryString());
714         }
715 
716         // OK, we have a BODY at this point
717 
718         boolean restletPath = false;
719         int idx = uri.indexOf("/schema");
720         if (idx >= 0 && uri.endsWith("/schema") || uri.contains("/schema/")) {
721           restletPath = true;
722         }
723         idx = uri.indexOf("/config");
724         if (idx >= 0 && uri.endsWith("/config") || uri.contains("/config/")) {
725           restletPath = true;
726         }
727 
728         if (restletPath) {
729           return parseQueryString(req.getQueryString());
730         }
731 
732         if ("PUT".equals(method) || "DELETE".equals(method)) {
733           throw new SolrException(ErrorCode.BAD_REQUEST, "Unsupported method: " + method + " for request " + req);
734         }
735       }
736 
737 
738       if (formdata.isFormData(req)) {
739         String userAgent = req.getHeader("User-Agent");
740         boolean isCurl = userAgent != null && userAgent.startsWith("curl/");
741         FastInputStream input = FastInputStream.wrap( req.getInputStream() );
742 
743         if (isCurl) {
744           SolrParams params = autodetect(req, streams, input);
745           if (params != null) return params;
746         }
747 
748         return formdata.parseParamsAndFillStreams(req, streams, input);
749       }
750 
751       if (ServletFileUpload.isMultipartContent(req)) {
752         return multipart.parseParamsAndFillStreams(req, streams);
753       }
754 
755       // some other content-type (json, XML, csv, etc)
756       return raw.parseParamsAndFillStreams(req, streams);
757     }
758   }

761   private static final long WS_MASK=(1L<<' ')|(1L<<'\t')|(1L<<'\r')|(1L<<'\n')|(1L<<'#')|(1L<<'/')|(0x01); // set 1 bit so 0xA0 will be flagged as possible whitespace
762 
763   /** Returns the parameter map if a different content type was auto-detected */
764   private static SolrParams autodetect(HttpServletRequest req, ArrayList<ContentStream> streams, FastInputStream in) throws IOException {
765     String detectedContentType = null;
766     boolean shouldClose = true;
767 
768     try {
769       in.peek();  // should cause some bytes to be read
770       byte[] arr = in.getBuffer();
771       int pos = in.getPositionInBuffer();
772       int end = in.getEndInBuffer();
773 
774       for (int i = pos; i < end - 1; i++) {  // we do "end-1" because we check "arr[i+1]" sometimes in the loop body
775         int ch = arr[i];
776         boolean isWhitespace = ((WS_MASK >> ch) & 0x01) != 0 && (ch <= ' ' || ch == 0xa0);
777         if (!isWhitespace) {
778           // first non-whitespace chars
779           if (ch == '#'                         // single line comment
780               || (ch == '/' && (arr[i + 1] == '/' || arr[i + 1] == '*'))  // single line or multi-line comment
781               || (ch == '{' || ch == '[')       // start of JSON object
782               )
783           {
784             detectedContentType = "application/json";
785           }
786           if (ch == '<') {
787             detectedContentType = "text/xml";
788           }
789           break;
790         }
791       }
792 
793       if (detectedContentType == null) {
794         shouldClose = false;
795         return null;
796       }
797 
798       Long size = null;
799       String v = req.getHeader("Content-Length");
800       if (v != null) {
801         size = Long.valueOf(v);
802       }
803       streams.add(new InputStreamContentStream(in, detectedContentType, size));
804 
805 
806       final Map<String, String[]> map = new HashMap<>();
807       // also add possible URL parameters and include into the map (parsed using UTF-8):
808       final String qs = req.getQueryString();
809       if (qs != null) {
810         parseQueryString(qs, map);
811       }
812 
813       return new MultiMapSolrParams(map);
814 
815     } catch (IOException ioe) {
816       throw new SolrException(ErrorCode.BAD_REQUEST, ioe);
817     } catch (IllegalStateException ise) {
818       throw (SolrException) FormDataRequestParser.getParameterIncompatibilityException().initCause(ise);
819     } finally {
820       if (shouldClose) {
821         IOUtils.closeWhileHandlingException(in);
822       }
823     }
824   }
