50  /**
51   * An adapter from {@link Lookup} API to {@link FSTCompletion}.
52   * 
53   * <p>This adapter differs from {@link FSTCompletion} in that it attempts
54   * to discretize any "weights" as passed from in {@link InputIterator#weight()}
55   * to match the number of buckets. For the rationale for bucketing, see
56   * {@link FSTCompletion}.
57   * 
58   * <p><b>Note:</b>Discretization requires an additional sorting pass.
59   * 
60   * <p>The range of weights for bucketing/ discretization is determined 
61   * by sorting the input by weight and then dividing into
62   * equal ranges. Then, scores within each range are assigned to that bucket. 
63   * 
64   * <p>Note that this means that even large differences in weights may be lost 
65   * during automaton construction, but the overall distinction between "classes"
66   * of weights will be preserved regardless of the distribution of weights. 
67   * 
68   * <p>For fine-grained control over which weights are assigned to which buckets,
69   * use {@link FSTCompletion} directly or {@link TSTLookup}, for example.
70   * 
71   * @see FSTCompletion
72   * @lucene.experimental
73   */
74  public class FSTCompletionLookup extends Lookup implements Accountable {
75    /** 
76     * An invalid bucket count if we're creating an object
77     * of this class from an existing FST.
78     * 
79     * @see #FSTCompletionLookup(FSTCompletion, boolean)
80     */
81    private static int INVALID_BUCKETS_COUNT = -1;
82    
83    /**
84     * Shared tail length for conflating in the created automaton. Setting this
85     * to larger values ({@link Integer#MAX_VALUE}) will create smaller (or minimal) 
86     * automata at the cost of RAM for keeping nodes hash in the {@link FST}. 
87     *  
88     * <p>Empirical pick.
89     */
90    private final static int sharedTailLength = 5;
91  
92    private int buckets;
93    private boolean exactMatchFirst;
94  
95    /**
96     * Automaton used for completions with higher weights reordering.
97     */
98    private FSTCompletion higherWeightsCompletion;
99  
100   /**
101    * Automaton used for normal completions.
102    */
103   private FSTCompletion normalCompletion;
104 
105   /** Number of entries the lookup was built with */
106   private long count = 0;
107 
108   /**
109    * This constructor prepares for creating a suggested FST using the
110    * {@link #build(InputIterator)} method. The number of weight
111    * discretization buckets is set to {@link FSTCompletion#DEFAULT_BUCKETS} and
112    * exact matches are promoted to the top of the suggestions list.
113    */
114   public FSTCompletionLookup() {
115     this(FSTCompletion.DEFAULT_BUCKETS, true);
116   }
117 
118   /**
119    * This constructor prepares for creating a suggested FST using the
120    * {@link #build(InputIterator)} method.
121    * 
122    * @param buckets
123    *          The number of weight discretization buckets (see
124    *          {@link FSTCompletion} for details).
125    * 
126    * @param exactMatchFirst
127    *          If <code>true</code> exact matches are promoted to the top of the
128    *          suggestions list. Otherwise they appear in the order of
129    *          discretized weight and alphabetical within the bucket.
130    */
131   public FSTCompletionLookup(int buckets, boolean exactMatchFirst) {
132     this.buckets = buckets;
133     this.exactMatchFirst = exactMatchFirst;
134   }
135 
136   /**
137    * This constructor takes a pre-built automaton.
138    * 
139    *  @param completion 
140    *          An instance of {@link FSTCompletion}.
141    *  @param exactMatchFirst
142    *          If <code>true</code> exact matches are promoted to the top of the
143    *          suggestions list. Otherwise they appear in the order of
144    *          discretized weight and alphabetical within the bucket.
145    */
146   public FSTCompletionLookup(FSTCompletion completion, boolean exactMatchFirst) {
147     this(INVALID_BUCKETS_COUNT, exactMatchFirst);
148     this.normalCompletion = new FSTCompletion(
149         completion.getFST(), false, exactMatchFirst);
150     this.higherWeightsCompletion =  new FSTCompletion(
151         completion.getFST(), true, exactMatchFirst);
152   }
153 
154   @Override
155   public void build(InputIterator iterator) throws IOException {
156     if (iterator.hasPayloads()) {
157       throw new IllegalArgumentException("this suggester doesn't support payloads");
158     }
159     if (iterator.hasContexts()) {
160       throw new IllegalArgumentException("this suggester doesn't support contexts");
161     }
162     Path tempInput = Files.createTempFile(
163         OfflineSorter.getDefaultTempDir(), FSTCompletionLookup.class.getSimpleName(), ".input");
164     Path tempSorted = Files.createTempFile(
165         OfflineSorter.getDefaultTempDir(), FSTCompletionLookup.class.getSimpleName(), ".sorted");
166 
167     OfflineSorter.ByteSequencesWriter writer = new OfflineSorter.ByteSequencesWriter(tempInput);
168     OfflineSorter.ByteSequencesReader reader = null;
169     ExternalRefSorter sorter = null;
170 
171     // Push floats up front before sequences to sort them. For now, assume they are non-negative.
172     // If negative floats are allowed some trickery needs to be done to find their byte order.
173     boolean success = false;
174     count = 0;
175     try {
176       byte [] buffer = new byte [0];
177       ByteArrayDataOutput output = new ByteArrayDataOutput(buffer);
178       BytesRef spare;
179       while ((spare = iterator.next()) != null) {
180         if (spare.length + 4 >= buffer.length) {
181           buffer = ArrayUtil.grow(buffer, spare.length + 4);
182         }
183 
184         output.reset(buffer);
185         output.writeInt(encodeWeight(iterator.weight()));
186         output.writeBytes(spare.bytes, spare.offset, spare.length);
187         writer.write(buffer, 0, output.getPosition());
188       }
189       writer.close();
190 
191       // We don't know the distribution of scores and we need to bucket them, so we'll sort
192       // and divide into equal buckets.
193       SortInfo info = new OfflineSorter().sort(tempInput, tempSorted);
194       Files.delete(tempInput);
195       FSTCompletionBuilder builder = new FSTCompletionBuilder(
196           buckets, sorter = new ExternalRefSorter(new OfflineSorter()), sharedTailLength);
197 
198       final int inputLines = info.lines;
199       reader = new OfflineSorter.ByteSequencesReader(tempSorted);
200       long line = 0;
201       int previousBucket = 0;
202       int previousScore = 0;
203       ByteArrayDataInput input = new ByteArrayDataInput();
204       BytesRefBuilder tmp1 = new BytesRefBuilder();
205       BytesRef tmp2 = new BytesRef();
206       while (reader.read(tmp1)) {
207         input.reset(tmp1.bytes());
208         int currentScore = input.readInt();
209 
210         int bucket;
211         if (line > 0 && currentScore == previousScore) {
212           bucket = previousBucket;
213         } else {
214           bucket = (int) (line * buckets / inputLines);
215         }
216         previousScore = currentScore;
217         previousBucket = bucket;
218 
219         // Only append the input, discard the weight.
220         tmp2.bytes = tmp1.bytes();
221         tmp2.offset = input.getPosition();
222         tmp2.length = tmp1.length() - input.getPosition();
223         builder.add(tmp2, bucket);
224 
225         line++;
226         count++;
227       }
228 
229       // The two FSTCompletions share the same automaton.
230       this.higherWeightsCompletion = builder.build();
231       this.normalCompletion = new FSTCompletion(
232           higherWeightsCompletion.getFST(), false, exactMatchFirst);
233       
234       success = true;
235     } finally {
236       IOUtils.closeWhileHandlingException(reader, writer, sorter);
237 
238       if (success) {
239         Files.delete(tempSorted);
240       } else {
241         IOUtils.deleteFilesIgnoringExceptions(tempInput, tempSorted);
242       }
243     }
244   }
245   
246   /** weight -&gt; cost */
247   private static int encodeWeight(long value) {
248     if (value < Integer.MIN_VALUE || value > Integer.MAX_VALUE) {
249       throw new UnsupportedOperationException("cannot encode value: " + value);
250     }
251     return (int)value;
252   }
253 
254   @Override
255   public List<LookupResult> lookup(CharSequence key, Set<BytesRef> contexts, boolean higherWeightsFirst, int num) {
256     if (contexts != null) {
257       throw new IllegalArgumentException("this suggester doesn't support contexts");
258     }
259     final List<Completion> completions;
260     if (higherWeightsFirst) {
261       completions = higherWeightsCompletion.lookup(key, num);
262     } else {
263       completions = normalCompletion.lookup(key, num);
264     }
265     
266     final ArrayList<LookupResult> results = new ArrayList<>(completions.size());
267     CharsRefBuilder spare = new CharsRefBuilder();
268     for (Completion c : completions) {
269       spare.copyUTF8Bytes(c.utf8);
270       results.add(new LookupResult(spare.toString(), c.bucket));
271     }
272     return results;
273   }
274 
275   /**
276    * Returns the bucket (weight) as a Long for the provided key if it exists,
277    * otherwise null if it does not.
278    */
279   public Object get(CharSequence key) {
280     final int bucket = normalCompletion.getBucket(key);
281     return bucket == -1 ? null : Long.valueOf(bucket);
282   }

285   @Override
286   public synchronized boolean store(DataOutput output) throws IOException {
287     output.writeVLong(count);
288     if (this.normalCompletion == null || normalCompletion.getFST() == null) 
289       return false;
290     normalCompletion.getFST().save(output);
291     return true;
292   }
293 
294   @Override
295   public synchronized boolean load(DataInput input) throws IOException {
296     count = input.readVLong();
297     this.higherWeightsCompletion = new FSTCompletion(new FST<>(
298         input, NoOutputs.getSingleton()));
299     this.normalCompletion = new FSTCompletion(
300         higherWeightsCompletion.getFST(), false, exactMatchFirst);
301     return true;
302   }
303 
304   @Override
305   public long ramBytesUsed() {
306     long mem = RamUsageEstimator.shallowSizeOf(this) + RamUsageEstimator.shallowSizeOf(normalCompletion) + RamUsageEstimator.shallowSizeOf(higherWeightsCompletion);
307     if (normalCompletion != null) {
308       mem += normalCompletion.getFST().ramBytesUsed();
309     }
310     if (higherWeightsCompletion != null && (normalCompletion == null || normalCompletion.getFST() != higherWeightsCompletion.getFST())) {
311       // the fst should be shared between the 2 completion instances, don't count it twice
312       mem += higherWeightsCompletion.getFST().ramBytesUsed();
313     }
314     return mem;
315   }
316 
317   @Override
318   public Collection<Accountable> getChildResources() {
319     List<Accountable> resources = new ArrayList<>();
320     if (normalCompletion != null) {
321       resources.add(Accountables.namedAccountable("fst", normalCompletion.getFST()));
322     }
323     if (higherWeightsCompletion != null && (normalCompletion == null || normalCompletion.getFST() != higherWeightsCompletion.getFST())) {
324       resources.add(Accountables.namedAccountable("higher weights fst", higherWeightsCompletion.getFST()));
325     }
326     return Collections.unmodifiableList(resources);
327   }
328 
329   @Override
330   public long getCount() {
331     return count;
332   }
