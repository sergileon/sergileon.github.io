69    @Override
70    public void init(NamedList args) {
71      SolrParams params = SolrParams.toSolrParams(args);
72      isMutable = params.getBool("mutable", false);
73      args.remove("mutable");
74      managedSchemaResourceName = params.get(MANAGED_SCHEMA_RESOURCE_NAME, DEFAULT_MANAGED_SCHEMA_RESOURCE_NAME);
75      args.remove(MANAGED_SCHEMA_RESOURCE_NAME);
76      if (SCHEMA_DOT_XML.equals(managedSchemaResourceName)) {
77        String msg = MANAGED_SCHEMA_RESOURCE_NAME + " can't be '" + SCHEMA_DOT_XML + "'";
78        log.error(msg);
79        throw new SolrException(ErrorCode.SERVER_ERROR, msg);
80      }
81      
82      if (args.size() > 0) {
83        String msg = "Unexpected arg(s): " + args;
84        log.error(msg);
85        throw new SolrException(ErrorCode.SERVER_ERROR, msg);
86      }
87    }

89    /**
90     * First, try to locate the managed schema file named in the managedSchemaResourceName
91     * param. If the managed schema file exists and is accessible, it is used to instantiate
92     * an IndexSchema.
93     *
94     * If the managed schema file can't be found, the resource named by the resourceName
95     * parameter is used to instantiate an IndexSchema.
96     *
97     * Once the IndexSchema is instantiated, if the managed schema file does not exist,
98     * the instantiated IndexSchema is persisted to the managed schema file named in the
99     * managedSchemaResourceName param, in the directory given by 
100    * {@link org.apache.solr.core.SolrResourceLoader#getConfigDir()}, or if configs are
101    * in ZooKeeper, under {@link org.apache.solr.cloud.ZkSolrResourceLoader#configSetZkPath}.
102    *
103    * After the managed schema file is persisted, the original schema file is
104    * renamed by appending the extension named in {@link #UPGRADED_SCHEMA_EXTENSION}.
105    */
106   @Override
107   public ManagedIndexSchema create(String resourceName, SolrConfig config) {
108     this.resourceName = resourceName;
109     this.config = config;
110     this.loader = config.getResourceLoader();
111     InputStream schemaInputStream = null;
112 
113     if (null == resourceName) {
114       resourceName = IndexSchema.DEFAULT_SCHEMA_FILE;
115     }
116 
117     int schemaZkVersion = -1;
118     if ( ! (loader instanceof ZkSolrResourceLoader)) {
119       schemaInputStream = readSchemaLocally();
120     } else { // ZooKeeper
121       final ZkSolrResourceLoader zkLoader = (ZkSolrResourceLoader)loader;
122       final SolrZkClient zkClient = zkLoader.getZkController().getZkClient();
123       final String managedSchemaPath = zkLoader.getConfigSetZkPath() + "/" + managedSchemaResourceName;
124       Stat stat = new Stat();
125       try {
126         // Attempt to load the managed schema
127         byte[] data = zkClient.getData(managedSchemaPath, null, stat, true);
128         schemaZkVersion = stat.getVersion();
129         schemaInputStream = new ByteArrayInputStream(data);
130         loadedResource = managedSchemaResourceName;
131         warnIfNonManagedSchemaExists();
132       } catch (InterruptedException e) {
133         // Restore the interrupted status
134         Thread.currentThread().interrupt();
135         log.warn("", e);
136       } catch (KeeperException.NoNodeException e) {
137         log.info("The schema is configured as managed, but managed schema resource " + managedSchemaResourceName
138                 + " not found - loading non-managed schema " + resourceName + " instead");
139       } catch (KeeperException e) {
140         String msg = "Error attempting to access " + managedSchemaPath;
141         log.error(msg, e);
142         throw new SolrException(ErrorCode.SERVER_ERROR, msg, e);
143       }
144       if (null == schemaInputStream) {
145         // The managed schema file could not be found - load the non-managed schema
146         try {
147           schemaInputStream = loader.openSchema(resourceName);
148           loadedResource = resourceName;
149           shouldUpgrade = true;
150         } catch (Exception e) {
151           try {
152             // Retry to load the managed schema, in case it was created since the first attempt
153             byte[] data = zkClient.getData(managedSchemaPath, null, stat, true);
154             schemaZkVersion = stat.getVersion();
155             schemaInputStream = new ByteArrayInputStream(data);
156             warnIfNonManagedSchemaExists();
157           } catch (Exception e1) {
158             if (e1 instanceof InterruptedException) {
159               Thread.currentThread().interrupt(); // Restore the interrupted status
160             }
161             final String msg = "Error loading both non-managed schema '" + resourceName + "' and managed schema '"
162                              + managedSchemaResourceName + "'";
163             log.error(msg, e);
164             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, msg, e);
165           }
166         }
167       }
168     }
169     InputSource inputSource = new InputSource(schemaInputStream);
170     inputSource.setSystemId(SystemIdResolver.createSystemIdFromResourceName(loadedResource));
171     try {
172       schema = new ManagedIndexSchema(config, loadedResource, inputSource, isMutable, 
173                                       managedSchemaResourceName, schemaZkVersion, getSchemaUpdateLock());
174     } catch (KeeperException e) {
175       final String msg = "Error instantiating ManagedIndexSchema";
176       log.error(msg, e);
177       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, msg, e);
178     } catch (InterruptedException e) {
179       // Restore the interrupted status
180       Thread.currentThread().interrupt();
181       log.warn("", e);
182     }
183 
184     if (shouldUpgrade) {
185       // Persist the managed schema if it doesn't already exist
186       upgradeToManagedSchema();
187     }
188 
189     return schema;
190   }

192   private InputStream readSchemaLocally() {
193     InputStream schemaInputStream = null;
194     try {
195       // Attempt to load the managed schema
196       schemaInputStream = loader.openSchema(managedSchemaResourceName);
197       loadedResource = managedSchemaResourceName;
198       warnIfNonManagedSchemaExists();
199     } catch (IOException e) {
200       log.info("The schema is configured as managed, but managed schema resource " + managedSchemaResourceName
201               + " not found - loading non-managed schema " + resourceName + " instead");
202     }
203     if (null == schemaInputStream) {
204       // The managed schema file could not be found - load the non-managed schema
205       try {
206         schemaInputStream = loader.openSchema(resourceName);
207         loadedResource = resourceName;
208         shouldUpgrade = true;
209       } catch (Exception e) {
210         final String msg = "Error loading both non-managed schema '" + resourceName + "' and managed schema '"
211                          + managedSchemaResourceName + "'";
212         log.error(msg, e);
213         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, msg, e);
214       }
215     }
216     return schemaInputStream; 
217   }

219   /**
220    * Return whether a non-managed schema exists, either in local storage or on ZooKeeper. 
221    */
222   private void warnIfNonManagedSchemaExists() {
223     if ( ! resourceName.equals(managedSchemaResourceName)) {
224       boolean exists = false;
225       SolrResourceLoader loader = config.getResourceLoader();
226       if (loader instanceof ZkSolrResourceLoader) {
227         ZkSolrResourceLoader zkLoader = (ZkSolrResourceLoader)loader;
228         String nonManagedSchemaPath = zkLoader.getConfigSetZkPath() + "/" + resourceName;
229         try {
230           exists = zkLoader.getZkController().pathExists(nonManagedSchemaPath);
231         } catch (InterruptedException e) {
232           Thread.currentThread().interrupt(); // Restore the interrupted status
233           log.warn("", e); // Log as warning and suppress the exception 
234         } catch (KeeperException e) {
235           // log as warning and suppress the exception
236           log.warn("Error checking for the existence of the non-managed schema " + resourceName, e);
237         }
238       } else { // Config is not in ZooKeeper
239         InputStream nonManagedSchemaInputStream = null;
240         try {
241           nonManagedSchemaInputStream = loader.openSchema(resourceName);
242           if (null != nonManagedSchemaInputStream) {
243             exists = true;
244           }
245         } catch (IOException e) {
246           // This is expected when the non-managed schema does not exist
247         } finally {
248           IOUtils.closeQuietly(nonManagedSchemaInputStream);
249         }
250       }
251       if (exists) {
252         log.warn("The schema has been upgraded to managed, but the non-managed schema " + resourceName
253                 + " is still loadable.  PLEASE REMOVE THIS FILE.");
254       }
255     }
256   }

258   /**
259    * Persist the managed schema and rename the non-managed schema 
260    * by appending {@link #UPGRADED_SCHEMA_EXTENSION}.
261    *
262    * Failure to rename the non-managed schema will be logged as a warning,
263    * and no exception will be thrown.
264    */
265   private void upgradeToManagedSchema() {
266     SolrResourceLoader loader = config.getResourceLoader();
267     if (loader instanceof ZkSolrResourceLoader) {
268       zkUgradeToManagedSchema();
269     } else {
270       // Configs are not on ZooKeeper
271       schema.persistManagedSchema(true);  // Only create it - don't update it if it already exists
272 
273       // After successfully persisting the managed schema, rename the non-managed
274       // schema file by appending UPGRADED_SCHEMA_EXTENSION to its name.
275 
276       if (resourceName.equals(managedSchemaResourceName)) {
277         log.info("On upgrading to managed schema, did not rename non-managed schema '"
278             + resourceName + "' because it's the same as the managed schema's name.");
279       } else {
280         final File nonManagedSchemaFile = locateConfigFile(resourceName);
281         if (null == nonManagedSchemaFile) {
282           // Don't throw an exception for failure to rename the non-managed schema
283           log.warn("On upgrading to managed schema, did not rename non-managed schema "
284               + resourceName + " because it's neither an absolute file "
285               + "nor under SolrConfig.getConfigDir() or the current directory."
286               + "  PLEASE REMOVE THIS FILE.");
287         } else {
288           File upgradedSchemaFile = new File(nonManagedSchemaFile + UPGRADED_SCHEMA_EXTENSION);
289           if (nonManagedSchemaFile.renameTo(upgradedSchemaFile)) {
290             // Set the resource name to the managed schema so that the CoreAdminHandler returns a findable filename 
291             schema.setResourceName(managedSchemaResourceName);
292 
293             log.info("After upgrading to managed schema, renamed the non-managed schema "
294                 + nonManagedSchemaFile + " to " + upgradedSchemaFile);
295           } else {
296             // Don't throw an exception for failure to rename the non-managed schema
297             log.warn("Can't rename " + nonManagedSchemaFile.toString() + " to "
298                 + upgradedSchemaFile.toString() + " - PLEASE REMOVE THIS FILE.");
299           }
300         }
301       }
302     }
303   }

305   /**
306    * Finds any resource by its name on the filesystem.  The classpath is not consulted.
307    *
308    * If the resource is not absolute, the resource is sought in $configDir and then in the current directory.
309    *
310    *@return the File for the named resource, or null if it can't be found
311    */
312   private File locateConfigFile(String resource) {
313     String location = config.getResourceLoader().resourceLocation(resource);
314     if (location.equals(resource) || location.startsWith("classpath:"))
315       return null;
316     return new File(location);
317   }

319   /**
320    * Persist the managed schema to ZooKeeper and rename the non-managed schema 
321    * by appending {@link #UPGRADED_SCHEMA_EXTENSION}.
322    *
323    * Failure to rename the non-managed schema will be logged as a warning,
324    * and no exception will be thrown.
325    */
326   private void zkUgradeToManagedSchema() {
327     schema.persistManagedSchemaToZooKeeper(true); // Only create, don't update it if it already exists
328 
329     // After successfully persisting the managed schema, rename the non-managed
330     // schema znode by appending UPGRADED_SCHEMA_EXTENSION to its name.
331 
332     if (resourceName.equals(managedSchemaResourceName)) {
333       log.info("On upgrading to managed schema, did not rename non-managed schema "
334           + resourceName + " because it's the same as the managed schema's name.");
335     } else {
336       // Rename the non-managed schema znode in ZooKeeper
337       ZkSolrResourceLoader zkLoader = (ZkSolrResourceLoader)loader;
338       final String nonManagedSchemaPath = zkLoader.getConfigSetZkPath() + "/" + resourceName;
339       try {
340         ZkController zkController = zkLoader.getZkController();
341         ZkCmdExecutor zkCmdExecutor = new ZkCmdExecutor(zkController.getClientTimeout());
342         if (zkController.pathExists(nonManagedSchemaPath)) {
343           // First, copy the non-managed schema znode content to the upgraded schema znode
344           byte[] bytes = zkController.getZkClient().getData(nonManagedSchemaPath, null, null, true);
345           final String upgradedSchemaPath = nonManagedSchemaPath + UPGRADED_SCHEMA_EXTENSION;
346           zkCmdExecutor.ensureExists(upgradedSchemaPath, zkController.getZkClient());
347           zkController.getZkClient().setData(upgradedSchemaPath, bytes, true);
348           // Then delete the non-managed schema znode
349           zkController.getZkClient().delete(nonManagedSchemaPath, -1, true);
350 
351           // Set the resource name to the managed schema so that the CoreAdminHandler returns a findable filename 
352           schema.setResourceName(managedSchemaResourceName);
353 
354           log.info("After upgrading to managed schema in ZooKeeper, renamed the non-managed schema "
355                   + nonManagedSchemaPath + " to " + upgradedSchemaPath);
356         } else {
357           log.info("After upgrading to managed schema in ZooKeeper, the non-managed schema "
358                   + nonManagedSchemaPath + " no longer exists.");
359         }
360       } catch (Exception e) {
361         if (e instanceof InterruptedException) {
362           Thread.currentThread().interrupt(); // Restore the interrupted status
363         }
364         final String msg = "Error persisting managed schema resource " + managedSchemaResourceName;
365         log.warn(msg, e); // Log as warning and suppress the exception
366       }
367     }
368   }

373   @Override
374   public void inform(SolrCore core) {
375     this.core = core;
376     if (loader instanceof ZkSolrResourceLoader) {
377       this.zkIndexSchemaReader = new ZkIndexSchemaReader(this);
378       ZkSolrResourceLoader zkLoader = (ZkSolrResourceLoader)loader;
379       zkLoader.setZkIndexSchemaReader(this.zkIndexSchemaReader);
380     } else {
381       this.zkIndexSchemaReader = null;
382     }
383   }
