103   public static class DebugServlet extends HttpServlet {
104     public static void clear() {

116     public static Map<String,String[]> parameters = null;
117     public static String queryString = null;

119     
120     public static void setErrorCode(Integer code) {
121       errorCode = code;
122     }
123     
124     @Override
125     protected void doDelete(HttpServletRequest req, HttpServletResponse resp)
126         throws ServletException, IOException {
127       lastMethod = "delete";
128       recordRequest(req, resp);
129     }
130     
131     @Override
132     protected void doGet(HttpServletRequest req, HttpServletResponse resp)
133         throws ServletException, IOException {
134       lastMethod = "get";
135       recordRequest(req, resp);

137     
138     @Override
139     protected void doHead(HttpServletRequest req, HttpServletResponse resp)
140         throws ServletException, IOException {
141       lastMethod = "head";

144     
145     private void setHeaders(HttpServletRequest req) {
146       Enumeration<String> headerNames = req.getHeaderNames();
147       headers = new HashMap<>();
148       while (headerNames.hasMoreElements()) {
149         final String name = headerNames.nextElement();
150         headers.put(name, req.getHeader(name));
151       }
152     }
153 
154     @SuppressForbidden(reason = "fake servlet only")
155     private void setParameters(HttpServletRequest req) {
156       parameters = req.getParameterMap();
157     }
158 
159     private void setQueryString(HttpServletRequest req) {
160       queryString = req.getQueryString();
161     }
162 
163     private void setCookies(HttpServletRequest req) {
164       javax.servlet.http.Cookie[] ck = req.getCookies();
165       cookies = req.getCookies();
166     }
167 
168     @Override
169     protected void doPost(HttpServletRequest req, HttpServletResponse resp)
170         throws ServletException, IOException {
171       lastMethod = "post";
172       recordRequest(req, resp);
173     }
174 
175     @Override
176     protected void doPut(HttpServletRequest req, HttpServletResponse resp)
177         throws ServletException, IOException {
178       lastMethod = "put";
179       recordRequest(req, resp);
180     }
181     
182     private void recordRequest(HttpServletRequest req, HttpServletResponse resp) {
183       setHeaders(req);
184       setParameters(req);
185       setQueryString(req);
186       setCookies(req);
187       if (null != errorCode) {
188         try { 
189           resp.sendError(errorCode); 
190         } catch (IOException e) {
191           throw new RuntimeException("sendError IO fail in DebugServlet", e);
192         }
193       }
194     }

525   @Test
526   public void testCompression() throws Exception {
527     try (HttpSolrClient client = new HttpSolrClient(jetty.getBaseUrl().toString() + "/debug/foo")) {
528       SolrQuery q = new SolrQuery("*:*");
529       
530       // verify request header gets set
531       DebugServlet.clear();
532       try {
533         client.query(q);
534       } catch (ParseException ignored) {}
535       assertNull(DebugServlet.headers.get("Accept-Encoding"));
536       client.setAllowCompression(true);
537       try {
538         client.query(q);
539       } catch (ParseException ignored) {}
540       assertNotNull(DebugServlet.headers.get("Accept-Encoding"));
541       client.setAllowCompression(false);
542       try {
543         client.query(q);
544       } catch (ParseException ignored) {}
545       assertNull(DebugServlet.headers.get("Accept-Encoding"));
546     }
547     
548     // verify server compresses output
549     HttpGet get = new HttpGet(jetty.getBaseUrl().toString() + "/collection1" +
550                               "/select?q=foo&wt=xml");
551     get.setHeader("Accept-Encoding", "gzip");
552     CloseableHttpClient httpclient = HttpClientUtil.createClient(null);
553     HttpEntity entity = null;
554     try {
555       HttpResponse response = httpclient.execute(get);
556       entity = response.getEntity();
557       Header ceheader = entity.getContentEncoding();
558       assertEquals("gzip", ceheader.getValue());
559     } finally {
560       if (entity != null) {
561         entity.getContent().close();
562       }
563       httpclient.close();
564     }
565     
566     // verify compressed response can be handled
567     try (HttpSolrClient client = new HttpSolrClient(jetty.getBaseUrl().toString() + "/collection1")) {
568       client.setAllowCompression(true);
569       SolrQuery q = new SolrQuery("foo");
570       QueryResponse response = client.query(q);
571       assertEquals(0, response.getStatus());
572     }
573   }

663   /**
664    * An interceptor setting a cookie
665    */
666   HttpRequestInterceptor cookieSettingRequestInterceptor = new HttpRequestInterceptor() {    
667     @Override
668     public void process(HttpRequest request, HttpContext context) throws HttpException,
669     IOException {
670       BasicClientCookie cookie = new BasicClientCookie(cookieName, cookieValue);
671       cookie.setVersion(0);
672       cookie.setPath("/");
673       cookie.setDomain(jetty.getBaseUrl().getHost());
674 
675       CookieStore cookieStore = new BasicCookieStore();        
676       CookieSpecRegistry registry = (CookieSpecRegistry) context.getAttribute(ClientContext.COOKIESPEC_REGISTRY);
677       String policy = HttpClientParams.getCookiePolicy(request.getParams());
678       CookieSpec cookieSpec = registry.getCookieSpec(policy, request.getParams());
679       // Add the cookies to the request
680       List<Header> headers = cookieSpec.formatCookies(Collections.singletonList((Cookie)cookie));
681       for (Header header : headers) {
682         request.addHeader(header);
683       }
684       context.setAttribute(ClientContext.COOKIE_STORE, cookieStore);
685       context.setAttribute(ClientContext.COOKIE_SPEC, cookieSpec);
686     }
687   };

698     HttpClientUtil.addRequestInterceptor(changeRequestInterceptor);
699     HttpClientUtil.addRequestInterceptor(cookieSettingRequestInterceptor);

701     try(HttpSolrClient server = new HttpSolrClient(jetty.getBaseUrl().toString() +

703 
704       SolrQuery q = new SolrQuery("foo");
705       q.setParam("a", "\u1234");

707         server.query(q, random().nextBoolean()?METHOD.POST:METHOD.GET);
708       } catch (Throwable t) {}
709 
710       // Assert cookies from UseContextCallback 
711       assertNotNull(DebugServlet.cookies);
712       boolean foundCookie = false;

726     } catch (IOException ex) {
727       throw new RuntimeException(ex);
728     } finally {
729       HttpClientUtil.removeRequestInterceptor(changeRequestInterceptor);
730       HttpClientUtil.removeRequestInterceptor(cookieSettingRequestInterceptor);

750   private void verifyServletState(HttpSolrClient client, SolrRequest request) {
751     // check query String
752     Iterator<String> paramNames = request.getParams().getParameterNamesIterator();
753     while (paramNames.hasNext()) {
754       String name = paramNames.next();
755       String [] values = request.getParams().getParams(name);
756       if (values != null) {
757         for (String value : values) {
758           boolean shouldBeInQueryString = client.getQueryParams().contains(name)
759             || (request.getQueryParams() != null && request.getQueryParams().contains(name));
760           assertEquals(shouldBeInQueryString, DebugServlet.queryString.contains(name + "=" + value));
761           // in either case, it should be in the parameters
762           assertNotNull(DebugServlet.parameters.get(name));
763           assertEquals(1, DebugServlet.parameters.get(name).length);
764           assertEquals(value, DebugServlet.parameters.get(name)[0]);
765         }
766       }
767     }
768   }

770   @Test
771   public void testQueryString() throws Exception {
772 
773     try (HttpSolrClient client = new HttpSolrClient(jetty.getBaseUrl().toString() + "/debug/foo")) {
774       // test without request query params
775       DebugServlet.clear();
776       client.setQueryParams(setOf("serverOnly"));
777       UpdateRequest req = new UpdateRequest();
778       setReqParamsOf(req, "serverOnly", "notServer");
779       try {
780         client.request(req);
781       } catch (ParseException ignored) {}
782       verifyServletState(client, req);
783   
784       // test without server query params
785       DebugServlet.clear();
786       client.setQueryParams(setOf());
787       req = new UpdateRequest();
788       req.setQueryParams(setOf("requestOnly"));
789       setReqParamsOf(req, "requestOnly", "notRequest");
790       try {
791         client.request(req);
792       } catch (ParseException ignored) {}
793       verifyServletState(client, req);
794   
795       // test with both request and server query params
796       DebugServlet.clear();
797       req = new UpdateRequest();
798       client.setQueryParams(setOf("serverOnly", "both"));
799       req.setQueryParams(setOf("requestOnly", "both"));
800       setReqParamsOf(req, "serverOnly", "requestOnly", "both", "neither");
801        try {
802         client.request(req);
803        } catch (ParseException ignored) {}
804       verifyServletState(client, req);
805   
806       // test with both request and server query params with single stream
807       DebugServlet.clear();
808       req = new UpdateRequest();
809       req.add(new SolrInputDocument());
810       client.setQueryParams(setOf("serverOnly", "both"));
811       req.setQueryParams(setOf("requestOnly", "both"));
812       setReqParamsOf(req, "serverOnly", "requestOnly", "both", "neither");
813        try {
814         client.request(req);
815        } catch (ParseException ignored) {}
816       // NOTE: single stream requests send all the params
817       // as part of the query string.  So add "neither" to the request
818       // so it passes the verification step.
819       req.setQueryParams(setOf("requestOnly", "both", "neither"));
820       verifyServletState(client, req);
821     }
