61    public SolrZkServer(String zkRun, String zkHost, String dataHome, String confHome, int solrPort) {
62      this.zkRun = zkRun;
63      this.zkHost = zkHost;
64      this.dataHome = dataHome;
65      this.confHome = confHome;
66      this.solrPort = solrPort;
67    }
68  
69    public String getClientString() {
70      if (zkHost != null) return zkHost;
71      
72      if (zkProps == null) return null;
73  
74      // if the string wasn't passed as zkHost, then use the standalone server we started
75      if (zkRun == null) return null;
76      return "localhost:" + zkProps.getClientPortAddress().getPort();
77    }
78  
79    public void parseConfig() {
80      if (zkProps == null) {
81        zkProps = new SolrZkServerProps();
82        // set default data dir
83        // TODO: use something based on IP+port???  support ensemble all from same solr home?
84        zkProps.setDataDir(dataHome);
85        zkProps.zkRun = zkRun;
86        zkProps.solrPort = Integer.toString(solrPort);
87      }
88      
89      try {
90        props = SolrZkServerProps.getProperties(confHome + '/' + "zoo.cfg");
91        SolrZkServerProps.injectServers(props, zkRun, zkHost);
92        zkProps.parseProperties(props);
93        if (zkProps.getClientPortAddress() == null) {
94          zkProps.setClientPort(solrPort + 1000);
95        }
96      } catch (QuorumPeerConfig.ConfigException | IOException e) {
97        if (zkRun != null)
98          throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, e);
99      }
100   }
101 
102   public Map<Long, QuorumPeer.QuorumServer> getServers() {
103     return zkProps.getServers();
104   }
105 
106   public void start() {
107     if (zkRun == null) return;
108 
109     zkThread = new Thread() {
110       @Override
111       public void run() {
112         try {
113           if (zkProps.getServers().size() > 1) {
114             QuorumPeerMain zkServer = new QuorumPeerMain();
115             zkServer.runFromConfig(zkProps);
116           } else {
117             ServerConfig sc = new ServerConfig();
118             sc.readFrom(zkProps);
119             ZooKeeperServerMain zkServer = new ZooKeeperServerMain();
120             zkServer.runFromConfig(sc);
121           }
122           log.info("ZooKeeper Server exited.");
123         } catch (Exception e) {
124           log.error("ZooKeeper Server ERROR", e);
125           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, e);
126         }
127       }
128     };
129 
130     if (zkProps.getServers().size() > 1) {
131       log.info("STARTING EMBEDDED ENSEMBLE ZOOKEEPER SERVER at port " + zkProps.getClientPortAddress().getPort());
132     } else {
133       log.info("STARTING EMBEDDED STANDALONE ZOOKEEPER SERVER at port " + zkProps.getClientPortAddress().getPort());
134     }
135 
136     zkThread.setDaemon(true);
137     zkThread.start();
138     try {
139       Thread.sleep(500); // pause for ZooKeeper to start
140     } catch (Exception e) {
141       log.error("STARTING ZOOKEEPER", e);
142     }
143   }
144 
145   public void stop() {
146     if (zkRun == null) return;
147     zkThread.interrupt();
148   }
149 }

155 // zoo.cfg (which validates that there is a dataDir)
156 class SolrZkServerProps extends QuorumPeerConfig {
157   private static final Logger LOG = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());

162   /**
163    * Parse a ZooKeeper configuration file
164    * @param path the patch of the configuration file
165    * @throws ConfigException error processing configuration
166    */
167   public static Properties getProperties(String path) throws ConfigException {
168     File configFile = new File(path);
169 
170     LOG.info("Reading configuration from: " + configFile);
171 
172     try {
173       if (!configFile.exists()) {
174         throw new IllegalArgumentException(configFile.toString()
175             + " file is missing");
176       }
177 
178       Properties cfg = new Properties();
179       FileInputStream in = new FileInputStream(configFile);
180       try {
181         cfg.load(new InputStreamReader(in, StandardCharsets.UTF_8));
182       } finally {
183         in.close();
184       }
185 
186       return cfg;
187 
188     } catch (IOException | IllegalArgumentException e) {
189       throw new ConfigException("Error processing " + path, e);
190     }
191   }

194   // Adds server.x if they don't exist, based on zkHost if it does exist.
195   // Given zkHost=localhost:1111,localhost:2222 this will inject
196   // server.0=localhost:1112:1113
197   // server.1=localhost:2223:2224
198   public static void injectServers(Properties props, String zkRun, String zkHost) {
199 
200     // if clientPort not already set, use zkRun
201     if (zkRun != null && props.getProperty("clientPort")==null) {
202       int portIdx = zkRun.lastIndexOf(':');
203       if (portIdx > 0) {
204         String portStr = zkRun.substring(portIdx+1);
205         props.setProperty("clientPort", portStr);
206       }
207     }
208 
209     boolean hasServers = hasServers(props);
210 
211     if (!hasServers && zkHost != null) {
212       int alg = Integer.parseInt(props.getProperty("electionAlg","3").trim());
213       String[] hosts = zkHost.split(",");
214       int serverNum = 0;
215       for (String hostAndPort : hosts) {
216         hostAndPort = hostAndPort.trim();
217         int portIdx = hostAndPort.lastIndexOf(':');
218         String clientPortStr = hostAndPort.substring(portIdx+1);
219         int clientPort = Integer.parseInt(clientPortStr);
220         String host = hostAndPort.substring(0,portIdx);
221 
222         String serverStr = host + ':' + (clientPort+1);
223         // zk leader election algorithms other than 0 need an extra port for leader election.
224         if (alg != 0) {
225           serverStr = serverStr + ':' + (clientPort+2);
226         }
227 
228         props.setProperty("server."+serverNum, serverStr);
229         serverNum++;
230       }
231     }
232   }
233 
234   public static boolean hasServers(Properties props) {
235     for (Object key : props.keySet())
236       if (((String)key).startsWith("server."))
237         return true;
238     return false;
239   }

241   // called by the modified version of parseProperties
242   // when the myid file is missing.
243   public Long getMyServerId() {
244     if (zkRun == null && solrPort == null) return null;
245 
246     Map<Long, QuorumPeer.QuorumServer> slist = getServers();
247 
248     String myHost = "localhost";
249     InetSocketAddress thisAddr = null;
250 
251     if (zkRun != null && zkRun.length()>0) {
252       String parts[] = zkRun.split(":");
253       myHost = parts[0];
254       thisAddr = new InetSocketAddress(myHost, Integer.parseInt(parts[1]) + 1);
255     } else {
256       // default to localhost:<solrPort+1001>
257       thisAddr = new InetSocketAddress(myHost, Integer.parseInt(solrPort)+1001);
258     }
259 
260 
261     // first try a straight match by host
262     Long me = null;
263     boolean multiple = false;
264     int port = 0;
265     for (QuorumPeer.QuorumServer server : slist.values()) {
266       if (server.addr.getHostName().equals(myHost)) {
267         multiple = me!=null;
268         me = server.id;
269         port = server.addr.getPort();
270       }
271     }
272 
273     if (!multiple) {
274       // only one host matched... assume it's me.
275       setClientPort(port - 1);
276       return me;
277     }
278 
279     if (me == null) {
280       // no hosts matched.
281       return null;
282     }
283 
284 
285     // multiple matches... try to figure out by port.
286     for (QuorumPeer.QuorumServer server : slist.values()) {
287       if (server.addr.equals(thisAddr)) {
288         if (clientPortAddress == null || clientPortAddress.getPort() <= 0)
289           setClientPort(server.addr.getPort() - 1);
290         return server.id;
291       }
292     }
293 
294     return null;
295   }

303   public void setClientPort(int clientPort) {
304     if (clientPortAddress != null) {
305       try {
306         this.clientPortAddress = new InetSocketAddress(
307                 InetAddress.getByName(clientPortAddress.getHostName()), clientPort);
308       } catch (UnknownHostException e) {
309         throw new RuntimeException(e);
310       }
311     } else {
312       this.clientPortAddress = new InetSocketAddress(clientPort);
313     }
314   }

317   // NOTE: copied from ZooKeeper 3.2
318   /**
319    * Parse config from a Properties.
320    * @param zkProp Properties to parse from.
321    */
322   @Override
323   public void parseProperties(Properties zkProp)
324       throws IOException, ConfigException {
325     for (Entry<Object, Object> entry : zkProp.entrySet()) {
326       String key = entry.getKey().toString().trim();
327       String value = entry.getValue().toString().trim();
328       if (key.equals("dataDir")) {
329         dataDir = value;
330       } else if (key.equals("dataLogDir")) {
331         dataLogDir = value;
332       } else if (key.equals("clientPort")) {
333         setClientPort(Integer.parseInt(value));
334       } else if (key.equals("tickTime")) {
335         tickTime = Integer.parseInt(value);
336       } else if (key.equals("initLimit")) {
337         initLimit = Integer.parseInt(value);
338       } else if (key.equals("syncLimit")) {
339         syncLimit = Integer.parseInt(value);
340       } else if (key.equals("electionAlg")) {
341         electionAlg = Integer.parseInt(value);
342       } else if (key.equals("maxClientCnxns")) {
343         maxClientCnxns = Integer.parseInt(value);
344       } else if (key.startsWith("server.")) {
345         int dot = key.indexOf('.');
346         long sid = Long.parseLong(key.substring(dot + 1));
347         String parts[] = value.split(":");
348         if ((parts.length != 2) && (parts.length != 3)) {
349           LOG.error(value
350               + " does not have the form host:port or host:port:port");
351         }
352         InetSocketAddress addr = new InetSocketAddress(parts[0],
353             Integer.parseInt(parts[1]));
354         if (parts.length == 2) {
355           servers.put(Long.valueOf(sid), new QuorumPeer.QuorumServer(sid, addr));
356         } else if (parts.length == 3) {
357           InetSocketAddress electionAddr = new InetSocketAddress(
358               parts[0], Integer.parseInt(parts[2]));
359           servers.put(Long.valueOf(sid), new QuorumPeer.QuorumServer(sid, addr,
360               electionAddr));
361         }
362       } else if (key.startsWith("group")) {
363         int dot = key.indexOf('.');
364         long gid = Long.parseLong(key.substring(dot + 1));
365 
366         numGroups++;
367 
368         String parts[] = value.split(":");
369         for(String s : parts){
370           long sid = Long.parseLong(s);
371           if(serverGroup.containsKey(sid))
372             throw new ConfigException("Server " + sid + "is in multiple groups");
373           else
374             serverGroup.put(sid, gid);
375         }
376 
377       } else if(key.startsWith("weight")) {
378         int dot = key.indexOf('.');
379         long sid = Long.parseLong(key.substring(dot + 1));
380         serverWeight.put(sid, Long.parseLong(value));
381       } else {
382         System.setProperty("zookeeper." + key, value);
383       }
384     }
385     if (dataDir == null) {
386       throw new IllegalArgumentException("dataDir is not set");
387     }
388     if (dataLogDir == null) {
389       dataLogDir = dataDir;
390     } else {
391       if (!new File(dataLogDir).isDirectory()) {
392         throw new IllegalArgumentException("dataLogDir " + dataLogDir
393             + " is missing.");
394       }
395     }
396 
397     if (tickTime == 0) {
398       throw new IllegalArgumentException("tickTime is not set");
399     }
400     if (servers.size() > 1) {
401       if (initLimit == 0) {
402         throw new IllegalArgumentException("initLimit is not set");
403       }
404       if (syncLimit == 0) {
405         throw new IllegalArgumentException("syncLimit is not set");
406       }
407       /*
408       * If using FLE, then every server requires a separate election
409       * port.
410       */
411       if (electionAlg != 0) {
412         for (QuorumPeer.QuorumServer s : servers.values()) {
413           if (s.electionAddr == null)
414             throw new IllegalArgumentException(
415                 "Missing election port for server: " + s.id);
416         }
417       }
418 
419       /*
420       * Default of quorum config is majority
421       */
422       if(serverGroup.size() > 0){
423         if(servers.size() != serverGroup.size())
424           throw new ConfigException("Every server must be in exactly one group");
425         /*
426          * The deafult weight of a server is 1
427          */
428         for(QuorumPeer.QuorumServer s : servers.values()){
429           if(!serverWeight.containsKey(s.id))
430             serverWeight.put(s.id, (long) 1);
431         }
432 
433         /*
434                       * Set the quorumVerifier to be QuorumHierarchical
435                       */
436         quorumVerifier = new QuorumHierarchical(numGroups,
437             serverWeight, serverGroup);
438       } else {
439         /*
440                       * The default QuorumVerifier is QuorumMaj
441                       */
442 
443         LOG.info("Defaulting to majority quorums");
444         quorumVerifier = new QuorumMaj(servers.size());
445       }
446 
447       File myIdFile = new File(dataDir, "myid");
448       if (!myIdFile.exists()) {
449         ///////////////// ADDED FOR SOLR //////
450         Long myid = getMyServerId();
451         if (myid != null) {
452           serverId = myid;
453           return;
454         }
455         if (zkRun == null) return;
456         //////////////// END ADDED FOR SOLR //////
457         throw new IllegalArgumentException(myIdFile.toString()
458             + " file is missing");
459       }
460 
461       BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(myIdFile), StandardCharsets.UTF_8));
462       String myIdString;
463       try {
464         myIdString = br.readLine();
465       } finally {
466         br.close();
467       }
468       try {
469         serverId = Long.parseLong(myIdString);
470       } catch (NumberFormatException e) {
471         throw new IllegalArgumentException("serverid " + myIdString
472             + " is not a number");
473       }
474     }
475   }
