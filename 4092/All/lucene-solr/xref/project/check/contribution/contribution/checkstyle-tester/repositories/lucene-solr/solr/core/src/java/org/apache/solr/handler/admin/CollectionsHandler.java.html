152   @Override
153   public void handleRequestBody(SolrQueryRequest req, SolrQueryResponse rsp) throws Exception {
154     // Make sure the cores is enabled
155     CoreContainer cores = getCoreContainer();
156     if (cores == null) {
157       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
158               "Core container instance missing");
159     }
160 
161     // Make sure that the core is ZKAware
162     if(!cores.isZooKeeperAware()) {
163       throw new SolrException(ErrorCode.BAD_REQUEST,
164           "Solr instance is not running in SolrCloud mode.");
165     }
166 
167     // Pick the action
168     SolrParams params = req.getParams();
169     String a = params.get(CoreAdminParams.ACTION);
170     if (a != null) {
171       CollectionAction action = CollectionAction.get(a);
172       if (action == null)
173         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Unknown action: " + a);
174       CollectionOperation operation = CollectionOperation.get(action);
175       log.info("Invoked Collection Action :{} with params {} ", action.toLower(), req.getParamString());
176       Map<String, Object> result = operation.call(req, rsp, this);
177       if (result != null) {
178         result.put(QUEUE_OPERATION, operation.action.toLower());
179         ZkNodeProps props = new ZkNodeProps(result);
180         if (operation.sendToOCPQueue) handleResponse(operation.action.toLower(), props, rsp, operation.timeOut);
181         else Overseer.getInQueue(coreContainer.getZkController().getZkClient()).offer(Utils.toJSON(props));
182 
183       }
184     } else {
185       throw new SolrException(ErrorCode.BAD_REQUEST, "action is a required param");
186 
187     }
188 
189     rsp.setHttpCaching(false);
190   }

204   private void handleResponse(String operation, ZkNodeProps m,
205       SolrQueryResponse rsp, long timeout) throws KeeperException, InterruptedException {
206     long time = System.nanoTime();
207 
208      if(m.containsKey(ASYNC) && m.get(ASYNC) != null) {
209 
210        String asyncId = m.getStr(ASYNC);
211 
212        if(asyncId.equals("-1")) {
213          throw new SolrException(ErrorCode.BAD_REQUEST, "requestid can not be -1. It is reserved for cleanup purposes.");
214        }
215 
216        NamedList<String> r = new NamedList<>();
217 
218        if (coreContainer.getZkController().getOverseerCompletedMap().contains(asyncId) ||
219            coreContainer.getZkController().getOverseerFailureMap().contains(asyncId) ||
220            coreContainer.getZkController().getOverseerRunningMap().contains(asyncId) ||
221            overseerCollectionQueueContains(asyncId)) {
222          r.add("error", "Task with the same requestid already exists.");
223 
224        } else {
225          coreContainer.getZkController().getOverseerCollectionQueue()
226              .offer(Utils.toJSON(m));
227        }
228        r.add(CoreAdminParams.REQUESTID, (String) m.get(ASYNC));
229        SolrResponse response = new OverseerSolrResponse(r);
230 
231        rsp.getValues().addAll(response.getResponse());
232 
233        return;
234      }
235 
236     QueueEvent event = coreContainer.getZkController()
237         .getOverseerCollectionQueue()
238         .offer(Utils.toJSON(m), timeout);
239     if (event.getBytes() != null) {
240       SolrResponse response = SolrResponse.deserialize(event.getBytes());
241       rsp.getValues().addAll(response.getResponse());
242       SimpleOrderedMap exp = (SimpleOrderedMap) response.getResponse().get("exception");
243       if (exp != null) {
244         Integer code = (Integer) exp.get("rspCode");
245         rsp.setException(new SolrException(code != null && code != -1 ? ErrorCode.getErrorCode(code) : ErrorCode.SERVER_ERROR, (String)exp.get("msg")));
246       }
247     } else {
248       if (System.nanoTime() - time >= TimeUnit.NANOSECONDS.convert(timeout, TimeUnit.MILLISECONDS)) {
249         throw new SolrException(ErrorCode.SERVER_ERROR, operation
250             + " the collection time out:" + timeout / 1000 + "s");
251       } else if (event.getWatchedEvent() != null) {
252         throw new SolrException(ErrorCode.SERVER_ERROR, operation
253             + " the collection error [Watcher fired on path: "
254             + event.getWatchedEvent().getPath() + " state: "
255             + event.getWatchedEvent().getState() + " type "
256             + event.getWatchedEvent().getType() + "]");
257       } else {
258         throw new SolrException(ErrorCode.SERVER_ERROR, operation
259             + " the collection unknown case");
260       }
261     }
262   }

264   private boolean overseerCollectionQueueContains(String asyncId) throws KeeperException, InterruptedException {
265     OverseerTaskQueue collectionQueue = coreContainer.getZkController().getOverseerCollectionQueue();
266     return collectionQueue.containsTaskWithRequestId(ASYNC, asyncId);
267   }

269   private static Map<String, Object> copyPropertiesWithPrefix(SolrParams params, Map<String, Object> props, String prefix) {
270     Iterator<String> iter =  params.getParameterNamesIterator();
271     while (iter.hasNext()) {
272       String param = iter.next();
273       if (param.startsWith(prefix)) {
274         props.put(param, params.get(param));
275       }
276     }
277     return props;
278   }

280   public static ModifiableSolrParams params(String... params) {
281     ModifiableSolrParams msp = new ModifiableSolrParams();
282     for (int i = 0; i < params.length; i += 2) {
283       msp.add(params[i], params[i + 1]);
284     }
285     return msp;

312             REPLICATION_FACTOR,

322             SNITCH);
323 
324         if (props.get(STATE_FORMAT) == null) {
325           props.put(STATE_FORMAT, "2");
326         }
327         addMapObject(props, RULE);
328         addMapObject(props, SNITCH);
329         verifyRuleParams(h.coreContainer, props);

331           //We must always create a .system collection with only a single shard
332           props.put(NUM_SLICES, 1);
333           props.remove(SHARDS_PROP);
334           createSysConfigSet(h.coreContainer);
335 
336         }
337         copyPropertiesWithPrefix(req.getParams(), props, COLL_PROP_PREFIX);
338         return copyPropertiesWithPrefix(req.getParams(), props, "router.");
339

342       private void createSysConfigSet(CoreContainer coreContainer) throws KeeperException, InterruptedException {
343         SolrZkClient zk = coreContainer.getZkController().getZkStateReader().getZkClient();
344         ZkCmdExecutor cmdExecutor = new ZkCmdExecutor(zk.getZkClientTimeout());
345         cmdExecutor.ensureExists(ZkStateReader.CONFIGS_ZKNODE, zk);
346         cmdExecutor.ensureExists(ZkStateReader.CONFIGS_ZKNODE + "/" + SYSTEM_COLL, zk);
347 
348         String path = ZkStateReader.CONFIGS_ZKNODE + "/" + SYSTEM_COLL + "/schema.xml";
349         byte[] data = BlobHandler.SCHEMA.replaceAll("'", "\"").getBytes(StandardCharsets.UTF_8);
350         cmdExecutor.ensureExists(path, data, CreateMode.PERSISTENT, zk);
351 
352         path = ZkStateReader.CONFIGS_ZKNODE + "/" + SYSTEM_COLL + "/solrconfig.xml";
353         data = BlobHandler.CONF.replaceAll("'", "\"").getBytes(StandardCharsets.UTF_8);
354         cmdExecutor.ensureExists(path, data, CreateMode.PERSISTENT, zk);
355       }

519     REQUESTSTATUS_OP(REQUESTSTATUS) {
520       @Override
521       Map<String, Object> call(SolrQueryRequest req, SolrQueryResponse rsp, CollectionsHandler h) throws Exception {
522         CoreContainer coreContainer = h.coreContainer;
523         req.getParams().required().check(REQUESTID);
524 
525         String requestId = req.getParams().get(REQUESTID);
526 
527         if (requestId.equals("-1")) {
528           // Special taskId (-1), clears up the request state maps.
529           if (requestId.equals("-1")) {
530             coreContainer.getZkController().getOverseerCompletedMap().clear();
531             coreContainer.getZkController().getOverseerFailureMap().clear();
532             return null;
533           }
534         } else {
535           NamedList<Object> results = new NamedList<>();
536           if (coreContainer.getZkController().getOverseerCompletedMap().contains(requestId)) {
537             DistributedMap.MapEvent mapEvent = coreContainer.getZkController().getOverseerCompletedMap().get(requestId);
538             SimpleOrderedMap success = new SimpleOrderedMap();
539             if(mapEvent != null) {
540               rsp.getValues().addAll(SolrResponse.deserialize(mapEvent.getBytes()).getResponse());
541             }
542             success.add("state", "completed");
543             success.add("msg", "found " + requestId + " in completed tasks");
544             results.add("status", success);
545           } else if (coreContainer.getZkController().getOverseerFailureMap().contains(requestId)) {
546             SimpleOrderedMap success = new SimpleOrderedMap();
547             DistributedMap.MapEvent mapEvent = coreContainer.getZkController().getOverseerFailureMap().get(requestId);

758   private static void forceLeaderElection(SolrQueryRequest req, CollectionsHandler handler) {
759     ClusterState clusterState = handler.coreContainer.getZkController().getClusterState();
760     String collection = req.getParams().required().get(COLLECTION_PROP);
761     String sliceId = req.getParams().required().get(SHARD_ID_PROP);
762 
763     log.info("Force leader invoked, state: {}", clusterState);
764     Slice slice = clusterState.getSlice(collection, sliceId);
765     if (slice == null) {
766       if (clusterState.hasCollection(collection)) {
767         throw new SolrException(ErrorCode.BAD_REQUEST,
768             "No shard with name " + sliceId + " exists for collection " + collection);
769       } else {
770         throw new SolrException(ErrorCode.BAD_REQUEST, "No collection with the specified name exists: " + collection);
771       }
772     }
773 
774     try {
775       // if an active replica is the leader, then all is fine already
776       Replica leader = slice.getLeader();
777       if (leader != null && leader.getState() == State.ACTIVE) {
778         throw new SolrException(ErrorCode.SERVER_ERROR,
779             "The shard already has an active leader. Force leader is not applicable. State: " + slice);
780       }
781 
782       // Clear out any LIR state
783       String lirPath = handler.coreContainer.getZkController().getLeaderInitiatedRecoveryZnodePath(collection, sliceId);
784       if (handler.coreContainer.getZkController().getZkClient().exists(lirPath, true)) {
785         StringBuilder sb = new StringBuilder();
786         handler.coreContainer.getZkController().getZkClient().printLayout(lirPath, 4, sb);
787         log.info("Cleaning out LIR data, which was: {}", sb);
788         handler.coreContainer.getZkController().getZkClient().clean(lirPath);
789       }
790 
791       // Call all live replicas to prepare themselves for leadership, e.g. set last published
792       // state to active.
793       for (Replica rep : slice.getReplicas()) {
794         if (clusterState.getLiveNodes().contains(rep.getNodeName())) {
795           ShardHandler shardHandler = handler.coreContainer.getShardHandlerFactory().getShardHandler();
796 
797           ModifiableSolrParams params = new ModifiableSolrParams();
798           params.set(CoreAdminParams.ACTION, CoreAdminAction.FORCEPREPAREFORLEADERSHIP.toString());
799           params.set(CoreAdminParams.CORE, rep.getStr("core"));
800           String nodeName = rep.getNodeName();
801 
802           OverseerCollectionMessageHandler.sendShardRequest(nodeName, params, shardHandler, null, null,
803               CommonParams.CORES_HANDLER_PATH, handler.coreContainer.getZkController().getZkStateReader()); // synchronous request
804         }
805       }
806 
807       // Wait till we have an active leader
808       boolean success = false;
809       for (int i = 0; i < 9; i++) {
810         Thread.sleep(5000);
811         clusterState = handler.coreContainer.getZkController().getClusterState();
812         slice = clusterState.getSlice(collection, sliceId);
813         if (slice.getLeader() != null && slice.getLeader().getState() == State.ACTIVE) {
814           success = true;
815           break;
816         }
817         log.warn("Force leader attempt {}. Waiting 5 secs for an active leader. State of the slice: {}", (i + 1), slice);
818       }
819 
820       if (success) {
821         log.info("Successfully issued FORCELEADER command for collection: {}, shard: {}", collection, sliceId);
822       } else {
823         log.info("Couldn't successfully force leader, collection: {}, shard: {}. Cluster state: {}", collection, sliceId, clusterState);
824       }
825     } catch (SolrException e) {
826       throw e;
827     } catch (Exception e) {
828       throw new SolrException(ErrorCode.SERVER_ERROR,
829           "Error executing FORCELEADER operation for collection: " + collection + " shard: " + sliceId, e);
830     }
831   }

834     List l = (List) m.get(RULE);
835     if (l != null) {
836       for (Object o : l) {
837         Map map = (Map) o;
838         try {
839           new Rule(map);
840         } catch (Exception e) {
841           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Error in rule " + m, e);
842         }
843       }
844     }
845     ReplicaAssigner.verifySnitchConf(cc, (List) m.get(SNITCH));
846   }

848   /**
849    * Converts a String of the form a:b,c:d to a Map
850    */
851   private static Map<String, Object> addMapObject(Map<String, Object> props, String key) {
852     Object v = props.get(key);
853     if (v == null) return props;
854     List<String> val = new ArrayList<>();
855     if (v instanceof String[]) {
856       val.addAll(Arrays.asList((String[]) v));
857     } else {
858       val.add(v.toString());
859     }
860     if (val.size() > 0) {
861       ArrayList<Map> l = new ArrayList<>();
862       for (String rule : val) l.add(Rule.parseRule(rule));
863       props.put(key, l);
864     }
865     return props;
866   }
