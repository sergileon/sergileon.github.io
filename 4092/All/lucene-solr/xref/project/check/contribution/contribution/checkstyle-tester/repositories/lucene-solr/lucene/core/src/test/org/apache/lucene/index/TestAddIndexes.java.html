52    public void testSimpleCase() throws IOException {
53      // main directory
54      Directory dir = newDirectory();
55      // two auxiliary directories
56      Directory aux = newDirectory();
57      Directory aux2 = newDirectory();
58  
59      IndexWriter writer = null;
60  
61      writer = newWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
62          .setOpenMode(OpenMode.CREATE));
63      // add 100 documents
64      addDocs(writer, 100);
65      assertEquals(100, writer.maxDoc());
66      writer.close();
67      TestUtil.checkIndex(dir);
68  
69      writer = newWriter(
70          aux,
71          newIndexWriterConfig(new MockAnalyzer(random())).
72              setOpenMode(OpenMode.CREATE).
73              setMergePolicy(newLogMergePolicy(false))
74      );
75      // add 40 documents in separate files
76      addDocs(writer, 40);
77      assertEquals(40, writer.maxDoc());
78      writer.close();
79  
80      writer = newWriter(aux2, newIndexWriterConfig(new MockAnalyzer(random())).setOpenMode(OpenMode.CREATE));
81      // add 50 documents in compound files
82      addDocs2(writer, 50);
83      assertEquals(50, writer.maxDoc());
84      writer.close();
85  
86      // test doc count before segments are merged
87      writer = newWriter(dir, newIndexWriterConfig(new MockAnalyzer(random())).setOpenMode(OpenMode.APPEND));
88      assertEquals(100, writer.maxDoc());
89      writer.addIndexes(aux, aux2);
90      assertEquals(190, writer.maxDoc());
91      writer.close();
92      TestUtil.checkIndex(dir);
93  
94      // make sure the old index is correct
95      verifyNumDocs(aux, 40);
96  
97      // make sure the new index is correct
98      verifyNumDocs(dir, 190);
99  
100     // now add another set in.
101     Directory aux3 = newDirectory();
102     writer = newWriter(aux3, newIndexWriterConfig(new MockAnalyzer(random())));
103     // add 40 documents
104     addDocs(writer, 40);
105     assertEquals(40, writer.maxDoc());
106     writer.close();
107 
108     // test doc count before segments are merged
109     writer = newWriter(dir, newIndexWriterConfig(new MockAnalyzer(random())).setOpenMode(OpenMode.APPEND));
110     assertEquals(190, writer.maxDoc());
111     writer.addIndexes(aux3);
112     assertEquals(230, writer.maxDoc());

114

157     // main directory
158     Directory dir = newDirectory();

161

165 
166     // Adds 10 docs, then replaces them with another 10

170       doc.add(newStringField("id", "" + (i % 10), Field.Store.NO));

179     writer.commit();
180 
181     verifyNumDocs(dir, 1039);
182     verifyTermDocs(dir, new Term("content", "aaa"), 1030);

573         addDocs(writer, 10);
574       }

579               setOpenMode(OpenMode.APPEND).

581               setMergePolicy(newLogMergePolicy(false, 10))
582       );
583     }
584     assertEquals(30, writer.maxDoc());
585     assertEquals(3, writer.getSegmentCount());
586     writer.close();
587   }

642   private abstract class RunAddIndexesThreads {
643 
644     Directory dir, dir2;
645     final static int NUM_INIT_DOCS = 17;
646     IndexWriter writer2;
647     final List<Throwable> failures = new ArrayList<>();
648     volatile boolean didClose;
649     final DirectoryReader[] readers;
650     final int NUM_COPY;
651     final static int NUM_THREADS = 5;
652     final Thread[] threads = new Thread[NUM_THREADS];
653 
654     public RunAddIndexesThreads(int numCopy) throws Throwable {
655       NUM_COPY = numCopy;
656       dir = new MockDirectoryWrapper(random(), new RAMDirectory());
657       IndexWriter writer = new IndexWriter(dir, new IndexWriterConfig(new MockAnalyzer(random()))
658           .setMaxBufferedDocs(2));
659       for (int i = 0; i < NUM_INIT_DOCS; i++)
660         addDoc(writer);
661       writer.close();
662 
663       dir2 = newDirectory();
664       writer2 = new IndexWriter(dir2, new IndexWriterConfig(new MockAnalyzer(random())));
665       writer2.commit();
666       
667 
668       readers = new DirectoryReader[NUM_COPY];
669       for(int i=0;i<NUM_COPY;i++)
670         readers[i] = DirectoryReader.open(dir);
671     }
672 
673     void launchThreads(final int numIter) {
674 
675       for(int i=0;i<NUM_THREADS;i++) {
676         threads[i] = new Thread() {
677             @Override
678             public void run() {
679               try {
680 
681                 final Directory[] dirs = new Directory[NUM_COPY];
682                 for(int k=0;k<NUM_COPY;k++)
683                   dirs[k] = new MockDirectoryWrapper(random(), TestUtil.ramCopyOf(dir));
684 
685                 int j=0;
686 
687                 while(true) {
688                   // System.out.println(Thread.currentThread().getName() + ": iter j=" + j);
689                   if (numIter > 0 && j == numIter)
690                     break;
691                   doBody(j++, dirs);
692                 }
693               } catch (Throwable t) {

710       didClose = true;
711       if (doWait == false) {
712         writer2.rollback();
713       } else {
714         writer2.close();
715       }
716     }

718     void closeDir() throws Throwable {
719       for(int i=0;i<NUM_COPY;i++)
720         readers[i].close();
721       dir2.close();
722     }

882       case 4:
883         if (VERBOSE) {
884           System.out.println("TEST: " + Thread.currentThread().getName() + ": commit");
885         }
886         writer2.commit();
887       }

890     @Override
891     void handle(Throwable t) {
892       boolean report = true;
893 
894       if (t instanceof AlreadyClosedException || t instanceof MergePolicy.MergeAbortedException || t instanceof NullPointerException) {
895         report = !didClose;
896       } else if (t instanceof FileNotFoundException || t instanceof NoSuchFileException)  {
897         report = !didClose;
898       } else if (t instanceof IOException)  {
899         Throwable t2 = t.getCause();
900         if (t2 instanceof MergePolicy.MergeAbortedException) {
901           report = !didClose;
902         }
903       }
904       if (report) {
905         t.printStackTrace(System.out);
906         synchronized(failures) {
907           failures.add(t);
908         }
909       }
910     }
