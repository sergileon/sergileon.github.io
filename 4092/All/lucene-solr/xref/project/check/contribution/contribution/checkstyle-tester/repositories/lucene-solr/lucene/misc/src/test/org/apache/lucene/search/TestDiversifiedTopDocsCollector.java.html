47  /**
48   * Demonstrates an application of the {@link DiversifiedTopDocsCollector} in
49   * assembling a collection of top results but without over-representation of any
50   * one source (in this case top-selling singles from the 60s without having them
51   * all be Beatles records...). Results are ranked by the number of weeks a
52   * single is top of the charts and de-duped by the artist name.
53   * 
54   */
55  public class TestDiversifiedTopDocsCollector extends LuceneTestCase {
56  
57    public void testNonDiversifiedResults() throws Exception {
58      int numberOfTracksOnCompilation = 10;
59      int expectedMinNumOfBeatlesHits = 5;
60      TopDocs res = searcher.search(getTestQuery(), numberOfTracksOnCompilation);
61      assertEquals(numberOfTracksOnCompilation, res.scoreDocs.length);
62      // due to randomization of segment merging in tests the exact number of Beatles hits 
63      // selected varies between 5 and 6 but we prove the point they are over-represented
64      // in our result set using a standard search.
65      assertTrue(getMaxNumRecordsPerArtist(res.scoreDocs) >= expectedMinNumOfBeatlesHits);
66    }

68    public void testFirstPageDiversifiedResults() throws Exception {
69      // Using a diversified collector we can limit the results from
70      // any one artist.
71      int requiredMaxHitsPerArtist = 2;
72      int numberOfTracksOnCompilation = 10;
73      DiversifiedTopDocsCollector tdc = doDiversifiedSearch(
74          numberOfTracksOnCompilation, requiredMaxHitsPerArtist);
75      ScoreDoc[] sd = tdc.topDocs(0).scoreDocs;
76      assertEquals(numberOfTracksOnCompilation, sd.length);
77      assertTrue(getMaxNumRecordsPerArtist(sd) <= requiredMaxHitsPerArtist);
78    }

80    public void testSecondPageResults() throws Exception {
81      int numberOfTracksPerCompilation = 10;
82      int numberOfCompilations = 2;
83      int requiredMaxHitsPerArtist = 1;
84  
85      // Volume 2 of our hits compilation - start at position 10
86      DiversifiedTopDocsCollector tdc = doDiversifiedSearch(
87          numberOfTracksPerCompilation * numberOfCompilations,
88          requiredMaxHitsPerArtist);
89      ScoreDoc[] volume2 = tdc.topDocs(numberOfTracksPerCompilation,
90          numberOfTracksPerCompilation).scoreDocs;
91      assertEquals(numberOfTracksPerCompilation, volume2.length);
92      assertTrue(getMaxNumRecordsPerArtist(volume2) <= requiredMaxHitsPerArtist);
93  
94    }

96    public void testInvalidArguments() throws Exception {
97      int numResults = 5;
98      DiversifiedTopDocsCollector tdc = doDiversifiedSearch(numResults, 15);
99  
100     // start < 0
101     assertEquals(0, tdc.topDocs(-1).scoreDocs.length);
102 
103     // start > pq.size()
104     assertEquals(0, tdc.topDocs(numResults + 1).scoreDocs.length);
105 
106     // start == pq.size()
107     assertEquals(0, tdc.topDocs(numResults).scoreDocs.length);
108 
109     // howMany < 0
110     assertEquals(0, tdc.topDocs(0, -1).scoreDocs.length);
111 
112     // howMany == 0
113     assertEquals(0, tdc.topDocs(0, 0).scoreDocs.length);
114

117   // Diversifying collector that looks up de-dup keys using SortedDocValues
118   // from a top-level Reader
119   private static final class DocValuesDiversifiedCollector extends
120       DiversifiedTopDocsCollector {
121     private final SortedDocValues sdv;
122 
123     public DocValuesDiversifiedCollector(int size, int maxHitsPerKey,
124         SortedDocValues sdv) {
125       super(size, maxHitsPerKey);
126       this.sdv = sdv;
127     }
128 
129     @Override
130     protected NumericDocValues getKeys(final LeafReaderContext context) {
131 
132       return new NumericDocValues() {
133         @Override
134         public long get(int docID) {
135           // Keys are always expressed as a long so we obtain the
136           // ordinal for our String-based artist name here
137           return sdv.getOrd(context.docBase + docID);
138         }
139       };
140     }
141   }
142 
143   // Alternative, faster implementation for converting String keys to longs
144   // but with the potential for hash collisions
145   private static final class HashedDocValuesDiversifiedCollector extends
146       DiversifiedTopDocsCollector {
147 
148     private final String field;
149     private BinaryDocValues vals;
150 
151     public HashedDocValuesDiversifiedCollector(int size, int maxHitsPerKey,
152         String field) {
153       super(size, maxHitsPerKey);
154       this.field = field;
155     }
156 
157     @Override
158     protected NumericDocValues getKeys(LeafReaderContext context) {
159       return new NumericDocValues() {
160         @Override
161         public long get(int docID) {
162           return vals == null ? -1 : vals.get(docID).hashCode();
163         }
164       };
165     }
166 
167     @Override
168     public LeafCollector getLeafCollector(LeafReaderContext context)
169         throws IOException {
170       this.vals = DocValues.getBinary(context.reader(), field);
171       return super.getLeafCollector(context);
172     }
173   }

256   static class Record {
257     String year;
258     String artist;
259     String song;
260     float weeks;
261     String id;
262 
263     public Record(String id, String year, String artist, String song,
264         float weeks) {
265       super();
266       this.id = id;
267       this.year = year;
268       this.artist = artist;
269       this.song = song;
270       this.weeks = weeks;
271     }
272 
273     @Override
274     public String toString() {
275       return "Record [id=" + id + ", artist=" + artist + ", weeks=" + weeks
276           + ", year=" + year + ", song=" + song + "]";
277     }

281   private DiversifiedTopDocsCollector doDiversifiedSearch(int numResults,
282       int maxResultsPerArtist) throws IOException {
283     // Alternate between implementations used for key lookups 
284     if (random().nextBoolean()) {
285       // Faster key lookup but with potential for collisions on larger datasets
286       return doFuzzyDiversifiedSearch(numResults, maxResultsPerArtist);
287     } else {
288       // Slower key lookup but 100% accurate
289       return doAccurateDiversifiedSearch(numResults, maxResultsPerArtist);
290     }
291   }
292 
293   private DiversifiedTopDocsCollector doFuzzyDiversifiedSearch(int numResults,
294       int maxResultsPerArtist) throws IOException {
295     DiversifiedTopDocsCollector tdc = new HashedDocValuesDiversifiedCollector(
296         numResults, maxResultsPerArtist, "artist");
297     searcher.search(getTestQuery(), tdc);
298     return tdc;
299   }
300 
301   private DiversifiedTopDocsCollector doAccurateDiversifiedSearch(
302       int numResults, int maxResultsPerArtist) throws IOException {
303     DiversifiedTopDocsCollector tdc = new DocValuesDiversifiedCollector(
304         numResults, maxResultsPerArtist, artistDocValues);
305     searcher.search(getTestQuery(), tdc);
306     return tdc;
307   }

322   @Override
323   public void setUp() throws Exception {
324     super.setUp();
325 
326     // populate an index with documents - artist, song and weeksAtNumberOne
327     dir = newDirectory();
328     RandomIndexWriter writer = new RandomIndexWriter(random(), dir);
329     Document doc = new Document();
330 
331     Field yearField = newTextField("year", "", Field.Store.NO);
332     SortedDocValuesField artistField = new SortedDocValuesField("artist",
333         new BytesRef(""));
334     Field weeksAtNumberOneField = new FloatDocValuesField("weeksAtNumberOne",
335         0.0F);
336     Field weeksStoredField = new FloatField("weeks", 0.0F, Store.YES);
337     Field idField = newStringField("id", "", Field.Store.YES);
338     Field songField = newTextField("song", "", Field.Store.NO);
339     Field storedArtistField = newTextField("artistName", "", Field.Store.NO);
340 
341     doc.add(idField);
342     doc.add(weeksAtNumberOneField);
343     doc.add(storedArtistField);
344     doc.add(songField);
345     doc.add(weeksStoredField);
346     doc.add(yearField);
347     doc.add(artistField);
348 
349     parsedRecords.clear();
350     for (int i = 0; i < hitsOfThe60s.length; i++) {
351       String cols[] = hitsOfThe60s[i].split("\t");
352       Record record = new Record(String.valueOf(i), cols[0], cols[1], cols[2],
353           Float.valueOf(cols[3]));
354       parsedRecords.put(record.id, record);
355       idField.setStringValue(record.id);
356       yearField.setStringValue(record.year);
357       storedArtistField.setStringValue(record.artist);
358       artistField.setBytesValue(new BytesRef(record.artist));
359       songField.setStringValue(record.song);
360       weeksStoredField.setFloatValue(record.weeks);
361       weeksAtNumberOneField.setFloatValue(record.weeks);
362       writer.addDocument(doc);
363       if (i % 10 == 0) {
364         // Causes the creation of multiple segments for our test
365         writer.commit();
366       }
367     }
368     reader = writer.getReader();
369     writer.close();
370     searcher = newSearcher(reader);
371     LeafReader ar = SlowCompositeReaderWrapper.wrap(reader);
372     artistDocValues = ar.getSortedDocValues("artist");
373 
374     // All searches sort by song popularity 
375     final Similarity base = searcher.getSimilarity(true);
376     searcher.setSimilarity(new DocValueSimilarity(base, "weeksAtNumberOne"));
377   }

387   private int getMaxNumRecordsPerArtist(ScoreDoc[] sd) throws IOException {
388     int result = 0;
389     HashMap<String, Integer> artistCounts = new HashMap<String, Integer>();
390     for (int i = 0; i < sd.length; i++) {
391       Document doc = reader.document(sd[i].doc);
392       Record record = parsedRecords.get(doc.get("id"));
393       Integer count = artistCounts.get(record.artist);
394       int newCount = 1;
395       if (count != null) {
396         newCount = count.intValue() + 1;
397       }
398       result = Math.max(result, newCount);
399       artistCounts.put(record.artist, newCount);
400     }
401     return result;
402   }

404   /**
405    * Similarity that wraps another similarity and replaces the final score
406    * according to whats in a docvalues field.
407    * 
408    * @lucene.experimental
409    */
410   static class DocValueSimilarity extends Similarity {
411     private final Similarity sim;
412     private final String scoreValueField;
413 
414     public DocValueSimilarity(Similarity sim, String scoreValueField) {
415       this.sim = sim;
416       this.scoreValueField = scoreValueField;
417     }
418 
419     @Override
420     public long computeNorm(FieldInvertState state) {
421       return sim.computeNorm(state);
422     }
423 
424     @Override
425     public SimWeight computeWeight(
426         CollectionStatistics collectionStats, TermStatistics... termStats) {
427       return sim.computeWeight(collectionStats, termStats);
428     }
429 
430     @Override
431     public SimScorer simScorer(SimWeight stats, LeafReaderContext context)
432         throws IOException {
433       final SimScorer sub = sim.simScorer(stats, context);
434       final NumericDocValues values = DocValues.getNumeric(context.reader(),
435           scoreValueField);
436 
437       return new SimScorer() {
438         @Override
439         public float score(int doc, float freq) {
440           return Float.intBitsToFloat((int) values.get(doc));
441         }
442 
443         @Override
444         public float computeSlopFactor(int distance) {
445           return sub.computeSlopFactor(distance);
446         }
447 
448         @Override
449         public float computePayloadFactor(int doc, int start, int end,
450             BytesRef payload) {
451           return sub.computePayloadFactor(doc, start, end, payload);
452         }
453 
454         @Override
455         public Explanation explain(int doc, Explanation freq) {
456           return Explanation.match(Float.intBitsToFloat((int) values.get(doc)),
457               "indexDocValue(" + scoreValueField + ")");
458         }
459       };
460     }
