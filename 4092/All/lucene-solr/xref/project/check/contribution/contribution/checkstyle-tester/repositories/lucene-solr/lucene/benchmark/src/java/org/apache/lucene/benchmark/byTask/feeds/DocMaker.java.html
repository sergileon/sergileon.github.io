48  /**
49   * Creates {@link Document} objects. Uses a {@link ContentSource} to generate
50   * {@link DocData} objects. Supports the following parameters:
51   * <ul>
52   * <li><b>content.source</b> - specifies the {@link ContentSource} class to use
53   * (default <b>SingleDocSource</b>).
54   * <li><b>doc.stored</b> - specifies whether fields should be stored (default
55   * <b>false</b>).
56   * <li><b>doc.body.stored</b> - specifies whether the body field should be stored (default
57   * = <b>doc.stored</b>).
58   * <li><b>doc.tokenized</b> - specifies whether fields should be tokenized
59   * (default <b>true</b>).
60   * <li><b>doc.body.tokenized</b> - specifies whether the
61   * body field should be tokenized (default = <b>doc.tokenized</b>).
62   * <li><b>doc.tokenized.norms</b> - specifies whether norms should be stored in
63   * the index or not. (default <b>false</b>).
64   * <li><b>doc.body.tokenized.norms</b> - specifies whether norms should be
65   * stored in the index for the body field. This can be set to true, while
66   * <code>doc.tokenized.norms</code> is set to false, to allow norms storing just
67   * for the body field. (default <b>true</b>).
68   * <li><b>doc.term.vector</b> - specifies whether term vectors should be stored
69   * for fields (default <b>false</b>).
70   * <li><b>doc.term.vector.positions</b> - specifies whether term vectors should
71   * be stored with positions (default <b>false</b>).
72   * <li><b>doc.term.vector.offsets</b> - specifies whether term vectors should be
73   * stored with offsets (default <b>false</b>).
74   * <li><b>doc.store.body.bytes</b> - specifies whether to store the raw bytes of
75   * the document's content in the document (default <b>false</b>).
76   * <li><b>doc.reuse.fields</b> - specifies whether Field and Document objects
77   * should be reused (default <b>true</b>).
78   * <li><b>doc.index.props</b> - specifies whether the properties returned by
79   * <li><b>doc.random.id.limit</b> - if specified, docs will be assigned random
80   * IDs from 0 to this limit.  This is useful with UpdateDoc
81   * for testing performance of IndexWriter.updateDocument.
82   * {@link DocData#getProps()} will be indexed. (default <b>false</b>).
83   * </ul>
84   */
85  public class DocMaker implements Closeable {

95    /**
96     * Document state, supports reuse of field instances
97     * across documents (see <code>reuseFields</code> parameter).
98     */
99    protected static class DocState {
100     
101     private final Map<String,Field> fields;
102     private final Map<String,Field> numericFields;
103     private final boolean reuseFields;
104     final Document doc;
105     DocData docData = new DocData();
106     
107     public DocState(boolean reuseFields, FieldType ft, FieldType bodyFt) {
108 
109       this.reuseFields = reuseFields;
110       
111       if (reuseFields) {
112         fields =  new HashMap<>();
113         numericFields = new HashMap<>();
114         
115         // Initialize the map with the default fields.
116         fields.put(BODY_FIELD, new Field(BODY_FIELD, "", bodyFt));
117         fields.put(TITLE_FIELD, new Field(TITLE_FIELD, "", ft));
118         fields.put(DATE_FIELD, new Field(DATE_FIELD, "", ft));
119         fields.put(ID_FIELD, new StringField(ID_FIELD, "", Field.Store.YES));
120         fields.put(NAME_FIELD, new Field(NAME_FIELD, "", ft));
121 
122         numericFields.put(DATE_MSEC_FIELD, new LongField(DATE_MSEC_FIELD, 0L, Field.Store.NO));
123         numericFields.put(TIME_SEC_FIELD, new IntField(TIME_SEC_FIELD, 0, Field.Store.NO));
124         
125         doc = new Document();
126       } else {
127         numericFields = null;
128         fields = null;
129         doc = null;
130       }
131     }
132 
133     /**
134      * Returns a field corresponding to the field name. If
135      * <code>reuseFields</code> was set to true, then it attempts to reuse a
136      * Field instance. If such a field does not exist, it creates a new one.
137      */
138     Field getField(String name, FieldType ft) {
139       if (!reuseFields) {
140         return new Field(name, "", ft);
141       }
142       
143       Field f = fields.get(name);
144       if (f == null) {
145         f = new Field(name, "", ft);
146         fields.put(name, f);
147       }
148       return f;
149     }
150 
151     Field getNumericField(String name, NumericType type) {
152       Field f;
153       if (reuseFields) {
154         f = numericFields.get(name);
155       } else {
156         f = null;
157       }
158       
159       if (f == null) {
160         switch(type) {
161         case INT:
162           f = new IntField(name, 0, Field.Store.NO);
163           break;
164         case LONG:
165           f = new LongField(name, 0L, Field.Store.NO);
166           break;
167         case FLOAT:
168           f = new FloatField(name, 0.0F, Field.Store.NO);
169           break;
170         case DOUBLE:
171           f = new DoubleField(name, 0.0, Field.Store.NO);
172           break;
173         default:
174           throw new AssertionError("Cannot get here");
175         }
176         if (reuseFields) {
177           numericFields.put(name, f);
178         }
179       }
180       return f;
181     }
182   }

186   private static class DateUtil {
187     public SimpleDateFormat parser = new SimpleDateFormat("dd-MMM-yyyy HH:mm:ss", Locale.ENGLISH);
188     public Calendar cal = Calendar.getInstance(TimeZone.getTimeZone("GMT"), Locale.ROOT);
189     public ParsePosition pos = new ParsePosition(0);
190     public DateUtil() {
191       parser.setLenient(true);
192     }
193   }

206   public static final String BYTES_FIELD = "bytes";
207   public static final String NAME_FIELD = "docname";

223   // create a doc
224   // use only part of the body, modify it to keep the rest (or use all if size==0).
225   // reset the docdata properties so they are not added more than once.
226   private Document createDocument(DocData docData, int size, int cnt) throws UnsupportedEncodingException {
227 
228     final DocState ds = getDocState();
229     final Document doc = reuseFields ? ds.doc : new Document();
230     doc.getFields().clear();
231     
232     // Set ID_FIELD
233     FieldType ft = new FieldType(valType);
234     ft.setStored(true);
235 
236     Field idField = ds.getField(ID_FIELD, ft);
237     int id;
238     if (r != null) {
239       id = r.nextInt(updateDocIDLimit);
240     } else {
241       id = docData.getID();
242       if (id == -1) {
243         id = numDocsCreated.getAndIncrement();
244       }
245     }
246     idField.setStringValue(Integer.toString(id));
247     doc.add(idField);
248     
249     // Set NAME_FIELD
250     String name = docData.getName();
251     if (name == null) name = "";
252     name = cnt < 0 ? name : name + "_" + cnt;
253     Field nameField = ds.getField(NAME_FIELD, valType);
254     nameField.setStringValue(name);
255     doc.add(nameField);
256     
257     // Set DATE_FIELD
258     DateUtil util = dateParsers.get();
259     if (util == null) {
260       util = new DateUtil();
261       dateParsers.set(util);
262     }
263     Date date = null;
264     String dateString = docData.getDate();
265     if (dateString != null) {
266       util.pos.setIndex(0);
267       date = util.parser.parse(dateString, util.pos);
268       //System.out.println(dateString + " parsed to " + date);
269     } else {
270       dateString = "";
271     }
272     Field dateStringField = ds.getField(DATE_FIELD, valType);
273     dateStringField.setStringValue(dateString);
274     doc.add(dateStringField);
275 
276     if (date == null) {
277       // just set to right now
278       date = new Date();
279     }
280 
281     Field dateField = ds.getNumericField(DATE_MSEC_FIELD, NumericType.LONG);
282     dateField.setLongValue(date.getTime());
283     doc.add(dateField);
284 
285     util.cal.setTime(date);
286     final int sec = util.cal.get(Calendar.HOUR_OF_DAY)*3600 + util.cal.get(Calendar.MINUTE)*60 + util.cal.get(Calendar.SECOND);
287 
288     Field timeSecField = ds.getNumericField(TIME_SEC_FIELD, NumericType.INT);
289     timeSecField.setIntValue(sec);
290     doc.add(timeSecField);
291     
292     // Set TITLE_FIELD
293     String title = docData.getTitle();
294     Field titleField = ds.getField(TITLE_FIELD, valType);
295     titleField.setStringValue(title == null ? "" : title);
296     doc.add(titleField);
297     
298     String body = docData.getBody();
299     if (body != null && body.length() > 0) {
300       String bdy;
301       if (size <= 0 || size >= body.length()) {
302         bdy = body; // use all
303         docData.setBody(""); // nothing left
304       } else {
305         // attempt not to break words - if whitespace found within next 20 chars...
306         for (int n = size - 1; n < size + 20 && n < body.length(); n++) {
307           if (Character.isWhitespace(body.charAt(n))) {
308             size = n;
309             break;
310           }
311         }
312         bdy = body.substring(0, size); // use part
313         docData.setBody(body.substring(size)); // some left
314       }
315       Field bodyField = ds.getField(BODY_FIELD, bodyValType);
316       bodyField.setStringValue(bdy);
317       doc.add(bodyField);
318       
319       if (storeBytes) {
320         Field bytesField = ds.getField(BYTES_FIELD, StringField.TYPE_STORED);
321         bytesField.setBytesValue(bdy.getBytes(StandardCharsets.UTF_8));
322         doc.add(bytesField);
323       }
324     }
325 
326     if (indexProperties) {
327       Properties props = docData.getProps();
328       if (props != null) {
329         for (final Map.Entry<Object,Object> entry : props.entrySet()) {
330           Field f = ds.getField((String) entry.getKey(), valType);
331           f.setStringValue((String) entry.getValue());
332           doc.add(f);
333         }
334         docData.setProps(null);
335       }
336     }
337     
338     //System.out.println("============== Created doc "+numDocsCreated+" :\n"+doc+"\n==========");
339     return doc;
340   }
341 
342   private void resetLeftovers() {
343     leftovr.set(null);
344   }
345 
346   protected DocState getDocState() {
347     DocState ds = docState.get();
348     if (ds == null) {
349       ds = new DocState(reuseFields, valType, bodyValType);
350       docState.set(ds);
351     }
352     return ds;
353   }

365   /**
366    * Creates a {@link Document} object ready for indexing. This method uses the
367    * {@link ContentSource} to get the next document from the source, and creates
368    * a {@link Document} object from the returned fields. If
369    * <code>reuseFields</code> was set to true, it will reuse {@link Document}
370    * and {@link Field} instances.
371    */
372   public Document makeDocument() throws Exception {
373     resetLeftovers();
374     DocData docData = source.getNextDocData(getDocState().docData);
375     Document doc = createDocument(docData, 0, -1);
376     return doc;
377   }
378 
379   /**
380    * Same as {@link #makeDocument()}, only this method creates a document of the
381    * given size input by <code>size</code>.
382    */
383   public Document makeDocument(int size) throws Exception {
384     LeftOver lvr = leftovr.get();
385     if (lvr == null || lvr.docdata == null || lvr.docdata.getBody() == null
386         || lvr.docdata.getBody().length() == 0) {
387       resetLeftovers();
388     }
389     DocData docData = getDocState().docData;
390     DocData dd = (lvr == null ? source.getNextDocData(docData) : lvr.docdata);
391     int cnt = (lvr == null ? 0 : lvr.cnt);
392     while (dd.getBody() == null || dd.getBody().length() < size) {
393       DocData dd2 = dd;
394       dd = source.getNextDocData(new DocData());
395       cnt = 0;
396       dd.setBody(dd2.getBody() + dd.getBody());
397     }
398     Document doc = createDocument(dd, size, cnt);
399     if (dd.getBody() == null || dd.getBody().length() == 0) {
400       resetLeftovers();
401     } else {
402       if (lvr == null) {
403         lvr = new LeftOver();
404         leftovr.set(lvr);
405       }
406       lvr.docdata = dd;
407       lvr.cnt = ++cnt;
408     }
409     return doc;
410   }

412   /** Reset inputs so that the test run would behave, input wise, as if it just started. */
413   public synchronized void resetInputs() throws IOException {
414     source.printStatistics("docs");
415     // re-initiate since properties by round may have changed.
416     setConfig(config, source);
417     source.resetInputs();
418     numDocsCreated.set(0);
419     resetLeftovers();
420   }

422   /** Set the configuration parameters of this doc maker. */
423   public void setConfig(Config config, ContentSource source) {
424     this.config = config;
425     this.source = source;
426 
427     boolean stored = config.get("doc.stored", false);
428     boolean bodyStored = config.get("doc.body.stored", stored);
429     boolean tokenized = config.get("doc.tokenized", true);
430     boolean bodyTokenized = config.get("doc.body.tokenized", tokenized);
431     boolean norms = config.get("doc.tokenized.norms", false);
432     boolean bodyNorms = config.get("doc.body.tokenized.norms", true);
433     boolean termVec = config.get("doc.term.vector", false);
434     boolean termVecPositions = config.get("doc.term.vector.positions", false);
435     boolean termVecOffsets = config.get("doc.term.vector.offsets", false);
436     
437     valType = new FieldType(TextField.TYPE_NOT_STORED);
438     valType.setStored(stored);
439     valType.setTokenized(tokenized);
440     valType.setOmitNorms(!norms);
441     valType.setStoreTermVectors(termVec);
442     valType.setStoreTermVectorPositions(termVecPositions);
443     valType.setStoreTermVectorOffsets(termVecOffsets);
444     valType.freeze();
445 
446     bodyValType = new FieldType(TextField.TYPE_NOT_STORED);
447     bodyValType.setStored(bodyStored);
448     bodyValType.setTokenized(bodyTokenized);
449     bodyValType.setOmitNorms(!bodyNorms);
450     bodyValType.setStoreTermVectors(termVec);
451     bodyValType.setStoreTermVectorPositions(termVecPositions);
452     bodyValType.setStoreTermVectorOffsets(termVecOffsets);
453     bodyValType.freeze();
454 
455     storeBytes = config.get("doc.store.body.bytes", false);
456     
457     reuseFields = config.get("doc.reuse.fields", true);
458 
459     // In a multi-rounds run, it is important to reset DocState since settings
460     // of fields may change between rounds, and this is the only way to reset
461     // the cache of all threads.
462     docState = new ThreadLocal<>();
463     
464     indexProperties = config.get("doc.index.props", false);
465 
466     updateDocIDLimit = config.get("doc.random.id.limit", -1);
467     if (updateDocIDLimit != -1) {
468       r = new Random(179);
469     }
470   }
