95    public static enum FacetMethod {
96      DV,  // DocValues
97      UIF, // UnInvertedField
98      ENUM,
99      STREAM,
100     SMART,
101     ;
102 
103     public static FacetMethod fromString(String method) {
104       if (method == null || method.length()==0) return null;
105       if ("dv".equals(method)) {
106         return DV;
107       } else if ("uif".equals(method)) {
108         return UIF;
109       } else if ("enum".equals(method)) {
110         return ENUM;
111       } else if ("smart".equals(method)) {
112         return SMART;
113       } else if ("stream".equals(method)) {
114         return STREAM;
115       }
116       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Unknown FacetField method " + method);
117     }
118   }

121   @Override
122   public FacetProcessor createFacetProcessor(FacetContext fcontext) {
123     SchemaField sf = fcontext.searcher.getSchema().getField(field);
124     FieldType ft = sf.getType();
125     boolean multiToken = sf.multiValued() || ft.multiValuedFieldCache();
126 
127     if (method == FacetMethod.ENUM && sf.indexed()) {
128       throw new UnsupportedOperationException();
129     } else if (method == FacetMethod.STREAM && sf.indexed()) {
130       return new FacetFieldProcessorStream(fcontext, this, sf);
131     }
132 
133     org.apache.lucene.document.FieldType.NumericType ntype = ft.getNumericType();
134 
135     if (sf.hasDocValues() && ntype==null) {
136       // single and multi-valued string docValues
137       return new FacetFieldProcessorDV(fcontext, this, sf);
138     }
139 
140     if (!multiToken) {
141       if (sf.getType().getNumericType() != null) {
142         // single valued numeric (docvalues or fieldcache)
143         return new FacetFieldProcessorNumeric(fcontext, this, sf);
144       } else {
145         // single valued string...
146         return new FacetFieldProcessorDV(fcontext, this, sf);
147       }
148     }
149 
150     // multivalued but field doesn't have docValues
151     if (method == FacetMethod.DV) {
152       return new FacetFieldProcessorDV(fcontext, this, sf);
153     }
154 
155     // Top-level multi-valued field cache (UIF)
156     return new FacetFieldProcessorUIF(fcontext, this, sf);
157   }

167 abstract class FacetFieldProcessor extends FacetProcessor<FacetField> {
168   SchemaField sf;
169   SlotAcc indexOrderAcc;
170   int effectiveMincount;
171 
172   Map<String,AggValueSource> deferredAggs;  // null if none
173 
174   // TODO: push any of this down to base class?
175 
176   //
177   // For sort="x desc", collectAcc would point to "x", and sortAcc would also point to "x".
178   // collectAcc would be used to accumulate all buckets, and sortAcc would be used to sort those buckets.
179   //
180   SlotAcc collectAcc;  // Accumulator to collect across entire domain (in addition to the countAcc).  May be null.
181   SlotAcc sortAcc;     // Accumulator to use for sorting *only* (i.e. not used for collection). May be an alias of countAcc, collectAcc, or indexOrderAcc
182   SlotAcc[] otherAccs; // Accumulators that do not need to be calculated across all buckets.
183 
184   SpecialSlotAcc allBucketsAcc;  // this can internally refer to otherAccs and/or collectAcc. setNextReader should be called on otherAccs directly if they exist.

188     super(fcontext, freq);
189     this.sf = sf;
190     this.effectiveMincount = (int)(fcontext.isShard() ? Math.min(1 , freq.mincount) : freq.mincount);
191   }

198   // This is used to create accs for second phase (or to create accs for all aggs)
199   @Override
200   protected void createAccs(int docCount, int slotCount) throws IOException {
201     if (accMap == null) {
202       accMap = new LinkedHashMap<>();
203     }
204 
205     // allow a custom count acc to be used
206     if (countAcc == null) {
207       countAcc = new CountSlotArrAcc(fcontext, slotCount);
208       countAcc.key = "count";
209     }
210 
211     if (accs != null) {
212       // reuse these accs, but reset them first
213       for (SlotAcc acc : accs) {
214         acc.reset();
215       }
216       return;
217     } else {
218       accs = new SlotAcc[ freq.getFacetStats().size() ];
219     }
220 
221     int accIdx = 0;
222     for (Map.Entry<String,AggValueSource> entry : freq.getFacetStats().entrySet()) {
223       SlotAcc acc = null;
224       if (slotCount == 1) {
225         acc = accMap.get(entry.getKey());
226         if (acc != null) {
227           acc.reset();
228         }
229       }
230       if (acc == null) {
231         acc = entry.getValue().createSlotAcc(fcontext, docCount, slotCount);
232         acc.key = entry.getKey();
233         accMap.put(acc.key, acc);
234       }
235       accs[accIdx++] = acc;
236     }
237   }
238 
239   void createCollectAcc(int numDocs, int numSlots) throws IOException {
240     accMap = new LinkedHashMap<>();
241 
242     // we always count...
243     // allow a subclass to set a custom counter.
244     if (countAcc == null) {
245       countAcc = new CountSlotArrAcc(fcontext, numSlots);
246     }
247 
248     if ("count".equals(freq.sortVariable)) {
249       sortAcc = countAcc;
250       deferredAggs = freq.getFacetStats();
251     } else if ("index".equals(freq.sortVariable)) {
252       // allow subclass to set indexOrderAcc first
253       if (indexOrderAcc == null) {
254         // This sorting accumulator just goes by the slot number, so does not need to be collected
255         // and hence does not need to find it's way into the accMap or accs array.
256         indexOrderAcc = new SortSlotAcc(fcontext);
257       }
258       sortAcc = indexOrderAcc;
259       deferredAggs = freq.getFacetStats();
260     } else {
261       AggValueSource sortAgg = freq.getFacetStats().get(freq.sortVariable);
262       if (sortAgg != null) {
263         collectAcc = sortAgg.createSlotAcc(fcontext, numDocs, numSlots);
264         collectAcc.key = freq.sortVariable; // TODO: improve this
265       }
266       sortAcc = collectAcc;
267       deferredAggs = new HashMap<>(freq.getFacetStats());
268       deferredAggs.remove(freq.sortVariable);
269     }
270 
271     if (deferredAggs.size() == 0) {
272       deferredAggs = null;
273     }
274 
275     boolean needOtherAccs = freq.allBuckets;  // TODO: use for missing too...
276 
277     if (!needOtherAccs) {
278       // we may need them later, but we don't want to create them now
279       // otherwise we won't know if we need to call setNextReader on them.
280       return;
281     }
282 
283     // create the deferred aggs up front for use by allBuckets
284     createOtherAccs(numDocs, 1);
285   }

288   void createOtherAccs(int numDocs, int numSlots) throws IOException {
289     if (otherAccs != null) {
290       // reuse existing accumulators
291       for (SlotAcc acc : otherAccs) {
292         acc.reset();  // todo - make reset take numDocs and numSlots?
293       }
294       return;
295     }
296 
297     int numDeferred = deferredAggs == null ? 0 : deferredAggs.size();
298     if (numDeferred <= 0) return;
299 
300     otherAccs = new SlotAcc[ numDeferred ];
301 
302     int otherAccIdx = 0;
303     for (Map.Entry<String,AggValueSource> entry : deferredAggs.entrySet()) {
304       AggValueSource agg = entry.getValue();
305       SlotAcc acc = agg.createSlotAcc(fcontext, numDocs, numSlots);
306       acc.key = entry.getKey();
307       accMap.put(acc.key, acc);
308       otherAccs[otherAccIdx++] = acc;
309     }
310 
311     if (numDeferred == freq.getFacetStats().size()) {
312       // accs and otherAccs are the same...
313       accs = otherAccs;
314     }
315   }

318   int collectFirstPhase(DocSet docs, int slot) throws IOException {
319     int num = -1;
320     if (collectAcc != null) {
321       num = collectAcc.collect(docs, slot);
322     }
323     if (allBucketsAcc != null) {
324       num = allBucketsAcc.collect(docs, slot);
325     }
326     return num >= 0 ? num : docs.size();
327   }

329   void collectFirstPhase(int segDoc, int slot) throws IOException {
330     if (collectAcc != null) {
331       collectAcc.collect(segDoc, slot);
332     }
333     if (allBucketsAcc != null) {
334       allBucketsAcc.collect(segDoc, slot);
335     }
336   }

339   void fillBucket(SimpleOrderedMap<Object> target, int count, int slotNum, DocSet subDomain, Query filter) throws IOException {
340     target.add("count", count);
341     if (count <= 0 && !freq.processEmpty) return;
342 
343     if (collectAcc != null && slotNum >= 0) {
344       collectAcc.setValues(target, slotNum);
345     }
346 
347     createOtherAccs(-1, 1);
348 
349     if (otherAccs == null && freq.subFacets.isEmpty()) return;
350 
351     if (subDomain == null) {
352       subDomain = fcontext.searcher.getDocSet(filter, fcontext.base);
353     }
354 
355     // if no subFacets, we only need a DocSet
356     // otherwise we need more?
357     // TODO: save something generic like "slotNum" in the context and use that to implement things like filter exclusion if necessary?
358     // Hmmm, but we need to look up some stuff anyway (for the label?)
359     // have a method like "DocSet applyConstraint(facet context, DocSet parent)"
360     // that's needed for domain changing things like joins anyway???
361 
362     if (otherAccs != null) {
363       // do acc at a time (traversing domain each time) or do all accs for each doc?
364       for (SlotAcc acc : otherAccs) {
365         acc.reset(); // TODO: only needed if we previously used for allBuckets or missing
366         acc.collect(subDomain, 0);
367         acc.setValues(target, 0);
368       }
369     }
370 
371     processSubs(target, filter, subDomain);
372   }

375   @Override
376   protected void processStats(SimpleOrderedMap<Object> bucket, DocSet docs, int docCount) throws IOException {
377     if (docCount == 0 && !freq.processEmpty || freq.getFacetStats().size() == 0) {
378       bucket.add("count", docCount);
379       return;
380     }
381     createAccs(docCount, 1);
382     int collected = collect(docs, 0);
383 
384     // countAcc.incrementCount(0, collected);  // should we set the counton the acc instead of just passing it?
385 
386     assert collected == docCount;
387     addStats(bucket, collected, 0);
388   }

393     if (count > 0 || freq.processEmpty) {
394       for (SlotAcc acc : accs) {
395         acc.setValues(target, slotNum);
396       }
397     }

516 // base class for FC style of facet counting (single and multi-valued strings)
517 abstract class FacetFieldProcessorFCBase extends FacetFieldProcessor {
518   BytesRefBuilder prefixRef;
519   int startTermIndex;
520   int endTermIndex;
521   int nTerms;
522   int nDocs;
523   int maxSlots;
524 
525   int allBucketsSlot = -1;  // slot for the primary Accs (countAcc, collectAcc)
526 
527   public FacetFieldProcessorFCBase(FacetContext fcontext, FacetField freq, SchemaField sf) {
528     super(fcontext, freq, sf);
529   }
530 
531   @Override
532   public void process() throws IOException {
533     super.process();
534     sf = fcontext.searcher.getSchema().getField(freq.field);
535     response = getFieldCacheCounts();
536   }
537 
538 
539   /** this BytesRef may be shared across calls and should be deep-cloned if necessary */
540   abstract protected BytesRef lookupOrd(int ord) throws IOException;
541   abstract protected void findStartAndEndOrds() throws IOException;
542   abstract protected void collectDocs() throws IOException;
543 
544 
545   public SimpleOrderedMap<Object> getFieldCacheCounts() throws IOException {
546     String prefix = freq.prefix;
547     if (prefix == null || prefix.length() == 0) {
548       prefixRef = null;
549     } else {
550       prefixRef = new BytesRefBuilder();
551       prefixRef.copyChars(prefix);
552     }
553 
554     findStartAndEndOrds();
555 
556     maxSlots = nTerms;
557 
558     if (freq.allBuckets) {
559       allBucketsSlot = maxSlots++;
560     }
561 
562     createCollectAcc(nDocs, maxSlots);
563 
564     if (freq.allBuckets) {
565       allBucketsAcc = new SpecialSlotAcc(fcontext, collectAcc, allBucketsSlot, otherAccs, 0);
566     }
567 
568     collectDocs();
569 
570     return findTopSlots();
571   }
572 
573 
574   protected SimpleOrderedMap<Object> findTopSlots() throws IOException {
575     SimpleOrderedMap<Object> res = new SimpleOrderedMap<>();
576 
577     int numBuckets = 0;
578     List<Object> bucketVals = null;
579     if (freq.numBuckets && fcontext.isShard()) {
580       bucketVals = new ArrayList(100);
581     }
582 
583     int off = fcontext.isShard() ? 0 : (int) freq.offset;
584     // add a modest amount of over-request if this is a shard request
585     int lim = freq.limit >= 0 ? (fcontext.isShard() ? (int)(freq.limit*1.1+4) : (int)freq.limit) : Integer.MAX_VALUE;
586 
587     int maxsize = (int)(freq.limit >= 0 ?  freq.offset + lim : Integer.MAX_VALUE - 1);
588     maxsize = Math.min(maxsize, nTerms);
589 
590     final int sortMul = freq.sortDirection.getMultiplier();
591     final SlotAcc sortAcc = this.sortAcc;
592 
593     PriorityQueue<Slot> queue = new PriorityQueue<Slot>(maxsize) {
594       @Override
595       protected boolean lessThan(Slot a, Slot b) {
596         int cmp = sortAcc.compare(a.slot, b.slot) * sortMul;
597         return cmp == 0 ? b.slot < a.slot : cmp < 0;
598       }
599     };
600 
601     Slot bottom = null;
602     for (int i = 0; i < nTerms; i++) {
603       // screen out buckets not matching mincount immediately (i.e. don't even increment numBuckets)
604       if (effectiveMincount > 0 && countAcc.getCount(i) < effectiveMincount) {
605         continue;
606       }
607 
608       numBuckets++;
609       if (bucketVals != null && bucketVals.size()<100) {
610         int ord = startTermIndex + i;
611         BytesRef br = lookupOrd(ord);
612         Object val = sf.getType().toObject(sf, br);
613         bucketVals.add(val);
614       }
615 
616 
617       if (bottom != null) {
618         if (sortAcc.compare(bottom.slot, i) * sortMul < 0) {
619           bottom.slot = i;
620           bottom = queue.updateTop();
621         }
622       } else if (lim > 0) {
623         // queue not full
624         Slot s = new Slot();
625         s.slot = i;
626         queue.add(s);
627         if (queue.size() >= maxsize) {
628           bottom = queue.top();
629         }
630       }
631     }
632 
633     if (freq.numBuckets) {
634       if (!fcontext.isShard()) {
635         res.add("numBuckets", numBuckets);
636       } else {
637         SimpleOrderedMap map = new SimpleOrderedMap(2);
638         map.add("numBuckets", numBuckets);
639         map.add("vals", bucketVals);
640         res.add("numBuckets", map);
641       }
642     }
643 
644     // if we are deep paging, we don't have to order the highest "offset" counts.
645     int collectCount = Math.max(0, queue.size() - off);
646     assert collectCount <= lim;
647     int[] sortedSlots = new int[collectCount];
648     for (int i = collectCount - 1; i >= 0; i--) {
649       sortedSlots[i] = queue.pop().slot;
650     }
651 
652     if (freq.allBuckets) {
653       SimpleOrderedMap<Object> allBuckets = new SimpleOrderedMap<>();
654       allBuckets.add("count", allBucketsAcc.getSpecialCount());
655       if (allBucketsAcc != null) {
656         allBucketsAcc.setValues(allBuckets, allBucketsSlot);
657       }
658       res.add("allBuckets", allBuckets);
659     }
660 
661     ArrayList bucketList = new ArrayList(collectCount);
662     res.add("buckets", bucketList);
663 
664 
665     // TODO: do this with a callback instead?
666     boolean needFilter = deferredAggs != null || freq.getSubFacets().size() > 0;
667 
668     for (int slotNum : sortedSlots) {
669       SimpleOrderedMap<Object> bucket = new SimpleOrderedMap<>();
670 
671       // get the ord of the slot...
672       int ord = startTermIndex + slotNum;
673 
674       BytesRef br = lookupOrd(ord);
675       Object val = sf.getType().toObject(sf, br);
676 
677       bucket.add("val", val);
678 
679       TermQuery filter = needFilter ? new TermQuery(new Term(sf.getName(), br)) : null;
680       fillBucket(bucket, countAcc.getCount(slotNum), slotNum, null, filter);
681 
682       bucketList.add(bucket);
683     }
684 
685     if (freq.missing) {
686       SimpleOrderedMap<Object> missingBucket = new SimpleOrderedMap<>();
687       fillBucket(missingBucket, getFieldMissingQuery(fcontext.searcher, freq.field), null);
688       res.add("missing", missingBucket);
689     }
690 
691     return res;
692   }

701 // UnInvertedField implementation of field faceting
702 class FacetFieldProcessorUIF extends FacetFieldProcessorFCBase {

712     uif = UnInvertedField.getUnInvertedField(freq.field, fcontext.searcher);
713     te = uif.getOrdTermsEnum( fcontext.searcher.getLeafReader() );    // "te" can be null
714

716     endTermIndex = uif.numTerms();  // one past the end
717 
718     if (prefixRef != null && te != null) {
719       if (te.seekCeil(prefixRef.get()) == TermsEnum.SeekStatus.END) {
720         startTermIndex = uif.numTerms();
721       } else {

724       prefixRef.append(UnicodeUtil.BIG_TERM);
725       if (te.seekCeil(prefixRef.get()) == TermsEnum.SeekStatus.END) {
726         endTermIndex = uif.numTerms();
727       } else {
728         endTermIndex = (int) te.ord();
729       }
730     }
731 
732     nTerms = endTermIndex - startTermIndex;
733   }

749   long bucketsToSkip;
750   long bucketsReturned;
751 
752   boolean closed;
753   boolean countOnly;
754   boolean hasSubFacets;  // true if there are subfacets
755   int minDfFilterCache;
756   DocSet docs;
757   DocSet fastForRandomSet;
758   TermsEnum termsEnum = null;

760   PostingsEnum postingsEnum;

768     super(fcontext, freq, sf);

771   @Override
772   public void close() throws IOException {
773     if (!closed) {
774       closed = true;
775       // fcontext.base.decref();  // OFF-HEAP
776     }
777   }

780   @Override
781   public void process() throws IOException {
782     super.process();
783 
784     // We need to keep the fcontext open after processing is done (since we will be streaming in the response writer).
785     // But if the connection is broken, we want to clean up.
786     // fcontext.base.incref();  // OFF-HEAP
787     fcontext.qcontext.addCloseHook(this);
788 
789     setup();
790     response = new SimpleOrderedMap<>();
791     response.add("buckets", new Iterator() {
792       boolean retrieveNext = true;
793       Object val;
794 
795       @Override
796       public boolean hasNext() {
797         if (retrieveNext) {
798           val = nextBucket();
799         }
800         retrieveNext = false;
801         return val != null;
802       }
803 
804       @Override
805       public Object next() {
806         if (retrieveNext) {
807           val = nextBucket();
808         }
809         retrieveNext = true;
810         if (val == null) {
811           // Last value, so clean up.  In the case that we are doing streaming facets within streaming facets,
812           // the number of close hooks could grow very large, so we want to remove ourselves.
813           boolean removed = fcontext.qcontext.removeCloseHook(FacetFieldProcessorStream.this);
814           assert removed;
815           try {
816             close();
817           } catch (IOException e) {
818             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Error during facet streaming close", e);
819           }
820         }
821         return val;
822       }
823 
824       @Override
825       public void remove() {
826         throw new UnsupportedOperationException();
827       }
828     });
829   }

833   public void setup() throws IOException {
834 
835     countOnly = freq.facetStats.size() == 0 || freq.facetStats.values().iterator().next() instanceof CountAgg;
836     hasSubFacets = freq.subFacets.size() > 0;
837     bucketsToSkip = freq.offset;
838 
839     createAccs(-1, 1);
840 
841     // Minimum term docFreq in order to use the filterCache for that term.
842     int defaultMinDf = Math.max(fcontext.searcher.maxDoc() >> 4, 3);  // (minimum of 3 is for test coverage purposes)
843     int minDfFilterCache = freq.cacheDf == 0 ? defaultMinDf : freq.cacheDf;
844     if (minDfFilterCache == -1) minDfFilterCache = Integer.MAX_VALUE;  // -1 means never cache
845 
846     docs = fcontext.base;
847     fastForRandomSet = null;
848 
849     if (freq.prefix != null) {
850       String indexedPrefix = sf.getType().toInternal(freq.prefix);
851       startTermBytes = new BytesRef(indexedPrefix);
852     } else if (sf.getType().getNumericType() != null) {
853       String triePrefix = TrieField.getMainValuePrefix(sf.getType());
854       if (triePrefix != null) {
855         startTermBytes = new BytesRef(triePrefix);
856       }
857     }
858 
859     Fields fields = fcontext.searcher.getLeafReader().fields();
860     Terms terms = fields == null ? null : fields.terms(sf.getName());
861 
862 
863     termsEnum = null;
864     deState = null;
865     term = null;
866 
867 
868     if (terms != null) {
869 
870       termsEnum = terms.iterator();
871 
872       // TODO: OPT: if seek(ord) is supported for this termsEnum, then we could use it for
873       // facet.offset when sorting by index order.
874 
875       if (startTermBytes != null) {
876         if (termsEnum.seekCeil(startTermBytes) == TermsEnum.SeekStatus.END) {
877           termsEnum = null;
878         } else {
879           term = termsEnum.term();
880         }
881       } else {
882         // position termsEnum on first term
883         term = termsEnum.next();
884       }
885     }
886 
887     List<LeafReaderContext> leafList = fcontext.searcher.getTopReaderContext().leaves();
888     leaves = leafList.toArray( new LeafReaderContext[ leafList.size() ]);
889   }

892   public SimpleOrderedMap<Object> nextBucket() {
893     try {
894       return _nextBucket();
895     } catch (Exception e) {
896       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Error during facet streaming", e);
897     }
898   }

900   public SimpleOrderedMap<Object> _nextBucket() throws IOException {
901     DocSet termSet = null;
902 
903     try {
904       while (term != null) {
905 
906         if (startTermBytes != null && !StringHelper.startsWith(term, startTermBytes)) {
907           break;
908         }
909 
910         int df = termsEnum.docFreq();
911         if (df < effectiveMincount) {
912           term = termsEnum.next();
913           continue;
914         }
915 
916         if (termSet != null) {
917           // termSet.decref(); // OFF-HEAP
918           termSet = null;
919         }
920 
921         int c = 0;
922 
923         if (hasSubFacets || df >= minDfFilterCache) {
924           // use the filter cache
925 
926           if (deState == null) {
927             deState = new SolrIndexSearcher.DocsEnumState();
928             deState.fieldName = sf.getName();
929             deState.liveDocs = fcontext.searcher.getLeafReader().getLiveDocs();
930             deState.termsEnum = termsEnum;
931             deState.postingsEnum = postingsEnum;
932             deState.minSetSizeCached = minDfFilterCache;
933           }
934 
935             if (hasSubFacets || !countOnly) {
936               DocSet termsAll = fcontext.searcher.getDocSet(deState);
937               termSet = docs.intersection(termsAll);
938               // termsAll.decref(); // OFF-HEAP
939               c = termSet.size();
940             } else {
941               c = fcontext.searcher.numDocs(docs, deState);
942             }
943             postingsEnum = deState.postingsEnum;
944 
945             resetStats();
946 
947             if (!countOnly) {
948               collect(termSet, 0);
949             }
950 
951         } else {
952           // We don't need the docset here (meaning no sub-facets).
953           // if countOnly, then we are calculating some other stats...
954           resetStats();
955 
956           // lazy convert to fastForRandomSet
957           if (fastForRandomSet == null) {
958             fastForRandomSet = docs;
959             if (docs instanceof SortedIntDocSet) {  // OFF-HEAP todo: also check for native version
960               SortedIntDocSet sset = (SortedIntDocSet) docs;
961               fastForRandomSet = new HashDocSet(sset.getDocs(), 0, sset.size());
962             }
963           }
964           // iterate over TermDocs to calculate the intersection
965           postingsEnum = termsEnum.postings(postingsEnum, PostingsEnum.NONE);
966 
967           if (postingsEnum instanceof MultiPostingsEnum) {
968             MultiPostingsEnum.EnumWithSlice[] subs = ((MultiPostingsEnum) postingsEnum).getSubs();
969             int numSubs = ((MultiPostingsEnum) postingsEnum).getNumSubs();
970             for (int subindex = 0; subindex < numSubs; subindex++) {
971               MultiPostingsEnum.EnumWithSlice sub = subs[subindex];
972               if (sub.postingsEnum == null) continue;
973               int base = sub.slice.start;
974               int docid;
975 
976               if (countOnly) {
977                 while ((docid = sub.postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
978                   if (fastForRandomSet.exists(docid + base)) c++;
979                 }
980               } else {
981                 setNextReader(leaves[sub.slice.readerIndex]);
982                 while ((docid = sub.postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
983                   if (fastForRandomSet.exists(docid + base)) {
984                     c++;
985                     collect(docid, 0);
986                   }
987                 }
988               }
989 
990             }
991           } else {
992             int docid;
993             if (countOnly) {
994               while ((docid = postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
995                 if (fastForRandomSet.exists(docid)) c++;
996               }
997             } else {
998               setNextReader(leaves[0]);
999               while ((docid = postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
1000                 if (fastForRandomSet.exists(docid)) {
1001                   c++;
1002                   collect(docid, 0);
1003                 }
1004               }
1005             }
1006           }
1007 
1008         }
1009 
1010 
1011 
1012         if (c < effectiveMincount) {
1013           term = termsEnum.next();
1014           continue;
1015         }
1016 
1017         // handle offset and limit
1018         if (bucketsToSkip > 0) {
1019           bucketsToSkip--;
1020           term = termsEnum.next();
1021           continue;
1022         }
1023 
1024         if (freq.limit >= 0 && ++bucketsReturned > freq.limit) {
1025           return null;
1026         }
1027 
1028         // set count in case other stats depend on it
1029         countAcc.incrementCount(0, c);
1030 
1031         // OK, we have a good bucket to return... first get bucket value before moving to next term
1032         Object bucketVal = sf.getType().toObject(sf, term);
1033         TermQuery bucketQuery = hasSubFacets ? new TermQuery(new Term(freq.field, term)) : null;
1034         term = termsEnum.next();
1035 
1036         SimpleOrderedMap<Object> bucket = new SimpleOrderedMap<>();
1037         bucket.add("val", bucketVal);
1038         addStats(bucket, 0);
1039         if (hasSubFacets) {
1040           processSubs(bucket, bucketQuery, termSet);
1041         }
1042 
1043         // TODO... termSet needs to stick around for streaming sub-facets?
1044 
1045         return bucket;
1046 
1047       }
1048 
1049     } finally {
1050       if (termSet != null) {
1051         // termSet.decref();  // OFF-HEAP
1052         termSet = null;
1053       }
1054     }
1055 
1056 
1057     // end of the iteration
1058     return null;
1059   }
