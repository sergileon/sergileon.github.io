20  import java.lang.reflect.Method;
21  import java.util.HashSet;
22  import java.util.Set;
23  
24  import org.apache.lucene.util.LuceneTestCase;
25  import org.junit.Test;
26  
27  public class TestFilterSpans extends LuceneTestCase {
28  
29    @Test
30    public void testOverrides() throws Exception {
31      // verify that all methods of Spans are overridden by FilterSpans,
32      // except those under the 'exclude' list
33      Set<Method> exclude = new HashSet<>();
34      exclude.add(FilterSpans.class.getMethod("freq"));
35      exclude.add(FilterSpans.class.getMethod("score"));
36      exclude.add(FilterSpans.class.getMethod("sloppyFreq"));
37      for (Method m : FilterSpans.class.getMethods()) {
38        if (m.getDeclaringClass() == Spans.class) {
39          assertTrue("method " + m.getName() + " not overridden!", exclude.contains(m));
40        }
41      }
42    }
43    
44  }
