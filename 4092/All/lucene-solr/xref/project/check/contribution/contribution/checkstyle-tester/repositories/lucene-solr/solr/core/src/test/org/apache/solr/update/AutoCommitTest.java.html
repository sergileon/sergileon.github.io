44  class NewSearcherListener implements SolrEventListener {
45  
46    enum TriggerOn {Both, Soft, Hard}
47  
48    private volatile boolean triggered = false;
49    private volatile TriggerOn lastType;
50    private volatile TriggerOn triggerOnType;
51    private volatile SolrIndexSearcher newSearcher;
52  
53    public NewSearcherListener() {
54      this(TriggerOn.Both);
55    }
56  
57    public NewSearcherListener(TriggerOn type) {
58      this.triggerOnType = type;
59    }
60  
61    @Override
62    public void init(NamedList args) {}
63  
64    @Override
65    public void newSearcher(SolrIndexSearcher newSearcher,
66        SolrIndexSearcher currentSearcher) {
67      if (triggerOnType == TriggerOn.Soft && lastType == TriggerOn.Soft) {
68        triggered = true;
69      } else if (triggerOnType == TriggerOn.Hard && lastType == TriggerOn.Hard) {
70        triggered = true;
71      } else if (triggerOnType == TriggerOn.Both) {
72        triggered = true;
73      }
74      this.newSearcher = newSearcher;
75      // log.info("TEST: newSearcher event: triggered="+triggered+" newSearcher="+newSearcher);
76    }
77  
78    @Override
79    public void postCommit() {
80      lastType = TriggerOn.Hard;
81    }
82  
83    @Override
84    public void postSoftCommit() {
85      lastType = TriggerOn.Soft;
86    }
87  
88    public void reset() {
89      triggered = false;
90      // log.info("TEST: trigger reset");
91    }
92  
93    boolean waitForNewSearcher(int timeoutMs) {
94      TimeOut timeout = new TimeOut(timeoutMs, TimeUnit.MILLISECONDS);
95      while (! timeout.hasTimedOut()) {
96        if (triggered) {
97          // check if the new searcher has been registered yet
98          RefCounted<SolrIndexSearcher> registeredSearcherH = newSearcher.getCore().getSearcher();
99          SolrIndexSearcher registeredSearcher = registeredSearcherH.get();
100         registeredSearcherH.decref();
101         if (registeredSearcher == newSearcher) return true;
102         // log.info("TEST: waiting for searcher " + newSearcher + " to be registered.  current=" + registeredSearcher);
103       }
104 
105       try {
106         Thread.sleep(250);
107       } catch (InterruptedException e) {}
108 
109     }
110     return false;
111   }
112 }

145   /**
146    * Take a string and make it an iterable ContentStream
147    * 
148    * This should be moved to a helper class. (it is useful for the client too!)
149    */
150   public static Collection<ContentStream> toContentStreams( final String str, final String contentType )
151   {
152     ArrayList<ContentStream> streams = new ArrayList<>();
153     ContentStreamBase stream = new ContentStreamBase.StringStream( str );
154     if (contentType != null) stream.setContentType( contentType );
155     streams.add( stream );
156     return streams;
157   }

159   public void testMaxDocs() throws Exception {
160     SolrCore core = h.getCore();
161     
162     NewSearcherListener trigger = new NewSearcherListener();
163 
164     DirectUpdateHandler2 updateHandler = (DirectUpdateHandler2)core.getUpdateHandler();
165     CommitTracker tracker = updateHandler.softCommitTracker;
166     tracker.setTimeUpperBound(-1);
167     tracker.setDocsUpperBound(14);
168     core.registerNewSearcherListener(trigger);
169   
170     
171     UpdateRequestHandler handler = new UpdateRequestHandler();
172     handler.init( null );
173     
174     MapSolrParams params = new MapSolrParams( new HashMap<String, String>() );
175     
176     // Add documents
177     SolrQueryResponse rsp = new SolrQueryResponse();
178     SolrQueryRequestBase req = new SolrQueryRequestBase( core, params ) {};
179     for( int i=0; i<14; i++ ) {
180       req.setContentStreams( toContentStreams(
181         adoc("id", Integer.toString(i), "subject", "info" ), null ) );
182       handler.handleRequest( req, rsp );
183     }
184     // It should not be there right away
185     assertQ("shouldn't find any", req("id:1") ,"//result[@numFound=0]" );
186     assertEquals( 0, tracker.getCommitCount());
187 
188     req.setContentStreams( toContentStreams(
189         adoc("id", "14", "subject", "info" ), null ) );
190     handler.handleRequest( req, rsp );
191 
192     assertTrue(trigger.waitForNewSearcher(15000));
193 
194     req.setContentStreams( toContentStreams(
195         adoc("id", "15", "subject", "info" ), null ) );
196     handler.handleRequest( req, rsp );
197       
198     // Now make sure we can find it
199     assertQ("should find one", req("id:14") ,"//result[@numFound=1]" );
200     assertEquals( 1, tracker.getCommitCount());
201     // But not the one added afterward
202     assertQ("should not find one", req("id:15") ,"//result[@numFound=0]" );
203     assertEquals( 1, tracker.getCommitCount());
204     
205   }

207   public void testMaxTime() throws Exception {
208     SolrCore core = h.getCore();
209     
210     NewSearcherListener trigger = new NewSearcherListener();    
211     core.registerNewSearcherListener(trigger);
212     DirectUpdateHandler2 updater = (DirectUpdateHandler2) core.getUpdateHandler();
213     CommitTracker tracker = updater.softCommitTracker;
214     // too low of a number can cause a slow host to commit before the test code checks that it
215     // isn't there... causing a failure at "shouldn't find any"
216     tracker.setTimeUpperBound(1000);
217     tracker.setDocsUpperBound(-1);
218     // updater.commitCallbacks.add(trigger);
219     
220     UpdateRequestHandler handler = new UpdateRequestHandler();
221     handler.init( null );
222     
223     MapSolrParams params = new MapSolrParams( new HashMap<String, String>() );
224     
225     // Add a single document
226     SolrQueryResponse rsp = new SolrQueryResponse();
227     SolrQueryRequestBase req = new SolrQueryRequestBase( core, params ) {};
228     req.setContentStreams( toContentStreams(
229       adoc("id", "529", "field_t", "what's inside?", "subject", "info"), null ) );
230     trigger.reset();
231     handler.handleRequest( req, rsp );
232 
233     // Check it it is in the index
234     assertQ("shouldn't find any", req("id:529") ,"//result[@numFound=0]" );
235 
236     // Wait longer than the autocommit time
237     assertTrue(trigger.waitForNewSearcher(45000));
238     trigger.reset();
239     req.setContentStreams( toContentStreams(
240       adoc("id", "530", "field_t", "what's inside?", "subject", "info"), null ) );
241     handler.handleRequest( req, rsp );
242       
243     // Now make sure we can find it
244     assertQ("should find one", req("id:529") ,"//result[@numFound=1]" );
245     // But not this one
246     assertQ("should find none", req("id:530") ,"//result[@numFound=0]" );
247     
248     // Delete the document
249     assertU( delI("529") );
250     assertQ("deleted, but should still be there", req("id:529") ,"//result[@numFound=1]" );
251     // Wait longer than the autocommit time
252     assertTrue(trigger.waitForNewSearcher(30000));
253     trigger.reset();
254     req.setContentStreams( toContentStreams(
255       adoc("id", "550", "field_t", "what's inside?", "subject", "info"), null ) );
256     handler.handleRequest( req, rsp );
257     assertEquals( 2, tracker.getCommitCount() );
258     assertQ("deleted and time has passed", req("id:529") ,"//result[@numFound=0]" );
259     
260     // now make the call 10 times really fast and make sure it 
261     // only commits once
262     req.setContentStreams( toContentStreams(
263         adoc("id", "500" ), null ) );
264     for( int i=0;i<10; i++ ) {
265       handler.handleRequest( req, rsp );
266     }
267     assertQ("should not be there yet", req("id:500") ,"//result[@numFound=0]" );
268     
269     // Wait longer than the autocommit time
270     assertTrue(trigger.waitForNewSearcher(45000));
271     trigger.reset();
272     
273     req.setContentStreams( toContentStreams(
274       adoc("id", "531", "field_t", "what's inside?", "subject", "info"), null ) );
275     handler.handleRequest( req, rsp );
276     assertEquals( 3, tracker.getCommitCount() );
277 
278     assertQ("now it should", req("id:500") ,"//result[@numFound=1]" );
279   }

281   public void testCommitWithin() throws Exception {
282     SolrCore core = h.getCore();
283     
284     NewSearcherListener trigger = new NewSearcherListener();    
285     core.registerNewSearcherListener(trigger);
286     DirectUpdateHandler2 updater = (DirectUpdateHandler2) core.getUpdateHandler();
287     CommitTracker tracker = updater.softCommitTracker;
288     tracker.setTimeUpperBound(0);
289     tracker.setDocsUpperBound(-1);
290     
291     UpdateRequestHandler handler = new UpdateRequestHandler();
292     handler.init( null );
293     
294     MapSolrParams params = new MapSolrParams( new HashMap<String, String>() );
295     
296     // Add a single document with commitWithin == 4 second
297     SolrQueryResponse rsp = new SolrQueryResponse();
298     SolrQueryRequestBase req = new SolrQueryRequestBase( core, params ) {};
299     req.setContentStreams( toContentStreams(
300       adoc(4000, "id", "529", "field_t", "what's inside?", "subject", "info"), null ) );
301     trigger.reset();
302     handler.handleRequest( req, rsp );
303 
304     // Check it isn't in the index
305     assertQ("shouldn't find any", req("id:529") ,"//result[@numFound=0]" );
306     
307     // Wait longer than the commitWithin time
308     assertTrue("commitWithin failed to commit", trigger.waitForNewSearcher(30000));
309 
310     // Add one document without commitWithin
311     req.setContentStreams( toContentStreams(
312         adoc("id", "530", "field_t", "what's inside?", "subject", "info"), null ) );
313       trigger.reset();
314       handler.handleRequest( req, rsp );
315       
316     // Check it isn't in the index
317     assertQ("shouldn't find any", req("id:530") ,"//result[@numFound=0]" );
318     
319     // Delete one document with commitWithin
320     req.setContentStreams( toContentStreams(
321       delI("529", "commitWithin", "1000"), null ) );
322     trigger.reset();
323     handler.handleRequest( req, rsp );
324       
325     // Now make sure we can find it
326     assertQ("should find one", req("id:529") ,"//result[@numFound=1]" );
327     
328     // Wait for the commit to happen
329     assertTrue("commitWithin failed to commit", trigger.waitForNewSearcher(30000));
330     
331     // Now we shouldn't find it
332     assertQ("should find none", req("id:529") ,"//result[@numFound=0]" );
333     // ... but we should find the new one
334     assertQ("should find one", req("id:530") ,"//result[@numFound=1]" );
335     
336     trigger.reset();
337     
338     // now make the call 10 times really fast and make sure it 
339     // only commits once
340     req.setContentStreams( toContentStreams(
341         adoc(2000, "id", "500" ), null ) );
342     for( int i=0;i<10; i++ ) {
343       handler.handleRequest( req, rsp );
344     }
345     assertQ("should not be there yet", req("id:500") ,"//result[@numFound=0]" );
346     
347     // the same for the delete
348     req.setContentStreams( toContentStreams(
349         delI("530", "commitWithin", "1000"), null ) );
350     for( int i=0;i<10; i++ ) {
351       handler.handleRequest( req, rsp );
352     }
353     assertQ("should be there", req("id:530") ,"//result[@numFound=1]" );
354     
355     assertTrue("commitWithin failed to commit", trigger.waitForNewSearcher(30000));
356     assertQ("should be there", req("id:500") ,"//result[@numFound=1]" );
357     assertQ("should not be there", req("id:530") ,"//result[@numFound=0]" );
358     
359     assertEquals(3, tracker.getCommitCount());
360   }
