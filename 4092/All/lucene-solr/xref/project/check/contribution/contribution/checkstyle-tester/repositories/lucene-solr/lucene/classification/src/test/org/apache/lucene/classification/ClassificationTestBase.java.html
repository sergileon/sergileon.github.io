90    protected void checkCorrectClassification(Classifier<T> classifier, String inputDoc, T expectedResult, Analyzer analyzer, String textFieldName, String classFieldName) throws Exception {
91      checkCorrectClassification(classifier, inputDoc, expectedResult, analyzer, textFieldName, classFieldName, null);
92    }

94    protected void checkCorrectClassification(Classifier<T> classifier, String inputDoc, T expectedResult, Analyzer analyzer, String textFieldName, String classFieldName, Query query) throws Exception {
95      LeafReader leafReader = null;
96      try {
97        leafReader = populateSampleIndex(analyzer);
98        classifier.train(leafReader, textFieldName, classFieldName, analyzer, query);
99        ClassificationResult<T> classificationResult = classifier.assignClass(inputDoc);
100       assertNotNull(classificationResult.getAssignedClass());
101       assertEquals("got an assigned class of " + classificationResult.getAssignedClass(), expectedResult, classificationResult.getAssignedClass());
102       assertTrue("got a not positive score " + classificationResult.getScore(), classificationResult.getScore() > 0);
103     } finally {
104       if (leafReader != null)
105         leafReader.close();
106     }
107   }
108   protected void checkOnlineClassification(Classifier<T> classifier, String inputDoc, T expectedResult, Analyzer analyzer, String textFieldName, String classFieldName) throws Exception {
109     checkOnlineClassification(classifier, inputDoc, expectedResult, analyzer, textFieldName, classFieldName, null);
110   }

112   protected void checkOnlineClassification(Classifier<T> classifier, String inputDoc, T expectedResult, Analyzer analyzer, String textFieldName, String classFieldName, Query query) throws Exception {
113     LeafReader leafReader = null;
114     try {
115       leafReader = populateSampleIndex(analyzer);
116       classifier.train(leafReader, textFieldName, classFieldName, analyzer, query);
117       ClassificationResult<T> classificationResult = classifier.assignClass(inputDoc);
118       assertNotNull(classificationResult.getAssignedClass());
119       assertEquals("got an assigned class of " + classificationResult.getAssignedClass(), expectedResult, classificationResult.getAssignedClass());
120       assertTrue("got a not positive score " + classificationResult.getScore(), classificationResult.getScore() > 0);
121       updateSampleIndex();
122       ClassificationResult<T> secondClassificationResult = classifier.assignClass(inputDoc);
123       assertEquals(classificationResult.getAssignedClass(), secondClassificationResult.getAssignedClass());
124       assertEquals(Double.valueOf(classificationResult.getScore()), Double.valueOf(secondClassificationResult.getScore()));
125 
126     } finally {
127       if (leafReader != null)
128         leafReader.close();
129     }
130   }

210   protected void checkPerformance(Classifier<T> classifier, Analyzer analyzer, String classFieldName) throws Exception {
211     LeafReader leafReader = null;
212     long trainStart = System.currentTimeMillis();
213     try {
214       populatePerformanceIndex(analyzer);
215       leafReader = SlowCompositeReaderWrapper.wrap(indexWriter.getReader());
216       classifier.train(leafReader, textFieldName, classFieldName, analyzer);
217       long trainEnd = System.currentTimeMillis();
218       long trainTime = trainEnd - trainStart;
219       assertTrue("training took more than 2 mins : " + trainTime / 1000 + "s", trainTime < 120000);
220     } finally {
221       if (leafReader != null)
222         leafReader.close();
223     }
224   }

226   private void populatePerformanceIndex(Analyzer analyzer) throws IOException {
227     indexWriter.close();
228     indexWriter = new RandomIndexWriter(random(), dir, newIndexWriterConfig(analyzer).setOpenMode(IndexWriterConfig.OpenMode.CREATE));
229     indexWriter.commit();
230 
231     FieldType ft = new FieldType(TextField.TYPE_STORED);
232     ft.setStoreTermVectors(true);
233     ft.setStoreTermVectorOffsets(true);
234     ft.setStoreTermVectorPositions(true);
235     int docs = 1000;
236     Random random = random();
237     for (int i = 0; i < docs; i++) {
238       boolean b = random.nextBoolean();
239       Document doc = new Document();
240       doc.add(new Field(textFieldName, createRandomString(random), ft));
241       doc.add(new Field(categoryFieldName, b ? "technology" : "politics", ft));
242       doc.add(new Field(booleanFieldName, String.valueOf(b), ft));
243       indexWriter.addDocument(doc);
244     }
245     indexWriter.commit();
246   }

248   private String createRandomString(Random random) {
249     StringBuilder builder = new StringBuilder();
250     for (int i = 0; i < 20; i++) {
251       builder.append(TestUtil.randomSimpleString(random, 5));
252       builder.append(" ");
253     }
254     return builder.toString();
255   }
