48  /**
49   * The class responsible for handling Tika events and translating them into {@link org.apache.solr.common.SolrInputDocument}s.
50   * <B>This class is not thread-safe.</B>
51   * <p>
52   * This class cannot be reused, you have to create a new instance per document!
53   * <p>
54   * User's may wish to override this class to provide their own functionality.
55   *
56   * @see org.apache.solr.handler.extraction.SolrContentHandlerFactory
57   * @see org.apache.solr.handler.extraction.ExtractingRequestHandler
58   * @see org.apache.solr.handler.extraction.ExtractingDocumentLoader
59   */
60  public class SolrContentHandler extends DefaultHandler implements ExtractingParams {
61    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
62  
63    public static final String contentFieldName = "content";
64  
65    protected final SolrInputDocument document;
66  
67    protected final Collection<String> dateFormats;
68  
69    protected final Metadata metadata;
70    protected final SolrParams params;
71    protected final StringBuilder catchAllBuilder = new StringBuilder(2048);
72    protected final IndexSchema schema;
73    protected final Map<String, StringBuilder> fieldBuilders;
74    private final Deque<StringBuilder> bldrStack = new ArrayDeque<>();
75  
76    protected final boolean captureAttribs;
77    protected final boolean lowerNames;
78    
79    protected final String unknownFieldPrefix;
80    protected final String defaultField;
81  
82    private final boolean literalsOverride;
83    
84    private Set<String> literalFieldNames = null;
85    
86    public SolrContentHandler(Metadata metadata, SolrParams params, IndexSchema schema) {
87      this(metadata, params, schema, DateUtil.DEFAULT_DATE_FORMATS);
88    }

91    public SolrContentHandler(Metadata metadata, SolrParams params,
92                              IndexSchema schema, Collection<String> dateFormats) {
93      this.document = new SolrInputDocument();
94      this.metadata = metadata;
95      this.params = params;
96      this.schema = schema;
97      this.dateFormats = dateFormats;
98  
99      this.lowerNames = params.getBool(LOWERNAMES, false);
100     this.captureAttribs = params.getBool(CAPTURE_ATTRIBUTES, false);
101     this.literalsOverride = params.getBool(LITERALS_OVERRIDE, true);
102     this.unknownFieldPrefix = params.get(UNKNOWN_FIELD_PREFIX, "");
103     this.defaultField = params.get(DEFAULT_FIELD, "");
104     
105     String[] captureFields = params.getParams(CAPTURE_ELEMENTS);
106     if (captureFields != null && captureFields.length > 0) {
107       fieldBuilders = new HashMap<>();
108       for (int i = 0; i < captureFields.length; i++) {
109         fieldBuilders.put(captureFields[i], new StringBuilder());
110       }
111     } else {
112       fieldBuilders = Collections.emptyMap();
113     }
114     bldrStack.add(catchAllBuilder);
115   }

118   /**
119    * This is called by a consumer when it is ready to deal with a new SolrInputDocument.  Overriding
120    * classes can use this hook to add in or change whatever they deem fit for the document at that time.
121    * The base implementation adds the metadata as fields, allowing for potential remapping.
122    *
123    * @return The {@link org.apache.solr.common.SolrInputDocument}.
124    *
125    * @see #addMetadata()
126    * @see #addCapturedContent()
127    * @see #addContent()
128    * @see #addLiterals()
129    */
130   public SolrInputDocument newDocument() {
131     //handle the literals from the params. NOTE: This MUST be called before the others in order for literals to override other values
132     addLiterals();
133 
134     //handle the metadata extracted from the document
135     addMetadata();
136 
137     //add in the content
138     addContent();
139 
140     //add in the captured content
141     addCapturedContent();
142 
143     if (log.isDebugEnabled()) {
144       log.debug("Doc: {}", document);
145     }
146     return document;
147   }
148 
149   /**
150    * Add the per field captured content to the Solr Document.  Default implementation uses the
151    * {@link #fieldBuilders} info
152    */
153   protected void addCapturedContent() {
154     for (Map.Entry<String, StringBuilder> entry : fieldBuilders.entrySet()) {
155       if (entry.getValue().length() > 0) {
156         String fieldName = entry.getKey();
157         if (literalsOverride && literalFieldNames.contains(fieldName))
158           continue;
159         addField(fieldName, entry.getValue().toString(), null);      }
160     }
161   }
162 
163   /**
164    * Add in the catch all content to the field.  Default impl. uses the {@link #contentFieldName}
165    * and the {@link #catchAllBuilder}
166    */
167   protected void addContent() {
168     if (literalsOverride && literalFieldNames.contains(contentFieldName))
169       return;
170     addField(contentFieldName, catchAllBuilder.toString(), null);
171   }
172 
173   /**
174    * Add in the literals to the document using the {@link #params} and the {@link #LITERALS_PREFIX}.
175    */
176   protected void addLiterals() {
177     Iterator<String> paramNames = params.getParameterNamesIterator();
178     literalFieldNames = new HashSet<>();
179     while (paramNames.hasNext()) {
180       String pname = paramNames.next();
181       if (!pname.startsWith(LITERALS_PREFIX)) continue;
182 
183       String name = pname.substring(LITERALS_PREFIX.length());
184       addField(name, null, params.getParams(pname));
185       literalFieldNames.add(name);
186     }
187   }
188 
189   /**
190    * Add in any metadata using {@link #metadata} as the source.
191    */
192   protected void addMetadata() {
193     for (String name : metadata.names()) {
194       if (literalsOverride && literalFieldNames.contains(name))
195         continue;
196       String[] vals = metadata.getValues(name);
197       addField(name, null, vals);
198     }
199   }
200 
201   // Naming rules:
202   // 1) optionally map names to nicenames (lowercase+underscores)
203   // 2) execute "map" commands
204   // 3) if resulting field is unknown, map it to a common prefix
205   protected void addField(String fname, String fval, String[] vals) {
206     if (lowerNames) {
207       StringBuilder sb = new StringBuilder();
208       for (int i=0; i<fname.length(); i++) {
209         char ch = fname.charAt(i);
210         if (!Character.isLetterOrDigit(ch)) ch='_';
211         else ch=Character.toLowerCase(ch);
212         sb.append(ch);
213       }
214       fname = sb.toString();
215     }    
216 
217     String name = findMappedName(fname);
218     SchemaField sf = schema.getFieldOrNull(name);
219     if (sf==null && unknownFieldPrefix.length() > 0) {
220       name = unknownFieldPrefix + name;
221       sf = schema.getFieldOrNull(name);
222     } else if (sf == null && defaultField.length() > 0 && name.equals(TikaMetadataKeys.RESOURCE_NAME_KEY) == false /*let the fall through below handle this*/){
223       name = defaultField;
224       sf = schema.getFieldOrNull(name);
225     }
226 
227     // Arguably we should handle this as a special case. Why? Because unlike basically
228     // all the other fields in metadata, this one was probably set not by Tika by in
229     // ExtractingDocumentLoader.load(). You shouldn't have to define a mapping for this
230     // field just because you specified a resource.name parameter to the handler, should
231     // you?
232     if (sf == null && unknownFieldPrefix.length()==0 && name == TikaMetadataKeys.RESOURCE_NAME_KEY) {
233       return;
234     }
235 
236     // normalize val params so vals.length>1
237     if (vals != null && vals.length==1) {
238       fval = vals[0];
239       vals = null;
240     }
241 
242     // single valued field with multiple values... catenate them.
243     if (sf != null && !sf.multiValued() && vals != null) {
244       StringBuilder builder = new StringBuilder();
245       boolean first=true;
246       for (String val : vals) {
247         if (first) {
248           first=false;
249         } else {
250           builder.append(' ');
251         }
252         builder.append(val);
253       }
254       fval = builder.toString();
255       vals=null;
256     }
257 
258     float boost = getBoost(name);
259 
260     if (fval != null) {
261       document.addField(name, transformValue(fval, sf), boost);
262     }
263 
264     if (vals != null) {
265       for (String val : vals) {
266         document.addField(name, transformValue(val, sf), boost);
267       }
268     }
269 
270     // no value set - throw exception for debugging
271     // if (vals==null && fval==null) throw new RuntimeException(name + " has no non-null value ");
272   }

274   @Override
275   public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
276     StringBuilder theBldr = fieldBuilders.get(localName);
277     if (theBldr != null) {
278       //we need to switch the currentBuilder
279       bldrStack.add(theBldr);
280     }
281     if (captureAttribs == true) {
282       for (int i = 0; i < attributes.getLength(); i++) {
283         addField(localName, attributes.getValue(i), null);
284       }
285     } else {
286       for (int i = 0; i < attributes.getLength(); i++) {
287         bldrStack.getLast().append(' ').append(attributes.getValue(i));
288       }
289     }
290     bldrStack.getLast().append(' ');
291   }
292 
293   @Override
294   public void endElement(String uri, String localName, String qName) throws SAXException {
295     StringBuilder theBldr = fieldBuilders.get(localName);
296     if (theBldr != null) {
297       //pop the stack
298       bldrStack.removeLast();
299       assert (bldrStack.size() >= 1);
300     }
301     bldrStack.getLast().append(' ');
302   }
303 
304 
305   @Override
306   public void characters(char[] chars, int offset, int length) throws SAXException {
307     bldrStack.getLast().append(chars, offset, length);
308   }
309 
310   /**
311    * Treat the same as any other characters
312    */
313   @Override
314   public void ignorableWhitespace(char[] chars, int offset, int length) throws SAXException {
315     characters(chars, offset, length);
316   }

318   /**
319    * Can be used to transform input values based on their {@link org.apache.solr.schema.SchemaField}
320    * <p>
321    * This implementation only formats dates using the {@link org.apache.solr.common.util.DateUtil}.
322    *
323    * @param val    The value to transform
324    * @param schFld The {@link org.apache.solr.schema.SchemaField}
325    * @return The potentially new value.
326    */
327   protected String transformValue(String val, SchemaField schFld) {
328     String result = val;
329     if (schFld != null && schFld.getType() instanceof TrieDateField) {
330       //try to transform the date
331       try {
332         Date date = DateUtil.parseDate(val, dateFormats);
333         DateFormat df = DateUtil.getThreadLocalDateFormat();
334         result = df.format(date);
335 
336       } catch (Exception e) {
337         // Let the specific fieldType handle errors
338         // throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Invalid value: " + val + " for field: " + schFld, e);
339       }
340     }
341     return result;
342   }

345   /**
346    * Get the value of any boost factor for the mapped name.
347    *
348    * @param name The name of the field to see if there is a boost specified
349    * @return The boost value
350    */
351   protected float getBoost(String name) {
352     return params.getFloat(BOOST_PREFIX + name, 1.0f);
353   }
354 
355   /**
356    * Get the name mapping
357    *
358    * @param name The name to check to see if there is a mapping
359    * @return The new name, if there is one, else <code>name</code>
360    */
361   protected String findMappedName(String name) {
362     return params.get(MAP_PREFIX + name, name);
363   }
