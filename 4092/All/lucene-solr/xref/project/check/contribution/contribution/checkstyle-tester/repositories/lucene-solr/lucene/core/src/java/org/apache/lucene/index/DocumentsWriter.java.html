40  /**
41   * This class accepts multiple added documents and directly
42   * writes segment files.
43   *
44   * Each added document is passed to the indexing chain,
45   * which in turn processes the document into the different
46   * codec formats.  Some formats write bytes to files
47   * immediately, e.g. stored fields and term vectors, while
48   * others are buffered by the indexing chain and written
49   * only on flush.
50   *
51   * Once we have used our allowed RAM buffer, or the number
52   * of added docs is large enough (in the case we are
53   * flushing by doc count instead of RAM usage), we create a
54   * real segment and flush it to the Directory.
55   *
56   * Threads:
57   *
58   * Multiple threads are allowed into addDocument at once.
59   * There is an initial synchronized call to getThreadState
60   * which allocates a ThreadState for this thread.  The same
61   * thread will get the same ThreadState over time (thread
62   * affinity) so that if there are consistent patterns (for
63   * example each thread is indexing a different content
64   * source) then we make better use of RAM.  Then
65   * processDocument is called on that ThreadState without
66   * synchronization (most of the "heavy lifting" is in this
67   * call).  Finally the synchronized "finishDocument" is
68   * called to flush changes to the directory.
69   *
70   * When flush is called by IndexWriter we forcefully idle
71   * all threads and flush only once they are all idle.  This
72   * means you can call flush with a given thread even while
73   * other threads are actively adding/deleting documents.
74   *
75   *
76   * Exceptions:
77   *
78   * Because this class directly updates in-memory posting
79   * lists, and flushes stored fields and term vectors
80   * directly to files in the directory, there are certain
81   * limited times when an exception can corrupt this state.
82   * For example, a disk full while flushing stored fields
83   * leaves this file in a corrupt state.  Or, an OOM
84   * exception while appending to the in-memory posting lists
85   * can corrupt that posting list.  We call such exceptions
86   * "aborting exceptions".  In these cases we must call
87   * abort() to discard all docs added since the last flush.
88   *
89   * All other exceptions ("non-aborting exceptions") can
90   * still partially update the index structures.  These
91   * updates are consistent, but, they represent only a part
92   * of the document seen up until the exception was hit.
93   * When this happens, we immediately mark the document as
94   * deleted so that the document is always atomically ("all
95   * or none") added to the index.
96   */

170   private boolean applyAllDeletes(DocumentsWriterDeleteQueue deleteQueue) throws IOException {
171     if (flushControl.getAndResetApplyAllDeletes()) {
172       if (deleteQueue != null && !flushControl.isFullFlush()) {
173         ticketQueue.addDeletes(deleteQueue);
174       }
175       putEvent(ApplyDeletesEvent.INSTANCE); // apply deletes event forces a purge
176       return true;
177     }
178     return false;
179   }

182     if (forced) {
183       return ticketQueue.forcePurge(writer);
184     } else {
185       return ticketQueue.tryPurge(writer);
186     }
187   }

201   /** Called if we hit an exception at a bad time (when
202    *  updating the index files) and must discard all
203    *  currently buffered docs.  This resets our state,
204    *  discarding any docs added since last flush. */
205   synchronized void abort(IndexWriter writer) {
206     assert !Thread.holdsLock(writer) : "IndexWriter lock should never be hold when aborting";
207     boolean success = false;
208     try {
209       deleteQueue.clear();
210       if (infoStream.isEnabled("DW")) {
211         infoStream.message("DW", "abort");
212       }
213       final int limit = perThreadPool.getActiveThreadStateCount();
214       for (int i = 0; i < limit; i++) {
215         final ThreadState perThread = perThreadPool.getThreadState(i);
216         perThread.lock();
217         try {
218           abortThreadState(perThread);
219         } finally {
220           perThread.unlock();
221         }
222       }
223       flushControl.abortPendingFlushes();
224       flushControl.waitForFlush();
225       success = true;
226     } finally {
227       if (infoStream.isEnabled("DW")) {
228         infoStream.message("DW", "done abort success=" + success);
229       }
230     }
231   }

233   /** Returns how many documents were aborted. */
234   synchronized long lockAndAbortAll(IndexWriter indexWriter) {
235     assert indexWriter.holdsFullFlushLock();
236     if (infoStream.isEnabled("DW")) {
237       infoStream.message("DW", "lockAndAbortAll");
238     }
239     long abortedDocCount = 0;
240     boolean success = false;
241     try {
242       deleteQueue.clear();
243       final int limit = perThreadPool.getMaxThreadStates();
244       perThreadPool.setAbort();
245       for (int i = 0; i < limit; i++) {
246         final ThreadState perThread = perThreadPool.getThreadState(i);
247         perThread.lock();
248         abortedDocCount += abortThreadState(perThread);
249       }
250       deleteQueue.clear();
251       flushControl.abortPendingFlushes();
252       flushControl.waitForFlush();
253       success = true;
254       return abortedDocCount;
255     } finally {
256       if (infoStream.isEnabled("DW")) {
257         infoStream.message("DW", "finished lockAndAbortAll success=" + success);
258       }
259       if (success == false) {
260         // if something happens here we unlock all states again
261         unlockAllAfterAbortAll(indexWriter);
262       }
263     }
264   }

266   /** Returns how many documents were aborted. */
267   private int abortThreadState(final ThreadState perThread) {
268     assert perThread.isHeldByCurrentThread();
269     if (perThread.isInitialized()) { 
270       try {
271         int abortedDocCount = perThread.dwpt.getNumDocsInRAM();
272         subtractFlushedNumDocs(abortedDocCount);
273         perThread.dwpt.abort();
274         return abortedDocCount;
275       } finally {
276         flushControl.doOnAbort(perThread);
277       }
278     } else {
279       flushControl.doOnAbort(perThread);
280       // This DWPT was never initialized so it has no indexed documents:
281       return 0;
282     }
283   }
284   
285   synchronized void unlockAllAfterAbortAll(IndexWriter indexWriter) {
286     assert indexWriter.holdsFullFlushLock();
287     if (infoStream.isEnabled("DW")) {
288       infoStream.message("DW", "unlockAll");
289     }
290     final int limit = perThreadPool.getMaxThreadStates();
291     perThreadPool.clearAbort();
292     for (int i = 0; i < limit; i++) {
293       try {
294         final ThreadState perThread = perThreadPool.getThreadState(i);
295         if (perThread.isHeldByCurrentThread()) {
296           perThread.unlock();
297         }
298       } catch (Throwable e) {
299         if (infoStream.isEnabled("DW")) {
300           infoStream.message("DW", "unlockAll: could not unlock state: " + i + " msg:" + e.getMessage());
301         }
302         // ignore & keep on unlocking
303       }
304     }
305   }

307   boolean anyChanges() {
308     /*
309      * changes are either in a DWPT or in the deleteQueue.
310      * yet if we currently flush deletes and / or dwpt there
311      * could be a window where all changes are in the ticket queue
312      * before they are published to the IW. ie we need to check if the 
313      * ticket queue has any tickets.
314      */
315     boolean anyChanges = numDocsInRAM.get() != 0 || anyDeletions() || ticketQueue.hasTickets() || pendingChangesInCurrentFullFlush;
316     if (infoStream.isEnabled("DW") && anyChanges) {
317       infoStream.message("DW", "anyChanges? numDocsInRam=" + numDocsInRAM.get()
318                          + " deletes=" + anyDeletions() + " hasTickets:"
319                          + ticketQueue.hasTickets() + " pendingChangesInFullFlush: "
320                          + pendingChangesInCurrentFullFlush);
321     }
322     return anyChanges;
323   }

344   private boolean preUpdate() throws IOException, AbortingException {
345     ensureOpen();
346     boolean hasEvents = false;
347     if (flushControl.anyStalledThreads() || flushControl.numQueuedFlushes() > 0) {
348       // Help out flushing any queued DWPTs so we can un-stall:
349       if (infoStream.isEnabled("DW")) {
350         infoStream.message("DW", "DocumentsWriter has queued dwpt; will hijack this thread to flush pending segment(s)");
351       }
352       do {
353         // Try pick up pending threads here if possible
354         DocumentsWriterPerThread flushingDWPT;
355         while ((flushingDWPT = flushControl.nextPendingFlush()) != null) {
356           // Don't push the delete here since the update could fail!
357           hasEvents |= doFlush(flushingDWPT);
358         }
359   
360         if (infoStream.isEnabled("DW") && flushControl.anyStalledThreads()) {
361           infoStream.message("DW", "WARNING DocumentsWriter has stalled threads; waiting");
362         }
363         
364         flushControl.waitIfStalled(); // block if stalled
365       } while (flushControl.numQueuedFlushes() != 0); // still queued DWPTs try help flushing
366 
367       if (infoStream.isEnabled("DW")) {
368         infoStream.message("DW", "continue indexing after helping out flushing DocumentsWriter is healthy");
369       }
370     }
371     return hasEvents;
372   }

374   private boolean postUpdate(DocumentsWriterPerThread flushingDWPT, boolean hasEvents) throws IOException, AbortingException {
375     hasEvents |= applyAllDeletes(deleteQueue);
376     if (flushingDWPT != null) {
377       hasEvents |= doFlush(flushingDWPT);
378     } else {
379       final DocumentsWriterPerThread nextPendingFlush = flushControl.nextPendingFlush();
380       if (nextPendingFlush != null) {
381         hasEvents |= doFlush(nextPendingFlush);
382       }
383     }
384 
385     return hasEvents;
386   }

389     if (state.dwpt == null) {
390       final FieldInfos.Builder infos = new FieldInfos.Builder(writer.globalFieldNumberMap);
391       state.dwpt = new DocumentsWriterPerThread(writer, writer.newSegmentName(), directoryOrig,

397   boolean updateDocuments(final Iterable<? extends Iterable<? extends IndexableField>> docs, final Analyzer analyzer,
398                           final Term delTerm) throws IOException, AbortingException {
399     boolean hasEvents = preUpdate();
400 
401     final ThreadState perThread = flushControl.obtainAndLock();
402     final DocumentsWriterPerThread flushingDWPT;
403     
404     try {
405       // This must happen after we've pulled the ThreadState because IW.close
406       // waits for all ThreadStates to be released:
407       ensureOpen();
408       ensureInitialized(perThread);
409       assert perThread.isInitialized();
410       final DocumentsWriterPerThread dwpt = perThread.dwpt;
411       final int dwptNumDocs = dwpt.getNumDocsInRAM();
412       try {
413         dwpt.updateDocuments(docs, analyzer, delTerm);
414       } catch (AbortingException ae) {
415         flushControl.doOnAbort(perThread);
416         dwpt.abort();
417         throw ae;
418       } finally {
419         // We don't know how many documents were actually
420         // counted as indexed, so we must subtract here to
421         // accumulate our separate counter:
422         numDocsInRAM.addAndGet(dwpt.getNumDocsInRAM() - dwptNumDocs);
423       }
424       final boolean isUpdate = delTerm != null;
425       flushingDWPT = flushControl.doAfterDocument(perThread, isUpdate);
426     } finally {
427       perThreadPool.release(perThread);
428     }
429 
430     return postUpdate(flushingDWPT, hasEvents);
431   }

433   boolean updateDocument(final Iterable<? extends IndexableField> doc, final Analyzer analyzer,
434       final Term delTerm) throws IOException, AbortingException {
435 
436     boolean hasEvents = preUpdate();
437 
438     final ThreadState perThread = flushControl.obtainAndLock();
439 
440     final DocumentsWriterPerThread flushingDWPT;
441     try {
442       // This must happen after we've pulled the ThreadState because IW.close
443       // waits for all ThreadStates to be released:
444       ensureOpen();
445       ensureInitialized(perThread);
446       assert perThread.isInitialized();
447       final DocumentsWriterPerThread dwpt = perThread.dwpt;
448       final int dwptNumDocs = dwpt.getNumDocsInRAM();
449       try {
450         dwpt.updateDocument(doc, analyzer, delTerm); 
451       } catch (AbortingException ae) {
452         flushControl.doOnAbort(perThread);
453         dwpt.abort();
454         throw ae;
455       } finally {
456         // We don't know whether the document actually
457         // counted as being indexed, so we must subtract here to
458         // accumulate our separate counter:
459         numDocsInRAM.addAndGet(dwpt.getNumDocsInRAM() - dwptNumDocs);
460       }
461       final boolean isUpdate = delTerm != null;
462       flushingDWPT = flushControl.doAfterDocument(perThread, isUpdate);
463     } finally {
464       perThreadPool.release(perThread);
465     }
466 
467     return postUpdate(flushingDWPT, hasEvents);
468   }

470   private boolean doFlush(DocumentsWriterPerThread flushingDWPT) throws IOException, AbortingException {
471     boolean hasEvents = false;
472     while (flushingDWPT != null) {
473       hasEvents = true;
474       boolean success = false;
475       SegmentFlushTicket ticket = null;
476       try {
477         assert currentFullFlushDelQueue == null
478             || flushingDWPT.deleteQueue == currentFullFlushDelQueue : "expected: "
479             + currentFullFlushDelQueue + "but was: " + flushingDWPT.deleteQueue
480             + " " + flushControl.isFullFlush();
481         /*
482          * Since with DWPT the flush process is concurrent and several DWPT
483          * could flush at the same time we must maintain the order of the
484          * flushes before we can apply the flushed segment and the frozen global
485          * deletes it is buffering. The reason for this is that the global
486          * deletes mark a certain point in time where we took a DWPT out of
487          * rotation and freeze the global deletes.
488          * 
489          * Example: A flush 'A' starts and freezes the global deletes, then
490          * flush 'B' starts and freezes all deletes occurred since 'A' has
491          * started. if 'B' finishes before 'A' we need to wait until 'A' is done
492          * otherwise the deletes frozen by 'B' are not applied to 'A' and we
493          * might miss to deletes documents in 'A'.
494          */
495         try {
496           // Each flush is assigned a ticket in the order they acquire the ticketQueue lock
497           ticket = ticketQueue.addFlushTicket(flushingDWPT);
498   
499           final int flushingDocsInRam = flushingDWPT.getNumDocsInRAM();
500           boolean dwptSuccess = false;
501           try {
502             // flush concurrently without locking
503             final FlushedSegment newSegment = flushingDWPT.flush();
504             ticketQueue.addSegment(ticket, newSegment);
505             dwptSuccess = true;
506           } finally {
507             subtractFlushedNumDocs(flushingDocsInRam);
508             if (!flushingDWPT.pendingFilesToDelete().isEmpty()) {
509               putEvent(new DeleteNewFilesEvent(flushingDWPT.pendingFilesToDelete()));
510               hasEvents = true;
511             }
512             if (!dwptSuccess) {
513               putEvent(new FlushFailedEvent(flushingDWPT.getSegmentInfo()));
514               hasEvents = true;
515             }
516           }
517           // flush was successful once we reached this point - new seg. has been assigned to the ticket!
518           success = true;
519         } finally {
520           if (!success && ticket != null) {
521             // In the case of a failure make sure we are making progress and
522             // apply all the deletes since the segment flush failed since the flush
523             // ticket could hold global deletes see FlushTicket#canPublish()
524             ticketQueue.markTicketFailed(ticket);
525           }
526         }
527         /*
528          * Now we are done and try to flush the ticket queue if the head of the
529          * queue has already finished the flush.
530          */
531         if (ticketQueue.getTicketCount() >= perThreadPool.getActiveThreadStateCount()) {
532           // This means there is a backlog: the one
533           // thread in innerPurge can't keep up with all
534           // other threads flushing segments.  In this case
535           // we forcefully stall the producers.
536           putEvent(ForcedPurgeEvent.INSTANCE);
537           break;
538         }
539       } finally {
540         flushControl.doAfterFlush(flushingDWPT);
541       }
542      
543       flushingDWPT = flushControl.nextPendingFlush();
544     }
545     if (hasEvents) {
546       putEvent(MergePendingEvent.INSTANCE);
547     }
548     // If deletes alone are consuming > 1/2 our RAM
549     // buffer, force them all to apply now. This is to
550     // prevent too-frequent flushing of a long tail of
551     // tiny segments:
552     final double ramBufferSizeMB = config.getRAMBufferSizeMB();
553     if (ramBufferSizeMB != IndexWriterConfig.DISABLE_AUTO_FLUSH &&
554         flushControl.getDeleteBytesUsed() > (1024*1024*ramBufferSizeMB/2)) {
555       hasEvents = true;
556       if (!this.applyAllDeletes(deleteQueue)) {
557         if (infoStream.isEnabled("DW")) {
558           infoStream.message("DW", String.format(Locale.ROOT, "force apply deletes bytesUsed=%.1f MB vs ramBuffer=%.1f MB",
559                                                  flushControl.getDeleteBytesUsed()/(1024.*1024.),
560                                                  ramBufferSizeMB));
561         }
562         putEvent(ApplyDeletesEvent.INSTANCE);
563       }
564     }
565 
566     return hasEvents;
567   }

569   void subtractFlushedNumDocs(int numFlushed) {
570     int oldValue = numDocsInRAM.get();
571     while (!numDocsInRAM.compareAndSet(oldValue, oldValue - numFlushed)) {
572       oldValue = numDocsInRAM.get();
573     }
574     assert numDocsInRAM.get() >= 0;
575   }

581   private synchronized boolean setFlushingDeleteQueue(DocumentsWriterDeleteQueue session) {

583     return true;

586   /*
587    * FlushAllThreads is synced by IW fullFlushLock. Flushing all threads is a
588    * two stage operation; the caller must ensure (in try/finally) that finishFlush
589    * is called after this method, to release the flush lock in DWFlushControl
590    */
591   boolean flushAllThreads()
592     throws IOException, AbortingException {
593     final DocumentsWriterDeleteQueue flushingDeleteQueue;
594     if (infoStream.isEnabled("DW")) {
595       infoStream.message("DW", "startFullFlush");
596     }
597     
598     synchronized (this) {
599       pendingChangesInCurrentFullFlush = anyChanges();
600       flushingDeleteQueue = deleteQueue;
601       /* Cutover to a new delete queue.  This must be synced on the flush control
602        * otherwise a new DWPT could sneak into the loop with an already flushing
603        * delete queue */
604       flushControl.markForFullFlush(); // swaps the delQueue synced on FlushControl
605       assert setFlushingDeleteQueue(flushingDeleteQueue);
606     }
607     assert currentFullFlushDelQueue != null;
608     assert currentFullFlushDelQueue != deleteQueue;
609     
610     boolean anythingFlushed = false;
611     try {
612       DocumentsWriterPerThread flushingDWPT;
613       // Help out with flushing:
614       while ((flushingDWPT = flushControl.nextPendingFlush()) != null) {
615         anythingFlushed |= doFlush(flushingDWPT);
616       }
617       // If a concurrent flush is still in flight wait for it
618       flushControl.waitForFlush();  
619       if (!anythingFlushed && flushingDeleteQueue.anyChanges()) { // apply deletes if we did not flush any document
620         if (infoStream.isEnabled("DW")) {
621           infoStream.message("DW", Thread.currentThread().getName() + ": flush naked frozen global deletes");
622         }
623         ticketQueue.addDeletes(flushingDeleteQueue);
624       } 
625       ticketQueue.forcePurge(writer);
626       assert !flushingDeleteQueue.anyChanges() && !ticketQueue.hasTickets();
627     } finally {
628       assert flushingDeleteQueue == currentFullFlushDelQueue;
629     }
630     return anythingFlushed;
631   }

633   void finishFullFlush(IndexWriter indexWriter, boolean success) {
634     assert indexWriter.holdsFullFlushLock();
635     try {
636       if (infoStream.isEnabled("DW")) {
637         infoStream.message("DW", Thread.currentThread().getName() + " finishFullFlush success=" + success);
638       }
639       assert setFlushingDeleteQueue(null);
640       if (success) {
641         // Release the flush lock
642         flushControl.finishFullFlush();
643       } else {
644         flushControl.abortFullFlushes();
645 
646       }
647     } finally {
648       pendingChangesInCurrentFullFlush = false;
649     }
