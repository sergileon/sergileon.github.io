68    @Override
69    public QParser createParser(String qstr, SolrParams localParams, SolrParams params, SolrQueryRequest req) {
70      return new QParser(qstr, localParams, params, req) {
71        
72        @Override
73        public Query parse() throws SyntaxError {
74          if(localParams!=null && localParams.get(ScoreJoinQParserPlugin.SCORE)!=null){
75            return new ScoreJoinQParserPlugin().createParser(qstr, localParams, params, req).parse();
76          }else{
77            return parseJoin();
78          }
79        }
80        
81        Query parseJoin() throws SyntaxError {
82          final String fromField = getParam("from");
83          final String fromIndex = getParam("fromIndex");
84          final String toField = getParam("to");
85          final String v = localParams.get("v");
86          final String coreName;
87  
88          Query fromQuery;
89          long fromCoreOpenTime = 0;
90  
91          if (fromIndex != null && !fromIndex.equals(req.getCore().getCoreDescriptor().getName()) ) {
92            CoreContainer container = req.getCore().getCoreDescriptor().getCoreContainer();
93  
94            // if in SolrCloud mode, fromIndex should be the name of a single-sharded collection
95            coreName = ScoreJoinQParserPlugin.getCoreName(fromIndex, container);
96  
97            final SolrCore fromCore = container.getCore(coreName);
98            if (fromCore == null) {
99              throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
100                 "Cross-core join: no such core " + coreName);
101           }
102 
103           RefCounted<SolrIndexSearcher> fromHolder = null;
104           LocalSolrQueryRequest otherReq = new LocalSolrQueryRequest(fromCore, params);
105           try {
106             QParser parser = QParser.getParser(v, "lucene", otherReq);
107             fromQuery = parser.getQuery();
108             fromHolder = fromCore.getRegisteredSearcher();
109             if (fromHolder != null) fromCoreOpenTime = fromHolder.get().getOpenNanoTime();
110           } finally {
111             otherReq.close();
112             fromCore.close();
113             if (fromHolder != null) fromHolder.decref();
114           }
115         } else {
116           coreName = null;
117           QParser fromQueryParser = subQuery(v, null);
118           fromQuery = fromQueryParser.getQuery();
119         }
120 
121         JoinQuery jq = new JoinQuery(fromField, toField, coreName == null ? fromIndex : coreName, fromQuery);
122         jq.fromCoreOpenTime = fromCoreOpenTime;
123         return jq;
124       }
125     };
126   }

158   private class JoinQueryWeight extends ConstantScoreWeight {
159     SolrIndexSearcher fromSearcher;
160     RefCounted<SolrIndexSearcher> fromRef;
161     SolrIndexSearcher toSearcher;
162     private Similarity similarity;
163     ResponseBuilder rb;
164 
165     public JoinQueryWeight(SolrIndexSearcher searcher) {
166       super(JoinQuery.this);
167       this.fromSearcher = searcher;
168       SolrRequestInfo info = SolrRequestInfo.getRequestInfo();
169       if (info != null) {
170         rb = info.getResponseBuilder();
171       }
172 
173       if (fromIndex == null) {
174         this.fromSearcher = searcher;
175       } else {
176         if (info == null) {
177           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Cross-core join must have SolrRequestInfo");
178         }
179 
180         CoreContainer container = searcher.getCore().getCoreDescriptor().getCoreContainer();
181         final SolrCore fromCore = container.getCore(fromIndex);
182 
183         if (fromCore == null) {
184           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Cross-core join: no such core " + fromIndex);
185         }
186 
187         if (info.getReq().getCore() == fromCore) {
188           // if this is the same core, use the searcher passed in... otherwise we could be warming and
189           // get an older searcher from the core.
190           fromSearcher = searcher;
191         } else {
192           // This could block if there is a static warming query with a join in it, and if useColdSearcher is true.
193           // Deadlock could result if two cores both had useColdSearcher and had joins that used eachother.
194           // This would be very predictable though (should happen every time if misconfigured)
195           fromRef = fromCore.getSearcher(false, true, null);
196 
197           // be careful not to do anything with this searcher that requires the thread local
198           // SolrRequestInfo in a manner that requires the core in the request to match
199           fromSearcher = fromRef.get();
200         }
201 
202         if (fromRef != null) {
203           final RefCounted<SolrIndexSearcher> ref = fromRef;
204           info.addCloseHook(new Closeable() {
205             @Override
206             public void close() {
207               ref.decref();
208             }
209           });
210         }
211 
212         info.addCloseHook(new Closeable() {
213           @Override
214           public void close() {
215             fromCore.close();
216           }
217         });
218 
219       }
220       this.toSearcher = searcher;
221     }

223     DocSet resultSet;
224     Filter filter;

227     @Override
228     public Scorer scorer(LeafReaderContext context) throws IOException {
229       if (filter == null) {
230         boolean debug = rb != null && rb.isDebug();
231         RTimer timer = (debug ? new RTimer() : null);
232         resultSet = getDocSet();
233         if (timer != null) timer.stop();
234 
235         if (debug) {
236           SimpleOrderedMap<Object> dbg = new SimpleOrderedMap<>();
237           dbg.add("time", (long) timer.getTime());
238           dbg.add("fromSetSize", fromSetSize);  // the input
239           dbg.add("toSetSize", resultSet.size());    // the output
240 
241           dbg.add("fromTermCount", fromTermCount);
242           dbg.add("fromTermTotalDf", fromTermTotalDf);
243           dbg.add("fromTermDirectCount", fromTermDirectCount);
244           dbg.add("fromTermHits", fromTermHits);
245           dbg.add("fromTermHitsTotalDf", fromTermHitsTotalDf);
246           dbg.add("toTermHits", toTermHits);
247           dbg.add("toTermHitsTotalDf", toTermHitsTotalDf);
248           dbg.add("toTermDirectCount", toTermDirectCount);
249           dbg.add("smallSetsDeferred", smallSetsDeferred);
250           dbg.add("toSetDocsAdded", resultListDocs);
251 
252           // TODO: perhaps synchronize  addDebug in the future...
253           rb.addDebug(dbg, "join", JoinQuery.this.toString());
254         }
255 
256         filter = resultSet.getTopFilter();
257       }
258 
259       // Although this set only includes live docs, other filters can be pushed down to queries.
260       DocIdSet readerSet = filter.getDocIdSet(context, null);
261       if (readerSet == null) {
262         return null;
263       }
264       DocIdSetIterator readerSetIterator = readerSet.iterator();
265       if (readerSetIterator == null) {
266         return null;
267       }
268       return new ConstantScoreScorer(this, score(), readerSetIterator);
269     }

272     int fromSetSize;          // number of docs in the fromSet (that match the from query)
273     long resultListDocs;      // total number of docs collected
274     int fromTermCount;
275     long fromTermTotalDf;
276     int fromTermDirectCount;  // number of fromTerms that were too small to use the filter cache
277     int fromTermHits;         // number of fromTerms that intersected the from query
278     long fromTermHitsTotalDf; // sum of the df of the matching terms
279     int toTermHits;           // num if intersecting from terms that match a term in the to field
280     long toTermHitsTotalDf;   // sum of the df for the toTermHits
281     int toTermDirectCount;    // number of toTerms that we set directly on a bitset rather than doing set intersections
282     int smallSetsDeferred;    // number of small sets collected to be used later to intersect w/ bitset or create another small set

285     public DocSet getDocSet() throws IOException {
286       FixedBitSet resultBits = null;
287 
288       // minimum docFreq to use the cache
289       int minDocFreqFrom = Math.max(5, fromSearcher.maxDoc() >> 13);
290       int minDocFreqTo = Math.max(5, toSearcher.maxDoc() >> 13);
291 
292       // use a smaller size than normal since we will need to sort and dedup the results
293       int maxSortedIntSize = Math.max(10, toSearcher.maxDoc() >> 10);
294 
295       DocSet fromSet = fromSearcher.getDocSet(q);
296       fromSetSize = fromSet.size();
297 
298       List<DocSet> resultList = new ArrayList<>(10);
299 
300       // make sure we have a set that is fast for random access, if we will use it for that
301       DocSet fastForRandomSet = fromSet;
302       if (minDocFreqFrom>0 && fromSet instanceof SortedIntDocSet) {
303         SortedIntDocSet sset = (SortedIntDocSet)fromSet;
304         fastForRandomSet = new HashDocSet(sset.getDocs(), 0, sset.size());
305       }
306 
307       Fields fromFields = fromSearcher.getLeafReader().fields();
308       Fields toFields = fromSearcher==toSearcher ? fromFields : toSearcher.getLeafReader().fields();
309       if (fromFields == null) return DocSet.EMPTY;
310       Terms terms = fromFields.terms(fromField);
311       Terms toTerms = toFields.terms(toField);
312       if (terms == null || toTerms==null) return DocSet.EMPTY;
313       String prefixStr = TrieField.getMainValuePrefix(fromSearcher.getSchema().getFieldType(fromField));
314       BytesRef prefix = prefixStr == null ? null : new BytesRef(prefixStr);
315 
316       BytesRef term = null;
317       TermsEnum  termsEnum = terms.iterator();
318       TermsEnum  toTermsEnum = toTerms.iterator();
319       SolrIndexSearcher.DocsEnumState fromDeState = null;
320       SolrIndexSearcher.DocsEnumState toDeState = null;
321 
322       if (prefix == null) {
323         term = termsEnum.next();
324       } else {
325         if (termsEnum.seekCeil(prefix) != TermsEnum.SeekStatus.END) {
326           term = termsEnum.term();
327         }
328       }
329 
330       Bits fromLiveDocs = fromSearcher.getLeafReader().getLiveDocs();
331       Bits toLiveDocs = fromSearcher == toSearcher ? fromLiveDocs : toSearcher.getLeafReader().getLiveDocs();
332 
333       fromDeState = new SolrIndexSearcher.DocsEnumState();
334       fromDeState.fieldName = fromField;
335       fromDeState.liveDocs = fromLiveDocs;
336       fromDeState.termsEnum = termsEnum;
337       fromDeState.postingsEnum = null;
338       fromDeState.minSetSizeCached = minDocFreqFrom;
339 
340       toDeState = new SolrIndexSearcher.DocsEnumState();
341       toDeState.fieldName = toField;
342       toDeState.liveDocs = toLiveDocs;
343       toDeState.termsEnum = toTermsEnum;
344       toDeState.postingsEnum = null;
345       toDeState.minSetSizeCached = minDocFreqTo;
346 
347       while (term != null) {
348         if (prefix != null && !StringHelper.startsWith(term, prefix))
349           break;
350 
351         fromTermCount++;
352 
353         boolean intersects = false;
354         int freq = termsEnum.docFreq();
355         fromTermTotalDf++;
356 
357         if (freq < minDocFreqFrom) {
358           fromTermDirectCount++;
359           // OK to skip liveDocs, since we check for intersection with docs matching query
360           fromDeState.postingsEnum = fromDeState.termsEnum.postings(fromDeState.postingsEnum, PostingsEnum.NONE);
361           PostingsEnum postingsEnum = fromDeState.postingsEnum;
362 
363           if (postingsEnum instanceof MultiPostingsEnum) {
364             MultiPostingsEnum.EnumWithSlice[] subs = ((MultiPostingsEnum) postingsEnum).getSubs();
365             int numSubs = ((MultiPostingsEnum) postingsEnum).getNumSubs();
366             outer: for (int subindex = 0; subindex<numSubs; subindex++) {
367               MultiPostingsEnum.EnumWithSlice sub = subs[subindex];
368               if (sub.postingsEnum == null) continue;
369               int base = sub.slice.start;
370               int docid;
371               while ((docid = sub.postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
372                 if (fastForRandomSet.exists(docid+base)) {
373                   intersects = true;
374                   break outer;
375                 }
376               }
377             }
378           } else {
379             int docid;
380             while ((docid = postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
381               if (fastForRandomSet.exists(docid)) {
382                 intersects = true;
383                 break;
384               }
385             }
386           }
387         } else {
388           // use the filter cache
389           DocSet fromTermSet = fromSearcher.getDocSet(fromDeState);
390           intersects = fromSet.intersects(fromTermSet);
391         }
392 
393         if (intersects) {
394           fromTermHits++;
395           fromTermHitsTotalDf++;
396           TermsEnum.SeekStatus status = toTermsEnum.seekCeil(term);
397           if (status == TermsEnum.SeekStatus.END) break;
398           if (status == TermsEnum.SeekStatus.FOUND) {
399             toTermHits++;
400             int df = toTermsEnum.docFreq();
401             toTermHitsTotalDf += df;
402             if (resultBits==null && df + resultListDocs > maxSortedIntSize && resultList.size() > 0) {
403               resultBits = new FixedBitSet(toSearcher.maxDoc());
404             }
405 
406             // if we don't have a bitset yet, or if the resulting set will be too large
407             // use the filterCache to get a DocSet
408             if (toTermsEnum.docFreq() >= minDocFreqTo || resultBits == null) {
409               // use filter cache
410               DocSet toTermSet = toSearcher.getDocSet(toDeState);
411               resultListDocs += toTermSet.size();
412               if (resultBits != null) {
413                 toTermSet.addAllTo(new BitDocSet(resultBits));
414               } else {
415                 if (toTermSet instanceof BitDocSet) {
416                   resultBits = ((BitDocSet)toTermSet).bits.clone();
417                 } else {
418                   resultList.add(toTermSet);
419                 }
420               }
421             } else {
422               toTermDirectCount++;
423 
424               // need to use liveDocs here so we don't map to any deleted ones
425               toDeState.postingsEnum = toDeState.termsEnum.postings(toDeState.postingsEnum, PostingsEnum.NONE);
426               toDeState.postingsEnum = BitsFilteredPostingsEnum.wrap(toDeState.postingsEnum, toDeState.liveDocs);
427               PostingsEnum postingsEnum = toDeState.postingsEnum;
428 
429               if (postingsEnum instanceof MultiPostingsEnum) {
430                 MultiPostingsEnum.EnumWithSlice[] subs = ((MultiPostingsEnum) postingsEnum).getSubs();
431                 int numSubs = ((MultiPostingsEnum) postingsEnum).getNumSubs();
432                 for (int subindex = 0; subindex<numSubs; subindex++) {
433                   MultiPostingsEnum.EnumWithSlice sub = subs[subindex];
434                   if (sub.postingsEnum == null) continue;
435                   int base = sub.slice.start;
436                   int docid;
437                   while ((docid = sub.postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
438                     resultListDocs++;
439                     resultBits.set(docid + base);
440                   }
441                 }
442               } else {
443                 int docid;
444                 while ((docid = postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
445                   resultListDocs++;
446                   resultBits.set(docid);
447                 }
448               }
449             }
450 
451           }
452         }
453 
454         term = termsEnum.next();
455       }
456 
457       smallSetsDeferred = resultList.size();
458 
459       if (resultBits != null) {
460         BitDocSet bitSet = new BitDocSet(resultBits);
461         for (DocSet set : resultList) {
462           set.addAllTo(bitSet);
463         }
464         return bitSet;
465       }
466 
467       if (resultList.size()==0) {
468         return DocSet.EMPTY;
469       }
470 
471       if (resultList.size() == 1) {
472         return resultList.get(0);
473       }
474 
475       int sz = 0;
476 
477       for (DocSet set : resultList)
478         sz += set.size();
479 
480       int[] docs = new int[sz];
481       int pos = 0;
482       for (DocSet set : resultList) {
483         System.arraycopy(((SortedIntDocSet)set).getDocs(), 0, docs, pos, set.size());
484         pos += set.size();
485       }
486       Arrays.sort(docs);
487       int[] dedup = new int[sz];
488       pos = 0;
489       int last = -1;
490       for (int doc : docs) {
491         if (doc != last)
492           dedup[pos++] = doc;
493         last = doc;
494       }
495 
496       if (pos != dedup.length) {
497         dedup = Arrays.copyOf(dedup, pos);
498       }
499 
500       return new SortedIntDocSet(dedup, dedup.length);
501     }
502 
503   }

505   @Override
506   public String toString(String field) {
507     return "{!join from="+fromField+" to="+toField
508         + (fromIndex != null ? " fromIndex="+fromIndex : "")
509         +"}"+q.toString();
510   }

512   @Override
513   public boolean equals(Object o) {
514     if (!super.equals(o)) return false;
515     JoinQuery other = (JoinQuery)o;
516     return this.fromField.equals(other.fromField)
517            && this.toField.equals(other.toField)
518            && this.q.equals(other.q)
519            && (this.fromIndex == other.fromIndex || this.fromIndex != null && this.fromIndex.equals(other.fromIndex))
520            && this.fromCoreOpenTime == other.fromCoreOpenTime
521         ;
522   }
523 
524   @Override
525   public int hashCode() {
526     int h = super.hashCode();
527     h = h * 31 + q.hashCode();
528     h = h * 31 + (int)fromCoreOpenTime;
529     h = h * 31 + fromField.hashCode();
530     h = h * 31 + toField.hashCode();
531     return h;
532   }
