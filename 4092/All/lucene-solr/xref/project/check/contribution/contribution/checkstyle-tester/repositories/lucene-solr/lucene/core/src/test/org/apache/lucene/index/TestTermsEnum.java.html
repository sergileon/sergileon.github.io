43    public void test() throws Exception {
44      Random random = new Random(random().nextLong());
45      final LineFileDocs docs = new LineFileDocs(random, true);
46      final Directory d = newDirectory();
47      MockAnalyzer analyzer = new MockAnalyzer(random());
48      analyzer.setMaxTokenLength(TestUtil.nextInt(random(), 1, IndexWriter.MAX_TERM_LENGTH));
49      final RandomIndexWriter w = new RandomIndexWriter(random(), d, analyzer);
50      final int numDocs = atLeast(10);
51      for(int docCount=0;docCount<numDocs;docCount++) {
52        w.addDocument(docs.nextDoc());
53      }
54      final IndexReader r = w.getReader();
55      w.close();
56  
57      final List<BytesRef> terms = new ArrayList<>();
58      final TermsEnum termsEnum = MultiFields.getTerms(r, "body").iterator();
59      BytesRef term;
60      while((term = termsEnum.next()) != null) {
61        terms.add(BytesRef.deepCopyOf(term));
62      }
63      if (VERBOSE) {
64        System.out.println("TEST: " + terms.size() + " terms");
65      }
66  
67      int upto = -1;
68      final int iters = atLeast(200);
69      for(int iter=0;iter<iters;iter++) {
70        final boolean isEnd;
71        if (upto != -1 && random().nextBoolean()) {
72          // next
73          if (VERBOSE) {
74            System.out.println("TEST: iter next");
75          }
76          isEnd = termsEnum.next() == null;
77          upto++;
78          if (isEnd) {
79            if (VERBOSE) {
80              System.out.println("  end");
81            }
82            assertEquals(upto, terms.size());
83            upto = -1;
84          } else {
85            if (VERBOSE) {
86              System.out.println("  got term=" + termsEnum.term().utf8ToString() + " expected=" + terms.get(upto).utf8ToString());
87            }
88            assertTrue(upto < terms.size());
89            assertEquals(terms.get(upto), termsEnum.term());
90          }
91        } else {
92  
93          final BytesRef target;
94          final String exists;
95          if (random().nextBoolean()) {
96            // likely fake term
97            if (random().nextBoolean()) {
98              target = new BytesRef(TestUtil.randomSimpleString(random()));
99            } else {
100             target = new BytesRef(TestUtil.randomRealisticUnicodeString(random()));
101           }
102           exists = "likely not";
103         } else {
104           // real term
105           target = terms.get(random().nextInt(terms.size()));
106           exists = "yes";
107         }
108 
109         upto = Collections.binarySearch(terms, target);
110 
111         if (random().nextBoolean()) {
112           if (VERBOSE) {
113             System.out.println("TEST: iter seekCeil target=" + target.utf8ToString() + " exists=" + exists);
114           }
115           // seekCeil
116           final TermsEnum.SeekStatus status = termsEnum.seekCeil(target);
117           if (VERBOSE) {
118             System.out.println("  got " + status);
119           }
120           
121           if (upto < 0) {
122             upto = -(upto+1);
123             if (upto >= terms.size()) {
124               assertEquals(TermsEnum.SeekStatus.END, status);
125               upto = -1;
126             } else {
127               assertEquals(TermsEnum.SeekStatus.NOT_FOUND, status);
128               assertEquals(terms.get(upto), termsEnum.term());
129             }
130           } else {
131             assertEquals(TermsEnum.SeekStatus.FOUND, status);
132             assertEquals(terms.get(upto), termsEnum.term());
133           }
134         } else {
135           if (VERBOSE) {
136             System.out.println("TEST: iter seekExact target=" + target.utf8ToString() + " exists=" + exists);
137           }
138           // seekExact

140           if (VERBOSE) {
141             System.out.println("  got " + result);

143           if (upto < 0) {

145             upto = -1;

147             assertTrue(result);

174   private boolean accepts(CompiledAutomaton c, BytesRef b) {
175     int state = c.runAutomaton.getInitialState();
176     for(int idx=0;idx<b.length;idx++) {
177       assertTrue(state != -1);
178       state = c.runAutomaton.step(state, b.bytes[b.offset+idx] & 0xff);
179     }
180     return c.runAutomaton.isAccept(state);
181   }

524   public void testRandomTerms() throws Exception {
525     final String[] terms = new String[TestUtil.nextInt(random(), 1, atLeast(1000))];
526     final Set<String> seen = new HashSet<>();
527 
528     final boolean allowEmptyString = random().nextBoolean();
529 
530     if (random().nextInt(10) == 7 && terms.length > 2) {
531       // Sometimes add a bunch of terms sharing a longish common prefix:
532       final int numTermsSamePrefix = random().nextInt(terms.length/2);
533       if (numTermsSamePrefix > 0) {
534         String prefix;
535         while(true) {
536           prefix = getRandomString();
537           if (prefix.length() < 5) {
538             continue;
539           } else {
540             break;
541           }
542         }
543         while(seen.size() < numTermsSamePrefix) {
544           final String t = prefix + getRandomString();
545           if (!seen.contains(t)) {
546             terms[seen.size()] = t;
547             seen.add(t);
548           }
549         }
550       }
551     }
552 
553     while(seen.size() < terms.length) {
554       final String t = getRandomString();
555       if (!seen.contains(t) && (allowEmptyString || t.length() != 0)) {
556         terms[seen.size()] = t;
557         seen.add(t);
558       }
559     }
560     r = makeIndex(terms);
561     testRandomSeeks(r, terms);
562     close();
563   }

601   private void testRandomSeeks(IndexReader r, String... validTermStrings) throws IOException {
602     final BytesRef[] validTerms = new BytesRef[validTermStrings.length];
603     for(int termIDX=0;termIDX<validTermStrings.length;termIDX++) {
604       validTerms[termIDX] = new BytesRef(validTermStrings[termIDX]);
605     }
606     Arrays.sort(validTerms);
607     if (VERBOSE) {
608       System.out.println("TEST: " + validTerms.length + " terms:");
609       for(BytesRef t : validTerms) {
610         System.out.println("  " + t.utf8ToString() + " " + t);
611       }
612     }
613     final TermsEnum te = MultiFields.getTerms(r, FIELD).iterator();
614 
615     final int END_LOC = -validTerms.length-1;
616     
617     final List<TermAndState> termStates = new ArrayList<>();
618 
619     for(int iter=0;iter<100*RANDOM_MULTIPLIER;iter++) {
620 
621       final BytesRef t;
622       int loc;
623       final TermState termState;
624       if (random().nextInt(6) == 4) {
625         // pick term that doens't exist:
626         t = getNonExistTerm(validTerms);
627         termState = null;
628         if (VERBOSE) {
629           System.out.println("\nTEST: invalid term=" + t.utf8ToString());
630         }
631         loc = Arrays.binarySearch(validTerms, t);
632       } else if (termStates.size() != 0 && random().nextInt(4) == 1) {
633         final TermAndState ts = termStates.get(random().nextInt(termStates.size()));
634         t = ts.term;
635         loc = Arrays.binarySearch(validTerms, t);
636         assertTrue(loc >= 0);
637         termState = ts.state;
638         if (VERBOSE) {
639           System.out.println("\nTEST: valid termState term=" + t.utf8ToString());
640         }
641       } else {
642         // pick valid term
643         loc = random().nextInt(validTerms.length);
644         t = BytesRef.deepCopyOf(validTerms[loc]);
645         termState = null;
646         if (VERBOSE) {
647           System.out.println("\nTEST: valid term=" + t.utf8ToString());
648         }
649       }
650 
651       // seekCeil or seekExact:
652       final boolean doSeekExact = random().nextBoolean();
653       if (termState != null) {
654         if (VERBOSE) {
655           System.out.println("  seekExact termState");
656         }
657         te.seekExact(t, termState);
658       } else if (doSeekExact) {
659         if (VERBOSE) {
660           System.out.println("  seekExact");
661         }
662         assertEquals(loc >= 0, te.seekExact(t));
663       } else {
664         if (VERBOSE) {
665           System.out.println("  seekCeil");
666         }
667 
668         final TermsEnum.SeekStatus result = te.seekCeil(t);
669         if (VERBOSE) {
670           System.out.println("  got " + result);
671         }
672 
673         if (loc >= 0) {
674           assertEquals(TermsEnum.SeekStatus.FOUND, result);
675         } else if (loc == END_LOC) {
676           assertEquals(TermsEnum.SeekStatus.END, result);
677         } else {
678           assert loc >= -validTerms.length;
679           assertEquals(TermsEnum.SeekStatus.NOT_FOUND, result);
680         }
681       }
682 
683       if (loc >= 0) {
684         assertEquals(t, te.term());
685       } else if (doSeekExact) {
686         // TermsEnum is unpositioned if seekExact returns false
687         continue;
688       } else if (loc == END_LOC) {
689         continue;
690       } else {
691         loc = -loc-1;
692         assertEquals(validTerms[loc], te.term());
693       }
694 
695       // Do a bunch of next's after the seek
696       final int numNext = random().nextInt(validTerms.length);
697 
698       for(int nextCount=0;nextCount<numNext;nextCount++) {
699         if (VERBOSE) {
700           System.out.println("\nTEST: next loc=" + loc + " of " + validTerms.length);
701         }
702         final BytesRef t2 = te.next();
703         loc++;
704         if (loc == validTerms.length) {
705           assertNull(t2);
706           break;
707         } else {
708           assertEquals(validTerms[loc], t2);
709           if (random().nextInt(40) == 17 && termStates.size() < 100) {
710             termStates.add(new TermAndState(validTerms[loc], te.termState()));
711           }
712         }
713       }
714     }
