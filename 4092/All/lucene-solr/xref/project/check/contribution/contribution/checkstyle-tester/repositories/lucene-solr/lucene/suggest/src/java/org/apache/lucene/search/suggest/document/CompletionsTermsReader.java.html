20  import java.io.IOException;
21  import java.util.Collection;
22  import java.util.Collections;
23  
24  import org.apache.lucene.store.IndexInput;
25  import org.apache.lucene.util.Accountable;
26  
27  /**
28   * Holder for suggester and field-level info
29   * for a suggest field
30   *
31   * @lucene.experimental
32   */
33  public final class CompletionsTermsReader implements Accountable {
34    /** Minimum entry weight for the suggester */
35    public final long minWeight;
36    /** Maximum entry weight for the suggester */
37    public final long maxWeight;
38    /** type of suggester (context-enabled or not) */
39    public final byte type;
40    private final IndexInput dictIn;
41    private final long offset;
42  
43    private NRTSuggester suggester;
44  
45    /**
46     * Creates a CompletionTermsReader to load a field-specific suggester
47     * from the index <code>dictIn</code> with <code>offset</code>
48     */
49    CompletionsTermsReader(IndexInput dictIn, long offset, long minWeight, long maxWeight, byte type) throws IOException {
50      assert minWeight <= maxWeight;
51      assert offset >= 0l && offset < dictIn.length();
52      this.dictIn = dictIn;
53      this.offset = offset;
54      this.minWeight = minWeight;
55      this.maxWeight = maxWeight;
56      this.type = type;
57    }
58  
59    /**
60     * Returns the suggester for a field, if not loaded already, loads
61     * the appropriate suggester from CompletionDictionary
62     */
63    public synchronized NRTSuggester suggester() throws IOException {
64      if (suggester == null) {
65        try (IndexInput dictClone = dictIn.clone()) { // let multiple fields load concurrently
66          dictClone.seek(offset);
67          suggester = NRTSuggester.load(dictClone);
68        }
69      }
70      return suggester;
71    }
72  
73    @Override
74    public long ramBytesUsed() {
75      return (suggester != null) ? suggester.ramBytesUsed() : 0;
76    }
77  
78    @Override
79    public Collection<Accountable> getChildResources() {
80      return Collections.emptyList();
81    }
82  }
