138   @Override
139   public Query parse() throws SyntaxError {
140 
141     parsed = true;
142     
143     /* the main query we will execute.  we disable the coord because
144      * this query is an artificial construct
145      */
146     BooleanQuery.Builder query = new BooleanQuery.Builder();
147     query.setDisableCoord(true);
148     
149     /* * * Main User Query * * */
150     parsedUserQuery = null;
151     String userQuery = getString();
152     altUserQuery = null;
153     if( userQuery == null || userQuery.trim().length() == 0 ) {
154       // If no query is specified, we may have an alternate
155       if (config.altQ != null) {
156         QParser altQParser = subQuery(config.altQ, null);
157         altUserQuery = altQParser.getQuery();
158         query.add( altUserQuery , BooleanClause.Occur.MUST );
159       } else {
160         return null;
161         // throw new SyntaxError("missing query string" );
162       }
163     } else {
164       // There is a valid query string
165       ExtendedSolrQueryParser up = createEdismaxQueryParser(this, IMPOSSIBLE_FIELD_NAME);
166       up.addAlias(IMPOSSIBLE_FIELD_NAME, config.tiebreaker, config.queryFields);
167       addAliasesFromRequest(up, config.tiebreaker);
168       up.setPhraseSlop(config.qslop);     // slop for explicit user phrase queries
169       up.setAllowLeadingWildcard(true);
170       
171       // defer escaping and only do if lucene parsing fails, or we need phrases
172       // parsing fails.  Need to sloppy phrase queries anyway though.
173       List<Clause> clauses = splitIntoClauses(userQuery, false);
174       
175       // Always rebuild mainUserQuery from clauses to catch modifications from splitIntoClauses
176       // This was necessary for userFields modifications to get propagated into the query.
177       // Convert lower or mixed case operators to uppercase if we saw them.
178       // only do this for the lucene query part and not for phrase query boosting
179       // since some fields might not be case insensitive.
180       // We don't use a regex for this because it might change and AND or OR in
181       // a phrase query in a case sensitive field.
182       String mainUserQuery = rebuildUserQuery(clauses, config.lowercaseOperators);
183       
184       // but always for unstructured implicit bqs created by getFieldQuery
185       up.minShouldMatch = config.minShouldMatch;
186       
187       parsedUserQuery = parseOriginalQuery(up, mainUserQuery, clauses, config);
188       
189       if (parsedUserQuery == null) {
190         parsedUserQuery = parseEscapedQuery(up, escapeUserQuery(clauses), config);
191       }
192       
193       query.add(parsedUserQuery, BooleanClause.Occur.MUST);
194       
195       addPhraseFieldQueries(query, clauses, config);
196       
197     }
198     
199     /* * * Boosting Query * * */
200     boostQueries = getBoostQueries();
201     for(Query f : boostQueries) {
202       query.add(f, BooleanClause.Occur.SHOULD);
203     }
204     
205     /* * * Boosting Functions * * */
206     List<Query> boostFunctions = getBoostFunctions();
207     for(Query f : boostFunctions) {
208       query.add(f, BooleanClause.Occur.SHOULD);
209     }
210     
211     //
212     // create a boosted query (scores multiplied by boosts)
213     //
214     Query topQuery = query.build();
215     List<ValueSource> boosts = getMultiplicativeBoosts();
216     if (boosts.size()>1) {
217       ValueSource prod = new ProductFloatFunction(boosts.toArray(new ValueSource[boosts.size()]));
218       topQuery = new BoostedQuery(topQuery, prod);
219     } else if (boosts.size() == 1) {
220       topQuery = new BoostedQuery(topQuery, boosts.get(0));
221     }
222     
223     return topQuery;
224   }

226   /**
227    * Adds shingled phrase queries to all the fields specified in the pf, pf2 anf pf3 parameters
228    * 
229    */
230   protected void addPhraseFieldQueries(BooleanQuery.Builder query, List<Clause> clauses,
231       ExtendedDismaxConfiguration config) throws SyntaxError {
232 
233     // sloppy phrase queries for proximity
234     List<FieldParams> allPhraseFields = config.getAllPhraseFields();
235     
236     if (allPhraseFields.size() > 0) {
237       // find non-field clauses
238       List<Clause> normalClauses = new ArrayList<>(clauses.size());
239       for (Clause clause : clauses) {
240         if (clause.field != null || clause.isPhrase) continue;
241         // check for keywords "AND,OR,TO"
242         if (clause.isBareWord()) {
243           String s = clause.val;
244           // avoid putting explicit operators in the phrase query
245           if ("OR".equals(s) || "AND".equals(s) || "NOT".equals(s) || "TO".equals(s)) continue;
246         }
247         normalClauses.add(clause);
248       }
249 
250       // create a map of {wordGram, [phraseField]}
251       Multimap<Integer, FieldParams> phraseFieldsByWordGram = Multimaps.index(allPhraseFields, WORD_GRAM_EXTRACTOR);
252 
253       // for each {wordGram, [phraseField]} entry, create and add shingled field queries to the main user query
254       for (Map.Entry<Integer, Collection<FieldParams>> phraseFieldsByWordGramEntry : phraseFieldsByWordGram.asMap().entrySet()) {
255 
256         // group the fields within this wordGram collection by their associated slop (it's possible that the same
257         // field appears multiple times for the same wordGram count but with different slop values. In this case, we
258         // should take the *sum* of those phrase queries, rather than the max across them).
259         Multimap<Integer, FieldParams> phraseFieldsBySlop = Multimaps.index(phraseFieldsByWordGramEntry.getValue(), PHRASE_SLOP_EXTRACTOR);
260         for (Map.Entry<Integer, Collection<FieldParams>> phraseFieldsBySlopEntry : phraseFieldsBySlop.asMap().entrySet()) {
261           addShingledPhraseQueries(query, normalClauses, phraseFieldsBySlopEntry.getValue(),
262               phraseFieldsByWordGramEntry.getKey(), config.tiebreaker, phraseFieldsBySlopEntry.getKey());
263         }
264       }
265     }
266   }

285   /**
286    * Parses an escaped version of the user's query.  This method is called 
287    * in the event that the original query encounters exceptions during parsing.
288    *
289    * @param up parser used
290    * @param escapedUserQuery query that is parsed, should already be escaped so that no trivial parse errors are encountered
291    * @param config Configuration options for this parse request
292    * @return the resulting query (flattened if needed) with "min should match" rules applied as specified in the config.
293    * @see #parseOriginalQuery
294    * @see SolrPluginUtils#flattenBooleanQuery
295    */
296   protected Query parseEscapedQuery(ExtendedSolrQueryParser up,
297       String escapedUserQuery, ExtendedDismaxConfiguration config) throws SyntaxError {
298     Query query = up.parse(escapedUserQuery);
299     
300     if (query instanceof BooleanQuery) {
301       BooleanQuery.Builder t = new BooleanQuery.Builder();
302       SolrPluginUtils.flattenBooleanQuery(t, (BooleanQuery)query);
303       SolrPluginUtils.setMinShouldMatch(t, config.minShouldMatch);
304       query = t.build();
305     }
306     return query;
307   }

309   /**
310    * Parses the user's original query.  This method attempts to cleanly parse the specified query string using the specified parser, any Exceptions are ignored resulting in null being returned.
311    *
312    * @param up parser used
313    * @param mainUserQuery query string that is parsed
314    * @param clauses used to dictate "min should match" logic
315    * @param config Configuration options for this parse request
316    * @return the resulting query with "min should match" rules applied as specified in the config.
317    * @see #parseEscapedQuery
318    */
319    protected Query parseOriginalQuery(ExtendedSolrQueryParser up,
320       String mainUserQuery, List<Clause> clauses, ExtendedDismaxConfiguration config) {
321     
322     Query query = null;
323     try {
324       up.setRemoveStopFilter(!config.stopwords);
325       up.exceptions = true;
326       query = up.parse(mainUserQuery);
327       
328       if (shouldRemoveStopFilter(config, query)) {
329         // if the query was all stop words, remove none of them
330         up.setRemoveStopFilter(true);
331         query = up.parse(mainUserQuery);          
332       }
333     } catch (Exception e) {
334       // ignore failure and reparse later after escaping reserved chars
335       up.exceptions = false;
336     }
337     
338     if(query == null) {
339       return null;
340     }
341     // For correct lucene queries, turn off mm processing if there
342     // were explicit operators (except for AND).
343     boolean doMinMatched = doMinMatched(clauses, config.lowercaseOperators);
344     if (doMinMatched && query instanceof BooleanQuery) {
345       query = SolrPluginUtils.setMinShouldMatch((BooleanQuery)query, config.minShouldMatch);
346     }
347     return query;
348   }

360   private String escapeUserQuery(List<Clause> clauses) {
361     StringBuilder sb = new StringBuilder();
362     for (Clause clause : clauses) {
363       
364       boolean doQuote = clause.isPhrase;
365       
366       String s=clause.val;
367       if (!clause.isPhrase && ("OR".equals(s) || "AND".equals(s) || "NOT".equals(s))) {
368         doQuote=true;
369       }
370       
371       if (clause.must != 0) {
372         sb.append(clause.must);
373       }
374       if (clause.field != null) {
375         sb.append(clause.field);
376         sb.append(':');
377       }
378       if (doQuote) {
379         sb.append('"');
380       }
381       sb.append(clause.val);
382       if (doQuote) {
383         sb.append('"');
384       }
385       if (clause.field != null) {
386         // Add the default user field boost, if any
387         Float boost = config.userFields.getBoost(clause.field);
388         if(boost != null)
389           sb.append("^").append(boost);
390       }
391       sb.append(' ');
392     }
393     return sb.toString();
394   }

396   /**
397    * Returns false if at least one of the clauses is an explicit operator (except for AND)
398    */
399   private boolean doMinMatched(List<Clause> clauses, boolean lowercaseOperators) {
400     for (Clause clause : clauses) {
401       if (clause.must == '+') return false;
402       if (clause.must == '-') return false;
403       if (clause.isBareWord()) {
404         String s = clause.val;
405         if ("OR".equals(s)) {
406           return false;
407         } else if ("NOT".equals(s)) {
408           return false;
409         } else if (lowercaseOperators && "or".equals(s)) {
410           return false;
411         }
412       }
413     }
414     return true;
415   }

417   /**
418    * Generates a query string from the raw clauses, uppercasing 
419    * 'and' and 'or' as needed.
420    * @param clauses the clauses of the query string to be rebuilt
421    * @param lowercaseOperators if true, lowercase 'and' and 'or' clauses will 
422    *        be recognized as operators and uppercased in the final query string.
423    * @return the generated query string.
424    */
425   protected String rebuildUserQuery(List<Clause> clauses, boolean lowercaseOperators) {
426     StringBuilder sb = new StringBuilder();
427     for (int i=0; i<clauses.size(); i++) {
428       Clause clause = clauses.get(i);
429       String s = clause.raw;
430       // and and or won't be operators at the start or end
431       if (lowercaseOperators && i>0 && i+1<clauses.size()) {
432         if ("AND".equalsIgnoreCase(s)) {
433           s="AND";
434         } else if ("OR".equalsIgnoreCase(s)) {
435           s="OR";
436         }
437       }
438       sb.append(s);
439       sb.append(' ');
440     }
441     return sb.toString();
442   }

444   /**
445    * Parses all multiplicative boosts
446    */
447   protected List<ValueSource> getMultiplicativeBoosts() throws SyntaxError {
448     List<ValueSource> boosts = new ArrayList<>();
449     if (config.hasMultiplicativeBoosts()) {
450       for (String boostStr : config.multBoosts) {
451         if (boostStr==null || boostStr.length()==0) continue;
452         Query boost = subQuery(boostStr, FunctionQParserPlugin.NAME).getQuery();
453         ValueSource vs;
454         if (boost instanceof FunctionQuery) {
455           vs = ((FunctionQuery)boost).getValueSource();
456         } else {
457           vs = new QueryValueSource(boost, 1.0f);
458         }
459         boosts.add(vs);
460       }
461     }
462     return boosts;
463   }

465   /**
466    * Parses all function queries
467    */
468   protected List<Query> getBoostFunctions() throws SyntaxError {
469     List<Query> boostFunctions = new LinkedList<>();
470     if (config.hasBoostFunctions()) {
471       for (String boostFunc : config.boostFuncs) {
472         if(null == boostFunc || "".equals(boostFunc)) continue;
473         Map<String,Float> ff = SolrPluginUtils.parseFieldBoosts(boostFunc);
474         for (String f : ff.keySet()) {
475           Query fq = subQuery(f, FunctionQParserPlugin.NAME).getQuery();
476           Float b = ff.get(f);
477           if (null != b && b.floatValue() != 1f) {
478             fq = new BoostQuery(fq, b);
479           }
480           boostFunctions.add(fq);
481         }
482       }
483     }
484     return boostFunctions;
485   }

487   /**
488    * Parses all boost queries
489    */
490   protected List<Query> getBoostQueries() throws SyntaxError {
491     List<Query> boostQueries = new LinkedList<>();
492     if (config.hasBoostParams()) {
493       for (String qs : config.boostParams) {
494         if (qs.trim().length()==0) continue;
495         Query q = subQuery(qs, null).getQuery();
496         boostQueries.add(q);
497       }
498     }
499     return boostQueries;
500   }

502   /**
503    * Extracts all the aliased fields from the requests and adds them to up
504    */
505   private void addAliasesFromRequest(ExtendedSolrQueryParser up, float tiebreaker) {
506     Iterator<String> it = config.solrParams.getParameterNamesIterator();
507     while(it.hasNext()) {
508       String param = it.next();
509       if(param.startsWith("f.") && param.endsWith(".qf")) {
510         // Add the alias
511         String fname = param.substring(2,param.length()-3);
512         String qfReplacement = config.solrParams.get(param);
513         Map<String,Float> parsedQf = SolrPluginUtils.parseFieldBoosts(qfReplacement);
514         if(parsedQf.size() == 0)
515           return;
516         up.addAlias(fname, tiebreaker, parsedQf);
517       }
518     }
519   }

521   /**
522    * Modifies the main query by adding a new optional Query consisting
523    * of shingled phrase queries across the specified clauses using the 
524    * specified field =&gt; boost mappings.
525    *
526    * @param mainQuery Where the phrase boosting queries will be added
527    * @param clauses Clauses that will be used to construct the phrases
528    * @param fields Field =&gt; boost mappings for the phrase queries
529    * @param shingleSize how big the phrases should be, 0 means a single phrase
530    * @param tiebreaker tie breaker value for the DisjunctionMaxQueries
531    */
532   protected void addShingledPhraseQueries(final BooleanQuery.Builder mainQuery, 
533       final List<Clause> clauses,
534       final Collection<FieldParams> fields,
535       int shingleSize,
536       final float tiebreaker,
537       final int slop)
538           throws SyntaxError {
539     
540     if (null == fields || fields.isEmpty() || 
541         null == clauses || clauses.size() < shingleSize ) 
542       return;
543     
544     if (0 == shingleSize) shingleSize = clauses.size();
545     
546     final int lastClauseIndex = shingleSize-1;
547     
548     StringBuilder userPhraseQuery = new StringBuilder();
549     for (int i=0; i < clauses.size() - lastClauseIndex; i++) {
550       userPhraseQuery.append('"');
551       for (int j=0; j <= lastClauseIndex; j++) {
552         userPhraseQuery.append(clauses.get(i + j).val);
553         userPhraseQuery.append(' ');
554       }
555       userPhraseQuery.append('"');
556       userPhraseQuery.append(' ');
557     }
558     
559     /* for parsing sloppy phrases using DisjunctionMaxQueries */
560     ExtendedSolrQueryParser pp = createEdismaxQueryParser(this, IMPOSSIBLE_FIELD_NAME);
561 
562     pp.addAlias(IMPOSSIBLE_FIELD_NAME, tiebreaker, getFieldBoosts(fields));
563     pp.setPhraseSlop(slop);
564     pp.setRemoveStopFilter(true);  // remove stop filter and keep stopwords
565     
566     /* :TODO: reevaluate using makeDismax=true vs false...
567      * 
568      * The DismaxQueryParser always used DisjunctionMaxQueries for the 
569      * pf boost, for the same reasons it used them for the qf fields.
570      * When Yonik first wrote the ExtendedDismaxQParserPlugin, he added
571      * the "makeDismax=false" property to use BooleanQueries instead, but 
572      * when asked why his response was "I honestly don't recall" ...
573      *
574      * https://issues.apache.org/jira/browse/SOLR-1553?focusedCommentId=12793813#action_12793813
575      *
576      * so for now, we continue to use dismax style queries because it 
577      * seems the most logical and is back compatible, but we should 
578      * try to figure out what Yonik was thinking at the time (because he 
579      * rarely does things for no reason)
580      */
581     pp.makeDismax = true; 
582     
583     
584     // minClauseSize is independent of the shingleSize because of stop words
585     // (if they are removed from the middle, so be it, but we need at least 
586     // two or there shouldn't be a boost)
587     pp.minClauseSize = 2;  
588     
589     // TODO: perhaps we shouldn't use synonyms either...
590     
591     Query phrase = pp.parse(userPhraseQuery.toString());
592     if (phrase != null) {
593       mainQuery.add(phrase, BooleanClause.Occur.SHOULD);
594     }
595   }

597   /**
598    * @return a {fieldName, fieldBoost} map for the given fields.
599    */
600   private Map<String, Float> getFieldBoosts(Collection<FieldParams> fields) {
601     Map<String, Float> fieldBoostMap = new LinkedHashMap<>(fields.size());
602 
603     for (FieldParams field : fields) {
604       fieldBoostMap.put(field.getField(), field.getBoost());
605     }
606 
607     return fieldBoostMap;
608   }

715       return must==0 && !isPhrase;

722     protected boolean hasSpecialSyntax;

729   public List<Clause> splitIntoClauses(String s, boolean ignoreQuote) {
730     ArrayList<Clause> lst = new ArrayList<>(4);
731     Clause clause;
732     
733     int pos=0;
734     int end=s.length();
735     char ch=0;
736     int start;
737     boolean disallowUserField;
738     while (pos < end) {
739       clause = new Clause();
740       disallowUserField = true;
741       
742       ch = s.charAt(pos);
743       
744       while (Character.isWhitespace(ch)) {
745         if (++pos >= end) break;
746         ch = s.charAt(pos);
747       }
748       
749       start = pos;      
750       
751       if ((ch=='+' || ch=='-') && (pos+1)<end) {
752         clause.must = ch;
753         pos++;
754       }
755       
756       clause.field = getFieldName(s, pos, end);
757       if(clause.field != null && !config.userFields.isAllowed(clause.field)) {
758         clause.field = null;
759       }
760       if (clause.field != null) {
761         disallowUserField = false;
762         int colon = s.indexOf(':',pos);
763         clause.rawField = s.substring(pos, colon);
764         pos += colon - pos; // skip the field name
765         pos++;  // skip the ':'
766       }
767       
768       if (pos>=end) break;
769       
770       
771       char inString=0;
772       
773       ch = s.charAt(pos);
774       if (!ignoreQuote && ch=='"') {
775         clause.isPhrase = true;
776         inString = '"';
777         pos++;
778       }
779       
780       StringBuilder sb = new StringBuilder();
781       while (pos < end) {
782         ch = s.charAt(pos++);
783         if (ch=='\\') {    // skip escaped chars, but leave escaped
784           sb.append(ch);
785           if (pos >= end) {
786             sb.append(ch); // double backslash if we are at the end of the string
787             break;
788           }
789           ch = s.charAt(pos++);
790           sb.append(ch);
791           continue;
792         } else if (inString != 0 && ch == inString) {
793           inString=0;
794           break;
795         } else if (Character.isWhitespace(ch)) {
796           clause.hasWhitespace=true;
797           if (inString == 0) {
798             // end of the token if we aren't in a string, backing
799             // up the position.
800             pos--;
801             break;
802           }
803         }
804         
805         if (inString == 0) {
806           switch (ch) {
807             case '!':
808             case '(':
809             case ')':
810             case ':':
811             case '^':
812             case '[':
813             case ']':
814             case '{':
815             case '}':
816             case '~':
817             case '*':
818             case '?':
819             case '"':
820             case '+':
821             case '-':
822             case '\\':
823             case '|':
824             case '&':
825             case '/':
826               clause.hasSpecialSyntax = true;
827               sb.append('\\');
828           }
829         } else if (ch=='"') {
830           // only char we need to escape in a string is double quote
831           sb.append('\\');
832         }
833         sb.append(ch);
834       }
835       clause.val = sb.toString();
836       
837       if (clause.isPhrase) {
838         if (inString != 0) {
839           // detected bad quote balancing... retry
840           // parsing with quotes like any other char
841           return splitIntoClauses(s, true);
842         }
843         
844         // special syntax in a string isn't special
845         clause.hasSpecialSyntax = false;        
846       } else {
847         // an empty clause... must be just a + or - on its own
848         if (clause.val.length() == 0) {
849           clause.syntaxError = true;
850           if (clause.must != 0) {
851             clause.val="\\"+clause.must;
852             clause.must = 0;
853             clause.hasSpecialSyntax = true;
854           } else {
855             // uh.. this shouldn't happen.
856             clause=null;
857           }
858         }
859       }
860       
861       if (clause != null) {
862         if(disallowUserField) {
863           clause.raw = s.substring(start, pos);
864           // escape colons, except for "match all" query
865           if(!"*:*".equals(clause.raw)) {
866             clause.raw = clause.raw.replaceAll("([^\\\\]):", "$1\\\\:");
867           }
868         } else {
869           clause.raw = s.substring(start, pos);
870           // Add default userField boost if no explicit boost exists
871           if(config.userFields.isAllowed(clause.field) && !clause.raw.contains("^")) {
872             Float boost = config.userFields.getBoost(clause.field);
873             if(boost != null)
874               clause.raw += "^" + boost;
875           }
876         }
877         lst.add(clause);
878       }
879     }
880     
881     return lst;
882   }

884   /** 
885    * returns a field name or legal field alias from the current 
886    * position of the string 
887    */
888   public String getFieldName(String s, int pos, int end) {
889     if (pos >= end) return null;
890     int p=pos;
891     int colon = s.indexOf(':',pos);
892     // make sure there is space after the colon, but not whitespace
893     if (colon<=pos || colon+1>=end || Character.isWhitespace(s.charAt(colon+1))) return null;
894     char ch = s.charAt(p++);
895     while ((ch=='(' || ch=='+' || ch=='-') && (pos<end)) {
896       ch = s.charAt(p++);
897       pos++;
898     }
899     if (!Character.isJavaIdentifierPart(ch)) return null;
900     while (p<colon) {
901       ch = s.charAt(p++);
902       if (!(Character.isJavaIdentifierPart(ch) || ch=='-' || ch=='.')) return null;
903     }
904     String fname = s.substring(pos, p);
905     boolean isInSchema = getReq().getSchema().getFieldTypeNoEx(fname) != null;
906     boolean isAlias = config.solrParams.get("f."+fname+".qf") != null;
907     boolean isMagic = (null != MagicFieldName.get(fname));
908     
909     return (isInSchema || isAlias || isMagic) ? fname : null;
910   }

912   public static List<String> split(String s, boolean ignoreQuote) {
913     ArrayList<String> lst = new ArrayList<>(4);
914     int pos=0, start=0, end=s.length();
915     char inString=0;
916     char ch=0;
917     while (pos < end) {
918       char prevChar=ch;
919       ch = s.charAt(pos++);
920       if (ch=='\\') {    // skip escaped chars
921         pos++;
922       } else if (inString != 0 && ch==inString) {
923         inString=0;
924       } else if (!ignoreQuote && ch=='"') {
925         // If char is directly preceeded by a number or letter
926         // then don't treat it as the start of a string.
927         if (!Character.isLetterOrDigit(prevChar)) {
928           inString=ch;
929         }
930       } else if (Character.isWhitespace(ch) && inString==0) {
931         lst.add(s.substring(start,pos-1));
932         start=pos;
933       }
934     }
935     if (start < end) {
936       lst.add(s.substring(start,end));
937     }
938     
939     if (inString != 0) {
940       // unbalanced quote... ignore them
941       return split(s, true);
942     }
943     
944     return lst;
945   }

1130     /**
1131      * Delegates to the super class unless the field has been specified
1132      * as an alias -- in which case we recurse on each of
1133      * the aliased fields, and the results are composed into a
1134      * DisjunctionMaxQuery.  (so yes: aliases which point at other
1135      * aliases should work)
1136      */
1137     protected Query getAliasedQuery() throws SyntaxError {
1138       Alias a = aliases.get(field);
1139       this.validateCyclicAliasing(field);
1140       if (a != null) {
1141         List<Query> lst = getQueries(a);
1142         if (lst == null || lst.size()==0)
1143           return getQuery();
1144         // make a DisjunctionMaxQuery in this case too... it will stop
1145         // the "mm" processing from making everything required in the case
1146         // that the query expanded to multiple clauses.
1147         // DisMaxQuery.rewrite() removes itself if there is just a single clause anyway.
1148         // if (lst.size()==1) return lst.get(0);
1149         
1150         if (makeDismax) {
1151           DisjunctionMaxQuery q = new DisjunctionMaxQuery(lst, a.tie);
1152           return q;
1153         } else {
1154           // should we disable coord?
1155           BooleanQuery.Builder q = new BooleanQuery.Builder();
1156           q.setDisableCoord(disableCoord);
1157           for (Query sub : lst) {
1158             q.add(sub, BooleanClause.Occur.SHOULD);
1159           }
1160           return q.build();
1161         }
1162       } else {
1163         
1164         // verify that a fielded query is actually on a field that exists... if not,
1165         // then throw an exception to get us out of here, and we'll treat it like a
1166         // literal when we try the escape+re-parse.
1167         if (exceptions) {
1168           FieldType ft = schema.getFieldTypeNoEx(field);
1169           if (ft == null && null == MagicFieldName.get(field)) {
1170             throw unknownField;
1171           }
1172         }
1173         
1174         return getQuery();
1175       }

1180      */

1182       Set<String> set = new HashSet<>();

1189     private boolean validateField(String field, Set<String> set) {
1190       if(this.getAlias(field) == null) {
1191         return false;
1192       }
1193       boolean hascycle = false;
1194       for(String referencedField:this.getAlias(field).fields.keySet()) {
1195         if(!set.add(referencedField)) {
1196           hascycle = true;
1197         } else {
1198           if(validateField(referencedField, set)) {
1199             hascycle = true;
1200           }
1201           set.remove(referencedField);
1202         }
1203       }
1204       return hascycle;
1205     }

1208       if (a == null) return null;
1209       if (a.fields.size()==0) return null;

1211

1222       }

1226     private Query getQuery() {
1227       try {
1228         
1229         switch (type) {
1230           case FIELD:  // fallthrough
1231           case PHRASE:
1232             Query query = super.getFieldQuery(field, val, type == QType.PHRASE);
1233             // A BooleanQuery is only possible from getFieldQuery if it came from
1234             // a single whitespace separated term. In this case, check the coordination
1235             // factor on the query: if it's enabled, that means we aren't a set of synonyms
1236             // but instead multiple terms from one whitespace-separated term, we must
1237             // apply minShouldMatch here so that it works correctly with other things
1238             // like aliasing.
1239             if (query instanceof BooleanQuery) {
1240               BooleanQuery bq = (BooleanQuery) query;
1241               if (!bq.isCoordDisabled()) {
1242                 query = SolrPluginUtils.setMinShouldMatch(bq, minShouldMatch);
1243               }
1244             }
1245             if (query instanceof PhraseQuery) {
1246               PhraseQuery pq = (PhraseQuery)query;
1247               if (minClauseSize > 1 && pq.getTerms().length < minClauseSize) return null;
1248               PhraseQuery.Builder builder = new PhraseQuery.Builder();
1249               Term[] terms = pq.getTerms();
1250               int[] positions = pq.getPositions();
1251               for (int i = 0; i < terms.length; ++i) {
1252                 builder.add(terms[i], positions[i]);
1253               }
1254               builder.setSlop(slop);
1255               query = builder.build();
1256             } else if (query instanceof MultiPhraseQuery) {
1257               MultiPhraseQuery pq = (MultiPhraseQuery)query;
1258               if (minClauseSize > 1 && pq.getTermArrays().size() < minClauseSize) return null;
1259               ((MultiPhraseQuery)query).setSlop(slop);
1260             } else if (minClauseSize > 1) {
1261               // if it's not a type of phrase query, it doesn't meet the minClauseSize requirements
1262               return null;
1263             }
1264             return query;
1265           case PREFIX: return super.getPrefixQuery(field, val);
1266           case WILDCARD: return super.getWildcardQuery(field, val);
1267           case FUZZY: return super.getFuzzyQuery(field, val, flt);
1268           case RANGE: return super.getRangeQuery(field, val, val2, bool, bool2);
1269         }
1270         return null;
1271         
1272       } catch (Exception e) {
1273         // an exception here is due to the field query not being compatible with the input text
1274         // for example, passing a string to a numeric field.
1275         return null;
1276       }
1277     }

1279     private Analyzer noStopwordFilterAnalyzer(String fieldName) {
1280       FieldType ft = parser.getReq().getSchema().getFieldType(fieldName);
1281       Analyzer qa = ft.getQueryAnalyzer();
1282       if (!(qa instanceof TokenizerChain)) {
1283         return qa;
1284       }
1285       
1286       TokenizerChain tcq = (TokenizerChain) qa;
1287       Analyzer ia = ft.getIndexAnalyzer();
1288       if (ia == qa || !(ia instanceof TokenizerChain)) {
1289         return qa;
1290       }
1291       TokenizerChain tci = (TokenizerChain) ia;
1292       
1293       // make sure that there isn't a stop filter in the indexer
1294       for (TokenFilterFactory tf : tci.getTokenFilterFactories()) {
1295         if (tf instanceof StopFilterFactory) {
1296           return qa;
1297         }
1298       }
1299       
1300       // now if there is a stop filter in the query analyzer, remove it
1301       int stopIdx = -1;
1302       TokenFilterFactory[] facs = tcq.getTokenFilterFactories();
1303       
1304       for (int i = 0; i < facs.length; i++) {
1305         TokenFilterFactory tf = facs[i];
1306         if (tf instanceof StopFilterFactory) {
1307           stopIdx = i;
1308           break;
1309         }
1310       }
1311       
1312       if (stopIdx == -1) {
1313         // no stop filter exists
1314         return qa;
1315       }
1316       
1317       TokenFilterFactory[] newtf = new TokenFilterFactory[facs.length - 1];
1318       for (int i = 0, j = 0; i < facs.length; i++) {
1319         if (i == stopIdx) continue;
1320         newtf[j++] = facs[i];
1321       }
1322       
1323       TokenizerChain newa = new TokenizerChain(tcq.getTokenizerFactory(), newtf);
1324       newa.setPositionIncrementGap(tcq.getPositionIncrementGap(fieldName));
1325       return newa;
1326     }
1327   }

1329   static boolean isEmpty(Query q) {
1330     if (q==null) return true;
1331     if (q instanceof BooleanQuery && ((BooleanQuery)q).clauses().size()==0) return true;
1332     return false;
1333   }

1335   /**
1336    * Class that encapsulates the input from userFields parameter and can answer whether
1337    * a field allowed or disallowed as fielded query in the query string
1338    */
1339   static class UserFields {
1340     private Map<String,Float> userFieldsMap;
1341     private DynamicField[] dynamicUserFields;
1342     private DynamicField[] negativeDynamicUserFields;
1343     
1344     UserFields(Map<String,Float> ufm) {
1345       userFieldsMap = ufm;
1346       if (0 == userFieldsMap.size()) {
1347         userFieldsMap.put("*", null);
1348       }
1349       
1350       // Process dynamic patterns in userFields
1351       ArrayList<DynamicField> dynUserFields = new ArrayList<>();
1352       ArrayList<DynamicField> negDynUserFields = new ArrayList<>();
1353       for(String f : userFieldsMap.keySet()) {
1354         if(f.contains("*")) {
1355           if(f.startsWith("-"))
1356             negDynUserFields.add(new DynamicField(f.substring(1)));
1357           else
1358             dynUserFields.add(new DynamicField(f));
1359         }
1360       }
1361       Collections.sort(dynUserFields);
1362       dynamicUserFields = dynUserFields.toArray(new DynamicField[dynUserFields.size()]);
1363       Collections.sort(negDynUserFields);
1364       negativeDynamicUserFields = negDynUserFields.toArray(new DynamicField[negDynUserFields.size()]);
1365     }
1366     
1367     /**
1368      * Is the given field name allowed according to UserFields spec given in the uf parameter?
1369      * @param fname the field name to examine
1370      * @return true if the fielded queries are allowed on this field
1371      */
1372     public boolean isAllowed(String fname) {
1373       boolean res = ((userFieldsMap.containsKey(fname) || isDynField(fname, false)) && 
1374           !userFieldsMap.containsKey("-"+fname) &&
1375           !isDynField(fname, true));
1376       return res;
1377     }
1378     
1379     private boolean isDynField(String field, boolean neg) {
1380       return getDynFieldForName(field, neg) == null ? false : true;
1381     }
1382     
1383     private String getDynFieldForName(String f, boolean neg) {
1384       for( DynamicField df : neg?negativeDynamicUserFields:dynamicUserFields ) {
1385         if( df.matches( f ) ) return df.wildcard;
1386       }
1387       return null;
1388     }
1389     
1390     /**
1391      * Finds the default user field boost associated with the given field.
1392      * This is parsed from the uf parameter, and may be specified as wildcards, e.g. *name^2.0 or *^3.0
1393      * @param field the field to find boost for
1394      * @return the float boost value associated with the given field or a wildcard matching the field
1395      */
1396     public Float getBoost(String field) {
1397       return (userFieldsMap.containsKey(field)) ?
1398           userFieldsMap.get(field) : // Exact field
1399             userFieldsMap.get(getDynFieldForName(field, false)); // Dynamic field
1400     }
1401   }

1403   /* Represents a dynamic field, for easier matching, inspired by same class in IndexSchema */
1404   static class DynamicField implements Comparable<DynamicField> {
1405     final static int STARTS_WITH=1;
1406     final static int ENDS_WITH=2;
1407     final static int CATCHALL=3;
1408     
1409     final String wildcard;
1410     final int type;
1411     
1412     final String str;
1413     
1414     protected DynamicField(String wildcard) {
1415       this.wildcard = wildcard;
1416       if (wildcard.equals("*")) {
1417         type=CATCHALL;
1418         str=null;
1419       }
1420       else if (wildcard.startsWith("*")) {
1421         type=ENDS_WITH;
1422         str=wildcard.substring(1);
1423       }
1424       else if (wildcard.endsWith("*")) {
1425         type=STARTS_WITH;
1426         str=wildcard.substring(0,wildcard.length()-1);
1427       }
1428       else {
1429         throw new RuntimeException("dynamic field name must start or end with *");
1430       }
1431     }
1432     
1433     /*
1434      * Returns true if the regex wildcard for this DynamicField would match the input field name
1435      */
1436     public boolean matches(String name) {
1437       if (type==CATCHALL) return true;
1438       else if (type==STARTS_WITH && name.startsWith(str)) return true;
1439       else if (type==ENDS_WITH && name.endsWith(str)) return true;
1440       else return false;
1441     }
1442     
1443     /**
1444      * Sort order is based on length of regex.  Longest comes first.
1445      * @param other The object to compare to.
1446      * @return a negative integer, zero, or a positive integer
1447      * as this object is less than, equal to, or greater than
1448      * the specified object.
1449      */
1450     @Override
1451     public int compareTo(DynamicField other) {
1452       return other.wildcard.length() - wildcard.length();
1453     }
1454     
1455     @Override
1456     public String toString() {
1457       return this.wildcard;
1458     }
1459   }
