96    /**
97     * read a set of queries from a resource file
98     */
99    private Set<String> readQueries(String resource) throws IOException {
100     Set<String> queries = new HashSet<>();
101     InputStream stream = getClass().getResourceAsStream(resource);
102     BufferedReader reader = new BufferedReader(new InputStreamReader(stream, StandardCharsets.UTF_8));
103     String line = null;
104     while ((line = reader.readLine()) != null) {
105       line = line.trim();
106       if (line.length() > 0 && !line.startsWith("#") && !line.startsWith("//")) {
107         queries.add(line);
108       }
109     }
110     return queries;
111   }

125    * Build a randomish document for both RAMDirectory and MemoryIndex,
126    * and run all the queries against it.
127    */
128   public void assertAgainstRAMDirectory(MemoryIndex memory) throws Exception {
129     memory.reset();
130     StringBuilder fooField = new StringBuilder();
131     StringBuilder termField = new StringBuilder();
132  
133     // add up to 250 terms to field "foo"
134     final int numFooTerms = random().nextInt(250 * RANDOM_MULTIPLIER);
135     for (int i = 0; i < numFooTerms; i++) {
136       fooField.append(" ");
137       fooField.append(randomTerm());
138     }
139 
140     // add up to 250 terms to field "term"
141     final int numTermTerms = random().nextInt(250 * RANDOM_MULTIPLIER);
142     for (int i = 0; i < numTermTerms; i++) {
143       termField.append(" ");
144       termField.append(randomTerm());
145     }
146     
147     Directory ramdir = new RAMDirectory();
148     Analyzer analyzer = randomAnalyzer();
149     IndexWriter writer = new IndexWriter(ramdir,
150                                          new IndexWriterConfig(analyzer).setCodec(
151                                              TestUtil.alwaysPostingsFormat(TestUtil.getDefaultPostingsFormat())));
152     Document doc = new Document();
153     Field field1 = newTextField("foo", fooField.toString(), Field.Store.NO);
154     Field field2 = newTextField("term", termField.toString(), Field.Store.NO);
155     doc.add(field1);
156     doc.add(field2);
157     writer.addDocument(doc);
158     writer.close();
159     
160     memory.addField("foo", fooField.toString(), analyzer);
161     memory.addField("term", termField.toString(), analyzer);
162     
163     LeafReader reader = (LeafReader) memory.createSearcher().getIndexReader();
164     TestUtil.checkReader(reader);
165     DirectoryReader competitor = DirectoryReader.open(ramdir);
166     duellReaders(competitor, reader);
167     IOUtils.close(reader, competitor);
168     assertAllQueries(memory, ramdir, analyzer);  
169     ramdir.close();    
170   }

172   private void duellReaders(CompositeReader other, LeafReader memIndexReader)
173       throws IOException {
174     LeafReader competitor = SlowCompositeReaderWrapper.wrap(other);
175     Fields memFields = memIndexReader.fields();
176     for (String field : competitor.fields()) {
177       Terms memTerms = memFields.terms(field);
178       Terms iwTerms = memIndexReader.terms(field);
179       if (iwTerms == null) {
180         assertNull(memTerms);
181       } else {
182         NumericDocValues normValues = competitor.getNormValues(field);
183         NumericDocValues memNormValues = memIndexReader.getNormValues(field);
184         if (normValues != null) {
185           // mem idx always computes norms on the fly
186           assertNotNull(memNormValues);
187           assertEquals(normValues.get(0), memNormValues.get(0));
188         }
189           
190         assertNotNull(memTerms);
191         assertEquals(iwTerms.getDocCount(), memTerms.getDocCount());
192         assertEquals(iwTerms.getSumDocFreq(), memTerms.getSumDocFreq());
193         assertEquals(iwTerms.getSumTotalTermFreq(), memTerms.getSumTotalTermFreq());
194         TermsEnum iwTermsIter = iwTerms.iterator();
195         TermsEnum memTermsIter = memTerms.iterator();
196         if (iwTerms.hasPositions()) {
197           final boolean offsets = iwTerms.hasOffsets() && memTerms.hasOffsets();
198          
199           while(iwTermsIter.next() != null) {
200             assertNotNull(memTermsIter.next());
201             assertEquals(iwTermsIter.term(), memTermsIter.term());
202             PostingsEnum iwDocsAndPos = iwTermsIter.postings(null, PostingsEnum.ALL);
203             PostingsEnum memDocsAndPos = memTermsIter.postings(null, PostingsEnum.ALL);
204             while(iwDocsAndPos.nextDoc() != PostingsEnum.NO_MORE_DOCS) {
205               assertEquals(iwDocsAndPos.docID(), memDocsAndPos.nextDoc());
206               assertEquals(iwDocsAndPos.freq(), memDocsAndPos.freq());
207               for (int i = 0; i < iwDocsAndPos.freq(); i++) {
208                 assertEquals("term: " + iwTermsIter.term().utf8ToString(), iwDocsAndPos.nextPosition(), memDocsAndPos.nextPosition());
209                 if (offsets) {
210                   assertEquals(iwDocsAndPos.startOffset(), memDocsAndPos.startOffset());
211                   assertEquals(iwDocsAndPos.endOffset(), memDocsAndPos.endOffset());
212                 }
213 
214                 if (iwTerms.hasPayloads()) {
215                   assertEquals(iwDocsAndPos.getPayload(), memDocsAndPos.getPayload());
216                 }
217               }
218               
219             }
220             
221           }
222         } else {
223           while(iwTermsIter.next() != null) {
224             assertEquals(iwTermsIter.term(), memTermsIter.term());
225             PostingsEnum iwDocsAndPos = iwTermsIter.postings(null);
226             PostingsEnum memDocsAndPos = memTermsIter.postings(null);
227             while(iwDocsAndPos.nextDoc() != PostingsEnum.NO_MORE_DOCS) {
228               assertEquals(iwDocsAndPos.docID(), memDocsAndPos.nextDoc());
229               assertEquals(iwDocsAndPos.freq(), memDocsAndPos.freq());
230             }
231           }
232         }
233       }
234       
235     }
236   }

238   /**
239    * Run all queries against both the RAMDirectory and MemoryIndex, ensuring they are the same.
240    */
241   public void assertAllQueries(MemoryIndex memory, Directory ramdir, Analyzer analyzer) throws Exception {
242     IndexReader reader = DirectoryReader.open(ramdir);
243     IndexSearcher ram = newSearcher(reader);
244     IndexSearcher mem = memory.createSearcher();
245     QueryParser qp = new QueryParser("foo", analyzer);
246     for (String query : queries) {
247       TopDocs ramDocs = ram.search(qp.parse(query), 1);
248       TopDocs memDocs = mem.search(qp.parse(query), 1);
249       assertEquals(query, ramDocs.totalHits, memDocs.totalHits);
250     }
251     reader.close();
252   }

255    * Return a random analyzer (Simple, Stop, Standard) to analyze the terms.
256    */
257   private Analyzer randomAnalyzer() {
258     switch(random().nextInt(4)) {
259       case 0: return new MockAnalyzer(random(), MockTokenizer.SIMPLE, true);
260       case 1: return new MockAnalyzer(random(), MockTokenizer.SIMPLE, true, MockTokenFilter.ENGLISH_STOPSET);
261       case 2: return new Analyzer() {
262         @Override
263         protected TokenStreamComponents createComponents(String fieldName) {
264           Tokenizer tokenizer = new MockTokenizer();
265           return new TokenStreamComponents(tokenizer, new CrazyTokenFilter(tokenizer));
266         }
267       };
268       default: return new MockAnalyzer(random(), MockTokenizer.WHITESPACE, false);
269     }
270   }

272   // a tokenfilter that makes all terms starting with 't' empty strings
273   static final class CrazyTokenFilter extends TokenFilter {
274     final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
275     
276     CrazyTokenFilter(TokenStream input) {
277       super(input);
278     }
279 
280     @Override
281     public boolean incrementToken() throws IOException {
282       if (input.incrementToken()) {
283         if (termAtt.length() > 0 && termAtt.buffer()[0] == 't') {
284           termAtt.setLength(0);
285         }
286         return true;
287       } else {
288         return false;
289       }
290     }
291   };

318     Analyzer analyzer = new MockAnalyzer(random());
319     MemoryIndex memory = new MemoryIndex(random().nextBoolean(), false, random().nextInt(50) * 1024 * 1024);

439     LineFileDocs lineFileDocs = new LineFileDocs(random());
440     int numDocs = atLeast(10);
441     MemoryIndex memory = randomMemoryIndex();
442     for (int i = 0; i < numDocs; i++) {
443       Directory dir = newDirectory();
444       MockAnalyzer mockAnalyzer = new MockAnalyzer(random());
445       mockAnalyzer.setMaxTokenLength(TestUtil.nextInt(random(), 1, IndexWriter.MAX_TERM_LENGTH));
446       IndexWriter writer = new IndexWriter(dir, newIndexWriterConfig(random(), mockAnalyzer));
447       Document nextDoc = lineFileDocs.nextDoc();
448       Document doc = new Document();
449       for (IndexableField field : nextDoc.getFields()) {
450         if (field.fieldType().indexOptions() != IndexOptions.NONE) {
451           doc.add(field);
452           if (random().nextInt(3) == 0) {
453             doc.add(field);  // randomly add the same field twice
454           }
455         }
456       }
457       
458       writer.addDocument(doc);
459       writer.close();
460       for (IndexableField field : doc.getFields()) {
461         memory.addField(field.name(), ((Field)field).stringValue(), mockAnalyzer);  
462       }
463       DirectoryReader competitor = DirectoryReader.open(dir);
464       LeafReader memIndexReader= (LeafReader) memory.createSearcher().getIndexReader();
465       TestUtil.checkReader(memIndexReader);
466       duellReaders(competitor, memIndexReader);
467       IOUtils.close(competitor, memIndexReader);
468       memory.reset();
469       dir.close();
470     }
471     lineFileDocs.close();
472   }

528   protected void compareTermVectors(Terms terms, Terms memTerms, String field_name) throws IOException {
529 
530     TermsEnum termEnum = terms.iterator();
531     TermsEnum memTermEnum = memTerms.iterator();
532 
533     while (termEnum.next() != null) {
534       assertNotNull(memTermEnum.next());
535       assertThat(termEnum.totalTermFreq(), equalTo(memTermEnum.totalTermFreq()));
536 
537       PostingsEnum docsPosEnum = termEnum.postings(null, PostingsEnum.POSITIONS);
538       PostingsEnum memDocsPosEnum = memTermEnum.postings(null, PostingsEnum.POSITIONS);
539       String currentTerm = termEnum.term().utf8ToString();
540 
541       assertThat("Token mismatch for field: " + field_name, currentTerm, equalTo(memTermEnum.term().utf8ToString()));
542 
543       docsPosEnum.nextDoc();
544       memDocsPosEnum.nextDoc();
545 
546       int freq = docsPosEnum.freq();
547       assertThat(freq, equalTo(memDocsPosEnum.freq()));
548       for (int i = 0; i < freq; i++) {
549         String failDesc = " (field:" + field_name + " term:" + currentTerm + ")";
550         int memPos = memDocsPosEnum.nextPosition();
551         int pos = docsPosEnum.nextPosition();
552         assertThat("Position test failed" + failDesc, memPos, equalTo(pos));
553         assertThat("Start offset test failed" + failDesc, memDocsPosEnum.startOffset(), equalTo(docsPosEnum.startOffset()));
554         assertThat("End offset test failed" + failDesc, memDocsPosEnum.endOffset(), equalTo(docsPosEnum.endOffset()));
555         assertThat("Missing payload test failed" + failDesc, docsPosEnum.getPayload(), equalTo(docsPosEnum.getPayload()));
556       }
557     }
558     assertNull("Still some tokens not processed", memTermEnum.next());
559   }
