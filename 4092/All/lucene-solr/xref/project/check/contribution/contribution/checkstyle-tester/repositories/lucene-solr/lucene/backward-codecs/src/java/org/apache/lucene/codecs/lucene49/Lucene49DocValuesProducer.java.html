111   /** expert: instantiates a new reader */
112   Lucene49DocValuesProducer(SegmentReadState state, String dataCodec, String dataExtension, String metaCodec, String metaExtension) throws IOException {
113     String metaName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, metaExtension);
114     this.maxDoc = state.segmentInfo.maxDoc();
115     merging = false;
116     
117     int version = -1;
118     int numFields = -1;
119     
120     // read in the entries from the metadata file.
121     try (ChecksumIndexInput in = state.directory.openChecksumInput(metaName, state.context)) {
122       Throwable priorE = null;
123       try {
124         version = CodecUtil.checkHeader(in, metaCodec, 
125                                             Lucene49DocValuesFormat.VERSION_START,
126                                             Lucene49DocValuesFormat.VERSION_CURRENT);
127         numFields = readFields(in, state.fieldInfos);
128       } catch (Throwable exception) {
129         priorE = exception;
130       } finally {
131         CodecUtil.checkFooter(in, priorE);
132       }
133     }
134     this.numFields = numFields;
135 
136     String dataName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, dataExtension);
137     this.data = state.directory.openInput(dataName, state.context);
138     boolean success = false;
139     try {
140       final int version2 = CodecUtil.checkHeader(data, dataCodec, 
141                                                  Lucene49DocValuesFormat.VERSION_START,
142                                                  Lucene49DocValuesFormat.VERSION_CURRENT);
143       if (version != version2) {
144         throw new CorruptIndexException("Format versions mismatch: meta=" + version + ", data=" + version2, data);
145       }
146       
147       // NOTE: data file is too costly to verify checksum against all the bytes on open,
148       // but for now we at least verify proper structure of the checksum footer: which looks
149       // for FOOTER_MAGIC + algorithmID. This is cheap and can detect some forms of corruption
150       // such as file truncation.
151       CodecUtil.retrieveChecksum(data);
152 
153       success = true;
154     } finally {
155       if (!success) {
156         IOUtils.closeWhileHandlingException(this.data);
157       }
158     }
159     
160     ramBytesUsed = new AtomicLong(RamUsageEstimator.shallowSizeOfInstance(getClass()));
161   }

163   private void readSortedField(FieldInfo info, IndexInput meta) throws IOException {
164     // sorted = binary + numeric
165     if (meta.readVInt() != info.number) {
166       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
167     }
168     if (meta.readByte() != Lucene49DocValuesFormat.BINARY) {
169       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
170     }
171     BinaryEntry b = readBinaryEntry(meta);
172     binaries.put(info.name, b);
173     
174     if (meta.readVInt() != info.number) {
175       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
176     }
177     if (meta.readByte() != Lucene49DocValuesFormat.NUMERIC) {
178       throw new CorruptIndexException("sorted entry for field: " + info.name + " is corrupt", meta);
179     }
180     NumericEntry n = readNumericEntry(meta);
181     ords.put(info.name, n);
182   }

184   private void readSortedSetFieldWithAddresses(FieldInfo info, IndexInput meta) throws IOException {
185     // sortedset = binary + numeric (addresses) + ordIndex
186     if (meta.readVInt() != info.number) {
187       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
188     }
189     if (meta.readByte() != Lucene49DocValuesFormat.BINARY) {
190       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
191     }
192     BinaryEntry b = readBinaryEntry(meta);
193     binaries.put(info.name, b);
194 
195     if (meta.readVInt() != info.number) {
196       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
197     }
198     if (meta.readByte() != Lucene49DocValuesFormat.NUMERIC) {
199       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
200     }
201     NumericEntry n1 = readNumericEntry(meta);
202     ords.put(info.name, n1);
203 
204     if (meta.readVInt() != info.number) {
205       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
206     }
207     if (meta.readByte() != Lucene49DocValuesFormat.NUMERIC) {
208       throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
209     }
210     NumericEntry n2 = readNumericEntry(meta);
211     ordIndexes.put(info.name, n2);
212   }

214   private int readFields(IndexInput meta, FieldInfos infos) throws IOException {
215     int numFields = 0;
216     int fieldNumber = meta.readVInt();
217     while (fieldNumber != -1) {
218       numFields++;
219       FieldInfo info = infos.fieldInfo(fieldNumber);
220       if (info == null) {
221         // trickier to validate more: because we use multiple entries for "composite" types like sortedset, etc.
222         throw new CorruptIndexException("Invalid field number: " + fieldNumber, meta);
223       }
224       byte type = meta.readByte();
225       if (type == Lucene49DocValuesFormat.NUMERIC) {
226         numerics.put(info.name, readNumericEntry(meta));
227       } else if (type == Lucene49DocValuesFormat.BINARY) {
228         BinaryEntry b = readBinaryEntry(meta);
229         binaries.put(info.name, b);
230       } else if (type == Lucene49DocValuesFormat.SORTED) {
231         readSortedField(info, meta);
232       } else if (type == Lucene49DocValuesFormat.SORTED_SET) {
233         SortedSetEntry ss = readSortedSetEntry(meta);
234         sortedSets.put(info.name, ss);
235         if (ss.format == SORTED_WITH_ADDRESSES) {
236           readSortedSetFieldWithAddresses(info, meta);
237         } else if (ss.format == SORTED_SINGLE_VALUED) {
238           if (meta.readVInt() != fieldNumber) {
239             throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
240           }
241           if (meta.readByte() != Lucene49DocValuesFormat.SORTED) {
242             throw new CorruptIndexException("sortedset entry for field: " + info.name + " is corrupt", meta);
243           }
244           readSortedField(info, meta);
245         } else {
246           throw new AssertionError();
247         }
248       } else if (type == Lucene49DocValuesFormat.SORTED_NUMERIC) {
249         SortedSetEntry ss = readSortedSetEntry(meta);
250         sortedNumerics.put(info.name, ss);
251         if (meta.readVInt() != fieldNumber) {
252           throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
253         }
254         if (meta.readByte() != Lucene49DocValuesFormat.NUMERIC) {
255           throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
256         }
257         numerics.put(info.name, readNumericEntry(meta));
258         if (ss.format == SORTED_WITH_ADDRESSES) {
259           if (meta.readVInt() != fieldNumber) {
260             throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
261           }
262           if (meta.readByte() != Lucene49DocValuesFormat.NUMERIC) {
263             throw new CorruptIndexException("sortednumeric entry for field: " + info.name + " is corrupt", meta);
264           }
265           NumericEntry ordIndex = readNumericEntry(meta);
266           ordIndexes.put(info.name, ordIndex);
267         } else if (ss.format != SORTED_SINGLE_VALUED) {
268           throw new AssertionError();
269         }
270       } else {
271         throw new CorruptIndexException("invalid type: " + type, meta);
272       }
273       fieldNumber = meta.readVInt();
274     }
275     return numFields;
276   }

278   static NumericEntry readNumericEntry(IndexInput meta) throws IOException {
279     NumericEntry entry = new NumericEntry();
280     entry.format = meta.readVInt();
281     entry.missingOffset = meta.readLong();
282     entry.offset = meta.readLong();
283     entry.count = meta.readVLong();
284     switch(entry.format) {
285       case GCD_COMPRESSED:
286         entry.minValue = meta.readLong();
287         entry.gcd = meta.readLong();
288         entry.bitsPerValue = meta.readVInt();
289         break;
290       case TABLE_COMPRESSED:
291         final int uniqueValues = meta.readVInt();
292         if (uniqueValues > 256) {
293           throw new CorruptIndexException("TABLE_COMPRESSED cannot have more than 256 distinct values, got=" + uniqueValues, meta);
294         }
295         entry.table = new long[uniqueValues];
296         for (int i = 0; i < uniqueValues; ++i) {
297           entry.table[i] = meta.readLong();
298         }
299         entry.bitsPerValue = meta.readVInt();
300         break;
301       case DELTA_COMPRESSED:
302         entry.minValue = meta.readLong();
303         entry.bitsPerValue = meta.readVInt();
304         break;
305       case MONOTONIC_COMPRESSED:
306         entry.packedIntsVersion = meta.readVInt();
307         entry.blockSize = meta.readVInt();
308         break;
309       default:
310         throw new CorruptIndexException("Unknown format: " + entry.format, meta);
311     }
312     entry.endOffset = meta.readLong();
313     return entry;
314   }

316   static BinaryEntry readBinaryEntry(IndexInput meta) throws IOException {
317     BinaryEntry entry = new BinaryEntry();
318     entry.format = meta.readVInt();
319     entry.missingOffset = meta.readLong();
320     entry.minLength = meta.readVInt();
321     entry.maxLength = meta.readVInt();
322     entry.count = meta.readVLong();
323     entry.offset = meta.readLong();
324     switch(entry.format) {
325       case BINARY_FIXED_UNCOMPRESSED:
326         break;
327       case BINARY_PREFIX_COMPRESSED:
328         entry.addressInterval = meta.readVInt();
329         entry.addressesOffset = meta.readLong();
330         entry.packedIntsVersion = meta.readVInt();
331         entry.blockSize = meta.readVInt();
332         break;
333       case BINARY_VARIABLE_UNCOMPRESSED:
334         entry.addressesOffset = meta.readLong();
335         entry.packedIntsVersion = meta.readVInt();
336         entry.blockSize = meta.readVInt();
337         break;
338       default:
339         throw new CorruptIndexException("Unknown format: " + entry.format, meta);
340     }
341     return entry;
342   }

382   LongValues getNumeric(NumericEntry entry) throws IOException {
383     RandomAccessInput slice = this.data.randomAccessSlice(entry.offset, entry.endOffset - entry.offset);
384     switch (entry.format) {
385       case DELTA_COMPRESSED:
386         final long delta = entry.minValue;
387         final LongValues values = DirectReader.getInstance(slice, entry.bitsPerValue);
388         return new LongValues() {
389           @Override
390           public long get(long id) {
391             return delta + values.get(id);
392           }
393         };
394       case GCD_COMPRESSED:
395         final long min = entry.minValue;
396         final long mult = entry.gcd;
397         final LongValues quotientReader = DirectReader.getInstance(slice, entry.bitsPerValue);
398         return new LongValues() {
399           @Override
400           public long get(long id) {
401             return min + mult * quotientReader.get(id);
402           }
403         };
404       case TABLE_COMPRESSED:
405         final long table[] = entry.table;
406         final LongValues ords = DirectReader.getInstance(slice, entry.bitsPerValue);
407         return new LongValues() {
408           @Override
409           public long get(long id) {
410             return table[(int) ords.get(id)];
411           }
412         };
413       default:
414         throw new AssertionError();
415     }
416   }

433   private BinaryDocValues getFixedBinary(FieldInfo field, final BinaryEntry bytes) {
434     final IndexInput data = this.data.clone();
435 
436     return new LongBinaryDocValues() {
437       final BytesRef term;
438       {
439         term = new BytesRef(bytes.maxLength);
440         term.offset = 0;
441         term.length = bytes.maxLength;
442       }
443       
444       @Override
445       public BytesRef get(long id) {
446         long address = bytes.offset + id * bytes.maxLength;
447         try {
448           data.seek(address);
449           data.readBytes(term.bytes, 0, term.length);
450           return term;
451         } catch (IOException e) {
452           throw new RuntimeException(e);
453         }
454       }
455     };
456   }

458   /** returns an address instance for variable-length binary values. */
459   private synchronized MonotonicBlockPackedReader getAddressInstance(IndexInput data, FieldInfo field, BinaryEntry bytes) throws IOException {
460     final MonotonicBlockPackedReader addresses;
461     MonotonicBlockPackedReader addrInstance = addressInstances.get(field.name);
462     if (addrInstance == null) {
463       data.seek(bytes.addressesOffset);
464       addrInstance = MonotonicBlockPackedReader.of(data, bytes.packedIntsVersion, bytes.blockSize, bytes.count+1, false);
465       if (!merging) {
466         addressInstances.put(field.name, addrInstance);
467         ramBytesUsed.addAndGet(addrInstance.ramBytesUsed() + RamUsageEstimator.NUM_BYTES_INT);
468       }
469     }
470     addresses = addrInstance;
471     return addresses;
472   }

474   private BinaryDocValues getVariableBinary(FieldInfo field, final BinaryEntry bytes) throws IOException {
475     final IndexInput data = this.data.clone();
476     
477     final MonotonicBlockPackedReader addresses = getAddressInstance(data, field, bytes);
478 
479     return new LongBinaryDocValues() {
480       final BytesRef term = new BytesRef(Math.max(0, bytes.maxLength));
481       
482       @Override
483       public BytesRef get(long id) {
484         long startAddress = bytes.offset + addresses.get(id);
485         long endAddress = bytes.offset + addresses.get(id+1);
486         int length = (int) (endAddress - startAddress);
487         try {
488           data.seek(startAddress);
489           data.readBytes(term.bytes, 0, length);
490           term.length = length;
491           return term;
492         } catch (IOException e) {
493           throw new RuntimeException(e);
494         }
495       }
496     };
497   }

499   /** returns an address instance for prefix-compressed binary values. */
500   private synchronized MonotonicBlockPackedReader getIntervalInstance(IndexInput data, FieldInfo field, BinaryEntry bytes) throws IOException {
501     final MonotonicBlockPackedReader addresses;
502     final long interval = bytes.addressInterval;
503     MonotonicBlockPackedReader addrInstance = addressInstances.get(field.name);
504     if (addrInstance == null) {
505       data.seek(bytes.addressesOffset);
506       final long size;
507       if (bytes.count % interval == 0) {
508         size = bytes.count / interval;
509       } else {
510         size = 1L + bytes.count / interval;
511       }
512       addrInstance = MonotonicBlockPackedReader.of(data, bytes.packedIntsVersion, bytes.blockSize, size, false);
513       if (!merging) {
514         addressInstances.put(field.name, addrInstance);
515         ramBytesUsed.addAndGet(addrInstance.ramBytesUsed() + RamUsageEstimator.NUM_BYTES_INT);
516       }
517     }
518     addresses = addrInstance;
519     return addresses;
520   }

591   @Override
592   public SortedNumericDocValues getSortedNumeric(FieldInfo field) throws IOException {
593     SortedSetEntry ss = sortedNumerics.get(field.name);
594     NumericEntry numericEntry = numerics.get(field.name);
595     final LongValues values = getNumeric(numericEntry);
596     if (ss.format == SORTED_SINGLE_VALUED) {
597       final Bits docsWithField = getMissingBits(numericEntry.missingOffset);
598       return DocValues.singleton(values, docsWithField);
599     } else if (ss.format == SORTED_WITH_ADDRESSES) {
600       final IndexInput data = this.data.clone();
601       final MonotonicBlockPackedReader ordIndex = getOrdIndexInstance(data, field, ordIndexes.get(field.name));
602       
603       return new SortedNumericDocValues() {
604         long startOffset;
605         long endOffset;
606         
607         @Override
608         public void setDocument(int doc) {
609           startOffset = ordIndex.get(doc);
610           endOffset = ordIndex.get(doc+1L);
611         }
612 
613         @Override
614         public long valueAt(int index) {
615           return values.get(startOffset + index);
616         }
617 
618         @Override
619         public int count() {
620           return (int) (endOffset - startOffset);
621         }
622       };
623     } else {
624       throw new AssertionError();
625     }
626   }

628   @Override
629   public SortedSetDocValues getSortedSet(FieldInfo field) throws IOException {
630     SortedSetEntry ss = sortedSets.get(field.name);
631     if (ss.format == SORTED_SINGLE_VALUED) {
632       final SortedDocValues values = getSorted(field);
633       return DocValues.singleton(values);
634     } else if (ss.format != SORTED_WITH_ADDRESSES) {
635       throw new AssertionError();
636     }
637 
638     final IndexInput data = this.data.clone();
639     final long valueCount = binaries.get(field.name).count;
640     // we keep the byte[]s and list of ords on disk, these could be large
641     final LongBinaryDocValues binary = (LongBinaryDocValues) getBinary(field);
642     final LongValues ordinals = getNumeric(ords.get(field.name));
643     // but the addresses to the ord stream are in RAM
644     final MonotonicBlockPackedReader ordIndex = getOrdIndexInstance(data, field, ordIndexes.get(field.name));
645     
646     return new RandomAccessOrds() {
647       long startOffset;
648       long offset;
649       long endOffset;
650       
651       @Override
652       public long nextOrd() {
653         if (offset == endOffset) {
654           return NO_MORE_ORDS;
655         } else {
656           long ord = ordinals.get(offset);
657           offset++;
658           return ord;
659         }
660       }
661 
662       @Override
663       public void setDocument(int docID) {
664         startOffset = offset = ordIndex.get(docID);
665         endOffset = ordIndex.get(docID+1L);
666       }
667 
668       @Override
669       public BytesRef lookupOrd(long ord) {
670         return binary.get(ord);
671       }
672 
673       @Override
674       public long getValueCount() {
675         return valueCount;
676       }
677       
678       @Override
679       public long lookupTerm(BytesRef key) {
680         if (binary instanceof CompressedBinaryDocValues) {
681           return ((CompressedBinaryDocValues)binary).lookupTerm(key);
682         } else {
683           return super.lookupTerm(key);
684         }
685       }
686 
687       @Override
688       public TermsEnum termsEnum() {
689         if (binary instanceof CompressedBinaryDocValues) {
690           return ((CompressedBinaryDocValues)binary).getTermsEnum();
691         } else {
692           return super.termsEnum();
693         }
694       }
695 
696       @Override
697       public long ordAt(int index) {
698         return ordinals.get(startOffset + index);
699       }
700 
701       @Override
702       public int cardinality() {
703         return (int) (endOffset - startOffset);
704       }
705     };
706   }

708   private Bits getMissingBits(final long offset) throws IOException {
709     if (offset == -1) {
710       return new Bits.MatchAllBits(maxDoc);
711     } else {
712       int length = (int) ((maxDoc + 7L) >>> 3);
713       final RandomAccessInput in = data.randomAccessSlice(offset, length);
714       return new Bits() {
715         @Override
716         public boolean get(int index) {
717           try {
718             return (in.readByte(index >> 3) & (1 << (index & 7))) != 0;
719           } catch (IOException e) {
720             throw new RuntimeException(e);
721           }
722         }
723 
724         @Override
725         public int length() {
726           return maxDoc;
727         }
728       };
729     }
730   }

733   public Bits getDocsWithField(FieldInfo field) throws IOException {
734     switch(field.getDocValuesType()) {
735       case SORTED_SET:
736         return DocValues.docsWithValue(getSortedSet(field), maxDoc);
737       case SORTED_NUMERIC:
738         return DocValues.docsWithValue(getSortedNumeric(field), maxDoc);
739       case SORTED:
740         return DocValues.docsWithValue(getSorted(field), maxDoc);
741       case BINARY:
742         BinaryEntry be = binaries.get(field.name);
743         return getMissingBits(be.missingOffset);
744       case NUMERIC:
745         NumericEntry ne = numerics.get(field.name);
746         return getMissingBits(ne.missingOffset);
747       default:
748         throw new AssertionError();
749     }

826   // in the compressed case, we add a few additional operations for
827   // more efficient reverse lookup and enumeration
828   static class CompressedBinaryDocValues extends LongBinaryDocValues {
829     final BinaryEntry bytes;
830     final long interval;
831     final long numValues;
832     final long numIndexValues;
833     final MonotonicBlockPackedReader addresses;
834     final IndexInput data;
835     final TermsEnum termsEnum;
836     
837     public CompressedBinaryDocValues(BinaryEntry bytes, MonotonicBlockPackedReader addresses, IndexInput data) throws IOException {
838       this.bytes = bytes;
839       this.interval = bytes.addressInterval;
840       this.addresses = addresses;
841       this.data = data;
842       this.numValues = bytes.count;
843       this.numIndexValues = addresses.size();
844       this.termsEnum = getTermsEnum(data);
845     }
846     
847     @Override
848     public BytesRef get(long id) {
849       try {
850         termsEnum.seekExact(id);
851         return termsEnum.term();
852       } catch (IOException e) {
853         throw new RuntimeException(e);
854       }
855     }
856     
857     long lookupTerm(BytesRef key) {
858       try {
859         SeekStatus status = termsEnum.seekCeil(key);
860         if (status == SeekStatus.END) {
861           return -numValues-1;
862         } else if (status == SeekStatus.FOUND) {
863           return termsEnum.ord();
864         } else {
865           return -termsEnum.ord()-1;
866         }
867       } catch (IOException bogus) {
868         throw new RuntimeException(bogus);
869       }
870     }
871     
872     TermsEnum getTermsEnum() {
873       try {
874         return getTermsEnum(data.clone());
875       } catch (IOException e) {
876         throw new RuntimeException(e);
877       }
878     }
879     
880     private TermsEnum getTermsEnum(final IndexInput input) throws IOException {
881       input.seek(bytes.offset);
882       
883       return new TermsEnum() {
884         private long currentOrd = -1;
885         // TODO: maxLength is negative when all terms are merged away...
886         private final BytesRef term = new BytesRef(bytes.maxLength < 0 ? 0 : bytes.maxLength);
887 
888         @Override
889         public BytesRef next() throws IOException {
890           if (++currentOrd >= numValues) {
891             return null;
892           } else {
893             int start = input.readVInt();
894             int suffix = input.readVInt();
895             input.readBytes(term.bytes, start, suffix);
896             term.length = start + suffix;
897             return term;
898           }
899         }
900 
901         @Override
902         public SeekStatus seekCeil(BytesRef text) throws IOException {
903           // binary-search just the index values to find the block,
904           // then scan within the block
905           long low = 0;
906           long high = numIndexValues-1;
907 
908           while (low <= high) {
909             long mid = (low + high) >>> 1;
910             seekExact(mid * interval);
911             int cmp = term.compareTo(text);
912 
913             if (cmp < 0) {
914               low = mid + 1;
915             } else if (cmp > 0) {
916               high = mid - 1;
917             } else {
918               // we got lucky, found an indexed term
919               return SeekStatus.FOUND;
920             }
921           }
922           
923           if (numIndexValues == 0) {
924             return SeekStatus.END;
925           }
926           
927           // block before insertion point
928           long block = low-1;
929           seekExact(block < 0 ? -1 : block * interval);
930           
931           while (next() != null) {
932             int cmp = term.compareTo(text);
933             if (cmp == 0) {
934               return SeekStatus.FOUND;
935             } else if (cmp > 0) {
936               return SeekStatus.NOT_FOUND;
937             }
938           }
939           
940           return SeekStatus.END;
941         }
942 
943         @Override
944         public void seekExact(long ord) throws IOException {
945           long block = ord / interval;
946 
947           if (ord >= currentOrd && block == currentOrd / interval) {
948             // seek within current block
949           } else {
950             // position before start of block
951             currentOrd = ord - ord % interval - 1;
952             input.seek(bytes.offset + addresses.get(block));
953           }
954           
955           while (currentOrd < ord) {
956             next();
957           }
958         }
959 
960         @Override
961         public BytesRef term() throws IOException {
962           return term;
963         }
964 
965         @Override
966         public long ord() throws IOException {
967           return currentOrd;
968         }
969 
970         @Override
971         public int docFreq() throws IOException {
972           throw new UnsupportedOperationException();
973         }
974 
975         @Override
976         public long totalTermFreq() throws IOException {
977           return -1;
978         }
979 
980         @Override
981         public PostingsEnum postings(PostingsEnum reuse, int flags) throws IOException {
982           throw new UnsupportedOperationException();
983         }
984       };
985     }
