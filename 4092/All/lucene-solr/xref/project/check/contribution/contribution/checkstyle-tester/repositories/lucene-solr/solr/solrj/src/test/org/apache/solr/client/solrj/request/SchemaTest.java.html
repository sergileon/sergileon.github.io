64      fieldAttributes.put("name", fieldName);
65      fieldAttributes.put("type", "string");
66      fieldAttributes.put("stored", true);
67      SchemaRequest.AddField addFieldRequest = new SchemaRequest.AddField(fieldAttributes);
68      addFieldRequest.process(solrClient);

71    private static SchemaRequest.AddFieldType createFieldTypeRequest(String fieldTypeName) {
72      FieldTypeDefinition fieldTypeDefinition = new FieldTypeDefinition();
73      Map<String, Object> fieldTypeAttributes = new LinkedHashMap<>();
74      fieldTypeAttributes.put("name", fieldTypeName);
75      fieldTypeAttributes.put("class", "solr.TextField");
76      fieldTypeDefinition.setAttributes(fieldTypeAttributes);
77      AnalyzerDefinition indexAnalyzerDefinition = new AnalyzerDefinition();
78      Map<String, Object> iTokenizerAttributes = new LinkedHashMap<>();
79      iTokenizerAttributes.put("class", "solr.PathHierarchyTokenizerFactory");
80      iTokenizerAttributes.put("delimiter", "/");
81      indexAnalyzerDefinition.setTokenizer(iTokenizerAttributes);
82      fieldTypeDefinition.setIndexAnalyzer(indexAnalyzerDefinition);
83      AnalyzerDefinition queryAnalyzerDefinition = new AnalyzerDefinition();
84      Map<String, Object> qTokenizerAttributes = new LinkedHashMap<>();
85      qTokenizerAttributes.put("class", "solr.KeywordTokenizerFactory");
86      queryAnalyzerDefinition.setTokenizer(qTokenizerAttributes);
87      fieldTypeDefinition.setQueryAnalyzer(queryAnalyzerDefinition);
88      return new SchemaRequest.AddFieldType(fieldTypeDefinition);
89    }

554     String fieldName = "accuracyField";
