35  /**
36   * An implementation of {@link AbstractGroupFacetCollector} that computes grouped facets based on the indexed terms
37   * from DocValues.
38   *
39   * @lucene.experimental
40   */
41  public abstract class TermGroupFacetCollector extends AbstractGroupFacetCollector {
42  
43    final List<GroupedFacetHit> groupedFacetHits;
44    final SentinelIntSet segmentGroupedFacetHits;
45  
46    SortedDocValues groupFieldTermsIndex;
47  
48    /**
49     * Factory method for creating the right implementation based on the fact whether the facet field contains
50     * multiple tokens per documents.
51     *
52     * @param groupField The group field
53     * @param facetField The facet field
54     * @param facetFieldMultivalued Whether the facet field has multiple tokens per document
55     * @param facetPrefix The facet prefix a facet entry should start with to be included.
56     * @param initialSize The initial allocation size of the internal int set and group facet list which should roughly
57     *                    match the total number of expected unique groups. Be aware that the heap usage is
58     *                    4 bytes * initialSize.
59     * @return <code>TermGroupFacetCollector</code> implementation
60     */
61    public static TermGroupFacetCollector createTermGroupFacetCollector(String groupField,
62                                                                        String facetField,
63                                                                        boolean facetFieldMultivalued,
64                                                                        BytesRef facetPrefix,
65                                                                        int initialSize) {
66      if (facetFieldMultivalued) {
67        return new MV(groupField, facetField, facetPrefix, initialSize);
68      } else {
69        return new SV(groupField, facetField, facetPrefix, initialSize);
70      }
71    }
72  
73    TermGroupFacetCollector(String groupField, String facetField, BytesRef facetPrefix, int initialSize) {
74      super(groupField, facetField, facetPrefix);
75      groupedFacetHits = new ArrayList<>(initialSize);
76      segmentGroupedFacetHits = new SentinelIntSet(initialSize, Integer.MIN_VALUE);
77    }
78    
79    @Override
80    public boolean needsScores() {
81      return true; // TODO, maybe we don't?
82    }
83  
84    // Implementation for single valued facet fields.
85    static class SV extends TermGroupFacetCollector {
86  
87      private SortedDocValues facetFieldTermsIndex;
88  
89      SV(String groupField, String facetField, BytesRef facetPrefix, int initialSize) {
90        super(groupField, facetField, facetPrefix, initialSize);
91      }
92  
93      @Override
94      public void collect(int doc) throws IOException {
95        int facetOrd = facetFieldTermsIndex.getOrd(doc);
96        if (facetOrd < startFacetOrd || facetOrd >= endFacetOrd) {
97          return;
98        }
99  
100       int groupOrd = groupFieldTermsIndex.getOrd(doc);
101       int segmentGroupedFacetsIndex = groupOrd * (facetFieldTermsIndex.getValueCount()+1) + facetOrd;
102       if (segmentGroupedFacetHits.exists(segmentGroupedFacetsIndex)) {
103         return;
104       }
105 
106       segmentTotalCount++;
107       segmentFacetCounts[facetOrd+1]++;
108 
109       segmentGroupedFacetHits.put(segmentGroupedFacetsIndex);
110 
111       BytesRef groupKey;
112       if (groupOrd == -1) {
113         groupKey = null;
114       } else {
115         groupKey = BytesRef.deepCopyOf(groupFieldTermsIndex.lookupOrd(groupOrd));
116       }
117 
118       BytesRef facetKey;
119       if (facetOrd == -1) {
120         facetKey = null;
121       } else {
122         facetKey = BytesRef.deepCopyOf(facetFieldTermsIndex.lookupOrd(facetOrd));
123       }
124 
125       groupedFacetHits.add(new GroupedFacetHit(groupKey, facetKey));
126     }
127 
128     @Override
129     protected void doSetNextReader(LeafReaderContext context) throws IOException {
130       if (segmentFacetCounts != null) {
131         segmentResults.add(createSegmentResult());
132       }
133 
134       groupFieldTermsIndex = DocValues.getSorted(context.reader(), groupField);
135       facetFieldTermsIndex = DocValues.getSorted(context.reader(), facetField);
136 
137       // 1+ to allow for the -1 "not set":
138       segmentFacetCounts = new int[facetFieldTermsIndex.getValueCount()+1];
139       segmentTotalCount = 0;
140 
141       segmentGroupedFacetHits.clear();
142       for (GroupedFacetHit groupedFacetHit : groupedFacetHits) {
143         int facetOrd = groupedFacetHit.facetValue == null ? -1 : facetFieldTermsIndex.lookupTerm(groupedFacetHit.facetValue);
144         if (groupedFacetHit.facetValue != null && facetOrd < 0) {
145           continue;
146         }
147 
148         int groupOrd = groupedFacetHit.groupValue == null ? -1 : groupFieldTermsIndex.lookupTerm(groupedFacetHit.groupValue);
149         if (groupedFacetHit.groupValue != null && groupOrd < 0) {
150           continue;
151         }
152 
153         int segmentGroupedFacetsIndex = groupOrd * (facetFieldTermsIndex.getValueCount()+1) + facetOrd;
154         segmentGroupedFacetHits.put(segmentGroupedFacetsIndex);
155       }
156 
157       if (facetPrefix != null) {
158         startFacetOrd = facetFieldTermsIndex.lookupTerm(facetPrefix);
159         if (startFacetOrd < 0) {
160           // Points to the ord one higher than facetPrefix
161           startFacetOrd = -startFacetOrd - 1;
162         }
163         BytesRefBuilder facetEndPrefix = new BytesRefBuilder();
164         facetEndPrefix.append(facetPrefix);
165         facetEndPrefix.append(UnicodeUtil.BIG_TERM);
166         endFacetOrd = facetFieldTermsIndex.lookupTerm(facetEndPrefix.get());
167         assert endFacetOrd < 0;
168         endFacetOrd = -endFacetOrd - 1; // Points to the ord one higher than facetEndPrefix
169       } else {
170         startFacetOrd = -1;
171         endFacetOrd = facetFieldTermsIndex.getValueCount();
172       }
173     }
174 
175     @Override
176     protected SegmentResult createSegmentResult() throws IOException {
177       return new SegmentResult(segmentFacetCounts, segmentTotalCount, facetFieldTermsIndex.termsEnum(), startFacetOrd, endFacetOrd);
178     }
179 
180     private static class SegmentResult extends AbstractGroupFacetCollector.SegmentResult {
181 
182       final TermsEnum tenum;
183 
184       SegmentResult(int[] counts, int total, TermsEnum tenum, int startFacetOrd, int endFacetOrd) throws IOException {
185         super(counts, total - counts[0], counts[0], endFacetOrd+1);
186         this.tenum = tenum;
187         this.mergePos = startFacetOrd == -1 ? 1 : startFacetOrd+1;
188         if (mergePos < maxTermPos) {
189           assert tenum != null;
190           tenum.seekExact(startFacetOrd == -1 ? 0 : startFacetOrd);
191           mergeTerm = tenum.term();
192         }
193       }
194 
195       @Override
196       protected void nextTerm() throws IOException {
197         mergeTerm = tenum.next();
198       }
199     }
200   }

202   // Implementation for multi valued facet fields.
203   static class MV extends TermGroupFacetCollector {
204 
205     private SortedSetDocValues facetFieldDocTermOrds;
206     private TermsEnum facetOrdTermsEnum;
207     private int facetFieldNumTerms;
208 
209     MV(String groupField, String facetField, BytesRef facetPrefix, int initialSize) {
210       super(groupField, facetField, facetPrefix, initialSize);
211     }
212 
213     @Override
214     public void collect(int doc) throws IOException {
215       int groupOrd = groupFieldTermsIndex.getOrd(doc);
216       if (facetFieldNumTerms == 0) {
217         int segmentGroupedFacetsIndex = groupOrd * (facetFieldNumTerms + 1);
218         if (facetPrefix != null || segmentGroupedFacetHits.exists(segmentGroupedFacetsIndex)) {
219           return;
220         }
221 
222         segmentTotalCount++;
223         segmentFacetCounts[facetFieldNumTerms]++;
224 
225         segmentGroupedFacetHits.put(segmentGroupedFacetsIndex);
226         BytesRef groupKey;
227         if (groupOrd == -1) {
228           groupKey = null;
229         } else {
230           groupKey = BytesRef.deepCopyOf(groupFieldTermsIndex.lookupOrd(groupOrd));
231         }
232         groupedFacetHits.add(new GroupedFacetHit(groupKey, null));
233         return;
234       }
235 
236       facetFieldDocTermOrds.setDocument(doc);
237       long ord;
238       boolean empty = true;
239       while ((ord = facetFieldDocTermOrds.nextOrd()) != SortedSetDocValues.NO_MORE_ORDS) {
240         process(groupOrd, (int) ord);
241         empty = false;
242       }
243       
244       if (empty) {
245         process(groupOrd, facetFieldNumTerms); // this facet ord is reserved for docs not containing facet field.
246       }
247     }
248     
249     private void process(int groupOrd, int facetOrd) {
250       if (facetOrd < startFacetOrd || facetOrd >= endFacetOrd) {
251         return;
252       }
253 
254       int segmentGroupedFacetsIndex = groupOrd * (facetFieldNumTerms + 1) + facetOrd;
255       if (segmentGroupedFacetHits.exists(segmentGroupedFacetsIndex)) {
256         return;
257       }
258 
259       segmentTotalCount++;
260       segmentFacetCounts[facetOrd]++;
261 
262       segmentGroupedFacetHits.put(segmentGroupedFacetsIndex);
263 
264       BytesRef groupKey;
265       if (groupOrd == -1) {
266         groupKey = null;
267       } else {
268         groupKey = BytesRef.deepCopyOf(groupFieldTermsIndex.lookupOrd(groupOrd));
269       }
270 
271       final BytesRef facetValue;
272       if (facetOrd == facetFieldNumTerms) {
273         facetValue = null;
274       } else {
275         facetValue = BytesRef.deepCopyOf(facetFieldDocTermOrds.lookupOrd(facetOrd));
276       }
277       groupedFacetHits.add(new GroupedFacetHit(groupKey, facetValue));
278     }
279 
280     @Override
281     protected void doSetNextReader(LeafReaderContext context) throws IOException {
282       if (segmentFacetCounts != null) {
283         segmentResults.add(createSegmentResult());
284       }
285 
286       groupFieldTermsIndex = DocValues.getSorted(context.reader(), groupField);
287       facetFieldDocTermOrds = DocValues.getSortedSet(context.reader(), facetField);
288       facetFieldNumTerms = (int) facetFieldDocTermOrds.getValueCount();
289       if (facetFieldNumTerms == 0) {
290         facetOrdTermsEnum = null;
291       } else {
292         facetOrdTermsEnum = facetFieldDocTermOrds.termsEnum();
293       }
294       // [facetFieldNumTerms() + 1] for all possible facet values and docs not containing facet field
295       segmentFacetCounts = new int[facetFieldNumTerms + 1];
296       segmentTotalCount = 0;
297 
298       segmentGroupedFacetHits.clear();
299       for (GroupedFacetHit groupedFacetHit : groupedFacetHits) {
300         int groupOrd = groupedFacetHit.groupValue == null ? -1 : groupFieldTermsIndex.lookupTerm(groupedFacetHit.groupValue);
301         if (groupedFacetHit.groupValue != null && groupOrd < 0) {
302           continue;
303         }
304 
305         int facetOrd;
306         if (groupedFacetHit.facetValue != null) {
307           if (facetOrdTermsEnum == null || !facetOrdTermsEnum.seekExact(groupedFacetHit.facetValue)) {
308             continue;
309           }
310           facetOrd = (int) facetOrdTermsEnum.ord();
311         } else {
312           facetOrd = facetFieldNumTerms;
313         }
314 
315         // (facetFieldDocTermOrds.numTerms() + 1) for all possible facet values and docs not containing facet field
316         int segmentGroupedFacetsIndex = groupOrd * (facetFieldNumTerms + 1) + facetOrd;
317         segmentGroupedFacetHits.put(segmentGroupedFacetsIndex);
318       }
319 
320       if (facetPrefix != null) {
321         TermsEnum.SeekStatus seekStatus;
322         if (facetOrdTermsEnum != null) {
323           seekStatus = facetOrdTermsEnum.seekCeil(facetPrefix);
324         } else {
325           seekStatus = TermsEnum.SeekStatus.END;
326         }
327 
328         if (seekStatus != TermsEnum.SeekStatus.END) {
329           startFacetOrd = (int) facetOrdTermsEnum.ord();
330         } else {
331           startFacetOrd = 0;
332           endFacetOrd = 0;
333           return;
334         }
335 
336         BytesRefBuilder facetEndPrefix = new BytesRefBuilder();
337         facetEndPrefix.append(facetPrefix);
338         facetEndPrefix.append(UnicodeUtil.BIG_TERM);
339         seekStatus = facetOrdTermsEnum.seekCeil(facetEndPrefix.get());
340         if (seekStatus != TermsEnum.SeekStatus.END) {
341           endFacetOrd = (int) facetOrdTermsEnum.ord();
342         } else {
343           endFacetOrd = facetFieldNumTerms; // Don't include null...
344         }
345       } else {
346         startFacetOrd = 0;
347         endFacetOrd = facetFieldNumTerms + 1;
348       }
349     }
350 
351     @Override
352     protected SegmentResult createSegmentResult() throws IOException {
353       return new SegmentResult(segmentFacetCounts, segmentTotalCount, facetFieldNumTerms, facetOrdTermsEnum, startFacetOrd, endFacetOrd);
354     }
355 
356     private static class SegmentResult extends AbstractGroupFacetCollector.SegmentResult {
357 
358       final TermsEnum tenum;
359 
360       SegmentResult(int[] counts, int total, int missingCountIndex, TermsEnum tenum, int startFacetOrd, int endFacetOrd) throws IOException {
361         super(counts, total - counts[missingCountIndex], counts[missingCountIndex],
362             endFacetOrd == missingCountIndex + 1 ?  missingCountIndex : endFacetOrd);
363         this.tenum = tenum;
364         this.mergePos = startFacetOrd;
365         if (tenum != null) {
366           tenum.seekExact(mergePos);
367           mergeTerm = tenum.term();
368         }
369       }
370 
371       @Override
372       protected void nextTerm() throws IOException {
373         mergeTerm = tenum.next();
374       }
375     }
376   }
377 }

379 class GroupedFacetHit {
380 
381   final BytesRef groupValue;
382   final BytesRef facetValue;
383 
384   GroupedFacetHit(BytesRef groupValue, BytesRef facetValue) {
385     this.groupValue = groupValue;
386     this.facetValue = facetValue;
387   }
