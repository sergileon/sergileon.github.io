32  /** Handles intersection of a shape with a BKD tree previously written with {@link BKDTreeWriter}.
33   *
34   * @lucene.experimental */
35  
36  final class BKDTreeReader implements Accountable {
37    final private int[] splitValues;
38    final private int leafNodeOffset;
39    final private long[] leafBlockFPs;
40    final int maxDoc;
41    final IndexInput in;
42  
43    enum Relation {CELL_INSIDE_SHAPE, SHAPE_CROSSES_CELL, SHAPE_OUTSIDE_CELL};
44  
45    interface LatLonFilter {
46      // TODO: move DVs/encoding out on top: this method should just take a docID
47      boolean accept(double lat, double lon);
48      // TODO: move DVs/encoding out on top: this method should take ints and do its own decode
49      Relation compare(double latMin, double latMax, double lonMin, double lonMax);
50    }
51  
52    public BKDTreeReader(IndexInput in, int maxDoc) throws IOException {
53  
54      // Read index:
55      int numLeaves = in.readVInt();
56      leafNodeOffset = numLeaves;
57  
58      // Tree is fully balanced binary tree, so number of nodes = numLeaves-1, except our nodeIDs are 1-based (splitValues[0] is unused):
59      splitValues = new int[numLeaves];
60      for(int i=0;i<numLeaves;i++) {
61        splitValues[i] = in.readInt();
62      }
63      leafBlockFPs = new long[numLeaves];
64      for(int i=0;i<numLeaves;i++) {
65        leafBlockFPs[i] = in.readVLong();
66      }
67  
68      this.maxDoc = maxDoc;
69      this.in = in;
70    }
71  
72    private static final class QueryState {
73      final IndexInput in;
74      byte[] scratch = new byte[16];
75      final ByteArrayDataInput scratchReader = new ByteArrayDataInput(scratch);
76      final DocIdSetBuilder docs;
77      final int latMinEnc;
78      final int latMaxEnc;
79      final int lonMinEnc;
80      final int lonMaxEnc;
81      final LatLonFilter latLonFilter;
82      final SortedNumericDocValues sndv;
83  
84      public QueryState(IndexInput in, int maxDoc,
85                        int latMinEnc, int latMaxEnc,
86                        int lonMinEnc, int lonMaxEnc,
87                        LatLonFilter latLonFilter,
88                        SortedNumericDocValues sndv) {
89        this.in = in;
90        this.docs = new DocIdSetBuilder(maxDoc);
91        this.latMinEnc = latMinEnc;
92        this.latMaxEnc = latMaxEnc;
93        this.lonMinEnc = lonMinEnc;
94        this.lonMaxEnc = lonMaxEnc;
95        this.latLonFilter = latLonFilter;
96        this.sndv = sndv;
97      }
98    }
99  
100   // TODO: move DVs/encoding out on top: this method should take ints, and encode should be done up above
101   public DocIdSet intersect(double latMin, double latMax, double lonMin, double lonMax, LatLonFilter filter, SortedNumericDocValues sndv) throws IOException {
102     if (BKDTreeWriter.validLat(latMin) == false) {
103       throw new IllegalArgumentException("invalid latMin: " + latMin);
104     }
105     if (BKDTreeWriter.validLat(latMax) == false) {
106       throw new IllegalArgumentException("invalid latMax: " + latMax);
107     }
108     if (BKDTreeWriter.validLon(lonMin) == false) {
109       throw new IllegalArgumentException("invalid lonMin: " + lonMin);
110     }
111     if (BKDTreeWriter.validLon(lonMax) == false) {
112       throw new IllegalArgumentException("invalid lonMax: " + lonMax);
113     }
114 
115     int latMinEnc = BKDTreeWriter.encodeLat(latMin);
116     int latMaxEnc = BKDTreeWriter.encodeLat(latMax);
117     int lonMinEnc = BKDTreeWriter.encodeLon(lonMin);
118     int lonMaxEnc = BKDTreeWriter.encodeLon(lonMax);
119 
120     QueryState state = new QueryState(in.clone(), maxDoc,
121                                       latMinEnc, latMaxEnc,
122                                       lonMinEnc, lonMaxEnc,
123                                       filter,
124                                       sndv);
125 
126     int hitCount = intersect(state, 1,
127                              BKDTreeWriter.encodeLat(-90.0),
128                              BKDTreeWriter.encodeLat(Math.nextAfter(90.0, Double.POSITIVE_INFINITY)),
129                              BKDTreeWriter.encodeLon(-180.0),
130                              BKDTreeWriter.encodeLon(Math.nextAfter(180.0, Double.POSITIVE_INFINITY)));
131 
132     // NOTE: hitCount is an over-estimate in the multi-valued case:
133     return state.docs.build(hitCount);
134   }

136   /** Fast path: this is called when the query rect fully encompasses all cells under this node. */
137   private int addAll(QueryState state, int nodeID) throws IOException {
138 
139     //long latRange = (long) cellLatMaxEnc - (long) cellLatMinEnc;
140     //long lonRange = (long) cellLonMaxEnc - (long) cellLonMinEnc;
141 
142     if (nodeID >= leafNodeOffset) {
143 
144       /*
145       System.out.println("A: " + BKDTreeWriter.decodeLat(cellLatMinEnc)
146                          + " " + BKDTreeWriter.decodeLat(cellLatMaxEnc)
147                          + " " + BKDTreeWriter.decodeLon(cellLonMinEnc)
148                          + " " + BKDTreeWriter.decodeLon(cellLonMaxEnc));
149       */
150 
151       // Leaf node
152       long fp = leafBlockFPs[nodeID-leafNodeOffset];
153       //System.out.println("    leaf nodeID=" + nodeID + " vs leafNodeOffset=" + leafNodeOffset + " fp=" + fp);
154       if (fp == 0) {
155         // Dead end node (adversary case):
156         return 0;
157       }
158       state.in.seek(fp);
159 
160       //System.out.println("    seek to leafFP=" + fp);
161       // How many points are stored in this leaf cell:
162       int count = state.in.readVInt();
163       state.docs.grow(count);
164       for(int i=0;i<count;i++) {
165         int docID = state.in.readInt();
166         state.docs.add(docID);
167       }
168 
169       return count;
170     } else {
171       int splitValue = splitValues[nodeID];
172 
173       if (splitValue == Integer.MAX_VALUE) {
174         // Dead end node (adversary case):
175         return 0;
176       }
177 
178       //System.out.println("  splitValue=" + splitValue);
179 
180       //System.out.println("  addAll: inner");
181       int count = addAll(state, 2*nodeID);
182       count += addAll(state, 2*nodeID+1);
183       //System.out.println("  addAll: return count=" + count);
184       return count;
185     }
186   }

188   private int intersect(QueryState state,
189                         int nodeID,
190                         int cellLatMinEnc, int cellLatMaxEnc, int cellLonMinEnc, int cellLonMaxEnc)
191     throws IOException {
192 
193     // 2.06 sec -> 1.52 sec for 225 OSM London queries:
194     if (state.latLonFilter != null) {
195 
196       // Only call the filter when the current cell does not fully contain the bbox:
197       if (cellLatMinEnc > state.latMinEnc || cellLatMaxEnc < state.latMaxEnc ||
198           cellLonMinEnc > state.lonMinEnc || cellLonMaxEnc < state.lonMaxEnc) {
199 
200         Relation r = state.latLonFilter.compare(BKDTreeWriter.decodeLat(cellLatMinEnc),
201                                                 BKDTreeWriter.decodeLat(cellLatMaxEnc),
202                                                 BKDTreeWriter.decodeLon(cellLonMinEnc),
203                                                 BKDTreeWriter.decodeLon(cellLonMaxEnc));
204         // System.out.println("BKD.intersect cellLat=" + BKDTreeWriter.decodeLat(cellLatMinEnc) + " TO " + BKDTreeWriter.decodeLat(cellLatMaxEnc) + ", cellLon=" + BKDTreeWriter.decodeLon(cellLonMinEnc) + " TO " + BKDTreeWriter.decodeLon(cellLonMaxEnc) + " compare=" + r);
205         if (r == Relation.SHAPE_OUTSIDE_CELL) {
206           // This cell is fully outside of the query shape: stop recursing
207           return 0;
208         } else if (r == Relation.CELL_INSIDE_SHAPE) {
209           // This cell is fully inside of the query shape: recursively add all points in this cell without filtering
210           return addAll(state, nodeID);
211         } else {
212           // The cell crosses the shape boundary, so we fall through and do full filtering
213         }
214       }
215     // TODO: clean this up: the bbox case should also just be a filter, and we should assert filter != null at the start
216     } else if (state.latMinEnc <= cellLatMinEnc && state.latMaxEnc >= cellLatMaxEnc && state.lonMinEnc <= cellLonMinEnc && state.lonMaxEnc >= cellLonMaxEnc) {
217       // Bbox query: optimize the case when the query fully contains this cell: we can
218       // recursively add all points without checking if they match the query:
219       return addAll(state, nodeID);
220     }
221 
222     long latRange = (long) cellLatMaxEnc - (long) cellLatMinEnc;
223     long lonRange = (long) cellLonMaxEnc - (long) cellLonMinEnc;
224 
225     int dim;
226     if (latRange >= lonRange) {
227       dim = 0;
228     } else {
229       dim = 1;
230     }
231 
232     //System.out.println("\nintersect node=" + nodeID + " vs " + leafNodeOffset);
233 
234     if (nodeID >= leafNodeOffset) {
235       // Leaf node; scan and filter all points in this block:
236       //System.out.println("    intersect leaf nodeID=" + nodeID + " vs leafNodeOffset=" + leafNodeOffset + " fp=" + leafBlockFPs[nodeID-leafNodeOffset]);
237       int hitCount = 0;
238 
239       long fp = leafBlockFPs[nodeID-leafNodeOffset];
240       if (fp == 0) {
241         // Dead end node (adversary case):
242         //System.out.println("    dead-end leaf");
243         return 0;
244       }
245 
246       /*
247       System.out.println("I: " + BKDTreeWriter.decodeLat(cellLatMinEnc)
248                          + " " + BKDTreeWriter.decodeLat(cellLatMaxEnc)
249                          + " " + BKDTreeWriter.decodeLon(cellLonMinEnc)
250                          + " " + BKDTreeWriter.decodeLon(cellLonMaxEnc));
251       */
252 
253       state.in.seek(fp);
254 
255       // How many points are stored in this leaf cell:
256       int count = state.in.readVInt();
257 
258       state.docs.grow(count);
259       for(int i=0;i<count;i++) {
260         int docID = state.in.readInt();
261         state.sndv.setDocument(docID);
262         // How many values this doc has:
263         int docValueCount = state.sndv.count();
264         for(int j=0;j<docValueCount;j++) {
265           long enc = state.sndv.valueAt(j);
266 
267           int latEnc = (int) ((enc>>32) & 0xffffffffL);
268           int lonEnc = (int) (enc & 0xffffffffL);
269 
270           if (latEnc >= state.latMinEnc &&
271               latEnc < state.latMaxEnc &&
272               lonEnc >= state.lonMinEnc &&
273               lonEnc < state.lonMaxEnc &&
274               (state.latLonFilter == null ||
275                state.latLonFilter.accept(BKDTreeWriter.decodeLat(latEnc), BKDTreeWriter.decodeLon(lonEnc)))) {
276             state.docs.add(docID);
277             hitCount++;
278 
279             // Stop processing values for this doc:
280             break;
281           }
282         }
283       }
284 
285       return hitCount;
286 
287     } else {
288 
289       int splitValue = splitValues[nodeID];
290 
291       if (splitValue == Integer.MAX_VALUE) {
292         // Dead end node (adversary case):
293         //System.out.println("    dead-end sub-tree");
294         return 0;
295       }
296 
297       //System.out.println("  splitValue=" + splitValue);
298 
299       int count = 0;
300 
301       if (dim == 0) {
302 
303         //System.out.println("  split on lat=" + splitValue);
304 
305         // Inner node split on lat:
306 
307         // Left node:
308         if (state.latMinEnc < splitValue) {
309           //System.out.println("  recurse left");
310           count += intersect(state,
311                              2*nodeID,
312                              cellLatMinEnc, splitValue, cellLonMinEnc, cellLonMaxEnc);
313         }
314 
315         // Right node:
316         if (state.latMaxEnc >= splitValue) {
317           //System.out.println("  recurse right");
318           count += intersect(state,
319                              2*nodeID+1,
320                              splitValue, cellLatMaxEnc, cellLonMinEnc, cellLonMaxEnc);
321         }
322 
323       } else {
324         // Inner node split on lon:
325         assert dim == 1;
326 
327         // System.out.println("  split on lon=" + splitValue);
328 
329         // Left node:
330         if (state.lonMinEnc < splitValue) {
331           // System.out.println("  recurse left");
332           count += intersect(state,
333                              2*nodeID,
334                              cellLatMinEnc, cellLatMaxEnc, cellLonMinEnc, splitValue);
335         }
336 
337         // Right node:
338         if (state.lonMaxEnc >= splitValue) {
339           // System.out.println("  recurse right");
340           count += intersect(state,
341                              2*nodeID+1,
342                              cellLatMinEnc, cellLatMaxEnc, splitValue, cellLonMaxEnc);
343         }
344       }
345 
346       return count;
347     }
348   }

350   @Override
351   public long ramBytesUsed() {
352     return splitValues.length * RamUsageEstimator.NUM_BYTES_INT +
353       leafBlockFPs.length * RamUsageEstimator.NUM_BYTES_LONG;
354   }

358     return Collections.emptyList();
