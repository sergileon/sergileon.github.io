546   static List<List<String>> buildShardUrls(List<Object> urls, Integer numShards) {
547     if (urls == null) return null;
548     List<List<String>> shardUrls = new ArrayList<>(urls.size());
549     List<String> list = null;
550     
551     int sz;
552     if (numShards == null) {
553       numShards = urls.size();
554     }
555     sz = (int) Math.ceil(urls.size() / (float)numShards);
556     for (int i = 0; i < urls.size(); i++) {
557       if (i % sz == 0) {
558         list = new ArrayList<>();
559         shardUrls.add(list);
560       }
561       list.add((String) urls.get(i));
562     }
563 
564     return shardUrls;
565   }

611   /** API for Java clients; visible for testing; may become a public API eventually */
612   int run(Options options) throws Exception {
613     if (getConf().getBoolean("isMR1", false) && "local".equals(getConf().get("mapred.job.tracker"))) {
614       throw new IllegalStateException(
615         "Running with LocalJobRunner (i.e. all of Hadoop inside a single JVM) is not supported " +
616         "because LocalJobRunner does not (yet) implement the Hadoop Distributed Cache feature, " +
617         "which is required for passing files via --files and --libjars");
618     }
619 
620     long programStartTime = System.nanoTime();
621     if (options.fairSchedulerPool != null) {
622       getConf().set("mapred.fairscheduler.pool", options.fairSchedulerPool);
623     }
624     getConf().setInt(SolrOutputFormat.SOLR_RECORD_WRITER_MAX_SEGMENTS, options.maxSegments);
625     
626     // switch off a false warning about allegedly not implementing Tool
627     // also see http://hadoop.6.n7.nabble.com/GenericOptionsParser-warning-td8103.html
628     // also see https://issues.apache.org/jira/browse/HADOOP-8183
629     getConf().setBoolean("mapred.used.genericoptionsparser", true);
630     
631     if (options.log4jConfigFile != null) {
632       Utils.setLogConfigFile(options.log4jConfigFile, getConf());
633       addDistributedCacheFile(options.log4jConfigFile, getConf());
634     }
635 
636     job = Job.getInstance(getConf());
637     job.setJarByClass(getClass());
638 
639     if (options.morphlineFile == null) {
640       throw new ArgumentParserException("Argument --morphline-file is required", null);
641     }
642     verifyGoLiveArgs(options, null);
643     verifyZKStructure(options, null);
644     
645     int mappers = new JobClient(job.getConfiguration()).getClusterStatus().getMaxMapTasks(); // MR1
646     //int mappers = job.getCluster().getClusterStatus().getMapSlotCapacity(); // Yarn only
647     LOG.info("Cluster reports {} mapper slots", mappers);
648     
649     if (options.mappers == -1) { 
650       mappers = 8 * mappers; // better accomodate stragglers
651     } else {
652       mappers = options.mappers;
653     }
654     if (mappers <= 0) {
655       throw new IllegalStateException("Illegal number of mappers: " + mappers);
656     }
657     options.mappers = mappers;
658     
659     FileSystem fs = options.outputDir.getFileSystem(job.getConfiguration());
660     if (fs.exists(options.outputDir) && !delete(options.outputDir, true, fs)) {
661       return -1;
662     }
663     Path outputResultsDir = new Path(options.outputDir, RESULTS_DIR);
664     Path outputReduceDir = new Path(options.outputDir, "reducers");
665     Path outputStep1Dir = new Path(options.outputDir, "tmp1");    
666     Path outputStep2Dir = new Path(options.outputDir, "tmp2");    
667     Path outputTreeMergeStep = new Path(options.outputDir, "mtree-merge-output");
668     Path fullInputList = new Path(outputStep1Dir, FULL_INPUT_LIST);
669     
670     LOG.debug("Creating list of input files for mappers: {}", fullInputList);
671     long numFiles = addInputFiles(options.inputFiles, options.inputLists, fullInputList, job.getConfiguration());
672     if (numFiles == 0) {
673       LOG.info("No input files found - nothing to process");
674       return 0;
675     }
676     int numLinesPerSplit = (int) ceilDivide(numFiles, mappers);
677     if (numLinesPerSplit < 0) { // numeric overflow from downcasting long to int?
678       numLinesPerSplit = Integer.MAX_VALUE;
679     }
680     numLinesPerSplit = Math.max(1, numLinesPerSplit);
681 
682     int realMappers = Math.min(mappers, (int) ceilDivide(numFiles, numLinesPerSplit));
683     calculateNumReducers(options, realMappers);
684     int reducers = options.reducers;
685     LOG.info("Using these parameters: " +
686         "numFiles: {}, mappers: {}, realMappers: {}, reducers: {}, shards: {}, fanout: {}, maxSegments: {}",
687         new Object[] {numFiles, mappers, realMappers, reducers, options.shards, options.fanout, options.maxSegments});
688         
689     
690     LOG.info("Randomizing list of {} input files to spread indexing load more evenly among mappers", numFiles);
691     long startTime = System.nanoTime();      
692     if (numFiles < job.getConfiguration().getInt(MAIN_MEMORY_RANDOMIZATION_THRESHOLD, 100001)) {
693       // If there are few input files reduce latency by directly running main memory randomization 
694       // instead of launching a high latency MapReduce job
695       randomizeFewInputFiles(fs, outputStep2Dir, fullInputList);
696     } else {
697       // Randomize using a MapReduce job. Use sequential algorithm below a certain threshold because there's no
698       // benefit in using many parallel mapper tasks just to randomize the order of a few lines each
699       int numLinesPerRandomizerSplit = Math.max(10 * 1000 * 1000, numLinesPerSplit);
700       Job randomizerJob = randomizeManyInputFiles(getConf(), fullInputList, outputStep2Dir, numLinesPerRandomizerSplit);
701       if (!waitForCompletion(randomizerJob, options.isVerbose)) {
702         return -1; // job failed
703       }
704     }
705     float secs = (System.nanoTime() - startTime) / (float)(10^9);
706     LOG.info("Done. Randomizing list of {} input files took {} secs", numFiles, secs);
707     
708     
709     job.setInputFormatClass(NLineInputFormat.class);
710     NLineInputFormat.addInputPath(job, outputStep2Dir);
711     NLineInputFormat.setNumLinesPerSplit(job, numLinesPerSplit);    
712     FileOutputFormat.setOutputPath(job, outputReduceDir);
713     
714     String mapperClass = job.getConfiguration().get(JobContext.MAP_CLASS_ATTR);
715     if (mapperClass == null) { // enable customization
716       Class clazz = MorphlineMapper.class;
717       mapperClass = clazz.getName();
718       job.setMapperClass(clazz);
719     }
720     job.setJobName(getClass().getName() + "/" + Utils.getShortClassName(mapperClass));
721     
722     if (job.getConfiguration().get(JobContext.REDUCE_CLASS_ATTR) == null) { // enable customization
723       job.setReducerClass(SolrReducer.class);
724     }
725     if (options.updateConflictResolver == null) {
726       throw new IllegalArgumentException("updateConflictResolver must not be null");
727     }
728     job.getConfiguration().set(SolrReducer.UPDATE_CONFLICT_RESOLVER, options.updateConflictResolver);
729     
730     if (options.zkHost != null) {
731       assert options.collection != null;
732       /*
733        * MapReduce partitioner that partitions the Mapper output such that each
734        * SolrInputDocument gets sent to the SolrCloud shard that it would have
735        * been sent to if the document were ingested via the standard SolrCloud
736        * Near Real Time (NRT) API.
737        * 
738        * In other words, this class implements the same partitioning semantics
739        * as the standard SolrCloud NRT API. This enables to mix batch updates
740        * from MapReduce ingestion with updates from standard NRT ingestion on
741        * the same SolrCloud cluster, using identical unique document keys.
742        */
743       if (job.getConfiguration().get(JobContext.PARTITIONER_CLASS_ATTR) == null) { // enable customization
744         job.setPartitionerClass(SolrCloudPartitioner.class);
745       }
746       job.getConfiguration().set(SolrCloudPartitioner.ZKHOST, options.zkHost);
747       job.getConfiguration().set(SolrCloudPartitioner.COLLECTION, options.collection);
748     }
749     job.getConfiguration().setInt(SolrCloudPartitioner.SHARDS, options.shards);
750 
751     job.setOutputFormatClass(SolrOutputFormat.class);
752     if (options.solrHomeDir != null) {
753       SolrOutputFormat.setupSolrHomeCache(options.solrHomeDir, job);
754     } else {
755       assert options.zkHost != null;
756       // use the config that this collection uses for the SolrHomeCache.
757       ZooKeeperInspector zki = new ZooKeeperInspector();
758       SolrZkClient zkClient = zki.getZkClient(options.zkHost);
759       try {
760         String configName = zki.readConfigName(zkClient, options.collection);
761         File tmpSolrHomeDir = zki.downloadConfigDir(zkClient, configName);
762         SolrOutputFormat.setupSolrHomeCache(tmpSolrHomeDir, job);
763         options.solrHomeDir = tmpSolrHomeDir;
764       } finally {
765         zkClient.close();
766       }
767     }
768     
769     MorphlineMapRunner runner = setupMorphline(options);
770     if (options.isDryRun && runner != null) {
771       LOG.info("Indexing {} files in dryrun mode", numFiles);
772       startTime = System.nanoTime();
773       dryRun(runner, fs, fullInputList);
774       secs = (System.nanoTime() - startTime) / (float)(10^9);
775       LOG.info("Done. Indexing {} files in dryrun mode took {} secs", numFiles, secs);
776       goodbye(null, programStartTime);
777       return 0;
778     }          
779     job.getConfiguration().set(MorphlineMapRunner.MORPHLINE_FILE_PARAM, options.morphlineFile.getName());
780 
781     job.setNumReduceTasks(reducers);  
782     job.setOutputKeyClass(Text.class);
783     job.setOutputValueClass(SolrInputDocumentWritable.class);
784     LOG.info("Indexing {} files using {} real mappers into {} reducers", new Object[] {numFiles, realMappers, reducers});
785     startTime = System.nanoTime();
786     if (!waitForCompletion(job, options.isVerbose)) {
787       return -1; // job failed
788     }
789 
790     secs = (System.nanoTime() - startTime) / (float)(10^9);
791     LOG.info("Done. Indexing {} files using {} real mappers into {} reducers took {} secs", new Object[] {numFiles, realMappers, reducers, secs});
792 
793     int mtreeMergeIterations = 0;
794     if (reducers > options.shards) {
795       mtreeMergeIterations = (int) Math.round(log(options.fanout, reducers / options.shards));
796     }
797     LOG.debug("MTree merge iterations to do: {}", mtreeMergeIterations);
798     int mtreeMergeIteration = 1;
799     while (reducers > options.shards) { // run a mtree merge iteration
800       job = Job.getInstance(getConf());
801       job.setJarByClass(getClass());
802       job.setJobName(getClass().getName() + "/" + Utils.getShortClassName(TreeMergeMapper.class));
803       job.setMapperClass(TreeMergeMapper.class);
804       job.setOutputFormatClass(TreeMergeOutputFormat.class);
805       job.setNumReduceTasks(0);  
806       job.setOutputKeyClass(Text.class);
807       job.setOutputValueClass(NullWritable.class);    
808       job.setInputFormatClass(NLineInputFormat.class);
809       
810       Path inputStepDir = new Path(options.outputDir, "mtree-merge-input-iteration" + mtreeMergeIteration);
811       fullInputList = new Path(inputStepDir, FULL_INPUT_LIST);    
812       LOG.debug("MTree merge iteration {}/{}: Creating input list file for mappers {}", new Object[] {mtreeMergeIteration, mtreeMergeIterations, fullInputList});
813       numFiles = createTreeMergeInputDirList(outputReduceDir, fs, fullInputList);    
814       if (numFiles != reducers) {
815         throw new IllegalStateException("Not same reducers: " + reducers + ", numFiles: " + numFiles);
816       }
817       NLineInputFormat.addInputPath(job, fullInputList);
818       NLineInputFormat.setNumLinesPerSplit(job, options.fanout);    
819       FileOutputFormat.setOutputPath(job, outputTreeMergeStep);
820       
821       LOG.info("MTree merge iteration {}/{}: Merging {} shards into {} shards using fanout {}", new Object[] { 
822           mtreeMergeIteration, mtreeMergeIterations, reducers, (reducers / options.fanout), options.fanout});
823       startTime = System.nanoTime();
824       if (!waitForCompletion(job, options.isVerbose)) {
825         return -1; // job failed
826       }
827       if (!renameTreeMergeShardDirs(outputTreeMergeStep, job, fs)) {
828         return -1;
829       }
830       secs = (System.nanoTime() - startTime) / (float)(10^9);
831       LOG.info("MTree merge iteration {}/{}: Done. Merging {} shards into {} shards using fanout {} took {} secs",
832           new Object[] {mtreeMergeIteration, mtreeMergeIterations, reducers, (reducers / options.fanout), options.fanout, secs});
833       
834       if (!delete(outputReduceDir, true, fs)) {
835         return -1;
836       }
837       if (!rename(outputTreeMergeStep, outputReduceDir, fs)) {
838         return -1;
839       }
840       assert reducers % options.fanout == 0;
841       reducers = reducers / options.fanout;
842       mtreeMergeIteration++;
843     }
844     assert reducers == options.shards;
845     
846     // normalize output shard dir prefix, i.e.
847     // rename part-r-00000 to part-00000 (stems from zero tree merge iterations)
848     // rename part-m-00000 to part-00000 (stems from > 0 tree merge iterations)
849     for (FileStatus stats : fs.listStatus(outputReduceDir)) {
850       String dirPrefix = SolrOutputFormat.getOutputName(job);
851       Path srcPath = stats.getPath();
852       if (stats.isDirectory() && srcPath.getName().startsWith(dirPrefix)) {
853         String dstName = dirPrefix + srcPath.getName().substring(dirPrefix.length() + "-m".length());
854         Path dstPath = new Path(srcPath.getParent(), dstName);
855         if (!rename(srcPath, dstPath, fs)) {
856           return -1;
857         }        
858       }
859     };    
860     
861     // publish results dir    
862     if (!rename(outputReduceDir, outputResultsDir, fs)) {
863       return -1;
864     }
865 
866     if (options.goLive && !new GoLive().goLive(options, listSortedOutputShardDirs(outputResultsDir, fs))) {
867       return -1;
868     }
869     
870     goodbye(job, programStartTime);    
871     return 0;
872   }

874   private void calculateNumReducers(Options options, int realMappers) throws IOException {
875     if (options.shards <= 0) {
876       throw new IllegalStateException("Illegal number of shards: " + options.shards);
877     }
878     if (options.fanout <= 1) {
879       throw new IllegalStateException("Illegal fanout: " + options.fanout);
880     }
881     if (realMappers <= 0) {
882       throw new IllegalStateException("Illegal realMappers: " + realMappers);
883     }
884     
885 
886     int reducers = new JobClient(job.getConfiguration()).getClusterStatus().getMaxReduceTasks(); // MR1
887     //reducers = job.getCluster().getClusterStatus().getReduceSlotCapacity(); // Yarn only      
888     LOG.info("Cluster reports {} reduce slots", reducers);
889 
890     if (options.reducers == -2) {
891       reducers = options.shards;
892     } else if (options.reducers == -1) {
893       reducers = Math.min(reducers, realMappers); // no need to use many reducers when using few mappers
894     } else {
895       if (options.reducers == 0) {
896         throw new IllegalStateException("Illegal zero reducers");
897       }
898       reducers = options.reducers;
899     }
900     reducers = Math.max(reducers, options.shards);
901     
902     if (reducers != options.shards) {
903       // Ensure fanout isn't misconfigured. fanout can't meaningfully be larger than what would be 
904       // required to merge all leaf shards in one single tree merge iteration into root shards
905       options.fanout = Math.min(options.fanout, (int) ceilDivide(reducers, options.shards));
906       
907       // Ensure invariant reducers == options.shards * (fanout ^ N) where N is an integer >= 1.
908       // N is the number of mtree merge iterations.
909       // This helps to evenly spread docs among root shards and simplifies the impl of the mtree merge algorithm.
910       int s = options.shards;
911       while (s < reducers) { 
912         s = s * options.fanout;
913       }
914       reducers = s;
915       assert reducers % options.fanout == 0;
916     }
917     options.reducers = reducers;
918   }

920   private long addInputFiles(List<Path> inputFiles, List<Path> inputLists, Path fullInputList, Configuration conf) 
921       throws IOException {
922     
923     long numFiles = 0;
924     FileSystem fs = fullInputList.getFileSystem(conf);
925     FSDataOutputStream out = fs.create(fullInputList);
926     try {
927       Writer writer = new BufferedWriter(new OutputStreamWriter(out, StandardCharsets.UTF_8));
928       
929       for (Path inputFile : inputFiles) {
930         FileSystem inputFileFs = inputFile.getFileSystem(conf);
931         if (inputFileFs.exists(inputFile)) {
932           PathFilter pathFilter = new PathFilter() {      
933             @Override
934             public boolean accept(Path path) { // ignore "hidden" files and dirs
935               return !(path.getName().startsWith(".") || path.getName().startsWith("_")); 
936             }
937           };
938           numFiles += addInputFilesRecursively(inputFile, writer, inputFileFs, pathFilter);
939         }
940       }
941 
942       for (Path inputList : inputLists) {
943         InputStream in;
944         if (inputList.toString().equals("-")) {
945           in = System.in;
946         } else if (inputList.isAbsoluteAndSchemeAuthorityNull()) {
947           in = new BufferedInputStream(new FileInputStream(inputList.toString()));
948         } else {
949           in = inputList.getFileSystem(conf).open(inputList);
950         }
951         try {
952           BufferedReader reader = new BufferedReader(new InputStreamReader(in, StandardCharsets.UTF_8));
953           String line;
954           while ((line = reader.readLine()) != null) {
955             writer.write(line + "\n");
956             numFiles++;
957           }
958           reader.close();
959         } finally {
960           in.close();
961         }
962       }
963       
964       writer.close();
965     } finally {
966       out.close();
967     }    
968     return numFiles;
969   }

990     List<String> lines = new ArrayList();
991     BufferedReader reader = new BufferedReader(new InputStreamReader(fs.open(fullInputList), StandardCharsets.UTF_8));
992     try {
993       String line;
994       while ((line = reader.readLine()) != null) {
995         lines.add(line);
996       }
997     } finally {
998       reader.close();
999     }
1000     
1001     Collections.shuffle(lines, new Random(421439783L)); // constant seed for reproducability

1126     }

1128     job.getConfiguration().set(MorphlineMapRunner.MORPHLINE_FILE_PARAM, options.morphlineFile.getPath());
1129     return new MorphlineMapRunner(

1175       IOException {
1176

1179       @Override

1201   /*
1202    * You can run MapReduceIndexerTool in Solrcloud mode, and once the MR job completes, you can use
1203    * the standard solrj Solrcloud API to send doc updates and deletes to SolrCloud, and those updates
1204    * and deletes will go to the right Solr shards, and it will work just fine.
1205    * 
1206    * The MapReduce framework doesn't guarantee that input split N goes to the map task with the
1207    * taskId = N. The job tracker and Yarn schedule and assign tasks, considering data locality
1208    * aspects, but without regard of the input split# withing the overall list of input splits. In
1209    * other words, split# != taskId can be true.
1210    * 
1211    * To deal with this issue, our mapper tasks write a little auxiliary metadata file (per task)
1212    * that tells the job driver which taskId processed which split#. Once the mapper-only job is
1213    * completed, the job driver renames the output dirs such that the dir name contains the true solr
1214    * shard id, based on these auxiliary files.
1215    * 
1216    * This way each doc gets assigned to the right Solr shard even with #reducers > #solrshards
1217    * 
1218    * Example for a merge with two shards:
1219    * 
1220    * part-m-00000 and part-m-00001 goes to outputShardNum = 0 and will end up in merged part-m-00000
1221    * part-m-00002 and part-m-00003 goes to outputShardNum = 1 and will end up in merged part-m-00001
1222    * part-m-00004 and part-m-00005 goes to outputShardNum = 2 and will end up in merged part-m-00002
1223    * ... and so on
1224    * 
1225    * Also see run() method above where it uses NLineInputFormat.setNumLinesPerSplit(job,
1226    * options.fanout)
1227    * 
1228    * Also see TreeMergeOutputFormat.TreeMergeRecordWriter.writeShardNumberFile()
1229    */
1230   private boolean renameTreeMergeShardDirs(Path outputTreeMergeStep, Job job, FileSystem fs) throws IOException {
1231     final String dirPrefix = SolrOutputFormat.getOutputName(job);
1232     FileStatus[] dirs = fs.listStatus(outputTreeMergeStep, new PathFilter() {      
1233       @Override
1234       public boolean accept(Path path) {
1235         return path.getName().startsWith(dirPrefix);
1236       }
1237     });
1238     
1239     for (FileStatus dir : dirs) {
1240       if (!dir.isDirectory()) {
1241         throw new IllegalStateException("Not a directory: " + dir.getPath());
1242       }
1243     }
1244 
1245     // Example: rename part-m-00004 to _part-m-00004
1246     for (FileStatus dir : dirs) {
1247       Path path = dir.getPath();
1248       Path renamedPath = new Path(path.getParent(), "_" + path.getName());
1249       if (!rename(path, renamedPath, fs)) {
1250         return false;
1251       }
1252     }
1253     
1254     // Example: rename _part-m-00004 to part-m-00002
1255     for (FileStatus dir : dirs) {
1256       Path path = dir.getPath();
1257       Path renamedPath = new Path(path.getParent(), "_" + path.getName());
1258       
1259       // read auxiliary metadata file (per task) that tells which taskId 
1260       // processed which split# aka solrShard
1261       Path solrShardNumberFile = new Path(renamedPath, TreeMergeMapper.SOLR_SHARD_NUMBER);
1262       InputStream in = fs.open(solrShardNumberFile);
1263       byte[] bytes = ByteStreams.toByteArray(in);
1264       in.close();
1265       Preconditions.checkArgument(bytes.length > 0);
1266       int solrShard = Integer.parseInt(new String(bytes, StandardCharsets.UTF_8));
1267       if (!delete(solrShardNumberFile, false, fs)) {
1268         return false;
1269       }
1270       
1271       // same as FileOutputFormat.NUMBER_FORMAT
1272       NumberFormat numberFormat = NumberFormat.getInstance(Locale.ENGLISH);
1273       numberFormat.setMinimumIntegerDigits(5);
1274       numberFormat.setGroupingUsed(false);
1275       Path finalPath = new Path(renamedPath.getParent(), dirPrefix + "-m-" + numberFormat.format(solrShard));
1276       
1277       LOG.info("MTree merge renaming solr shard: " + solrShard + " from dir: " + dir.getPath() + " to dir: " + finalPath);
1278       if (!rename(renamedPath, finalPath, fs)) {
1279         return false;
1280       }
1281     }
1282     return true;
1283   }
