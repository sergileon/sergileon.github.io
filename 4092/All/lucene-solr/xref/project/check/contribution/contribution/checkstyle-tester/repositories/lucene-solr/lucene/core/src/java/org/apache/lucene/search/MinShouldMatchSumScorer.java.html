20  import java.io.IOException;
21  import java.util.ArrayList;
22  import java.util.Collection;
23  import java.util.Collections;
24  import java.util.List;

26  import org.apache.lucene.util.PriorityQueue;
27  
28  import static org.apache.lucene.search.DisiPriorityQueue.leftNode;
29  import static org.apache.lucene.search.DisiPriorityQueue.parentNode;
30  import static org.apache.lucene.search.DisiPriorityQueue.rightNode;

32  /**
33   * A {@link Scorer} for {@link BooleanQuery} when
34   * {@link BooleanQuery.Builder#setMinimumNumberShouldMatch(int) minShouldMatch} is
35   * between 2 and the total number of clauses.
36   *
37   * This implementation keeps sub scorers in 3 different places:
38   *  - lead: a linked list of scorer that are positioned on the desired doc ID
39   *  - tail: a heap that contains at most minShouldMatch - 1 scorers that are
40   *    behind the desired doc ID. These scorers are ordered by cost so that we
41   *    can advance the least costly ones first.
42   *  - head: a heap that contains scorers which are beyond the desired doc ID,
43   *    ordered by doc ID in order to move quickly to the next candidate.
44   *
45   * Finding the next match consists of first setting the desired doc ID to the
46   * least entry in 'head' and then advance 'tail' until there is a match.
47   */
48  final class MinShouldMatchSumScorer extends Scorer {
49  
50    private static long cost(Collection<Scorer> scorers, int minShouldMatch) {
51      // the idea here is the following: a boolean query c1,c2,...cn with minShouldMatch=m
52      // could be rewritten to:
53      // (c1 AND (c2..cn|msm=m-1)) OR (!c1 AND (c2..cn|msm=m))
54      // if we assume that clauses come in ascending cost, then
55      // the cost of the first part is the cost of c1 (because the cost of a conjunction is
56      // the cost of the least costly clause)
57      // the cost of the second part is the cost of finding m matches among the c2...cn
58      // remaining clauses
59      // since it is a disjunction overall, the total cost is the sum of the costs of these
60      // two parts
61  
62      // If we recurse infinitely, we find out that the cost of a msm query is the sum of the
63      // costs of the num_scorers - minShouldMatch + 1 least costly scorers
64      final PriorityQueue<Scorer> pq = new PriorityQueue<Scorer>(scorers.size() - minShouldMatch + 1) {
65        @Override
66        protected boolean lessThan(Scorer a, Scorer b) {
67          return a.cost() > b.cost();
68        }
69      };
70      for (Scorer scorer : scorers) {
71        pq.insertWithOverflow(scorer);
72      }
73      long cost = 0;
74      for (Scorer scorer = pq.pop(); scorer != null; scorer = pq.pop()) {
75        cost += scorer.cost();
76      }
77      return cost;
78    }
79  
80    final int minShouldMatch;
81    final float[] coord;
82  
83    // list of scorers which 'lead' the iteration and are currently
84    // positioned on 'doc'
85    DisiWrapper<Scorer> lead;
86    int doc;  // current doc ID of the leads
87    int freq; // number of scorers on the desired doc ID
88  
89    // priority queue of scorers that are too advanced compared to the current
90    // doc. Ordered by doc ID.
91    final DisiPriorityQueue<Scorer> head;
92  
93    // priority queue of scorers which are behind the current doc.
94    // Ordered by cost.
95    final DisiWrapper<Scorer>[] tail;
96    int tailSize;
97  
98    final Collection<ChildScorer> childScorers;
99    final long cost;
100 
101   @SuppressWarnings({"unchecked","rawtypes"})
102   MinShouldMatchSumScorer(Weight weight, Collection<Scorer> scorers, int minShouldMatch, float[] coord) {
103     super(weight);
104 
105     if (minShouldMatch > scorers.size()) {
106       throw new IllegalArgumentException("minShouldMatch should be <= the number of scorers");
107     }
108     if (minShouldMatch < 1) {
109       throw new IllegalArgumentException("minShouldMatch should be >= 1");
110     }
111 
112     this.minShouldMatch = minShouldMatch;
113     this.coord = coord;
114     this.doc = -1;
115 
116     head = new DisiPriorityQueue<Scorer>(scorers.size() - minShouldMatch + 1);
117     // there can be at most minShouldMatch - 1 scorers beyond the current position
118     // otherwise we might be skipping over matching documents
119     tail = new DisiWrapper[minShouldMatch - 1];
120 
121     for (Scorer scorer : scorers) {
122       addLead(new DisiWrapper<Scorer>(scorer));
123     }
124 
125     List<ChildScorer> children = new ArrayList<>();
126     for (Scorer scorer : scorers) {
127       children.add(new ChildScorer(scorer, "SHOULD"));
128     }
129     this.childScorers = Collections.unmodifiableCollection(children);
130     this.cost = cost(scorers, minShouldMatch);
131   }
132 
133   @Override
134   public long cost() {
135     return cost;
136   }
137 
138   @Override
139   public final Collection<ChildScorer> getChildren() {
140     return childScorers;
141   }
142 
143   @Override
144   public int nextDoc() throws IOException {
145     // We are moving to the next doc ID, so scorers in 'lead' need to go in
146     // 'tail'. If there is not enough space in 'tail', then we take the least
147     // costly scorers and advance them.
148     for (DisiWrapper<Scorer> s = lead; s != null; s = s.next) {
149       final DisiWrapper<Scorer> evicted = insertTailWithOverFlow(s);
150       if (evicted != null) {
151         if (evicted.doc == doc) {
152           evicted.doc = evicted.iterator.nextDoc();
153         } else {
154           evicted.doc = evicted.iterator.advance(doc + 1);
155         }
156         head.add(evicted);
157       }
158     }
159 
160     setDocAndFreq();
161     return doNext();
162   }
163 
164   @Override
165   public int advance(int target) throws IOException {
166     // Same logic as in nextDoc
167     for (DisiWrapper<Scorer> s = lead; s != null; s = s.next) {
168       final DisiWrapper<Scorer> evicted = insertTailWithOverFlow(s);
169       if (evicted != null) {
170         evicted.doc = evicted.iterator.advance(target);
171         head.add(evicted);
172       }
173     }
174 
175     // But this time there might also be scorers in 'head' behind the desired
176     // target so we need to do the same thing that we did on 'lead' on 'head'
177     DisiWrapper<Scorer> headTop = head.top();
178     while (headTop.doc < target) {
179       final DisiWrapper<Scorer> evicted = insertTailWithOverFlow(headTop);
180       // We know that the tail is full since it contains at most
181       // minShouldMatch - 1 entries and we just moved at least minShouldMatch
182       // entries to it, so evicted is not null
183       evicted.doc = evicted.iterator.advance(target);
184       headTop = head.updateTop(evicted);
185     }
186 
187     setDocAndFreq();
188     return doNext();
189   }
190 
191   private void addLead(DisiWrapper<Scorer> lead) {
192     lead.next = this.lead;
193     this.lead = lead;
194     freq += 1;
195   }
196 
197   private void pushBackLeads() throws IOException {
198     for (DisiWrapper<Scorer> s = lead; s != null; s = s.next) {
199       addTail(s);
200     }
201   }
202 
203   private void advanceTail(DisiWrapper<Scorer> top) throws IOException {
204     top.doc = top.iterator.advance(doc);
205     if (top.doc == doc) {
206       addLead(top);
207     } else {
208       head.add(top);
209     }
210   }
211 
212   private void advanceTail() throws IOException {
213     final DisiWrapper<Scorer> top = popTail();
214     advanceTail(top);
215   }
216 
217   /** Reinitializes head, freq and doc from 'head' */
218   private void setDocAndFreq() {
219     assert head.size() > 0;
220 
221     // The top of `head` defines the next potential match
222     // pop all documents which are on this doc
223     lead = head.pop();
224     lead.next = null;
225     freq = 1;
226     doc = lead.doc;
227     while (head.size() > 0 && head.top().doc == doc) {
228       addLead(head.pop());
229     }
230   }
231 
232   /** Advance tail to the lead until there is a match. */
233   private int doNext() throws IOException {
234     while (freq < minShouldMatch) {
235       assert freq > 0;
236       if (freq + tailSize >= minShouldMatch) {
237         // a match on doc is still possible, try to
238         // advance scorers from the tail
239         advanceTail();
240       } else {
241         // no match on doc is possible anymore, move to the next potential match
242         pushBackLeads();
243         setDocAndFreq();
244       }
245     }
246 
247     return doc;
248   }
249 
250   /** Advance all entries from the tail to know about all matches on the
251    *  current doc. */
252   private void updateFreq() throws IOException {
253     assert freq >= minShouldMatch;
254     // we return the next doc when there are minShouldMatch matching clauses
255     // but some of the clauses in 'tail' might match as well
256     // in general we want to advance least-costly clauses first in order to
257     // skip over non-matching documents as fast as possible. However here,
258     // we are advancing everything anyway so iterating over clauses in
259     // (roughly) cost-descending order might help avoid some permutations in
260     // the head heap
261     for (int i = tailSize - 1; i >= 0; --i) {
262       advanceTail(tail[i]);
263     }
264     tailSize = 0;
265   }
266 
267   @Override
268   public int freq() throws IOException {
269     // we need to know about all matches
270     updateFreq();
271     return freq;
272   }
273 
274   @Override
275   public float score() throws IOException {
276     // we need to know about all matches
277     updateFreq();
278     double score = 0;
279     for (DisiWrapper<Scorer> s = lead; s != null; s = s.next) {
280       score += s.iterator.score();
281     }
282     return coord[freq] * (float) score;
283   }
284 
285   @Override
286   public int docID() {
287     assert doc == lead.doc;
288     return doc;
289   }
290 
291   /** Insert an entry in 'tail' and evict the least-costly scorer if full. */
292   private DisiWrapper<Scorer> insertTailWithOverFlow(DisiWrapper<Scorer> s) {
293     if (tailSize < tail.length) {
294       addTail(s);
295       return null;
296     } else if (tail.length >= 1) {
297       final DisiWrapper<Scorer> top = tail[0];
298       if (top.cost < s.cost) {
299         tail[0] = s;
300         downHeapCost(tail, tailSize);
301         return top;
302       }
303     }
304     return s;
305   }
306 
307   /** Add an entry to 'tail'. Fails if over capacity. */
308   private void addTail(DisiWrapper<Scorer> s) {
309     tail[tailSize] = s;
310     upHeapCost(tail, tailSize);
311     tailSize += 1;
312   }
313 
314   /** Pop the least-costly scorer from 'tail'. */
315   private DisiWrapper<Scorer> popTail() {
316     assert tailSize > 0;
317     final DisiWrapper<Scorer> result = tail[0];
318     tail[0] = tail[--tailSize];
319     downHeapCost(tail, tailSize);
320     return result;
321   }
322 
323   /** Heap helpers */
324 
325   private static void upHeapCost(DisiWrapper<Scorer>[] heap, int i) {
326     final DisiWrapper<Scorer> node = heap[i];
327     final long nodeCost = node.cost;
328     int j = parentNode(i);
329     while (j >= 0 && nodeCost < heap[j].cost) {
330       heap[i] = heap[j];
331       i = j;
332       j = parentNode(j);
333     }
334     heap[i] = node;
335   }
336 
337   private static void downHeapCost(DisiWrapper<Scorer>[] heap, int size) {
338     int i = 0;
339     final DisiWrapper<Scorer> node = heap[0];
340     int j = leftNode(i);
341     if (j < size) {
342       int k = rightNode(j);
343       if (k < size && heap[k].cost < heap[j].cost) {
344         j = k;
345       }
346       if (heap[j].cost < node.cost) {
347         do {
348           heap[i] = heap[j];
349           i = j;
350           j = leftNode(i);
351           k = rightNode(j);
352           if (k < size && heap[k].cost < heap[j].cost) {
353             j = k;
354           }
355         } while (j < size && heap[j].cost < node.cost);
356         heap[i] = node;
357       }
358     }
359   }
