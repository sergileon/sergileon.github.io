40    /** numeric default lucene sort (relative to presumed default value of 0) */
41    private void checkSortMissingDefault(final String field,
42                                         final String negative,
43                                         final String positive) {
44        assertU(adoc("id", "0")); // missing
45        assertU(adoc("id", "1", field, negative));
46        assertU(adoc("id", "2", field, positive));
47        assertU(commit());
48        assertQ(req("q", "*:*", "sort", field+" asc"),
49                "//result/doc[1]/str[@name='id'][.=1]",
50                "//result/doc[2]/str[@name='id'][.=0]",
51                "//result/doc[3]/str[@name='id'][.=2]");
52        assertQ(req("q", "*:*", "sort", field+" desc"),
53                "//result/doc[1]/str[@name='id'][.=2]",
54                "//result/doc[2]/str[@name='id'][.=0]",
55                "//result/doc[3]/str[@name='id'][.=1]");
56    }

96    /** function query based on missing */
97    private void checkSortMissingFunction(final String field,
98                                          final String low,
99                                          final String high) {
100     assertU(adoc("id", "0")); // missing
101     assertU(adoc("id", "1", field, low));
102     assertU(adoc("id", "2", field, high));
103     assertU(commit());
104     assertQ(req("q", "*:*", "fl", "e:exists("+field+")", "sort", "id asc"),
105             "//result/doc[1]/bool[@name='e'][.='false']",
106             "//result/doc[2]/bool[@name='e'][.='true']",
107             "//result/doc[3]/bool[@name='e'][.='true']");
108   }

110   /** missing facet count */
111   private void checkSortMissingFacet(final String field,
112                                      final String low,
113                                      final String high) {
114     assertU(adoc("id", "0")); // missing
115     assertU(adoc("id", "1")); // missing
116     assertU(adoc("id", "2", field, low));
117     assertU(adoc("id", "3", field, high));
118     assertU(commit());
119     assertQ(req("q", "*:*", "facet", "true", "facet.field", field, 
120                 "facet.mincount", "1", "facet.missing", "true"),
121             "//lst[@name='facet_fields']/lst[@name='"+field+"']/int[@name='"+low+"'][.=1]",
122             "//lst[@name='facet_fields']/lst[@name='"+field+"']/int[@name='"+high+"'][.=1]",
123             "//lst[@name='facet_fields']/lst[@name='"+field+"']/int[.=2]");
124   }

359   /** string (and dynamic string) with default lucene sort (treats as "") */
360   public void testStringSort() throws Exception {
361 
362     // note: cant use checkSortMissingDefault because 
363     // nothing sorts lower then the default of ""
364     for (String field : new String[] {"stringdv","dyn_stringdv"}) {
365       assertU(adoc("id", "0")); // missing
366       assertU(adoc("id", "1", field, "a"));
367       assertU(adoc("id", "2", field, "z"));
368       assertU(commit());
369       assertQ(req("q", "*:*", "sort", field+" asc"),
370               "//result/doc[1]/str[@name='id'][.=0]",
371               "//result/doc[2]/str[@name='id'][.=1]",
372               "//result/doc[3]/str[@name='id'][.=2]");
373       assertQ(req("q", "*:*", "sort", field+" desc"),
374               "//result/doc[1]/str[@name='id'][.=2]",
375               "//result/doc[2]/str[@name='id'][.=1]",
376               "//result/doc[3]/str[@name='id'][.=0]");
377     }
378   }
