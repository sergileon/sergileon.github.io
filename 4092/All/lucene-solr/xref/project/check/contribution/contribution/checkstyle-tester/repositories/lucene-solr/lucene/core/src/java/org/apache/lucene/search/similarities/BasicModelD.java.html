20  import static org.apache.lucene.search.similarities.SimilarityBase.log2;
21  
22  /**
23   * Implements the approximation of the binomial model with the divergence
24   * for DFR. The formula used in Lucene differs slightly from the one in the
25   * original paper: to avoid underflow for small values of {@code N} and
26   * {@code F}, {@code N} is increased by {@code 1} and
27   * {@code F} is always increased by {@code tfn+1}.
28   * <p>
29   * WARNING: for terms that do not meet the expected random distribution
30   * (e.g. stopwords), this model may give poor performance, such as
31   * abnormally high scores for low tf values.
32   * @lucene.experimental
33   */
34  public class BasicModelD extends BasicModel {
35    
36    /** Sole constructor: parameter-free */
37    public BasicModelD() {}
38    
39    @Override
40    public final float score(BasicStats stats, float tfn) {
41      // we have to ensure phi is always < 1 for tiny TTF values, otherwise nphi can go negative,
42      // resulting in NaN. cleanest way is to unconditionally always add tfn to totalTermFreq
43      // to create a 'normalized' F.
44      double F = stats.getTotalTermFreq() + 1 + tfn;
45      double phi = (double)tfn / F;
46      double nphi = 1 - phi;
47      double p = 1.0 / (stats.getNumberOfDocuments() + 1);
48      double D = phi * log2(phi / p) + nphi * log2(nphi / (1 - p));
49      return (float)(D * F + 0.5 * log2(1 + 2 * Math.PI * tfn * nphi));
50    }
51    
52    @Override
53    public String toString() {
54      return "D";
55    }
56  }
