20  /**
21   * Bounding box wider than PI but limited on three sides (
22   * bottom lat, left lon, right lon).
23   *
24   * @lucene.internal
25   */
26  public class GeoWideNorthRectangle extends GeoBaseBBox {
27    /** Bottom latitude */
28    protected final double bottomLat;
29    /** Left longitude */
30    protected final double leftLon;
31    /** Right longitude */
32    protected final double rightLon;
33  
34    /** The cosine of the middle latitude */
35    protected final double cosMiddleLat;
36  
37    /** The lower right hand corner point */
38    protected final GeoPoint LRHC;
39    /** The lower left hand corner point */
40    protected final GeoPoint LLHC;
41  
42    /** The bottom plane */
43    protected final SidedPlane bottomPlane;
44    /** The left plane */
45    protected final SidedPlane leftPlane;
46    /** The right plane */
47    protected final SidedPlane rightPlane;
48  
49    /** Notable points for the bottom plane */
50    protected final GeoPoint[] bottomPlanePoints;
51    /** Notable points for the left plane */
52    protected final GeoPoint[] leftPlanePoints;
53    /** Notable points for the right plane */
54    protected final GeoPoint[] rightPlanePoints;
55  
56    /** Center point */
57    protected final GeoPoint centerPoint;
58  
59    /** Composite left/right bounds */
60    protected final EitherBound eitherBound;
61  
62    /** A point on the edge */
63    protected final GeoPoint[] edgePoints;
64  
65    /**
66     * Accepts only values in the following ranges: lat: {@code -PI/2 -> PI/2}, lon: {@code -PI -> PI}.
67     * Horizontal angle must be greater than or equal to PI.
68     */
69    public GeoWideNorthRectangle(final PlanetModel planetModel, final double bottomLat, final double leftLon, double rightLon) {
70      super(planetModel);
71      // Argument checking
72      if (bottomLat > Math.PI * 0.5 || bottomLat < -Math.PI * 0.5)
73        throw new IllegalArgumentException("Bottom latitude out of range");
74      if (leftLon < -Math.PI || leftLon > Math.PI)
75        throw new IllegalArgumentException("Left longitude out of range");
76      if (rightLon < -Math.PI || rightLon > Math.PI)
77        throw new IllegalArgumentException("Right longitude out of range");
78      double extent = rightLon - leftLon;
79      if (extent < 0.0) {
80        extent += 2.0 * Math.PI;
81      }
82      if (extent < Math.PI)
83        throw new IllegalArgumentException("Width of rectangle too small");
84  
85      this.bottomLat = bottomLat;
86      this.leftLon = leftLon;
87      this.rightLon = rightLon;
88  
89      final double sinBottomLat = Math.sin(bottomLat);
90      final double cosBottomLat = Math.cos(bottomLat);
91      final double sinLeftLon = Math.sin(leftLon);
92      final double cosLeftLon = Math.cos(leftLon);
93      final double sinRightLon = Math.sin(rightLon);
94      final double cosRightLon = Math.cos(rightLon);
95  
96      // Now build the four points
97      this.LRHC = new GeoPoint(planetModel, sinBottomLat, sinRightLon, cosBottomLat, cosRightLon, bottomLat, rightLon);
98      this.LLHC = new GeoPoint(planetModel, sinBottomLat, sinLeftLon, cosBottomLat, cosLeftLon, bottomLat, leftLon);
99  
100     final double middleLat = (Math.PI * 0.5 + bottomLat) * 0.5;
101     final double sinMiddleLat = Math.sin(middleLat);
102     this.cosMiddleLat = Math.cos(middleLat);
103     // Normalize
104     while (leftLon > rightLon) {
105       rightLon += Math.PI * 2.0;
106     }
107     final double middleLon = (leftLon + rightLon) * 0.5;
108     final double sinMiddleLon = Math.sin(middleLon);
109     final double cosMiddleLon = Math.cos(middleLon);
110 
111     this.centerPoint = new GeoPoint(planetModel, sinMiddleLat, sinMiddleLon, cosMiddleLat, cosMiddleLon);
112 
113     this.bottomPlane = new SidedPlane(centerPoint, planetModel, sinBottomLat);
114     this.leftPlane = new SidedPlane(centerPoint, cosLeftLon, sinLeftLon);
115     this.rightPlane = new SidedPlane(centerPoint, cosRightLon, sinRightLon);
116 
117     this.bottomPlanePoints = new GeoPoint[]{LLHC, LRHC};
118     this.leftPlanePoints = new GeoPoint[]{planetModel.NORTH_POLE, LLHC};
119     this.rightPlanePoints = new GeoPoint[]{planetModel.NORTH_POLE, LRHC};
120 
121     this.eitherBound = new EitherBound();
122     this.edgePoints = new GeoPoint[]{planetModel.NORTH_POLE};
123   }
124 
125   @Override
126   public GeoBBox expand(final double angle) {
127     final double newTopLat = Math.PI * 0.5;
128     final double newBottomLat = bottomLat - angle;
129     // Figuring out when we escalate to a special case requires some prefiguring
130     double currentLonSpan = rightLon - leftLon;
131     if (currentLonSpan < 0.0)
132       currentLonSpan += Math.PI * 2.0;
133     double newLeftLon = leftLon - angle;
134     double newRightLon = rightLon + angle;
135     if (currentLonSpan + 2.0 * angle >= Math.PI * 2.0) {
136       newLeftLon = -Math.PI;
137       newRightLon = Math.PI;
138     }
139     return GeoBBoxFactory.makeGeoBBox(planetModel, newTopLat, newBottomLat, newLeftLon, newRightLon);
140   }
141 
142   @Override
143   public boolean isWithin(final double x, final double y, final double z) {
144     return
145         bottomPlane.isWithin(x, y, z) &&
146             (leftPlane.isWithin(x, y, z) ||
147                 rightPlane.isWithin(x, y, z));
148   }
149 
150   @Override
151   public double getRadius() {
152     // Here we compute the distance from the middle point to one of the corners.  However, we need to be careful
153     // to use the longest of three distances: the distance to a corner on the top; the distnace to a corner on the bottom, and
154     // the distance to the right or left edge from the center.
155     final double centerAngle = (rightLon - (rightLon + leftLon) * 0.5) * cosMiddleLat;
156     final double bottomAngle = centerPoint.arcDistance(LLHC);
157     return Math.max(centerAngle, bottomAngle);
158   }
159 
160   @Override
161   public GeoPoint getCenter() {
162     return centerPoint;
163   }
164 
165   @Override
166   public GeoPoint[] getEdgePoints() {
167     return edgePoints;
168   }
169 
170   @Override
171   public boolean intersects(final Plane p, final GeoPoint[] notablePoints, final Membership... bounds) {
172     // Right and left bounds are essentially independent hemispheres; crossing into the wrong part of one
173     // requires crossing into the right part of the other.  So intersection can ignore the left/right bounds.
174     return
175         p.intersects(planetModel, bottomPlane, notablePoints, bottomPlanePoints, bounds, eitherBound) ||
176             p.intersects(planetModel, leftPlane, notablePoints, leftPlanePoints, bounds, bottomPlane) ||
177             p.intersects(planetModel, rightPlane, notablePoints, rightPlanePoints, bounds, bottomPlane);
178   }
179 
180   @Override
181   public void getBounds(Bounds bounds) {
182     super.getBounds(bounds);
183     bounds.isWide()
184       .addHorizontalPlane(planetModel, bottomLat, bottomPlane, eitherBound)
185       .addVerticalPlane(planetModel, leftLon, leftPlane, bottomPlane)
186       .addVerticalPlane(planetModel, rightLon, rightPlane, bottomPlane)
187       .addPoint(LLHC).addPoint(LRHC).addPoint(planetModel.NORTH_POLE);
188   }
189 
190   @Override
191   public int getRelationship(final GeoShape path) {
192     //System.err.println(this+" comparing to "+path);
193     final int insideRectangle = isShapeInsideBBox(path);
194     if (insideRectangle == SOME_INSIDE) {
195       //System.err.println(" some inside");
196       return OVERLAPS;
197     }
198 
199     final boolean insideShape = path.isWithin(planetModel.NORTH_POLE);
200 
201     if (insideRectangle == ALL_INSIDE && insideShape) {
202       //System.err.println(" both inside each other");
203       return OVERLAPS;
204     }
205 
206     if (
207         path.intersects(bottomPlane, bottomPlanePoints, eitherBound) ||
208             path.intersects(leftPlane, leftPlanePoints, bottomPlane) ||
209             path.intersects(rightPlane, rightPlanePoints, bottomPlane)) {
210       //System.err.println(" edges intersect");
211       return OVERLAPS;
212     }
213 
214     if (insideRectangle == ALL_INSIDE) {
215       //System.err.println(" shape inside rectangle");
216       return WITHIN;
217     }
218 
219     if (insideShape) {
220       //System.err.println(" rectangle inside shape");
221       return CONTAINS;
222     }
223 
224     //System.err.println(" disjoint");
225     return DISJOINT;
226   }

228   @Override
229   protected double outsideDistance(final DistanceStyle distanceStyle, final double x, final double y, final double z) {
230     final double bottomDistance = distanceStyle.computeDistance(planetModel, bottomPlane, x,y,z, eitherBound);
231     // Because the rectangle exceeds 180 degrees, it is safe to compute the horizontally 
232     // unbounded distance to both the left and the right and only take the minimum of the two.
233     final double leftDistance = distanceStyle.computeDistance(planetModel, leftPlane, x,y,z, bottomPlane);
234     final double rightDistance = distanceStyle.computeDistance(planetModel, rightPlane, x,y,z, bottomPlane);
235     
236     final double LRHCDistance = distanceStyle.computeDistance(LRHC, x,y,z);
237     final double LLHCDistance = distanceStyle.computeDistance(LLHC, x,y,z);
238     
239     return Math.min(
240       Math.min(
241         bottomDistance,
242         Math.min(leftDistance, rightDistance)),
243       Math.min(LRHCDistance, LLHCDistance));
244   }
245 
246   @Override
247   public boolean equals(Object o) {
248     if (!(o instanceof GeoWideNorthRectangle))
249       return false;
250     GeoWideNorthRectangle other = (GeoWideNorthRectangle) o;
251     return super.equals(other) && other.LLHC.equals(LLHC) && other.LRHC.equals(LRHC);
252   }
253 
254   @Override
255   public int hashCode() {
256     int result = super.hashCode();
257     result = 31 * result + LLHC.hashCode();
258     result = 31 * result + LRHC.hashCode();
259     return result;
260   }
261 
262   @Override
263   public String toString() {
264     return "GeoWideNorthRectangle: {planetmodel="+planetModel+", bottomlat=" + bottomLat + "(" + bottomLat * 180.0 / Math.PI + "), leftlon=" + leftLon + "(" + leftLon * 180.0 / Math.PI + "), rightlon=" + rightLon + "(" + rightLon * 180.0 / Math.PI + ")}";
265   }
266 
267   /** Membership implementation representing a wide (more than 180 degree) bound.
268    */
269   protected class EitherBound implements Membership {
270     /** Constructor.
271      */
272     public EitherBound() {
273     }
274 
275     @Override
276     public boolean isWithin(final Vector v) {
277       return leftPlane.isWithin(v) || rightPlane.isWithin(v);
278     }
279 
280     @Override
281     public boolean isWithin(final double x, final double y, final double z) {
282       return leftPlane.isWithin(x, y, z) || rightPlane.isWithin(x, y, z);
283     }
284   }
