301   /**
302    * Executes one query using interval faceting and compares with the same query using
303    * facet query with the same range
304    */
305   @SuppressWarnings("unchecked")
306   private void doTestQuery(int cardinality, String[] fields) throws Exception {
307     String[] startOptions = new String[]{"(", "["};
308     String[] endOptions = new String[]{")", "]"};
309     // the query should match some documents in most cases
310     Integer[] qRange = getRandomRange(cardinality, "id");
311     ModifiableSolrParams params = new ModifiableSolrParams();
312     params.set("q", "id:[" + qRange[0] + " TO " + qRange[1] + "]");
313     params.set("facet", "true");
314     String field = fields[random().nextInt(fields.length)]; //choose from any of the fields
315     params.set("facet.interval", field);
316     // number of intervals
317     for (int i = 0; i < 1 + random().nextInt(20); i++) {
318       Integer[] interval = getRandomRange(cardinality, field);
319       String open = startOptions[interval[0] % 2];
320       String close = endOptions[interval[1] % 2];
321       params.add("f." + field + ".facet.interval.set", open + interval[0] + "," + interval[1] + close);
322       params.add("facet.query", field + ":" + open.replace('(', '{') + interval[0] + " TO " + interval[1] + close.replace(')', '}'));
323     }
324     SolrQueryRequest req = req(params);
325     try {
326       SolrQueryResponse rsp = h.queryAndResponse("standard", req);
327       NamedList<Object> facetQueries = (NamedList<Object>) ((NamedList<Object>) rsp.getValues().get("facet_counts")).get("facet_queries");
328       NamedList<Object> facetIntervals = (NamedList<Object>) ((NamedList<Object>) (NamedList<Object>) ((NamedList<Object>) rsp.getValues().get("facet_counts"))
329           .get("facet_intervals")).get(field);
330       assertEquals("Responses don't have the same number of facets: \n" + facetQueries + "\n" + facetIntervals,
331           facetQueries.size(), facetIntervals.size());
332       for (int i = 0; i < facetIntervals.size(); i++) {
333         assertEquals("Interval did not match: " + facetIntervals.getName(i), facetIntervals.getVal(i).toString(),
334             facetQueries.get(field + ":" + facetIntervals.getName(i).replace(",", " TO ").replace('(', '{').replace(')', '}')).toString());
335       }
336     } finally {
337       req.close();
338     }
339 
340   }

342   /**
343    * Returns a random range. It's guaranteed that the first
344    * number will be lower than the second, and both of them
345    * between 0 (inclusive) and <code>max</code> (exclusive).
346    * If the fieldName is "test_s_dv" or "test_ss_dv" (the
347    * two fields used for Strings), the comparison will be done
348    * alphabetically
349    */
350   private Integer[] getRandomRange(int max, String fieldName) {
351     Integer[] values = new Integer[2];
352     values[0] = random().nextInt(max);
353     values[1] = random().nextInt(max);
354     if (fieldName.startsWith("test_s")) {
355       Arrays.sort(values, new Comparator<Integer>() {
356 
357         @Override
358         public int compare(Integer o1, Integer o2) {
359           return String.valueOf(o1).compareTo(String.valueOf(o2));
360         }
361       });
362     } else {
363       Arrays.sort(values);
364     }
365     return values;
366   }

579     SchemaField f = h.getCore().getLatestSchema().getField(fieldName);
580     FacetInterval interval = new FacetInterval(f, intervalStr, new ModifiableSolrParams());
581     for (long l : included) {

583           IntervalCompareResult.INCLUDED, interval.includes(l));

585     for (long l : lowerThanStart) {
586       assertEquals("Value " + l + " should be LOWER_THAN_START for inteval " + interval,
587           IntervalCompareResult.LOWER_THAN_START, interval.includes(l));

589     for (long l : graterThanEnd) {
590       assertEquals("Value " + l + " should be GRATER_THAN_END for inteval " + interval,
591           IntervalCompareResult.GREATER_THAN_END, interval.includes(l));
592     }

599     SchemaField f = h.getCore().getLatestSchema().getField(fieldName);
600     ModifiableSolrParams solrParams = new ModifiableSolrParams();

606

969   /**
970    * Will run a match all query, and ask for interval facets in the specified field.
971    * The intervals facet are indicated in the <code>intervals</code> parameter, followed
972    * by the expected count result. For example:
973    * <code>assertIntervalQuery("my_string_field", "[0,10]", "3", "(20,*), "12");</code>
974    *
975    * @param field     The field in which the interval facet should be asked
976    * @param intervals a variable array of intervals followed by the expected count (also a string)
977    */
978   private void assertIntervalQuery(String field, String... intervals) {
979     assertIntervalQuery(field, "*:*", -1, intervals);
980   }
981 
982   private void assertIntervalQuery(String field, String query, int resultCount, String... intervals) {
983     assert (intervals.length & 1) == 0;
984     int idx = 0;
985     String[] params = new String[intervals.length + 6];
986     params[idx++] = "q";
987     params[idx++] = query;
988     params[idx++] = "facet";
989     params[idx++] = "true";
990     params[idx++] = "facet.interval";
991     params[idx++] = field;
992 
993     for (int i = 0; i < intervals.length; i += 2) {
994       params[idx++] = "f." + field + ".facet.interval.set";
995       params[idx++] = intervals[i];
996     }
997 
998     String[] tests = new String[intervals.length / 2 + (resultCount > 0 ? 1 : 0)];
999     idx = 0;
1000     for (int i = 0; i < intervals.length; i += 2) {
1001       tests[idx++] = "//lst[@name='facet_intervals']/lst[@name='" + field + "']/int[@name='" + intervals[i] + "'][.=" + intervals[i + 1] + "]";
1002     }
1003     if (resultCount >= 0) {
1004       tests[idx++] = "//*[@numFound='" + resultCount + "']";
1005     }
1006 
1007     assertQ("Unexpected facet iterval count. Field:" + field + ", Intervals: " + Arrays.toString(intervals) + "Query: " + query,
1008         req(params), tests);
1009   }
