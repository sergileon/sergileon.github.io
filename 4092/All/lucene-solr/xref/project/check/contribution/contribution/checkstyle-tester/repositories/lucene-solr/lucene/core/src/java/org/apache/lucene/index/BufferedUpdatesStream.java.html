43  /* Tracks the stream of {@link BufferedDeletes}.
44   * When DocumentsWriterPerThread flushes, its buffered
45   * deletes and updates are appended to this stream.  We later
46   * apply them (resolve them to the actual
47   * docIDs, per segment) when a merge is started
48   * (only to the to-be-merged segments).  We
49   * also apply to all segments when NRT reader is pulled,
50   * commit/close is called, or when too many deletes or  updates are
51   * buffered and must be flushed (by RAM usage or by count).
52   *
53   * Each packet is assigned a generation, and each flushed or
54   * merged segment is also assigned a generation, so we can
55   * track which BufferedDeletes packets to apply to any given
56   * segment. */
57  
58  class BufferedUpdatesStream implements Accountable {

79    // Appends a new packet of buffered deletes to the stream,
80    // setting its generation:
81    public synchronized long push(FrozenBufferedUpdates packet) {
82      /*
83       * The insert operation must be atomic. If we let threads increment the gen
84       * and push the packet afterwards we risk that packets are out of order.
85       * With DWPT this is possible if two or more flushes are racing for pushing
86       * updates. If the pushed packets get our of order would loose documents
87       * since deletes are applied to the wrong segments.
88       */
89      packet.setDelGen(nextGen++);
90      assert packet.any();
91      assert checkDeleteStats();
92      assert packet.delGen() < nextGen;
93      assert updates.isEmpty() || updates.get(updates.size()-1).delGen() < packet.delGen() : "Delete packets must be in order";
94      updates.add(packet);
95      numTerms.addAndGet(packet.numTermDeletes);
96      bytesUsed.addAndGet(packet.bytesUsed);
97      if (infoStream.isEnabled("BD")) {
98        infoStream.message("BD", "push deletes " + packet + " segmentPrivate?=" + packet.isSegmentPrivate + " delGen=" + packet.delGen() + " packetCount=" + updates.size() + " totBytesUsed=" + bytesUsed.get());
99      }
100     assert checkDeleteStats();
101     return packet.delGen();
102   }
103 
104   public synchronized void clear() {
105     updates.clear();
106     nextGen = 1;
107     numTerms.set(0);
108     bytesUsed.set(0);
109   }

132     public final boolean anyDeletes;

147   // Sorts SegmentInfos from smallest to biggest bufferedDelGen:
148   private static final Comparator<SegmentCommitInfo> sortSegInfoByDelGen = new Comparator<SegmentCommitInfo>() {
149     @Override
150     public int compare(SegmentCommitInfo si1, SegmentCommitInfo si2) {
151       return Long.compare(si1.getBufferedDeletesGen(), si2.getBufferedDeletesGen());
152     }

155   /** Resolves the buffered deleted Term/Query/docIDs, into
156    *  actual deleted docIDs in the liveDocs MutableBits for
157    *  each SegmentReader. */
158   public synchronized ApplyDeletesResult applyDeletesAndUpdates(IndexWriter.ReaderPool pool, List<SegmentCommitInfo> infos) throws IOException {
159     final long t0 = System.currentTimeMillis();
160 
161     final long gen = nextGen++;
162 
163     if (infos.size() == 0) {
164       return new ApplyDeletesResult(false, gen, null);
165     }
166 
167     // We only init these on demand, when we find our first deletes that need to be applied:
168     SegmentState[] segStates = null;
169 
170     long totDelCount = 0;
171     long totTermVisitedCount = 0;
172 
173     boolean success = false;
174 
175     ApplyDeletesResult result = null;
176 
177     try {
178       if (infoStream.isEnabled("BD")) {
179         infoStream.message("BD", String.format(Locale.ROOT, "applyDeletes: open segment readers took %d msec", System.currentTimeMillis()-t0));
180       }
181 
182       assert checkDeleteStats();
183 
184       if (!any()) {
185         if (infoStream.isEnabled("BD")) {
186           infoStream.message("BD", "applyDeletes: no segments; skipping");
187         }
188         return new ApplyDeletesResult(false, gen, null);
189       }
190 
191       if (infoStream.isEnabled("BD")) {
192         infoStream.message("BD", "applyDeletes: infos=" + infos + " packetCount=" + updates.size());
193       }
194 
195       infos = sortByDelGen(infos);
196 
197       CoalescedUpdates coalescedUpdates = null;
198       int infosIDX = infos.size()-1;
199       int delIDX = updates.size()-1;
200 
201       // Backwards merge sort the segment delGens with the packet delGens in the buffered stream:
202       while (infosIDX >= 0) {
203         final FrozenBufferedUpdates packet = delIDX >= 0 ? updates.get(delIDX) : null;
204         final SegmentCommitInfo info = infos.get(infosIDX);
205         final long segGen = info.getBufferedDeletesGen();
206 
207         if (packet != null && segGen < packet.delGen()) {
208           if (!packet.isSegmentPrivate && packet.any()) {
209             /*
210              * Only coalesce if we are NOT on a segment private del packet: the segment private del packet
211              * must only apply to segments with the same delGen.  Yet, if a segment is already deleted
212              * from the SI since it had no more documents remaining after some del packets younger than
213              * its segPrivate packet (higher delGen) have been applied, the segPrivate packet has not been
214              * removed.
215              */
216             if (coalescedUpdates == null) {
217               coalescedUpdates = new CoalescedUpdates();
218             }
219             coalescedUpdates.update(packet);
220           }
221 
222           delIDX--;
223         } else if (packet != null && segGen == packet.delGen()) {
224           assert packet.isSegmentPrivate : "Packet and Segments deletegen can only match on a segment private del packet gen=" + segGen;
225 
226           if (segStates == null) {
227             segStates = openSegmentStates(pool, infos);
228           }
229 
230           SegmentState segState = segStates[infosIDX];
231 
232           // Lock order: IW -> BD -> RP
233           assert pool.infoIsLive(info);
234           int delCount = 0;
235           final DocValuesFieldUpdates.Container dvUpdates = new DocValuesFieldUpdates.Container();
236           if (coalescedUpdates != null) {
237             delCount += applyQueryDeletes(coalescedUpdates.queriesIterable(), segState);
238             applyDocValuesUpdates(coalescedUpdates.numericDVUpdates, segState, dvUpdates);
239             applyDocValuesUpdates(coalescedUpdates.binaryDVUpdates, segState, dvUpdates);
240           }
241           delCount += applyQueryDeletes(packet.queriesIterable(), segState);
242           applyDocValuesUpdates(Arrays.asList(packet.numericDVUpdates), segState, dvUpdates);
243           applyDocValuesUpdates(Arrays.asList(packet.binaryDVUpdates), segState, dvUpdates);
244           if (dvUpdates.any()) {
245             segState.rld.writeFieldUpdates(info.info.dir, dvUpdates);
246           }
247 
248           totDelCount += delCount;
249 
250           /*
251            * Since we are on a segment private del packet we must not
252            * update the coalescedUpdates here! We can simply advance to the 
253            * next packet and seginfo.
254            */
255           delIDX--;
256           infosIDX--;
257 
258         } else {
259           if (coalescedUpdates != null) {
260             if (segStates == null) {
261               segStates = openSegmentStates(pool, infos);
262             }
263             SegmentState segState = segStates[infosIDX];
264             // Lock order: IW -> BD -> RP
265             assert pool.infoIsLive(info);
266             int delCount = 0;
267             delCount += applyQueryDeletes(coalescedUpdates.queriesIterable(), segState);
268             DocValuesFieldUpdates.Container dvUpdates = new DocValuesFieldUpdates.Container();
269             applyDocValuesUpdates(coalescedUpdates.numericDVUpdates, segState, dvUpdates);
270             applyDocValuesUpdates(coalescedUpdates.binaryDVUpdates, segState, dvUpdates);
271             if (dvUpdates.any()) {
272               segState.rld.writeFieldUpdates(info.info.dir, dvUpdates);
273             }
274 
275             totDelCount += delCount;
276           }
277 
278           infosIDX--;
279         }
280       }
281 
282       // Now apply all term deletes:
283       if (coalescedUpdates != null && coalescedUpdates.totalTermCount != 0) {
284         if (segStates == null) {
285           segStates = openSegmentStates(pool, infos);
286         }
287         totTermVisitedCount += applyTermDeletes(coalescedUpdates, segStates);
288       }
289 
290       assert checkDeleteStats();
291 
292       success = true;
293 
294     } finally {
295       if (segStates != null) {
296         result = closeSegmentStates(pool, segStates, success, gen);
297       }
298     }
299 
300     if (result == null) {
301       result = new ApplyDeletesResult(false, gen, null);      
302     }
303 
304     if (infoStream.isEnabled("BD")) {
305       infoStream.message("BD",
306                          String.format(Locale.ROOT,
307                                        "applyDeletes took %d msec for %d segments, %d newly deleted docs (query deletes), %d visited terms, allDeleted=%s",
308                                        System.currentTimeMillis()-t0, infos.size(), totDelCount, totTermVisitedCount, result.allDeleted));
309     }
310 
311     return result;
312   }

314   private List<SegmentCommitInfo> sortByDelGen(List<SegmentCommitInfo> infos) {
315     infos = new ArrayList<>(infos);
316     // Smaller delGens come first:
317     Collections.sort(infos, sortSegInfoByDelGen);
318     return infos;
319   }

322     return nextGen++;

325   // Lock order IW -> BD
326   /* Removes any BufferedDeletes that we no longer need to
327    * store because all segments in the index have had the
328    * deletes applied. */
329   public synchronized void prune(SegmentInfos segmentInfos) {
330     assert checkDeleteStats();
331     long minGen = Long.MAX_VALUE;
332     for(SegmentCommitInfo info : segmentInfos) {
333       minGen = Math.min(info.getBufferedDeletesGen(), minGen);
334     }
335 
336     if (infoStream.isEnabled("BD")) {
337       infoStream.message("BD", "prune sis=" + segmentInfos + " minGen=" + minGen + " packetCount=" + updates.size());
338     }
339     final int limit = updates.size();
340     for(int delIDX=0;delIDX<limit;delIDX++) {
341       if (updates.get(delIDX).delGen() >= minGen) {
342         prune(delIDX);
343         assert checkDeleteStats();
344         return;
345       }
346     }
347 
348     // All deletes pruned
349     prune(limit);
350     assert !any();
351     assert checkDeleteStats();
352   }
353 
354   private synchronized void prune(int count) {
355     if (count > 0) {
356       if (infoStream.isEnabled("BD")) {
357         infoStream.message("BD", "pruneDeletes: prune " + count + " packets; " + (updates.size() - count) + " packets remain");
358       }
359       for(int delIDX=0;delIDX<count;delIDX++) {
360         final FrozenBufferedUpdates packet = updates.get(delIDX);
361         numTerms.addAndGet(-packet.numTermDeletes);
362         assert numTerms.get() >= 0;
363         bytesUsed.addAndGet(-packet.bytesUsed);
364         assert bytesUsed.get() >= 0;
365       }
366       updates.subList(0, count).clear();
367     }
368   }

370   static class SegmentState {
371     final long delGen;
372     final ReadersAndUpdates rld;
373     final SegmentReader reader;
374     final int startDelCount;
375 
376     TermsEnum termsEnum;
377     PostingsEnum postingsEnum;
378     BytesRef term;
379     boolean any;
380 
381     public SegmentState(IndexWriter.ReaderPool pool, SegmentCommitInfo info) throws IOException {
382       rld = pool.get(info, true);
383       startDelCount = rld.getPendingDeleteCount();
384       reader = rld.getReader(IOContext.READ);
385       delGen = info.getBufferedDeletesGen();
386     }
387 
388     public void finish(IndexWriter.ReaderPool pool) throws IOException {
389       try {
390         rld.release(reader);
391       } finally {
392         pool.release(rld);
393       }
394     }
395   }

397   /** Does a merge sort by current term across all segments. */
398   static class SegmentQueue extends PriorityQueue<SegmentState> {
399     public SegmentQueue(int size) {
400       super(size);
401     }
402 
403     @Override
404     protected boolean lessThan(SegmentState a, SegmentState b) {
405       return a.term.compareTo(b.term) < 0;
406     }
407   }

409   /** Opens SegmentReader and inits SegmentState for each segment. */
410   private SegmentState[] openSegmentStates(IndexWriter.ReaderPool pool, List<SegmentCommitInfo> infos) throws IOException {
411     int numReaders = infos.size();
412     SegmentState[] segStates = new SegmentState[numReaders];
413     boolean success = false;
414     try {
415       for(int i=0;i<numReaders;i++) {
416         segStates[i] = new SegmentState(pool, infos.get(i));
417       }
418       success = true;
419     } finally {
420       if (success == false) {
421         for(int j=0;j<numReaders;j++) {
422           if (segStates[j] != null) {
423             try {
424               segStates[j].finish(pool);
425             } catch (Throwable th) {
426               // suppress so we keep throwing original exc
427             }
428           }
429         }
430       }
431     }
432 
433     return segStates;
434   }

436   /** Close segment states previously opened with openSegmentStates. */
437   private ApplyDeletesResult closeSegmentStates(IndexWriter.ReaderPool pool, SegmentState[] segStates, boolean success, long gen) throws IOException {
438     int numReaders = segStates.length;
439     Throwable firstExc = null;
440     List<SegmentCommitInfo> allDeleted = null;
441     long totDelCount = 0;
442     for (int j=0;j<numReaders;j++) {
443       SegmentState segState = segStates[j];
444       if (success) {
445         totDelCount += segState.rld.getPendingDeleteCount() - segState.startDelCount;
446         segState.reader.getSegmentInfo().setBufferedDeletesGen(gen);
447         int fullDelCount = segState.rld.info.getDelCount() + segState.rld.getPendingDeleteCount();
448         assert fullDelCount <= segState.rld.info.info.maxDoc();
449         if (fullDelCount == segState.rld.info.info.maxDoc()) {
450           if (allDeleted == null) {
451             allDeleted = new ArrayList<>();
452           }
453           allDeleted.add(segState.reader.getSegmentInfo());
454         }
455       }
456       try {
457         segStates[j].finish(pool);
458       } catch (Throwable th) {
459         if (firstExc != null) {
460           firstExc = th;
461         }
462       }
463     }
464 
465     if (success) {
466       // Does nothing if firstExc is null:
467       IOUtils.reThrow(firstExc);
468     }
469 
470     if (infoStream.isEnabled("BD")) {
471       infoStream.message("BD", "applyDeletes: " + totDelCount + " new deleted documents");
472     }
473 
474     return new ApplyDeletesResult(totDelCount > 0, gen, allDeleted);      
475   }

477   /** Merge sorts the deleted terms and all segments to resolve terms to docIDs for deletion. */
478   private synchronized long applyTermDeletes(CoalescedUpdates updates, SegmentState[] segStates) throws IOException {
479 
480     long startNS = System.nanoTime();
481 
482     int numReaders = segStates.length;
483 
484     long delTermVisitedCount = 0;
485     long segTermVisitedCount = 0;
486 
487     FieldTermIterator iter = updates.termIterator();
488 
489     String field = null;
490     SegmentQueue queue = null;
491 
492     BytesRef term;
493 
494     while ((term = iter.next()) != null) {
495 
496       if (iter.field() != field) {
497         // field changed
498         field = iter.field();
499 
500         queue = new SegmentQueue(numReaders);
501 
502         long segTermCount = 0;
503         for(int i=0;i<numReaders;i++) {
504           SegmentState state = segStates[i];
505           Terms terms = state.reader.fields().terms(field);
506           if (terms != null) {
507             segTermCount += terms.size();
508             state.termsEnum = terms.iterator();
509             state.term = state.termsEnum.next();
510             if (state.term != null) {
511               queue.add(state);
512             }
513           }
514         }
515 
516         assert checkDeleteTerm(null);
517       }
518 
519       assert checkDeleteTerm(term);
520 
521       delTermVisitedCount++;
522 
523       long delGen = iter.delGen();
524 
525       while (queue.size() != 0) {
526 
527         // Get next term merged across all segments
528         SegmentState state = queue.top();
529         segTermVisitedCount++;
530 
531         int cmp = term.compareTo(state.term);
532 
533         if (cmp < 0) {
534           break;
535         } else if (cmp == 0) {
536           // fall through
537         } else {
538           TermsEnum.SeekStatus status = state.termsEnum.seekCeil(term);
539           if (status == TermsEnum.SeekStatus.FOUND) {
540             // fallthrough
541           } else {
542             if (status == TermsEnum.SeekStatus.NOT_FOUND) {
543               state.term = state.termsEnum.term();
544               queue.updateTop();
545             } else {
546               // No more terms in this segment
547               queue.pop();
548             }
549 
550             continue;
551           }
552         }
553 
554         assert state.delGen != delGen;
555 
556         if (state.delGen < delGen) {
557 
558           // we don't need term frequencies for this
559           final Bits acceptDocs = state.rld.getLiveDocs();
560           state.postingsEnum = state.termsEnum.postings(state.postingsEnum, PostingsEnum.NONE);
561 
562           assert state.postingsEnum != null;
563 
564           while (true) {
565             final int docID = state.postingsEnum.nextDoc();
566             if (docID == DocIdSetIterator.NO_MORE_DOCS) {
567               break;
568             }
569             if (acceptDocs != null && acceptDocs.get(docID) == false) {
570               continue;
571             }
572             if (!state.any) {
573               state.rld.initWritableLiveDocs();
574               state.any = true;
575             }
576 
577             // NOTE: there is no limit check on the docID
578             // when deleting by Term (unlike by Query)
579             // because on flush we apply all Term deletes to
580             // each segment.  So all Term deleting here is
581             // against prior segments:
582             state.rld.delete(docID);
583           }
584         }
585 
586         state.term = state.termsEnum.next();
587         if (state.term == null) {
588           queue.pop();
589         } else {
590           queue.updateTop();
591         }
592       }
593     }
594 
595     if (infoStream.isEnabled("BD")) {
596       infoStream.message("BD",
597                          String.format(Locale.ROOT, "applyTermDeletes took %.1f msec for %d segments and %d packets; %d del terms visited; %d seg terms visited",
598                                        (System.nanoTime()-startNS)/1000000.,
599                                        numReaders,
600                                        updates.terms.size(),
601                                        delTermVisitedCount, segTermVisitedCount));
602     }
603 
604     return delTermVisitedCount;
605   }

607   // DocValues updates
608   private synchronized void applyDocValuesUpdates(Iterable<? extends DocValuesUpdate> updates, 
609       SegmentState segState, DocValuesFieldUpdates.Container dvUpdatesContainer) throws IOException {
610     Fields fields = segState.reader.fields();
611 
612     // TODO: we can process the updates per DV field, from last to first so that
613     // if multiple terms affect same document for the same field, we add an update
614     // only once (that of the last term). To do that, we can keep a bitset which
615     // marks which documents have already been updated. So e.g. if term T1
616     // updates doc 7, and then we process term T2 and it updates doc 7 as well,
617     // we don't apply the update since we know T1 came last and therefore wins
618     // the update.
619     // We can also use that bitset as 'liveDocs' to pass to TermEnum.docs(), so
620     // that these documents aren't even returned.
621     
622     String currentField = null;
623     TermsEnum termsEnum = null;
624     PostingsEnum postingsEnum = null;
625     
626     for (DocValuesUpdate update : updates) {
627       Term term = update.term;
628       int limit = update.docIDUpto;
629       
630       // TODO: we traverse the terms in update order (not term order) so that we
631       // apply the updates in the correct order, i.e. if two terms udpate the
632       // same document, the last one that came in wins, irrespective of the
633       // terms lexical order.
634       // we can apply the updates in terms order if we keep an updatesGen (and
635       // increment it with every update) and attach it to each NumericUpdate. Note
636       // that we cannot rely only on docIDUpto because an app may send two updates
637       // which will get same docIDUpto, yet will still need to respect the order
638       // those updates arrived.
639       
640       if (!term.field().equals(currentField)) {
641         // if we change the code to process updates in terms order, enable this assert
642 //        assert currentField == null || currentField.compareTo(term.field()) < 0;
643         currentField = term.field();
644         Terms terms = fields.terms(currentField);
645         if (terms != null) {
646           termsEnum = terms.iterator();
647         } else {
648           termsEnum = null;
649         }
650       }
651 
652       if (termsEnum == null) {
653         // no terms in this field
654         continue;
655       }
656 
657       if (termsEnum.seekExact(term.bytes())) {
658         // we don't need term frequencies for this
659         final Bits acceptDocs = segState.rld.getLiveDocs();
660         postingsEnum = termsEnum.postings(postingsEnum, PostingsEnum.NONE);
661 
662         DocValuesFieldUpdates dvUpdates = dvUpdatesContainer.getUpdates(update.field, update.type);
663         if (dvUpdates == null) {
664           dvUpdates = dvUpdatesContainer.newUpdates(update.field, update.type, segState.reader.maxDoc());
665         }
666         int doc;
667         while ((doc = postingsEnum.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
668           if (doc >= limit) {
669             break; // no more docs that can be updated for this term
670           }
671           if (acceptDocs != null && acceptDocs.get(doc) == false) {
672             continue;
673           }
674           dvUpdates.add(doc, update.value);
675         }
676       }
677     }
678   }

681     public final Query query;
682     public final int limit;
683     public QueryAndLimit(Query query, int limit) {
684       this.query = query;
685       this.limit = limit;
686     }
687   }

689   // Delete by query
690   private static long applyQueryDeletes(Iterable<QueryAndLimit> queriesIter, SegmentState segState) throws IOException {
691     long delCount = 0;
692     final LeafReaderContext readerContext = segState.reader.getContext();
693     for (QueryAndLimit ent : queriesIter) {
694       Query query = ent.query;
695       int limit = ent.limit;
696       final IndexSearcher searcher = new IndexSearcher(readerContext.reader());
697       searcher.setQueryCache(null);
698       final Weight weight = searcher.createNormalizedWeight(query, false);
699       final DocIdSetIterator it = weight.scorer(readerContext);
700       if (it != null) {
701         final Bits liveDocs = readerContext.reader().getLiveDocs();
702         while (true)  {
703           int doc = it.nextDoc();
704           if (doc >= limit) {
705             break;
706           }
707           if (liveDocs != null && liveDocs.get(doc) == false) {
708             continue;
709           }
710 
711           if (!segState.any) {
712             segState.rld.initWritableLiveDocs();
713             segState.any = true;
714           }
715           if (segState.rld.delete(doc)) {
716             delCount++;
717           }
718         }
719       }
720     }
721 
722     return delCount;
723   }

725   // used only by assert
726   private boolean checkDeleteTerm(BytesRef term) {
727     if (term != null) {
728       assert lastDeleteTerm == null || term.compareTo(lastDeleteTerm) >= 0: "lastTerm=" + lastDeleteTerm + " vs term=" + term;
729     }
730     // TODO: we re-use term now in our merged iterable, but we shouldn't clone, instead copy for this assert
731     lastDeleteTerm = term == null ? null : BytesRef.deepCopyOf(term);
732     return true;
733   }
