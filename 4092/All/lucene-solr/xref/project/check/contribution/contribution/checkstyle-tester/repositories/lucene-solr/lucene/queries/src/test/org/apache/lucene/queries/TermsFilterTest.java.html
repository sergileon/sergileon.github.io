59    public void testCachability() throws Exception {
60      TermsFilter a = termsFilter(random().nextBoolean(), new Term("field1", "a"), new Term("field1", "b"));
61      HashSet<Filter> cachedFilters = new HashSet<>();
62      cachedFilters.add(a);
63      TermsFilter b = termsFilter(random().nextBoolean(), new Term("field1", "b"), new Term("field1", "a"));
64      assertTrue("Must be cached", cachedFilters.contains(b));
65      //duplicate term
66      assertTrue("Must be cached", cachedFilters.contains(termsFilter(true, new Term("field1", "a"), new Term("field1", "a"), new Term("field1", "b"))));
67      assertFalse("Must not be cached", cachedFilters.contains(termsFilter(random().nextBoolean(), new Term("field1", "a"), new Term("field1", "a"), new Term("field1", "b"),  new Term("field1", "v"))));
68    }

70    public void testMissingTerms() throws Exception {
71      String fieldName = "field1";
72      Directory rd = newDirectory();
73      RandomIndexWriter w = new RandomIndexWriter(random(), rd);
74      for (int i = 0; i < 100; i++) {
75        Document doc = new Document();
76        int term = i * 10; //terms are units of 10;
77        doc.add(newStringField(fieldName, "" + term, Field.Store.YES));
78        w.addDocument(doc);
79      }
80      IndexReader reader = SlowCompositeReaderWrapper.wrap(w.getReader());
81      assertTrue(reader.getContext() instanceof LeafReaderContext);
82      LeafReaderContext context = (LeafReaderContext) reader.getContext();
83      w.close();
84  
85      List<Term> terms = new ArrayList<>();
86      terms.add(new Term(fieldName, "19"));
87      BitDocIdSet bits = (BitDocIdSet) termsFilter(random().nextBoolean(), terms).getDocIdSet(context, context.reader().getLiveDocs());
88      assertNull("Must match nothing", bits);
89  
90      terms.add(new Term(fieldName, "20"));
91      bits = (BitDocIdSet) termsFilter(random().nextBoolean(), terms).getDocIdSet(context, context.reader().getLiveDocs());
92      assertEquals("Must match 1", 1, bits.bits().cardinality());
93  
94      terms.add(new Term(fieldName, "10"));
95      bits = (BitDocIdSet) termsFilter(random().nextBoolean(), terms).getDocIdSet(context, context.reader().getLiveDocs());
96      assertEquals("Must match 2", 2, bits.bits().cardinality());
97  
98      terms.add(new Term(fieldName, "00"));
99      bits = (BitDocIdSet) termsFilter(random().nextBoolean(), terms).getDocIdSet(context, context.reader().getLiveDocs());
100     assertEquals("Must match 2", 2, bits.bits().cardinality());
101 
102     reader.close();
103     rd.close();
104   }

106   public void testMissingField() throws Exception {
107     String fieldName = "field1";
108     Directory rd1 = newDirectory();
109     RandomIndexWriter w1 = new RandomIndexWriter(random(), rd1);
110     Document doc = new Document();
111     doc.add(newStringField(fieldName, "content1", Field.Store.YES));
112     w1.addDocument(doc);
113     IndexReader reader1 = w1.getReader();
114     w1.close();
115     
116     fieldName = "field2";
117     Directory rd2 = newDirectory();
118     RandomIndexWriter w2 = new RandomIndexWriter(random(), rd2);
119     doc = new Document();
120     doc.add(newStringField(fieldName, "content2", Field.Store.YES));
121     w2.addDocument(doc);
122     IndexReader reader2 = w2.getReader();
123     w2.close();
124     
125     TermsFilter tf = new TermsFilter(new Term(fieldName, "content1"));
126     MultiReader multi = new MultiReader(reader1, reader2);
127     for (LeafReaderContext context : multi.leaves()) {
128       DocIdSet docIdSet = tf.getDocIdSet(context, context.reader().getLiveDocs());
129       if (context.reader().docFreq(new Term(fieldName, "content1")) == 0) {
130         assertNull(docIdSet);
131       } else {
132         BitDocIdSet bits = (BitDocIdSet) docIdSet;
133         assertTrue("Must be >= 0", bits.bits().cardinality() >= 0);      
134       }
135     }
136     multi.close();
137     reader1.close();
138     reader2.close();
139     rd1.close();
140     rd2.close();
141   }

143   public void testFieldNotPresent() throws IOException {
144     Directory dir = newDirectory();
145     RandomIndexWriter w = new RandomIndexWriter(random(), dir);
146     int num = atLeast(3);
147     int skip = random().nextInt(num);
148     List<Term> terms = new ArrayList<>();
149     for (int i = 0; i < num; i++) {
150       terms.add(new Term("field" + i, "content1"));
151       Document doc = new Document();
152       if (skip == i) {
153         continue;
154       }
155       doc.add(newStringField("field" + i, "content1", Field.Store.YES));
156       w.addDocument(doc);  
157     }
158     
159     w.forceMerge(1);
160     IndexReader reader = w.getReader();
161     w.close();
162     assertEquals(1, reader.leaves().size());
163     
164     
165     
166     LeafReaderContext context = reader.leaves().get(0);
167     TermsFilter tf = new TermsFilter(terms);
168 
169     BitDocIdSet bits = (BitDocIdSet) tf.getDocIdSet(context, context.reader().getLiveDocs());
170     assertEquals("Must be num fields - 1 since we skip only one field", num-1, bits.bits().cardinality());  
171     reader.close();
172     dir.close();
173   }

175   public void testSkipField() throws IOException {
176     Directory dir = newDirectory();
177     RandomIndexWriter w = new RandomIndexWriter(random(), dir);
178     int num = atLeast(10);
179     Set<Term> terms = new HashSet<>();
180     for (int i = 0; i < num; i++) {
181       String field = "field" + random().nextInt(100);
182       terms.add(new Term(field, "content1"));
183       Document doc = new Document();
184       doc.add(newStringField(field, "content1", Field.Store.YES));
185       w.addDocument(doc);
186     }
187     int randomFields = random().nextInt(10);
188     for (int i = 0; i < randomFields; i++) {
189       while (true) {
190         String field = "field" + random().nextInt(100);
191         Term t = new Term(field, "content1");
192         if (!terms.contains(t)) {
193           terms.add(t);
194           break;
195         }
196       }
197     }
198     w.forceMerge(1);
199     IndexReader reader = w.getReader();
200     w.close();
201     assertEquals(1, reader.leaves().size());
202     LeafReaderContext context = reader.leaves().get(0);
203     TermsFilter tf = new TermsFilter(new ArrayList<>(terms));
204 
205     BitDocIdSet bits = (BitDocIdSet) tf.getDocIdSet(context, context.reader().getLiveDocs());
206     assertEquals(context.reader().numDocs(), bits.bits().cardinality());  
207     reader.close();
208     dir.close();
209   }

211   public void testRandom() throws IOException {
212     Directory dir = newDirectory();
213     RandomIndexWriter w = new RandomIndexWriter(random(), dir);
214     int num = atLeast(100);
215     final boolean singleField = random().nextBoolean();
216     List<Term> terms = new ArrayList<>();
217     for (int i = 0; i < num; i++) {
218       String field = "field" + (singleField ? "1" : random().nextInt(100));
219       String string = TestUtil.randomRealisticUnicodeString(random());
220       terms.add(new Term(field, string));
221       Document doc = new Document();
222       doc.add(newStringField(field, string, Field.Store.YES));
223       w.addDocument(doc);
224     }
225     IndexReader reader = w.getReader();
226     w.close();
227     
228     IndexSearcher searcher = newSearcher(reader);
229     
230     int numQueries = atLeast(10);
231     for (int i = 0; i < numQueries; i++) {
232       Collections.shuffle(terms, random());
233       int numTerms = 1 + random().nextInt(
234           Math.min(BooleanQuery.getMaxClauseCount(), terms.size()));
235       BooleanQuery.Builder bq = new BooleanQuery.Builder();
236       for (int j = 0; j < numTerms; j++) {
237         bq.add(new BooleanClause(new TermQuery(terms.get(j)), Occur.SHOULD));
238       }
239       TopDocs queryResult = searcher.search(new ConstantScoreQuery(bq.build()), reader.maxDoc());
240       
241       MatchAllDocsQuery matchAll = new MatchAllDocsQuery();
242       final TermsFilter filter = termsFilter(singleField, terms.subList(0, numTerms));;
243       TopDocs filterResult = searcher.search(matchAll, filter, reader.maxDoc());
244       assertEquals(filterResult.totalHits, queryResult.totalHits);
245       ScoreDoc[] scoreDocs = filterResult.scoreDocs;
246       for (int j = 0; j < scoreDocs.length; j++) {
247         assertEquals(scoreDocs[j].doc, queryResult.scoreDocs[j].doc);
248       }
249     }
250     
251     reader.close();
252     dir.close();
253   }

255   private TermsFilter termsFilter(boolean singleField, Term...terms) {
256     return termsFilter(singleField, Arrays.asList(terms));
257   }
258 
259   private TermsFilter termsFilter(boolean singleField, Collection<Term> termList) {
260     if (!singleField) {
261       return new TermsFilter(new ArrayList<>(termList));
262     }
263     final TermsFilter filter;
264     List<BytesRef> bytes = new ArrayList<>();
265     String field = null;
266     for (Term term : termList) {
267         bytes.add(term.bytes());
268         if (field != null) {
269           assertEquals(term.field(), field);
270         }
271         field = term.field();
272     }
273     assertNotNull(field);
274     filter = new TermsFilter(field, bytes);
275     return filter;
276   }

278   public void testHashCodeAndEquals() {
279     int num = atLeast(100);
280     final boolean singleField = random().nextBoolean();
281     List<Term> terms = new ArrayList<>();
282     Set<Term> uniqueTerms = new HashSet<>();
283     for (int i = 0; i < num; i++) {
284       String field = "field" + (singleField ? "1" : random().nextInt(100));
285       String string = TestUtil.randomRealisticUnicodeString(random());
286       terms.add(new Term(field, string));
287       uniqueTerms.add(new Term(field, string));
288       TermsFilter left = termsFilter(singleField ? random().nextBoolean() : false, uniqueTerms);
289       Collections.shuffle(terms, random());
290       TermsFilter right = termsFilter(singleField ? random().nextBoolean() : false, terms);
291       assertEquals(right, left);
292       assertEquals(right.hashCode(), left.hashCode());
293       if (uniqueTerms.size() > 1) {
294         List<Term> asList = new ArrayList<>(uniqueTerms);
295         asList.remove(0);
296         TermsFilter notEqual = termsFilter(singleField ? random().nextBoolean() : false, asList);
297         assertFalse(left.equals(notEqual));
298         assertFalse(right.equals(notEqual));
299       }
300     }
301   }

303   public void testSingleFieldEquals() {
304     // Two terms with the same hash code
305     assertEquals("AaAaBB".hashCode(), "BBBBBB".hashCode());
306     TermsFilter left = termsFilter(true, new Term("id", "AaAaAa"), new Term("id", "AaAaBB"));
307     TermsFilter right = termsFilter(true, new Term("id", "AaAaAa"), new Term("id", "BBBBBB"));
308     assertFalse(left.equals(right));
309   }

342   public void testRamBytesUsed() {
343     List<Term> terms = new ArrayList<>();
344     final int numTerms = 1000 + random().nextInt(1000);
345     for (int i = 0; i < numTerms; ++i) {
346       terms.add(new Term("f", RandomStrings.randomUnicodeOfLength(random(), 10)));
347     }
348     TermsFilter filter = new TermsFilter(terms);
349     final long actualRamBytesUsed = RamUsageTester.sizeOf(filter);
350     final long expectedRamBytesUsed = filter.ramBytesUsed();
351     // error margin within 1%
352     assertEquals(actualRamBytesUsed, expectedRamBytesUsed, actualRamBytesUsed / 100);
