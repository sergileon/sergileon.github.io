31  /**
32   * <p>
33   * A class used to represent a set of many, potentially large, values (e.g. many
34   * long strings such as URLs), using a significantly smaller amount of memory.
35   * </p>
36   * <p>
37   * The set is "lossy" in that it cannot definitively state that is does contain
38   * a value but it <em>can</em> definitively say if a value is <em>not</em> in
39   * the set. It can therefore be used as a Bloom Filter.
40   * </p> 
41   * Another application of the set is that it can be used to perform fuzzy counting because
42   * it can estimate reasonably accurately how many unique values are contained in the set. 
43   * <p>This class is NOT threadsafe.</p>
44   * <p>
45   * Internally a Bitset is used to record values and once a client has finished recording
46   * a stream of values the {@link #downsize(float)} method can be used to create a suitably smaller set that
47   * is sized appropriately for the number of values recorded and desired saturation levels. 
48   * 
49   * </p>
50   * @lucene.experimental
51   */
52  public class FuzzySet implements Accountable {
53  
54    public static final int VERSION_SPI = 1; // HashFunction used to be loaded through a SPI
55    public static final int VERSION_START = VERSION_SPI;
56    public static final int VERSION_CURRENT = 2;
57  
58    public static HashFunction hashFunctionForVersion(int version) {
59      if (version < VERSION_START) {
60        throw new IllegalArgumentException("Version " + version + " is too old, expected at least " + VERSION_START);
61      } else if (version > VERSION_CURRENT) {
62        throw new IllegalArgumentException("Version " + version + " is too new, expected at most " + VERSION_CURRENT);
63      }
64      return MurmurHash2.INSTANCE;
65    }
66  
67    /**
68     * Result from {@link FuzzySet#contains(BytesRef)}:
69     * can never return definitively YES (always MAYBE), 
70     * but can sometimes definitely return NO.
71     */
72    public enum ContainsResult {
73      MAYBE, NO
74    };
75    private HashFunction hashFunction;
76    private FixedBitSet filter;
77    private int bloomSize;
78    
79    //The sizes of BitSet used are all numbers that, when expressed in binary form,
80    //are all ones. This is to enable fast downsizing from one bitset to another
81    //by simply ANDing each set index in one bitset with the size of the target bitset
82    // - this provides a fast modulo of the number. Values previously accumulated in
83    // a large bitset and then mapped to a smaller set can be looked up using a single
84    // AND operation of the query term's hash rather than needing to perform a 2-step
85    // translation of the query term that mirrors the stored content's reprojections.
86    static final int usableBitSetSizes[];
87    static
88    {
89      usableBitSetSizes=new int[30];
90      int mask=1;
91      int size=mask;
92      for (int i = 0; i < usableBitSetSizes.length; i++) {
93          size=(size<<1)|mask;
94          usableBitSetSizes[i]=size;
95      }    
96    }
97  
98    /**
99     * Rounds down required maxNumberOfBits to the nearest number that is made up
100    * of all ones as a binary number.  
101    * Use this method where controlling memory use is paramount.
102    */
103   public static int getNearestSetSize(int maxNumberOfBits)
104   {
105     int result=usableBitSetSizes[0];
106     for (int i = 0; i < usableBitSetSizes.length; i++) {
107       if(usableBitSetSizes[i]<=maxNumberOfBits)
108       {
109         result=usableBitSetSizes[i];
110       }
111     }
112     return result;
113   }
114 
115   /**
116    * Use this method to choose a set size where accuracy (low content saturation) is more important
117    * than deciding how much memory to throw at the problem.
118    * @param desiredSaturation A number between 0 and 1 expressing the % of bits set once all values have been recorded
119    * @return The size of the set nearest to the required size
120    */
121   public static int getNearestSetSize(int maxNumberOfValuesExpected,
122       float desiredSaturation) {
123     // Iterate around the various scales of bitset from smallest to largest looking for the first that
124     // satisfies value volumes at the chosen saturation level
125     for (int i = 0; i < usableBitSetSizes.length; i++) {
126       int numSetBitsAtDesiredSaturation = (int) (usableBitSetSizes[i] * desiredSaturation);
127       int estimatedNumUniqueValues = getEstimatedNumberUniqueValuesAllowingForCollisions(
128           usableBitSetSizes[i], numSetBitsAtDesiredSaturation);
129       if (estimatedNumUniqueValues > maxNumberOfValuesExpected) {
130         return usableBitSetSizes[i];
131       }
132     }
133     return -1;    
134   }
135   
136   public static FuzzySet createSetBasedOnMaxMemory(int maxNumBytes)
137   {
138       int setSize=getNearestSetSize(maxNumBytes);
139       return new FuzzySet(new FixedBitSet(setSize+1),setSize, hashFunctionForVersion(VERSION_CURRENT));
140   }
141   
142   public static FuzzySet createSetBasedOnQuality(int maxNumUniqueValues, float desiredMaxSaturation)
143   {
144       int setSize=getNearestSetSize(maxNumUniqueValues,desiredMaxSaturation);
145       return new FuzzySet(new FixedBitSet(setSize+1),setSize, hashFunctionForVersion(VERSION_CURRENT));
146   }

151   private FuzzySet(FixedBitSet filter, int bloomSize, HashFunction hashFunction) {
152     super();
153     this.filter = filter;
154     this.bloomSize = bloomSize;
155     this.hashFunction=hashFunction;
156   }
157   
158   /**
159    * The main method required for a Bloom filter which, given a value determines set membership.
160    * Unlike a conventional set, the fuzzy set returns NO or MAYBE rather than true or false.
161    * @return NO or MAYBE
162    */
163   public ContainsResult contains(BytesRef value) {
164     int hash = hashFunction.hash(value);
165     if (hash < 0) {
166       hash = hash * -1;
167     }
168     return mayContainValue(hash);
169   }
170   
171   /**
172    * Serializes the data set to file using the following format:
173    * <ul>
174    *  <li>FuzzySet --&gt;FuzzySetVersion,HashFunctionName,BloomSize,
175    * NumBitSetWords,BitSetWord<sup>NumBitSetWords</sup></li> 
176    * <li>HashFunctionName --&gt; {@link DataOutput#writeString(String) String} The
177    * name of a ServiceProvider registered {@link HashFunction}</li>
178    * <li>FuzzySetVersion --&gt; {@link DataOutput#writeInt Uint32} The version number of the {@link FuzzySet} class</li>
179    * <li>BloomSize --&gt; {@link DataOutput#writeInt Uint32} The modulo value used
180    * to project hashes into the field's Bitset</li>
181    * <li>NumBitSetWords --&gt; {@link DataOutput#writeInt Uint32} The number of
182    * longs (as returned from {@link FixedBitSet#getBits})</li>
183    * <li>BitSetWord --&gt; {@link DataOutput#writeLong Long} A long from the array
184    * returned by {@link FixedBitSet#getBits}</li>
185    * </ul>
186    * @param out Data output stream
187    * @throws IOException If there is a low-level I/O error
188    */
189   public void serialize(DataOutput out) throws IOException
190   {
191       out.writeInt(VERSION_CURRENT);
192       out.writeInt(bloomSize);
193       long[] bits = filter.getBits();
194       out.writeInt(bits.length);
195       for (int i = 0; i < bits.length; i++) {
196         // Can't used VLong encoding because cant cope with negative numbers
197         // output by FixedBitSet
198         out.writeLong(bits[i]);
199       }
200   }
201   public static FuzzySet deserialize(DataInput in) throws IOException
202   {
203     int version=in.readInt();
204     if (version == VERSION_SPI) {
205       in.readString();
206     }
207     final HashFunction hashFunction = hashFunctionForVersion(version);
208     int bloomSize=in.readInt();
209     int numLongs=in.readInt();
210     long[]longs=new long[numLongs];
211     for (int i = 0; i < numLongs; i++) {
212       longs[i]=in.readLong();
213     }
214     FixedBitSet bits = new FixedBitSet(longs,bloomSize+1);
215     return new FuzzySet(bits,bloomSize,hashFunction);
216   }
217   
218   private ContainsResult mayContainValue(int positiveHash) {
219     assert positiveHash >= 0;
220     // Bloom sizes are always base 2 and so can be ANDed for a fast modulo
221     int pos = positiveHash & bloomSize;
222     if (filter.get(pos)) {
223       // This term may be recorded in this index (but could be a collision)
224       return ContainsResult.MAYBE;
225     }
226     // definitely NOT in this segment
227     return ContainsResult.NO;
228   }
229   
230   /**
231    * Records a value in the set. The referenced bytes are hashed and then modulo n'd where n is the
232    * chosen size of the internal bitset.
233    * @param value the key value to be hashed
234    * @throws IOException If there is a low-level I/O error
235    */
236   public void addValue(BytesRef value) throws IOException {    
237       int hash = hashFunction.hash(value);
238       if (hash < 0) {
239         hash = hash * -1;
240       }
241       // Bitmasking using bloomSize is effectively a modulo operation.
242       int bloomPos = hash & bloomSize;
243       filter.set(bloomPos);
244   }  
245   
246   
247   /**
248    * 
249    * @param targetMaxSaturation A number between 0 and 1 describing the % of bits that would ideally be set in the 
250    * result. Lower values have better accuracy but require more space.
251    * @return a smaller FuzzySet or null if the current set is already over-saturated
252    */
253   public FuzzySet downsize(float targetMaxSaturation)
254   {
255     int numBitsSet = filter.cardinality();
256     FixedBitSet rightSizedBitSet = filter;
257     int rightSizedBitSetSize = bloomSize;
258     //Hopefully find a smaller size bitset into which we can project accumulated values while maintaining desired saturation level
259     for (int i = 0; i < usableBitSetSizes.length; i++) {
260       int candidateBitsetSize = usableBitSetSizes[i];
261       float candidateSaturation = (float) numBitsSet
262           / (float) candidateBitsetSize;
263       if (candidateSaturation <= targetMaxSaturation) {
264         rightSizedBitSetSize = candidateBitsetSize;
265         break;
266       }
267     }
268     // Re-project the numbers to a smaller space if necessary
269     if (rightSizedBitSetSize < bloomSize) {
270       // Reset the choice of bitset to the smaller version
271       rightSizedBitSet = new FixedBitSet(rightSizedBitSetSize + 1);
272       // Map across the bits from the large set to the smaller one
273       int bitIndex = 0;
274       do {
275         bitIndex = filter.nextSetBit(bitIndex);
276         if (bitIndex != DocIdSetIterator.NO_MORE_DOCS) {
277           // Project the larger number into a smaller one effectively
278           // modulo-ing by using the target bitset size as a mask
279           int downSizedBitIndex = bitIndex & rightSizedBitSetSize;
280           rightSizedBitSet.set(downSizedBitIndex);
281           bitIndex++;
282         }
283       } while ( (bitIndex >= 0)&&(bitIndex<=bloomSize));
284     } else {
285       return null;
286     }
287     return new FuzzySet(rightSizedBitSet,rightSizedBitSetSize, hashFunction);
288   }
289   
290   public int getEstimatedUniqueValues()
291   {
292        return getEstimatedNumberUniqueValuesAllowingForCollisions(bloomSize, filter.cardinality());
293   }
294   
295   // Given a set size and a the number of set bits, produces an estimate of the number of unique values recorded
296   public static int getEstimatedNumberUniqueValuesAllowingForCollisions(
297       int setSize, int numRecordedBits) {
298     double setSizeAsDouble = setSize;
299     double numRecordedBitsAsDouble = numRecordedBits;
300     double saturation = numRecordedBitsAsDouble / setSizeAsDouble;
301     double logInverseSaturation = Math.log(1 - saturation) * -1;
302     return (int) (setSizeAsDouble * logInverseSaturation);
303   }
304 
305   public float getSaturation() {
306     int numBitsSet = filter.cardinality();
307     return (float) numBitsSet / (float) bloomSize;
308   }

312     return RamUsageEstimator.sizeOf(filter.getBits());
