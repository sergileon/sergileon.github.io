43  public class LRUCache<K,V> extends SolrCacheBase implements SolrCache<K,V>, Accountable {
44    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
45  
46    static final long BASE_RAM_BYTES_USED = RamUsageEstimator.shallowSizeOfInstance(LRUCache.class);
47  
48    ///  Copied from Lucene's LRUQueryCache
49  
50    // memory usage of a simple term query
51    static final long DEFAULT_RAM_BYTES_USED = 192;
52  
53    static final long HASHTABLE_RAM_BYTES_PER_ENTRY =
54        2 * RamUsageEstimator.NUM_BYTES_OBJECT_REF // key + value
55            * 2; // hash tables need to be oversized to avoid collisions, assume 2x capacity
56  
57    static final long LINKED_HASHTABLE_RAM_BYTES_PER_ENTRY =
58        HASHTABLE_RAM_BYTES_PER_ENTRY
59            + 2 * RamUsageEstimator.NUM_BYTES_OBJECT_REF; // previous & next references
60    /// End copied code
61  
62    /* An instance of this class will be shared across multiple instances
63     * of an LRUCache at the same time.  Make sure everything is thread safe.
64     */
65    private static class CumulativeStats {
66      AtomicLong lookups = new AtomicLong();
67      AtomicLong hits = new AtomicLong();
68      AtomicLong inserts = new AtomicLong();
69      AtomicLong evictions = new AtomicLong();
70      AtomicLong evictionsRamUsage = new AtomicLong();
71    }
72  
73    private CumulativeStats stats;
74  
75    // per instance stats.  The synchronization used for the map will also be
76    // used for updating these statistics (and hence they are not AtomicLongs
77    private long lookups;
78    private long hits;
79    private long inserts;
80    private long evictions;
81    private long evictionsRamUsage;
82  
83    private long warmupTime = 0;
84  
85    private Map<K,V> map;
86    private String description="LRU Cache";
87  
88    private long maxRamBytes = Long.MAX_VALUE;
89    // The synchronization used for the map will be used to update this,
90    // hence not an AtomicLong
91    private long ramBytesUsed = 0;
92  
93    @Override
94    public Object init(Map args, Object persistence, CacheRegenerator regenerator) {
95      super.init(args, regenerator);
96      String str = (String)args.get("size");
97      final int limit = str==null ? 1024 : Integer.parseInt(str);
98      str = (String)args.get("initialSize");
99      final int initialSize = Math.min(str==null ? 1024 : Integer.parseInt(str), limit);
100     str = (String) args.get("maxRamMB");
101     final long maxRamBytes = this.maxRamBytes = str == null ? Long.MAX_VALUE : (long) (Double.parseDouble(str) * 1024L * 1024L);
102     description = generateDescription(limit, initialSize);
103 
104     map = new LinkedHashMap<K,V>(initialSize, 0.75f, true) {
105         @Override
106         protected boolean removeEldestEntry(Map.Entry eldest) {
107           if (size() > limit || ramBytesUsed > maxRamBytes) {
108             if (maxRamBytes != Long.MAX_VALUE && ramBytesUsed > maxRamBytes) {
109               long bytesToDecrement = 0;
110 
111               Iterator<Map.Entry<K, V>> iterator = entrySet().iterator();
112               do {
113                 Map.Entry<K, V> entry = iterator.next();
114                 if (entry.getKey() != null) {
115                   if (entry.getKey() instanceof Accountable) {
116                     bytesToDecrement += ((Accountable) entry.getKey()).ramBytesUsed();
117                   } else  {
118                     bytesToDecrement += DEFAULT_RAM_BYTES_USED;
119                   }
120                 }
121                 if (entry.getValue() != null) {
122                   bytesToDecrement += ((Accountable) entry.getValue()).ramBytesUsed();
123                 }
124                 bytesToDecrement += LINKED_HASHTABLE_RAM_BYTES_PER_ENTRY;
125                 ramBytesUsed -= bytesToDecrement;
126                 iterator.remove();
127                 evictions++;
128                 evictionsRamUsage++;
129                 stats.evictions.incrementAndGet();
130                 stats.evictionsRamUsage.incrementAndGet();
131               } while (iterator.hasNext() && ramBytesUsed > maxRamBytes);
132               // must return false according to javadocs of removeEldestEntry if we're modifying
133               // the map ourselves
134               return false;
135             } else  {
136               // increment evictions regardless of state.
137               // this doesn't need to be synchronized because it will
138               // only be called in the context of a higher level synchronized block.
139               evictions++;
140               stats.evictions.incrementAndGet();
141               return true;
142             }
143           }
144           return false;
145         }
146       };
147 
148     if (persistence==null) {
149       // must be the first time a cache of this type is being created
150       persistence = new CumulativeStats();
151     }
152 
153     stats = (CumulativeStats)persistence;
154 
155     return persistence;
156   }

158   /**
159    * 
160    * @return Returns the description of this cache. 
161    */
162   private String generateDescription(int limit, int initialSize) {
163     String description = "LRU Cache(maxSize=" + limit + ", initialSize=" + initialSize;
164     if (isAutowarmingOn()) {
165       description += ", " + getAutowarmDescription();
166     }
167     if (maxRamBytes != Long.MAX_VALUE)  {
168       description += ", maxRamMB=" + (maxRamBytes / 1024L / 1024L);
169     }
170     description += ')';
171     return description;
172   }

174   @Override
175   public int size() {
176     synchronized(map) {
177       return map.size();
178     }
179   }

181   @Override
182   public V put(K key, V value) {
183     synchronized (map) {
184       if (getState() == State.LIVE) {
185         stats.inserts.incrementAndGet();
186       }
187 
188       // increment local inserts regardless of state???
189       // it does make it more consistent with the current size...
190       inserts++;
191 
192       // important to calc and add new ram bytes first so that removeEldestEntry can compare correctly
193       long keySize = DEFAULT_RAM_BYTES_USED;
194       if (maxRamBytes != Long.MAX_VALUE) {
195         if (key != null && key instanceof Accountable) {
196           keySize = ((Accountable) key).ramBytesUsed();
197         }
198         long valueSize = 0;
199         if (value != null) {
200           if (value instanceof Accountable) {
201             Accountable accountable = (Accountable) value;
202             valueSize = accountable.ramBytesUsed();
203           } else {
204             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Cache: "
205                 + getName() + " is configured with maxRamBytes=" + RamUsageEstimator.humanReadableUnits(maxRamBytes)
206                 + " but its values do not implement org.apache.lucene.util.Accountable");
207           }
208         }
209         ramBytesUsed += keySize + valueSize + LINKED_HASHTABLE_RAM_BYTES_PER_ENTRY;
210       }
211       V old = map.put(key, value);
212       if (maxRamBytes != Long.MAX_VALUE && old != null) {
213         long bytesToDecrement = ((Accountable) old).ramBytesUsed();
214         // the key existed in the map but we added its size before the put, so let's back out
215         bytesToDecrement += LINKED_HASHTABLE_RAM_BYTES_PER_ENTRY;
216         if (key != null) {
217           if (key instanceof Accountable) {
218             Accountable aKey = (Accountable) key;
219             bytesToDecrement += aKey.ramBytesUsed();
220           } else {
221             bytesToDecrement += DEFAULT_RAM_BYTES_USED;
222           }
223         }
224         ramBytesUsed -= bytesToDecrement;
225       }
226       return old;
227     }
228   }
229 
230   @Override
231   public V get(K key) {
232     synchronized (map) {
233       V val = map.get(key);
234       if (getState() == State.LIVE) {
235         // only increment lookups and hits if we are live.
236         lookups++;
237         stats.lookups.incrementAndGet();
238         if (val!=null) {
239           hits++;
240           stats.hits.incrementAndGet();
241         }
242       }
243       return val;
244     }
245   }
246 
247   @Override
248   public void clear() {
249     synchronized(map) {
250       map.clear();
251       ramBytesUsed = 0;
252     }
253   }

255   @Override
256   public void warm(SolrIndexSearcher searcher, SolrCache<K,V> old) {
257     if (regenerator==null) return;
258     long warmingStartTime = System.nanoTime();
259     LRUCache<K,V> other = (LRUCache<K,V>)old;
260 
261     // warm entries
262     if (isAutowarmingOn()) {
263       Object[] keys,vals = null;
264 
265       // Don't do the autowarming in the synchronized block, just pull out the keys and values.
266       synchronized (other.map) {
267         
268         int sz = autowarm.getWarmCount(other.map.size());
269         
270         keys = new Object[sz];
271         vals = new Object[sz];
272 
273         Iterator<Map.Entry<K, V>> iter = other.map.entrySet().iterator();
274 
275         // iteration goes from oldest (least recently used) to most recently used,
276         // so we need to skip over the oldest entries.
277         int skip = other.map.size() - sz;
278         for (int i=0; i<skip; i++) iter.next();
279 
280 
281         for (int i=0; i<sz; i++) {
282           Map.Entry<K,V> entry = iter.next();
283           keys[i]=entry.getKey();
284           vals[i]=entry.getValue();
285         }
286       }
287 
288       // autowarm from the oldest to the newest entries so that the ordering will be
289       // correct in the new cache.
290       for (int i=0; i<keys.length; i++) {
291         try {
292           boolean continueRegen = regenerator.regenerateItem(searcher, this, old, keys[i], vals[i]);
293           if (!continueRegen) break;
294         }
295         catch (Exception e) {
296           SolrException.log(log,"Error during auto-warming of key:" + keys[i], e);
297         }
298       }
299     }
300 
301     warmupTime = TimeUnit.MILLISECONDS.convert(System.nanoTime() - warmingStartTime, TimeUnit.NANOSECONDS);
302   }

328   @Override
329   public NamedList getStatistics() {
330     NamedList lst = new SimpleOrderedMap();
331     synchronized (map) {
332       lst.add("lookups", lookups);
333       lst.add("hits", hits);
334       lst.add("hitratio", calcHitRatio(lookups,hits));
335       lst.add("inserts", inserts);
336       lst.add("evictions", evictions);
337       lst.add("size", map.size());
338       if (maxRamBytes != Long.MAX_VALUE)  {
339         lst.add("maxRamMB", maxRamBytes / 1024L / 1024L);
340         lst.add("ramBytesUsed", ramBytesUsed());
341         lst.add("evictionsRamUsage", evictionsRamUsage);
342       }
343     }
344     lst.add("warmupTime", warmupTime);
345     
346     long clookups = stats.lookups.get();
347     long chits = stats.hits.get();
348     lst.add("cumulative_lookups", clookups);
349     lst.add("cumulative_hits", chits);
350     lst.add("cumulative_hitratio", calcHitRatio(clookups, chits));
351     lst.add("cumulative_inserts", stats.inserts.get());
352     lst.add("cumulative_evictions", stats.evictions.get());
353     if (maxRamBytes != Long.MAX_VALUE)  {
354       lst.add("cumulative_evictionsRamUsage", stats.evictionsRamUsage.get());
355     }
356     
357     return lst;
358   }

372   @Override
373   public Collection<Accountable> getChildResources() {
374     if (maxRamBytes != Long.MAX_VALUE)  {
375       synchronized (map)  {
376         return Accountables.namedAccountables(getName(), (Map<?, ? extends Accountable>) map);
377       }
378     } else  {
379       return Collections.emptyList();
