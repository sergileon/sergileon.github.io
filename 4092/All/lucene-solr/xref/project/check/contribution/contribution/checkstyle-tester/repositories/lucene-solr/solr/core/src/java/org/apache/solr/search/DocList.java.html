21  /**
22   * <code>DocList</code> represents the result of a query: an ordered list of document ids with optional score.
23   * This list contains a subset of the complete list of documents actually matched: <code>size()</code>
24   * document ids starting at <code>offset()</code>.
25   *
26   *
27   * @since solr 0.9
28   */
29  public interface DocList extends DocSet {
30  
31    /**
32     * Returns the zero based offset of this list within the total ordered list of matches to the query.
33     */
34    public int offset();
35  
36    /**
37     * Returns the number of ids in this list.
38     */
39    @Override
40    public int size();
41  
42    /**
43     * Returns the total number of matches for the search
44     * (as opposed to just the number collected according
45     * to <code>offset()</code> and <code>size()</code>).
46     * Hence it's always true that matches() &gt;= size()
47     * @return number of matches for the search(query &amp; any filters)
48     */
49    public int matches();
50  
51  
52    /***
53    public int getDoc(int pos);
54    ***/
55  
56    // hmmm, what if a different slice could be generated from an existing DocSet
57    // (and was before)...
58  
59    // how to distinguish cached values from logical values?
60    // docSet could represent docs 10-20, but actually contain 0-100
61    // should the big slice be cached independently, and a new class called
62    // DocListSubset be created to refer to a range within the DocList?
63  
64    /**
65     * Get a subset of an existing DocList.
66     * Returns null if not possible.
67     */
68    public DocList subset(int offset, int len);
69  
70    /**
71     * Returns an iterator that may be used to iterate over the documents in this DocList
72     *
73     * <p>
74     * The order of the documents returned by this iterator is based on the
75     * Sort order of the search that produced it.  The Scoring information
76     * is meaningful only if <code>hasScores()</code> returns true.
77     * </p>
78     * @see #hasScores
79     */
80    @Override
81    public DocIterator iterator();
82      
83    /** True if scores were retained */
84    public boolean hasScores();
85  
86    /** The maximum score for the search... only valid if
87     * scores were retained (if hasScores()==true)
88     */
89    public float maxScore();
90  }
