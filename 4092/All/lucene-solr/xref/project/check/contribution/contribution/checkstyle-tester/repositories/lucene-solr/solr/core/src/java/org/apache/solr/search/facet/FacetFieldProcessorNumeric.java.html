37  class FacetFieldProcessorNumeric extends FacetFieldProcessor {
38    static int MAXIMUM_STARTING_TABLE_SIZE=1024;  // must be a power of two, non-final to support setting by tests
39  
40    static class LongCounts {
41  
42      static final float LOAD_FACTOR = 0.7f;
43  
44      long numAdds;
45      long[] vals;
46      int[] counts;  // maintain the counts here since we need them to tell if there was actually a value anyway
47      int[] oldToNewMapping;
48  
49      int cardinality;
50      int threshold;
51  
52      /** sz must be a power of two */
53      LongCounts(int sz) {
54        vals = new long[sz];
55        counts = new int[sz];
56        threshold = (int) (sz * LOAD_FACTOR);
57      }
58  
59      /** Current number of slots in the hash table */
60      public int numSlots() {
61        return vals.length;
62      }
63  
64      private int hash(long val) {
65        // For floats: exponent bits start at bit 23 for single precision,
66        // and bit 52 for double precision.
67        // Many values will only have significant bits just to the right of that,
68        // and the leftmost bits will all be zero.
69  
70        // For now, lets just settle to get first 8 significant mantissa bits of double or float in the lowest bits of our hash
71        // The upper bits of our hash will be irrelevant.
72        int h = (int) (val + (val >>> 44) + (val >>> 15));
73        return h;
74      }
75  
76      /** returns the slot */
77      int add(long val) {
78        if (cardinality >= threshold) {
79          rehash();
80        }
81  
82        numAdds++;
83        int h = hash(val);
84        for (int slot = h & (vals.length-1);  ;slot = (slot + ((h>>7)|1)) & (vals.length-1)) {
85          int count = counts[slot];
86          if (count == 0) {
87            counts[slot] = 1;
88            vals[slot] = val;
89            cardinality++;
90            return slot;
91          } else if (vals[slot] == val) {
92            // val is already in the set
93            counts[slot] = count + 1;
94            return slot;
95          }
96        }
97      }
98  
99      protected void rehash() {
100       long[] oldVals = vals;
101       int[] oldCounts = counts;  // after retrieving the count, this array is reused as a mapping to new array
102       int newCapacity = vals.length << 1;
103       vals = new long[newCapacity];
104       counts = new int[newCapacity];
105       threshold = (int) (newCapacity * LOAD_FACTOR);
106 
107       for (int i=0; i<oldVals.length; i++) {
108         int count = oldCounts[i];
109         if (count == 0) {
110           oldCounts[i] = -1;
111           continue;
112         }
113 
114         long val = oldVals[i];
115 
116         int h = hash(val);
117         int slot = h & (vals.length-1);
118         while (counts[slot] != 0) {
119           slot = (slot + ((h>>7)|1)) & (vals.length-1);
120         }
121         counts[slot] = count;
122         vals[slot] = val;
123         oldCounts[i] = slot;
124       }
125 
126       oldToNewMapping = oldCounts;
127     }
128 
129     int cardinality() {
130       return cardinality;
131     }
132 
133   }

137   FacetFieldProcessorNumeric(FacetContext fcontext, FacetField freq, SchemaField sf) {
138     super(fcontext, freq, sf);
139   }

141   int allBucketsSlot = -1;
142 
143   @Override
144   public void process() throws IOException {
145     super.process();
146     response = calcFacets();
147   }
148 
149   private void doRehash(LongCounts table) {
150     if (collectAcc == null && allBucketsAcc == null) return;
151 
152     // Our "count" acc is backed by the hash table and will already be rehashed
153     // otherAccs don't need to be rehashed
154 
155     int newTableSize = table.numSlots();
156     int numSlots = newTableSize;
157     final int oldAllBucketsSlot = allBucketsSlot;
158     if (oldAllBucketsSlot >= 0) {
159       allBucketsSlot = numSlots++;
160     }
161 
162     final int finalNumSlots = numSlots;
163     final int[] mapping = table.oldToNewMapping;
164 
165     SlotAcc.Resizer resizer = new SlotAcc.Resizer() {
166       @Override
167       public int getNewSize() {
168         return finalNumSlots;
169       }
170 
171       @Override
172       public int getNewSlot(int oldSlot) {
173         if (oldSlot < mapping.length) {
174           return mapping[oldSlot];
175         }
176         if (oldSlot == oldAllBucketsSlot) {
177           return allBucketsSlot;
178         }
179         return -1;
180       }
181     };
182 
183     // NOTE: resizing isn't strictly necessary for missing/allBuckets... we could just set the new slot directly
184     if (collectAcc != null) {
185       collectAcc.resize(resizer);
186     }
187     if (allBucketsAcc != null) {
188       allBucketsAcc.resize(resizer);
189     }
190   }
191 
192   public SimpleOrderedMap<Object> calcFacets() throws IOException {
193 
194 
195     final FacetRangeProcessor.Calc calc = FacetRangeProcessor.getNumericCalc(sf);
196 
197 
198     // TODO: it would be really nice to know the number of unique values!!!!
199 
200     int possibleValues = fcontext.base.size();
201     // size smaller tables so that no resize will be necessary
202     int currHashSize = BitUtil.nextHighestPowerOfTwo((int) (possibleValues * (1 / LongCounts.LOAD_FACTOR) + 1));
203     currHashSize = Math.min(currHashSize, MAXIMUM_STARTING_TABLE_SIZE);
204     final LongCounts table = new LongCounts(currHashSize) {
205       @Override
206       protected void rehash() {
207         super.rehash();
208         doRehash(this);
209         oldToNewMapping = null; // allow for gc
210       }
211     };
212 
213     int numSlots = currHashSize;
214 
215     int numMissing = 0;
216 
217 
218     if (freq.allBuckets) {
219       allBucketsSlot = numSlots++;
220     }
221 
222     indexOrderAcc = new SlotAcc(fcontext) {
223       @Override
224       public void collect(int doc, int slot) throws IOException {
225       }
226 
227       @Override
228       public int compare(int slotA, int slotB) {
229         long s1 = calc.bitsToSortableBits(table.vals[slotA]);
230         long s2 = calc.bitsToSortableBits(table.vals[slotB]);
231         return Long.compare(s1, s2);
232       }
233 
234       @Override
235       public Object getValue(int slotNum) throws IOException {
236         return null;
237       }
238 
239       @Override
240       public void reset() {
241       }
242 
243       @Override
244       public void resize(Resizer resizer) {
245       }
246     };
247 
248     countAcc = new CountSlotAcc(fcontext) {
249       @Override
250       public void incrementCount(int slot, int count) {
251         throw new UnsupportedOperationException();
252       }
253 
254       @Override
255       public int getCount(int slot) {
256         return table.counts[slot];
257       }
258 
259       @Override
260       public Object getValue(int slotNum) {
261         return getCount(slotNum);
262       }
263 
264       @Override
265       public void reset() {
266         throw new UnsupportedOperationException();
267       }
268 
269       @Override
270       public void collect(int doc, int slot) throws IOException {
271         throw new UnsupportedOperationException();
272       }
273 
274       @Override
275       public int compare(int slotA, int slotB) {
276         return Integer.compare( table.counts[slotA], table.counts[slotB] );
277       }
278 
279       @Override
280       public void resize(Resizer resizer) {
281         throw new UnsupportedOperationException();
282       }
283     };
284 
285     // we set the countAcc & indexAcc first so generic ones won't be created for us.
286     createCollectAcc(fcontext.base.size(), numSlots);
287 
288     if (freq.allBuckets) {
289       allBucketsAcc = new SpecialSlotAcc(fcontext, collectAcc, allBucketsSlot, otherAccs, 0);
290     }
291 
292     NumericDocValues values = null;
293     Bits docsWithField = null;
294 
295     // TODO: factor this code out so it can be shared...
296     final List<LeafReaderContext> leaves = fcontext.searcher.getIndexReader().leaves();
297     final Iterator<LeafReaderContext> ctxIt = leaves.iterator();
298     LeafReaderContext ctx = null;
299     int segBase = 0;
300     int segMax;
301     int adjustedMax = 0;
302     for (DocIterator docsIt = fcontext.base.iterator(); docsIt.hasNext(); ) {
303       final int doc = docsIt.nextDoc();
304       if (doc >= adjustedMax) {
305         do {
306           ctx = ctxIt.next();
307           segBase = ctx.docBase;
308           segMax = ctx.reader().maxDoc();
309           adjustedMax = segBase + segMax;
310         } while (doc >= adjustedMax);
311         assert doc >= ctx.docBase;
312         setNextReaderFirstPhase(ctx);
313 
314         values = DocValues.getNumeric(ctx.reader(), sf.getName());
315         docsWithField = DocValues.getDocsWithField(ctx.reader(), sf.getName());
316       }
317 
318       int segDoc = doc - segBase;
319       long val = values.get(segDoc);
320       if (val != 0 || docsWithField.get(segDoc)) {
321         int slot = table.add(val);  // this can trigger a rehash rehash
322 
323         // countAcc.incrementCount(slot, 1);
324         // our countAcc is virtual, so this is not needed
325 
326         collectFirstPhase(segDoc, slot);
327       }
328     }
329 
330 
331     //
332     // collection done, time to find the top slots
333     //
334 
335     int numBuckets = 0;
336     List<Object> bucketVals = null;
337     if (freq.numBuckets && fcontext.isShard()) {
338       bucketVals = new ArrayList(100);
339     }
340 
341     int off = fcontext.isShard() ? 0 : (int) freq.offset;
342     // add a modest amount of over-request if this is a shard request
343     int lim = freq.limit >= 0 ? (fcontext.isShard() ? (int)(freq.limit*1.1+4) : (int)freq.limit) : Integer.MAX_VALUE;
344 
345     int maxsize = (int)(freq.limit >= 0 ?  freq.offset + lim : Integer.MAX_VALUE - 1);
346     maxsize = Math.min(maxsize, table.cardinality);
347 
348     final int sortMul = freq.sortDirection.getMultiplier();
349 
350     PriorityQueue<Slot> queue = new PriorityQueue<Slot>(maxsize) {
351       @Override
352       protected boolean lessThan(Slot a, Slot b) {
353         // TODO: sort-by-index-order
354         int cmp = sortAcc.compare(a.slot, b.slot) * sortMul;
355         return cmp == 0 ? (indexOrderAcc.compare(a.slot, b.slot) > 0) : cmp < 0;
356       }
357     };
358 
359     // TODO: create a countAcc that wrapps the table so we can reuse more code?
360 
361     Slot bottom = null;
362     for (int i=0; i<table.counts.length; i++) {
363       int count = table.counts[i];
364       if (count < effectiveMincount) {
365         // either not a valid slot, or count not high enough
366         continue;
367       }
368       numBuckets++;  // can be different from the table cardinality if mincount > 1
369 
370       long val = table.vals[i];
371       if (bucketVals != null && bucketVals.size()<100) {
372         bucketVals.add( calc.bitsToValue(val) );
373       }
374 
375       if (bottom == null) {
376         bottom = new Slot();
377       }
378       bottom.slot = i;
379 
380       bottom = queue.insertWithOverflow(bottom);
381     }
382 
383 
384     SimpleOrderedMap res = new SimpleOrderedMap();
385     if (freq.numBuckets) {
386       if (!fcontext.isShard()) {
387         res.add("numBuckets", numBuckets);
388       } else {
389         SimpleOrderedMap map = new SimpleOrderedMap(2);
390         map.add("numBuckets", numBuckets);
391         map.add("vals", bucketVals);
392         res.add("numBuckets", map);
393       }
394     }
395 
396     if (freq.allBuckets) {
397       SimpleOrderedMap<Object> allBuckets = new SimpleOrderedMap<>();
398       // countAcc.setValues(allBuckets, allBucketsSlot);
399       allBuckets.add("count", table.numAdds);
400       allBucketsAcc.setValues(allBuckets, -1);
401       // allBuckets currently doesn't execute sub-facets (because it doesn't change the domain?)
402       res.add("allBuckets", allBuckets);
403     }
404 
405     if (freq.missing) {
406       // TODO: it would be more efficient to buid up a missing DocSet if we need it here anyway.
407 
408       SimpleOrderedMap<Object> missingBucket = new SimpleOrderedMap<>();
409       fillBucket(missingBucket, getFieldMissingQuery(fcontext.searcher, freq.field), null);
410       res.add("missing", missingBucket);
411     }
412 
413     // if we are deep paging, we don't have to order the highest "offset" counts.
414     int collectCount = Math.max(0, queue.size() - off);
415     assert collectCount <= lim;
416     int[] sortedSlots = new int[collectCount];
417     for (int i = collectCount - 1; i >= 0; i--) {
418       sortedSlots[i] = queue.pop().slot;
419     }
420 
421     ArrayList bucketList = new ArrayList(collectCount);
422     res.add("buckets", bucketList);
423 
424     boolean needFilter = deferredAggs != null || freq.getSubFacets().size() > 0;
425 
426     for (int slotNum : sortedSlots) {
427       SimpleOrderedMap<Object> bucket = new SimpleOrderedMap<>();
428       Comparable val = calc.bitsToValue(table.vals[slotNum]);
429       bucket.add("val", val);
430 
431       Query filter = needFilter ? sf.getType().getFieldQuery(null, sf, calc.formatValue(val)) : null;
432 
433       fillBucket(bucket, table.counts[slotNum], slotNum, null, filter);
434 
435       bucketList.add(bucket);
436     }
437 
438 
439 
440     return res;
441   }
