49  /**
50   * A collection of segmentInfo objects with methods for operating on those
51   * segments in relation to the file system.
52   * <p>
53   * The active segments in the index are stored in the segment info file,
54   * <tt>segments_N</tt>. There may be one or more <tt>segments_N</tt> files in
55   * the index; however, the one with the largest generation is the active one
56   * (when older segments_N files are present it's because they temporarily cannot
57   * be deleted, or a custom {@link IndexDeletionPolicy} is in
58   * use). This file lists each segment by name and has details about the codec
59   * and generation of deletes.
60   * </p>
61   * <p>
62   * Files:
63   * <ul>
64   * <li><tt>segments_N</tt>: Header, LuceneVersion, Version, NameCounter, SegCount, MinSegmentLuceneVersion, &lt;SegName,
65   * HasSegID, SegID, SegCodec, DelGen, DeletionCount, FieldInfosGen, DocValuesGen,
66   * UpdatesFiles&gt;<sup>SegCount</sup>, CommitUserData, Footer
67   * </ul>
68   * Data types:
69   * <ul>
70   * <li>Header --&gt; {@link CodecUtil#writeIndexHeader IndexHeader}</li>
71   * <li>LuceneVersion --&gt; Which Lucene code {@link Version} was used for this commit, written as three {@link DataOutput#writeVInt vInt}: major, minor, bugfix
72   * <li>MinSegmentLuceneVersion --&gt; Lucene code {@link Version} of the oldest segment, written as three {@link DataOutput#writeVInt vInt}: major, minor, bugfix; this is only
73   *   written only if there's at least one segment
74   * <li>NameCounter, SegCount, DeletionCount --&gt;
75   * {@link DataOutput#writeInt Int32}</li>
76   * <li>Generation, Version, DelGen, Checksum, FieldInfosGen, DocValuesGen --&gt;
77   * {@link DataOutput#writeLong Int64}</li>
78   * <li>HasSegID --&gt; {@link DataOutput#writeByte Int8}</li>
79   * <li>SegID --&gt; {@link DataOutput#writeByte Int8<sup>ID_LENGTH</sup>}</li>
80   * <li>SegName, SegCodec --&gt; {@link DataOutput#writeString String}</li>
81   * <li>CommitUserData --&gt; {@link DataOutput#writeMapOfStrings
82   * Map&lt;String,String&gt;}</li>
83   * <li>UpdatesFiles --&gt; Map&lt;{@link DataOutput#writeInt Int32},
84   * {@link DataOutput#writeSetOfStrings(Set) Set&lt;String&gt;}&gt;</li>
85   * <li>Footer --&gt; {@link CodecUtil#writeFooter CodecFooter}</li>
86   * </ul>
87   * Field Descriptions:
88   * <ul>
89   * <li>Version counts how often the index has been changed by adding or deleting
90   * documents.</li>
91   * <li>NameCounter is used to generate names for new segment files.</li>
92   * <li>SegName is the name of the segment, and is used as the file name prefix
93   * for all of the files that compose the segment's index.</li>
94   * <li>DelGen is the generation count of the deletes file. If this is -1, there
95   * are no deletes. Anything above zero means there are deletes stored by
96   * {@link LiveDocsFormat}.</li>
97   * <li>DeletionCount records the number of deleted documents in this segment.</li>
98   * <li>SegCodec is the {@link Codec#getName() name} of the Codec that encoded
99   * this segment.</li>
100  * <li>HasSegID is nonzero if the segment has an identifier. Otherwise, when it is 0
101  * the identifier is {@code null} and no SegID is written. Null only happens for Lucene
102  * 4.x segments referenced in commits.</li>
103  * <li>SegID is the identifier of the Codec that encoded this segment. </li>
104  * <li>CommitUserData stores an optional user-supplied opaque
105  * Map&lt;String,String&gt; that was passed to
106  * {@link IndexWriter#setCommitData(java.util.Map)}.</li>
107  * <li>FieldInfosGen is the generation count of the fieldInfos file. If this is
108  * -1, there are no updates to the fieldInfos in that segment. Anything above
109  * zero means there are updates to fieldInfos stored by {@link FieldInfosFormat}
110  * .</li>
111  * <li>DocValuesGen is the generation count of the updatable DocValues. If this
112  * is -1, there are no updates to DocValues in that segment. Anything above zero
113  * means there are updates to DocValues stored by {@link DocValuesFormat}.</li>
114  * <li>UpdatesFiles stores the set of files that were updated in that segment
115  * per field.</li>
116  * </ul>
117  * 
118  * @lucene.experimental
119  */
120 public final class SegmentInfos implements Cloneable, Iterable<SegmentCommitInfo> {

189   /**
190    * Get the generation of the most recent commit to the
191    * list of index files (N in the segments_N file).
192    *
193    * @param files -- array of file names to check
194    */
195   public static long getLastCommitGeneration(String[] files) {
196     long max = -1;
197     for (String file : files) {
198       if (file.startsWith(IndexFileNames.SEGMENTS) && !file.equals(IndexFileNames.OLD_SEGMENTS_GEN)) {
199         long gen = generationFromSegmentsFileName(file);
200         if (gen > max) {
201           max = gen;
202         }
203       }
204     }
205     return max;
206   }

252   /**
253    * Parse the generation off the segments file name and
254    * return it.
255    */
256   public static long generationFromSegmentsFileName(String fileName) {
257     if (fileName.equals(IndexFileNames.SEGMENTS)) {
258       return 0;
259     } else if (fileName.startsWith(IndexFileNames.SEGMENTS)) {
260       return Long.parseLong(fileName.substring(1+IndexFileNames.SEGMENTS.length()),
261                             Character.MAX_RADIX);
262     } else {
263       throw new IllegalArgumentException("fileName \"" + fileName + "\" is not a segments file");
264     }
265   }

267   /** return generation of the next pending_segments_N that will be written */
268   private long getNextPendingGeneration() {
269     if (generation == -1) {
270       return 1;
271     } else {
272       return generation+1;
273     }
274   }

282   /**
283    * Read a particular segmentFileName.  Note that this may
284    * throw an IOException if a commit is in process.
285    *
286    * @param directory -- directory containing the segments file
287    * @param segmentFileName -- segment file to load
288    * @throws CorruptIndexException if the index is corrupt
289    * @throws IOException if there is a low-level IO error
290    */
291   public static final SegmentInfos readCommit(Directory directory, String segmentFileName) throws IOException {
292 
293     long generation = generationFromSegmentsFileName(segmentFileName);
294     try (ChecksumIndexInput input = directory.openChecksumInput(segmentFileName, IOContext.READ)) {
295       // NOTE: as long as we want to throw indexformattooold (vs corruptindexexception), we need
296       // to read the magic ourselves.
297       int magic = input.readInt();
298       if (magic != CodecUtil.CODEC_MAGIC) {
299         throw new IndexFormatTooOldException(input, magic, CodecUtil.CODEC_MAGIC, CodecUtil.CODEC_MAGIC);
300       }
301       // 4.0+
302       int format = CodecUtil.checkHeaderNoMagic(input, "segments", VERSION_40, VERSION_CURRENT);
303       // 5.0+
304       byte id[] = null;
305       if (format >= VERSION_50) {
306         id = new byte[StringHelper.ID_LENGTH];
307         input.readBytes(id, 0, id.length);
308         CodecUtil.checkIndexHeaderSuffix(input, Long.toString(generation, Character.MAX_RADIX));
309       }
310       
311       SegmentInfos infos = new SegmentInfos();
312       infos.id = id;
313       infos.generation = generation;
314       infos.lastGeneration = generation;
315       if (format >= VERSION_53) {
316         // TODO: in the future (7.0?  sigh) we can use this to throw IndexFormatTooOldException ... or just rely on the
317         // minSegmentLuceneVersion check instead:
318         infos.luceneVersion = Version.fromBits(input.readVInt(), input.readVInt(), input.readVInt());
319       } else {
320         // else compute the min version down below in the for loop
321       }
322 
323       infos.version = input.readLong();
324       infos.counter = input.readInt();
325       int numSegments = input.readInt();
326       if (numSegments < 0) {
327         throw new CorruptIndexException("invalid segment count: " + numSegments, input);
328       }
329 
330       if (format >= VERSION_53) {
331         if (numSegments > 0) {
332           infos.minSegmentLuceneVersion = Version.fromBits(input.readVInt(), input.readVInt(), input.readVInt());
333           if (infos.minSegmentLuceneVersion.onOrAfter(Version.LUCENE_4_0_0_ALPHA) == false) {
334             throw new IndexFormatTooOldException(input, "this index contains a too-old segment (version: " + infos.minSegmentLuceneVersion + ")");
335           }
336         } else {
337           // else leave as null: no segments
338         }
339       } else {
340         // else we recompute it below as we visit segments; it can't be used for throwing IndexFormatTooOldExc, but consumers of
341         // SegmentInfos can maybe still use it for other reasons
342       }
343 
344       long totalDocs = 0;
345       for (int seg = 0; seg < numSegments; seg++) {
346         String segName = input.readString();
347         final byte segmentID[];
348         if (format >= VERSION_50) {
349           byte hasID = input.readByte();
350           if (hasID == 1) {
351             segmentID = new byte[StringHelper.ID_LENGTH];
352             input.readBytes(segmentID, 0, segmentID.length);
353           } else if (hasID == 0) {
354             segmentID = null; // 4.x segment, doesn't have an ID
355           } else {
356             throw new CorruptIndexException("invalid hasID byte, got: " + hasID, input);
357           }
358         } else {
359           segmentID = null;
360         }
361         Codec codec = readCodec(input, format < VERSION_53);
362         SegmentInfo info = codec.segmentInfoFormat().read(directory, segName, segmentID, IOContext.READ);
363         info.setCodec(codec);
364         totalDocs += info.maxDoc();
365         long delGen = input.readLong();
366         int delCount = input.readInt();
367         if (delCount < 0 || delCount > info.maxDoc()) {
368           throw new CorruptIndexException("invalid deletion count: " + delCount + " vs maxDoc=" + info.maxDoc(), input);
369         }
370         long fieldInfosGen = -1;
371         if (format >= VERSION_46) {
372           fieldInfosGen = input.readLong();
373         }
374         long dvGen = -1;
375         if (format >= VERSION_49) {
376           dvGen = input.readLong();
377         } else {
378           dvGen = fieldInfosGen;
379         }
380         SegmentCommitInfo siPerCommit = new SegmentCommitInfo(info, delCount, delGen, fieldInfosGen, dvGen);
381         if (format >= VERSION_46) {
382           if (format < VERSION_49) {
383             // Recorded per-generation files, which were buggy (see
384             // LUCENE-5636). We need to read and keep them so we continue to
385             // reference those files. Unfortunately it means that the files will
386             // be referenced even if the fields are updated again, until the
387             // segment is merged.
388             final int numGensUpdatesFiles = input.readInt();
389             final Map<Long,Set<String>> genUpdatesFiles;
390             if (numGensUpdatesFiles == 0) {
391               genUpdatesFiles = Collections.emptyMap();
392             } else {
393               genUpdatesFiles = new HashMap<>(numGensUpdatesFiles);
394               for (int i = 0; i < numGensUpdatesFiles; i++) {
395                 genUpdatesFiles.put(input.readLong(), input.readStringSet());
396               }
397             }
398             siPerCommit.setGenUpdatesFiles(genUpdatesFiles);
399           } else {
400             if (format >= VERSION_51) {
401               siPerCommit.setFieldInfosFiles(input.readSetOfStrings());
402             } else {
403               siPerCommit.setFieldInfosFiles(Collections.unmodifiableSet(input.readStringSet()));
404             }
405             final Map<Integer,Set<String>> dvUpdateFiles;
406             final int numDVFields = input.readInt();
407             if (numDVFields == 0) {
408               dvUpdateFiles = Collections.emptyMap();
409             } else {
410               Map<Integer,Set<String>> map = new HashMap<>(numDVFields);
411               for (int i = 0; i < numDVFields; i++) {
412                 if (format >= VERSION_51) {
413                   map.put(input.readInt(), input.readSetOfStrings());
414                 } else {
415                   map.put(input.readInt(), Collections.unmodifiableSet(input.readStringSet()));
416                 }
417               }
418               dvUpdateFiles = Collections.unmodifiableMap(map);
419             }
420             siPerCommit.setDocValuesUpdatesFiles(dvUpdateFiles);
421           }
422         }
423         infos.add(siPerCommit);
424 
425         Version segmentVersion = info.getVersion();
426         if (format < VERSION_53) {
427           if (infos.minSegmentLuceneVersion == null || segmentVersion.onOrAfter(infos.minSegmentLuceneVersion) == false) {
428             infos.minSegmentLuceneVersion = segmentVersion;
429           }
430         } else if (segmentVersion.onOrAfter(infos.minSegmentLuceneVersion) == false) {
431           throw new CorruptIndexException("segments file recorded minSegmentLuceneVersion=" + infos.minSegmentLuceneVersion + " but segment=" + info + " has older version=" + segmentVersion, input);
432         }
433       }
434 
435       if (format >= VERSION_51) {
436         infos.userData = input.readMapOfStrings();
437       } else {
438         infos.userData = Collections.unmodifiableMap(input.readStringStringMap());
439       }
440 
441       if (format >= VERSION_48) {
442         CodecUtil.checkFooter(input);
443       } else {
444         final long checksumNow = input.getChecksum();
445         final long checksumThen = input.readLong();
446         if (checksumNow != checksumThen) {
447           throw new CorruptIndexException("checksum failed (hardware problem?) : expected=" + Long.toHexString(checksumThen) +  
448                                           " actual=" + Long.toHexString(checksumNow), input);
449         }
450         CodecUtil.checkEOF(input);
451       }
452 
453       // LUCENE-6299: check we are in bounds
454       if (totalDocs > IndexWriter.getActualMaxDocs()) {
455         throw new CorruptIndexException("Too many documents: an index cannot exceed " + IndexWriter.getActualMaxDocs() + " but readers have total maxDoc=" + totalDocs, input);
456       }
457       
458       return infos;
459     }
460   }

462   private static final List<String> unsupportedCodecs = Arrays.asList(
463       "Lucene3x"
464   );

466   private static Codec readCodec(DataInput input, boolean unsupportedAllowed) throws IOException {
467     final String name = input.readString();
468     try {
469       return Codec.forName(name);
470     } catch (IllegalArgumentException e) {
471       // give better error messages if we can, first check if this is a legacy codec
472       if (unsupportedCodecs.contains(name)) {
473         // We should only get here on pre-5.3 indices, but we can't test this until 7.0 when 5.x indices become too old:
474         assert unsupportedAllowed;
475         IOException newExc = new IndexFormatTooOldException(input, "Codec '" + name + "' is too old");
476         newExc.initCause(e);
477         throw newExc;
478       }
479       // or maybe it's an old default codec that moved
480       if (name.startsWith("Lucene")) {
481         throw new IllegalArgumentException("Could not load codec '" + name + "'.  Did you forget to add lucene-backward-codecs.jar?", e);
482       }
483       throw e;
484     }
485   }

489   public static final SegmentInfos readLatestCommit(Directory directory) throws IOException {

502   private void write(Directory directory) throws IOException {
503 
504     long nextGeneration = getNextPendingGeneration();
505     String segmentFileName = IndexFileNames.fileNameFromGeneration(IndexFileNames.PENDING_SEGMENTS,
506                                                                    "",
507                                                                    nextGeneration);
508     
509     // Always advance the generation on write:
510     generation = nextGeneration;
511     
512     IndexOutput segnOutput = null;
513     boolean success = false;
514 
515     try {
516       segnOutput = directory.createOutput(segmentFileName, IOContext.DEFAULT);
517       CodecUtil.writeIndexHeader(segnOutput, "segments", VERSION_CURRENT, 
518                                  StringHelper.randomId(), Long.toString(nextGeneration, Character.MAX_RADIX));
519       segnOutput.writeVInt(Version.LATEST.major);
520       segnOutput.writeVInt(Version.LATEST.minor);
521       segnOutput.writeVInt(Version.LATEST.bugfix);
522 
523       segnOutput.writeLong(version); 
524       segnOutput.writeInt(counter); // write counter
525       segnOutput.writeInt(size());
526 
527       if (size() > 0) {
528 
529         Version minSegmentVersion = null;
530 
531         // We do a separate loop up front so we can write the minSegmentVersion before
532         // any SegmentInfo; this makes it cleaner to throw IndexFormatTooOldExc at read time:
533         for (SegmentCommitInfo siPerCommit : this) {
534           Version segmentVersion = siPerCommit.info.getVersion();
535           if (minSegmentVersion == null || segmentVersion.onOrAfter(minSegmentVersion) == false) {
536             minSegmentVersion = segmentVersion;
537           }
538         }
539 
540         segnOutput.writeVInt(minSegmentVersion.major);
541         segnOutput.writeVInt(minSegmentVersion.minor);
542         segnOutput.writeVInt(minSegmentVersion.bugfix);
543       }
544 
545       // write infos
546       for (SegmentCommitInfo siPerCommit : this) {
547         SegmentInfo si = siPerCommit.info;
548         segnOutput.writeString(si.name);
549         byte segmentID[] = si.getId();
550         // TODO: remove this in lucene 6, we don't need to include 4.x segments in commits anymore
551         if (segmentID == null) {
552           segnOutput.writeByte((byte)0);
553         } else {
554           if (segmentID.length != StringHelper.ID_LENGTH) {
555             throw new IllegalStateException("cannot write segment: invalid id segment=" + si.name + "id=" + StringHelper.idToString(segmentID));
556           }
557           segnOutput.writeByte((byte)1);
558           segnOutput.writeBytes(segmentID, segmentID.length);
559         }
560         segnOutput.writeString(si.getCodec().getName());
561         segnOutput.writeLong(siPerCommit.getDelGen());
562         int delCount = siPerCommit.getDelCount();
563         if (delCount < 0 || delCount > si.maxDoc()) {
564           throw new IllegalStateException("cannot write segment: invalid maxDoc segment=" + si.name + " maxDoc=" + si.maxDoc() + " delCount=" + delCount);
565         }
566         segnOutput.writeInt(delCount);
567         segnOutput.writeLong(siPerCommit.getFieldInfosGen());
568         segnOutput.writeLong(siPerCommit.getDocValuesGen());
569         segnOutput.writeSetOfStrings(siPerCommit.getFieldInfosFiles());
570         final Map<Integer,Set<String>> dvUpdatesFiles = siPerCommit.getDocValuesUpdatesFiles();
571         segnOutput.writeInt(dvUpdatesFiles.size());
572         for (Entry<Integer,Set<String>> e : dvUpdatesFiles.entrySet()) {
573           segnOutput.writeInt(e.getKey());
574           segnOutput.writeSetOfStrings(e.getValue());
575         }
576       }
577       segnOutput.writeMapOfStrings(userData);
578       CodecUtil.writeFooter(segnOutput);
579       segnOutput.close();
580       directory.sync(Collections.singleton(segmentFileName));
581       success = true;
582     } finally {
583       if (success) {
584         pendingCommit = true;
585       } else {
586         // We hit an exception above; try to close the file
587         // but suppress any exception:
588         IOUtils.closeWhileHandlingException(segnOutput);
589         // Try not to leave a truncated segments_N file in
590         // the index:
591         IOUtils.deleteFilesIgnoringExceptions(directory, segmentFileName);
592       }
593     }
594   }

602   public SegmentInfos clone() {
603     try {
604       final SegmentInfos sis = (SegmentInfos) super.clone();

606       sis.segments = new ArrayList<>(size());
607       for(final SegmentCommitInfo info : this) {

612       sis.userData = new HashMap<>(userData);
613       return sis;
614     } catch (CloneNotSupportedException e) {
615       throw new RuntimeException("should not happen", e);
616     }

662   /**
663    * Utility class for executing code that needs to do
664    * something with the current segments file.  This is
665    * necessary with lock-less commits because from the time
666    * you locate the current segments file name, until you
667    * actually open it, read its contents, or check modified
668    * time, etc., it could have been deleted due to a writer
669    * commit finishing.
670    */
671   public abstract static class FindSegmentsFile<T> {
672 
673     final Directory directory;
674 
675     /** Sole constructor. */ 
676     public FindSegmentsFile(Directory directory) {
677       this.directory = directory;
678     }
679 
680     /** Locate the most recent {@code segments} file and
681      *  run {@link #doBody} on it. */
682     public T run() throws IOException {
683       return run(null);
684     }
685     
686     /** Run {@link #doBody} on the provided commit. */
687     public T run(IndexCommit commit) throws IOException {
688       if (commit != null) {
689         if (directory != commit.getDirectory())
690           throw new IOException("the specified commit does not match the specified Directory");
691         return doBody(commit.getSegmentsFileName());
692       }
693 
694       long lastGen = -1;
695       long gen = -1;
696       IOException exc = null;
697 
698       // Loop until we succeed in calling doBody() without
699       // hitting an IOException.  An IOException most likely
700       // means an IW deleted our commit while opening
701       // the time it took us to load the now-old infos files
702       // (and segments files).  It's also possible it's a
703       // true error (corrupt index).  To distinguish these,
704       // on each retry we must see "forward progress" on
705       // which generation we are trying to load.  If we
706       // don't, then the original error is real and we throw
707       // it.
708       
709       for (;;) {
710         lastGen = gen;
711         String files[] = directory.listAll();
712         String files2[] = directory.listAll();
713         Arrays.sort(files);
714         Arrays.sort(files2);
715         if (!Arrays.equals(files, files2)) {
716           // listAll() is weakly consistent, this means we hit "concurrent modification exception"
717           continue;
718         }
719         gen = getLastCommitGeneration(files);
720         
721         if (infoStream != null) {
722           message("directory listing gen=" + gen);
723         }
724         
725         if (gen == -1) {
726           throw new IndexNotFoundException("no segments* file found in " + directory + ": files: " + Arrays.toString(files));
727         } else if (gen > lastGen) {
728           String segmentFileName = IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS, "", gen);
729         
730           try {
731             T t = doBody(segmentFileName);
732             if (infoStream != null) {
733               message("success on " + segmentFileName);
734             }
735             return t;
736           } catch (IOException err) {
737             // Save the original root cause:
738             if (exc == null) {
739               exc = err;
740             }
741 
742             if (infoStream != null) {
743               message("primary Exception on '" + segmentFileName + "': " + err + "'; will retry: gen = " + gen);
744             }
745           }
746         } else {
747           throw exc;
748         }
749       }
750     }
751 
752     /**
753      * Subclass must implement this.  The assumption is an
754      * IOException will be thrown if something goes wrong
755      * during the processing that could have been caused by
756      * a writer committing.
757      */
758     protected abstract T doBody(String segmentFileName) throws IOException;
759   }

779   final void rollbackCommit(Directory dir) {
780     if (pendingCommit) {
781       pendingCommit = false;
782       
783       // we try to clean up our pending_segments_N
784 
785       // Must carefully compute fileName from "generation"
786       // since lastGeneration isn't incremented:
787       final String pending = IndexFileNames.fileNameFromGeneration(IndexFileNames.PENDING_SEGMENTS, "", generation);
788 
789       // Suppress so we keep throwing the original exception
790       // in our caller
791       IOUtils.deleteFilesIgnoringExceptions(dir, pending);
792     }
793   }

795   /** Call this to start a commit.  This writes the new
796    *  segments file, but writes an invalid checksum at the
797    *  end, so that it is not visible to readers.  Once this
798    *  is called you must call {@link #finishCommit} to complete
799    *  the commit or {@link #rollbackCommit} to abort it.
800    *  <p>
801    *  Note: {@link #changed()} should be called prior to this
802    *  method if changes have been made to this {@link SegmentInfos} instance
803    *  </p>  
804    **/
805   final void prepareCommit(Directory dir) throws IOException {
806     if (pendingCommit) {
807       throw new IllegalStateException("prepareCommit was already called");
808     }
809     write(dir);
810   }

821   /** Returns all file names referenced by SegmentInfo.
822    *  The returned collection is recomputed on each
823    *  invocation.  */
824   public Collection<String> files(boolean includeSegmentsFile) throws IOException {
825     HashSet<String> files = new HashSet<>();
826     if (includeSegmentsFile) {
827       final String segmentFileName = getSegmentsFileName();
828       if (segmentFileName != null) {
829         files.add(segmentFileName);
830       }
831     }
832     final int size = size();
833     for(int i=0;i<size;i++) {
834       final SegmentCommitInfo info = info(i);

841   /** Returns the committed segments_N filename. */
842   final String finishCommit(Directory dir) throws IOException {
843     if (pendingCommit == false) {
844       throw new IllegalStateException("prepareCommit was not called");
845     }
846     boolean success = false;
847     final String dest;
848     try {
849       final String src =  IndexFileNames.fileNameFromGeneration(IndexFileNames.PENDING_SEGMENTS, "", generation);
850       dest = IndexFileNames.fileNameFromGeneration(IndexFileNames.SEGMENTS, "", generation);
851       dir.renameFile(src, dest);
852       success = true;
853     } finally {
854       if (!success) {
855         // deletes pending_segments_N:
856         rollbackCommit(dir);
857       }
858     }
859 
860     pendingCommit = false;
861     lastGeneration = generation;
862     return dest;
863   }

869    *  method if changes have been made to this {@link SegmentInfos} instance
870    *  </p>  
871    **/
872   final void commit(Directory dir) throws IOException {
873     prepareCommit(dir);

947   /** applies all changes caused by committing a merge to this SegmentInfos */
948   void applyMergeChanges(MergePolicy.OneMerge merge, boolean dropSegment) {
949     final Set<SegmentCommitInfo> mergedAway = new HashSet<>(merge.segments);
950     boolean inserted = false;
951     int newSegIdx = 0;
952     for (int segIdx = 0, cnt = segments.size(); segIdx < cnt; segIdx++) {
953       assert segIdx >= newSegIdx;
954       final SegmentCommitInfo info = segments.get(segIdx);
955       if (mergedAway.contains(info)) {
956         if (!inserted && !dropSegment) {
957           segments.set(segIdx, merge.info);
958           inserted = true;
959           newSegIdx++;
960         }
961       } else {
962         segments.set(newSegIdx, info);
963         newSegIdx++;
964       }
965     }
966 
967     // the rest of the segments in list are duplicates, so don't remove from map, only list!
968     segments.subList(newSegIdx, segments.size()).clear();
969     
970     // Either we found place to insert segment, or, we did
971     // not, but only because all segments we merged becamee
972     // deleted while we are merging, in which case it should
973     // be the case that the new segment is also all deleted,
974     // we insert it at the beginning if it should not be dropped:
975     if (!inserted && !dropSegment) {
976       segments.add(0, merge.info);
977     }
978   }
