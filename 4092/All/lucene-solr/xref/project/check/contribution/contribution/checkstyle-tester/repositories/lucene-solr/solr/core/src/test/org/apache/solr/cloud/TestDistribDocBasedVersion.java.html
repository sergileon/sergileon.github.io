245   void vdelete(String id, long version, String... params) throws Exception {
246     UpdateRequest req = new UpdateRequest();
247     req.deleteById(id);
248     req.setParam("del_version", Long.toString(version));
249     for (int i=0; i<params.length; i+=2) {
250       req.setParam( params[i], params[i+1]);
251     }
252     solrClient.request(req);
253     // req.process(cloudClient);
254   }

265   void vaddFail(String id, long version, int errCode, String... params) throws Exception {
266     boolean failed = false;
267     try {
268       vadd(id, version, params);
269     } catch (SolrException e) {
270       failed = true;
271       assertEquals(errCode, e.code());

273       log.error("ERROR", e);

292   void doQuery(String expectedDocs, String... queryParams) throws Exception {
293 
294     List<String> strs = StrUtils.splitSmart(expectedDocs, ",", true);
295     Map<String, Object> expectedIds = new HashMap<>();
296     for (int i=0; i<strs.size(); i+=2) {
297       String id = strs.get(i);
298       String vS = strs.get(i+1);
299       Long v = Long.valueOf(vS);
300       expectedIds.put(id,v);
301     }
302 
303     QueryResponse rsp = cloudClient.query(params(queryParams));
304     Map<String, Object> obtainedIds = new HashMap<>();
305     for (SolrDocument doc : rsp.getResults()) {
306       obtainedIds.put((String) doc.get("id"), doc.get(vfield));
307     }
308 
309     assertEquals(expectedIds, obtainedIds);
310   }

313   void doRTG(String ids, String versions) throws Exception {
314     Map<String, Object> expectedIds = new HashMap<>();
315     List<String> strs = StrUtils.splitSmart(ids, ",", true);
316     List<String> verS = StrUtils.splitSmart(versions, ",", true);
317     for (int i=0; i<strs.size(); i++) {
318       expectedIds.put(strs.get(i), Long.valueOf(verS.get(i)));
319     }
320 
321     solrClient.query(params("qt", "/get", "ids", ids));
322 
323     QueryResponse rsp = cloudClient.query(params("qt","/get", "ids",ids));
324     Map<String, Object> obtainedIds = new HashMap<>();
325     for (SolrDocument doc : rsp.getResults()) {
326       obtainedIds.put((String) doc.get("id"), doc.get(vfield));
327     }
328 
329     assertEquals(expectedIds, obtainedIds);
330   }

332   void doRTG(String ids) throws Exception {
333     solrClient.query(params("qt", "/get", "ids", ids));
334 
335     Set<String> expectedIds = new HashSet<>( StrUtils.splitSmart(ids, ",", true) );
336 
337     QueryResponse rsp = cloudClient.query(params("qt","/get", "ids",ids));
338     Set<String> obtainedIds = new HashSet<>();
339     for (SolrDocument doc : rsp.getResults()) {
340       obtainedIds.add((String) doc.get("id"));
341     }
342 
343     assertEquals(expectedIds, obtainedIds);
344   }
