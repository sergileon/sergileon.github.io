209   @Override
210   protected void doAfterWriter(final ExecutorService es) throws Exception {
211     final double minReopenSec = 0.01 + 0.05 * random().nextDouble();
212     final double maxReopenSec = minReopenSec * (1.0 + 10 * random().nextDouble());
213 
214     if (VERBOSE) {
215       System.out.println("TEST: make SearcherManager maxReopenSec=" + maxReopenSec + " minReopenSec=" + minReopenSec);
216     }
217 
218     genWriter = new TrackingIndexWriter(writer);
219 
220     final SearcherFactory sf = new SearcherFactory() {
221         @Override
222         public IndexSearcher newSearcher(IndexReader r, IndexReader previous) throws IOException {
223           TestControlledRealTimeReopenThread.this.warmCalled = true;
224           IndexSearcher s = new IndexSearcher(r, es);
225           s.search(new TermQuery(new Term("body", "united")), 10);
226           return s;
227         }
228       };
229 
230     nrtNoDeletes = new SearcherManager(writer, false, sf);
231     nrtDeletes = new SearcherManager(writer, true, sf);
232                          
233     nrtDeletesThread = new ControlledRealTimeReopenThread<>(genWriter, nrtDeletes, maxReopenSec, minReopenSec);
234     nrtDeletesThread.setName("NRTDeletes Reopen Thread");
235     nrtDeletesThread.setPriority(Math.min(Thread.currentThread().getPriority()+2, Thread.MAX_PRIORITY));
236     nrtDeletesThread.setDaemon(true);
237     nrtDeletesThread.start();
238 
239     nrtNoDeletesThread = new ControlledRealTimeReopenThread<>(genWriter, nrtNoDeletes, maxReopenSec, minReopenSec);
240     nrtNoDeletesThread.setName("NRTNoDeletes Reopen Thread");
241     nrtNoDeletesThread.setPriority(Math.min(Thread.currentThread().getPriority()+2, Thread.MAX_PRIORITY));
242     nrtNoDeletesThread.setDaemon(true);
243     nrtNoDeletesThread.start();
244   }

303   public void testThreadStarvationNoDeleteNRTReader() throws IOException, InterruptedException {
304     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
305     conf.setMergePolicy(NoMergePolicy.INSTANCE);
306     Directory d = newDirectory();
307     final CountDownLatch latch = new CountDownLatch(1);
308     final CountDownLatch signal = new CountDownLatch(1);
309 
310     LatchedIndexWriter _writer = new LatchedIndexWriter(d, conf, latch, signal);
311     final TrackingIndexWriter writer = new TrackingIndexWriter(_writer);
312     final SearcherManager manager = new SearcherManager(_writer, false, null);
313     Document doc = new Document();
314     doc.add(newTextField("test", "test", Field.Store.YES));
315     writer.addDocument(doc);
316     manager.maybeRefresh();
317     Thread t = new Thread() {
318       @Override
319       public void run() {
320         try {
321           signal.await();
322           manager.maybeRefresh();
323           writer.deleteDocuments(new TermQuery(new Term("foo", "barista")));
324           manager.maybeRefresh(); // kick off another reopen so we inc. the internal gen
325         } catch (Exception e) {
326           e.printStackTrace();
327         } finally {
328           latch.countDown(); // let the add below finish
329         }
330       }
331     };
332     t.start();
333     _writer.waitAfterUpdate = true; // wait in addDocument to let some reopens go through
334     final long lastGen = writer.updateDocument(new Term("foo", "bar"), doc); // once this returns the doc is already reflected in the last reopen
335 
336     assertFalse(manager.isSearcherCurrent()); // false since there is a delete in the queue
337     
338     IndexSearcher searcher = manager.acquire();
339     try {
340       assertEquals(2, searcher.getIndexReader().numDocs());
341     } finally {
342       manager.release(searcher);
343     }
344     final ControlledRealTimeReopenThread<IndexSearcher> thread = new ControlledRealTimeReopenThread<>(writer, manager, 0.01, 0.01);
345     thread.start(); // start reopening
346     if (VERBOSE) {
347       System.out.println("waiting now for generation " + lastGen);
348     }
349     
350     final AtomicBoolean finished = new AtomicBoolean(false);
351     Thread waiter = new Thread() {
352       @Override
353       public void run() {
354         try {
355           thread.waitForGeneration(lastGen);
356         } catch (InterruptedException ie) {
357           Thread.currentThread().interrupt();
358           throw new RuntimeException(ie);
359         }
360         finished.set(true);
361       }
362     };
363     waiter.start();
364     manager.maybeRefresh();
365     waiter.join(1000);
366     if (!finished.get()) {
367       waiter.interrupt();
368       fail("thread deadlocked on waitForGeneration");
369     }
370     thread.close();
371     thread.join();
372     _writer.close();
373     IOUtils.close(manager, d);
374   }

376   public static class LatchedIndexWriter extends IndexWriter {
377 
378     private CountDownLatch latch;
379     boolean waitAfterUpdate = false;
380     private CountDownLatch signal;
381 
382     public LatchedIndexWriter(Directory d, IndexWriterConfig conf,
383         CountDownLatch latch, CountDownLatch signal)
384         throws IOException {
385       super(d, conf);
386       this.latch = latch;
387       this.signal = signal;
388 
389     }
390 
391     @Override
392     public void updateDocument(Term term,
393         Iterable<? extends IndexableField> doc)
394         throws IOException {
395       super.updateDocument(term, doc);
396       try {
397         if (waitAfterUpdate) {
398           signal.countDown();
399           latch.await();
400         }
401       } catch (InterruptedException e) {
402         throw new ThreadInterruptedException(e);
403       }
404     }
405   }

458   // Relies on wall clock time, so it can easily false-fail when the machine is otherwise busy:
459   @AwaitsFix(bugUrl = "https://issues.apache.org/jira/browse/LUCENE-5737")
460   // LUCENE-5461
461   public void testCRTReopen() throws Exception {
462     //test behaving badly
463 
464     //should be high enough
465     int maxStaleSecs = 20;
466 
467     //build crap data just to store it.
468     String s = "        abcdefghijklmnopqrstuvwxyz     ";
469     char[] chars = s.toCharArray();
470     StringBuilder builder = new StringBuilder(2048);
471     for (int i = 0; i < 2048; i++) {
472       builder.append(chars[random().nextInt(chars.length)]);
473     }
474     String content = builder.toString();
475 
476     final SnapshotDeletionPolicy sdp = new SnapshotDeletionPolicy(new KeepOnlyLastCommitDeletionPolicy());
477     final Directory dir = new NRTCachingDirectory(newFSDirectory(createTempDir("nrt")), 5, 128);
478     IndexWriterConfig config = new IndexWriterConfig(new MockAnalyzer(random()));
479     config.setCommitOnClose(true);
480     config.setIndexDeletionPolicy(sdp);
481     config.setOpenMode(IndexWriterConfig.OpenMode.CREATE_OR_APPEND);
482     final IndexWriter iw = new IndexWriter(dir, config);
483     SearcherManager sm = new SearcherManager(iw, true, new SearcherFactory());
484     final TrackingIndexWriter tiw = new TrackingIndexWriter(iw);
485     ControlledRealTimeReopenThread<IndexSearcher> controlledRealTimeReopenThread =
486       new ControlledRealTimeReopenThread<>(tiw, sm, maxStaleSecs, 0);
487 
488     controlledRealTimeReopenThread.setDaemon(true);
489     controlledRealTimeReopenThread.start();
490 
491     List<Thread> commitThreads = new ArrayList<>();
492 
493     for (int i = 0; i < 500; i++) {
494       if (i > 0 && i % 50 == 0) {
495         Thread commitThread =  new Thread(new Runnable() {
496             @Override
497             public void run() {
498               try {
499                 iw.commit();
500                 IndexCommit ic = sdp.snapshot();
501                 for (String name : ic.getFileNames()) {
502                   //distribute, and backup
503                   //System.out.println(names);
504                   assertTrue(slowFileExists(dir, name));
505                 }
506               } catch (Exception e) {
507                 throw new RuntimeException(e);
508               }
509             }
510           });
511         commitThread.start();
512         commitThreads.add(commitThread);
513       }
514       Document d = new Document();
515       d.add(new TextField("count", i + "", Field.Store.NO));
516       d.add(new TextField("content", content, Field.Store.YES));
517       long start = System.currentTimeMillis();
518       long l = tiw.addDocument(d);
519       controlledRealTimeReopenThread.waitForGeneration(l);
520       long wait = System.currentTimeMillis() - start;
521       assertTrue("waited too long for generation " + wait,
522                  wait < (maxStaleSecs *1000));
523       IndexSearcher searcher = sm.acquire();
524       TopDocs td = searcher.search(new TermQuery(new Term("count", i + "")), 10);
525       sm.release(searcher);
526       assertEquals(1, td.totalHits);
527     }
528 
529     for(Thread commitThread : commitThreads) {
530       commitThread.join();
531     }
532 
533     controlledRealTimeReopenThread.close();
534     sm.close();
535     iw.close();
536     dir.close();
537   }
