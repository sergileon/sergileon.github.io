62    @BeforeClass
63    public static void beforeClass() throws Exception {
64      initCore("solrconfig-basic.xml","schema-analytics.xml");
65      h.update("<delete><query>*:*</query></delete>");
66      defaults.put("int_id", new Integer(0));
67      defaults.put("long_ld", new Long(0));
68      defaults.put("float_fd", new Float(0));
69      defaults.put("double_dd", new Double(0));
70      defaults.put("date_dtd", "1800-12-31T23:59:59Z");
71      defaults.put("string_sd", "str0");
72      
73      intTestStart = new ArrayList<>();
74      longTestStart = new ArrayList<>();
75      floatTestStart = new ArrayList<>();
76      doubleTestStart = new ArrayList<>();
77      dateTestStart = new ArrayList<>();
78      stringTestStart = new ArrayList<>();
79      
80      for (int j = 0; j < NUM_LOOPS; ++j) {
81        int i = j%INT;
82        long l = j%LONG;
83        float f = j%FLOAT;
84        double d = j%DOUBLE;
85        String dt = (1800+j%DATE) + "-12-31T23:59:59Z";
86        String s = "str" + (j%STRING);
87        List<String> fields = new ArrayList<>();
88        fields.add("id"); fields.add("1000"+j);
89        
90        if( i != 0 ){
91          fields.add("int_id"); fields.add("" + i);
92          intTestStart.add(i);
93        } else intMissing++;
94        
95        if( l != 0l ){
96          fields.add("long_ld"); fields.add("" + l);
97          longTestStart.add(l);
98        } else longMissing++;
99        
100       if( f != 0.0f ){
101         fields.add("float_fd"); fields.add("" + f);
102         floatTestStart.add(f);
103       } else floatMissing++;
104       
105       if( d != 0.0d ){
106         fields.add("double_dd"); fields.add("" + d);
107         doubleTestStart.add(d);
108       } else doubleMissing++;
109       
110       if( (j%DATE) != 0 ){
111         fields.add("date_dtd"); fields.add(dt);
112         dateTestStart.add(dt);
113       } else dateMissing++;
114       
115       if( (j%STRING) != 0 ){
116         fields.add("string_sd"); fields.add(s);
117         stringTestStart.add(s);
118       } else stringMissing++;
119       
120       fields.add("int_i"); fields.add("" + i);
121       fields.add("long_l"); fields.add("" + l);
122       fields.add("float_f"); fields.add("" + f);
123       fields.add("double_d"); fields.add("" + d);
124       
125       assertU(adoc(fields.toArray(new String[0])));
126       
127       
128       if (usually()) {
129         assertU(commit());  // to have several segments
130       }
131     }
132     
133     assertU(commit()); 
134     
135     //Sort ascending tests
136     setResponse(h.query(request(fileToStringArr(NoFacetTest.class, fileName))));
137   }
