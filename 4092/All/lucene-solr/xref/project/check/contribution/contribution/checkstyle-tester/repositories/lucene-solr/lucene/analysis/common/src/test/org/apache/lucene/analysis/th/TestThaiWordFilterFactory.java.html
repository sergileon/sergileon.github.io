20  import java.io.Reader;
21  import java.io.StringReader;
22  
23  import org.apache.lucene.analysis.MockTokenizer;
24  import org.apache.lucene.analysis.TokenStream;
25  import org.apache.lucene.analysis.th.ThaiWordFilter;
26  import org.apache.lucene.analysis.util.BaseTokenStreamFactoryTestCase;
27  
28  /**
29   * Simple tests to ensure the Thai word filter factory is working.
30   */
31  @Deprecated
32  public class TestThaiWordFilterFactory extends BaseTokenStreamFactoryTestCase {
33    /**
34     * Ensure the filter actually decomposes text.
35     */
36    public void testWordBreak() throws Exception {
37      assumeTrue("JRE does not support Thai dictionary-based BreakIterator", ThaiWordFilter.DBBI_AVAILABLE);
38      Reader reader = new StringReader("การที่ได้ต้องแสดงว่างานดี");
39      TokenStream stream = whitespaceMockTokenizer(reader);
40      stream = tokenFilterFactory("ThaiWord").create(stream);
41      assertTokenStreamContents(stream, new String[] {"การ", "ที่", "ได้",
42          "ต้อง", "แสดง", "ว่า", "งาน", "ดี"});
43    }
44    
45    /** Test that bogus arguments result in exception */
46    public void testBogusArguments() throws Exception {
47      assumeTrue("JRE does not support Thai dictionary-based BreakIterator", ThaiWordFilter.DBBI_AVAILABLE);
48      try {
49        tokenFilterFactory("ThaiWord", "bogusArg", "bogusValue");
50        fail();
51      } catch (IllegalArgumentException expected) {
52        assertTrue(expected.getMessage().contains("Unknown parameters"));
53      }
54    }
55  }
