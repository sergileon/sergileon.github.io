33  /**
34   * Traverses a {@link SpatialPrefixTree} indexed field, using the template and
35   * visitor design patterns for subclasses to guide the traversal and collect
36   * matching documents.
37   * <p>
38   * Subclasses implement {@link #getDocIdSet(org.apache.lucene.index.LeafReaderContext)}
39   * by instantiating a custom {@link VisitorTemplate} subclass (i.e. an anonymous inner class)
40   * and implement the required methods.
41   *
42   * @lucene.internal
43   */
44  public abstract class AbstractVisitingPrefixTreeQuery extends AbstractPrefixTreeQuery {
45  
46    //Historical note: this code resulted from a refactoring of RecursivePrefixTreeQuery,
47    // which in turn came out of SOLR-2155
48  
49    //This class perhaps could have been implemented in terms of FilteredTermsEnum & MultiTermQuery.
50    //  Maybe so for simple Intersects predicate but not for when we want to collect terms
51    //  differently depending on cell state like IsWithin and for fuzzy/accurate collection planned improvements.  At
52    //  least it would just make things more complicated.
53  
54    protected final int prefixGridScanLevel;//at least one less than grid.getMaxLevels()
55  
56    public AbstractVisitingPrefixTreeQuery(Shape queryShape, String fieldName, SpatialPrefixTree grid,
57                                           int detailLevel, int prefixGridScanLevel) {
58      super(queryShape, fieldName, grid, detailLevel);
59      this.prefixGridScanLevel = Math.max(0, Math.min(prefixGridScanLevel, grid.getMaxLevels() - 1));
60      assert detailLevel <= grid.getMaxLevels();
61    }
62  
63    /**
64     * An abstract class designed to make it easy to implement predicates or
65     * other operations on a {@link SpatialPrefixTree} indexed field. An instance
66     * of this class is not designed to be re-used across LeafReaderContext
67     * instances so simply create a new one per-leaf.
68     * The {@link #getDocIdSet()} method here starts the work. It first checks
69     * that there are indexed terms; if not it quickly returns null. Then it calls
70     * {@link #start()} so a subclass can set up a return value, like an
71     * {@link org.apache.lucene.util.FixedBitSet}. Then it starts the traversal
72     * process, calling {@link #findSubCellsToVisit(org.apache.lucene.spatial.prefix.tree.Cell)}
73     * which by default finds the top cells that intersect {@code queryShape}. If
74     * there isn't an indexed cell for a corresponding cell returned for this
75     * method then it's short-circuited until it finds one, at which point
76     * {@link #visitPrefix(org.apache.lucene.spatial.prefix.tree.Cell)} is called. At
77     * some depths, of the tree, the algorithm switches to a scanning mode that
78     * calls {@link #visitScanned(org.apache.lucene.spatial.prefix.tree.Cell)}
79     * for each leaf cell found.
80     *
81     * @lucene.internal
82     */
83    public abstract class VisitorTemplate extends BaseTermsEnumTraverser {
84  
85    /* Future potential optimizations:
86  
87    * Can a polygon query shape be optimized / made-simpler at recursive depths
88      (e.g. intersection of shape + cell box)
89  
90    * RE "scan" vs divide & conquer performance decision:
91      We should use termsEnum.docFreq() as an estimate on the number of places at
92      this depth.  It would be nice if termsEnum knew how many terms
93      start with the current term without having to repeatedly next() & test to find out.
94  
95    * Perhaps don't do intermediate seek()'s to cells above detailLevel that have Intersects
96      relation because we won't be collecting those docs any way.  However seeking
97      does act as a short-circuit.  So maybe do some percent of the time or when the level
98      is above some threshold.
99  
100   */

103     //  TODO MAJOR REFACTOR SIMPLIFICATION BASED ON TreeCellIterator  TODO

106     private VNode curVNode;//current pointer, derived from query shape
107     private BytesRef curVNodeTerm = new BytesRef();//curVNode.cell's term, without leaf. in main loop only
108 
109     private BytesRef thisTerm;//the result of termsEnum.term()
110     private Cell indexedCell;//Cell wrapper of thisTerm. Always updated when thisTerm is.
111 
112     public VisitorTemplate(LeafReaderContext context) throws IOException {
113       super(context);
114     }
115 
116     public DocIdSet getDocIdSet() throws IOException {
117       assert curVNode == null : "Called more than once?";
118       if (termsEnum == null)
119         return null;
120       if (!nextTerm()) {//advances
121         return null;
122       }
123 
124       curVNode = new VNode(null);
125       curVNode.reset(grid.getWorldCell());
126 
127       start();
128 
129       addIntersectingChildren();
130 
131       main: while (thisTerm != null) {//terminates for other reasons too!
132 
133         //Advance curVNode pointer
134         if (curVNode.children != null) {
135           //-- HAVE CHILDREN: DESCEND
136           assert curVNode.children.hasNext();//if we put it there then it has something
137           preSiblings(curVNode);
138           curVNode = curVNode.children.next();
139         } else {
140           //-- NO CHILDREN: ADVANCE TO NEXT SIBLING
141           VNode parentVNode = curVNode.parent;
142           while (true) {
143             if (parentVNode == null)
144               break main; // all done
145             if (parentVNode.children.hasNext()) {
146               //advance next sibling
147               curVNode = parentVNode.children.next();
148               break;
149             } else {
150               //reached end of siblings; pop up
151               postSiblings(parentVNode);
152               parentVNode.children = null;//GC
153               parentVNode = parentVNode.parent;
154             }
155           }
156         }
157 
158         //Seek to curVNode's cell (or skip if termsEnum has moved beyond)
159         final int compare = indexedCell.compareToNoLeaf(curVNode.cell);
160         if (compare > 0) {
161           // The indexed cell is after; continue loop to next query cell
162           continue;
163         }
164         if (compare < 0) {
165           // The indexed cell is before; seek ahead to query cell:
166           //      Seek !
167           curVNode.cell.getTokenBytesNoLeaf(curVNodeTerm);
168           TermsEnum.SeekStatus seekStatus = termsEnum.seekCeil(curVNodeTerm);
169           if (seekStatus == TermsEnum.SeekStatus.END)
170             break; // all done
171           thisTerm = termsEnum.term();
172           indexedCell = grid.readCell(thisTerm, indexedCell);
173           if (seekStatus == TermsEnum.SeekStatus.NOT_FOUND) {
174             // Did we find a leaf of the cell we were looking for or something after?
175             if (!indexedCell.isLeaf() || indexedCell.compareToNoLeaf(curVNode.cell) != 0)
176               continue; // The indexed cell is after; continue loop to next query cell
177           }
178         }
179         // indexedCell == queryCell (disregarding leaf).
180 
181         // If indexedCell is a leaf then there's no prefix (prefix sorts before) -- just visit and continue
182         if (indexedCell.isLeaf()) {
183           visitLeaf(indexedCell);//TODO or query cell? Though shouldn't matter.
184           if (!nextTerm()) break;
185           continue;
186         }
187         // If a prefix (non-leaf) then visit; see if we descend.
188         final boolean descend = visitPrefix(curVNode.cell);//need to use curVNode.cell not indexedCell
189         if (!nextTerm()) break;
190         // Check for adjacent leaf with the same prefix
191         if (indexedCell.isLeaf() && indexedCell.getLevel() == curVNode.cell.getLevel()) {
192           visitLeaf(indexedCell);//TODO or query cell? Though shouldn't matter.
193           if (!nextTerm()) break;
194         }
195 
196 
197         if (descend) {
198           addIntersectingChildren();
199         }
200 
201       }//main loop
202 
203       return finish();
204     }
205 
206     /** Called initially, and whenever {@link #visitPrefix(org.apache.lucene.spatial.prefix.tree.Cell)}
207      * returns true. */
208     private void addIntersectingChildren() throws IOException {
209       assert thisTerm != null;
210       Cell cell = curVNode.cell;
211       if (cell.getLevel() >= detailLevel)
212         throw new IllegalStateException("Spatial logic error");
213 
214       //Decide whether to continue to divide & conquer, or whether it's time to
215       // scan through terms beneath this cell.
216       // Scanning is a performance optimization trade-off.
217 
218       //TODO use termsEnum.docFreq() as heuristic
219       boolean scan = cell.getLevel() >= prefixGridScanLevel;//simple heuristic
220 
221       if (!scan) {
222         //Divide & conquer (ultimately termsEnum.seek())
223 
224         Iterator<Cell> subCellsIter = findSubCellsToVisit(cell);
225         if (!subCellsIter.hasNext())//not expected
226           return;
227         curVNode.children = new VNodeCellIterator(subCellsIter, new VNode(curVNode));
228 
229       } else {
230         //Scan (loop of termsEnum.next())
231 
232         scan(detailLevel);
233       }
234     }
235 
236     /**
237      * Called when doing a divide and conquer to find the next intersecting cells
238      * of the query shape that are beneath {@code cell}. {@code cell} is
239      * guaranteed to have an intersection and thus this must return some number
240      * of nodes.
241      */
242     protected CellIterator findSubCellsToVisit(Cell cell) {
243       return cell.getNextLevelCells(queryShape);
244     }
245 
246     /**
247      * Scans ({@code termsEnum.next()}) terms until a term is found that does
248      * not start with curVNode's cell. If it finds a leaf cell or a cell at
249      * level {@code scanDetailLevel} then it calls {@link
250      * #visitScanned(org.apache.lucene.spatial.prefix.tree.Cell)}.
251      */
252     protected void scan(int scanDetailLevel) throws IOException {
253       //note: this can be a do-while instead in 6x; 5x has a back-compat with redundant leaves -- LUCENE-4942
254       while (curVNode.cell.isPrefixOf(indexedCell)) {
255         if (indexedCell.getLevel() == scanDetailLevel
256             || (indexedCell.getLevel() < scanDetailLevel && indexedCell.isLeaf())) {
257           visitScanned(indexedCell);
258         }
259         //advance
260         if (!nextTerm()) break;
261       }
262     }
263 
264     private boolean nextTerm() throws IOException {
265       if ((thisTerm = termsEnum.next()) == null)
266         return false;
267       indexedCell = grid.readCell(thisTerm, indexedCell);
268       return true;
269     }

271     /** Used for {@link VNode#children}. */
272     private class VNodeCellIterator implements Iterator<VNode> {
273 
274       final Iterator<Cell> cellIter;
275       private final VNode vNode;
276 
277       VNodeCellIterator(Iterator<Cell> cellIter, VNode vNode) {
278         this.cellIter = cellIter;
279         this.vNode = vNode;
280       }
281 
282       @Override
283       public boolean hasNext() {
284         return cellIter.hasNext();
285       }
286 
287       @Override
288       public VNode next() {
289         assert hasNext();
290         vNode.reset(cellIter.next());
291         return vNode;
292       }
293 
294       @Override
295       public void remove() {//it always removes
296       }
297     }

299     /** Called first to setup things. */
300     protected abstract void start() throws IOException;
301 
302     /** Called last to return the result. */
303     protected abstract DocIdSet finish() throws IOException;
304 
305     /**
306      * Visit an indexed non-leaf cell. The presence of a prefix cell implies
307      * there are leaf cells at further levels. The cell passed should have it's
308      * {@link org.apache.lucene.spatial.prefix.tree.Cell#getShapeRel()} set
309      * relative to the filtered shape.
310      *
311      * @param cell An intersecting cell; not a leaf.
312      * @return true to descend to more levels.
313      */
314     protected abstract boolean visitPrefix(Cell cell) throws IOException;
315 
316     /**
317      * Called when an indexed leaf cell is found. An
318      * indexed leaf cell usually means associated documents won't be found at
319      * further detail levels.  However, if a document has
320      * multiple overlapping shapes at different resolutions, then this isn't true.
321      */
322     protected abstract void visitLeaf(Cell cell) throws IOException;
323 
324     /**
325      * The cell is either indexed as a leaf or is the last level of detail. It
326      * might not even intersect the query shape, so be sure to check for that.
327      * The default implementation will check that and if passes then call
328      * {@link #visitLeaf(org.apache.lucene.spatial.prefix.tree.Cell)} or
329      * {@link #visitPrefix(org.apache.lucene.spatial.prefix.tree.Cell)}.
330      */
331     protected void visitScanned(Cell cell) throws IOException {
332       final SpatialRelation relate = cell.getShape().relate(queryShape);
333       if (relate.intersects()) {
334         cell.setShapeRel(relate);//just being pedantic
335         if (cell.isLeaf()) {
336           visitLeaf(cell);
337         } else {
338           visitPrefix(cell);
339         }
340       }
341     }

343     protected void preSiblings(VNode vNode) throws IOException {
344     }
345 
346     protected void postSiblings(VNode vNode) throws IOException {
347     }

350   /**
351    * A visitor node/cell found via the query shape for {@link VisitorTemplate}.
352    * Sometimes these are reset(cell). It's like a LinkedList node but forms a
353    * tree.
354    *
355    * @lucene.internal
356    */
357   protected static class VNode {
358     //Note: The VNode tree adds more code to debug/maintain v.s. a flattened
359     // LinkedList that we used to have. There is more opportunity here for
360     // custom behavior (see preSiblings & postSiblings) but that's not
361     // leveraged yet. Maybe this is slightly more GC friendly.
362 
363     final VNode parent;//only null at the root
364     Iterator<VNode> children;//null, then sometimes set, then null
365     Cell cell;//not null (except initially before reset())
366 
367     /**
368      * call reset(cell) after to set the cell.
369      */
370     VNode(VNode parent) { // remember to call reset(cell) after
371       this.parent = parent;
372     }
373 
374     void reset(Cell cell) {
375       assert cell != null;
376       this.cell = cell;
377       assert children == null;
378     }
