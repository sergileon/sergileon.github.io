47  /** A block-based terms index and dictionary that assigns
48   *  terms to variable length blocks according to how they
49   *  share prefixes.  The terms index is a prefix trie
50   *  whose leaves are term blocks.  The advantage of this
51   *  approach is that seekExact is often able to
52   *  determine a term cannot exist without doing any IO, and
53   *  intersection with Automata is very fast.  Note that this
54   *  terms dictionary has its own fixed terms index (ie, it
55   *  does not support a pluggable terms index
56   *  implementation).
57   *
58   *  <p><b>NOTE</b>: this terms dictionary supports
59   *  min/maxItemsPerBlock during indexing to control how
60   *  much memory the terms index uses.</p>
61   *
62   *  <p>If auto-prefix terms were indexed (see
63   *  {@link BlockTreeTermsWriter}), then the {@link Terms#intersect}
64   *  implementation here will make use of these terms only if the
65   *  automaton has a binary sink state, i.e. an accept state
66   *  which has a transition to itself accepting all byte values.
67   *  For example, both {@link PrefixQuery} and {@link TermRangeQuery}
68   *  pass such automata to {@link Terms#intersect}.</p>
69   *
70   *  <p>The data structure used by this implementation is very
71   *  similar to a burst trie
72   *  (http://citeseer.ist.psu.edu/viewdoc/summary?doi=10.1.1.18.3499),
73   *  but with added logic to break up too-large blocks of all
74   *  terms sharing a given prefix into smaller ones.</p>
75   *
76   *  <p>Use {@link org.apache.lucene.index.CheckIndex} with the <code>-verbose</code>
77   *  option to see summary statistics on the blocks in the
78   *  dictionary.
79   *
80   *  See {@link BlockTreeTermsWriter}.
81   *
82   * @lucene.experimental
83   */
84  
85  public final class BlockTreeTermsReader extends FieldsProducer {

87    static final Outputs<BytesRef> FST_OUTPUTS = ByteSequenceOutputs.getSingleton();
88    
89    static final BytesRef NO_OUTPUT = FST_OUTPUTS.getNoOutput();

91    static final int OUTPUT_FLAGS_NUM_BITS = 2;
92    static final int OUTPUT_FLAGS_MASK = 0x3;
93    static final int OUTPUT_FLAG_IS_FLOOR = 0x1;
94    static final int OUTPUT_FLAG_HAS_TERMS = 0x2;

96    /** Extension of terms file */
97    static final String TERMS_EXTENSION = "tim";
98    final static String TERMS_CODEC_NAME = "BlockTreeTermsDict";
99  
100   /** Initial terms format. */
101   public static final int VERSION_START = 0;
102 
103   /** Auto-prefix terms. */
104   public static final int VERSION_AUTO_PREFIX_TERMS = 1;
105 
106   /** Conditional auto-prefix terms: we record at write time whether
107    *  this field did write any auto-prefix terms. */
108   public static final int VERSION_AUTO_PREFIX_TERMS_COND = 2;
109 
110   /** Current terms format. */
111   public static final int VERSION_CURRENT = VERSION_AUTO_PREFIX_TERMS_COND;
112 
113   /** Extension of terms index file */
114   static final String TERMS_INDEX_EXTENSION = "tip";
115   final static String TERMS_INDEX_CODEC_NAME = "BlockTreeTermsIndex";
116 
117   // Open input to the main terms dict file (_X.tib)
118   final IndexInput termsIn;
119 
120   //private static final boolean DEBUG = BlockTreeTermsWriter.DEBUG;
121 
122   // Reads the terms dict entries, to gather state to
123   // produce DocsEnum on demand
124   final PostingsReaderBase postingsReader;
125 
126   private final TreeMap<String,FieldReader> fields = new TreeMap<>();
127 
128   /** File offset where the directory starts in the terms file. */
129   private long dirOffset;
130 
131   /** File offset where the directory starts in the index file. */
132   private long indexDirOffset;
133 
134   final String segment;
135   
136   final int version;
137 
138   final boolean anyAutoPrefixTerms;
139 
140   /** Sole constructor. */
141   public BlockTreeTermsReader(PostingsReaderBase postingsReader, SegmentReadState state) throws IOException {
142     boolean success = false;
143     IndexInput indexIn = null;
144     
145     this.postingsReader = postingsReader;
146     this.segment = state.segmentInfo.name;
147     
148     String termsName = IndexFileNames.segmentFileName(segment, state.segmentSuffix, TERMS_EXTENSION);
149     try {
150       termsIn = state.directory.openInput(termsName, state.context);
151       version = CodecUtil.checkIndexHeader(termsIn, TERMS_CODEC_NAME, VERSION_START, VERSION_CURRENT, state.segmentInfo.getId(), state.segmentSuffix);
152 
153       if (version < VERSION_AUTO_PREFIX_TERMS) {
154         // Old (pre-5.2.0) index, no auto-prefix terms:
155         this.anyAutoPrefixTerms = false;
156       } else if (version == VERSION_AUTO_PREFIX_TERMS) {
157         // 5.2.x index, might have auto-prefix terms:
158         this.anyAutoPrefixTerms = true;
159       } else {
160         // 5.3.x index, we record up front if we may have written any auto-prefix terms:
161         assert version >= VERSION_AUTO_PREFIX_TERMS_COND;
162         byte b = termsIn.readByte();
163         if (b == 0) {
164           this.anyAutoPrefixTerms = false;
165         } else if (b == 1) {
166           this.anyAutoPrefixTerms = true;
167         } else {
168           throw new CorruptIndexException("invalid anyAutoPrefixTerms: expected 0 or 1 but got " + b, termsIn);
169         }
170       }
171 
172       String indexName = IndexFileNames.segmentFileName(segment, state.segmentSuffix, TERMS_INDEX_EXTENSION);
173       indexIn = state.directory.openInput(indexName, state.context);
174       CodecUtil.checkIndexHeader(indexIn, TERMS_INDEX_CODEC_NAME, version, version, state.segmentInfo.getId(), state.segmentSuffix);
175       CodecUtil.checksumEntireFile(indexIn);
176 
177       // Have PostingsReader init itself
178       postingsReader.init(termsIn, state);
179       
180       // NOTE: data file is too costly to verify checksum against all the bytes on open,
181       // but for now we at least verify proper structure of the checksum footer: which looks
182       // for FOOTER_MAGIC + algorithmID. This is cheap and can detect some forms of corruption
183       // such as file truncation.
184       CodecUtil.retrieveChecksum(termsIn);
185 
186       // Read per-field details
187       seekDir(termsIn, dirOffset);
188       seekDir(indexIn, indexDirOffset);
189 
190       final int numFields = termsIn.readVInt();
191       if (numFields < 0) {
192         throw new CorruptIndexException("invalid numFields: " + numFields, termsIn);
193       }
194 
195       for (int i = 0; i < numFields; ++i) {
196         final int field = termsIn.readVInt();
197         final long numTerms = termsIn.readVLong();
198         if (numTerms <= 0) {
199           throw new CorruptIndexException("Illegal numTerms for field number: " + field, termsIn);
200         }
201         final int numBytes = termsIn.readVInt();
202         if (numBytes < 0) {
203           throw new CorruptIndexException("invalid rootCode for field number: " + field + ", numBytes=" + numBytes, termsIn);
204         }
205         final BytesRef rootCode = new BytesRef(new byte[numBytes]);
206         termsIn.readBytes(rootCode.bytes, 0, numBytes);
207         rootCode.length = numBytes;
208         final FieldInfo fieldInfo = state.fieldInfos.fieldInfo(field);
209         if (fieldInfo == null) {
210           throw new CorruptIndexException("invalid field number: " + field, termsIn);
211         }
212         final long sumTotalTermFreq = fieldInfo.getIndexOptions() == IndexOptions.DOCS ? -1 : termsIn.readVLong();
213         final long sumDocFreq = termsIn.readVLong();
214         final int docCount = termsIn.readVInt();
215         final int longsSize = termsIn.readVInt();
216         if (longsSize < 0) {
217           throw new CorruptIndexException("invalid longsSize for field: " + fieldInfo.name + ", longsSize=" + longsSize, termsIn);
218         }
219         BytesRef minTerm = readBytesRef(termsIn);
220         BytesRef maxTerm = readBytesRef(termsIn);
221         if (docCount < 0 || docCount > state.segmentInfo.maxDoc()) { // #docs with field must be <= #docs
222           throw new CorruptIndexException("invalid docCount: " + docCount + " maxDoc: " + state.segmentInfo.maxDoc(), termsIn);
223         }
224         if (sumDocFreq < docCount) {  // #postings must be >= #docs with field
225           throw new CorruptIndexException("invalid sumDocFreq: " + sumDocFreq + " docCount: " + docCount, termsIn);
226         }
227         if (sumTotalTermFreq != -1 && sumTotalTermFreq < sumDocFreq) { // #positions must be >= #postings
228           throw new CorruptIndexException("invalid sumTotalTermFreq: " + sumTotalTermFreq + " sumDocFreq: " + sumDocFreq, termsIn);
229         }
230         final long indexStartFP = indexIn.readVLong();
231         FieldReader previous = fields.put(fieldInfo.name,       
232                                           new FieldReader(this, fieldInfo, numTerms, rootCode, sumTotalTermFreq, sumDocFreq, docCount,
233                                                           indexStartFP, longsSize, indexIn, minTerm, maxTerm));
234         if (previous != null) {
235           throw new CorruptIndexException("duplicate field: " + fieldInfo.name, termsIn);
236         }
237       }
238       
239       indexIn.close();
240       success = true;
241     } finally {
242       if (!success) {
243         // this.close() will close in:
244         IOUtils.closeWhileHandlingException(indexIn, this);
245       }
246     }
247   }
248 
249   private static BytesRef readBytesRef(IndexInput in) throws IOException {
250     BytesRef bytes = new BytesRef();
251     bytes.length = in.readVInt();
252     bytes.bytes = new byte[bytes.length];
253     in.readBytes(bytes.bytes, 0, bytes.length);
254     return bytes;
255   }
256 
257   /** Seek {@code input} to the directory offset. */
258   private void seekDir(IndexInput input, long dirOffset)
259       throws IOException {
260     input.seek(input.length() - CodecUtil.footerLength() - 8);
261     dirOffset = input.readLong();
262     input.seek(dirOffset);
263   }

270   @Override
271   public void close() throws IOException {
272     try {
273       IOUtils.close(termsIn, postingsReader);
274     } finally { 
275       // Clear so refs to terms index is GCable even if
276       // app hangs onto us:
277       fields.clear();
278     }
279   }
280 
281   @Override
282   public Iterator<String> iterator() {
283     return Collections.unmodifiableSet(fields.keySet()).iterator();
284   }
285 
286   @Override
287   public Terms terms(String field) throws IOException {
288     assert field != null;
289     return fields.get(field);
290   }
291 
292   @Override
293   public int size() {
294     return fields.size();
295   }
296 
297   // for debugging
298   String brToString(BytesRef b) {
299     if (b == null) {
300       return "null";
301     } else {
302       try {
303         return b.utf8ToString() + " " + b;
304       } catch (Throwable t) {
305         // If BytesRef isn't actually UTF8, or it's eg a
306         // prefix of UTF8 that ends mid-unicode-char, we
307         // fallback to hex:
308         return b.toString();
309       }
310     }
311   }
312 
313   @Override
314   public long ramBytesUsed() {
315     long sizeInBytes = postingsReader.ramBytesUsed();
316     for(FieldReader reader : fields.values()) {
317       sizeInBytes += reader.ramBytesUsed();
318     }
319     return sizeInBytes;
320   }
321 
322   @Override
323   public Collection<Accountable> getChildResources() {
324     List<Accountable> resources = new ArrayList<>();
325     resources.addAll(Accountables.namedAccountables("field", fields));
326     resources.add(Accountables.namedAccountable("delegate", postingsReader));
327     return Collections.unmodifiableList(resources);
328   }
329 
330   @Override
331   public void checkIntegrity() throws IOException { 
332     // term dictionary
333     CodecUtil.checksumEntireFile(termsIn);
334       
335     // postings
336     postingsReader.checkIntegrity();
337   }
338 
339   @Override
340   public String toString() {
341     return getClass().getSimpleName() + "(fields=" + fields.size() + ",delegate=" + postingsReader + ")";
342   }
