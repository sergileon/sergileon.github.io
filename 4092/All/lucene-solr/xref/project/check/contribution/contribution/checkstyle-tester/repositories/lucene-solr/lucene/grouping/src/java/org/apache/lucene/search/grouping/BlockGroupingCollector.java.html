40  // TODO: this sentence is too long for the class summary.
41  /** BlockGroupingCollector performs grouping with a
42   *  single pass collector, as long as you are grouping by a
43   *  doc block field, ie all documents sharing a given group
44   *  value were indexed as a doc block using the atomic
45   *  {@link IndexWriter#addDocuments IndexWriter.addDocuments()} 
46   *  or {@link IndexWriter#updateDocuments IndexWriter.updateDocuments()} 
47   *  API.
48   *
49   *  <p>This results in faster performance (~25% faster QPS)
50   *  than the two-pass grouping collectors, with the tradeoff
51   *  being that the documents in each group must always be
52   *  indexed as a block.  This collector also fills in
53   *  TopGroups.totalGroupCount without requiring the separate
54   *  {@link org.apache.lucene.search.grouping.term.TermAllGroupsCollector}.  However, this collector does
55   *  not fill in the groupValue of each group; this field
56   *  will always be null.
57   *
58   *  <p><b>NOTE</b>: this collector makes no effort to verify
59   *  the docs were in fact indexed as a block, so it's up to
60   *  you to ensure this was the case.
61   *
62   *  <p>See {@link org.apache.lucene.search.grouping} for more
63   *  details including a full code example.</p>
64   *
65   * @lucene.experimental
66   */
67

99    private static final class OneGroup {
100     LeafReaderContext readerContext;
101     //int groupOrd;
102     int topGroupDoc;
103     int[] docs;
104     float[] scores;
105     int count;
106     int comparatorSlot;

109   // Sorts by groupSort.  Not static -- uses comparators, reversed
110   private final class GroupQueue extends PriorityQueue<OneGroup> {
111 
112     public GroupQueue(int size) {
113       super(size);
114     }
115 
116     @Override
117     protected boolean lessThan(final OneGroup group1, final OneGroup group2) {
118 
119       //System.out.println("    ltcheck");
120       assert group1 != group2;
121       assert group1.comparatorSlot != group2.comparatorSlot;
122 
123       final int numComparators = comparators.length;
124       for (int compIDX=0;compIDX < numComparators; compIDX++) {
125         final int c = reversed[compIDX] * comparators[compIDX].compare(group1.comparatorSlot, group2.comparatorSlot);
126         if (c != 0) {
127           // Short circuit
128           return c > 0;
129         }
130       }
131 
132       // Break ties by docID; lower docID is always sorted first
133       return group1.topGroupDoc > group2.topGroupDoc;
134     }
135   }
136 
137   // Called when we transition to another group; if the
138   // group is competitive we insert into the group queue
139   private void processGroup() {
140     totalGroupCount++;
141     //System.out.println("    processGroup ord=" + lastGroupOrd + " competes=" + groupCompetes + " count=" + subDocUpto + " groupDoc=" + topGroupDoc);
142     if (groupCompetes) {
143       if (!queueFull) {
144         // Startup transient: always add a new OneGroup
145         final OneGroup og = new OneGroup();
146         og.count = subDocUpto;
147         og.topGroupDoc = docBase + topGroupDoc;
148         og.docs = pendingSubDocs;
149         pendingSubDocs = new int[10];
150         if (needsScores) {
151           og.scores = pendingSubScores;
152           pendingSubScores = new float[10];
153         }
154         og.readerContext = currentReaderContext;
155         //og.groupOrd = lastGroupOrd;
156         og.comparatorSlot = bottomSlot;
157         final OneGroup bottomGroup = groupQueue.add(og);
158         //System.out.println("      ADD group=" + getGroupString(lastGroupOrd) + " newBottom=" + getGroupString(bottomGroup.groupOrd));
159         queueFull = groupQueue.size() == topNGroups;
160         if (queueFull) {
161           // Queue just became full; now set the real bottom
162           // in the comparators:
163           bottomSlot = bottomGroup.comparatorSlot;
164           //System.out.println("    set bottom=" + bottomSlot);
165           for (int i = 0; i < comparators.length; i++) {
166             leafComparators[i].setBottom(bottomSlot);
167           }
168           //System.out.println("     QUEUE FULL");
169         } else {
170           // Queue not full yet -- just advance bottomSlot:
171           bottomSlot = groupQueue.size();
172         }
173       } else {
174         // Replace bottom element in PQ and then updateTop
175         final OneGroup og = groupQueue.top();
176         assert og != null;
177         og.count = subDocUpto;
178         og.topGroupDoc = docBase + topGroupDoc;
179         // Swap pending docs
180         final int[] savDocs = og.docs;
181         og.docs = pendingSubDocs;
182         pendingSubDocs = savDocs;
183         if (needsScores) {
184           // Swap pending scores
185           final float[] savScores = og.scores;
186           og.scores = pendingSubScores;
187           pendingSubScores = savScores;
188         }
189         og.readerContext = currentReaderContext;
190         //og.groupOrd = lastGroupOrd;
191         bottomSlot = groupQueue.updateTop().comparatorSlot;
192 
193         //System.out.println("    set bottom=" + bottomSlot);
194         for (int i = 0; i < comparators.length; i++) {
195           leafComparators[i].setBottom(bottomSlot);
196         }
197       }
198     }
199     subDocUpto = 0;
200   }

202   /**
203    * Create the single pass collector.
204    *
205    *  @param groupSort The {@link Sort} used to sort the
206    *    groups.  The top sorted document within each group
207    *    according to groupSort, determines how that group
208    *    sorts against other groups.  This must be non-null,
209    *    ie, if you want to groupSort by relevance use
210    *    Sort.RELEVANCE.
211    *  @param topNGroups How many top groups to keep.
212    *  @param needsScores true if the collected documents
213    *    require scores, either because relevance is included
214    *    in the withinGroupSort or because you plan to pass true
215    *    for either getSscores or getMaxScores to {@link
216    *    #getTopGroups}
217    *  @param lastDocPerGroup a {@link Weight} that marks the
218    *    last document in each group.
219    */
220   public BlockGroupingCollector(Sort groupSort, int topNGroups, boolean needsScores, Weight lastDocPerGroup) throws IOException {
221 
222     if (topNGroups < 1) {
223       throw new IllegalArgumentException("topNGroups must be >= 1 (got " + topNGroups + ")");
224     }
225 
226     groupQueue = new GroupQueue(topNGroups);
227     pendingSubDocs = new int[10];
228     if (needsScores) {
229       pendingSubScores = new float[10];
230     }
231 
232     this.needsScores = needsScores;
233     this.lastDocPerGroup = lastDocPerGroup;
234     // TODO: allow null groupSort to mean "by relevance",
235     // and specialize it?
236     this.groupSort = groupSort;
237     
238     this.topNGroups = topNGroups;
239 
240     final SortField[] sortFields = groupSort.getSort();
241     comparators = new FieldComparator<?>[sortFields.length];
242     leafComparators = new LeafFieldComparator[sortFields.length];
243     compIDXEnd = comparators.length - 1;
244     reversed = new int[sortFields.length];
245     for (int i = 0; i < sortFields.length; i++) {
246       final SortField sortField = sortFields[i];
247       comparators[i] = sortField.getComparator(topNGroups, i);
248       reversed[i] = sortField.getReverse() ? -1 : 1;
249     }
250   }

257   /** Returns the grouped results.  Returns null if the
258    *  number of groups collected is &lt;= groupOffset.
259    *
260    *  <p><b>NOTE</b>: This collector is unable to compute
261    *  the groupValue per group so it will always be null.
262    *  This is normally not a problem, as you can obtain the
263    *  value just like you obtain other values for each
264    *  matching document (eg, via stored fields, via
265    *  DocValues, etc.)
266    *
267    *  @param withinGroupSort The {@link Sort} used to sort
268    *    documents within each group.  Passing null is
269    *    allowed, to sort by relevance.
270    *  @param groupOffset Which group to start from
271    *  @param withinGroupOffset Which document to start from
272    *    within each group
273    *  @param maxDocsPerGroup How many top documents to keep
274    *     within each group.
275    *  @param fillSortFields If true then the Comparable
276    *     values for the sort fields will be set
277    */
278   public TopGroups<?> getTopGroups(Sort withinGroupSort, int groupOffset, int withinGroupOffset, int maxDocsPerGroup, boolean fillSortFields) throws IOException {
279 
280     //if (queueFull) {
281     //System.out.println("getTopGroups groupOffset=" + groupOffset + " topNGroups=" + topNGroups);
282     //}
283     if (subDocUpto != 0) {
284       processGroup();
285     }
286     if (groupOffset >= groupQueue.size()) {
287       return null;
288     }
289     int totalGroupedHitCount = 0;
290 
291     final FakeScorer fakeScorer = new FakeScorer();
292 
293     float maxScore = Float.MIN_VALUE;
294 
295     @SuppressWarnings({"unchecked","rawtypes"})
296     final GroupDocs<Object>[] groups = new GroupDocs[groupQueue.size() - groupOffset];
297     for(int downTo=groupQueue.size()-groupOffset-1;downTo>=0;downTo--) {
298       final OneGroup og = groupQueue.pop();
299 
300       // At this point we hold all docs w/ in each group,
301       // unsorted; we now sort them:
302       final TopDocsCollector<?> collector;
303       if (withinGroupSort == null) {
304         // Sort by score
305         if (!needsScores) {
306           throw new IllegalArgumentException("cannot sort by relevance within group: needsScores=false");
307         }
308         collector = TopScoreDocCollector.create(maxDocsPerGroup);
309       } else {
310         // Sort by fields
311         collector = TopFieldCollector.create(withinGroupSort, maxDocsPerGroup, fillSortFields, needsScores, needsScores);
312       }
313 
314       LeafCollector leafCollector = collector.getLeafCollector(og.readerContext);
315       leafCollector.setScorer(fakeScorer);
316       for(int docIDX=0;docIDX<og.count;docIDX++) {
317         final int doc = og.docs[docIDX];
318         fakeScorer.doc = doc;
319         if (needsScores) {
320           fakeScorer.score = og.scores[docIDX];
321         }
322         leafCollector.collect(doc);
323       }
324       totalGroupedHitCount += og.count;
325 
326       final Object[] groupSortValues;
327 
328       if (fillSortFields) {
329         groupSortValues = new Comparable<?>[comparators.length];
330         for(int sortFieldIDX=0;sortFieldIDX<comparators.length;sortFieldIDX++) {
331           groupSortValues[sortFieldIDX] = comparators[sortFieldIDX].value(og.comparatorSlot);
332         }
333       } else {
334         groupSortValues = null;
335       }
336 
337       final TopDocs topDocs = collector.topDocs(withinGroupOffset, maxDocsPerGroup);
338 
339       // TODO: we could aggregate scores across children
340       // by Sum/Avg instead of passing NaN:
341       groups[downTo] = new GroupDocs<>(Float.NaN,
342                                              topDocs.getMaxScore(),
343                                              og.count,
344                                              topDocs.scoreDocs,
345                                              null,
346                                              groupSortValues);
347       maxScore = Math.max(maxScore, topDocs.getMaxScore());
348     }
349 
350     /*
351     while (groupQueue.size() != 0) {
352       final OneGroup og = groupQueue.pop();
353       //System.out.println("  leftover: og ord=" + og.groupOrd + " count=" + og.count);
354       totalGroupedHitCount += og.count;
355     }
356     */
357 
358     return new TopGroups<>(new TopGroups<>(groupSort.getSort(),
359                                        withinGroupSort == null ? null : withinGroupSort.getSort(),
360                                        totalHitCount, totalGroupedHitCount, groups, maxScore),
361                          totalGroupCount);
362   }

364   @Override
365   public void setScorer(Scorer scorer) throws IOException {
366     this.scorer = scorer;
367     for (LeafFieldComparator comparator : leafComparators) {
368       comparator.setScorer(scorer);
369     }
370   }

372   @Override
373   public void collect(int doc) throws IOException {
374 
375     // System.out.println("C " + doc);
376 
377     if (doc > groupEndDocID) {
378       // Group changed
379       if (subDocUpto != 0) {
380         processGroup();
381       }
382       groupEndDocID = lastDocPerGroupBits.advance(doc);
383       //System.out.println("  adv " + groupEndDocID + " " + lastDocPerGroupBits);
384       subDocUpto = 0;
385       groupCompetes = !queueFull;
386     }
387 
388     totalHitCount++;
389 
390     // Always cache doc/score within this group:
391     if (subDocUpto == pendingSubDocs.length) {
392       pendingSubDocs = ArrayUtil.grow(pendingSubDocs);
393     }
394     pendingSubDocs[subDocUpto] = doc;
395     if (needsScores) {
396       if (subDocUpto == pendingSubScores.length) {
397         pendingSubScores = ArrayUtil.grow(pendingSubScores);
398       }
399       pendingSubScores[subDocUpto] = scorer.score();
400     }
401     subDocUpto++;
402 
403     if (groupCompetes) {
404       if (subDocUpto == 1) {
405         assert !queueFull;
406 
407         //System.out.println("    init copy to bottomSlot=" + bottomSlot);
408         for (LeafFieldComparator fc : leafComparators) {
409           fc.copy(bottomSlot, doc);
410           fc.setBottom(bottomSlot);
411         }        
412         topGroupDoc = doc;
413       } else {
414         // Compare to bottomSlot
415         for (int compIDX = 0;; compIDX++) {
416           final int c = reversed[compIDX] * leafComparators[compIDX].compareBottom(doc);
417           if (c < 0) {
418             // Definitely not competitive -- done
419             return;
420           } else if (c > 0) {
421             // Definitely competitive.
422             break;
423           } else if (compIDX == compIDXEnd) {
424             // Ties with bottom, except we know this docID is
425             // > docID in the queue (docs are visited in
426             // order), so not competitive:
427             return;
428           }
429         }
430 
431         //System.out.println("       best w/in group!");
432         
433         for (LeafFieldComparator fc : leafComparators) {
434           fc.copy(bottomSlot, doc);
435           // Necessary because some comparators cache
436           // details of bottom slot; this forces them to
437           // re-cache:
438           fc.setBottom(bottomSlot);
439         }        
440         topGroupDoc = doc;
441       }
442     } else {
443       // We're not sure this group will make it into the
444       // queue yet
445       for (int compIDX = 0;; compIDX++) {
446         final int c = reversed[compIDX] * leafComparators[compIDX].compareBottom(doc);
447         if (c < 0) {
448           // Definitely not competitive -- done
449           //System.out.println("    doc doesn't compete w/ top groups");
450           return;
451         } else if (c > 0) {
452           // Definitely competitive.
453           break;
454         } else if (compIDX == compIDXEnd) {
455           // Ties with bottom, except we know this docID is
456           // > docID in the queue (docs are visited in
457           // order), so not competitive:
458           //System.out.println("    doc doesn't compete w/ top groups");
459           return;
460         }
461       }
462       groupCompetes = true;
463       for (LeafFieldComparator fc : leafComparators) {
464         fc.copy(bottomSlot, doc);
465         // Necessary because some comparators cache
466         // details of bottom slot; this forces them to
467         // re-cache:
468         fc.setBottom(bottomSlot);
469       }
470       topGroupDoc = doc;
471       //System.out.println("        doc competes w/ top groups");
472     }
473   }

475   @Override
476   protected void doSetNextReader(LeafReaderContext readerContext) throws IOException {
477     if (subDocUpto != 0) {
478       processGroup();
479     }
480     subDocUpto = 0;
481     docBase = readerContext.docBase;
482     //System.out.println("setNextReader base=" + docBase + " r=" + readerContext.reader);
483     lastDocPerGroupBits = lastDocPerGroup.scorer(readerContext);
484     groupEndDocID = -1;
485 
486     currentReaderContext = readerContext;
487     for (int i=0; i<comparators.length; i++) {
488       leafComparators[i] = comparators[i].getLeafComparator(readerContext);
489     }
490   }

492   @Override
493   public boolean needsScores() {
494     return needsScores;
495   }
