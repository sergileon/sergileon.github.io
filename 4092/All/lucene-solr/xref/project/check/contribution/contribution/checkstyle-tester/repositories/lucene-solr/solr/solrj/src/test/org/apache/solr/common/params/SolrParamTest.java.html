35    public void testParamIterators() {
36  
37      ModifiableSolrParams aaa = new ModifiableSolrParams();
38      aaa.add("foo", "a1");
39      aaa.add("foo", "a2");
40  
41      assertIterSize("aaa: foo", 1, aaa);
42      assertIterSize("required aaa: foo", 1, aaa.required());
43  
44      assertEquals(new String[] { "a1", "a2" }, aaa.getParams("foo"));
45  
46      aaa.add("yak", "a3");
47  
48      assertIterSize("aaa: foo & yak", 2, aaa);
49      assertIterSize("required aaa: foo & yak", 2, aaa.required());
50  
51      assertEquals(new String[] { "a1", "a2" }, aaa.getParams("foo"));
52      assertEquals(new String[] { "a3" }, aaa.getParams("yak"));
53  
54      ModifiableSolrParams bbb = new ModifiableSolrParams();
55      bbb.add("foo", "b1");
56      bbb.add("foo", "b2");
57      bbb.add("zot", "b3");
58  
59      assertIterSize("bbb: foo & zot", 2, bbb);
60      assertIterSize("required bbb: foo & zot", 2, bbb.required());
61  
62      assertEquals(new String[] { "b1", "b2" }, bbb.getParams("foo"));
63      assertEquals(new String[] { "b3" }, bbb.getParams("zot"));
64  
65      SolrParams def = SolrParams.wrapDefaults(aaa, bbb);
66  
67      assertIterSize("def: aaa + bbb", 3, def);
68      assertIterSize("required def: aaa + bbb", 3, def.required());
69  
70      assertEquals(new String[] { "a1", "a2" }, def.getParams("foo"));
71      assertEquals(new String[] { "a3" }, def.getParams("yak"));
72      assertEquals(new String[] { "b3" }, def.getParams("zot"));
73  
74      SolrParams append = SolrParams.wrapAppended(aaa, bbb);
75  
76      assertIterSize("append: aaa + bbb", 3, append);
77      assertIterSize("required appended: aaa + bbb", 3, append.required());
78  
79      assertEquals(new String[] { "a1", "a2", "b1", "b2", }, append.getParams("foo"));
80      assertEquals(new String[] { "a3" }, append.getParams("yak"));
81      assertEquals(new String[] { "b3" }, append.getParams("zot"));
82  
83    }

85    public void testMultiValues() {
86      NamedList nl = new NamedList();
87      nl.add("x", "X1");
88      nl.add("x", "X2");
89      nl.add("x", new String[]{"X3", "X4"});
90      Map<String, String[]> m = SolrParams.toMultiMap(nl);
91      String[] r = m.get("x");
92      assertTrue(Arrays.asList(r).containsAll(Arrays.asList(new String[]{"X1", "X2", "X3", "X4"})));
93    }

95    public void testGetAll() {
96      ModifiableSolrParams params = new ModifiableSolrParams();
97      params.add("x", "X1");
98      params.add("x", "X2");
99      params.add("y", "Y");
100     Map<String, Object> m = params.getAll(null, "x", "y");
101     String[] x = (String[]) m.get("x");
102     assertEquals(2, x.length);
103     assertEquals("X1", x[0]);
104     assertEquals("X2", x[1]);
105     assertEquals("Y", m.get("y"));
106     try {
107       params.required().getAll(null, "z");
108       fail("Error expected");
109     } catch (SolrException e) {
110       assertEquals(e.code(), SolrException.ErrorCode.BAD_REQUEST.code);
111 
112     }
113   }

141   public void testGetParams() {
142     Map<String,String> pmap = new HashMap<>();
143     pmap.put( "str"        , "string"   );
144     pmap.put( "bool"       , "true"     );
145     pmap.put( "true-0"     , "true"     );
146     pmap.put( "true-1"     , "yes"      );
147     pmap.put( "true-2"     , "on"       );
148     pmap.put( "false-0"    , "false"    );
149     pmap.put( "false-1"    , "off"      );
150     pmap.put( "false-2"    , "no"       );
151     pmap.put( "int"        , "100"      );
152     pmap.put( "float"      , "10.6"     );
153     pmap.put( "f.fl.str"   , "string"   );
154     pmap.put( "f.fl.bool"  , "true"     );
155     pmap.put( "f.fl.int"   , "100"      );
156     pmap.put( "f.fl.float" , "10.6"     );
157     pmap.put( "f.bad.bool" , "notbool"  );
158     pmap.put( "f.bad.int"  , "notint"   );
159     pmap.put( "f.bad.float", "notfloat" );
160     final SolrParams params = new MapSolrParams( pmap );
161     
162     // Test the string values we put in directly
163     assertEquals(  "string"   , params.get( "str"       ) );
164     assertEquals(  "true"     , params.get( "bool"      ) );
165     assertEquals(  "100"      , params.get( "int"       ) );
166     assertEquals(  "10.6"     , params.get( "float"     ) );
167     assertEquals(  "string"   , params.get( "f.fl.str"    ) );
168     assertEquals(  "true"     , params.get( "f.fl.bool"   ) );
169     assertEquals(  "100"      , params.get( "f.fl.int"    ) );
170     assertEquals(  "10.6"     , params.get( "f.fl.float"  ) );
171     assertEquals(  "notbool"  , params.get( "f.bad.bool"  ) );
172     assertEquals(  "notint"   , params.get( "f.bad.int"   ) );
173     assertEquals(  "notfloat" , params.get( "f.bad.float" ) );
174     
175     final String  pstr = "string";
176     final Boolean pbool = Boolean.TRUE;
177     final Integer pint = new Integer( 100 );
178     final Float   pfloat = new Float( 10.6f );
179     
180     // Make sure they parse ok
181     assertEquals( pstr   , params.get(      "str"      ) );
182     assertEquals( pbool  , params.getBool(  "bool"     ) );
183     assertEquals( pint   , params.getInt(   "int"      ) );
184     assertEquals( pfloat , params.getFloat( "float"    ) );
185     assertEquals( pbool  , params.getBool(  "f.fl.bool"  ) );
186     assertEquals( pint   , params.getInt(   "f.fl.int"   ) );
187     assertEquals( pfloat , params.getFloat( "f.fl.float" ) );
188     assertEquals( pstr   , params.getFieldParam( "fl", "str"  ) );
189     assertEquals( pbool  , params.getFieldBool(  "fl", "bool" ) );
190     assertEquals( pint   , params.getFieldInt(   "fl", "int"  ) );
191     assertEquals( pfloat , params.getFieldFloat( "fl", "float" ) );
192     
193     // Test field defaulting (fall through to non-field-specific value)
194     assertEquals( pint   , params.getFieldInt( "fff",  "int"      ) );
195     
196     // test boolean parsing
197     for( int i=0; i<3; i++ ) {
198       // Must use Boolean rather than boolean reference value to prevent
199       // auto-unboxing ambiguity
200       assertEquals( Boolean.TRUE,  params.getBool( "true-"+i  ) );
201       assertEquals( Boolean.FALSE, params.getBool( "false-"+i ) );
202     }
203     
204     // Malformed params: These should throw a 400
205     assertEquals( 400, getReturnCode( new Runnable() { @Override
206     public void run() { params.getInt(   "f.bad.int" ); } } ) );
207     assertEquals( 400, getReturnCode( new Runnable() { @Override
208     public void run() { params.getBool(  "f.bad.bool" ); } } ) );
209     assertEquals( 400, getReturnCode( new Runnable() { @Override
210     public void run() { params.getFloat( "f.bad.float" ); } } ) );
211     
212     // Ask for params that arent there
213     assertNull( params.get( "asagdsaga" ) );
214     assertNull( params.getBool( "asagdsaga" ) );
215     assertNull( params.getInt( "asagdsaga" ) );
216     assertNull( params.getFloat( "asagdsaga" ) );
217     
218     // Get things with defaults
219     assertEquals( pstr                  , params.get(          "xxx", pstr   ) );
220     assertEquals( pbool.booleanValue()  , params.getBool(      "xxx", pbool   ) );
221     assertEquals( pint.intValue()       , params.getInt(       "xxx", pint   ) );
222     assertEquals( pfloat.floatValue()   , params.getFloat(     "xxx", pfloat  ), 0.1);
223     assertEquals( pbool.booleanValue()  , params.getFieldBool( "xxx", "bool", pbool ) );
224     assertEquals( pint.intValue()       , params.getFieldInt(  "xxx", "int", pint  ) );
225     assertEquals( pfloat.floatValue()   , params.getFieldFloat("xxx", "float", pfloat  ), 0.1);
226     assertEquals( pstr                  , params.getFieldParam("xxx", "str", pstr  ) );
227 
228     // Required params testing uses decorator
229     final SolrParams required = params.required();
230     
231     // Required params which are present should test same as above
232     assertEquals( pstr   , required.get(      "str"      ) );
233     assertEquals( pbool  , required.getBool(  "bool"     ) );
234     assertEquals( pint   , required.getInt(   "int"      ) );
235     assertEquals( pfloat , required.getFloat( "float"    ) );
236     
237     // field value present
238     assertEquals( pbool  , required.getFieldBool(  "fl", "bool" ) );
239     // field defaulting (fall through to non-field-specific value)
240     assertEquals( pstr   , required.getFieldParams("fakefield", "str")[0] );
241     assertEquals( pstr   , required.getFieldParam( "fakefield", "str"   ) );
242     assertEquals( pbool  , required.getFieldBool(  "fakefield", "bool"  ) );
243     assertEquals( pint   , required.getFieldInt(   "fakefield", "int"   ) );
244     assertEquals( pfloat , required.getFieldFloat( "fakefield", "float" ) );
245     
246     // Required params which are missing: These should throw a 400
247     assertEquals( 400, getReturnCode( new Runnable() { @Override
248     public void run() { required.get( "aaaa" ); } } ) );
249     assertEquals( 400, getReturnCode( new Runnable() { @Override
250     public void run() { required.getInt(   "f.bad.int" ); } } ) );
251     assertEquals( 400, getReturnCode( new Runnable() { @Override
252     public void run() { required.getBool(  "f.bad.bool" ); } } ) );
253     assertEquals( 400, getReturnCode( new Runnable() { @Override
254     public void run() { required.getFloat( "f.bad.float" ); } } ) );
255     assertEquals( 400, getReturnCode( new Runnable() { @Override
256     public void run() { required.getInt(   "aaa" ); } } ) );
257     assertEquals( 400, getReturnCode( new Runnable() { @Override
258     public void run() { required.getBool(  "aaa" ); } } ) );
259     assertEquals( 400, getReturnCode( new Runnable() { @Override
260     public void run() { required.getFloat( "aaa" ); } } ) );
261     assertEquals( 400, getReturnCode( new Runnable() { @Override
262     public void run() { params.getFieldBool(  "bad", "bool" ); } } ) );
263     assertEquals( 400, getReturnCode( new Runnable() { @Override
264     public void run() { params.getFieldInt(   "bad", "int"  ); } } ) );
265 
266     // Fields with default use their parent value:
267     assertEquals(
268         params.get(   "aaaa", "str" ), 
269         required.get( "aaaa", "str" ) );
270     assertEquals(
271         params.getInt(   "f.bad.nnnn", pint ), 
272         required.getInt( "f.bad.nnnn", pint ) );
273     
274     // Check default SolrParams
275     Map<String,String> dmap = new HashMap<>();
276     // these are not defined in params
277     dmap.put( "dstr"               , "default"   );
278     dmap.put( "dint"               , "123"       );
279     // these are defined in params
280     dmap.put( "int"                , "456"       );
281     SolrParams defaults = SolrParams.wrapDefaults(params, new MapSolrParams(dmap));
282   
283     // in params, not in default
284     assertEquals( pstr                  , defaults.get( "str"      ) );
285     // in default, not in params
286     assertEquals( "default"             , defaults.get( "dstr"      ) );
287     assertEquals( new Integer(123)      , defaults.getInt(  "dint"     ) );
288     // in params, overriding defaults
289     assertEquals( pint                  , defaults.getInt(   "int"      ) );
290     // in neither params nor defaults
291     assertNull( defaults.get( "asagdsaga" ) );
292   }

294   public static int getReturnCode( Runnable runnable )
295   {
296     try {
297       runnable.run();
298     }
299     catch( SolrException sx ) {
300       return sx.code();
301     }
302     catch( Exception ex ) {
303       ex.printStackTrace();
304       return 500;
305     }
306     return 200;
307   }

309   static <T> List<T> iterToList(Iterator<T> iter) {
310     List<T> result = new ArrayList<>();
311     while (iter.hasNext()) {
312       result.add(iter.next());
313     }
314     return result;
315   }

317   static void assertIterSize(String msg, int expectedSize, SolrParams p) {
318     List<String> keys = iterToList(p.getParameterNamesIterator());
319     assertEquals(msg + " " + keys.toString(), expectedSize, keys.size());
320   }
