37  /**
38   * @lucene.experimental
39   */
40  public class BlockDirectory extends FilterDirectory {
41    private static final Logger LOG = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
42    
43    public static final long BLOCK_SHIFT = 13; // 2^13 = 8,192 bytes per block
44    public static final long BLOCK_MOD = 0x1FFF;
45    public static final int BLOCK_SIZE = 1 << BLOCK_SHIFT;
46    
47    public static long getBlock(long pos) {
48      return pos >>> BLOCK_SHIFT;
49    }
50    
51    public static long getPosition(long pos) {
52      return pos & BLOCK_MOD;
53    }
54    
55    public static long getRealPosition(long block, long positionInBlock) {
56      return (block << BLOCK_SHIFT) + positionInBlock;
57    }
58    
59    public static Cache NO_CACHE = new Cache() {
60      
61      @Override
62      public void update(String name, long blockId, int blockOffset,
63          byte[] buffer, int offset, int length) {}
64      
65      @Override
66      public boolean fetch(String name, long blockId, int blockOffset, byte[] b,
67          int off, int lengthToReadInBlock) {
68        return false;
69      }
70      
71      @Override
72      public void delete(String name) {
73        
74      }
75      
76      @Override
77      public long size() {
78        return 0;
79      }
80      
81      @Override
82      public void renameCacheFile(String source, String dest) {}
83  
84      @Override
85      public void releaseResources() {}
86    };
87    
88    private final int blockSize;
89    private final String dirName;
90    private final Cache cache;
91    private final Set<String> blockCacheFileTypes;
92    private final boolean blockCacheReadEnabled;
93    private final boolean blockCacheWriteEnabled;
94  
95    public BlockDirectory(String dirName, Directory directory, Cache cache,
96        Set<String> blockCacheFileTypes, boolean blockCacheReadEnabled,
97        boolean blockCacheWriteEnabled) throws IOException {
98      super(directory);
99      this.dirName = dirName;
100     blockSize = BLOCK_SIZE;
101     this.cache = cache;
102     if (blockCacheFileTypes == null || blockCacheFileTypes.isEmpty()) {
103       this.blockCacheFileTypes = null;
104     } else {
105       this.blockCacheFileTypes = blockCacheFileTypes;
106     }
107     this.blockCacheReadEnabled = blockCacheReadEnabled;
108     if (!blockCacheReadEnabled) {
109       LOG.info("Block cache on read is disabled");
110     }
111     this.blockCacheWriteEnabled = blockCacheWriteEnabled;
112     if (!blockCacheWriteEnabled) {
113       LOG.info("Block cache on write is disabled");
114     }
115   }
116   
117   private IndexInput openInput(String name, int bufferSize, IOContext context)
118       throws IOException {
119     final IndexInput source = super.openInput(name, context);
120     if (useReadCache(name, context)) {
121       return new CachedIndexInput(source, blockSize, name,
122           getFileCacheName(name), cache, bufferSize);
123     }
124     return source;
125   }
126   
127   private boolean isCachableFile(String name) {
128     for (String ext : blockCacheFileTypes) {
129       if (name.endsWith(ext)) {
130         return true;
131       }
132     }
133     return false;
134   }
135   
136   @Override
137   public IndexInput openInput(final String name, IOContext context)
138       throws IOException {
139     return openInput(name, blockSize, context);
140   }
141   
142   static class CachedIndexInput extends CustomBufferedIndexInput {
143     private final Store store;
144     private IndexInput source;
145     private final int blockSize;
146     private final long fileLength;
147     private final String cacheName;
148     private final Cache cache;
149     
150     public CachedIndexInput(IndexInput source, int blockSize, String name,
151         String cacheName, Cache cache, int bufferSize) {
152       super(name, bufferSize);
153       this.source = source;
154       this.blockSize = blockSize;
155       fileLength = source.length();
156       this.cacheName = cacheName;
157       this.cache = cache;
158       store = BufferStore.instance(blockSize);
159     }
160     
161     @Override
162     public IndexInput clone() {
163       CachedIndexInput clone = (CachedIndexInput) super.clone();
164       clone.source = (IndexInput) source.clone();
165       return clone;
166     }
167     
168     @Override
169     public long length() {
170       return source.length();
171     }
172     
173     @Override
174     protected void seekInternal(long pos) throws IOException {}
175     
176     @Override
177     protected void readInternal(byte[] b, int off, int len) throws IOException {
178       long position = getFilePointer();
179       while (len > 0) {
180         int length = fetchBlock(position, b, off, len);
181         position += length;
182         len -= length;
183         off += length;
184       }
185     }
186     
187     private int fetchBlock(long position, byte[] b, int off, int len)
188         throws IOException {
189       // read whole block into cache and then provide needed data
190       long blockId = getBlock(position);
191       int blockOffset = (int) getPosition(position);
192       int lengthToReadInBlock = Math.min(len, blockSize - blockOffset);
193       if (checkCache(blockId, blockOffset, b, off, lengthToReadInBlock)) {
194         return lengthToReadInBlock;
195       } else {
196         readIntoCacheAndResult(blockId, blockOffset, b, off,
197             lengthToReadInBlock);
198       }
199       return lengthToReadInBlock;
200     }
201     
202     private void readIntoCacheAndResult(long blockId, int blockOffset,
203         byte[] b, int off, int lengthToReadInBlock) throws IOException {
204       long position = getRealPosition(blockId, 0);
205       int length = (int) Math.min(blockSize, fileLength - position);
206       source.seek(position);
207       
208       byte[] buf = store.takeBuffer(blockSize);
209       source.readBytes(buf, 0, length);
210       System.arraycopy(buf, blockOffset, b, off, lengthToReadInBlock);
211       cache.update(cacheName, blockId, 0, buf, 0, blockSize);
212       store.putBuffer(buf);
213     }
214     
215     private boolean checkCache(long blockId, int blockOffset, byte[] b,
216         int off, int lengthToReadInBlock) {
217       return cache.fetch(cacheName, blockId, blockOffset, b, off,
218           lengthToReadInBlock);
219     }
220     
221     @Override
222     protected void closeInternal() throws IOException {
223       source.close();
224     }
225   }
226   
227   @Override
228   public void close() throws IOException {
229     try {
230       String[] files = listAll();
231       
232       for (String file : files) {
233         cache.delete(getFileCacheName(file));
234       }
235       
236     } catch (FileNotFoundException e) {
237       // the local file system folder may be gone
238     } finally {
239       super.close();
240       cache.releaseResources();
241     }
242   }
243   
244   String getFileCacheName(String name) throws IOException {
245     return getFileCacheLocation(name) + ":" + getFileModified(name);
246   }
247   
248   private long getFileModified(String name) throws IOException {
249     if (in instanceof FSDirectory) {
250       File directory = ((FSDirectory) in).getDirectory().toFile();
251       File file = new File(directory, name);
252       if (!file.exists()) {
253         throw new FileNotFoundException("File [" + name + "] not found");
254       }
255       return file.lastModified();
256     } else if (in instanceof HdfsDirectory) {
257       return ((HdfsDirectory) in).fileModified(name);
258     } else {
259       throw new UnsupportedOperationException();
260     }
261   }
262   
263   String getFileCacheLocation(String name) {
264     return dirName + "/" + name;
265   }

276   /**
277    * Determine whether read caching should be used for a particular
278    * file/context.
279    */
280   boolean useReadCache(String name, IOContext context) {
281     if (!blockCacheReadEnabled) {
282       return false;
283     }
284     if (blockCacheFileTypes != null && !isCachableFile(name)) {
285       return false;
286     }
287     switch (context.context) {
288       default: {
289         return true;
290       }
291     }
292   }
293   
294   /**
295    * Determine whether write caching should be used for a particular
296    * file/context.
297    */
298   boolean useWriteCache(String name, IOContext context) {
299     if (!blockCacheWriteEnabled || name.startsWith(IndexFileNames.PENDING_SEGMENTS)) {
300       // for safety, don't bother caching pending commits.
301       // the cache does support renaming (renameCacheFile), but thats a scary optimization.
302       return false;
303     }
304     if (blockCacheFileTypes != null && !isCachableFile(name)) {
305       return false;
306     }
307     switch (context.context) {
308       case MERGE: {
309         // we currently don't cache any merge context writes
310         return false;
311       }
312       default: {
313         return true;
314       }
315     }
316   }
317   
318   @Override
319   public IndexOutput createOutput(String name, IOContext context)
320       throws IOException {
321     final IndexOutput dest = super.createOutput(name, context);
322     if (useWriteCache(name, context)) {
323       return new CachedIndexOutput(this, dest, blockSize, name, cache, blockSize);
324     }
325     return dest;
326   }
327   
328   public void deleteFile(String name) throws IOException {
329     cache.delete(getFileCacheName(name));
330     super.deleteFile(name);
331   }
332     
333   public boolean isBlockCacheReadEnabled() {
334     return blockCacheReadEnabled;
335   }

337   public boolean isBlockCacheWriteEnabled() {
338     return blockCacheWriteEnabled;
