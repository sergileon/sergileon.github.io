49   * Adds debugging information to a request.
50   * 
51   *
52   * @since solr 1.3

54  public class DebugComponent extends SearchComponent

56    public static final String COMPONENT_NAME = "debug";
57    
58    /**
59     * A counter to ensure that no RID is equal, even if they fall in the same millisecond
60     */
61    private static final AtomicLong ridCounter = new AtomicLong();
62    
63    /**
64     * Map containing all the possible stages as key and
65     * the corresponding readable purpose as value
66     */
67    private static final Map<Integer, String> stages;
68  
69    static {
70        Map<Integer, String> map = new TreeMap<>();
71        map.put(ResponseBuilder.STAGE_START, "START");
72        map.put(ResponseBuilder.STAGE_PARSE_QUERY, "PARSE_QUERY");
73        map.put(ResponseBuilder.STAGE_TOP_GROUPS, "TOP_GROUPS");
74        map.put(ResponseBuilder.STAGE_EXECUTE_QUERY, "EXECUTE_QUERY");
75        map.put(ResponseBuilder.STAGE_GET_FIELDS, "GET_FIELDS");
76        map.put(ResponseBuilder.STAGE_DONE, "DONE");
77        stages = Collections.unmodifiableMap(map);
78    }

80    @Override
81    public void prepare(ResponseBuilder rb) throws IOException
82    {
83      if(rb.isDebugTrack() && rb.isDistrib) {
84        doDebugTrack(rb);
85      }
86    }

88    @SuppressWarnings("unchecked")
89    @Override
90    public void process(ResponseBuilder rb) throws IOException
91    {
92      if( rb.isDebug() ) {
93        DocList results = null;
94        //some internal grouping requests won't have results value set
95        if(rb.getResults() != null) {
96          results = rb.getResults().docList;
97        }
98  
99        NamedList stdinfo = SolrPluginUtils.doStandardDebug( rb.req,
100           rb.getQueryString(), rb.wrap(rb.getQuery()), results, rb.isDebugQuery(), rb.isDebugResults());
101       
102       NamedList info = rb.getDebugInfo();
103       if( info == null ) {
104         rb.setDebugInfo( stdinfo );
105         info = stdinfo;
106       }
107       else {
108         info.addAll( stdinfo );
109       }
110 
111       if (rb.req.getJSON() != null) {
112         info.add(JSON, rb.req.getJSON());
113       }
114 
115       if (rb.isDebugQuery() && rb.getQparser() != null) {
116         rb.getQparser().addDebugInfo(rb.getDebugInfo());
117       }
118       
119       if (null != rb.getDebugInfo() ) {
120         if (rb.isDebugQuery() && null != rb.getFilters() ) {
121           info.add("filter_queries",rb.req.getParams().getParams(FQ));
122           List<String> fqs = new ArrayList<>(rb.getFilters().size());
123           for (Query fq : rb.getFilters()) {
124             fqs.add(QueryParsing.toString(fq, rb.req.getSchema()));
125           }
126           info.add("parsed_filter_queries",fqs);
127         }
128         
129         // Add this directly here?
130         rb.rsp.add("debug", rb.getDebugInfo() );
131       }
132     }
133   }

136   private void doDebugTrack(ResponseBuilder rb) {
137     SolrQueryRequest req = rb.req;
138     String rid = req.getParams().get(CommonParams.REQUEST_ID);
139     if(rid == null || "".equals(rid)) {
140       rid = generateRid(rb);
141       ModifiableSolrParams params = new ModifiableSolrParams(req.getParams());
142       params.add(CommonParams.REQUEST_ID, rid);//add rid to the request so that shards see it
143       req.setParams(params);
144     }
145     rb.addDebug(rid, "track", CommonParams.REQUEST_ID);//to see it in the response
146     rb.rsp.addToLog(CommonParams.REQUEST_ID, rid); //to see it in the logs of the landing core
147     
148   }
149 
150   @SuppressForbidden(reason = "Need currentTimeMillis, only used for naming")
151   private String generateRid(ResponseBuilder rb) {
152     String hostName = rb.req.getCore().getCoreDescriptor().getCoreContainer().getHostName();
153     return hostName + "-" + rb.req.getCore().getName() + "-" + System.currentTimeMillis() + "-" + ridCounter.getAndIncrement();
154   }

156   @Override
157   public void modifyRequest(ResponseBuilder rb, SearchComponent who, ShardRequest sreq) {
158     if (!rb.isDebug()) return;
159     
160     // Turn on debug to get explain only when retrieving fields
161     if ((sreq.purpose & ShardRequest.PURPOSE_GET_FIELDS) != 0) {
162       sreq.purpose |= ShardRequest.PURPOSE_GET_DEBUG;
163       if (rb.isDebugAll()) {
164         sreq.params.set(CommonParams.DEBUG_QUERY, "true");
165       } else {
166         if (rb.isDebugQuery()){
167           sreq.params.add(CommonParams.DEBUG, CommonParams.QUERY);
168         }
169         if (rb.isDebugResults()){
170           sreq.params.add(CommonParams.DEBUG, CommonParams.RESULTS);
171         }
172       }
173     } else {
174       sreq.params.set(CommonParams.DEBUG_QUERY, "false");
175       sreq.params.set(CommonParams.DEBUG, "false");
176     }
177     if (rb.isDebugTimings()) {
178       sreq.params.add(CommonParams.DEBUG, CommonParams.TIMING);
179     } 
180     if (rb.isDebugTrack()) {
181       sreq.params.add(CommonParams.DEBUG, CommonParams.TRACK);
182       sreq.params.set(CommonParams.REQUEST_ID, rb.req.getParams().get(CommonParams.REQUEST_ID));
183       sreq.params.set(CommonParams.REQUEST_PURPOSE, SolrPluginUtils.getRequestPurpose(sreq.purpose));
184     }
185   }

187   @Override
188   public void handleResponses(ResponseBuilder rb, ShardRequest sreq) {
189     if (rb.isDebugTrack() && rb.isDistrib && !rb.finished.isEmpty()) {
190       @SuppressWarnings("unchecked")
191       NamedList<Object> stageList = (NamedList<Object>) ((NamedList<Object>)rb.getDebugInfo().get("track")).get(stages.get(rb.stage));
192       if(stageList == null) {
193         stageList = new SimpleOrderedMap<>();
194         rb.addDebug(stageList, "track", stages.get(rb.stage));
195       }
196       for(ShardResponse response: sreq.responses) {
197         stageList.add(response.getShard(), getTrackResponse(response));
198       }
199     }
200   }

202   private final static Set<String> EXCLUDE_SET = Collections.unmodifiableSet(new HashSet<>(Arrays.asList("explain")));
203 
204   @Override
205   public void finishStage(ResponseBuilder rb) {
206     if (rb.isDebug() && rb.stage == ResponseBuilder.STAGE_GET_FIELDS) {
207       NamedList<Object> info = rb.getDebugInfo();
208       NamedList<Object> explain = new SimpleOrderedMap<>();
209 
210       Map.Entry<String, Object>[]  arr =  new NamedList.NamedListEntry[rb.resultIds.size()];
211       // Will be set to true if there is at least one response with PURPOSE_GET_DEBUG
212       boolean hasGetDebugResponses = false;
213 
214       for (ShardRequest sreq : rb.finished) {
215         for (ShardResponse srsp : sreq.responses) {
216           if (srsp.getException() != null) {
217             // can't expect the debug content if there was an exception for this request
218             // this should only happen when using shards.tolerant=true
219             continue;
220           }
221           NamedList sdebug = (NamedList)srsp.getSolrResponse().getResponse().get("debug");
222           info = (NamedList)merge(sdebug, info, EXCLUDE_SET);
223           if ((sreq.purpose & ShardRequest.PURPOSE_GET_DEBUG) != 0) {
224             hasGetDebugResponses = true;
225             if (rb.isDebugResults()) {
226               NamedList sexplain = (NamedList)sdebug.get("explain");
227               for (int i = 0; i < sexplain.size(); i++) {
228                 String id = sexplain.getName(i);
229                 // TODO: lookup won't work for non-string ids... String vs Float
230                 ShardDoc sdoc = rb.resultIds.get(id);
231                 int idx = sdoc.positionInResponse;
232                 arr[idx] = new NamedList.NamedListEntry<>(id, sexplain.getVal(i));
233               }
234             }
235           }
236         }
237       }
238 
239       if (rb.isDebugResults()) {
240          explain = SolrPluginUtils.removeNulls(arr, new SimpleOrderedMap<>());
241       }
242 
243       if (!hasGetDebugResponses) {
244         if (info == null) {
245           info = new SimpleOrderedMap<>();
246         }
247         // No responses were received from shards. Show local query info.
248         SolrPluginUtils.doStandardQueryDebug(
249                 rb.req, rb.getQueryString(),  rb.wrap(rb.getQuery()), rb.isDebugQuery(), info);
250         if (rb.isDebugQuery() && rb.getQparser() != null) {
251           rb.getQparser().addDebugInfo(info);
252         }
253       }
254       if (rb.isDebugResults()) {
255         int idx = info.indexOf("explain",0);
256         if (idx>=0) {
257           info.setVal(idx, explain);
258         } else {
259           info.add("explain", explain);
260         }
261       }
262 
263       rb.setDebugInfo(info);
264       rb.rsp.add("debug", rb.getDebugInfo() );
265     }
266     
267   }

270   private NamedList<String> getTrackResponse(ShardResponse shardResponse) {
271     NamedList<String> namedList = new SimpleOrderedMap<>();
272     if (shardResponse.getException() != null) {
273       namedList.add("Exception", shardResponse.getException().getMessage());
274       return namedList;
275     }
276     NamedList<Object> responseNL = shardResponse.getSolrResponse().getResponse();
277     @SuppressWarnings("unchecked")
278     NamedList<Object> responseHeader = (NamedList<Object>)responseNL.get("responseHeader");
279     if(responseHeader != null) {
280       namedList.add("QTime", responseHeader.get("QTime").toString());
281     }
282     namedList.add("ElapsedTime", String.valueOf(shardResponse.getSolrResponse().getElapsedTime()));
283     namedList.add("RequestPurpose", shardResponse.getShardRequest().params.get(CommonParams.REQUEST_PURPOSE));
284     SolrDocumentList docList = (SolrDocumentList)shardResponse.getSolrResponse().getResponse().get("response");
285     if(docList != null) {
286       namedList.add("NumFound", String.valueOf(docList.getNumFound()));
287     }
288     namedList.add("Response", String.valueOf(responseNL));
289     return namedList;
290   }

292   Object merge(Object source, Object dest, Set<String> exclude) {
293     if (source == null) return dest;
294     if (dest == null) {
295       if (source instanceof NamedList) {
296         dest = source instanceof SimpleOrderedMap ? new SimpleOrderedMap() : new NamedList();
297       } else {
298         return source;
299       }
300     } else {
301 
302       if (dest instanceof Collection) {
303         if (source instanceof Collection) {
304           ((Collection)dest).addAll((Collection)source);
305         } else {
306           ((Collection)dest).add(source);
307         }
308         return dest;
309       } else if (source instanceof Number) {
310         if (dest instanceof Number) {
311           if (source instanceof Double || dest instanceof Double) {
312             return ((Number)source).doubleValue() + ((Number)dest).doubleValue();
313           }
314           return ((Number)source).longValue() + ((Number)dest).longValue();
315         }
316         // fall through
317       } else if (source instanceof String) {
318         if (source.equals(dest)) {
319           return dest;
320         }
321         // fall through
322       }
323     }
324 
325 
326     if (source instanceof NamedList && dest instanceof NamedList) {
327       NamedList<Object> tmp = new NamedList<>();
328       @SuppressWarnings("unchecked")
329       NamedList<Object> sl = (NamedList<Object>)source;
330       @SuppressWarnings("unchecked")
331       NamedList<Object> dl = (NamedList<Object>)dest;
332       for (int i=0; i<sl.size(); i++) {
333         String skey = sl.getName(i);
334         if (exclude != null && exclude.contains(skey)) continue;
335         Object sval = sl.getVal(i);
336         int didx = -1;
337 
338         // optimize case where elements are in same position
339         if (i < dl.size()) {
340           String dkey = dl.getName(i);
341           if (skey == dkey || (skey!=null && skey.equals(dkey))) {
342             didx = i;
343           }
344         }
345 
346         if (didx == -1) {
347           didx = dl.indexOf(skey, 0);
348         }
349 
350         if (didx == -1) {
351           tmp.add(skey, merge(sval, null, null));
352         } else {
353           dl.setVal(didx, merge(sval, dl.getVal(didx), null));
354         }
355       }
356       dl.addAll(tmp);
357       return dl;
358     }
359 
360     // only add to list if JSON is different
361     if (source.equals(dest)) return source;
362 
363     // merge unlike elements in a list
364     List<Object> t = new ArrayList<>();
365     t.add(dest);
366     t.add(source);
367     return t;
368   }
