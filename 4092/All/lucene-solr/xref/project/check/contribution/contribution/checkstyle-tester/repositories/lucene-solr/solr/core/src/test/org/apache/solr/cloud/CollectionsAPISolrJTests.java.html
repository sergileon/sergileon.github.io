73    protected void testCreateAndDeleteCollection() throws Exception {
74      String collectionName = "solrj_test";
75      CollectionAdminRequest.Create createCollectionRequest = new CollectionAdminRequest.Create()
76              .setCollectionName(collectionName)
77              .setNumShards(2)
78              .setReplicationFactor(2)
79              .setConfigName("conf1")
80              .setRouterField("myOwnField")
81              .setStateFormat(1);
82      CollectionAdminResponse response = createCollectionRequest.process(cloudClient);
83  
84      assertEquals(0, response.getStatus());
85      assertTrue(response.isSuccess());
86      Map<String, NamedList<Integer>> coresStatus = response.getCollectionCoresStatus();
87      assertEquals(4, coresStatus.size());
88      for (int i=0; i<4; i++) {
89        NamedList<Integer> status = coresStatus.get(collectionName + "_shard" + (i/2+1) + "_replica" + (i%2+1));
90        assertEquals(0, (int)status.get("status"));
91        assertTrue(status.get("QTime") > 0);
92      }
93  
94      cloudClient.setDefaultCollection(collectionName);
95      CollectionAdminRequest.Delete deleteCollectionRequest = new CollectionAdminRequest.Delete()
96              .setCollectionName(collectionName);
97      response = deleteCollectionRequest.process(cloudClient);
98  
99      assertEquals(0, response.getStatus());
100     assertTrue(response.isSuccess());
101     Map<String,NamedList<Integer>> nodesStatus = response.getCollectionNodesStatus();
102     assertNull("Deleted collection " + collectionName + "still exists",
103         cloudClient.getZkStateReader().getClusterState().getCollectionOrNull(collectionName));
104     assertEquals(4, nodesStatus.size());
105     
106     // Test Creating a collection with new stateformat.
107     collectionName = "solrj_newstateformat";
108     createCollectionRequest = new CollectionAdminRequest.Create()
109             .setCollectionName(collectionName)
110             .setNumShards(2)
111             .setConfigName("conf1")
112             .setStateFormat(2);
113 
114     response = createCollectionRequest.process(cloudClient);
115     assertEquals(0, response.getStatus());
116     assertTrue(response.isSuccess());
117 
118     waitForRecoveriesToFinish(collectionName, false);
119     assertTrue("Collection state does not exist",
120         cloudClient.getZkStateReader().getZkClient()
121             .exists(ZkStateReader.getCollectionPath(collectionName), true));
122 
123   }

231   private void testCreateCollectionWithPropertyParam() throws Exception {
232     String collectionName = "solrj_test_core_props";
233     
234     File tmpDir = createTempDir("testPropertyParamsForCreate").toFile();
235     File instanceDir = new File(tmpDir, "instanceDir-" + TestUtil.randomSimpleString(random(), 1, 5));
236     File dataDir = new File(tmpDir, "dataDir-" + TestUtil.randomSimpleString(random(), 1, 5));
237     File ulogDir = new File(tmpDir, "ulogDir-" + TestUtil.randomSimpleString(random(), 1, 5));
238 
239     Properties properties = new Properties();
240     properties.put(CoreAdminParams.INSTANCE_DIR, instanceDir.getAbsolutePath());
241     properties.put(CoreAdminParams.DATA_DIR, dataDir.getAbsolutePath());
242     properties.put(CoreAdminParams.ULOG_DIR, ulogDir.getAbsolutePath());
243 
244     CollectionAdminRequest.Create createReq = new CollectionAdminRequest.Create()
245             .setCollectionName(collectionName)
246             .setNumShards(1)
247             .setConfigName("conf1")
248             .setProperties(properties);
249 
250     CollectionAdminResponse response = createReq.process(cloudClient);
251     assertEquals(0, response.getStatus());
252     assertTrue(response.isSuccess());
253     Map<String, NamedList<Integer>> coresStatus = response.getCollectionCoresStatus();
254     assertEquals(1, coresStatus.size());
255 
256     DocCollection testCollection = cloudClient.getZkStateReader()
257         .getClusterState().getCollection(collectionName);
258 
259     Replica replica1 = testCollection.getReplica("core_node1");
260 
261     try (HttpSolrClient client = new HttpSolrClient(replica1.getStr("base_url"))) {
262       CoreAdminResponse status = CoreAdminRequest.getStatus(replica1.getStr("core"), client);
263       NamedList<Object> coreStatus = status.getCoreStatus(replica1.getStr("core"));
264       String dataDirStr = (String) coreStatus.get("dataDir");
265       String instanceDirStr = (String) coreStatus.get("instanceDir");
266       assertEquals("Instance dir does not match param passed in property.instanceDir syntax",
267           new File(instanceDirStr).getAbsolutePath(), instanceDir.getAbsolutePath());
268       assertEquals("Data dir does not match param given in property.dataDir syntax",
269           new File(dataDirStr).getAbsolutePath(), dataDir.getAbsolutePath());
270     }
271 
272     CollectionAdminRequest.Delete deleteCollectionRequest = new CollectionAdminRequest.Delete();
273     deleteCollectionRequest.setCollectionName(collectionName);
274     deleteCollectionRequest.process(cloudClient);
275   }

277   private void testAddAndDeleteReplica() throws Exception {
278     String collectionName = "solrj_replicatests";
279     createCollection(collectionName, cloudClient, 1, 2);
280 
281     cloudClient.setDefaultCollection(collectionName);
282 
283     String newReplicaName = Assign.assignNode(collectionName, cloudClient.getZkStateReader().getClusterState());
284     ArrayList<String> nodeList = new ArrayList<>(cloudClient.getZkStateReader().getClusterState().getLiveNodes());
285     Collections.shuffle(nodeList, random());
286     CollectionAdminRequest.AddReplica addReplica = new CollectionAdminRequest.AddReplica()
287             .setCollectionName(collectionName)
288             .setShardName("shard1")
289             .setNode(nodeList.get(0));
290     CollectionAdminResponse response = addReplica.process(cloudClient);
291 
292     assertEquals(0, response.getStatus());
293     assertTrue(response.isSuccess());
294 
295     TimeOut timeout = new TimeOut(3, TimeUnit.SECONDS);
296     Replica newReplica = null;
297 
298     while (! timeout.hasTimedOut() && newReplica == null) {
299       Slice slice = cloudClient.getZkStateReader().getClusterState().getSlice(collectionName, "shard1");
300       newReplica = slice.getReplica(newReplicaName);
301     }
302 
303     assertNotNull(newReplica);
304 
305     assertEquals("Replica should be created on the right node",
306         cloudClient.getZkStateReader().getBaseUrlForNodeName(nodeList.get(0)),
307         newReplica.getStr(ZkStateReader.BASE_URL_PROP)
308     );
309     
310     // Test DELETEREPLICA
311     CollectionAdminRequest.DeleteReplica deleteReplicaRequest = new CollectionAdminRequest.DeleteReplica()
312             .setCollectionName(collectionName)
313             .setShardName("shard1")
314             .setReplica(newReplicaName);
315     response = deleteReplicaRequest.process(cloudClient);
316 
317     assertEquals(0, response.getStatus());
318 
319     timeout = new TimeOut(3, TimeUnit.SECONDS);
320 
321     while (! timeout.hasTimedOut() && newReplica != null) {
322       Slice slice = cloudClient.getZkStateReader().getClusterState().getSlice(collectionName, "shard1");
323       newReplica = slice.getReplica(newReplicaName);
324     }
325 
326     assertNull(newReplica);
327   }

329   private void testClusterProp() throws InterruptedException, IOException, SolrServerException {
330     CollectionAdminRequest.ClusterProp clusterPropRequest = new CollectionAdminRequest.ClusterProp()
331             .setPropertyName(ZkStateReader.LEGACY_CLOUD)
332             .setPropertyValue("false");
333     CollectionAdminResponse response = clusterPropRequest.process(cloudClient);
334 
335     assertEquals(0, response.getStatus());
336 
337     TimeOut timeout = new TimeOut(3, TimeUnit.SECONDS);
338     boolean changed = false;
339     
340     while(! timeout.hasTimedOut()){
341       Thread.sleep(10);
342       changed = Objects.equals("false",
343           cloudClient.getZkStateReader().getClusterProps().get(ZkStateReader.LEGACY_CLOUD));
344       if(changed) break;
345     }
346     assertTrue("The Cluster property wasn't set", changed);
347     
348     // Unset ClusterProp that we set.
349     clusterPropRequest = new CollectionAdminRequest.ClusterProp()
350             .setPropertyName(ZkStateReader.LEGACY_CLOUD)
351             .setPropertyValue(null);
352     clusterPropRequest.process(cloudClient);
353 
354     timeout = new TimeOut(3, TimeUnit.SECONDS);
355     changed = false;
356     while(! timeout.hasTimedOut()) {
357       Thread.sleep(10);
358       changed = (cloudClient.getZkStateReader().getClusterProps().get(ZkStateReader.LEGACY_CLOUD) == null);
359       if(changed)  
360         break;
361     }
362     assertTrue("The Cluster property wasn't unset", changed);
363   }

365   private void testAddAndRemoveRole() throws InterruptedException, IOException, SolrServerException {
366     cloudClient.setDefaultCollection(DEFAULT_COLLECTION);
367     Replica replica = cloudClient.getZkStateReader().getLeaderRetry(DEFAULT_COLLECTION, SHARD1);
368     CollectionAdminRequest.AddRole addRoleRequest = new CollectionAdminRequest.AddRole()
369             .setNode(replica.getNodeName())
370             .setRole("overseer");
371     addRoleRequest.process(cloudClient);
372 
373     CollectionAdminRequest.ClusterStatus clusterStatusRequest = new CollectionAdminRequest.ClusterStatus()
374             .setCollectionName(DEFAULT_COLLECTION);
375     CollectionAdminResponse response = clusterStatusRequest.process(cloudClient);
376 
377     NamedList<Object> rsp = response.getResponse();
378     NamedList<Object> cluster = (NamedList<Object>) rsp.get("cluster");
379     assertNotNull("Cluster state should not be null", cluster);
380     Map<String, Object> roles = (Map<String, Object>) cluster.get("roles");
381     assertNotNull("Role information should not be null", roles);
382     List<String> overseer = (List<String>) roles.get("overseer");
383     assertNotNull(overseer);
384     assertEquals(1, overseer.size());
385     assertTrue(overseer.contains(replica.getNodeName()));
386     
387     // Remove role
388     new CollectionAdminRequest.RemoveRole()
389             .setNode(replica.getNodeName())
390             .setRole("overseer")
391             .process(cloudClient);
392 
393     clusterStatusRequest = new CollectionAdminRequest.ClusterStatus();
394     clusterStatusRequest.setCollectionName(DEFAULT_COLLECTION);
395     response = clusterStatusRequest.process(cloudClient);
396 
397     rsp = response.getResponse();
398     cluster = (NamedList<Object>) rsp.get("cluster");
399     assertNotNull("Cluster state should not be null", cluster);
400     roles = (Map<String, Object>) cluster.get("roles");
401     assertNotNull("Role information should not be null", roles);
402     overseer = (List<String>) roles.get("overseer");
403     assertFalse(overseer.contains(replica.getNodeName()));
404   }

418   private void testAddAndDeleteReplicaProp() throws InterruptedException, IOException, SolrServerException {
419     Replica replica = cloudClient.getZkStateReader().getLeaderRetry(DEFAULT_COLLECTION, SHARD1);
420     CollectionAdminResponse response = new CollectionAdminRequest.AddReplicaProp()
421             .setCollectionName(DEFAULT_COLLECTION)
422             .setShardName(SHARD1)
423             .setReplica(replica.getName())
424             .setPropertyName("preferredleader")
425             .setPropertyValue("true").process(cloudClient);
426     assertEquals(0, response.getStatus());
427 
428     TimeOut timeout = new TimeOut(20, TimeUnit.SECONDS);
429     String propertyValue = null;
430     
431     String replicaName = replica.getName();
432     while (! timeout.hasTimedOut()) {
433       ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
434       replica = clusterState.getReplica(DEFAULT_COLLECTION, replicaName);
435       propertyValue = replica.getStr("property.preferredleader"); 
436       if(StringUtils.equals("true", propertyValue))
437         break;
438       Thread.sleep(50);
439     }
440     
441     assertEquals("Replica property was not updated, Latest value: " +
442         cloudClient.getZkStateReader().getClusterState().getReplica(DEFAULT_COLLECTION, replicaName),
443         "true",
444         propertyValue);
445 
446     response = new CollectionAdminRequest.DeleteReplicaProp()
447             .setCollectionName(DEFAULT_COLLECTION)
448             .setShardName(SHARD1)
449             .setReplica(replicaName)
450             .setPropertyName("property.preferredleader").process(cloudClient);
451     assertEquals(0, response.getStatus());
452 
453     timeout = new TimeOut(20, TimeUnit.SECONDS);
454     boolean updated = false;
455 
456     while (! timeout.hasTimedOut()) {
457       ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
458       replica = clusterState.getReplica(DEFAULT_COLLECTION, replicaName);
459       updated = replica.getStr("property.preferredleader") == null;
460       if(updated)
461         break;
462       Thread.sleep(50);
463     }
464 
465     assertTrue("Replica property was not removed", updated);
466     
467   }
