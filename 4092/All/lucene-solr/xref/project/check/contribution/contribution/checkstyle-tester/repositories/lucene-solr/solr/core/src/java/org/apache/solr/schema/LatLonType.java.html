56  /**
57   * Represents a Latitude/Longitude as a 2 dimensional point.  Latitude is <b>always</b> specified first.
58   */
59  public class LatLonType extends AbstractSubTypeFieldType implements SpatialQueryable {
60    protected static final int LAT = 0;
61    protected static final int LON = 1;
62  
63    @Override
64    protected void init(IndexSchema schema, Map<String, String> args) {
65      super.init(schema, args);
66      //TODO: refactor this, as we are creating the suffix cache twice, since the super.init does it too
67      createSuffixCache(3);//we need three extra fields: one for the storage field, two for the lat/lon
68    }

70    @Override
71    public List<IndexableField> createFields(SchemaField field, Object value, float boost) {
72      String externalVal = value.toString();
73      //we could have 3 fields (two for the lat & lon, one for storage)
74      List<IndexableField> f = new ArrayList<>(3);
75      if (field.indexed()) {
76        Point point = SpatialUtils.parsePointSolrException(externalVal, SpatialContext.GEO);
77        //latitude
78        SchemaField subLatSF = subField(field, LAT, schema);
79        f.add(subLatSF.createField(String.valueOf(point.getY()), subLatSF.indexed() && !subLatSF.omitNorms() ? boost : 1f));
80        //longitude
81        SchemaField subLonSF = subField(field, LON, schema);
82        f.add(subLonSF.createField(String.valueOf(point.getX()), subLonSF.indexed() && !subLonSF.omitNorms() ? boost : 1f));
83      }
84  
85      if (field.stored()) {
86        FieldType customType = new FieldType();
87        customType.setStored(true);
88        f.add(createField(field.getName(), externalVal, customType, 1f));
89      }
90      return f;
91    }

94    @Override
95    public Query getRangeQuery(QParser parser, SchemaField field, String part1, String part2, boolean minInclusive, boolean maxInclusive) {
96      Point p1 = SpatialUtils.parsePointSolrException(part1, SpatialContext.GEO);
97      Point p2 = SpatialUtils.parsePointSolrException(part2, SpatialContext.GEO);
98  
99      SchemaField latSF = subField(field, LAT, parser.getReq().getSchema());
100     SchemaField lonSF = subField(field, LON, parser.getReq().getSchema());
101     BooleanQuery.Builder result = new BooleanQuery.Builder();
102     result.setDisableCoord(true);
103     // points must currently be ordered... should we support specifying any two opposite corner points?
104     result.add(latSF.getType().getRangeQuery(parser, latSF,
105         Double.toString(p1.getY()), Double.toString(p2.getY()), minInclusive, maxInclusive), BooleanClause.Occur.MUST);
106     result.add(lonSF.getType().getRangeQuery(parser, lonSF,
107         Double.toString(p1.getX()), Double.toString(p2.getX()), minInclusive, maxInclusive), BooleanClause.Occur.MUST);
108     return result.build();
109   }

111   @Override
112   public Query getFieldQuery(QParser parser, SchemaField field, String externalVal) {
113     Point p1 = SpatialUtils.parsePointSolrException(externalVal, SpatialContext.GEO);
114 
115     SchemaField latSF = subField(field, LAT, parser.getReq().getSchema());
116     SchemaField lonSF = subField(field, LON, parser.getReq().getSchema());
117     BooleanQuery.Builder result = new BooleanQuery.Builder();
118     result.setDisableCoord(true);
119     result.add(latSF.getType().getFieldQuery(parser, latSF,
120         Double.toString(p1.getY())), BooleanClause.Occur.MUST);
121     result.add(lonSF.getType().getFieldQuery(parser, lonSF,
122         Double.toString(p1.getX())), BooleanClause.Occur.MUST);
123     return result.build();
124   }

128   @Override
129   public Query createSpatialQuery(QParser parser, SpatialOptions options) {
130     Point point = SpatialUtils.parsePointSolrException(options.pointStr, SpatialContext.GEO);
131 
132     // lat & lon in degrees
133     double latCenter = point.getY();
134     double lonCenter = point.getX();
135     
136     double distDeg = DistanceUtils.dist2Degrees(options.distance, options.radius);
137     Rectangle bbox = DistanceUtils.calcBoxByDistFromPtDEG(latCenter, lonCenter, distDeg, SpatialContext.GEO, null);
138     double latMin = bbox.getMinY();
139     double latMax = bbox.getMaxY();
140     double lonMin, lonMax, lon2Min, lon2Max;
141     if (bbox.getCrossesDateLine()) {
142        lonMin = -180;
143        lonMax = bbox.getMaxX();
144        lon2Min = bbox.getMinX();
145        lon2Max = 180;
146     } else {
147        lonMin = bbox.getMinX();
148        lonMax = bbox.getMaxX();
149        lon2Min = -180;
150        lon2Max = 180;
151     }
152     
153     IndexSchema schema = parser.getReq().getSchema();
154     
155     // Now that we've figured out the ranges, build them!
156     SchemaField latSF = subField(options.field, LAT, schema);
157     SchemaField lonSF = subField(options.field, LON, schema);
158 
159     SpatialDistanceQuery spatial = new SpatialDistanceQuery();
160 
161 
162     if (options.bbox) {
163       BooleanQuery.Builder result = new BooleanQuery.Builder();
164 
165       Query latRange = latSF.getType().getRangeQuery(parser, latSF,
166                 String.valueOf(latMin),
167                 String.valueOf(latMax),
168                 true, true);
169       result.add(latRange, BooleanClause.Occur.MUST);
170 
171       if (lonMin != -180 || lonMax != 180) {
172         Query lonRange = lonSF.getType().getRangeQuery(parser, lonSF,
173                 String.valueOf(lonMin),
174                 String.valueOf(lonMax),
175                 true, true);
176         if (lon2Min != -180 || lon2Max != 180) {
177           // another valid longitude range
178           BooleanQuery.Builder bothLons = new BooleanQuery.Builder();
179           bothLons.add(lonRange, BooleanClause.Occur.SHOULD);
180 
181           lonRange = lonSF.getType().getRangeQuery(parser, lonSF,
182                 String.valueOf(lon2Min),
183                 String.valueOf(lon2Max),
184                 true, true);
185           bothLons.add(lonRange, BooleanClause.Occur.SHOULD);
186 
187           lonRange = bothLons.build();
188         }
189 
190         result.add(lonRange, BooleanClause.Occur.MUST);
191       }
192 
193       spatial.bboxQuery = result.build();
194     }
195 
196 
197     spatial.origField = options.field.getName();
198     spatial.latSource = latSF.getType().getValueSource(latSF, parser);
199     spatial.lonSource = lonSF.getType().getValueSource(lonSF, parser);
200     spatial.latMin = latMin;
201     spatial.latMax = latMax;
202     spatial.lonMin = lonMin;
203     spatial.lonMax = lonMax;
204     spatial.lon2Min = lon2Min;
205     spatial.lon2Max = lon2Max;
206     spatial.lon2 = lon2Min != -180 || lon2Max != 180;
207 
208     spatial.latCenter = latCenter;
209     spatial.lonCenter = lonCenter;
210     spatial.dist = options.distance;
211     spatial.planetRadius = options.radius;
212 
213     spatial.calcDist = !options.bbox;
214 
215     return spatial;
216   }

289 class SpatialDistanceQuery extends ExtendedQueryBase implements PostFilter {
290   String origField;
291   ValueSource latSource;
292   ValueSource lonSource;
293   double lonMin, lonMax, lon2Min, lon2Max, latMin, latMax;
294   boolean lon2;
295 
296   boolean calcDist;  // actually calculate the distance with haversine
297   Query bboxQuery;
298 
299   double latCenter;
300   double lonCenter;
301   double dist;
302   double planetRadius;
303 
304 
305   @Override
306   public Query rewrite(IndexReader reader) throws IOException {
307     if (getBoost() != 1f) {
308       return super.rewrite(reader);
309     }
310     return bboxQuery != null ? bboxQuery.rewrite(reader) : this;
311   }

313   protected class SpatialWeight extends ConstantScoreWeight {
314     protected IndexSearcher searcher;
315     protected Map latContext;
316     protected Map lonContext;
317 
318     public SpatialWeight(IndexSearcher searcher) throws IOException {
319       super(SpatialDistanceQuery.this);
320       this.searcher = searcher;
321       this.latContext = ValueSource.newContext(searcher);
322       this.lonContext = ValueSource.newContext(searcher);
323       latSource.createWeight(latContext, searcher);
324       lonSource.createWeight(lonContext, searcher);
325     }
326 
327     @Override
328     public Scorer scorer(LeafReaderContext context) throws IOException {
329       return new SpatialScorer(context, this, score());
330     }
331 
332     @Override
333     public Explanation explain(LeafReaderContext context, int doc) throws IOException {
334       return ((SpatialScorer)scorer(context)).explain(super.explain(context, doc), doc);
335     }
336   }

338   protected class SpatialScorer extends Scorer {
339     final IndexReader reader;
340     final SpatialWeight weight;
341     final int maxDoc;
342     final float qWeight;
343     int doc=-1;
344     final FunctionValues latVals;
345     final FunctionValues lonVals;
346 
347 
348     final double lonMin, lonMax, lon2Min, lon2Max, latMin, latMax;
349     final boolean lon2;
350     final boolean calcDist;
351     
352     final double latCenterRad;
353     final double lonCenterRad;
354     final double latCenterRad_cos;
355     final double dist;
356     final double planetRadius;
357 
358     int lastDistDoc;
359     double lastDist;
360 
361     public SpatialScorer(LeafReaderContext readerContext, SpatialWeight w, float qWeight) throws IOException {
362       super(w);
363       this.weight = w;
364       this.qWeight = qWeight;
365       this.reader = readerContext.reader();
366       this.maxDoc = reader.maxDoc();
367       latVals = latSource.getValues(weight.latContext, readerContext);
368       lonVals = lonSource.getValues(weight.lonContext, readerContext);
369 
370       this.lonMin = SpatialDistanceQuery.this.lonMin;
371       this.lonMax = SpatialDistanceQuery.this.lonMax;
372       this.lon2Min = SpatialDistanceQuery.this.lon2Min;
373       this.lon2Max = SpatialDistanceQuery.this.lon2Max;
374       this.latMin = SpatialDistanceQuery.this.latMin;
375       this.latMax = SpatialDistanceQuery.this.latMax;
376       this.lon2 = SpatialDistanceQuery.this.lon2;
377       this.calcDist = SpatialDistanceQuery.this.calcDist;
378 
379       this.latCenterRad = SpatialDistanceQuery.this.latCenter * DistanceUtils.DEGREES_TO_RADIANS;
380       this.lonCenterRad = SpatialDistanceQuery.this.lonCenter * DistanceUtils.DEGREES_TO_RADIANS;
381       this.latCenterRad_cos = this.calcDist ? Math.cos(latCenterRad) : 0;
382       this.dist = SpatialDistanceQuery.this.dist;
383       this.planetRadius = SpatialDistanceQuery.this.planetRadius;
384 
385     }
386 
387     boolean match() {
388       // longitude should generally be more restrictive than latitude
389       // (e.g. in the US, it immediately separates the coasts, and in world search separates
390       // US from Europe from Asia, etc.
391       double lon = lonVals.doubleVal(doc);
392       if (! ((lon >= lonMin && lon <=lonMax) || (lon2 && lon >= lon2Min && lon <= lon2Max)) ) {
393         return false;
394       }
395 
396       double lat = latVals.doubleVal(doc);
397       if ( !(lat >= latMin && lat <= latMax) ) {
398         return false;
399       }
400 
401       if (!calcDist) return true;
402 
403       // TODO: test for internal box where we wouldn't need to calculate the distance
404 
405       return dist(lat, lon) <= dist;
406     }
407 
408     double dist(double lat, double lon) {
409       double latRad = lat * DistanceUtils.DEGREES_TO_RADIANS;
410       double lonRad = lon * DistanceUtils.DEGREES_TO_RADIANS;
411       
412       // haversine, specialized to avoid a cos() call on latCenterRad
413       double diffX = latCenterRad - latRad;
414       double diffY = lonCenterRad - lonRad;
415       double hsinX = Math.sin(diffX * 0.5);
416       double hsinY = Math.sin(diffY * 0.5);
417       double h = hsinX * hsinX +
418               (latCenterRad_cos * Math.cos(latRad) * hsinY * hsinY);
419       double result = (planetRadius * 2 * Math.atan2(Math.sqrt(h), Math.sqrt(1 - h)));
420 
421       // save the results of this calculation
422       lastDistDoc = doc;
423       lastDist = result;
424       
425       return result;
426     }
427 
428     @Override
429     public int docID() {
430       return doc;
431     }
432 
433     // instead of matching all docs, we could also embed a query.
434     // the score could either ignore the subscore, or boost it.
435     // Containment:  floatline(foo:myTerm, "myFloatField", 1.0, 0.0f)
436     // Boost:        foo:myTerm^floatline("myFloatField",1.0,0.0f)
437     @Override
438     public int nextDoc() throws IOException {
439       for(;;) {
440         ++doc;
441         if (doc>=maxDoc) {
442           return doc=NO_MORE_DOCS;
443         }
444         if (!match()) continue;
445         return doc;
446       }
447     }
448 
449     @Override
450     public int advance(int target) throws IOException {
451       // this will work even if target==NO_MORE_DOCS
452       doc=target-1;
453       return nextDoc();
454     }
455 
456     @Override
457     public float score() throws IOException {
458       double dist = (doc == lastDistDoc) ? lastDist : dist(latVals.doubleVal(doc), lonVals.doubleVal(doc));
459       return (float)(dist * qWeight);
460     }
461 
462     @Override
463     public int freq() throws IOException {
464       return 1;
465     }
466 
467     @Override
468     public long cost() {
469       return maxDoc;
470     }
471 
472     public Explanation explain(Explanation base, int doc) throws IOException {
473       if (base.isMatch() == false) {
474         return base;
475       }
476       double dist = dist(latVals.doubleVal(doc), lonVals.doubleVal(doc));
477 
478       String description = SpatialDistanceQuery.this.toString();
479       return Explanation.match((float) (base.getValue() * dist), description + " product of:",
480           base, Explanation.match((float) dist, "hsin("+latVals.doubleVal(doc)+","+lonVals.doubleVal(doc)));
481     }
482 
483   }

506       spatialScorer.doc = doc;

552             && this.latMin == other.latMin
553             && this.latMax == other.latMax

556             && this.lon2Min == other.lon2Min

558             && this.dist == other.dist

564   }

566   /** Returns a hash code value for this object. */
567   @Override
568   public int hashCode() {
569     // don't bother making the hash expensive - the center latitude + min longitude will be very unique
570     long hash = Double.doubleToLongBits(latCenter);
571     hash = hash * 31 + Double.doubleToLongBits(lonMin);
572     hash = hash * 31 + (long)super.hashCode();
573     return (int)(hash >> 32 + hash);
574   }
