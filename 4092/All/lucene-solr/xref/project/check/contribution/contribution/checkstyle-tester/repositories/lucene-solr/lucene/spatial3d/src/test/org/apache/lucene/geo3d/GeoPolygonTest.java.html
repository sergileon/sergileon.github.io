32    @Test
33    public void testPolygonPointWithin() {
34      GeoMembershipShape c;
35      GeoPoint gp;
36      List<GeoPoint> points;
37  
38      points = new ArrayList<GeoPoint>();
39      points.add(new GeoPoint(PlanetModel.SPHERE, -0.1, -0.5));
40      points.add(new GeoPoint(PlanetModel.SPHERE, 0.0, -0.6));
41      points.add(new GeoPoint(PlanetModel.SPHERE, 0.1, -0.5));
42      points.add(new GeoPoint(PlanetModel.SPHERE, 0.0, -0.4));
43  
44      c = GeoPolygonFactory.makeGeoPolygon(PlanetModel.SPHERE, points, 0);
45      // Sample some points within
46      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.5);
47      assertTrue(c.isWithin(gp));
48      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.55);
49      assertTrue(c.isWithin(gp));
50      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.45);
51      assertTrue(c.isWithin(gp));
52      gp = new GeoPoint(PlanetModel.SPHERE, -0.05, -0.5);
53      assertTrue(c.isWithin(gp));
54      gp = new GeoPoint(PlanetModel.SPHERE, 0.05, -0.5);
55      assertTrue(c.isWithin(gp));
56      // Sample some nearby points outside
57      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.65);
58      assertFalse(c.isWithin(gp));
59      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.35);
60      assertFalse(c.isWithin(gp));
61      gp = new GeoPoint(PlanetModel.SPHERE, -0.15, -0.5);
62      assertFalse(c.isWithin(gp));
63      gp = new GeoPoint(PlanetModel.SPHERE, 0.15, -0.5);
64      assertFalse(c.isWithin(gp));
65      // Random points outside
66      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, 0.0);
67      assertFalse(c.isWithin(gp));
68      gp = new GeoPoint(PlanetModel.SPHERE, Math.PI * 0.5, 0.0);
69      assertFalse(c.isWithin(gp));
70      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, Math.PI);
71      assertFalse(c.isWithin(gp));
72  
73      points = new ArrayList<GeoPoint>();
74      points.add(new GeoPoint(PlanetModel.SPHERE, -0.1, -0.5));
75      points.add(new GeoPoint(PlanetModel.SPHERE, -0.01, -0.6));
76      points.add(new GeoPoint(PlanetModel.SPHERE, -0.1, -0.7));
77      points.add(new GeoPoint(PlanetModel.SPHERE, 0.0, -0.8));
78      points.add(new GeoPoint(PlanetModel.SPHERE, 0.1, -0.7));
79      points.add(new GeoPoint(PlanetModel.SPHERE, 0.01, -0.6));
80      points.add(new GeoPoint(PlanetModel.SPHERE, 0.1, -0.5));
81      points.add(new GeoPoint(PlanetModel.SPHERE, 0.0, -0.4));
82          
83          /*
84          System.out.println("Points: ");
85          for (GeoPoint p : points) {
86              System.out.println(" "+p);
87          }
88          */
89  
90      c = GeoPolygonFactory.makeGeoPolygon(PlanetModel.SPHERE, points, 0);
91      // Sample some points within
92      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.5);
93      assertTrue(c.isWithin(gp));
94      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.55);
95      assertTrue(c.isWithin(gp));
96      gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.45);
97      assertTrue(c.isWithin(gp));
98      gp = new GeoPoint(PlanetModel.SPHERE, -0.05, -0.5);
99      assertTrue(c.isWithin(gp));
100     gp = new GeoPoint(PlanetModel.SPHERE, 0.05, -0.5);
101     assertTrue(c.isWithin(gp));
102     gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.7);
103     assertTrue(c.isWithin(gp));
104     // Sample some nearby points outside
105     gp = new GeoPoint(PlanetModel.SPHERE, 0.0, -0.35);
106     assertFalse(c.isWithin(gp));
107     gp = new GeoPoint(PlanetModel.SPHERE, -0.15, -0.5);
108     assertFalse(c.isWithin(gp));
109     gp = new GeoPoint(PlanetModel.SPHERE, 0.15, -0.5);
110     assertFalse(c.isWithin(gp));
111     // Random points outside
112     gp = new GeoPoint(PlanetModel.SPHERE, 0.0, 0.0);
113     assertFalse(c.isWithin(gp));
114     gp = new GeoPoint(PlanetModel.SPHERE, Math.PI * 0.5, 0.0);
115     assertFalse(c.isWithin(gp));
116     gp = new GeoPoint(PlanetModel.SPHERE, 0.0, Math.PI);
117     assertFalse(c.isWithin(gp));

121   @Test
122   public void testPolygonBounds() {
123     GeoMembershipShape c;
124     LatLonBounds b;
125     List<GeoPoint> points;
126     XYZBounds xyzb;
127     GeoPoint point;
128     GeoArea area;
129     
130     // BKD failure
131     points = new ArrayList<GeoPoint>();
132     points.add(new GeoPoint(PlanetModel.WGS84, -0.36716183577912814, 1.4836349969188696));
133     points.add(new GeoPoint(PlanetModel.WGS84, 0.7846038240742979, -0.02743348424931823));
134     points.add(new GeoPoint(PlanetModel.WGS84, -0.7376479402362607, -0.5072961758807019));
135     points.add(new GeoPoint(PlanetModel.WGS84, -0.3760415907667887, 1.4970455334565513));
136     
137     c = GeoPolygonFactory.makeGeoPolygon(PlanetModel.WGS84, points, 1);
138     
139     point = new GeoPoint(PlanetModel.WGS84, -0.01580760332365284, -0.03956004622490505);
140     assertTrue(c.isWithin(point));
141     xyzb = new XYZBounds();
142     c.getBounds(xyzb);
143     area = GeoAreaFactory.makeGeoArea(PlanetModel.WGS84,
144       xyzb.getMinimumX(), xyzb.getMaximumX(), xyzb.getMinimumY(), xyzb.getMaximumY(), xyzb.getMinimumZ(), xyzb.getMaximumZ());
145     assertTrue(area.isWithin(point));
146     
147     points = new ArrayList<GeoPoint>();
148     points.add(new GeoPoint(PlanetModel.SPHERE, -0.1, -0.5));
149     points.add(new GeoPoint(PlanetModel.SPHERE, 0.0, -0.6));
150     points.add(new GeoPoint(PlanetModel.SPHERE, 0.1, -0.5));
151     points.add(new GeoPoint(PlanetModel.SPHERE, 0.0, -0.4));
152 
153     c = GeoPolygonFactory.makeGeoPolygon(PlanetModel.SPHERE, points, 0);
154 
155     b = new LatLonBounds();
156     c.getBounds(b);
157     assertFalse(b.checkNoLongitudeBound());
158     assertFalse(b.checkNoTopLatitudeBound());
159     assertFalse(b.checkNoBottomLatitudeBound());
160     assertEquals(-0.6, b.getLeftLongitude(), 0.000001);
161     assertEquals(-0.4, b.getRightLongitude(), 0.000001);
162     assertEquals(-0.1, b.getMinLatitude(), 0.000001);
163     assertEquals(0.1, b.getMaxLatitude(), 0.000001);
164   }
