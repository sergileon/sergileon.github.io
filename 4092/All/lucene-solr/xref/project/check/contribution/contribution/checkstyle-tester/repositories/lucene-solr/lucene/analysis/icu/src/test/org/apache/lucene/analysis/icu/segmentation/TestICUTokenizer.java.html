38    public void testHugeDoc() throws IOException {
39      StringBuilder sb = new StringBuilder();
40      char whitespace[] = new char[4094];
41      Arrays.fill(whitespace, ' ');
42      sb.append(whitespace);
43      sb.append("testing 1234");
44      String input = sb.toString();
45      ICUTokenizer tokenizer = new ICUTokenizer(newAttributeFactory(), new DefaultICUTokenizerConfig(false));
46      tokenizer.setReader(new StringReader(input));
47      assertTokenStreamContents(tokenizer, new String[] { "testing", "1234" });
48    }

50    public void testHugeTerm2() throws IOException {
51      StringBuilder sb = new StringBuilder();
52      for (int i = 0; i < 40960; i++) {
53        sb.append('a');
54      }
55      String input = sb.toString();
56      ICUTokenizer tokenizer = new ICUTokenizer(newAttributeFactory(), new DefaultICUTokenizerConfig(false));
57      tokenizer.setReader(new StringReader(input));
58      char token[] = new char[4096];
59      Arrays.fill(token, 'a');
60      String expectedToken = new String(token);
61      String expected[] = { 
62          expectedToken, expectedToken, expectedToken, 
63          expectedToken, expectedToken, expectedToken,
64          expectedToken, expectedToken, expectedToken,
65          expectedToken
66      };
67      assertTokenStreamContents(tokenizer, expected);
68    }

276   public void testTokenAttributes() throws Exception {
277     try (TokenStream ts = a.tokenStream("dummy", "This is a test")) {
278       ScriptAttribute scriptAtt = ts.addAttribute(ScriptAttribute.class);
279       ts.reset();
280       while (ts.incrementToken()) {
281         assertEquals(UScript.LATIN, scriptAtt.getCode());
282         assertEquals(UScript.getName(UScript.LATIN), scriptAtt.getName());
283         assertEquals(UScript.getShortName(UScript.LATIN), scriptAtt.getShortName());
284         assertTrue(ts.reflectAsString(false).contains("script=Latin"));
285       }
286       ts.end();
287     }

290   /** test for bugs like http://bugs.icu-project.org/trac/ticket/10767 */
291   public void testICUConcurrency() throws Exception {
292     int numThreads = 8;
293     final CountDownLatch startingGun = new CountDownLatch(1);
294     Thread threads[] = new Thread[numThreads];
295     for (int i = 0; i < threads.length; i++) {
296       threads[i] = new Thread() {
297         @Override
298         public void run() {
299           try {
300             startingGun.await();
301             long tokenCount = 0;
302             final String contents = "英 เบียร์ ビール ເບຍ abc";
303             for (int i = 0; i < 1000; i++) {
304               try (Tokenizer tokenizer = new ICUTokenizer()) {
305                 tokenizer.setReader(new StringReader(contents));
306                 tokenizer.reset();
307                 while (tokenizer.incrementToken()) {
308                   tokenCount++;
309                 }
310                 tokenizer.end();
311               }
312             }
313             if (VERBOSE) {
314               System.out.println(tokenCount);
315             }
316           } catch (Exception e) {
317             throw new RuntimeException(e);
318           }
319         } 
320       };
321       threads[i].start();
322     }
323     startingGun.countDown();
324     for (int i = 0; i < threads.length; i++) {
325       threads[i].join();
326     }
327   }
