20  public class TestStringHelper extends LuceneTestCase {
21    
22    public void testBytesDifference() {
23      BytesRef left = new BytesRef("foobar");
24      BytesRef right = new BytesRef("foozo");
25      assertEquals(3, StringHelper.bytesDifference(left, right));
26    }
27  
28    public void testEquals() {
29      assertTrue(StringHelper.equals("foo", "foo"));
30      assertFalse(StringHelper.equals("foo", null));
31      assertFalse(StringHelper.equals(null, "foo"));
32      assertTrue(StringHelper.equals(null, null));
33    }
34    
35    public void testStartsWith() {
36      BytesRef ref = new BytesRef("foobar");
37      BytesRef slice = new BytesRef("foo");
38      assertTrue(StringHelper.startsWith(ref, slice));
39    }
40    
41    public void testEndsWith() {
42      BytesRef ref = new BytesRef("foobar");
43      BytesRef slice = new BytesRef("bar");
44      assertTrue(StringHelper.endsWith(ref, slice));
45    }
46  
47    public void testStartsWithWhole() {
48      BytesRef ref = new BytesRef("foobar");
49      BytesRef slice = new BytesRef("foobar");
50      assertTrue(StringHelper.startsWith(ref, slice));
51    }
52    
53    public void testEndsWithWhole() {
54      BytesRef ref = new BytesRef("foobar");
55      BytesRef slice = new BytesRef("foobar");
56      assertTrue(StringHelper.endsWith(ref, slice));
57    }
58  
59    public void testMurmurHash3() throws Exception {
60      // Hashes computed using murmur3_32 from https://code.google.com/p/pyfasthash
61      assertEquals(0xf6a5c420, StringHelper.murmurhash3_x86_32(new BytesRef("foo"), 0));
62      assertEquals(0xcd018ef6, StringHelper.murmurhash3_x86_32(new BytesRef("foo"), 16));
63      assertEquals(0x111e7435, StringHelper.murmurhash3_x86_32(new BytesRef("You want weapons? We're in a library! Books! The best weapons in the world!"), 0));
64      assertEquals(0x2c628cd0, StringHelper.murmurhash3_x86_32(new BytesRef("You want weapons? We're in a library! Books! The best weapons in the world!"), 3476));
65    }
66    
67    public void testSortKeyLength() throws Exception {
68      assertEquals(3, StringHelper.sortKeyLength(new BytesRef("foo"), new BytesRef("for")));
69      assertEquals(3, StringHelper.sortKeyLength(new BytesRef("foo1234"), new BytesRef("for1234")));
70      assertEquals(2, StringHelper.sortKeyLength(new BytesRef("foo"), new BytesRef("fz")));
71      assertEquals(1, StringHelper.sortKeyLength(new BytesRef("foo"), new BytesRef("g")));
72      assertEquals(4, StringHelper.sortKeyLength(new BytesRef("foo"), new BytesRef("food")));
73    }
74  }
