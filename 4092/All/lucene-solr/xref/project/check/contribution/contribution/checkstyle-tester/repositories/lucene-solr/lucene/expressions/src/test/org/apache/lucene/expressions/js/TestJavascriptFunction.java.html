19  import org.apache.lucene.expressions.Expression;
20  import org.apache.lucene.util.LuceneTestCase;

22  public class TestJavascriptFunction extends LuceneTestCase {
23    private static double DELTA = 0.0000001;
24    
25    private void assertEvaluatesTo(String expression, double expected) throws Exception {
26      Expression evaluator = JavascriptCompiler.compile(expression);
27      double actual = evaluator.evaluate(0, null);
28      assertEquals(expected, actual, DELTA);
29    }
30    
31    public void testAbsMethod() throws Exception {
32      assertEvaluatesTo("abs(0)", 0);
33      assertEvaluatesTo("abs(119)", 119);
34      assertEvaluatesTo("abs(119)", 119);
35      assertEvaluatesTo("abs(1)", 1);
36      assertEvaluatesTo("abs(-1)", 1);
37    }

39    public void testAcosMethod() throws Exception {
40      assertEvaluatesTo("acos(-1)", Math.PI);
41      assertEvaluatesTo("acos(-0.8660254)", Math.PI*5/6);
42      assertEvaluatesTo("acos(-0.7071068)", Math.PI*3/4);
43      assertEvaluatesTo("acos(-0.5)", Math.PI*2/3);
44      assertEvaluatesTo("acos(0)", Math.PI/2);
45      assertEvaluatesTo("acos(0.5)", Math.PI/3);
46      assertEvaluatesTo("acos(0.7071068)", Math.PI/4);
47      assertEvaluatesTo("acos(0.8660254)", Math.PI/6);
48      assertEvaluatesTo("acos(1)", 0);
49    }

79    public void testAtanMethod() throws Exception {

89    public void testAtan2Method() throws Exception {
90      assertEvaluatesTo("atan2(+0,+0)", +0.0);
91      assertEvaluatesTo("atan2(+0,-0)", +Math.PI);
92      assertEvaluatesTo("atan2(-0,+0)", -0.0);
93      assertEvaluatesTo("atan2(-0,-0)", -Math.PI);
94      assertEvaluatesTo("atan2(2,2)", Math.PI/4);
95      assertEvaluatesTo("atan2(-2,2)", -Math.PI/4);
96      assertEvaluatesTo("atan2(2,-2)", Math.PI*3/4);
97      assertEvaluatesTo("atan2(-2,-2)", -Math.PI*3/4);
98    }
