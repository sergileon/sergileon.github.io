127   public void doTestNumbers(boolean withPayloads) throws Exception {
128     Directory dir = newDirectory();
129     Analyzer analyzer = withPayloads ? new MockPayloadAnalyzer() : new MockAnalyzer(random());
130     iwc = newIndexWriterConfig(analyzer);
131     iwc.setMergePolicy(newLogMergePolicy()); // will rely on docids a bit for skipping
132     RandomIndexWriter w = new RandomIndexWriter(random(), dir, iwc);
133     
134     FieldType ft = new FieldType(TextField.TYPE_STORED);
135     ft.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
136     if (random().nextBoolean()) {
137       ft.setStoreTermVectors(true);
138       ft.setStoreTermVectorOffsets(random().nextBoolean());
139       ft.setStoreTermVectorPositions(random().nextBoolean());
140     }
141     
142     int numDocs = atLeast(500);
143     for (int i = 0; i < numDocs; i++) {
144       Document doc = new Document();
145       doc.add(new Field("numbers", English.intToEnglish(i), ft));
146       doc.add(new Field("oddeven", (i % 2) == 0 ? "even" : "odd", ft));
147       doc.add(new StringField("id", "" + i, Field.Store.NO));
148       w.addDocument(doc);
149     }
150     
151     IndexReader reader = w.getReader();
152     w.close();
153     
154     String terms[] = { "one", "two", "three", "four", "five", "six", "seven", "eight", "nine", "ten", "hundred" };
155     
156     for (String term : terms) {
157       PostingsEnum dp = MultiFields.getTermPositionsEnum(reader, "numbers", new BytesRef(term));
158       int doc;
159       while((doc = dp.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
160         String storedNumbers = reader.document(doc).get("numbers");
161         int freq = dp.freq();
162         for (int i = 0; i < freq; i++) {
163           dp.nextPosition();
164           int start = dp.startOffset();
165           assert start >= 0;
166           int end = dp.endOffset();
167           assert end >= 0 && end >= start;
168           // check that the offsets correspond to the term in the src text
169           assertTrue(storedNumbers.substring(start, end).equals(term));
170           if (withPayloads) {
171             // check that we have a payload and it starts with "pos"
172             assertNotNull(dp.getPayload());
173             BytesRef payload = dp.getPayload();
174             assertTrue(payload.utf8ToString().startsWith("pos:"));
175           } // note: withPayloads=false doesnt necessarily mean we dont have them from MockAnalyzer!
176         }
177       }
178     }
179     
180     // check we can skip correctly
181     int numSkippingTests = atLeast(50);
182     
183     for (int j = 0; j < numSkippingTests; j++) {
184       int num = TestUtil.nextInt(random(), 100, Math.min(numDocs - 1, 999));
185       PostingsEnum dp = MultiFields.getTermPositionsEnum(reader, "numbers", new BytesRef("hundred"));
186       int doc = dp.advance(num);
187       assertEquals(num, doc);
188       int freq = dp.freq();
189       for (int i = 0; i < freq; i++) {
190         String storedNumbers = reader.document(doc).get("numbers");
191         dp.nextPosition();
192         int start = dp.startOffset();
193         assert start >= 0;
194         int end = dp.endOffset();
195         assert end >= 0 && end >= start;
196         // check that the offsets correspond to the term in the src text
197         assertTrue(storedNumbers.substring(start, end).equals("hundred"));
198         if (withPayloads) {
199           // check that we have a payload and it starts with "pos"
200           assertNotNull(dp.getPayload());
201           BytesRef payload = dp.getPayload();
202           assertTrue(payload.utf8ToString().startsWith("pos:"));
203         } // note: withPayloads=false doesnt necessarily mean we dont have them from MockAnalyzer!
204       }
205     }
206     
207     // check that other fields (without offsets) work correctly
208     
209     for (int i = 0; i < numDocs; i++) {
210       PostingsEnum dp = MultiFields.getTermDocsEnum(reader, "id", new BytesRef("" + i), 0);
211       assertEquals(i, dp.nextDoc());
212       assertEquals(DocIdSetIterator.NO_MORE_DOCS, dp.nextDoc());
213     }
214     
215     reader.close();
216     dir.close();
217   }

219   public void testRandom() throws Exception {
220     // token -> docID -> tokens
221     final Map<String,Map<Integer,List<Token>>> actualTokens = new HashMap<>();
222 
223     Directory dir = newDirectory();
224     RandomIndexWriter w = new RandomIndexWriter(random(), dir, iwc);
225 
226     final int numDocs = atLeast(20);
227     //final int numDocs = atLeast(5);
228 
229     FieldType ft = new FieldType(TextField.TYPE_NOT_STORED);
230 
231     // TODO: randomize what IndexOptions we use; also test
232     // changing this up in one IW buffered segment...:
233     ft.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
234     if (random().nextBoolean()) {
235       ft.setStoreTermVectors(true);
236       ft.setStoreTermVectorOffsets(random().nextBoolean());
237       ft.setStoreTermVectorPositions(random().nextBoolean());
238     }
239 
240     for(int docCount=0;docCount<numDocs;docCount++) {
241       Document doc = new Document();
242       doc.add(new IntField("id", docCount, Field.Store.YES));
243       doc.add(new NumericDocValuesField("id", docCount));
244       List<Token> tokens = new ArrayList<>();
245       final int numTokens = atLeast(100);
246       //final int numTokens = atLeast(20);
247       int pos = -1;
248       int offset = 0;
249       //System.out.println("doc id=" + docCount);
250       for(int tokenCount=0;tokenCount<numTokens;tokenCount++) {
251         final String text;
252         if (random().nextBoolean()) {
253           text = "a";
254         } else if (random().nextBoolean()) {
255           text = "b";
256         } else if (random().nextBoolean()) {
257           text = "c";
258         } else {
259           text = "d";
260         }       
261         
262         int posIncr = random().nextBoolean() ? 1 : random().nextInt(5);
263         if (tokenCount == 0 && posIncr == 0) {
264           posIncr = 1;
265         }
266         final int offIncr = random().nextBoolean() ? 0 : random().nextInt(5);
267         final int tokenOffset = random().nextInt(5);
268 
269         final Token token = makeToken(text, posIncr, offset+offIncr, offset+offIncr+tokenOffset);
270         if (!actualTokens.containsKey(text)) {
271           actualTokens.put(text, new HashMap<Integer,List<Token>>());
272         }
273         final Map<Integer,List<Token>> postingsByDoc = actualTokens.get(text);
274         if (!postingsByDoc.containsKey(docCount)) {
275           postingsByDoc.put(docCount, new ArrayList<Token>());
276         }
277         postingsByDoc.get(docCount).add(token);
278         tokens.add(token);
279         pos += posIncr;
280         // stuff abs position into type:
281         token.setType(""+pos);
282         offset += offIncr + tokenOffset;
283         //System.out.println("  " + token + " posIncr=" + token.getPositionIncrement() + " pos=" + pos + " off=" + token.startOffset() + "/" + token.endOffset() + " (freq=" + postingsByDoc.get(docCount).size() + ")");
284       }
285       doc.add(new Field("content", new CannedTokenStream(tokens.toArray(new Token[tokens.size()])), ft));
286       w.addDocument(doc);
287     }
288     final DirectoryReader r = w.getReader();
289     w.close();
290 
291     final String[] terms = new String[] {"a", "b", "c", "d"};
292     for(LeafReaderContext ctx : r.leaves()) {
293       // TODO: improve this
294       LeafReader sub = ctx.reader();
295       //System.out.println("\nsub=" + sub);
296       final TermsEnum termsEnum = sub.fields().terms("content").iterator();
297       PostingsEnum docs = null;
298       PostingsEnum docsAndPositions = null;
299       PostingsEnum docsAndPositionsAndOffsets = null;
300       final NumericDocValues docIDToID = DocValues.getNumeric(sub, "id");
301       for(String term : terms) {
302         //System.out.println("  term=" + term);
303         if (termsEnum.seekExact(new BytesRef(term))) {
304           docs = termsEnum.postings(docs);
305           assertNotNull(docs);
306           int doc;
307           //System.out.println("    doc/freq");
308           while((doc = docs.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
309             final List<Token> expected = actualTokens.get(term).get((int) docIDToID.get(doc));
310             //System.out.println("      doc=" + docIDToID.get(doc) + " docID=" + doc + " " + expected.size() + " freq");
311             assertNotNull(expected);
312             assertEquals(expected.size(), docs.freq());
313           }
314 
315           // explicitly exclude offsets here
316           docsAndPositions = termsEnum.postings(docsAndPositions, PostingsEnum.ALL);
317           assertNotNull(docsAndPositions);
318           //System.out.println("    doc/freq/pos");
319           while((doc = docsAndPositions.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
320             final List<Token> expected = actualTokens.get(term).get((int) docIDToID.get(doc));
321             //System.out.println("      doc=" + docIDToID.get(doc) + " " + expected.size() + " freq");
322             assertNotNull(expected);
323             assertEquals(expected.size(), docsAndPositions.freq());
324             for(Token token : expected) {
325               int pos = Integer.parseInt(token.type());
326               //System.out.println("        pos=" + pos);
327               assertEquals(pos, docsAndPositions.nextPosition());
328             }
329           }
330 
331           docsAndPositionsAndOffsets = termsEnum.postings(docsAndPositions, PostingsEnum.ALL);
332           assertNotNull(docsAndPositionsAndOffsets);
333           //System.out.println("    doc/freq/pos/offs");
334           while((doc = docsAndPositionsAndOffsets.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
335             final List<Token> expected = actualTokens.get(term).get((int) docIDToID.get(doc));
336             //System.out.println("      doc=" + docIDToID.get(doc) + " " + expected.size() + " freq");
337             assertNotNull(expected);
338             assertEquals(expected.size(), docsAndPositionsAndOffsets.freq());
339             for(Token token : expected) {
340               int pos = Integer.parseInt(token.type());
341               //System.out.println("        pos=" + pos);
342               assertEquals(pos, docsAndPositionsAndOffsets.nextPosition());
343               assertEquals(token.startOffset(), docsAndPositionsAndOffsets.startOffset());
344               assertEquals(token.endOffset(), docsAndPositionsAndOffsets.endOffset());
345             }
346           }
347         }
348       }        
349       // TODO: test advance:
350     }
351     r.close();
352     dir.close();
353   }

355   public void testWithUnindexedFields() throws Exception {
356     Directory dir = newDirectory();
357     RandomIndexWriter riw = new RandomIndexWriter(random(), dir, iwc);
358     for (int i = 0; i < 100; i++) {
359       Document doc = new Document();
360       // ensure at least one doc is indexed with offsets
361       if (i < 99 && random().nextInt(2) == 0) {
362         // stored only
363         FieldType ft = new FieldType();
364         ft.setStored(true);
365         doc.add(new Field("foo", "boo!", ft));
366       } else {
367         FieldType ft = new FieldType(TextField.TYPE_STORED);
368         ft.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
369         if (random().nextBoolean()) {
370           // store some term vectors for the checkindex cross-check
371           ft.setStoreTermVectors(true);
372           ft.setStoreTermVectorPositions(true);
373           ft.setStoreTermVectorOffsets(true);
374         }
375         doc.add(new Field("foo", "bar", ft));
376       }
377       riw.addDocument(doc);
378     }
379     CompositeReader ir = riw.getReader();
380     LeafReader slow = SlowCompositeReaderWrapper.wrap(ir);
381     FieldInfos fis = slow.getFieldInfos();
382     assertEquals(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS, fis.fieldInfo("foo").getIndexOptions());
383     slow.close();
384     ir.close();
385     riw.close();
386     dir.close();
387   }

519   private void checkTokens(Token[] field1, Token[] field2) throws IOException {
520     Directory dir = newDirectory();
521     RandomIndexWriter riw = new RandomIndexWriter(random(), dir, iwc);
522     boolean success = false;
523     try {
524       FieldType ft = new FieldType(TextField.TYPE_NOT_STORED);
525       ft.setIndexOptions(IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS);
526       // store some term vectors for the checkindex cross-check
527       ft.setStoreTermVectors(true);
528       ft.setStoreTermVectorPositions(true);
529       ft.setStoreTermVectorOffsets(true);
530      
531       Document doc = new Document();
532       doc.add(new Field("body", new CannedTokenStream(field1), ft));
533       doc.add(new Field("body", new CannedTokenStream(field2), ft));
534       riw.addDocument(doc);
535       riw.close();
536       success = true;
537     } finally {
538       if (success) {
539         IOUtils.close(dir);
540       } else {
541         IOUtils.closeWhileHandlingException(riw, dir);
542       }
543     }
544   }

573     final Token t = new Token();
574     t.append(text);
575     t.setPositionIncrement(posIncr);
576     t.setOffset(startOffset, endOffset);
