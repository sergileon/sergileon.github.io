155   /** test for constant score + boolean query + filter, the other tests only use the constant score mode */
156   private void testRange(int precisionStep) throws Exception {
157     String field="field"+precisionStep;
158     int count=3000;
159     long lower=(distance*3/2)+startOffset, upper=lower + count*distance + (distance/3);
160     NumericRangeQuery<Long> q = NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, true, true);
161     for (byte i=0; i<2; i++) {
162       TopDocs topDocs;
163       String type;
164       switch (i) {
165         case 0:
166           type = " (constant score filter rewrite)";
167           q.setRewriteMethod(MultiTermQuery.CONSTANT_SCORE_REWRITE);
168           topDocs = searcher.search(q, noDocs, Sort.INDEXORDER);
169           break;
170         case 1:
171           type = " (constant score boolean rewrite)";
172           q.setRewriteMethod(MultiTermQuery.CONSTANT_SCORE_BOOLEAN_REWRITE);
173           topDocs = searcher.search(q, noDocs, Sort.INDEXORDER);
174           break;
175         default:
176           return;
177       }
178       ScoreDoc[] sd = topDocs.scoreDocs;
179       assertNotNull(sd);
180       assertEquals("Score doc count"+type, count, sd.length );
181       Document doc=searcher.doc(sd[0].doc);
182       assertEquals("First doc"+type, 2*distance+startOffset, doc.getField(field).numericValue().longValue() );
183       doc=searcher.doc(sd[sd.length-1].doc);
184       assertEquals("Last doc"+type, (1+count)*distance+startOffset, doc.getField(field).numericValue().longValue() );
185     }
186   }

378   private void testRandomTrieAndClassicRangeQuery(int precisionStep) throws Exception {
379     String field="field"+precisionStep;
380     int totalTermCountT=0,totalTermCountC=0,termCountT,termCountC;
381     int num = TestUtil.nextInt(random(), 10, 20);
382     for (int i = 0; i < num; i++) {
383       long lower=(long)(random().nextDouble()*noDocs*distance)+startOffset;
384       long upper=(long)(random().nextDouble()*noDocs*distance)+startOffset;
385       if (lower>upper) {
386         long a=lower; lower=upper; upper=a;
387       }
388       final BytesRef lowerBytes, upperBytes;
389       BytesRefBuilder b = new BytesRefBuilder();
390       NumericUtils.longToPrefixCodedBytes(lower, 0, b);
391       lowerBytes = b.toBytesRef();
392       NumericUtils.longToPrefixCodedBytes(upper, 0, b);
393       upperBytes = b.toBytesRef();
394       
395       // test inclusive range
396       NumericRangeQuery<Long> tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, true, true);
397       TermRangeQuery cq=new TermRangeQuery(field, lowerBytes, upperBytes, true, true);
398       TopDocs tTopDocs = searcher.search(tq, 1);
399       TopDocs cTopDocs = searcher.search(cq, 1);
400       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
401       totalTermCountT += termCountT = countTerms(tq);
402       totalTermCountC += termCountC = countTerms(cq);
403       checkTermCounts(precisionStep, termCountT, termCountC);
404       // test exclusive range
405       tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, false, false);
406       cq=new TermRangeQuery(field, lowerBytes, upperBytes, false, false);
407       tTopDocs = searcher.search(tq, 1);
408       cTopDocs = searcher.search(cq, 1);
409       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
410       totalTermCountT += termCountT = countTerms(tq);
411       totalTermCountC += termCountC = countTerms(cq);
412       checkTermCounts(precisionStep, termCountT, termCountC);
413       // test left exclusive range
414       tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, false, true);
415       cq=new TermRangeQuery(field, lowerBytes, upperBytes, false, true);
416       tTopDocs = searcher.search(tq, 1);
417       cTopDocs = searcher.search(cq, 1);
418       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
419       totalTermCountT += termCountT = countTerms(tq);
420       totalTermCountC += termCountC = countTerms(cq);
421       checkTermCounts(precisionStep, termCountT, termCountC);
422       // test right exclusive range
423       tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, true, false);
424       cq=new TermRangeQuery(field, lowerBytes, upperBytes, true, false);
425       tTopDocs = searcher.search(tq, 1);
426       cTopDocs = searcher.search(cq, 1);
427       assertEquals("Returned count for NumericRangeQuery and TermRangeQuery must be equal", cTopDocs.totalHits, tTopDocs.totalHits );
428       totalTermCountT += termCountT = countTerms(tq);
429       totalTermCountC += termCountC = countTerms(cq);
430       checkTermCounts(precisionStep, termCountT, termCountC);
431     }
432     
433     checkTermCounts(precisionStep, totalTermCountT, totalTermCountC);
434     if (VERBOSE && precisionStep != Integer.MAX_VALUE) {
435       System.out.println("Average number of terms during random search on '" + field + "':");
436       System.out.println(" Numeric query: " + (((double)totalTermCountT)/(num * 4)));
437       System.out.println(" Classical query: " + (((double)totalTermCountC)/(num * 4)));
438     }
439   }

456   private int countTerms(MultiTermQuery q) throws Exception {
457     final Terms terms = MultiFields.getTerms(reader, q.getField());
458     if (terms == null)
459       return 0;
460     final TermsEnum termEnum = q.getTermsEnum(terms);
461     assertNotNull(termEnum);
462     int count = 0;
463     BytesRef cur, last = null;
464     while ((cur = termEnum.next()) != null) {
465       count++;
466       if (last != null) {
467         assertTrue(last.compareTo(cur) < 0);
468       }
469       last = BytesRef.deepCopyOf(cur);
470     } 
471     // LUCENE-3314: the results after next() already returned null are undefined,
472     // assertNull(termEnum.next());
473     return count;
474   }

476   private void checkTermCounts(int precisionStep, int termCountT, int termCountC) {
477     if (precisionStep == Integer.MAX_VALUE) {
478       assertEquals("Number of terms should be equal for unlimited precStep", termCountC, termCountT);
479     } else {
480       assertTrue("Number of terms for NRQ should be <= compared to classical TRQ", termCountT <= termCountC);
481     }
482   }

509   private void testRangeSplit(int precisionStep) throws Exception {
510     String field="ascfield"+precisionStep;
511     // 10 random tests
512     int num = TestUtil.nextInt(random(), 10, 20);
513     for (int i = 0; i < num; i++) {
514       long lower=(long)(random().nextDouble()*noDocs - noDocs/2);
515       long upper=(long)(random().nextDouble()*noDocs - noDocs/2);
516       if (lower>upper) {
517         long a=lower; lower=upper; upper=a;
518       }
519       // test inclusive range
520       Query tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, true, true);
521       TopDocs tTopDocs = searcher.search(tq, 1);
522       assertEquals("Returned count of range query must be equal to inclusive range length", upper-lower+1, tTopDocs.totalHits );
523       // test exclusive range
524       tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, false, false);
525       tTopDocs = searcher.search(tq, 1);
526       assertEquals("Returned count of range query must be equal to exclusive range length", Math.max(upper-lower-1, 0), tTopDocs.totalHits );
527       // test left exclusive range
528       tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, false, true);
529       tTopDocs = searcher.search(tq, 1);
530       assertEquals("Returned count of range query must be equal to half exclusive range length", upper-lower, tTopDocs.totalHits );
531       // test right exclusive range
532       tq=NumericRangeQuery.newLongRange(field, precisionStep, lower, upper, true, false);
533       tTopDocs = searcher.search(tq, 1);
534       assertEquals("Returned count of range query must be equal to half exclusive range length", upper-lower, tTopDocs.totalHits );
535     }
536   }
