868   public static Object buildGroupedResult(IndexSchema schema, List<Grp> sortedGroups, int start, int rows, int group_offset, int group_limit, boolean includeNGroups) {
869     Map<String,Object> result = new LinkedHashMap<>();
870 
871     long matches = 0;
872     for (Grp grp : sortedGroups) {
873       matches += grp.docs.size();
874     }
875     result.put("matches", matches);
876     if (includeNGroups) {
877       result.put("ngroups", sortedGroups.size());
878     }
879     List groupList = new ArrayList();
880     result.put("groups", groupList);
881 
882     for (int i=start; i<sortedGroups.size(); i++) {
883       if (rows != -1 && groupList.size() >= rows) break;  // directly test rather than calculating, so we can catch any calc errors in the real code
884       Map<String,Object> group = new LinkedHashMap<>();
885       groupList.add(group);
886 
887       Grp grp = sortedGroups.get(i);
888       group.put("groupValue", grp.groupValue);
889 
890       Map<String,Object> resultSet = new LinkedHashMap<>();
891       group.put("doclist", resultSet);
892       resultSet.put("numFound", grp.docs.size());
893       resultSet.put("start", group_offset);
894       List docs = new ArrayList();
895       resultSet.put("docs", docs);
896       for (int j=group_offset; j<grp.docs.size(); j++) {
897         if (group_limit != -1 && docs.size() >= group_limit) break;
898         docs.add( grp.docs.get(j).toObject(schema) );
899       }
900     }
901 
902     return result;
903   }

930   public static Map<Comparable, Grp> groupBy(Collection<Doc> docs, String field) {
931     Map<Comparable, Grp> groups = new HashMap<>();
932     for (Doc doc : docs) {
933       List<Comparable> vals = doc.getValues(field);
934       if (vals == null) {
935         Grp grp = groups.get(null);
936         if (grp == null) {
937           grp = new Grp();
938           grp.groupValue = null;
939           grp.docs = new ArrayList<>();
940           groups.put(null, grp);
941         }
942         grp.docs.add(doc);
943       } else {
944         for (Comparable val : vals) {
945 
946           Grp grp = groups.get(val);
947           if (grp == null) {
948             grp = new Grp();
949             grp.groupValue = val;
950             grp.docs = new ArrayList<>();
951             groups.put(grp.groupValue, grp);
952           }
953           grp.docs.add(doc);
954         }
955       }
956     }
957     return groups;
958   }
