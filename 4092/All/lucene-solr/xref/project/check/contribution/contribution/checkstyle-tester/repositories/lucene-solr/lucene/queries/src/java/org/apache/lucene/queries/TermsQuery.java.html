61  /**
62   * Specialization for a disjunction over many terms that behaves like a
63   * {@link ConstantScoreQuery} over a {@link BooleanQuery} containing only
64   * {@link org.apache.lucene.search.BooleanClause.Occur#SHOULD} clauses.
65   * <p>For instance in the following example, both @{code q1} and {@code q2}
66   * would yield the same scores:
67   * <pre class="prettyprint">
68   * Query q1 = new TermsQuery(new Term("field", "foo"), new Term("field", "bar"));
69   *
70   * BooleanQuery bq = new BooleanQuery();
71   * bq.add(new TermQuery(new Term("field", "foo")), Occur.SHOULD);
72   * bq.add(new TermQuery(new Term("field", "bar")), Occur.SHOULD);
73   * Query q2 = new ConstantScoreQuery(bq);
74   * </pre>
75   * <p>When there are few terms, this query executes like a regular disjunction.
76   * However, when there are many terms, instead of merging iterators on the fly,
77   * it will populate a bit set with matching docs and return a {@link Scorer}
78   * over this bit set.
79   * <p>NOTE: This query produces scores that are equal to its boost
80   */
81  public class TermsQuery extends Query implements Accountable {
82  
83    private static final long BASE_RAM_BYTES_USED = RamUsageEstimator.shallowSizeOfInstance(TermsQuery.class);
84    // Same threshold as MultiTermQueryConstantScoreWrapper
85    static final int BOOLEAN_REWRITE_TERM_COUNT_THRESHOLD = 16;
86  
87    private final PrefixCodedTerms termData;
88    private final int termDataHashCode; // cached hashcode of termData
89  
90    private static Term[] toTermArray(String field, List<BytesRef> termBytes) {
91      Term[] array = new Term[termBytes.size()];
92      int i = 0;
93      for (BytesRef t : termBytes) {
94        array[i++] = new Term(field, t);
95      }
96      return array;
97    }
98  
99    /**
100    * Creates a new {@link TermsQuery} from the given list. The list
101    * can contain duplicate terms and multiple fields.
102    */
103   public TermsQuery(final List<Term> terms) {
104     Term[] sortedTerms = terms.toArray(new Term[terms.size()]);
105     ArrayUtil.timSort(sortedTerms);
106     PrefixCodedTerms.Builder builder = new PrefixCodedTerms.Builder();
107     Term previous = null;
108     for (Term term : sortedTerms) {
109       if (term.equals(previous) == false) {
110         builder.add(term);
111       }
112       previous = term;
113     }
114     termData = builder.finish();
115     termDataHashCode = termData.hashCode();
116   }
117 
118   /**
119    * Creates a new {@link TermsQuery} from the given {@link BytesRef} list for
120    * a single field.
121    */
122   public TermsQuery(final String field, final List<BytesRef> terms) {
123     this(toTermArray(field, terms));
124   }
125 
126   /**
127    * Creates a new {@link TermsQuery} from the given {@link BytesRef} array for
128    * a single field.
129    */
130   public TermsQuery(final String field, final BytesRef...terms) {
131     // this ctor prevents unnecessary Term creations
132    this(field, Arrays.asList(terms));
133   }
134 
135   /**
136    * Creates a new {@link TermsQuery} from the given array. The array can
137    * contain duplicate terms and multiple fields.
138    */
139   public TermsQuery(final Term... terms) {
140     this(Arrays.asList(terms));
141   }

143   @Override
144   public Query rewrite(IndexReader reader) throws IOException {
145     if (getBoost() != 1f) {
146       return super.rewrite(reader);
147     }
148     final int threshold = Math.min(BOOLEAN_REWRITE_TERM_COUNT_THRESHOLD, BooleanQuery.getMaxClauseCount());
149     if (termData.size() <= threshold) {
150       BooleanQuery.Builder bq = new BooleanQuery.Builder();
151       TermIterator iterator = termData.iterator();
152       for (BytesRef term = iterator.next(); term != null; term = iterator.next()) {
153         bq.add(new TermQuery(new Term(iterator.field(), BytesRef.deepCopyOf(term))), Occur.SHOULD);
154       }
155       return new ConstantScoreQuery(bq.build());
156     }
157     return super.rewrite(reader);
158   }
159 
160   @Override
161   public boolean equals(Object obj) {
162     if (this == obj) {
163       return true;
164     }
165     if (!super.equals(obj)) {
166       return false;
167     }
168     TermsQuery that = (TermsQuery) obj;
169     // termData might be heavy to compare so check the hash code first
170     return termDataHashCode == that.termDataHashCode
171         && termData.equals(that.termData);
172   }
173 
174   @Override
175   public int hashCode() {
176     return 31 * super.hashCode() + termDataHashCode;
177   }
178 
179   @Override
180   public String toString(String defaultField) {
181     StringBuilder builder = new StringBuilder();
182     boolean first = true;
183     TermIterator iterator = termData.iterator();
184     for (BytesRef term = iterator.next(); term != null; term = iterator.next()) {
185       if (!first) {
186         builder.append(' ');
187       }
188       first = false;
189       builder.append(new Term(iterator.field(), term).toString());
190     }
191     builder.append(ToStringUtils.boost(getBoost()));
192     return builder.toString();
193   }
194 
195   @Override
196   public long ramBytesUsed() {
197     return BASE_RAM_BYTES_USED + termData.ramBytesUsed();
198   }
199 
200   @Override
201   public Collection<Accountable> getChildResources() {
202     return Collections.emptyList();
203   }

205   private static class TermAndState {
206     final String field;
207     final TermsEnum termsEnum;
208     final BytesRef term;
209     final TermState state;
210     final int docFreq;
211     final long totalTermFreq;
212 
213     TermAndState(String field, TermsEnum termsEnum) throws IOException {
214       this.field = field;
215       this.termsEnum = termsEnum;
216       this.term = BytesRef.deepCopyOf(termsEnum.term());
217       this.state = termsEnum.termState();
218       this.docFreq = termsEnum.docFreq();
219       this.totalTermFreq = termsEnum.totalTermFreq();
220     }
221   }
222 
223   private static class WeightOrDocIdSet {
224     final Weight weight;
225     final DocIdSet set;
226 
227     WeightOrDocIdSet(Weight weight) {
228       this.weight = Objects.requireNonNull(weight);
229       this.set = null;
230     }
231 
232     WeightOrDocIdSet(DocIdSet bitset) {
233       this.set = bitset;
234       this.weight = null;
235     }
236   }

238   @Override
239   public Weight createWeight(final IndexSearcher searcher, final boolean needsScores) throws IOException {
240     return new ConstantScoreWeight(this) {
241 
242       @Override
243       public void extractTerms(Set<Term> terms) {
244         // no-op
245         // This query is for abuse cases when the number of terms is too high to
246         // run efficiently as a BooleanQuery. So likewise we hide its terms in
247         // order to protect highlighters
248       }
249 
250       /**
251        * On the given leaf context, try to either rewrite to a disjunction if
252        * there are few matching terms, or build a bitset containing matching docs.
253        */
254       private WeightOrDocIdSet rewrite(LeafReaderContext context) throws IOException {
255         final LeafReader reader = context.reader();
256 
257         // We will first try to collect up to 'threshold' terms into 'matchingTerms'
258         // if there are two many terms, we will fall back to building the 'builder'
259         final int threshold = Math.min(BOOLEAN_REWRITE_TERM_COUNT_THRESHOLD, BooleanQuery.getMaxClauseCount());
260         assert termData.size() > threshold : "Query should have been rewritten";
261         List<TermAndState> matchingTerms = new ArrayList<>(threshold);
262         DocIdSetBuilder builder = null;
263 
264         final Fields fields = reader.fields();
265         String lastField = null;
266         Terms terms = null;
267         TermsEnum termsEnum = null;
268         PostingsEnum docs = null;
269         TermIterator iterator = termData.iterator();
270         for (BytesRef term = iterator.next(); term != null; term = iterator.next()) {
271           String field = iterator.field();
272           // comparing references is fine here
273           if (field != lastField) {
274             terms = fields.terms(field);
275             if (terms == null) {
276               termsEnum = null;
277             } else {
278               termsEnum = terms.iterator();
279             }
280             lastField = field;
281           }
282           if (termsEnum != null && termsEnum.seekExact(term)) {
283             if (matchingTerms == null) {
284               docs = termsEnum.postings(docs, PostingsEnum.NONE);
285               builder.add(docs);
286             } else if (matchingTerms.size() < threshold) {
287               matchingTerms.add(new TermAndState(field, termsEnum));
288             } else {
289               assert matchingTerms.size() == threshold;
290               builder = new DocIdSetBuilder(reader.maxDoc());
291               docs = termsEnum.postings(docs, PostingsEnum.NONE);
292               builder.add(docs);
293               for (TermAndState t : matchingTerms) {
294                 t.termsEnum.seekExact(t.term, t.state);
295                 docs = t.termsEnum.postings(docs, PostingsEnum.NONE);
296                 builder.add(docs);
297               }
298               matchingTerms = null;
299             }
300           }
301         }
302         if (matchingTerms != null) {
303           assert builder == null;
304           BooleanQuery.Builder bq = new BooleanQuery.Builder();
305           for (TermAndState t : matchingTerms) {
306             final TermContext termContext = new TermContext(searcher.getTopReaderContext());
307             termContext.register(t.state, context.ord, t.docFreq, t.totalTermFreq);
308             bq.add(new TermQuery(new Term(t.field, t.term), termContext), Occur.SHOULD);
309           }
310           Query q = new ConstantScoreQuery(bq.build());
311           final Weight weight = searcher.rewrite(q).createWeight(searcher, needsScores);
312           weight.normalize(1f, score());
313           return new WeightOrDocIdSet(weight);
314         } else {
315           assert builder != null;
316           return new WeightOrDocIdSet(builder.build());
317         }
318       }
319 
320       private Scorer scorer(DocIdSet set) throws IOException {
321         if (set == null) {
322           return null;
323         }
324         final DocIdSetIterator disi = set.iterator();
325         if (disi == null) {
326           return null;
327         }
328         return new ConstantScoreScorer(this, score(), disi);
329       }
330 
331       @Override
332       public BulkScorer bulkScorer(LeafReaderContext context) throws IOException {
333         final WeightOrDocIdSet weightOrBitSet = rewrite(context);
334         if (weightOrBitSet.weight != null) {
335           return weightOrBitSet.weight.bulkScorer(context);
336         } else {
337           final Scorer scorer = scorer(weightOrBitSet.set);
338           if (scorer == null) {
339             return null;
340           }
341           return new DefaultBulkScorer(scorer);
342         }
343       }
344 
345       @Override
346       public Scorer scorer(LeafReaderContext context) throws IOException {
347         final WeightOrDocIdSet weightOrBitSet = rewrite(context);
348         if (weightOrBitSet.weight != null) {
349           return weightOrBitSet.weight.scorer(context);
350         } else {
351           return scorer(weightOrBitSet.set);
352         }
353       }
354     };
355   }
