47    @Test
48    public void testRangeFilterId() throws IOException {
49  
50      IndexReader reader = signedIndexReader;
51      IndexSearcher search = newSearcher(reader);
52  
53      int medId = ((maxId - minId) / 2);
54          
55      String minIP = pad(minId);
56      String maxIP = pad(maxId);
57      String medIP = pad(medId);
58      
59      int numDocs = reader.numDocs();
60          
61      assertEquals("num of docs", numDocs, 1+ maxId - minId);
62          
63      ScoreDoc[] result;
64      Query q = new TermQuery(new Term("body","body"));
65  
66      // test id, bounded on both ends
67      result = search.search(q, DocValuesRangeFilter.newStringRange("id",minIP,maxIP,T,T), numDocs).scoreDocs;
68      assertEquals("find all", numDocs, result.length);
69  
70      result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,maxIP,T,F), numDocs).scoreDocs;
71      assertEquals("all but last", numDocs-1, result.length);
72  
73      result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,maxIP,F,T), numDocs).scoreDocs;
74      assertEquals("all but first", numDocs-1, result.length);
75          
76      result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,maxIP,F,F), numDocs).scoreDocs;
77      assertEquals("all but ends", numDocs-2, result.length);
78      
79      result = search.search(q,DocValuesRangeFilter.newStringRange("id",medIP,maxIP,T,T), numDocs).scoreDocs;
80      assertEquals("med and up", 1+ maxId-medId, result.length);
81          
82      result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,medIP,T,T), numDocs).scoreDocs;
83      assertEquals("up to med", 1+ medId-minId, result.length);
84  
85      // unbounded id
86  
87      result = search.search(q,DocValuesRangeFilter.newStringRange("id",null,null,T,T), numDocs).scoreDocs;
88      assertEquals("find all", numDocs, result.length);
89  
90      result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,null,T,F), numDocs).scoreDocs;
91      assertEquals("min and up", numDocs, result.length);
92  
93      result = search.search(q,DocValuesRangeFilter.newStringRange("id",null,maxIP,F,T), numDocs).scoreDocs;
94      assertEquals("max and down", numDocs, result.length);
95  
96      result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,null,F,F), numDocs).scoreDocs;
97      assertEquals("not min, but up", numDocs-1, result.length);
98          
99      result = search.search(q,DocValuesRangeFilter.newStringRange("id",null,maxIP,F,F), numDocs).scoreDocs;
100     assertEquals("not max, but down", numDocs-1, result.length);
101         
102     result = search.search(q,DocValuesRangeFilter.newStringRange("id",medIP,maxIP,T,F), numDocs).scoreDocs;
103     assertEquals("med and up, not max", maxId-medId, result.length);
104         
105     result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,medIP,F,T), numDocs).scoreDocs;
106     assertEquals("not min, up to med", medId-minId, result.length);
107 
108     // very small sets
109 
110     result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,minIP,F,F), numDocs).scoreDocs;
111     assertEquals("min,min,F,F", 0, result.length);
112     result = search.search(q,DocValuesRangeFilter.newStringRange("id",medIP,medIP,F,F), numDocs).scoreDocs;
113     assertEquals("med,med,F,F", 0, result.length);
114     result = search.search(q,DocValuesRangeFilter.newStringRange("id",maxIP,maxIP,F,F), numDocs).scoreDocs;
115     assertEquals("max,max,F,F", 0, result.length);
116                      
117     result = search.search(q,DocValuesRangeFilter.newStringRange("id",minIP,minIP,T,T), numDocs).scoreDocs;
118     assertEquals("min,min,T,T", 1, result.length);
119     result = search.search(q,DocValuesRangeFilter.newStringRange("id",null,minIP,F,T), numDocs).scoreDocs;
120     assertEquals("nul,min,F,T", 1, result.length);
121 
122     result = search.search(q,DocValuesRangeFilter.newStringRange("id",maxIP,maxIP,T,T), numDocs).scoreDocs;
123     assertEquals("max,max,T,T", 1, result.length);
124     result = search.search(q,DocValuesRangeFilter.newStringRange("id",maxIP,null,T,F), numDocs).scoreDocs;
125     assertEquals("max,nul,T,T", 1, result.length);
126 
127     result = search.search(q,DocValuesRangeFilter.newStringRange("id",medIP,medIP,T,T), numDocs).scoreDocs;
128     assertEquals("med,med,T,T", 1, result.length);
129   }

436   @Test
437   public void testSparseIndex() throws IOException {
438     Directory dir = newDirectory();
439     IndexWriter writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random())));
440 
441     for (int d = -20; d <= 20; d++) {
442       Document doc = new Document();
443       doc.add(new IntField("id_int", d, Field.Store.NO));
444       doc.add(new NumericDocValuesField("id_int", d));
445       doc.add(newStringField("body", "body", Field.Store.NO));
446       writer.addDocument(doc);
447     }
448     
449     writer.forceMerge(1);
450     BytesRefBuilder term0 = new BytesRefBuilder();
451     NumericUtils.intToPrefixCoded(0, 0, term0);
452     writer.deleteDocuments(new Term("id_int", term0.get()));
453     writer.close();
454 
455     IndexReader reader = DirectoryReader.open(dir);
456     IndexSearcher search = newSearcher(reader);
457     assertTrue(reader.hasDeletions());
458 
459     ScoreDoc[] result;
460     Query q = new TermQuery(new Term("body","body"));
461 
462     result = search.search(q,DocValuesRangeFilter.newIntRange("id_int",-20,20,T,T), 100).scoreDocs;
463     assertEquals("find all", 40, result.length);
464 
465     result = search.search(q,DocValuesRangeFilter.newIntRange("id_int",0,20,T,T), 100).scoreDocs;
466     assertEquals("find all", 20, result.length);
467 
468     result = search.search(q,DocValuesRangeFilter.newIntRange("id_int",-20,0,T,T), 100).scoreDocs;
469     assertEquals("find all", 20, result.length);
470 
471     result = search.search(q,DocValuesRangeFilter.newIntRange("id_int",10,20,T,T), 100).scoreDocs;
472     assertEquals("find all", 11, result.length);
473 
474     result = search.search(q,DocValuesRangeFilter.newIntRange("id_int",-20,-10,T,T), 100).scoreDocs;
475     assertEquals("find all", 11, result.length);
476     reader.close();
477     dir.close();
478   }
