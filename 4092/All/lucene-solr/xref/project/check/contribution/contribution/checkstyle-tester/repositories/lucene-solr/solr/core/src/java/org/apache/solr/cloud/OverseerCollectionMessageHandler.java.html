206   @Override
207   @SuppressWarnings("unchecked")
208   public SolrResponse processMessage(ZkNodeProps message, String operation) {
209     log.info("OverseerCollectionMessageHandler.processMessage : "+ operation + " , "+ message.toString());
210 
211     NamedList results = new NamedList();
212     try {
213       // force update the cluster state
214       zkStateReader.updateClusterState();
215       CollectionParams.CollectionAction action = CollectionParams.CollectionAction.get(operation);
216       if (action == null) {
217         throw new SolrException(ErrorCode.BAD_REQUEST, "Unknown operation:" + operation);
218       }
219       switch (action) {
220         case CREATE:
221           createCollection(zkStateReader.getClusterState(), message, results);
222           break;
223         case DELETE:
224           deleteCollection(message, results);
225           break;
226         case RELOAD:
227           ModifiableSolrParams params = new ModifiableSolrParams();
228           params.set(CoreAdminParams.ACTION, CoreAdminAction.RELOAD.toString());
229           collectionCmd(zkStateReader.getClusterState(), message, params, results, Replica.State.ACTIVE);
230           break;
231         case CREATEALIAS:
232           createAlias(zkStateReader.getAliases(), message);
233           break;
234         case DELETEALIAS:
235           deleteAlias(zkStateReader.getAliases(), message);
236           break;
237         case SPLITSHARD:
238           splitShard(zkStateReader.getClusterState(), message, results);
239           break;
240         case DELETESHARD:
241           deleteShard(zkStateReader.getClusterState(), message, results);
242           break;
243         case CREATESHARD:
244           createShard(zkStateReader.getClusterState(), message, results);
245           break;
246         case DELETEREPLICA:
247           deleteReplica(zkStateReader.getClusterState(), message, results);
248           break;
249         case MIGRATE:
250           migrate(zkStateReader.getClusterState(), message, results);
251           break;
252         case ADDROLE:
253           processRoleCommand(message, operation);
254           break;
255         case REMOVEROLE:
256           processRoleCommand(message, operation);
257           break;
258         case ADDREPLICA:
259           addReplica(zkStateReader.getClusterState(), message, results);
260           break;
261         case OVERSEERSTATUS:
262           getOverseerStatus(message, results);
263           break;
264         case CLUSTERSTATUS://TODO . deprecated. OCP does not need to do it .remove in a later release
265           new ClusterStatus(zkStateReader, message).getClusterStatus(results);
266           break;
267         case ADDREPLICAPROP:
268           processReplicaAddPropertyCommand(message);
269           break;
270         case DELETEREPLICAPROP:
271           processReplicaDeletePropertyCommand(message);
272           break;
273         case BALANCESHARDUNIQUE:
274           balanceProperty(message);
275           break;
276         case REBALANCELEADERS:
277           processRebalanceLeaders(message);
278           break;
279         case MODIFYCOLLECTION:
280           overseer.getInQueue(zkStateReader.getZkClient()).offer(Utils.toJSON(message));
281           break;
282         case MIGRATESTATEFORMAT:
283           migrateStateFormat(message, results);
284           break;
285         default:
286           throw new SolrException(ErrorCode.BAD_REQUEST, "Unknown operation:"
287               + operation);
288       }
289     } catch (Exception e) {
290       String collName = message.getStr("collection");
291       if (collName == null) collName = message.getStr(NAME);
292 
293       if (collName == null) {
294         SolrException.log(log, "Operation " + operation + " failed", e);
295       } else  {
296         SolrException.log(log, "Collection: " + collName + " operation: " + operation
297             + " failed", e);
298       }
299 
300       results.add("Operation " + operation + " caused exception:", e);
301       SimpleOrderedMap nl = new SimpleOrderedMap();
302       nl.add("msg", e.getMessage());
303       nl.add("rspCode", e instanceof SolrException ? ((SolrException)e).code() : -1);
304       results.add("exception", nl);
305     }
306     return new OverseerSolrResponse(results);
307   }

310   private void processRebalanceLeaders(ZkNodeProps message) throws KeeperException, InterruptedException {
311     checkRequired(message, COLLECTION_PROP, SHARD_ID_PROP, CORE_NAME_PROP, ELECTION_NODE_PROP,

317     params.set(REJOIN_AT_HEAD_PROP, message.getStr(REJOIN_AT_HEAD_PROP));
318     params.set(CoreAdminParams.ACTION, CoreAdminAction.REJOINLEADERELECTION.toString());
319     params.set(CORE_NAME_PROP, message.getStr(CORE_NAME_PROP));
320     params.set(CORE_NODE_NAME_PROP, message.getStr(CORE_NODE_NAME_PROP));
321     params.set(ELECTION_NODE_PROP, message.getStr(ELECTION_NODE_PROP));
322     params.set(BASE_URL_PROP, message.getStr(BASE_URL_PROP));
323 
324     String baseUrl = message.getStr(BASE_URL_PROP);
325     ShardRequest sreq = new ShardRequest();
326     sreq.nodeName = message.getStr(ZkStateReader.CORE_NAME_PROP);
327     // yes, they must use same admin handler path everywhere...
328     params.set("qt", adminPath);
329     sreq.purpose = ShardRequest.PURPOSE_PRIVATE;
330     sreq.shards = new String[] {baseUrl};
331     sreq.actualShards = sreq.shards;
332     sreq.params = params;
333     ShardHandler shardHandler = shardHandlerFactory.getShardHandler();
334     shardHandler.submit(sreq, baseUrl, sreq.params);
335   }

375   @SuppressWarnings("unchecked")
376   private void getOverseerStatus(ZkNodeProps message, NamedList results) throws KeeperException, InterruptedException {
377     String leaderNode = OverseerTaskProcessor.getLeaderNode(zkStateReader.getZkClient());
378     results.add("leader", leaderNode);
379     Stat stat = new Stat();
380     zkStateReader.getZkClient().getData("/overseer/queue",null, stat, true);
381     results.add("overseer_queue_size", stat.getNumChildren());
382     stat = new Stat();
383     zkStateReader.getZkClient().getData("/overseer/queue-work",null, stat, true);
384     results.add("overseer_work_queue_size", stat.getNumChildren());
385     stat = new Stat();
386     zkStateReader.getZkClient().getData("/overseer/collection-queue-work",null, stat, true);
387     results.add("overseer_collection_queue_size", stat.getNumChildren());
388 
389     NamedList overseerStats = new NamedList();
390     NamedList collectionStats = new NamedList();
391     NamedList stateUpdateQueueStats = new NamedList();
392     NamedList workQueueStats = new NamedList();
393     NamedList collectionQueueStats = new NamedList();
394     for (Map.Entry<String, Overseer.Stat> entry : stats.getStats().entrySet()) {
395       String key = entry.getKey();
396       NamedList<Object> lst = new SimpleOrderedMap<>();
397       if (key.startsWith("collection_"))  {
398         collectionStats.add(key.substring(11), lst);
399         int successes = stats.getSuccessCount(entry.getKey());
400         int errors = stats.getErrorCount(entry.getKey());
401         lst.add("requests", successes);
402         lst.add("errors", errors);
403         List<Overseer.FailedOp> failureDetails = stats.getFailureDetails(key);
404         if (failureDetails != null) {
405           List<SimpleOrderedMap<Object>> failures = new ArrayList<>();
406           for (Overseer.FailedOp failedOp : failureDetails) {
407             SimpleOrderedMap<Object> fail = new SimpleOrderedMap<>();
408             fail.add("request", failedOp.req.getProperties());
409             fail.add("response", failedOp.resp.getResponse());
410             failures.add(fail);
411           }
412           lst.add("recent_failures", failures);
413         }
414       } else if (key.startsWith("/overseer/queue_"))  {
415         stateUpdateQueueStats.add(key.substring(16), lst);
416       } else if (key.startsWith("/overseer/queue-work_"))  {
417         workQueueStats.add(key.substring(21), lst);
418       } else if (key.startsWith("/overseer/collection-queue-work_"))  {
419         collectionQueueStats.add(key.substring(32), lst);
420       } else  {
421         // overseer stats
422         overseerStats.add(key, lst);
423         int successes = stats.getSuccessCount(entry.getKey());
424         int errors = stats.getErrorCount(entry.getKey());
425         lst.add("requests", successes);
426         lst.add("errors", errors);
427       }
428       Timer timer = entry.getValue().requestTime;
429       Snapshot snapshot = timer.getSnapshot();
430       lst.add("totalTime", timer.getSum());
431       lst.add("avgRequestsPerMinute", timer.getMeanRate());
432       lst.add("5minRateRequestsPerMinute", timer.getFiveMinuteRate());
433       lst.add("15minRateRequestsPerMinute", timer.getFifteenMinuteRate());
434       lst.add("avgTimePerRequest", timer.getMean());
435       lst.add("medianRequestTime", snapshot.getMedian());
436       lst.add("75thPctlRequestTime", snapshot.get75thPercentile());
437       lst.add("95thPctlRequestTime", snapshot.get95thPercentile());
438       lst.add("99thPctlRequestTime", snapshot.get99thPercentile());
439       lst.add("999thPctlRequestTime", snapshot.get999thPercentile());
440     }
441     results.add("overseer_operations", overseerStats);
442     results.add("collection_operations", collectionStats);
443     results.add("overseer_queue", stateUpdateQueueStats);
444     results.add("overseer_internal_queue", workQueueStats);
445     results.add("collection_queue", collectionQueueStats);
446 
447   }

449   @SuppressWarnings("unchecked")
450   private void getClusterStatus(ClusterState clusterState, ZkNodeProps message, NamedList results) throws KeeperException, InterruptedException {
451     String collection = message.getStr(ZkStateReader.COLLECTION_PROP);
452 
453     // read aliases
454     Aliases aliases = zkStateReader.getAliases();
455     Map<String, List<String>> collectionVsAliases = new HashMap<>();
456     Map<String, String> aliasVsCollections = aliases.getCollectionAliasMap();
457     if (aliasVsCollections != null) {
458       for (Map.Entry<String, String> entry : aliasVsCollections.entrySet()) {
459         List<String> colls = StrUtils.splitSmart(entry.getValue(), ',');
460         String alias = entry.getKey();
461         for (String coll : colls) {
462           if (collection == null || collection.equals(coll))  {
463             List<String> list = collectionVsAliases.get(coll);
464             if (list == null) {
465               list = new ArrayList<>();
466               collectionVsAliases.put(coll, list);
467             }
468             list.add(alias);
469           }
470         }
471       }
472     }
473 
474     Map roles = null;
475     if (zkStateReader.getZkClient().exists(ZkStateReader.ROLES, true)) {
476       roles = (Map) Utils.fromJSON(zkStateReader.getZkClient().getData(ZkStateReader.ROLES, null, null, true));
477     }
478 
479     // convert cluster state into a map of writable types
480     byte[] bytes = Utils.toJSON(clusterState);
481     Map<String, Object> stateMap = (Map<String,Object>) Utils.fromJSON(bytes);
482 
483     Set<String> collections = new HashSet<>();
484     String routeKey = message.getStr(ShardParams._ROUTE_);
485     String shard = message.getStr(ZkStateReader.SHARD_ID_PROP);
486     if (collection == null) {
487       collections = new HashSet<>(clusterState.getCollections());
488     } else  {
489       collections = Collections.singleton(collection);
490     }
491 
492     NamedList<Object> collectionProps = new SimpleOrderedMap<Object>();
493 
494     for (String name : collections) {
495       Map<String, Object> collectionStatus = null;
496       DocCollection clusterStateCollection = clusterState.getCollection(name);
497 
498       Set<String> requestedShards = new HashSet<>();
499       if (routeKey != null) {
500         DocRouter router = clusterStateCollection.getRouter();
501         Collection<Slice> slices = router.getSearchSlices(routeKey, null, clusterStateCollection);
502         for (Slice slice : slices) {
503           requestedShards.add(slice.getName());
504         }
505       }
506       if (shard != null) {
507         requestedShards.add(shard);
508       }
509 
510       if (clusterStateCollection.getStateFormat() > 1) {
511         bytes = Utils.toJSON(clusterStateCollection);
512         Map<String, Object> docCollection = (Map<String, Object>) Utils.fromJSON(bytes);
513         collectionStatus = getCollectionStatus(docCollection, name, requestedShards);
514       } else {
515         collectionStatus = getCollectionStatus((Map<String, Object>) stateMap.get(name), name, requestedShards);
516       }
517 
518       collectionStatus.put("znodeVersion", clusterStateCollection.getZNodeVersion());
519       if (collectionVsAliases.containsKey(name) && !collectionVsAliases.get(name).isEmpty()) {
520         collectionStatus.put("aliases", collectionVsAliases.get(name));
521       }
522       String configName = zkStateReader.readConfigName(name);
523       collectionStatus.put("configName", configName);
524       collectionProps.add(name, collectionStatus);
525     }
526 
527     List<String> liveNodes = zkStateReader.getZkClient().getChildren(ZkStateReader.LIVE_NODES_ZKNODE, null, true);
528 
529     // now we need to walk the collectionProps tree to cross-check replica state with live nodes
530     crossCheckReplicaStateWithLiveNodes(liveNodes, collectionProps);
531 
532     NamedList<Object> clusterStatus = new SimpleOrderedMap<>();
533     clusterStatus.add("collections", collectionProps);
534 
535     // read cluster properties
536     Map clusterProps = zkStateReader.getClusterProps();
537     if (clusterProps != null && !clusterProps.isEmpty())  {
538       clusterStatus.add("properties", clusterProps);
539     }
540 
541     // add the alias map too
542     if (aliasVsCollections != null && !aliasVsCollections.isEmpty())  {
543       clusterStatus.add("aliases", aliasVsCollections);
544     }
545 
546     // add the roles map
547     if (roles != null)  {
548       clusterStatus.add("roles", roles);
549     }
550 
551     // add live_nodes
552     clusterStatus.add("live_nodes", liveNodes);
553 
554     results.add("cluster", clusterStatus);
555   }

557   /**
558    * Walks the tree of collection status to verify that any replicas not reporting a "down" status is
559    * on a live node, if any replicas reporting their status as "active" but the node is not live is
560    * marked as "down"; used by CLUSTERSTATUS.
561    * @param liveNodes List of currently live node names.
562    * @param collectionProps Map of collection status information pulled directly from ZooKeeper.
563    */
564 
565   @SuppressWarnings("unchecked")
566   protected void crossCheckReplicaStateWithLiveNodes(List<String> liveNodes, NamedList<Object> collectionProps) {
567     Iterator<Map.Entry<String,Object>> colls = collectionProps.iterator();
568     while (colls.hasNext()) {
569       Map.Entry<String,Object> next = colls.next();
570       Map<String,Object> collMap = (Map<String,Object>)next.getValue();
571       Map<String,Object> shards = (Map<String,Object>)collMap.get("shards");
572       for (Object nextShard : shards.values()) {
573         Map<String,Object> shardMap = (Map<String,Object>)nextShard;
574         Map<String,Object> replicas = (Map<String,Object>)shardMap.get("replicas");
575         for (Object nextReplica : replicas.values()) {
576           Map<String,Object> replicaMap = (Map<String,Object>)nextReplica;
577           if (Replica.State.getState((String) replicaMap.get(ZkStateReader.STATE_PROP)) != Replica.State.DOWN) {
578             // not down, so verify the node is live
579             String node_name = (String)replicaMap.get(ZkStateReader.NODE_NAME_PROP);
580             if (!liveNodes.contains(node_name)) {
581               // node is not live, so this replica is actually down
582               replicaMap.put(ZkStateReader.STATE_PROP, Replica.State.DOWN.toString());
583             }
584           }
585         }
586       }
587     }
588   }

617       }

666   @SuppressWarnings("unchecked")
667   private void deleteReplica(ClusterState clusterState, ZkNodeProps message, NamedList results)
668       throws KeeperException, InterruptedException {
669     checkRequired(message, COLLECTION_PROP, SHARD_ID_PROP, REPLICA_PROP);
670     String collectionName = message.getStr(COLLECTION_PROP);
671     String shard = message.getStr(SHARD_ID_PROP);
672     String replicaName = message.getStr(REPLICA_PROP);
673     
674     DocCollection coll = clusterState.getCollection(collectionName);
675     Slice slice = coll.getSlice(shard);
676     ShardHandler shardHandler = shardHandlerFactory.getShardHandler();
677     if (slice == null) {
678       throw new SolrException(ErrorCode.BAD_REQUEST,
679           "Invalid shard name : " + shard + " in collection : " + collectionName);
680     }
681     Replica replica = slice.getReplica(replicaName);
682     if (replica == null) {
683       ArrayList<String> l = new ArrayList<>();
684       for (Replica r : slice.getReplicas())
685         l.add(r.getName());
686       throw new SolrException(ErrorCode.BAD_REQUEST, "Invalid replica : " + replicaName + " in shard/collection : "
687           + shard + "/" + collectionName + " available replicas are " + StrUtils.join(l, ','));
688     }
689     
690     // If users are being safe and only want to remove a shard if it is down, they can specify onlyIfDown=true
691     // on the command.
692     if (Boolean.parseBoolean(message.getStr(ONLY_IF_DOWN)) && replica.getState() != Replica.State.DOWN) {
693       throw new SolrException(ErrorCode.BAD_REQUEST,
694           "Attempted to remove replica : " + collectionName + "/" + shard + "/" + replicaName
695               + " with onlyIfDown='true', but state is '" + replica.getStr(ZkStateReader.STATE_PROP) + "'");
696     }
697     
698     String baseUrl = replica.getStr(ZkStateReader.BASE_URL_PROP);
699     String core = replica.getStr(ZkStateReader.CORE_NAME_PROP);
700     
701     // assume the core exists and try to unload it
702     Map m = makeMap("qt", adminPath, CoreAdminParams.ACTION, CoreAdminAction.UNLOAD.toString(), CoreAdminParams.CORE,
703         core, CoreAdminParams.DELETE_INSTANCE_DIR, "true", CoreAdminParams.DELETE_DATA_DIR, "true");
704         
705     ShardRequest sreq = new ShardRequest();
706     sreq.purpose = 1;
707     sreq.shards = new String[] {baseUrl};
708     sreq.actualShards = sreq.shards;
709     sreq.params = new ModifiableSolrParams(new MapSolrParams(m));
710     try {
711       shardHandler.submit(sreq, baseUrl, sreq.params);
712     } catch (Exception e) {
713       log.warn("Exception trying to unload core " + sreq, e);
714     }
715     
716     collectShardResponses(replica.getState() != Replica.State.ACTIVE ? new NamedList() : results, false, null,
717         shardHandler);
718         
719     if (waitForCoreNodeGone(collectionName, shard, replicaName, 5000)) return;// check if the core unload removed the
720                                                                               // corenode zk enry
721     deleteCoreNode(collectionName, replicaName, replica, core); // try and ensure core info is removed from clusterstate
722     if (waitForCoreNodeGone(collectionName, shard, replicaName, 30000)) return;
723     
724     throw new SolrException(ErrorCode.SERVER_ERROR,
725         "Could not  remove replica : " + collectionName + "/" + shard + "/" + replicaName);
726         
727   }

944   private boolean createShard(ClusterState clusterState, ZkNodeProps message, NamedList results)

959         createNodeSetStr, overseer.getZkController().getCoreContainer());

966       Thread.sleep(100);

968       if (created) break;

971       throw new SolrException(ErrorCode.SERVER_ERROR, "Could not fully create shard: " + message.getStr(NAME));
972       
973     String configName = message.getStr(COLL_CONF);
974     for (int j = 1; j <= repFactor; j++) {

979           
980       // Need to create new params for each request
981       ModifiableSolrParams params = new ModifiableSolrParams();
982       params.set(CoreAdminParams.ACTION, CoreAdminAction.CREATE.toString());
983       
984       params.set(CoreAdminParams.NAME, shardName);
985       params.set(COLL_CONF, configName);
986       params.set(CoreAdminParams.COLLECTION, collectionName);
987       params.set(CoreAdminParams.SHARD, sliceName);
988       params.set(ZkStateReader.NUM_SHARDS_PROP, numSlices);

992       params.set("qt", adminPath);

994       String replica = zkStateReader.getBaseUrlForNodeName(nodeName);
995       sreq.shards = new String[] {replica};
996       sreq.actualShards = sreq.shards;
997       sreq.params = params;
998       
999       shardHandler.submit(sreq, replica, sreq.params);
1000       
1001     }
1002     
1003     processResponses(results, shardHandler);
1004     
1005     log.info("Finished create command on all shards for collection: " + collectionName);
1006     
1007     return true;
1008   }

1011   private boolean splitShard(ClusterState clusterState, ZkNodeProps message, NamedList results) {
1012     String collectionName = message.getStr("collection");
1013     String slice = message.getStr(ZkStateReader.SHARD_ID_PROP);
1014     
1015     log.info("Split shard invoked");
1016     String splitKey = message.getStr("split.key");
1017     ShardHandler shardHandler = shardHandlerFactory.getShardHandler();
1018     
1019     DocCollection collection = clusterState.getCollection(collectionName);
1020     DocRouter router = collection.getRouter() != null ? collection.getRouter() : DocRouter.DEFAULT;
1021     
1022     Slice parentSlice = null;
1023     
1024     if (slice == null) {
1025       if (router instanceof CompositeIdRouter) {
1026         Collection<Slice> searchSlices = router.getSearchSlicesSingle(splitKey, new ModifiableSolrParams(), collection);
1027         if (searchSlices.isEmpty()) {
1028           throw new SolrException(ErrorCode.BAD_REQUEST, "Unable to find an active shard for split.key: " + splitKey);
1029         }
1030         if (searchSlices.size() > 1) {
1031           throw new SolrException(ErrorCode.BAD_REQUEST,
1032               "Splitting a split.key: " + splitKey + " which spans multiple shards is not supported");
1033         }
1034         parentSlice = searchSlices.iterator().next();
1035         slice = parentSlice.getName();
1036         log.info("Split by route.key: {}, parent shard is: {} ", splitKey, slice);
1037       } else {
1038         throw new SolrException(ErrorCode.BAD_REQUEST,
1039             "Split by route key can only be used with CompositeIdRouter or subclass. Found router: "
1040                 + router.getClass().getName());
1041       }
1042     } else {
1043       parentSlice = clusterState.getSlice(collectionName, slice);
1044     }
1045     
1046     if (parentSlice == null) {
1047       if (clusterState.hasCollection(collectionName)) {
1048         throw new SolrException(ErrorCode.BAD_REQUEST, "No shard with the specified name exists: " + slice);
1049       } else {
1050         throw new SolrException(ErrorCode.BAD_REQUEST,
1051             "No collection with the specified name exists: " + collectionName);
1052       }
1053     }
1054     
1055     // find the leader for the shard
1056     Replica parentShardLeader = null;
1057     try {
1058       parentShardLeader = zkStateReader.getLeaderRetry(collectionName, slice, 10000);
1059     } catch (InterruptedException e) {
1060       Thread.currentThread().interrupt();
1061     }
1062     
1063     DocRouter.Range range = parentSlice.getRange();
1064     if (range == null) {
1065       range = new PlainIdRouter().fullRange();
1066     }
1067     
1068     List<DocRouter.Range> subRanges = null;
1069     String rangesStr = message.getStr(CoreAdminParams.RANGES);
1070     if (rangesStr != null) {
1071       String[] ranges = rangesStr.split(",");
1072       if (ranges.length == 0 || ranges.length == 1) {
1073         throw new SolrException(ErrorCode.BAD_REQUEST, "There must be at least two ranges specified to split a shard");
1074       } else {
1075         subRanges = new ArrayList<>(ranges.length);
1076         for (int i = 0; i < ranges.length; i++) {
1077           String r = ranges[i];
1078           try {
1079             subRanges.add(DocRouter.DEFAULT.fromString(r));
1080           } catch (Exception e) {
1081             throw new SolrException(ErrorCode.BAD_REQUEST, "Exception in parsing hexadecimal hash range: " + r, e);
1082           }
1083           if (!subRanges.get(i).isSubsetOf(range)) {
1084             throw new SolrException(ErrorCode.BAD_REQUEST,
1085                 "Specified hash range: " + r + " is not a subset of parent shard's range: " + range.toString());
1086           }
1087         }
1088         List<DocRouter.Range> temp = new ArrayList<>(subRanges); // copy to preserve original order
1089         Collections.sort(temp);
1090         if (!range.equals(new DocRouter.Range(temp.get(0).min, temp.get(temp.size() - 1).max))) {
1091           throw new SolrException(ErrorCode.BAD_REQUEST,
1092               "Specified hash ranges: " + rangesStr + " do not cover the entire range of parent shard: " + range);
1093         }
1094         for (int i = 1; i < temp.size(); i++) {
1095           if (temp.get(i - 1).max + 1 != temp.get(i).min) {
1096             throw new SolrException(ErrorCode.BAD_REQUEST, "Specified hash ranges: " + rangesStr
1097                 + " either overlap with each other or " + "do not cover the entire range of parent shard: " + range);
1098           }
1099         }
1100       }
1101     } else if (splitKey != null) {
1102       if (router instanceof CompositeIdRouter) {
1103         CompositeIdRouter compositeIdRouter = (CompositeIdRouter) router;
1104         subRanges = compositeIdRouter.partitionRangeByKey(splitKey, range);
1105         if (subRanges.size() == 1) {
1106           throw new SolrException(ErrorCode.BAD_REQUEST, "The split.key: " + splitKey
1107               + " has a hash range that is exactly equal to hash range of shard: " + slice);
1108         }
1109         for (DocRouter.Range subRange : subRanges) {
1110           if (subRange.min == subRange.max) {
1111             throw new SolrException(ErrorCode.BAD_REQUEST, "The split.key: " + splitKey + " must be a compositeId");
1112           }
1113         }
1114         log.info("Partitioning parent shard " + slice + " range: " + parentSlice.getRange() + " yields: " + subRanges);
1115         rangesStr = "";
1116         for (int i = 0; i < subRanges.size(); i++) {
1117           DocRouter.Range subRange = subRanges.get(i);
1118           rangesStr += subRange.toString();
1119           if (i < subRanges.size() - 1) rangesStr += ',';
1120         }
1121       }
1122     } else {
1123       // todo: fixed to two partitions?
1124       subRanges = router.partitionRange(2, range);
1125     }
1126     
1127     try {
1128       List<String> subSlices = new ArrayList<>(subRanges.size());
1129       List<String> subShardNames = new ArrayList<>(subRanges.size());
1130       String nodeName = parentShardLeader.getNodeName();
1131       for (int i = 0; i < subRanges.size(); i++) {
1132         String subSlice = slice + "_" + i;
1133         subSlices.add(subSlice);
1134         String subShardName = collectionName + "_" + subSlice + "_replica1";
1135         subShardNames.add(subShardName);
1136         
1137         Slice oSlice = clusterState.getSlice(collectionName, subSlice);
1138         if (oSlice != null) {
1139           final Slice.State state = oSlice.getState();
1140           if (state == Slice.State.ACTIVE) {
1141             throw new SolrException(ErrorCode.BAD_REQUEST,
1142                 "Sub-shard: " + subSlice + " exists in active state. Aborting split shard.");
1143           } else if (state == Slice.State.CONSTRUCTION || state == Slice.State.RECOVERY) {
1144             // delete the shards
1145             for (String sub : subSlices) {
1146               log.info("Sub-shard: {} already exists therefore requesting its deletion", sub);
1147               Map<String,Object> propMap = new HashMap<>();
1148               propMap.put(Overseer.QUEUE_OPERATION, "deleteshard");
1149               propMap.put(COLLECTION_PROP, collectionName);
1150               propMap.put(SHARD_ID_PROP, sub);
1151               ZkNodeProps m = new ZkNodeProps(propMap);
1152               try {
1153                 deleteShard(clusterState, m, new NamedList());
1154               } catch (Exception e) {
1155                 throw new SolrException(ErrorCode.SERVER_ERROR, "Unable to delete already existing sub shard: " + sub,
1156                     e);
1157               }
1158             }
1159           }
1160         }
1161       }
1162       
1163       // do not abort splitshard if the unloading fails
1164       // this can happen because the replicas created previously may be down
1165       // the only side effect of this is that the sub shard may end up having more replicas than we want
1166       collectShardResponses(results, false, null, shardHandler);
1167       
1168       final String asyncId = message.getStr(ASYNC);
1169       HashMap<String,String> requestMap = new HashMap<>();
1170       
1171       for (int i = 0; i < subRanges.size(); i++) {
1172         String subSlice = subSlices.get(i);
1173         String subShardName = subShardNames.get(i);
1174         DocRouter.Range subRange = subRanges.get(i);
1175         
1176         log.info("Creating slice " + subSlice + " of collection " + collectionName + " on " + nodeName);
1177         
1178         Map<String,Object> propMap = new HashMap<>();
1179         propMap.put(Overseer.QUEUE_OPERATION, CREATESHARD.toLower());
1180         propMap.put(ZkStateReader.SHARD_ID_PROP, subSlice);
1181         propMap.put(ZkStateReader.COLLECTION_PROP, collectionName);
1182         propMap.put(ZkStateReader.SHARD_RANGE_PROP, subRange.toString());
1183         propMap.put(ZkStateReader.SHARD_STATE_PROP, Slice.State.CONSTRUCTION.toString());
1184         propMap.put(ZkStateReader.SHARD_PARENT_PROP, parentSlice.getName());
1185         DistributedQueue inQueue = Overseer.getInQueue(zkStateReader.getZkClient());
1186         inQueue.offer(Utils.toJSON(new ZkNodeProps(propMap)));
1187         
1188         // wait until we are able to see the new shard in cluster state
1189         waitForNewShard(collectionName, subSlice);
1190         
1191         // refresh cluster state
1192         clusterState = zkStateReader.getClusterState();
1193         
1194         log.info("Adding replica " + subShardName + " as part of slice " + subSlice + " of collection " + collectionName
1195             + " on " + nodeName);
1196         propMap = new HashMap<>();
1197         propMap.put(Overseer.QUEUE_OPERATION, ADDREPLICA.toLower());
1198         propMap.put(COLLECTION_PROP, collectionName);
1199         propMap.put(SHARD_ID_PROP, subSlice);
1200         propMap.put("node", nodeName);
1201         propMap.put(CoreAdminParams.NAME, subShardName);
1202         // copy over property params:
1203         for (String key : message.keySet()) {
1204           if (key.startsWith(COLL_PROP_PREFIX)) {
1205             propMap.put(key, message.getStr(key));
1206           }
1207         }
1208         // add async param
1209         if (asyncId != null) {
1210           propMap.put(ASYNC, asyncId);
1211         }
1212         addReplica(clusterState, new ZkNodeProps(propMap), results);
1213       }
1214       
1215       collectShardResponses(results, true, "SPLITSHARD failed to create subshard leaders", shardHandler);
1216       
1217       completeAsyncRequest(asyncId, requestMap, results);
1218       
1219       for (String subShardName : subShardNames) {
1220         // wait for parent leader to acknowledge the sub-shard core
1221         log.info("Asking parent leader to wait for: " + subShardName + " to be alive on: " + nodeName);
1222         String coreNodeName = waitForCoreNodeName(collectionName, nodeName, subShardName);
1223         CoreAdminRequest.WaitForState cmd = new CoreAdminRequest.WaitForState();
1224         cmd.setCoreName(subShardName);
1225         cmd.setNodeName(nodeName);
1226         cmd.setCoreNodeName(coreNodeName);
1227         cmd.setState(Replica.State.ACTIVE);
1228         cmd.setCheckLive(true);
1229         cmd.setOnlyIfLeader(true);
1230         
1231         ModifiableSolrParams p = new ModifiableSolrParams(cmd.getParams());
1232         sendShardRequest(nodeName, p, shardHandler, asyncId, requestMap);
1233       }
1234       
1235       collectShardResponses(results, true, "SPLITSHARD timed out waiting for subshard leaders to come up",
1236           shardHandler);
1237           
1238       completeAsyncRequest(asyncId, requestMap, results);
1239       
1240       log.info("Successfully created all sub-shards for collection " + collectionName + " parent shard: " + slice
1241           + " on: " + parentShardLeader);
1242           
1243       log.info("Splitting shard " + parentShardLeader.getName() + " as part of slice " + slice + " of collection "
1244           + collectionName + " on " + parentShardLeader);
1245           
1246       ModifiableSolrParams params = new ModifiableSolrParams();
1247       params.set(CoreAdminParams.ACTION, CoreAdminAction.SPLIT.toString());
1248       params.set(CoreAdminParams.CORE, parentShardLeader.getStr("core"));
1249       for (int i = 0; i < subShardNames.size(); i++) {
1250         String subShardName = subShardNames.get(i);
1251         params.add(CoreAdminParams.TARGET_CORE, subShardName);
1252       }
1253       params.set(CoreAdminParams.RANGES, rangesStr);
1254       
1255       sendShardRequest(parentShardLeader.getNodeName(), params, shardHandler, asyncId, requestMap);
1256       
1257       collectShardResponses(results, true, "SPLITSHARD failed to invoke SPLIT core admin command", shardHandler);
1258       completeAsyncRequest(asyncId, requestMap, results);
1259       
1260       log.info("Index on shard: " + nodeName + " split into two successfully");
1261       
1262       // apply buffered updates on sub-shards
1263       for (int i = 0; i < subShardNames.size(); i++) {
1264         String subShardName = subShardNames.get(i);
1265         
1266         log.info("Applying buffered updates on : " + subShardName);
1267         
1268         params = new ModifiableSolrParams();
1269         params.set(CoreAdminParams.ACTION, CoreAdminAction.REQUESTAPPLYUPDATES.toString());
1270         params.set(CoreAdminParams.NAME, subShardName);
1271         
1272         sendShardRequest(nodeName, params, shardHandler, asyncId, requestMap);
1273       }
1274       
1275       collectShardResponses(results, true, "SPLITSHARD failed while asking sub shard leaders to apply buffered updates",
1276           shardHandler);
1277           
1278       completeAsyncRequest(asyncId, requestMap, results);
1279       
1280       log.info("Successfully applied buffered updates on : " + subShardNames);
1281       
1282       // Replica creation for the new Slices
1283       
1284       // look at the replication factor and see if it matches reality
1285       // if it does not, find best nodes to create more cores
1286       
1287       // TODO: Have replication factor decided in some other way instead of numShards for the parent
1288       
1289       int repFactor = clusterState.getSlice(collectionName, slice).getReplicas().size();
1290       
1291       // we need to look at every node and see how many cores it serves
1292       // add our new cores to existing nodes serving the least number of cores
1293       // but (for now) require that each core goes on a distinct node.
1294       
1295       // TODO: add smarter options that look at the current number of cores per
1296       // node?
1297       // for now we just go random
1298       Set<String> nodes = clusterState.getLiveNodes();
1299       List<String> nodeList = new ArrayList<>(nodes.size());
1300       nodeList.addAll(nodes);
1301       
1302       // TODO: Have maxShardsPerNode param for this operation?
1303       
1304       // Remove the node that hosts the parent shard for replica creation.
1305       nodeList.remove(nodeName);
1306       
1307       // TODO: change this to handle sharding a slice into > 2 sub-shards.
1308 
1309       List<Map<String, Object>> replicas = new ArrayList<>((repFactor - 1) * 2);
1310       for (int i = 1; i <= subSlices.size(); i++) {
1311         Collections.shuffle(nodeList, RANDOM);
1312         String sliceName = subSlices.get(i - 1);
1313         for (int j = 2; j <= repFactor; j++) {
1314           String subShardNodeName = nodeList.get((repFactor * (i - 1) + (j - 2)) % nodeList.size());
1315           String shardName = collectionName + "_" + sliceName + "_replica" + (j);
1316 
1317           log.info("Creating replica shard " + shardName + " as part of slice " + sliceName + " of collection "
1318               + collectionName + " on " + subShardNodeName);
1319 
1320           ZkNodeProps props = new ZkNodeProps(Overseer.QUEUE_OPERATION, ADDREPLICA.toLower(),
1321               ZkStateReader.COLLECTION_PROP, collectionName,
1322               ZkStateReader.SHARD_ID_PROP, sliceName,
1323               ZkStateReader.CORE_NAME_PROP, shardName,
1324               ZkStateReader.STATE_PROP, Replica.State.DOWN.toString(),
1325               ZkStateReader.BASE_URL_PROP, zkStateReader.getBaseUrlForNodeName(subShardNodeName),
1326               ZkStateReader.NODE_NAME_PROP, subShardNodeName);
1327           Overseer.getInQueue(zkStateReader.getZkClient()).offer(Utils.toJSON(props));
1328 
1329           HashMap<String,Object> propMap = new HashMap<>();
1330           propMap.put(Overseer.QUEUE_OPERATION, ADDREPLICA.toLower());
1331           propMap.put(COLLECTION_PROP, collectionName);
1332           propMap.put(SHARD_ID_PROP, sliceName);
1333           propMap.put("node", subShardNodeName);
1334           propMap.put(CoreAdminParams.NAME, shardName);
1335           // copy over property params:
1336           for (String key : message.keySet()) {
1337             if (key.startsWith(COLL_PROP_PREFIX)) {
1338               propMap.put(key, message.getStr(key));
1339             }
1340           }
1341           // add async param
1342           if (asyncId != null) {
1343             propMap.put(ASYNC, asyncId);
1344           }
1345           // special flag param to instruct addReplica not to create the replica in cluster state again
1346           propMap.put(SKIP_CREATE_REPLICA_IN_CLUSTER_STATE, "true");
1347 
1348           replicas.add(propMap);
1349         }
1350       }
1351 
1352       // we must set the slice state into recovery before actually creating the replica cores
1353       // this ensures that the logic inside Overseer to update sub-shard state to 'active'
1354       // always gets a chance to execute. See SOLR-7673
1355 
1356       if (repFactor == 1) {
1357         // switch sub shard states to 'active'
1358         log.info("Replication factor is 1 so switching shard states");
1359         DistributedQueue inQueue = Overseer.getInQueue(zkStateReader.getZkClient());
1360         Map<String,Object> propMap = new HashMap<>();
1361         propMap.put(Overseer.QUEUE_OPERATION, OverseerAction.UPDATESHARDSTATE.toLower());
1362         propMap.put(slice, Slice.State.INACTIVE.toString());
1363         for (String subSlice : subSlices) {
1364           propMap.put(subSlice, Slice.State.ACTIVE.toString());
1365         }
1366         propMap.put(ZkStateReader.COLLECTION_PROP, collectionName);
1367         ZkNodeProps m = new ZkNodeProps(propMap);
1368         inQueue.offer(Utils.toJSON(m));
1369       } else {
1370         log.info("Requesting shard state be set to 'recovery'");
1371         DistributedQueue inQueue = Overseer.getInQueue(zkStateReader.getZkClient());
1372         Map<String,Object> propMap = new HashMap<>();
1373         propMap.put(Overseer.QUEUE_OPERATION, OverseerAction.UPDATESHARDSTATE.toLower());
1374         for (String subSlice : subSlices) {
1375           propMap.put(subSlice, Slice.State.RECOVERY.toString());
1376         }
1377         propMap.put(ZkStateReader.COLLECTION_PROP, collectionName);
1378         ZkNodeProps m = new ZkNodeProps(propMap);
1379         inQueue.offer(Utils.toJSON(m));
1380       }
1381 
1382       // now actually create replica cores on sub shard nodes
1383       for (Map<String, Object> replica : replicas) {
1384         addReplica(clusterState, new ZkNodeProps(replica), results);
1385       }
1386       
1387       collectShardResponses(results, true,
1388           "SPLITSHARD failed to create subshard replicas", shardHandler);
1389           
1390       completeAsyncRequest(asyncId, requestMap, results);
1391       
1392       log.info("Successfully created all replica shards for all sub-slices " + subSlices);
1393       
1394       commit(results, slice, parentShardLeader);
1395       
1396       return true;
1397     } catch (SolrException e) {
1398       throw e;
1399     } catch (Exception e) {
1400       log.error("Error executing split operation for collection: " + collectionName + " parent shard: " + slice, e);
1401       throw new SolrException(ErrorCode.SERVER_ERROR, null, e);
1402     }
1403   }

1487   private void collectShardResponses(NamedList results, boolean abortOnError,
1488                                      String msgOnError,
1489                                      ShardHandler shardHandler) {
1490     ShardResponse srsp;

1494         processResponse(results, srsp);

1497           // drain pending requests
1498           while (srsp != null)  {
1499             srsp = shardHandler.takeCompletedOrError();
1500           }

1502         }
1503       }
1504     } while (srsp != null);

1507   private void deleteShard(ClusterState clusterState, ZkNodeProps message, NamedList results) {
1508     String collection = message.getStr(ZkStateReader.COLLECTION_PROP);
1509     String sliceId = message.getStr(ZkStateReader.SHARD_ID_PROP);
1510     
1511     log.info("Delete shard invoked");
1512     Slice slice = clusterState.getSlice(collection, sliceId);
1513     
1514     if (slice == null) {
1515       if (clusterState.hasCollection(collection)) {
1516         throw new SolrException(ErrorCode.BAD_REQUEST,
1517             "No shard with name " + sliceId + " exists for collection " + collection);
1518       } else {
1519         throw new SolrException(ErrorCode.BAD_REQUEST, "No collection with the specified name exists: " + collection);
1520       }
1521     }
1522     // For now, only allow for deletions of Inactive slices or custom hashes (range==null).
1523     // TODO: Add check for range gaps on Slice deletion
1524     final Slice.State state = slice.getState();
1525     if (!(slice.getRange() == null || state == Slice.State.INACTIVE || state == Slice.State.RECOVERY
1526         || state == Slice.State.CONSTRUCTION)) {
1527       throw new SolrException(ErrorCode.BAD_REQUEST, "The slice: " + slice.getName() + " is currently " + state
1528           + ". Only non-active (or custom-hashed) slices can be deleted.");
1529     }
1530     ShardHandler shardHandler = shardHandlerFactory.getShardHandler();
1531     
1532     try {
1533       ModifiableSolrParams params = new ModifiableSolrParams();
1534       params.set(CoreAdminParams.ACTION, CoreAdminAction.UNLOAD.toString());
1535       params.set(CoreAdminParams.DELETE_INDEX, "true");
1536       sliceCmd(clusterState, params, null, slice, shardHandler);
1537       
1538       processResponses(results, shardHandler);
1539       
1540       ZkNodeProps m = new ZkNodeProps(Overseer.QUEUE_OPERATION, DELETESHARD.toLower(), ZkStateReader.COLLECTION_PROP,
1541           collection, ZkStateReader.SHARD_ID_PROP, sliceId);
1542       Overseer.getInQueue(zkStateReader.getZkClient()).offer(Utils.toJSON(m));
1543       
1544       // wait for a while until we don't see the shard
1545       TimeOut timeout = new TimeOut(30, TimeUnit.SECONDS);
1546       boolean removed = false;
1547       while (! timeout.hasTimedOut()) {
1548         Thread.sleep(100);
1549         removed = zkStateReader.getClusterState().getSlice(collection, sliceId) == null;
1550         if (removed) {
1551           Thread.sleep(100); // just a bit of time so it's more likely other readers see on return
1552           break;
1553         }
1554       }
1555       if (!removed) {
1556         throw new SolrException(ErrorCode.SERVER_ERROR,
1557             "Could not fully remove collection: " + collection + " shard: " + sliceId);
1558       }
1559       
1560       log.info("Successfully deleted collection: " + collection + ", shard: " + sliceId);
1561       
1562     } catch (SolrException e) {
1563       throw e;
1564     } catch (Exception e) {
1565       throw new SolrException(ErrorCode.SERVER_ERROR,
1566           "Error executing delete operation for collection: " + collection + " shard: " + sliceId, e);
1567     }
1568   }

1570   private void migrate(ClusterState clusterState, ZkNodeProps message, NamedList results) throws KeeperException, InterruptedException {
1571     String sourceCollectionName = message.getStr("collection");
1572     String splitKey = message.getStr("split.key");
1573     String targetCollectionName = message.getStr("target.collection");
1574     int timeout = message.getInt("forward.timeout", 10 * 60) * 1000;
1575 
1576     DocCollection sourceCollection = clusterState.getCollection(sourceCollectionName);
1577     if (sourceCollection == null) {
1578       throw new SolrException(ErrorCode.BAD_REQUEST, "Unknown source collection: " + sourceCollectionName);
1579     }
1580     DocCollection targetCollection = clusterState.getCollection(targetCollectionName);
1581     if (targetCollection == null) {
1582       throw new SolrException(ErrorCode.BAD_REQUEST, "Unknown target collection: " + sourceCollectionName);
1583     }
1584     if (!(sourceCollection.getRouter() instanceof CompositeIdRouter))  {
1585       throw new SolrException(ErrorCode.BAD_REQUEST, "Source collection must use a compositeId router");
1586     }
1587     if (!(targetCollection.getRouter() instanceof CompositeIdRouter))  {

1596     }

1608       for (Slice targetSlice : targetSlices) {
1609         log.info("Migrating source shard: {} to target shard: {} for split.key = " + splitKey, sourceSlice, targetSlice);
1610         migrateKey(clusterState, sourceCollection, sourceSlice, targetCollection, targetSlice, splitKey,
1611             timeout, results, asyncId, message);
1612       }
1613     }

1852   private DocRouter.Range intersect(DocRouter.Range a, DocRouter.Range b) {
1853     if (a == null || b == null || !a.overlaps(b)) {
1854       return null;
1855     } else if (a.isSubsetOf(b))
1856       return a;
1857     else if (b.isSubsetOf(a))
1858       return b;
1859     else if (b.includes(a.max)) {
1860       return new DocRouter.Range(b.min, a.max);
1861     } else  {
1862       return new DocRouter.Range(a.min, b.max);
1863     }
1864   }

1928   private void createCollection(ClusterState clusterState, ZkNodeProps message, NamedList results) throws KeeperException, InterruptedException {
1929     final String collectionName = message.getStr(NAME);
1930     if (clusterState.hasCollection(collectionName)) {
1931       throw new SolrException(ErrorCode.BAD_REQUEST, "collection already exists: " + collectionName);
1932     }
1933 
1934     String configName = getConfigName(collectionName, message);
1935     if (configName == null) {
1936       throw new SolrException(ErrorCode.BAD_REQUEST, "No config set found to associate with the collection.");
1937     } else if (!validateConfig(configName)) {
1938       throw new SolrException(ErrorCode.BAD_REQUEST, "Can not find the specified config set: " + configName);
1939     }
1940 
1941     try {
1942       // look at the replication factor and see if it matches reality
1943       // if it does not, find best nodes to create more cores
1944 
1945       int repFactor = message.getInt(REPLICATION_FACTOR, 1);
1946 
1947       ShardHandler shardHandler = shardHandlerFactory.getShardHandler();
1948       final String async = message.getStr(ASYNC);
1949 
1950       Integer numSlices = message.getInt(NUM_SLICES, null);
1951       String router = message.getStr("router.name", DocRouter.DEFAULT_NAME);
1952       List<String> shardNames = new ArrayList<>();
1953       if(ImplicitDocRouter.NAME.equals(router)){
1954         ClusterStateMutator.getShardNames(shardNames, message.getStr("shards", null));
1955         numSlices = shardNames.size();
1956       } else {
1957         if (numSlices == null ) {
1958           throw new SolrException(ErrorCode.BAD_REQUEST, NUM_SLICES + " is a required param (when using CompositeId router).");
1959         }
1960         ClusterStateMutator.getShardNames(numSlices, shardNames);
1961       }
1962 
1963       int maxShardsPerNode = message.getInt(MAX_SHARDS_PER_NODE, 1);
1964 
1965       if (repFactor <= 0) {
1966         throw new SolrException(ErrorCode.BAD_REQUEST, REPLICATION_FACTOR + " must be greater than 0");
1967       }
1968 
1969       if (numSlices <= 0) {
1970         throw new SolrException(ErrorCode.BAD_REQUEST, NUM_SLICES + " must be > 0");
1971       }
1972 
1973       // we need to look at every node and see how many cores it serves
1974       // add our new cores to existing nodes serving the least number of cores
1975       // but (for now) require that each core goes on a distinct node.
1976 
1977       final List<String> nodeList = getLiveOrLiveAndCreateNodeSetList(clusterState.getLiveNodes(), message, RANDOM);
1978       Map<Position, String> positionVsNodes;
1979       if (nodeList.isEmpty()) {
1980         log.warn("It is unusual to create a collection ("+collectionName+") without cores.");
1981

1986               + REPLICATION_FACTOR

1990               + collectionName
1991               + " is higher than or equal to the number of Solr instances currently live or live and part of your " + CREATE_NODE_SET + "("
1992               + nodeList.size()
1993               + "). It's unusual to run two replica of the same slice on the same Solr-instance.");
1994         }
1995         
1996         int maxShardsAllowedToCreate = maxShardsPerNode * nodeList.size();
1997         int requestedShardsToCreate = numSlices * repFactor;
1998         if (maxShardsAllowedToCreate < requestedShardsToCreate) {
1999           throw new SolrException(ErrorCode.BAD_REQUEST, "Cannot create collection " + collectionName + ". Value of "
2000               + MAX_SHARDS_PER_NODE + " is " + maxShardsPerNode
2001               + ", and the number of nodes currently live or live and part of your "+CREATE_NODE_SET+" is " + nodeList.size()
2002               + ". This allows a maximum of " + maxShardsAllowedToCreate
2003               + " to be created. Value of " + NUM_SLICES + " is " + numSlices
2004               + " and value of " + REPLICATION_FACTOR + " is " + repFactor
2005               + ". This requires " + requestedShardsToCreate
2006               + " shards to be created (higher than the allowed number)");
2007         }

2039           collectionName, shardNames, repFactor));

2046             , coreName, position.shard, collectionName, nodeName));

2054               Overseer.QUEUE_OPERATION, ADDREPLICA.toString(),
2055               ZkStateReader.COLLECTION_PROP, collectionName,
2056               ZkStateReader.SHARD_ID_PROP, position.shard,
2057               ZkStateReader.CORE_NAME_PROP, coreName,
2058               ZkStateReader.STATE_PROP, Replica.State.DOWN.toString(),
2059               ZkStateReader.BASE_URL_PROP, baseUrl);
2060           Overseer.getInQueue(zkStateReader.getZkClient()).offer(Utils.toJSON(props));
2061         }

2065         params.set(CoreAdminParams.ACTION, CoreAdminAction.CREATE.toString());
