45  /**
46   * Leader Election process. This class contains the logic by which a
47   * leader is chosen. First call * {@link #setup(ElectionContext)} to ensure
48   * the election process is init'd. Next call
49   * {@link #joinElection(ElectionContext, boolean)} to start the leader election.
50   * 
51   * The implementation follows the classic ZooKeeper recipe of creating an
52   * ephemeral, sequential node for each candidate and then looking at the set
53   * of such nodes - if the created node is the lowest sequential node, the
54   * candidate that created the node is the leader. If not, the candidate puts
55   * a watch on the next lowest node it finds, and if that node goes down, 
56   * starts the whole process over by checking if it's the lowest sequential node, etc.
57   * 
58   */
59  public  class LeaderElector {
60    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
61    
62    static final String ELECTION_NODE = "/election";
63    
64    public final static Pattern LEADER_SEQ = Pattern.compile(".*?/?.*?-n_(\\d+)");
65    private final static Pattern SESSION_ID = Pattern.compile(".*?/?(.*?-.*?)-n_\\d+");
66    private final static Pattern  NODE_NAME = Pattern.compile(".*?/?(.*?-)(.*?)-n_\\d+");

95    /**
96     * Check if the candidate with the given n_* sequence number is the leader.
97     * If it is, set the leaderId on the leader zk node. If it is not, start
98     * watching the candidate that is in line before this one - if it goes down, check
99     * if this candidate is the leader again.
100    *
101    * @param replacement has someone else been the leader already?
102    */
103   private void checkIfIamLeader(final ElectionContext context, boolean replacement) throws KeeperException,
104       InterruptedException, IOException {
105     context.checkIfIamLeaderFired();
106     // get all other numbers...
107     final String holdElectionPath = context.electionPath + ELECTION_NODE;
108     List<String> seqs = zkClient.getChildren(holdElectionPath, null, true);
109 
110     sortSeqs(seqs);
111     List<Integer> intSeqs = getSeqs(seqs);
112     if (intSeqs.size() == 0) {
113       log.warn("Our node is no longer in line to be leader");
114       return;
115     }
116     // We can't really rely on the sequence number stored in the old watcher, it may be stale, thus this check.
117 
118     int seq = -1;
119 
120     // See if we've already been re-added, and this is an old context. In which case, use our current sequence number.
121     String newLeaderSeq = "";
122     for (String elec : seqs) {
123       if (getNodeName(elec).equals(getNodeName(context.leaderSeqPath)) && seq < getSeq(elec)) {
124         seq = getSeq(elec); // so use the current sequence number.
125         newLeaderSeq = elec;
126         break;
127       }
128     }
129 
130     // Now, if we've been re-added, presumably we've also set up watchers and all that kind of thing, so we're done
131     if (StringUtils.isNotBlank(newLeaderSeq) && seq > getSeq(context.leaderSeqPath)) {
132       log.info("Node " + context.leaderSeqPath + " already in queue as " + newLeaderSeq + " nothing to do.");
133       return;
134     }
135 
136     // Fallback in case we're all coming in here fresh and there is no node for this core already in the election queue.
137     if (seq == -1) {
138       seq = getSeq(context.leaderSeqPath);
139     }
140 
141     if (seq <= intSeqs.get(0)) {
142       if (seq == intSeqs.get(0) && !context.leaderSeqPath.equals(holdElectionPath + "/" + seqs.get(0))) {//somebody else already  became the leader with the same sequence id , not me
143         log.info("was going to be leader {} , seq(0) {}", context.leaderSeqPath, holdElectionPath + "/" + seqs.get(0));//but someone else jumped the line
144 
145         // The problem is that deleting the ZK node that's watched by others
146         // results in an unpredictable sequencing of the events and sometime the context that comes in for checking
147         // this happens to be after the node has already taken over leadership. So just leave out of here.
148         // This caused one of the tests to fail on having two nodes with the same name in the queue. I'm not sure
149         // the assumption that this is a bad state is valid.
150         if (getNodeName(context.leaderSeqPath).equals(getNodeName(seqs.get(0)))) {
151           return;
152         }
153         retryElection(context, false);//join at the tail again
154         return;
155       }
156 
157       try {
158         runIamLeaderProcess(context, replacement);
159       } catch (KeeperException.NodeExistsException e) {
160         log.error("node exists",e);
161         retryElection(context, false);
162         return;
163       }
164     } else {
165       // I am not the leader - watch the node below me
166       int toWatch = -1;
167       for (int idx = 0; idx < intSeqs.size(); idx++) {
168         if (intSeqs.get(idx) < seq && ! getNodeName(context.leaderSeqPath).equals(getNodeName(seqs.get(idx)))) {
169           toWatch = idx;
170         }
171         if (intSeqs.get(idx) >= seq) {
172           break;
173         }
174       }
175       if (toWatch < 0) {
176         log.warn("Our node is no longer in line to be leader");
177         return;
178       }
179       try {
180         String watchedNode = holdElectionPath + "/" + seqs.get(toWatch);
181 
182         zkClient.getData(watchedNode, watcher = new ElectionWatcher(context.leaderSeqPath , watchedNode,seq, context) , null, true);
183         log.info("Watching path {} to know if I could be the leader", watchedNode);
184       } catch (KeeperException.SessionExpiredException e) {
185         throw e;
186       } catch (KeeperException e) {
187         log.warn("Failed setting watch", e);
188         // we couldn't set our watch - the node before us may already be down?
189         // we need to check if we are the leader again
190         checkIfIamLeader(context, true);
191       }
192     }
193   }

201   /**
202    * Returns int given String of form n_0000000001 or n_0000000003, etc.
203    * 
204    * @return sequence number
205    */
206   public static int getSeq(String nStringSequence) {
207     int seq = 0;
208     Matcher m = LEADER_SEQ.matcher(nStringSequence);
209     if (m.matches()) {
210       seq = Integer.parseInt(m.group(1));
211     } else {
212       throw new IllegalStateException("Could not find regex match in:"
213           + nStringSequence);
214     }
215     return seq;
216   }
217   
218   private String getNodeId(String nStringSequence) {
219     String id;
220     Matcher m = SESSION_ID.matcher(nStringSequence);
221     if (m.matches()) {
222       id = m.group(1);
223     } else {
224       throw new IllegalStateException("Could not find regex match in:"
225           + nStringSequence);
226     }
227     return id;
228   }
229 
230   public static String getNodeName(String nStringSequence){
231     String result;
232     Matcher m = NODE_NAME.matcher(nStringSequence);
233     if (m.matches()) {
234       result = m.group(2);
235     } else {
236       throw new IllegalStateException("Could not find regex match in:"
237           + nStringSequence);
238     }
239     return result;
240 
241   }

243   /**
244    * Returns int list given list of form n_0000000001, n_0000000003, etc.
245    * 
246    * @return int seqs
247    */
248   private List<Integer> getSeqs(List<String> seqs) {
249     List<Integer> intSeqs = new ArrayList<>(seqs.size());
250     for (String seq : seqs) {
251       intSeqs.add(getSeq(seq));
252     }
253     return intSeqs;
254   }
255   public int joinElection(ElectionContext context, boolean replacement) throws KeeperException, InterruptedException, IOException {
256     return joinElection(context,replacement, false);
257   }

259     /**
260      * Begin participating in the election process. Gets a new sequential number
261      * and begins watching the node with the sequence number before it, unless it
262      * is the lowest number, in which case, initiates the leader process. If the
263      * node that is watched goes down, check if we are the new lowest node, else
264      * watch the next lowest numbered node.
265      *
266      * @return sequential node number
267      */
268   public int joinElection(ElectionContext context, boolean replacement,boolean joinAtHead) throws KeeperException, InterruptedException, IOException {
269     context.joinedElectionFired();
270     
271     final String shardsElectZkPath = context.electionPath + LeaderElector.ELECTION_NODE;
272     
273     long sessionId = zkClient.getSolrZooKeeper().getSessionId();
274     String id = sessionId + "-" + context.id;
275     String leaderSeqPath = null;
276     boolean cont = true;
277     int tries = 0;
278     while (cont) {
279       try {
280         if(joinAtHead){
281           log.info("Node {} trying to join election at the head", id);
282           List<String> nodes = OverseerTaskProcessor.getSortedElectionNodes(zkClient, shardsElectZkPath);
283           if(nodes.size() <2){
284             leaderSeqPath = zkClient.create(shardsElectZkPath + "/" + id + "-n_", null,
285                 CreateMode.EPHEMERAL_SEQUENTIAL, false);
286           } else {
287             String firstInLine = nodes.get(1);
288             log.info("The current head: {}", firstInLine);
289             Matcher m = LEADER_SEQ.matcher(firstInLine);
290             if (!m.matches()) {
291               throw new IllegalStateException("Could not find regex match in:"
292                   + firstInLine);
293             }
294             leaderSeqPath = shardsElectZkPath + "/" + id + "-n_"+ m.group(1);
295             zkClient.create(leaderSeqPath, null, CreateMode.EPHEMERAL, false);
296           }
297         } else {
298           leaderSeqPath = zkClient.create(shardsElectZkPath + "/" + id + "-n_", null,
299               CreateMode.EPHEMERAL_SEQUENTIAL, false);
300         }
301 
302         log.info("Joined leadership election with path: {}", leaderSeqPath);
303         context.leaderSeqPath = leaderSeqPath;
304         cont = false;
305       } catch (ConnectionLossException e) {
306         // we don't know if we made our node or not...
307         List<String> entries = zkClient.getChildren(shardsElectZkPath, null, true);
308         
309         boolean foundId = false;
310         for (String entry : entries) {
311           String nodeId = getNodeId(entry);
312           if (id.equals(nodeId)) {
313             // we did create our node...
314             foundId  = true;
315             break;
316           }
317         }
318         if (!foundId) {
319           cont = true;
320           if (tries++ > 20) {
321             throw new ZooKeeperException(SolrException.ErrorCode.SERVER_ERROR,
322                 "", e);
323           }
324           try {
325             Thread.sleep(50);
326           } catch (InterruptedException e2) {
327             Thread.currentThread().interrupt();
328           }
329         }
330 
331       } catch (KeeperException.NoNodeException e) {
332         // we must have failed in creating the election node - someone else must
333         // be working on it, lets try again
334         if (tries++ > 20) {
335           context = null;
336           throw new ZooKeeperException(SolrException.ErrorCode.SERVER_ERROR,
337               "", e);
338         }
339         cont = true;
340         try {
341           Thread.sleep(50);
342         } catch (InterruptedException e2) {
343           Thread.currentThread().interrupt();
344         }
345       }
346     }
347     checkIfIamLeader(context, replacement);
348 
349     return getSeq(context.leaderSeqPath);
350   }

352   private class ElectionWatcher implements Watcher {
353     final String myNode,watchedNode;
354     final ElectionContext context;
355 
356     private boolean canceled = false;
357 
358     private ElectionWatcher(String myNode, String watchedNode, int seq, ElectionContext context) {
359       this.myNode = myNode;
360       this.watchedNode = watchedNode;
361       this.context = context;
362     }
363 
364     void cancel() {
365       canceled = true;
366 
367     }
368 
369     @Override
370     public void process(WatchedEvent event) {
371       // session events are not change events,
372       // and do not remove the watcher
373       if (EventType.None.equals(event.getType())) {
374         return;
375       }
376       if (canceled) {
377         log.info("This watcher is not active anymore {}", myNode);
378         try {
379           zkClient.delete(myNode, -1, true);
380         } catch (KeeperException.NoNodeException nne) {
381           // expected . don't do anything
382         } catch (Exception e) {
383           log.warn("My watched node still exists and can't remove " + myNode, e);
384         }
385         return;
386       }
387       try {
388         // am I the next leader?
389         checkIfIamLeader(context, true);
390       } catch (Exception e) {
391         log.warn("", e);
392       }
393     }
394   }

396   /**
397    * Set up any ZooKeeper nodes needed for leader election.
398    */
399   public void setup(final ElectionContext context) throws InterruptedException,
400       KeeperException {
401     String electZKPath = context.electionPath + LeaderElector.ELECTION_NODE;
402     
403     zkCmdExecutor.ensureExists(electZKPath, zkClient);
404     this.context = context;
405   }

407   /**
408    * Sort n string sequence list.
409    */
410   public static void sortSeqs(List<String> seqs) {
411     Collections.sort(seqs, new Comparator<String>() {
412       
413       @Override
414       public int compare(String o1, String o2) {
415         int i = Integer.valueOf(getSeq(o1)).compareTo(
416             Integer.valueOf(getSeq(o2)));
417         return i == 0 ? o1.compareTo(o2) : i ;
418       }
419     });
420   }

422   void retryElection(ElectionContext context, boolean joinAtHead) throws KeeperException, InterruptedException, IOException {
423     ElectionWatcher watcher = this.watcher;
424     ElectionContext ctx = context.copy();
425     if (electionContexts != null) {
426       electionContexts.put(contextKey, ctx);
427     }
428     if (watcher != null) watcher.cancel();
429     this.context.cancelElection();
430     this.context = ctx;
431     joinElection(ctx, true, joinAtHead);
432   }
