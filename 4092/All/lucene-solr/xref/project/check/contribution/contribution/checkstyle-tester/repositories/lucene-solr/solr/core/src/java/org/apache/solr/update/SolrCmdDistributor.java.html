96    private void doRetriesIfNeeded() {
97      // NOTE: retries will be forwards to a single url
98      
99      List<Error> errors = new ArrayList<>(this.errors);
100     errors.addAll(clients.getErrors());
101     List<Error> resubmitList = new ArrayList<>();
102 
103     for (Error err : errors) {
104       try {
105         String oldNodeUrl = err.req.node.getUrl();
106         
107         // if there is a retry url, we want to retry...
108         boolean isRetry = err.req.node.checkRetry();
109         
110         boolean doRetry = false;
111         int rspCode = err.statusCode;
112         
113         if (testing_errorHook != null) Diagnostics.call(testing_errorHook,
114             err.e);
115         
116         // this can happen in certain situations such as close
117         if (isRetry) {
118           if (rspCode == 404 || rspCode == 403 || rspCode == 503) {
119             doRetry = true;
120           }
121           
122           // if it's a connect exception, lets try again
123           if (err.e instanceof SolrServerException) {
124             if (((SolrServerException) err.e).getRootCause() instanceof ConnectException) {
125               doRetry = true;
126             }
127           }
128           
129           if (err.e instanceof ConnectException) {
130             doRetry = true;
131           }
132           
133           if (err.req.retries < maxRetriesOnForward && doRetry) {
134             err.req.retries++;
135             
136             SolrException.log(SolrCmdDistributor.log, "forwarding update to "
137                 + oldNodeUrl + " failed - retrying ... retries: "
138                 + err.req.retries + " " + err.req.cmdString + " params:"
139                 + err.req.uReq.getParams() + " rsp:" + rspCode, err.e);
140             try {
141               Thread.sleep(retryPause);
142             } catch (InterruptedException e) {
143               Thread.currentThread().interrupt();
144               log.warn(null, e);
145             }
146             
147             resubmitList.add(err);
148           } else {
149             allErrors.add(err);
150           }
151         } else {
152           allErrors.add(err);
153         }
154       } catch (Exception e) {
155         // continue on
156         log.error("Unexpected Error while doing request retries", e);
157       }
158     }
159     
160     clients.clearErrors();
161     this.errors.clear();
162     for (Error err : resubmitList) {
163       submit(err.req, false);
164     }
165     
166     if (resubmitList.size() > 0) {
167       blockAndDoRetries();
168     }
169   }

171   public void distribDelete(DeleteUpdateCommand cmd, List<Node> nodes, ModifiableSolrParams params) throws IOException {
172     distribDelete(cmd, nodes, params, false);
173   }

175   public void distribDelete(DeleteUpdateCommand cmd, List<Node> nodes, ModifiableSolrParams params, boolean sync) throws IOException {
176     
177     for (Node node : nodes) {
178       UpdateRequest uReq = new UpdateRequest();
179       uReq.setParams(params);
180       uReq.setCommitWithin(cmd.commitWithin);
181       if (cmd.isDeleteById()) {
182         uReq.deleteById(cmd.getId(), cmd.getRoute(), cmd.getVersion());
183       } else {
184         uReq.deleteByQuery(cmd.query);
185       }
186       
187       submit(new Req(cmd.toString(), node, uReq, sync), false);
188     }
189   }

199   public void distribAdd(AddUpdateCommand cmd, List<Node> nodes, ModifiableSolrParams params, boolean synchronous, RequestReplicationTracker rrt) throws IOException {  
200     String cmdStr = cmd.toString();
201     for (Node node : nodes) {
202       UpdateRequest uReq = new UpdateRequest();
203       if (cmd.isLastDocInBatch)
204         uReq.lastDocInBatch();
205       uReq.setParams(params);
206       uReq.add(cmd.solrDoc, cmd.commitWithin, cmd.overwrite);
207       submit(new Req(cmdStr, node, uReq, synchronous, rrt, cmd.pollQueueTime), false);
208     }
209     
210   }

212   public void distribCommit(CommitUpdateCommand cmd, List<Node> nodes,
213       ModifiableSolrParams params) throws IOException {
214     
215     // we need to do any retries before commit...
216     blockAndDoRetries();
217     
218     UpdateRequest uReq = new UpdateRequest();
219     uReq.setParams(params);
220     
221     addCommit(uReq, cmd);
222     
223     log.debug("Distrib commit to: {} params: {}", nodes, params);
224     
225     for (Node node : nodes) {
226       submit(new Req(cmd.toString(), node, uReq, false), true);
227     }
228     
229   }

231   private void blockAndDoRetries() {
232     clients.blockUntilFinished();
233     
234     // wait for any async commits to complete
235     while (pending != null && pending.size() > 0) {
236       Future<Object> future = null;
237       try {
238         future = completionService.take();
239       } catch (InterruptedException e) {
240         Thread.currentThread().interrupt();
241         log.error("blockAndDoRetries interrupted", e);
242       }
243       if (future == null) break;
244       pending.remove(future);
245     }
246     doRetriesIfNeeded();
247 
248   }
249   
250   void addCommit(UpdateRequest ureq, CommitUpdateCommand cmd) {
251     if (cmd == null) return;
252     ureq.setAction(cmd.optimize ? AbstractUpdateRequest.ACTION.OPTIMIZE
253         : AbstractUpdateRequest.ACTION.COMMIT, false, cmd.waitSearcher, cmd.maxOptimizeSegments, cmd.softCommit, cmd.expungeDeletes, cmd.openSearcher);
254   }

256   private void submit(final Req req, boolean isCommit) {
257     if (req.synchronous) {
258       blockAndDoRetries();
259 
260       try (HttpSolrClient client = new HttpSolrClient(req.node.getUrl(), clients.getHttpClient())) {
261         client.request(req.uReq);
262       } catch (Exception e) {
263         throw new SolrException(ErrorCode.SERVER_ERROR, "Failed synchronous update on shard " + req.node + " update: " + req.uReq , e);
264       }
265       
266       return;
267     }
268     
269     if (log.isDebugEnabled()) {
270       log.debug("sending update to "
271           + req.node.getUrl() + " retry:"
272           + req.retries + " " + req.cmdString + " params:" + req.uReq.getParams());
273     }
274     
275     if (isCommit) {
276       // a commit using ConncurrentUpdateSolrServer is not async,
277       // so we make it async to prevent commits from happening
278       // serially across multiple nodes
279       pending.add(completionService.submit(new Callable<Object>() {
280         
281         @Override
282         public Object call() throws Exception {
283           doRequest(req);
284           return null;
285         }
286 
287       }));
288     } else {
289       doRequest(req);
290     }
291   }
292   
293   private void doRequest(final Req req) {
294     try {
295       SolrClient solrClient = clients.getSolrClient(req);
296       solrClient.request(req.uReq);
297     } catch (Exception e) {
298       SolrException.log(log, e);
299       Error error = new Error();
300       error.e = e;
301       error.req = req;
302       if (e instanceof SolrException) {
303         error.statusCode = ((SolrException) e).code();
304       }
305       errors.add(error);
306     }
307   }

310     public Node node;
311     public UpdateRequest uReq;
312     public int retries;
313     public boolean synchronous;
314     public String cmdString;
315     public RequestReplicationTracker rfTracker;
316     public int pollQueueTime;
317 
318     public Req(String cmdString, Node node, UpdateRequest uReq, boolean synchronous) {
319       this(cmdString, node, uReq, synchronous, null, 0);
320     }
321     
322     public Req(String cmdString, Node node, UpdateRequest uReq, boolean synchronous, RequestReplicationTracker rfTracker, int pollQueueTime) {
323       this.node = node;
324       this.uReq = uReq;
325       this.synchronous = synchronous;
326       this.cmdString = cmdString;
327       this.rfTracker = rfTracker;
328       this.pollQueueTime = pollQueueTime;
329     }

331     public String toString() {
332       StringBuilder sb = new StringBuilder();
333       sb.append("SolrCmdDistributor$Req: cmd=").append(String.valueOf(cmdString));
334       sb.append("; node=").append(String.valueOf(node));
335       return sb.toString();
336     }

338     public void trackRequestResult(HttpResponse resp, boolean success) {      
339       if (rfTracker != null) {
340         Integer rf = null;
341         if (resp != null) {
342           // need to parse out the rf from requests that were forwards to another leader
343           InputStream inputStream = null;
344           try {
345             inputStream = resp.getEntity().getContent();
346             BinaryResponseParser brp = new BinaryResponseParser();
347             NamedList<Object> nl= brp.processResponse(inputStream, null);
348             Object hdr = nl.get("responseHeader");
349             if (hdr != null && hdr instanceof NamedList) {
350               NamedList<Object> hdrList = (NamedList<Object>)hdr;
351               Object rfObj = hdrList.get(UpdateRequest.REPFACT);
352               if (rfObj != null && rfObj instanceof Integer) {
353                 rf = (Integer)rfObj;
354               }
355             }
356           } catch (Exception e) {
357             log.warn("Failed to parse response from "+node+" during replication factor accounting due to: "+e);
358           } finally {
359             if (inputStream != null) {
360               try {
361                 inputStream.close();
362               } catch (Exception ignore){}
363             }
364           }
365         }
366         rfTracker.trackRequestResult(node, success, rf);
367       }
368     }

491   // RetryNodes are used in the case of 'forward to leader' where we want
492   // to try the latest leader on a fail in the case the leader just went down.
493   public static class RetryNode extends StdNode {
494     
495     private ZkStateReader zkStateReader;
496     
497     public RetryNode(ZkCoreNodeProps nodeProps, ZkStateReader zkStateReader, String collection, String shardId) {
498       super(nodeProps, collection, shardId);
499       this.zkStateReader = zkStateReader;
500       this.collection = collection;
501       this.shardId = shardId;
502     }
503 
504     @Override
505     public boolean checkRetry() {
506       ZkCoreNodeProps leaderProps;
507       try {
508         leaderProps = new ZkCoreNodeProps(zkStateReader.getLeaderRetry(
509             collection, shardId));
510       } catch (InterruptedException e) {
511         Thread.currentThread().interrupt();
512         return false;
513       } catch (Exception e) {
514         // we retry with same info
515         log.warn(null, e);
516         return true;
517       }
518      
519       this.nodeProps = leaderProps;
520       
521       return true;
522     }
523 
524     @Override
525     public int hashCode() {
526       final int prime = 31;
527       int result = super.hashCode();
528       result = prime * result
529           + ((collection == null) ? 0 : collection.hashCode());
530       result = prime * result + ((shardId == null) ? 0 : shardId.hashCode());
531       return result;
532     }
533 
534     @Override
535     public boolean equals(Object obj) {
536       if (this == obj) return true;
537       if (!super.equals(obj)) return false;
538       if (getClass() != obj.getClass()) return false;
539       RetryNode other = (RetryNode) obj;
540       if (nodeProps.getCoreUrl() == null) {
541         if (other.nodeProps.getCoreUrl() != null) return false;
542       } else if (!nodeProps.getCoreUrl().equals(other.nodeProps.getCoreUrl())) return false;
543 
544       return true;
545     }
546   }
