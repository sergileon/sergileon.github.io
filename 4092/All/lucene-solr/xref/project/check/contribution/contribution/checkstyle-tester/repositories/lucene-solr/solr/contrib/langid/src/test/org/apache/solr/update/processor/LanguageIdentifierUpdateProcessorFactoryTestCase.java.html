85    @Test
86    public void testMapFieldName() throws Exception {
87      ModifiableSolrParams parameters = new ModifiableSolrParams();
88      parameters.add("langid.fl", "name");
89      parameters.add("langid.map.lcmap", "jp:s zh:cjk ko:cjk");
90      parameters.set("langid.enforceSchema", "false");
91      liProcessor = createLangIdProcessor(parameters);
92      
93      assertEquals("test_no", liProcessor.getMappedField("test", "no"));
94      assertEquals("test_en", liProcessor.getMappedField("test", "en"));
95      assertEquals("test_s", liProcessor.getMappedField("test", "jp"));
96      assertEquals("test_cjk", liProcessor.getMappedField("test", "zh"));
97      assertEquals("test_cjk", liProcessor.getMappedField("test", "ko"));
98  
99      // Test that enforceSchema correctly catches illegal field and returns null
100     parameters.set("langid.enforceSchema", "true");
101     liProcessor = createLangIdProcessor(parameters);
102     assertEquals(null, liProcessor.getMappedField("inputfield", "sv"));
103 
104     // Prove support for other mapping regex, still with enforceSchema=true
105     parameters.add("langid.map.pattern", "text_(.*?)_field");
106     parameters.add("langid.map.replace", "$1_{lang}_s");
107     liProcessor = createLangIdProcessor(parameters);
108     assertEquals("title_no_s", liProcessor.getMappedField("text_title_field", "no"));
109     assertEquals("body_sv_s", liProcessor.getMappedField("text_body_field", "sv"));
110   }

236   @Test
237   public void testResolveLanguage() throws Exception {
238     List<DetectedLanguage> langs;
239     ModifiableSolrParams parameters = new ModifiableSolrParams();
240     parameters.add("langid.fl", "text");
241     parameters.add("langid.langField", "language");
242     liProcessor = createLangIdProcessor(parameters);
243 
244     // No detected languages
245     langs = new ArrayList<>();
246     assertEquals("", liProcessor.resolveLanguage(langs, null));
247     assertEquals("fallback", liProcessor.resolveLanguage(langs, "fallback"));
248 
249     // One detected language
250     langs.add(new DetectedLanguage("one", 1.0));
251     assertEquals("one", liProcessor.resolveLanguage(langs, "fallback"));    
252 
253     // One detected language under default threshold
254     langs = new ArrayList<>();
255     langs.add(new DetectedLanguage("under", 0.1));
256     assertEquals("fallback", liProcessor.resolveLanguage(langs, "fallback"));    
257   }

288   protected abstract LanguageIdentifierUpdateProcessor createLangIdProcessor(ModifiableSolrParams parameters) throws Exception;
289 
290   protected void assertLang(String langCode, String... fieldsAndValues) throws Exception {
291     if(liProcessor == null)
292       throw new Exception("Processor must be initialized before calling assertLang()");
293     SolrInputDocument doc = sid(fieldsAndValues);
294     assertEquals(langCode, liProcessor.process(doc).getFieldValue(liProcessor.langField));
295   }
296   
297   private SolrInputDocument sid(String... fieldsAndValues) {
298     SolrInputDocument doc = new SolrInputDocument();
299     for (int i = 0; i < fieldsAndValues.length; i+=2) {
300       doc.addField(fieldsAndValues[i], fieldsAndValues[i+1]);
301     }
302     return doc;
303   }
