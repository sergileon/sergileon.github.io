20  import java.util.StringTokenizer;
21  
22  
23  /**
24   * Some useful constants.
25   **/
26  
27  public final class Constants {
28    private Constants() {}  // can't construct
29  
30    /** JVM vendor info. */
31    public static final String JVM_VENDOR = System.getProperty("java.vm.vendor");
32    public static final String JVM_VERSION = System.getProperty("java.vm.version");
33    public static final String JVM_NAME = System.getProperty("java.vm.name");
34    public static final String JVM_SPEC_VERSION = System.getProperty("java.specification.version");
35  
36    /** The value of <tt>System.getProperty("java.version")</tt>. **/
37    public static final String JAVA_VERSION = System.getProperty("java.version");
38   
39    /** The value of <tt>System.getProperty("os.name")</tt>. **/
40    public static final String OS_NAME = System.getProperty("os.name");
41    /** True iff running on Linux. */
42    public static final boolean LINUX = OS_NAME.startsWith("Linux");
43    /** True iff running on Windows. */
44    public static final boolean WINDOWS = OS_NAME.startsWith("Windows");
45    /** True iff running on SunOS. */
46    public static final boolean SUN_OS = OS_NAME.startsWith("SunOS");
47    /** True iff running on Mac OS X */
48    public static final boolean MAC_OS_X = OS_NAME.startsWith("Mac OS X");
49    /** True iff running on FreeBSD */
50    public static final boolean FREE_BSD = OS_NAME.startsWith("FreeBSD");
51  
52    public static final String OS_ARCH = System.getProperty("os.arch");
53    public static final String OS_VERSION = System.getProperty("os.version");
54    public static final String JAVA_VENDOR = System.getProperty("java.vendor");
55    
56    private static final int JVM_MAJOR_VERSION;
57    private static final int JVM_MINOR_VERSION;
58   
59    /** True iff running on a 64bit JVM */
60    public static final boolean JRE_IS_64BIT;
61    
62    static {
63      final StringTokenizer st = new StringTokenizer(JVM_SPEC_VERSION, ".");
64      JVM_MAJOR_VERSION = Integer.parseInt(st.nextToken());
65      if (st.hasMoreTokens()) {
66        JVM_MINOR_VERSION = Integer.parseInt(st.nextToken());
67      } else {
68        JVM_MINOR_VERSION = 0;
69      }
70      boolean is64Bit = false;
71      final String x = System.getProperty("sun.arch.data.model");
72      if (x != null) {
73        is64Bit = x.contains("64");
74      } else {
75        if (OS_ARCH != null && OS_ARCH.contains("64")) {
76          is64Bit = true;
77        } else {
78          is64Bit = false;
79        }
80      }
81      JRE_IS_64BIT = is64Bit;
82    }
83    
84    public static final boolean JRE_IS_MINIMUM_JAVA8 = JVM_MAJOR_VERSION > 1 || (JVM_MAJOR_VERSION == 1 && JVM_MINOR_VERSION >= 8);
85    public static final boolean JRE_IS_MINIMUM_JAVA9 = JVM_MAJOR_VERSION > 1 || (JVM_MAJOR_VERSION == 1 && JVM_MINOR_VERSION >= 9);
86  
87    /**
88     * This is the internal Lucene version, including bugfix versions, recorded into each segment.
89     * @deprecated Use {@link Version#LATEST}
90     */
91    @Deprecated
92    public static final String LUCENE_MAIN_VERSION = Version.LATEST.toString();
93    
94    /**
95     * Don't use this constant because the name is not self-describing!
96     * @deprecated Use {@link Version#LATEST}
97     */
98    @Deprecated
99    public static final String LUCENE_VERSION = Version.LATEST.toString();
100   
101 }
