45  /**
46   * A general-purpose Analyzer that can be created with a builder-style API.
47   * Under the hood it uses the factory classes {@link TokenizerFactory},
48   * {@link TokenFilterFactory}, and {@link CharFilterFactory}.
49   * <p>You can create an instance of this Analyzer using the builder:
50   * <pre class="prettyprint">
51   * Analyzer ana = CustomAnalyzer.builder(Paths.get(&quot;/path/to/config/dir&quot;))
52   *   .withTokenizer(&quot;standard&quot;)
53   *   .addTokenFilter(&quot;standard&quot;)
54   *   .addTokenFilter(&quot;lowercase&quot;)
55   *   .addTokenFilter(&quot;stop&quot;, &quot;ignoreCase&quot;, &quot;false&quot;, &quot;words&quot;, &quot;stopwords.txt&quot;, &quot;format&quot;, &quot;wordset&quot;)
56   *   .build();
57   * </pre>
58   * The parameters passed to components are also used by Apache Solr and are documented
59   * on their corresponding factory classes. Refer to documentation of subclasses
60   * of {@link TokenizerFactory}, {@link TokenFilterFactory}, and {@link CharFilterFactory}.
61   * <p>The list of names to be used for components can be looked up through:
62   * {@link TokenizerFactory#availableTokenizers()}, {@link TokenFilterFactory#availableTokenFilters()},
63   * and {@link CharFilterFactory#availableCharFilters()}.
64   */
65  public final class CustomAnalyzer extends Analyzer {
66    
67    /** Returns a builder for custom analyzers that loads all resources from classpath.
68     * All path names given must be absolute with package prefixes. */
69    public static Builder builder() {
70      return builder(new ClasspathResourceLoader());
71    }
72    
73    /** Returns a builder for custom analyzers that loads all resources from the given
74     * file system base directory. Place, e.g., stop word files there.
75     * Files that are not in the given directory are loaded from classpath. */
76    public static Builder builder(Path configDir) {
77      return builder(new FilesystemResourceLoader(configDir));
78    }
79    
80    /** Returns a builder for custom analyzers that loads all resources using the given {@link ResourceLoader}. */
81    public static Builder builder(ResourceLoader loader) {
82      return new Builder(loader);
83    }
84    
85    private final CharFilterFactory[] charFilters;
86    private final TokenizerFactory tokenizer;
87    private final TokenFilterFactory[] tokenFilters;
88    private final Integer posIncGap, offsetGap;
89  
90    CustomAnalyzer(Version defaultMatchVersion, CharFilterFactory[] charFilters, TokenizerFactory tokenizer, TokenFilterFactory[] tokenFilters, Integer posIncGap, Integer offsetGap) {
91      this.charFilters = charFilters;
92      this.tokenizer = tokenizer;
93      this.tokenFilters = tokenFilters;
94      this.posIncGap = posIncGap;
95      this.offsetGap = offsetGap;
96      if (defaultMatchVersion != null) {
97        setVersion(defaultMatchVersion);
98      }
99    }
100   
101   @Override
102   protected Reader initReader(String fieldName, Reader reader) {
103     for (final CharFilterFactory charFilter : charFilters) {
104       reader = charFilter.create(reader);
105     }
106     return reader;
107   }
108 
109   @Override
110   protected TokenStreamComponents createComponents(String fieldName) {
111     final Tokenizer tk = tokenizer.create();
112     TokenStream ts = tk;
113     for (final TokenFilterFactory filter : tokenFilters) {
114       ts = filter.create(ts);
115     }
116     return new TokenStreamComponents(tk, ts);
117   }
118   
119   @Override
120   public int getPositionIncrementGap(String fieldName) {
121     // use default from Analyzer base class if null
122     return (posIncGap == null) ? super.getPositionIncrementGap(fieldName) : posIncGap.intValue();
123   }
124   
125   @Override
126   public int getOffsetGap(String fieldName) {
127     // use default from Analyzer base class if null
128     return (offsetGap == null) ? super.getOffsetGap(fieldName) : offsetGap.intValue();
129   }
130   
131   /** Returns the list of char filters that are used in this analyzer. */
132   public List<CharFilterFactory> getCharFilterFactories() {
133     return Collections.unmodifiableList(Arrays.asList(charFilters));
134   }
135   
136   /** Returns the tokenizer that is used in this analyzer. */
137   public TokenizerFactory getTokenizerFactory() {
138     return tokenizer;
139   }
140   
141   /** Returns the list of token filters that are used in this analyzer. */
142   public List<TokenFilterFactory> getTokenFilterFactories() {
143     return Collections.unmodifiableList(Arrays.asList(tokenFilters));
144   }
145 
146   @Override
147   public String toString() {
148     final StringBuilder sb = new StringBuilder(getClass().getSimpleName()).append('(');
149     for (final CharFilterFactory filter : charFilters) {
150       sb.append(filter).append(',');
151     }
152     sb.append(tokenizer);
153     for (final TokenFilterFactory filter : tokenFilters) {
154       sb.append(',').append(filter);
155     }
156     return sb.append(')').toString();
157   }
158 
159   /** Builder for {@link CustomAnalyzer}.
160    * @see CustomAnalyzer#builder()
161    * @see CustomAnalyzer#builder(Path)
162    * @see CustomAnalyzer#builder(ResourceLoader)
163    */
164   public static final class Builder {
165     private final ResourceLoader loader;
166     private final SetOnce<Version> defaultMatchVersion = new SetOnce<>();
167     private final List<CharFilterFactory> charFilters = new ArrayList<>();
168     private final SetOnce<TokenizerFactory> tokenizer = new SetOnce<>();
169     private final List<TokenFilterFactory> tokenFilters = new ArrayList<>();
170     private final SetOnce<Integer> posIncGap = new SetOnce<>();
171     private final SetOnce<Integer> offsetGap = new SetOnce<>();
172     
173     private boolean componentsAdded = false;
174     
175     Builder(ResourceLoader loader) {
176       this.loader = loader;
177     }
178     
179     /** This match version is passed as default to all tokenizers or filters. It is used unless you
180      * pass the parameter {code luceneMatchVersion} explicitly. It defaults to undefined, so the
181      * underlying factory will (in most cases) use {@link Version#LATEST}. */
182     public Builder withDefaultMatchVersion(Version version) {
183       Objects.requireNonNull(version, "version may not be null");
184       if (componentsAdded) {
185         throw new IllegalStateException("You may only set the default match version before adding tokenizers, "+
186             "token filters, or char filters.");
187       }
188       this.defaultMatchVersion.set(version);
189       return this;
190     }
191     
192     /** Sets the position increment gap of the analyzer.
193      * The default is defined in the analyzer base class.
194      * @see Analyzer#getPositionIncrementGap(String)
195      */
196     public Builder withPositionIncrementGap(int posIncGap) {
197       if (posIncGap < 0) {
198         throw new IllegalArgumentException("posIncGap must be >= 0");
199       }
200       this.posIncGap.set(posIncGap);
201       return this;
202     }
203     
204     /** Sets the offset gap of the analyzer. The default is defined
205      * in the analyzer base class.
206      * @see Analyzer#getOffsetGap(String)
207      */
208     public Builder withOffsetGap(int offsetGap) {
209       if (offsetGap < 0) {
210         throw new IllegalArgumentException("offsetGap must be >= 0");
211       }
212       this.offsetGap.set(offsetGap);
213       return this;
214     }
215     
216     /** Uses the given tokenizer.
217      * @param name is used to look up the factory with {@link TokenizerFactory#forName(String, Map)}.
218      *  The list of possible names can be looked up with {@link TokenizerFactory#availableTokenizers()}.
219      * @param params a list of factory string params as key/value pairs.
220      *  The number of parameters must be an even number, as they are pairs.
221      */
222     public Builder withTokenizer(String name, String... params) throws IOException {
223       return withTokenizer(name, paramsToMap(params));
224     }
225     
226     /** Uses the given tokenizer.
227      * @param name is used to look up the factory with {@link TokenizerFactory#forName(String, Map)}.
228      *  The list of possible names can be looked up with {@link TokenizerFactory#availableTokenizers()}.
229      * @param params the map of parameters to be passed to factory. The map must be modifiable.
230      */
231     public Builder withTokenizer(String name, Map<String,String> params) throws IOException {
232       Objects.requireNonNull(name, "Tokenizer name may not be null");
233       tokenizer.set(applyResourceLoader(TokenizerFactory.forName(name, applyDefaultParams(params))));
234       componentsAdded = true;
235       return this;
236     }
237     
238     /** Adds the given token filter.
239      * @param name is used to look up the factory with {@link TokenFilterFactory#forName(String, Map)}.
240      *  The list of possible names can be looked up with {@link TokenFilterFactory#availableTokenFilters()}.
241      * @param params a list of factory string params as key/value pairs.
242      *  The number of parameters must be an even number, as they are pairs.
243      */
244     public Builder addTokenFilter(String name, String... params) throws IOException {
245       return addTokenFilter(name, paramsToMap(params));
246     }
247     
248     /** Adds the given token filter.
249      * @param name is used to look up the factory with {@link TokenFilterFactory#forName(String, Map)}.
250      *  The list of possible names can be looked up with {@link TokenFilterFactory#availableTokenFilters()}.
251      * @param params the map of parameters to be passed to factory. The map must be modifiable.
252      */
253     public Builder addTokenFilter(String name, Map<String,String> params) throws IOException {
254       Objects.requireNonNull(name, "TokenFilter name may not be null");
255       tokenFilters.add(applyResourceLoader(TokenFilterFactory.forName(name, applyDefaultParams(params))));
256       componentsAdded = true;
257       return this;
258     }
259     
260     /** Adds the given char filter.
261      * @param name is used to look up the factory with {@link CharFilterFactory#forName(String, Map)}.
262      *  The list of possible names can be looked up with {@link CharFilterFactory#availableCharFilters()}.
263      * @param params a list of factory string params as key/value pairs.
264      *  The number of parameters must be an even number, as they are pairs.
265      */
266     public Builder addCharFilter(String name, String... params) throws IOException {
267       return addCharFilter(name, paramsToMap(params));
268     }
269     
270     /** Adds the given char filter.
271      * @param name is used to look up the factory with {@link CharFilterFactory#forName(String, Map)}.
272      *  The list of possible names can be looked up with {@link CharFilterFactory#availableCharFilters()}.
273      * @param params the map of parameters to be passed to factory. The map must be modifiable.
274      */
275     public Builder addCharFilter(String name, Map<String,String> params) throws IOException {
276       Objects.requireNonNull(name, "CharFilter name may not be null");
277       charFilters.add(applyResourceLoader(CharFilterFactory.forName(name, applyDefaultParams(params))));
278       componentsAdded = true;
279       return this;
280     }
281     
282     /** Builds the analyzer. */
283     public CustomAnalyzer build() {
284       if (tokenizer.get() == null) {
285         throw new IllegalStateException("You have to set at least a tokenizer.");
286       }
287       return new CustomAnalyzer(
288         defaultMatchVersion.get(),
289         charFilters.toArray(new CharFilterFactory[charFilters.size()]),
290         tokenizer.get(), 
291         tokenFilters.toArray(new TokenFilterFactory[tokenFilters.size()]),
292         posIncGap.get(),
293         offsetGap.get()
294       );
295     }
296     
297     private Map<String,String> applyDefaultParams(Map<String,String> map) {
298       if (defaultMatchVersion.get() != null && !map.containsKey(AbstractAnalysisFactory.LUCENE_MATCH_VERSION_PARAM)) {
299         map.put(AbstractAnalysisFactory.LUCENE_MATCH_VERSION_PARAM, defaultMatchVersion.get().toString());
300       }
301       return map;
302     }
303     
304     private Map<String, String> paramsToMap(String... params) {
305       if (params.length % 2 != 0) {
306         throw new IllegalArgumentException("Key-value pairs expected, so the number of params must be even.");
307       }
308       final Map<String, String> map = new HashMap<>();
309       for (int i = 0; i < params.length; i += 2) {
310         Objects.requireNonNull(params[i], "Key of param may not be null.");
311         map.put(params[i], params[i + 1]);
312       }
313       return map;
314     }
315     
316     private <T> T applyResourceLoader(T factory) throws IOException {
317       if (factory instanceof ResourceLoaderAware) {
318         ((ResourceLoaderAware) factory).inform(loader);
319       }
320       return factory;
321     }
322   }
