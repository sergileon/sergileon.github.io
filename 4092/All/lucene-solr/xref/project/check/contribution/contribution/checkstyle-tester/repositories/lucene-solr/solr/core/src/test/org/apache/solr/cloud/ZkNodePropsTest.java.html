20  import java.io.IOException;
21  import java.util.HashMap;
22  import java.util.Map;
23  
24  import org.apache.solr.SolrTestCaseJ4;
25  import org.apache.solr.common.cloud.ZkNodeProps;
26  import org.apache.solr.common.util.Utils;
27  import org.junit.Test;
28  
29  public class ZkNodePropsTest extends SolrTestCaseJ4 {
30    @Test
31    public void testBasic() throws IOException {
32      
33      Map<String,Object> props = new HashMap<>();
34      props.put("prop1", "value1");
35      props.put("prop2", "value2");
36      props.put("prop3", "value3");
37      props.put("prop4", "value4");
38      props.put("prop5", "value5");
39      props.put("prop6", "value6");
40      
41      ZkNodeProps zkProps = new ZkNodeProps(props);
42      byte[] bytes = Utils.toJSON(zkProps);
43      
44      ZkNodeProps props2 = ZkNodeProps.load(bytes);
45      assertEquals("value1", props2.getStr("prop1"));
46      assertEquals("value2", props2.getStr("prop2"));
47      assertEquals("value3", props2.getStr("prop3"));
48      assertEquals("value4", props2.getStr("prop4"));
49      assertEquals("value5", props2.getStr("prop5"));
50      assertEquals("value6", props2.getStr("prop6"));
51    }
52  }
