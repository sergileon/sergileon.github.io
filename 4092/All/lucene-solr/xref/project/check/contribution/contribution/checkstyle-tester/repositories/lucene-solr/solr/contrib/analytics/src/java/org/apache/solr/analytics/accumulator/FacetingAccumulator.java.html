92    public FacetingAccumulator(SolrIndexSearcher searcher, DocSet docs, AnalyticsRequest request, SolrQueryRequest queryRequest) throws IOException {
93      // The parent Basic Accumulator keeps track of overall stats while
94      // the Faceting Accumulator only manages the facet stats
95      super(searcher, docs, request);
96      this.analyticsRequest = request;
97      this.queryRequest = queryRequest;
98      basicsAndFieldFacetsComputed = false;
99      List<FieldFacetRequest> fieldFreqs = request.getFieldFacets();
100     List<RangeFacetRequest> rangeFreqs = request.getRangeFacets();
101     List<QueryFacetRequest> queryFreqs = request.getQueryFacets();
102 
103     this.fieldFacetExpressions = new TreeMap<>();
104     this.rangeFacetExpressions = new LinkedHashMap<>(rangeFreqs.size());
105     this.queryFacetExpressions = new LinkedHashMap<>(queryFreqs.size());
106     this.fieldFacetCollectors = new LinkedHashMap<>(fieldFreqs.size());
107     this.rangeFacetCollectors = new LinkedHashMap<>(rangeFreqs.size());
108     this.queryFacetCollectors = new LinkedHashMap<>(queryFreqs.size());
109     this.facetAccumulators = new ArrayList<>();
110     this.hiddenFieldFacets = new HashSet<>();
111     
112     /**
113      * For each field facet request add a bucket to the {@link Expression} map and {@link StatsCollector} map.
114      * Field facets are computed during the initial collection of documents, therefore
115      * the FieldFacetAccumulators are created initially.
116      */
117     for( FieldFacetRequest freq : fieldFreqs ){
118       final FieldFacetRequest fr = (FieldFacetRequest) freq;
119       if (fr.isHidden()) {
120         hiddenFieldFacets.add(fr.getName());
121       }
122       final SchemaField ff = fr.getField();
123       final FieldFacetAccumulator facc = FieldFacetAccumulator.create(searcher, this, ff);
124       facetAccumulators.add(facc);
125       fieldFacetExpressions.put(freq.getName(), new TreeMap<String, Expression[]>() );
126       fieldFacetCollectors.put(freq.getName(), new TreeMap<String,StatsCollector[]>());
127     }
128     /**
129      * For each range and query facet request add a bucket to the corresponding
130      * {@link Expression} map and {@link StatsCollector} map.
131      * Range and Query Facets are computed in the post processing, so the accumulators
132      * are not created initially.
133      */
134     for( RangeFacetRequest freq : rangeFreqs ){
135       if( rangeFacets == null ) rangeFacets = new ArrayList<>();
136       rangeFacets.add(freq);
137       rangeFacetExpressions.put(freq.getName(), new LinkedHashMap<String,Expression[]>() );
138       rangeFacetCollectors.put(freq.getName(), new LinkedHashMap<String,StatsCollector[]>());
139     }
140     for( QueryFacetRequest freq : queryFreqs ){
141       if( queryFacets == null ) queryFacets = new ArrayList<>();
142       queryFacets.add(freq);
143       queryFacetExpressions.put(freq.getName(), new LinkedHashMap<String,Expression[]>() );
144       queryFacetCollectors.put(freq.getName(), new LinkedHashMap<String,StatsCollector[]>());
145     }
146     this.queryCount = 0l;
147   }

149   public static FacetingAccumulator create(SolrIndexSearcher searcher, DocSet docs, AnalyticsRequest request, SolrQueryRequest queryRequest) throws IOException {
150     return new FacetingAccumulator(searcher,docs,request,queryRequest);
151   }

218   /**
219    * Given a document, fieldFacet field and facetValue, adds the document to the
220    * {@link StatsCollector}s held in the bucket corresponding to the fieldFacet field and facetValue.
221    * Called during initial document collection.
222    */
223   @Override
224   public void collectField(int doc, String facetField, String facetValue) throws IOException {
225     Map<String,StatsCollector[]> map = fieldFacetCollectors.get(facetField);
226     StatsCollector[] statsCollectors = map.get(facetValue);
227     // If the facetValue has not been seen yet, a StatsCollector array is
228     // created and associated with that bucket.
229     if( statsCollectors == null ){
230       statsCollectors = statsCollectorArraySupplier.get();
231       map.put(facetValue,statsCollectors);
232       fieldFacetExpressions.get(facetField).put(facetValue,makeExpressions(statsCollectors));
233       for (StatsCollector statsCollector : statsCollectors) {
234         statsCollector.setNextReader(context);
235       }
236     }
237     for (StatsCollector statsCollector : statsCollectors) {
238       statsCollector.collect(doc);
239     }
240   }

290   /**
291    * A comparator to compare expression values for field facet sorting.
292    */
293   public static class EntryComparator implements Comparator<Entry<String,Expression[]>> {
294     private final Comparator<Expression> comp;
295     private final int comparatorExpressionPlace;
296    
297     public EntryComparator(Comparator<Expression> comp, int comparatorExpressionPlace) {
298       this.comp = comp;
299       this.comparatorExpressionPlace = comparatorExpressionPlace;
300     }
301 
302     @Override
303     public int compare(Entry<String,Expression[]> o1, Entry<String,Expression[]> o2) {
304       return comp.compare(o1.getValue()[comparatorExpressionPlace], o2.getValue()[comparatorExpressionPlace]);
305     }
306   }

350   /**
351    * Returns the value of an expression to use in a range or query facet.
352    * @param expressionName the name of the expression
353    * @param fieldFacet the facet field
354    * @param facetValue the facet value
355    * @return String String representation of pivot value
356    */
357   @SuppressWarnings({ "deprecation", "rawtypes" })
358   public String getResult(String expressionName, String fieldFacet, String facetValue) {
359     if (facetValue.contains(AnalyticsParams.RESULT) && !facetValue.contains(AnalyticsParams.QUERY_RESULT)) {
360       try {
361         String[] pivotStr = ExpressionFactory.getArguments(facetValue.substring(facetValue.indexOf('(')+1,facetValue.lastIndexOf(')')).trim());
362         if (pivotStr.length==1) {
363           facetValue = getResult(pivotStr[0]);
364         } else if (pivotStr.length==3) {
365           facetValue = getResult(pivotStr[0],pivotStr[1],pivotStr[2]);
366         } else {
367           throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+facetValue+" has an invalid amount of arguments.");
368         }
369       } catch (IndexOutOfBoundsException e) {
370         throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+facetValue+" is invalid. Lacks parentheses.",e);
371       }
372     } 
373     if (fieldFacetExpressions.get(fieldFacet)!=null) {
374       Expression[] facetExpressions = fieldFacetExpressions.get(fieldFacet).get(facetValue);
375       for (int count = 0; count < expressionNames.length; count++) {
376         if (expressionName.equals(expressionNames[count])) {
377           Comparable value = facetExpressions[count].getValue();
378           if (value.getClass().equals(Date.class)) {
379             return DateFormatUtil.formatExternal((Date)value);
380           } else {
381             return value.toString();
382           }
383         }
384       }
385     }
386     throw new SolrException(ErrorCode.BAD_REQUEST,"Field Facet Pivot expression "+expressionName+" not found.");
387   }

443   @Override
444   @SuppressWarnings("unchecked")
445   public NamedList<?> export() {
446     final NamedList<Object> base = (NamedList<Object>)super.export();
447     NamedList<NamedList<?>> facetList = new NamedList<>();
448     
449     // Add the field facet buckets to the output
450     base.add("fieldFacets",facetList);
451     for( FieldFacetRequest freq : request.getFieldFacets() ){
452       final String name = freq.getName();
453       if (hiddenFieldFacets.contains(name)) {
454         continue;
455       }
456       final Map<String,Expression[]> buckets = fieldFacetExpressions.get(name);
457       final NamedList<Object> bucketBase = new NamedList<>();
458 
459       Iterable<Entry<String,Expression[]>> iter = buckets.entrySet();
460       
461       final FieldFacetRequest fr = (FieldFacetRequest) freq;
462      
463       final FacetSortSpecification sort = fr.getSort();
464       final int limit = fr.getLimit();
465       final int offset = fr.getOffset();
466       final boolean showMissing = fr.showsMissing();
467       if (!showMissing) {
468         buckets.remove(MISSING_VALUE);
469       }
470       // Sorting the buckets if a sort specification is provided
471       if( sort != null && buckets.values().iterator().hasNext()){
472         int sortPlace = Arrays.binarySearch(expressionNames, sort.getStatistic());
473         final Expression first = buckets.values().iterator().next()[sortPlace];
474         final Comparator<Expression> comp = (Comparator<Expression>) first.comparator(sort.getDirection());
475         
476         final List<Entry<String,Expression[]>> sorted = new ArrayList<>(buckets.size());
477         Iterables.addAll(sorted, iter);
478         Collections.sort(sorted, new EntryComparator(comp,sortPlace));
479         iter = sorted;
480       }
481       // apply the limit
482       if( limit > AnalyticsContentHandler.DEFAULT_FACET_LIMIT ){
483         if( offset > 0 ){
484           iter = Iterables.skip(iter, offset);
485         }
486         iter = Iterables.limit(iter, limit);
487       }
488       
489       // Export each expression in the bucket.
490       for( Entry<String,Expression[]> bucket : iter ){
491         bucketBase.add(bucket.getKey(),export(bucket.getValue()));
492       }
493       
494       facetList.add(name, bucketBase);
495     }
496 
497     // Add the range facet buckets to the output
498     facetList = new NamedList<>();
499     base.add("rangeFacets",facetList);
500     for( RangeFacetRequest freq : request.getRangeFacets() ){
501       final String name = freq.getName();
502       final Map<String,Expression[]> buckets = rangeFacetExpressions.get(name);
503       final NamedList<Object> bucketBase = new NamedList<>();
504 
505       Iterable<Entry<String,Expression[]>> iter = buckets.entrySet();
506       
507       for( Entry<String,Expression[]> bucket : iter ){
508         bucketBase.add(bucket.getKey(),export(bucket.getValue()));
509       }
510       
511       facetList.add(name, bucketBase);
512     }
513     
514     // Add the query facet buckets to the output
515     facetList = new NamedList<>();
516     base.add("queryFacets",facetList);
517     for( QueryFacetRequest freq : request.getQueryFacets() ){
518       final String name = freq.getName();
519       final Map<String,Expression[]> buckets = queryFacetExpressions.get(name);
520       final NamedList<Object> bucketBase = new NamedList<>();
521 
522       Iterable<Entry<String,Expression[]>> iter = buckets.entrySet();
523       
524       for( Entry<String,Expression[]> bucket : iter ){
525         bucketBase.add(bucket.getKey(),export(bucket.getValue()));
526       }
527       
528       facetList.add(name, bucketBase);
529     }
530 
531     return base;
532   }

534   /**
535    * Exports a list of expressions as a NamedList
536    * @param expressionArr an array of expressions
537    * @return named list of expressions
538    */
539   public NamedList<?> export(Expression[] expressionArr) {
540     NamedList<Object> base = new NamedList<>();
541     for (int count = 0; count < expressionArr.length; count++) {
542       if (!hiddenExpressions.contains(expressionNames[count])) {
543         base.add(expressionNames[count], expressionArr[count].getValue());
544       }
545     }
546     return base;
547   }

549   /**
550    * Processes the query and range facets.
551    * Must be called if range and/or query facets are supported.
552    */
553   @Override
554   public void postProcess() throws IOException {
555     super.compute();
556     for( Map<String, StatsCollector[]> f : fieldFacetCollectors.values() ){
557       for( StatsCollector[] arr : f.values() ){
558         for( StatsCollector b : arr ){
559           b.compute();
560         }
561       }
562     }
563     basicsAndFieldFacetsComputed = true;
564     final Filter filter = docs.getTopFilter();
565     if( rangeFacets != null ){
566       processRangeFacets(filter); 
567     }
568     if( queryFacets != null ){
569       processQueryFacets(filter); 
570     }
571   }

573   /**
574    * Initiates the collecting of query facets
575    * @param filter the base filter to work against
576    * @throws IOException if searching failed
577    */
578   public void processQueryFacets(final Filter filter) throws IOException {
579     for( QueryFacetRequest qfr : queryFacets ){
580       for( String query : qfr.getQueries() ){
581         if (query.contains(AnalyticsParams.RESULT) && !query.contains(AnalyticsParams.QUERY_RESULT)) {
582           try {
583             String[] pivotStr = ExpressionFactory.getArguments(query.substring(query.indexOf('(')+1,query.lastIndexOf(')')).trim());
584             if (pivotStr.length==1) {
585               query = getResult(pivotStr[0]);
586             } else if (pivotStr.length==3) {
587               query = getResult(pivotStr[0],pivotStr[1],pivotStr[2]);
588             } else {
589               throw new SolrException(ErrorCode.BAD_REQUEST,"Result request "+query+" has an invalid amount of arguments.");
590             }
591           } catch (IndexOutOfBoundsException e) {
592             throw new SolrException(ErrorCode.BAD_REQUEST,"Result request "+query+" is invalid. Lacks parentheses.",e);
593           }
594         } else if (query.contains(AnalyticsParams.QUERY_RESULT)) {
595           try {
596             String[] pivotStr = ExpressionFactory.getArguments(query.substring(query.indexOf('(')+1,query.lastIndexOf(')')).trim());
597             if (pivotStr.length==3) {
598               query = getQueryResult(qfr.getName(),pivotStr[0],pivotStr[1],pivotStr[2]);
599             } else {
600               throw new SolrException(ErrorCode.BAD_REQUEST,"Result request "+query+" has an invalid amount of arguments.");
601             }
602           } catch (IndexOutOfBoundsException e) {
603             throw new SolrException(ErrorCode.BAD_REQUEST,"Result request "+query+" is invalid. Lacks parentheses.",e);
604           }
605         }
606         QueryFacetAccumulator qAcc = new QueryFacetAccumulator(this,qfr.getName(),query);
607         final Query q;
608         try {
609           q = QParser.getParser(query, null, queryRequest).getQuery();
610         } catch( SyntaxError e ){
611           throw new SolrException(ErrorCode.BAD_REQUEST,"Invalid query '"+query+"'",e);
612         }
613         // The searcher sends docIds to the QueryFacetAccumulator which forwards
614         // them to <code>collectQuery()</code> in this class for collection.
615         searcher.search(new FilteredQuery(q, filter), qAcc);
616         computeQueryFacet(qfr.getName());
617         queryCount++;
618       }
619     }
620   }
621   
622   @Override
623   public long getNumQueries() {
624     return queryCount;
625   }

627   /**
628    * Initiates the collecting of range facets
629    * @param filter the base filter to use
630    * @throws IOException if searching fails
631    */
632   public void processRangeFacets(final Filter filter) throws IOException {
633     for( RangeFacetRequest rfr : rangeFacets ){
634       String[] pivotStr;
635       String start = rfr.getStart();
636       if (start.contains(AnalyticsParams.QUERY_RESULT)) {
637         throw new SolrException(ErrorCode.BAD_REQUEST,"Query result requests can not be used in Range Facets");
638       } else if (start.contains(AnalyticsParams.RESULT)) {
639         try {
640           pivotStr = ExpressionFactory.getArguments(start.substring(start.indexOf('(')+1,start.indexOf(')')).trim());
641           if (pivotStr.length==1) {
642             rfr.setStart(getResult(pivotStr[0]));
643           } else if (pivotStr.length==3) {
644             rfr.setStart(getResult(pivotStr[0],pivotStr[1],pivotStr[2]));
645           } else {
646             throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+start+" has an invalid amount of arguments.");
647           }
648         } catch (IndexOutOfBoundsException e) {
649           throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+start+" is invalid. Lacks parentheses.",e);
650         }
651       }
652       String end = rfr.getEnd();
653       if (end.contains(AnalyticsParams.QUERY_RESULT)) {
654         throw new SolrException(ErrorCode.BAD_REQUEST, "Query result requests can not be used in Range Facets");
655       } else if (end.contains(AnalyticsParams.RESULT)) {
656         try {
657           pivotStr = ExpressionFactory.getArguments(end.substring(end.indexOf('(')+1,end.indexOf(')')).trim());
658           if (pivotStr.length==1) {
659             rfr.setEnd(getResult(pivotStr[0]));
660           } else if (pivotStr.length==3) {
661             rfr.setEnd(getResult(pivotStr[0],pivotStr[1],pivotStr[2]));
662           } else {
663             throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+end+" has an invalid amount of arguments.");
664           }
665         } catch (IndexOutOfBoundsException e) {
666           throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+end+" is invalid. Lacks parentheses.",e);
667         }
668       }
669       String[] gaps = rfr.getGaps();
670       for (int count = 0; count<gaps.length; count++){
671         String gap = gaps[count];
672         if (gap.contains(AnalyticsParams.QUERY_RESULT)) {
673           throw new SolrException(ErrorCode.BAD_REQUEST, "Query result requests can not be used in Range Facets");
674         } else if (gap.contains(AnalyticsParams.RESULT)) {
675           try {
676             pivotStr = ExpressionFactory.getArguments(gap.substring(gap.indexOf('(')+1,gap.indexOf(')')).trim());
677             if (pivotStr.length==1) {
678               gaps[count]=getResult(pivotStr[0]);
679             } else if (pivotStr.length==3) {
680               gaps[count]=getResult(pivotStr[0],pivotStr[1],pivotStr[2]);
681             } else {
682               throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+gap+" has an invalid amount of arguments.");
683             }
684           } catch (IndexOutOfBoundsException e) {
685             throw new SolrException(ErrorCode.BAD_REQUEST, "Result request "+gap+" is invalid. Lacks parentheses.",e);
686           }
687         }
688       }
689       // Computes the end points of the ranges in the rangeFacet
690       final RangeEndpointCalculator<? extends Comparable<?>> rec = RangeEndpointCalculator.create(rfr);
691       final SchemaField sf = rfr.getField();
692       
693       // Create a rangeFacetAccumulator for each range and 
694       // collect the documents for that range.
695       for( FacetRange range : rec.getRanges() ){
696         final String upper;
697         final String lower;
698         String facetValue = "";
699         if( range.lower == null ){
700           facetValue = "(*";
701           lower = null;
702         } else {
703           lower = range.lower;
704           facetValue = ((range.includeLower)?"[":"(") + range.lower;
705         }
706         facetValue+=" TO ";
707         if( range.upper == null ){
708           upper = null;
709           facetValue += "*)";
710         } else {
711           upper = range.upper;
712           facetValue += range.upper + ((range.includeUpper)?"]":")");
713         }
714         
715         Query q = sf.getType().getRangeQuery(null, sf, lower, upper, range.includeLower,range.includeUpper);
716         RangeFacetAccumulator rAcc = new RangeFacetAccumulator(this,rfr.getName(),facetValue);
717         // The searcher sends docIds to the RangeFacetAccumulator which forwards
718         // them to <code>collectRange()</code> in this class for collection.
719         searcher.search(new FilteredQuery(q, filter), rAcc);
720         computeRangeFacet(sf.getName());
721       }
722     }
723   }
