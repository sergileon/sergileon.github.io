27  /**
28   * This tree structure stores the hyphenation patterns in an efficient way for
29   * fast lookup. It provides the provides the method to hyphenate a word.
30   * 
31   * This class has been taken from the Apache FOP project (http://xmlgraphics.apache.org/fop/). They have been slightly modified. 
32   */
33  public class HyphenationTree extends TernaryTree implements PatternConsumer {
34  
35    /**
36     * value space: stores the interletter values
37     */
38    protected ByteVector vspace;
39  
40    /**
41     * This map stores hyphenation exceptions
42     */
43    protected HashMap<String,ArrayList<Object>> stoplist;
44  
45    /**
46     * This map stores the character classes
47     */
48    protected TernaryTree classmap;
49  
50    /**
51     * Temporary map to store interletter values on pattern loading.
52     */
53    private transient TernaryTree ivalues;

55    public HyphenationTree() {
56      stoplist = new HashMap<>(23); // usually a small table
57      classmap = new TernaryTree();
58      vspace = new ByteVector();
59      vspace.alloc(1); // this reserves index 0, which we don't use
60    }

62    /**
63     * Packs the values by storing them in 4 bits, two values into a byte Values
64     * range is from 0 to 9. We use zero as terminator, so we'll add 1 to the
65     * value.
66     * 
67     * @param values a string of digits from '0' to '9' representing the
68     *        interletter values.
69     * @return the index into the vspace array where the packed values are stored.
70     */
71    protected int packValues(String values) {
72      int i, n = values.length();
73      int m = (n & 1) == 1 ? (n >> 1) + 2 : (n >> 1) + 1;
74      int offset = vspace.alloc(m);
75      byte[] va = vspace.getArray();
76      for (i = 0; i < n; i++) {
77        int j = i >> 1;
78        byte v = (byte) ((values.charAt(i) - '0' + 1) & 0x0f);
79        if ((i & 1) == 1) {
80          va[j + offset] = (byte) (va[j + offset] | v);
81        } else {
82          va[j + offset] = (byte) (v << 4); // big endian
83        }
84      }
85      va[m - 1 + offset] = 0; // terminator
86      return offset;
87    }
88  
89    protected String unpackValues(int k) {
90      StringBuilder buf = new StringBuilder();
91      byte v = vspace.get(k++);
92      while (v != 0) {
93        char c = (char) ((v >>> 4) - 1 + '0');
94        buf.append(c);
95        c = (char) (v & 0x0f);
96        if (c == 0) {
97          break;
98        }
99        c = (char) (c - 1 + '0');
100       buf.append(c);
101       v = vspace.get(k++);
102     }
103     return buf.toString();
104   }

106   /**
107    * Read hyphenation patterns from an XML file.
108    * 
109    * @param source the InputSource for the file
110    * @throws IOException In case the parsing fails
111    */
112   public void loadPatterns(InputSource source) throws IOException {
113     PatternParser pp = new PatternParser(this);
114     ivalues = new TernaryTree();
115 
116     pp.parse(source);
117 
118     // patterns/values should be now in the tree
119     // let's optimize a bit
120     trimToSize();
121     vspace.trimToSize();
122     classmap.trimToSize();
123 
124     // get rid of the auxiliary map
125     ivalues = null;
126   }

128   public String findPattern(String pat) {
129     int k = super.find(pat);
130     if (k >= 0) {
131       return unpackValues(k);
132     }
133     return "";
134   }

136   /**
137    * String compare, returns 0 if equal or t is a substring of s
138    */
139   protected int hstrcmp(char[] s, int si, char[] t, int ti) {
140     for (; s[si] == t[ti]; si++, ti++) {
141       if (s[si] == 0) {
142         return 0;
143       }
144     }
145     if (t[ti] == 0) {
146       return 0;
147     }
148     return s[si] - t[ti];
149   }

151   protected byte[] getValues(int k) {
152     StringBuilder buf = new StringBuilder();
153     byte v = vspace.get(k++);
154     while (v != 0) {
155       char c = (char) ((v >>> 4) - 1);
156       buf.append(c);
157       c = (char) (v & 0x0f);
158       if (c == 0) {
159         break;
160       }
161       c = (char) (c - 1);
162       buf.append(c);
163       v = vspace.get(k++);
164     }
165     byte[] res = new byte[buf.length()];
166     for (int i = 0; i < res.length; i++) {
167       res[i] = (byte) buf.charAt(i);
168     }
169     return res;
170   }

172   /**
173    * <p>
174    * Search for all possible partial matches of word starting at index an update
175    * interletter values. In other words, it does something like:
176    * </p>
177    * <code>
178    * for(i=0; i&lt;patterns.length; i++) {
179    * if ( word.substring(index).startsWidth(patterns[i]) )
180    * update_interletter_values(patterns[i]);
181    * }
182    * </code>
183    * <p>
184    * But it is done in an efficient way since the patterns are stored in a
185    * ternary tree. In fact, this is the whole purpose of having the tree: doing
186    * this search without having to test every single pattern. The number of
187    * patterns for languages such as English range from 4000 to 10000. Thus,
188    * doing thousands of string comparisons for each word to hyphenate would be
189    * really slow without the tree. The tradeoff is memory, but using a ternary
190    * tree instead of a trie, almost halves the the memory used by Lout or TeX.
191    * It's also faster than using a hash table
192    * </p>
193    * 
194    * @param word null terminated word to match
195    * @param index start index from word
196    * @param il interletter values array to update
197    */
198   protected void searchPatterns(char[] word, int index, byte[] il) {
199     byte[] values;
200     int i = index;
201     char p, q;
202     char sp = word[i];
203     p = root;
204 
205     while (p > 0 && p < sc.length) {
206       if (sc[p] == 0xFFFF) {
207         if (hstrcmp(word, i, kv.getArray(), lo[p]) == 0) {
208           values = getValues(eq[p]); // data pointer is in eq[]
209           int j = index;
210           for (int k = 0; k < values.length; k++) {
211             if (j < il.length && values[k] > il[j]) {
212               il[j] = values[k];
213             }
214             j++;
215           }
216         }
217         return;
218       }
219       int d = sp - sc[p];
220       if (d == 0) {
221         if (sp == 0) {
222           break;
223         }
224         sp = word[++i];
225         p = eq[p];
226         q = p;
227 
228         // look for a pattern ending at this position by searching for
229         // the null char ( splitchar == 0 )
230         while (q > 0 && q < sc.length) {
231           if (sc[q] == 0xFFFF) { // stop at compressed branch
232             break;
233           }
234           if (sc[q] == 0) {
235             values = getValues(eq[q]);
236             int j = index;
237             for (int k = 0; k < values.length; k++) {
238               if (j < il.length && values[k] > il[j]) {
239                 il[j] = values[k];
240               }
241               j++;
242             }
243             break;
244           } else {
245             q = lo[q];
246 
247             /**
248              * actually the code should be: q = sc[q] < 0 ? hi[q] : lo[q]; but
249              * java chars are unsigned
250              */
251           }
252         }
253       } else {
254         p = d < 0 ? lo[p] : hi[p];
255       }
256     }
257   }

259   /**
260    * Hyphenate word and return a Hyphenation object.
261    * 
262    * @param word the word to be hyphenated
263    * @param remainCharCount Minimum number of characters allowed before the
264    *        hyphenation point.
265    * @param pushCharCount Minimum number of characters allowed after the
266    *        hyphenation point.
267    * @return a {@link Hyphenation Hyphenation} object representing the
268    *         hyphenated word or null if word is not hyphenated.
269    */
270   public Hyphenation hyphenate(String word, int remainCharCount,
271       int pushCharCount) {
272     char[] w = word.toCharArray();
273     return hyphenate(w, 0, w.length, remainCharCount, pushCharCount);
274   }

276   /**
277    * w = "****nnllllllnnn*****", where n is a non-letter, l is a letter, all n
278    * may be absent, the first n is at offset, the first l is at offset +
279    * iIgnoreAtBeginning; word = ".llllll.'\0'***", where all l in w are copied
280    * into word. In the first part of the routine len = w.length, in the second
281    * part of the routine len = word.length. Three indices are used: index(w),
282    * the index in w, index(word), the index in word, letterindex(word), the
283    * index in the letter part of word. The following relations exist: index(w) =
284    * offset + i - 1 index(word) = i - iIgnoreAtBeginning letterindex(word) =
285    * index(word) - 1 (see first loop). It follows that: index(w) - index(word) =
286    * offset - 1 + iIgnoreAtBeginning index(w) = letterindex(word) + offset +
287    * iIgnoreAtBeginning
288    */
289 
290   /**
291    * Hyphenate word and return an array of hyphenation points.
292    * 
293    * @param w char array that contains the word
294    * @param offset Offset to first character in word
295    * @param len Length of word
296    * @param remainCharCount Minimum number of characters allowed before the
297    *        hyphenation point.
298    * @param pushCharCount Minimum number of characters allowed after the
299    *        hyphenation point.
300    * @return a {@link Hyphenation Hyphenation} object representing the
301    *         hyphenated word or null if word is not hyphenated.
302    */
303   public Hyphenation hyphenate(char[] w, int offset, int len,
304       int remainCharCount, int pushCharCount) {
305     int i;
306     char[] word = new char[len + 3];
307 
308     // normalize word
309     char[] c = new char[2];
310     int iIgnoreAtBeginning = 0;
311     int iLength = len;
312     boolean bEndOfLetters = false;
313     for (i = 1; i <= len; i++) {
314       c[0] = w[offset + i - 1];
315       int nc = classmap.find(c, 0);
316       if (nc < 0) { // found a non-letter character ...
317         if (i == (1 + iIgnoreAtBeginning)) {
318           // ... before any letter character
319           iIgnoreAtBeginning++;
320         } else {
321           // ... after a letter character
322           bEndOfLetters = true;
323         }
324         iLength--;
325       } else {
326         if (!bEndOfLetters) {
327           word[i - iIgnoreAtBeginning] = (char) nc;
328         } else {
329           return null;
330         }
331       }
332     }
333     len = iLength;
334     if (len < (remainCharCount + pushCharCount)) {
335       // word is too short to be hyphenated
336       return null;
337     }
338     int[] result = new int[len + 1];
339     int k = 0;
340 
341     // check exception list first
342     String sw = new String(word, 1, len);
343     if (stoplist.containsKey(sw)) {
344       // assume only simple hyphens (Hyphen.pre="-", Hyphen.post = Hyphen.no =
345       // null)
346       ArrayList<Object> hw = stoplist.get(sw);
347       int j = 0;
348       for (i = 0; i < hw.size(); i++) {
349         Object o = hw.get(i);
350         // j = index(sw) = letterindex(word)?
351         // result[k] = corresponding index(w)
352         if (o instanceof String) {
353           j += ((String) o).length();
354           if (j >= remainCharCount && j < (len - pushCharCount)) {
355             result[k++] = j + iIgnoreAtBeginning;
356           }
357         }
358       }
359     } else {
360       // use algorithm to get hyphenation points
361       word[0] = '.'; // word start marker
362       word[len + 1] = '.'; // word end marker
363       word[len + 2] = 0; // null terminated
364       byte[] il = new byte[len + 3]; // initialized to zero
365       for (i = 0; i < len + 1; i++) {
366         searchPatterns(word, i, il);
367       }
368 
369       // hyphenation points are located where interletter value is odd
370       // i is letterindex(word),
371       // i + 1 is index(word),
372       // result[k] = corresponding index(w)
373       for (i = 0; i < len; i++) {
374         if (((il[i + 1] & 1) == 1) && i >= remainCharCount
375             && i <= (len - pushCharCount)) {
376           result[k++] = i + iIgnoreAtBeginning;
377         }
378       }
379     }
380 
381     if (k > 0) {
382       // trim result array
383       int[] res = new int[k+2];
384       System.arraycopy(result, 0, res, 1, k);
385       // We add the synthetical hyphenation points
386       // at the beginning and end of the word
387       res[0]=0;
388       res[k+1]=len;
389       return new Hyphenation(res);
390     } else {
391       return null;
392     }
393   }

395   /**
396    * Add a character class to the tree. It is used by
397    * {@link PatternParser PatternParser} as callback to add character classes.
398    * Character classes define the valid word characters for hyphenation. If a
399    * word contains a character not defined in any of the classes, it is not
400    * hyphenated. It also defines a way to normalize the characters in order to
401    * compare them with the stored patterns. Usually pattern files use only lower
402    * case characters, in this case a class for letter 'a', for example, should
403    * be defined as "aA", the first character being the normalization char.
404    */
405   @Override
406   public void addClass(String chargroup) {
407     if (chargroup.length() > 0) {
408       char equivChar = chargroup.charAt(0);
409       char[] key = new char[2];
410       key[1] = 0;
411       for (int i = 0; i < chargroup.length(); i++) {
412         key[0] = chargroup.charAt(i);
413         classmap.insert(key, 0, equivChar);
414       }
415     }
416   }

418   /**
419    * Add an exception to the tree. It is used by
420    * {@link PatternParser PatternParser} class as callback to store the
421    * hyphenation exceptions.
422    * 
423    * @param word normalized word
424    * @param hyphenatedword a vector of alternating strings and
425    *        {@link Hyphen hyphen} objects.
426    */
427   @Override
428   public void addException(String word, ArrayList<Object> hyphenatedword) {
429     stoplist.put(word, hyphenatedword);
430   }

432   /**
433    * Add a pattern to the tree. Mainly, to be used by
434    * {@link PatternParser PatternParser} class as callback to add a pattern to
435    * the tree.
436    * 
437    * @param pattern the hyphenation pattern
438    * @param ivalue interletter weight values indicating the desirability and
439    *        priority of hyphenating at a given point within the pattern. It
440    *        should contain only digit characters. (i.e. '0' to '9').
441    */
442   @Override
443   public void addPattern(String pattern, String ivalue) {
444     int k = ivalues.find(ivalue);
445     if (k <= 0) {
446       k = packValues(ivalue);
447       ivalues.insert(ivalue, (char) k);
448     }
449     insert(pattern, (char) k);
450   }
