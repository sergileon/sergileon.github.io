884   /**
885    * Refills the input buffer.
886    *
887    * @return      <code>false</code>, iff there was new input.
888    * 
889    * @exception   java.io.IOException  if any I/O-Error occurs
890    */
891   private boolean zzRefill() throws java.io.IOException {
892 
893     /* first: make room (if you can) */
894     if (zzStartRead > 0) {
895       System.arraycopy(zzBuffer, zzStartRead,
896                        zzBuffer, 0,
897                        zzEndRead-zzStartRead);
898 
899       /* translate stored positions */
900       zzEndRead-= zzStartRead;
901       zzCurrentPos-= zzStartRead;
902       zzMarkedPos-= zzStartRead;
903       zzStartRead = 0;
904     }
905 
906     /* is the buffer big enough? */
907     if (zzCurrentPos >= zzBuffer.length) {
908       /* if not: blow it up */
909       char newBuffer[] = new char[zzCurrentPos*2];
910       System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
911       zzBuffer = newBuffer;
912     }
913 
914     /* finally: fill the buffer with new input */
915     int numRead = zzReader.read(zzBuffer, zzEndRead,
916                                             zzBuffer.length-zzEndRead);
917 
918     if (numRead > 0) {
919       zzEndRead+= numRead;
920       return false;
921     }
922     // unlikely but not impossible: read 0 characters, but not at end of stream    
923     if (numRead == 0) {
924       int c = zzReader.read();
925       if (c == -1) {
926         return true;
927       } else {
928         zzBuffer[zzEndRead++] = (char) c;
929         return false;
930       }     
931     }
932 
933     // numRead < 0
934     return true;
935   }

1069   /**
1070    * Resumes scanning until the next regular expression is matched,
1071    * the end of input is encountered or an I/O-Error occurs.
1072    *
1073    * @return      the next token
1074    * @exception   java.io.IOException  if any I/O-Error occurs
1075    */
1076   public int getNextToken() throws java.io.IOException {
1077     int zzInput;
1078     int zzAction;
1079 
1080     // cached fields:
1081     int zzCurrentPosL;
1082     int zzMarkedPosL;
1083     int zzEndReadL = zzEndRead;
1084     char [] zzBufferL = zzBuffer;
1085     char [] zzCMapL = ZZ_CMAP;
1086 
1087     int [] zzTransL = ZZ_TRANS;
1088     int [] zzRowMapL = ZZ_ROWMAP;
1089     int [] zzAttrL = ZZ_ATTRIBUTE;
1090 
1091     while (true) {
1092       zzMarkedPosL = zzMarkedPos;
1093 
1094       yychar+= zzMarkedPosL-zzStartRead;
1095 
1096       zzAction = -1;
1097 
1098       zzCurrentPosL = zzCurrentPos = zzStartRead = zzMarkedPosL;
1099   
1100       zzState = ZZ_LEXSTATE[zzLexicalState];
1101 
1102       // set up zzAction for empty match case:
1103       int zzAttributes = zzAttrL[zzState];
1104       if ( (zzAttributes & 1) == 1 ) {
1105         zzAction = zzState;
1106       }
1107 
1108 
1109       zzForAction: {
1110         while (true) {
1111     
1112           if (zzCurrentPosL < zzEndReadL)
1113             zzInput = zzBufferL[zzCurrentPosL++];
1114           else if (zzAtEOF) {
1115             zzInput = YYEOF;
1116             break zzForAction;
1117           }
1118           else {
1119             // store back cached positions
1120             zzCurrentPos  = zzCurrentPosL;
1121             zzMarkedPos   = zzMarkedPosL;
1122             boolean eof = zzRefill();
1123             // get translated positions and possibly new buffer
1124             zzCurrentPosL  = zzCurrentPos;
1125             zzMarkedPosL   = zzMarkedPos;
1126             zzBufferL      = zzBuffer;
1127             zzEndReadL     = zzEndRead;
1128             if (eof) {
1129               zzInput = YYEOF;
1130               break zzForAction;
1131             }
1132             else {
1133               zzInput = zzBufferL[zzCurrentPosL++];
1134             }
1135           }
1136           int zzNext = zzTransL[ zzRowMapL[zzState] + zzCMapL[zzInput] ];
1137           if (zzNext == -1) break zzForAction;
1138           zzState = zzNext;
1139 
1140           zzAttributes = zzAttrL[zzState];
1141           if ( (zzAttributes & 1) == 1 ) {
1142             zzAction = zzState;
1143             zzMarkedPosL = zzCurrentPosL;
1144             if ( (zzAttributes & 8) == 8 ) break zzForAction;
1145           }
1146 
1147         }
1148       }
1149 
1150       // store back cached position
1151       zzMarkedPos = zzMarkedPosL;
1152 
1153       switch (zzAction < 0 ? zzAction : ZZ_ACTION[zzAction]) {
1154         case 1: 
1155           { /* Break so we don't hit fall-through warning: */ break; /* Not numeric, word, ideographic, hiragana, or SE Asian -- ignore it. */
1156           }
1157         case 9: break;
1158         case 2: 
1159           { return WORD_TYPE;
1160           }
1161         case 10: break;
1162         case 3: 
1163           { return NUMERIC_TYPE;
1164           }
1165         case 11: break;
1166         case 4: 
1167           { return KATAKANA_TYPE;
1168           }
1169         case 12: break;
1170         case 5: 
1171           { return SOUTH_EAST_ASIAN_TYPE;
1172           }
1173         case 13: break;
1174         case 6: 
1175           { return IDEOGRAPHIC_TYPE;
1176           }
1177         case 14: break;
1178         case 7: 
1179           { return HIRAGANA_TYPE;
1180           }
1181         case 15: break;
1182         case 8: 
1183           { return HANGUL_TYPE;
1184           }
1185         case 16: break;
1186         default: 
1187           if (zzInput == YYEOF && zzStartRead == zzCurrentPos) {
1188             zzAtEOF = true;
1189               {
1190                 return YYEOF;
1191               }
1192           } 
1193           else {
1194             zzScanError(ZZ_NO_MATCH);
1195           }
1196       }
1197     }
