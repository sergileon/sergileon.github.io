64  /**
65   * ConcurrentUpdateSolrClient buffers all added documents and writes
66   * them into open HTTP connections. This class is thread safe.
67   * 
68   * Params from {@link UpdateRequest} are converted to http request
69   * parameters. When params change between UpdateRequests a new HTTP
70   * request is started.
71   * 
72   * Although any SolrClient request can be made with this implementation, it is
73   * only recommended to use ConcurrentUpdateSolrClient with /update
74   * requests. The class {@link HttpSolrClient} is better suited for the
75   * query interface.
76   */
77  public class ConcurrentUpdateSolrClient extends SolrClient {
78    private static final long serialVersionUID = 1L;
79    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
80    private HttpSolrClient client;
81    final BlockingQueue<UpdateRequest> queue;
82    final ExecutorService scheduler;
83    final Queue<Runner> runners;
84    volatile CountDownLatch lock = null; // used to block everything
85    final int threadCount;
86    boolean shutdownExecutor = false;
87    int pollQueueTime = 250;
88    private final boolean streamDeletes;

121   /**
122    * Uses the supplied HttpClient to send documents to the Solr server.
123    */
124   public ConcurrentUpdateSolrClient(String solrServerUrl,
125                                     HttpClient client, int queueSize, int threadCount, ExecutorService es, boolean streamDeletes) {
126     this.client = new HttpSolrClient(solrServerUrl, client);
127     this.client.setFollowRedirects(false);
128     queue = new LinkedBlockingQueue<>(queueSize);
129     this.threadCount = threadCount;
130     runners = new LinkedList<>();
131     scheduler = es;
132     this.streamDeletes = streamDeletes;
133   }

147   /**
148    * Opens a connection and sends everything...
149    */
150   class Runner implements Runnable {
151     @Override
152     public void run() {
153       log.debug("starting runner: {}", this);
154 
155       // This loop is so we can continue if an element was added to the queue after the last runner exited.
156       for (;;) {
157 
158         try {
159 
160           sendUpdateStream();
161 
162         } catch (Throwable e) {
163           if (e instanceof OutOfMemoryError) {
164             throw (OutOfMemoryError) e;
165           }
166           handleError(e);
167         } finally {
168 
169           synchronized (runners) {
170             // check to see if anything else was added to the queue
171             if (runners.size() == 1 && !queue.isEmpty() && !scheduler.isShutdown()) {
172               // If there is something else to process, keep last runner alive by staying in the loop.
173             } else {
174               runners.remove(this);
175               if (runners.isEmpty()) {
176                 // notify anyone waiting in blockUntilFinished
177                 runners.notifyAll();
178               }
179               break;
180             }
181           }
182 
183         }
184       }
185 
186       log.debug("finished: {}", this);
187     }
188 
189     //
190     // Pull from the queue multiple times and streams over a single connection.
191     // Exits on exception, interruption, or an empty queue to pull from.
192     //
193     void sendUpdateStream() throws Exception {
194       while (!queue.isEmpty()) {
195         HttpPost method = null;
196         HttpResponse response = null;
197 
198         try {
199           final UpdateRequest updateRequest =
200               queue.poll(pollQueueTime, TimeUnit.MILLISECONDS);
201           if (updateRequest == null)
202             break;
203 
204           String contentType = client.requestWriter.getUpdateContentType();
205           final boolean isXml = ClientUtils.TEXT_XML.equals(contentType);
206 
207           final ModifiableSolrParams origParams = new ModifiableSolrParams(updateRequest.getParams());
208 
209           EntityTemplate template = new EntityTemplate(new ContentProducer() {
210 
211             @Override
212             public void writeTo(OutputStream out) throws IOException {
213               try {
214                 if (isXml) {
215                   out.write("<stream>".getBytes(StandardCharsets.UTF_8)); // can be anything
216                 }
217                 UpdateRequest req = updateRequest;
218                 while (req != null) {
219                   SolrParams currentParams = new ModifiableSolrParams(req.getParams());
220                   if (!origParams.toNamedList().equals(currentParams.toNamedList())) {
221                     queue.add(req); // params are different, push back to queue
222                     break;
223                   }
224 
225                   client.requestWriter.write(req, out);
226                   if (isXml) {
227                     // check for commit or optimize
228                     SolrParams params = req.getParams();
229                     if (params != null) {
230                       String fmt = null;
231                       if (params.getBool(UpdateParams.OPTIMIZE, false)) {
232                         fmt = "<optimize waitSearcher=\"%s\" />";
233                       } else if (params.getBool(UpdateParams.COMMIT, false)) {
234                         fmt = "<commit waitSearcher=\"%s\" />";
235                       }
236                       if (fmt != null) {
237                         byte[] content = String.format(Locale.ROOT,
238                             fmt,
239                             params.getBool(UpdateParams.WAIT_SEARCHER, false)
240                                 + "").getBytes(StandardCharsets.UTF_8);
241                         out.write(content);
242                       }
243                     }
244                   }
245                   out.flush();
246 
247                   if (pollQueueTime > 0 && threadCount == 1 && req.isLastDocInBatch()) {
248                     // no need to wait to see another doc in the queue if we've hit the last doc in a batch
249                     req = queue.poll(0, TimeUnit.MILLISECONDS);
250                   } else {
251                     req = queue.poll(pollQueueTime, TimeUnit.MILLISECONDS);
252                   }
253 
254                 }
255 
256                 if (isXml) {
257                   out.write("</stream>".getBytes(StandardCharsets.UTF_8));
258                 }
259 
260               } catch (InterruptedException e) {
261                 Thread.currentThread().interrupt();
262                 log.warn("", e);
263               }
264             }
265           });
266 
267           // The parser 'wt=' and 'version=' params are used instead of the
268           // original params
269           ModifiableSolrParams requestParams = new ModifiableSolrParams(origParams);
270           requestParams.set(CommonParams.WT, client.parser.getWriterType());
271           requestParams.set(CommonParams.VERSION, client.parser.getVersion());
272 
273           method = new HttpPost(client.getBaseURL() + "/update"
274               + requestParams.toQueryString());
275           method.setEntity(template);
276           method.addHeader("User-Agent", HttpSolrClient.AGENT);
277           method.addHeader("Content-Type", contentType);
278 
279           response = client.getHttpClient().execute(method);
280           int statusCode = response.getStatusLine().getStatusCode();
281           if (statusCode != HttpStatus.SC_OK) {
282             StringBuilder msg = new StringBuilder();
283             msg.append(response.getStatusLine().getReasonPhrase());
284             msg.append("\n\n\n\n");
285             msg.append("request: ").append(method.getURI());
286 
287             SolrException solrExc = new SolrException(ErrorCode.getErrorCode(statusCode), msg.toString());
288             // parse out the metadata from the SolrException
289             try {
290               NamedList<Object> resp =
291                   client.parser.processResponse(response.getEntity().getContent(),
292                       response.getEntity().getContentType().getValue());
293               NamedList<Object> error = (NamedList<Object>) resp.get("error");
294               if (error != null)
295                 solrExc.setMetadata((NamedList<String>) error.get("metadata"));
296             } catch (Exception exc) {
297               // don't want to fail to report error if parsing the response fails
298               log.warn("Failed to parse error response from " + client.getBaseURL() + " due to: " + exc);
299             }
300 
301             handleError(solrExc);
302           } else {
303             onSuccess(response);
304           }
305         } finally {
306           try {
307             if (response != null) {
308               response.getEntity().getContent().close();
309             }
310           } catch (Exception ex) {
311             log.warn("", ex);
312           }
313         }
314       }
315     }
316   }

318   // *must* be called with runners monitor held, e.g. synchronized(runners){ addRunner() }
319   private void addRunner() {
320     MDC.put("ConcurrentUpdateSolrClient.url", client.getBaseURL());
321     try {
322       Runner r = new Runner();
323       runners.add(r);
324       scheduler.execute(r);  // this can throw an exception if the scheduler has been shutdown, but that should be fine.
325     } finally {
326       MDC.remove("ConcurrentUpdateSolrClient.url");
327     }
328   }

330   @Override
331   public NamedList<Object> request(final SolrRequest request, String collection)
332       throws SolrServerException, IOException {
333     if (!(request instanceof UpdateRequest)) {
334       return client.request(request, collection);
335     }
336     UpdateRequest req = (UpdateRequest) request;
337 
338     // this happens for commit...
339     if (streamDeletes) {
340       if ((req.getDocuments() == null || req.getDocuments().isEmpty())
341           && (req.getDeleteById() == null || req.getDeleteById().isEmpty())
342           && (req.getDeleteByIdMap() == null || req.getDeleteByIdMap().isEmpty())) {
343         if (req.getDeleteQuery() == null) {
344           blockUntilFinished();
345           return client.request(request, collection);
346         }
347       }
348     } else {
349       if ((req.getDocuments() == null || req.getDocuments().isEmpty())) {
350         blockUntilFinished();
351         return client.request(request, collection);
352       }
353     }
354 
355 
356     SolrParams params = req.getParams();
357     if (params != null) {
358       // check if it is waiting for the searcher
359       if (params.getBool(UpdateParams.WAIT_SEARCHER, false)) {
360         log.info("blocking for commit/optimize");
361         blockUntilFinished(); // empty the queue
362         return client.request(request, collection);
363       }
364     }
365 
366     try {
367       CountDownLatch tmpLock = lock;
368       if (tmpLock != null) {
369         tmpLock.await();
370       }
371 
372       boolean success = queue.offer(req);
373 
374       for (;;) {
375         synchronized (runners) {
376           // see if queue is half full and we can add more runners
377           // special case: if only using a threadCount of 1 and the queue
378           // is filling up, allow 1 add'l runner to help process the queue
379           if (runners.isEmpty() || (queue.remainingCapacity() < queue.size() && runners.size() < threadCount))
380           {
381             // We need more runners, so start a new one.
382             addRunner();
383           } else {
384             // break out of the retry loop if we added the element to the queue
385             // successfully, *and*
386             // while we are still holding the runners lock to prevent race
387             // conditions.
388             if (success)
389               break;
390           }
391         }
392 
393         // Retry to add to the queue w/o the runners lock held (else we risk
394         // temporary deadlock)
395         // This retry could also fail because
396         // 1) existing runners were not able to take off any new elements in the
397         // queue
398         // 2) the queue was filled back up since our last try
399         // If we succeed, the queue may have been completely emptied, and all
400         // runners stopped.
401         // In all cases, we should loop back to the top to see if we need to
402         // start more runners.
403         //
404         if (!success) {
405           success = queue.offer(req, 100, TimeUnit.MILLISECONDS);
406         }
407       }
408     } catch (InterruptedException e) {
409       log.error("interrupted", e);
410       throw new IOException(e.getLocalizedMessage());
411     }
412 
413     // RETURN A DUMMY result
414     NamedList<Object> dummy = new NamedList<>();
415     dummy.add("NOTE", "the request is processed in a background stream");
416     return dummy;
417   }

419   public synchronized void blockUntilFinished() {
420     lock = new CountDownLatch(1);
421     try {
422       synchronized (runners) {
423 
424         // NOTE: if the executor is shut down, runners may never become empty (a scheduled task may never be run,
425         // which means it would never remove itself from the runners list.  This is why we don't wait forever
426         // and periodically check if the scheduler is shutting down.
427         while (!runners.isEmpty()) {
428           try {
429             runners.wait(250);
430           } catch (InterruptedException e) {
431             Thread.interrupted();
432           }
433           
434           if (scheduler.isShutdown())
435             break;
436                       
437           // Need to check if the queue is empty before really considering this is finished (SOLR-4260)
438           int queueSize = queue.size();
439           if (queueSize > 0 && runners.isEmpty()) {
440             // TODO: can this still happen?
441             log.warn("No more runners, but queue still has "+
442               queueSize+" adding more runners to process remaining requests on queue");
443             addRunner();
444           }
445         }
446       }
447     } finally {
448       lock.countDown();
449       lock = null;
450     }
451   }

469   @Override
470   @Deprecated
471   public void shutdown() {
472     client.shutdown();
473     if (shutdownExecutor) {
474       scheduler.shutdown();
475       try {
476         if (!scheduler.awaitTermination(60, TimeUnit.SECONDS)) {
477           scheduler.shutdownNow();
478           if (!scheduler.awaitTermination(60, TimeUnit.SECONDS)) log
479               .error("ExecutorService did not terminate");
480         }
481       } catch (InterruptedException ie) {
482         scheduler.shutdownNow();
483         Thread.currentThread().interrupt();
484       }
485     }
486   }

500   public void shutdownNow() {
501     client.shutdown();
502     if (shutdownExecutor) {
503       scheduler.shutdownNow(); // Cancel currently executing tasks
504       try {
505         if (!scheduler.awaitTermination(30, TimeUnit.SECONDS)) 
506           log.error("ExecutorService did not terminate");
507       } catch (InterruptedException ie) {
508         scheduler.shutdownNow();
509         Thread.currentThread().interrupt();
510       }
511     }    
512   }
