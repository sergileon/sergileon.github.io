20  import org.apache.lucene.analysis.MockAnalyzer;
21  import org.apache.lucene.document.Document;
22  import org.apache.lucene.document.Field;
23  import org.apache.lucene.document.IntField;
24  import org.apache.lucene.index.LeafReader;
25  import org.apache.lucene.index.DirectoryReader;
26  import org.apache.lucene.index.IndexWriter;
27  import org.apache.lucene.index.NumericDocValues;
28  import org.apache.lucene.store.Directory;
29  import org.apache.lucene.util.LuceneTestCase;
30  
31  public class TestFieldCacheReopen extends LuceneTestCase {
32    
33    // TODO: make a version of this that tests the same thing with UninvertingReader.wrap()
34    
35    // LUCENE-1579: Ensure that on a reopened reader, that any
36    // shared segments reuse the doc values arrays in
37    // FieldCache
38    public void testFieldCacheReuseAfterReopen() throws Exception {
39      Directory dir = newDirectory();
40      IndexWriter writer = new IndexWriter(
41          dir,
42          newIndexWriterConfig(new MockAnalyzer(random())).
43              setMergePolicy(newLogMergePolicy(10))
44      );
45      Document doc = new Document();
46      doc.add(new IntField("number", 17, Field.Store.NO));
47      writer.addDocument(doc);
48      writer.commit();
49    
50      // Open reader1
51      DirectoryReader r = DirectoryReader.open(dir);
52      LeafReader r1 = getOnlySegmentReader(r);
53      final NumericDocValues ints = FieldCache.DEFAULT.getNumerics(r1, "number", FieldCache.NUMERIC_UTILS_INT_PARSER, false);
54      assertEquals(17, ints.get(0));
55    
56      // Add new segment
57      writer.addDocument(doc);
58      writer.commit();
59    
60      // Reopen reader1 --> reader2
61      DirectoryReader r2 = DirectoryReader.openIfChanged(r);
62      assertNotNull(r2);
63      r.close();
64      LeafReader sub0 = r2.leaves().get(0).reader();
65      final NumericDocValues ints2 = FieldCache.DEFAULT.getNumerics(sub0, "number", FieldCache.NUMERIC_UTILS_INT_PARSER, false);
66      r2.close();
67      assertTrue(ints == ints2);
68    
69      writer.close();
70      dir.close();
71    }
72  }
