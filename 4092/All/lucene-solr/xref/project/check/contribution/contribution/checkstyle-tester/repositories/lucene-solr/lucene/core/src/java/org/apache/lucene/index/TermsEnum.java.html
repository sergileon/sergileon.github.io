20  import java.io.IOException;
21  
22  import org.apache.lucene.util.AttributeSource;
23  import org.apache.lucene.util.Bits;
24  import org.apache.lucene.util.BytesRef;
25  import org.apache.lucene.util.BytesRefIterator;
26  
27  /** Iterator to seek ({@link #seekCeil(BytesRef)}, {@link
28   * #seekExact(BytesRef)}) or step through ({@link
29   * #next} terms to obtain frequency information ({@link
30   * #docFreq}), {@link PostingsEnum} or {@link
31   * PostingsEnum} for the current term ({@link
32   * #postings}.
33   * 
34   * <p>Term enumerations are always ordered by
35   * BytesRef.compareTo, which is Unicode sort
36   * order if the terms are UTF-8 bytes.  Each term in the
37   * enumeration is greater than the one before it.</p>
38   *
39   * <p>The TermsEnum is unpositioned when you first obtain it
40   * and you must first successfully call {@link #next} or one
41   * of the <code>seek</code> methods.
42   *
43   * @lucene.experimental */
44  public abstract class TermsEnum implements BytesRefIterator {
45  
46    private AttributeSource atts = null;
47  
48    /** Sole constructor. (For invocation by subclass 
49     *  constructors, typically implicit.) */
50    protected TermsEnum() {
51    }
52  
53    /** Returns the related attributes. */
54    public AttributeSource attributes() {
55      if (atts == null) atts = new AttributeSource();
56      return atts;
57    }
58    
59    /** Represents returned result from {@link #seekCeil}. */
60    public static enum SeekStatus {
61      /** The term was not found, and the end of iteration was hit. */
62      END,
63      /** The precise term was found. */
64      FOUND,
65      /** A different term was found after the requested term */
66      NOT_FOUND
67    };
68  
69    /** Attempts to seek to the exact term, returning
70     *  true if the term is found.  If this returns false, the
71     *  enum is unpositioned.  For some codecs, seekExact may
72     *  be substantially faster than {@link #seekCeil}. */
73    public boolean seekExact(BytesRef text) throws IOException {
74      return seekCeil(text) == SeekStatus.FOUND;
75    }
76  
77    /** Seeks to the specified term, if it exists, or to the
78     *  next (ceiling) term.  Returns SeekStatus to
79     *  indicate whether exact term was found, a different
80     *  term was found, or EOF was hit.  The target term may
81     *  be before or after the current term.  If this returns
82     *  SeekStatus.END, the enum is unpositioned. */
83    public abstract SeekStatus seekCeil(BytesRef text) throws IOException;
84  
85    /** Seeks to the specified term by ordinal (position) as
86     *  previously returned by {@link #ord}.  The target ord
87     *  may be before or after the current ord, and must be
88     *  within bounds. */
89    public abstract void seekExact(long ord) throws IOException;
90  
91    /**
92     * Expert: Seeks a specific position by {@link TermState} previously obtained
93     * from {@link #termState()}. Callers should maintain the {@link TermState} to
94     * use this method. Low-level implementations may position the TermsEnum
95     * without re-seeking the term dictionary.
96     * <p>
97     * Seeking by {@link TermState} should only be used iff the state was obtained 
98     * from the same {@link TermsEnum} instance. 
99     * <p>
100    * NOTE: Using this method with an incompatible {@link TermState} might leave
101    * this {@link TermsEnum} in undefined state. On a segment level
102    * {@link TermState} instances are compatible only iff the source and the
103    * target {@link TermsEnum} operate on the same field. If operating on segment
104    * level, TermState instances must not be used across segments.
105    * <p>
106    * NOTE: A seek by {@link TermState} might not restore the
107    * {@link AttributeSource}'s state. {@link AttributeSource} states must be
108    * maintained separately if this method is used.
109    * @param term the term the TermState corresponds to
110    * @param state the {@link TermState}
111    * */
112   public void seekExact(BytesRef term, TermState state) throws IOException {
113     if (!seekExact(term)) {
114       throw new IllegalArgumentException("term=" + term + " does not exist");
115     }
116   }
117 
118   /** Returns current term. Do not call this when the enum
119    *  is unpositioned. */
120   public abstract BytesRef term() throws IOException;
121 
122   /** Returns ordinal position for current term.  This is an
123    *  optional method (the codec may throw {@link
124    *  UnsupportedOperationException}).  Do not call this
125    *  when the enum is unpositioned. */
126   public abstract long ord() throws IOException;
127 
128   /** Returns the number of documents containing the current
129    *  term.  Do not call this when the enum is unpositioned.
130    *  {@link SeekStatus#END}.*/
131   public abstract int docFreq() throws IOException;
132 
133   /** Returns the total number of occurrences of this term
134    *  across all documents (the sum of the freq() for each
135    *  doc that has this term).  This will be -1 if the
136    *  codec doesn't support this measure.  Note that, like
137    *  other term measures, this measure does not take
138    *  deleted documents into account. */
139   public abstract long totalTermFreq() throws IOException;
140 
141   /** Get {@link PostingsEnum} for the current term.  Do not
142    *  call this when the enum is unpositioned.  This method
143    *  will not return null.
144    *  <p>
145    *  <b>NOTE</b>: the returned iterator may return deleted documents, so
146    *  deleted documents have to be checked on top of the {@link PostingsEnum}.
147    *  <p>
148    *  Use this method if you only require documents and frequencies,
149    *  and do not need any proximity data.
150    *  This method is equivalent to 
151    *  {@link #postings(PostingsEnum, int) postings(reuse, PostingsEnum.FREQS)}
152    *
153    * @param reuse pass a prior PostingsEnum for possible reuse 
154    * @see #postings(PostingsEnum, int)
155    */
156   public final PostingsEnum postings(PostingsEnum reuse) throws IOException {
157     return postings(reuse, PostingsEnum.FREQS);
158   }
159 
160   /** Get {@link PostingsEnum} for the current term, with
161    *  control over whether freqs, positions, offsets or payloads
162    *  are required.  Do not call this when the enum is
163    *  unpositioned.  This method may return null if the postings
164    *  information required is not available from the index
165    *  <p>
166    *  <b>NOTE</b>: the returned iterator may return deleted documents, so
167    *  deleted documents have to be checked on top of the {@link PostingsEnum}.
168    *
169    * @param reuse pass a prior PostingsEnum for possible reuse
170    * @param flags specifies which optional per-document values
171    *        you require; see {@link PostingsEnum#FREQS}
172    */
173   public abstract PostingsEnum postings(PostingsEnum reuse, int flags) throws IOException;
174 
175   /**
176    * Expert: Returns the TermsEnums internal state to position the TermsEnum
177    * without re-seeking the term dictionary.
178    * <p>
179    * NOTE: A seek by {@link TermState} might not capture the
180    * {@link AttributeSource}'s state. Callers must maintain the
181    * {@link AttributeSource} states separately
182    * 
183    * @see TermState
184    * @see #seekExact(BytesRef, TermState)
185    */
186   public TermState termState() throws IOException {
187     return new TermState() {
188       @Override
189       public void copyFrom(TermState other) {
190         throw new UnsupportedOperationException();
191       }
192     };
193   }
194 
195   /** An empty TermsEnum for quickly returning an empty instance e.g.
196    * in {@link org.apache.lucene.search.MultiTermQuery}
197    * <p><em>Please note:</em> This enum should be unmodifiable,
198    * but it is currently possible to add Attributes to it.
199    * This should not be a problem, as the enum is always empty and
200    * the existence of unused Attributes does not matter.
201    */
202   public static final TermsEnum EMPTY = new TermsEnum() {    
203     @Override
204     public SeekStatus seekCeil(BytesRef term) { return SeekStatus.END; }
205     
206     @Override
207     public void seekExact(long ord) {}
208     
209     @Override
210     public BytesRef term() {
211       throw new IllegalStateException("this method should never be called");
212     }
213 
214     @Override
215     public int docFreq() {
216       throw new IllegalStateException("this method should never be called");
217     }
218 
219     @Override
220     public long totalTermFreq() {
221       throw new IllegalStateException("this method should never be called");
222     }
223       
224     @Override
225     public long ord() {
226       throw new IllegalStateException("this method should never be called");
227     }
228 
229     @Override
230     public PostingsEnum postings(PostingsEnum reuse, int flags) {
231       throw new IllegalStateException("this method should never be called");
232     }
233       
234     @Override
235     public BytesRef next() {
236       return null;
237     }
238     
239     @Override // make it synchronized here, to prevent double lazy init
240     public synchronized AttributeSource attributes() {
241       return super.attributes();
242     }
243 
244     @Override
245     public TermState termState() {
246       throw new IllegalStateException("this method should never be called");
247     }
248 
249     @Override
250     public void seekExact(BytesRef term, TermState state) {
251       throw new IllegalStateException("this method should never be called");
252     }
253 
254   };

256   /** Get {@link DocsEnum} for the current term.  Do not
257    *  call this when the enum is unpositioned.  This method
258    *  will not return null.
259    *  
260    * @param liveDocs unset bits are documents that should not
261    * be returned
262    * @param reuse pass a prior DocsEnum for possible reuse 
263    * @deprecated Use {@link #postings(PostingsEnum)} instead */
264   @Deprecated
265   public final DocsEnum docs(Bits liveDocs, DocsEnum reuse) throws IOException {
266     return docs(liveDocs, reuse, DocsEnum.FLAG_FREQS);
267   }
268 
269   /** Get {@link DocsEnum} for the current term, with
270    *  control over whether freqs are required.  Do not
271    *  call this when the enum is unpositioned.  This method
272    *  will not return null.
273    *  
274    * @param liveDocs unset bits are documents that should not
275    * be returned
276    * @param reuse pass a prior DocsEnum for possible reuse
277    * @param flags specifies which optional per-document values
278    *        you require; see {@link DocsEnum#FLAG_FREQS} 
279    * @see #docs(Bits, DocsEnum, int) 
280    * @deprecated Use {@link #postings(PostingsEnum, int)} instead */
281   @Deprecated
282   public final DocsEnum docs(Bits liveDocs, DocsEnum reuse, int flags) throws IOException {
283     final int newFlags;
284     if (flags == DocsEnum.FLAG_FREQS) {
285       newFlags = PostingsEnum.FREQS;
286     } else if (flags == DocsEnum.FLAG_NONE) {
287       newFlags = PostingsEnum.NONE;
288     } else {
289       throw new IllegalArgumentException("Invalid legacy docs flags: " + flags);
290     }
291     PostingsEnum actualReuse = DocsAndPositionsEnum.unwrap(reuse);
292     PostingsEnum postings = postings(actualReuse, newFlags);
293     if (postings == null) {
294       throw new AssertionError();
295     } else if (postings == actualReuse
296         && liveDocs == DocsAndPositionsEnum.unwrapliveDocs(reuse)) {
297       return reuse;
298     } else {
299       return DocsAndPositionsEnum.wrap(postings, liveDocs);
300     }
301   };
302 
303   /** Get {@link DocsAndPositionsEnum} for the current term.
304    *  Do not call this when the enum is unpositioned.  This
305    *  method will return null if positions were not
306    *  indexed.
307    *  
308    *  @param liveDocs unset bits are documents that should not
309    *  be returned
310    *  @param reuse pass a prior DocsAndPositionsEnum for possible reuse
311    *  @see #docsAndPositions(Bits, DocsAndPositionsEnum, int)
312    *  @deprecated Use {@link #postings(PostingsEnum, int)} instead */
313   @Deprecated
314   public final DocsAndPositionsEnum docsAndPositions(Bits liveDocs, DocsAndPositionsEnum reuse) throws IOException {
315     return docsAndPositions(liveDocs, reuse, DocsAndPositionsEnum.FLAG_OFFSETS | DocsAndPositionsEnum.FLAG_PAYLOADS);
316   }
317 
318   /** Get {@link DocsAndPositionsEnum} for the current term,
319    *  with control over whether offsets and payloads are
320    *  required.  Some codecs may be able to optimize their
321    *  implementation when offsets and/or payloads are not required.
322    *  Do not call this when the enum is unpositioned.  This
323    *  will return null if positions were not indexed.
324 
325    *  @param liveDocs unset bits are documents that should not
326    *  be returned
327    *  @param reuse pass a prior DocsAndPositionsEnum for possible reuse
328    *  @param flags specifies which optional per-position values you
329    *         require; see {@link DocsAndPositionsEnum#FLAG_OFFSETS} and 
330    *         {@link DocsAndPositionsEnum#FLAG_PAYLOADS}. 
331    *  @deprecated Use {@link #postings(PostingsEnum, int)} instead */
332   @Deprecated
333   public final DocsAndPositionsEnum docsAndPositions(Bits liveDocs, DocsAndPositionsEnum reuse, int flags) throws IOException {
334     final int newFlags;
335     if (flags == (DocsAndPositionsEnum.FLAG_OFFSETS | DocsAndPositionsEnum.FLAG_PAYLOADS)) {
336       newFlags = PostingsEnum.OFFSETS | PostingsEnum.PAYLOADS; 
337     } else if (flags == DocsAndPositionsEnum.FLAG_OFFSETS) {
338       newFlags = PostingsEnum.OFFSETS;
339     } else if (flags == DocsAndPositionsEnum.FLAG_PAYLOADS) {
340       newFlags = PostingsEnum.PAYLOADS;
341     } else if (flags == DocsAndPositionsEnum.FLAG_NONE) {
342       newFlags = PostingsEnum.POSITIONS;
343     } else {
344       throw new IllegalArgumentException("Invalid legacy docsAndPositions flags: " + flags);
345     }
346     PostingsEnum actualReuse = DocsAndPositionsEnum.unwrap(reuse);
347     PostingsEnum postings = postings(actualReuse, newFlags | DocsAndPositionsEnum.OLD_NULL_SEMANTICS);
348     if (postings == null) {
349       return null; // if no positions were indexed
350     } else if (postings == actualReuse
351         && liveDocs == DocsAndPositionsEnum.unwrapliveDocs(reuse)) {
352       return reuse;
353     } else {
354       return DocsAndPositionsEnum.wrap(postings, liveDocs);
355     }
356   }
