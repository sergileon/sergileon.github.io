62    public void testBasic() throws Exception {
63      Directory dir = newDirectory();
64      Directory taxoDir = newDirectory();
65  
66      // Writes facet ords to a separate directory from the
67      // main index:
68      DirectoryTaxonomyWriter taxoWriter = new DirectoryTaxonomyWriter(taxoDir, IndexWriterConfig.OpenMode.CREATE);
69  
70      FacetsConfig config = new FacetsConfig();
71      config.setHierarchical("Publish Date", true);
72  
73      RandomIndexWriter writer = new RandomIndexWriter(random(), dir);
74  
75      Document doc = new Document();
76      doc.add(new FacetField("Author", "Bob"));
77      doc.add(new FacetField("Publish Date", "2010", "10", "15"));
78      writer.addDocument(config.build(taxoWriter, doc));
79  
80      doc = new Document();
81      doc.add(new FacetField("Author", "Lisa"));
82      doc.add(new FacetField("Publish Date", "2010", "10", "20"));
83      writer.addDocument(config.build(taxoWriter, doc));
84  
85      doc = new Document();
86      doc.add(new FacetField("Author", "Lisa"));
87      doc.add(new FacetField("Publish Date", "2012", "1", "1"));
88      writer.addDocument(config.build(taxoWriter, doc));
89  
90      doc = new Document();
91      doc.add(new FacetField("Author", "Susan"));
92      doc.add(new FacetField("Publish Date", "2012", "1", "7"));
93      writer.addDocument(config.build(taxoWriter, doc));
94  
95      doc = new Document();
96      doc.add(new FacetField("Author", "Frank"));
97      doc.add(new FacetField("Publish Date", "1999", "5", "5"));
98      writer.addDocument(config.build(taxoWriter, doc));
99  
100     // NRT open
101     IndexSearcher searcher = newSearcher(writer.getReader());
102 
103     // NRT open
104     TaxonomyReader taxoReader = new DirectoryTaxonomyReader(taxoWriter);
105 
106     // Aggregate the facet counts:
107     FacetsCollector c = new FacetsCollector();
108 
109     // MatchAllDocsQuery is for "browsing" (counts facets
110     // for all non-deleted docs in the index); normally
111     // you'd use a "normal" query, and use MultiCollector to
112     // wrap collecting the "normal" hits and also facets:
113     searcher.search(new MatchAllDocsQuery(), c);
114 
115     Facets facets = new FastTaxonomyFacetCounts(taxoReader, config, c);
116 
117     // Retrieve & verify results:
118     assertEquals("dim=Publish Date path=[] value=5 childCount=3\n  2010 (2)\n  2012 (2)\n  1999 (1)\n", facets.getTopChildren(10, "Publish Date").toString());
119     assertEquals("dim=Author path=[] value=5 childCount=4\n  Lisa (2)\n  Bob (1)\n  Susan (1)\n  Frank (1)\n", facets.getTopChildren(10, "Author").toString());
120 
121     // Now user drills down on Publish Date/2010:
122     DrillDownQuery q2 = new DrillDownQuery(config);
123     q2.add("Publish Date", "2010");
124     c = new FacetsCollector();
125     searcher.search(q2, c);
126     facets = new FastTaxonomyFacetCounts(taxoReader, config, c);
127     assertEquals("dim=Author path=[] value=2 childCount=2\n  Bob (1)\n  Lisa (1)\n", facets.getTopChildren(10, "Author").toString());
128 
129     assertEquals(1, facets.getSpecificValue("Author", "Lisa"));
130 
131     assertNull(facets.getTopChildren(10, "Non exitent dim"));
132 
133     // Smoke test PrintTaxonomyStats:
134     ByteArrayOutputStream bos = new ByteArrayOutputStream();
135     PrintTaxonomyStats.printStats(taxoReader, new PrintStream(bos, false, IOUtils.UTF_8), true);
136     String result = bos.toString(IOUtils.UTF_8);
137     assertTrue(result.indexOf("/Author: 4 immediate children; 5 total categories") != -1);
138     assertTrue(result.indexOf("/Publish Date: 3 immediate children; 12 total categories") != -1);
139     // Make sure at least a few nodes of the tree came out:
140     assertTrue(result.indexOf("  /1999") != -1);
141     assertTrue(result.indexOf("  /2012") != -1);
142     assertTrue(result.indexOf("      /20") != -1);
143 
144     writer.close();
145     IOUtils.close(taxoWriter, searcher.getIndexReader(), taxoReader, taxoDir, dir);
146   }

687   public void testRandom() throws Exception {
688     String[] tokens = getRandomTokens(10);
689     Directory indexDir = newDirectory();
690     Directory taxoDir = newDirectory();
691 
692     RandomIndexWriter w = new RandomIndexWriter(random(), indexDir);
693     DirectoryTaxonomyWriter tw = new DirectoryTaxonomyWriter(taxoDir);
694     FacetsConfig config = new FacetsConfig();
695     int numDocs = atLeast(1000);
696     int numDims = TestUtil.nextInt(random(), 1, 7);
697     List<TestDoc> testDocs = getRandomDocs(tokens, numDocs, numDims);
698     for(TestDoc testDoc : testDocs) {
699       Document doc = new Document();
700       doc.add(newStringField("content", testDoc.content, Field.Store.NO));
701       for(int j=0;j<numDims;j++) {
702         if (testDoc.dims[j] != null) {
703           doc.add(new FacetField("dim" + j, testDoc.dims[j]));
704         }
705       }
706       w.addDocument(config.build(tw, doc));
707     }
708 
709     // NRT open
710     IndexSearcher searcher = newSearcher(w.getReader());
711     
712     // NRT open
713     TaxonomyReader tr = new DirectoryTaxonomyReader(tw);
714 
715     int iters = atLeast(100);
716     for(int iter=0;iter<iters;iter++) {
717       String searchToken = tokens[random().nextInt(tokens.length)];
718       if (VERBOSE) {
719         System.out.println("\nTEST: iter content=" + searchToken);
720       }
721       FacetsCollector fc = new FacetsCollector();
722       FacetsCollector.search(searcher, new TermQuery(new Term("content", searchToken)), 10, fc);
723       Facets facets = getTaxonomyFacetCounts(tr, config, fc);
724 
725       // Slow, yet hopefully bug-free, faceting:
726       @SuppressWarnings({"rawtypes","unchecked"}) Map<String,Integer>[] expectedCounts = new HashMap[numDims];
727       for(int i=0;i<numDims;i++) {
728         expectedCounts[i] = new HashMap<>();
729       }
730 
731       for(TestDoc doc : testDocs) {
732         if (doc.content.equals(searchToken)) {
733           for(int j=0;j<numDims;j++) {
734             if (doc.dims[j] != null) {
735               Integer v = expectedCounts[j].get(doc.dims[j]);
736               if (v == null) {
737                 expectedCounts[j].put(doc.dims[j], 1);
738               } else {
739                 expectedCounts[j].put(doc.dims[j], v.intValue() + 1);
740               }
741             }
742           }
743         }
744       }
745 
746       List<FacetResult> expected = new ArrayList<>();
747       for(int i=0;i<numDims;i++) {
748         List<LabelAndValue> labelValues = new ArrayList<>();
749         int totCount = 0;
750         for(Map.Entry<String,Integer> ent : expectedCounts[i].entrySet()) {
751           labelValues.add(new LabelAndValue(ent.getKey(), ent.getValue()));
752           totCount += ent.getValue();
753         }
754         sortLabelValues(labelValues);
755         if (totCount > 0) {
756           expected.add(new FacetResult("dim" + i, new String[0], totCount, labelValues.toArray(new LabelAndValue[labelValues.size()]), labelValues.size()));
757         }
758       }
759 
760       // Sort by highest value, tie break by value:
761       sortFacetResults(expected);
762 
763       List<FacetResult> actual = facets.getAllDims(10);
764 
765       // Messy: fixup ties
766       sortTies(actual);
767 
768       assertEquals(expected, actual);
769     }
770 
771     w.close();
772     IOUtils.close(tw, searcher.getIndexReader(), tr, indexDir, taxoDir);
773   }
