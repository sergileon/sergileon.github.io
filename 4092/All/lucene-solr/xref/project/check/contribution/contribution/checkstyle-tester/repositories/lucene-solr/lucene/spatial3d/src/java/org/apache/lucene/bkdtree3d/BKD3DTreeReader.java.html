31  /** Handles intersection of a shape with a BKD tree previously written with {@link BKD3DTreeWriter}.
32   *
33   * @lucene.experimental */
34  
35  final class BKD3DTreeReader implements Accountable {
36    final private int[] splitValues; 
37    final private int leafNodeOffset;
38    final private long[] leafBlockFPs;
39    final int maxDoc;
40    final IndexInput in;
41  
42    enum Relation {CELL_INSIDE_SHAPE, SHAPE_CROSSES_CELL, SHAPE_OUTSIDE_CELL, SHAPE_INSIDE_CELL};
43  
44    interface ValueFilter {
45      boolean accept(int docID);
46      Relation compare(int cellXMin, int cellXMax, int cellYMin, int cellYMax, int cellZMin, int cellZMax);
47    }
48  
49    public BKD3DTreeReader(IndexInput in, int maxDoc) throws IOException {
50  
51      // Read index:
52      int numLeaves = in.readVInt();
53      leafNodeOffset = numLeaves;
54  
55      // Tree is fully balanced binary tree, so number of nodes = numLeaves-1, except our nodeIDs are 1-based (splitValues[0] is unused):
56      splitValues = new int[numLeaves];
57      for(int i=0;i<numLeaves;i++) {
58        splitValues[i] = in.readInt();
59      }
60      leafBlockFPs = new long[numLeaves];
61      for(int i=0;i<numLeaves;i++) {
62        leafBlockFPs[i] = in.readVLong();
63      }
64  
65      this.maxDoc = maxDoc;
66      this.in = in;
67    }
68  
69    private static final class QueryState {
70      final IndexInput in;
71      byte[] scratch = new byte[16];
72      final ByteArrayDataInput scratchReader = new ByteArrayDataInput(scratch);
73      final DocIdSetBuilder docs;
74      final int xMin;
75      final int xMax;
76      final int yMin;
77      final int yMax;
78      final int zMin;
79      final int zMax;
80      final ValueFilter valueFilter;
81  
82      public QueryState(IndexInput in, int maxDoc,
83                        int xMin, int xMax,
84                        int yMin, int yMax,
85                        int zMin, int zMax,
86                        ValueFilter valueFilter) {
87        this.in = in;
88        this.docs = new DocIdSetBuilder(maxDoc);
89        this.xMin = xMin;
90        this.xMax = xMax;
91        this.yMin = yMin;
92        this.yMax = yMax;
93        this.zMin = zMin;
94        this.zMax = zMax;
95        this.valueFilter = valueFilter;
96      }
97    }
98  
99    public DocIdSet intersect(ValueFilter filter) throws IOException {
100     return intersect(Integer.MIN_VALUE, Integer.MAX_VALUE,
101                      Integer.MIN_VALUE, Integer.MAX_VALUE,
102                      Integer.MIN_VALUE, Integer.MAX_VALUE,
103                      filter);
104   }
105 
106   /** Optimized intersect which takes the 3D bbox for the query and uses that to avoid filter.compare calls
107   *   when cells are clearly outside the bbox. */
108   public DocIdSet intersect(int xMin, int xMax, int yMin, int yMax, int zMin, int zMax, ValueFilter filter) throws IOException {
109 
110     QueryState state = new QueryState(in.clone(), maxDoc,
111                                       xMin, xMax,
112                                       yMin, yMax,
113                                       zMin, zMax,
114                                       filter);
115 
116     int hitCount = intersect(state, 1,
117                              Integer.MIN_VALUE, Integer.MAX_VALUE,
118                              Integer.MIN_VALUE, Integer.MAX_VALUE,
119                              Integer.MIN_VALUE, Integer.MAX_VALUE);
120 
121     // NOTE: hitCount is an over-estimate in the multi-valued case:
122     return state.docs.build(hitCount);
123   }
124 
125   /** Fast path: this is called when the query rect fully encompasses all cells under this node. */
126   private int addAll(QueryState state, int nodeID) throws IOException {
127     //System.out.println("  addAll nodeID=" + nodeID + " leafNodeOffset=" + leafNodeOffset);
128 
129     if (nodeID >= leafNodeOffset) {
130 
131       /*
132       System.out.println("A: " + BKDTreeWriter.decodeLat(cellLatMinEnc)
133                          + " " + BKDTreeWriter.decodeLat(cellLatMaxEnc)
134                          + " " + BKDTreeWriter.decodeLon(cellLonMinEnc)
135                          + " " + BKDTreeWriter.decodeLon(cellLonMaxEnc));
136       */
137 
138       // Leaf node
139       long fp = leafBlockFPs[nodeID-leafNodeOffset];
140       //System.out.println("    leaf fp=" + fp);
141       state.in.seek(fp);
142       
143       //System.out.println("    seek to leafFP=" + fp);
144       // How many points are stored in this leaf cell:
145       int count = state.in.readVInt();
146       //System.out.println("    count=" + count);
147       state.docs.grow(count);
148       for(int i=0;i<count;i++) {
149         int docID = state.in.readInt();
150         state.docs.add(docID);
151 
152         // Up above in the recursion we asked valueFilter to relate our cell, and it returned Relation.CELL_INSIDE_SHAPE
153         // so all docs inside this cell better be accepted by the filter:
154 
155         // NOTE: this is too anal, because we lost precision in the pack/unpack (8 bytes to 4 bytes), a point that's a bit above/below the
156         // earth's surface due to that quantization may incorrectly evaluate as not inside the shape:
157         // assert state.valueFilter.accept(docID);
158       }
159 
160       return count;
161     } else {
162       int count = addAll(state, 2*nodeID);
163       count += addAll(state, 2*nodeID+1);
164       return count;
165     }
166   }
167 
168   private int intersect(QueryState state,
169                         int nodeID,
170                         int cellXMin, int cellXMax,
171                         int cellYMin, int cellYMax,
172                         int cellZMin, int cellZMax)
173     throws IOException {
174 
175     //System.out.println("BKD3D.intersect nodeID=" + nodeID + " cellX=" + cellXMin + " TO " + cellXMax + ", cellY=" + cellYMin + " TO " + cellYMax + ", cellZ=" + cellZMin + " TO " + cellZMax);
176 
177     if (cellXMin >= state.xMin ||
178         cellXMax <= state.xMax ||
179         cellYMin >= state.yMin ||
180         cellYMax <= state.yMax ||
181         cellZMin >= state.zMin ||
182         cellZMax <= state.zMax) {
183 
184       // Only call the filter when the current cell does not fully contain the bbox:
185       Relation r = state.valueFilter.compare(cellXMin, cellXMax,
186                                              cellYMin, cellYMax,
187                                              cellZMin, cellZMax);
188       //System.out.println("  relation: " + r);
189 
190       if (r == Relation.SHAPE_OUTSIDE_CELL) {
191         // This cell is fully outside of the query shape: stop recursing
192         return 0;
193       } else if (r == Relation.CELL_INSIDE_SHAPE) {
194         // This cell is fully inside of the query shape: recursively add all points in this cell without filtering
195         
196         /*
197         System.out.println(Thread.currentThread() + ": switch to addAll at cell" +
198                            " x=" + Geo3DDocValuesFormat.decodeValue(cellXMin) + " to " + Geo3DDocValuesFormat.decodeValue(cellXMax) +
199                            " y=" + Geo3DDocValuesFormat.decodeValue(cellYMin) + " to " + Geo3DDocValuesFormat.decodeValue(cellYMax) +
200                            " z=" + Geo3DDocValuesFormat.decodeValue(cellZMin) + " to " + Geo3DDocValuesFormat.decodeValue(cellZMax));
201         */
202         return addAll(state, nodeID);
203       } else {
204         // The cell crosses the shape boundary, so we fall through and do full filtering
205       }
206     } else {
207       // The whole point of the incoming bbox (state.xMin/xMax/etc.) is that it is
208       // supposed to fully enclose the shape, so this cell we are visiting, which
209       // fully contains the query's bbox, better in turn fully contain the shape!
210       assert state.valueFilter.compare(cellXMin, cellXMax, cellYMin, cellYMax, cellZMin, cellZMax) == Relation.SHAPE_INSIDE_CELL: "got " + state.valueFilter.compare(cellXMin, cellXMax, cellYMin, cellYMax, cellZMin, cellZMax);
211     }
212 
213     //System.out.println("\nintersect node=" + nodeID + " vs " + leafNodeOffset);
214 
215     if (nodeID >= leafNodeOffset) {
216       //System.out.println("  leaf");
217       // Leaf node; scan and filter all points in this block:
218       //System.out.println("    intersect leaf nodeID=" + nodeID + " vs leafNodeOffset=" + leafNodeOffset + " fp=" + leafBlockFPs[nodeID-leafNodeOffset]);
219       int hitCount = 0;
220 
221       long fp = leafBlockFPs[nodeID-leafNodeOffset];
222 
223       /*
224       System.out.println("I: " + BKDTreeWriter.decodeLat(cellLatMinEnc)
225                          + " " + BKDTreeWriter.decodeLat(cellLatMaxEnc)
226                          + " " + BKDTreeWriter.decodeLon(cellLonMinEnc)
227                          + " " + BKDTreeWriter.decodeLon(cellLonMaxEnc));
228       */
229 
230       state.in.seek(fp);
231 
232       // How many points are stored in this leaf cell:
233       int count = state.in.readVInt();
234 
235       state.docs.grow(count);
236       //System.out.println("  count=" + count);
237       for(int i=0;i<count;i++) {
238         int docID = state.in.readInt();
239         //System.out.println("  check docID=" + docID);
240         if (state.valueFilter.accept(docID)) {
241           state.docs.add(docID);
242           hitCount++;
243         }
244       }
245 
246       return hitCount;
247 
248     } else {
249 
250       //System.out.println("  non-leaf");
251 
252       int splitDim = BKD3DTreeWriter.getSplitDim(cellXMin, cellXMax,
253                                                  cellYMin, cellYMax,
254                                                  cellZMin, cellZMax);
255 
256       int splitValue = splitValues[nodeID];
257 
258       int count = 0;
259 
260       if (splitDim == 0) {
261 
262         //System.out.println("  split on lat=" + splitValue);
263 
264         // Inner node split on x:
265 
266         // Left node:
267         if (state.xMin <= splitValue) {
268           //System.out.println("  recurse left");
269           count += intersect(state,
270                              2*nodeID,
271                              cellXMin, splitValue,
272                              cellYMin, cellYMax,
273                              cellZMin, cellZMax);
274         }
275 
276         // Right node:
277         if (state.xMax >= splitValue) {
278           //System.out.println("  recurse right");
279           count += intersect(state,
280                              2*nodeID+1,
281                              splitValue, cellXMax,
282                              cellYMin, cellYMax,
283                              cellZMin, cellZMax);
284         }
285 
286       } else if (splitDim == 1) {
287         // Inner node split on y:
288 
289         // System.out.println("  split on lon=" + splitValue);
290 
291         // Left node:
292         if (state.yMin <= splitValue) {
293           // System.out.println("  recurse left");
294           count += intersect(state,
295                              2*nodeID,
296                              cellXMin, cellXMax,
297                              cellYMin, splitValue,
298                              cellZMin, cellZMax);
299         }
300 
301         // Right node:
302         if (state.yMax >= splitValue) {
303           // System.out.println("  recurse right");
304           count += intersect(state,
305                              2*nodeID+1,
306                              cellXMin, cellXMax,
307                              splitValue, cellYMax,
308                              cellZMin, cellZMax);
309         }
310       } else {
311         // Inner node split on z:
312 
313         // System.out.println("  split on lon=" + splitValue);
314 
315         // Left node:
316         if (state.zMin <= splitValue) {
317           // System.out.println("  recurse left");
318           count += intersect(state,
319                              2*nodeID,
320                              cellXMin, cellXMax,
321                              cellYMin, cellYMax,
322                              cellZMin, splitValue);
323         }
324 
325         // Right node:
326         if (state.zMax >= splitValue) {
327           // System.out.println("  recurse right");
328           count += intersect(state,
329                              2*nodeID+1,
330                              cellXMin, cellXMax,
331                              cellYMin, cellYMax,
332                              splitValue, cellZMax);
333         }
334       }
335 
336       return count;
337     }
338   }
339 
340   @Override
341   public long ramBytesUsed() {
342     return splitValues.length * RamUsageEstimator.NUM_BYTES_INT + 
343       leafBlockFPs.length * RamUsageEstimator.NUM_BYTES_LONG;
344   }
345 
346   @Override
347   public Collection<Accountable> getChildResources() {
348     return Collections.emptyList();
349   }
