50  /**
51   * Pre-analyzed field type provides a way to index a serialized token stream,
52   * optionally with an independent stored value of a field.
53   */
54  public class PreAnalyzedField extends FieldType {
55    private static final Logger LOG = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
56  
57    /** Init argument name. Value is a fully-qualified class name of the parser
58     * that implements {@link PreAnalyzedParser}.
59     */
60    public static final String PARSER_IMPL = "parserImpl";
61    
62    private static final String DEFAULT_IMPL = JsonPreAnalyzedParser.class.getName();
63  
64    
65    private PreAnalyzedParser parser;
66    private Analyzer analyzer;
67    
68    @Override
69    public void init(IndexSchema schema, Map<String, String> args) {
70      super.init(schema, args);
71      String implName = args.get(PARSER_IMPL);
72      if (implName == null) {
73        parser = new JsonPreAnalyzedParser();
74      } else {
75        // short name
76        if (JSON.equalsIgnoreCase(implName)) {
77          parser = new JsonPreAnalyzedParser();
78        } else if ("simple".equalsIgnoreCase(implName)) {
79          parser = new SimplePreAnalyzedParser();
80        } else {
81          try {
82            Class<? extends PreAnalyzedParser> implClazz = schema.getResourceLoader().findClass(implName, PreAnalyzedParser.class);
83            Constructor<?> c = implClazz.getConstructor(new Class<?>[0]);
84            parser = (PreAnalyzedParser) c.newInstance(new Object[0]);
85          } catch (Exception e) {
86            LOG.warn("Can't use the configured PreAnalyzedParser class '" + implName +
87                "', using default " + DEFAULT_IMPL, e);
88            parser = new JsonPreAnalyzedParser();
89          }
90        }
91        args.remove(PARSER_IMPL);
92      }
93      // create Analyzer instance for reuse:
94      analyzer = new SolrAnalyzer() {
95        @Override
96        protected TokenStreamComponents createComponents(String fieldName) {
97          return new TokenStreamComponents(new PreAnalyzedTokenizer(parser));
98        }
99      };
100   }
101 
102   @Override
103   public Analyzer getIndexAnalyzer() {
104     return analyzer;
105   }
106   
107   @Override
108   public Analyzer getQueryAnalyzer() {
109     return getIndexAnalyzer();
110   }
111 
112   @Override
113   public IndexableField createField(SchemaField field, Object value,
114           float boost) {
115     IndexableField f = null;
116     try {
117       f = fromString(field, String.valueOf(value), boost);
118     } catch (Exception e) {
119       LOG.warn("Error parsing pre-analyzed field '" + field.getName() + "'", e);
120       return null;
121     }
122     return f;
123   }
124   
125   @Override
126   public SortField getSortField(SchemaField field, boolean top) {
127     field.checkSortability();
128     return Sorting.getTextSortField(field.getName(), top, field.sortMissingLast(), field.sortMissingFirst());
129   }
130   
131   @Override
132   public ValueSource getValueSource(SchemaField field, QParser parser) {
133     return new SortedSetFieldSource(field.getName());
134   }
135 
136   @Override
137   public Type getUninversionType(SchemaField sf) {
138     return Type.SORTED_SET_BINARY;
139   }
140 
141   @Override
142   public void write(TextResponseWriter writer, String name, IndexableField f)
143           throws IOException {
144     writer.writeStr(name, f.stringValue(), true);
145   }

147   /** Utility method to convert a field to a string that is parse-able by this
148    * class.
149    * @param f field to convert
150    * @return string that is compatible with the serialization format
151    * @throws IOException If there is a low-level I/O error.
152    */
153   public String toFormattedString(Field f) throws IOException {
154     return parser.toFormattedString(f);
155   }

157   /**
158    * Utility method to create a {@link org.apache.lucene.document.FieldType}
159    * based on the {@link SchemaField}
160    */
161   public static org.apache.lucene.document.FieldType createFieldType(SchemaField field) {
162     if (!field.indexed() && !field.stored()) {
163       if (LOG.isTraceEnabled())
164         LOG.trace("Ignoring unindexed/unstored field: " + field);
165       return null;
166     }
167     org.apache.lucene.document.FieldType newType = new org.apache.lucene.document.FieldType();
168     newType.setTokenized(field.isTokenized());
169     newType.setStored(field.stored());
170     newType.setOmitNorms(field.omitNorms());
171     IndexOptions options = IndexOptions.DOCS_AND_FREQS_AND_POSITIONS;
172     if (field.omitTermFreqAndPositions()) {
173       options = IndexOptions.DOCS;
174     } else if (field.omitPositions()) {
175       options = IndexOptions.DOCS_AND_FREQS;
176     } else if (field.storeOffsetsWithPositions()) {
177       options = IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS;
178     }
179     newType.setIndexOptions(options);
180     newType.setStoreTermVectors(field.storeTermVector());
181     newType.setStoreTermVectorOffsets(field.storeTermOffsets());
182     newType.setStoreTermVectorPositions(field.storeTermPositions());
183     newType.setStoreTermVectorPayloads(field.storeTermPayloads());
184     return newType;
185   }

187   /**
188    * This is a simple holder of a stored part and the collected states (tokens with attributes).
189    */
190   public static class ParseResult {
191     public String str;
192     public byte[] bin;
193     public List<State> states = new LinkedList<>();
194   }
195   
196   /**
197    * Parse the input and return the stored part and the tokens with attributes.
198    */
199   public static interface PreAnalyzedParser {
200     /**
201      * Parse input.
202      * @param reader input to read from
203      * @param parent parent who will own the resulting states (tokens with attributes)
204      * @return parse result, with possibly null stored and/or states fields.
205      * @throws IOException if a parsing error or IO error occurs
206      */
207     public ParseResult parse(Reader reader, AttributeSource parent) throws IOException;
208     
209     /**
210      * Format a field so that the resulting String is valid for parsing with {@link #parse(Reader, AttributeSource)}.
211      * @param f field instance
212      * @return formatted string
213      * @throws IOException If there is a low-level I/O error.
214      */
215     public String toFormattedString(Field f) throws IOException;
216   }
217   
218   
219   public IndexableField fromString(SchemaField field, String val, float boost) throws Exception {
220     if (val == null || val.trim().length() == 0) {
221       return null;
222     }
223     PreAnalyzedTokenizer parse = new PreAnalyzedTokenizer(parser);
224     parse.setReader(new StringReader(val));
225     parse.reset(); // consume
226     org.apache.lucene.document.FieldType type = createFieldType(field);
227     if (type == null) {
228       parse.close();
229       return null;
230     }
231     Field f = null;
232     if (parse.getStringValue() != null) {
233       if (field.stored()) {
234         f = new Field(field.getName(), parse.getStringValue(), type);
235       } else {
236         type.setStored(false);
237       }
238     } else if (parse.getBinaryValue() != null) {
239       if (field.isBinary()) {
240         f = new Field(field.getName(), parse.getBinaryValue(), type);
241       }
242     } else {
243       type.setStored(false);
244     }
245     
246     if (parse.hasTokenStream()) {
247       if (field.indexed()) {
248         type.setTokenized(true);
249         if (f != null) {
250           f.setTokenStream(parse);
251         } else {
252           f = new Field(field.getName(), parse, type);
253         }
254       } else {
255         if (f != null) {
256           f.fieldType().setIndexOptions(IndexOptions.NONE);
257           f.fieldType().setTokenized(false);
258         }
259       }
260     }
261     if (f != null) {
262       f.setBoost(boost);
263     }
264     return f;
265   }

267   /**
268    * Token stream that works from a list of saved states.
269    */
270   private static class PreAnalyzedTokenizer extends Tokenizer {
271     private final List<AttributeSource.State> cachedStates = new LinkedList<>();
272     private Iterator<AttributeSource.State> it = null;
273     private String stringValue = null;
274     private byte[] binaryValue = null;
275     private PreAnalyzedParser parser;
276     
277     public PreAnalyzedTokenizer(PreAnalyzedParser parser) {
278       // we don't pack attributes: since we are used for (de)serialization and dont want bloat.
279       super(AttributeFactory.DEFAULT_ATTRIBUTE_FACTORY);
280       this.parser = parser;
281     }
282     
283     public boolean hasTokenStream() {
284       return !cachedStates.isEmpty();
285     }
286     
287     public String getStringValue() {
288       return stringValue;
289     }
290     
291     public byte[] getBinaryValue() {
292       return binaryValue;
293     }
294 
295     @Override
296     public final boolean incrementToken() {
297       // lazy init the iterator
298       if (it == null) {
299         it = cachedStates.iterator();
300       }
301     
302       if (!it.hasNext()) {
303         return false;
304       }
305       
306       AttributeSource.State state = (State) it.next();
307       restoreState(state.clone());
308       return true;
309     }
310   
311     @Override
312     public final void reset() throws IOException {
313       // NOTE: this acts like rewind if you call it again
314       if (it == null) {
315         super.reset();
316         cachedStates.clear();
317         stringValue = null;
318         binaryValue = null;
319         ParseResult res = parser.parse(input, this);
320         if (res != null) {
321           stringValue = res.str;
322           binaryValue = res.bin;
323           if (res.states != null) {
324             cachedStates.addAll(res.states);
325           }
326         }
327       }
328       it = cachedStates.iterator();
329     }
330   }
