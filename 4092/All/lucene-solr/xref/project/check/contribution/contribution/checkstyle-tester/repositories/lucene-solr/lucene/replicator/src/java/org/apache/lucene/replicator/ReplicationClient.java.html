45  /**
46   * A client which monitors and obtains new revisions from a {@link Replicator}.
47   * It can be used to either periodically check for updates by invoking
48   * {@link #startUpdateThread}, or manually by calling {@link #updateNow()}.
49   * <p>
50   * Whenever a new revision is available, the {@link #requiredFiles(Map)} are
51   * copied to the {@link Directory} specified by {@link PerSessionDirectoryFactory} and
52   * a handler is notified.
53   * 
54   * @lucene.experimental
55   */
56  public class ReplicationClient implements Closeable {
57    
58    private class ReplicationThread extends Thread {
59      
60      private final long interval;
61      
62      // client uses this to stop us
63      final CountDownLatch stop = new CountDownLatch(1);
64      
65      public ReplicationThread(long interval) {
66        this.interval = interval;
67      }
68      
69      @SuppressWarnings("synthetic-access")
70      @Override
71      public void run() {
72        while (true) {
73          long time = System.currentTimeMillis();
74          updateLock.lock();
75          try {
76            doUpdate();
77          } catch (Throwable t) {
78            handleUpdateException(t);
79          } finally {
80            updateLock.unlock();
81          }
82          time = System.currentTimeMillis() - time;
83          
84          // adjust timeout to compensate the time spent doing the replication.
85          final long timeout = interval - time;
86          if (timeout > 0) {
87            try {
88              // this will return immediately if we were ordered to stop (count=0)
89              // or the timeout has elapsed. if it returns true, it means count=0,
90              // so terminate.
91              if (stop.await(timeout, TimeUnit.MILLISECONDS)) {
92                return;
93              }
94            } catch (InterruptedException e) {
95              // if we were interruted, somebody wants to terminate us, so just
96              // throw the exception further.
97              Thread.currentThread().interrupt();
98              throw new ThreadInterruptedException(e);
99            }
100         }
101       }
102     }
103     
104   }

106   /** Handler for revisions obtained by the client. */
107   public static interface ReplicationHandler {
108     
109     /** Returns the current revision files held by the handler. */
110     public Map<String,List<RevisionFile>> currentRevisionFiles();
111     
112     /** Returns the current revision version held by the handler. */
113     public String currentVersion();
114     
115     /**
116      * Called when a new revision was obtained and is available (i.e. all needed
117      * files were successfully copied).
118      * 
119      * @param version
120      *          the version of the {@link Revision} that was copied
121      * @param revisionFiles
122      *          the files contained by this {@link Revision}
123      * @param copiedFiles
124      *          the files that were actually copied
125      * @param sourceDirectory
126      *          a mapping from a source of files to the {@link Directory} they
127      *          were copied into
128      */
129     public void revisionReady(String version, Map<String,List<RevisionFile>> revisionFiles, 
130         Map<String,List<String>> copiedFiles, Map<String, Directory> sourceDirectory) throws IOException;
131   }

133   /**
134    * Resolves a session and source into a {@link Directory} to use for copying
135    * the session files to.
136    */
137   public static interface SourceDirectoryFactory {
138     
139     /**
140      * Called to denote that the replication actions for this session were finished and the directory is no longer needed. 
141      */
142     public void cleanupSession(String sessionID) throws IOException;
143     
144     /**
145      * Returns the {@link Directory} to use for the given session and source.
146      * Implementations may e.g. return different directories for different
147      * sessions, or the same directory for all sessions. In that case, it is
148      * advised to clean the directory before it is used for a new session.
149      * 
150      * @see #cleanupSession(String)
151      */
152     public Directory getDirectory(String sessionID, String source) throws IOException;
153     
154   }

156   /** The component name to use with {@link InfoStream#isEnabled(String)}. */
157   public static final String INFO_STREAM_COMPONENT = "ReplicationThread";
158   
159   private final Replicator replicator;
160   private final ReplicationHandler handler;
161   private final SourceDirectoryFactory factory;
162   private final byte[] copyBuffer = new byte[16384];
163   private final Lock updateLock = new ReentrantLock();
164   
165   private volatile ReplicationThread updateThread;
166   private volatile boolean closed = false;
167   private volatile InfoStream infoStream = InfoStream.getDefault();
168   
169   /**
170    * Constructor.
171    * 
172    * @param replicator the {@link Replicator} used for checking for updates
173    * @param handler notified when new revisions are ready
174    * @param factory returns a {@link Directory} for a given source and session 
175    */
176   public ReplicationClient(Replicator replicator, ReplicationHandler handler, SourceDirectoryFactory factory) {
177     this.replicator = replicator;
178     this.handler = handler;
179     this.factory = factory;
180   }
181   
182   private void copyBytes(IndexOutput out, InputStream in) throws IOException {
183     int numBytes;
184     while ((numBytes = in.read(copyBuffer)) > 0) {
185       out.writeBytes(copyBuffer, 0, numBytes);
186     }
187   }
188   
189   private void doUpdate() throws IOException {
190     SessionToken session = null;
191     final Map<String,Directory> sourceDirectory = new HashMap<>();
192     final Map<String,List<String>> copiedFiles = new HashMap<>();
193     boolean notify = false;
194     try {
195       final String version = handler.currentVersion();
196       session = replicator.checkForUpdate(version);
197       if (infoStream.isEnabled(INFO_STREAM_COMPONENT)) {
198         infoStream.message(INFO_STREAM_COMPONENT, "doUpdate(): handlerVersion=" + version + " session=" + session);
199       }
200       if (session == null) {
201         // already up to date
202         return;
203       }
204       Map<String,List<RevisionFile>> requiredFiles = requiredFiles(session.sourceFiles);
205       if (infoStream.isEnabled(INFO_STREAM_COMPONENT)) {
206         infoStream.message(INFO_STREAM_COMPONENT, "doUpdate(): requiredFiles=" + requiredFiles);
207       }
208       for (Entry<String,List<RevisionFile>> e : requiredFiles.entrySet()) {
209         String source = e.getKey();
210         Directory dir = factory.getDirectory(session.id, source);
211         sourceDirectory.put(source, dir);
212         List<String> cpFiles = new ArrayList<>();
213         copiedFiles.put(source, cpFiles);
214         for (RevisionFile file : e.getValue()) {
215           if (closed) {
216             // if we're closed, abort file copy
217             if (infoStream.isEnabled(INFO_STREAM_COMPONENT)) {
218               infoStream.message(INFO_STREAM_COMPONENT, "doUpdate(): detected client was closed); abort file copy");
219             }
220             return;
221           }
222           InputStream in = null;
223           IndexOutput out = null;
224           try {
225             in = replicator.obtainFile(session.id, source, file.fileName);
226             out = dir.createOutput(file.fileName, IOContext.DEFAULT);
227             copyBytes(out, in);
228             cpFiles.add(file.fileName);
229             // TODO add some validation, on size / checksum
230           } finally {
231             IOUtils.close(in, out);
232           }
233         }
234       }
235       // only notify if all required files were successfully obtained.
236       notify = true;
237     } finally {
238       if (session != null) {
239         try {
240           replicator.release(session.id);
241         } finally {
242           if (!notify) { // cleanup after ourselves
243             IOUtils.close(sourceDirectory.values());
244             factory.cleanupSession(session.id);
245           }
246         }
247       }
248     }
249     
250     // notify outside the try-finally above, so the session is released sooner.
251     // the handler may take time to finish acting on the copied files, but the
252     // session itself is no longer needed.
253     try {
254       if (notify && !closed ) { // no use to notify if we are closed already
255         handler.revisionReady(session.version, session.sourceFiles, copiedFiles, sourceDirectory);
256       }
257     } finally {
258       IOUtils.close(sourceDirectory.values());
259       if (session != null) {
260         factory.cleanupSession(session.id);
261       }
262     }
263   }
264   
265   /** Throws {@link AlreadyClosedException} if the client has already been closed. */
266   protected final void ensureOpen() {
267     if (closed) {
268       throw new AlreadyClosedException("this update client has already been closed");
269     }
270   }
271   
272   /**
273    * Called when an exception is hit by the replication thread. The default
274    * implementation prints the full stacktrace to the {@link InfoStream} set in
275    * {@link #setInfoStream(InfoStream)}, or the {@link InfoStream#getDefault()
276    * default} one. You can override to log the exception elswhere.
277    * <p>
278    * <b>NOTE:</b> if you override this method to throw the exception further,
279    * the replication thread will be terminated. The only way to restart it is to
280    * call {@link #stopUpdateThread()} followed by
281    * {@link #startUpdateThread(long, String)}.
282    */
283   protected void handleUpdateException(Throwable t) {
284     final StringWriter sw = new StringWriter();
285     t.printStackTrace(new PrintWriter(sw));
286     if (infoStream.isEnabled(INFO_STREAM_COMPONENT)) {
287       infoStream.message(INFO_STREAM_COMPONENT, "an error occurred during revision update: " + sw.toString());
288     }
289   }
290   
291   /**
292    * Returns the files required for replication. By default, this method returns
293    * all files that exist in the new revision, but not in the handler.
294    */
295   protected Map<String,List<RevisionFile>> requiredFiles(Map<String,List<RevisionFile>> newRevisionFiles) {
296     Map<String,List<RevisionFile>> handlerRevisionFiles = handler.currentRevisionFiles();
297     if (handlerRevisionFiles == null) {
298       return newRevisionFiles;
299     }
300     
301     Map<String,List<RevisionFile>> requiredFiles = new HashMap<>();
302     for (Entry<String,List<RevisionFile>> e : handlerRevisionFiles.entrySet()) {
303       // put the handler files in a Set, for faster contains() checks later
304       Set<String> handlerFiles = new HashSet<>();
305       for (RevisionFile file : e.getValue()) {
306         handlerFiles.add(file.fileName);
307       }
308       
309       // make sure to preserve revisionFiles order
310       ArrayList<RevisionFile> res = new ArrayList<>();
311       String source = e.getKey();
312       assert newRevisionFiles.containsKey(source) : "source not found in newRevisionFiles: " + newRevisionFiles;
313       for (RevisionFile file : newRevisionFiles.get(source)) {
314         if (!handlerFiles.contains(file.fileName)) {
315           res.add(file);
316         }
317       }
318       requiredFiles.put(source, res);
319     }
320     
321     return requiredFiles;
322   }

324   @Override
325   public synchronized void close() {
326     if (!closed) {
327       stopUpdateThread();
328       closed = true;
329     }
330   }
331   
332   /**
333    * Start the update thread with the specified interval in milliseconds. For
334    * debugging purposes, you can optionally set the name to set on
335    * {@link Thread#setName(String)}. If you pass {@code null}, a default name
336    * will be set.
337    * 
338    * @throws IllegalStateException if the thread has already been started
339    */
340   public synchronized void startUpdateThread(long intervalMillis, String threadName) {
341     ensureOpen();
342     if (updateThread != null && updateThread.isAlive()) {
343       throw new IllegalStateException(
344           "cannot start an update thread when one is running, must first call 'stopUpdateThread()'");
345     }
346     threadName = threadName == null ? INFO_STREAM_COMPONENT : "ReplicationThread-" + threadName;
347     updateThread = new ReplicationThread(intervalMillis);
348     updateThread.setName(threadName);
349     updateThread.start();
350     // we rely on isAlive to return true in isUpdateThreadAlive, assert to be on the safe side
351     assert updateThread.isAlive() : "updateThread started but not alive?";
352   }
353   
354   /**
355    * Stop the update thread. If the update thread is not running, silently does
356    * nothing. This method returns after the update thread has stopped.
357    */
358   public synchronized void stopUpdateThread() {
359     if (updateThread != null) {
360       // this will trigger the thread to terminate if it awaits the lock.
361       // otherwise, if it's in the middle of replication, we wait for it to
362       // stop.
363       updateThread.stop.countDown();
364       try {
365         updateThread.join();
366       } catch (InterruptedException e) {
367         Thread.currentThread().interrupt();
368         throw new ThreadInterruptedException(e);
369       }
370       updateThread = null;
371     }
372   }
373   
374   /**
375    * Returns true if the update thread is alive. The update thread is alive if
376    * it has been {@link #startUpdateThread(long, String) started} and not
377    * {@link #stopUpdateThread() stopped}, as well as didn't hit an error which
378    * caused it to terminate (i.e. {@link #handleUpdateException(Throwable)}
379    * threw the exception further).
380    */
381   public synchronized boolean isUpdateThreadAlive() {
382     return updateThread != null && updateThread.isAlive();
383   }
384   
385   @Override
386   public String toString() {
387     String res = "ReplicationClient";
388     if (updateThread != null) {
389       res += " (" + updateThread.getName() + ")";
390     }
391     return res;
392   }
393   
394   /**
395    * Executes the update operation immediately, irregardess if an update thread
396    * is running or not.
397    */
398   public void updateNow() throws IOException {
399     ensureOpen();
400     updateLock.lock();
401     try {
402       doUpdate();
403     } finally {
404       updateLock.unlock();
405     }
406   }

408   /** Sets the {@link InfoStream} to use for logging messages. */
409   public void setInfoStream(InfoStream infoStream) {
410     if (infoStream == null) {
411       infoStream = InfoStream.NO_OUTPUT;
412     }
413     this.infoStream = infoStream;
