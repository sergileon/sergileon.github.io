17  package org.apache.solr.internal.csv;
18  
19  import java.io.StringReader;
20  
21  import junit.framework.TestCase;
22  
23  /**
24   * CSVStrategyTest
25   *
26   * The test are organized in three different sections:
27   * The 'setter/getter' section, the lexer section and finally the strategy 
28   * section. In case a test fails, you should follow a top-down approach for 
29   * fixing a potential bug (it's likely that the strategy itself fails if the lexer
30   * has problems...).
31   */
32  public class CSVStrategyTest extends TestCase {
33  
34    // ======================================================
35    //   getters / setters
36    // ======================================================
37    public void testGetSetCommentStart() {
38      CSVStrategy strategy = (CSVStrategy)CSVStrategy.DEFAULT_STRATEGY.clone();
39      strategy.setCommentStart('#');
40      assertEquals(strategy.getCommentStart(), '#');
41      strategy.setCommentStart('!');
42      assertEquals(strategy.getCommentStart(), '!');
43    }
44  
45    public void testGetSetEncapsulator() {
46      CSVStrategy strategy = (CSVStrategy)CSVStrategy.DEFAULT_STRATEGY.clone();
47      strategy.setEncapsulator('"');
48      assertEquals(strategy.getEncapsulator(), '"');
49      strategy.setEncapsulator('\'');
50      assertEquals(strategy.getEncapsulator(), '\'');
51    }
52  
53    public void testGetSetDelimiter() {
54      CSVStrategy strategy = (CSVStrategy)CSVStrategy.DEFAULT_STRATEGY.clone();
55      strategy.setDelimiter(';');
56      assertEquals(strategy.getDelimiter(), ';');
57      strategy.setDelimiter(',');
58      assertEquals(strategy.getDelimiter(), ',');
59      strategy.setDelimiter('\t');
60      assertEquals(strategy.getDelimiter(), '\t');
61    }
62  
63    public void testSetCSVStrategy() {
64      CSVStrategy strategy = CSVStrategy.DEFAULT_STRATEGY;
65      // default settings
66      assertEquals(strategy.getDelimiter(), ',');
67      assertEquals(strategy.getEncapsulator(), '"');
68      assertEquals(strategy.getCommentStart(), CSVStrategy.COMMENTS_DISABLED);
69      assertEquals(true,  strategy.getIgnoreLeadingWhitespaces());
70      assertEquals(false, strategy.getUnicodeEscapeInterpretation());
71      assertEquals(true,  strategy.getIgnoreEmptyLines());
72      // explicit csv settings
73      assertEquals(strategy.getDelimiter(), ',');
74      assertEquals(strategy.getEncapsulator(), '"');
75      assertEquals(strategy.getCommentStart(), CSVStrategy.COMMENTS_DISABLED);
76      assertEquals(true,  strategy.getIgnoreLeadingWhitespaces());
77      assertEquals(false, strategy.getUnicodeEscapeInterpretation());
78      assertEquals(true,  strategy.getIgnoreEmptyLines());
79    }
80    
81    public void testSetExcelStrategy() {
82      CSVStrategy strategy = CSVStrategy.EXCEL_STRATEGY;
83      assertEquals(strategy.getDelimiter(), ',');
84      assertEquals(strategy.getEncapsulator(), '"');
85      assertEquals(strategy.getCommentStart(), CSVStrategy.COMMENTS_DISABLED);
86      assertEquals(false,  strategy.getIgnoreLeadingWhitespaces());
87      assertEquals(false, strategy.getUnicodeEscapeInterpretation());
88      assertEquals(false, strategy.getIgnoreEmptyLines());
89    }
90    
91  }
