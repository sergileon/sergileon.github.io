43  /**
44   * A simplistic Lucene based NaiveBayes classifier, see <code>http://en.wikipedia.org/wiki/Naive_Bayes_classifier</code>;
45   *
46   * @lucene.experimental

106   /**
107    * {@inheritDoc}
108    */
109   @Override
110   public void train(LeafReader leafReader, String[] textFieldNames, String classFieldName, Analyzer analyzer, Query query)
111       throws IOException {
112     this.leafReader = leafReader;
113     this.indexSearcher = new IndexSearcher(this.leafReader);
114     this.textFieldNames = textFieldNames;
115     this.classFieldName = classFieldName;
116     this.analyzer = analyzer;
117     this.query = query;

120   /**
121    * {@inheritDoc}
122    */
123   @Override
124   public ClassificationResult<BytesRef> assignClass(String inputDocument) throws IOException {
125     List<ClassificationResult<BytesRef>> doclist = assignClassNormalizedList(inputDocument);
126     ClassificationResult<BytesRef> retval = null;
127     double maxscore = -Double.MAX_VALUE;
128     for (ClassificationResult<BytesRef> element : doclist) {
129       if (element.getScore() > maxscore) {
130         retval = element;
131         maxscore = element.getScore();
132       }
133     }
134     return retval;
135   }
136 
137   /**
138    * {@inheritDoc}
139    */
140   @Override
141   public List<ClassificationResult<BytesRef>> getClasses(String text) throws IOException {
142     List<ClassificationResult<BytesRef>> doclist = assignClassNormalizedList(text);
143     Collections.sort(doclist);
144     return doclist;
145   }
146 
147   /**
148    * {@inheritDoc}
149    */
150   @Override
151   public List<ClassificationResult<BytesRef>> getClasses(String text, int max) throws IOException {
152     List<ClassificationResult<BytesRef>> doclist = assignClassNormalizedList(text);
153     Collections.sort(doclist);
154     return doclist.subList(0, max);
155   }

157   private List<ClassificationResult<BytesRef>> assignClassNormalizedList(String inputDocument) throws IOException {
158     if (leafReader == null) {
159       throw new IOException("You must first call Classifier#train");
160     }
161     List<ClassificationResult<BytesRef>> dataList = new ArrayList<>();
162 
163     Terms terms = MultiFields.getTerms(leafReader, classFieldName);
164     TermsEnum termsEnum = terms.iterator();
165     BytesRef next;
166     String[] tokenizedDoc = tokenizeDoc(inputDocument);
167     int docsWithClassSize = countDocsWithClass();
168     while ((next = termsEnum.next()) != null) {
169       if (next.length > 0) {
170         // We are passing the term to IndexSearcher so we need to make sure it will not change over time
171         Term term = new Term(this.classFieldName, next);
172         double clVal = calculateLogPrior(term, docsWithClassSize) + calculateLogLikelihood(tokenizedDoc, term, docsWithClassSize);
173         dataList.add(new ClassificationResult<>(term.bytes(), clVal));
174       }
175     }
176 
177     // normalization; the values transforms to a 0-1 range
178     ArrayList<ClassificationResult<BytesRef>> returnList = new ArrayList<>();
179     if (!dataList.isEmpty()) {
180       Collections.sort(dataList);
181       // this is a negative number closest to 0 = a
182       double smax = dataList.get(0).getScore();
183 
184       double sumLog = 0;
185       // log(sum(exp(x_n-a)))
186       for (ClassificationResult<BytesRef> cr : dataList) {
187         // getScore-smax <=0 (both negative, smax is the smallest abs()
188         sumLog += Math.exp(cr.getScore() - smax);
189       }
190       // loga=a+log(sum(exp(x_n-a))) = log(sum(exp(x_n)))
191       double loga = smax;
192       loga += Math.log(sumLog);
193 
194       // 1/sum*x = exp(log(x))*1/sum = exp(log(x)-log(sum))
195       for (ClassificationResult<BytesRef> cr : dataList) {
196         returnList.add(new ClassificationResult<>(cr.getAssignedClass(), Math.exp(cr.getScore() - loga)));
197       }
198     }
199 
200     return returnList;
201   }

203   /**
204    * count the number of documents in the index having at least a value for the 'class' field
205    *
206    * @return the no. of documents having a value for the 'class' field
207    * @throws IOException if accessing to term vectors or search fails
208    */
209   protected int countDocsWithClass() throws IOException {
210     int docCount = MultiFields.getTerms(this.leafReader, this.classFieldName).getDocCount();
211     if (docCount == -1) { // in case codec doesn't support getDocCount
212       TotalHitCountCollector totalHitCountCollector = new TotalHitCountCollector();
213       BooleanQuery.Builder q = new BooleanQuery.Builder();
214       q.add(new BooleanClause(new WildcardQuery(new Term(classFieldName, String.valueOf(WildcardQuery.WILDCARD_STRING))), BooleanClause.Occur.MUST));
215       if (query != null) {
216         q.add(query, BooleanClause.Occur.MUST);
217       }
218       indexSearcher.search(q.build(),
219           totalHitCountCollector);
220       docCount = totalHitCountCollector.getTotalHits();
221     }
222     return docCount;
223   }

225   /**
226    * tokenize a <code>String</code> on this classifier's text fields and analyzer
227    *
228    * @param doc the <code>String</code> representing an input text (to be classified)
229    * @return a <code>String</code> array of the resulting tokens
230    * @throws IOException if tokenization fails
231    */
232   protected String[] tokenizeDoc(String doc) throws IOException {
233     Collection<String> result = new LinkedList<>();
234     for (String textFieldName : textFieldNames) {
235       try (TokenStream tokenStream = analyzer.tokenStream(textFieldName, doc)) {
236         CharTermAttribute charTermAttribute = tokenStream.addAttribute(CharTermAttribute.class);
237         tokenStream.reset();
238         while (tokenStream.incrementToken()) {
239           result.add(charTermAttribute.toString());
240         }
241         tokenStream.end();
242       }
243     }
244     return result.toArray(new String[result.size()]);
245   }
246 
247   private double calculateLogLikelihood(String[] tokenizedText, Term term, int docsWithClass) throws IOException {
248     // for each word
249     double result = 0d;
250     for (String word : tokenizedText) {
251       // search with text:word AND class:c
252       int hits = getWordFreqForClass(word, term);
253 
254       // num : count the no of times the word appears in documents of class c (+1)
255       double num = hits + 1; // +1 is added because of add 1 smoothing
256 
257       // den : for the whole dictionary, count the no of times a word appears in documents of class c (+|V|)
258       double den = getTextTermFreqForClass(term) + docsWithClass;
259 
260       // P(w|c) = num/den
261       double wordProbability = num / den;
262       result += Math.log(wordProbability);
263     }
264 
265     // log(P(d|c)) = log(P(w1|c))+...+log(P(wn|c))
266     return result;
267   }

269   /**
270    * Returns the average number of unique terms times the number of docs belonging to the input class
271    * @param term the term representing the class
272    * @return the average number of unique terms
273    * @throws IOException if a low level I/O problem happens
274    */
275   private double getTextTermFreqForClass(Term term) throws IOException {
276     double avgNumberOfUniqueTerms = 0;
277     for (String textFieldName : textFieldNames) {
278       Terms terms = MultiFields.getTerms(leafReader, textFieldName);
279       long numPostings = terms.getSumDocFreq(); // number of term/doc pairs
280       avgNumberOfUniqueTerms += numPostings / (double) terms.getDocCount(); // avg # of unique terms per doc
281     }
282     int docsWithC = leafReader.docFreq(term);
283     return avgNumberOfUniqueTerms * docsWithC; // avg # of unique terms in text fields per doc * # docs with c
284   }

286   /**
287    * Returns the number of documents of the input class ( from the whole index or from a subset)
288    * that contains the word ( in a specific field or in all the fields if no one selected)
289    * @param word the token produced by the analyzer
290    * @param term the term representing the class
291    * @return the number of documents of the input class
292    * @throws IOException if a low level I/O problem happens
293    */
294   private int getWordFreqForClass(String word, Term term) throws IOException {
295     BooleanQuery.Builder booleanQuery = new BooleanQuery.Builder();
296     BooleanQuery.Builder subQuery = new BooleanQuery.Builder();
297     for (String textFieldName : textFieldNames) {
298       subQuery.add(new BooleanClause(new TermQuery(new Term(textFieldName, word)), BooleanClause.Occur.SHOULD));
299     }
300     booleanQuery.add(new BooleanClause(subQuery.build(), BooleanClause.Occur.MUST));
301     booleanQuery.add(new BooleanClause(new TermQuery(term), BooleanClause.Occur.MUST));
302     if (query != null) {
303       booleanQuery.add(query, BooleanClause.Occur.MUST);
304     }
305     TotalHitCountCollector totalHitCountCollector = new TotalHitCountCollector();
306     indexSearcher.search(booleanQuery.build(), totalHitCountCollector);
307     return totalHitCountCollector.getTotalHits();
308   }
309 
310   private double calculateLogPrior(Term term, int docsWithClassSize) throws IOException {
311     return Math.log((double) docCount(term)) - Math.log(docsWithClassSize);
312   }
313 
314   private int docCount(Term term) throws IOException {
315     return leafReader.docFreq(term);
316   }
