113   /** Persist the schema to local storage or to ZooKeeper */
114   boolean persistManagedSchema(boolean createOnly) {
115     if (loader instanceof ZkSolrResourceLoader) {
116       return persistManagedSchemaToZooKeeper(createOnly);
117     }
118     // Persist locally
119     File managedSchemaFile = new File(loader.getConfigDir(), managedSchemaResourceName);
120     OutputStreamWriter writer = null;
121     try {
122       File parentDir = managedSchemaFile.getParentFile();
123       if ( ! parentDir.isDirectory()) {
124         if ( ! parentDir.mkdirs()) {
125           final String msg = "Can't create managed schema directory " + parentDir.getAbsolutePath();
126           log.error(msg);
127           throw new SolrException(ErrorCode.SERVER_ERROR, msg);
128         }
129       }
130       final FileOutputStream out = new FileOutputStream(managedSchemaFile);
131       writer = new OutputStreamWriter(out, StandardCharsets.UTF_8);
132       persist(writer);
133       log.info("Upgraded to managed schema at " + managedSchemaFile.getPath());
134     } catch (IOException e) {
135       final String msg = "Error persisting managed schema " + managedSchemaFile;
136       log.error(msg, e);
137       throw new SolrException(ErrorCode.SERVER_ERROR, msg, e);
138     } finally {
139       IOUtils.closeQuietly(writer);
140       try {
141         FileUtils.sync(managedSchemaFile);
142       } catch (IOException e) {
143         final String msg = "Error syncing the managed schema file " + managedSchemaFile;
144         log.error(msg, e);
145       }
146     }
147     return true;
148   }

150   /**
151    * Persists the managed schema to ZooKeeper using optimistic concurrency.
152    * <p/>
153    * If createOnly is true, success is when the schema is created or if it previously existed.
154    * <p/>
155    * If createOnly is false, success is when the schema is persisted - this will only happen
156    * if schemaZkVersion matches the version in ZooKeeper.
157    * 
158    * @return true on success 
159    */
160   boolean persistManagedSchemaToZooKeeper(boolean createOnly) {
161     final ZkSolrResourceLoader zkLoader = (ZkSolrResourceLoader)loader;
162     final ZkController zkController = zkLoader.getZkController();
163     final SolrZkClient zkClient = zkController.getZkClient();
164     final String managedSchemaPath = zkLoader.getConfigSetZkPath() + "/" + managedSchemaResourceName;
165     boolean success = true;
166     boolean schemaChangedInZk = false;
167     try {
168       // Persist the managed schema
169       StringWriter writer = new StringWriter();
170       persist(writer);
171 
172       final byte[] data = writer.toString().getBytes(StandardCharsets.UTF_8);
173       if (createOnly) {
174         try {
175           zkClient.create(managedSchemaPath, data, CreateMode.PERSISTENT, true);
176           schemaZkVersion = 0;
177           log.info("Created and persisted managed schema znode at " + managedSchemaPath);
178         } catch (KeeperException.NodeExistsException e) {
179           // This is okay - do nothing and fall through
180           log.info("Managed schema znode at " + managedSchemaPath + " already exists - no need to create it");
181         }
182       } else {
183         try {
184           // Assumption: the path exists
185           Stat stat = zkClient.setData(managedSchemaPath, data, schemaZkVersion, true);
186           schemaZkVersion = stat.getVersion();
187           log.info("Persisted managed schema version "+schemaZkVersion+" at " + managedSchemaPath);
188         } catch (KeeperException.BadVersionException e) {
189 
190           log.error("Bad version when trying to persist schema using "+schemaZkVersion+" due to: "+e);
191 
192           success = false;
193           schemaChangedInZk = true;
194         }
195       }
196     } catch (Exception e) {
197       if (e instanceof InterruptedException) {
198         Thread.currentThread().interrupt(); // Restore the interrupted status
199       }
200       final String msg = "Error persisting managed schema at " + managedSchemaPath;
201       log.error(msg, e);
202       throw new SolrException(ErrorCode.SERVER_ERROR, msg, e);
203     }
204     if (schemaChangedInZk) {
205       String msg = "Failed to persist managed schema at " + managedSchemaPath
206         + " - version mismatch";
207       log.info(msg);
208       throw new SchemaChangedInZkException(ErrorCode.CONFLICT, msg + ", retry.");
209     }
210     return success; 
211   }

213   /**
214    * Block up to a specified maximum time until we see agreement on the schema
215    * version in ZooKeeper across all replicas for a collection.
216    */
217   public static void waitForSchemaZkVersionAgreement(String collection, String localCoreNodeName,
218                                                      int schemaZkVersion, ZkController zkController, int maxWaitSecs)
219   {
220     RTimer timer = new RTimer();
221 
222     // get a list of active replica cores to query for the schema zk version (skipping this core of course)
223     List<GetZkSchemaVersionCallable> concurrentTasks = new ArrayList<>();
224     for (String coreUrl : getActiveReplicaCoreUrls(zkController, collection, localCoreNodeName))
225       concurrentTasks.add(new GetZkSchemaVersionCallable(coreUrl, schemaZkVersion));
226     if (concurrentTasks.isEmpty())
227       return; // nothing to wait for ...
228 
229 
230     log.info("Waiting up to "+maxWaitSecs+" secs for "+concurrentTasks.size()+
231         " replicas to apply schema update version "+schemaZkVersion+" for collection "+collection);
232 
233     // use an executor service to invoke schema zk version requests in parallel with a max wait time
234     int poolSize = Math.min(concurrentTasks.size(), 10);
235     ExecutorService parallelExecutor =
236         ExecutorUtil.newMDCAwareFixedThreadPool(poolSize, new DefaultSolrThreadFactory("managedSchemaExecutor"));
237     try {
238       List<Future<Integer>> results =
239           parallelExecutor.invokeAll(concurrentTasks, maxWaitSecs, TimeUnit.SECONDS);
240 
241       // determine whether all replicas have the update
242       List<String> failedList = null; // lazily init'd
243       for (int f=0; f < results.size(); f++) {
244         int vers = -1;
245         Future<Integer> next = results.get(f);
246         if (next.isDone() && !next.isCancelled()) {
247           // looks to have finished, but need to check the version value too
248           try {
249             vers = next.get();
250           } catch (ExecutionException e) {
251             // shouldn't happen since we checked isCancelled
252           }
253         }
254 
255         if (vers == -1) {
256           String coreUrl = concurrentTasks.get(f).coreUrl;
257           log.warn("Core "+coreUrl+" version mismatch! Expected "+schemaZkVersion+" but got "+vers);
258           if (failedList == null) failedList = new ArrayList<>();
259           failedList.add(coreUrl);
260         }
261       }
262 
263       // if any tasks haven't completed within the specified timeout, it's an error
264       if (failedList != null)
265         throw new SolrException(ErrorCode.SERVER_ERROR, failedList.size()+" out of "+(concurrentTasks.size() + 1)+
266             " replicas failed to update their schema to version "+schemaZkVersion+" within "+
267             maxWaitSecs+" seconds! Failed cores: "+failedList);
268 
269     } catch (InterruptedException ie) {
270       log.warn("Core "+localCoreNodeName+" was interrupted waiting for schema version "+schemaZkVersion+
271           " to propagate to "+concurrentTasks.size()+" replicas for collection "+collection);
272 
273       Thread.currentThread().interrupt();
274     } finally {
275       if (!parallelExecutor.isShutdown())
276         parallelExecutor.shutdown();
277     }
278 
279     log.info("Took {}ms for {} replicas to apply schema update version {} for collection {}",
280         timer.getTime(), concurrentTasks.size(), schemaZkVersion, collection);
281   }

283   protected static List<String> getActiveReplicaCoreUrls(ZkController zkController, String collection, String localCoreNodeName) {
284     List<String> activeReplicaCoreUrls = new ArrayList<>();
285     ZkStateReader zkStateReader = zkController.getZkStateReader();
286     ClusterState clusterState = zkStateReader.getClusterState();
287     Set<String> liveNodes = clusterState.getLiveNodes();
288     Collection<Slice> activeSlices = clusterState.getActiveSlices(collection);
289     if (activeSlices != null && activeSlices.size() > 0) {
290       for (Slice next : activeSlices) {
291         Map<String, Replica> replicasMap = next.getReplicasMap();
292         if (replicasMap != null) {
293           for (Map.Entry<String, Replica> entry : replicasMap.entrySet()) {
294             Replica replica = entry.getValue();
295             if (!localCoreNodeName.equals(replica.getName()) &&
296                 replica.getState() == Replica.State.ACTIVE &&
297                 liveNodes.contains(replica.getNodeName())) {
298               ZkCoreNodeProps replicaCoreProps = new ZkCoreNodeProps(replica);
299               activeReplicaCoreUrls.add(replicaCoreProps.getCoreUrl());
300             }
301           }
302         }
303       }
304     }
305     return activeReplicaCoreUrls;
306   }

308   private static class GetZkSchemaVersionCallable extends SolrRequest implements Callable<Integer> {
309 
310     private String coreUrl;
311     private int expectedZkVersion;
312 
313     GetZkSchemaVersionCallable(String coreUrl, int expectedZkVersion) {
314       super(METHOD.GET, "/schema/zkversion");
315 
316       this.coreUrl = coreUrl;
317       this.expectedZkVersion = expectedZkVersion;
318     }
319 
320     @Override
321     public SolrParams getParams() {
322       ModifiableSolrParams wparams = new ModifiableSolrParams();
323       wparams.set("refreshIfBelowVersion", expectedZkVersion);
324       return wparams;
325     }
326 
327     @Override
328     public Integer call() throws Exception {
329       int remoteVersion = -1;
330       try (HttpSolrClient solr = new HttpSolrClient(coreUrl)) {
331         // eventually, this loop will get killed by the ExecutorService's timeout
332         while (remoteVersion == -1 || remoteVersion < expectedZkVersion) {
333           try {
334             HttpSolrClient.HttpUriRequestResponse mrr = solr.httpUriRequest(this);
335             NamedList<Object> zkversionResp = mrr.future.get();
336             if (zkversionResp != null)
337               remoteVersion = (Integer)zkversionResp.get("zkversion");
338 
339             if (remoteVersion < expectedZkVersion) {
340               // rather than waiting and re-polling, let's be proactive and tell the replica
341               // to refresh its schema from ZooKeeper, if that fails, then the
342               //Thread.sleep(1000); // slight delay before requesting version again
343               log.error("Replica "+coreUrl+" returned schema version "+
344                   remoteVersion+" and has not applied schema version "+expectedZkVersion);
345             }
346 
347           } catch (Exception e) {
348             if (e instanceof InterruptedException) {
349               break; // stop looping
350             } else {
351               log.warn("Failed to get /schema/zkversion from " + coreUrl + " due to: " + e);
352             }
353           }
354         }
355       }
356       return remoteVersion;
357     }
358 
359     @Override
360     public Collection<ContentStream> getContentStreams() throws IOException {
361       return null;
362     }
363 
364     @Override
365     protected SolrResponse createResponse(SolrClient client) {
366       return null;
367     }

385   @Override
386   public ManagedIndexSchema addFields(Collection<SchemaField> newFields,
387                                       Map<String, Collection<String>> copyFieldNames,
388                                       boolean persist) {
389     ManagedIndexSchema newSchema;
390     if (isMutable) {
391       boolean success = false;
392       if (copyFieldNames == null){
393         copyFieldNames = Collections.emptyMap();
394       }
395       newSchema = shallowCopy(true);
396 
397       for (SchemaField newField : newFields) {
398         if (null != newSchema.getFieldOrNull(newField.getName())) {
399           String msg = "Field '" + newField.getName() + "' already exists.";
400           throw new FieldExistsException(ErrorCode.BAD_REQUEST, msg);
401         }
402         newSchema.fields.put(newField.getName(), newField);
403 
404         if (null != newField.getDefaultValue()) {
405           log.debug(newField.getName() + " contains default value: " + newField.getDefaultValue());
406           newSchema.fieldsWithDefaultValue.add(newField);
407         }
408         if (newField.isRequired()) {
409           log.debug("{} is required in this schema", newField.getName());
410           newSchema.requiredFields.add(newField);
411         }
412         Collection<String> copyFields = copyFieldNames.get(newField.getName());
413         if (copyFields != null) {
414           for (String copyField : copyFields) {
415             newSchema.registerCopyField(newField.getName(), copyField);
416           }
417         }
418       }
419 
420       newSchema.postReadInform();
421 
422       newSchema.refreshAnalyzers();
423 
424       if(persist) {
425         success = newSchema.persistManagedSchema(false); // don't just create - update it if it already exists
426         if (success) {
427           log.debug("Added field(s): {}", newFields);
428         } else {
429           log.error("Failed to add field(s): {}", newFields);
430           newSchema = null;
431         }
432       }
433     } else {
434       String msg = "This ManagedIndexSchema is not mutable.";
435       log.error(msg);
436       throw new SolrException(ErrorCode.SERVER_ERROR, msg);
437     }
438     return newSchema;
439   }

441   @Override
442   public ManagedIndexSchema deleteFields(Collection<String> names) {
443     ManagedIndexSchema newSchema;
444     if (isMutable) {
445       newSchema = shallowCopy(true);
446       for (String name : names) {
447         SchemaField field = getFieldOrNull(name); 
448         if (null != field) {
449           String message = "Can't delete field '" + name
450               + "' because it's referred to by at least one copy field directive.";
451           if (newSchema.copyFieldsMap.containsKey(name) || newSchema.isCopyFieldTarget(field)) {
452             throw new SolrException(ErrorCode.BAD_REQUEST, message);
453           }
454           for (int i = 0 ; i < newSchema.dynamicCopyFields.length ; ++i) {
455             DynamicCopy dynamicCopy = newSchema.dynamicCopyFields[i];
456             if (name.equals(dynamicCopy.getRegex())) {
457               throw new SolrException(ErrorCode.BAD_REQUEST, message);
458             }
459           }
460           newSchema.fields.remove(name);
461           newSchema.fieldsWithDefaultValue.remove(field);
462           newSchema.requiredFields.remove(field);
463         } else {
464           String msg = "The field '" + name + "' is not present in this schema, and so cannot be deleted.";
465           throw new SolrException(ErrorCode.BAD_REQUEST, msg);
466         }
467       }
468       newSchema.postReadInform();
469       newSchema.refreshAnalyzers();
470     } else {
471       String msg = "This ManagedIndexSchema is not mutable.";
472       log.error(msg);
473       throw new SolrException(ErrorCode.SERVER_ERROR, msg);
474     }
475     return newSchema;
476   }

478   @Override
479   public ManagedIndexSchema replaceField
480       (String fieldName, FieldType replacementFieldType, Map<String,?> replacementArgs) {
481     ManagedIndexSchema newSchema;
482     if (isMutable) {
483       SchemaField oldField = fields.get(fieldName);
484       if (null == oldField) {
485         String msg = "The field '" + fieldName + "' is not present in this schema, and so cannot be replaced.";
486         throw new SolrException(ErrorCode.BAD_REQUEST, msg);
487       }
488       newSchema = shallowCopy(true);
489       // clone data structures before modifying them
490       newSchema.copyFieldsMap = cloneCopyFieldsMap(copyFieldsMap);
491       newSchema.copyFieldTargetCounts
492           = (Map<SchemaField,Integer>)((HashMap<SchemaField,Integer>)copyFieldTargetCounts).clone();
493       newSchema.dynamicCopyFields = new DynamicCopy[dynamicCopyFields.length];
494       System.arraycopy(dynamicCopyFields, 0, newSchema.dynamicCopyFields, 0, dynamicCopyFields.length);
495 
496       // Drop the old field
497       newSchema.fields.remove(fieldName);
498       newSchema.fieldsWithDefaultValue.remove(oldField);
499       newSchema.requiredFields.remove(oldField);
500 
501       // Add the replacement field
502       SchemaField replacementField = SchemaField.create(fieldName, replacementFieldType, replacementArgs);
503       newSchema.fields.put(fieldName, replacementField);
504       if (null != replacementField.getDefaultValue()) {
505         log.debug(replacementField.getName() + " contains default value: " + replacementField.getDefaultValue());
506         newSchema.fieldsWithDefaultValue.add(replacementField);
507       }
508       if (replacementField.isRequired()) {
509         log.debug("{} is required in this schema", replacementField.getName());
510         newSchema.requiredFields.add(replacementField);
511       }
512 
513       List<CopyField> copyFieldsToRebuild = new ArrayList<>();
514       newSchema.removeCopyFieldSource(fieldName, copyFieldsToRebuild);
515 
516       newSchema.copyFieldTargetCounts.remove(oldField); // zero out target count for this field
517 
518       // Remove copy fields where the target is this field; remember them to rebuild
519       for (Map.Entry<String,List<CopyField>> entry : newSchema.copyFieldsMap.entrySet()) {
520         List<CopyField> perSourceCopyFields = entry.getValue();
521         Iterator<CopyField> checkDestCopyFieldsIter = perSourceCopyFields.iterator();
522         while (checkDestCopyFieldsIter.hasNext()) {
523           CopyField checkDestCopyField = checkDestCopyFieldsIter.next();
524           if (fieldName.equals(checkDestCopyField.getDestination().getName())) {
525             checkDestCopyFieldsIter.remove();
526             copyFieldsToRebuild.add(checkDestCopyField);
527           }
528         }
529       }
530       newSchema.rebuildCopyFields(copyFieldsToRebuild);
531 
532       // Find dynamic copy fields where the source or destination is this field; remember them to rebuild
533       List<DynamicCopy> dynamicCopyFieldsToRebuild = new ArrayList<>();
534       List<DynamicCopy> newDynamicCopyFields = new ArrayList<>();
535       for (int i = 0 ; i < newSchema.dynamicCopyFields.length ; ++i) {
536         DynamicCopy dynamicCopy = newSchema.dynamicCopyFields[i];
537         SchemaField destinationPrototype = dynamicCopy.getDestination().getPrototype();
538         if (fieldName.equals(dynamicCopy.getRegex()) || fieldName.equals(destinationPrototype.getName())) {
539           dynamicCopyFieldsToRebuild.add(dynamicCopy);
540         } else {
541           newDynamicCopyFields.add(dynamicCopy);
542         }
543       }
544       // Rebuild affected dynamic copy fields
545       if (dynamicCopyFieldsToRebuild.size() > 0) {
546         newSchema.dynamicCopyFields = newDynamicCopyFields.toArray(new DynamicCopy[newDynamicCopyFields.size()]);
547         for (DynamicCopy dynamicCopy : dynamicCopyFieldsToRebuild) {
548           newSchema.registerCopyField(dynamicCopy.getRegex(), dynamicCopy.getDestFieldName(), dynamicCopy.getMaxChars());
549         }
550       }
551 
552       newSchema.postReadInform();
553       newSchema.refreshAnalyzers();
554     } else {
555       String msg = "This ManagedIndexSchema is not mutable.";
556       log.error(msg);
557       throw new SolrException(ErrorCode.SERVER_ERROR, msg);
558     }
559     return newSchema;
560   }

562   @Override
563   public ManagedIndexSchema addDynamicFields(Collection<SchemaField> newDynamicFields, 
564                                              Map<String,Collection<String>> copyFieldNames, boolean persist) {
565     ManagedIndexSchema newSchema;
566     if (isMutable) {
567       boolean success = false;
568       if (copyFieldNames == null){
569         copyFieldNames = Collections.emptyMap();
570       }
571       newSchema = shallowCopy(true);
572 
573       for (SchemaField newDynamicField : newDynamicFields) {
574         List<DynamicField> dFields = new ArrayList<>(Arrays.asList(newSchema.dynamicFields));
575         if (isDuplicateDynField(dFields, newDynamicField)) {
576           String msg = "Dynamic field '" + newDynamicField.getName() + "' already exists.";
577           throw new FieldExistsException(ErrorCode.BAD_REQUEST, msg);
578         }
579         dFields.add(new DynamicField(newDynamicField));
580         newSchema.dynamicFields = dynamicFieldListToSortedArray(dFields);
581 
582         Collection<String> copyFields = copyFieldNames.get(newDynamicField.getName());
583         if (copyFields != null) {
584           for (String copyField : copyFields) {
585             newSchema.registerCopyField(newDynamicField.getName(), copyField);
586           }
587         }
588       }
589 
590       newSchema.postReadInform();
591       newSchema.refreshAnalyzers();
592       if (persist) {
593         success = newSchema.persistManagedSchema(false); // don't just create - update it if it already exists
594         if (success) {
595           log.debug("Added dynamic field(s): {}", newDynamicFields);
596         } else {
597           log.error("Failed to add dynamic field(s): {}", newDynamicFields);
598         }
599       }
600     } else {
601       String msg = "This ManagedIndexSchema is not mutable.";
602       log.error(msg);
603       throw new SolrException(ErrorCode.SERVER_ERROR, msg);
604     }
605     return newSchema;
606   }

608   @Override
609   public ManagedIndexSchema deleteDynamicFields(Collection<String> fieldNamePatterns) {
610     ManagedIndexSchema newSchema;
611     if (isMutable) {
612       newSchema = shallowCopy(true);
613 
614       newSchema.dynamicCopyFields = new DynamicCopy[dynamicCopyFields.length];
615       System.arraycopy(dynamicCopyFields, 0, newSchema.dynamicCopyFields, 0, dynamicCopyFields.length);
616 
617       List<DynamicCopy> dynamicCopyFieldsToRebuild = new ArrayList<>();
618       List<DynamicCopy> newDynamicCopyFields = new ArrayList<>();
619 
620       for (String fieldNamePattern : fieldNamePatterns) {
621         DynamicField dynamicField = null;
622         int dfPos = 0;
623         for ( ; dfPos < newSchema.dynamicFields.length ; ++dfPos) {
624           DynamicField df = newSchema.dynamicFields[dfPos];
625           if (df.getRegex().equals(fieldNamePattern)) {
626             dynamicField = df;
627             break;
628           }
629         }
630         if (null == dynamicField) {
631           String msg = "The dynamic field '" + fieldNamePattern
632               + "' is not present in this schema, and so cannot be deleted.";
633           throw new SolrException(ErrorCode.BAD_REQUEST, msg);
634         }          
635         for (int i = 0 ; i < newSchema.dynamicCopyFields.length ; ++i) {
636           DynamicCopy dynamicCopy = newSchema.dynamicCopyFields[i];
637           DynamicField destDynamicBase = dynamicCopy.getDestDynamicBase();
638           DynamicField sourceDynamicBase = dynamicCopy.getSourceDynamicBase();
639           if ((null != destDynamicBase && fieldNamePattern.equals(destDynamicBase.getRegex()))
640               || (null != sourceDynamicBase && fieldNamePattern.equals(sourceDynamicBase.getRegex()))
641               || dynamicField.matches(dynamicCopy.getRegex())
642               || dynamicField.matches(dynamicCopy.getDestFieldName())) {
643             dynamicCopyFieldsToRebuild.add(dynamicCopy);
644             newSchema.decrementCopyFieldTargetCount(dynamicCopy.getDestination().getPrototype());
645             // don't add this dynamic copy field to newDynamicCopyFields - effectively removing it
646           } else {
647             newDynamicCopyFields.add(dynamicCopy);
648           }
649         }
650         if (newSchema.dynamicFields.length > 1) {
651           DynamicField[] temp = new DynamicField[newSchema.dynamicFields.length - 1];
652           System.arraycopy(newSchema.dynamicFields, 0, temp, 0, dfPos);
653           // skip over the dynamic field to be deleted
654           System.arraycopy(newSchema.dynamicFields, dfPos + 1, temp, dfPos, newSchema.dynamicFields.length - dfPos - 1);
655           newSchema.dynamicFields = temp;
656         } else {
657           newSchema.dynamicFields = new DynamicField[0];
658         }
659       }
660       // After removing all dynamic fields, rebuild affected dynamic copy fields.
661       // This may trigger an exception, if one of the deleted dynamic fields was the only matching source or target.
662       if (dynamicCopyFieldsToRebuild.size() > 0) {
663         newSchema.dynamicCopyFields = newDynamicCopyFields.toArray(new DynamicCopy[newDynamicCopyFields.size()]);
664         for (DynamicCopy dynamicCopy : dynamicCopyFieldsToRebuild) {
665           newSchema.registerCopyField(dynamicCopy.getRegex(), dynamicCopy.getDestFieldName(), dynamicCopy.getMaxChars());
666         }
667       }
668 
669       newSchema.postReadInform();
670       newSchema.refreshAnalyzers();
671     } else {
672       String msg = "This ManagedIndexSchema is not mutable.";
673       log.error(msg);
674       throw new SolrException(ErrorCode.SERVER_ERROR, msg);
675     }
676     return newSchema;
677   }

824   private void deleteCopyField(String source, String dest) {
825     // Assumption: a copy field directive will exist only if the source & destination (dynamic) fields exist
826     SchemaField destSchemaField = fields.get(dest);
827     SchemaField sourceSchemaField = fields.get(source);
828 
829     final String invalidGlobMessage = "is an invalid glob: either it contains more than one asterisk,"
830         + " or the asterisk occurs neither at the start nor at the end.";
831     if (source.contains("*") && ! isValidFieldGlob(source)) {
832       String msg = "copyField source '" + source + "' " + invalidGlobMessage;
833       throw new SolrException(ErrorCode.BAD_REQUEST, msg);
834     }
835     if (dest.contains("*") && ! isValidFieldGlob(dest)) {
836       String msg = "copyField dest '" + dest + "' " + invalidGlobMessage;
837       throw new SolrException(ErrorCode.BAD_REQUEST, msg);
838     }
839 
840     boolean found = false;
841 
842     if (null == destSchemaField || null == sourceSchemaField) { // Must be dynamic copy field
843       if (dynamicCopyFields != null) {
844         for (int i = 0 ; i < dynamicCopyFields.length ; ++i) {
845           DynamicCopy dynamicCopy = dynamicCopyFields[i];
846           if (source.equals(dynamicCopy.getRegex()) && dest.equals(dynamicCopy.getDestFieldName())) {
847             found = true;
848             SchemaField destinationPrototype = dynamicCopy.getDestination().getPrototype();
849             if (copyFieldTargetCounts.containsKey(destinationPrototype)) {
850               decrementCopyFieldTargetCount(destinationPrototype);
851             }
852             if (dynamicCopyFields.length > 1) {
853               DynamicCopy[] temp = new DynamicCopy[dynamicCopyFields.length - 1];
854               System.arraycopy(dynamicCopyFields, 0, temp, 0, i);
855               // skip over the dynamic copy field to be deleted
856               System.arraycopy(dynamicCopyFields, i + 1, temp, i, dynamicCopyFields.length - i - 1);
857               dynamicCopyFields = temp;
858             } else {
859               dynamicCopyFields = null;
860             }
861             break;
862           }
863         }
864       }
865     } else { // non-dynamic copy field directive
866       List<CopyField> copyFieldList = copyFieldsMap.get(source);
867       if (copyFieldList != null) {
868         for (Iterator<CopyField> iter = copyFieldList.iterator() ; iter.hasNext() ; ) {
869           CopyField copyField = iter.next();
870           if (dest.equals(copyField.getDestination().getName())) {
871             found = true;
872             decrementCopyFieldTargetCount(copyField.getDestination());
873             iter.remove();
874             if (copyFieldList.isEmpty()) {
875               copyFieldsMap.remove(source);
876             }
877             break;
878           }
879         }
880       }
881     }
882     if ( ! found) {
883       throw new SolrException(ErrorCode.BAD_REQUEST,
884           "Copy field directive not found: '" + source + "' -> '" + dest + "'");
885     }
886   }

1029   @Override
1030   public ManagedIndexSchema replaceFieldType(String typeName, String replacementClassName, Map<String,Object> replacementArgs) {
1031     ManagedIndexSchema newSchema;
1032     if (isMutable) {
1033       if ( ! fieldTypes.containsKey(typeName)) {
1034         String msg = "The field type '" + typeName + "' is not present in this schema, and so cannot be replaced.";
1035         throw new SolrException(ErrorCode.BAD_REQUEST, msg);
1036       }
1037       newSchema = shallowCopy(true);
1038       // clone data structures before modifying them
1039       newSchema.fieldTypes = (Map<String,FieldType>)((HashMap<String,FieldType>)fieldTypes).clone();
1040       newSchema.copyFieldsMap = cloneCopyFieldsMap(copyFieldsMap);
1041       newSchema.copyFieldTargetCounts
1042           = (Map<SchemaField,Integer>)((HashMap<SchemaField,Integer>)copyFieldTargetCounts).clone();
1043       newSchema.dynamicCopyFields = new DynamicCopy[dynamicCopyFields.length];
1044       System.arraycopy(dynamicCopyFields, 0, newSchema.dynamicCopyFields, 0, dynamicCopyFields.length);
1045       newSchema.dynamicFields = new DynamicField[dynamicFields.length];
1046       System.arraycopy(dynamicFields, 0, newSchema.dynamicFields, 0, dynamicFields.length);
1047       
1048       newSchema.fieldTypes.remove(typeName);
1049       FieldType replacementFieldType = newSchema.newFieldType(typeName, replacementClassName, replacementArgs);
1050       newSchema.fieldTypes.put(typeName, replacementFieldType);
1051 
1052       // Rebuild fields of the type being replaced
1053       List<CopyField> copyFieldsToRebuild = new ArrayList<>();
1054       List<SchemaField> replacementFields = new ArrayList<>();
1055       Iterator<Map.Entry<String,SchemaField>> fieldsIter = newSchema.fields.entrySet().iterator();
1056       while (fieldsIter.hasNext()) {
1057         Map.Entry<String,SchemaField> entry = fieldsIter.next();
1058         SchemaField oldField = entry.getValue();
1059         if (oldField.getType().getTypeName().equals(typeName)) {
1060           String fieldName = oldField.getName();
1061           
1062           // Drop the old field
1063           fieldsIter.remove();
1064           newSchema.fieldsWithDefaultValue.remove(oldField);
1065           newSchema.requiredFields.remove(oldField);
1066           
1067           // Add the replacement field
1068           SchemaField replacementField = SchemaField.create(fieldName, replacementFieldType, oldField.getArgs());
1069           replacementFields.add(replacementField); // Save the new field to be added after iteration is finished
1070           if (null != replacementField.getDefaultValue()) {
1071             log.debug(replacementField.getName() + " contains default value: " + replacementField.getDefaultValue());
1072             newSchema.fieldsWithDefaultValue.add(replacementField);
1073           }
1074           if (replacementField.isRequired()) {
1075             log.debug("{} is required in this schema", replacementField.getName());
1076             newSchema.requiredFields.add(replacementField);
1077           }
1078           newSchema.removeCopyFieldSource(fieldName, copyFieldsToRebuild);
1079         }
1080       }
1081       for (SchemaField replacementField : replacementFields) {
1082         newSchema.fields.put(replacementField.getName(), replacementField);
1083       }
1084       // Remove copy fields where the target is of the type being replaced; remember them to rebuild
1085       Iterator<Map.Entry<String,List<CopyField>>> copyFieldsMapIter = newSchema.copyFieldsMap.entrySet().iterator();
1086       while (copyFieldsMapIter.hasNext()) {
1087         Map.Entry<String,List<CopyField>> entry = copyFieldsMapIter.next();
1088         List<CopyField> perSourceCopyFields = entry.getValue();
1089         Iterator<CopyField> checkDestCopyFieldsIter = perSourceCopyFields.iterator();
1090         while (checkDestCopyFieldsIter.hasNext()) {
1091           CopyField checkDestCopyField = checkDestCopyFieldsIter.next();
1092           SchemaField destination = checkDestCopyField.getDestination();
1093           if (typeName.equals(destination.getType().getTypeName())) {
1094             checkDestCopyFieldsIter.remove();
1095             copyFieldsToRebuild.add(checkDestCopyField);
1096             newSchema.copyFieldTargetCounts.remove(destination); // zero out target count
1097           }
1098         }
1099         if (perSourceCopyFields.isEmpty()) {
1100           copyFieldsMapIter.remove();
1101         }
1102       }
1103       // Rebuild dynamic fields of the type being replaced
1104       for (int i = 0; i < newSchema.dynamicFields.length; ++i) {
1105         SchemaField prototype = newSchema.dynamicFields[i].getPrototype();
1106         if (typeName.equals(prototype.getType().getTypeName())) {
1107           newSchema.dynamicFields[i] = new DynamicField
1108               (SchemaField.create(prototype.getName(), replacementFieldType, prototype.getArgs()));
1109         }
1110       }
1111       // Find dynamic copy fields where the destination field's type is being replaced
1112       // or the source dynamic base's type is being replaced; remember them to rebuild
1113       List<DynamicCopy> dynamicCopyFieldsToRebuild = new ArrayList<>();
1114       List<DynamicCopy> newDynamicCopyFields = new ArrayList<>();
1115       for (int i = 0 ; i < newSchema.dynamicCopyFields.length ; ++i) {
1116         DynamicCopy dynamicCopy = newSchema.dynamicCopyFields[i];
1117         DynamicField sourceDynamicBase = dynamicCopy.getSourceDynamicBase();
1118         SchemaField destinationPrototype = dynamicCopy.getDestination().getPrototype();
1119         if (typeName.equals(destinationPrototype.getType().getTypeName())
1120             || (null != sourceDynamicBase && typeName.equals(sourceDynamicBase.getPrototype().getType().getTypeName()))) {
1121           dynamicCopyFieldsToRebuild.add(dynamicCopy);
1122           if (newSchema.copyFieldTargetCounts.containsKey(destinationPrototype)) {
1123             newSchema.decrementCopyFieldTargetCount(destinationPrototype);
1124           }
1125           // don't add this dynamic copy field to newDynamicCopyFields - effectively removing it
1126         } else {
1127           newDynamicCopyFields.add(dynamicCopy);
1128         }
1129       }
1130       // Rebuild affected dynamic copy fields
1131       if (dynamicCopyFieldsToRebuild.size() > 0) {
1132         newSchema.dynamicCopyFields = newDynamicCopyFields.toArray(new DynamicCopy[newDynamicCopyFields.size()]);
1133         for (DynamicCopy dynamicCopy : dynamicCopyFieldsToRebuild) {
1134           newSchema.registerCopyField(dynamicCopy.getRegex(), dynamicCopy.getDestFieldName(), dynamicCopy.getMaxChars());
1135         }
1136       }
1137       newSchema.rebuildCopyFields(copyFieldsToRebuild);
1138 
1139       newSchema.postReadInform();
1140       newSchema.refreshAnalyzers();
1141     } else {
1142       String msg = "This ManagedIndexSchema is not mutable.";
1143       log.error(msg);
1144       throw new SolrException(ErrorCode.SERVER_ERROR, msg);
1145     }
1146     return newSchema;
1147   }

1157   /**
1158    * Informs analyzers used by a fieldType.
1159    */
1160   protected void informResourceLoaderAwareObjectsForFieldType(FieldType fieldType) {
1161     // must inform any sub-components used in the
1162     // tokenizer chain if they are ResourceLoaderAware
1163     if (!fieldType.supportsAnalyzers())
1164       return;
1165 
1166     Analyzer indexAnalyzer = fieldType.getIndexAnalyzer();
1167     if (indexAnalyzer != null && indexAnalyzer instanceof TokenizerChain)
1168       informResourceLoaderAwareObjectsInChain((TokenizerChain)indexAnalyzer);
1169 
1170     Analyzer queryAnalyzer = fieldType.getQueryAnalyzer();
1171     // ref comparison is correct here (vs. equals) as they may be the same
1172     // object in which case, we don't need to inform twice ... however, it's
1173     // actually safe to call inform multiple times on an object anyway
1174     if (queryAnalyzer != null &&
1175         queryAnalyzer != indexAnalyzer &&
1176         queryAnalyzer instanceof TokenizerChain)
1177       informResourceLoaderAwareObjectsInChain((TokenizerChain)queryAnalyzer);
1178 
1179     // if fieldType is a TextField, it might have a multi-term analyzer
1180     if (fieldType instanceof TextField) {
1181       TextField textFieldType = (TextField)fieldType;
1182       Analyzer multiTermAnalyzer = textFieldType.getMultiTermAnalyzer();
1183       if (multiTermAnalyzer != null && multiTermAnalyzer != indexAnalyzer &&

1264

1266       String msg = "Field type '" + typeName + "' already exists.";

1268       throw new SolrException(ErrorCode.BAD_REQUEST, msg);
1269     }
1270 
1271     // build the new FieldType using the existing FieldTypePluginLoader framework
1272     // which expects XML, so we use a JSON to XML adapter to transform the JSON object
1273     // provided in the request into the XML format supported by the plugin loader
1274     Map<String,FieldType> newFieldTypes = new HashMap<>();
1275     List<SchemaAware> schemaAwareList = new ArrayList<>();
1276     FieldTypePluginLoader typeLoader = new FieldTypePluginLoader(this, newFieldTypes, schemaAwareList);
1277     typeLoader.loadSingle(loader, FieldTypeXmlAdapter.toNode(options));
1278     FieldType ft = newFieldTypes.get(typeName);
1279     if (!schemaAwareList.isEmpty())
1280       schemaAware.addAll(schemaAwareList);
1281 
1282     return ft;

1285   /**
1286    * After creating a new FieldType, it may contain components that implement
1287    * the ResourceLoaderAware interface, which need to be informed after they
1288    * are loaded (as they depend on this callback to complete initialization work)
1289    */
1290   protected void informResourceLoaderAwareObjectsInChain(TokenizerChain chain) {
1291     CharFilterFactory[] charFilters = chain.getCharFilterFactories();
1292     for (CharFilterFactory next : charFilters) {
1293       if (next instanceof ResourceLoaderAware) {
1294         try {
1295           ((ResourceLoaderAware) next).inform(loader);
1296         } catch (IOException e) {
1297           throw new SolrException(ErrorCode.SERVER_ERROR, e);
1298         }
1299         }
1300     }
1301 
1302     TokenizerFactory tokenizerFactory = chain.getTokenizerFactory();
1303     if (tokenizerFactory instanceof ResourceLoaderAware) {
1304       try {
1305         ((ResourceLoaderAware) tokenizerFactory).inform(loader);
1306       } catch (IOException e) {
1307         throw new SolrException(ErrorCode.SERVER_ERROR, e);
1308       }
1309     }
1310 
1311     TokenFilterFactory[] filters = chain.getTokenFilterFactories();
1312     for (TokenFilterFactory next : filters) {
1313       if (next instanceof ResourceLoaderAware) {
1314         try {
1315           ((ResourceLoaderAware) next).inform(loader);
1316         } catch (IOException e) {
1317           throw new SolrException(ErrorCode.SERVER_ERROR, e);
1318         }
1319       }
1320     }
1321   }

1334    * Makes a shallow copy of this schema.
1335    * 
1336    * Not copied: analyzers 
1337    * 
1338    * @param includeFieldDataStructures if true, fields, fieldsWithDefaultValue, and requiredFields
1339    *                                   are copied; otherwise, they are not.
1340    * @return A shallow copy of this schema
1341    */
1342    ManagedIndexSchema shallowCopy(boolean includeFieldDataStructures) {

1344     try {
1345       newSchema = new ManagedIndexSchema
1346           (solrConfig, loader, isMutable, managedSchemaResourceName, schemaZkVersion, getSchemaUpdateLock());

1348       final String msg = "Error instantiating ManagedIndexSchema";
1349       log.error(msg, e);
1350       throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, msg, e);
1351     } catch (InterruptedException e) {
1352       // Restore the interrupted status
1353       Thread.currentThread().interrupt();
1354       log.warn("", e);
1355     }
1356 
1357     assert newSchema != null;
1358     
1359     newSchema.name = name;
1360     newSchema.version = version;
1361     newSchema.defaultSearchFieldName = defaultSearchFieldName;
1362     newSchema.queryParserDefaultOperator = queryParserDefaultOperator;
1363     newSchema.isExplicitQueryParserDefaultOperator = isExplicitQueryParserDefaultOperator;
1364     newSchema.similarity = similarity;

1366     newSchema.isExplicitSimilarity = isExplicitSimilarity;
1367     newSchema.uniqueKeyField = uniqueKeyField;
1368     newSchema.uniqueKeyFieldName = uniqueKeyFieldName;
1369     newSchema.uniqueKeyFieldType = uniqueKeyFieldType;
1370     
1371     // After the schema is persisted, resourceName is the same as managedSchemaResourceName
1372     newSchema.resourceName = managedSchemaResourceName;
1373 
1374     if (includeFieldDataStructures) {
1375       // These need new collections, since addFields() can add members to them
1376       newSchema.fields.putAll(fields);
1377       newSchema.fieldsWithDefaultValue.addAll(fieldsWithDefaultValue);
1378       newSchema.requiredFields.addAll(requiredFields);

1380

1384     newSchema.dynamicCopyFields = dynamicCopyFields;
1385     newSchema.copyFieldsMap = copyFieldsMap;
