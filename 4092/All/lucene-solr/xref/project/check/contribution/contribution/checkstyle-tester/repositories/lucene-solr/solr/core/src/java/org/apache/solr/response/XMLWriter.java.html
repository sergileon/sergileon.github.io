62    public static void writeResponse(Writer writer, SolrQueryRequest req, SolrQueryResponse rsp) throws IOException {
63      XMLWriter xmlWriter = null;
64      try {
65        xmlWriter = new XMLWriter(writer, req, rsp);
66        xmlWriter.writeResponse();
67      } finally {
68        xmlWriter.close();
69      }
70    }
71  
72    public XMLWriter(Writer writer, SolrQueryRequest req, SolrQueryResponse rsp) {
73      super(writer, req, rsp);
74  
75      String version = req.getParams().get(CommonParams.VERSION);
76      float ver = version==null? CURRENT_VERSION : Float.parseFloat(version);
77      this.version = (int)(ver*1000);
78      if( this.version < 2200 ) {
79        throw new SolrException( SolrException.ErrorCode.BAD_REQUEST,
80            "XMLWriter does not support version: "+version );
81      }
82    }

86    public void writeResponse() throws IOException {
87      writer.write(XML_START1);
88  
89      String stylesheet = req.getParams().get("stylesheet");
90      if (stylesheet != null && stylesheet.length() > 0) {
91        writer.write(XML_STYLESHEET);
92        XML.escapeAttributeValue(stylesheet, writer);
93        writer.write(XML_STYLESHEET_END);
94      }
95  
96      /*
97      String noSchema = req.getParams().get("noSchema");
98      // todo - change when schema becomes available?
99      if (false && noSchema == null)
100       writer.write(XML_START2_SCHEMA);
101     else
102       writer.write(XML_START2_NOSCHEMA);
103      ***/
104     writer.write(XML_START2_NOSCHEMA);
105 
106     // dump response values
107     NamedList<?> lst = rsp.getValues();
108     Boolean omitHeader = req.getParams().getBool(CommonParams.OMIT_HEADER);
109     if(omitHeader != null && omitHeader) lst.remove("responseHeader");
110     int sz = lst.size();
111     int start=0;
112 
113     for (int i=start; i<sz; i++) {
114       writeVal(lst.getName(i),lst.getVal(i));
115     }
116 
117     writer.write("\n</response>\n");
118   }

124   /** Writes the XML attribute name/val. A null val means that the attribute is missing. */
125   private void writeAttr(String name, String val) throws IOException {
126     writeAttr(name, val, true);
127   }
128 
129   public void writeAttr(String name, String val, boolean escape) throws IOException{
130     if (val != null) {
131       writer.write(' ');
132       writer.write(name);
133       writer.write("=\"");
134       if(escape){
135         XML.escapeAttributeValue(val, writer);
136       } else {
137         writer.write(val);
138       }
139       writer.write('"');
140     }
141   }
142 
143   void startTag(String tag, String name, boolean closeTag) throws IOException {
144     if (doIndent) indent();
145 
146     writer.write('<');
147     writer.write(tag);
148     if (name!=null) {
149       writeAttr(NAME, name);
150       if (closeTag) {
151         writer.write("/>");
152       } else {
153         writer.write(">");
154       }
155     } else {
156       if (closeTag) {
157         writer.write("/>");
158       } else {
159         writer.write('>');
160       }
161     }
162   }

165   @Override
166   public void writeStartDocumentList(String name, 
167       long start, int size, long numFound, Float maxScore) throws IOException
168   {
169     if (doIndent) indent();
170 
171     writer.write("<result");
172     writeAttr(NAME, name);
173     writeAttr("numFound",Long.toString(numFound));
174     writeAttr("start",Long.toString(start));
175     if(maxScore!=null) {
176       writeAttr("maxScore",Float.toString(maxScore));
177     }
178     writer.write(">");
179     
180     incLevel();
181   }

184   /**
185    * The SolrDocument should already have multivalued fields implemented as
186    * Collections -- this will not rewrite to &lt;arr&gt;
187    */ 
188   @Override
189   public void writeSolrDocument(String name, SolrDocument doc, ReturnFields returnFields, int idx ) throws IOException {
190     startTag("doc", name, false);
191     incLevel();
192 
193     for (String fname : doc.getFieldNames()) {
194       if (returnFields!= null && !returnFields.wantsField(fname)) {
195         continue;
196       }
197 
198       Object val = doc.getFieldValue(fname);
199       if( "_explain_".equals( fname ) ) {
200         System.out.println( val );
201       }
202       writeVal(fname, val);
203     }
204 
205     if(doc.hasChildDocuments()) {
206       for(SolrDocument childDoc : doc.getChildDocuments()) {
207         writeSolrDocument(null, childDoc, new SolrReturnFields(), idx);
208       }
209     }
210     
211     decLevel();
212     writer.write("</doc>");
213   }
214   
215   @Override
216   public void writeEndDocumentList() throws IOException
217   {
218     decLevel();
219     if (doIndent) indent();
220     writer.write("</result>");
221   }

229   @Override
230   public void writeNamedList(String name, NamedList val) throws IOException {
231     int sz = val.size();
232     startTag("lst", name, sz<=0);
233 
234     incLevel();
235     for (int i=0; i<sz; i++) {
236       writeVal(val.getName(i),val.getVal(i));
237     }
238     decLevel();
239 
240     if (sz > 0) {
241       if (doIndent) indent();
242       writer.write("</lst>");
243     }
244   }
245 
246   @Override
247   public void writeMap(String name, Map map, boolean excludeOuter, boolean isFirstVal) throws IOException {
248     int sz = map.size();
249 
250     if (!excludeOuter) {
251       startTag("lst", name, sz<=0);
252       incLevel();
253     }
254 
255     for (Map.Entry entry : (Set<Map.Entry>)map.entrySet()) {
256       Object k = entry.getKey();
257       Object v = entry.getValue();
258       // if (sz<indentThreshold) indent();
259       writeVal( null == k ? null : k.toString(), v);
260     }
261 
262     if (!excludeOuter) {
263       decLevel();
264       if (sz > 0) {
265         if (doIndent) indent();
266         writer.write("</lst>");
267       }
268     }
269   }

271   @Override
272   public void writeArray(String name, Object[] val) throws IOException {
273     writeArray(name, Arrays.asList(val).iterator());
274   }

276   @Override
277   public void writeArray(String name, Iterator iter) throws IOException {
278     if( iter.hasNext() ) {
279       startTag("arr", name, false );
280       incLevel();
281       while( iter.hasNext() ) {
282         writeVal(null, iter.next());
283       }
284       decLevel();
285       if (doIndent) indent();
286       writer.write("</arr>");
287     }
288     else {
289       startTag("arr", name, true );
290     }
291   }

349   //
350   // OPT - specific writeInt, writeFloat, methods might be faster since
351   // there would be less write calls (write("<int name=\"" + name + ... + </int>)
352   //
353   private void writePrim(String tag, String name, String val, boolean escape) throws IOException {
354     int contentLen = val==null ? 0 : val.length();
355 
356     startTag(tag, name, contentLen==0);
357     if (contentLen==0) return;
358 
359     if (escape) {
360       XML.escapeCharData(val,writer);
361     } else {
362       writer.write(val,0,contentLen);
363     }
364 
365     writer.write('<');
366     writer.write('/');
367     writer.write(tag);
368     writer.write('>');
369   }
