271   private void simpleQuery(Object... queryParams) throws SolrServerException, IOException {
272     ModifiableSolrParams params = new ModifiableSolrParams();
273     for (int i = 0; i < queryParams.length; i += 2) {
274       params.add(queryParams[i].toString(), queryParams[i + 1].toString());
275     }
276     params.set("shards", shards);
277     queryServer(params);
278   }
