324     int i = 0;       /* index in packed string  */
325     int j = offset;  /* index in unpacked array */
326     int l = packed.length();
327     while (i < l) {
328       int count = packed.charAt(i++);

331       do result[j++] = value; while (--count > 0);
332     }
333     return j;

543   /**
544    * Refills the input buffer.
545    *
546    * @return      <code>false</code>, iff there was new input.
547    * 
548    * @exception   java.io.IOException  if any I/O-Error occurs
549    */
550   private boolean zzRefill() throws java.io.IOException {
551 
552     /* first: make room (if you can) */
553     if (zzStartRead > 0) {
554       zzEndRead += zzFinalHighSurrogate;
555       zzFinalHighSurrogate = 0;
556       System.arraycopy(zzBuffer, zzStartRead,
557                        zzBuffer, 0,
558                        zzEndRead-zzStartRead);
559 
560       /* translate stored positions */
561       zzEndRead-= zzStartRead;
562       zzCurrentPos-= zzStartRead;
563       zzMarkedPos-= zzStartRead;
564       zzStartRead = 0;
565     }
566 
567     /* is the buffer big enough? */
568     if (zzCurrentPos >= zzBuffer.length - zzFinalHighSurrogate) {
569       /* if not: blow it up */
570       char newBuffer[] = new char[zzBuffer.length*2];
571       System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
572       zzBuffer = newBuffer;
573       zzEndRead += zzFinalHighSurrogate;
574       zzFinalHighSurrogate = 0;
575     }
576 
577     /* fill the buffer with new input */
578     int requested = zzBuffer.length - zzEndRead;           
579     int totalRead = 0;
580     while (totalRead < requested) {
581       int numRead = zzReader.read(zzBuffer, zzEndRead + totalRead, requested - totalRead);
582       if (numRead == -1) {
583         break;
584       }
585       totalRead += numRead;
586     }
587 
588     if (totalRead > 0) {
589       zzEndRead += totalRead;
590       if (totalRead == requested) { /* possibly more input available */
591         if (Character.isHighSurrogate(zzBuffer[zzEndRead - 1])) {
592           --zzEndRead;
593           zzFinalHighSurrogate = 1;
594         }
595       }
596       return false;
597     }
598 
599     // totalRead = 0: End of stream
600     return true;
601   }

736   /**
737    * Resumes scanning until the next regular expression is matched,
738    * the end of input is encountered or an I/O-Error occurs.
739    *
740    * @return      the next token
741    * @exception   java.io.IOException  if any I/O-Error occurs
742    */
743   public int getNextToken() throws java.io.IOException {
744     int zzInput;
745     int zzAction;
746 
747     // cached fields:
748     int zzCurrentPosL;
749     int zzMarkedPosL;
750     int zzEndReadL = zzEndRead;
751     char [] zzBufferL = zzBuffer;
752     char [] zzCMapL = ZZ_CMAP;
753 
754     int [] zzTransL = ZZ_TRANS;
755     int [] zzRowMapL = ZZ_ROWMAP;
756     int [] zzAttrL = ZZ_ATTRIBUTE;
757 
758     while (true) {
759       zzMarkedPosL = zzMarkedPos;
760 
761       yychar+= zzMarkedPosL-zzStartRead;
762 
763       zzAction = -1;
764 
765       zzCurrentPosL = zzCurrentPos = zzStartRead = zzMarkedPosL;
766   
767       zzState = ZZ_LEXSTATE[zzLexicalState];
768 
769       // set up zzAction for empty match case:
770       int zzAttributes = zzAttrL[zzState];
771       if ( (zzAttributes & 1) == 1 ) {
772         zzAction = zzState;
773       }
774 
775 
776       zzForAction: {
777         while (true) {
778     
779           if (zzCurrentPosL < zzEndReadL) {
780             zzInput = Character.codePointAt(zzBufferL, zzCurrentPosL, zzEndReadL);
781             zzCurrentPosL += Character.charCount(zzInput);
782           }
783           else if (zzAtEOF) {
784             zzInput = YYEOF;
785             break zzForAction;
786           }
787           else {
788             // store back cached positions
789             zzCurrentPos  = zzCurrentPosL;
790             zzMarkedPos   = zzMarkedPosL;
791             boolean eof = zzRefill();
792             // get translated positions and possibly new buffer
793             zzCurrentPosL  = zzCurrentPos;
794             zzMarkedPosL   = zzMarkedPos;
795             zzBufferL      = zzBuffer;
796             zzEndReadL     = zzEndRead;
797             if (eof) {
798               zzInput = YYEOF;
799               break zzForAction;
800             }
801             else {
802               zzInput = Character.codePointAt(zzBufferL, zzCurrentPosL, zzEndReadL);
803               zzCurrentPosL += Character.charCount(zzInput);
804             }
805           }
806           int zzNext = zzTransL[ zzRowMapL[zzState] + zzCMapL[zzInput] ];

808           zzState = zzNext;
809 
810           zzAttributes = zzAttrL[zzState];

812             zzAction = zzState;
813             zzMarkedPosL = zzCurrentPosL;
814             if ( (zzAttributes & 8) == 8 ) break zzForAction;
815           }
