62  /**
63   * Writer for {@link MemoryDocValuesFormat}
64   */
65  class MemoryDocValuesConsumer extends DocValuesConsumer {
66    IndexOutput data, meta;
67    final int maxDoc;
68    final float acceptableOverheadRatio;
69    
70    MemoryDocValuesConsumer(SegmentWriteState state, String dataCodec, String dataExtension, String metaCodec, String metaExtension, float acceptableOverheadRatio) throws IOException {
71      this.acceptableOverheadRatio = acceptableOverheadRatio;
72      maxDoc = state.segmentInfo.maxDoc();
73      boolean success = false;
74      try {
75        String dataName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, dataExtension);
76        data = state.directory.createOutput(dataName, state.context);
77        CodecUtil.writeIndexHeader(data, dataCodec, VERSION_CURRENT, state.segmentInfo.getId(), state.segmentSuffix);
78        String metaName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, metaExtension);
79        meta = state.directory.createOutput(metaName, state.context);
80        CodecUtil.writeIndexHeader(meta, metaCodec, VERSION_CURRENT, state.segmentInfo.getId(), state.segmentSuffix);
81        success = true;
82      } finally {
83        if (!success) {
84          IOUtils.closeWhileHandlingException(this);
85        }
86      }
87    }

94    void addNumericField(FieldInfo field, Iterable<Number> values, boolean optimizeStorage) throws IOException {
95      meta.writeVInt(field.number);
96      meta.writeByte(NUMBER);
97      meta.writeLong(data.getFilePointer());
98      long minValue = Long.MAX_VALUE;
99      long maxValue = Long.MIN_VALUE;
100     long blockSum = 0;
101     long gcd = 0;
102     boolean missing = false;
103     // TODO: more efficient?
104     HashSet<Long> uniqueValues = null;
105     long count = 0;
106 
107     if (optimizeStorage) {
108       uniqueValues = new HashSet<>();
109 
110       long currentBlockMin = Long.MAX_VALUE;
111       long currentBlockMax = Long.MIN_VALUE;
112       for (Number nv : values) {
113         final long v;
114         if (nv == null) {
115           v = 0;
116           missing = true;
117         } else {
118           v = nv.longValue();
119         }
120 
121         if (gcd != 1) {
122           if (v < Long.MIN_VALUE / 2 || v > Long.MAX_VALUE / 2) {
123             // in that case v - minValue might overflow and make the GCD computation return
124             // wrong results. Since these extreme values are unlikely, we just discard
125             // GCD computation for them
126             gcd = 1;
127           } else if (count != 0) { // minValue needs to be set first
128             gcd = MathUtil.gcd(gcd, v - minValue);
129           }
130         }
131 
132         currentBlockMin = Math.min(minValue, v);
133         currentBlockMax = Math.max(maxValue, v);
134         
135         minValue = Math.min(minValue, v);
136         maxValue = Math.max(maxValue, v);
137 
138         if (uniqueValues != null) {
139           if (uniqueValues.add(v)) {
140             if (uniqueValues.size() > 256) {
141               uniqueValues = null;
142             }
143           }
144         }
145 
146         ++count;
147         if (count % BLOCK_SIZE == 0) {
148           final long blockDelta = currentBlockMax - currentBlockMin;
149           final int blockDeltaRequired = PackedInts.unsignedBitsRequired(blockDelta);
150           final int blockBPV = PackedInts.fastestFormatAndBits(BLOCK_SIZE, blockDeltaRequired, acceptableOverheadRatio).bitsPerValue;
151           blockSum += blockBPV;
152           currentBlockMax = Long.MIN_VALUE;
153           currentBlockMin = Long.MAX_VALUE;
154         }
155       }
156     } else {
157       for (Number nv : values) {
158         long v = nv.longValue();
159         maxValue = Math.max(v, maxValue);
160         minValue = Math.min(v, minValue);
161         count++;
162       }
163     }
164     
165     if (missing) {
166       long start = data.getFilePointer();
167       writeMissingBitset(values);
168       meta.writeLong(start);
169       meta.writeLong(data.getFilePointer() - start);
170     } else {
171       meta.writeLong(-1L);
172     }
173     
174     final long delta = maxValue - minValue;
175     final int deltaRequired = delta < 0 ? 64 : PackedInts.bitsRequired(delta);
176     final FormatAndBits deltaBPV = PackedInts.fastestFormatAndBits(maxDoc, deltaRequired, acceptableOverheadRatio);
177         
178     final FormatAndBits tableBPV;
179     if (count < Integer.MAX_VALUE && uniqueValues != null) {
180       tableBPV = PackedInts.fastestFormatAndBits(maxDoc, PackedInts.bitsRequired(uniqueValues.size()-1), acceptableOverheadRatio);
181     } else {
182       tableBPV = null;
183     }
184     
185     final FormatAndBits gcdBPV;
186     if (count < Integer.MAX_VALUE && gcd != 0 && gcd != 1) {
187       final long gcdDelta = (maxValue - minValue) / gcd;
188       final int gcdRequired = gcdDelta < 0 ? 64 : PackedInts.bitsRequired(gcdDelta);
189       gcdBPV = PackedInts.fastestFormatAndBits(maxDoc, gcdRequired, acceptableOverheadRatio);
190     } else {
191       gcdBPV = null;
192     }
193     
194     boolean doBlock = false;
195     if (blockSum != 0) {
196       int numBlocks = maxDoc / BLOCK_SIZE;
197       float avgBPV = blockSum / (float)numBlocks;
198       // just a heuristic, with tiny amounts of blocks our estimate is skewed as we ignore the final "incomplete" block.
199       // with at least 4 blocks it's pretty accurate. The difference must also be significant (according to acceptable overhead).
200       if (numBlocks >= 4 && (avgBPV+avgBPV*acceptableOverheadRatio) < deltaBPV.bitsPerValue) {
201         doBlock = true;
202       }
203     }
204     // blockpackedreader allows us to read in huge streams of ints
205     if (count >= Integer.MAX_VALUE) {
206       doBlock = true;
207     }
208     
209     if (tableBPV != null && (tableBPV.bitsPerValue+tableBPV.bitsPerValue*acceptableOverheadRatio) < deltaBPV.bitsPerValue) {
210       // small number of unique values
211       meta.writeByte(TABLE_COMPRESSED); // table-compressed
212       Long[] decode = uniqueValues.toArray(new Long[uniqueValues.size()]);
213       final HashMap<Long,Integer> encode = new HashMap<>();
214       int length = 1 << tableBPV.bitsPerValue;
215       data.writeVInt(length);
216       for (int i = 0; i < decode.length; i++) {
217         data.writeLong(decode[i]);
218         encode.put(decode[i], i);
219       }
220       for (int i = decode.length; i < length; i++) {
221         data.writeLong(0);
222       }
223       
224       meta.writeVInt(PackedInts.VERSION_CURRENT);
225       meta.writeLong(count);
226       data.writeVInt(tableBPV.format.getId());
227       data.writeVInt(tableBPV.bitsPerValue);
228       
229       final PackedInts.Writer writer = PackedInts.getWriterNoHeader(data, tableBPV.format, (int)count, tableBPV.bitsPerValue, PackedInts.DEFAULT_BUFFER_SIZE);
230       for(Number nv : values) {
231         writer.add(encode.get(nv == null ? 0 : nv.longValue()));
232       }
233       writer.finish();
234     } else if (gcdBPV != null && (gcdBPV.bitsPerValue+gcdBPV.bitsPerValue*acceptableOverheadRatio) < deltaBPV.bitsPerValue) {
235       meta.writeByte(GCD_COMPRESSED);
236       meta.writeVInt(PackedInts.VERSION_CURRENT);
237       meta.writeLong(count);
238       data.writeLong(minValue);
239       data.writeLong(gcd);
240       data.writeVInt(gcdBPV.format.getId());
241       data.writeVInt(gcdBPV.bitsPerValue);
242 
243       final PackedInts.Writer writer = PackedInts.getWriterNoHeader(data, gcdBPV.format, (int)count, gcdBPV.bitsPerValue, PackedInts.DEFAULT_BUFFER_SIZE);
244       for (Number nv : values) {
245         long value = nv == null ? 0 : nv.longValue();
246         writer.add((value - minValue) / gcd);
247       }
248       writer.finish();
249     } else if (doBlock) {
250       meta.writeByte(BLOCK_COMPRESSED); // block delta-compressed
251       meta.writeVInt(PackedInts.VERSION_CURRENT);
252       meta.writeLong(count);
253       data.writeVInt(BLOCK_SIZE);
254       final BlockPackedWriter writer = new BlockPackedWriter(data, BLOCK_SIZE);
255       for (Number nv : values) {
256         writer.add(nv == null ? 0 : nv.longValue());
257       }
258       writer.finish();
259     } else {
260       meta.writeByte(DELTA_COMPRESSED); // delta-compressed
261       meta.writeVInt(PackedInts.VERSION_CURRENT);
262       meta.writeLong(count);
263       final long minDelta = deltaBPV.bitsPerValue == 64 ? 0 : minValue;
264       data.writeLong(minDelta);
265       data.writeVInt(deltaBPV.format.getId());
266       data.writeVInt(deltaBPV.bitsPerValue);
267 
268       final PackedInts.Writer writer = PackedInts.getWriterNoHeader(data, deltaBPV.format, (int)count, deltaBPV.bitsPerValue, PackedInts.DEFAULT_BUFFER_SIZE);
269       for (Number nv : values) {
270         long v = nv == null ? 0 : nv.longValue();
271         writer.add(v - minDelta);
272       }
273       writer.finish();
274     }
275   }

277   @Override
278   public void close() throws IOException {
279     boolean success = false;
280     try {
281       if (meta != null) {
282         meta.writeVInt(-1); // write EOF marker
283         CodecUtil.writeFooter(meta); // write checksum
284       }
285       if (data != null) {
286         CodecUtil.writeFooter(data);
287       }
288       success = true;
289     } finally {
290       if (success) {
291         IOUtils.close(data, meta);
292       } else {
293         IOUtils.closeWhileHandlingException(data, meta);
294       }
295       data = meta = null;
296     }
297   }

299   @Override
300   public void addBinaryField(FieldInfo field, final Iterable<BytesRef> values) throws IOException {
301     // write the byte[] data
302     meta.writeVInt(field.number);
303     meta.writeByte(BYTES);
304     int minLength = Integer.MAX_VALUE;
305     int maxLength = Integer.MIN_VALUE;
306     final long startFP = data.getFilePointer();
307     boolean missing = false;
308     int upto = 0;
309     for(BytesRef v : values) {
310       final int length;
311       if (v == null) {
312         length = 0;
313         missing = true;
314       } else {
315         length = v.length;
316       }
317       if (length > MemoryDocValuesFormat.MAX_BINARY_FIELD_LENGTH) {
318         throw new IllegalArgumentException("DocValuesField \"" + field.name + "\" is too large, must be <= " + MemoryDocValuesFormat.MAX_BINARY_FIELD_LENGTH + " but got length=" + length + " v=" + v + "; upto=" + upto + " values=" + values);
319       }
320       upto++;
321       minLength = Math.min(minLength, length);
322       maxLength = Math.max(maxLength, length);
323       if (v != null) {
324         data.writeBytes(v.bytes, v.offset, v.length);
325       }
326     }
327     meta.writeLong(startFP);
328     meta.writeLong(data.getFilePointer() - startFP);
329     if (missing) {
330       long start = data.getFilePointer();
331       writeMissingBitset(values);
332       meta.writeLong(start);
333       meta.writeLong(data.getFilePointer() - start);
334     } else {
335       meta.writeLong(-1L);
336     }
337     meta.writeVInt(minLength);
338     meta.writeVInt(maxLength);
339     
340     // if minLength == maxLength, it's a fixed-length byte[], we are done (the addresses are implicit)
341     // otherwise, we need to record the length fields...
342     if (minLength != maxLength) {
343       meta.writeVInt(PackedInts.VERSION_CURRENT);
344       meta.writeVInt(BLOCK_SIZE);
345 
346       final MonotonicBlockPackedWriter writer = new MonotonicBlockPackedWriter(data, BLOCK_SIZE);
347       long addr = 0;
348       for (BytesRef v : values) {
349         if (v != null) {
350           addr += v.length;
351         }
352         writer.add(addr);
353       }
354       writer.finish();
355     }
356   }

358   private void writeFST(FieldInfo field, Iterable<BytesRef> values) throws IOException {
359     meta.writeVInt(field.number);
360     meta.writeByte(FST);
361     meta.writeLong(data.getFilePointer());
362     PositiveIntOutputs outputs = PositiveIntOutputs.getSingleton();
363     Builder<Long> builder = new Builder<>(INPUT_TYPE.BYTE1, outputs);
364     IntsRefBuilder scratch = new IntsRefBuilder();
365     long ord = 0;
366     for (BytesRef v : values) {
367       builder.add(Util.toIntsRef(v, scratch), ord);
368       ord++;
369     }
370     FST<Long> fst = builder.finish();
371     if (fst != null) {
372       fst.save(data);
373     }
374     meta.writeVLong(ord);
375   }

377   // TODO: in some cases representing missing with minValue-1 wouldn't take up additional space and so on,
378   // but this is very simple, and algorithms only check this for values of 0 anyway (doesnt slow down normal decode)
379   void writeMissingBitset(Iterable<?> values) throws IOException {
380     long bits = 0;
381     int count = 0;
382     for (Object v : values) {
383       if (count == 64) {
384         data.writeLong(bits);
385         count = 0;
386         bits = 0;
387       }
388       if (v != null) {
389         bits |= 1L << (count & 0x3f);
390       }
391       count++;
392     }
393     if (count > 0) {
394       data.writeLong(bits);
395     }
396   }

398   @Override
399   public void addSortedField(FieldInfo field, Iterable<BytesRef> values, Iterable<Number> docToOrd) throws IOException {
400     // write the ordinals as numerics
401     addNumericField(field, docToOrd, false);
402     
403     // write the values as FST
404     writeFST(field, values);
405   }

407   @Override
408   public void addSortedNumericField(FieldInfo field, Iterable<Number> docToValueCount, Iterable<Number> values) throws IOException {
409     meta.writeVInt(field.number);
410     
411     if (isSingleValued(docToValueCount)) {
412       meta.writeByte(SORTED_NUMERIC_SINGLETON);
413       addNumericField(field, singletonView(docToValueCount, values, null), true);
414     } else {
415       meta.writeByte(SORTED_NUMERIC);
416       
417       // write the addresses:
418       meta.writeVInt(PackedInts.VERSION_CURRENT);
419       meta.writeVInt(BLOCK_SIZE);
420       meta.writeLong(data.getFilePointer());
421       final MonotonicBlockPackedWriter writer = new MonotonicBlockPackedWriter(data, BLOCK_SIZE);
422       long addr = 0;
423       writer.add(addr);
424       for (Number v : docToValueCount) {
425         addr += v.longValue();
426         writer.add(addr);
427       }
428       writer.finish();
429       long valueCount = writer.ord();
430       meta.writeLong(valueCount);
431       
432       // write the values
433       addNumericField(field, values, true);
434     }
435   }

437   // note: this might not be the most efficient... but it's fairly simple
438   @Override
439   public void addSortedSetField(FieldInfo field, Iterable<BytesRef> values, final Iterable<Number> docToOrdCount, final Iterable<Number> ords) throws IOException {
440     meta.writeVInt(field.number);
441     
442     if (isSingleValued(docToOrdCount)) {
443       meta.writeByte(SORTED_SET_SINGLETON);
444       addSortedField(field, values, singletonView(docToOrdCount, ords, -1L));
445     } else {
446       meta.writeByte(SORTED_SET);
447       // write the ordinals as a binary field
448       addBinaryField(field, new Iterable<BytesRef>() {
449         @Override
450         public Iterator<BytesRef> iterator() {
451           return new SortedSetIterator(docToOrdCount.iterator(), ords.iterator());
452         }
453       });
454       
455       // write the values as FST
456       writeFST(field, values);
457     }
458   }

460   // per-document vint-encoded byte[]
461   static class SortedSetIterator implements Iterator<BytesRef> {
462     byte[] buffer = new byte[10];
463     ByteArrayDataOutput out = new ByteArrayDataOutput();
464     BytesRef ref = new BytesRef();
465     
466     final Iterator<Number> counts;
467     final Iterator<Number> ords;
468     
469     SortedSetIterator(Iterator<Number> counts, Iterator<Number> ords) {
470       this.counts = counts;
471       this.ords = ords;
472     }
473     
474     @Override
475     public boolean hasNext() {
476       return counts.hasNext();
477     }
478 
479     @Override
480     public BytesRef next() {
481       if (!hasNext()) {
482         throw new NoSuchElementException();
483       }
484       
485       int count = counts.next().intValue();
486       int maxSize = count*9; // worst case
487       if (maxSize > buffer.length) {
488         buffer = ArrayUtil.grow(buffer, maxSize);
489       }
490       
491       try {
492         encodeValues(count);
493       } catch (IOException bogus) {
494         throw new RuntimeException(bogus);
495       }
496       
497       ref.bytes = buffer;
498       ref.offset = 0;
499       ref.length = out.getPosition();
500 
501       return ref;
502     }
503     
504     // encodes count values to buffer
505     private void encodeValues(int count) throws IOException {
506       out.reset(buffer);
507       long lastOrd = 0;
508       for (int i = 0; i < count; i++) {
509         long ord = ords.next().longValue();
510         out.writeVLong(ord - lastOrd);
511         lastOrd = ord;
512       }
513     }
514 
515     @Override
516     public void remove() {
517       throw new UnsupportedOperationException();
518     }
519   }
