102   public void testTrailingOperators() throws Exception {
103     // really just test that exceptions aren't thrown by
104     // single + -
105 
106     assertJQ(req("defType","edismax", "q","-")
107         ,"/response==");
108 
109     assertJQ(req("defType","edismax", "q","+")
110         ,"/response==");
111 
112     assertJQ(req("defType","edismax", "q","+ - +")
113         ,"/response==");
114 
115     assertJQ(req("defType","edismax", "q","- + -")
116         ,"/response==");
117 
118     assertJQ(req("defType","edismax", "q","id:47 +")
119         ,"/response/numFound==1");
120 
121     assertJQ(req("defType","edismax", "q","id:47 -")
122         ,"/response/numFound==1");
123 
124     Random r = random();
125     for (int i=0; i<100; i++) {
126       StringBuilder sb = new StringBuilder();
127       for (int j=0; j<r.nextInt(10); j++) {
128         switch (r.nextInt(3)) {
129           case 0: sb.append(' '); break;
130           case 1: sb.append('+'); break;
131           case 2: sb.append('-'); break;
132           case 3: sb.append((char)r.nextInt(127)); break;
133         }
134       }
135 
136       String q = sb.toString();
137       assertJQ(req("defType","edismax", "q",q)
138           ,"/response==");
139     }

443   public void testUserFields() {
444     String allr = "*[count(//doc)=10]";
445     String oner = "*[count(//doc)=1]";
446     String nor = "*[count(//doc)=0]";

448     // User fields
449     // Default is allow all "*"
450     // If a list of fields are given, only those are allowed "foo bar"
451     // Possible to invert with "-" syntax:
452     //   Disallow all: "-*"
453     //   Allow all but id: "* -id"
454     // Also supports "dynamic" field name wildcarding
455     assertQ(req("defType","edismax", "q","id:42"),

537     String oner = "*[count(//doc)=1]";
538     String twor = "*[count(//doc)=2]";
539     String nor = "*[count(//doc)=0]";
540     
541  // Aliasing
542     // Single field
543     assertQ(req("defType","edismax", "q","myalias:Zapp"),
544         nor);
545     
546     assertQ(req("defType","edismax", "q","myalias:Zapp", "f.myalias.qf","name"),
547         oner);
548     
549     // Multi field
550     assertQ(req("defType","edismax", "uf", "myalias", "q","myalias:(Zapp Obnoxious)", "f.myalias.qf","name^2.0 mytrait_ss^5.0", "mm", "50%"),
551         oner);
552     
553     // Multi field
554     assertQ(req("defType","edismax", "q","Zapp Obnoxious", "f.myalias.qf","name^2.0 mytrait_ss^5.0"),

1056   private boolean containsClause(Query query, String field, String value,
1057       int boost, boolean fuzzy) {
1058 
1059     float queryBoost = 1f;
1060     if (query instanceof BoostQuery) {
1061       BoostQuery bq = (BoostQuery) query;
1062       query = bq.getQuery();
1063       queryBoost = bq.getBoost();
1064     }
1065 
1066     if(query instanceof BooleanQuery) {
1067       return containsClause((BooleanQuery)query, field, value, boost, fuzzy);
1068     }
1069     if(query instanceof DisjunctionMaxQuery) {
1070       return containsClause((DisjunctionMaxQuery)query, field, value, boost, fuzzy);
1071     }
1072     if (boost != queryBoost) {
1073       return false;
1074     }
1075     if(query instanceof TermQuery && !fuzzy) {
1076       return containsClause((TermQuery)query, field, value);
1077     }
1078     if(query instanceof FuzzyQuery && fuzzy) {
1079       return containsClause((FuzzyQuery)query, field, value);
1080     }
1081     return false;
1082   }

1086        query.getTerm().bytes().utf8ToString().equals(value)) {

1095         return true;

1112         return true;
