34  /**
35   * Test for {@link org.apache.solr.handler.component.TrackingShardHandlerFactory}
36   * See SOLR-7147 for more information
37   */
38  @SolrTestCaseJ4.SuppressSSL
39  public class TestTrackingShardHandlerFactory extends AbstractFullDistribZkTestBase {
40  
41    public TestTrackingShardHandlerFactory() {
42      schemaString = "schema15.xml"; // we need a string id
43    }
44  
45    @Override
46    protected String getSolrXml() {
47      return "solr-trackingshardhandler.xml";
48    }
49  
50    @Test
51    @BaseDistributedSearchTestCase.ShardsFixed(num = 2)
52    public void testRequestTracking() throws Exception {
53      String collectionName = "testTwoPhase";
54  
55      List<JettySolrRunner> runners = new ArrayList<>(jettys);
56      runners.add(controlJetty);
57  
58      TrackingShardHandlerFactory.RequestTrackingQueue trackingQueue = new TrackingShardHandlerFactory.RequestTrackingQueue();
59      TrackingShardHandlerFactory.setTrackingQueue(runners, trackingQueue);
60  
61      for (JettySolrRunner runner : runners) {
62        CoreContainer container = runner.getCoreContainer();
63        ShardHandlerFactory factory = container.getShardHandlerFactory();
64        assert factory instanceof TrackingShardHandlerFactory;
65        TrackingShardHandlerFactory trackingShardHandlerFactory = (TrackingShardHandlerFactory) factory;
66        assertSame(trackingQueue, trackingShardHandlerFactory.getTrackingQueue());
67      }
68  
69      createCollection(collectionName, 2, 1, 1);
70  
71      waitForRecoveriesToFinish(collectionName, true);
72  
73      List<TrackingShardHandlerFactory.ShardRequestAndParams> coreAdminRequests = trackingQueue.getCoreAdminRequests();
74      assertNotNull(coreAdminRequests);
75      assertEquals("Unexpected number of core admin requests were found", 2, coreAdminRequests.size());
76  
77      CloudSolrClient client = cloudClient;
78  
79      client.setDefaultCollection(collectionName);
80          /*
81          hash of b is 95de7e03 high bits=2 shard=shard1
82          hash of e is 656c4367 high bits=1 shard=shard2
83           */
84      for (int i = 0; i < 10; i++) {
85        SolrInputDocument doc = new SolrInputDocument();
86        doc.addField("id", (i % 2 == 0 ? "b!" : "e!") + i);
87        doc.addField("a_i", i);
88        doc.addField("a_t", "text_" + i);
89        client.add(doc);
90      }
91      client.commit();
92  
93      client.query(new SolrQuery("*:*"));
94  
95      TrackingShardHandlerFactory.ShardRequestAndParams getTopIdsRequest = trackingQueue.getShardRequestByPurpose(client.getZkStateReader(), collectionName, "shard1", ShardRequest.PURPOSE_GET_TOP_IDS);
96      assertNotNull(getTopIdsRequest);
97      getTopIdsRequest = trackingQueue.getShardRequestByPurpose(client.getZkStateReader(), collectionName, "shard2", ShardRequest.PURPOSE_GET_TOP_IDS);
98      assertNotNull(getTopIdsRequest);
99  
100     TrackingShardHandlerFactory.ShardRequestAndParams getFieldsRequest = trackingQueue.getShardRequestByPurpose(client.getZkStateReader(), collectionName, "shard1", ShardRequest.PURPOSE_GET_FIELDS);
101     assertNotNull(getFieldsRequest);
102     getFieldsRequest = trackingQueue.getShardRequestByPurpose(client.getZkStateReader(), collectionName, "shard2", ShardRequest.PURPOSE_GET_FIELDS);
103     assertNotNull(getFieldsRequest);
104 
105     int numRequests = 0;
106     Map<String, List<TrackingShardHandlerFactory.ShardRequestAndParams>> allRequests = trackingQueue.getAllRequests();
107     for (Map.Entry<String, List<TrackingShardHandlerFactory.ShardRequestAndParams>> entry : allRequests.entrySet()) {
108       numRequests += entry.getValue().size();
109     }
110     // 4 shard requests + 2 core admin requests (invoked by create collection API)
111     assertEquals("Total number of requests do not match expected", 6, numRequests);
112 
113     // reset
114     TrackingShardHandlerFactory.setTrackingQueue(runners, null);
115 
116     for (JettySolrRunner runner : runners) {
117       CoreContainer container = runner.getCoreContainer();
118       ShardHandlerFactory factory = container.getShardHandlerFactory();
119       assert factory instanceof TrackingShardHandlerFactory;
120       TrackingShardHandlerFactory trackingShardHandlerFactory = (TrackingShardHandlerFactory) factory;
121       assertFalse(trackingShardHandlerFactory.isTracking());
122     }
123 
124     // make another request and verify
125     client.query(new SolrQuery("*:*"));
126     numRequests = 0;
127     allRequests = trackingQueue.getAllRequests();
128     for (Map.Entry<String, List<TrackingShardHandlerFactory.ShardRequestAndParams>> entry : allRequests.entrySet()) {
129       numRequests += entry.getValue().size();
130     }
131     // should still be 6
132     assertEquals("Total number of shard requests do not match expected", 6, numRequests);
133   }
134 }
