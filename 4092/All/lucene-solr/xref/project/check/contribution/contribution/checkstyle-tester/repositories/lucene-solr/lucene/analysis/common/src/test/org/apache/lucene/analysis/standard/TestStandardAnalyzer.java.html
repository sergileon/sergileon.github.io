108   public void testHugeDoc() throws IOException {
109     StringBuilder sb = new StringBuilder();
110     char whitespace[] = new char[4094];
111     Arrays.fill(whitespace, ' ');
112     sb.append(whitespace);
113     sb.append("testing 1234");
114     String input = sb.toString();
115     StandardTokenizer tokenizer = new StandardTokenizer();
116     tokenizer.setReader(new StringReader(input));
117     BaseTokenStreamTestCase.assertTokenStreamContents(tokenizer, new String[] { "testing", "1234" });
118   }

329   /**
330    * Multiple consecutive chars in \p{WB:MidLetter}, \p{WB:MidNumLet},
331    * and/or \p{MidNum} should trigger a token split.
332    */
333   public void testMid() throws Exception {
334     // ':' is in \p{WB:MidLetter}, which should trigger a split unless there is a Letter char on both sides
335     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:B", new String[] { "A:B" });
336     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A::B", new String[] { "A", "B" });
337 
338     // '.' is in \p{WB:MidNumLet}, which should trigger a split unless there is a Letter or Numeric char on both sides
339     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.2", new String[] { "1.2" });
340     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.B", new String[] { "A.B" });
341     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1..2", new String[] { "1", "2" });
342     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A..B", new String[] { "A", "B" });
343 
344     // ',' is in \p{WB:MidNum}, which should trigger a split unless there is a Numeric char on both sides
345     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,2", new String[] { "1,2" });
346     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,,2", new String[] { "1", "2" });
347 
348     // Mixed consecutive \p{WB:MidLetter} and \p{WB:MidNumLet} should trigger a split
349     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.:B", new String[] { "A", "B" });
350     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:.B", new String[] { "A", "B" });
351 
352     // Mixed consecutive \p{WB:MidNum} and \p{WB:MidNumLet} should trigger a split
353     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,.2", new String[] { "1", "2" });
354     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.,2", new String[] { "1", "2" });
355 
356     // '_' is in \p{WB:ExtendNumLet}
357 
358     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:B_A:B", new String[] { "A:B_A:B" });
359     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A:B_A::B", new String[] { "A:B_A", "B" });
360 
361     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.2_1.2", new String[] { "1.2_1.2" });
362     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.B_A.B", new String[] { "A.B_A.B" });
363     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1.2_1..2", new String[] { "1.2_1", "2" });
364     BaseTokenStreamTestCase.assertAnalyzesTo(a, "A.B_A..B", new String[] { "A.B_A", "B" });
365 
366     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,2_1,2", new String[] { "1,2_1,2" });
367     BaseTokenStreamTestCase.assertAnalyzesTo(a, "1,2_1,,2", new String[] { "1,2_1", "2" });
368 
369     BaseTokenStreamTestCase.assertAnalyzesTo(a, "C_A.:B", new String[] { "C_A", "B" });
370     BaseTokenStreamTestCase.assertAnalyzesTo(a, "C_A:.B", new String[] { "C_A", "B" });
371 
372     BaseTokenStreamTestCase.assertAnalyzesTo(a, "3_1,.2", new String[] { "3_1", "2" });
373     BaseTokenStreamTestCase.assertAnalyzesTo(a, "3_1.,2", new String[] { "3_1", "2" });
374   }
