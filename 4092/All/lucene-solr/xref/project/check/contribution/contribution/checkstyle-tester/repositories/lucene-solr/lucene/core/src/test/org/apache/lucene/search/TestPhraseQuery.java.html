308   public void testSlopScoring() throws IOException {
309     Directory directory = newDirectory();
310     RandomIndexWriter writer = new RandomIndexWriter(random(), directory, 
311         newIndexWriterConfig(new MockAnalyzer(random()))
312           .setMergePolicy(newLogMergePolicy())
313           .setSimilarity(new DefaultSimilarity()));
314 
315     Document doc = new Document();
316     doc.add(newTextField("field", "foo firstname lastname foo", Field.Store.YES));
317     writer.addDocument(doc);
318     
319     Document doc2 = new Document();
320     doc2.add(newTextField("field", "foo firstname zzz lastname foo", Field.Store.YES));
321     writer.addDocument(doc2);
322     
323     Document doc3 = new Document();
324     doc3.add(newTextField("field", "foo firstname zzz yyy lastname foo", Field.Store.YES));
325     writer.addDocument(doc3);
326     
327     IndexReader reader = writer.getReader();
328     writer.close();
329 
330     IndexSearcher searcher = newSearcher(reader);
331     searcher.setSimilarity(new DefaultSimilarity());
332     PhraseQuery query = new PhraseQuery(Integer.MAX_VALUE, "field", "firstname", "lastname");
333     ScoreDoc[] hits = searcher.search(query, 1000).scoreDocs;
334     assertEquals(3, hits.length);
335     // Make sure that those matches where the terms appear closer to
336     // each other get a higher score:
337     assertEquals(0.71, hits[0].score, 0.01);
338     assertEquals(0, hits[0].doc);
339     assertEquals(0.44, hits[1].score, 0.01);
340     assertEquals(1, hits[1].doc);
341     assertEquals(0.31, hits[2].score, 0.01);
342     assertEquals(2, hits[2].doc);
343     QueryUtils.check(random(), query,searcher);
344     reader.close();
345     directory.close();
346   }

348   public void testToString() throws Exception {
349     PhraseQuery q = new PhraseQuery("field", new String[0]);
350     assertEquals("\"\"", q.toString());
351 
352     PhraseQuery.Builder builder = new PhraseQuery.Builder();
353     builder.add(new Term("field", "hi"), 1);
354     q = builder.build();
355     assertEquals("field:\"? hi\"", q.toString());
356 
357     
358     builder = new PhraseQuery.Builder();
359     builder.add(new Term("field", "hi"), 1);
360     builder.add(new Term("field", "test"), 5);
361     q = builder.build(); // Query "this hi this is a test is"
362 
363     assertEquals("field:\"? hi ? ? ? test\"", q.toString());
364 
365     builder = new PhraseQuery.Builder();
366     builder.add(new Term("field", "hi"), 1);
367     builder.add(new Term("field", "hello"), 1);
368     builder.add(new Term("field", "test"), 5);
369     q = builder.build();
370     assertEquals("field:\"? hi|hello ? ? ? test\"", q.toString());
371 
372     builder = new PhraseQuery.Builder();
373     builder.add(new Term("field", "hi"), 1);
374     builder.add(new Term("field", "hello"), 1);
375     builder.add(new Term("field", "test"), 5);
376     builder.setSlop(5);
377     q = builder.build();
378     assertEquals("field:\"? hi|hello ? ? ? test\"~5", q.toString());
379   }

602   public void testRandomPhrases() throws Exception {
603     Directory dir = newDirectory();
604     Analyzer analyzer = new MockAnalyzer(random());
605 
606     RandomIndexWriter w  = new RandomIndexWriter(random(), dir, newIndexWriterConfig(analyzer).setMergePolicy(newLogMergePolicy()));
607     List<List<String>> docs = new ArrayList<>();
608     Document d = new Document();
609     Field f = newTextField("f", "", Field.Store.NO);
610     d.add(f);
611 
612     Random r = random();
613 
614     int NUM_DOCS = atLeast(10);
615     for (int i = 0; i < NUM_DOCS; i++) {
616       // must be > 4096 so it spans multiple chunks
617       int termCount = TestUtil.nextInt(random(), 4097, 8200);
618 
619       List<String> doc = new ArrayList<>();
620 
621       StringBuilder sb = new StringBuilder();
622       while(doc.size() < termCount) {
623         if (r.nextInt(5) == 1 || docs.size() == 0) {
624           // make new non-empty-string term
625           String term;
626           while(true) {
627             term = TestUtil.randomUnicodeString(r);
628             if (term.length() > 0) {
629               break;
630             }
631           }
632           try (TokenStream ts = analyzer.tokenStream("ignore", term)) {
633             CharTermAttribute termAttr = ts.addAttribute(CharTermAttribute.class);
634             ts.reset();
635             while(ts.incrementToken()) {
636               String text = termAttr.toString();
637               doc.add(text);
638               sb.append(text).append(' ');
639             }
640             ts.end();
641           }
642         } else {
643           // pick existing sub-phrase
644           List<String> lastDoc = docs.get(r.nextInt(docs.size()));
645           int len = TestUtil.nextInt(r, 1, 10);
646           int start = r.nextInt(lastDoc.size()-len);
647           for(int k=start;k<start+len;k++) {
648             String t = lastDoc.get(k);
649             doc.add(t);
650             sb.append(t).append(' ');
651           }
652         }
653       }
654       docs.add(doc);
655       f.setStringValue(sb.toString());
656       w.addDocument(d);
657     }
658 
659     IndexReader reader = w.getReader();
660     IndexSearcher s = newSearcher(reader);
661     w.close();
662 
663     // now search
664     int num = atLeast(10);
665     for(int i=0;i<num;i++) {
666       int docID = r.nextInt(docs.size());
667       List<String> doc = docs.get(docID);
668       
669       final int numTerm = TestUtil.nextInt(r, 2, 20);
670       final int start = r.nextInt(doc.size()-numTerm);
671       PhraseQuery.Builder builder = new PhraseQuery.Builder();
672       StringBuilder sb = new StringBuilder();
673       for(int t=start;t<start+numTerm;t++) {
674         builder.add(new Term("f", doc.get(t)), t);
675         sb.append(doc.get(t)).append(' ');
676       }
677       PhraseQuery pq = builder.build();
678 
679       TopDocs hits = s.search(pq, NUM_DOCS);
680       boolean found = false;
681       for(int j=0;j<hits.scoreDocs.length;j++) {
682         if (hits.scoreDocs[j].doc == docID) {
683           found = true;
684           break;
685         }
686       }
687 
688       assertTrue("phrase '" + sb + "' not found; start=" + start + ", it=" + i + ", expected doc " + docID, found);
689     }
690 
691     reader.close();
692     dir.close();
693   }
