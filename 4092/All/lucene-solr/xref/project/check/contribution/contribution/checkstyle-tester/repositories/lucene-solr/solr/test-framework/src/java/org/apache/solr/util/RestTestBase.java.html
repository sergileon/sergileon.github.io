87    /** Checks the success or failure of an update message
88     */
89    private static void checkUpdateU(String message, String update, boolean shouldSucceed) {
90      try {
91        String m = (null == message) ? "" : message + " ";
92        if (shouldSucceed) {
93          String response = restTestHarness.validateUpdate(update);
94          if (response != null) fail(m + "update was not successful: " + response);
95        } else {
96          String response = restTestHarness.validateErrorUpdate(update);
97          if (response != null) fail(m + "update succeeded, but should have failed: " + response);
98        }
99      } catch (SAXException e) {
100       throw new RuntimeException("Invalid XML", e);
101     }
102   }

105    * Validates a query matches some XPath test expressions
106    * 
107    * @param request a URL path with optional query params, e.g. "/schema/fields?fl=id,_version_" 
108    */
109   public static void assertQ(String request, String... tests) {
110     try {
111       int queryStartPos = request.indexOf('?');
112       String query;
113       String path;
114       if (-1 == queryStartPos) {
115         query = "";
116         path = request;
117       } else {
118         query = request.substring(queryStartPos + 1);
119         path = request.substring(0, queryStartPos);
120       }
121       if ( ! query.matches(".*wt=schema\\.xml.*")) { // don't overwrite wt=schema.xml
122         query = setParam(query, "wt", "xml");
123       }
124       request = path + '?' + setParam(query, "indent", "on");
125 
126       String response = restTestHarness.query(request);
127 
128       // TODO: should the facet handling below be converted to parse the URL?
129       /*
130       if (req.getParams().getBool("facet", false)) {
131         // add a test to ensure that faceting did not throw an exception
132         // internally, where it would be added to facet_counts/exception
133         String[] allTests = new String[tests.length+1];
134         System.arraycopy(tests,0,allTests,1,tests.length);
135         allTests[0] = "*[count(//lst[@name='facet_counts']/*[@name='exception'])=0]";
136         tests = allTests;
137       }
138       */
139 
140       String results = TestHarness.validateXPath(response, tests);
141 
142       if (null != results) {
143         String msg = "REQUEST FAILED: xpath=" + results
144             + "\n\txml response was: " + response
145             + "\n\trequest was:" + request;
146 
147         log.error(msg);
148         throw new RuntimeException(msg);
149       }
150 
151     } catch (XPathExpressionException e1) {
152       throw new RuntimeException("XPath is invalid", e1);
153     } catch (Exception e2) {
154       SolrException.log(log, "REQUEST FAILED: " + request, e2);
155       throw new RuntimeException("Exception during query", e2);
156     }
157   }

201   /**
202    * Validates a query matches some JSON test expressions and closes the
203    * query. The text expression is of the form path:JSON.  To facilitate
204    * easy embedding in Java strings, the JSON can have double quotes
205    * replaced with single quotes.
206    * <p>
207    * Please use this with care: this makes it easy to match complete
208    * structures, but doing so can result in fragile tests if you are
209    * matching more than what you want to test.
210    * </p>
211    * @param request a URL path with optional query params, e.g. "/schema/fields?fl=id,_version_"
212    * @param delta tolerance allowed in comparing float/double values
213    * @param tests JSON path expression + '==' + expected value
214    */
215   public static void assertJQ(String request, double delta, String... tests) throws Exception {
216     int queryStartPos = request.indexOf('?');
217     String query;
218     String path;
219     if (-1 == queryStartPos) {
220       query = "";
221       path = request;
222     } else {
223       query = request.substring(queryStartPos + 1);
224       path = request.substring(0, queryStartPos);
225     }
226     query = setParam(query, "wt", "json");
227     request = path + '?' + setParam(query, "indent", "on");
228 
229     String response;
230     boolean failed = true;
231     try {
232       response = restTestHarness.query(request);
233       failed = false;
234     } finally {
235       if (failed) {
236         log.error("REQUEST FAILED: " + request);
237       }
238     }
239 
240     for (String test : tests) {
241       if (null == test || 0 == test.length()) continue;
242       String testJSON = json(test);
243 
244       try {
245         failed = true;
246         String err = JSONTestUtil.match(response, testJSON, delta);
247         failed = false;
248         if (err != null) {
249           log.error("query failed JSON validation. error=" + err +
250               "\n expected =" + testJSON +
251               "\n response = " + response +
252               "\n request = " + request + "\n"
253           );
254           throw new RuntimeException(err);
255         }
256       } finally {
257         if (failed) {
258           log.error("JSON query validation threw an exception." +
259               "\n expected =" + testJSON +
260               "\n response = " + response +
261               "\n request = " + request + "\n"
262           );
263         }
264       }
265     }
266   }

495   /**
496    * Insures that the given param is included in the query with the given value.
497    *
498    * <ol>
499    *   <li>If the param is already included with the given value, the request is returned unchanged.</li>
500    *   <li>If the param is not already included, it is added with the given value.</li>
501    *   <li>If the param is already included, but with a different value, the value is replaced with the given value.</li>
502    *   <li>If the param is already included multiple times, they are replaced with a single param with given value.</li>
503    * </ol>
504    *
505    * The passed-in valueToSet should NOT be URL encoded, as it will be URL encoded by this method.
506    *
507    * @param query The query portion of a request URL, e.g. "wt=json&indent=on&fl=id,_version_"
508    * @param paramToSet The parameter name to insure the presence of in the returned request 
509    * @param valueToSet The parameter value to insure in the returned request
510    * @return The query with the given param set to the given value 
511    */
512   private static String setParam(String query, String paramToSet, String valueToSet) {
513     if (null == valueToSet) {
514       valueToSet = "";
515     }
516     try {
517       StringBuilder builder = new StringBuilder();
518       if (null == query || query.trim().isEmpty()) {
519         // empty query -> return "paramToSet=valueToSet"
520         builder.append(paramToSet);
521         builder.append('=');
522         StrUtils.partialURLEncodeVal(builder, valueToSet);
523         return builder.toString();
524       }
525       MultiMapSolrParams requestParams = SolrRequestParsers.parseQueryString(query);
526       String[] values = requestParams.getParams(paramToSet);
527       if (null == values) {
528         // paramToSet isn't present in the request -> append "&paramToSet=valueToSet"
529         builder.append(query);
530         builder.append('&');
531         builder.append(paramToSet);
532         builder.append('=');
533         StrUtils.partialURLEncodeVal(builder, valueToSet);
534         return builder.toString();
535       }
536       if (1 == values.length && valueToSet.equals(values[0])) {
537         // paramToSet=valueToSet is already in the query - just return the query as-is.
538         return query;
539       }
540       // More than one value for paramToSet on the request, or paramToSet's value is not valueToSet
541       // -> rebuild the query
542       boolean isFirst = true;
543       for (Map.Entry<String,String[]> entry : requestParams.getMap().entrySet()) {
544         String key = entry.getKey();
545         String[] valarr = entry.getValue();
546 
547         if ( ! key.equals(paramToSet)) {
548           for (String val : valarr) {
549             builder.append(isFirst ? "" : '&');
550             isFirst = false;
551             builder.append(key);
552             builder.append('=');
553             StrUtils.partialURLEncodeVal(builder, null == val ? "" : val);
554           }
555         }
556       }
557       builder.append(isFirst ? "" : '&');
558       builder.append(paramToSet);
559       builder.append('=');
560       StrUtils.partialURLEncodeVal(builder, valueToSet);
561       return builder.toString();
562     } catch (IOException e) {
563       throw new RuntimeException(e);
564     }
565   }
