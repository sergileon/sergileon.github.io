47   * The default implementation of return fields parsing for Solr.

49  public class SolrReturnFields extends ReturnFields {
50    // Special Field Keys
51    public static final String SCORE = "score";
52  
53    private final List<String> globs = new ArrayList<>(1);
54  
55    // The lucene field names to request from the SolrIndexSearcher
56    private final Set<String> fields = new HashSet<>();
57  
58    // Field names that are OK to include in the response.
59    // This will include pseudo fields, lucene fields, and matching globs
60    private Set<String> okFieldNames = new HashSet<>();
61  
62    // The list of explicitly requested fields
63    // Order is important for CSVResponseWriter
64    private Set<String> reqFieldNames = null;
65    
66    protected DocTransformer transformer;
67    protected boolean _wantsScore = false;
68    protected boolean _wantsAllFields = false;
69  
70    public SolrReturnFields() {
71      _wantsAllFields = true;
72    }

74    public SolrReturnFields(SolrQueryRequest req) {
75      this( req.getParams().getParams(CommonParams.FL), req );
76    }

78    public SolrReturnFields(String fl, SolrQueryRequest req) {
79  //    this( (fl==null)?null:SolrPluginUtils.split(fl), req );
80      if( fl == null ) {
81        parseFieldList((String[])null, req);
82      }
83      else {
84        if( fl.trim().length() == 0 ) {
85          // legacy thing to support fl='  ' => fl=*,score!
86          // maybe time to drop support for this?
87          // See ConvertedLegacyTest
88          _wantsScore = true;
89          _wantsAllFields = true;
90          transformer = new ScoreAugmenter(SCORE);
91        }
92        else {
93          parseFieldList( new String[]{fl}, req);
94        }
95      }
96    }
97  
98    public SolrReturnFields(String[] fl, SolrQueryRequest req) {
99      parseFieldList(fl, req);
100   }

102   private void parseFieldList(String[] fl, SolrQueryRequest req) {
103     _wantsScore = false;
104     _wantsAllFields = false;
105     if (fl == null || fl.length == 0 || fl.length == 1 && fl[0].length()==0) {
106       _wantsAllFields = true;
107       return;
108     }
109 
110     NamedList<String> rename = new NamedList<>();
111     DocTransformers augmenters = new DocTransformers();
112     for (String fieldList : fl) {
113       add(fieldList,rename,augmenters,req);
114     }
115     for( int i=0; i<rename.size(); i++ ) {
116       String from = rename.getName(i);
117       String to = rename.getVal(i);
118       okFieldNames.add( to );
119       boolean copy = (reqFieldNames!=null && reqFieldNames.contains(from));
120       if(!copy) {
121         // Check that subsequent copy/rename requests have the field they need to copy
122         for(int j=i+1; j<rename.size(); j++) {
123           if(from.equals(rename.getName(j))) {
124             rename.setName(j, to); // copy from the current target
125             if(reqFieldNames==null) {
126               reqFieldNames = new LinkedHashSet<>();
127             }
128             reqFieldNames.add(to); // don't rename our current target
129           }
130         }
131       }
132       augmenters.addTransformer( new RenameFieldTransformer( from, to, copy ) );
133     }
134 
135     if( !_wantsAllFields ) {
136       if( !globs.isEmpty() ) {
137         // TODO??? need to fill up the fields with matching field names in the index
138         // and add them to okFieldNames?
139         // maybe just get all fields?
140         // this would disable field selection optimization... i think thatis OK
141         fields.clear(); // this will get all fields, and use wantsField to limit
142       }
143       okFieldNames.addAll( fields );
144     }
145 
146     if( augmenters.size() == 1 ) {
147       transformer = augmenters.getTransformer(0);
148     }
149     else if( augmenters.size() > 1 ) {
150       transformer = augmenters;
151     }
152   }

154   // like getId, but also accepts dashes for legacy fields
155   public static String getFieldName(StrParser sp) {
156     sp.eatws();
157     int id_start = sp.pos;
158     char ch;
159     if (sp.pos < sp.end && (ch = sp.val.charAt(sp.pos)) != '$' && Character.isJavaIdentifierStart(ch)) {
160       sp.pos++;
161       while (sp.pos < sp.end) {
162         ch = sp.val.charAt(sp.pos);
163         if (!Character.isJavaIdentifierPart(ch) && ch != '.' && ch != '-') {
164           break;
165         }
166         sp.pos++;
167       }
168       return sp.val.substring(id_start, sp.pos);
169     }
170 
171     return null;
172   }

174   private void add(String fl, NamedList<String> rename, DocTransformers augmenters, SolrQueryRequest req) {
175     if( fl == null ) {
176       return;
177     }
178     try {
179       StrParser sp = new StrParser(fl);
180 
181       for(;;) {
182         sp.opt(',');
183         sp.eatws();
184         if (sp.pos >= sp.end) break;
185 
186         int start = sp.pos;
187 
188         // short circuit test for a really simple field name
189         String key = null;
190         String field = getFieldName(sp);
191         char ch = sp.ch();
192 
193         if (field != null) {
194           if (sp.opt(':')) {
195             // this was a key, not a field name
196             key = field;
197             field = null;
198             sp.eatws();
199             start = sp.pos;
200           } else {
201             if (Character.isWhitespace(ch) || ch == ',' || ch==0) {
202               addField(field, key, augmenters, false);
203               continue;
204             }
205             // an invalid field name... reset the position pointer to retry
206             sp.pos = start;
207             field = null;
208           }
209         }
210 
211         if (key != null) {
212           // we read "key : "
213           field = sp.getId(null);
214           ch = sp.ch();
215           if (field != null && (Character.isWhitespace(ch) || ch == ',' || ch==0)) {
216             rename.add(field, key);
217             addField(field, key, augmenters, false);
218             continue;
219           }
220           // an invalid field name... reset the position pointer to retry
221           sp.pos = start;
222           field = null;
223         }
224 
225         if (field == null) {
226           // We didn't find a simple name, so let's see if it's a globbed field name.
227           // Globbing only works with field names of the recommended form (roughly like java identifiers)
228 
229           field = sp.getGlobbedId(null);
230           ch = sp.ch();
231           if (field != null && (Character.isWhitespace(ch) || ch == ',' || ch==0)) {
232             // "*" looks and acts like a glob, but we give it special treatment
233             if ("*".equals(field)) {
234               _wantsAllFields = true;
235             } else {
236               globs.add(field);
237             }
238             continue;
239           }
240 
241           // an invalid glob
242           sp.pos = start;
243         }
244 
245         String funcStr = sp.val.substring(start);
246 
247         // Is it an augmenter of the form [augmenter_name foo=1 bar=myfield]?
248         // This is identical to localParams syntax except it uses [] instead of {!}
249 
250         if (funcStr.startsWith("[")) {
251           Map<String,String> augmenterArgs = new HashMap<>();
252           int end = QueryParsing.parseLocalParams(funcStr, 0, augmenterArgs, req.getParams(), "[", ']');
253           sp.pos += end;
254 
255           // [foo] is short for [type=foo] in localParams syntax
256           String augmenterName = augmenterArgs.remove("type");
257           String disp = key;
258           if( disp == null ) {
259             disp = '['+augmenterName+']';
260           }
261 
262           TransformerFactory factory = req.getCore().getTransformerFactory( augmenterName );
263           if( factory != null ) {
264             MapSolrParams augmenterParams = new MapSolrParams( augmenterArgs );
265             DocTransformer t = factory.create(disp, augmenterParams, req);
266             if(t!=null) {
267               if(!_wantsAllFields) {
268                 String[] extra = t.getExtraRequestFields();
269                 if(extra!=null) {
270                   for(String f : extra) {
271                     fields.add(f); // also request this field from IndexSearcher
272                   }
273                 }
274               }
275               augmenters.addTransformer( t );
276             }
277           }
278           else {
279             //throw new SolrException(ErrorCode.BAD_REQUEST, "Unknown DocTransformer: "+augmenterName);
280           }
281           addField(field, disp, augmenters, true);
282           continue;
283         }
284 
285 
286         // let's try it as a function instead
287         QParser parser = QParser.getParser(funcStr, FunctionQParserPlugin.NAME, req);
288         Query q = null;
289         ValueSource vs = null;
290 
291         try {
292           if (parser instanceof FunctionQParser) {
293             FunctionQParser fparser = (FunctionQParser)parser;
294             fparser.setParseMultipleSources(false);
295             fparser.setParseToEnd(false);
296 
297             q = fparser.getQuery();
298 
299             if (fparser.localParams != null) {
300               if (fparser.valFollowedParams) {
301                 // need to find the end of the function query via the string parser
302                 int leftOver = fparser.sp.end - fparser.sp.pos;
303                 sp.pos = sp.end - leftOver;   // reset our parser to the same amount of leftover
304               } else {
305                 // the value was via the "v" param in localParams, so we need to find
306                 // the end of the local params themselves to pick up where we left off
307                 sp.pos = start + fparser.localParamsEnd;
308               }
309             } else {
310               // need to find the end of the function query via the string parser
311               int leftOver = fparser.sp.end - fparser.sp.pos;
312               sp.pos = sp.end - leftOver;   // reset our parser to the same amount of leftover
313             }
314           } else {
315             // A QParser that's not for function queries.
316             // It must have been specified via local params.
317             q = parser.getQuery();
318 
319             assert parser.getLocalParams() != null;
320             sp.pos = start + parser.localParamsEnd;
321           }
322           funcStr = sp.val.substring(start, sp.pos);
323 
324 
325           if (q instanceof FunctionQuery) {
326             vs = ((FunctionQuery)q).getValueSource();
327           } else {
328             vs = new QueryValueSource(q, 0.0f);
329           }
330 
331           if (key==null) {
332             SolrParams localParams = parser.getLocalParams();
333             if (localParams != null) {
334               key = localParams.get("key");
335             }
336           }
337 
338           if (key==null) {
339             key = funcStr;
340           }
341           addField(funcStr, key, augmenters, true);
342           augmenters.addTransformer( new ValueSourceAugmenter( key, parser, vs ) );
343         }
344         catch (SyntaxError e) {
345           // try again, simple rules for a field name with no whitespace
346           sp.pos = start;
347           field = sp.getSimpleString();
348 
349           if (req.getSchema().getFieldOrNull(field) != null) {
350             // OK, it was an oddly named field
351             addField(field, key, augmenters, false);
352             if( key != null ) {
353               rename.add(field, key);
354             }
355           } else {
356             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Error parsing fieldname: " + e.getMessage(), e);
357           }
358         }
359 
360         // end try as function
361 
362       } // end for(;;)
363     } catch (SyntaxError e) {
364       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Error parsing fieldname", e);
365     }
366   }

368   private void addField(String field, String key, DocTransformers augmenters, boolean isPseudoField)
369   {
370     if(reqFieldNames==null) {
371       reqFieldNames = new LinkedHashSet<>();
372     }
373     
374     if(key==null) {
375       reqFieldNames.add(field);
376     }
377     else {
378       reqFieldNames.add(key);
379     }
380 
381     if ( ! isPseudoField) {
382       // fields is returned by getLuceneFieldNames(), to be used to select which real fields
383       // to return, so pseudo-fields should not be added
384       fields.add(field);
385     }
386 
387     okFieldNames.add( field );
388     okFieldNames.add( key );
389     // a valid field name
390     if(SCORE.equals(field)) {
391       _wantsScore = true;
392 
393       String disp = (key==null) ? field : key;
394       augmenters.addTransformer( new ScoreAugmenter( disp ) );
395     }
396   }

398   @Override
399   public Set<String> getLuceneFieldNames()
400   {
401     return (_wantsAllFields || fields.isEmpty()) ? null : fields;
402   }
403 
404   @Override
405   public Set<String> getRequestedFieldNames() {
406     if(_wantsAllFields || reqFieldNames==null || reqFieldNames.isEmpty()) {
407       return null;
408     }
409     return reqFieldNames;
410   }
411   
412   @Override
413   public boolean hasPatternMatching() {
414     return !globs.isEmpty();
415   }
416 
417   @Override
418   public boolean wantsField(String name)
419   {
420     if( _wantsAllFields || okFieldNames.contains( name ) ){
421       return true;
422     }
423     for( String s : globs ) {
424       // TODO something better?
425       if( FilenameUtils.wildcardMatch(name, s) ) {
426         okFieldNames.add(name); // Don't calculate it again
427         return true;
428       }
429     }
430     return false;
431   }
432   
433   @Override
434   public boolean wantsAllFields()
435   {
436     return _wantsAllFields;
437   }
438 
439   @Override
440   public boolean wantsScore()
441   {
442     return _wantsScore;
443   }
444 
445   @Override
446   public DocTransformer getTransformer()
447   {
448     return transformer;
449   }
