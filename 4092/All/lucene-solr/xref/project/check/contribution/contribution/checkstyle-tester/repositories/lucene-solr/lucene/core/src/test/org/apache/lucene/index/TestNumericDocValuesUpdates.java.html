297     Directory dir = newDirectory();
298     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
299     conf.setMaxBufferedDocs(10); // prevent merges
300     IndexWriter writer = new IndexWriter(dir, conf);
301

303       Document doc = new Document();
304       doc.add(new StringField("dvUpdateKey", "dv", Store.NO));
305       doc.add(new NumericDocValuesField("ndv", i));
306       doc.add(new BinaryDocValuesField("bdv", new BytesRef(Integer.toString(i))));
307       doc.add(new SortedDocValuesField("sdv", new BytesRef(Integer.toString(i))));
308       doc.add(new SortedSetDocValuesField("ssdv", new BytesRef(Integer.toString(i))));
309       doc.add(new SortedSetDocValuesField("ssdv", new BytesRef(Integer.toString(i * 2))));
310       writer.addDocument(doc);
311     }
312     writer.commit();
313     
314     // update all docs' ndv field
315     writer.updateNumericDocValue(new Term("dvUpdateKey", "dv"), "ndv", 17L);
316     writer.close();
317     
318     final DirectoryReader reader = DirectoryReader.open(dir);
319     LeafReader r = reader.leaves().get(0).reader();
320     NumericDocValues ndv = r.getNumericDocValues("ndv");
321     BinaryDocValues bdv = r.getBinaryDocValues("bdv");
322     SortedDocValues sdv = r.getSortedDocValues("sdv");
323     SortedSetDocValues ssdv = r.getSortedSetDocValues("ssdv");
324     for (int i = 0; i < r.maxDoc(); i++) {
325       assertEquals(17, ndv.get(i));
326       BytesRef term = bdv.get(i);
327       assertEquals(new BytesRef(Integer.toString(i)), term);
328       term = sdv.get(i);
329       assertEquals(new BytesRef(Integer.toString(i)), term);
330       ssdv.setDocument(i);
331       long ord = ssdv.nextOrd();
332       term = ssdv.lookupOrd(ord);
333       assertEquals(i, Integer.parseInt(term.utf8ToString()));
334       if (i != 0) {
335         ord = ssdv.nextOrd();
336         term = ssdv.lookupOrd(ord);
337         assertEquals(i * 2, Integer.parseInt(term.utf8ToString()));
338       }
339       assertEquals(SortedSetDocValues.NO_MORE_ORDS, ssdv.nextOrd());
340     }
341     
342     reader.close();
343     dir.close();
344   }

510   @Test
511   public void testSegmentMerges() throws Exception {
512     Directory dir = newDirectory();
513     Random random = random();
514     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random));
515     IndexWriter writer = new IndexWriter(dir, conf);
516     
517     int docid = 0;
518     int numRounds = atLeast(10);
519     for (int rnd = 0; rnd < numRounds; rnd++) {
520       Document doc = new Document();
521       doc.add(new StringField("key", "doc", Store.NO));
522       doc.add(new NumericDocValuesField("ndv", -1));
523       int numDocs = atLeast(30);
524       for (int i = 0; i < numDocs; i++) {
525         doc.removeField("id");
526         doc.add(new StringField("id", Integer.toString(docid++), Store.NO));
527         writer.addDocument(doc);
528       }
529       
530       long value = rnd + 1;
531       writer.updateNumericDocValue(new Term("key", "doc"), "ndv", value);
532       
533       if (random.nextDouble() < 0.2) { // randomly delete some docs
534         writer.deleteDocuments(new Term("id", Integer.toString(random.nextInt(docid))));
535       }
536       
537       // randomly commit or reopen-IW (or nothing), before forceMerge
538       if (random.nextDouble() < 0.4) {
539         writer.commit();
540       } else if (random.nextDouble() < 0.1) {
541         writer.close();
542         conf = newIndexWriterConfig(new MockAnalyzer(random));
543         writer = new IndexWriter(dir, conf);
544       }
545 
546       // add another document with the current value, to be sure forceMerge has
547       // something to merge (for instance, it could be that CMS finished merging
548       // all segments down to 1 before the delete was applied, so when
549       // forceMerge is called, the index will be with one segment and deletes
550       // and some MPs might now merge it, thereby invalidating test's
551       // assumption that the reader has no deletes).
552       doc = new Document();
553       doc.add(new StringField("id", Integer.toString(docid++), Store.NO));
554       doc.add(new StringField("key", "doc", Store.NO));
555       doc.add(new NumericDocValuesField("ndv", value));
556       writer.addDocument(doc);
557 
558       writer.forceMerge(1, true);
559       final DirectoryReader reader;
560       if (random.nextBoolean()) {
561         writer.commit();
562         reader = DirectoryReader.open(dir);
563       } else {
564         reader = DirectoryReader.open(writer, true);
565       }
566       
567       assertEquals(1, reader.leaves().size());
568       final LeafReader r = reader.leaves().get(0).reader();
569       assertNull("index should have no deletes after forceMerge", r.getLiveDocs());
570       NumericDocValues ndv = r.getNumericDocValues("ndv");
571       assertNotNull(ndv);
572       for (int i = 0; i < r.maxDoc(); i++) {
573         assertEquals(value, ndv.get(i));
574       }
575       reader.close();
576     }
577     
578     writer.close();
579     dir.close();
580   }

908   @Test
909   public void testStressMultiThreading() throws Exception {
910     final Directory dir = newDirectory();
911     IndexWriterConfig conf = newIndexWriterConfig(new MockAnalyzer(random()));
912     final IndexWriter writer = new IndexWriter(dir, conf);
913     
914     // create index
915     final int numFields = TestUtil.nextInt(random(), 1, 4);
916     final int numDocs = atLeast(2000);
917     for (int i = 0; i < numDocs; i++) {
918       Document doc = new Document();
919       doc.add(new StringField("id", "doc" + i, Store.NO));
920       double group = random().nextDouble();
921       String g;
922       if (group < 0.1) g = "g0";
923       else if (group < 0.5) g = "g1";
924       else if (group < 0.8) g = "g2";
925       else g = "g3";
926       doc.add(new StringField("updKey", g, Store.NO));
927       for (int j = 0; j < numFields; j++) {
928         long value = random().nextInt();
929         doc.add(new NumericDocValuesField("f" + j, value));
930         doc.add(new NumericDocValuesField("cf" + j, value * 2)); // control, always updated to f * 2
931       }
932       writer.addDocument(doc);
933     }
934     
935     final int numThreads = TestUtil.nextInt(random(), 3, 6);
936     final CountDownLatch done = new CountDownLatch(numThreads);
937     final AtomicInteger numUpdates = new AtomicInteger(atLeast(100));
938     
939     // same thread updates a field as well as reopens
940     Thread[] threads = new Thread[numThreads];
941     for (int i = 0; i < threads.length; i++) {
942       threads[i] = new Thread("UpdateThread-" + i) {
943         @Override
944         public void run() {
945           DirectoryReader reader = null;
946           boolean success = false;
947           try {
948             Random random = random();
949             while (numUpdates.getAndDecrement() > 0) {
950               double group = random.nextDouble();
951               Term t;
952               if (group < 0.1) t = new Term("updKey", "g0");
953               else if (group < 0.5) t = new Term("updKey", "g1");
954               else if (group < 0.8) t = new Term("updKey", "g2");
955               else t = new Term("updKey", "g3");
956 
957               final int field = random().nextInt(numFields);
958               final String f = "f" + field;
959               final String cf = "cf" + field;
960 //              System.out.println("[" + Thread.currentThread().getName() + "] numUpdates=" + numUpdates + " updateTerm=" + t + " field=" + field);
961               long updValue = random.nextInt();
962               writer.updateDocValues(t, new NumericDocValuesField(f, updValue), new NumericDocValuesField(cf, updValue*2));
963               
964               if (random.nextDouble() < 0.2) {
965                 // delete a random document
966                 int doc = random.nextInt(numDocs);
967 //                System.out.println("[" + Thread.currentThread().getName() + "] deleteDoc=doc" + doc);
968                 writer.deleteDocuments(new Term("id", "doc" + doc));
969               }
970   
971               if (random.nextDouble() < 0.05) { // commit every 20 updates on average
972 //                  System.out.println("[" + Thread.currentThread().getName() + "] commit");
973                 writer.commit();
974               }
975               
976               if (random.nextDouble() < 0.1) { // reopen NRT reader (apply updates), on average once every 10 updates
977                 if (reader == null) {
978 //                  System.out.println("[" + Thread.currentThread().getName() + "] open NRT");
979                   reader = DirectoryReader.open(writer, true);
980                 } else {
981 //                  System.out.println("[" + Thread.currentThread().getName() + "] reopen NRT");
982                   DirectoryReader r2 = DirectoryReader.openIfChanged(reader, writer, true);
983                   if (r2 != null) {
984                     reader.close();
985                     reader = r2;
986                   }
987                 }
988               }
989             }
990 //            System.out.println("[" + Thread.currentThread().getName() + "] DONE");
991             success = true;
992           } catch (IOException e) {
993             throw new RuntimeException(e);
994           } finally {
995             if (reader != null) {
996               try {
997                 reader.close();
998               } catch (IOException e) {
999                 if (success) { // suppress this exception only if there was another exception
1000                   throw new RuntimeException(e);
1001                 }
1002               }
1003             }
1004             done.countDown();
1005           }
1006         }
1007       };
1008     }
1009     
1010     for (Thread t : threads) t.start();
1011     done.await();
1012     writer.close();
1013     
1014     DirectoryReader reader = DirectoryReader.open(dir);
1015     for (LeafReaderContext context : reader.leaves()) {
1016       LeafReader r = context.reader();
1017       for (int i = 0; i < numFields; i++) {
1018         NumericDocValues ndv = r.getNumericDocValues("f" + i);
1019         NumericDocValues control = r.getNumericDocValues("cf" + i);
1020         Bits docsWithNdv = r.getDocsWithField("f" + i);
1021         Bits docsWithControl = r.getDocsWithField("cf" + i);
1022         Bits liveDocs = r.getLiveDocs();
1023         for (int j = 0; j < r.maxDoc(); j++) {
1024           if (liveDocs == null || liveDocs.get(j)) {
1025             assertTrue(docsWithNdv.get(j));
1026             assertTrue(docsWithControl.get(j));
1027             assertEquals(control.get(j), ndv.get(j) * 2);
1028           }
1029         }
1030       }
1031     }
1032     reader.close();
1033     
1034     dir.close();
1035   }
