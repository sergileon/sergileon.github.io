32    public void testUAX29URLEmailTokenizer() throws Exception {
33      Reader reader = new StringReader("Wha\u0301t's this thing do?");
34      Tokenizer stream = tokenizerFactory("UAX29URLEmail").create(newAttributeFactory());
35      stream.setReader(reader);
36      assertTokenStreamContents(stream, 
37          new String[] { "Wha\u0301t's", "this", "thing", "do" });
38    }

151   public void testMaxTokenLength() throws Exception {
152     StringBuilder builder = new StringBuilder();
153     for (int i = 0 ; i < 100 ; ++i) {
154       builder.append("abcdefg"); // 7 * 100 = 700 char "word"
155     }
156     String longWord = builder.toString();
157     String content = "one two three " + longWord + " four five six";
158     Reader reader = new StringReader(content);
159     Tokenizer stream = tokenizerFactory("UAX29URLEmail",
160         "maxTokenLength", "1000").create(newAttributeFactory());
161     stream.setReader(reader);
162     assertTokenStreamContents(stream, 
163         new String[] {"one", "two", "three", longWord, "four", "five", "six" });
164   }

166   /** Test that bogus arguments result in exception */
167   public void testBogusArguments() throws Exception {
168     try {
169       tokenizerFactory("UAX29URLEmail", "bogusArg", "bogusValue");
170       fail();
171     } catch (IllegalArgumentException expected) {
172       assertTrue(expected.getMessage().contains("Unknown parameters"));
173     }
174   }

176  public void testIllegalArguments() throws Exception {
177     try {
178       tokenizerFactory("UAX29URLEmail", "maxTokenLength", "-1").create();
179       fail();
180     } catch (IllegalArgumentException expected) {
181       assertTrue(expected.getMessage().contains("maxTokenLength must be greater than zero"));
182     }
183   }
