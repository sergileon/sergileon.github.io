54  /**
55   * Supports runtime mapping of REST API endpoints to ManagedResource 
56   * implementations; endpoints can be registered at either the /schema
57   * or /config base paths, depending on which base path is more appropriate
58   * for the type of managed resource.

64    public static final String SCHEMA_BASE_PATH = "/schema";
65    public static final String MANAGED_ENDPOINT = "/managed";

67    // used for validating resourceIds provided during registration
68    private static final Pattern resourceIdRegex = Pattern.compile("(/config|/schema)(/.*)");

72    /**
73     * Used internally to keep track of registrations during core initialization
74     */
75    private static class ManagedResourceRegistration {
76      String resourceId;
77      Class<? extends ManagedResource> implClass;
78      List<ManagedResourceObserver> observers = new ArrayList<>();
79  
80      private ManagedResourceRegistration(String resourceId,
81                                          Class<? extends ManagedResource> implClass, 
82                                          ManagedResourceObserver observer)
83      {
84        this.resourceId = resourceId;
85        this.implClass = implClass;
86        
87        if (observer != null) {
88          this.observers.add(observer);
89        }
90      }  
91  
92      /** Returns resourceId, class, and number of observers of this registered resource */
93      public Map<String,String> getInfo() {
94        Map<String,String> info = new HashMap<>();
95        info.put("resourceId", resourceId);
96        info.put("class", implClass.getName());
97        info.put("numObservers", String.valueOf(observers.size()));
98        return info;
99      }    
100   }

102   /**
103    * Per-core registry of ManagedResources found during core initialization.
104    * 
105    * Registering of managed resources can happen before the RestManager is
106    * fully initialized. To avoid timing issues, resources register themselves
107    * and then the RestManager initializes all ManagedResources before the core
108    * is activated.  
109    */
110   public static class Registry {
111     
112     private Map<String,ManagedResourceRegistration> registered = new TreeMap<>();
113     
114     // maybe null until there is a restManager
115     private RestManager initializedRestManager = null;
116 
117     // REST API endpoints that need to be protected against dynamic endpoint creation
118     private final Set<String> reservedEndpoints = new HashSet<>();
119     private final Pattern reservedEndpointsPattern;
120 
121     public Registry() {
122       reservedEndpoints.add(SCHEMA_BASE_PATH + MANAGED_ENDPOINT);
123 
124       for (String reservedEndpoint : SolrSchemaRestApi.getReservedEndpoints()) {
125         reservedEndpoints.add(reservedEndpoint);
126       }
127 
128       reservedEndpointsPattern = getReservedEndpointsPattern();
129     }
130 
131     /**
132      * Returns the set of non-registerable endpoints.
133      */
134     public Set<String> getReservedEndpoints() {
135       return Collections.unmodifiableSet(reservedEndpoints);
136     }
137 
138     /**
139      * Returns a Pattern, to be used with Matcher.matches(), that will recognize
140      * prefixes or full matches against reserved endpoints that need to be protected
141      * against dynamic endpoint registration.  group(1) will contain the match
142      * regardless of whether it's a full match or a prefix.
143      */
144     private Pattern getReservedEndpointsPattern() {
145       // Match any of the reserved endpoints exactly, or followed by a slash and more stuff
146       StringBuilder builder = new StringBuilder();
147       builder.append("(");
148       boolean notFirst = false;
149       for (String reservedEndpoint : reservedEndpoints) {
150         if (notFirst) {
151           builder.append("|");
152         } else {
153           notFirst = true;
154         }
155         builder.append(reservedEndpoint);
156       }
157       builder.append(")(?:|/.*)");
158       return Pattern.compile(builder.toString());
159     }

166       return Collections.unmodifiableCollection(registered.values());

169     /**
170      * Register the need to use a ManagedResource; this method is typically called
171      * by a Solr component during core initialization to register itself as an 
172      * observer of a specific type of ManagedResource. As many Solr components may
173      * share the same ManagedResource, this method only serves to associate the
174      * observer with an endpoint and implementation class. The actual construction
175      * of the ManagedResource and loading of data from storage occurs later once
176      * the RestManager is fully initialized.
177      * @param resourceId - An endpoint in the Rest API to manage the resource; must
178      * start with /config and /schema.
179      * @param implClass - Class that implements ManagedResource.
180      * @param observer - Solr component that needs to know when the data being managed
181      * by the ManagedResource is loaded, such as a TokenFilter.
182      */
183     public synchronized void registerManagedResource(String resourceId, 
184         Class<? extends ManagedResource> implClass, ManagedResourceObserver observer) {
185       
186       if (resourceId == null)
187         throw new IllegalArgumentException(
188             "Must provide a non-null resourceId to register a ManagedResource!");
189 
190       Matcher resourceIdValidator = resourceIdRegex.matcher(resourceId);
191       if (!resourceIdValidator.matches()) {
192         String errMsg = String.format(Locale.ROOT,
193             "Invalid resourceId '%s'; must start with  %s.",
194             resourceId,  SCHEMA_BASE_PATH);
195         throw new SolrException(ErrorCode.SERVER_ERROR, errMsg);        
196       }
197          
198       // protect reserved REST API endpoints from being used by another
199       Matcher reservedEndpointsMatcher = reservedEndpointsPattern.matcher(resourceId);
200       if (reservedEndpointsMatcher.matches()) {
201         throw new SolrException(ErrorCode.SERVER_ERROR,
202             reservedEndpointsMatcher.group(1)
203             + " is a reserved endpoint used by the Solr REST API!");
204       }
205 
206       // IMPORTANT: this code should assume there is no RestManager at this point
207       
208       // it's ok to re-register the same class for an existing path
209       ManagedResourceRegistration reg = registered.get(resourceId);
210       if (reg != null) {
211         if (!reg.implClass.equals(implClass)) {
212           String errMsg = String.format(Locale.ROOT,
213               "REST API path %s already registered to instances of %s",
214               resourceId, reg.implClass.getName());
215           throw new SolrException(ErrorCode.SERVER_ERROR, errMsg);          
216         } 
217         
218         if (observer != null) {
219           reg.observers.add(observer);
220           log.info("Added observer of type {} to existing ManagedResource {}", 
221               observer.getClass().getName(), resourceId);
222         }
223       } else {
224         registered.put(resourceId, 
225             new ManagedResourceRegistration(resourceId, implClass, observer));
226         log.info("Registered ManagedResource impl {} for path {}", 
227             implClass.getName(), resourceId);
228       }
229       
230       // there may be a RestManager, in which case, we want to add this new ManagedResource immediately
231       if (initializedRestManager != null) {
232         initializedRestManager.addRegisteredResource(registered.get(resourceId));
233       }
234     }    
235   }

237   /**
238    * Locates the RestManager using ThreadLocal SolrRequestInfo.
239    */
240   public static RestManager getRestManager(SolrRequestInfo solrRequestInfo) {
241     if (solrRequestInfo == null)
242       throw new ResourceException(Status.SERVER_ERROR_INTERNAL, 
243           "No SolrRequestInfo in this Thread!");
244 
245     SolrQueryRequest req = solrRequestInfo.getReq();
246     RestManager restManager = 
247         (req != null) ? req.getCore().getRestManager() : null;
248     
249     if (restManager == null)
250       throw new ResourceException(Status.SERVER_ERROR_INTERNAL, 
251           "No RestManager found!");
252     
253     return restManager;
254   }

256   /**
257    * The Restlet router needs a lightweight extension of ServerResource to delegate a request
258    * to. ManagedResource implementations are heavy-weight objects that live for the duration of
259    * a SolrCore, so this class acts as the proxy between Restlet and a ManagedResource when
260    * doing request processing.
261    *
262    */
263   public static class ManagedEndpoint extends BaseSolrResource
264       implements GETable, PUTable, POSTable, DELETEable
265   {
266     /**
267      * Determines the ManagedResource resourceId from the Restlet request.
268      */
269     public static String resolveResourceId(Request restletReq)  {
270       String resourceId = restletReq.getResourceRef().
271           getRelativeRef(restletReq.getRootRef().getParentRef()).getPath(DECODE);
272       
273       // all resources are registered with the leading slash
274       if (!resourceId.startsWith("/"))
275         resourceId = "/"+resourceId;
276 
277 
278       return resourceId;
279     }

281     protected ManagedResource managedResource;
282     protected String childId;    
283     
284     /**
285      * Initialize objects needed to handle a request to the REST API. Specifically,
286      * we lookup the RestManager using the ThreadLocal SolrRequestInfo and then
287      * dynamically locate the ManagedResource associated with the request URI.
288      */
289     @Override
290     public void doInit() throws ResourceException {
291       super.doInit();      
292       
293       // get the relative path to the requested resource, which is
294       // needed to locate ManagedResource impls at runtime
295       String resourceId = resolveResourceId(getRequest());
296 
297       // supports a request for a registered resource or its child
298       RestManager restManager = 
299           RestManager.getRestManager(SolrRequestInfo.getRequestInfo());
300       
301       managedResource = restManager.getManagedResourceOrNull(resourceId);      
302       if (managedResource == null) {
303         // see if we have a registered endpoint one-level up ...
304         int lastSlashAt = resourceId.lastIndexOf('/');
305         if (lastSlashAt != -1) {
306           String parentResourceId = resourceId.substring(0,lastSlashAt);          
307           log.info("Resource not found for {}, looking for parent: {}",
308               resourceId, parentResourceId);          
309           managedResource = restManager.getManagedResourceOrNull(parentResourceId);
310           if (managedResource != null) {
311             // verify this resource supports child resources
312             if (!(managedResource instanceof ManagedResource.ChildResourceSupport)) {
313               String errMsg = String.format(Locale.ROOT,
314                   "%s does not support child resources!", managedResource.getResourceId());
315               throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, errMsg);
316             }
317             
318             childId = resourceId.substring(lastSlashAt+1);
319             log.info("Found parent resource {} for child: {}", 
320                 parentResourceId, childId);
321           }
322         }
323       }    
324       
325       if (managedResource == null) {
326         if (Method.PUT.equals(getMethod()) || Method.POST.equals(getMethod())) {
327           // delegate create requests to the RestManager
328           managedResource = restManager.endpoint;
329         } else {        
330           throw new ResourceException(Status.CLIENT_ERROR_NOT_FOUND, 
331               "No REST managed resource registered for path "+resourceId);
332         }
333       }
334       
335       log.info("Found ManagedResource ["+managedResource+"] for "+resourceId);      
336     }

338     @Override
339     public Representation put(Representation entity) {
340       try {
341         managedResource.doPut(this, entity, parseJsonFromRequestBody(entity));
342       } catch (Exception e) {
343         getSolrResponse().setException(e);        
344       }
345       handlePostExecution(log);
346       return new SolrOutputRepresentation();    
347     }

349     @Override
350     public Representation post(Representation entity) {
351       try {
352         managedResource.doPost(this, entity, parseJsonFromRequestBody(entity));
353       } catch (Exception e) {
354         getSolrResponse().setException(e);        
355       }
356       handlePostExecution(log);
357       return new SolrOutputRepresentation();    
358     }    
359 
360     @Override
361     public Representation delete() {
362       // only delegate delete child resources to the ManagedResource
363       // as deleting the actual resource is best handled by the
364       // RestManager
365       if (childId != null) {        
366         try {
367           managedResource.doDeleteChild(this, childId);
368         } catch (Exception e) {
369           getSolrResponse().setException(e);        
370         }
371       } else {
372         try {
373           RestManager restManager = 
374               RestManager.getRestManager(SolrRequestInfo.getRequestInfo());
375           restManager.deleteManagedResource(managedResource);
376         } catch (Exception e) {
377           getSolrResponse().setException(e);        
378         }
379       }
380       handlePostExecution(log);
381       return new SolrOutputRepresentation();    
382     }    
383         
384     @Override
385     public Representation get() { 
386       try {
387         managedResource.doGet(this, childId);
388       } catch (Exception e) {
389         getSolrResponse().setException(e);        
390       }
391       handlePostExecution(log);
392       return new SolrOutputRepresentation();    
393     }

395     /**
396      * Parses and validates the JSON passed from the to the ManagedResource. 
397      */
398     protected Object parseJsonFromRequestBody(Representation entity) {
399       if (entity.getMediaType() == null) {
400         entity.setMediaType(MediaType.APPLICATION_JSON);
401       }
402       
403       if (!entity.getMediaType().equals(MediaType.APPLICATION_JSON, true)) {
404         String errMsg = String.format(Locale.ROOT,
405             "Invalid content type %s; only %s is supported.",
406             entity.getMediaType(), MediaType.APPLICATION_JSON.toString());
407         log.error(errMsg);
408         throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, errMsg);
409       }
410       
411       String text = null;
412       try {
413         text = entity.getText();
414       } catch (IOException ioExc) {
415         String errMsg = "Failed to read entity text due to: "+ioExc;
416         log.error(errMsg, ioExc);
417         throw new ResourceException(Status.SERVER_ERROR_INTERNAL, errMsg, ioExc);
418       }
419       
420       if (text == null || text.trim().length() == 0) {
421         throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, "Empty request body!");      
422       }
423 
424       Object parsedJson = null;
425       try {
426         parsedJson = ObjectBuilder.fromJSON(text);
427       } catch (IOException ioExc) {
428         String errMsg = String.format(Locale.ROOT,
429             "Failed to parse request [%s] into JSON due to: %s",
430             text, ioExc.toString());
431         log.error(errMsg, ioExc);
432         throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, errMsg, ioExc);
433       }
434       return parsedJson;
435     }

438   /**
439    * The RestManager itself supports some endpoints for creating and listing managed resources.
440    * Effectively, this resource provides the API endpoint for doing CRUD on the registry.
441    */
442   private static class RestManagerManagedResource extends ManagedResource {
443 
444     private static final String REST_MANAGER_STORAGE_ID = "/rest/managed";
445 
446     private final RestManager restManager;
447 
448 
449     public RestManagerManagedResource(RestManager restManager) throws SolrException {
450       super(REST_MANAGER_STORAGE_ID, restManager.loader, restManager.storageIO);
451       this.restManager = restManager;
452     }
453 
454     /**
455      * Overrides the parent impl to handle FileNotFoundException better
456      */
457     @Override
458     protected synchronized void reloadFromStorage() throws SolrException {
459       String resourceId = getResourceId();
460       Object data = null;
461       try {
462         data = storage.load(resourceId);
463       } catch (FileNotFoundException fnf) {
464         // this is ok - simply means there are no managed components added yet
465       } catch (IOException ioExc) {
466         throw new SolrException(ErrorCode.SERVER_ERROR,
467             "Failed to load stored data for "+resourceId+" due to: "+ioExc, ioExc);
468       }
469 
470       Object managedData = processStoredData(data);
471 
472       if (managedInitArgs == null)
473         managedInitArgs = new NamedList<>();
474 
475       if (managedData != null)
476         onManagedDataLoadedFromStorage(managedInitArgs, managedData);
477     }
478 
479     /**
480      * Loads and initializes any ManagedResources that have been created but
481      * are not associated with any Solr components.
482      */
483     @SuppressWarnings("unchecked")
484     @Override
485     protected void onManagedDataLoadedFromStorage(NamedList<?> managedInitArgs, Object managedData)
486         throws SolrException {
487 
488       if (managedData == null) {
489         // this is ok - just means no managed components have been added yet
490         return;
491       }
492       
493       List<Object> managedList = (List<Object>)managedData;
494       for (Object next : managedList) {
495         Map<String,String> info = (Map<String,String>)next;        
496         String implClass = info.get("class");
497         String resourceId = info.get("resourceId");
498         Class<? extends ManagedResource> clazz = solrResourceLoader.findClass(implClass, ManagedResource.class);
499         ManagedResourceRegistration existingReg = restManager.registry.registered.get(resourceId);
500         if (existingReg == null) {
501           restManager.registry.registerManagedResource(resourceId, clazz, null);
502         } // else already registered, no need to take any action        
503       }      
504     }
505             
506     /**
507      * Creates a new ManagedResource in the RestManager.
508      */
509     @SuppressWarnings("unchecked")
510     @Override
511     public synchronized void doPut(BaseSolrResource endpoint, Representation entity, Object json) {      
512       if (json instanceof Map) {
513         String resourceId = ManagedEndpoint.resolveResourceId(endpoint.getRequest());
514         Map<String,String> info = (Map<String,String>)json;
515         info.put("resourceId", resourceId);
516         storeManagedData(applyUpdatesToManagedData(json));
517       } else {
518         throw new ResourceException(Status.CLIENT_ERROR_BAD_REQUEST, 
519             "Expected Map to create a new ManagedResource but received a "+json.getClass().getName());
520       }          
521       // PUT just returns success status code with an empty body
522     }
523 
524     /**
525      * Registers a new {@link ManagedResource}.
526      *
527      * Called during PUT/POST processing to apply updates to the managed data passed from the client.
528      */
529     @SuppressWarnings("unchecked")
530     @Override
531     protected Object applyUpdatesToManagedData(Object updates) {
532       Map<String,String> info = (Map<String,String>)updates;      
533       // this is where we'd register a new ManagedResource
534       String implClass = info.get("class");
535       String resourceId = info.get("resourceId");
536       log.info("Creating a new ManagedResource of type {} at path {}", 
537           implClass, resourceId);
538       Class<? extends ManagedResource> clazz = 
539           solrResourceLoader.findClass(implClass, ManagedResource.class);
540       
541       // add this new resource to the RestManager
542       restManager.addManagedResource(resourceId, clazz);
543 
544       // we only store ManagedResources that don't have observers as those that do
545       // are already implicitly defined
546       List<Map<String,String>> managedList = new ArrayList<>();
547       for (ManagedResourceRegistration reg : restManager.registry.getRegistered()) {
548         if (reg.observers.isEmpty()) {
549           managedList.add(reg.getInfo());
550         }
551       }          
552       return managedList;
553     }
554 
555     /**
556      * Deleting of child resources not supported by this implementation.
557      */
558     @Override
559     public void doDeleteChild(BaseSolrResource endpoint, String childId) {
560       throw new ResourceException(Status.SERVER_ERROR_NOT_IMPLEMENTED);
561     }
562 
563     @Override
564     public void doGet(BaseSolrResource endpoint, String childId) {
565       
566       // filter results by /schema or /config
567       String path = ManagedEndpoint.resolveResourceId(endpoint.getRequest());
568       Matcher resourceIdMatcher = resourceIdRegex.matcher(path);
569       if (!resourceIdMatcher.matches()) {
570         // extremely unlikely but didn't want to squelch it either
571         throw new ResourceException(Status.SERVER_ERROR_NOT_IMPLEMENTED, path);
572       }
573       
574       String filter = resourceIdMatcher.group(1);
575             
576       List<Map<String,String>> regList = new ArrayList<>();
577       for (ManagedResourceRegistration reg : restManager.registry.getRegistered()) {
578         if (!reg.resourceId.startsWith(filter))
579           continue; // doesn't match filter
580         
581         if (RestManagerManagedResource.class.isAssignableFrom(reg.implClass))
582           continue; // internal, no need to expose to outside
583         
584         regList.add(reg.getInfo());          
585       }
586       
587       endpoint.getSolrResponse().add("managedResources", regList);      
588     }

601   /**
602    * Initializes the RestManager with the storageIO being optionally created outside of this implementation
603    * such as to use ZooKeeper instead of the local FS. 
604    */
605   public void init(SolrResourceLoader loader,
606                    NamedList<String> initArgs, 
607                    StorageIO storageIO) 
608       throws SolrException
609   {
610     log.info("Initializing RestManager with initArgs: "+initArgs);
611 
612     if (storageIO == null)
613       throw new IllegalArgumentException(
614           "Must provide a valid StorageIO implementation to the RestManager!");
615     
616     this.storageIO = storageIO;
617     this.loader = loader;
618     
619     registry = loader.getManagedResourceRegistry();
620     
621     // the RestManager provides metadata about managed resources via the /managed endpoint
622     // and allows you to create new ManagedResources dynamically by PUT'ing to this endpoint
623     endpoint = new RestManagerManagedResource(this);
624     endpoint.loadManagedDataAndNotify(null); // no observers for my endpoint
625     // responds to requests to /config/managed and /schema/managed
626     managed.put(SCHEMA_BASE_PATH+MANAGED_ENDPOINT, endpoint);
627             
628     // init registered managed resources
629     log.info("Initializing {} registered ManagedResources", registry.registered.size());
630     for (ManagedResourceRegistration reg : registry.registered.values()) {
631       // keep track of this for lookups during request processing
632       managed.put(reg.resourceId, createManagedResource(reg));
633     }
634     
635     // this is for any new registrations that don't come through the API
636     // such as from adding a new fieldType to a managed schema that uses a ManagedResource
637     registry.initializedRestManager = this;
638   }

640   /**
641    * If not already registered, registers the given {@link ManagedResource} subclass
642    * at the given resourceId, creates an instance, and attaches it to the appropriate
643    * Restlet router.  Returns the corresponding instance.
644    */
645   public synchronized ManagedResource addManagedResource(String resourceId, Class<? extends ManagedResource> clazz) {
646     ManagedResource res = null;
647     ManagedResourceRegistration existingReg = registry.registered.get(resourceId);
648     if (existingReg == null) {
649       registry.registerManagedResource(resourceId, clazz, null);
650       addRegisteredResource(registry.registered.get(resourceId));
651     } else {
652       res = getManagedResource(resourceId);
653     }
654     return res;
655   }

657   // used internally to create and attach a ManagedResource to the Restlet router
658   // the registry also uses this method directly, which is slightly hacky but necessary
659   // in order to support dynamic adding of new fieldTypes using the managed-schema API
660   private synchronized ManagedResource addRegisteredResource(ManagedResourceRegistration reg) {
661     String resourceId = reg.resourceId;
662     ManagedResource res = createManagedResource(reg);
663     managed.put(resourceId, res);
664     log.info("Registered new managed resource {}", resourceId);
665     
666     // attach this new resource to the Restlet router
667     Matcher resourceIdValidator = resourceIdRegex.matcher(resourceId);
668     boolean validated = resourceIdValidator.matches();
669     assert validated : "managed resourceId '" + resourceId
670                      + "' should already be validated by registerManagedResource()";
671     String routerPath = resourceIdValidator.group(1);      
672     String path = resourceIdValidator.group(2);
673     Router router = SCHEMA_BASE_PATH.equals(routerPath) ? schemaRouter : configRouter;
674     if (router != null) {
675       attachManagedResource(res, path, router);
676     }
677     return res;
678   }

682    * Creates a ManagedResource using registration information. 
683    */
684   protected ManagedResource createManagedResource(ManagedResourceRegistration reg) throws SolrException {
685     ManagedResource res = null;
686     try {
687       Constructor<? extends ManagedResource> ctor = 
688           reg.implClass.getConstructor(String.class, SolrResourceLoader.class, StorageIO.class);
689       res = ctor.newInstance(reg.resourceId, loader, storageIO);
690       res.loadManagedDataAndNotify(reg.observers);
691     } catch (Exception e) {
692       String errMsg = 
693           String.format(Locale.ROOT,
694               "Failed to create new ManagedResource %s of type %s due to: %s",
695               reg.resourceId, reg.implClass.getName(), e);      
696       throw new SolrException(ErrorCode.SERVER_ERROR, errMsg, e);
697     }
698     return res;
699   }

701   /**
702    * Returns the {@link ManagedResource} subclass instance corresponding
703    * to the given resourceId from the registry.
704    *
705    * @throws ResourceException if no managed resource is registered with
706    *  the given resourceId.
707    */
708   public ManagedResource getManagedResource(String resourceId) {
709     ManagedResource res = getManagedResourceOrNull(resourceId);
710     if (res == null) {
711       throw new ResourceException(Status.SERVER_ERROR_INTERNAL, 
712           "No ManagedResource registered for path: "+resourceId);
713     }
714     return res;
715   }

726   /**
727    * Deletes a managed resource if it is not being used by any Solr components. 
728    */
729   public synchronized void deleteManagedResource(ManagedResource res) {
730     String resourceId = res.getResourceId();
731     ManagedResourceRegistration existingReg = registry.registered.get(resourceId);
732     int numObservers = existingReg.observers.size();
733     if (numObservers > 0) {
734       String errMsg = 
735           String.format(Locale.ROOT,
736               "Cannot delete managed resource %s as it is being used by %d Solr components",
737               resourceId, numObservers);
738       throw new SolrException(ErrorCode.FORBIDDEN, errMsg);
739     }
740     
741     registry.registered.remove(resourceId);
742     managed.remove(resourceId);
743     try {
744       res.onResourceDeleted();
745     } catch (IOException e) {
746       // the resource is already deleted so just log this
747       log.error("Error when trying to clean-up after deleting "+resourceId, e);
748     }
749   }

751   /**
752    * Attach managed resource paths to the given Restlet Router. 
753    * @param router - Restlet Router
754    */
755   public synchronized void attachManagedResources(String routerPath, Router router) {
756     if (SCHEMA_BASE_PATH.equals(routerPath)) {
757       this.schemaRouter = router;
758     } else {
759       throw new SolrException(ErrorCode.SERVER_ERROR, 
760           routerPath+" not supported by the RestManager");
761     }      
762     
763     int numAttached = 0;
764     for (String resourceId : managed.keySet()) {
765       if (resourceId.startsWith(routerPath)) {
766         // the way restlet works is you attach a path w/o the routerPath
767         String path = resourceId.substring(routerPath.length());
768         attachManagedResource(managed.get(resourceId), path, router);
769         ++numAttached;
770       }
771     }
772     
773     log.info("Attached {} ManagedResource endpoints to Restlet router: {}", 
774         numAttached, routerPath);
775   }

777   /**
778    * Attaches a ManagedResource and optionally a path for child resources
779    * to the given Restlet Router.
780    */
781   protected void attachManagedResource(ManagedResource res, String path, Router router) {
782     router.attach(path, res.getServerResourceClass());
783     log.info("Attached managed resource at path: {}",path);
784     
785     // Determine if we should also route requests for child resources
786     // ManagedResource.ChildResourceSupport is a marker interface that
787     // indicates the ManagedResource also manages child resources at
788     // a path one level down from the main resourceId
789     if (ManagedResource.ChildResourceSupport.class.isAssignableFrom(res.getClass())) {
790       router.attach(path+"/{child}", res.getServerResourceClass());
791     }    
792   }
