98    @Override
99    public void handleRequestBody(SolrQueryRequest req, SolrQueryResponse rsp) throws Exception 
100   {
101     SolrParams params = req.getParams();
102 
103     long timeAllowed = (long)params.getInt( CommonParams.TIME_ALLOWED, -1 );
104     if(timeAllowed > 0) {
105       SolrQueryTimeoutImpl.set(timeAllowed);
106     }
107       try {
108 
109         // Set field flags
110         ReturnFields returnFields = new SolrReturnFields(req);
111         rsp.setReturnFields(returnFields);
112         int flags = 0;
113         if (returnFields.wantsScore()) {
114           flags |= SolrIndexSearcher.GET_SCORES;
115         }
116 
117         String defType = params.get(QueryParsing.DEFTYPE, QParserPlugin.DEFAULT_QTYPE);
118         String q = params.get(CommonParams.Q);
119         Query query = null;
120         SortSpec sortSpec = null;
121         List<Query> filters = null;
122 
123         try {
124           if (q != null) {
125             QParser parser = QParser.getParser(q, defType, req);
126             query = parser.getQuery();
127             sortSpec = parser.getSort(true);
128           }
129 
130           String[] fqs = req.getParams().getParams(CommonParams.FQ);
131           if (fqs != null && fqs.length != 0) {
132             filters = new ArrayList<>();
133             for (String fq : fqs) {
134               if (fq != null && fq.trim().length() != 0) {
135                 QParser fqp = QParser.getParser(fq, null, req);
136                 filters.add(fqp.getQuery());
137               }
138             }
139           }
140         } catch (SyntaxError e) {
141           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, e);
142         }
143 
144         SolrIndexSearcher searcher = req.getSearcher();
145 
146         MoreLikeThisHelper mlt = new MoreLikeThisHelper(params, searcher);
147 
148         // Hold on to the interesting terms if relevant
149         TermStyle termStyle = TermStyle.get(params.get(MoreLikeThisParams.INTERESTING_TERMS));
150         List<InterestingTerm> interesting = (termStyle == TermStyle.NONE)
151             ? null : new ArrayList<InterestingTerm>(mlt.mlt.getMaxQueryTerms());
152 
153         DocListAndSet mltDocs = null;
154 
155         // Parse Required Params
156         // This will either have a single Reader or valid query
157         Reader reader = null;
158         try {
159           if (q == null || q.trim().length() < 1) {
160             Iterable<ContentStream> streams = req.getContentStreams();
161             if (streams != null) {
162               Iterator<ContentStream> iter = streams.iterator();
163               if (iter.hasNext()) {
164                 reader = iter.next().getReader();
165               }
166               if (iter.hasNext()) {
167                 throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
168                     "MoreLikeThis does not support multiple ContentStreams");
169               }
170             }
171           }
172 
173           int start = params.getInt(CommonParams.START, 0);
174           int rows = params.getInt(CommonParams.ROWS, 10);
175 
176           // Find documents MoreLikeThis - either with a reader or a query
177           // --------------------------------------------------------------------------------
178           if (reader != null) {
179             mltDocs = mlt.getMoreLikeThis(reader, start, rows, filters,
180                 interesting, flags);
181           } else if (q != null) {
182             // Matching options
183             boolean includeMatch = params.getBool(MoreLikeThisParams.MATCH_INCLUDE,
184                 true);
185             int matchOffset = params.getInt(MoreLikeThisParams.MATCH_OFFSET, 0);
186             // Find the base match
187             DocList match = searcher.getDocList(query, null, null, matchOffset, 1,
188                 flags); // only get the first one...
189             if (includeMatch) {
190               rsp.add("match", match);
191             }
192 
193             // This is an iterator, but we only handle the first match
194             DocIterator iterator = match.iterator();
195             if (iterator.hasNext()) {
196               // do a MoreLikeThis query for each document in results
197               int id = iterator.nextDoc();
198               mltDocs = mlt.getMoreLikeThis(id, start, rows, filters, interesting,
199                   flags);
200             }
201           } else {
202             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
203                 "MoreLikeThis requires either a query (?q=) or text to find similar documents.");
204           }
205 
206         } finally {
207           if (reader != null) {
208             reader.close();
209           }
210         }
211 
212         if (mltDocs == null) {
213           mltDocs = new DocListAndSet(); // avoid NPE
214         }
215         rsp.add("response", mltDocs.docList);
216 
217 
218         if (interesting != null) {
219           if (termStyle == TermStyle.DETAILS) {
220             NamedList<Float> it = new NamedList<>();
221             for (InterestingTerm t : interesting) {
222               it.add(t.term.toString(), t.boost);
223             }
224             rsp.add("interestingTerms", it);
225           } else {
226             List<String> it = new ArrayList<>(interesting.size());
227             for (InterestingTerm t : interesting) {
228               it.add(t.term.text());
229             }
230             rsp.add("interestingTerms", it);
231           }
232         }
233 
234         // maybe facet the results
235         if (params.getBool(FacetParams.FACET, false)) {
236           if (mltDocs.docSet == null) {
237             rsp.add("facet_counts", null);
238           } else {
239             SimpleFacets f = new SimpleFacets(req, mltDocs.docSet, params);
240             rsp.add("facet_counts", FacetComponent.getFacetCounts(f));
241           }
242         }
243         boolean dbg = req.getParams().getBool(CommonParams.DEBUG_QUERY, false);
244 
245         boolean dbgQuery = false, dbgResults = false;
246         if (dbg == false) {//if it's true, we are doing everything anyway.
247           String[] dbgParams = req.getParams().getParams(CommonParams.DEBUG);
248           if (dbgParams != null) {
249             for (int i = 0; i < dbgParams.length; i++) {
250               if (dbgParams[i].equals(CommonParams.QUERY)) {
251                 dbgQuery = true;
252               } else if (dbgParams[i].equals(CommonParams.RESULTS)) {
253                 dbgResults = true;
254               }
255             }
256           }
257         } else {
258           dbgQuery = true;
259           dbgResults = true;
260         }
261         // Copied from StandardRequestHandler... perhaps it should be added to doStandardDebug?
262         if (dbg == true) {
263           try {
264             NamedList<Object> dbgInfo = SolrPluginUtils.doStandardDebug(req, q, mlt.getRawMLTQuery(), mltDocs.docList, dbgQuery, dbgResults);
265             if (null != dbgInfo) {
266               if (null != filters) {
267                 dbgInfo.add("filter_queries", req.getParams().getParams(CommonParams.FQ));
268                 List<String> fqs = new ArrayList<>(filters.size());
269                 for (Query fq : filters) {
270                   fqs.add(QueryParsing.toString(fq, req.getSchema()));
271                 }
272                 dbgInfo.add("parsed_filter_queries", fqs);
273               }
274               rsp.add("debug", dbgInfo);
275             }
276           } catch (Exception e) {
277             SolrException.log(log, "Exception during debug", e);
278             rsp.add("exception_during_debug", SolrException.toStr(e));
279           }
280         }
281       } catch (ExitableDirectoryReader.ExitingReaderException ex) {
282         log.warn( "Query: " + req.getParamString() + "; " + ex.getMessage());
283       } finally {
284         SolrQueryTimeoutImpl.reset();
285       }
286   }

288   public static class InterestingTerm
289   {
290     public Term term;
291     public float boost;
292         
293     public static Comparator<InterestingTerm> BOOST_ORDER = new Comparator<InterestingTerm>() {
294       @Override
295       public int compare(InterestingTerm t1, InterestingTerm t2) {
296         float d = t1.boost - t2.boost;
297         if( d == 0 ) {
298           return 0;
299         }
300         return (d>0)?1:-1;
301       }
302     };
303   }

305   /**
306    * Helper class for MoreLikeThis that can be called from other request handlers
307    */
308   public static class MoreLikeThisHelper 
309   { 
310     final SolrIndexSearcher searcher;
311     final MoreLikeThis mlt;
312     final IndexReader reader;
313     final SchemaField uniqueKeyField;
314     final boolean needDocSet;
315     Map<String,Float> boostFields;
316     
317     public MoreLikeThisHelper( SolrParams params, SolrIndexSearcher searcher )
318     {
319       this.searcher = searcher;
320       this.reader = searcher.getIndexReader();
321       this.uniqueKeyField = searcher.getSchema().getUniqueKeyField();
322       this.needDocSet = params.getBool(FacetParams.FACET,false);
323       
324       SolrParams required = params.required();
325       String[] fl = required.getParams(MoreLikeThisParams.SIMILARITY_FIELDS);
326       List<String> list = new ArrayList<>();
327       for (String f : fl) {
328         if (!StringUtils.isEmpty(f))  {
329           String[] strings = splitList.split(f);
330           for (String string : strings) {
331             if (!StringUtils.isEmpty(string)) {
332               list.add(string);
333             }
334           }
335         }
336       }
337       String[] fields = list.toArray(new String[list.size()]);
338       if( fields.length < 1 ) {
339         throw new SolrException( SolrException.ErrorCode.BAD_REQUEST, 
340             "MoreLikeThis requires at least one similarity field: "+MoreLikeThisParams.SIMILARITY_FIELDS );
341       }
342       
343       this.mlt = new MoreLikeThis( reader ); // TODO -- after LUCENE-896, we can use , searcher.getSimilarity() );
344       mlt.setFieldNames(fields);
345       mlt.setAnalyzer( searcher.getSchema().getIndexAnalyzer() );
346       
347       // configurable params
348       
349       mlt.setMinTermFreq(       params.getInt(MoreLikeThisParams.MIN_TERM_FREQ,         MoreLikeThis.DEFAULT_MIN_TERM_FREQ));
350       mlt.setMinDocFreq(        params.getInt(MoreLikeThisParams.MIN_DOC_FREQ,          MoreLikeThis.DEFAULT_MIN_DOC_FREQ));
351       mlt.setMaxDocFreq(        params.getInt(MoreLikeThisParams.MAX_DOC_FREQ,          MoreLikeThis.DEFAULT_MAX_DOC_FREQ));
352       mlt.setMinWordLen(        params.getInt(MoreLikeThisParams.MIN_WORD_LEN,          MoreLikeThis.DEFAULT_MIN_WORD_LENGTH));
353       mlt.setMaxWordLen(        params.getInt(MoreLikeThisParams.MAX_WORD_LEN,          MoreLikeThis.DEFAULT_MAX_WORD_LENGTH));
354       mlt.setMaxQueryTerms(     params.getInt(MoreLikeThisParams.MAX_QUERY_TERMS,       MoreLikeThis.DEFAULT_MAX_QUERY_TERMS));
355       mlt.setMaxNumTokensParsed(params.getInt(MoreLikeThisParams.MAX_NUM_TOKENS_PARSED, MoreLikeThis.DEFAULT_MAX_NUM_TOKENS_PARSED));
356       mlt.setBoost(            params.getBool(MoreLikeThisParams.BOOST, false ) );
357       boostFields = SolrPluginUtils.parseFieldBoosts(params.getParams(MoreLikeThisParams.QF));
358     }

376     private Query getBoostedQuery(Query mltquery) {
377       BooleanQuery boostedQuery = (BooleanQuery)mltquery;
378       if (boostFields.size() > 0) {
379         BooleanQuery.Builder newQ = new BooleanQuery.Builder();
380         newQ.setDisableCoord(boostedQuery.isCoordDisabled());
381         newQ.setMinimumNumberShouldMatch(boostedQuery.getMinimumNumberShouldMatch());
382         for (BooleanClause clause : boostedQuery) {
383           Query q = clause.getQuery();
384           Float b = this.boostFields.get(((TermQuery) q).getTerm().field());
385           if (b != null) {
386             q = new BoostQuery(q, b);
387           }
388           newQ.add(q, clause.getOccur());
389         }
390         boostedQuery = newQ.build();
391       }
392       return boostedQuery;
393     }

395     public DocListAndSet getMoreLikeThis( int id, int start, int rows, List<Query> filters, List<InterestingTerm> terms, int flags ) throws IOException
396     {
397       Document doc = reader.document(id);
398       rawMLTQuery = mlt.like(id);
399       boostedMLTQuery = getBoostedQuery( rawMLTQuery );
400       if( terms != null ) {
401         fillInterestingTermsFromMLTQuery( rawMLTQuery, terms );
402       }
403 
404       // exclude current document from results
405       BooleanQuery.Builder realMLTQuery = new BooleanQuery.Builder();
406       realMLTQuery.add(boostedMLTQuery, BooleanClause.Occur.MUST);
407       realMLTQuery.add(
408           new TermQuery(new Term(uniqueKeyField.getName(), uniqueKeyField.getType().storedToIndexed(doc.getField(uniqueKeyField.getName())))), 
409             BooleanClause.Occur.MUST_NOT);
410       this.realMLTQuery = realMLTQuery.build();
411       
412       DocListAndSet results = new DocListAndSet();
413       if (this.needDocSet) {
414         results = searcher.getDocListAndSet(this.realMLTQuery, filters, null, start, rows, flags);
415       } else {
416         results.docList = searcher.getDocList(this.realMLTQuery, filters, null, start, rows, flags);
417       }
418       return results;
419     }

421     public DocListAndSet getMoreLikeThis( Reader reader, int start, int rows, List<Query> filters, List<InterestingTerm> terms, int flags ) throws IOException
422     {
423       // analyzing with the first field: previous (stupid) behavior
424       rawMLTQuery = mlt.like(mlt.getFieldNames()[0], reader);
425       boostedMLTQuery = getBoostedQuery( rawMLTQuery );
426       if( terms != null ) {
427         fillInterestingTermsFromMLTQuery( boostedMLTQuery, terms );
428       }
429       DocListAndSet results = new DocListAndSet();
430       if (this.needDocSet) {
431         results = searcher.getDocListAndSet( boostedMLTQuery, filters, null, start, rows, flags);
432       } else {
433         results.docList = searcher.getDocList( boostedMLTQuery, filters, null, start, rows, flags);
434       }
435       return results;
436     }

455     public NamedList<BooleanQuery> getMoreLikeTheseQuery(DocList docs)
456         throws IOException {
457       IndexSchema schema = searcher.getSchema();
458       NamedList<BooleanQuery> result = new NamedList<>();
459       DocIterator iterator = docs.iterator();
460       while (iterator.hasNext()) {
461         int id = iterator.nextDoc();
462         String uniqueId = schema.printableUniqueKey(reader.document(id));
463 
464         BooleanQuery mltquery = (BooleanQuery) mlt.like(id);
465         if (mltquery.clauses().size() == 0) {
466           return result;
467         }
468         mltquery = (BooleanQuery) getBoostedQuery(mltquery);
469         
470         // exclude current document from results
471         BooleanQuery.Builder mltQuery = new BooleanQuery.Builder();
472         mltQuery.add(mltquery, BooleanClause.Occur.MUST);
473         
474         mltQuery.add(
475             new TermQuery(new Term(uniqueKeyField.getName(), uniqueId)), BooleanClause.Occur.MUST_NOT);
476         result.add(uniqueId, mltQuery.build());
477       }
478 
479       return result;
480     }
481     
482     private void fillInterestingTermsFromMLTQuery( Query query, List<InterestingTerm> terms )
483     { 
484       Collection<BooleanClause> clauses = ((BooleanQuery)query).clauses();
485       for( BooleanClause o : clauses ) {
486         Query q = o.getQuery();
487         float boost = 1f;
488         if (query instanceof BoostQuery) {
489           BoostQuery bq = (BoostQuery) q;
490           q = bq.getQuery();
491           boost = bq.getBoost();
492         }
493         InterestingTerm it = new InterestingTerm();
494         it.boost = boost;
495         it.term = ((TermQuery) q).getTerm();
496         terms.add( it );
497       } 
498       // alternatively we could use
499       // mltquery.extractTerms( terms );
500     }
