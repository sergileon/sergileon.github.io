58  /**
59   * <p>
60   * Randomized testing of Pivot Faceting using SolrCloud.
61   * </p>
62   * <p>
63   * After indexing a bunch of random docs, picks some random fields to pivot facet on, 
64   * and then confirms that the resulting counts match the results of filtering on those 
65   * values.  This gives us strong assertions on the correctness of the total counts for 
66   * each pivot value, but no assertions that the correct "top" counts were chosen.
67   * </p>
68   * <p>
69   * NOTE: this test ignores the control collection and only deals with the 
70   * CloudSolrServer - this is because the randomized field values make it very easy for 
71   * the term stats to miss values even with the overrequest.
72   * (because so many values will tie for "1").  What we care about here is 
73   * that the counts we get back are correct and match what we get when filtering on those 
74   * constraints.
75   * </p>
76   *
77   *
78   *
79   */

105   @Test
106   public void test() throws Exception {
107 
108     sanityCheckAssertNumerics();
109 
110     waitForThingsToLevelOut(30000); // TODO: why whould we have to wait?
111     // 
112     handle.clear();
113     handle.put("QTime", SKIPVAL);
114     handle.put("timestamp", SKIPVAL);
115     
116     final Set<String> fieldNameSet = new HashSet<>();
117     
118     // build up a randomized index
119     final int numDocs = atLeast(500);
120     log.info("numDocs: {}", numDocs);
121 
122     for (int i = 1; i <= numDocs; i++) {
123       SolrInputDocument doc = buildRandomDocument(i);
124 
125       // not efficient, but it guarantees that even if people change buildRandomDocument
126       // we'll always have the full list of fields w/o needing to keep code in sync
127       fieldNameSet.addAll(doc.getFieldNames());
128 
129       cloudClient.add(doc);
130     }
131     cloudClient.commit();
132 
133     fieldNameSet.remove("id");
134     assertTrue("WTF, bogus field exists?", fieldNameSet.add("bogus_not_in_any_doc_s"));
135 
136     final String[] fieldNames = fieldNameSet.toArray(new String[fieldNameSet.size()]);
137     Arrays.sort(fieldNames); // need determinism when picking random fields
138 
139 
140     for (int i = 0; i < 5; i++) {
141 
142       String q = "*:*";
143       if (random().nextBoolean()) {
144         q = "id:[* TO " + TestUtil.nextInt(random(),300,numDocs) + "]";
145       }
146       ModifiableSolrParams baseP = params("rows", "0", "q", q);
147       
148       if (random().nextBoolean()) {
149         baseP.add("fq", "id:[* TO " + TestUtil.nextInt(random(),200,numDocs) + "]");
150       }
151 
152       final boolean stats = random().nextBoolean();
153       if (stats) {
154         baseP.add(StatsParams.STATS, "true");
155         
156         // if we are doing stats, then always generated the same # of STATS_FIELD
157         // params, using multiple tags from a fixed set, but with diff fieldName values.
158         // later, each pivot will randomly pick a tag.
159         baseP.add(StatsParams.STATS_FIELD, "{!key=sk1 tag=st1,st2}" +
160                   pickRandomStatsFields(fieldNames));
161         baseP.add(StatsParams.STATS_FIELD, "{!key=sk2 tag=st2,st3}" +
162                   pickRandomStatsFields(fieldNames));
163         baseP.add(StatsParams.STATS_FIELD, "{!key=sk3 tag=st3,st4}" +
164                   pickRandomStatsFields(fieldNames));
165         // NOTE: there's a chance that some of those stats field names
166         // will be the same, but if so, all the better to test that edge case
167       }
168       
169       ModifiableSolrParams pivotP = params(FACET,"true");
170 
171       // put our FACET_PIVOT params in a set in case we just happen to pick the same one twice
172       LinkedHashSet<String> pivotParamValues = new LinkedHashSet<String>();
173       pivotParamValues.add(buildPivotParamValue(buildRandomPivot(fieldNames)));
174                  
175       if (random().nextBoolean()) {
176         pivotParamValues.add(buildPivotParamValue(buildRandomPivot(fieldNames)));
177       }
178       pivotP.set(FACET_PIVOT, pivotParamValues.toArray(new String[pivotParamValues.size()]));
179 
180       // keep limit low - lots of unique values, and lots of depth in pivots
181       pivotP.add(FACET_LIMIT, ""+TestUtil.nextInt(random(),1,17));
182 
183       // sometimes use an offset
184       if (random().nextBoolean()) {
185         pivotP.add(FACET_OFFSET, ""+TestUtil.nextInt(random(),0,7));
186       }
187 
188       if (random().nextBoolean()) {
189         String min = ""+TestUtil.nextInt(random(),0,numDocs+10);
190         pivotP.add(FACET_PIVOT_MINCOUNT, min);
191         // trace param for validation
192         baseP.add(TRACE_MIN, min);
193       }
194 
195       if (random().nextBoolean()) {
196         String missing = ""+random().nextBoolean();
197         pivotP.add(FACET_MISSING, missing);
198         // trace param for validation
199         baseP.add(TRACE_MISS, missing);
200       }
201 
202       if (random().nextBoolean()) {
203         String sort = random().nextBoolean() ? "index" : "count";
204         pivotP.add(FACET_SORT, sort);
205         // trace param for validation
206         baseP.add(TRACE_SORT, sort);
207       }
208 
209       // overrequest
210       //
211       // NOTE: since this test focuses on accuracy of refinement, and doesn't do 
212       // control collection comparisons, there isn't a lot of need for excessive
213       // overrequesting -- we focus here on trying to exercise the various edge cases
214       // involved as different values are used with overrequest
215       if (0 == TestUtil.nextInt(random(),0,4)) {
216         // we want a decent chance of no overrequest at all
217         pivotP.add(FACET_OVERREQUEST_COUNT, "0");
218         pivotP.add(FACET_OVERREQUEST_RATIO, "0");
219       } else {
220         if (random().nextBoolean()) {
221           pivotP.add(FACET_OVERREQUEST_COUNT, ""+TestUtil.nextInt(random(),0,5));
222         }
223         if (random().nextBoolean()) {
224           // sometimes give a ratio less then 1, code should be smart enough to deal
225           float ratio = 0.5F + random().nextFloat();
226           // sometimes go negative
227           if (random().nextBoolean()) {
228             ratio *= -1;
229           }
230           pivotP.add(FACET_OVERREQUEST_RATIO, ""+ratio);
231         }
232       }
233       
234       assertPivotCountsAreCorrect(baseP, pivotP);
235     }

238   /**
239    * Given some query params, executes the request against the cloudClient and 
240    * then walks the pivot facet values in the response, treating each one as a 
241    * filter query to assert the pivot counts are correct.
242    */
243   private void assertPivotCountsAreCorrect(SolrParams baseParams, 
244                                            SolrParams pivotParams) 
245     throws SolrServerException {
246     
247     SolrParams initParams = SolrParams.wrapAppended(pivotParams, baseParams);
248 
249     log.info("Doing full run: {}", initParams);
250     countNumFoundChecks = 0;
251 
252     NamedList<List<PivotField>> pivots = null;
253     try {
254       QueryResponse initResponse = cloudClient.query(initParams);
255       pivots = initResponse.getFacetPivot();
256       assertNotNull(initParams + " has null pivots?", pivots);
257       assertEquals(initParams + " num pivots", 
258                    initParams.getParams("facet.pivot").length, pivots.size());
259     } catch (Exception e) {
260       throw new RuntimeException("init query failed: " + initParams + ": " + 
261                                  e.getMessage(), e);
262     }
263     try {
264       for (Map.Entry<String,List<PivotField>> pivot : pivots) {
265         final String pivotKey = pivot.getKey();
266         // :HACK: for counting the max possible pivot depth
267         final int maxDepth = 1 + pivotKey.length() - pivotKey.replace(",","").length();
268 
269         assertTraceOk(pivotKey, baseParams, pivot.getValue());
270 
271         // NOTE: we can't make any assumptions/assertions about the number of
272         // constraints here because of the random data - which means if pivotting is
273         // completely broken and there are no constrains this loop could be a No-Op
274         // but in that case we just have to trust that DistributedFacetPivotTest
275         // will catch it.
276         for (PivotField constraint : pivot.getValue()) {
277           int depth = assertPivotCountsAreCorrect(pivotKey, baseParams, constraint);
278           
279           // we can't assert that the depth reached is the same as the depth requested
280           // because the fq and/or mincount may have pruned the tree too much
281           assertTrue("went too deep: "+depth+": " + pivotKey + " ==> " + pivot,
282                      depth <= maxDepth);
283 
284         }
285       }
286     } catch (AssertionError e) {
287       throw new AssertionError(initParams + " ==> " + e.getMessage(), e);
288     } finally {
289       log.info("Ending full run (countNumFoundChecks={}): {}", 
290                countNumFoundChecks, initParams);
291     }
292   }

294   /**
295    * Recursive Helper method for asserting that pivot constraint counts match
296    * results when filtering on those constraints. Returns the recursive depth reached 
297    * (for sanity checking)
298    */
299   private int assertPivotCountsAreCorrect(String pivotName,
300                                           SolrParams baseParams, 
301                                           PivotField constraint) 
302     throws SolrServerException {
303 
304     SolrParams p = SolrParams.wrapAppended(baseParams,
305                                            params("fq", buildFilter(constraint)));
306     List<PivotField> subPivots = null;
307     try {
308       assertPivotData(pivotName, constraint, p); 
309       subPivots = constraint.getPivot();
310     } catch (Exception e) {
311       throw new RuntimeException(pivotName + ": count query failed: " + p + ": " + 
312                                  e.getMessage(), e);
313     }
314     int depth = 0;
315     if (null != subPivots) {
316       assertTraceOk(pivotName, baseParams, subPivots);
317 
318       for (PivotField subPivot : subPivots) {
319         depth = assertPivotCountsAreCorrect(pivotName, p, subPivot);
320       }
321     }
322     return depth + 1;
323   }

325   /**
326    * Executes a query and compares the results with the data available in the 
327    * {@link PivotField} constraint -- this method is not recursive, and doesn't 
328    * check anything about the sub-pivots (if any).
329    *
330    * @param pivotName pivot name
331    * @param constraint filters on pivot
332    * @param params base solr parameters
333    */
334   private void assertPivotData(String pivotName, PivotField constraint, SolrParams params)
335       throws SolrServerException, IOException {
336     
337     SolrParams p = SolrParams.wrapDefaults(params("rows","0"), params);
338     QueryResponse res = cloudClient.query(p);
339     String msg = pivotName + ": " + p;
340 
341     assertNumFound(msg, constraint.getCount(), res);
342 
343     if ( p.getBool(StatsParams.STATS, false) ) {
344       // only check stats if stats expected
345       assertPivotStats(msg, constraint, res);
346     }
347   }

349   /**
350    * Compare top level stats in response with stats from pivot constraint
351    */
352   private void assertPivotStats(String message, PivotField constraint, QueryResponse response) throws SolrServerException {
353 
354     if (null == constraint.getFieldStatsInfo()) {
355       // no stats for this pivot, nothing to check
356 
357       // TODO: use a trace param to know if/how-many to expect ?
358       log.info("No stats to check for => " + message);
359       return;
360     }
361     
362     Map<String, FieldStatsInfo> actualFieldStatsInfoMap = response.getFieldStatsInfo();
363 
364     for (FieldStatsInfo pivotStats : constraint.getFieldStatsInfo().values()) {
365       String statsKey = pivotStats.getName();
366 
367       FieldStatsInfo actualStats = actualFieldStatsInfoMap.get(statsKey);
368 
369       if (actualStats == null) {
370         // handle case for not found stats (using stats query)
371         //
372         // these has to be a special case check due to the legacy behavior of "top level" 
373         // StatsComponent results being "null" (and not even included in the 
374         // getFieldStatsInfo() Map due to specila SolrJ logic) 
375 
376         log.info("Requested stats missing in verification query, pivot stats: " + pivotStats);
377         assertEquals("Special Count", 0L, pivotStats.getCount().longValue());
378         assertEquals("Special Missing", 
379                      constraint.getCount(), pivotStats.getMissing().longValue());
380 
381       } else {
382         // regular stats, compare everything...
383 
384         assert actualStats != null;
385         String msg = " of " + statsKey + " => " + message;
386 
387         // no wiggle room, these should always be exactly equals, regardless of field type
388         assertEquals("Count" + msg, pivotStats.getCount(), actualStats.getCount());
389         assertEquals("Missing" + msg, pivotStats.getMissing(), actualStats.getMissing());
390         assertEquals("Min" + msg, pivotStats.getMin(), actualStats.getMin());
391         assertEquals("Max" + msg, pivotStats.getMax(), actualStats.getMax());
392 
393         // precision loss can affect these in some field types depending on shards used
394         // and the order that values are accumulated
395         assertNumerics("Sum" + msg, pivotStats.getSum(), actualStats.getSum());
396         assertNumerics("Mean" + msg, pivotStats.getMean(), actualStats.getMean());
397         assertNumerics("Stddev" + msg, pivotStats.getStddev(), actualStats.getStddev());
398         assertNumerics("SumOfSquares" + msg, 
399                       pivotStats.getSumOfSquares(), actualStats.getSumOfSquares());
400       }
401     }
402 
403     if (constraint.getFieldStatsInfo().containsKey("sk2")) { // cheeseball hack
404       // if "sk2" was one of hte stats we computed, then we must have also seen
405       // sk1 or sk3 because of the way the tags are fixed
406       assertEquals("had stats sk2, but not another stat?", 
407                    2, constraint.getFieldStatsInfo().size());
408     } else {
409       // if we did not see "sk2", then 1 of the others must be alone
410       assertEquals("only expected 1 stat",
411                    1, constraint.getFieldStatsInfo().size());
412       assertTrue("not sk1 or sk3", 
413                  constraint.getFieldStatsInfo().containsKey("sk1") ||
414                  constraint.getFieldStatsInfo().containsKey("sk3"));
415     }
416 
417   }

419   /**
420    * Verify that the PivotFields we're lookin at doesn't violate any of the expected 
421    * behaviors based on the <code>TRACE_*</code> params found in the base params
422    */
423   private void assertTraceOk(String pivotName, SolrParams baseParams, List<PivotField> constraints) {
424     if (null == constraints || 0 == constraints.size()) {
425       return;
426     }
427     final int maxIdx = constraints.size() - 1;
428       
429     final int min = baseParams.getInt(TRACE_MIN, -1);
430     final boolean expectMissing = baseParams.getBool(TRACE_MISS, false);
431     final boolean checkCount = "count".equals(baseParams.get(TRACE_SORT, "count"));
432 
433     int prevCount = Integer.MAX_VALUE;
434 
435     for (int i = 0; i <= maxIdx; i++) {
436       final PivotField constraint = constraints.get(i);
437       final int count = constraint.getCount();
438 
439       if (0 < min) {
440         assertTrue(pivotName + ": val #"+i +" of " + maxIdx + 
441                    ": count("+count+") < facet.mincount("+min+"): " + constraint,
442                    min <= count);
443       }
444       // missing value must always come last, but only if facet.missing was used
445       // and may not exist at all (mincount, none missing for this sub-facet, etc...)
446       if ((i < maxIdx) || (!expectMissing)) {
447         assertNotNull(pivotName + ": val #"+i +" of " + maxIdx + 
448                       " has null value: " + constraint,
449                       constraint.getValue());
450       }
451       // if we are expecting count based sort, then the count of each constraint 
452       // must be lt-or-eq the count that came before -- or it must be the last value and 
453       // be "missing"
454       if (checkCount) {
455         assertTrue(pivotName + ": val #"+i +" of" + maxIdx + 
456                    ": count("+count+") > prevCount("+prevCount+"): " + constraint,
457                    ((count <= prevCount)
458                     || (expectMissing && i == maxIdx && null == constraint.getValue())));
459         prevCount = count;
460       }
461     }
462   }

464   /**
465    * Given a PivotField constraint, generate a query for the field+value
466    * for use in an <code>fq</code> to verify the constraint count
467    */
468   private static String buildFilter(PivotField constraint) {
469     Object value = constraint.getValue();
470     if (null == value) {
471       // facet.missing, exclude any indexed term
472       return "-" + constraint.getField() + ":[* TO *]";
473     }
474     // otherwise, build up a term filter...
475     String prefix = "{!term f=" + constraint.getField() + "}";
476     if (value instanceof Date) {
477       return prefix + DateFormatUtil.formatExternal((Date)value);
478     } else {
479       return prefix + value;
480     }
481   }

484   /**
485    * Creates a random facet.pivot param string using some of the specified fieldNames
486    */
487   private static String buildRandomPivot(String[] fieldNames) {
488     final int depth = TestUtil.nextInt(random(), 1, 3);
489     String [] fields = new String[depth];
490     for (int i = 0; i < depth; i++) {
491       // yes this means we might use the same field twice
492       // makes it a robust test (especially for multi-valued fields)
493       fields[i] = fieldNames[TestUtil.nextInt(random(),0,fieldNames.length-1)];
494     }
495     return StringUtils.join(fields, ",");
496   }

498   /**
499    * Picks a random field to use for Stats
500    */
501   private static String pickRandomStatsFields(String[] fieldNames) {
502     // we need to skip boolean fields when computing stats
503     String fieldName;
504     do {
505       fieldName = fieldNames[TestUtil.nextInt(random(),0,fieldNames.length-1)];
506     }
507     while(fieldName.endsWith("_b") || fieldName.endsWith("_b1")) ;
508           
509     return fieldName;
510   }

512   /**
513    * Generates a random {@link FacetParams#FACET_PIVOT} value w/ local params 
514    * using the specified pivotValue.
515    */
516   private static String buildPivotParamValue(String pivotValue) {
517     // randomly decide which stat tag to use
518 
519     // if this is 0, or stats aren't enabled, we'll be asking for a tag that doesn't exist
520     // ...which should be fine (just like excluding a taged fq that doesn't exist)
521     final int statTag = TestUtil.nextInt(random(), -1, 4);
522       
523     if (0 <= statTag) {
524       // only use 1 tag name in the 'stats' localparam - see SOLR-6663
525       return "{!stats=st"+statTag+"}" + pivotValue;
526     } else {
527       // statTag < 0 == sanity check the case of a pivot w/o any stats
528       return pivotValue;
529     }

532   /**
533    * Creates a document with randomized field values, some of which be missing values, 
534    * some of which will be multi-valued (per the schema) and some of which will be 
535    * skewed so that small subsets of the ranges will be more common (resulting in an 
536    * increased likelihood of duplicate values)
537    * 
538    * @see #buildRandomPivot
539    */
540   private static SolrInputDocument buildRandomDocument(int id) {
541     SolrInputDocument doc = sdoc("id", id);
542     // most fields are in most docs
543     // if field is in a doc, then "skewed" chance val is from a dense range
544     // (hopefully with lots of duplication)
545     for (String prefix : new String[] { "pivot_i", "pivot_ti" }) {
546       if (useField()) {
547         doc.addField(prefix+"1", skewed(TestUtil.nextInt(random(), 20, 50),
548                                         random().nextInt()));
549                                         
550       }
551       if (useField()) {
552         int numMulti = atLeast(1);
553         while (0 < numMulti--) {
554           doc.addField(prefix, skewed(TestUtil.nextInt(random(), 20, 50), 
555                                       random().nextInt()));
556         }
557       }
558     }
559     for (String prefix : new String[] { "pivot_l", "pivot_tl" }) {
560       if (useField()) {
561         doc.addField(prefix+"1", skewed(TestUtil.nextInt(random(), 5000, 5100),
562                                         random().nextLong()));
563       }
564       if (useField()) {
565         int numMulti = atLeast(1);
566         while (0 < numMulti--) {
567           doc.addField(prefix, skewed(TestUtil.nextInt(random(), 5000, 5100), 
568                                       random().nextLong()));
569         }
570       }
571     }
572     for (String prefix : new String[] { "pivot_f", "pivot_tf" }) {
573       if (useField()) {
574         doc.addField(prefix+"1", skewed(1.0F / random().nextInt(13),
575                                         random().nextFloat() * random().nextInt()));
576       }
577       if (useField()) {
578         int numMulti = atLeast(1);
579         while (0 < numMulti--) {
580           doc.addField(prefix, skewed(1.0F / random().nextInt(13),
581                                       random().nextFloat() * random().nextInt()));
582         }
583       }
584     }
585     for (String prefix : new String[] { "pivot_d", "pivot_td" }) {
586       if (useField()) {
587         doc.addField(prefix+"1", skewed(1.0D / random().nextInt(19),
588                                         random().nextDouble() * random().nextInt()));
589       }
590       if (useField()) {
591         int numMulti = atLeast(1);
592         while (0 < numMulti--) {
593           doc.addField(prefix, skewed(1.0D / random().nextInt(19),
594                                       random().nextDouble() * random().nextInt()));
595         }
596       }
597     }
598     for (String prefix : new String[] { "pivot_dt", "pivot_tdt" }) {
599       if (useField()) {
600         doc.addField(prefix+"1", skewed(randomSkewedDate(), randomDate()));
601                                         
602       }
603       if (useField()) {
604         int numMulti = atLeast(1);
605         while (0 < numMulti--) {
606           doc.addField(prefix, skewed(randomSkewedDate(), randomDate()));
607                                       
608         }
609       }
610     }
611     {
612       String prefix = "pivot_b";
613       if (useField()) {
614         doc.addField(prefix+"1", random().nextBoolean() ? "t" : "f");
615       }
616       if (useField()) {
617         int numMulti = atLeast(1);
618         while (0 < numMulti--) {
619           doc.addField(prefix, random().nextBoolean() ? "t" : "f");
620         }
621       }
622     }
623     for (String prefix : new String[] { "pivot_x_s", "pivot_y_s", "pivot_z_s"}) {
624       if (useField()) {
625         doc.addField(prefix+"1", skewed(TestUtil.randomSimpleString(random(), 1, 1),
626                                         randomXmlUsableUnicodeString()));
627       }
628       if (useField()) {
629         int numMulti = atLeast(1);
630         while (0 < numMulti--) {
631           doc.addField(prefix, skewed(TestUtil.randomSimpleString(random(), 1, 1),
632                                       randomXmlUsableUnicodeString()));
633         }
634       }
635     }
636 
637     //
638     // for the remaining fields, make every doc have a value in a dense range
639     //
640 
641     for (String prefix : new String[] { "dense_pivot_x_s", "dense_pivot_y_s" }) {
642       if (useField()) {
643         doc.addField(prefix+"1", TestUtil.randomSimpleString(random(), 1, 1));
644       }
645       if (useField()) {
646         int numMulti = atLeast(1);
647         while (0 < numMulti--) {
648           doc.addField(prefix, TestUtil.randomSimpleString(random(), 1, 1));
649         }
650       }
651     }
652     for (String prefix : new String[] { "dense_pivot_i", "dense_pivot_ti" }) {
653       if (useField()) {
654         doc.addField(prefix+"1", TestUtil.nextInt(random(), 20, 50));
655       }
656       if (useField()) {
657         int numMulti = atLeast(1);
658         while (0 < numMulti--) {
659           doc.addField(prefix, TestUtil.nextInt(random(), 20, 50));
660         }
661       }
662     }
663 
664     return doc;
665   }

689   /**
690    * Given two objects returned as stat values asserts that they are they are either both <code>null</code> 
691    * or all of the following are true:
692    * <ul>
693    *  <li>They have the exact same class</li>
694    *  <li>They are both Numbers or they are both Dates -- in the later case, their millisecond's 
695    *      since epoch are used for all subsequent comparisons
696    *  </li>
697    *  <li>Either:
698    *   <ul>
699    *    <li>They are Integer or Long objects with the exact same <code>longValue()</code></li>
700    *    <li>They are Float or Double objects and their <code>doubleValue()</code>s
701    *        are equally-ish with a "small" epsilon (relative to the scale of the expected value)
702    *    </li>
703    *   </ul>
704    *  </li>
705    * <ul>
706    *
707    * @see Date#getTime
708    * @see Number#doubleValue
709    * @see Number#longValue
710    * @see #assertEquals(String,double,double,double)
711    */
712   private void assertNumerics(String msg, Object expected, Object actual) {
713     if (null == expected || null == actual) {
714       assertEquals(msg, expected, actual);
715       return;
716     }
717     
718     assertEquals(msg + " ... values do not have the same type: " + expected + " vs " + actual,
719                  expected.getClass(), actual.getClass());
720 
721     if (expected instanceof Date) {
722       expected = ((Date)expected).getTime();
723       actual = ((Date)actual).getTime();
724       msg = msg + " (w/dates converted to ms)";
725     }
726     
727     assertTrue(msg + " ... expected is not a Number: " + 
728                expected + "=>" + expected.getClass(),
729                expected instanceof Number);
730         
731     if (expected instanceof Long || expected instanceof Integer) {
732       assertEquals(msg, ((Number)expected).longValue(), ((Number)actual).longValue());
733       
734     } else if (expected instanceof Float || expected instanceof Double) {
735       // compute an epsilon relative to the size of the expected value
736       double expect = ((Number)expected).doubleValue();
737       double epsilon = Math.abs(expect * 0.1E-7D);
738 
739       assertEquals(msg, expect, ((Number)actual).doubleValue(), epsilon);
740       
741     } else {
742       fail(msg + " ... where did this come from: " + expected.getClass());
743     }
744   }
