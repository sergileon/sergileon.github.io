137   private char savedBuffer[] = new char[1024];
138   private int savedStartOffset;

180   @Override
181   public boolean incrementToken() throws IOException {
182     while (true) {
183       if (!hasSavedState) {
184         // process a new input word
185         if (!input.incrementToken()) {
186           return false;
187         }
188 
189         int termLength = termAttribute.length();
190         char[] termBuffer = termAttribute.buffer();
191         
192         accumPosInc += posIncAttribute.getPositionIncrement();
193 
194         iterator.setText(termBuffer, termLength);
195         iterator.next();
196 
197         // word of no delimiters, or protected word: just return it
198         if ((iterator.current == 0 && iterator.end == termLength) ||
199             (protWords != null && protWords.contains(termBuffer, 0, termLength))) {
200           posIncAttribute.setPositionIncrement(accumPosInc);
201           accumPosInc = 0;
202           return true;
203         }
204         
205         // word of simply delimiters
206         if (iterator.end == WordDelimiterIterator.DONE && !has(PRESERVE_ORIGINAL)) {
207           // if the posInc is 1, simply ignore it in the accumulation
208           if (posIncAttribute.getPositionIncrement() == 1) {
209             accumPosInc--;
210           }
211           continue;
212         }
213 
214         saveState();
215 
216         hasOutputToken = false;
217         hasOutputFollowingOriginal = !has(PRESERVE_ORIGINAL);
218         lastConcatCount = 0;
219         
220         if (has(PRESERVE_ORIGINAL)) {
221           posIncAttribute.setPositionIncrement(accumPosInc);
222           accumPosInc = 0;
223           return true;
224         }
225       }
226       
227       // at the end of the string, output any concatenations
228       if (iterator.end == WordDelimiterIterator.DONE) {
229         if (!concat.isEmpty()) {
230           if (flushConcatenation(concat)) {
231             return true;
232           }
233         }
234         
235         if (!concatAll.isEmpty()) {
236           // only if we haven't output this same combo above!
237           if (concatAll.subwordCount > lastConcatCount) {
238             concatAll.writeAndClear();
239             return true;
240           }
241           concatAll.clear();
242         }
243         
244         // no saved concatenations, on to the next input word
245         hasSavedState = false;
246         continue;
247       }
248       
249       // word surrounded by delimiters: always output
250       if (iterator.isSingleWord()) {
251         generatePart(true);
252         iterator.next();
253         return true;
254       }
255       
256       int wordType = iterator.type();
257       
258       // do we already have queued up incompatible concatenations?
259       if (!concat.isEmpty() && (concat.type & wordType) == 0) {
260         if (flushConcatenation(concat)) {
261           hasOutputToken = false;
262           return true;
263         }
264         hasOutputToken = false;
265       }
266       
267       // add subwords depending upon options
268       if (shouldConcatenate(wordType)) {
269         if (concat.isEmpty()) {
270           concat.type = wordType;
271         }
272         concatenate(concat);
273       }
274       
275       // add all subwords (catenateAll)
276       if (has(CATENATE_ALL)) {
277         concatenate(concatAll);
278       }
279       
280       // if we should output the word or number part
281       if (shouldGenerateParts(wordType)) {
282         generatePart(false);
283         iterator.next();
284         return true;
285       }
286         
287       iterator.next();
288     }
289   }

305   /**
306    * Saves the existing attribute states
307    */
308   private void saveState() {
309     // otherwise, we have delimiters, save state
310     savedStartOffset = offsetAttribute.startOffset();
311     savedEndOffset = offsetAttribute.endOffset();
312     // if length by start + end offsets doesn't match the term text then assume this is a synonym and don't adjust the offsets.
313     hasIllegalOffsets = (savedEndOffset - savedStartOffset != termAttribute.length());
314     savedType = typeAttribute.type();
315 
316     if (savedBuffer.length < termAttribute.length()) {
317       savedBuffer = new char[ArrayUtil.oversize(termAttribute.length(), RamUsageEstimator.NUM_BYTES_CHAR)];
318     }
319 
320     System.arraycopy(termAttribute.buffer(), 0, savedBuffer, 0, termAttribute.length());
321     iterator.text = savedBuffer;
322 
323     hasSavedState = true;
324   }

326   /**
327    * Flushes the given WordDelimiterConcatenation by either writing its concat and then clearing, or just clearing.
328    *
329    * @param concatenation WordDelimiterConcatenation that will be flushed
330    * @return {@code true} if the concatenation was written before it was cleared, {@code false} otherwise
331    */
332   private boolean flushConcatenation(WordDelimiterConcatenation concatenation) {
333     lastConcatCount = concatenation.subwordCount;
334     if (concatenation.subwordCount != 1 || !shouldGenerateParts(concatenation.type)) {
335       concatenation.writeAndClear();
336       return true;
337     }
338     concatenation.clear();
339     return false;
340   }

342   /**
343    * Determines whether to concatenate a word or number if the current word is the given type
344    *
345    * @param wordType Type of the current word used to determine if it should be concatenated
346    * @return {@code true} if concatenation should occur, {@code false} otherwise
347    */
348   private boolean shouldConcatenate(int wordType) {
349     return (has(CATENATE_WORDS) && isAlpha(wordType)) || (has(CATENATE_NUMBERS) && isDigit(wordType));
350   }
351 
352   /**
353    * Determines whether a word/number part should be generated for a word of the given type
354    *
355    * @param wordType Type of the word used to determine if a word/number part should be generated
356    * @return {@code true} if a word/number part should be generated, {@code false} otherwise
357    */
358   private boolean shouldGenerateParts(int wordType) {
359     return (has(GENERATE_WORD_PARTS) && isAlpha(wordType)) || (has(GENERATE_NUMBER_PARTS) && isDigit(wordType));
360   }

362   /**
363    * Concatenates the saved buffer to the given WordDelimiterConcatenation
364    *
365    * @param concatenation WordDelimiterConcatenation to concatenate the buffer to
366    */
367   private void concatenate(WordDelimiterConcatenation concatenation) {
368     if (concatenation.isEmpty()) {
369       concatenation.startOffset = savedStartOffset + iterator.current;
370     }
371     concatenation.append(savedBuffer, iterator.current, iterator.end - iterator.current);
372     concatenation.endOffset = savedStartOffset + iterator.end;
373   }
374 
375   /**
376    * Generates a word/number part, updating the appropriate attributes
377    *
378    * @param isSingleWord {@code true} if the generation is occurring from a single word, {@code false} otherwise
379    */
380   private void generatePart(boolean isSingleWord) {
381     clearAttributes();
382     termAttribute.copyBuffer(savedBuffer, iterator.current, iterator.end - iterator.current);
383 
384     int startOffset = savedStartOffset + iterator.current;
385     int endOffset = savedStartOffset + iterator.end;
386     
387     if (hasIllegalOffsets) {
388       // historically this filter did this regardless for 'isSingleWord', 
389       // but we must do a sanity check:
390       if (isSingleWord && startOffset <= savedEndOffset) {
391         offsetAttribute.setOffset(startOffset, savedEndOffset);
392       } else {
393         offsetAttribute.setOffset(savedStartOffset, savedEndOffset);
394       }
395     } else {
396       offsetAttribute.setOffset(startOffset, endOffset);
397     }
398     posIncAttribute.setPositionIncrement(position(false));
399     typeAttribute.setType(savedType);
400   }

402   /**
403    * Get the position increment gap for a subword or concatenation
404    *
405    * @param inject true if this token wants to be injected
406    * @return position increment gap
407    */
408   private int position(boolean inject) {
409     int posInc = accumPosInc;
410 
411     if (hasOutputToken) {
412       accumPosInc = 0;
413       return inject ? 0 : Math.max(1, posInc);
414     }
415 
416     hasOutputToken = true;
417     
418     if (!hasOutputFollowingOriginal) {
419       // the first token following the original is 0 regardless
420       hasOutputFollowingOriginal = true;
421       return 0;
422     }
423     // clear the accumulated position increment
424     accumPosInc = 0;
425     return Math.max(1, posInc);
426   }

480   /**
481    * A WDF concatenated 'run'
482    */
483   final class WordDelimiterConcatenation {
484     final StringBuilder buffer = new StringBuilder();
485     int startOffset;
486     int endOffset;
487     int type;
488     int subwordCount;
489 
490     /**
491      * Appends the given text of the given length, to the concetenation at the given offset
492      *
493      * @param text Text to append
494      * @param offset Offset in the concetenation to add the text
495      * @param length Length of the text to append
496      */
497     void append(char text[], int offset, int length) {
498       buffer.append(text, offset, length);
499       subwordCount++;
500     }
501 
502     /**
503      * Writes the concatenation to the attributes
504      */
505     void write() {
506       clearAttributes();
507       if (termAttribute.length() < buffer.length()) {
508         termAttribute.resizeBuffer(buffer.length());
509       }
510       char termbuffer[] = termAttribute.buffer();
511       
512       buffer.getChars(0, buffer.length(), termbuffer, 0);
513       termAttribute.setLength(buffer.length());
514         
515       if (hasIllegalOffsets) {
516         offsetAttribute.setOffset(savedStartOffset, savedEndOffset);
517       }
518       else {
519         offsetAttribute.setOffset(startOffset, endOffset);
520       }
521       posIncAttribute.setPositionIncrement(position(true));
522       typeAttribute.setType(savedType);
523       accumPosInc = 0;
524     }
525 
526     /**
527      * Determines if the concatenation is empty
528      *
529      * @return {@code true} if the concatenation is empty, {@code false} otherwise
530      */
531     boolean isEmpty() {
532       return buffer.length() == 0;
533     }
534 
535     /**
536      * Clears the concatenation and resets its state
537      */
538     void clear() {
539       buffer.setLength(0);
540       startOffset = endOffset = type = subwordCount = 0;
541     }
542 
543     /**
544      * Convenience method for the common scenario of having to write the concetenation and then clearing its state
545      */
546     void writeAndClear() {
547       write();
548       clear();
549     }
550   }
