20  import org.apache.lucene.index.IndexReader.ReaderClosedListener;
21  import org.apache.lucene.util.Bits;
22  
23  import java.io.IOException;
24  
25  /** {@code LeafReader} is an abstract class, providing an interface for accessing an
26   index.  Search of an index is done entirely through this abstract interface,
27   so that any subclass which implements it is searchable. IndexReaders implemented
28   by this subclass do not consist of several sub-readers,
29   they are atomic. They support retrieval of stored fields, doc values, terms,
30   and postings.
31  
32   <p>For efficiency, in this API documents are often referred to via
33   <i>document numbers</i>, non-negative integers which each name a unique
34   document in the index.  These document numbers are ephemeral -- they may change
35   as documents are added to and deleted from an index.  Clients should thus not
36   rely on a given document having the same number between sessions.
37  
38   <p>
39   <a name="thread-safety"></a><p><b>NOTE</b>: {@link
40   IndexReader} instances are completely thread
41   safe, meaning multiple threads can call any of its methods,
42   concurrently.  If your application requires external
43   synchronization, you should <b>not</b> synchronize on the
44   <code>IndexReader</code> instance; use your own
45   (non-Lucene) objects instead.
46  */
47  public abstract class LeafReader extends IndexReader {
48  
49    private final LeafReaderContext readerContext = new LeafReaderContext(this);
50  
51    /** Sole constructor. (For invocation by subclass
52     *  constructors, typically implicit.) */
53    protected LeafReader() {
54      super();
55    }
56  
57    @Override
58    public final LeafReaderContext getContext() {
59      ensureOpen();
60      return readerContext;
61    }
62  
63    /**
64     * Called when the shared core for this {@link LeafReader}
65     * is closed.
66     * <p>
67     * If this {@link LeafReader} impl has the ability to share
68     * resources across instances that might only vary through
69     * deleted documents and doc values updates, then this listener
70     * will only be called when the shared core is closed.
71     * Otherwise, this listener will be called when this reader is
72     * closed.</p>
73     * <p>
74     * This is typically useful to manage per-segment caches: when
75     * the listener is called, it is safe to evict this reader from
76     * any caches keyed on {@link #getCoreCacheKey}.</p>
77     *
78     * @lucene.experimental
79     */
80    public static interface CoreClosedListener {
81      /** Invoked when the shared core of the original {@code
82       *  SegmentReader} has closed. The provided {@code
83       *  ownerCoreCacheKey} will be the same key as the one
84       *  returned by {@link LeafReader#getCoreCacheKey()}. */
85      public void onClose(Object ownerCoreCacheKey) throws IOException;
86    }
87  
88    private static class CoreClosedListenerWrapper implements ReaderClosedListener {
89  
90      private final CoreClosedListener listener;
91  
92      CoreClosedListenerWrapper(CoreClosedListener listener) {
93        this.listener = listener;
94      }
95  
96      @Override
97      public void onClose(IndexReader reader) throws IOException {
98        listener.onClose(reader.getCoreCacheKey());
99      }
100 
101     @Override
102     public int hashCode() {
103       return listener.hashCode();
104     }
105 
106     @Override
107     public boolean equals(Object other) {
108       if (!(other instanceof CoreClosedListenerWrapper)) {
109         return false;
110       }
111       return listener.equals(((CoreClosedListenerWrapper) other).listener);
112     }
113 
114   }
115 
116   /** Add a {@link CoreClosedListener} as a {@link ReaderClosedListener}. This
117    * method is typically useful for {@link LeafReader} implementations that
118    * don't have the concept of a core that is shared across several
119    * {@link LeafReader} instances in which case the {@link CoreClosedListener}
120    * is called when closing the reader. */
121   protected static void addCoreClosedListenerAsReaderClosedListener(IndexReader reader, CoreClosedListener listener) {
122     reader.addReaderClosedListener(new CoreClosedListenerWrapper(listener));
123   }
124 
125   /** Remove a {@link CoreClosedListener} which has been added with
126    * {@link #addCoreClosedListenerAsReaderClosedListener(IndexReader, CoreClosedListener)}. */
127   protected static void removeCoreClosedListenerAsReaderClosedListener(IndexReader reader, CoreClosedListener listener) {
128     reader.removeReaderClosedListener(new CoreClosedListenerWrapper(listener));
129   }
130 
131   /** Expert: adds a CoreClosedListener to this reader's shared core
132    *  @lucene.experimental */
133   public abstract void addCoreClosedListener(CoreClosedListener listener);
134 
135   /** Expert: removes a CoreClosedListener from this reader's shared core
136    *  @lucene.experimental */
137   public abstract void removeCoreClosedListener(CoreClosedListener listener);
138 
139   /**
140    * Returns {@link Fields} for this reader.
141    * This method will not return null.
142    */
143   public abstract Fields fields() throws IOException;
144 
145   @Override
146   public final int docFreq(Term term) throws IOException {
147     final Terms terms = terms(term.field());
148     if (terms == null) {
149       return 0;
150     }
151     final TermsEnum termsEnum = terms.iterator();
152     if (termsEnum.seekExact(term.bytes())) {
153       return termsEnum.docFreq();
154     } else {
155       return 0;
156     }
157   }
158 
159   /** Returns the number of documents containing the term
160    * <code>t</code>.  This method returns 0 if the term or
161    * field does not exists.  This method does not take into
162    * account deleted documents that have not yet been merged
163    * away. */
164   @Override
165   public final long totalTermFreq(Term term) throws IOException {
166     final Terms terms = terms(term.field());
167     if (terms == null) {
168       return 0;
169     }
170     final TermsEnum termsEnum = terms.iterator();
171     if (termsEnum.seekExact(term.bytes())) {
172       return termsEnum.totalTermFreq();
173     } else {
174       return 0;
175     }
176   }
177 
178   @Override
179   public final long getSumDocFreq(String field) throws IOException {
180     final Terms terms = terms(field);
181     if (terms == null) {
182       return 0;
183     }
184     return terms.getSumDocFreq();
185   }
186 
187   @Override
188   public final int getDocCount(String field) throws IOException {
189     final Terms terms = terms(field);
190     if (terms == null) {
191       return 0;
192     }
193     return terms.getDocCount();
194   }
195 
196   @Override
197   public final long getSumTotalTermFreq(String field) throws IOException {
198     final Terms terms = terms(field);
199     if (terms == null) {
200       return 0;
201     }
202     return terms.getSumTotalTermFreq();
203   }
204 
205   /** This may return null if the field does not exist.*/
206   public final Terms terms(String field) throws IOException {
207     return fields().terms(field);
208   }
209 
210   /** Returns {@link PostingsEnum} for the specified term.
211    *  This will return null if either the field or
212    *  term does not exist.
213    *  <p><b>NOTE:</b> The returned {@link PostingsEnum} may contain deleted docs.
214    *  @see TermsEnum#postings(PostingsEnum) */
215   public final PostingsEnum postings(Term term, int flags) throws IOException {
216     assert term.field() != null;
217     assert term.bytes() != null;
218     final Terms terms = terms(term.field());
219     if (terms != null) {
220       final TermsEnum termsEnum = terms.iterator();
221       if (termsEnum.seekExact(term.bytes())) {
222         return termsEnum.postings(null, flags);
223       }
224     }
225     return null;
226   }
227 
228   /** Returns {@link PostingsEnum} for the specified term
229    *  with {@link PostingsEnum#FREQS}.
230    *  <p>
231    *  Use this method if you only require documents and frequencies,
232    *  and do not need any proximity data.
233    *  This method is equivalent to 
234    *  {@link #postings(Term, int) postings(term, PostingsEnum.FREQS)}
235    *  <p><b>NOTE:</b> The returned {@link PostingsEnum} may contain deleted docs.
236    *  @see #postings(Term, int)
237    */
238   public final PostingsEnum postings(Term term) throws IOException {
239     return postings(term, PostingsEnum.FREQS);
240   }
241 
242   /** Returns {@link NumericDocValues} for this field, or
243    *  null if no {@link NumericDocValues} were indexed for
244    *  this field.  The returned instance should only be
245    *  used by a single thread. */
246   public abstract NumericDocValues getNumericDocValues(String field) throws IOException;
247 
248   /** Returns {@link BinaryDocValues} for this field, or
249    *  null if no {@link BinaryDocValues} were indexed for
250    *  this field.  The returned instance should only be
251    *  used by a single thread. */
252   public abstract BinaryDocValues getBinaryDocValues(String field) throws IOException;
253 
254   /** Returns {@link SortedDocValues} for this field, or
255    *  null if no {@link SortedDocValues} were indexed for
256    *  this field.  The returned instance should only be
257    *  used by a single thread. */
258   public abstract SortedDocValues getSortedDocValues(String field) throws IOException;
259   
260   /** Returns {@link SortedNumericDocValues} for this field, or
261    *  null if no {@link SortedNumericDocValues} were indexed for
262    *  this field.  The returned instance should only be
263    *  used by a single thread. */
264   public abstract SortedNumericDocValues getSortedNumericDocValues(String field) throws IOException;
265 
266   /** Returns {@link SortedSetDocValues} for this field, or
267    *  null if no {@link SortedSetDocValues} were indexed for
268    *  this field.  The returned instance should only be
269    *  used by a single thread. */
270   public abstract SortedSetDocValues getSortedSetDocValues(String field) throws IOException;
271 
272   /** Returns a {@link Bits} at the size of <code>reader.maxDoc()</code>,
273    *  with turned on bits for each docid that does have a value for this field,
274    *  or null if no DocValues were indexed for this field. The
275    *  returned instance should only be used by a single thread */
276   public abstract Bits getDocsWithField(String field) throws IOException;
277 
278   /** Returns {@link NumericDocValues} representing norms
279    *  for this field, or null if no {@link NumericDocValues}
280    *  were indexed. The returned instance should only be
281    *  used by a single thread. */
282   public abstract NumericDocValues getNormValues(String field) throws IOException;
283 
284   /**
285    * Get the {@link FieldInfos} describing all fields in
286    * this reader.
287    * @lucene.experimental
288    */
289   public abstract FieldInfos getFieldInfos();
290 
291   /** Returns the {@link Bits} representing live (not
292    *  deleted) docs.  A set bit indicates the doc ID has not
293    *  been deleted.  If this method returns null it means
294    *  there are no deleted documents (all documents are
295    *  live).
296    *
297    *  The returned instance has been safely published for
298    *  use by multiple threads without additional
299    *  synchronization.
300    */
301   public abstract Bits getLiveDocs();
302 
303   /**
304    * Checks consistency of this reader.
305    * <p>
306    * Note that this may be costly in terms of I/O, e.g.
307    * may involve computing a checksum value against large data files.
308    * @lucene.internal
309    */
310   public abstract void checkIntegrity() throws IOException;
311   
312   /** Returns {@link DocsEnum} for the specified term.
313    *  This will return null if either the field or
314    *  term does not exist.
315    *  @deprecated use {@link #postings(Term)} instead */
316   @Deprecated
317   public final DocsEnum termDocsEnum(Term term) throws IOException {
318     assert term.field() != null;
319     assert term.bytes() != null;
320     final Terms terms = terms(term.field());
321     if (terms != null) {
322       final TermsEnum termsEnum = terms.iterator();
323       if (termsEnum.seekExact(term.bytes())) {
324         return termsEnum.docs(getLiveDocs(), null);
325       }
326     }
327     return null;
328   }
329 
330   /** Returns {@link DocsAndPositionsEnum} for the specified
331    *  term.  This will return null if the
332    *  field or term does not exist or positions weren't indexed.
333    *  @deprecated use {@link #postings(Term, int)} instead */
334   @Deprecated
335   public final DocsAndPositionsEnum termPositionsEnum(Term term) throws IOException {
336     assert term.field() != null;
337     assert term.bytes() != null;
338     final Terms terms = terms(term.field());
339     if (terms != null) {
340       final TermsEnum termsEnum = terms.iterator();
341       if (termsEnum.seekExact(term.bytes())) {
342         return termsEnum.docsAndPositions(getLiveDocs(), null);
343       }
344     }
345     return null;
346   }
