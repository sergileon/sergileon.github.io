112   @Test
113   public void testIntersectFilter() throws Exception {
114     setupDocs();
115     
116     //Try some edge cases
117     checkHits(fieldName, "1,1", 175, DistanceUtils.EARTH_MEAN_RADIUS_KM, 3, 5, 6, 7);
118     checkHits(fieldName, "0,179.8", 200, DistanceUtils.EARTH_MEAN_RADIUS_KM, 2, 8, 9);
119     checkHits(fieldName, "89.8, 50", 200, DistanceUtils.EARTH_MEAN_RADIUS_KM, 2, 10, 11);//this goes over the north pole
120     checkHits(fieldName, "-89.8, 50", 200, DistanceUtils.EARTH_MEAN_RADIUS_KM, 2, 12, 13);//this goes over the south pole
121     //try some normal cases
122     checkHits(fieldName, "33.0,-80.0", 300, DistanceUtils.EARTH_MEAN_RADIUS_KM, 2);
123     //large distance
124     checkHits(fieldName, "1,1", 5000, DistanceUtils.EARTH_MEAN_RADIUS_KM, 3, 5, 6, 7);
125     //Because we are generating a box based on the west/east longitudes and the south/north latitudes, which then
126     //translates to a range query, which is slightly more inclusive.  Thus, even though 0.0 is 15.725 kms away,
127     //it will be included, b/c of the box calculation.
128     checkHits(fieldName, false, "0.1,0.1", 15, DistanceUtils.EARTH_MEAN_RADIUS_KM, 2, 5, 6);
129 
130     //try some more
131     clearIndex();
132     assertU(adoc("id", "14", fieldName, "0,5"));
133     assertU(adoc("id", "15", fieldName, "0,15"));
134     //3000KM from 0,0, see http://www.movable-type.co.uk/scripts/latlong.html
135     assertU(adoc("id", "16", fieldName, "18.71111,19.79750"));
136     assertU(adoc("id", "17", fieldName, "44.043900,-95.436643"));
137     assertU(commit());
138 
139     checkHits(fieldName, "0,0", 1000, DistanceUtils.EARTH_MEAN_RADIUS_KM, 1, 14);
140     checkHits(fieldName, "0,0", 2000, DistanceUtils.EARTH_MEAN_RADIUS_KM, 2, 14, 15);
141     checkHits(fieldName, false, "0,0", 3000, DistanceUtils.EARTH_MEAN_RADIUS_KM, 3, 14, 15, 16);
142     checkHits(fieldName, "0,0", 3001, DistanceUtils.EARTH_MEAN_RADIUS_KM, 3, 14, 15, 16);
143     checkHits(fieldName, "0,0", 3000.1, DistanceUtils.EARTH_MEAN_RADIUS_KM, 3, 14, 15, 16);
144 
145     //really fine grained distance and reflects some of the vagaries of how we are calculating the box
146     checkHits(fieldName, "43.517030,-96.789603", 109, DistanceUtils.EARTH_MEAN_RADIUS_KM, 0);
147 
148     //falls outside of the real distance, but inside the bounding box
149     checkHits(fieldName, true,  "43.517030,-96.789603", 110, DistanceUtils.EARTH_MEAN_RADIUS_KM, 0);
150     checkHits(fieldName, false, "43.517030,-96.789603", 110, DistanceUtils.EARTH_MEAN_RADIUS_KM, 1, 17);
151   }

173   private void checkHits(String fieldName, String pt, double distKM, double sphereRadius, int count, int ... docIds) throws ParseException {
174     checkHits(fieldName, true, pt, distKM, sphereRadius, count, docIds);
175   }
176 
177   private void checkHits(String fieldName, boolean exact, String ptStr, double distKM, double sphereRadius, int count, int ... docIds) throws ParseException {
178     if (exact && fieldName.equalsIgnoreCase("bbox")) {
179       return; // bbox field only supports rectangular query
180     }
181     String [] tests = new String[docIds != null && docIds.length > 0 ? docIds.length + 1 : 1];
182     //test for presence of required ids first
183     int i = 0;
184     if (docIds != null && docIds.length > 0) {
185       for (int docId : docIds) {
186         tests[i++] = "//result/doc/*[@name='id'][.='" + docId + "']";
187       }
188     }
189     //check total length last; maybe response includes ids it shouldn't.  Nicer to check this last instead of first so
190     // that there may be a more specific detailed id to investigate.
191     tests[i++] = "*[count(//doc)=" + count + "]";
192 
193     //Test using the Lucene spatial syntax
194     {
195       //never actually need the score but lets test
196       String score = new String[]{null, "none","distance","recipDistance"}[random().nextInt(4)];
197 
198       double distDEG = DistanceUtils.dist2Degrees(distKM, DistanceUtils.EARTH_MEAN_RADIUS_KM);
199       Point point = SpatialUtils.parsePoint(ptStr, SpatialContext.GEO);
200       String circleStr = "BUFFER(POINT(" + point.getX()+" "+point.getY()+")," + distDEG + ")";
201       String shapeStr;
202       if (exact) {
203         shapeStr = circleStr;
204       } else {//bbox
205         //the GEO is an assumption
206         SpatialContext ctx = SpatialContext.GEO;
207         Rectangle bbox = ctx.readShapeFromWkt(circleStr).getBoundingBox();
208         shapeStr = "ENVELOPE(" + bbox.getMinX() + ", " + bbox.getMaxX() +
209             ", " + bbox.getMaxY() + ", " + bbox.getMinY() + ")";
210       }
211 
212       //FYI default distErrPct=0.025 works with the tests in this file
213       assertQ(req(
214             "fl", "id", "q","*:*", "rows", "1000",
215             "fq", "{!field f=" + fieldName + (score==null?"":" score="+score)
216               + "}Intersects(" + shapeStr + ")"),
217           tests);
218     }
219     //Test using geofilt
220     {
221       assertQ(req(
222           "fl", "id", "q", "*:*", "rows", "1000",
223           "fq", "{!" + (exact ? "geofilt" : "bbox") + " sfield=" + fieldName + " pt='" + ptStr + "' d=" + distKM + " sphere_radius=" + sphereRadius + "}"),
224           tests);
225     }
226 
227   }

245   @Test
246   public void testSort() throws Exception {
247     assertU(adoc("id", "100", fieldName, "1,2"));
248     assertU(adoc("id", "101", fieldName, "4,-1"));
249     assertU(adoc("id", "999", fieldName, "70,70"));//far away from these queries
250     assertU(commit());
251 
252     //test absence of score=distance means it doesn't score
253 
254     assertJQ(req(
255         "q", radiusQuery(3, 4, 9, null, null),
256         "fl","id,score")
257         , 1e-9
258         , "/response/docs/[0]/score==1.0"
259         , "/response/docs/[1]/score==1.0"
260     );
261 
262     //score by distance
263     assertJQ(req(
264         "q", radiusQuery(3, 4, 9, "distance", null),
265         "fl","id,score",
266         "sort","score asc")//want ascending due to increasing distance
267         , 1e-3
268         , "/response/docs/[0]/id=='100'"
269         , "/response/docs/[0]/score==2.827493"
270         , "/response/docs/[1]/id=='101'"
271         , "/response/docs/[1]/score==5.089807"
272     );
273     //score by recipDistance
274     assertJQ(req(
275         "q", radiusQuery(3, 4, 9, "recipDistance", null),
276         "fl","id,score",
277         "sort","score desc")//want descending
278         , 1e-3
279         , "/response/docs/[0]/id=='100'"
280         , "/response/docs/[0]/score==0.3099695"
281         , "/response/docs/[1]/id=='101'"
282         , "/response/docs/[1]/score==0.19970943"
283     );
284 
285     //score by distance and don't filter
286     assertJQ(req(
287         //circle radius is small and shouldn't match either, but we disable filtering
288         "q", radiusQuery(3, 4, 0.000001, "distance", "false"),
289         "fl","id,score",
290         "sort","score asc")//want ascending due to increasing distance
291         , 1e-3
292         , "/response/docs/[0]/id=='100'"
293         , "/response/docs/[0]/score==2.827493"
294         , "/response/docs/[1]/id=='101'"
295         , "/response/docs/[1]/score==5.089807"
296     );
297 
298     //query again with the query point closer to #101, and check the new ordering
299     assertJQ(req(
300         "q", radiusQuery(4, 0, 9, "distance", null),
301         "fl","id,score",
302         "sort","score asc")//want ascending due to increasing distance
303         , 1e-4
304         , "/response/docs/[0]/id=='101'"
305         , "/response/docs/[1]/id=='100'"
306     );
307 
308     //use sort=query(...)
309     assertJQ(req(
310         "q","-id:999",//exclude that doc
311         "fl","id,score",
312         "sort","query($sortQuery) asc", //want ascending due to increasing distance
313         "sortQuery", radiusQuery(3, 4, 9, "distance", null))
314         , 1e-4
315         , "/response/docs/[0]/id=='100'"
316         , "/response/docs/[1]/id=='101'"  );
317 
318     //check reversed direction with query point closer to #101
319     assertJQ(req(
320         "q","-id:999",//exclude that doc
321         "fl","id,score",
322         "sort","query($sortQuery) asc", //want ascending due to increasing distance
323         "sortQuery", radiusQuery(4, 0, 9, "distance", null))
324         , 1e-4
325         , "/response/docs/[0]/id=='101'"
326         , "/response/docs/[1]/id=='100'"  );
327   }

329   private String radiusQuery(double lat, double lon, double dDEG, String score, String filter) {
330     //Choose between the Solr/Geofilt syntax, and the Lucene spatial module syntax
331     if (fieldName.equals("bbox") || random().nextBoolean()) {
332       //we cheat for bbox strategy which doesn't do radius, only rect.
333       final String qparser = fieldName.equals("bbox") ? "bbox" : "geofilt";
334       return "{!" + qparser + " " +
335           "sfield=" + fieldName + " "
336           + (score != null ? "score="+score : "") + " "
337           + (filter != null ? "filter="+filter : "") + " "
338           + "pt=" + lat + "," + lon + " d=" + (dDEG /* DistanceUtils.DEG_TO_KM*/) + "}";
339     } else {
340       return "{! "
341           + (score != null ? "score="+score : "") + " "
342           + (filter != null ? "filter="+filter : "") + " "
343           + "}" + fieldName + ":\"Intersects(BUFFER(POINT(" + lon + " " + lat + ")," + dDEG + "))\"";
344     }
345   }

375   @Test
376   public void testSpatialConfig() throws Exception {
377     try (SolrCore core = h.getCoreInc())  {
378       IndexSchema schema = core.getLatestSchema();
379 
380       // BBox Config
381       // Make sure the subfields are not stored
382       SchemaField sub = schema.getField("bbox"+BBoxStrategy.SUFFIX_MINX);
383       assertFalse(sub.stored());
384 
385       // Make sure solr field type is also not stored
386       BBoxField bbox = (BBoxField)schema.getField("bbox").getType();
387       BBoxStrategy strategy = bbox.getStrategy("bbox");
388       assertFalse(strategy.getFieldType().stored());
389     }
