44    public void testSolrQueryMethods() {
45      SolrQuery q = new SolrQuery("dog");
46      boolean b = false;
47      
48      q.setFacetLimit(10);
49      q.addFacetField("price");
50      q.addFacetField("state");
51      Assert.assertEquals(q.getFacetFields().length, 2);
52      q.addFacetQuery("instock:true");
53      q.addFacetQuery("instock:false");
54      q.addFacetQuery("a:b");
55      Assert.assertEquals(q.getFacetQuery().length, 3);
56      
57      b = q.removeFacetField("price");
58      Assert.assertEquals(b, true);
59      b = q.removeFacetField("price2");
60      Assert.assertEquals(b, false);
61      b = q.removeFacetField("state");
62      Assert.assertEquals(b, true);
63      Assert.assertEquals(null, q.getFacetFields());
64      
65      b = q.removeFacetQuery("instock:true");
66      Assert.assertEquals(b, true);
67      b = q.removeFacetQuery("instock:false");
68      b = q.removeFacetQuery("a:c");
69      Assert.assertEquals(b, false);
70      b = q.removeFacetQuery("a:b");
71      Assert.assertEquals(null, q.getFacetQuery());   
72      
73      q.addSort("price", SolrQuery.ORDER.asc);
74      q.addSort("date", SolrQuery.ORDER.desc);
75      q.addSort("qty", SolrQuery.ORDER.desc);
76      q.removeSort(new SortClause("date", SolrQuery.ORDER.desc));
77      Assert.assertEquals(2, q.getSorts().size());
78      q.removeSort(new SortClause("price", SolrQuery.ORDER.asc));
79      q.removeSort(new SortClause("qty", SolrQuery.ORDER.desc));
80      Assert.assertEquals(0, q.getSorts().size());
81      
82      q.addHighlightField("hl1");
83      q.addHighlightField("hl2");
84      q.setHighlightSnippets(2);
85      Assert.assertEquals(2, q.getHighlightFields().length);
86      Assert.assertEquals(100, q.getHighlightFragsize());
87      Assert.assertEquals(q.getHighlightSnippets(), 2);
88      q.removeHighlightField("hl1");
89      q.removeHighlightField("hl3");
90      Assert.assertEquals(1, q.getHighlightFields().length);
91      q.removeHighlightField("hl2");
92      Assert.assertEquals(null, q.getHighlightFields());
93      
94      // check to see that the removes are properly clearing the cgi params
95      Assert.assertEquals(q.toString(), "q=dog");
96  
97      //Add time allowed param
98      q.setTimeAllowed(1000);
99      Assert.assertEquals((Integer)1000, q.getTimeAllowed() );
100     //Adding a null should remove it
101     q.setTimeAllowed(null);
102     Assert.assertEquals(null, q.getTimeAllowed() ); 
103     
104     // System.out.println(q);
105   }

161   public void testSortClause() {
162     new SolrQuery.SortClause("rating", SolrQuery.ORDER.desc);
163     new SolrQuery.SortClause("rating", SolrQuery.ORDER.valueOf("desc"));
164     new SolrQuery.SortClause("rating", SolrQuery.ORDER.valueOf("desc"));
165     SolrQuery.SortClause.create("rating", SolrQuery.ORDER.desc);
166     SolrQuery.SortClause.create("rating", SolrQuery.ORDER.desc);
167     SolrQuery.SortClause.create("rating", SolrQuery.ORDER.desc);
168 
169     SolrQuery.SortClause sc1a = SolrQuery.SortClause.asc("sc1");
170     SolrQuery.SortClause sc1b = SolrQuery.SortClause.asc("sc1");
171     Assert.assertEquals(sc1a, sc1b);
172     Assert.assertEquals(sc1a.hashCode(), sc1b.hashCode());
173 
174     SolrQuery.SortClause sc2a = SolrQuery.SortClause.asc("sc2");
175     SolrQuery.SortClause sc2b = SolrQuery.SortClause.desc("sc2");
176     Assert.assertFalse(sc2a.equals(sc2b));
177 
178     SolrQuery.SortClause sc3a = SolrQuery.SortClause.asc("sc2");
179     SolrQuery.SortClause sc3b = SolrQuery.SortClause.asc("not sc2");
180     Assert.assertFalse(sc3a.equals(sc3b));
181   }

186   public void testSort() throws IOException {
187 
188     SolrQuery q = new SolrQuery("dog");
189 
190     // Simple adds
191     q.addSort("price", SolrQuery.ORDER.asc);
192     q.addSort("date", SolrQuery.ORDER.desc);
193     q.addSort("qty", SolrQuery.ORDER.desc);
194     Assert.assertEquals(3, q.getSorts().size());
195     Assert.assertEquals("price asc,date desc,qty desc", q.get(CommonParams.SORT));
196 
197     // Remove one (middle)
198     q.removeSort("date");
199     Assert.assertEquals(2, q.getSorts().size());
200     Assert.assertEquals("price asc,qty desc", q.get(CommonParams.SORT));
201 
202     // Remove remaining (last, first)
203     q.removeSort("price");
204     q.removeSort("qty");
205     Assert.assertTrue(q.getSorts().isEmpty());
206     Assert.assertNull(q.get(CommonParams.SORT));
207 
208     // Clear sort
209     q.addSort("price", SolrQuery.ORDER.asc);
210     q.clearSorts();
211     Assert.assertTrue(q.getSorts().isEmpty());
212     Assert.assertNull(q.get(CommonParams.SORT));
213 
214     // Add vs update
215     q.clearSorts();
216     q.addSort("1", SolrQuery.ORDER.asc);
217     q.addSort("2", SolrQuery.ORDER.asc);
218     q.addSort("3", SolrQuery.ORDER.asc);
219     q.addOrUpdateSort("2", SolrQuery.ORDER.desc);
220     q.addOrUpdateSort("4", SolrQuery.ORDER.desc);
221     Assert.assertEquals("1 asc,2 desc,3 asc,4 desc", q.get(CommonParams.SORT));
222 
223     // Using SortClause
224     q.clearSorts();
225     q.addSort(new SortClause("1", SolrQuery.ORDER.asc));
226     q.addSort(new SortClause("2", SolrQuery.ORDER.asc));
227     q.addSort(new SortClause("3", SolrQuery.ORDER.asc));
228     q.addOrUpdateSort(SortClause.desc("2"));
229     q.addOrUpdateSort(SortClause.asc("4"));
230     Assert.assertEquals("1 asc,2 desc,3 asc,4 asc", q.get(CommonParams.SORT));
231     q.setSort(SortClause.asc("A"));
232     q.addSort(SortClause.asc("B"));
233     q.addSort(SortClause.asc("C"));
234     q.addSort(SortClause.asc("D"));
235     Assert.assertEquals("A asc,B asc,C asc,D asc", q.get(CommonParams.SORT));
236 
237     // removeSort should ignore the ORDER
238     q.setSort(SortClause.asc("A"));
239     q.addSort(SortClause.asc("B"));
240     q.addSort(SortClause.asc("C"));
241     q.addSort(SortClause.asc("D"));
242     q.removeSort("A");
243     q.removeSort(SortClause.asc("C"));
244     q.removeSort(SortClause.desc("B"));
245     Assert.assertEquals("D asc", q.get(CommonParams.SORT));
246 
247     // Verify that a query containing a SortClause is serializable
248     q.clearSorts();
249     q.addSort("1", SolrQuery.ORDER.asc);
250     ObjectOutputStream out = new ObjectOutputStream(new ByteArrayOutputStream());
251     out.writeObject(q);
252     out.close();
253   }

295   public void testFacetDateRange() {
296     SolrQuery q = new SolrQuery("dog");
297     Calendar calendar = Calendar.getInstance(TimeZone.getTimeZone("UTC"), Locale.UK);
298     calendar.set(2010, 1, 1);
299     Date start = calendar.getTime();
300     calendar.set(2011, 1, 1);
301     Date end = calendar.getTime();
302     q.addDateRangeFacet("field", start, end, "+1MONTH");
303     assertEquals("true", q.get(FacetParams.FACET));
304     assertEquals("field", q.get(FacetParams.FACET_RANGE));
305     assertEquals(DateUtil.getThreadLocalDateFormat().format(start), q.get("f.field." + FacetParams.FACET_RANGE_START));
306     assertEquals(DateUtil.getThreadLocalDateFormat().format(end), q.get("f.field." + FacetParams.FACET_RANGE_END));
307     assertEquals("+1MONTH", q.get("f.field." + FacetParams.FACET_RANGE_GAP));
308   }

406   public void testFacetInterval() {
407     SolrQuery solrQuery = new SolrQuery();
408     solrQuery.addIntervalFacets("field1", new String[]{});
409     assertTrue(solrQuery.getBool(FacetParams.FACET));
410     assertEquals("field1", solrQuery.get(FacetParams.FACET_INTERVAL));
411     
412     solrQuery.addIntervalFacets("field2", new String[]{"[1,10]"});
413     assertArrayEquals(new String[]{"field1", "field2"}, solrQuery.getParams(FacetParams.FACET_INTERVAL));
414     assertEquals("[1,10]", solrQuery.get("f.field2.facet.interval.set"));
415     
416     solrQuery.addIntervalFacets("field3", new String[]{"[1,10]", "(10,100]", "(100,1000]", "(1000,*]"});
417     assertArrayEquals(new String[]{"field1", "field2", "field3"}, solrQuery.getParams(FacetParams.FACET_INTERVAL));
418     assertArrayEquals(new String[]{"[1,10]", "(10,100]", "(100,1000]", "(1000,*]"}, solrQuery.getParams("f.field3.facet.interval.set"));
419     
420     //Validate adding more intervals for an existing field
421     solrQuery.addIntervalFacets("field2", new String[]{"[10,100]"});
422     assertArrayEquals(new String[]{"[1,10]", "[10,100]"}, solrQuery.getParams("f.field2.facet.interval.set"));
423     
424     assertNull(solrQuery.removeIntervalFacets("field1"));
425     assertArrayEquals(new String[]{"field2", "field3", "field2"}, solrQuery.getParams(FacetParams.FACET_INTERVAL));
426     assertNull(solrQuery.getParams("f.field1.facet.interval.set"));
427     
428     assertArrayEquals(new String[]{"[1,10]", "[10,100]"}, solrQuery.removeIntervalFacets("field2"));
429     assertArrayEquals(new String[]{"field3"}, solrQuery.getParams(FacetParams.FACET_INTERVAL));
430     assertNull(solrQuery.getParams("f.field2.facet.interval.set"));
431     
432     assertArrayEquals(new String[]{"[1,10]", "(10,100]", "(100,1000]", "(1000,*]"}, solrQuery.removeIntervalFacets("field3"));
433     assertNull(solrQuery.getParams(FacetParams.FACET_INTERVAL));
434     assertNull(solrQuery.getParams("f.field3.facet.interval.set"));
