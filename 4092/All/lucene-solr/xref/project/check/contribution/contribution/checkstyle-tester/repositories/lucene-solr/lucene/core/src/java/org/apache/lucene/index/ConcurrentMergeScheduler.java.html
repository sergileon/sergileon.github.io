32  /** A {@link MergeScheduler} that runs each merge using a
33   *  separate thread.
34   *
35   *  <p>Specify the max number of threads that may run at
36   *  once, and the maximum number of simultaneous merges
37   *  with {@link #setMaxMergesAndThreads}.</p>
38   *
39   *  <p>If the number of merges exceeds the max number of threads 
40   *  then the largest merges are paused until one of the smaller
41   *  merges completes.</p>
42   *
43   *  <p>If more than {@link #getMaxMergeCount} merges are
44   *  requested then this class will forcefully throttle the
45   *  incoming threads by pausing until one more more merges
46   *  complete.</p>
47   *
48   *  <p>This class attempts to detect whether the index is
49   *  on rotational storage (traditional hard drive) or not
50   *  (e.g. solid-state disk) and changes the default max merge
51   *  and thread count accordingly.  This detection is currently
52   *  Linux-only, and relies on the OS to put the right value
53   *  into /sys/block/&lt;dev&gt;/block/rotational.  For all
54   *  other operating systems it currently assumes a rotational
55   *  disk for backwards compatibility.  To enable default
56   *  settings for spinning or solid state disks for such
57   *  operating systems, use {@link #setDefaultMaxMergesAndThreads(boolean)}.
58   */ 
59  
60  public class ConcurrentMergeScheduler extends MergeScheduler {
61  
62    /** Dynamic default for {@code maxThreadCount} and {@code maxMergeCount},
63     *  used to detect whether the index is backed by an SSD or rotational disk and
64     *  set {@code maxThreadCount} accordingly.  If it's an SSD,
65     *  {@code maxThreadCount} is set to {@code max(1, min(4, cpuCoreCount/2))},
66     *  otherwise 1.  Note that detection only currently works on
67     *  Linux; other platforms will assume the index is not on an SSD. */
68    public static final int AUTO_DETECT_MERGES_AND_THREADS = -1;

83    // Max number of merge threads allowed to be running at
84    // once.  When there are more merges then this, we
85    // forcefully pause the larger ones, letting the smaller
86    // ones run, up until maxMergeCount merges at which point
87    // we forcefully pause incoming threads (that presumably
88    // are the ones causing so much merging).
89    private int maxThreadCount = AUTO_DETECT_MERGES_AND_THREADS;
90  
91    // Max number of merges we accept before forcefully
92    // throttling the incoming threads
93    private int maxMergeCount = AUTO_DETECT_MERGES_AND_THREADS;

95    /** How many {@link MergeThread}s have kicked off (this is use
96     *  to name them). */
97    protected int mergeThreadCount;

99    /** Floor for IO write rate limit (we will never go any lower than this) */
100   private static final double MIN_MERGE_MB_PER_SEC = 5.0;
101 
102   /** Ceiling for IO write rate limit (we will never go any higher than this) */
103   private static final double MAX_MERGE_MB_PER_SEC = 10240.0;
104 
105   /** Initial value for IO write rate limit when doAutoIOThrottle is true */
106   private static final double START_MB_PER_SEC = 20.0;
107 
108   /** Merges below this size are not counted in the maxThreadCount, i.e. they can freely run in their own thread (up until maxMergeCount). */
109   private static final double MIN_BIG_MERGE_MB = 50.0;
110 
111   /** Current IO writes throttle rate */
112   protected double targetMBPerSec = START_MB_PER_SEC;

114   /** true if we should rate-limit writes for each merge */
115   private boolean doAutoIOThrottle = true;

117   private double forceMergeMBPerSec = Double.POSITIVE_INFINITY;

124   /**
125    * Expert: directly set the maximum number of merge threads and
126    * simultaneous merges allowed.
127    * 
128    * @param maxMergeCount the max # simultaneous merges that are allowed.
129    *       If a merge is necessary yet we already have this many
130    *       threads running, the incoming thread (that is calling
131    *       add/updateDocument) will block until a merge thread
132    *       has completed.  Note that we will only run the
133    *       smallest <code>maxThreadCount</code> merges at a time.
134    * @param maxThreadCount the max # simultaneous merge threads that should
135    *       be running at once.  This must be &lt;= <code>maxMergeCount</code>
136    */
137   public synchronized void setMaxMergesAndThreads(int maxMergeCount, int maxThreadCount) {
138     if (maxMergeCount == AUTO_DETECT_MERGES_AND_THREADS && maxThreadCount == AUTO_DETECT_MERGES_AND_THREADS) {
139       // OK
140       this.maxMergeCount = AUTO_DETECT_MERGES_AND_THREADS;
141       this.maxThreadCount = AUTO_DETECT_MERGES_AND_THREADS;
142     } else if (maxMergeCount == AUTO_DETECT_MERGES_AND_THREADS) {
143       throw new IllegalArgumentException("both maxMergeCount and maxThreadCount must be AUTO_DETECT_MERGES_AND_THREADS");
144     } else if (maxThreadCount == AUTO_DETECT_MERGES_AND_THREADS) {
145       throw new IllegalArgumentException("both maxMergeCount and maxThreadCount must be AUTO_DETECT_MERGES_AND_THREADS");
146     } else {
147       if (maxThreadCount < 1) {
148         throw new IllegalArgumentException("maxThreadCount should be at least 1");
149       }
150       if (maxMergeCount < 1) {
151         throw new IllegalArgumentException("maxMergeCount should be at least 1");
152       }
153       if (maxThreadCount > maxMergeCount) {
154         throw new IllegalArgumentException("maxThreadCount should be <= maxMergeCount (= " + maxMergeCount + ")");
155       }
156       this.maxThreadCount = maxThreadCount;
157       this.maxMergeCount = maxMergeCount;
158     }
159   }

161   /** Sets max merges and threads to proper defaults for rotational
162    *  or non-rotational storage.
163    *
164    * @param spins true to set defaults best for traditional rotatational storage (spinning disks), 
165    *        else false (e.g. for solid-state disks)
166    */
167   public synchronized void setDefaultMaxMergesAndThreads(boolean spins) {
168     if (spins) {
169       maxThreadCount = 1;
170       maxMergeCount = 6;
171     } else {
172       int coreCount = Runtime.getRuntime().availableProcessors();
173 
174       // Let tests override this to help reproducing a failure on a machine that has a different
175       // core count than the one where the test originally failed:
176       try {
177         String value = System.getProperty(DEFAULT_CPU_CORE_COUNT_PROPERTY);
178         if (value != null) {
179           coreCount = Integer.parseInt(value);
180         }
181       } catch (Throwable ignored) {
182       }
183 
184       maxThreadCount = Math.max(1, Math.min(4, coreCount/2));
185       maxMergeCount = maxThreadCount+5;
186     }
187   }

244   /** Removes the calling thread from the active merge threads. */
245   synchronized void removeMergeThread() {
246     Thread currentThread = Thread.currentThread();
247     // Paranoia: don't trust Thread.equals:
248     for(int i=0;i<mergeThreads.size();i++) {
249       if (mergeThreads.get(i) == currentThread) {
250         mergeThreads.remove(i);
251         return;
252       }
253     }
254       
255     assert false: "merge thread " + currentThread + " was not found";
256   }

258   /**
259    * Called whenever the running merges have changed, to set merge IO limits.
260    * This method sorts the merge threads by their merge size in
261    * descending order and then pauses/unpauses threads from first to last --
262    * that way, smaller merges are guaranteed to run before larger ones.
263    */
264 
265   protected synchronized void updateMergeThreads() {
266 
267     // Only look at threads that are alive & not in the
268     // process of stopping (ie have an active merge):
269     final List<MergeThread> activeMerges = new ArrayList<>();
270 
271     int threadIdx = 0;
272     while (threadIdx < mergeThreads.size()) {
273       final MergeThread mergeThread = mergeThreads.get(threadIdx);
274       if (!mergeThread.isAlive()) {
275         // Prune any dead threads
276         mergeThreads.remove(threadIdx);
277         continue;
278       }
279       activeMerges.add(mergeThread);
280       threadIdx++;
281     }
282 
283     // Sort the merge threads, largest first:
284     CollectionUtil.timSort(activeMerges);
285 
286     final int activeMergeCount = activeMerges.size();
287 
288     int bigMergeCount = 0;
289 
290     for (threadIdx=activeMergeCount-1;threadIdx>=0;threadIdx--) {
291       MergeThread mergeThread = activeMerges.get(threadIdx);
292       if (mergeThread.merge.estimatedMergeBytes > MIN_BIG_MERGE_MB*1024*1024) {
293         bigMergeCount = 1+threadIdx;
294         break;
295       }
296     }
297 
298     long now = System.nanoTime();
299 
300     StringBuilder message;
301     if (verbose()) {
302       message = new StringBuilder();
303       message.append(String.format(Locale.ROOT, "updateMergeThreads ioThrottle=%s targetMBPerSec=%.1f MB/sec", doAutoIOThrottle, targetMBPerSec));
304     } else {
305       message = null;
306     }
307 
308     for (threadIdx=0;threadIdx<activeMergeCount;threadIdx++) {
309       MergeThread mergeThread = activeMerges.get(threadIdx);
310 
311       OneMerge merge = mergeThread.merge;
312 
313       // pause the thread if maxThreadCount is smaller than the number of merge threads.
314       final boolean doPause = threadIdx < bigMergeCount - maxThreadCount;
315 
316       double newMBPerSec;
317       if (doPause) {
318         newMBPerSec = 0.0;
319       } else if (merge.maxNumSegments != -1) {
320         newMBPerSec = forceMergeMBPerSec;
321       } else if (doAutoIOThrottle == false) {
322         newMBPerSec = Double.POSITIVE_INFINITY;
323       } else if (merge.estimatedMergeBytes < MIN_BIG_MERGE_MB*1024*1024) {
324         // Don't rate limit small merges:
325         newMBPerSec = Double.POSITIVE_INFINITY;
326       } else {
327         newMBPerSec = targetMBPerSec;
328       }
329 
330       double curMBPerSec = merge.rateLimiter.getMBPerSec();
331       
332       if (verbose()) {
333         long mergeStartNS = merge.mergeStartNS;
334         if (mergeStartNS == -1) {
335           // IndexWriter didn't start the merge yet:
336           mergeStartNS = now;
337         }
338         message.append('\n');
339         message.append(String.format(Locale.ROOT, "merge thread %s estSize=%.1f MB (written=%.1f MB) runTime=%.1fs (stopped=%.1fs, paused=%.1fs) rate=%s\n",
340                                      mergeThread.getName(),
341                                      bytesToMB(merge.estimatedMergeBytes),
342                                      bytesToMB(merge.rateLimiter.totalBytesWritten),
343                                      nsToSec(now - mergeStartNS),
344                                      nsToSec(merge.rateLimiter.getTotalStoppedNS()),
345                                      nsToSec(merge.rateLimiter.getTotalPausedNS()),
346                                      rateToString(merge.rateLimiter.getMBPerSec())));
347 
348         if (newMBPerSec != curMBPerSec) {
349           if (newMBPerSec == 0.0) {
350             message.append("  now stop");
351           } else if (curMBPerSec == 0.0) {
352             if (newMBPerSec == Double.POSITIVE_INFINITY) {
353               message.append("  now resume");
354             } else {
355               message.append(String.format(Locale.ROOT, "  now resume to %.1f MB/sec", newMBPerSec));
356             }
357           } else {
358             message.append(String.format(Locale.ROOT, "  now change from %.1f MB/sec to %.1f MB/sec", curMBPerSec, newMBPerSec));
359           }
360         } else if (curMBPerSec == 0.0) {
361           message.append("  leave stopped");
362         } else {
363           message.append(String.format(Locale.ROOT, "  leave running at %.1f MB/sec", curMBPerSec));
364         }
365       }
366 
367       merge.rateLimiter.setMBPerSec(newMBPerSec);
368     }
369     if (verbose()) {
370       message(message.toString());
371     }
372   }

374   private synchronized void initDynamicDefaults(IndexWriter writer) throws IOException {
375     if (maxThreadCount == AUTO_DETECT_MERGES_AND_THREADS) {
376       boolean spins = IOUtils.spins(writer.getDirectory());
377 
378       // Let tests override this to help reproducing a failure on a machine that has a different
379       // core count than the one where the test originally failed:
380       try {
381         String value = System.getProperty(DEFAULT_SPINS_PROPERTY);
382         if (value != null) {
383           spins = Boolean.parseBoolean(value);
384         }
385       } catch (Throwable ignored) {
386       }
387       setDefaultMaxMergesAndThreads(spins);
388       if (verbose()) {
389         message("initDynamicDefaults spins=" + spins + " maxThreadCount=" + maxThreadCount + " maxMergeCount=" + maxMergeCount);
390       }
391     }
392   }

394   private static String rateToString(double mbPerSec) {
395     if (mbPerSec == 0.0) {
396       return "stopped";
397     } else if (mbPerSec == Double.POSITIVE_INFINITY) {
398       return "unlimited";
399     } else {
400       return String.format(Locale.ROOT, "%.1f MB/sec", mbPerSec);
401     }
402   }

409   /** Wait for any running merge threads to finish. This call is not interruptible as used by {@link #close()}. */
410   public void sync() {
411     boolean interrupted = false;
412     try {
413       while (true) {
414         MergeThread toSync = null;
415         synchronized (this) {
416           for (MergeThread t : mergeThreads) {
417             // In case a merge thread is calling us, don't try to sync on
418             // itself, since that will never finish!
419             if (t.isAlive() && t != Thread.currentThread()) {
420               toSync = t;
421               break;
422             }
423           }
424         }
425         if (toSync != null) {
426           try {
427             toSync.join();
428           } catch (InterruptedException ie) {
429             // ignore this Exception, we will retry until all threads are dead
430             interrupted = true;
431           }
432         } else {
433           break;
434         }
435       }
436     } finally {
437       // finally, restore interrupt status:
438       if (interrupted) Thread.currentThread().interrupt();
439     }
440   }

442   /**
443    * Returns the number of merge threads that are alive, ignoring the calling thread
444    * if it is a merge thread.  Note that this number is &le; {@link #mergeThreads} size.
445    *
446    * @lucene.internal
447    */
448   public synchronized int mergeThreadCount() {
449     Thread currentThread = Thread.currentThread();
450     int count = 0;
451     for (MergeThread mergeThread : mergeThreads) {
452       if (currentThread != mergeThread && mergeThread.isAlive() && mergeThread.merge.rateLimiter.getAbort() == false) {
453         count++;
454       }
455     }
456     return count;
457   }

459   @Override
460   public synchronized void merge(IndexWriter writer, MergeTrigger trigger, boolean newMergesFound) throws IOException {
461 
462     assert !Thread.holdsLock(writer);
463 
464     initDynamicDefaults(writer);
465 
466     if (trigger == MergeTrigger.CLOSING) {
467       // Disable throttling on close:
468       targetMBPerSec = MAX_MERGE_MB_PER_SEC;
469       updateMergeThreads();
470     }
471 
472     // First, quickly run through the newly proposed merges
473     // and add any orthogonal merges (ie a merge not
474     // involving segments already pending to be merged) to
475     // the queue.  If we are way behind on merging, many of
476     // these newly proposed merges will likely already be
477     // registered.
478 
479     if (verbose()) {
480       message("now merge");
481       message("  index: " + writer.segString());
482     }
483     
484     // Iterate, pulling from the IndexWriter's queue of
485     // pending merges, until it's empty:
486     while (true) {
487 
488       if (maybeStall(writer) == false) {
489         break;
490       }
491 
492       OneMerge merge = writer.getNextMerge();
493       if (merge == null) {
494         if (verbose()) {
495           message("  no more merges pending; now return");
496         }
497         return;
498       }
499 
500       updateIOThrottle(merge);
501 
502       boolean success = false;
503       try {
504         if (verbose()) {
505           message("  consider merge " + writer.segString(merge.segments));
506         }
507 
508         // OK to spawn a new merge thread to handle this
509         // merge:
510         final MergeThread merger = getMergeThread(writer, merge);
511         mergeThreads.add(merger);
512 
513         if (verbose()) {
514           message("    launch new thread [" + merger.getName() + "]");
515         }
516 
517         merger.start();
518         updateMergeThreads();
519 
520         success = true;
521       } finally {
522         if (!success) {
523           writer.mergeFinish(merge);
524         }
525       }
526     }
527   }

529   /** This is invoked by {@link #merge} to possibly stall the incoming
530    *  thread when there are too many merges running or pending.  The 
531    *  default behavior is to force this thread, which is producing too
532    *  many segments for merging to keep up, to wait until merges catch
533    *  up. Applications that can take other less drastic measures, such
534    *  as limiting how many threads are allowed to index, can do nothing
535    *  here and throttle elsewhere.
536    *
537    *  If this method wants to stall but the calling thread is a merge
538    *  thread, it should return false to tell caller not to kick off
539    *  any new merges. */
540 
541   protected synchronized boolean maybeStall(IndexWriter writer) {
542     long startStallTime = 0;
543     while (writer.hasPendingMerges() && mergeThreadCount() >= maxMergeCount) {
544 
545       // This means merging has fallen too far behind: we
546       // have already created maxMergeCount threads, and
547       // now there's at least one more merge pending.
548       // Note that only maxThreadCount of
549       // those created merge threads will actually be
550       // running; the rest will be paused (see
551       // updateMergeThreads).  We stall this producer
552       // thread to prevent creation of new segments,
553       // until merging has caught up:
554 
555       if (mergeThreads.contains(Thread.currentThread())) {
556         // Never stall a merge thread since this blocks the thread from
557         // finishing and calling updateMergeThreads, and blocking it
558         // accomplishes nothing anyway (it's not really a segment producer):
559         return false;
560       }
561 
562       if (verbose() && startStallTime == 0) {
563         message("    too many merges; stalling...");
564       }
565       startStallTime = System.currentTimeMillis();
566       doStall();
567     }
568 
569     if (verbose() && startStallTime != 0) {
570       message("  stalled for " + (System.currentTimeMillis()-startStallTime) + " msec");
571     }
572 
573     return true;
574   }
575 
576   /** Called from {@link #maybeStall} to pause the calling thread for a bit. */
577   protected synchronized void doStall() {
578     try {
579       // Defensively wait for only .25 seconds in case we are missing a .notify/All somewhere:
580       wait(250);
581     } catch (InterruptedException ie) {
582       throw new ThreadInterruptedException(ie);
583     }
584   }

591   /** Create and return a new MergeThread */
592   protected synchronized MergeThread getMergeThread(IndexWriter writer, OneMerge merge) throws IOException {
593     final MergeThread thread = new MergeThread(writer, merge);
594     thread.setDaemon(true);
595     thread.setName("Lucene Merge Thread #" + mergeThreadCount++);
596     return thread;
597   }

599   /** Runs a merge thread to execute a single merge, then exits. */
600   protected class MergeThread extends Thread implements Comparable<MergeThread> {
601 
602     final IndexWriter writer;
603     final OneMerge merge;
604 
605     /** Sole constructor. */
606     public MergeThread(IndexWriter writer, OneMerge merge) {
607       this.writer = writer;
608       this.merge = merge;
609     }
610     
611     @Override
612     public int compareTo(MergeThread other) {
613       // Larger merges sort first:
614       return Long.compare(other.merge.estimatedMergeBytes, merge.estimatedMergeBytes);
615     }
616 
617     @Override
618     public void run() {
619 
620       try {
621 
622         if (verbose()) {
623           message("  merge thread: start");
624         }
625 
626         doMerge(writer, merge);
627 
628         if (verbose()) {
629           message("  merge thread: done");
630         }
631 
632         // Let CMS run new merges if necessary:
633         try {
634           merge(writer, MergeTrigger.MERGE_FINISHED, true);
635         } catch (AlreadyClosedException ace) {
636           // OK
637         } catch (IOException ioe) {
638           throw new RuntimeException(ioe);
639         }
640 
641       } catch (Throwable exc) {
642 
643         if (exc instanceof MergePolicy.MergeAbortedException) {
644           // OK to ignore
645         } else if (suppressExceptions == false) {
646           // suppressExceptions is normally only set during
647           // testing.
648           handleMergeException(writer.getDirectory(), exc);
649         }
650 
651       } finally {
652         synchronized(ConcurrentMergeScheduler.this) {
653           removeMergeThread();
654 
655           updateMergeThreads();
656 
657           // In case we had stalled indexing, we can now wake up
658           // and possibly unstall:
659           ConcurrentMergeScheduler.this.notifyAll();
660         }
661       }
662     }
663   }
664 
665   /** Called when an exception is hit in a background merge
666    *  thread */
667   protected void handleMergeException(Directory dir, Throwable exc) {
668     throw new MergePolicy.MergeException(exc, dir);
669   }

692   private boolean isBacklog(long now, OneMerge merge) {
693     double mergeMB = bytesToMB(merge.estimatedMergeBytes);
694     for (MergeThread mergeThread : mergeThreads) {
695       long mergeStartNS = mergeThread.merge.mergeStartNS;
696       if (mergeThread.isAlive() && mergeThread.merge != merge &&
697           mergeStartNS != -1 &&
698           mergeThread.merge.estimatedMergeBytes >= MIN_BIG_MERGE_MB*1024*1024 &&
699           nsToSec(now-mergeStartNS) > 3.0) {
700         double otherMergeMB = bytesToMB(mergeThread.merge.estimatedMergeBytes);
701         double ratio = otherMergeMB / mergeMB;
702         if (ratio > 0.3 && ratio < 3.0) {
703           return true;
704         }
705       }
706     }
707 
708     return false;
709   }

711   /** Tunes IO throttle when a new merge starts. */
712   private synchronized void updateIOThrottle(OneMerge newMerge) throws IOException {
713     if (doAutoIOThrottle == false) {
714       return;
715     }
716 
717     double mergeMB = bytesToMB(newMerge.estimatedMergeBytes);
718     if (mergeMB < MIN_BIG_MERGE_MB) {
719       // Only watch non-trivial merges for throttling; this is safe because the MP must eventually
720       // have to do larger merges:
721       return;
722     }
723 
724     long now = System.nanoTime();
725 
726     // Simplistic closed-loop feedback control: if we find any other similarly
727     // sized merges running, then we are falling behind, so we bump up the
728     // IO throttle, else we lower it:
729     boolean newBacklog = isBacklog(now, newMerge);
730 
731     boolean curBacklog = false;
732 
733     if (newBacklog == false) {
734       if (mergeThreads.size() > maxThreadCount) {
735         // If there are already more than the maximum merge threads allowed, count that as backlog:
736         curBacklog = true;
737       } else {
738         // Now see if any still-running merges are backlog'd:
739         for (MergeThread mergeThread : mergeThreads) {
740           if (isBacklog(now, mergeThread.merge)) {
741             curBacklog = true;
742             break;
743           }
744         }
745       }
746     }
747 
748     double curMBPerSec = targetMBPerSec;
749 
750     if (newBacklog) {
751       // This new merge adds to the backlog: increase IO throttle by 20%
752       targetMBPerSec *= 1.20;
753       if (targetMBPerSec > MAX_MERGE_MB_PER_SEC) {
754         targetMBPerSec = MAX_MERGE_MB_PER_SEC;
755       }
756       if (verbose()) {
757         if (curMBPerSec == targetMBPerSec) {
758           message(String.format(Locale.ROOT, "io throttle: new merge backlog; leave IO rate at ceiling %.1f MB/sec", targetMBPerSec));
759         } else {
760           message(String.format(Locale.ROOT, "io throttle: new merge backlog; increase IO rate to %.1f MB/sec", targetMBPerSec));
761         }
762       }
763     } else if (curBacklog) {
764       // We still have an existing backlog; leave the rate as is:
765       if (verbose()) {
766         message(String.format(Locale.ROOT, "io throttle: current merge backlog; leave IO rate at %.1f MB/sec",
767                               targetMBPerSec));
768       }
769     } else {
770       // We are not falling behind: decrease IO throttle by 10%
771       targetMBPerSec /= 1.10;
772       if (targetMBPerSec < MIN_MERGE_MB_PER_SEC) {
773         targetMBPerSec = MIN_MERGE_MB_PER_SEC;
774       }
775       if (verbose()) {
776         if (curMBPerSec == targetMBPerSec) {
777           message(String.format(Locale.ROOT, "io throttle: no merge backlog; leave IO rate at floor %.1f MB/sec", targetMBPerSec));
778         } else {
779           message(String.format(Locale.ROOT, "io throttle: no merge backlog; decrease IO rate to %.1f MB/sec", targetMBPerSec));
780         }
781       }
782     }
783 
784     double rate;
785 
786     if (newMerge.maxNumSegments != -1) {
787       rate = forceMergeMBPerSec;
788     } else {
789       rate = targetMBPerSec;
790     }
791     newMerge.rateLimiter.setMBPerSec(rate);
792     targetMBPerSecChanged();
793   }

797   }
