20  import java.io.IOException;
21  
22  import org.apache.lucene.analysis.BaseTokenStreamTestCase;
23  import org.apache.lucene.analysis.Analyzer;
24  import org.apache.lucene.analysis.util.CharArraySet;
25  import org.apache.lucene.util.Version;
26  
27  /**
28   * Test the CzechAnalyzer
29   * 
30   * Before Lucene 3.1, CzechAnalyzer was a StandardAnalyzer with a custom 
31   * stopword list. As of 3.1 it also includes a stemmer.
32   *
33   */
34  public class TestCzechAnalyzer extends BaseTokenStreamTestCase {
35    
36    /** This test fails with NPE when the 
37     * stopwords file is missing in classpath */
38    public void testResourcesAvailable() {
39      new CzechAnalyzer().close();
40    }
41    
42    public void testStopWord() throws Exception {
43      Analyzer analyzer = new CzechAnalyzer();
44      assertAnalyzesTo(analyzer, "Pokud mluvime o volnem", 
45          new String[] { "mluvim", "voln" });
46      analyzer.close();
47    }
48    
49    public void testReusableTokenStream() throws Exception {
50      Analyzer analyzer = new CzechAnalyzer();
51      assertAnalyzesTo(analyzer, "Pokud mluvime o volnem", new String[] { "mluvim", "voln" });
52      assertAnalyzesTo(analyzer, "Česká Republika", new String[] { "česk", "republik" });
53      analyzer.close();
54    }
55  
56    public void testWithStemExclusionSet() throws IOException{
57      CharArraySet set = new CharArraySet(1, true);
58      set.add("hole");
59      CzechAnalyzer cz = new CzechAnalyzer(CharArraySet.EMPTY_SET, set);
60      assertAnalyzesTo(cz, "hole desek", new String[] {"hole", "desk"});
61      cz.close();
62    }
63    
64    /** blast some random strings through the analyzer */
65    public void testRandomStrings() throws Exception {
66      Analyzer analyzer = new CzechAnalyzer();
67      checkRandomData(random(), analyzer, 1000*RANDOM_MULTIPLIER);
68      analyzer.close();
69    }
70  
71    public void testBackcompat40() throws IOException {
72      CzechAnalyzer a = new CzechAnalyzer();
73      a.setVersion(Version.LUCENE_4_6_1);
74      // this is just a test to see the correct unicode version is being used, not actually testing hebrew
75      assertAnalyzesTo(a, "א\"א", new String[] {"א", "א"});
76    }
77  }
