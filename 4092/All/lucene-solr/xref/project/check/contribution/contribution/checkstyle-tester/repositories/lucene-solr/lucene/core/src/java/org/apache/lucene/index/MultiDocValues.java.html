39  /**
40   * A wrapper for CompositeIndexReader providing access to DocValues.
41   * 
42   * <p><b>NOTE</b>: for multi readers, you'll get better
43   * performance by gathering the sub readers using
44   * {@link IndexReader#getContext()} to get the
45   * atomic leaves and then operate per-LeafReader,
46   * instead of using this class.
47   * 
48   * <p><b>NOTE</b>: This is very costly.
49   *
50   * @lucene.experimental
51   * @lucene.internal
52   */

58    /** Returns a NumericDocValues for a reader's norms (potentially merging on-the-fly).
59     * <p>
60     * This is a slow way to access normalization values. Instead, access them per-segment
61     * with {@link LeafReader#getNormValues(String)}
62     * </p> 
63     */
64    public static NumericDocValues getNormValues(final IndexReader r, final String field) throws IOException {
65      final List<LeafReaderContext> leaves = r.leaves();
66      final int size = leaves.size();
67      if (size == 0) {
68        return null;
69      } else if (size == 1) {
70        return leaves.get(0).reader().getNormValues(field);
71      }
72      FieldInfo fi = MultiFields.getMergedFieldInfos(r).fieldInfo(field);
73      if (fi == null || fi.hasNorms() == false) {
74        return null;
75      }
76  
77      boolean anyReal = false;
78      final NumericDocValues[] values = new NumericDocValues[size];
79      final int[] starts = new int[size+1];
80      for (int i = 0; i < size; i++) {
81        LeafReaderContext context = leaves.get(i);
82        NumericDocValues v = context.reader().getNormValues(field);
83        if (v == null) {
84          v = DocValues.emptyNumeric();
85        } else {
86          anyReal = true;
87        }
88        values[i] = v;
89        starts[i] = context.docBase;
90      }
91      starts[size] = r.maxDoc();
92      
93      assert anyReal;
94  
95      return new NumericDocValues() {
96        @Override
97        public long get(int docID) {
98          int subIndex = ReaderUtil.subIndex(docID, starts);
99          return values[subIndex].get(docID - starts[subIndex]);
100       }
101     };
102   }

148   /** Returns a Bits for a reader's docsWithField (potentially merging on-the-fly) 
149    * <p>
150    * This is a slow way to access this bitset. Instead, access them per-segment
151    * with {@link LeafReader#getDocsWithField(String)}
152    * </p> 
153    * */
154   public static Bits getDocsWithField(final IndexReader r, final String field) throws IOException {
155     final List<LeafReaderContext> leaves = r.leaves();
156     final int size = leaves.size();
157     if (size == 0) {
158       return null;
159     } else if (size == 1) {
160       return leaves.get(0).reader().getDocsWithField(field);
161     }
162 
163     boolean anyReal = false;
164     boolean anyMissing = false;
165     final Bits[] values = new Bits[size];
166     final int[] starts = new int[size+1];
167     for (int i = 0; i < size; i++) {
168       LeafReaderContext context = leaves.get(i);
169       Bits v = context.reader().getDocsWithField(field);
170       if (v == null) {
171         v = new Bits.MatchNoBits(context.reader().maxDoc());
172         anyMissing = true;
173       } else {
174         anyReal = true;
175         if (v instanceof Bits.MatchAllBits == false) {
176           anyMissing = true;
177         }
178       }
179       values[i] = v;
180       starts[i] = context.docBase;
181     }
182     starts[size] = r.maxDoc();
183 
184     if (!anyReal) {
185       return null;
186     } else if (!anyMissing) {
187       return new Bits.MatchAllBits(r.maxDoc());
188     } else {
189       return new MultiBits(values, starts, false);
190     }
191   }

375   /** maps per-segment ordinals to/from global ordinal space */
376   // TODO: we could also have a utility method to merge Terms[] and use size() as a weight when we need it
377   // TODO: use more efficient packed ints structures?
378   // TODO: pull this out? it's pretty generic (maps between N ord()-enabled TermsEnums) 
379   public static class OrdinalMap implements Accountable {
380 
381     private static class SegmentMap implements Accountable {
382       private static final long BASE_RAM_BYTES_USED = RamUsageEstimator.shallowSizeOfInstance(SegmentMap.class);
383 
384       /** Build a map from an index into a sorted view of `weights` to an index into `weights`. */
385       private static int[] map(final long[] weights) {
386         final int[] newToOld = new int[weights.length];
387         for (int i = 0; i < weights.length; ++i) {
388           newToOld[i] = i;
389         }
390         new InPlaceMergeSorter() {
391           @Override
392           protected void swap(int i, int j) {
393             final int tmp = newToOld[i];
394             newToOld[i] = newToOld[j];
395             newToOld[j] = tmp;
396           }
397           @Override
398           protected int compare(int i, int j) {
399             // j first since we actually want higher weights first
400             return Long.compare(weights[newToOld[j]], weights[newToOld[i]]);
401           }
402         }.sort(0, weights.length);
403         return newToOld;
404       }
405 
406       /** Inverse the map. */
407       private static int[] inverse(int[] map) {
408         final int[] inverse = new int[map.length];
409         for (int i = 0; i < map.length; ++i) {
410           inverse[map[i]] = i;
411         }
412         return inverse;
413       }
414 
415       private final int[] newToOld, oldToNew;
416 
417       SegmentMap(long[] weights) {
418         newToOld = map(weights);
419         oldToNew = inverse(newToOld);
420         assert Arrays.equals(newToOld, inverse(oldToNew));
421       }
422 
423       int newToOld(int segment) {
424         return newToOld[segment];
425       }
426 
427       int oldToNew(int segment) {
428         return oldToNew[segment];
429       }
430 
431       @Override
432       public long ramBytesUsed() {
433         return BASE_RAM_BYTES_USED + RamUsageEstimator.sizeOf(newToOld) + RamUsageEstimator.sizeOf(oldToNew);
434       }
435 
436       @Override
437       public Collection<Accountable> getChildResources() {
438         return Collections.emptyList();
439       }
440     }

472     /** 
473      * Creates an ordinal map that allows mapping ords to/from a merged
474      * space from <code>subs</code>.
475      * @param owner a cache key
476      * @param subs TermsEnums that support {@link TermsEnum#ord()}. They need
477      *             not be dense (e.g. can be FilteredTermsEnums}.
478      * @param weights a weight for each sub. This is ideally correlated with
479      *             the number of unique terms that each sub introduces compared
480      *             to the other subs
481      * @throws IOException if an I/O error occurred.
482      */
483     public static OrdinalMap build(Object owner, TermsEnum subs[], long[] weights, float acceptableOverheadRatio) throws IOException {
484       if (subs.length != weights.length) {
485         throw new IllegalArgumentException("subs and weights must have the same length");
486       }
487 
488       // enums are not sorted, so let's sort to save memory
489       final SegmentMap segmentMap = new SegmentMap(weights);
490       return new OrdinalMap(owner, subs, segmentMap, acceptableOverheadRatio);
491     }
492 
493     private static final long BASE_RAM_BYTES_USED = RamUsageEstimator.shallowSizeOfInstance(OrdinalMap.class);
494 
495     // cache key of whoever asked for this awful thing
496     final Object owner;
497     // globalOrd -> (globalOrd - segmentOrd) where segmentOrd is the the ordinal in the first segment that contains this term
498     final PackedLongValues globalOrdDeltas;
499     // globalOrd -> first segment container
500     final PackedLongValues firstSegments;
501     // for every segment, segmentOrd -> globalOrd
502     final LongValues segmentToGlobalOrds[];
503     // the map from/to segment ids
504     final SegmentMap segmentMap;
505     // ram usage
506     final long ramBytesUsed;
507     
508     OrdinalMap(Object owner, TermsEnum subs[], SegmentMap segmentMap, float acceptableOverheadRatio) throws IOException {
509       // create the ordinal mappings by pulling a termsenum over each sub's 
510       // unique terms, and walking a multitermsenum over those
511       this.owner = owner;
512       this.segmentMap = segmentMap;
513       // even though we accept an overhead ratio, we keep these ones with COMPACT
514       // since they are only used to resolve values given a global ord, which is
515       // slow anyway
516       PackedLongValues.Builder globalOrdDeltas = PackedLongValues.monotonicBuilder(PackedInts.COMPACT);
517       PackedLongValues.Builder firstSegments = PackedLongValues.packedBuilder(PackedInts.COMPACT);
518       final PackedLongValues.Builder[] ordDeltas = new PackedLongValues.Builder[subs.length];
519       for (int i = 0; i < ordDeltas.length; i++) {
520         ordDeltas[i] = PackedLongValues.monotonicBuilder(acceptableOverheadRatio);
521       }
522       long[] ordDeltaBits = new long[subs.length];
523       long segmentOrds[] = new long[subs.length];
524       ReaderSlice slices[] = new ReaderSlice[subs.length];
525       TermsEnumIndex indexes[] = new TermsEnumIndex[slices.length];
526       for (int i = 0; i < slices.length; i++) {
527         slices[i] = new ReaderSlice(0, 0, i);
528         indexes[i] = new TermsEnumIndex(subs[segmentMap.newToOld(i)], i);
529       }
530       MultiTermsEnum mte = new MultiTermsEnum(slices);
531       mte.reset(indexes);
532       long globalOrd = 0;
533       while (mte.next() != null) {        
534         TermsEnumWithSlice matches[] = mte.getMatchArray();
535         int firstSegmentIndex = Integer.MAX_VALUE;
536         long globalOrdDelta = Long.MAX_VALUE;
537         for (int i = 0; i < mte.getMatchCount(); i++) {
538           int segmentIndex = matches[i].index;
539           long segmentOrd = matches[i].terms.ord();
540           long delta = globalOrd - segmentOrd;
541           // We compute the least segment where the term occurs. In case the
542           // first segment contains most (or better all) values, this will
543           // help save significant memory
544           if (segmentIndex < firstSegmentIndex) {
545             firstSegmentIndex = segmentIndex;
546             globalOrdDelta = delta;
547           }
548           // for each per-segment ord, map it back to the global term.
549           while (segmentOrds[segmentIndex] <= segmentOrd) {
550             ordDeltaBits[segmentIndex] |= delta;
551             ordDeltas[segmentIndex].add(delta);
552             segmentOrds[segmentIndex]++;
553           }
554         }
555         // for each unique term, just mark the first segment index/delta where it occurs
556         assert firstSegmentIndex < segmentOrds.length;
557         firstSegments.add(firstSegmentIndex);
558         globalOrdDeltas.add(globalOrdDelta);
559         globalOrd++;
560       }
561       this.firstSegments = firstSegments.build();
562       this.globalOrdDeltas = globalOrdDeltas.build();
563       // ordDeltas is typically the bottleneck, so let's see what we can do to make it faster
564       segmentToGlobalOrds = new LongValues[subs.length];
565       long ramBytesUsed = BASE_RAM_BYTES_USED + this.globalOrdDeltas.ramBytesUsed()
566           + this.firstSegments.ramBytesUsed() + RamUsageEstimator.shallowSizeOf(segmentToGlobalOrds)
567           + segmentMap.ramBytesUsed();
568       for (int i = 0; i < ordDeltas.length; ++i) {
569         final PackedLongValues deltas = ordDeltas[i].build();
570         if (ordDeltaBits[i] == 0L) {
571           // segment ords perfectly match global ordinals
572           // likely in case of low cardinalities and large segments
573           segmentToGlobalOrds[i] = LongValues.IDENTITY;
574         } else {
575           final int bitsRequired = ordDeltaBits[i] < 0 ? 64 : PackedInts.bitsRequired(ordDeltaBits[i]);
576           final long monotonicBits = deltas.ramBytesUsed() * 8;
577           final long packedBits = bitsRequired * deltas.size();
578           if (deltas.size() <= Integer.MAX_VALUE
579               && packedBits <= monotonicBits * (1 + acceptableOverheadRatio)) {
580             // monotonic compression mostly adds overhead, let's keep the mapping in plain packed ints
581             final int size = (int) deltas.size();
582             final PackedInts.Mutable newDeltas = PackedInts.getMutable(size, bitsRequired, acceptableOverheadRatio);
583             final PackedLongValues.Iterator it = deltas.iterator();
584             for (int ord = 0; ord < size; ++ord) {
585               newDeltas.set(ord, it.next());
586             }
587             assert !it.hasNext();
588             segmentToGlobalOrds[i] = new LongValues() {
589               @Override
590               public long get(long ord) {
591                 return ord + newDeltas.get((int) ord);
592               }
593             };
594             ramBytesUsed += newDeltas.ramBytesUsed();
595           } else {
596             segmentToGlobalOrds[i] = new LongValues() {
597               @Override
598               public long get(long ord) {
599                 return ord + deltas.get(ord);
600               }
601             };
602             ramBytesUsed += deltas.ramBytesUsed();
603           }
604           ramBytesUsed += RamUsageEstimator.shallowSizeOf(segmentToGlobalOrds[i]);
605         }
606       }
607       this.ramBytesUsed = ramBytesUsed;
608     }

658    * Implements SortedDocValues over n subs, using an OrdinalMap
659    * @lucene.internal
660    */
661   public static class MultiSortedDocValues extends SortedDocValues {
662     /** docbase for each leaf: parallel with {@link #values} */
663     public final int docStarts[];
664     /** leaf values */
665     public final SortedDocValues values[];
666     /** ordinal map mapping ords from <code>values</code> to global ord space */
667     public final OrdinalMap mapping;
668   
669     /** Creates a new MultiSortedDocValues over <code>values</code> */
670     MultiSortedDocValues(SortedDocValues values[], int docStarts[], OrdinalMap mapping) throws IOException {
671       assert docStarts.length == values.length + 1;
672       this.values = values;
673       this.docStarts = docStarts;
674       this.mapping = mapping;
675     }
676        
677     @Override
678     public int getOrd(int docID) {
679       int subIndex = ReaderUtil.subIndex(docID, docStarts);
680       int segmentOrd = values[subIndex].getOrd(docID - docStarts[subIndex]);
681       return segmentOrd == -1 ? segmentOrd : (int) mapping.getGlobalOrds(subIndex).get(segmentOrd);
682     }

686       int subIndex = mapping.getFirstSegmentNumber(ord);

688       return values[subIndex].lookupOrd(segmentOrd);
689     }

692     public int getValueCount() {

694     }
