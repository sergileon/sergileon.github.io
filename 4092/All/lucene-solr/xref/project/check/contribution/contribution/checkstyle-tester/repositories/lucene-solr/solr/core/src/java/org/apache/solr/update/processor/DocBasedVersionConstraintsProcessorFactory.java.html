52  /**
53   * <p>
54   * This Factory generates an UpdateProcessor that helps to enforce Version 
55   * constraints on documents based on per-document version numbers using a configured 
56   * name of a <code>versionField</code>.  It should be configured on the "default"
57   * update processor somewhere before the DistributedUpdateProcessorFactory.
58   * As an example, see the solrconfig.xml that the tests use:
59   * solr/core/src/test-files/solr/collection1/conf/solrconfig-externalversionconstraint.xml
60   * </p>
61   * <p>
62   * When documents are added through this processor, if a document with the same 
63   * unique key already exists in the collection, then the value of the 
64   * <code>versionField</code> in the <i>existing</i> document is not less then the 
65   * field value in the <i>new</i> document then the new document is rejected with a 
66   * 409 Version Conflict error.
67   * </p>
68   * <p>
69   * In addition to the mandatory <code>versionField</code> init param, two additional 
70   * optional init params affect the behavior of this factory:
71   * </p>
72   * <ul>
73   *   <li><code>deleteVersionParam</code> - This string parameter controls whether this 
74   *     processor will intercept and inspect Delete By Id commands in addition to adding 
75   *     documents.  If specified, then the value will specify the name of a request 
76   *     paramater which becomes  mandatory for all Delete By Id commands.  This param 
77   *     must then be used to specify the document version associated with the delete.
78   *     If the version specified using this param is not greater then the value in the 
79   *     <code>versionField</code> for any existing document, then the delete will fail 
80   *     with a 409 Version Conflict error.  When using this param, Any Delete By Id 
81   *     command with a high enough document version number to succeed will be internally 
82   *     converted into an Add Document command that replaces the existing document with 
83   *     a new one which is empty except for the Unique Key and <code>versionField</code> 
84   *     to keeping a record of the deleted version so future Add Document commands will 
85   *     fail if their "new" version is not high enough.</li>
86   *
87   *   <li><code>ignoreOldUpdates</code> - This boolean parameter defaults to 
88   *     <code>false</code>, but if set to <code>true</code> causes any update with a 
89   *     document version that is not great enough to be silently ignored (and return 
90   *     a status 200 to the client) instead of generating a 409 Version Conflict error.
91   *   </li>
92   * </ul>
93   */
94  public class DocBasedVersionConstraintsProcessorFactory extends UpdateRequestProcessorFactory implements SolrCoreAware, UpdateRequestProcessorFactory.RunAlways {
95    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
96  
97    private boolean ignoreOldUpdates = false;
98    private String versionField = null;
99    private String deleteVersionParamName = null;
100   private boolean useFieldCache;
101 
102   @Override
103   public void init( NamedList args )  {
104 
105     Object tmp = args.remove("versionField");
106     if (null == tmp) {
107       throw new SolrException(SERVER_ERROR, 
108                               "'versionField' must be configured");
109     }
110     if (! (tmp instanceof String) ) {
111       throw new SolrException(SERVER_ERROR, 
112                               "'versionField' must be configured as a <str>");
113     }
114     versionField = tmp.toString();
115 
116     // optional
117     tmp = args.remove("deleteVersionParam");
118     if (null != tmp) {
119       if (! (tmp instanceof String) ) {
120         throw new SolrException(SERVER_ERROR, 
121                                 "'deleteVersionParam' must be configured as a <str>");
122       }
123       deleteVersionParamName = tmp.toString();
124     }
125 
126     // optional - defaults to false
127     tmp = args.remove("ignoreOldUpdates");
128     if (null != tmp) {
129       if (! (tmp instanceof Boolean) ) {
130         throw new SolrException(SERVER_ERROR, 
131                                 "'ignoreOldUpdates' must be configured as a <bool>");
132       }
133       ignoreOldUpdates = ((Boolean)tmp).booleanValue();
134     }
135     super.init(args);
136   }
137   
138 
139   public UpdateRequestProcessor getInstance(SolrQueryRequest req, 
140                                             SolrQueryResponse rsp, 
141                                             UpdateRequestProcessor next ) {
142     return new DocBasedVersionConstraintsProcessor(versionField, 
143                                                    ignoreOldUpdates,
144                                                    deleteVersionParamName,
145                                                    useFieldCache,
146                                                    req, rsp, next);
147   }
148 
149   @Override
150   public void inform(SolrCore core) {
151 
152     if (core.getUpdateHandler().getUpdateLog() == null) {
153       throw new SolrException(SERVER_ERROR,
154           "updateLog must be enabled.");
155     }
156 
157     if (core.getLatestSchema().getUniqueKeyField() == null) {
158       throw new SolrException(SERVER_ERROR,
159           "schema must have uniqueKey defined.");
160     }
161 
162     SchemaField userVersionField = core.getLatestSchema().getField(versionField);
163     if (userVersionField == null || !userVersionField.stored() || userVersionField.multiValued()) {
164       throw new SolrException(SERVER_ERROR,
165           "field " + versionField + " must be defined in schema, be stored, and be single valued.");
166     }
167 
168     try {
169       ValueSource vs = userVersionField.getType().getValueSource(userVersionField, null);
170       useFieldCache = true;
171     } catch (Exception e) {
172       log.warn("Can't use fieldcache/valuesource: " + e.getMessage());
173     }
174   }

178   private static class DocBasedVersionConstraintsProcessor
179     extends UpdateRequestProcessor {
180 
181     private final String versionFieldName;
182     private final SchemaField userVersionField;
183     private final SchemaField solrVersionField;
184     private final boolean ignoreOldUpdates;
185     private final String deleteVersionParamName;
186     private final SolrCore core;
187 
188     private long oldSolrVersion;  // current _version_ of the doc in the index/update log
189     private DistributedUpdateProcessor distribProc;  // the distributed update processor following us
190     private DistributedUpdateProcessor.DistribPhase phase;
191     private boolean useFieldCache;
192 
193     public DocBasedVersionConstraintsProcessor(String versionField,
194                                                boolean ignoreOldUpdates,
195                                                String deleteVersionParamName,
196                                                boolean useFieldCache,
197                                                SolrQueryRequest req, 
198                                                SolrQueryResponse rsp, 
199                                                UpdateRequestProcessor next ) {
200       super(next);
201       this.ignoreOldUpdates = ignoreOldUpdates;
202       this.deleteVersionParamName = deleteVersionParamName;
203       this.core = req.getCore();
204       this.versionFieldName = versionField;
205       this.userVersionField = core.getLatestSchema().getField(versionField);
206       this.solrVersionField = core.getLatestSchema().getField(VersionInfo.VERSION_FIELD);
207       this.useFieldCache = useFieldCache;
208 
209       for (UpdateRequestProcessor proc = next ;proc != null; proc = proc.next) {
210         if (proc instanceof DistributedUpdateProcessor) {
211           distribProc = (DistributedUpdateProcessor)proc;
212           break;
213         }
214       }
215 
216       if (distribProc == null) {
217         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "DistributedUpdateProcessor must follow DocBasedVersionConstraintsProcessor");
218       }
219 
220       phase = DistributedUpdateProcessor.DistribPhase.parseParam(req.getParams().get(DISTRIB_UPDATE_PARAM));
221 
222     }
223     
224     /**
225      * Inspects a raw field value (which may come from a doc in the index, or a 
226      * doc in the UpdateLog that still has String values, or a String sent by 
227      * the user as a param) and if it is a String, asks the versionField FieldType 
228      * to convert it to an Object suitable for comparison.
229      */
230     private Object convertFieldValueUsingType(SchemaField sf, final Object rawValue) {
231       if (rawValue instanceof CharSequence) {
232         // in theory, the FieldType might still be CharSequence based,
233         // but in that case trust it to do an identiy conversion...
234         FieldType fieldType = userVersionField.getType();
235         BytesRefBuilder term = new BytesRefBuilder();
236         fieldType.readableToIndexed((CharSequence)rawValue, term);
237         return fieldType.toObject(userVersionField, term.get());
238       }
239       // else...
240       return rawValue;
241     }
242 
243 
244     /**
245      * Returns true if the specified new version value is greater the the one
246      * already known to exist for the document, or the document does not already
247      * exist.
248      * Returns false if the specified new version is not high enough but the
249      * processor has been configured with ignoreOldUpdates=true
250      * Throws a SolrException if the version is not high enough and
251      * ignoreOldUpdates=false
252      */
253     private boolean isVersionNewEnough(BytesRef indexedDocId,
254                                        Object newUserVersion) throws IOException {
255       assert null != indexedDocId;
256       assert null != newUserVersion;
257 
258       oldSolrVersion = -1;
259       // log.info("!!!!!!!!! isVersionNewEnough being called for " + indexedDocId.utf8ToString() + " newVersion=" + newUserVersion);
260       newUserVersion = convertFieldValueUsingType(userVersionField, newUserVersion);
261       Object oldUserVersion = null;
262       SolrInputDocument oldDoc = null;
263 
264       if (useFieldCache) {
265         oldDoc = RealTimeGetComponent.getInputDocumentFromTlog(core, indexedDocId);
266         if (oldDoc == RealTimeGetComponent.DELETED) {
267           return true;
268         }
269         if (oldDoc == null) {
270           // need to look up in index now...
271           RefCounted<SolrIndexSearcher> newestSearcher = core.getRealtimeSearcher();
272           try {
273             SolrIndexSearcher searcher = newestSearcher.get();
274             long lookup = searcher.lookupId(indexedDocId);
275             if (lookup < 0) {
276               // doc not in index either...
277               return true;
278             }
279 
280             ValueSource vs = solrVersionField.getType().getValueSource(solrVersionField, null);
281             Map context = ValueSource.newContext(searcher);
282             vs.createWeight(context, searcher);
283             FunctionValues fv = vs.getValues(context, searcher.getTopReaderContext().leaves().get((int)(lookup>>32)));
284             oldSolrVersion = fv.longVal((int)lookup);
285 
286             vs = userVersionField.getType().getValueSource(userVersionField, null);
287             context = ValueSource.newContext(searcher);
288             vs.createWeight(context, searcher);
289             fv = vs.getValues(context, searcher.getTopReaderContext().leaves().get((int)(lookup>>32)));
290             oldUserVersion = fv.objectVal((int)lookup);
291 
292           } catch (IOException e) {
293             throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Error reading version from index", e);
294           } finally {
295             if (newestSearcher != null) {
296               newestSearcher.decref();
297             }
298           }
299         }
300       } else {
301         // stored fields only...
302 
303         oldDoc = RealTimeGetComponent.getInputDocument(core, indexedDocId);
304 
305         if (null == oldDoc) {
306           // log.info("VERSION no doc found, returning true");
307           return true;
308         }
309       }
310 
311 
312       if (oldDoc != null) {
313         oldUserVersion = oldDoc.getFieldValue(versionFieldName);
314         // Make the FieldType resolve any conversion we need.
315         oldUserVersion = convertFieldValueUsingType(userVersionField, oldUserVersion);
316 
317         Object o = oldDoc.getFieldValue(solrVersionField.getName());
318         if (o == null) {
319           throw new SolrException(SERVER_ERROR, "No _version_ for document "+ oldDoc);
320         }
321         oldSolrVersion = o instanceof Number ? ((Number) o).longValue() : Long.parseLong(o.toString());
322       }
323 
324       // log.info("VERSION old=" + oldUserVersion + " new=" +newUserVersion );
325 
326       if ( null == oldUserVersion) {
327         // could happen if they turn this feature on after building an index
328         // w/o the versionField
329         throw new SolrException(SERVER_ERROR,
330             "Doc exists in index, but has null versionField: "
331                 + versionFieldName);
332       }
333 
334 
335       if (! (oldUserVersion instanceof Comparable && newUserVersion instanceof Comparable) ) {
336         throw new SolrException(BAD_REQUEST,
337             "old version and new version are not comparable: " +
338                 oldUserVersion.getClass()+" vs "+newUserVersion.getClass());
339       }
340 
341       try {
342         if (0 < ((Comparable)newUserVersion).compareTo((Comparable) oldUserVersion)) {
343           // log.info("VERSION returning true (proceed with update)" );
344           return true;
345         }
346         if (ignoreOldUpdates) {
347           if (log.isDebugEnabled()) {
348             log.debug("Dropping update since user version is not high enough: " + newUserVersion + "; old user version=" + oldUserVersion);
349           }
350           // log.info("VERSION returning false (dropping update)" );
351           return false;
352         } else {
353           // log.info("VERSION will throw conflict" );
354           throw new SolrException(CONFLICT,
355               "user version is not high enough: " + newUserVersion);
356         }
357       } catch (ClassCastException e) {
358         throw new SolrException(BAD_REQUEST,
359             "old version and new version are not comparable: " +
360                 oldUserVersion.getClass()+" vs "+newUserVersion.getClass() +
361                 ": " + e.getMessage(), e);
362 
363       }
364     }

368     public boolean isLeader(UpdateCommand cmd) {
369       if ((cmd.getFlags() & (UpdateCommand.REPLAY | UpdateCommand.PEER_SYNC)) != 0) {
370         return false;
371       }
372       if (phase == DistributedUpdateProcessor.DistribPhase.FROMLEADER) {
373         return false;
374       }
375       // if phase==TOLEADER, we can't just assume we are the leader... let the normal logic check.
376       boolean x = distribProc.isLeader(cmd);
377       // log.info("VERSION: checking if we are leader:" + x);
378       return x;
379     }
380 
381     public void processAdd(AddUpdateCommand cmd) throws IOException {
382       if (!isLeader(cmd)) {
383         super.processAdd(cmd);
384         return;
385       }
386 
387       final SolrInputDocument newDoc = cmd.getSolrInputDocument();
388 
389       Object newVersion = newDoc.getFieldValue(versionFieldName);
390       if ( null == newVersion ) {
391         throw new SolrException(BAD_REQUEST, "Doc does not have versionField: " + versionFieldName);
392       }
393 
394       for (int i=0; ;i++) {
395         // Log a warning every 256 retries.... even a few retries should normally be very unusual.
396         if ((i&0xff) == 0xff) {
397           log.warn("Unusual number of optimistic concurrency retries: retries=" + i + " cmd=" + cmd);
398         }
399 
400         if (!isVersionNewEnough(cmd.getIndexedId(), newVersion)) {
401           // drop older update
402           return;
403         }
404 
405         try {
406           cmd.setVersion(oldSolrVersion);  // use optimistic concurrency to ensure that the doc has not changed in the meantime
407           super.processAdd(cmd);
408           return;
409         } catch (SolrException e) {
410           if (e.code() == 409) {
411             // log.info ("##################### CONFLICT ADDING newDoc=" + newDoc + " newVersion=" + newVersion );
412             continue;  // if a version conflict, retry
413           }
414           throw e;  // rethrow
415         }
416 
417       }
418     }
419 
420     public void processDelete(DeleteUpdateCommand cmd) throws IOException {
421       if (null == deleteVersionParamName) {
422         // not suppose to look at deletes at all
423         super.processDelete(cmd);
424         return;
425       }
426 
427       if ( ! cmd.isDeleteById() ) {
428         // nothing to do
429         super.processDelete(cmd);
430         return;
431       }
432 
433       String deleteParamValue = cmd.getReq().getParams().get(deleteVersionParamName);
434       if (null == deleteParamValue) {
435         throw new SolrException(BAD_REQUEST,
436             "Delete by ID must specify doc version param: " +
437                 deleteVersionParamName);
438       }
439 
440 
441       if (!isLeader(cmd)) {
442         // transform delete to add earlier rather than later
443 
444         SolrInputDocument newDoc = new SolrInputDocument();
445         newDoc.setField(core.getLatestSchema().getUniqueKeyField().getName(),
446             cmd.getId());
447         newDoc.setField(versionFieldName, deleteParamValue);
448 
449         AddUpdateCommand newCmd = new AddUpdateCommand(cmd.getReq());
450         newCmd.solrDoc = newDoc;
451         newCmd.commitWithin = cmd.commitWithin;
452         super.processAdd(newCmd);
453         return;
454       }
455 
456 
457       for (int i=0; ;i++) {
458         // Log a warning every 256 retries.... even a few retries should normally be very unusual.
459         if ((i&0xff) == 0xff) {
460           log.warn("Unusual number of optimistic concurrency retries: retries=" + i + " cmd=" + cmd);
461         }
462 
463         if (!isVersionNewEnough(cmd.getIndexedId(), deleteParamValue)) {
464           // drop this older update
465           return;
466         }
467 
468         // :TODO: should this logic be split and driven by two params?
469         //   - deleteVersionParam to do a version check
470         //   - some new boolean param to determine if a stub document gets added in place?
471         try {
472           // drop the delete, and instead propogate an AddDoc that
473           // replaces the doc with a new "empty" one that records the deleted version
474 
475           SolrInputDocument newDoc = new SolrInputDocument();
476           newDoc.setField(core.getLatestSchema().getUniqueKeyField().getName(),
477               cmd.getId());
478           newDoc.setField(versionFieldName, deleteParamValue);
479 
480           AddUpdateCommand newCmd = new AddUpdateCommand(cmd.getReq());
481           newCmd.solrDoc = newDoc;
482           newCmd.commitWithin = cmd.commitWithin;
483 
484           newCmd.setVersion(oldSolrVersion);  // use optimistic concurrency to ensure that the doc has not changed in the meantime
485           super.processAdd(newCmd);
486           return;
487         } catch (SolrException e) {
488           if (e.code() == 409) {
489             continue;  // if a version conflict, retry
490           }
491           throw e;  // rethrow
492         }
493 
494       }
495     }
