47    private static final char[] XML_DECLARATION = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>".toCharArray();

49        = "<!-- Solr managed schema - automatically generated - DO NOT EDIT -->".toCharArray();

51    private boolean emitManagedSchemaDoNotEditWarning = false;
52    public void setEmitManagedSchemaDoNotEditWarning(boolean emitManagedSchemaDoNotEditWarning) { 
53      this.emitManagedSchemaDoNotEditWarning = emitManagedSchemaDoNotEditWarning; 
54    }

56    public static void writeResponse(Writer writer, SolrQueryRequest req, SolrQueryResponse rsp) throws IOException {
57      SchemaXmlWriter schemaXmlWriter = null;
58      try {
59        schemaXmlWriter = new SchemaXmlWriter(writer, req, rsp);
60        schemaXmlWriter.writeResponse();
61      } finally {
62        if (null != schemaXmlWriter) {
63          schemaXmlWriter.close();
64        }
65      }
66    }

68    public SchemaXmlWriter(Writer writer, SolrQueryRequest req, SolrQueryResponse rsp) {
69      super(writer, req, rsp);
70    }

72    public void writeResponse() throws IOException {
73      
74      writer.write(XML_DECLARATION);
75      if (emitManagedSchemaDoNotEditWarning) {
76        if (doIndent) {
77          writer.write('\n');
78        }
79        writer.write(MANAGED_SCHEMA_DO_NOT_EDIT_WARNING);
80      }
81  
82      @SuppressWarnings("unchecked") SimpleOrderedMap<Object> schemaProperties
83          = (SimpleOrderedMap<Object>)rsp.getValues().get(IndexSchema.SCHEMA);
84  
85      openStartTag(IndexSchema.SCHEMA);
86      writeAttr(IndexSchema.NAME, schemaProperties.get(IndexSchema.NAME).toString());
87      writeAttr(IndexSchema.VERSION, schemaProperties.get(IndexSchema.VERSION).toString());
88      closeStartTag(false);
89      incLevel();
90  
91      for (int schemaPropNum = 0 ; schemaPropNum < schemaProperties.size() ; ++schemaPropNum) {
92        String schemaPropName = schemaProperties.getName(schemaPropNum);
93        if (schemaPropName.equals(IndexSchema.NAME) || schemaPropName.equals(IndexSchema.VERSION)) {
94          continue;
95        }
96        if (schemaPropName.equals(IndexSchema.UNIQUE_KEY)) {
97          openStartTag(IndexSchema.UNIQUE_KEY);
98          closeStartTag(false);
99          writer.write(schemaProperties.getVal(schemaPropNum).toString());
100         endTag(IndexSchema.UNIQUE_KEY, false);
101       } else if (schemaPropName.equals(IndexSchema.DEFAULT_SEARCH_FIELD)) {
102         openStartTag(IndexSchema.DEFAULT_SEARCH_FIELD);
103         closeStartTag(false);
104         writer.write(schemaProperties.getVal(schemaPropNum).toString());
105         endTag(IndexSchema.DEFAULT_SEARCH_FIELD, false);
106       } else if (schemaPropName.equals(IndexSchema.SOLR_QUERY_PARSER)) {
107         openStartTag(IndexSchema.SOLR_QUERY_PARSER);
108         @SuppressWarnings("unchecked") SimpleOrderedMap<Object> solrQueryParserProperties
109             = (SimpleOrderedMap<Object>)schemaProperties.getVal(schemaPropNum);
110         writeAttr(IndexSchema.DEFAULT_OPERATOR, solrQueryParserProperties.get(IndexSchema.DEFAULT_OPERATOR).toString());
111         closeStartTag(true);
112       } else if (schemaPropName.equals(IndexSchema.SIMILARITY)) {
113         writeSimilarity((SimpleOrderedMap<Object>) schemaProperties.getVal(schemaPropNum));
114       } else if (schemaPropName.equals(IndexSchema.FIELD_TYPES)) {
115         writeFieldTypes((List<SimpleOrderedMap<Object>>)schemaProperties.getVal(schemaPropNum));
116       } else if (schemaPropName.equals(IndexSchema.FIELDS)) {
117         @SuppressWarnings("unchecked") List<SimpleOrderedMap<Object>> fieldPropertiesList
118             = (List<SimpleOrderedMap<Object>>)schemaProperties.getVal(schemaPropNum);
119         for (SimpleOrderedMap<Object> fieldProperties : fieldPropertiesList) {
120           openStartTag(IndexSchema.FIELD);
121           for (int fieldPropNum = 0 ; fieldPropNum < fieldProperties.size() ; ++fieldPropNum) {
122             writeAttr(fieldProperties.getName(fieldPropNum), fieldProperties.getVal(fieldPropNum).toString());
123           }
124           closeStartTag(true);
125         }
126       } else if (schemaPropName.equals(IndexSchema.DYNAMIC_FIELDS)) {
127         @SuppressWarnings("unchecked") List<SimpleOrderedMap<Object>> dynamicFieldPropertiesList 
128             = (List<SimpleOrderedMap<Object>>)schemaProperties.getVal(schemaPropNum);
129         for (SimpleOrderedMap<Object> dynamicFieldProperties : dynamicFieldPropertiesList) {
130           openStartTag(IndexSchema.DYNAMIC_FIELD);
131           for (int dynamicFieldPropNum = 0 ; dynamicFieldPropNum < dynamicFieldProperties.size() ; ++dynamicFieldPropNum) {
132             writeAttr(dynamicFieldProperties.getName(dynamicFieldPropNum), 
133                       dynamicFieldProperties.getVal(dynamicFieldPropNum).toString());
134           }
135           closeStartTag(true);
136         }
137       } else if (schemaPropName.equals(IndexSchema.COPY_FIELDS)) {
138         @SuppressWarnings("unchecked") List<SimpleOrderedMap<Object>> copyFieldPropertiesList
139             = (List<SimpleOrderedMap<Object>>)schemaProperties.getVal(schemaPropNum);
140         for (SimpleOrderedMap<Object> copyFieldProperties : copyFieldPropertiesList) {
141           openStartTag(IndexSchema.COPY_FIELD);
142           for (int copyFieldPropNum = 0 ; copyFieldPropNum < copyFieldProperties.size() ; ++ copyFieldPropNum) {
143             writeAttr(copyFieldProperties.getName(copyFieldPropNum), 
144                       copyFieldProperties.getVal(copyFieldPropNum).toString());
145           }
146           closeStartTag(true);
147         }
148       } else {
149         log.warn("Unknown schema component '" + schemaPropName + "'");
150       }
151     }
152     decLevel();
153     endTag(IndexSchema.SCHEMA);
154   }

156   private void writeFieldTypes(List<SimpleOrderedMap<Object>> fieldTypePropertiesList) throws IOException {
157     for (SimpleOrderedMap<Object> fieldTypeProperties : fieldTypePropertiesList) {
158       SimpleOrderedMap<Object> analyzerProperties = null;
159       SimpleOrderedMap<Object> indexAnalyzerProperties = null;
160       SimpleOrderedMap<Object> queryAnalyzerProperties = null;
161       SimpleOrderedMap<Object> multiTermAnalyzerProperties = null;
162       SimpleOrderedMap<Object> perFieldSimilarityProperties = null;
163       openStartTag(IndexSchema.FIELD_TYPE);
164       for (int fieldTypePropNum = 0 ; fieldTypePropNum < fieldTypeProperties.size() ; ++fieldTypePropNum) {
165         String fieldTypePropName = fieldTypeProperties.getName(fieldTypePropNum);
166         if (fieldTypePropName.equals(FieldType.ANALYZER)) {
167           analyzerProperties = (SimpleOrderedMap<Object>)fieldTypeProperties.getVal(fieldTypePropNum);
168         } else if (fieldTypePropName.equals(FieldType.INDEX_ANALYZER)) {
169           indexAnalyzerProperties =  (SimpleOrderedMap<Object>)fieldTypeProperties.getVal(fieldTypePropNum); 
170         } else if (fieldTypePropName.equals(FieldType.QUERY_ANALYZER)) {
171           queryAnalyzerProperties = (SimpleOrderedMap<Object>)fieldTypeProperties.getVal(fieldTypePropNum);
172         } else if (fieldTypePropName.equals(FieldType.MULTI_TERM_ANALYZER)) {
173           multiTermAnalyzerProperties = (SimpleOrderedMap<Object>)fieldTypeProperties.getVal(fieldTypePropNum);
174         } else if (fieldTypePropName.equals(FieldType.SIMILARITY)) {
175           perFieldSimilarityProperties = (SimpleOrderedMap<Object>)fieldTypeProperties.getVal(fieldTypePropNum);
176         } else {
177           writeAttr(fieldTypePropName, fieldTypeProperties.getVal(fieldTypePropNum).toString());
178         }
179       }
180       boolean isEmptyTag = null == analyzerProperties           && null == indexAnalyzerProperties
181                         && null == queryAnalyzerProperties      && null == multiTermAnalyzerProperties
182                         && null == perFieldSimilarityProperties;
183       if (isEmptyTag) {
184         closeStartTag(true);
185       } else {
186         closeStartTag(false);
187         incLevel();
188         if (null != analyzerProperties)           writeAnalyzer(analyzerProperties, null);
189         if (null != indexAnalyzerProperties)      writeAnalyzer(indexAnalyzerProperties, FieldType.INDEX);
190         if (null != queryAnalyzerProperties)      writeAnalyzer(queryAnalyzerProperties, FieldType.QUERY);
191         if (null != multiTermAnalyzerProperties)  writeAnalyzer(multiTermAnalyzerProperties, FieldType.MULTI_TERM);
192         if (null != perFieldSimilarityProperties) writeSimilarity(perFieldSimilarityProperties);
193         decLevel();
194         endTag(IndexSchema.FIELD_TYPE);
195       }
196     }
197   }

199   private void writeSimilarity(SimpleOrderedMap<Object> similarityProperties) throws IOException {
200     openStartTag(IndexSchema.SIMILARITY);
201     writeAttr(SimilarityFactory.CLASS_NAME, similarityProperties.get(SimilarityFactory.CLASS_NAME).toString());
202     if (similarityProperties.size() > 1) {
203       closeStartTag(false);
204       incLevel();
205       writeNamedList(null, similarityProperties);
206       decLevel();
207       endTag(IndexSchema.SIMILARITY);
208     } else {
209       closeStartTag(true);
210     }
211   }

213   private void writeAnalyzer(SimpleOrderedMap<Object> analyzerProperties, String analyzerType) throws IOException {
214     openStartTag(FieldType.ANALYZER);
215     if (null != analyzerType) {
216       writeAttr(FieldType.TYPE, analyzerType);
217     }
218     List<SimpleOrderedMap<Object>> charFilterPropertiesList = null;
219     SimpleOrderedMap<Object> tokenizerProperties = null;
220     List<SimpleOrderedMap<Object>> filterPropertiesList = null;
221     for (int i = 0 ; i < analyzerProperties.size() ; ++i) {
222       String name = analyzerProperties.getName(i);
223       if (name.equals(FieldType.CHAR_FILTERS)) {
224         charFilterPropertiesList = (List<SimpleOrderedMap<Object>>)analyzerProperties.getVal(i);
225       } else if (name.equals(FieldType.TOKENIZER)) {
226         tokenizerProperties = (SimpleOrderedMap<Object>)analyzerProperties.getVal(i);
227       } else if (name.equals(FieldType.FILTERS)) {
228         filterPropertiesList = (List<SimpleOrderedMap<Object>>)analyzerProperties.getVal(i);
229       } else if (name.equals(FieldType.CLASS_NAME)) {
230         if ( ! "solr.TokenizerChain".equals(analyzerProperties.getVal(i))) {
231           writeAttr(name, analyzerProperties.getVal(i).toString());
232         }
233       }
234     }
235     boolean isEmptyTag
236         = null == charFilterPropertiesList && null == tokenizerProperties && null == filterPropertiesList;
237     if (isEmptyTag) {
238       closeStartTag(true);
239     } else {
240       closeStartTag(false);
241       incLevel();
242       if (null != charFilterPropertiesList) {
243         for (SimpleOrderedMap<Object> charFilterProperties : charFilterPropertiesList) {
244           openStartTag(FieldType.CHAR_FILTER);
245           for (int i = 0 ; i < charFilterProperties.size() ; ++i) {
246             writeAttr(charFilterProperties.getName(i), charFilterProperties.getVal(i).toString());
247           }
248           closeStartTag(true);
249         }
250       }
251       if (null != tokenizerProperties) {
252         openStartTag(FieldType.TOKENIZER);
253         for (int i = 0 ; i < tokenizerProperties.size() ; ++i) {
254           writeAttr(tokenizerProperties.getName(i), tokenizerProperties.getVal(i).toString());
255         }
256         closeStartTag(true);
257       }
258       if (null != filterPropertiesList) {
259         for (SimpleOrderedMap<Object> filterProperties : filterPropertiesList) {
260           openStartTag(FieldType.FILTER);
261           for (int i = 0 ; i < filterProperties.size() ; ++i) {
262             writeAttr(filterProperties.getName(i), filterProperties.getVal(i).toString());
263           }
264           closeStartTag(true);
265         }
266       }
267       decLevel();
268       endTag(FieldType.ANALYZER);
269     }
270   }

272   void openStartTag(String tag) throws IOException {
273     if (doIndent) indent();
274     writer.write('<');
275     writer.write(tag);
276   }
277   
278   void closeStartTag(boolean isEmptyTag) throws IOException {
279     if (isEmptyTag) writer.write('/');
280     writer.write('>');
281   }
282 
283   void endTag(String tag) throws IOException {
284     endTag(tag, true);
285   }
286   
287   void endTag(String tag, boolean indentThisTag) throws IOException {
288     if (doIndent && indentThisTag) indent();
289 
290     writer.write('<');
291     writer.write('/');
292     writer.write(tag);
293     writer.write('>');
294   }

296   /** Writes the XML attribute name/val. A null val means that the attribute is missing. */
297   private void writeAttr(String name, String val) throws IOException {
298     writeAttr(name, val, true);
299   }
300 
301   public void writeAttr(String name, String val, boolean escape) throws IOException{
302     if (val != null) {
303       writer.write(' ');
304       writer.write(name);
305       writer.write("=\"");
306       if (escape){
307         XML.escapeAttributeValue(val, writer);
308       } else {
309         writer.write(val);
310       }
311       writer.write('"');
312     }
313   }

315   @Override
316   public void writeNamedList(String name, NamedList val) throws IOException {
317     // name is ignored - this method is only used for SimilarityFactory
318     int sz = val.size();
319     for (int i=0; i<sz; i++) {
320       String valName = val.getName(i);
321       if ( ! valName.equals(SimilarityFactory.CLASS_NAME)) {
322         writeVal(valName, val.getVal(i));
323       }
324     }
325   }

327   void startTag(String tag, String name, boolean closeTag) throws IOException {
328     if (doIndent) indent();
329 
330     writer.write('<');
331     writer.write(tag);
332     if (name!=null) {
333       writeAttr(NAME, name);
334       if (closeTag) {
335         writer.write("/>");
336       } else {
337         writer.write(">");
338       }
339     } else {
340       if (closeTag) {
341         writer.write("/>");
342       } else {
343         writer.write('>');
344       }
345     }
346   }

349   @Override
350   public void writeMap(String name, Map map, boolean excludeOuter, boolean isFirstVal) throws IOException {
351     int sz = map.size();
352 
353     if (!excludeOuter) {
354       startTag("lst", name, sz<=0);
355       incLevel();
356     }
357 
358     for (Map.Entry entry : (Set<Map.Entry>)map.entrySet()) {
359       Object k = entry.getKey();
360       Object v = entry.getValue();
361       // if (sz<indentThreshold) indent();
362       writeVal( null == k ? null : k.toString(), v);
363     }
364 
365     if (!excludeOuter) {
366       decLevel();
367       if (sz > 0) {
368         if (doIndent) indent();
369         writer.write("</lst>");
370       }
371     }
372   }

379   @Override
380   public void writeArray(String name, Iterator iter) throws IOException {
381     if( iter.hasNext() ) {
382       startTag("arr", name, false );
383       incLevel();
384       while( iter.hasNext() ) {
385         writeVal(null, iter.next());
386       }
387       decLevel();
388       if (doIndent) indent();
389       writer.write("</arr>");
390     }
391     else {
392       startTag("arr", name, true );
393     }
394   }

452   //
453   // OPT - specific writeInt, writeFloat, methods might be faster since
454   // there would be less write calls (write("<int name=\"" + name + ... + </int>)
455   //
456   private void writePrim(String tag, String name, String val, boolean escape) throws IOException {
457     int contentLen = val==null ? 0 : val.length();
458 
459     startTag(tag, name, contentLen==0);
460     if (contentLen==0) return;
461 
462     if (escape) {
463       XML.escapeCharData(val,writer);
464     } else {
465       writer.write(val,0,contentLen);
466     }
467 
468     writer.write('<');
469     writer.write('/');
470     writer.write(tag);
471     writer.write('>');
472   }
