55    public static int count(Term t, IndexReader r) throws IOException {
56      int count = 0;
57      PostingsEnum td = TestUtil.docs(random(), r,
58          t.field(), new BytesRef(t.text()),
59          null,
60          0);
61  
62      if (td != null) {
63        final Bits liveDocs = MultiFields.getLiveDocs(r);
64        while (td.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
65          td.docID();
66          if (liveDocs == null || liveDocs.get(td.docID())) {
67            count++;
68          }
69        }
70      }
71      return count;
72    }

200     Directory dir = newDirectory();
201     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
202     
203     IndexWriter writer = new IndexWriter(dir, iwc);
204     Document doc = new Document();
205     doc.add(newTextField("field", "a b c", Field.Store.NO));
206     writer.addDocument(doc);
207     writer.close();
208     
209     iwc = newIndexWriterConfig(new MockAnalyzer(random()));
210     writer = new IndexWriter(dir, iwc);
211     doc = new Document();
212     doc.add(newTextField("field", "a b c", Field.Store.NO));
213     DirectoryReader nrtReader = writer.getReader();
214     assertTrue(nrtReader.isCurrent());
215     writer.addDocument(doc);
216     assertFalse(nrtReader.isCurrent()); // should see the changes
217     writer.forceMerge(1); // make sure we don't have a merge going on
218     assertFalse(nrtReader.isCurrent());
219     nrtReader.close();
220     
221     DirectoryReader dirReader = DirectoryReader.open(dir);
222     nrtReader = writer.getReader();
223     
224     assertTrue(dirReader.isCurrent());
225     assertTrue(nrtReader.isCurrent()); // nothing was committed yet so we are still current
226     assertEquals(2, nrtReader.maxDoc()); // sees the actual document added
227     assertEquals(1, dirReader.maxDoc());
228     writer.close(); // close is actually a commit both should see the changes
229     assertFalse(nrtReader.isCurrent()); 
230     assertFalse(dirReader.isCurrent()); // this reader has been opened before the writer was closed / committed
231     
232     dirReader.close();
233     nrtReader.close();
234     dir.close();

407   private class AddDirectoriesThreads {
408     Directory addDir;
409     final static int NUM_INIT_DOCS = 100;
410     int numDirs;
411     final Thread[] threads = new Thread[numThreads];
412     IndexWriter mainWriter;
413     final List<Throwable> failures = new ArrayList<>();
414     DirectoryReader[] readers;
415     boolean didClose = false;
416     AtomicInteger count = new AtomicInteger(0);
417     AtomicInteger numaddIndexes = new AtomicInteger(0);
418     
419     public AddDirectoriesThreads(int numDirs, IndexWriter mainWriter) throws Throwable {
420       this.numDirs = numDirs;
421       this.mainWriter = mainWriter;
422       addDir = newDirectory();
423       IndexWriter writer = new IndexWriter(addDir, newIndexWriterConfig(new MockAnalyzer(random()))
424                                                      .setMaxBufferedDocs(2));
425       TestUtil.reduceOpenFiles(writer);
426       for (int i = 0; i < NUM_INIT_DOCS; i++) {
427         Document doc = DocHelper.createDocument(i, "addindex", 4);
428         writer.addDocument(doc);
429       }
430         
431       writer.close();
432       
433       readers = new DirectoryReader[numDirs];
434       for (int i = 0; i < numDirs; i++)
435         readers[i] = DirectoryReader.open(addDir);
436     }
437     
438     void joinThreads() {
439       for (int i = 0; i < numThreads; i++)
440         try {
441           threads[i].join();
442         } catch (InterruptedException ie) {
443           throw new ThreadInterruptedException(ie);
444         }
445     }
446 
447     void close(boolean doWait) throws Throwable {
448       didClose = true;
449       if (doWait) {
450         mainWriter.close();
451       } else {
452         mainWriter.rollback();
453       }
454     }
455 
456     void closeDir() throws Throwable {
457       for (int i = 0; i < numDirs; i++) {
458         readers[i].close();
459       }
460       addDir.close();
461     }
462     
463     void handle(Throwable t) {
464       t.printStackTrace(System.out);
465       synchronized (failures) {
466         failures.add(t);
467       }
468     }
469     
470     void launchThreads(final int numIter) {
471       for (int i = 0; i < numThreads; i++) {
472         threads[i] = new Thread() {
473           @Override
474           public void run() {
475             try {
476               final Directory[] dirs = new Directory[numDirs];
477               for (int k = 0; k < numDirs; k++)
478                 dirs[k] = new MockDirectoryWrapper(random(), TestUtil.ramCopyOf(addDir));
479               //int j = 0;
480               //while (true) {
481                 // System.out.println(Thread.currentThread().getName() + ": iter
482                 // j=" + j);
483                 for (int x=0; x < numIter; x++) {
484                   // only do addIndexes
485                   doBody(x, dirs);
486                 }
487                 //if (numIter > 0 && j == numIter)
488                 //  break;
489                 //doBody(j++, dirs);
490                 //doBody(5, dirs);
491               //}
492             } catch (Throwable t) {
493               handle(t);
494             }
495           }
496         };
497       }
498       for (int i = 0; i < numThreads; i++)
499         threads[i].start();
500     }
501     
502     void doBody(int j, Directory[] dirs) throws Throwable {
503       switch (j % 4) {
504         case 0:
505           mainWriter.addIndexes(dirs);
506           mainWriter.forceMerge(1);
507           break;
508         case 1:
509           mainWriter.addIndexes(dirs);
510           numaddIndexes.incrementAndGet();
511           break;
512         case 2:
513           TestUtil.addIndexesSlowly(mainWriter, readers);
514           break;
515         case 3:
516           mainWriter.commit();
517       }
518       count.addAndGet(dirs.length*NUM_INIT_DOCS);
519     }
520   }

739     }
740 
741     DirectoryReader r = writer.getReader();
742 
743     final int numIterations = 10;
744     final List<Throwable> excs = Collections.synchronizedList(new ArrayList<Throwable>());
745 
746     // Only one thread can addIndexes at a time, because
747     // IndexWriter acquires a write lock in each directory:

766           }
767         };
768       threads[i].setDaemon(true);
769       threads[i].start();
770     }
771

779         IndexSearcher searcher = newSearcher(r);
780         final int count = searcher.search(q, 10).totalHits;
781         assertTrue(count >= lastCount);
782         lastCount = count;
783       }
784     }
785 
786     for(int i=0;i<threads.length;i++) {
787       threads[i].join();
788     }
789     // final check
790     DirectoryReader r2 = DirectoryReader.openIfChanged(r);
791     if (r2 != null) {
792       r.close();
793       r = r2;
794     }
795     Query q = new TermQuery(new Term("indexname", "test"));
796     IndexSearcher searcher = newSearcher(r);
797     final int count = searcher.search(q, 10).totalHits;
798     assertTrue(count >= lastCount);
799 
800     assertEquals(0, excs.size());
801     r.close();
802     if (dir1 instanceof MockDirectoryWrapper) {
803       final Collection<String> openDeletedFiles = ((MockDirectoryWrapper)dir1).getOpenDeletedFiles();
804       assertEquals("openDeleted=" + openDeletedFiles, 0, openDeletedFiles.size());
