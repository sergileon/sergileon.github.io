20  import org.apache.lucene.util.LuceneTestCase;
21  
22  public class SimpleBoundaryScannerTest extends LuceneTestCase {
23    static final String TEXT =
24      "Apache Lucene(TM) is a high-performance, full-featured\ntext search engine library written entirely in Java.";
25  
26    public void testFindStartOffset() throws Exception {
27      StringBuilder text = new StringBuilder(TEXT);
28      BoundaryScanner scanner = new SimpleBoundaryScanner();
29      
30      // test out of range
31      int start = TEXT.length() + 1;
32      assertEquals(start, scanner.findStartOffset(text, start));
33      start = 0;
34      assertEquals(start, scanner.findStartOffset(text, start));
35      
36      start = TEXT.indexOf("formance");
37      int expected = TEXT.indexOf("high-performance");
38      assertEquals(expected, scanner.findStartOffset(text, start));
39      
40      start = TEXT.indexOf("che");
41      expected = TEXT.indexOf("Apache");
42      assertEquals(expected, scanner.findStartOffset(text, start));
43    }
44  
45    public void testFindEndOffset() throws Exception {
46      StringBuilder text = new StringBuilder(TEXT);
47      BoundaryScanner scanner = new SimpleBoundaryScanner();
48      
49      // test out of range
50      int start = TEXT.length() + 1;
51      assertEquals(start, scanner.findEndOffset(text, start));
52      start = -1;
53      assertEquals(start, scanner.findEndOffset(text, start));
54      
55      start = TEXT.indexOf("full-");
56      int expected = TEXT.indexOf("\ntext");
57      assertEquals(expected, scanner.findEndOffset(text, start));
58    }
59  }
