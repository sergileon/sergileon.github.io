40  /**
41   * A generalized version of {@link PhraseQuery}, with an added
42   * method {@link #add(Term[])} for adding more than one term at the same position
43   * that are treated as a disjunction (OR).
44   * To use this class to search for the phrase "Microsoft app*" first use
45   * {@link #add(Term)} on the term "microsoft" (assuming lowercase analysis), then
46   * find all terms that have "app" as prefix using {@link LeafReader#terms(String)},
47   * seeking to "app" then iterating and collecting terms until there is no longer
48   * that prefix, and finally use {@link #add(Term[])} to add them to the query.
49   */
50  public class MultiPhraseQuery extends Query {
51    private String field;// becomes non-null on first add() then is unmodified
52    private final ArrayList<Term[]> termArrays = new ArrayList<>();
53    private final ArrayList<Integer> positions = new ArrayList<>();
54  
55    private int slop = 0;
56  
57    /** Sets the phrase slop for this query.
58     * @see PhraseQuery#getSlop()
59     */
60    public void setSlop(int s) {
61      if (s < 0) {
62        throw new IllegalArgumentException("slop value cannot be negative");
63      }
64      slop = s; 
65    }
66  
67    /** Sets the phrase slop for this query.
68     * @see PhraseQuery#getSlop()
69     */
70    public int getSlop() { return slop; }
71  
72    /** Add a single term at the next position in the phrase.
73     */
74    public void add(Term term) { add(new Term[]{term}); }
75  
76    /** Add multiple terms at the next position in the phrase.  Any of the terms
77     * may match (a disjunction).
78     * The array is not copied or mutated, the caller should consider it
79     * immutable subsequent to calling this method.
80     */
81    public void add(Term[] terms) {
82      int position = 0;
83      if (positions.size() > 0)
84        position = positions.get(positions.size() - 1) + 1;
85  
86      add(terms, position);
87    }
88  
89    /**
90     * Allows to specify the relative position of terms within the phrase.
91     * The array is not copied or mutated, the caller should consider it
92     * immutable subsequent to calling this method.
93     */
94    public void add(Term[] terms, int position) {
95      Objects.requireNonNull(terms, "Term array must not be null");
96      if (termArrays.size() == 0)
97        field = terms[0].field();
98  
99      for (Term term : terms) {
100       if (!term.field().equals(field)) {
101         throw new IllegalArgumentException(
102             "All phrase terms must be in the same field (" + field + "): " + term);
103       }
104     }
105 
106     termArrays.add(terms);
107     positions.add(position);
108   }
109 
110   /**
111    * Returns a List of the terms in the multi-phrase.
112    * Do not modify the List or its contents.
113    */
114   public List<Term[]> getTermArrays() {
115     return Collections.unmodifiableList(termArrays);
116   }
117 
118   /**
119    * Returns the relative positions of terms in this phrase.
120    */
121   public int[] getPositions() {
122     int[] result = new int[positions.size()];
123     for (int i = 0; i < positions.size(); i++)
124       result[i] = positions.get(i);
125     return result;
126   }

129   private class MultiPhraseWeight extends Weight {
130     private final Similarity similarity;
131     private final Similarity.SimWeight stats;
132     private final Map<Term,TermContext> termContexts = new HashMap<>();
133     private final boolean needsScores;
134 
135     public MultiPhraseWeight(IndexSearcher searcher, boolean needsScores)
136       throws IOException {
137       super(MultiPhraseQuery.this);
138       this.needsScores = needsScores;
139       this.similarity = searcher.getSimilarity(needsScores);
140       final IndexReaderContext context = searcher.getTopReaderContext();
141       
142       // compute idf
143       ArrayList<TermStatistics> allTermStats = new ArrayList<>();
144       for(final Term[] terms: termArrays) {
145         for (Term term: terms) {
146           TermContext termContext = termContexts.get(term);
147           if (termContext == null) {
148             termContext = TermContext.build(context, term);
149             termContexts.put(term, termContext);
150           }
151           allTermStats.add(searcher.termStatistics(term, termContext));
152         }
153       }
154       stats = similarity.computeWeight(
155           searcher.collectionStatistics(field), 
156           allTermStats.toArray(new TermStatistics[allTermStats.size()]));
157     }
158 
159     @Override
160     public void extractTerms(Set<Term> terms) {
161       for (final Term[] arr : termArrays) {
162         Collections.addAll(terms, arr);
163       }
164     }
165 
166     @Override
167     public float getValueForNormalization() {
168       return stats.getValueForNormalization();
169     }
170 
171     @Override
172     public void normalize(float queryNorm, float boost) {
173       stats.normalize(queryNorm, boost);
174     }
175 
176     @Override
177     public Scorer scorer(LeafReaderContext context) throws IOException {
178       assert !termArrays.isEmpty();
179       final LeafReader reader = context.reader();
180       
181       PhraseQuery.PostingsAndFreq[] postingsFreqs = new PhraseQuery.PostingsAndFreq[termArrays.size()];
182 
183       final Terms fieldTerms = reader.terms(field);
184       if (fieldTerms == null) {
185         return null;
186       }
187 
188       // TODO: move this check to createWeight to happen earlier to the user?
189       if (fieldTerms.hasPositions() == false) {
190         throw new IllegalStateException("field \"" + field + "\" was indexed without position data;" +
191             " cannot run MultiPhraseQuery (phrase=" + getQuery() + ")");
192       }
193 
194       // Reuse single TermsEnum below:
195       final TermsEnum termsEnum = fieldTerms.iterator();
196       float totalMatchCost = 0;
197 
198       for (int pos=0; pos<postingsFreqs.length; pos++) {
199         Term[] terms = termArrays.get(pos);
200         List<PostingsEnum> postings = new ArrayList<>();
201         
202         for (Term term : terms) {
203           TermState termState = termContexts.get(term).get(context.ord);
204           if (termState != null) {
205             termsEnum.seekExact(term.bytes(), termState);
206             postings.add(termsEnum.postings(null, PostingsEnum.POSITIONS));
207             totalMatchCost += PhraseQuery.termPositionsCost(termsEnum);
208           }
209         }
210         
211         if (postings.isEmpty()) {
212           return null;
213         }
214         
215         final PostingsEnum postingsEnum;
216         if (postings.size() == 1) {
217           postingsEnum = postings.get(0);
218         } else {
219           postingsEnum = new UnionPostingsEnum(postings);
220         }
221 
222         postingsFreqs[pos] = new PhraseQuery.PostingsAndFreq(postingsEnum, positions.get(pos).intValue(), terms);
223       }
224 
225       // sort by increasing docFreq order
226       if (slop == 0) {
227         ArrayUtil.timSort(postingsFreqs);
228       }
229 
230       if (slop == 0) {
231         return new ExactPhraseScorer(this, postingsFreqs,
232                                       similarity.simScorer(stats, context),
233                                       needsScores, totalMatchCost);
234       } else {
235         return new SloppyPhraseScorer(this, postingsFreqs, slop,
236                                         similarity.simScorer(stats, context),
237                                         needsScores, totalMatchCost);
238       }
239     }
240 
241     @Override
242     public Explanation explain(LeafReaderContext context, int doc) throws IOException {
243       Scorer scorer = scorer(context);
244       if (scorer != null) {
245         int newDoc = scorer.advance(doc);
246         if (newDoc == doc) {
247           float freq = slop == 0 ? scorer.freq() : ((SloppyPhraseScorer)scorer).sloppyFreq();
248           SimScorer docScorer = similarity.simScorer(stats, context);
249           Explanation freqExplanation = Explanation.match(freq, "phraseFreq=" + freq);
250           Explanation scoreExplanation = docScorer.explain(doc, freqExplanation);
251           return Explanation.match(
252               scoreExplanation.getValue(),
253               "weight("+getQuery()+" in "+doc+") [" + similarity.getClass().getSimpleName() + "], result of:",
254               scoreExplanation);
255         }
256       }
257       
258       return Explanation.noMatch("no matching term");
259     }
260   }
261 
262   @Override
263   public Query rewrite(IndexReader reader) throws IOException {
264     if (getBoost() != 1f) {
265       return super.rewrite(reader);
266     }
267     if (termArrays.isEmpty()) {
268       return new MatchNoDocsQuery();
269     } else if (termArrays.size() == 1) {                 // optimize one-term case
270       Term[] terms = termArrays.get(0);
271       BooleanQuery.Builder builder = new BooleanQuery.Builder();
272       builder.setDisableCoord(true);
273       for (Term term : terms) {
274         builder.add(new TermQuery(term), BooleanClause.Occur.SHOULD);
275       }
276       return builder.build();
277     } else {
278       return super.rewrite(reader);
279     }
280   }

282   @Override
283   public Weight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
284     return new MultiPhraseWeight(searcher, needsScores);
285   }

287   /** Prints a user-readable version of this query. */
288   @Override
289   public final String toString(String f) {
290     StringBuilder buffer = new StringBuilder();
291     if (field == null || !field.equals(f)) {
292       buffer.append(field);
293       buffer.append(":");
294     }
295 
296     buffer.append("\"");
297     int k = 0;
298     Iterator<Term[]> i = termArrays.iterator();
299     int lastPos = -1;
300     boolean first = true;
301     while (i.hasNext()) {
302       Term[] terms = i.next();
303       int position = positions.get(k);
304       if (first) {
305         first = false;
306       } else {
307         buffer.append(" ");
308         for (int j=1; j<(position-lastPos); j++) {
309           buffer.append("? ");
310         }
311       }
312       if (terms.length > 1) {
313         buffer.append("(");
314         for (int j = 0; j < terms.length; j++) {
315           buffer.append(terms[j].text());
316           if (j < terms.length-1)
317             buffer.append(" ");
318         }
319         buffer.append(")");
320       } else {
321         buffer.append(terms[0].text());
322       }
323       lastPos = position;
324       ++k;
325     }
326     buffer.append("\"");
327 
328     if (slop != 0) {
329       buffer.append("~");
330       buffer.append(slop);
331     }
332 
333     buffer.append(ToStringUtils.boost(getBoost()));
334     return buffer.toString();
335   }

338   /** Returns true if <code>o</code> is equal to this. */
339   @Override
340   public boolean equals(Object o) {
341     if (!(o instanceof MultiPhraseQuery)) return false;
342     MultiPhraseQuery other = (MultiPhraseQuery)o;
343     return super.equals(o)
344       && this.slop == other.slop
345       && termArraysEquals(this.termArrays, other.termArrays)
346       && this.positions.equals(other.positions);
347   }

349   /** Returns a hash code value for this object.*/
350   @Override
351   public int hashCode() {
352     return super.hashCode()
353       ^ slop
354       ^ termArraysHashCode()
355       ^ positions.hashCode();
356   }

358   // Breakout calculation of the termArrays hashcode
359   private int termArraysHashCode() {
360     int hashCode = 1;
361     for (final Term[] termArray: termArrays) {
362       hashCode = 31 * hashCode
363           + (termArray == null ? 0 : Arrays.hashCode(termArray));
364     }
365     return hashCode;
366   }
367 
368   // Breakout calculation of the termArrays equals
369   private boolean termArraysEquals(List<Term[]> termArrays1, List<Term[]> termArrays2) {
370     if (termArrays1.size() != termArrays2.size()) {
371       return false;
372     }
373     ListIterator<Term[]> iterator1 = termArrays1.listIterator();
374     ListIterator<Term[]> iterator2 = termArrays2.listIterator();
375     while (iterator1.hasNext()) {
376       Term[] termArray1 = iterator1.next();
377       Term[] termArray2 = iterator2.next();
378       if (!(termArray1 == null ? termArray2 == null : Arrays.equals(termArray1,
379           termArray2))) {
380         return false;
381       }
382     }
383     return true;
384   }

386   /** 
387    * Takes the logical union of multiple PostingsEnum iterators.
388    * <p>
389    * Note: positions are merged during freq()
390    */
391   static class UnionPostingsEnum extends PostingsEnum {
392     /** queue ordered by docid */
393     final DocsQueue docsQueue;
394     /** cost of this enum: sum of its subs */
395     final long cost;
396     
397     /** queue ordered by position for current doc */
398     final PositionsQueue posQueue = new PositionsQueue();
399     /** current doc posQueue is working */
400     int posQueueDoc = -2;
401     /** list of subs (unordered) */
402     final PostingsEnum[] subs;
403     
404     UnionPostingsEnum(Collection<PostingsEnum> subs) {
405       docsQueue = new DocsQueue(subs.size());
406       long cost = 0;
407       for (PostingsEnum sub : subs) {
408         docsQueue.add(sub);
409         cost += sub.cost();
410       }
411       this.cost = cost;
412       this.subs = subs.toArray(new PostingsEnum[subs.size()]);
413     }
414 
415     @Override
416     public int freq() throws IOException {
417       int doc = docID();
418       if (doc != posQueueDoc) {
419         posQueue.clear();
420         for (PostingsEnum sub : subs) {
421           if (sub.docID() == doc) {
422             int freq = sub.freq();
423             for (int i = 0; i < freq; i++) {
424               posQueue.add(sub.nextPosition());
425             }
426           }
427         }
428         posQueue.sort();
429         posQueueDoc = doc;
430       }
431       return posQueue.size();
432     }
433 
434     @Override
435     public int nextPosition() throws IOException {
436       return posQueue.next();
437     }
438 
439     @Override
440     public int docID() {
441       return docsQueue.top().docID();
442     }
443 
444     @Override
445     public int nextDoc() throws IOException {
446       PostingsEnum top = docsQueue.top();
447       int doc = top.docID();
448       
449       do {
450         top.nextDoc();
451         top = docsQueue.updateTop();
452       } while (top.docID() == doc);
453 
454       return top.docID();
455     }
456 
457     @Override
458     public int advance(int target) throws IOException {
459       PostingsEnum top = docsQueue.top();
460       
461       do {
462         top.advance(target);
463         top = docsQueue.updateTop();
464       } while (top.docID() < target);
465 
466       return top.docID();
467     }
468 
469     @Override
470     public long cost() {
471       return cost;
472     }
473     
474     @Override
475     public int startOffset() throws IOException {
476       return -1; // offsets are unsupported
477     }

479     @Override
480     public int endOffset() throws IOException {
481       return -1; // offsets are unsupported
482     }

487     }

489     /** 
490      * disjunction of postings ordered by docid.
491      */
492     static class DocsQueue extends PriorityQueue<PostingsEnum> {
493       DocsQueue(int size) {
494         super(size);
495       }
496 
497       @Override
498       public final boolean lessThan(PostingsEnum a, PostingsEnum b) {
499         return a.docID() < b.docID();
500       }
501     }
502     
503     /** 
504      * queue of terms for a single document. its a sorted array of
505      * all the positions from all the postings
506      */
507     static class PositionsQueue {
508       private int arraySize = 16;
509       private int index = 0;
510       private int size = 0;
511       private int[] array = new int[arraySize];
512       
513       void add(int i) {
514         if (size == arraySize)
515           growArray();
516 
517         array[size++] = i;
518       }
519 
520       int next() {
521         return array[index++];
522       }
523 
524       void sort() {
525         Arrays.sort(array, index, size);
526       }
527 
528       void clear() {
529         index = 0;
530         size = 0;
531       }
532 
533       int size() {
534         return size;
535       }
536 
537       private void growArray() {
538         int[] newArray = new int[arraySize * 2];
539         System.arraycopy(array, 0, newArray, 0, arraySize);
540         array = newArray;
541         arraySize *= 2;
542       }
543     }
