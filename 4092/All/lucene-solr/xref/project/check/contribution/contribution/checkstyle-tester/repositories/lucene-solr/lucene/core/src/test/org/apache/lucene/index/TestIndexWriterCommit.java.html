40     * Simple test for "commit on close": open writer then
41     * add a bunch of docs, making sure reader does not see
42     * these docs until writer is closed.
43     */
44    public void testCommitOnClose() throws IOException {
45        Directory dir = newDirectory();
46        IndexWriter writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random())));
47        for (int i = 0; i < 14; i++) {
48          TestIndexWriter.addDoc(writer);
49        }
50        writer.close();
51  
52        Term searchTerm = new Term("content", "aaa");
53        DirectoryReader reader = DirectoryReader.open(dir);
54        IndexSearcher searcher = newSearcher(reader);
55        ScoreDoc[] hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
56        assertEquals("first number of hits", 14, hits.length);
57        reader.close();
58  
59        reader = DirectoryReader.open(dir);
60  
61        writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random())));
62        for(int i=0;i<3;i++) {
63          for(int j=0;j<11;j++) {
64            TestIndexWriter.addDoc(writer);
65          }
66          IndexReader r = DirectoryReader.open(dir);
67          searcher = newSearcher(r);
68          hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
69          assertEquals("reader incorrectly sees changes from writer", 14, hits.length);
70          r.close();
71          assertTrue("reader should have still been current", reader.isCurrent());
72        }
73  
74        // Now, close the writer:
75        writer.close();
76        assertFalse("reader should not be current now", reader.isCurrent());
77  
78        IndexReader r = DirectoryReader.open(dir);
79        searcher = newSearcher(r);
80        hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
81        assertEquals("reader did not see changes after writer was closed", 47, hits.length);

87    /*
88     * Simple test for "commit on close": open writer, then
89     * add a bunch of docs, making sure reader does not see
90     * them until writer has closed.  Then instead of
91     * closing the writer, call abort and verify reader sees
92     * nothing was added.  Then verify we can open the index
93     * and add docs to it.
94     */
95    public void testCommitOnCloseAbort() throws IOException {
96      Directory dir = newDirectory();
97      if (dir instanceof MockDirectoryWrapper) {
98        // test uses IW unref'ed check which is unaware of retries
99        ((MockDirectoryWrapper)dir).setEnableVirusScanner(false);
100     }
101     IndexWriter writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
102                                                 .setMaxBufferedDocs(10));
103     for (int i = 0; i < 14; i++) {
104       TestIndexWriter.addDoc(writer);
105     }
106     writer.close();
107 
108     Term searchTerm = new Term("content", "aaa");
109     IndexReader reader = DirectoryReader.open(dir);
110     IndexSearcher searcher = newSearcher(reader);
111     ScoreDoc[] hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
112     assertEquals("first number of hits", 14, hits.length);
113     reader.close();
114 
115     writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
116                                     .setOpenMode(OpenMode.APPEND)
117                                     .setMaxBufferedDocs(10));
118     for(int j=0;j<17;j++) {
119       TestIndexWriter.addDoc(writer);
120     }
121     // Delete all docs:
122     writer.deleteDocuments(searchTerm);
123 
124     reader = DirectoryReader.open(dir);
125     searcher = newSearcher(reader);
126     hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
127     assertEquals("reader incorrectly sees changes from writer", 14, hits.length);
128     reader.close();
129 
130     // Now, close the writer:
131     writer.rollback();
132 
133     TestIndexWriter.assertNoUnreferencedFiles(dir, "unreferenced files remain after rollback()");
134 
135     reader = DirectoryReader.open(dir);
136     searcher = newSearcher(reader);
137     hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
138     assertEquals("saw changes after writer.abort", 14, hits.length);
139     reader.close();
140 
141     // Now make sure we can re-open the index, add docs,
142     // and all is good:
143     writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
144                                     .setOpenMode(OpenMode.APPEND)
145                                     .setMaxBufferedDocs(10));
146 
147     // On abort, writer in fact may write to the same
148     // segments_N file:
149     if (dir instanceof MockDirectoryWrapper) {
150       ((MockDirectoryWrapper)dir).setPreventDoubleWrite(false);
151     }
152 
153     for(int i=0;i<12;i++) {
154       for(int j=0;j<17;j++) {
155         TestIndexWriter.addDoc(writer);
156       }
157       IndexReader r = DirectoryReader.open(dir);
158       searcher = newSearcher(r);
159       hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
160       assertEquals("reader incorrectly sees changes from writer", 14, hits.length);
161       r.close();
162     }
163 
164     writer.close();
165     IndexReader r = DirectoryReader.open(dir);
166     searcher = newSearcher(r);
167     hits = searcher.search(new TermQuery(searchTerm), 1000).scoreDocs;
168     assertEquals("didn't see changes after close", 218, hits.length);
169     r.close();
170 
171     dir.close();
172   }

174   /*
175    * Verify that a writer with "commit on close" indeed
176    * cleans up the temp segments created after opening
177    * that are not referenced by the starting segments
178    * file.  We check this by using MockDirectoryWrapper to
179    * measure max temp disk space used.
180    */
181   public void testCommitOnCloseDiskUsage() throws IOException {
182     // MemoryCodec, since it uses FST, is not necessarily
183     // "additive", ie if you add up N small FSTs, then merge
184     // them, the merged result can easily be larger than the
185     // sum because the merged FST may use array encoding for
186     // some arcs (which uses more space):
187 
188     final String idFormat = TestUtil.getPostingsFormat("id");
189     final String contentFormat = TestUtil.getPostingsFormat("content");
190     assumeFalse("This test cannot run with Memory codec", idFormat.equals("Memory") || contentFormat.equals("Memory"));
191     MockDirectoryWrapper dir = newMockDirectory();
192     if (dir instanceof MockDirectoryWrapper) {
193       // the virus scanner can use up too much disk space :)
194       // an alternative is to expose MDW.triedToDelete and discount it
195       dir.setEnableVirusScanner(false);
196     }
197     Analyzer analyzer;
198     if (random().nextBoolean()) {
199       // no payloads
200      analyzer = new Analyzer() {
201         @Override
202         public TokenStreamComponents createComponents(String fieldName) {
203           return new TokenStreamComponents(new MockTokenizer(MockTokenizer.WHITESPACE, true));
204         }
205       };
206     } else {
207       // fixed length payloads
208       final int length = random().nextInt(200);
209       analyzer = new Analyzer() {
210         @Override
211         public TokenStreamComponents createComponents(String fieldName) {
212           Tokenizer tokenizer = new MockTokenizer(MockTokenizer.WHITESPACE, true);
213           return new TokenStreamComponents(tokenizer, new MockFixedLengthPayloadFilter(random(), tokenizer, length));
214         }
215       };
216     }
217     
218     IndexWriter writer  = new IndexWriter(
219         dir,
220         newIndexWriterConfig(analyzer)
221             .setMaxBufferedDocs(10)
222             .setReaderPooling(false)
223             .setMergePolicy(newLogMergePolicy(10))
224     );
225     for(int j=0;j<30;j++) {
226       TestIndexWriter.addDocWithIndex(writer, j);
227     }
228     writer.close();
229     dir.resetMaxUsedSizeInBytes();
230 
231     dir.setTrackDiskUsage(true);
232     long startDiskUsage = dir.getMaxUsedSizeInBytes();
233     writer = new IndexWriter(
234         dir,
235         newIndexWriterConfig(analyzer)
236             .setOpenMode(OpenMode.APPEND)
237             .setMaxBufferedDocs(10)
238             .setMergeScheduler(new SerialMergeScheduler())
239             .setReaderPooling(false)
240             .setMergePolicy(newLogMergePolicy(10))
241 
242     );
243     for(int j=0;j<1470;j++) {
244       TestIndexWriter.addDocWithIndex(writer, j);
245     }
246     long midDiskUsage = dir.getMaxUsedSizeInBytes();
247     dir.resetMaxUsedSizeInBytes();
248     writer.forceMerge(1);
249     writer.close();
250 
251     DirectoryReader.open(dir).close();
252 
253     long endDiskUsage = dir.getMaxUsedSizeInBytes();
254 
255     // Ending index is 50X as large as starting index; due
256     // to 3X disk usage normally we allow 150X max
257     // transient usage.  If something is wrong w/ deleter
258     // and it doesn't delete intermediate segments then it
259     // will exceed this 150X:
260     // System.out.println("start " + startDiskUsage + "; mid " + midDiskUsage + ";end " + endDiskUsage);
261     assertTrue("writer used too much space while adding documents: mid=" + midDiskUsage + " start=" + startDiskUsage + " end=" + endDiskUsage + " max=" + (startDiskUsage*150),
262                midDiskUsage < 150*startDiskUsage);
263     assertTrue("writer used too much space after close: endDiskUsage=" + endDiskUsage + " startDiskUsage=" + startDiskUsage + " max=" + (startDiskUsage*150),
264                endDiskUsage < 150*startDiskUsage);
265     dir.close();
266   }

269   /*
270    * Verify that calling forceMerge when writer is open for
271    * "commit on close" works correctly both for rollback()
272    * and close().
273    */
274   public void testCommitOnCloseForceMerge() throws IOException {
275     Directory dir = newDirectory();
276     // Must disable throwing exc on double-write: this
277     // test uses IW.rollback which easily results in
278     // writing to same file more than once
279     if (dir instanceof MockDirectoryWrapper) {
280       ((MockDirectoryWrapper)dir).setPreventDoubleWrite(false);
281     }
282     if (dir instanceof MockDirectoryWrapper) {
283       // test uses IW unref'ed check which is unaware of retries
284       ((MockDirectoryWrapper)dir).setEnableVirusScanner(false);
285     }
286     IndexWriter writer = new IndexWriter(
287         dir,
288         newIndexWriterConfig(new MockAnalyzer(random()))
289             .setMaxBufferedDocs(10)
290             .setMergePolicy(newLogMergePolicy(10))
291     );
292     for(int j=0;j<17;j++) {
293       TestIndexWriter.addDocWithIndex(writer, j);
294     }
295     writer.close();
296 
297     writer  = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
298                                      .setOpenMode(OpenMode.APPEND));
299     writer.forceMerge(1);
300 
301     // Open a reader before closing (commiting) the writer:
302     DirectoryReader reader = DirectoryReader.open(dir);
303 
304     // Reader should see index as multi-seg at this
305     // point:
306     assertTrue("Reader incorrectly sees one segment", reader.leaves().size() > 1);
307     reader.close();
308 
309     // Abort the writer:
310     writer.rollback();
311     TestIndexWriter.assertNoUnreferencedFiles(dir, "aborted writer after forceMerge");
312 
313     // Open a reader after aborting writer:
314     reader = DirectoryReader.open(dir);
315 
316     // Reader should still see index as multi-segment
317     assertTrue("Reader incorrectly sees one segment", reader.leaves().size() > 1);
318     reader.close();
319 
320     if (VERBOSE) {
321       System.out.println("TEST: do real full merge");
322     }
323     writer = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
324                                     .setOpenMode(OpenMode.APPEND));
325     writer.forceMerge(1);
326     writer.close();
327 
328     if (VERBOSE) {
329       System.out.println("TEST: writer closed");
330     }
331     TestIndexWriter.assertNoUnreferencedFiles(dir, "aborted writer after forceMerge");
332 
333     // Open a reader after aborting writer:
334     reader = DirectoryReader.open(dir);
335 
336     // Reader should see index as one segment
337     assertEquals("Reader incorrectly sees more than one segment", 1, reader.leaves().size());
338     reader.close();
339     dir.close();
340   }

342   // LUCENE-2095: make sure with multiple threads commit
343   // doesn't return until all changes are in fact in the
344   // index
345   public void testCommitThreadSafety() throws Throwable {
346     final int NUM_THREADS = 5;
347     final double RUN_SEC = 0.5;
348     final Directory dir = newDirectory();
349     final RandomIndexWriter w = new RandomIndexWriter(random(), dir, newIndexWriterConfig(new MockAnalyzer(random()))
350                                                                        .setMergePolicy(newLogMergePolicy()));
351     TestUtil.reduceOpenFiles(w.w);
352     w.commit();
353     final AtomicBoolean failed = new AtomicBoolean();
354     Thread[] threads = new Thread[NUM_THREADS];
355     final long endTime = System.currentTimeMillis()+((long) (RUN_SEC*1000));
356     for(int i=0;i<NUM_THREADS;i++) {
357       final int finalI = i;
358       threads[i] = new Thread() {
359           @Override
360           public void run() {
361             try {
362               final Document doc = new Document();
363               DirectoryReader r = DirectoryReader.open(dir);
364               Field f = newStringField("f", "", Field.Store.NO);
365               doc.add(f);
366               int count = 0;
367               do {
368                 if (failed.get()) break;
369                 for(int j=0;j<10;j++) {
370                   final String s = finalI + "_" + String.valueOf(count++);
371                   f.setStringValue(s);
372                   w.addDocument(doc);
373                   w.commit();
374                   DirectoryReader r2 = DirectoryReader.openIfChanged(r);
375                   assertNotNull(r2);
376                   assertTrue(r2 != r);
377                   r.close();
378                   r = r2;
379                   assertEquals("term=f:" + s + "; r=" + r, 1, r.docFreq(new Term("f", s)));
380                 }
381               } while(System.currentTimeMillis() < endTime);
382               r.close();
383             } catch (Throwable t) {
384               failed.set(true);
385               throw new RuntimeException(t);
386             }
387           }
388         };
389       threads[i].start();
390     }
391     for(int i=0;i<NUM_THREADS;i++) {
392       threads[i].join();
393     }
394     assertFalse(failed.get());
395     w.close();
396     dir.close();
397   }

440     Directory dir = newDirectory();
441 
442     IndexWriter w = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
443                                            .setIndexDeletionPolicy(NoDeletionPolicy.INSTANCE));
444     Document doc = new Document();
445     w.addDocument(doc);
446 
447     // commit to "first"
448     Map<String,String> commitData = new HashMap<>();
449     commitData.put("tag", "first");
450     w.setCommitData(commitData);
451     w.commit();
452 
453     // commit to "second"
454     w.addDocument(doc);
455     commitData.put("tag", "second");
456     w.setCommitData(commitData);
457     w.close();
458 
459     // open "first" with IndexWriter
460     IndexCommit commit = null;
461     for(IndexCommit c : DirectoryReader.listCommits(dir)) {
462       if (c.getUserData().get("tag").equals("first")) {
463         commit = c;
464         break;
465       }
466     }
467 
468     assertNotNull(commit);
469 
470     w = new IndexWriter(dir, newIndexWriterConfig(new MockAnalyzer(random()))
471                                .setIndexDeletionPolicy(NoDeletionPolicy.INSTANCE)
472                                .setIndexCommit(commit));
473 
474     assertEquals(1, w.numDocs());
475 
476     // commit IndexWriter to "third"
477     w.addDocument(doc);
478     commitData.put("tag", "third");
479     w.setCommitData(commitData);
480     w.close();
481 
482     // make sure "second" commit is still there
483     commit = null;
484     for(IndexCommit c : DirectoryReader.listCommits(dir)) {
485       if (c.getUserData().get("tag").equals("second")) {
486         commit = c;
487         break;
488       }
489     }
490 
491     assertNotNull(commit);
492 
493     dir.close();
494   }
