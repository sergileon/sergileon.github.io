63  @Deprecated
64  final class Lucene42DocValuesConsumer extends DocValuesConsumer {
65    final IndexOutput data, meta;
66    final int maxDoc;
67    final float acceptableOverheadRatio;
68    
69    Lucene42DocValuesConsumer(SegmentWriteState state, String dataCodec, String dataExtension, String metaCodec, String metaExtension, float acceptableOverheadRatio) throws IOException {
70      this.acceptableOverheadRatio = acceptableOverheadRatio;
71      maxDoc = state.segmentInfo.maxDoc();
72      boolean success = false;
73      try {
74        String dataName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, dataExtension);
75        data = state.directory.createOutput(dataName, state.context);
76        // this writer writes the format 4.2 did!
77        CodecUtil.writeHeader(data, dataCodec, VERSION_GCD_COMPRESSION);
78        String metaName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, metaExtension);
79        meta = state.directory.createOutput(metaName, state.context);
80        CodecUtil.writeHeader(meta, metaCodec, VERSION_GCD_COMPRESSION);
81        success = true;
82      } finally {
83        if (!success) {
84          IOUtils.closeWhileHandlingException(this);
85        }
86      }
87    }

89    @Override
90    public void addNumericField(FieldInfo field, Iterable<Number> values) throws IOException {
91      if (field.getDocValuesGen() != -1) {
92        throw new UnsupportedOperationException("4.2 does not support dv updates");
93      }
94      addNumericField(field, values, true);
95    }

97    void addNumericField(FieldInfo field, Iterable<Number> values, boolean optimizeStorage) throws IOException {
98      meta.writeVInt(field.number);
99      meta.writeByte(NUMBER);
100     meta.writeLong(data.getFilePointer());
101     long minValue = Long.MAX_VALUE;
102     long maxValue = Long.MIN_VALUE;
103     long gcd = 0;
104     // TODO: more efficient?
105     HashSet<Long> uniqueValues = null;
106     if (optimizeStorage) {
107       uniqueValues = new HashSet<>();
108 
109       long count = 0;
110       for (Number nv : values) {
111         // TODO: support this as MemoryDVFormat (and be smart about missing maybe)
112         final long v = nv == null ? 0 : nv.longValue();
113 
114         if (gcd != 1) {
115           if (v < Long.MIN_VALUE / 2 || v > Long.MAX_VALUE / 2) {
116             // in that case v - minValue might overflow and make the GCD computation return
117             // wrong results. Since these extreme values are unlikely, we just discard
118             // GCD computation for them
119             gcd = 1;
120           } else if (count != 0) { // minValue needs to be set first
121             gcd = MathUtil.gcd(gcd, v - minValue);
122           }
123         }
124 
125         minValue = Math.min(minValue, v);
126         maxValue = Math.max(maxValue, v);
127 
128         if (uniqueValues != null) {
129           if (uniqueValues.add(v)) {
130             if (uniqueValues.size() > 256) {
131               uniqueValues = null;
132             }
133           }
134         }
135 
136         ++count;
137       }
138       assert count == maxDoc;
139     }
140 
141     if (uniqueValues != null) {
142       // small number of unique values
143       final int bitsPerValue = PackedInts.bitsRequired(uniqueValues.size()-1);
144       FormatAndBits formatAndBits = PackedInts.fastestFormatAndBits(maxDoc, bitsPerValue, acceptableOverheadRatio);
145       if (formatAndBits.bitsPerValue == 8 && minValue >= Byte.MIN_VALUE && maxValue <= Byte.MAX_VALUE) {
146         meta.writeByte(UNCOMPRESSED); // uncompressed
147         for (Number nv : values) {
148           data.writeByte(nv == null ? 0 : (byte) nv.longValue());
149         }
150       } else {
151         meta.writeByte(TABLE_COMPRESSED); // table-compressed
152         Long[] decode = uniqueValues.toArray(new Long[uniqueValues.size()]);
153         final HashMap<Long,Integer> encode = new HashMap<>();
154         data.writeVInt(decode.length);
155         for (int i = 0; i < decode.length; i++) {
156           data.writeLong(decode[i]);
157           encode.put(decode[i], i);
158         }
159 
160         meta.writeVInt(PackedInts.VERSION_CURRENT);
161         data.writeVInt(formatAndBits.format.getId());
162         data.writeVInt(formatAndBits.bitsPerValue);
163 
164         final PackedInts.Writer writer = PackedInts.getWriterNoHeader(data, formatAndBits.format, maxDoc, formatAndBits.bitsPerValue, PackedInts.DEFAULT_BUFFER_SIZE);
165         for(Number nv : values) {
166           writer.add(encode.get(nv == null ? 0 : nv.longValue()));
167         }
168         writer.finish();
169       }
170     } else if (gcd != 0 && gcd != 1) {
171       meta.writeByte(GCD_COMPRESSED);
172       meta.writeVInt(PackedInts.VERSION_CURRENT);
173       data.writeLong(minValue);
174       data.writeLong(gcd);
175       data.writeVInt(BLOCK_SIZE);
176 
177       final BlockPackedWriter writer = new BlockPackedWriter(data, BLOCK_SIZE);
178       for (Number nv : values) {
179         long value = nv == null ? 0 : nv.longValue();
180         writer.add((value - minValue) / gcd);
181       }
182       writer.finish();
183     } else {
184       meta.writeByte(DELTA_COMPRESSED); // delta-compressed
185 
186       meta.writeVInt(PackedInts.VERSION_CURRENT);
187       data.writeVInt(BLOCK_SIZE);
188 
189       final BlockPackedWriter writer = new BlockPackedWriter(data, BLOCK_SIZE);
190       for (Number nv : values) {
191         writer.add(nv == null ? 0 : nv.longValue());
192       }
193       writer.finish();
194     }
195   }

197   private boolean closed;

199   @Override
200   public void close() throws IOException {
201     if (closed) {
202       return;
203     }
204     closed = true;
205     
206     boolean success = false;
207     try {
208       meta.writeVInt(-1); // write EOF marker
209       success = true;
210     } finally {
211       if (success) {
212         IOUtils.close(data, meta);
213       } else {
214         IOUtils.closeWhileHandlingException(data, meta);
215       }
216     }
217   }

219   @Override
220   public void addBinaryField(FieldInfo field, final Iterable<BytesRef> values) throws IOException {
221     if (field.getDocValuesGen() != -1) {
222       throw new UnsupportedOperationException("4.2 does not support dv updates");
223     }
224     // write the byte[] data
225     meta.writeVInt(field.number);
226     meta.writeByte(BYTES);
227     int minLength = Integer.MAX_VALUE;
228     int maxLength = Integer.MIN_VALUE;
229     final long startFP = data.getFilePointer();
230     for(BytesRef v : values) {
231       final int length = v == null ? 0 : v.length;
232       if (length > Lucene42DocValuesFormat.MAX_BINARY_FIELD_LENGTH) {
233         throw new IllegalArgumentException("DocValuesField \"" + field.name + "\" is too large, must be <= " + Lucene42DocValuesFormat.MAX_BINARY_FIELD_LENGTH);
234       }
235       minLength = Math.min(minLength, length);
236       maxLength = Math.max(maxLength, length);
237       if (v != null) {
238         data.writeBytes(v.bytes, v.offset, v.length);
239       }
240     }
241     meta.writeLong(startFP);
242     meta.writeLong(data.getFilePointer() - startFP);
243     meta.writeVInt(minLength);
244     meta.writeVInt(maxLength);
245     
246     // if minLength == maxLength, its a fixed-length byte[], we are done (the addresses are implicit)
247     // otherwise, we need to record the length fields...
248     if (minLength != maxLength) {
249       meta.writeVInt(PackedInts.VERSION_CURRENT);
250       meta.writeVInt(BLOCK_SIZE);
251 
252       final MonotonicBlockPackedWriter writer = new MonotonicBlockPackedWriter(data, BLOCK_SIZE);
253       long addr = 0;
254       for (BytesRef v : values) {
255         if (v != null) {
256           addr += v.length;
257         }
258         writer.add(addr);
259       }
260       writer.finish();
261     }
262   }

264   private void writeFST(FieldInfo field, Iterable<BytesRef> values) throws IOException {
265     meta.writeVInt(field.number);
266     meta.writeByte(FST);
267     meta.writeLong(data.getFilePointer());
268     PositiveIntOutputs outputs = PositiveIntOutputs.getSingleton();
269     Builder<Long> builder = new Builder<>(INPUT_TYPE.BYTE1, outputs);
270     IntsRefBuilder scratch = new IntsRefBuilder();
271     long ord = 0;
272     for (BytesRef v : values) {
273       builder.add(Util.toIntsRef(v, scratch), ord);
274       ord++;
275     }
276     FST<Long> fst = builder.finish();
277     if (fst != null) {
278       fst.save(data);
279     }
280     meta.writeVLong(ord);
281   }

283   @Override
284   public void addSortedField(FieldInfo field, Iterable<BytesRef> values, Iterable<Number> docToOrd) throws IOException {
285     if (field.getDocValuesGen() != -1) {
286       throw new UnsupportedOperationException("4.2 does not support dv updates");
287     }
288     // three cases for simulating the old writer:
289     // 1. no missing
290     // 2. missing (and empty string in use): remap ord=-1 -> ord=0
291     // 3. missing (and empty string not in use): remap all ords +1, insert empty string into values
292     boolean anyMissing = false;
293     for (Number n : docToOrd) {
294       if (n.longValue() == -1) {
295         anyMissing = true;
296         break;
297       }
298     }
299     
300     boolean hasEmptyString = false;
301     for (BytesRef b : values) {
302       hasEmptyString = b.length == 0;
303       break;
304     }
305     
306     if (!anyMissing) {
307       // nothing to do
308     } else if (hasEmptyString) {
309       docToOrd = MissingOrdRemapper.mapMissingToOrd0(docToOrd);
310     } else {
311       docToOrd = MissingOrdRemapper.mapAllOrds(docToOrd);
312       values = MissingOrdRemapper.insertEmptyValue(values);
313     }
314     
315     // write the ordinals as numerics
316     addNumericField(field, docToOrd, false);
317     
318     // write the values as FST
319     writeFST(field, values);
320   }

323   @Override
324   public void addSortedSetField(FieldInfo field, Iterable<BytesRef> values, final Iterable<Number> docToOrdCount, final Iterable<Number> ords) throws IOException {
325     assert field.getDocValuesGen() == -1;
326     // write the ordinals as a binary field
327     addBinaryField(field, new Iterable<BytesRef>() {
328       @Override
329       public Iterator<BytesRef> iterator() {
330         return new SortedSetIterator(docToOrdCount.iterator(), ords.iterator());
331       }
332     });
333       
334     // write the values as FST
335     writeFST(field, values);
336   }

338   // per-document vint-encoded byte[]
339   static class SortedSetIterator implements Iterator<BytesRef> {
340     byte[] buffer = new byte[10];
341     ByteArrayDataOutput out = new ByteArrayDataOutput();
342     BytesRef ref = new BytesRef();
343     
344     final Iterator<Number> counts;
345     final Iterator<Number> ords;
346     
347     SortedSetIterator(Iterator<Number> counts, Iterator<Number> ords) {
348       this.counts = counts;
349       this.ords = ords;
350     }
351     
352     @Override
353     public boolean hasNext() {
354       return counts.hasNext();
355     }
356 
357     @Override
358     public BytesRef next() {
359       if (!hasNext()) {
360         throw new NoSuchElementException();
361       }
362       
363       int count = counts.next().intValue();
364       int maxSize = count*9; // worst case
365       if (maxSize > buffer.length) {
366         buffer = ArrayUtil.grow(buffer, maxSize);
367       }
368       
369       try {
370         encodeValues(count);
371       } catch (IOException bogus) {
372         throw new RuntimeException(bogus);
373       }
374       
375       ref.bytes = buffer;
376       ref.offset = 0;
377       ref.length = out.getPosition();
378 
379       return ref;
380     }
381     
382     // encodes count values to buffer
383     private void encodeValues(int count) throws IOException {
384       out.reset(buffer);
385       long lastOrd = 0;
386       for (int i = 0; i < count; i++) {
387         long ord = ords.next().longValue();
388         out.writeVLong(ord - lastOrd);
389         lastOrd = ord;
390       }
391     }
392 
393     @Override
394     public void remove() {
395       throw new UnsupportedOperationException();
396     }
397   }
