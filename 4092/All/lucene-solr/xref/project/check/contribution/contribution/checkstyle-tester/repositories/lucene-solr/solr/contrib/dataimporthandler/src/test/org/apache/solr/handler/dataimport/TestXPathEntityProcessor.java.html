238   @Test
239   public void withFieldsAndXpathStream() throws Exception {
240     final Object monitor = new Object();
241     final boolean[] done = new boolean[1];
242     
243     Map entityAttrs = createMap("name", "e", "url", "cd.xml",
244         XPathEntityProcessor.FOR_EACH, "/catalog/cd", "stream", "true", "batchSize","1");
245     List fields = new ArrayList();
246     fields.add(createMap("column", "title", "xpath", "/catalog/cd/title"));
247     fields.add(createMap("column", "artist", "xpath", "/catalog/cd/artist"));
248     fields.add(createMap("column", "year", "xpath", "/catalog/cd/year"));
249     Context c = getContext(null,
250         new VariableResolver(), getDataSource(cdData), Context.FULL_DUMP, fields, entityAttrs);
251     XPathEntityProcessor xPathEntityProcessor = new XPathEntityProcessor() {
252       private int count;
253       
254       @Override
255       protected Map<String, Object> readRow(Map<String, Object> record,
256           String xpath) {
257         synchronized (monitor) {
258           if (simulateSlowReader && !done[0]) {
259             try {
260               monitor.wait(100);
261             } catch (InterruptedException e) {
262               throw new RuntimeException(e);
263             }
264           }
265         }
266         
267         return super.readRow(record, xpath);
268       }
269     };
270     
271     if (simulateSlowResultProcessor) {
272       xPathEntityProcessor.blockingQueueSize = 1;
273     }
274     xPathEntityProcessor.blockingQueueTimeOut = 1;
275     xPathEntityProcessor.blockingQueueTimeOutUnits = TimeUnit.MICROSECONDS;
276     
277     xPathEntityProcessor.init(c);
278     List<Map<String, Object>> result = new ArrayList<>();
279     while (true) {
280       if (rowsToRead >= 0 && result.size() >= rowsToRead) {
281         Thread.currentThread().interrupt();
282       }
283       Map<String, Object> row = xPathEntityProcessor.nextRow();
284       if (row == null)
285         break;
286       result.add(row);
287       if (simulateSlowResultProcessor) {
288         synchronized (xPathEntityProcessor.publisherThread) {
289           if (xPathEntityProcessor.publisherThread.isAlive()) {
290             xPathEntityProcessor.publisherThread.wait(1000);
291           }
292         }
293       }
294     }
295     
296     synchronized (monitor) {
297       done[0] = true;
298       monitor.notify();
299     }
300     
301     // confirm that publisher thread stops.
302     xPathEntityProcessor.publisherThread.join(1000);
303     assertEquals("Expected thread to stop", false, xPathEntityProcessor.publisherThread.isAlive());
304     
305     assertEquals(rowsToRead < 0 ? 3 : rowsToRead, result.size());
306     
307     if (rowsToRead < 0) {
308       assertEquals("Empire Burlesque", result.get(0).get("title"));
309       assertEquals("Bonnie Tyler", result.get(1).get("artist"));
310       assertEquals("1982", result.get(2).get("year"));
311     }
312   }
