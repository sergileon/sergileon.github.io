37    /**
38     * Some explains methods calculate their values though a slightly
39     * different  order of operations from the actual scoring method ...
40     * this allows for a small amount of relative variation
41     */
42    public static float EXPLAIN_SCORE_TOLERANCE_DELTA = 0.001f;
43    
44    /**
45     * In general we use a relative epsilon, but some tests do crazy things
46     * like boost documents with 0, creating tiny tiny scores where the
47     * relative difference is large but the absolute difference is tiny.
48     * we ensure the the epsilon is always at least this big.
49     */
50    public static float EXPLAIN_SCORE_TOLERANCE_MINIMUM = 1e-6f;

52    /**
53     * Tests that all documents up to maxDoc which are *not* in the
54     * expected result set, have an explanation which indicates that 
55     * the document does not match
56     */
57    public static void checkNoMatchExplanations(Query q, String defaultFieldName,
58                                                IndexSearcher searcher, int[] results)
59      throws IOException {
60  
61      String d = q.toString(defaultFieldName);
62      Set<Integer> ignore = new TreeSet<>();
63      for (int i = 0; i < results.length; i++) {
64        ignore.add(Integer.valueOf(results[i]));
65      }
66      
67      int maxDoc = searcher.getIndexReader().maxDoc();
68      for (int doc = 0; doc < maxDoc; doc++) {
69        if (ignore.contains(Integer.valueOf(doc))) continue;
70  
71        Explanation exp = searcher.explain(q, doc);
72        Assert.assertNotNull("Explanation of [["+d+"]] for #"+doc+" is null",
73                               exp);
74        Assert.assertFalse("Explanation of [["+d+"]] for #"+doc+
75                           " doesn't indicate non-match: " + exp.toString(),
76                           exp.isMatch());
77      }
78      
79    }

81    /**
82     * Tests that a query matches the an expected set of documents using a
83     * HitCollector.
84     *
85     * <p>
86     * Note that when using the HitCollector API, documents will be collected
87     * if they "match" regardless of what their score is.
88     * </p>
89     * @param query the query to test
90     * @param searcher the searcher to test the query against
91     * @param defaultFieldName used for displaying the query in assertion messages
92     * @param results a list of documentIds that must match the query
93     * @see #checkHits
94     */
95    public static void checkHitCollector(Random random, Query query, String defaultFieldName,
96                                         IndexSearcher searcher, int[] results)
97      throws IOException {
98  
99      QueryUtils.check(random,query,searcher);
100     
101     Set<Integer> correct = new TreeSet<>();
102     for (int i = 0; i < results.length; i++) {
103       correct.add(Integer.valueOf(results[i]));
104     }
105     final Set<Integer> actual = new TreeSet<>();
106     final Collector c = new SetCollector(actual);
107 
108     searcher.search(query, c);
109     Assert.assertEquals("Simple: " + query.toString(defaultFieldName), 
110                         correct, actual);
111 
112     for (int i = -1; i < 2; i++) {
113       actual.clear();
114       IndexSearcher s = QueryUtils.wrapUnderlyingReader
115         (random, searcher, i);
116       s.search(query, c);
117       Assert.assertEquals("Wrap Reader " + i + ": " +
118                           query.toString(defaultFieldName),
119                           correct, actual);
120     }
121   }

123   /**
124    * Just collects document ids into a set.
125    */
126   public static class SetCollector extends SimpleCollector {
127     final Set<Integer> bag;
128     public SetCollector(Set<Integer> bag) {
129       this.bag = bag;
130     }
131     private int base = 0;
132     @Override
133     public void setScorer(Scorer scorer) throws IOException {}
134     @Override
135     public void collect(int doc) {
136       bag.add(Integer.valueOf(doc + base));
137     }
138     @Override
139     protected void doSetNextReader(LeafReaderContext context) throws IOException {
140       base = context.docBase;
141     }
142

144     public boolean needsScores() {
145       return false;
146     }

149   /**
150    * Tests that a query matches the an expected set of documents using Hits.
151    *
152    * <p>
153    * Note that when using the Hits API, documents will only be returned
154    * if they have a positive normalized score.
155    * </p>
156    * @param query the query to test
157    * @param searcher the searcher to test the query against
158    * @param defaultFieldName used for displaing the query in assertion messages
159    * @param results a list of documentIds that must match the query
160    * @see #checkHitCollector
161    */
162   public static void checkHits(
163         Random random,
164         Query query,
165         String defaultFieldName,
166         IndexSearcher searcher,
167         int[] results)
168           throws IOException {
169 
170     ScoreDoc[] hits = searcher.search(query, 1000).scoreDocs;
171 
172     Set<Integer> correct = new TreeSet<>();
173     for (int i = 0; i < results.length; i++) {
174       correct.add(Integer.valueOf(results[i]));
175     }
176 
177     Set<Integer> actual = new TreeSet<>();
178     for (int i = 0; i < hits.length; i++) {
179       actual.add(Integer.valueOf(hits[i].doc));
180     }
181 
182     Assert.assertEquals(query.toString(defaultFieldName), correct, actual);
183 
184     QueryUtils.check(random, query,searcher, LuceneTestCase.rarely(random));
185   }

187   /** Tests that a Hits has an expected order of documents */
188   public static void checkDocIds(String mes, int[] results, ScoreDoc[] hits) {
189     Assert.assertEquals(mes + " nr of hits", hits.length, results.length);
190     for (int i = 0; i < results.length; i++) {
191       Assert.assertEquals(mes + " doc nrs for hit " + i, results[i], hits[i].doc);
192     }

196    * and that the two queries have the same score values.

198   public static void checkHitsQuery(
199         Query query,
200         ScoreDoc[] hits1,
201         ScoreDoc[] hits2,
202         int[] results) {
203 
204     checkDocIds("hits1", results, hits1);
205     checkDocIds("hits2", results, hits2);
206     checkEqual(query, hits1, hits2);

209   public static void checkEqual(Query query, ScoreDoc[] hits1, ScoreDoc[] hits2) {
210      final float scoreTolerance = 1.0e-6f;
211      if (hits1.length != hits2.length) {
212        Assert.fail("Unequal lengths: hits1="+hits1.length+",hits2="+hits2.length);
213      }
214     for (int i = 0; i < hits1.length; i++) {
215       if (hits1[i].doc != hits2[i].doc) {
216         Assert.fail("Hit " + i + " docnumbers don't match\n"
217                 + hits2str(hits1, hits2,0,0)
218                 + "for query:" + query.toString());
219       }
220 
221       if ((hits1[i].doc != hits2[i].doc)
222           || Math.abs(hits1[i].score -  hits2[i].score) > scoreTolerance)
223       {
224         Assert.fail("Hit " + i + ", doc nrs " + hits1[i].doc + " and " + hits2[i].doc
225                       + "\nunequal       : " + hits1[i].score
226                       + "\n           and: " + hits2[i].score
227                       + "\nfor query:" + query.toString());
228       }
229     }
230   }

232   public static String hits2str(ScoreDoc[] hits1, ScoreDoc[] hits2, int start, int end) {
233     StringBuilder sb = new StringBuilder();
234     int len1=hits1==null ? 0 : hits1.length;
235     int len2=hits2==null ? 0 : hits2.length;
236     if (end<=0) {
237       end = Math.max(len1,len2);
238     }
239 
240       sb.append("Hits length1=").append(len1).append("\tlength2=").append(len2);
241 
242     sb.append('\n');
243     for (int i=start; i<end; i++) {
244         sb.append("hit=").append(i).append(':');
245       if (i<len1) {
246           sb.append(" doc").append(hits1[i].doc).append('=').append(hits1[i].score);
247       } else {
248         sb.append("               ");
249       }
250       sb.append(",\t");
251       if (i<len2) {
252         sb.append(" doc").append(hits2[i].doc).append('=').append(hits2[i].score);
253       }
254       sb.append('\n');
255     }
256     return sb.toString();
257   }

260   public static String topdocsString(TopDocs docs, int start, int end) {
261     StringBuilder sb = new StringBuilder();
262       sb.append("TopDocs totalHits=").append(docs.totalHits).append(" top=").append(docs.scoreDocs.length).append('\n');
263     if (end<=0) end=docs.scoreDocs.length;
264     else end=Math.min(end,docs.scoreDocs.length);
265     for (int i=start; i<end; i++) {
266       sb.append('\t');
267       sb.append(i);
268       sb.append(") doc=");
269       sb.append(docs.scoreDocs[i].doc);
270       sb.append("\tscore=");
271       sb.append(docs.scoreDocs[i].score);
272       sb.append('\n');
273     }
274     return sb.toString();
275   }

317   /** returns a reasonable epsilon for comparing two floats,
318    *  where minor differences are acceptable such as score vs. explain */
319   public static float explainToleranceDelta(float f1, float f2) {
320     return Math.max(EXPLAIN_SCORE_TOLERANCE_MINIMUM, Math.max(Math.abs(f1), Math.abs(f2)) * EXPLAIN_SCORE_TOLERANCE_DELTA);
321   }

323   /** 
324    * Assert that an explanation has the expected score, and optionally that its
325    * sub-details max/sum/factor match to that score.
326    *
327    * @param q String representation of the query for assertion messages
328    * @param doc Document ID for assertion messages
329    * @param score Real score value of doc with query q
330    * @param deep indicates whether a deep comparison of sub-Explanation details should be executed
331    * @param expl The Explanation to match against score
332    */
333   public static void verifyExplanation(String q, 
334                                        int doc, 
335                                        float score,
336                                        boolean deep,
337                                        Explanation expl) {
338     float value = expl.getValue();
339     Assert.assertEquals(q+": score(doc="+doc+")="+score+
340         " != explanationScore="+value+" Explanation: "+expl,
341         score,value,explainToleranceDelta(score, value));
342 
343     if (!deep) return;
344 
345     Explanation detail[] = expl.getDetails();
346     // TODO: can we improve this entire method? it's really geared to work only with TF/IDF
347     if (expl.getDescription().endsWith("computed from:")) {
348       return; // something more complicated.
349     }
350     if (detail.length > 0) {
351       if (detail.length==1) {
352         // simple containment, unless it's a freq of: (which lets a query explain how the freq is calculated), 
353         // just verify contained expl has same score
354         if (!expl.getDescription().endsWith("with freq of:"))
355           verifyExplanation(q,doc,score,deep,detail[0]);
356       } else {
357         // explanation must either:
358         // - end with one of: "product of:", "sum of:", "max of:", or
359         // - have "max plus <x> times others" (where <x> is float).
360         float x = 0;
361         String descr = expl.getDescription().toLowerCase(Locale.ROOT);
362         boolean productOf = descr.endsWith("product of:");
363         boolean sumOf = descr.endsWith("sum of:");
364         boolean maxOf = descr.endsWith("max of:");
365         boolean maxTimesOthers = false;
366         if (!(productOf || sumOf || maxOf)) {
367           // maybe 'max plus x times others'
368           int k1 = descr.indexOf("max plus ");
369           if (k1>=0) {
370             k1 += "max plus ".length();
371             int k2 = descr.indexOf(" ",k1);
372             try {
373               x = Float.parseFloat(descr.substring(k1,k2).trim());
374               if (descr.substring(k2).trim().equals("times others of:")) {
375                 maxTimesOthers = true;
376               }
377             } catch (NumberFormatException e) {
378             }
379           }
380         }
381         // TODO: this is a TERRIBLE assertion!!!!
382         Assert.assertTrue(
383             q+": multi valued explanation description=\""+descr
384             +"\" must be 'max of plus x times others' or end with 'product of'"
385             +" or 'sum of:' or 'max of:' - "+expl,
386             productOf || sumOf || maxOf || maxTimesOthers);
387         float sum = 0;
388         float product = 1;
389         float max = 0;
390         for (int i=0; i<detail.length; i++) {
391           float dval = detail[i].getValue();
392           verifyExplanation(q,doc,dval,deep,detail[i]);
393           product *= dval;
394           sum += dval;
395           max = Math.max(max,dval);
396         }
397         float combined = 0;
398         if (productOf) {
399           combined = product;
400         } else if (sumOf) {
401           combined = sum;
402         } else if (maxOf) {
403           combined = max;
404         } else if (maxTimesOthers) {
405           combined = max + x * (sum - max);
406         } else {
407             Assert.assertTrue("should never get here!",false);
408         }
409         Assert.assertEquals(q+": actual subDetails combined=="+combined+
410             " != value="+value+" Explanation: "+expl,
411             combined,value,explainToleranceDelta(combined, value));
412       }
413     }
414   }

416   /**
417    * an IndexSearcher that implicitly checks hte explanation of every match
418    * whenever it executes a search.
419    *
420    * @see ExplanationAsserter
421    */
422   public static class ExplanationAssertingSearcher extends IndexSearcher {
423     public ExplanationAssertingSearcher(IndexReader r) {
424       super(r);
425     }
426     protected void checkExplanations(Query q) throws IOException {
427       super.search(q,
428                    new ExplanationAsserter
429                    (q, null, this));
430     }
431     @Override
432     public TopFieldDocs search(Query query,
433                                int n,
434                                Sort sort) throws IOException {
435       
436       checkExplanations(query);
437       return super.search(query,n,sort);
438     }
439     @Override
440     public void search(Query query, Collector results) throws IOException {
441       checkExplanations(query);
442       super.search(query, results);
443     }
444     @Override
445     public TopDocs search(Query query, int n) throws IOException {
446 
447       checkExplanations(query);
448       return super.search(query, n);
449     }
450   }

452   /**
453    * Asserts that the score explanation for every document matching a
454    * query corresponds with the true score.
455    *
456    * NOTE: this HitCollector should only be used with the Query and Searcher
457    * specified at when it is constructed.
458    *
459    * @see CheckHits#verifyExplanation
460    */
461   public static class ExplanationAsserter extends SimpleCollector {
462 
463     Query q;
464     IndexSearcher s;
465     String d;
466     boolean deep;
467     
468     Scorer scorer;
469     private int base = 0;
470 
471     /** Constructs an instance which does shallow tests on the Explanation */
472     public ExplanationAsserter(Query q, String defaultFieldName, IndexSearcher s) {
473       this(q,defaultFieldName,s,false);
474     }      
475     public ExplanationAsserter(Query q, String defaultFieldName, IndexSearcher s, boolean deep) {
476       this.q=q;
477       this.s=s;
478       this.d = q.toString(defaultFieldName);
479       this.deep=deep;
480     }      
481     
482     @Override
483     public void setScorer(Scorer scorer) throws IOException {
484       this.scorer = scorer;     
485     }
486     
487     @Override
488     public void collect(int doc) throws IOException {
489       Explanation exp = null;
490       doc = doc + base;
491       try {
492         exp = s.explain(q, doc);
493       } catch (IOException e) {
494         throw new RuntimeException
495           ("exception in hitcollector of [["+d+"]] for #"+doc, e);
496       }
497       
498       Assert.assertNotNull("Explanation of [["+d+"]] for #"+doc+" is null", exp);
499       verifyExplanation(d,doc,scorer.score(),deep,exp);
500       Assert.assertTrue("Explanation of [["+d+"]] for #"+ doc + 
501                         " does not indicate match: " + exp.toString(), 
502                         exp.isMatch());
503     }
504     @Override
505     protected void doSetNextReader(LeafReaderContext context) throws IOException {
506       base = context.docBase;
507     }
508     
509     @Override
510     public boolean needsScores() {
511       return true;
512     }
513   }
