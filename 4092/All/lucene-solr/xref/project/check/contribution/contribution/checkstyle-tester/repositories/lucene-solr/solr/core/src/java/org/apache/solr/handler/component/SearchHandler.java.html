100   /**
101    * Initialize the components based on name.  Note, if using <code>INIT_FIRST_COMPONENTS</code> or <code>INIT_LAST_COMPONENTS</code>,
102    * then the {@link DebugComponent} will always occur last.  If this is not desired, then one must explicitly declare all components using
103    * the <code>INIT_COMPONENTS</code> syntax.
104    */
105   @Override
106   @SuppressWarnings("unchecked")
107   public void inform(SolrCore core)
108   {
109     this.core = core;
110     Set<String> missing = new HashSet<>();
111     List<String> c = (List<String>) initArgs.get(INIT_COMPONENTS);
112     missing.addAll(core.getSearchComponents().checkContains(c));
113     List<String> first = (List<String>) initArgs.get(INIT_FIRST_COMPONENTS);
114     missing.addAll(core.getSearchComponents().checkContains(first));
115     List<String> last = (List<String>) initArgs.get(INIT_LAST_COMPONENTS);
116     missing.addAll(core.getSearchComponents().checkContains(last));
117     if (!missing.isEmpty()) throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
118         "Missing SearchComponents named : " + missing);
119     if (c != null && (first != null || last != null)) throw new SolrException(SolrException.ErrorCode.SERVER_ERROR,
120         "First/Last components only valid if you do not declare 'components'");
121 
122     if (shfInfo == null) {
123       shardHandlerFactory = core.getCoreDescriptor().getCoreContainer().getShardHandlerFactory();
124     } else {
125       shardHandlerFactory = core.createInitInstance(shfInfo, ShardHandlerFactory.class, null, null);
126       core.addCloseHook(new CloseHook() {
127         @Override
128         public void preClose(SolrCore core) {
129           shardHandlerFactory.close();
130         }
131 
132         @Override
133         public void postClose(SolrCore core) {
134         }
135       });
136     }
137

140   private void initComponents() {
141     Object declaredComponents = initArgs.get(INIT_COMPONENTS);
142     List<String> first = (List<String>) initArgs.get(INIT_FIRST_COMPONENTS);
143     List<String> last  = (List<String>) initArgs.get(INIT_LAST_COMPONENTS);
144 
145     List<String> list = null;
146     boolean makeDebugLast = true;
147     if( declaredComponents == null ) {
148       // Use the default component list
149       list = getDefaultComponents();
150 
151       if( first != null ) {
152         List<String> clist = first;
153         clist.addAll( list );
154         list = clist;
155       }
156 
157       if( last != null ) {
158         list.addAll( last );
159       }
160     }
161     else {
162       list = (List<String>)declaredComponents;
163       if( first != null || last != null ) {
164         throw new SolrException( SolrException.ErrorCode.SERVER_ERROR,
165             "First/Last components only valid if you do not declare 'components'");
166       }
167       makeDebugLast = false;
168     }
169 
170     // Build the component list
171     List<SearchComponent> components = new ArrayList<>(list.size());
172     DebugComponent dbgCmp = null;
173     for(String c : list){
174       SearchComponent comp = core.getSearchComponent( c );
175       if (comp instanceof DebugComponent && makeDebugLast == true){
176         dbgCmp = (DebugComponent) comp;
177       } else {
178         components.add(comp);
179         log.debug("Adding  component:"+comp);
180       }
181     }
182     if (makeDebugLast == true && dbgCmp != null){
183       components.add(dbgCmp);
184       log.debug("Adding  debug component:" + dbgCmp);
185     }
186     this.components = components;
187   }
188 
189   public List<SearchComponent> getComponents() {
190     List<SearchComponent> result = components;  // volatile read
191     if (result == null) {
192       synchronized (this) {
193         if (components == null) {
194           initComponents();
195         }
196         result = components;
197       }
198     }
199     return result;
200   }

202   private ShardHandler getAndPrepShardHandler(SolrQueryRequest req, ResponseBuilder rb) {
203     ShardHandler shardHandler = null;
204 
205     rb.isDistrib = req.getParams().getBool("distrib", req.getCore().getCoreDescriptor()
206         .getCoreContainer().isZooKeeperAware());
207     if (!rb.isDistrib) {
208       // for back compat, a shards param with URLs like localhost:8983/solr will mean that this
209       // search is distributed.
210       final String shards = req.getParams().get(ShardParams.SHARDS);
211       rb.isDistrib = ((shards != null) && (shards.indexOf('/') > 0));
212     }
213     
214     if (rb.isDistrib) {
215       shardHandler = shardHandlerFactory.getShardHandler();
216       shardHandler.prepDistributed(rb);
217       if (!rb.isDistrib) {
218         shardHandler = null; // request is not distributed after all and so the shard handler is not needed
219       }
220     }
221 
222     return shardHandler;
223   }

225   @Override
226   public void handleRequestBody(SolrQueryRequest req, SolrQueryResponse rsp) throws Exception
227   {
228     List<SearchComponent> components  = getComponents();
229     ResponseBuilder rb = new ResponseBuilder(req, rsp, components);
230     if (rb.requestInfo != null) {
231       rb.requestInfo.setResponseBuilder(rb);
232     }
233 
234     boolean dbg = req.getParams().getBool(CommonParams.DEBUG_QUERY, false);
235     rb.setDebug(dbg);
236     if (dbg == false){//if it's true, we are doing everything anyway.
237       SolrPluginUtils.getDebugInterests(req.getParams().getParams(CommonParams.DEBUG), rb);
238     }
239 
240     final RTimerTree timer = rb.isDebug() ? req.getRequestTimer() : null;
241 
242     final ShardHandler shardHandler1 = getAndPrepShardHandler(req, rb); // creates a ShardHandler object only if it's needed
243     
244     if (timer == null) {
245       // non-debugging prepare phase
246       for( SearchComponent c : components ) {
247         c.prepare(rb);
248       }
249     } else {
250       // debugging prepare phase
251       RTimerTree subt = timer.sub( "prepare" );
252       for( SearchComponent c : components ) {
253         rb.setTimer( subt.sub( c.getName() ) );
254         c.prepare(rb);
255         rb.getTimer().stop();
256       }
257       subt.stop();
258     }
259 
260     if (!rb.isDistrib) {
261       // a normal non-distributed request
262 
263       long timeAllowed = req.getParams().getLong(CommonParams.TIME_ALLOWED, -1L);
264       if (timeAllowed > 0L) {
265         SolrQueryTimeoutImpl.set(timeAllowed);
266       }
267       try {
268         // The semantics of debugging vs not debugging are different enough that
269         // it makes sense to have two control loops
270         if(!rb.isDebug()) {
271           // Process
272           for( SearchComponent c : components ) {
273             c.process(rb);
274           }
275         }
276         else {
277           // Process
278           RTimerTree subt = timer.sub( "process" );
279           for( SearchComponent c : components ) {
280             rb.setTimer( subt.sub( c.getName() ) );
281             c.process(rb);
282             rb.getTimer().stop();
283           }
284           subt.stop();
285 
286           // add the timing info
287           if (rb.isDebugTimings()) {
288             rb.addDebugInfo("timing", timer.asNamedList() );
289           }
290         }
291       } catch (ExitableDirectoryReader.ExitingReaderException ex) {
292         log.warn( "Query: " + req.getParamString() + "; " + ex.getMessage());
293         SolrDocumentList r = (SolrDocumentList) rb.rsp.getValues().get("response");
294         if(r == null)
295           r = new SolrDocumentList();
296         r.setNumFound(0);
297         rb.rsp.add("response", r);
298         if(rb.isDebug()) {
299           NamedList debug = new NamedList();
300           debug.add("explain", new NamedList());
301           rb.rsp.add("debug", debug);
302         }
303         rb.rsp.getResponseHeader().add("partialResults", Boolean.TRUE);
304       } finally {
305         SolrQueryTimeoutImpl.reset();
306       }
307     } else {
308       // a distributed request
309 
310       if (rb.outgoing == null) {
311         rb.outgoing = new LinkedList<>();
312       }
313       rb.finished = new ArrayList<>();
314 
315       int nextStage = 0;
316       do {
317         rb.stage = nextStage;
318         nextStage = ResponseBuilder.STAGE_DONE;
319 
320         // call all components
321         for( SearchComponent c : components ) {
322           // the next stage is the minimum of what all components report
323           nextStage = Math.min(nextStage, c.distributedProcess(rb));
324         }
325 
326 
327         // check the outgoing queue and send requests
328         while (rb.outgoing.size() > 0) {
329 
330           // submit all current request tasks at once
331           while (rb.outgoing.size() > 0) {
332             ShardRequest sreq = rb.outgoing.remove(0);
333             sreq.actualShards = sreq.shards;
334             if (sreq.actualShards==ShardRequest.ALL_SHARDS) {
335               sreq.actualShards = rb.shards;
336             }
337             sreq.responses = new ArrayList<>(sreq.actualShards.length); // presume we'll get a response from each shard we send to
338 
339             // TODO: map from shard to address[]
340             for (String shard : sreq.actualShards) {
341               ModifiableSolrParams params = new ModifiableSolrParams(sreq.params);
342               params.remove(ShardParams.SHARDS);      // not a top-level request
343               params.set(CommonParams.DISTRIB, "false");               // not a top-level request
344               params.remove("indent");
345               params.remove(CommonParams.HEADER_ECHO_PARAMS);
346               params.set(ShardParams.IS_SHARD, true);  // a sub (shard) request
347               params.set(ShardParams.SHARDS_PURPOSE, sreq.purpose);
348               params.set(ShardParams.SHARD_URL, shard); // so the shard knows what was asked
349               if (rb.requestInfo != null) {
350                 // we could try and detect when this is needed, but it could be tricky
351                 params.set("NOW", Long.toString(rb.requestInfo.getNOW().getTime()));
352               }
353               String shardQt = params.get(ShardParams.SHARDS_QT);
354               if (shardQt != null) {
355                 params.set(CommonParams.QT, shardQt);
356               } else {
357                 // for distributed queries that don't include shards.qt, use the original path
358                 // as the default but operators need to update their luceneMatchVersion to enable
359                 // this behavior since it did not work this way prior to 5.1
360                 if (req.getCore().getSolrConfig().luceneMatchVersion.onOrAfter(Version.LUCENE_5_1_0)) {
361                   String reqPath = (String) req.getContext().get(PATH);
362                   if (!"/select".equals(reqPath)) {
363                     params.set(CommonParams.QT, reqPath);
364                   } // else if path is /select, then the qt gets passed thru if set
365                 } else {
366                   // this is the pre-5.1 behavior, which translates to sending the shard request to /select
367                   params.remove(CommonParams.QT);
368                 }
369               }
370               shardHandler1.submit(sreq, shard, params, rb.preferredHostAddress);
371             }
372           }
373 
374 
375           // now wait for replies, but if anyone puts more requests on
376           // the outgoing queue, send them out immediately (by exiting
377           // this loop)
378           boolean tolerant = rb.req.getParams().getBool(ShardParams.SHARDS_TOLERANT, false);
379           while (rb.outgoing.size() == 0) {
380             ShardResponse srsp = tolerant ? 
381                 shardHandler1.takeCompletedIncludingErrors():
382                 shardHandler1.takeCompletedOrError();
383             if (srsp == null) break;  // no more requests to wait for
384 
385             // Was there an exception?  
386             if (srsp.getException() != null) {
387               // If things are not tolerant, abort everything and rethrow
388               if(!tolerant) {
389                 shardHandler1.cancelAll();
390                 if (srsp.getException() instanceof SolrException) {
391                   throw (SolrException)srsp.getException();
392                 } else {
393                   throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, srsp.getException());
394                 }
395               } else {
396                 if(rsp.getResponseHeader().get("partialResults") == null) {
397                   rsp.getResponseHeader().add("partialResults", Boolean.TRUE);
398                 }
399               }
400             }
401 
402             rb.finished.add(srsp.getShardRequest());
403 
404             // let the components see the responses to the request
405             for(SearchComponent c : components) {
406               c.handleResponses(rb, srsp.getShardRequest());
407             }
408           }
409         }
410 
411         for(SearchComponent c : components) {
412           c.finishStage(rb);
413         }
414 
415         // we are done when the next stage is MAX_VALUE
416       } while (nextStage != Integer.MAX_VALUE);
417     }
418     
419     // SOLR-5550: still provide shards.info if requested even for a short circuited distrib request
420     if(!rb.isDistrib && req.getParams().getBool(ShardParams.SHARDS_INFO, false) && rb.shortCircuitedURL != null) {  
421       NamedList<Object> shardInfo = new SimpleOrderedMap<Object>();
422       SimpleOrderedMap<Object> nl = new SimpleOrderedMap<Object>();        
423       if (rsp.getException() != null) {
424         Throwable cause = rsp.getException();
425         if (cause instanceof SolrServerException) {
426           cause = ((SolrServerException)cause).getRootCause();
427         } else {
428           if (cause.getCause() != null) {
429             cause = cause.getCause();
430           }          
431         }
432         nl.add("error", cause.toString() );
433         StringWriter trace = new StringWriter();
434         cause.printStackTrace(new PrintWriter(trace));
435         nl.add("trace", trace.toString() );
436       }
437       else {
438         nl.add("numFound", rb.getResults().docList.matches());
439         nl.add("maxScore", rb.getResults().docList.maxScore());
440       }
441       nl.add("shardAddress", rb.shortCircuitedURL);
442       nl.add("time", req.getRequestTimer().getTime()); // elapsed time of this request so far
443       
444       int pos = rb.shortCircuitedURL.indexOf("://");        
445       String shardInfoName = pos != -1 ? rb.shortCircuitedURL.substring(pos+3) : rb.shortCircuitedURL;
446       shardInfo.add(shardInfoName, nl);   
447       rsp.getValues().add(ShardParams.SHARDS_INFO,shardInfo);            
448     }
449   }

453   @Override
454   public String getDescription() {
455     StringBuilder sb = new StringBuilder();
456     sb.append("Search using components: ");
457     if( components != null ) {
458       for(SearchComponent c : components){
459         sb.append(c.getName());
460         sb.append(",");
461       }
462     }
463     return sb.toString();
464   }
