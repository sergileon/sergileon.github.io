77    /** verify expected error msgs from bad client behavior */
78    public void testBadInputs() throws Exception {
79      // sometimes seed some data, othertimes use an empty index
80      if (random().nextBoolean()) {
81        assertU(adoc("id", "42", "str", "z", "float", "99.99", "int", "42"));
82        assertU(adoc("id", "66", "str", "x", "float", "22.00", "int", "-66"));
83      } else {
84        assertU(commit());
85      }
86        assertU(commit());
87  
88      // empty, blank, or bogus cursor
89      for (String c : new String[] { "", "   ", "all the docs please!"}) {
90        assertFail(params("q", "*:*", 
91                          "sort", "id desc", 
92                          CURSOR_MARK_PARAM, c),
93                   ErrorCode.BAD_REQUEST, "Unable to parse");
94      }
95  
96      // no id in sort
97      assertFail(params("q", "*:*", 
98                        "sort", "score desc", 
99                        CURSOR_MARK_PARAM, CURSOR_MARK_START),
100                ErrorCode.BAD_REQUEST, "uniqueKey field");
101     // _docid_
102     assertFail(params("q", "*:*", 
103                       "sort", "_docid_ asc, id desc", 
104                       CURSOR_MARK_PARAM, CURSOR_MARK_START),
105                ErrorCode.BAD_REQUEST, "_docid_");
106 
107     // using cursor w/ timeAllowed
108     assertFail(params("q", "*:*", 
109                       "sort", "id desc", 
110                       CommonParams.TIME_ALLOWED, "1000",
111                       CURSOR_MARK_PARAM, CURSOR_MARK_START),
112                ErrorCode.BAD_REQUEST, CommonParams.TIME_ALLOWED);
113 
114     // using cursor w/ grouping
115     assertFail(params("q", "*:*", 
116                       "sort", "id desc", 
117                       GroupParams.GROUP, "true",
118                       GroupParams.GROUP_FIELD, "str",
119                       CURSOR_MARK_PARAM, CURSOR_MARK_START),
120                ErrorCode.BAD_REQUEST, "Grouping");

503    * test that our assumptions about how caches are affected hold true
504    */
505   public void testCacheImpacts() throws Exception {
506     // cursor queryies can't live in the queryResultCache, but independent filters
507     // should still be cached & reused
508 
509     // don't add in order of any field to ensure we aren't inadvertantly 
510     // counting on internal docid ordering
511     assertU(adoc("id", "9", "str", "c", "float", "-3.2", "int", "42"));
512     assertU(adoc("id", "7", "str", "c", "float", "-3.2", "int", "-1976"));
513     assertU(adoc("id", "2", "str", "c", "float", "-3.2", "int", "666"));
514     assertU(adoc("id", "0", "str", "b", "float", "64.5", "int", "-42"));
515     assertU(adoc("id", "5", "str", "b", "float", "64.5", "int", "2001"));
516     assertU(adoc("id", "8", "str", "b", "float", "64.5", "int", "4055"));
517     assertU(adoc("id", "6", "str", "a", "float", "64.5", "int", "7"));
518     assertU(adoc("id", "1", "str", "a", "float", "64.5", "int", "7"));
519     assertU(adoc("id", "4", "str", "a", "float", "11.1", "int", "6"));
520     assertU(adoc("id", "3", "str", "a", "float", "11.1", "int", "3"));
521     assertU(commit());
522 
523     final Collection<String> allFieldNames = getAllSortFieldNames();
524 
525     final SolrInfoMBean filterCacheStats 
526       = h.getCore().getInfoRegistry().get("filterCache");
527     assertNotNull(filterCacheStats);
528     final SolrInfoMBean queryCacheStats 
529       = h.getCore().getInfoRegistry().get("queryResultCache");
530     assertNotNull(queryCacheStats);
531 
532     final long preQcIn = (Long) queryCacheStats.getStatistics().get("inserts");
533     final long preFcIn = (Long) filterCacheStats.getStatistics().get("inserts");

655   /**
656    * Given a set of params, executes a cursor query using {@link CursorMarkParams#CURSOR_MARK_START}
657    * and then continuously walks the results using {@link CursorMarkParams#CURSOR_MARK_START} as long
658    * as a non-0 number of docs ar returned.  This method records the the set of all id's
659    * (must be positive ints) encountered and throws an assertion failure if any id is
660    * encountered more than once, or if the set grows above maxSize
661    */
662   public SentinelIntSet assertFullWalkNoDups(int maxSize, SolrParams params) 
663     throws Exception {
664 
665     SentinelIntSet ids = new SentinelIntSet(maxSize, -1);
666     String cursorMark = CURSOR_MARK_START;
667     int docsOnThisPage = Integer.MAX_VALUE;
668     while (0 < docsOnThisPage) {
669       String json = assertJQ(req(params,
670                                  CURSOR_MARK_PARAM, cursorMark));
671       Map rsp = (Map) ObjectBuilder.fromJSON(json);
672       assertTrue("response doesn't contain " + CURSOR_MARK_NEXT + ": " + json,
673                  rsp.containsKey(CURSOR_MARK_NEXT));
674       String nextCursorMark = (String)rsp.get(CURSOR_MARK_NEXT);
675       assertNotNull(CURSOR_MARK_NEXT + " is null", nextCursorMark);
676       List<Map<Object,Object>> docs = (List) (((Map)rsp.get("response")).get("docs"));
677       docsOnThisPage = docs.size();
678       if (null != params.getInt(CommonParams.ROWS)) {
679         int rows = params.getInt(CommonParams.ROWS);
680         assertTrue("Too many docs on this page: " + rows + " < " + docsOnThisPage,
681                    docsOnThisPage <= rows);
682       }
683       if (0 == docsOnThisPage) {
684         assertEquals("no more docs, but "+CURSOR_MARK_NEXT+" isn't same",
685                      cursorMark, nextCursorMark);
686       }
687       for (Map<Object,Object> doc : docs) {
688         int id = ((Long)doc.get("id")).intValue();
689         assertFalse("walk already seen: " + id, ids.exists(id));
690         ids.put(id);
691         assertFalse("id set bigger then max allowed ("+maxSize+"): " + ids.size(),
692                     maxSize < ids.size());
693       }
694       cursorMark = nextCursorMark;
695     }
696     return ids;
697   }

737   /**
738    * Given a set of params, executes a cursor query using {@link CursorMarkParams#CURSOR_MARK_START}
739    * and then continuously walks the results using {@link CursorMarkParams#CURSOR_MARK_START} as long
740    * as a non-0 number of docs ar returned.  This method records the the set of all id's
741    * (must be positive ints) encountered and throws an assertion failure if any id is
742    * encountered more than once, or if the set grows above maxSize.
743    *
744    * Also checks that facets are the same with each page, and that they are correct.
745    */
746   public SentinelIntSet assertFullWalkNoDupsWithFacets(int maxSize, SolrParams params)
747       throws Exception {
748 
749     final String facetField = params.get("facet.field");
750     assertNotNull("facet.field param not specified", facetField);
751     assertFalse("facet.field param contains multiple values", facetField.contains(","));
752     assertEquals("facet.limit param not set to -1", "-1", params.get("facet.limit"));
753     final Map<String,MutableValueInt> facetCounts = new HashMap<>();
754     SentinelIntSet ids = new SentinelIntSet(maxSize, -1);
755     String cursorMark = CURSOR_MARK_START;
756     int docsOnThisPage = Integer.MAX_VALUE;
757     List previousFacets = null;
758     while (0 < docsOnThisPage) {
759       String json = assertJQ(req(params, CURSOR_MARK_PARAM, cursorMark));
760       Map rsp = (Map) ObjectBuilder.fromJSON(json);
761       assertTrue("response doesn't contain " + CURSOR_MARK_NEXT + ": " + json,
762                  rsp.containsKey(CURSOR_MARK_NEXT));
763       String nextCursorMark = (String)rsp.get(CURSOR_MARK_NEXT);
764       assertNotNull(CURSOR_MARK_NEXT + " is null", nextCursorMark);
765       List<Map<Object,Object>> docs = (List)(((Map)rsp.get("response")).get("docs"));
766       docsOnThisPage = docs.size();
767       if (null != params.getInt(CommonParams.ROWS)) {
768         int rows = params.getInt(CommonParams.ROWS);
769         assertTrue("Too many docs on this page: " + rows + " < " + docsOnThisPage,
770                    docsOnThisPage <= rows);
771       }
772       if (0 == docsOnThisPage) {
773         assertEquals("no more docs, but "+CURSOR_MARK_NEXT+" isn't same",
774                      cursorMark, nextCursorMark);
775       }
776       for (Map<Object,Object> doc : docs) {
777         int id = ((Long)doc.get("id")).intValue();
778         assertFalse("walk already seen: " + id, ids.exists(id));
779         ids.put(id);
780         assertFalse("id set bigger then max allowed ("+maxSize+"): " + ids.size(),
781                     maxSize < ids.size());
782         Object facet = doc.get(facetField);
783         String facetString = null == facet ? null : facet.toString(); // null: missing facet value
784         MutableValueInt count = facetCounts.get(facetString);
785         if (null == count) {
786           count = new MutableValueInt();
787           facetCounts.put(facetString, count);
788         }
789         ++count.value;
790       }
791       cursorMark = nextCursorMark;
792 
793       Map facetFields = (Map)((Map)rsp.get("facet_counts")).get("facet_fields");
794       List facets = (List)facetFields.get(facetField);
795       if (null != previousFacets) {
796         assertEquals("Facets not the same as on previous page:\nprevious page facets: "
797             + Arrays.toString(facets.toArray(new Object[facets.size()]))
798             + "\ncurrent page facets: "
799             + Arrays.toString(previousFacets.toArray(new Object[previousFacets.size()])),
800             previousFacets, facets);
801       }
802       previousFacets = facets;
803     }
804 
805     assertNotNull("previousFacets is null", previousFacets);
806     assertEquals("Mismatch in number of facets: ", facetCounts.size(), previousFacets.size() / 2);
807     int pos;
808     for (pos = 0 ; pos < previousFacets.size() ; pos += 2) {
809       String label = (String)previousFacets.get(pos);
810       int expectedCount = ((Number)previousFacets.get(pos + 1)).intValue();
811       MutableValueInt count = facetCounts.get(label);
812       assertNotNull("Expected facet label #" + (pos / 2) + " not found: '" + label + "'", count);
813       assertEquals("Facet count mismatch for label #" + (pos / 2) + " '" + label + "'", expectedCount,
814                    facetCounts.get(label).value);
815       pos += 2;
816     }
817     return ids;
818   }

856    * Creates a document with randomized field values, some of which be missing values, 
857    * and some of which will be skewed so that small subsets of the ranges will be 
858    * more common (resulting in an increased likelihood of duplicate values)
859    * 
860    * @see #buildRandomQuery
861    */
862   public static SolrInputDocument buildRandomDocument(int id) {
863     SolrInputDocument doc = sdoc("id", id);
864     // most fields are in most docs
865     // if field is in a doc, then "skewed" chance val is from a dense range
866     // (hopefully with lots of duplication)
867     if (useField()) {
868       doc.addField("int", skewed(random().nextInt(), 
869                                  TestUtil.nextInt(random(), 20, 50)));
870     }
871     if (useField()) {
872       doc.addField("long", skewed(random().nextLong(), 
873                                   TestUtil.nextInt(random(), 5000, 5100)));
874     }
875     if (useField()) {
876       doc.addField("float", skewed(random().nextFloat() * random().nextInt(), 
877                                    1.0F / random().nextInt(23)));
878     }
879     if (useField()) {
880       doc.addField("double", skewed(random().nextDouble() * random().nextInt(), 
881                                     1.0D / random().nextInt(37)));
882     }
883     if (useField()) {
884       doc.addField("str", skewed(randomXmlUsableUnicodeString(),
885                                  TestUtil.randomSimpleString(random(), 1, 1)));
886     }
887     if (useField()) {
888       int numBytes = (int) skewed(TestUtil.nextInt(random(), 20, 50), 2);
889       byte[] randBytes = new byte[numBytes];
890       random().nextBytes(randBytes);
891       doc.addField("bin", ByteBuffer.wrap(randBytes));
892     }
893     if (useField()) {
894       doc.addField("date", skewed(randomDate(), randomSkewedDate()));
895     }
896     if (useField()) {
897       doc.addField("uuid", UUID.randomUUID().toString());
898     }
899     if (useField()) {
900       doc.addField("currency", skewed("" + (random().nextInt() / 100.) + "," + randomCurrency(),
901                                       "" + TestUtil.nextInt(random(), 250, 320) + ",USD"));
902     }
903     if (useField()) {
904       doc.addField("bool", random().nextBoolean() ? "t" : "f");
905     }
906     if (useField()) {
907       doc.addField("enum", randomEnumValue());
908     }
909     return doc;
910   }

913    * Generates a random query using the fields populated by 
914    * {@link #buildRandomDocument}.  Queries will typically be fairly simple, but 
915    * won't be so trivial that the scores are completely constant.
916    */
917   public static String buildRandomQuery() {
918     List<String> numericFields = Arrays.asList("int","long","float","double");
919     Collections.shuffle(numericFields, random());
920     if (random().nextBoolean()) {
921       // simple function query across one field.
922       return "{!func}" + numericFields.get(0);
923     } else {
924       // several SHOULD clauses on range queries
925       int low = TestUtil.nextInt(random(), -2379, 2);
926       int high = TestUtil.nextInt(random(), 4, 5713);
927       return 
928         numericFields.get(0) + ":[* TO 0] " +
929         numericFields.get(1) + ":[0 TO *] " +
930         numericFields.get(2) + ":[" + low + " TO " + high + "]";
931     }
932   }

944   /**
945    * Given a list of fieldNames, builds up a random sort string which is guaranteed to
946    * have at least 3 clauses, ending with the "id" field for tie breaking
947    */
948   public static String buildRandomSort(final Collection<String> fieldNames) {
949 
950     ArrayList<String> shuffledNames = new ArrayList<>(fieldNames);
951     Collections.replaceAll(shuffledNames, "id", "score");
952     Collections.shuffle(shuffledNames, random());
953 
954     final StringBuilder result = new StringBuilder();
955     final int numClauses = TestUtil.nextInt(random(), 2, 5);
956 
957     for (int i = 0; i < numClauses; i++) {
958       String field = shuffledNames.get(i);
959 
960       // wrap in a function sometimes
961       if ( ! "score".equals(field) && 0 == TestUtil.nextInt(random(), 0, 7)) {
962         // specific function doesn't matter, just proving that we can handle the concept.
963         // but we do have to be careful with non numeric fields
964         if (field.contains("float") || field.contains("double")
965             || field.contains("int") || field.contains("long")) {
966           field = "abs(" + field + ")";
967         } else {
968           field = "if(exists(" + field + "),47,83)";
969         }
970       }
971       result.append(field).append(random().nextBoolean() ? " asc, " : " desc, ");
972     }
973     result.append("id").append(random().nextBoolean() ? " asc" : " desc");
974     return result.toString();
975   }
