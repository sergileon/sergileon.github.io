30  public final class RevertDefaultThreadHandlerRule implements TestRule {
31    private final static AtomicBoolean applied = new AtomicBoolean();
32    
33    @Override
34    public Statement apply(Statement s, Description d) {
35      return new StatementAdapter(s) {
36        @Override
37        protected void before() throws Throwable {
38          if (!applied.getAndSet(true)) {
39            UncaughtExceptionHandler p = Thread.getDefaultUncaughtExceptionHandler();
40            try {
41              // Try to initialize a zookeeper class that reinitializes default exception handler.
42              Class<?> cl = NIOServerCnxnFactory.class;
43              // Make sure static initializers have been called.
44              Class.forName(cl.getName(), true, cl.getClassLoader());
45            } finally {
46              if (p == Thread.getDefaultUncaughtExceptionHandler()) {
47              //  throw new RuntimeException("Zookeeper no longer resets default thread handler.");
48              }
49              Thread.setDefaultUncaughtExceptionHandler(p);
50            }
51          }
52        }
53      };
54    }
55  }
