41    private class MapperCache implements Cache {
42      public Map<String, byte[]> map = new ConcurrentLinkedHashMap.Builder<String, byte[]>().maximumWeightedCapacity(8).build();
43  
44      @Override
45      public void update(String name, long blockId, int blockOffset, byte[] buffer, int offset, int length) {
46        byte[] cached = map.get(name + blockId);
47        if (cached != null) {
48          int newlen = Math.max(cached.length, blockOffset + length);
49          byte[] b = new byte[newlen];
50          System.arraycopy(cached, 0, b, 0, cached.length);
51          System.arraycopy(buffer, offset, b, blockOffset, length);
52          cached = b;
53        } else {
54          cached = copy(blockOffset, buffer, offset, length);
55        }
56        map.put(name + blockId, cached);
57      }
58  
59      private byte[] copy(int blockOffset, byte[] buffer, int offset, int length) {
60        byte[] b = new byte[length + blockOffset];
61        System.arraycopy(buffer, offset, b, blockOffset, length);
62        return b;
63      }
64  
65      @Override
66      public boolean fetch(String name, long blockId, int blockOffset, byte[] b, int off, int lengthToReadInBlock) {
67        // return false;
68        byte[] data = map.get(name + blockId);
69        if (data == null) {
70          return false;
71        }
72        System.arraycopy(data, blockOffset, b, off, lengthToReadInBlock);
73        return true;
74      }
75  
76      @Override
77      public void delete(String name) {
78  
79      }
80  
81      @Override
82      public long size() {
83        return map.size();
84      }
85  
86      @Override
87      public void renameCacheFile(String source, String dest) {
88      }
89  
90      @Override
91      public void releaseResources() {}
92    }

135     IndexInput input = directory.openInput(name, new IOContext());
136     try {
137     input.seek(length);

139         input.readByte();
140         fail("should throw eof");
141       } catch (IOException e) {
142       }
143     } finally {
144       input.close();
145     }

174   private void assertInputsEquals(String name, Directory fsDir, Directory hdfs) throws IOException {
175     int reads = random.nextInt(MAX_NUMBER_OF_READS);
176     IndexInput fsInput = fsDir.openInput(name, new IOContext());
177     IndexInput hdfsInput = hdfs.openInput(name, new IOContext());
178     assertEquals(fsInput.length(), hdfsInput.length());
179     int fileLength = (int) fsInput.length();
180     for (int i = 0; i < reads; i++) {
181       int rnd;
182       if (fileLength == 0) {
183         rnd = 0;
184       } else {
185         rnd = random.nextInt(Math.min(MAX_BUFFER_SIZE - MIN_BUFFER_SIZE, fileLength));
186       }
187 
188       byte[] fsBuf = new byte[rnd + MIN_BUFFER_SIZE];
189       byte[] hdfsBuf = new byte[fsBuf.length];
190       int offset = random.nextInt(fsBuf.length);
191       int length = random.nextInt(fsBuf.length - offset);
192       
193       int pos;
194       if (fileLength == 0) {
195         pos = 0;
196       } else {
197         pos = random.nextInt(fileLength - length);
198       }
199     
200       fsInput.seek(pos);
201       fsInput.readBytes(fsBuf, offset, length);
202       hdfsInput.seek(pos);
203       hdfsInput.readBytes(hdfsBuf, offset, length);
204       for (int f = offset; f < length; f++) {
205         if (fsBuf[f] != hdfsBuf[f]) {
206           fail("read [" + i + "]");
207         }
208       }
209     }
210     fsInput.close();
211     hdfsInput.close();
212   }

214   private void createFile(String name, Directory fsDir, Directory hdfs) throws IOException {
215     int writes = random.nextInt(MAX_NUMBER_OF_WRITES);
216     int fileLength = random.nextInt(MAX_FILE_SIZE - MIN_FILE_SIZE) + MIN_FILE_SIZE;
217     IndexOutput fsOutput = fsDir.createOutput(name, IOContext.DEFAULT);
218     IndexOutput hdfsOutput = hdfs.createOutput(name, IOContext.DEFAULT);
219     for (int i = 0; i < writes; i++) {
220       byte[] buf = new byte[random.nextInt(Math.min(MAX_BUFFER_SIZE - MIN_BUFFER_SIZE, fileLength)) + MIN_BUFFER_SIZE];
221       random.nextBytes(buf);
222       int offset = random.nextInt(buf.length);
223       int length = random.nextInt(buf.length - offset);
224       fsOutput.writeBytes(buf, offset, length);
225       hdfsOutput.writeBytes(buf, offset, length);
226     }
227     fsOutput.close();
228     hdfsOutput.close();
229   }

244   /**
245    * Verify the configuration options for the block cache are handled
246    * appropriately.
247    */
248   @Test
249   public void ensureCacheConfigurable() throws Exception {
250     IOContext mergeContext = new IOContext(new MergeInfo(1,1,false,1));
251 
252     BlockDirectory d = directory;
253     assertTrue(d.useReadCache("", IOContext.DEFAULT));
254     assertTrue(d.useWriteCache("", IOContext.DEFAULT));
255     assertFalse(d.useWriteCache("", mergeContext));
256 
257     d = new BlockDirectory("test", directory, mapperCache, null, true, false);
258     assertTrue(d.useReadCache("", IOContext.DEFAULT));
259     assertFalse(d.useWriteCache("", IOContext.DEFAULT));
260     assertFalse(d.useWriteCache("", mergeContext));
261 
262     d = new BlockDirectory("test", directory, mapperCache, null, false, true);
263     assertFalse(d.useReadCache("", IOContext.DEFAULT));
264     assertTrue(d.useWriteCache("", IOContext.DEFAULT));
265     assertFalse(d.useWriteCache("", mergeContext));
266   }
