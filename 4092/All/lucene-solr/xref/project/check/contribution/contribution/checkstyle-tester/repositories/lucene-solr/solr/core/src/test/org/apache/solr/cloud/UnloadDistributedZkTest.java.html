75    private void testUnloadShardAndCollection() throws Exception{
76      // create one leader and one replica
77      Create createCmd = new Create();
78      createCmd.setCoreName("test_unload_shard_and_collection_1");
79      String collection = "test_unload_shard_and_collection";
80      createCmd.setCollection(collection);
81      String coreDataDir = createTempDir().toFile().getAbsolutePath();
82      createCmd.setDataDir(getDataDir(coreDataDir));
83      createCmd.setNumShards(2);
84      
85      SolrClient client = clients.get(0);
86      String url1 = getBaseUrl(client);
87  
88      try (HttpSolrClient adminClient = new HttpSolrClient(url1)) {
89        adminClient.setConnectionTimeout(15000);
90        adminClient.setSoTimeout(60000);
91        adminClient.request(createCmd);
92  
93        createCmd = new Create();
94        createCmd.setCoreName("test_unload_shard_and_collection_2");
95        collection = "test_unload_shard_and_collection";
96        createCmd.setCollection(collection);
97        coreDataDir = createTempDir().toFile().getAbsolutePath();
98        createCmd.setDataDir(getDataDir(coreDataDir));
99  
100       adminClient.request(createCmd);
101 
102       // does not mean they are active and up yet :*
103       waitForRecoveriesToFinish(collection, false);
104 
105       // now unload one of the two
106       Unload unloadCmd = new Unload(false);
107       unloadCmd.setCoreName("test_unload_shard_and_collection_2");
108       adminClient.request(unloadCmd);
109 
110       // there should be only one shard
111       int slices = getCommonCloudSolrClient().getZkStateReader().getClusterState().getSlices(collection).size();
112       final TimeOut timeout = new TimeOut(45, TimeUnit.SECONDS);
113       while (slices != 1) {
114         if (timeout.hasTimedOut()) {
115           printLayout();
116           fail("Expected to find only one slice in " + collection);
117         }
118 
119         Thread.sleep(1000);
120         slices = getCommonCloudSolrClient().getZkStateReader().getClusterState().getSlices(collection).size();
121       }
122 
123       // now unload one of the other
124       unloadCmd = new Unload(false);
125       unloadCmd.setCoreName("test_unload_shard_and_collection_1");
126       adminClient.request(unloadCmd);
127     }
128 
129     //printLayout();
130     // the collection should be gone
131     final TimeOut timeout = new TimeOut(30, TimeUnit.SECONDS);
132     while (getCommonCloudSolrClient().getZkStateReader().getClusterState().hasCollection(collection)) {
133       if (timeout.hasTimedOut()) {
134         printLayout();
135         fail("Still found collection");
136       }
137       
138       Thread.sleep(50);
139     }
140     
141   }

147     File tmpDir = createTempDir().toFile();
148 
149     String core1DataDir = tmpDir.getAbsolutePath() + File.separator + System.nanoTime() + "unloadcollection1" + "_1n";
150 
151     // create a new collection collection
152     SolrClient client = clients.get(0);
153     String url1 = getBaseUrl(client);
154     try (HttpSolrClient adminClient = new HttpSolrClient(url1)) {
155       adminClient.setConnectionTimeout(15000);
156       adminClient.setSoTimeout(60000);
157 
158       Create createCmd = new Create();
159       createCmd.setCoreName("unloadcollection1");
160       createCmd.setCollection("unloadcollection");
161       createCmd.setNumShards(1);
162       createCmd.setDataDir(getDataDir(core1DataDir));
163       adminClient.request(createCmd);
164     }
165     ZkStateReader zkStateReader = getCommonCloudSolrClient().getZkStateReader();
166     
167     zkStateReader.updateClusterState();
168 
169     int slices = zkStateReader.getClusterState().getCollection("unloadcollection").getSlices().size();
170     assertEquals(1, slices);
171     
172     client = clients.get(1);
173     String url2 = getBaseUrl(client);
174     try (HttpSolrClient adminClient = new HttpSolrClient(url2)) {
175 
176       Create createCmd = new Create();
177       createCmd.setCoreName("unloadcollection2");
178       createCmd.setCollection("unloadcollection");
179       String core2dataDir = tmpDir.getAbsolutePath() + File.separator + System.nanoTime() + "unloadcollection1" + "_2n";
180       createCmd.setDataDir(getDataDir(core2dataDir));
181       adminClient.request(createCmd);
182     }
183     zkStateReader.updateClusterState();
184     slices = zkStateReader.getClusterState().getCollection("unloadcollection").getSlices().size();
185     assertEquals(1, slices);
186     
187     waitForRecoveriesToFinish("unloadcollection", zkStateReader, false);
188     
189     ZkCoreNodeProps leaderProps = getLeaderUrlFromZk("unloadcollection", "shard1");
190     
191     Random random = random();
192     if (random.nextBoolean()) {
193       try (HttpSolrClient collectionClient = new HttpSolrClient(leaderProps.getCoreUrl())) {
194         // lets try and use the solrj client to index and retrieve a couple
195         // documents
196         SolrInputDocument doc1 = getDoc(id, 6, i1, -600, tlong, 600, t1,
197             "humpty dumpy sat on a wall");
198         SolrInputDocument doc2 = getDoc(id, 7, i1, -600, tlong, 600, t1,
199             "humpty dumpy3 sat on a walls");
200         SolrInputDocument doc3 = getDoc(id, 8, i1, -600, tlong, 600, t1,
201             "humpty dumpy2 sat on a walled");
202         collectionClient.add(doc1);
203         collectionClient.add(doc2);
204         collectionClient.add(doc3);
205         collectionClient.commit();
206       }
207     }
208 
209     // create another replica for our collection
210     client = clients.get(2);
211     String url3 = getBaseUrl(client);
212     try (HttpSolrClient adminClient = new HttpSolrClient(url3)) {
213       Create createCmd = new Create();
214       createCmd.setCoreName("unloadcollection3");
215       createCmd.setCollection("unloadcollection");
216       String core3dataDir = tmpDir.getAbsolutePath() + File.separator + System.nanoTime() + "unloadcollection" + "_3n";
217       createCmd.setDataDir(getDataDir(core3dataDir));
218       adminClient.request(createCmd);
219     }
220     
221     waitForRecoveriesToFinish("unloadcollection", zkStateReader, false);
222     
223     // so that we start with some versions when we reload...
224     DirectUpdateHandler2.commitOnClose = false;
225     
226     try (HttpSolrClient addClient = new HttpSolrClient(url3 + "/unloadcollection3")) {
227       addClient.setConnectionTimeout(30000);
228 
229       // add a few docs
230       for (int x = 20; x < 100; x++) {
231         SolrInputDocument doc1 = getDoc(id, x, i1, -600, tlong, 600, t1,
232             "humpty dumpy sat on a wall");
233         addClient.add(doc1);
234       }
235     }
236     // don't commit so they remain in the tran log
237     //collectionClient.commit();
238     
239     // unload the leader
240     try (HttpSolrClient collectionClient = new HttpSolrClient(leaderProps.getBaseUrl())) {
241       collectionClient.setConnectionTimeout(15000);
242       collectionClient.setSoTimeout(30000);
243 
244       Unload unloadCmd = new Unload(false);
245       unloadCmd.setCoreName(leaderProps.getCoreName());
246       ModifiableSolrParams p = (ModifiableSolrParams) unloadCmd.getParams();
247 
248       collectionClient.request(unloadCmd);
249     }
250 //    Thread.currentThread().sleep(500);
251 //    printLayout();
252     
253     int tries = 50;
254     while (leaderProps.getCoreUrl().equals(zkStateReader.getLeaderUrl("unloadcollection", "shard1", 15000))) {
255       Thread.sleep(100);
256       if (tries-- == 0) {
257         fail("Leader never changed");
258       }
259     }
260     
261     // ensure there is a leader
262     zkStateReader.getLeaderRetry("unloadcollection", "shard1", 15000);
263     
264     try (HttpSolrClient addClient = new HttpSolrClient(url2 + "/unloadcollection2")) {
265       addClient.setConnectionTimeout(30000);
266       addClient.setSoTimeout(90000);
267 
268       // add a few docs while the leader is down
269       for (int x = 101; x < 200; x++) {
270         SolrInputDocument doc1 = getDoc(id, x, i1, -600, tlong, 600, t1,
271             "humpty dumpy sat on a wall");
272         addClient.add(doc1);
273       }
274     }
275     
276     // create another replica for our collection
277     client = clients.get(3);
278     String url4 = getBaseUrl(client);
279     try (HttpSolrClient adminClient = new HttpSolrClient(url4)) {
280       adminClient.setConnectionTimeout(15000);
281       adminClient.setSoTimeout(30000);
282 
283       Create createCmd = new Create();
284       createCmd.setCoreName("unloadcollection4");
285       createCmd.setCollection("unloadcollection");
286       String core4dataDir = tmpDir.getAbsolutePath() + File.separator + System.nanoTime() + "unloadcollection" + "_4n";
287       createCmd.setDataDir(getDataDir(core4dataDir));
288       adminClient.request(createCmd);
289     }
290     waitForRecoveriesToFinish("unloadcollection", zkStateReader, false);
291     
292     // unload the leader again
293     leaderProps = getLeaderUrlFromZk("unloadcollection", "shard1");
294     try (HttpSolrClient collectionClient = new HttpSolrClient(leaderProps.getBaseUrl())) {
295       collectionClient.setConnectionTimeout(15000);
296       collectionClient.setSoTimeout(30000);
297 
298       Unload unloadCmd = new Unload(false);
299       unloadCmd.setCoreName(leaderProps.getCoreName());
300       SolrParams p = (ModifiableSolrParams) unloadCmd.getParams();
301       collectionClient.request(unloadCmd);
302     }
303     tries = 50;
304     while (leaderProps.getCoreUrl().equals(zkStateReader.getLeaderUrl("unloadcollection", "shard1", 15000))) {
305       Thread.sleep(100);
306       if (tries-- == 0) {
307         fail("Leader never changed");
308       }
309     }
310     
311     zkStateReader.getLeaderRetry("unloadcollection", "shard1", 15000);
312     
313     
314     // set this back
315     DirectUpdateHandler2.commitOnClose = true;
316     
317     // bring the downed leader back as replica
318     try (HttpSolrClient adminClient = new HttpSolrClient(leaderProps.getBaseUrl())) {
319       adminClient.setConnectionTimeout(15000);
320       adminClient.setSoTimeout(30000);
321 
322       Create createCmd = new Create();
323       createCmd.setCoreName(leaderProps.getCoreName());
324       createCmd.setCollection("unloadcollection");
325       createCmd.setDataDir(getDataDir(core1DataDir));
326       adminClient.request(createCmd);
327     }
328     waitForRecoveriesToFinish("unloadcollection", zkStateReader, false);
329 
330     long found1, found3;
331     
332     try (HttpSolrClient adminClient = new HttpSolrClient(url2 + "/unloadcollection")) {
333       adminClient.setConnectionTimeout(15000);
334       adminClient.setSoTimeout(30000);
335       adminClient.commit();
336       SolrQuery q = new SolrQuery("*:*");
337       q.set("distrib", false);
338       found1 = adminClient.query(q).getResults().getNumFound();
339     }
340     try (HttpSolrClient adminClient = new HttpSolrClient(url3 + "/unloadcollection")) {
341       adminClient.setConnectionTimeout(15000);
342       adminClient.setSoTimeout(30000);
343       adminClient.commit();
344       SolrQuery q = new SolrQuery("*:*");
345       q.set("distrib", false);
346       found3 = adminClient.query(q).getResults().getNumFound();
347     }
348 
349     try (HttpSolrClient adminClient = new HttpSolrClient(url4 + "/unloadcollection")) {
350       adminClient.setConnectionTimeout(15000);
351       adminClient.setSoTimeout(30000);
352       adminClient.commit();
353       SolrQuery q = new SolrQuery("*:*");
354       q.set("distrib", false);
355       long found4 = adminClient.query(q).getResults().getNumFound();
356 
357       // all 3 shards should now have the same number of docs
358       assertEquals(found1, found3);
359       assertEquals(found3, found4);
360     }
361     
362   }

364   private void testUnloadLotsOfCores() throws Exception {
365     SolrClient client = clients.get(2);
366     String url3 = getBaseUrl(client);
367     try (final HttpSolrClient adminClient = new HttpSolrClient(url3)) {
368       adminClient.setConnectionTimeout(15000);
369       adminClient.setSoTimeout(60000);
370       int cnt = atLeast(3);
371       ThreadPoolExecutor executor = new ExecutorUtil.MDCAwareThreadPoolExecutor(0, Integer.MAX_VALUE,
372           5, TimeUnit.SECONDS, new SynchronousQueue<Runnable>(),
373           new DefaultSolrThreadFactory("testExecutor"));
374       try {
375         // create the cores
376         createCores(adminClient, executor, "multiunload", 2, cnt);
377       } finally {
378         ExecutorUtil.shutdownAndAwaitTermination(executor);
379       }
380 
381       executor = new ExecutorUtil.MDCAwareThreadPoolExecutor(0, Integer.MAX_VALUE, 5,
382           TimeUnit.SECONDS, new SynchronousQueue<Runnable>(),
383           new DefaultSolrThreadFactory("testExecutor"));
384       try {
385         for (int j = 0; j < cnt; j++) {
386           final int freezeJ = j;
387           executor.execute(new Runnable() {
388             @Override
389             public void run() {
390               Unload unloadCmd = new Unload(true);
391               unloadCmd.setCoreName("multiunload" + freezeJ);
392               try {
393                 adminClient.request(unloadCmd);
394               } catch (SolrServerException | IOException e) {
395                 throw new RuntimeException(e);
396               }
397             }
398           });
399           Thread.sleep(random().nextInt(50));
400         }
401       } finally {
402         ExecutorUtil.shutdownAndAwaitTermination(executor);
403       }
404     }
