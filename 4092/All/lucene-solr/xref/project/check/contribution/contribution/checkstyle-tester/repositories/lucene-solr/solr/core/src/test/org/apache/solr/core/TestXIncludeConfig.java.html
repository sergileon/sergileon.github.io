20  import javax.xml.parsers.DocumentBuilderFactory;
21  
22  import org.apache.solr.schema.IndexSchema;
23  import org.apache.solr.update.processor.RegexReplaceProcessorFactory;
24  import org.apache.solr.update.processor.UpdateRequestProcessorChain;
25  import org.apache.solr.util.AbstractSolrTestCase;
26  import org.junit.Assume;
27  import org.junit.BeforeClass;
28  
29  /** 
30   * Test both XInclude as well as more old school "entity includes"
31   */
32  public class TestXIncludeConfig extends AbstractSolrTestCase {
33  
34    @BeforeClass
35    public static void beforeClass() throws Exception {
36      initCore("solrconfig-xinclude.xml", "schema-xinclude.xml");
37    }
38  
39    @Override
40    public void setUp() throws Exception {
41      javax.xml.parsers.DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
42      try {
43        //see whether it even makes sense to run this test
44        dbf.setXIncludeAware(true);
45        dbf.setNamespaceAware(true);
46      } catch (UnsupportedOperationException e) {
47        Assume.assumeTrue(false);
48      }
49      super.setUp();
50    }
51  
52    public void testXInclude() throws Exception {
53      SolrCore core = h.getCore();
54  
55      assertNotNull("includedHandler is null", 
56                    core.getRequestHandler("includedHandler"));
57  
58      UpdateRequestProcessorChain chain 
59        = core.getUpdateProcessingChain("special-include");
60      assertNotNull("chain is missing included processor", chain);
61      assertEquals("chain with inclued processor is wrong size", 
62                   1, chain.getFactories().length);
63      assertEquals("chain has wrong included processor",
64                   RegexReplaceProcessorFactory.class,
65                   chain.getFactories()[0].getClass());
66  
67      IndexSchema schema = core.getLatestSchema();
68      
69      // xinclude
70      assertNotNull("ft-included is null", schema.getFieldTypeByName("ft-included"));
71      assertNotNull("field-included is null", schema.getFieldOrNull("field-included"));
72  
73      // entity include
74      assertNotNull("ft-entity-include1 is null", 
75                    schema.getFieldTypeByName("ft-entity-include1"));
76      assertNotNull("ft-entity-include2 is null", 
77                    schema.getFieldTypeByName("ft-entity-include2"));
78  
79      // sanity check
80      assertNull("ft-entity-include3 is not null",  // Does Not Exist Anywhere
81                 schema.getFieldTypeByName("ft-entity-include3"));
82  
83    }
84  }
