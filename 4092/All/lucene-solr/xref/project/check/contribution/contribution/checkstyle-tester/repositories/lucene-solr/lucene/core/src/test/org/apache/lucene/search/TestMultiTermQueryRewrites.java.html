42    static Directory dir, sdir1, sdir2;
43    static IndexReader reader, multiReader, multiReaderDupls;
44    static IndexSearcher searcher, multiSearcher, multiSearcherDupls;
45  
46    @BeforeClass
47    public static void beforeClass() throws Exception {
48      dir = newDirectory();
49      sdir1 = newDirectory();
50      sdir2 = newDirectory();
51      final RandomIndexWriter writer = new RandomIndexWriter(random(), dir, new MockAnalyzer(random()));
52      final RandomIndexWriter swriter1 = new RandomIndexWriter(random(), sdir1, new MockAnalyzer(random()));
53      final RandomIndexWriter swriter2 = new RandomIndexWriter(random(), sdir2, new MockAnalyzer(random()));
54  
55      for (int i = 0; i < 10; i++) {
56        Document doc = new Document();
57        doc.add(newStringField("data", Integer.toString(i), Field.Store.NO));
58        writer.addDocument(doc);
59        ((i % 2 == 0) ? swriter1 : swriter2).addDocument(doc);
60      }
61      writer.forceMerge(1); swriter1.forceMerge(1); swriter2.forceMerge(1);
62      writer.close(); swriter1.close(); swriter2.close();
63      
64      reader = DirectoryReader.open(dir);
65      searcher = newSearcher(reader);
66      
67      multiReader = new MultiReader(new IndexReader[] {
68        DirectoryReader.open(sdir1), DirectoryReader.open(sdir2) 
69      }, true);
70      multiSearcher = newSearcher(multiReader);
71      
72      multiReaderDupls = new MultiReader(new IndexReader[] {
73        DirectoryReader.open(sdir1), DirectoryReader.open(dir) 
74      }, true);
75      multiSearcherDupls = newSearcher(multiReaderDupls);
76    }
77  
78    @AfterClass
79    public static void afterClass() throws Exception {
80      reader.close();
81      multiReader.close();
82      multiReaderDupls.close();
83      dir.close(); sdir1.close(); sdir2.close();
84      reader = multiReader = multiReaderDupls = null;
85      searcher = multiSearcher = multiSearcherDupls = null;
86      dir = sdir1 = sdir2 = null;
87    }

89    private Query extractInnerQuery(Query q) {
90      if (q instanceof ConstantScoreQuery) {
91        // wrapped as ConstantScoreQuery
92        q = ((ConstantScoreQuery) q).getQuery();
93      }
94      return q;
95    }
96    
97    private Term extractTerm(Query q) {
98      q = extractInnerQuery(q);
99      return ((TermQuery) q).getTerm();
100   }
101   
102   private void checkBooleanQueryOrder(Query q) {
103     q = extractInnerQuery(q);
104     final BooleanQuery bq = (BooleanQuery) q;
105     Term last = null, act;
106     for (BooleanClause clause : bq.clauses()) {
107       act = extractTerm(clause.getQuery());
108       if (last != null) {
109         assertTrue("sort order of terms in BQ violated", last.compareTo(act) < 0);
110       }
111       last = act;
112     }
113   }
114   
115   private void checkDuplicateTerms(MultiTermQuery.RewriteMethod method) throws Exception {
116     final MultiTermQuery mtq = TermRangeQuery.newStringRange("data", "2", "7", true, true);
117     mtq.setRewriteMethod(method);
118     final Query q1 = searcher.rewrite(mtq);
119     final Query q2 = multiSearcher.rewrite(mtq);
120     final Query q3 = multiSearcherDupls.rewrite(mtq);
121     if (VERBOSE) {
122       System.out.println();
123       System.out.println("single segment: " + q1);
124       System.out.println("multi segment: " + q2);
125       System.out.println("multi segment with duplicates: " + q3);
126     }
127     assertEquals("The multi-segment case must produce same rewritten query", q1, q2);
128     assertEquals("The multi-segment case with duplicates must produce same rewritten query", q1, q3);
129     checkBooleanQueryOrder(q1);
130     checkBooleanQueryOrder(q2);
131     checkBooleanQueryOrder(q3);
132   }

134   public void testRewritesWithDuplicateTerms() throws Exception {
135     checkDuplicateTerms(MultiTermQuery.SCORING_BOOLEAN_REWRITE);
136     
137     checkDuplicateTerms(MultiTermQuery.CONSTANT_SCORE_BOOLEAN_REWRITE);
138     
139     // use a large PQ here to only test duplicate terms and dont mix up when all scores are equal
140     checkDuplicateTerms(new MultiTermQuery.TopTermsScoringBooleanQueryRewrite(1024));
141     checkDuplicateTerms(new MultiTermQuery.TopTermsBoostOnlyBooleanQueryRewrite(1024));
142   }

144   private void checkBooleanQueryBoosts(BooleanQuery bq) {
145     for (BooleanClause clause : bq.clauses()) {
146       final BoostQuery boostQ = (BoostQuery) clause.getQuery();
147       final TermQuery mtq = (TermQuery) boostQ.getQuery();
148       assertEquals("Parallel sorting of boosts in rewrite mode broken",
149         Float.parseFloat(mtq.getTerm().text()), boostQ.getBoost(), 0);
150     }
151   }
152   
153   private void checkBoosts(MultiTermQuery.RewriteMethod method) throws Exception {
154     final MultiTermQuery mtq = new MultiTermQuery("data") {
155       @Override
156       protected TermsEnum getTermsEnum(Terms terms, AttributeSource atts) throws IOException {
157         return new FilteredTermsEnum(terms.iterator()) {
158 
159           final BoostAttribute boostAtt =
160             attributes().addAttribute(BoostAttribute.class);
161         
162           @Override
163           protected AcceptStatus accept(BytesRef term) {
164             boostAtt.setBoost(Float.parseFloat(term.utf8ToString()));
165             if (term.length == 0) {
166               return AcceptStatus.NO;
167             }
168             char c = (char) (term.bytes[term.offset] & 0xff);
169             if (c >= '2') {
170               if (c <= '7') {
171                 return AcceptStatus.YES;
172               } else {
173                 return AcceptStatus.END;
174               }
175             } else {
176               return AcceptStatus.NO;
177             }
178           }
179         };
180       }
181       
182       @Override
183       public String toString(String field) {
184         return "dummy";
185       }
186     };
187     mtq.setRewriteMethod(method);
188     final Query q1 = searcher.rewrite(mtq);
189     final Query q2 = multiSearcher.rewrite(mtq);
190     final Query q3 = multiSearcherDupls.rewrite(mtq);
191     if (VERBOSE) {
192       System.out.println();
193       System.out.println("single segment: " + q1);
194       System.out.println("multi segment: " + q2);
195       System.out.println("multi segment with duplicates: " + q3);
196     }
197     assertEquals("The multi-segment case must produce same rewritten query", q1, q2);
198     assertEquals("The multi-segment case with duplicates must produce same rewritten query", q1, q3);
199     checkBooleanQueryBoosts((BooleanQuery) q1);
200     checkBooleanQueryBoosts((BooleanQuery) q2);
201     checkBooleanQueryBoosts((BooleanQuery) q3);
202   }
203   
204   public void testBoosts() throws Exception {
205     checkBoosts(MultiTermQuery.SCORING_BOOLEAN_REWRITE);
206 
207     // use a large PQ here to only test boosts and dont mix up when all scores are equal
208     checkBoosts(new MultiTermQuery.TopTermsScoringBooleanQueryRewrite(1024));
209   }

211   private void checkMaxClauseLimitation(MultiTermQuery.RewriteMethod method) throws Exception {
212     int savedMaxClauseCount = BooleanQuery.getMaxClauseCount();
213     BooleanQuery.setMaxClauseCount(3);
214     
215     final MultiTermQuery mtq = TermRangeQuery.newStringRange("data", "2", "7", true, true);
216     mtq.setRewriteMethod(method);
217     try {
218       multiSearcherDupls.rewrite(mtq);
219       fail("Should throw BooleanQuery.TooManyClauses");
220     } catch (BooleanQuery.TooManyClauses e) {
221       //  Maybe remove this assert in later versions, when internal API changes:
222       assertEquals("Should throw BooleanQuery.TooManyClauses with a stacktrace containing checkMaxClauseCount()",
223         "checkMaxClauseCount", e.getStackTrace()[0].getMethodName());
224     } finally {
225       BooleanQuery.setMaxClauseCount(savedMaxClauseCount);
226     }
227   }
228   
229   private void checkNoMaxClauseLimitation(MultiTermQuery.RewriteMethod method) throws Exception {
230     int savedMaxClauseCount = BooleanQuery.getMaxClauseCount();
231     BooleanQuery.setMaxClauseCount(3);
232     
233     final MultiTermQuery mtq = TermRangeQuery.newStringRange("data", "2", "7", true, true);
234     mtq.setRewriteMethod(method);
235     try {
236       multiSearcherDupls.rewrite(mtq);
237     } finally {
238       BooleanQuery.setMaxClauseCount(savedMaxClauseCount);
239     }
240   }

242   public void testMaxClauseLimitations() throws Exception {
243     checkMaxClauseLimitation(MultiTermQuery.SCORING_BOOLEAN_REWRITE);
244     checkMaxClauseLimitation(MultiTermQuery.CONSTANT_SCORE_BOOLEAN_REWRITE);
245     
246     checkNoMaxClauseLimitation(MultiTermQuery.CONSTANT_SCORE_REWRITE);
247     checkNoMaxClauseLimitation(new MultiTermQuery.TopTermsScoringBooleanQueryRewrite(1024));
248     checkNoMaxClauseLimitation(new MultiTermQuery.TopTermsBoostOnlyBooleanQueryRewrite(1024));
249   }
