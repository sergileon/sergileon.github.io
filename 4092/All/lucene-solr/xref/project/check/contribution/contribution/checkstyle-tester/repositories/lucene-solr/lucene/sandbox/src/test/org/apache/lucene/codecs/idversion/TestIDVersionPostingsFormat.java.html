190   public void testRandom() throws Exception {
191     Directory dir = newDirectory();
192     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
193     int minItemsInBlock = TestUtil.nextInt(random(), 2, 50);
194     int maxItemsInBlock = 2*(minItemsInBlock-1) + random().nextInt(50);
195     iwc.setCodec(TestUtil.alwaysPostingsFormat(new IDVersionPostingsFormat(minItemsInBlock, maxItemsInBlock)));
196     RandomIndexWriter w = new RandomIndexWriter(random(), dir, iwc);
197     //IndexWriter w = new IndexWriter(dir, iwc);
198     int numDocs = atLeast(1000);
199     Map<String,Long> idValues = new HashMap<String,Long>();
200     int docUpto = 0;
201     if (VERBOSE) {
202       System.out.println("TEST: numDocs=" + numDocs);
203     }
204 
205     IDSource ids = getRandomIDs();
206     String idPrefix;
207     if (random().nextBoolean()) {
208       idPrefix = "";
209     } else {
210       idPrefix = TestUtil.randomSimpleString(random());
211       if (VERBOSE) {
212         System.out.println("TEST: use id prefix: " + idPrefix);
213       }
214     }
215 
216     boolean useMonotonicVersion = random().nextBoolean();
217     if (VERBOSE) {
218       System.out.println("TEST: useMonotonicVersion=" + useMonotonicVersion);
219     }
220 
221     List<String> idsList = new ArrayList<>();
222 
223     long version = 0;
224     while (docUpto < numDocs) {
225       String idValue = idPrefix + ids.next();
226       if (idValues.containsKey(idValue)) {
227         continue;
228       }
229       if (useMonotonicVersion) {
230         version += TestUtil.nextInt(random(), 1, 10);
231       } else {
232         version = random().nextLong() & 0x3fffffffffffffffL;
233       }
234       idValues.put(idValue, version);
235       if (VERBOSE) {
236         System.out.println("  " + idValue + " -> " + version);
237       }
238       Document doc = new Document();
239       doc.add(makeIDField(idValue, version));
240       w.addDocument(doc);
241       idsList.add(idValue);
242 
243       if (idsList.size() > 0 && random().nextInt(7) == 5) {
244         // Randomly delete or update a previous ID
245         idValue = idsList.get(random().nextInt(idsList.size()));
246         if (random().nextBoolean()) {
247           if (useMonotonicVersion) {
248             version += TestUtil.nextInt(random(), 1, 10);
249           } else {
250             version = random().nextLong() & 0x3fffffffffffffffL;
251           }
252           doc = new Document();
253           doc.add(makeIDField(idValue, version));
254           if (VERBOSE) {
255             System.out.println("  update " + idValue + " -> " + version);
256           }
257           w.updateDocument(new Term("id", idValue), doc);
258           idValues.put(idValue, version);
259         } else {
260           if (VERBOSE) {
261             System.out.println("  delete " + idValue);
262           }
263           w.deleteDocuments(new Term("id", idValue));
264           idValues.remove(idValue);
265         }        
266       }
267 
268       docUpto++;
269     }
270 
271     IndexReader r = w.getReader();
272     //IndexReader r = DirectoryReader.open(w, true);
273     PerThreadVersionPKLookup lookup = new PerThreadVersionPKLookup(r, "id");
274 
275     List<Map.Entry<String,Long>> idValuesList = new ArrayList<>(idValues.entrySet());
276     int iters = numDocs * 5;
277     for(int iter=0;iter<iters;iter++) {
278       String idValue;
279 
280       if (random().nextBoolean()) {
281         idValue = idValuesList.get(random().nextInt(idValuesList.size())).getKey();
282       } else if (random().nextBoolean()) {
283         idValue = ids.next();
284       } else {
285         idValue = idPrefix + TestUtil.randomSimpleString(random());
286       }
287 
288       BytesRef idValueBytes = new BytesRef(idValue);
289 
290       Long expectedVersion = idValues.get(idValue);
291 
292       if (VERBOSE) {
293         System.out.println("\nTEST: iter=" + iter + " id=" + idValue + " expectedVersion=" + expectedVersion);
294       }
295       
296       if (expectedVersion == null) {
297         assertEquals("term should not have been found (doesn't exist)", -1, lookup.lookup(idValueBytes));
298       } else {
299         if (random().nextBoolean()) {
300           if (VERBOSE) {
301             System.out.println("  lookup exact version (should be found)");
302           }
303           assertTrue("term should have been found (version too old)", lookup.lookup(idValueBytes, expectedVersion.longValue()) != -1);
304           assertEquals(expectedVersion.longValue(), lookup.getVersion());
305         } else {
306           if (VERBOSE) {
307             System.out.println("  lookup version+1 (should not be found)");
308           }
309           assertEquals("term should not have been found (version newer)", -1, lookup.lookup(idValueBytes, expectedVersion.longValue()+1));
310         }
311       }
312     }
313 
314     r.close();
315     w.close();
316     dir.close();
317   }

319   private static class PerThreadVersionPKLookup extends PerThreadPKLookup {
320     public PerThreadVersionPKLookup(IndexReader r, String field) throws IOException {
321       super(r, field);
322     }
323 
324     long lastVersion;
325 
326     /** Returns docID if found, else -1. */
327     public int lookup(BytesRef id, long version) throws IOException {
328       for(int seg=0;seg<numSegs;seg++) {
329         if (((IDVersionSegmentTermsEnum) termsEnums[seg]).seekExact(id, version)) {
330           if (VERBOSE) {
331             System.out.println("  found in seg=" + termsEnums[seg]);
332           }
333           postingsEnums[seg] = termsEnums[seg].postings(postingsEnums[seg], 0);
334           int docID = postingsEnums[seg].nextDoc();
335           if (docID != PostingsEnum.NO_MORE_DOCS && (liveDocs[seg] == null || liveDocs[seg].get(docID))) {
336             lastVersion = ((IDVersionSegmentTermsEnum) termsEnums[seg]).getVersion();
337             return docBases[seg] + docID;
338           }
339           assert hasDeletions;
340         }
341       }
342 
343       return -1;
344     }
345 
346     /** Only valid if lookup returned a valid docID. */
347     public long getVersion() {
348       return lastVersion;
349     }
350   }

633   // Simulates optimistic concurrency in a distributed indexing app and confirms the latest version always wins:
634   public void testGlobalVersions() throws Exception {
635     Directory dir = newDirectory();
636     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
637     iwc.setCodec(TestUtil.alwaysPostingsFormat(new IDVersionPostingsFormat()));
638     final RandomIndexWriter w = new RandomIndexWriter(random(), dir, iwc);
639 
640     IDSource idsSource = getRandomIDs();
641     int numIDs = atLeast(100);
642     if (VERBOSE) {
643       System.out.println("TEST: " + numIDs + " ids");
644     }
645     Set<String> idsSeen = new HashSet<String>();
646     while (idsSeen.size() < numIDs) {
647       idsSeen.add(idsSource.next());
648     }
649     final String[] ids = idsSeen.toArray(new String[numIDs]);
650 
651     final Object[] locks = new Object[ids.length];
652     for(int i=0;i<locks.length;i++) {
653       locks[i] = new Object();
654     }
655 
656     final AtomicLong nextVersion = new AtomicLong();
657 
658     final SearcherManager mgr = new SearcherManager(w.w, true, new SearcherFactory());
659 
660     final Long missingValue = -1L;
661 
662     final LiveFieldValues<IndexSearcher,Long> versionValues = new LiveFieldValues<IndexSearcher,Long>(mgr, missingValue) {
663       @Override
664       protected Long lookupFromSearcher(IndexSearcher s, String id) {
665         // TODO: would be cleaner if we could do our PerThreadLookup here instead of "up above":
666         // We always return missing: the caller then does a lookup against the current reader
667         return missingValue;
668       }
669     };
670 
671     // Maps to the version the id was lasted indexed with:
672     final Map<String,Long> truth = new ConcurrentHashMap<>();
673 
674     final CountDownLatch startingGun = new CountDownLatch(1);
675 
676     Thread[] threads = new Thread[TestUtil.nextInt(random(), 2, 7)];
677 
678     final int versionType = random().nextInt(3);
679 
680     if (VERBOSE) {
681       if (versionType == 0) {
682         System.out.println("TEST: use random versions");
683       } else if (versionType == 1) {
684         System.out.println("TEST: use monotonic versions");
685       } else {
686         System.out.println("TEST: use nanotime versions");
687       }
688     }
689 
690     // Run for 3 sec in normal tests, else 60 seconds for nightly:
691     final long stopTime = System.currentTimeMillis() + (TEST_NIGHTLY ? 60000 : 3000);
692 
693     for(int i=0;i<threads.length;i++) {
694       threads[i] = new Thread() {
695           @Override
696           public void run() {
697             try {
698               runForReal();
699             } catch (Exception e) {
700               throw new RuntimeException(e);
701             }
702           }
703 
704           private void runForReal() throws IOException, InterruptedException {
705             startingGun.await();
706             PerThreadVersionPKLookup lookup = null;
707             IndexReader lookupReader = null;
708             while (System.currentTimeMillis() < stopTime) {
709 
710               // Intentionally pull version first, and then sleep/yield, to provoke version conflicts:
711               long newVersion;
712               if (versionType == 0) {
713                 // Random:
714                 newVersion = random().nextLong() & 0x3fffffffffffffffL;
715               } else if (versionType == 1) {
716                 // Monotonic
717                 newVersion = nextVersion.getAndIncrement();
718               } else {
719                 newVersion = System.nanoTime();
720               }
721 
722               if (versionType != 0) {
723                 if (random().nextBoolean()) {
724                   Thread.yield();
725                 } else {
726                   Thread.sleep(TestUtil.nextInt(random(), 1, 4));
727                 }
728               }
729 
730               int x = random().nextInt(ids.length);
731 
732               // TODO: we could relax this, if e.g. we assign indexer thread based on ID.  This would ensure a given ID cannot be indexed at
733               // the same time in multiple threads:
734 
735               // Only one thread can update an ID at once:
736               synchronized (locks[x]) {
737 
738                 String id = ids[x];
739 
740                 // We will attempt to index id with newVersion, but only do so if id wasn't yet indexed, or it was indexed with an older
741                 // version (< newVersion):
742 
743                 // Must lookup the RT value before pulling from the index, in case a reopen happens just after we lookup:
744                 Long currentVersion = versionValues.get(id);
745 
746                 IndexSearcher s = mgr.acquire();
747                 try {
748                   if (VERBOSE) System.out.println("\n" + Thread.currentThread().getName() + ": update id=" + id + " newVersion=" + newVersion);
749 
750                   if (lookup == null || lookupReader != s.getIndexReader()) {
751                     // TODO: sort of messy; we could add reopen to PerThreadVersionPKLookup?
752                     // TODO: this is thin ice .... that we don't incRef/decRef this reader we are implicitly holding onto:
753                     lookupReader = s.getIndexReader();
754                     if (VERBOSE) System.out.println(Thread.currentThread().getName() + ": open new PK lookup reader=" + lookupReader);
755                     lookup = new PerThreadVersionPKLookup(lookupReader, "id");
756                   }
757 
758                   Long truthVersion = truth.get(id);
759                   if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   truthVersion=" + truthVersion);
760 
761                   boolean doIndex;
762                   if (currentVersion == missingValue) {
763                     if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   id not in RT cache");
764                     int otherDocID = lookup.lookup(new BytesRef(id), newVersion+1);
765                     if (otherDocID == -1) {
766                       if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   id not in index, or version is <= newVersion; will index");
767                       doIndex = true;
768                     } else {
769                       if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   id is in index with version=" + lookup.getVersion() + "; will not index");
770                       doIndex = false;
771                       if (truthVersion.longValue() !=lookup.getVersion()) {
772                         System.out.println(Thread.currentThread() + ": now fail0!");
773                       }
774                       assertEquals(truthVersion.longValue(), lookup.getVersion());
775                     }
776                   } else {
777                     if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   id is in RT cache: currentVersion=" + currentVersion);
778                     doIndex = newVersion > currentVersion;
779                   }
780 
781                   if (doIndex) {
782                     if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   now index");
783                     boolean passes = truthVersion == null || truthVersion.longValue() <= newVersion;
784                     if (passes == false) {
785                       System.out.println(Thread.currentThread() + ": now fail!");
786                     }
787                     assertTrue(passes);
788                     Document doc = new Document();
789                     doc.add(makeIDField(id, newVersion));
790                     w.updateDocument(new Term("id", id), doc);
791                     truth.put(id, newVersion);
792                     versionValues.add(id, newVersion);
793                   } else {
794                     if (VERBOSE) System.out.println(Thread.currentThread().getName() + ":   skip index");
795                     assertNotNull(truthVersion);
796                     assertTrue(truthVersion.longValue() >= newVersion);
797                   }
798                 } finally {
799                   mgr.release(s);
800                 }
801               }
802             }
803           }
804         };
805       threads[i].start();
806     }
807 
808     startingGun.countDown();
809 
810     // Keep reopening the NRT reader until all indexing threads are done:
811     refreshLoop:
812     while (true) {
813       Thread.sleep(TestUtil.nextInt(random(), 1, 10));
814       mgr.maybeRefresh();
815       for (Thread thread : threads) {
816         if (thread.isAlive()) {
817           continue refreshLoop;
818         }
819       }
820 
821       break;
822     }
823 
824     // Verify final index against truth:
825     for(int i=0;i<2;i++) {
826       mgr.maybeRefresh();
827       IndexSearcher s = mgr.acquire();
828       try {
829         IndexReader r = s.getIndexReader();
830         // cannot assert this: maybe not all IDs were indexed
831         /*
832         assertEquals(numIDs, r.numDocs());
833         if (i == 1) {
834           // After forceMerge no deleted docs:
835           assertEquals(numIDs, r.maxDoc());
836         }
837         */
838         PerThreadVersionPKLookup lookup = new PerThreadVersionPKLookup(r, "id");
839         for(Map.Entry<String,Long> ent : truth.entrySet()) {
840           assertTrue(lookup.lookup(new BytesRef(ent.getKey()), -1L) != -1);
841           assertEquals(ent.getValue().longValue(), lookup.getVersion());
842         }
843       } finally {
844         mgr.release(s);
845       }
846 
847       if (i == 1) {
848         break;
849       }
850 
851       // forceMerge and verify again
852       w.forceMerge(1);
853     }
854 
855     mgr.close();
856     w.close();
857     dir.close();
858   }
