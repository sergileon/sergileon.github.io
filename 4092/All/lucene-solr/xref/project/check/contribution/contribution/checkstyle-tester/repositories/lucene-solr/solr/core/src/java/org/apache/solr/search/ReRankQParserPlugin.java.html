62  public class ReRankQParserPlugin extends QParserPlugin {
63  
64    public static final String NAME = "rerank";
65    private static Query defaultQuery = new MatchAllDocsQuery();
66  
67    public void init(NamedList args) {
68    }

70    public QParser createParser(String query, SolrParams localParams, SolrParams params, SolrQueryRequest req) {
71      return new ReRankQParser(query, localParams, params, req);
72    }
73  
74    private class ReRankQParser extends QParser  {
75  
76      public ReRankQParser(String query, SolrParams localParams, SolrParams params, SolrQueryRequest req) {
77        super(query, localParams, params, req);
78      }
79  
80      public Query parse() throws SyntaxError {
81        String reRankQueryString = localParams.get("reRankQuery");
82        if (reRankQueryString == null || reRankQueryString.trim().length() == 0)  {
83          throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "reRankQuery parameter is mandatory");
84        }
85        QParser reRankParser = QParser.getParser(reRankQueryString, null, req);
86        Query reRankQuery = reRankParser.parse();
87  
88        int reRankDocs  = localParams.getInt("reRankDocs", 200);
89        reRankDocs = Math.max(1, reRankDocs); //
90  
91        double reRankWeight = localParams.getDouble("reRankWeight",2.0d);
92  
93        int start = params.getInt(CommonParams.START,0);
94        int rows = params.getInt(CommonParams.ROWS,10);
95        int length = start+rows;
96        return new ReRankQuery(reRankQuery, reRankDocs, reRankWeight, length);
97      }
98    }

100   private final class ReRankQuery extends RankQuery {
101     private Query mainQuery = defaultQuery;
102     private Query reRankQuery;
103     private int reRankDocs;
104     private int length;
105     private double reRankWeight;
106     private Map<BytesRef, Integer> boostedPriority;
107 
108     public int hashCode() {
109       return 31 * super.hashCode() + mainQuery.hashCode()+reRankQuery.hashCode()+(int)reRankWeight+reRankDocs;
110     }
111 
112     public boolean equals(Object o) {
113       if (super.equals(o) == false) {
114         return false;
115       }
116       ReRankQuery rrq = (ReRankQuery)o;
117       return mainQuery.equals(rrq.mainQuery) &&
118              reRankQuery.equals(rrq.reRankQuery) &&
119              reRankWeight == rrq.reRankWeight &&
120              reRankDocs == rrq.reRankDocs;
121     }
122 
123     public ReRankQuery(Query reRankQuery, int reRankDocs, double reRankWeight, int length) {
124       this.reRankQuery = reRankQuery;
125       this.reRankDocs = reRankDocs;
126       this.reRankWeight = reRankWeight;
127       this.length = length;
128     }
129 
130     public RankQuery wrap(Query _mainQuery) {
131       if(_mainQuery != null){
132         this.mainQuery = _mainQuery;
133       }
134       return  this;
135     }
136 
137     public MergeStrategy getMergeStrategy() {
138       return null;
139     }
140 
141     public TopDocsCollector getTopDocsCollector(int len, SolrIndexSearcher.QueryCommand cmd, IndexSearcher searcher) throws IOException {
142 
143       if(this.boostedPriority == null) {
144         SolrRequestInfo info = SolrRequestInfo.getRequestInfo();
145         if(info != null) {
146           Map context = info.getReq().getContext();
147           this.boostedPriority = (Map<BytesRef, Integer>)context.get(QueryElevationComponent.BOOSTED_PRIORITY);
148         }
149       }
150 
151       return new ReRankCollector(reRankDocs, length, reRankQuery, reRankWeight, cmd, searcher, boostedPriority);
152     }
153 
154     @Override
155     public String toString(String s) {
156       return "{!rerank mainQuery='"+mainQuery.toString()+
157              "' reRankQuery='"+reRankQuery.toString()+
158              "' reRankDocs="+reRankDocs+
159              " reRankWeight="+reRankWeight+"}";
160     }
161 
162     public Query rewrite(IndexReader reader) throws IOException {
163       if (getBoost() != 1f) {
164         return super.rewrite(reader);
165       }
166       Query q = mainQuery.rewrite(reader);
167       if (q != mainQuery) {
168         return new ReRankQuery(reRankQuery, reRankDocs, reRankWeight, length).wrap(q);
169       }
170       return super.rewrite(reader);
171     }
172 
173     public Weight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException{
174       return new ReRankWeight(mainQuery, reRankQuery, reRankWeight, searcher, needsScores);
175     }
176   }

178   private class ReRankWeight extends Weight{
179     private Query reRankQuery;
180     private IndexSearcher searcher;
181     private Weight mainWeight;
182     private double reRankWeight;
183 
184     public ReRankWeight(Query mainQuery, Query reRankQuery, double reRankWeight, IndexSearcher searcher, boolean needsScores) throws IOException {
185       super(mainQuery);
186       this.reRankQuery = reRankQuery;
187       this.searcher = searcher;
188       this.reRankWeight = reRankWeight;
189       this.mainWeight = mainQuery.createWeight(searcher, needsScores);
190     }
191 
192     @Override
193     public void extractTerms(Set<Term> terms) {
194       this.mainWeight.extractTerms(terms);
195 
196     }
197 
198     public float getValueForNormalization() throws IOException {
199       return mainWeight.getValueForNormalization();
200     }
201 
202     public Scorer scorer(LeafReaderContext context) throws IOException {
203       return mainWeight.scorer(context);
204     }
205 
206     public void normalize(float norm, float topLevelBoost) {
207       mainWeight.normalize(norm, topLevelBoost);
208     }
209 
210     public Explanation explain(LeafReaderContext context, int doc) throws IOException {
211       Explanation mainExplain = mainWeight.explain(context, doc);
212       return new QueryRescorer(reRankQuery) {
213         @Override
214         protected float combine(float firstPassScore, boolean secondPassMatches, float secondPassScore) {
215           float score = firstPassScore;
216           if (secondPassMatches) {
217             score += reRankWeight * secondPassScore;
218           }
219           return score;
220         }
221       }.explain(searcher, mainExplain, context.docBase+doc);
222     }
223   }

225   private class ReRankCollector extends TopDocsCollector {
226 
227     private Query reRankQuery;
228     private TopDocsCollector  mainCollector;
229     private IndexSearcher searcher;
230     private int reRankDocs;
231     private int length;
232     private double reRankWeight;
233     private Map<BytesRef, Integer> boostedPriority;
234 
235 
236     public ReRankCollector(int reRankDocs,
237                            int length,
238                            Query reRankQuery,
239                            double reRankWeight,
240                            SolrIndexSearcher.QueryCommand cmd,
241                            IndexSearcher searcher,
242                            Map<BytesRef, Integer> boostedPriority) throws IOException {
243       super(null);
244       this.reRankQuery = reRankQuery;
245       this.reRankDocs = reRankDocs;
246       this.length = length;
247       this.boostedPriority = boostedPriority;
248       Sort sort = cmd.getSort();
249       if(sort == null) {
250         this.mainCollector = TopScoreDocCollector.create(Math.max(this.reRankDocs, length));
251       } else {
252         sort = sort.rewrite(searcher);
253         this.mainCollector = TopFieldCollector.create(sort, Math.max(this.reRankDocs, length), false, true, true);
254       }
255       this.searcher = searcher;
256       this.reRankWeight = reRankWeight;
257     }
258 
259     public int getTotalHits() {
260       return mainCollector.getTotalHits();
261     }
262 
263     @Override
264     public LeafCollector getLeafCollector(LeafReaderContext context) throws IOException {
265       return mainCollector.getLeafCollector(context);
266     }
267 
268     @Override
269     public boolean needsScores() {
270       return true;
271     }
272 
273     public TopDocs topDocs(int start, int howMany) {
274 
275       try {
276 
277         TopDocs mainDocs = mainCollector.topDocs(0,  Math.max(reRankDocs, length));
278 
279         if(mainDocs.totalHits == 0 || mainDocs.scoreDocs.length == 0) {
280           return mainDocs;
281         }
282 
283         if(boostedPriority != null) {
284           SolrRequestInfo info = SolrRequestInfo.getRequestInfo();
285           Map requestContext = null;
286           if(info != null) {
287             requestContext = info.getReq().getContext();
288           }
289 
290           IntIntHashMap boostedDocs = QueryElevationComponent.getBoostDocs((SolrIndexSearcher)searcher, boostedPriority, requestContext);
291 
292           ScoreDoc[] mainScoreDocs = mainDocs.scoreDocs;
293           ScoreDoc[] reRankScoreDocs = new ScoreDoc[Math.min(mainScoreDocs.length, reRankDocs)];
294           System.arraycopy(mainScoreDocs,0,reRankScoreDocs,0,reRankScoreDocs.length);
295 
296           mainDocs.scoreDocs = reRankScoreDocs;
297 
298           TopDocs rescoredDocs = new QueryRescorer(reRankQuery) {
299             @Override
300             protected float combine(float firstPassScore, boolean secondPassMatches, float secondPassScore) {
301               float score = firstPassScore;
302               if (secondPassMatches) {
303                 score += reRankWeight * secondPassScore;
304               }
305               return score;
306             }
307           }.rescore(searcher, mainDocs, mainDocs.scoreDocs.length);
308 
309           Arrays.sort(rescoredDocs.scoreDocs, new BoostedComp(boostedDocs, mainDocs.scoreDocs, rescoredDocs.getMaxScore()));
310 
311           //Lower howMany if we've collected fewer documents.
312           howMany = Math.min(howMany, mainScoreDocs.length);
313 
314           if(howMany == rescoredDocs.scoreDocs.length) {
315             return rescoredDocs; // Just return the rescoredDocs
316           } else if(howMany > rescoredDocs.scoreDocs.length) {
317             //We need to return more then we've reRanked, so create the combined page.
318             ScoreDoc[] scoreDocs = new ScoreDoc[howMany];
319             System.arraycopy(mainScoreDocs, 0, scoreDocs, 0, scoreDocs.length); //lay down the initial docs
320             System.arraycopy(rescoredDocs.scoreDocs, 0, scoreDocs, 0, rescoredDocs.scoreDocs.length);//overlay the re-ranked docs.
321             rescoredDocs.scoreDocs = scoreDocs;
322             return rescoredDocs;
323           } else {
324             //We've rescored more then we need to return.
325             ScoreDoc[] scoreDocs = new ScoreDoc[howMany];
326             System.arraycopy(rescoredDocs.scoreDocs, 0, scoreDocs, 0, howMany);
327             rescoredDocs.scoreDocs = scoreDocs;
328             return rescoredDocs;
329           }
330 
331         } else {
332 
333           ScoreDoc[] mainScoreDocs   = mainDocs.scoreDocs;
334 
335           /*
336           *  Create the array for the reRankScoreDocs.
337           */
338           ScoreDoc[] reRankScoreDocs = new ScoreDoc[Math.min(mainScoreDocs.length, reRankDocs)];
339 
340           /*
341           *  Copy the initial results into the reRankScoreDocs array.
342           */
343           System.arraycopy(mainScoreDocs, 0, reRankScoreDocs, 0, reRankScoreDocs.length);
344 
345           mainDocs.scoreDocs = reRankScoreDocs;
346 
347           TopDocs rescoredDocs = new QueryRescorer(reRankQuery) {
348             @Override
349             protected float combine(float firstPassScore, boolean secondPassMatches, float secondPassScore) {
350               float score = firstPassScore;
351               if (secondPassMatches) {
352                 score += reRankWeight * secondPassScore;
353               }
354               return score;
355             }
356           }.rescore(searcher, mainDocs, mainDocs.scoreDocs.length);
357 
358           //Lower howMany to return if we've collected fewer documents.
359           howMany = Math.min(howMany, mainScoreDocs.length);
360 
361           if(howMany == rescoredDocs.scoreDocs.length) {
362             return rescoredDocs; // Just return the rescoredDocs
363           } else if(howMany > rescoredDocs.scoreDocs.length) {
364 
365             //We need to return more then we've reRanked, so create the combined page.
366             ScoreDoc[] scoreDocs = new ScoreDoc[howMany];
367             //lay down the initial docs
368             System.arraycopy(mainScoreDocs, 0, scoreDocs, 0, scoreDocs.length);
369             //overlay the rescoreds docs
370             System.arraycopy(rescoredDocs.scoreDocs, 0, scoreDocs, 0, rescoredDocs.scoreDocs.length);
371             rescoredDocs.scoreDocs = scoreDocs;
372             return rescoredDocs;
373           } else {
374             //We've rescored more then we need to return.
375             ScoreDoc[] scoreDocs = new ScoreDoc[howMany];
376             System.arraycopy(rescoredDocs.scoreDocs, 0, scoreDocs, 0, howMany);
377             rescoredDocs.scoreDocs = scoreDocs;
378             return rescoredDocs;
379           }
380         }
381       } catch (Exception e) {
382         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, e);
383       }
384     }

388   public class BoostedComp implements Comparator {
389     IntFloatHashMap boostedMap;
390 
391     public BoostedComp(IntIntHashMap boostedDocs, ScoreDoc[] scoreDocs, float maxScore) {
392       this.boostedMap = new IntFloatHashMap(boostedDocs.size()*2);
393 
394       for(int i=0; i<scoreDocs.length; i++) {
395         final int idx;
396         if((idx = boostedDocs.indexOf(scoreDocs[i].doc)) >= 0) {
397           boostedMap.put(scoreDocs[i].doc, maxScore+boostedDocs.indexGet(idx));
398         } else {
399           break;
400         }
401       }
402     }
403 
404     public int compare(Object o1, Object o2) {
405       ScoreDoc doc1 = (ScoreDoc) o1;
406       ScoreDoc doc2 = (ScoreDoc) o2;
407       float score1 = doc1.score;
408       float score2 = doc2.score;
409       int idx;
410       if((idx = boostedMap.indexOf(doc1.doc)) >= 0) {
411         score1 = boostedMap.indexGet(idx);
412       }
413 
414       if((idx = boostedMap.indexOf(doc2.doc)) >= 0) {
415         score2 = boostedMap.indexGet(idx);
416       }
417 
418       return -Float.compare(score1, score2);
419     }
