173   public void test() throws Exception {
174     Path tmp = LuceneTestCase.createTempDir();
175     
176     for (int i = 0; i < tests.length; i += 3) {
177       Path f = DICTIONARY_HOME.resolve(tests[i]);
178       assert Files.exists(f);
179       
180       IOUtils.rm(tmp);
181       Files.createDirectory(tmp);
182       
183       try (InputStream in = Files.newInputStream(f)) {
184         TestUtil.unzip(in, tmp);
185         Path dicEntry = tmp.resolve(tests[i+1]);
186         Path affEntry = tmp.resolve(tests[i+2]);
187       
188         try (InputStream dictionary = Files.newInputStream(dicEntry);
189              InputStream affix = Files.newInputStream(affEntry)) {
190           Dictionary dic = new Dictionary(affix, dictionary);
191           System.out.println(tests[i] + "\t" + RamUsageTester.humanSizeOf(dic) + "\t(" +
192                              "words=" + RamUsageTester.humanSizeOf(dic.words) + ", " +
193                              "flags=" + RamUsageTester.humanSizeOf(dic.flagLookup) + ", " +
194                              "strips=" + RamUsageTester.humanSizeOf(dic.stripData) + ", " +
195                              "conditions=" + RamUsageTester.humanSizeOf(dic.patterns) + ", " +
196                              "affixData=" + RamUsageTester.humanSizeOf(dic.affixData) + ", " +
197                              "prefixes=" + RamUsageTester.humanSizeOf(dic.prefixes) + ", " +
198                              "suffixes=" + RamUsageTester.humanSizeOf(dic.suffixes) + ")");
199         }
200       }
201     }
202   }
