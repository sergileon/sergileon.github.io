384   public void testLineDocFile() throws Exception {
385     Path lineFile = createTempFile("test.reuters.lines", ".txt");
386 
387     // We will call WriteLineDocs this many times
388     final int NUM_TRY_DOCS = 50;
389 
390     // Creates a line file with first 50 docs from SingleDocSource
391     String algLines1[] = {
392       "# ----- properties ",
393       "content.source=org.apache.lucene.benchmark.byTask.feeds.SingleDocSource",
394       "content.source.forever=true",
395       "line.file.out=" + lineFile.toAbsolutePath().toString().replace('\\', '/'),
396       "# ----- alg ",
397       "{WriteLineDoc()}:" + NUM_TRY_DOCS,
398     };
399 
400     // Run algo
401     Benchmark benchmark = execBenchmark(algLines1);
402 
403     BufferedReader r = Files.newBufferedReader(lineFile, StandardCharsets.UTF_8);
404     int numLines = 0;
405     String line;
406     while((line = r.readLine()) != null) {
407       if (numLines==0 && line.startsWith(WriteLineDocTask.FIELDS_HEADER_INDICATOR)) {
408         continue; // do not count the header line as a doc 
409       }
410       numLines++;
411     }
412     r.close();
413     assertEquals("did not see the right number of docs; should be " + NUM_TRY_DOCS + " but was " + numLines, NUM_TRY_DOCS, numLines);
414     
415     // Index the line docs
416     String algLines2[] = {
417       "# ----- properties ",
418       "analyzer=org.apache.lucene.analysis.core.WhitespaceAnalyzer",
419       "content.source=org.apache.lucene.benchmark.byTask.feeds.LineDocSource",
420       "docs.file=" + lineFile.toAbsolutePath().toString().replace('\\', '/'),
421       "content.source.forever=false",
422       "doc.reuse.fields=false",
423       "ram.flush.mb=4",
424       "# ----- alg ",
425       "ResetSystemErase",
426       "CreateIndex",
427       "{AddDoc}: *",
428       "CloseIndex",
429     };
430     
431     // Run algo
432     benchmark = execBenchmark(algLines2);
433 
434     // now we should be able to open the index for write. 
435     IndexWriter iw = new IndexWriter(benchmark.getRunData().getDirectory(),
436         new IndexWriterConfig(new MockAnalyzer(random()))
437             .setOpenMode(OpenMode.APPEND));
438     iw.close();
439 
440     IndexReader ir = DirectoryReader.open(benchmark.getRunData().getDirectory());
441     assertEquals(numLines + " lines were created but " + ir.numDocs() + " docs are in the index", numLines, ir.numDocs());
442     ir.close();
443 
444     Files.delete(lineFile);
445   }

447   /**
448    * Test ReadTokensTask
449    */
450   public void testReadTokens() throws Exception {
451 
452     // We will call ReadTokens on this many docs
453     final int NUM_DOCS = 20;
454 
455     // Read tokens from first NUM_DOCS docs from Reuters and
456     // then build index from the same docs
457     String algLines1[] = {
458       "# ----- properties ",
459       "analyzer=org.apache.lucene.analysis.core.WhitespaceAnalyzer",
460       "content.source=org.apache.lucene.benchmark.byTask.feeds.LineDocSource",
461       "docs.file=" + getReuters20LinesFile(),
462       "# ----- alg ",
463       "{ReadTokens}: " + NUM_DOCS,
464       "ResetSystemErase",
465       "CreateIndex",
466       "{AddDoc}: " + NUM_DOCS,
467       "CloseIndex",
468     };
469 
470     // Run algo
471     Benchmark benchmark = execBenchmark(algLines1);
472 
473     List<TaskStats> stats = benchmark.getRunData().getPoints().taskStats();
474 
475     // Count how many tokens all ReadTokens saw
476     int totalTokenCount1 = 0;
477     for (final TaskStats stat : stats) {
478       if (stat.getTask().getName().equals("ReadTokens")) {
479         totalTokenCount1 += stat.getCount();
480       }
481     }
482 
483     // Separately count how many tokens are actually in the index:
484     IndexReader reader = DirectoryReader.open(benchmark.getRunData().getDirectory());
485     assertEquals(NUM_DOCS, reader.numDocs());
486 
487     int totalTokenCount2 = 0;
488 
489     Fields fields = MultiFields.getFields(reader);
490 
491     for (String fieldName : fields) {
492       if (fieldName.equals(DocMaker.ID_FIELD) || fieldName.equals(DocMaker.DATE_MSEC_FIELD) || fieldName.equals(DocMaker.TIME_SEC_FIELD)) {
493         continue;
494       }
495       Terms terms = fields.terms(fieldName);
496       if (terms == null) {
497         continue;
498       }
499       TermsEnum termsEnum = terms.iterator();
500       PostingsEnum docs = null;
501       while(termsEnum.next() != null) {
502         docs = TestUtil.docs(random(), termsEnum, docs, PostingsEnum.FREQS);
503         while(docs.nextDoc() != DocIdSetIterator.NO_MORE_DOCS) {
504           totalTokenCount2 += docs.freq();
505         }
506       }
507     }
508     reader.close();
509 
510     // Make sure they are the same
511     assertEquals(totalTokenCount1, totalTokenCount2);
512   }

967   private void assertEqualCollation(Analyzer a1, Analyzer a2, String text)
968       throws Exception {
969     TokenStream ts1 = a1.tokenStream("bogus", text);
970     TokenStream ts2 = a2.tokenStream("bogus", text);
971     ts1.reset();
972     ts2.reset();
973     TermToBytesRefAttribute termAtt1 = ts1.addAttribute(TermToBytesRefAttribute.class);
974     TermToBytesRefAttribute termAtt2 = ts2.addAttribute(TermToBytesRefAttribute.class);
975     assertTrue(ts1.incrementToken());
976     assertTrue(ts2.incrementToken());
977     BytesRef bytes1 = termAtt1.getBytesRef();
978     BytesRef bytes2 = termAtt2.getBytesRef();
979     assertEquals(bytes1, bytes2);
980     assertFalse(ts1.incrementToken());
981     assertFalse(ts2.incrementToken());
982     ts1.close();
983     ts2.close();
984   }
