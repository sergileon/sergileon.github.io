204   /**
205    * Execute the task.
206    */
207   @Override
208   public void execute() throws BuildException {
209     log("Starting scan.", verboseLevel);
210     long start = System.currentTimeMillis();
211 
212     setupIvy();
213 
214     int numErrors = 0;
215     if ( ! verifySortedCoordinatesPropertiesFile(centralizedVersionsFile)) {
216       ++numErrors;
217     }
218     if ( ! verifySortedCoordinatesPropertiesFile(ignoreConflictsFile)) {
219       ++numErrors;
220     }
221     collectDirectDependencies();
222     if ( ! collectVersionConflictsToIgnore()) {
223       ++numErrors;
224     }
225 
226     int numChecked = 0;
227 
228     @SuppressWarnings("unchecked")
229     Iterator<Resource> iter = (Iterator<Resource>)ivyXmlResources.iterator();
230     while (iter.hasNext()) {
231       final Resource resource = iter.next();
232       if ( ! resource.isExists()) {
233         throw new BuildException("Resource does not exist: " + resource.getName());
234       }
235       if ( ! (resource instanceof FileResource)) {
236         throw new BuildException("Only filesystem resources are supported: " 
237             + resource.getName() + ", was: " + resource.getClass().getName());
238       }
239 
240       File ivyXmlFile = ((FileResource)resource).getFile();
241       try {
242         if ( ! checkIvyXmlFile(ivyXmlFile)) {
243           ++numErrors;
244         }
245         if ( ! resolveTransitively(ivyXmlFile)) {
246           ++numErrors;
247         }
248         if ( ! findLatestConflictVersions()) {
249           ++numErrors;
250         }
251       } catch (Exception e) {
252         throw new BuildException("Exception reading file " + ivyXmlFile.getPath() + " - " + e.toString(), e);
253       }
254       ++numChecked;
255     }
256 
257     log("Checking for orphans in " + centralizedVersionsFile.getName(), verboseLevel);
258     for (Map.Entry<String,Dependency> entry : directDependencies.entrySet()) {
259       String coordinateKey = entry.getKey();
260       if ( ! entry.getValue().directlyReferenced) {
261         log("ORPHAN coordinate key '" + coordinateKey + "' in " + centralizedVersionsFile.getName()
262             + " is not found in any " + IVY_XML_FILENAME + " file.",
263             Project.MSG_ERR);
264         ++numErrors;
265       }
266     }
267 
268     int numConflicts = emitConflicts();
269 
270     int messageLevel = numErrors > 0 ? Project.MSG_ERR : Project.MSG_INFO;
271     log("Checked that " + centralizedVersionsFile.getName() + " and " + ignoreConflictsFile.getName()
272         + " have lexically sorted '/org/name' keys and no duplicates or orphans.",
273         messageLevel);
274     log("Scanned " + numChecked + " " + IVY_XML_FILENAME + " files for rev=\"${/org/name}\" format.",
275         messageLevel);
276     log("Found " + numConflicts + " indirect dependency version conflicts.");
277     log(String.format(Locale.ROOT, "Completed in %.2fs., %d error(s).",
278                       (System.currentTimeMillis() - start) / 1000.0, numErrors),
279         messageLevel);
280 
281     if (numConflicts > 0 || numErrors > 0) {
282       throw new BuildException("Lib versions check failed. Check the logs.");
283     }
284   }

286   private boolean findLatestConflictVersions() {
287     boolean success = true;
288     StringBuilder latestIvyXml = new StringBuilder();
289     latestIvyXml.append("<ivy-module version=\"2.0\">\n");
290     latestIvyXml.append("  <info organisation=\"org.apache.lucene\" module=\"core-tools-find-latest-revision\"/>\n");
291     latestIvyXml.append("  <configurations>\n");
292     latestIvyXml.append("    <conf name=\"default\" transitive=\"false\"/>\n");
293     latestIvyXml.append("  </configurations>\n");
294     latestIvyXml.append("  <dependencies>\n");
295     for (Map.Entry<String, Dependency> directDependency : directDependencies.entrySet()) {
296       Dependency dependency = directDependency.getValue();
297       if (dependency.conflictLocations.entrySet().isEmpty()) {
298         continue;
299       }
300       latestIvyXml.append("    <dependency org=\"");
301       latestIvyXml.append(dependency.org);
302       latestIvyXml.append("\" name=\"");
303       latestIvyXml.append(dependency.name);
304       latestIvyXml.append("\" rev=\"latest.release\" conf=\"default->*\"/>\n");
305     }
306     latestIvyXml.append("  </dependencies>\n");
307     latestIvyXml.append("</ivy-module>\n");
308     File buildDir = new File(commonBuildDir, "ivy-transitive-resolve");
309     if ( ! buildDir.exists() && ! buildDir.mkdirs()) {
310       throw new BuildException("Could not create temp directory " + buildDir.getPath());
311     }
312     File findLatestIvyXmlFile = new File(buildDir, "find.latest.conflicts.ivy.xml");
313     try {
314       try (Writer writer = new OutputStreamWriter(new FileOutputStream(findLatestIvyXmlFile), StandardCharsets.UTF_8)) {
315         writer.write(latestIvyXml.toString());
316       }
317       ResolveOptions options = new ResolveOptions();
318       options.setDownload(false);           // Download only module descriptors, not artifacts
319       options.setTransitive(false);         // Resolve only direct dependencies
320       options.setUseCacheOnly(false);       // Download the internet!
321       options.setOutputReport(false);       // Don't print to the console
322       options.setLog(LogOptions.LOG_QUIET); // Don't log to the console
323       options.setConfs(new String[] {"*"}); // Resolve all configurations
324       ResolveReport resolveReport = ivy.resolve(findLatestIvyXmlFile.toURI().toURL(), options);
325       IvyNodeElement root = IvyNodeElementAdapter.adapt(resolveReport);
326       for (IvyNodeElement element : root.getDependencies()) {
327         String coordinate = "/" + element.getOrganization() + "/" + element.getName();
328         Dependency dependency = directDependencies.get(coordinate);
329         if (null == dependency) {
330           log("ERROR: the following coordinate key does not appear in "
331               + centralizedVersionsFile.getName() + ": " + coordinate, Project.MSG_ERR);
332           success = false;
333         } else {
334           dependency.latestVersion = element.getRevision();
335         }
336       }
337     } catch (IOException e) {
338       log("Exception writing to " + findLatestIvyXmlFile.getPath() + ": " + e.toString(), Project.MSG_ERR);
339       success = false;
340     } catch (ParseException e) {
341       log("Exception parsing filename " + findLatestIvyXmlFile.getPath() + ": " + e.toString(), Project.MSG_ERR);
342       success = false;
343     }
344     return success;
345   }

347   /**
348    * Collects indirect dependency version conflicts to ignore 
349    * in ivy-ignore-conflicts.properties, and also checks for orphans
350    * (coordinates not included in ivy-versions.properties).
351    * 
352    * Returns true if no orphans are found.
353    */
354   private boolean collectVersionConflictsToIgnore() {
355     log("Checking for orphans in " + ignoreConflictsFile.getName(), verboseLevel);
356     boolean orphansFound = false;
357     InterpolatedProperties properties = new InterpolatedProperties();
358     try (InputStream inputStream = new FileInputStream(ignoreConflictsFile);
359          Reader reader = new InputStreamReader(inputStream, StandardCharsets.UTF_8)) {
360       properties.load(reader);
361     } catch (IOException e) {
362       throw new BuildException("Exception reading " + ignoreConflictsFile + ": " + e.toString(), e);
363     }
364     for (Object obj : properties.keySet()) {
365       String coordinate = (String)obj;
366       if (COORDINATE_KEY_PATTERN.matcher(coordinate).matches()) {
367         if ( ! directDependencies.containsKey(coordinate)) {
368           orphansFound = true;
369           log("ORPHAN coordinate key '" + coordinate + "' in " + ignoreConflictsFile.getName()
370                   + " is not found in " + centralizedVersionsFile.getName(),
371               Project.MSG_ERR);
372         } else {
373           String versionsToIgnore = properties.getProperty(coordinate);
374           List<String> ignore = Arrays.asList(versionsToIgnore.trim().split("\\s*,\\s*|\\s+"));
375           ignoreConflictVersions.put(coordinate, new HashSet<>(ignore));
376         }
377       }
378     }
379     return ! orphansFound;
380   }

382   private void collectDirectDependencies() {
383     InterpolatedProperties properties = new InterpolatedProperties();
384     try (InputStream inputStream = new FileInputStream(centralizedVersionsFile);
385          Reader reader = new InputStreamReader(inputStream, StandardCharsets.UTF_8)) {
386       properties.load(reader);
387     } catch (IOException e) {
388       throw new BuildException("Exception reading " + centralizedVersionsFile + ": " + e.toString(), e);
389     }
390     for (Object obj : properties.keySet()) {
391       String coordinate = (String)obj;
392       Matcher matcher = COORDINATE_KEY_PATTERN.matcher(coordinate);
393       if (matcher.matches()) {
394         String org = matcher.group(2);
395         String name = matcher.group(3);
396         String directVersion = properties.getProperty(coordinate);
397         Dependency dependency = new Dependency(org, name, directVersion);
398         directDependencies.put(coordinate, dependency);
399       }
400     }
401   }

403   /**
404    * Transitively resolves all dependencies in the given ivy.xml file,
405    * looking for indirect dependencies with versions that conflict
406    * with those of direct dependencies.  Dependency conflict when a
407    * direct dependency's version is older than that of an indirect
408    * dependency with the same /org/name.
409    * 
410    * Returns true if no version conflicts are found and no resolution
411    * errors occurred, false otherwise.
412    */
413   private boolean resolveTransitively(File ivyXmlFile) {
414     boolean success = true;
415 
416     ResolveOptions options = new ResolveOptions();
417     options.setDownload(false);           // Download only module descriptors, not artifacts
418     options.setTransitive(true);          // Resolve transitively, if not already specified in the ivy.xml file
419     options.setUseCacheOnly(false);       // Download the internet!
420     options.setOutputReport(false);       // Don't print to the console
421     options.setLog(LogOptions.LOG_QUIET); // Don't log to the console
422     options.setConfs(new String[] {"*"}); // Resolve all configurations
423 
424     // Rewrite the ivy.xml, replacing all 'transitive="false"' with 'transitive="true"'
425     // The Ivy API is file-based, so we have to write the result to the filesystem.
426     String moduleName = "unknown";
427     String ivyXmlContent = xmlToString(ivyXmlFile);
428     Matcher matcher = MODULE_NAME_PATTERN.matcher(ivyXmlContent);
429     if (matcher.find()) {
430       moduleName = matcher.group(1);
431     }
432     ivyXmlContent = ivyXmlContent.replaceAll("\\btransitive\\s*=\\s*[\"']false[\"']", "transitive=\"true\"");
433     File transitiveIvyXmlFile = null;
434     try {
435       File buildDir = new File(commonBuildDir, "ivy-transitive-resolve");
436       if ( ! buildDir.exists() && ! buildDir.mkdirs()) {
437         throw new BuildException("Could not create temp directory " + buildDir.getPath());
438       }
439       matcher = MODULE_DIRECTORY_PATTERN.matcher(ivyXmlFile.getCanonicalPath());
440       if ( ! matcher.matches()) {
441         throw new BuildException("Unknown ivy.xml module directory: " + ivyXmlFile.getCanonicalPath());
442       }
443       String moduleDirPrefix = matcher.group(1).replaceAll("[/\\\\]", ".");
444       transitiveIvyXmlFile = new File(buildDir, "transitive." + moduleDirPrefix + ".ivy.xml");
445       try (Writer writer = new OutputStreamWriter(new FileOutputStream(transitiveIvyXmlFile), StandardCharsets.UTF_8)) {
446         writer.write(ivyXmlContent);
447       }
448       ResolveReport resolveReport = ivy.resolve(transitiveIvyXmlFile.toURI().toURL(), options);
449       IvyNodeElement root = IvyNodeElementAdapter.adapt(resolveReport);
450       for (IvyNodeElement directDependency : root.getDependencies()) {
451         String coordinate = "/" + directDependency.getOrganization() + "/" + directDependency.getName();
452         Dependency dependency = directDependencies.get(coordinate);
453         if (null == dependency) {
454           log("ERROR: the following coordinate key does not appear in " 
455               + centralizedVersionsFile.getName() + ": " + coordinate);
456           success = false;
457         } else {
458           dependency.directlyReferenced = true;
459           if (collectConflicts(directDependency, directDependency, moduleName)) {
460             success = false;
461           }
462         }
463       }
464     } catch (ParseException | IOException e) {
465       if (null != transitiveIvyXmlFile) {
466         log("Exception reading " + transitiveIvyXmlFile.getPath() + ": " + e.toString());
467       }
468       success = false;
469     }
470     return success;
471   }

473   /**
474    * Recursively finds indirect dependencies that have a version conflict with a direct dependency.
475    * Returns true if one or more conflicts are found, false otherwise
476    */
477   private boolean collectConflicts(IvyNodeElement root, IvyNodeElement parent, String moduleName) {
478     boolean conflicts = false;
479     for (IvyNodeElement child : parent.getDependencies()) {
480       String coordinate = "/" + child.getOrganization() + "/" + child.getName();
481       Dependency dependency = directDependencies.get(coordinate);
482       if (null != dependency) { // Ignore this indirect dependency if it's not also a direct dependency
483         String indirectVersion = child.getRevision();
484         if (isConflict(coordinate, dependency.directVersion, indirectVersion)) {
485           conflicts = true;
486           Set<String> moduleNames = dependency.conflictLocations.get(root);
487           if (null == moduleNames) {
488             moduleNames = new HashSet<>();
489             dependency.conflictLocations.put(root, moduleNames);
490           }
491           moduleNames.add(moduleName);
492         }
493         conflicts |= collectConflicts(root, child, moduleName);
494       }
495     }
496     return conflicts;
497   }

499   /**
500    * Copy-pasted from Ivy's 
501    * org.apache.ivy.plugins.latest.LatestRevisionStrategy
502    * with minor modifications
503    */
504   private static final Map<String,Integer> SPECIAL_MEANINGS;
505   static {
506     SPECIAL_MEANINGS = new HashMap<>();
507     SPECIAL_MEANINGS.put("dev", -1);
508     SPECIAL_MEANINGS.put("rc", 1);
509     SPECIAL_MEANINGS.put("final", 2);
510   }

512   /**
513    * Copy-pasted from Ivy's 
514    * org.apache.ivy.plugins.latest.LatestRevisionStrategy.MridComparator
515    * with minor modifications
516    */
517   private static class LatestVersionComparator implements Comparator<String> {
518     @Override
519     public int compare(String rev1, String rev2) {
520       rev1 = rev1.replaceAll("([a-zA-Z])(\\d)", "$1.$2");
521       rev1 = rev1.replaceAll("(\\d)([a-zA-Z])", "$1.$2");
522       rev2 = rev2.replaceAll("([a-zA-Z])(\\d)", "$1.$2");
523       rev2 = rev2.replaceAll("(\\d)([a-zA-Z])", "$1.$2");
524 
525       String[] parts1 = rev1.split("[-._+]");
526       String[] parts2 = rev2.split("[-._+]");
527 
528       int i = 0;
529       for (; i < parts1.length && i < parts2.length; i++) {
530         if (parts1[i].equals(parts2[i])) {
531           continue;
532         }
533         boolean is1Number = isNumber(parts1[i]);
534         boolean is2Number = isNumber(parts2[i]);
535         if (is1Number && !is2Number) {
536           return 1;
537         }
538         if (is2Number && !is1Number) {
539           return -1;
540         }
541         if (is1Number && is2Number) {
542           return Long.valueOf(parts1[i]).compareTo(Long.valueOf(parts2[i]));
543         }
544         // both are strings, we compare them taking into account special meaning
545         Integer sm1 = SPECIAL_MEANINGS.get(parts1[i].toLowerCase(Locale.ROOT));
546         Integer sm2 = SPECIAL_MEANINGS.get(parts2[i].toLowerCase(Locale.ROOT));
547         if (sm1 != null) {
548           sm2 = sm2 == null ? new Integer(0) : sm2;
549           return sm1.compareTo(sm2);
550         }
551         if (sm2 != null) {
552           return new Integer(0).compareTo(sm2);
553         }
554         return parts1[i].compareTo(parts2[i]);
555       }
556       if (i < parts1.length) {
557         return isNumber(parts1[i]) ? 1 : -1;
558       }
559       if (i < parts2.length) {
560         return isNumber(parts2[i]) ? -1 : 1;
561       }
562       return 0;
563     }
564 
565     private static final Pattern IS_NUMBER = Pattern.compile("\\d+");
566     private static boolean isNumber(String str) {
567       return IS_NUMBER.matcher(str).matches();
568     }
569   }
570   private static LatestVersionComparator LATEST_VERSION_COMPARATOR = new LatestVersionComparator();

572   /**
573    * Returns true if directVersion is less than indirectVersion, and 
574    * coordinate=indirectVersion is not present in ivy-ignore-conflicts.properties. 
575    */
576   private boolean isConflict(String coordinate, String directVersion, String indirectVersion) {
577     boolean isConflict = LATEST_VERSION_COMPARATOR.compare(directVersion, indirectVersion) < 0;
578     if (isConflict) {
579       Set<String> ignoredVersions = ignoreConflictVersions.get(coordinate);
580       if (null != ignoredVersions && ignoredVersions.contains(indirectVersion)) {
581         isConflict = false;
582       }
583     }
584     return isConflict;
585   }

587   /**
588    * Returns the number of direct dependencies in conflict with indirect
589    * dependencies.
590    */
591   private int emitConflicts() {
592     int conflicts = 0;
593     StringBuilder builder = new StringBuilder();
594     for (Map.Entry<String,Dependency> directDependency : directDependencies.entrySet()) {
595       String coordinate = directDependency.getKey();
596       Set<Map.Entry<IvyNodeElement,Set<String>>> entrySet
597           = directDependency.getValue().conflictLocations.entrySet();
598       if (entrySet.isEmpty()) {
599         continue;
600       }
601       ++conflicts;
602       Map.Entry<IvyNodeElement,Set<String>> first = entrySet.iterator().next();
603       int notPrinted = entrySet.size() - 1;
604       builder.append("VERSION CONFLICT: transitive dependency in module(s) ");
605       boolean isFirst = true;
606       for (String moduleName : first.getValue()) {
607         if (isFirst) {
608           isFirst = false;
609         } else {
610           builder.append(", ");
611         }
612         builder.append(moduleName);
613       }
614       builder.append(":\n");
615       IvyNodeElement element = first.getKey();
616       builder.append('/').append(element.getOrganization()).append('/').append(element.getName())
617              .append('=').append(element.getRevision()).append('\n');
618       emitConflict(builder, coordinate, first.getKey(), 1);
619         
620       if (notPrinted > 0) {
621         builder.append("... and ").append(notPrinted).append(" more\n");
622       }
623       builder.append("\n");
624     }
625     if (builder.length() > 0) {
626       log(builder.toString());
627     }
628     return conflicts;
629   }
630   
631   private boolean emitConflict(StringBuilder builder, String conflictCoordinate, IvyNodeElement parent, int depth) {
632     for (IvyNodeElement child : parent.getDependencies()) {
633       String indirectCoordinate = "/" + child.getOrganization() + "/" + child.getName();
634       if (conflictCoordinate.equals(indirectCoordinate)) {
635         Dependency dependency = directDependencies.get(conflictCoordinate);
636         String directVersion = dependency.directVersion;
637         if (isConflict(conflictCoordinate, directVersion, child.getRevision())) {
638           for (int i = 0 ; i < depth - 1 ; ++i) {
639             builder.append("    ");
640           }
641           builder.append("+-- ");
642           builder.append(indirectCoordinate).append("=").append(child.getRevision());
643           builder.append(" <<< Conflict (direct=").append(directVersion);
644           builder.append(", latest=").append(dependency.latestVersion).append(")\n");
645           return true;
646         }
647       } else if (hasConflicts(conflictCoordinate, child)) {
648         for (int i = 0 ; i < depth -1 ; ++i) {
649           builder.append("    ");
650         }
651         builder.append("+-- ");
652         builder.append(indirectCoordinate).append("=").append(child.getRevision()).append("\n");
653         if (emitConflict(builder, conflictCoordinate, child, depth + 1)) {
654           return true;
655         }
656       }
657     }
658     return false;
659   }
660   
661   private boolean hasConflicts(String conflictCoordinate, IvyNodeElement parent) {
662     // the element itself will never be in conflict, since its coordinate is different
663     for (IvyNodeElement child : parent.getDependencies()) {
664       String indirectCoordinate = "/" + child.getOrganization() + "/" + child.getName();
665       if (conflictCoordinate.equals(indirectCoordinate)) {
666         Dependency dependency = directDependencies.get(conflictCoordinate);
667         if (isConflict(conflictCoordinate, dependency.directVersion, child.getRevision())) {
668           return true;
669         }
670       } else if (hasConflicts(conflictCoordinate, child)) {
671         return true;
672       }
673     }
674     return false;
675   }

705   /**
706    * Returns true if the "/org/name" coordinate keys in the given
707    * properties file are lexically sorted and are not duplicates.
708    */
709   private boolean verifySortedCoordinatesPropertiesFile(File coordinatePropertiesFile) {
710     log("Checking for lexically sorted non-duplicated '/org/name' keys in: " + coordinatePropertiesFile, verboseLevel);
711     boolean success = true;
712     String line = null;
713     String currentKey = null;
714     String previousKey = null;
715     try (InputStream stream = new FileInputStream(coordinatePropertiesFile);
716          Reader reader = new InputStreamReader(stream, StandardCharsets.ISO_8859_1);
717          BufferedReader bufferedReader = new BufferedReader(reader)) {
718       while (null != (line = readLogicalPropertiesLine(bufferedReader))) {
719         final Matcher keyMatcher = COORDINATE_KEY_PATTERN.matcher(line);
720         if ( ! keyMatcher.lookingAt()) {
721           continue; // Ignore keys that don't look like "/org/name"
722         }
723         currentKey = keyMatcher.group(1);
724         if (null != previousKey) {
725           int comparison = currentKey.compareTo(previousKey);
726           if (0 == comparison) {
727             log("DUPLICATE coordinate key '" + currentKey + "' in " + coordinatePropertiesFile.getName(),
728                 Project.MSG_ERR);
729             success = false;
730           } else if (comparison < 0) {
731             log("OUT-OF-ORDER coordinate key '" + currentKey + "' in " + coordinatePropertiesFile.getName(),
732                 Project.MSG_ERR);
733             success = false;
734           }
735         }
736         previousKey = currentKey;
737       }
738     } catch (IOException e) {
739       throw new BuildException("Exception reading " + coordinatePropertiesFile.getPath() + ": " + e.toString(), e);
740     }
741     return success;
742   }

744   /**
745    * Builds up logical {@link java.util.Properties} lines, composed of one non-blank,
746    * non-comment initial line, either:
747    * 
748    * 1. without a non-escaped trailing slash; or
749    * 2. with a non-escaped trailing slash, followed by
750    *    zero or more lines with a non-escaped trailing slash, followed by
751    *    one or more lines without a non-escaped trailing slash
752    *
753    * All leading non-escaped whitespace and trailing non-escaped whitespace +
754    * non-escaped slash are trimmed from each line before concatenating.
755    * 
756    * After composing the logical line, escaped characters are un-escaped.
757    * 
758    * null is returned if there are no lines left to read. 
759    */
760   private String readLogicalPropertiesLine(BufferedReader reader) throws IOException {
761     final StringBuilder logicalLine = new StringBuilder();
762     String line;
763     do {
764       line = reader.readLine();
765       if (null == line) { 
766         return null;
767       }
768     } while (BLANK_OR_COMMENT_LINE_PATTERN.matcher(line).matches());
769 
770     Matcher backslashMatcher = TRAILING_BACKSLASH_PATTERN.matcher(line); 
771     // Check for a non-escaped backslash
772     if (backslashMatcher.find() && 1 == (backslashMatcher.group(1).length() % 2)) {
773       final Matcher firstLineMatcher = TRAILING_WHITESPACE_BACKSLASH_PATTERN.matcher(line);
774       if (firstLineMatcher.matches()) {
775         logicalLine.append(firstLineMatcher.group(1)); // trim trailing backslash and any preceding whitespace
776       }
777       line = reader.readLine();
778       while (null != line
779              && (backslashMatcher = TRAILING_BACKSLASH_PATTERN.matcher(line)).find()
780              && 1 == (backslashMatcher.group(1).length() % 2)) {
781         // Trim leading whitespace, the trailing backslash and any preceding whitespace
782         final Matcher goodStuffMatcher = WHITESPACE_GOODSTUFF_WHITESPACE_BACKSLASH_PATTERN.matcher(line);
783         if (goodStuffMatcher.matches()) {
784           logicalLine.append(goodStuffMatcher.group(1));
785         }
786         line = reader.readLine();
787       }
788       if (null != line) {
789         // line can't have a non-escaped trailing backslash
790         final Matcher leadingWhitespaceMatcher = LEADING_WHITESPACE_PATTERN.matcher(line);
791         if (leadingWhitespaceMatcher.matches()) {
792           line = leadingWhitespaceMatcher.group(1); // trim leading whitespace
793         }
794         logicalLine.append(line);
795       }
796     } else {
797       logicalLine.append(line);
798     }
799     // trim non-escaped leading whitespace
800     final Matcher leadingWhitespaceMatcher = LEADING_WHITESPACE_PATTERN.matcher(logicalLine);
801     final CharSequence leadingWhitespaceStripped = leadingWhitespaceMatcher.matches()
802                                                  ? leadingWhitespaceMatcher.group(1)
803                                                  : logicalLine;
804 
805     // unescape all chars in the logical line
806     StringBuilder output = new StringBuilder();
807     final int numChars = leadingWhitespaceStripped.length();
808     for (int pos = 0 ; pos < numChars - 1 ; ++pos) {
809       char ch = leadingWhitespaceStripped.charAt(pos);
810       if (ch == '\\') {
811         ch = leadingWhitespaceStripped.charAt(++pos);
812       }
813       output.append(ch);
814     }
815     if (numChars > 0) {
816       output.append(leadingWhitespaceStripped.charAt(numChars - 1));
817     }
818 
819     return output.toString();
820   }

822   /**
823    * Check a single ivy.xml file for dependencies' versions in rev="${/org/name}"
824    * format.  Returns false if problems are found, true otherwise.
825    */
826   private boolean checkIvyXmlFile(File ivyXmlFile)
827       throws ParserConfigurationException, SAXException, IOException {
828     log("Scanning: " + ivyXmlFile.getPath(), verboseLevel);
829     XMLReader xmlReader = XMLReaderFactory.createXMLReader();
830     DependencyRevChecker revChecker = new DependencyRevChecker(ivyXmlFile); 
831     xmlReader.setContentHandler(revChecker);
832     xmlReader.setErrorHandler(revChecker);
833     xmlReader.parse(new InputSource(ivyXmlFile.getAbsolutePath()));
834     return ! revChecker.fail;
835   }

837   private class DependencyRevChecker extends DefaultHandler {
838     private final File ivyXmlFile;
839     private final Stack<String> tags = new Stack<>();
840     
841     public boolean fail = false;
842 
843     public DependencyRevChecker(File ivyXmlFile) {
844       this.ivyXmlFile = ivyXmlFile;
845     }
846 
847     @Override
848     public void startElement(String uri, String localName, String qName, Attributes attributes) throws SAXException {
849       if (localName.equals("dependency") && insideDependenciesTag()) {
850         String org = attributes.getValue("org");
851         boolean foundAllAttributes = true;
852         if (null == org) {
853           log("MISSING 'org' attribute on <dependency> in " + ivyXmlFile.getPath(), Project.MSG_ERR);
854           fail = true;
855           foundAllAttributes = false;
856         }
857         String name = attributes.getValue("name");
858         if (null == name) {
859           log("MISSING 'name' attribute on <dependency> in " + ivyXmlFile.getPath(), Project.MSG_ERR);
860           fail = true;
861           foundAllAttributes = false;
862         }
863         String rev = attributes.getValue("rev");
864         if (null == rev) {
865           log("MISSING 'rev' attribute on <dependency> in " + ivyXmlFile.getPath(), Project.MSG_ERR);
866           fail = true;
867           foundAllAttributes = false;
868         }
869         if (foundAllAttributes) {
870           String coordinateKey = "/" + org + '/' + name;
871           String expectedRev = "${" + coordinateKey + '}';
872           if ( ! rev.equals(expectedRev)) {
873             log("BAD <dependency> 'rev' attribute value '" + rev + "' - expected '" + expectedRev + "'"
874                 + " in " + ivyXmlFile.getPath(), Project.MSG_ERR);
875             fail = true;
876           }
877           if ( ! directDependencies.containsKey(coordinateKey)) {
878             log("MISSING key '" + coordinateKey + "' in " + centralizedVersionsFile.getPath(), Project.MSG_ERR);
879             fail = true;
880           }
881         }
882       }
883       tags.push(localName);
884     }
885 
886     @Override
887     public void endElement (String uri, String localName, String qName) throws SAXException {
888       tags.pop();
889     }
890 
891     private boolean insideDependenciesTag() {
892       return tags.size() == 2 && tags.get(0).equals("ivy-module") && tags.get(1).equals("dependencies");
893     }
894   }
