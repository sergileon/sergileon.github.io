20  import java.io.IOException;
21  
22  import org.apache.lucene.analysis.Analyzer;
23  import org.apache.lucene.analysis.MockAnalyzer;
24  import org.apache.lucene.document.Document;
25  import org.apache.lucene.document.Field;
26  import org.apache.lucene.index.DirectoryReader;
27  import org.apache.lucene.index.IndexReader;
28  import org.apache.lucene.index.Term;
29  import org.apache.lucene.index.RandomIndexWriter;
30  import org.apache.lucene.search.*;
31  import org.apache.lucene.store.Directory;
32  import org.apache.lucene.util.LuceneTestCase;
33  
34  /**
35   * A very simple demo used in the API documentation (src/java/overview.html).
36   *
37   * Please try to keep src/java/overview.html up-to-date when making changes
38   * to this class.
39   */
40  public class TestDemo extends LuceneTestCase {
41  
42    public void testDemo() throws IOException {
43      Analyzer analyzer = new MockAnalyzer(random());
44  
45      // Store the index in memory:
46      Directory directory = newDirectory();
47      // To store an index on disk, use this instead:
48      // Directory directory = FSDirectory.open(new File("/tmp/testindex"));
49      RandomIndexWriter iwriter = new RandomIndexWriter(random(), directory, analyzer);
50      Document doc = new Document();
51      String longTerm = "longtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongtermlongterm";
52      String text = "This is the text to be indexed. " + longTerm;
53      doc.add(newTextField("fieldname", text, Field.Store.YES));
54      iwriter.addDocument(doc);
55      iwriter.close();
56      
57      // Now search the index:
58      IndexReader ireader = DirectoryReader.open(directory); // read-only=true
59      IndexSearcher isearcher = newSearcher(ireader);
60  
61      assertEquals(1, isearcher.search(new TermQuery(new Term("fieldname", longTerm)), 1).totalHits);
62      Query query = new TermQuery(new Term("fieldname", "text"));
63      TopDocs hits = isearcher.search(query, 1);
64      assertEquals(1, hits.totalHits);
65      // Iterate through the results:
66      for (int i = 0; i < hits.scoreDocs.length; i++) {
67        Document hitDoc = isearcher.doc(hits.scoreDocs[i].doc);
68        assertEquals(text, hitDoc.get("fieldname"));
69      }
70  
71      // Test simple phrase query
72      PhraseQuery phraseQuery = new PhraseQuery("fieldname", "to", "be");
73      assertEquals(1, isearcher.search(phraseQuery, 1).totalHits);
74  
75      ireader.close();
76      directory.close();
77      analyzer.close();
78    }
79  }
