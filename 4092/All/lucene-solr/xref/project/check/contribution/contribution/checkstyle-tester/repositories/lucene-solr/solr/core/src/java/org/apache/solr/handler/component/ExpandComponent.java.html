95  /**
96   * The ExpandComponent is designed to work with the CollapsingPostFilter.
97   * The CollapsingPostFilter collapses a result set on a field.
98   * <p>
99   * The ExpandComponent expands the collapsed groups for a single page.
100  * <p>
101  * http parameters:
102  * <p>
103  * expand=true <br>
104  * expand.rows=5 <br>
105  * expand.sort=field asc|desc<br>
106  * expand.q=*:* (optional, overrides the main query)<br>
107  * expand.fq=type:child (optional, overrides the main filter queries)<br>
108  * expand.field=field (mandatory if the not used with the CollapsingQParserPlugin)<br>
109  */
110 public class ExpandComponent extends SearchComponent implements PluginInfoInitialized, SolrCoreAware {
111   public static final String COMPONENT_NAME = "expand";
112   private static final int finishingStage = ResponseBuilder.STAGE_GET_FIELDS;
113   private PluginInfo info = PluginInfo.EMPTY_INFO;
114

132   @SuppressWarnings("unchecked")
133   @Override
134   public void process(ResponseBuilder rb) throws IOException {
135 
136     if (!rb.doExpand) {
137       return;
138     }
139 
140     SolrQueryRequest req = rb.req;
141     SolrParams params = req.getParams();
142 
143     String field = params.get(ExpandParams.EXPAND_FIELD);
144     String hint = null;
145     if (field == null) {
146       List<Query> filters = rb.getFilters();
147       if (filters != null) {
148         for (Query q : filters) {
149           if (q instanceof CollapsingQParserPlugin.CollapsingPostFilter) {
150             CollapsingQParserPlugin.CollapsingPostFilter cp = (CollapsingQParserPlugin.CollapsingPostFilter) q;
151             field = cp.getField();
152             hint = cp.hint;
153           }
154         }
155       }
156     }
157 
158     if (field == null) {
159       throw new IOException("Expand field is null.");
160     }
161 
162     String sortParam = params.get(ExpandParams.EXPAND_SORT);
163     String[] fqs = params.getParams(ExpandParams.EXPAND_FQ);
164     String qs = params.get(ExpandParams.EXPAND_Q);
165     int limit = params.getInt(ExpandParams.EXPAND_ROWS, 5);
166 
167     Sort sort = null;
168 
169     if (sortParam != null) {
170       sort = SortSpecParsing.parseSortSpec(sortParam, rb.req).getSort();
171     }
172 
173     Query query;
174     if (qs == null) {
175       query = rb.getQuery();
176     } else {
177       try {
178         QParser parser = QParser.getParser(qs, null, req);
179         query = parser.getQuery();
180       } catch (Exception e) {
181         throw new IOException(e);
182       }
183     }
184 
185     List<Query> newFilters = new ArrayList<>();
186 
187     if (fqs == null) {
188       List<Query> filters = rb.getFilters();
189       if (filters != null) {
190         for (Query q : filters) {
191           if (!(q instanceof CollapsingQParserPlugin.CollapsingPostFilter)) {
192             newFilters.add(q);
193           }
194         }
195       }
196     } else {
197       try {
198         for (String fq : fqs) {
199           if (fq != null && fq.trim().length() != 0 && !fq.equals("*:*")) {
200             QParser fqp = QParser.getParser(fq, null, req);
201             newFilters.add(fqp.getQuery());
202           }
203         }
204       } catch (Exception e) {
205         throw new IOException(e);
206       }
207     }
208 
209     SolrIndexSearcher searcher = req.getSearcher();
210     LeafReader reader = searcher.getLeafReader();
211 
212     FieldType fieldType = searcher.getSchema().getField(field).getType();
213 
214     SortedDocValues values = null;
215     long nullValue = 0;
216 
217     if(fieldType instanceof StrField) {
218       //Get The Top Level SortedDocValues
219       if(CollapsingQParserPlugin.HINT_TOP_FC.equals(hint)) {
220         Map<String, UninvertingReader.Type> mapping = new HashMap();
221         mapping.put(field, UninvertingReader.Type.SORTED);
222         UninvertingReader uninvertingReader = new UninvertingReader(new ReaderWrapper(searcher.getLeafReader(), field), mapping);
223         values = uninvertingReader.getSortedDocValues(field);
224       } else {
225         values = DocValues.getSorted(reader, field);
226       }
227     } else {
228       //Get the nullValue for the numeric collapse field
229       String defaultValue = searcher.getSchema().getField(field).getDefaultValue();
230       if(defaultValue != null) {
231         if(fieldType instanceof TrieIntField || fieldType instanceof TrieLongField) {
232           nullValue = Long.parseLong(defaultValue);
233         } else if(fieldType instanceof TrieFloatField){
234           nullValue = Float.floatToIntBits(Float.parseFloat(defaultValue));
235         } else if(fieldType instanceof TrieDoubleField){
236           nullValue = Double.doubleToLongBits(Double.parseDouble(defaultValue));
237         }
238       } else {
239         if(fieldType instanceof TrieFloatField){
240           nullValue = Float.floatToIntBits(0.0f);
241         } else if(fieldType instanceof TrieDoubleField){
242           nullValue = Double.doubleToLongBits(0.0f);
243         }
244       }
245     }
246 
247     FixedBitSet groupBits = null;
248     LongHashSet groupSet = null;
249     DocList docList = rb.getResults().docList;
250     IntHashSet collapsedSet = new IntHashSet(docList.size() * 2);
251 
252     //Gather the groups for the current page of documents
253     DocIterator idit = docList.iterator();
254     int[] globalDocs = new int[docList.size()];
255     int docsIndex = -1;
256     while (idit.hasNext()) {
257       globalDocs[++docsIndex] = idit.nextDoc();
258     }
259 
260     Arrays.sort(globalDocs);
261     Query groupQuery = null;
262 
263     /*
264     * This code gathers the group information for the current page.
265     */
266     List<LeafReaderContext> contexts = searcher.getTopReaderContext().leaves();
267     int currentContext = 0;
268     int currentDocBase = contexts.get(currentContext).docBase;
269     int nextDocBase = (currentContext+1)<contexts.size() ? contexts.get(currentContext+1).docBase : Integer.MAX_VALUE;
270     IntObjectHashMap<BytesRef> ordBytes = null;
271     if(values != null) {
272       groupBits = new FixedBitSet(values.getValueCount());
273       MultiDocValues.OrdinalMap ordinalMap = null;
274       SortedDocValues[] sortedDocValues = null;
275       LongValues segmentOrdinalMap = null;
276       SortedDocValues currentValues = null;
277       if(values instanceof  MultiDocValues.MultiSortedDocValues) {
278         ordinalMap = ((MultiDocValues.MultiSortedDocValues)values).mapping;
279         sortedDocValues = ((MultiDocValues.MultiSortedDocValues)values).values;
280         currentValues = sortedDocValues[currentContext];
281         segmentOrdinalMap = ordinalMap.getGlobalOrds(currentContext);
282       }
283       int count = 0;
284 
285       ordBytes = new IntObjectHashMap<>();
286 
287       for(int i=0; i<globalDocs.length; i++) {
288         int globalDoc = globalDocs[i];
289         while(globalDoc >= nextDocBase) {
290           currentContext++;
291           currentDocBase = contexts.get(currentContext).docBase;
292           nextDocBase = (currentContext+1) < contexts.size() ? contexts.get(currentContext+1).docBase : Integer.MAX_VALUE;
293           if(ordinalMap != null) {
294             currentValues = sortedDocValues[currentContext];
295             segmentOrdinalMap = ordinalMap.getGlobalOrds(currentContext);
296           }
297         }
298 
299         int contextDoc = globalDoc - currentDocBase;
300         if(ordinalMap != null) {
301           int ord = currentValues.getOrd(contextDoc);
302           if(ord > -1) {
303             ++count;
304             BytesRef ref = currentValues.lookupOrd(ord);
305             ord = (int)segmentOrdinalMap.get(ord);
306             ordBytes.put(ord, BytesRef.deepCopyOf(ref));
307             groupBits.set(ord);
308             collapsedSet.add(globalDoc);
309           }
310         } else {
311           int ord = values.getOrd(globalDoc);
312           if(ord > -1) {
313             ++count;
314             BytesRef ref = values.lookupOrd(ord);
315             ordBytes.put(ord, BytesRef.deepCopyOf(ref));
316             groupBits.set(ord);
317             collapsedSet.add(globalDoc);
318           }
319         }
320       }
321 
322       if(count > 0 && count < 200) {
323         try {
324           groupQuery = getGroupQuery(field, count, ordBytes);
325         } catch(Exception e) {
326           throw new IOException(e);
327         }
328       }
329     } else {
330       groupSet = new LongHashSet(docList.size());
331       NumericDocValues collapseValues = contexts.get(currentContext).reader().getNumericDocValues(field);
332       int count = 0;
333       for(int i=0; i<globalDocs.length; i++) {
334         int globalDoc = globalDocs[i];
335         while(globalDoc >= nextDocBase) {
336           currentContext++;
337           currentDocBase = contexts.get(currentContext).docBase;
338           nextDocBase = currentContext+1 < contexts.size() ? contexts.get(currentContext+1).docBase : Integer.MAX_VALUE;
339           collapseValues = contexts.get(currentContext).reader().getNumericDocValues(field);
340         }
341         int contextDoc = globalDoc - currentDocBase;
342         long value = collapseValues.get(contextDoc);
343         if(value != nullValue) {
344           ++count;
345           groupSet.add(value);
346           collapsedSet.add(globalDoc);
347         }
348       }
349 
350       if(count > 0 && count < 200) {
351         groupQuery = getGroupQuery(field, fieldType, count, groupSet);
352       }
353     }
354 
355     Collector collector;
356     if (sort != null)
357       sort = sort.rewrite(searcher);
358 
359 
360     Collector groupExpandCollector = null;
361 
362     if(values != null) {
363       groupExpandCollector = new GroupExpandCollector(values, groupBits, collapsedSet, limit, sort);
364     } else {
365       groupExpandCollector = new NumericGroupExpandCollector(field, nullValue, groupSet, collapsedSet, limit, sort);
366     }
367 
368     if(groupQuery !=  null) {
369       //Limits the results to documents that are in the same group as the documents in the page.
370       newFilters.add(groupQuery);
371     }
372 
373     SolrIndexSearcher.ProcessedFilter pfilter = searcher.getProcessedFilter(null, newFilters);
374     if (pfilter.postFilter != null) {
375       pfilter.postFilter.setLastDelegate(groupExpandCollector);
376       collector = pfilter.postFilter;
377     } else {
378       collector = groupExpandCollector;
379     }
380 
381     if (pfilter.filter == null) {
382       searcher.search(query, collector);
383     } else {
384       searcher.search(new FilteredQuery(query, pfilter.filter), collector);
385     }
386     LongObjectMap<Collector> groups = ((GroupCollector) groupExpandCollector).getGroups();
387     NamedList outMap = new SimpleOrderedMap();
388     CharsRefBuilder charsRef = new CharsRefBuilder();
389     for (LongObjectCursor<Collector> cursor : groups) {
390       long groupValue = cursor.key;
391       TopDocsCollector<?> topDocsCollector = TopDocsCollector.class.cast(cursor.value);
392       TopDocs topDocs = topDocsCollector.topDocs();
393       ScoreDoc[] scoreDocs = topDocs.scoreDocs;
394       if (scoreDocs.length > 0) {
395         int[] docs = new int[scoreDocs.length];
396         float[] scores = new float[scoreDocs.length];
397         for (int i = 0; i < docs.length; i++) {
398           ScoreDoc scoreDoc = scoreDocs[i];
399           docs[i] = scoreDoc.doc;
400           scores[i] = scoreDoc.score;
401         }
402         DocSlice slice = new DocSlice(0, docs.length, docs, scores, topDocs.totalHits, topDocs.getMaxScore());
403 
404         if(fieldType instanceof StrField) {
405           final BytesRef bytesRef = ordBytes.get((int)groupValue);
406           fieldType.indexedToReadable(bytesRef, charsRef);
407           String group = charsRef.toString();
408           outMap.add(group, slice);
409         } else {
410           if(fieldType instanceof TrieIntField || fieldType instanceof TrieLongField ) {
411             outMap.add(Long.toString(groupValue), slice);
412           } else if(fieldType instanceof TrieFloatField) {
413             outMap.add(Float.toString(Float.intBitsToFloat((int) groupValue)), slice);
414           } else if(fieldType instanceof TrieDoubleField) {
415             outMap.add(Double.toString(Double.longBitsToDouble(groupValue)), slice);
416           }
417         }
418       }
419     }
420 
421     rb.rsp.add("expanded", outMap);
422   }

424   @Override
425   public int distributedProcess(ResponseBuilder rb) throws IOException {
426     if (rb.doExpand && rb.stage < finishingStage) {
427       return finishingStage;
428     }
429     return ResponseBuilder.STAGE_DONE;
430   }
431     
432   @Override
433   public void modifyRequest(ResponseBuilder rb, SearchComponent who, ShardRequest sreq) {
434     SolrParams params = rb.req.getParams();
435     if (!params.getBool(COMPONENT_NAME, false)) return;
436     if (!rb.onePassDistributedQuery && (sreq.purpose & ShardRequest.PURPOSE_GET_FIELDS) == 0) {
437       sreq.params.set(COMPONENT_NAME, "false");
438     } else {
439       sreq.params.set(COMPONENT_NAME, "true");
440     }
441   }
442 
443   @SuppressWarnings("unchecked")
444   @Override
445   public void handleResponses(ResponseBuilder rb, ShardRequest sreq) {
446 
447     if (!rb.doExpand) {
448       return;
449     }
450 
451     if ((sreq.purpose & ShardRequest.PURPOSE_GET_FIELDS) != 0) {
452       SolrQueryRequest req = rb.req;
453       NamedList expanded = (NamedList) req.getContext().get("expanded");
454       if (expanded == null) {
455         expanded = new SimpleOrderedMap();
456         req.getContext().put("expanded", expanded);
457       }
458 
459       for (ShardResponse srsp : sreq.responses) {
460         NamedList response = srsp.getSolrResponse().getResponse();
461         NamedList ex = (NamedList) response.get("expanded");
462         for (int i=0; i<ex.size(); i++) {
463           String name = ex.getName(i);
464           SolrDocumentList val = (SolrDocumentList) ex.getVal(i);
465           expanded.add(name, val);
466         }
467       }
468     }
469   }
470 
471   @Override
472   public void finishStage(ResponseBuilder rb) {
473 
474     if (!rb.doExpand) {
475       return;
476     }
477 
478     if (rb.stage != finishingStage) {
479       return;
480     }
481 
482     NamedList expanded = (NamedList) rb.req.getContext().get("expanded");
483     if (expanded == null) {
484       expanded = new SimpleOrderedMap();
485     }
486 
487     rb.rsp.add("expanded", expanded);
488   }

490   private class GroupExpandCollector implements Collector, GroupCollector {
491     private SortedDocValues docValues;
492     private MultiDocValues.OrdinalMap ordinalMap;
493     private SortedDocValues segmentValues;
494     private LongValues segmentOrdinalMap;
495     private MultiDocValues.MultiSortedDocValues multiSortedDocValues;
496 
497     private LongObjectMap<Collector> groups;
498     private FixedBitSet groupBits;
499     private IntHashSet collapsedSet;
500 
501     public GroupExpandCollector(SortedDocValues docValues, FixedBitSet groupBits, IntHashSet collapsedSet, int limit, Sort sort) throws IOException {
502       int numGroups = collapsedSet.size();
503       groups = new LongObjectHashMap<>(numGroups);
504       DocIdSetIterator iterator = new BitSetIterator(groupBits, 0); // cost is not useful here
505       int group;
506       while ((group = iterator.nextDoc()) != DocIdSetIterator.NO_MORE_DOCS) {
507         Collector collector = (sort == null) ? TopScoreDocCollector.create(limit) : TopFieldCollector.create(sort, limit, false, false, false);
508         groups.put(group, collector);
509       }
510 
511       this.collapsedSet = collapsedSet;
512       this.groupBits = groupBits;
513       this.docValues = docValues;
514       if(docValues instanceof MultiDocValues.MultiSortedDocValues) {
515         this.multiSortedDocValues = (MultiDocValues.MultiSortedDocValues)docValues;
516         this.ordinalMap = multiSortedDocValues.mapping;
517       }
518     }
519 
520     @Override
521     public boolean needsScores() {
522       return true; // TODO: is this always true?
523     }
524 
525     public LeafCollector getLeafCollector(LeafReaderContext context) throws IOException {
526       final int docBase = context.docBase;
527 
528       if(ordinalMap != null) {
529         this.segmentValues = this.multiSortedDocValues.values[context.ord];
530         this.segmentOrdinalMap = ordinalMap.getGlobalOrds(context.ord);
531       }
532 
533       final LongObjectMap<LeafCollector> leafCollectors = new LongObjectHashMap<>();
534       for (LongObjectCursor<Collector> entry : groups) {
535         leafCollectors.put(entry.key, entry.value.getLeafCollector(context));
536       }
537       return new LeafCollector() {
538 
539         @Override
540         public void setScorer(Scorer scorer) throws IOException {
541           for (ObjectCursor<LeafCollector> c : leafCollectors.values()) {
542             c.value.setScorer(scorer);
543           }
544         }
545 
546         @Override
547         public void collect(int docId) throws IOException {
548           int globalDoc = docId + docBase;
549           int ord = -1;
550           if(ordinalMap != null) {
551             ord = segmentValues.getOrd(docId);
552             if(ord > -1) {
553               ord = (int)segmentOrdinalMap.get(ord);
554             }
555           } else {
556             ord = docValues.getOrd(globalDoc);
557           }
558 
559           if (ord > -1 && groupBits.get(ord) && !collapsedSet.contains(globalDoc)) {
560             LeafCollector c = leafCollectors.get(ord);
561             c.collect(docId);
562           }
563         }
564       };
565     }
566 
567     public LongObjectMap<Collector> getGroups() {
568       return groups;
569     }
570   }

572   private class NumericGroupExpandCollector implements Collector, GroupCollector {
573     private NumericDocValues docValues;
574 
575     private String field;
576     private LongObjectHashMap<Collector> groups;
577 
578     private IntHashSet collapsedSet;
579     private long nullValue;
580 
581     public NumericGroupExpandCollector(String field, long nullValue, LongHashSet groupSet, IntHashSet collapsedSet, int limit, Sort sort) throws IOException {
582       int numGroups = collapsedSet.size();
583       this.nullValue = nullValue;
584       groups = new LongObjectHashMap<>(numGroups);
585       Iterator<LongCursor> iterator = groupSet.iterator();
586       while (iterator.hasNext()) {
587         LongCursor cursor = iterator.next();
588         Collector collector = (sort == null) ? TopScoreDocCollector.create(limit) : TopFieldCollector.create(sort, limit, false, false, false);
589         groups.put(cursor.value, collector);
590       }
591 
592       this.field = field;
593       this.collapsedSet = collapsedSet;
594     }
595     
596     @Override
597     public boolean needsScores() {
598       return true; // TODO: is this always true?
599     }
600 
601     public LeafCollector getLeafCollector(LeafReaderContext context) throws IOException {
602       final int docBase = context.docBase;
603       this.docValues = context.reader().getNumericDocValues(this.field);
604 
605       final LongObjectHashMap<LeafCollector> leafCollectors = new LongObjectHashMap<>();
606 
607       for (LongObjectCursor<Collector> entry : groups) {
608         leafCollectors.put(entry.key, entry.value.getLeafCollector(context));
609       }
610 
611       return new LeafCollector() {
612 
613         @Override
614         public void setScorer(Scorer scorer) throws IOException {
615           for (ObjectCursor<LeafCollector> c : leafCollectors.values()) {
616             c.value.setScorer(scorer);
617           }
618         }
619 
620         @Override
621         public void collect(int docId) throws IOException {
622           long value = docValues.get(docId);
623           final int index;
624           if (value != nullValue && 
625               (index = leafCollectors.indexOf(value)) >= 0 && 
626               !collapsedSet.contains(docId + docBase)) {
627             leafCollectors.indexGet(index).collect(docId);
628           }
629         }
630       };
631     }
632 
633     public LongObjectHashMap<Collector> getGroups() {
634       return groups;
635     }

643   private Query getGroupQuery(String fname,
644                            FieldType ft,
645                            int size,
646                            LongHashSet groupSet) {
647 
648     BytesRef[] bytesRefs = new BytesRef[size];
649     BytesRefBuilder term = new BytesRefBuilder();
650     Iterator<LongCursor> it = groupSet.iterator();
651     int index = -1;
652     String stringVal =  null;
653     while (it.hasNext()) {
654       LongCursor cursor = it.next();
655       if(ft instanceof TrieIntField || ft instanceof TrieLongField) {
656         stringVal = Long.toString(cursor.value);
657       } else {
658         if(ft instanceof TrieFloatField) {
659           stringVal = Float.toString(Float.intBitsToFloat((int)cursor.value));
660         } else {
661           stringVal = Double.toString(Double.longBitsToDouble(cursor.value));
662         }
663       }
664       ft.readableToIndexed(stringVal, term);
665       bytesRefs[++index] = term.toBytesRef();
666     }
667 
668     return new SolrConstantScoreQuery(new QueryWrapperFilter(new TermsQuery(fname, bytesRefs)));
669   }

681     return new SolrConstantScoreQuery(new QueryWrapperFilter(new TermsQuery(fname, bytesRefs)));
682   }
