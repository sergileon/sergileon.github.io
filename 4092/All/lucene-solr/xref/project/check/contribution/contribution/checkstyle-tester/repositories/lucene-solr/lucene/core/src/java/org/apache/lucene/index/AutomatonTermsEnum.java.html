22  import org.apache.lucene.util.BytesRef;
23  import org.apache.lucene.util.BytesRefBuilder;
24  import org.apache.lucene.util.IntsRefBuilder;
25  import org.apache.lucene.util.StringHelper;

31  /**
32   * A FilteredTermsEnum that enumerates terms based upon what is accepted by a
33   * DFA.
34   * <p>
35   * The algorithm is such:
36   * <ol>
37   *   <li>As long as matches are successful, keep reading sequentially.
38   *   <li>When a match fails, skip to the next string in lexicographic order that
39   * does not enter a reject state.
40   * </ol>
41   * <p>
42   * The algorithm does not attempt to actually skip to the next string that is
43   * completely accepted. This is not possible when the language accepted by the
44   * FSM is not finite (i.e. * operator).
45   * </p>
46   * @lucene.internal
47   */
48  public class AutomatonTermsEnum extends FilteredTermsEnum {
49    // a tableized array-based form of the DFA
50    private final ByteRunAutomaton runAutomaton;
51    // common suffix of the automaton
52    private final BytesRef commonSuffixRef;
53    // true if the automaton accepts a finite language
54    private final boolean finite;
55    // array of sorted transitions for each state, indexed by state number
56    private final Automaton automaton;
57    // for path tracking: each long records gen when we last
58    // visited the state; we use gens to avoid having to clear
59    private final long[] visited;
60    private long curGen;
61    // the reference used for seeking forwards through the term dictionary
62    private final BytesRefBuilder seekBytesRef = new BytesRefBuilder(); 
63    // true if we are enumerating an infinite portion of the DFA.
64    // in this case it is faster to drive the query based on the terms dictionary.
65    // when this is true, linearUpperBound indicate the end of range
66    // of terms where we should simply do sequential reads instead.
67    private boolean linear = false;
68    private final BytesRef linearUpperBound = new BytesRef(10);
69  
70    /**
71     * Construct an enumerator based upon an automaton, enumerating the specified
72     * field, working on a supplied TermsEnum
73     *
74     * @lucene.experimental 
75     * @param compiled CompiledAutomaton
76     */
77    public AutomatonTermsEnum(TermsEnum tenum, CompiledAutomaton compiled) {
78      super(tenum);
79      this.finite = compiled.finite;
80      this.runAutomaton = compiled.runAutomaton;
81      assert this.runAutomaton != null;
82      this.commonSuffixRef = compiled.commonSuffixRef;
83      this.automaton = compiled.automaton;
84  
85      // used for path tracking, where each bit is a numbered state.
86      visited = new long[runAutomaton.getSize()];
87    }
88    
89    /**
90     * Returns true if the term matches the automaton. Also stashes away the term
91     * to assist with smart enumeration.
92     */
93    @Override
94    protected AcceptStatus accept(final BytesRef term) {
95      if (commonSuffixRef == null || StringHelper.endsWith(term, commonSuffixRef)) {
96        if (runAutomaton.run(term.bytes, term.offset, term.length))
97          return linear ? AcceptStatus.YES : AcceptStatus.YES_AND_SEEK;
98        else
99          return (linear && term.compareTo(linearUpperBound) < 0) ? 
100             AcceptStatus.NO : AcceptStatus.NO_AND_SEEK;
101     } else {
102       return (linear && term.compareTo(linearUpperBound) < 0) ? 
103           AcceptStatus.NO : AcceptStatus.NO_AND_SEEK;
104     }
105   }
106   
107   @Override
108   protected BytesRef nextSeekTerm(final BytesRef term) throws IOException {
109     //System.out.println("ATE.nextSeekTerm term=" + term);
110     if (term == null) {
111       assert seekBytesRef.length() == 0;
112       // return the empty term, as it's valid
113       if (runAutomaton.isAccept(runAutomaton.getInitialState())) {   
114         return seekBytesRef.get();
115       }
116     } else {
117       seekBytesRef.copyBytes(term);
118     }
119 
120     // seek to the next possible string;
121     if (nextString()) {
122       return seekBytesRef.get();  // reposition
123     } else {
124       return null;          // no more possible strings can match
125     }
126   }
127 
128   private Transition transition = new Transition();
129 
130   /**
131    * Sets the enum to operate in linear fashion, as we have found
132    * a looping transition at position: we set an upper bound and 
133    * act like a TermRangeQuery for this portion of the term space.
134    */
135   private void setLinear(int position) {
136     assert linear == false;
137     
138     int state = runAutomaton.getInitialState();
139     assert state == 0;
140     int maxInterval = 0xff;
141     //System.out.println("setLinear pos=" + position + " seekbytesRef=" + seekBytesRef);
142     for (int i = 0; i < position; i++) {
143       state = runAutomaton.step(state, seekBytesRef.byteAt(i) & 0xff);
144       assert state >= 0: "state=" + state;
145     }
146     final int numTransitions = automaton.getNumTransitions(state);
147     automaton.initTransition(state, transition);
148     for (int i = 0; i < numTransitions; i++) {
149       automaton.getNextTransition(transition);
150       if (transition.min <= (seekBytesRef.byteAt(position) & 0xff) && 
151           (seekBytesRef.byteAt(position) & 0xff) <= transition.max) {
152         maxInterval = transition.max;
153         break;
154       }
155     }
156     // 0xff terms don't get the optimization... not worth the trouble.
157     if (maxInterval != 0xff)
158       maxInterval++;
159     int length = position + 1; /* position + maxTransition */
160     if (linearUpperBound.bytes.length < length)
161       linearUpperBound.bytes = new byte[length];
162     System.arraycopy(seekBytesRef.bytes(), 0, linearUpperBound.bytes, 0, position);
163     linearUpperBound.bytes[position] = (byte) maxInterval;
164     linearUpperBound.length = length;
165     
166     linear = true;
167   }
168 
169   private final IntsRefBuilder savedStates = new IntsRefBuilder();
170   
171   /**
172    * Increments the byte buffer to the next String in binary order after s that will not put
173    * the machine into a reject state. If such a string does not exist, returns
174    * false.
175    * 
176    * The correctness of this method depends upon the automaton being deterministic,
177    * and having no transitions to dead states.
178    * 
179    * @return true if more possible solutions exist for the DFA
180    */
181   private boolean nextString() {
182     int state;
183     int pos = 0;
184     savedStates.grow(seekBytesRef.length()+1);
185     savedStates.setIntAt(0, runAutomaton.getInitialState());
186     
187     while (true) {
188       curGen++;
189       linear = false;
190       // walk the automaton until a character is rejected.
191       for (state = savedStates.intAt(pos); pos < seekBytesRef.length(); pos++) {
192         visited[state] = curGen;
193         int nextState = runAutomaton.step(state, seekBytesRef.byteAt(pos) & 0xff);
194         if (nextState == -1)
195           break;
196         savedStates.setIntAt(pos+1, nextState);
197         // we found a loop, record it for faster enumeration
198         if (!finite && !linear && visited[nextState] == curGen) {
199           setLinear(pos);
200         }
201         state = nextState;
202       }
203 
204       // take the useful portion, and the last non-reject state, and attempt to
205       // append characters that will match.
206       if (nextString(state, pos)) {
207         return true;
208       } else { /* no more solutions exist from this useful portion, backtrack */
209         if ((pos = backtrack(pos)) < 0) /* no more solutions at all */
210           return false;
211         final int newState = runAutomaton.step(savedStates.intAt(pos), seekBytesRef.byteAt(pos) & 0xff);
212         if (newState >= 0 && runAutomaton.isAccept(newState))
213           /* String is good to go as-is */
214           return true;
215         /* else advance further */
216         // TODO: paranoia? if we backtrack thru an infinite DFA, the loop detection is important!
217         // for now, restart from scratch for all infinite DFAs 
218         if (!finite) pos = 0;
219       }
220     }
221   }
222   
223   /**
224    * Returns the next String in lexicographic order that will not put
225    * the machine into a reject state. 
226    * 
227    * This method traverses the DFA from the given position in the String,
228    * starting at the given state.
229    * 
230    * If this cannot satisfy the machine, returns false. This method will
231    * walk the minimal path, in lexicographic order, as long as possible.
232    * 
233    * If this method returns false, then there might still be more solutions,
234    * it is necessary to backtrack to find out.
235    * 
236    * @param state current non-reject state
237    * @param position useful portion of the string
238    * @return true if more possible solutions exist for the DFA from this
239    *         position
240    */
241   private boolean nextString(int state, int position) {
242     /* 
243      * the next lexicographic character must be greater than the existing
244      * character, if it exists.
245      */
246     int c = 0;
247     if (position < seekBytesRef.length()) {
248       c = seekBytesRef.byteAt(position) & 0xff;
249       // if the next byte is 0xff and is not part of the useful portion,
250       // then by definition it puts us in a reject state, and therefore this
251       // path is dead. there cannot be any higher transitions. backtrack.
252       if (c++ == 0xff)
253         return false;
254     }
255 
256     seekBytesRef.setLength(position);
257     visited[state] = curGen;
258 
259     final int numTransitions = automaton.getNumTransitions(state);
260     automaton.initTransition(state, transition);
261     // find the minimal path (lexicographic order) that is >= c
262     
263     for (int i = 0; i < numTransitions; i++) {
264       automaton.getNextTransition(transition);
265       if (transition.max >= c) {
266         int nextChar = Math.max(c, transition.min);
267         // append either the next sequential char, or the minimum transition
268         seekBytesRef.grow(seekBytesRef.length() + 1);
269         seekBytesRef.append((byte) nextChar);
270         state = transition.dest;
271         /* 
272          * as long as is possible, continue down the minimal path in
273          * lexicographic order. if a loop or accept state is encountered, stop.
274          */
275         while (visited[state] != curGen && !runAutomaton.isAccept(state)) {
276           visited[state] = curGen;
277           /* 
278            * Note: we work with a DFA with no transitions to dead states.
279            * so the below is ok, if it is not an accept state,
280            * then there MUST be at least one transition.
281            */
282           automaton.initTransition(state, transition);
283           automaton.getNextTransition(transition);
284           state = transition.dest;
285           
286           // append the minimum transition
287           seekBytesRef.grow(seekBytesRef.length() + 1);
288           seekBytesRef.append((byte) transition.min);
289           
290           // we found a loop, record it for faster enumeration
291           if (!finite && !linear && visited[state] == curGen) {
292             setLinear(seekBytesRef.length()-1);
293           }
294         }
295         return true;
296       }
297     }
298     return false;
299   }
300   
301   /**
302    * Attempts to backtrack thru the string after encountering a dead end
303    * at some given position. Returns false if no more possible strings 
304    * can match.
305    * 
306    * @param position current position in the input String
307    * @return {@code position >= 0} if more possible solutions exist for the DFA
308    */
309   private int backtrack(int position) {
310     while (position-- > 0) {
311       int nextChar = seekBytesRef.byteAt(position) & 0xff;
312       // if a character is 0xff it's a dead-end too,
313       // because there is no higher character in binary sort order.
314       if (nextChar++ != 0xff) {
315         seekBytesRef.setByteAt(position, (byte) nextChar);
316         seekBytesRef.setLength(position+1);
317         return position;
318       }
319     }
320     return -1; /* all solutions exhausted */
321   }
