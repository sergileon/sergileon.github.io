43  /**
44   * This class is a helper that enables users to easily use the Lucene query
45   * parser.
46   * <p>
47   * To construct a Query object from a query string, use the
48   * {@link #parse(String, String)} method:
49   * <pre class="prettyprint">
50   * StandardQueryParser queryParserHelper = new StandardQueryParser();
51   * Query query = queryParserHelper.parse("a AND b", "defaultField");
52   * </pre>
53   * <p>
54   * To change any configuration before parsing the query string do, for example:
55   * <br>
56   * <pre class="prettyprint">
57   * // the query config handler returned by {@link StandardQueryParser} is a {@link StandardQueryConfigHandler}
58   * queryParserHelper.getQueryConfigHandler().setAnalyzer(new WhitespaceAnalyzer());
59   * </pre>
60   * <p>
61   * The syntax for query strings is as follows (copied from the old QueryParser
62   * javadoc):
63   * A Query is a series of clauses. A clause may be prefixed by:
64   * <ul>
65   * <li>a plus (<code>+</code>) or a minus (<code>-</code>) sign, indicating that
66   * the clause is required or prohibited respectively; or
67   * <li>a term followed by a colon, indicating the field to be searched. This
68   * enables one to construct queries which search multiple fields.
69   * </ul>
70   * 
71   * A clause may be either:
72   * <ul>
73   * <li>a term, indicating all the documents that contain this term; or
74   * <li>a nested query, enclosed in parentheses. Note that this may be used with
75   * a <code>+</code>/<code>-</code> prefix to require any of a set of terms.
76   * </ul>
77   * 
78   * Thus, in BNF, the query grammar is:
79   * 
80   * <pre>
81   *   Query  ::= ( Clause )*
82   *   Clause ::= [&quot;+&quot;, &quot;-&quot;] [&lt;TERM&gt; &quot;:&quot;] ( &lt;TERM&gt; | &quot;(&quot; Query &quot;)&quot; )
83   * </pre>
84   * 
85   * <p>
86   * Examples of appropriately formatted queries can be found in the <a
87   * href="{@docRoot}/org/apache/lucene/queryparser/classic/package-summary.html#package_description">
88   * query syntax documentation</a>.
89   * </p>
90   * <p>
91   * The text parser used by this helper is a {@link StandardSyntaxParser}.
92   * <p>
93   * The query node processor used by this helper is a
94   * {@link StandardQueryNodeProcessorPipeline}.
95   * <p>
96   * The builder used by this helper is a {@link StandardQueryTreeBuilder}.
97   * 
98   * @see StandardQueryParser
99   * @see StandardQueryConfigHandler
100  * @see StandardSyntaxParser
101  * @see StandardQueryNodeProcessorPipeline
102  * @see StandardQueryTreeBuilder
103  */
104 public class StandardQueryParser extends QueryParserHelper implements CommonQueryParserConfiguration {
105   
106   /**
107    * Constructs a {@link StandardQueryParser} object.
108    */
109   public StandardQueryParser() {
110     super(new StandardQueryConfigHandler(), new StandardSyntaxParser(),
111         new StandardQueryNodeProcessorPipeline(null),
112         new StandardQueryTreeBuilder());
113     setEnablePositionIncrements(true);
114   }

122    * qp.getQueryConfigHandler().setAnalyzer(analyzer);

126    *          the analyzer to be used by this query parser helper

140   /**

142    * return object to {@link Query}. For more reference about this method, check
143    * {@link QueryParserHelper#parse(String, String)}.
144    * 
145    * @param query
146    *          the query string
147    * @param defaultField
148    *          the default field used by the text parser
149    * 
150    * @return the object built from the query
151    * 
152    * @throws QueryNodeException
153    *           if something wrong happens along the three phases
154    */
155   @Override
156   public Query parse(String query, String defaultField)
157       throws QueryNodeException {
158     
159     return (Query) super.parse(query, defaultField);

163   /**
164    * Gets implicit operator setting, which will be either {@link Operator#AND}
165    * or {@link Operator#OR}.
166    */
167   public StandardQueryConfigHandler.Operator getDefaultOperator() {
168     return getQueryConfigHandler().get(ConfigurationKeys.DEFAULT_OPERATOR);
169   }

171   /**
172    * Sets the boolean operator of the QueryParser. In default mode (
173    * {@link Operator#OR}) terms without any modifiers are considered optional:
174    * for example <code>capital of Hungary</code> is equal to
175    * <code>capital OR of OR Hungary</code>.<br>
176    * In {@link Operator#AND} mode terms are considered to be in conjunction: the
177    * above mentioned query is parsed as <code>capital AND of AND Hungary</code>
178    */
179   public void setDefaultOperator(StandardQueryConfigHandler.Operator operator) {
180     getQueryConfigHandler().set(ConfigurationKeys.DEFAULT_OPERATOR, operator);
181   }

183   /**
184    * Set to <code>true</code> to allow leading wildcard characters.
185    * <p>
186    * When set, <code>*</code> or <code>?</code> are allowed as the first
187    * character of a PrefixQuery and WildcardQuery. Note that this can produce
188    * very slow queries on big indexes.
189    * <p>
190    * Default: false.
191    */
192   @Override
193   public void setLowercaseExpandedTerms(boolean lowercaseExpandedTerms) {
194     getQueryConfigHandler().set(ConfigurationKeys.LOWERCASE_EXPANDED_TERMS, lowercaseExpandedTerms);
195   }

198    * @see #setLowercaseExpandedTerms(boolean)
199    */

202     Boolean lowercaseExpandedTerms = getQueryConfigHandler().get(ConfigurationKeys.LOWERCASE_EXPANDED_TERMS);
203     
204     if (lowercaseExpandedTerms == null) {
205       return true;
206       
207     } else {
208       return lowercaseExpandedTerms;

257   /**
258    * By default, it uses
259    * {@link MultiTermQuery#CONSTANT_SCORE_REWRITE} when creating a
260    * prefix, wildcard and range queries. This implementation is generally
261    * preferable because it a) Runs faster b) Does not have the scarcity of terms
262    * unduly influence score c) avoids any {@link TooManyListenersException}
263    * exception. However, if your application really needs to use the
264    * old-fashioned boolean queries expansion rewriting and the above points are
265    * not relevant then use this change the rewrite method.
266    */
267   @Override
268   public void setMultiTermRewriteMethod(MultiTermQuery.RewriteMethod method) {
269     getQueryConfigHandler().set(ConfigurationKeys.MULTI_TERM_REWRITE_METHOD, method);
270   }

273    * @see #setMultiTermRewriteMethod(org.apache.lucene.search.MultiTermQuery.RewriteMethod)
274    */
275   @Override
276   public MultiTermQuery.RewriteMethod getMultiTermRewriteMethod() {
277     return getQueryConfigHandler().get(ConfigurationKeys.MULTI_TERM_REWRITE_METHOD);
278   }

318   /**
319    * Set the prefix length for fuzzy queries. Default is 0.
320    * 
321    * @param fuzzyPrefixLength
322    *          The fuzzyPrefixLength to set.
323    */
324   @Override
325   public void setFuzzyPrefixLength(int fuzzyPrefixLength) {
326     QueryConfigHandler config = getQueryConfigHandler();
327     FuzzyConfig fuzzyConfig = config.get(ConfigurationKeys.FUZZY_CONFIG);
328     
329     if (fuzzyConfig == null) {
330       fuzzyConfig = new FuzzyConfig();
331       config.set(ConfigurationKeys.FUZZY_CONFIG, fuzzyConfig);
332     }
333 
334     fuzzyConfig.setPrefixLength(fuzzyPrefixLength);
335     
336   }
