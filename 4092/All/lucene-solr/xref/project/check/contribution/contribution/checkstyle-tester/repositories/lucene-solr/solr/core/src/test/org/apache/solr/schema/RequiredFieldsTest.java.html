18  package org.apache.solr.schema;
19  
20  import java.util.Collection;
21  
22  import org.apache.solr.SolrTestCaseJ4;
23  import org.apache.solr.core.SolrCore;
24  import org.junit.BeforeClass;
25  import org.junit.Test;
26  /**
27   * 
28   */
29  public class RequiredFieldsTest extends SolrTestCaseJ4 {
30    @BeforeClass
31    public static void beforeClass() throws Exception {
32      initCore("solrconfig.xml","schema-required-fields.xml");
33    }
34    
35    @Override
36    public void setUp() throws Exception {
37      super.setUp();
38      clearIndex();
39    }
40    
41    @Test
42    public void testRequiredFieldsConfig() {
43      SolrCore core = h.getCore();
44      IndexSchema schema = core.getLatestSchema();
45      SchemaField uniqueKey = schema.getUniqueKeyField();
46  
47      // Make sure the uniqueKey is required
48      assertTrue( uniqueKey.isRequired() ); 
49      assertTrue( schema.getRequiredFields().contains( uniqueKey ) );
50      
51      // we specified one required field, but all devault valued fields are also required
52      Collection<SchemaField> requiredFields =schema.getRequiredFields();
53      int numDefaultFields = schema.getFieldsWithDefaultValue().size();
54      assertEquals( numDefaultFields+1+1, requiredFields.size()); // also the uniqueKey
55    }
56    
57    @Test
58    public void testRequiredFieldsSingleAdd() {      
59      SolrCore core = h.getCore();     
60      // Add a single document
61      assertU("adding document",
62        adoc("id", "529", "name", "document with id, name, and subject", "field_t", "what's inside?", "subject", "info"));
63      assertU(commit());
64      
65      // Check it it is in the index
66      assertQ("should find one", req("id:529") ,"//result[@numFound=1]" );
67  
68      // Add another document without the required subject field, which
69      // has a configured defaultValue of "Stuff"
70      assertU("adding a doc without field w/ configured default",
71            adoc("id", "530", "name", "document with id and name", "field_t", "what's inside?"));
72      assertU(commit());
73  
74      // Add another document without a subject, which has a default in schema
75      String subjectDefault = core.getLatestSchema().getField("subject").getDefaultValue();
76      assertNotNull("subject has no default value", subjectDefault);
77      assertQ("should find one with subject="+subjectDefault, req("id:530 subject:"+subjectDefault) ,"//result[@numFound=1]" );
78  
79      // Add another document without a required name, which has no default
80      assertNull(core.getLatestSchema().getField("name").getDefaultValue());
81      ignoreException("missing required field");
82      assertFailedU("adding doc without required field",
83            adoc("id", "531", "subject", "no name document", "field_t", "what's inside?") );
84      resetExceptionIgnores();
85      assertU(commit());
86      
87      // Check to make sure this submission did not succeed
88      assertQ("should not find any", req("id:531") ,"//result[@numFound=0]" ); 
89    }
90    
91    @Test
92    public void testAddMultipleDocumentsWithErrors() {
93      //Add three documents at once to make sure the baseline succeeds
94      assertU("adding 3 documents",
95        "<add>" +doc("id", "601", "name", "multiad one", "field_t", "what's inside?", "subject", "info") +
96        doc("id", "602", "name", "multiad two", "field_t", "what's inside?", "subject", "info") +
97          doc("id", "603", "name", "multiad three", "field_t", "what's inside?", "subject", "info") +
98          "</add>");
99      assertU(commit());
100 
101     // Check that they are in the index
102     assertQ("should find three", req("name:multiad") ,"//result[@numFound=3]" );
103     
104     // Add three documents at once, with the middle one missing a field that has a default
105     assertU("adding 3 docs, with 2nd one missing a field that has a default value",
106       "<add>" +doc("id", "601", "name", "nosubject batch one", "field_t", "what's inside?", "subject", "info") +
107       doc("id", "602", "name", "nosubject batch two", "field_t", "what's inside?") +
108         doc("id", "603", "name", "nosubject batch three", "field_t", "what's inside?", "subject", "info") +
109         "</add>");
110     assertU(commit());
111     
112     // Since the missing field had a devault value,
113     // All three should have made it into the index
114     assertQ("should find three", req("name:nosubject") ,"//result[@numFound=3]" );
115     
116 
117     // Add three documents at once, with the middle with a bad field definition,
118     // to establish the baselinie behavior for errors in a multi-ad submission
119     assertFailedU("adding 3 documents, with 2nd one with undefined field",
120           "<add>" +doc("id", "801", "name", "baddef batch one", "field_t", "what's inside?", "subject", "info") +
121           doc("id", "802", "name", "baddef batch two", "missing_field_ignore_exception", "garbage") +
122             doc("id", "803", "name", "baddef batch three", "field_t", "what's inside?", "subject", "info") +
123             "</add>");
124     assertU(commit());    
125 
126     // Check that only docs before the error should be in the index
127     assertQ("should find one", req("name:baddef") ,"//result[@numFound=1]" );
128 
129     ignoreException("missing required field");
130     // Add three documents at once, with the middle one missing a required field that has no default
131     assertFailedU("adding 3 docs, with 2nd one missing required field",
132       "<add>" +doc("id", "701", "name", "noname batch one", "field_t", "what's inside?", "subject", "info") +
133       doc("id", "702", "field_t", "what's inside?", "subject", "info") +
134         doc("id", "703", "name", "noname batch batch three", "field_t", "what's inside?", "subject", "info") +
135         "</add>");
136     resetExceptionIgnores();
137 
138     assertU(commit());
139 
140     // Check that only docs before the error should be in the index
141     assertQ("should find one", req("name:noname") ,"//result[@numFound=1]" );
142   }
