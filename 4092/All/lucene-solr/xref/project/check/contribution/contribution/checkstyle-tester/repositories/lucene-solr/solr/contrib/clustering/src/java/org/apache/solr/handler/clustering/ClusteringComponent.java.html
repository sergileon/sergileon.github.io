51  /**
52   * Provides a plugin for performing cluster analysis. This can either be applied to 
53   * search results (e.g., via <a href="http://project.carrot2.org">Carrot<sup>2</sup></a>) or for
54   * clustering documents (e.g., via <a href="http://mahout.apache.org/">Mahout</a>).
55   * <p>
56   * See Solr example for configuration examples.</p>
57   * 
58   * @lucene.experimental
59   */
60  public class ClusteringComponent extends SearchComponent implements SolrCoreAware {
61    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
62  
63    /**
64     * Base name for all component parameters. This name is also used to
65     * register this component with SearchHandler.
66     */
67    public static final String COMPONENT_NAME = "clustering";
68  
69    /**
70     * Declaration-order list of search clustering engines.
71     */
72    private final LinkedHashMap<String, SearchClusteringEngine> searchClusteringEngines = Maps.newLinkedHashMap();
73    
74    /**
75     * Declaration order list of document clustering engines.
76     */
77    private final LinkedHashMap<String, DocumentClusteringEngine> documentClusteringEngines = Maps.newLinkedHashMap();
78  
79    /**
80     * An unmodifiable view of {@link #searchClusteringEngines}.
81     */
82    private final Map<String, SearchClusteringEngine> searchClusteringEnginesView = Collections.unmodifiableMap(searchClusteringEngines);

84    /**
85     * Initialization parameters temporarily saved here, the component
86     * is initialized in {@link #inform(SolrCore)} because we need to know
87     * the core's {@link SolrResourceLoader}.
88     * 
89     * @see #init(NamedList)
90     */
91    private NamedList<Object> initParams;
92  
93    @Override
94    @SuppressWarnings({"rawtypes", "unchecked"})
95    public void init(NamedList args) {
96      this.initParams = args;
97      super.init(args);
98    }

100   @SuppressWarnings("unchecked")
101   @Override
102   public void inform(SolrCore core) {
103     if (initParams != null) {
104       log.info("Initializing Clustering Engines");
105 
106       // Our target list of engines, split into search-results and document clustering.
107       SolrResourceLoader loader = core.getResourceLoader();
108   
109       for (Map.Entry<String,Object> entry : initParams) {
110         if ("engine".equals(entry.getKey())) {
111           NamedList<Object> engineInitParams = (NamedList<Object>) entry.getValue();
112           Boolean optional = engineInitParams.getBooleanArg("optional");
113           optional = (optional == null ? Boolean.FALSE : optional);
114 
115           String engineClassName = StringUtils.defaultIfBlank( 
116               (String) engineInitParams.get("classname"),
117               CarrotClusteringEngine.class.getName()); 
118   
119           // Instantiate the clustering engine and split to appropriate map. 
120           final ClusteringEngine engine = loader.newInstance(engineClassName, ClusteringEngine.class);
121           final String name = StringUtils.defaultIfBlank(engine.init(engineInitParams, core), "");
122 
123           if (!engine.isAvailable()) {
124             if (optional) {
125               log.info("Optional clustering engine not available: " + name);
126             } else {
127               throw new SolrException(ErrorCode.SERVER_ERROR, 
128                   "A required clustering engine failed to initialize, check the logs: " + name);
129             }
130           }
131           
132           final ClusteringEngine previousEntry;
133           if (engine instanceof SearchClusteringEngine) {
134             previousEntry = searchClusteringEngines.put(name, (SearchClusteringEngine) engine);
135           } else if (engine instanceof DocumentClusteringEngine) {
136             previousEntry = documentClusteringEngines.put(name, (DocumentClusteringEngine) engine);
137           } else {
138             log.warn("Unknown type of a clustering engine for class: " + engineClassName);
139             continue;
140           }
141           if (previousEntry != null) {
142             log.warn("Duplicate clustering engine component named '" + name + "'.");
143           }
144         }
145       }
146 
147       // Set up the default engine key for both types of engines.
148       setupDefaultEngine("search results clustering", searchClusteringEngines);
149       setupDefaultEngine("document clustering", documentClusteringEngines);
150 
151       log.info("Finished Initializing Clustering Engines");
152     }
153   }

155   @Override
156   public void prepare(ResponseBuilder rb) throws IOException {
157     SolrParams params = rb.req.getParams();
158     if (!params.getBool(COMPONENT_NAME, false)) {
159       return;
160     }
161   }
162 
163   @Override
164   public void process(ResponseBuilder rb) throws IOException {
165     SolrParams params = rb.req.getParams();
166     if (!params.getBool(COMPONENT_NAME, false)) {
167       return;
168     }
169 
170     final String name = getClusteringEngineName(rb);
171     boolean useResults = params.getBool(ClusteringParams.USE_SEARCH_RESULTS, false);
172     if (useResults == true) {
173       SearchClusteringEngine engine = searchClusteringEngines.get(name);
174       if (engine != null) {
175         checkAvailable(name, engine);
176         DocListAndSet results = rb.getResults();
177         Map<SolrDocument,Integer> docIds = Maps.newHashMapWithExpectedSize(results.docList.size());
178         SolrDocumentList solrDocList = SolrPluginUtils.docListToSolrDocumentList(
179             results.docList, rb.req.getSearcher(), engine.getFieldsToLoad(rb.req), docIds);
180         Object clusters = engine.cluster(rb.getQuery(), solrDocList, docIds, rb.req);
181         rb.rsp.add("clusters", clusters);
182       } else {
183         log.warn("No engine named: " + name);
184       }
185     }
186 
187     boolean useCollection = params.getBool(ClusteringParams.USE_COLLECTION, false);
188     if (useCollection == true) {
189       DocumentClusteringEngine engine = documentClusteringEngines.get(name);
190       if (engine != null) {
191         checkAvailable(name, engine);
192         boolean useDocSet = params.getBool(ClusteringParams.USE_DOC_SET, false);
193         NamedList<?> nl = null;
194 
195         // TODO: This likely needs to be made into a background task that runs in an executor
196         if (useDocSet == true) {
197           nl = engine.cluster(rb.getResults().docSet, params);
198         } else {
199           nl = engine.cluster(params);
200         }
201         rb.rsp.add("clusters", nl);
202       } else {
203         log.warn("No engine named: " + name);
204       }
205     }
206   }
207 
208   private void checkAvailable(String name, ClusteringEngine engine) {
209     if (!engine.isAvailable()) {
210       throw new SolrException(ErrorCode.SERVER_ERROR, 
211           "Clustering engine declared, but not available, check the logs: " + name);
212     }
213   }

215   private String getClusteringEngineName(ResponseBuilder rb){
216     return rb.req.getParams().get(ClusteringParams.ENGINE_NAME, ClusteringEngine.DEFAULT_ENGINE_NAME);
217   }

219   @Override
220   public void modifyRequest(ResponseBuilder rb, SearchComponent who, ShardRequest sreq) {
221     SolrParams params = rb.req.getParams();
222     if (!params.getBool(COMPONENT_NAME, false) || !params.getBool(ClusteringParams.USE_SEARCH_RESULTS, false)) {
223       return;
224     }
225     sreq.params.remove(COMPONENT_NAME);
226     if( ( sreq.purpose & ShardRequest.PURPOSE_GET_FIELDS ) != 0 ){
227       String fl = sreq.params.get(CommonParams.FL,"*");
228       // if fl=* then we don't need to check.
229       if (fl.indexOf('*') >= 0) { 
230         return;
231       }
232 
233       String name = getClusteringEngineName(rb);
234       SearchClusteringEngine engine = searchClusteringEngines.get(name);
235       if (engine != null) {
236         checkAvailable(name, engine);
237         Set<String> fields = engine.getFieldsToLoad(rb.req);
238         if (fields == null || fields.size() == 0) { 
239           return;
240         }
241   
242         StringBuilder sb = new StringBuilder();
243         String[] flparams = fl.split( "[,\\s]+" );
244         Set<String> flParamSet = new HashSet<>(flparams.length);
245         for (String flparam : flparams) {
246           // no need trim() because of split() by \s+
247           flParamSet.add(flparam);
248         }
249         for (String aFieldToLoad : fields) {
250           if (!flParamSet.contains(aFieldToLoad )) {
251             sb.append(',').append(aFieldToLoad);
252           }
253         }
254         if (sb.length() > 0) {
255           sreq.params.set(CommonParams.FL, fl + sb.toString());
256         }
257       } else {
258         log.warn("No engine named: " + name);
259       }
260     }
261   }

263   @Override
264   public void finishStage(ResponseBuilder rb) {
265     SolrParams params = rb.req.getParams();
266     if (!params.getBool(COMPONENT_NAME, false) || 
267         !params.getBool(ClusteringParams.USE_SEARCH_RESULTS, false)) {
268       return;
269     }
270 
271     if (rb.stage == ResponseBuilder.STAGE_GET_FIELDS) {
272       String name = getClusteringEngineName(rb);
273       SearchClusteringEngine engine = searchClusteringEngines.get(name);
274       if (engine != null) {
275         checkAvailable(name, engine);
276         SolrDocumentList solrDocList = (SolrDocumentList) rb.rsp.getValues().get("response");
277         // TODO: Currently, docIds is set to null in distributed environment.
278         // This causes CarrotParams.PRODUCE_SUMMARY doesn't work.
279         // To work CarrotParams.PRODUCE_SUMMARY under distributed mode, we can choose either one of:
280         // (a) In each shard, ClusteringComponent produces summary and finishStage()
281         //     merges these summaries.
282         // (b) Adding doHighlighting(SolrDocumentList, ...) method to SolrHighlighter and
283         //     making SolrHighlighter uses "external text" rather than stored values to produce snippets.
284         Map<SolrDocument,Integer> docIds = null;
285         Object clusters = engine.cluster(rb.getQuery(), solrDocList, docIds, rb.req);
286         rb.rsp.add("clusters", clusters);
287       } else {
288         log.warn("No engine named: " + name);
289       }
290     }
291   }

305   /**
306    * Setup the default clustering engine.
307    * @see "https://issues.apache.org/jira/browse/SOLR-5219"
308    */
309   private static <T extends ClusteringEngine> void setupDefaultEngine(String type, LinkedHashMap<String,T> map) {
310     // If there's already a default algorithm, leave it as is.
311     String engineName = ClusteringEngine.DEFAULT_ENGINE_NAME;
312     T defaultEngine = map.get(engineName);
313 
314     if (defaultEngine == null ||
315         !defaultEngine.isAvailable()) {
316       // If there's no default algorithm, and there are any algorithms available, 
317       // the first definition becomes the default algorithm.
318       for (Map.Entry<String, T> e : map.entrySet()) {
319         if (e.getValue().isAvailable()) {
320           engineName = e.getKey();
321           defaultEngine = e.getValue();
322           map.put(ClusteringEngine.DEFAULT_ENGINE_NAME, defaultEngine);
323           break;
324         }
325       }
326     }
327 
328     if (defaultEngine != null) {
329       log.info("Default engine for " + type + ": " + engineName + " [" + defaultEngine.getClass().getSimpleName() + "]");
330     } else {
331       log.warn("No default engine for " + type + ".");
332     }
