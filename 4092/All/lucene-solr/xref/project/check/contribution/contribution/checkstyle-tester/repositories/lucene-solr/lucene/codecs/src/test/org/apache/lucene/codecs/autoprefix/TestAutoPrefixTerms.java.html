202   private static BytesRef intToBytes(int v) {
203     int sortableBits = v ^ 0x80000000;
204     BytesRef token = new BytesRef(4);
205     token.length = 4;
206     int index = 3;
207     while (index >= 0) {
208       token.bytes[index] = (byte) (sortableBits & 0xff);
209       index--;
210       sortableBits >>>= 8;
211     }
212     return token;
213   }

507   /** Helper class to ensure auto-prefix terms 1) never overlap one another, and 2) are used when they should be. */
508   private static class VerifyAutoPrefixTerms {
509     final FixedBitSet allHits;
510     private final Map<BytesRef,Integer> prefixCounts = new HashMap<>();
511     private int totPrefixCount;
512     private final BytesRef[] bounds;
513     private int totTermCount;
514     private BytesRef lastTerm;
515 
516     public VerifyAutoPrefixTerms(int maxDoc, BytesRef... bounds) {
517       allHits = new FixedBitSet(maxDoc);
518       assert bounds.length > 0;
519       this.bounds = bounds;
520     }
521 
522     public void sawTerm(BytesRef term) {
523       //System.out.println("saw term=" + term);
524       if (lastTerm != null) {
525         assertTrue(lastTerm.compareTo(term) < 0);
526       }
527       lastTerm = BytesRef.deepCopyOf(term);
528       totTermCount++;
529       totPrefixCount += term.length;
530       for(int i=1;i<=term.length;i++) {
531         BytesRef prefix = BytesRef.deepCopyOf(term);
532         prefix.length = i;
533         Integer count = prefixCounts.get(prefix);
534         if (count == null) {
535           count = 1;
536         } else {
537           count += 1;
538         }
539         prefixCounts.put(prefix, count);
540       }
541     }
542 
543     public void sawDoc(int docID) {
544       // The auto-prefix terms should never "overlap" one another, so we should only ever see a given docID one time:
545       assertFalse(allHits.getAndSet(docID));
546     }
547 
548     public void finish(int expectedNumHits, int maxPrefixCount) {
549 
550       if (maxPrefixCount != -1) {
551         // Auto-prefix terms were used in this test
552         long allowedMaxTerms;
553 
554         if (bounds.length == 1) {
555           // Simple prefix query: we should never see more than maxPrefixCount terms, except for the empty string:
556           if (bounds[0].length == 0) {
557             allowedMaxTerms = Integer.MAX_VALUE;
558           } else {
559             allowedMaxTerms = maxPrefixCount;
560           }
561         } else {
562           // Trickier: we need to allow for maxPrefixTerms for each different leading byte in the min and max:
563           assert bounds.length == 2;
564           BytesRef minTerm = bounds[0];
565           BytesRef maxTerm = bounds[1];
566 
567           int commonPrefix = 0;
568           for(int i=0;i<minTerm.length && i<maxTerm.length;i++) {
569             if (minTerm.bytes[minTerm.offset+i] != maxTerm.bytes[maxTerm.offset+i]) {
570               commonPrefix = i;
571               break;
572             }
573           }
574 
575           allowedMaxTerms = maxPrefixCount * (long) ((minTerm.length-commonPrefix) + (maxTerm.length-commonPrefix));
576           if (commonPrefix == 0) {
577             int min;
578             if (minTerm.length == 0) {
579               min = 0;
580             } else {
581               min = minTerm.bytes[minTerm.offset] & 0xff;
582             }
583             int max;
584             if (maxTerm.length == 0) {
585               max = 0;
586             } else {
587               max = maxTerm.bytes[maxTerm.offset] & 0xff;
588             }
589             if (max > min) {
590               // When maxPrefixCount is small (< 16), each byte of the term can require more than one "level" of auto-prefixing:
591               // NOTE: this is still only approximate ... it's tricky to get a closed form max bound that's "tight"
592               allowedMaxTerms += MathUtil.log(max-min, maxPrefixCount);
593             }
594           }
595         }
596 
597         assertTrue("totTermCount=" + totTermCount + " is > allowedMaxTerms=" + allowedMaxTerms, totTermCount <= allowedMaxTerms);
598       }
599 
600       assertEquals(expectedNumHits, allHits.cardinality());
601       int sum = 0;
602       for(Map.Entry<BytesRef,Integer> ent : prefixCounts.entrySet()) {
603 
604         BytesRef prefix = ent.getKey();
605         if (VERBOSE) {
606           System.out.println("  verify prefix=" + TestUtil.bytesRefToString(prefix) + " count=" + ent.getValue());
607         }
608 
609         if (maxPrefixCount != -1) {
610           // Auto-prefix terms were used in this test
611 
612           int sumLeftoverSuffix = 0;
613           for(BytesRef bound : bounds) {
614 
615             int minSharedLength = Math.min(bound.length, prefix.length);
616             int commonPrefix = minSharedLength;
617             for(int i=0;i<minSharedLength;i++) {
618               if (bound.bytes[bound.offset+i] != prefix.bytes[prefix.offset+i]) {
619                 commonPrefix = i;
620                 break;
621               }
622             }
623             sumLeftoverSuffix += bound.length - commonPrefix;
624           }
625 
626           long limit = (1+sumLeftoverSuffix) * (long) maxPrefixCount;
627 
628           assertTrue("maxPrefixCount=" + maxPrefixCount + " prefix=" + prefix + " sumLeftoverSuffix=" + sumLeftoverSuffix + " limit=" + limit + " vs actual=" +ent.getValue(),
629                      ent.getValue() <= limit);
630         }
631 
632         sum += ent.getValue();
633       }
634 
635       // Make sure no test bug:
636       assertEquals(totPrefixCount, sum);
637     }
