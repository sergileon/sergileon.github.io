44    @Override
45    public void setUp() throws Exception {
46      super.setUp();
47      dir = newDirectory();
48      iw = new RandomIndexWriter(random(), dir);
49      Document doc = new Document();
50      Field idField = new StringField("id", "", Field.Store.NO);
51      doc.add(idField);
52      // add 500 docs with id 0..499
53      for (int i = 0; i < 500; i++) {
54        idField.setStringValue(Integer.toString(i));
55        iw.addDocument(doc);
56      }
57      // delete 20 of them
58      for (int i = 0; i < 20; i++) {
59        iw.deleteDocuments(new Term("id", Integer.toString(random().nextInt(iw.maxDoc()))));
60      }
61      ir = iw.getReader();
62      is = newSearcher(ir);
63    }
64  
65    @Override
66    public void tearDown() throws Exception {
67      iw.close();
68      IOUtils.close(ir, dir);
69      super.tearDown();
70    }

72    private void assertQueryEquals(Query f1, Query f2) throws Exception {
73      // wrap into CSQ so that scores are not needed
74      TopDocs hits1 = is.search(new ConstantScoreQuery(f1), ir.maxDoc());
75      TopDocs hits2 = is.search(new ConstantScoreQuery(f2), ir.maxDoc());
76      assertEquals(hits1.totalHits, hits2.totalHits);
77      CheckHits.checkEqual(f1, hits1.scoreDocs, hits2.scoreDocs);
78      // now do it again to confirm caching works
79      TopDocs hits3 = is.search(new ConstantScoreQuery(f1), ir.maxDoc());
80      TopDocs hits4 = is.search(new ConstantScoreQuery(f2), ir.maxDoc());
81      assertEquals(hits3.totalHits, hits4.totalHits);
82      CheckHits.checkEqual(f1, hits3.scoreDocs, hits4.scoreDocs);
83    }

130   private static class MockQuery extends Query {
131 
132     private final AtomicBoolean wasCalled = new AtomicBoolean();
133 
134     public boolean wasCalled() {
135       return wasCalled.get();
136     }
137 
138     public void clear() {
139       wasCalled.set(false);
140     }
141 
142     @Override
143     public String toString(String field) {
144       return "Mock";
145     }
146 
147     @Override
148     public Weight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
149       return new ConstantScoreWeight(this) {
150         @Override
151         public Scorer scorer(LeafReaderContext context) throws IOException {
152           wasCalled.set(true);
153           return new ConstantScoreScorer(this, score(), DocIdSetIterator.all(context.reader().maxDoc()));
154         }
155       };
156     }
157   }

159   public void testCachingWorks() throws Exception {
160     Directory dir = newDirectory();
161     RandomIndexWriter writer = new RandomIndexWriter(random(), dir);
162     writer.close();
163 
164     IndexReader reader = SlowCompositeReaderWrapper.wrap(DirectoryReader.open(dir));
165     IndexSearcher searcher = newSearcher(reader);
166     LeafReaderContext context = (LeafReaderContext) reader.getContext();
167     MockQuery filter = new MockQuery();
168     CachingWrapperQuery cacher = new CachingWrapperQuery(filter, QueryCachingPolicy.ALWAYS_CACHE);
169 
170     // first time, nested filter is called
171     searcher.rewrite(cacher).createWeight(searcher, false).scorer(context);
172     assertTrue("first time", filter.wasCalled());
173 
174     // make sure no exception if cache is holding the wrong docIdSet
175     searcher.rewrite(cacher).createWeight(searcher, false).scorer(context);
176 
177     // second time, nested filter should not be called
178     filter.clear();
179     searcher.rewrite(cacher).createWeight(searcher, false).scorer(context);
180     assertFalse("second time", filter.wasCalled());
181 
182     reader.close();
183     dir.close();
184   }

186   public void testEnforceDeletions() throws Exception {
187     Directory dir = newDirectory();
188     RandomIndexWriter writer = new RandomIndexWriter(
189         random(),
190         dir,
191         newIndexWriterConfig(new MockAnalyzer(random())).
192             setMergeScheduler(new SerialMergeScheduler()).
193             // asserts below requires no unexpected merges:
194             setMergePolicy(newLogMergePolicy(10))
195     );
196 
197     // NOTE: cannot use writer.getReader because RIW (on
198     // flipping a coin) may give us a newly opened reader,
199     // but we use .reopen on this reader below and expect to
200     // (must) get an NRT reader:
201     DirectoryReader reader = DirectoryReader.open(writer.w, true);
202     // same reason we don't wrap?
203     IndexSearcher searcher = newSearcher(reader, false);
204 
205     // add a doc, refresh the reader, and check that it's there
206     Document doc = new Document();
207     doc.add(newStringField("id", "1", Field.Store.YES));
208     writer.addDocument(doc);
209 
210     reader = refreshReader(reader);
211     searcher = newSearcher(reader, false);
212 
213     TopDocs docs = searcher.search(new MatchAllDocsQuery(), 1);
214     assertEquals("Should find a hit...", 1, docs.totalHits);
215 
216     final Query startQuery = new TermQuery(new Term("id", "1"));
217 
218     CachingWrapperQuery query = new CachingWrapperQuery(startQuery, QueryCachingPolicy.ALWAYS_CACHE);
219 
220     docs = searcher.search(new ConstantScoreQuery(query), 1);
221     assertTrue(query.ramBytesUsed() > 0);
222 
223     assertEquals("[query + filter] Should find a hit...", 1, docs.totalHits);
224 
225     Query constantScore = new ConstantScoreQuery(query);
226     docs = searcher.search(constantScore, 1);
227     assertEquals("[just filter] Should find a hit...", 1, docs.totalHits);
228 
229     // make sure we get a cache hit when we reopen reader
230     // that had no change to deletions
231 
232     // fake delete (deletes nothing):
233     writer.deleteDocuments(new Term("foo", "bar"));
234 
235     IndexReader oldReader = reader;
236     reader = refreshReader(reader);
237     assertTrue(reader == oldReader);
238     int missCount = query.missCount;
239     docs = searcher.search(constantScore, 1);
240     assertEquals("[just filter] Should find a hit...", 1, docs.totalHits);
241 
242     // cache hit:
243     assertEquals(missCount, query.missCount);
244 
245     // now delete the doc, refresh the reader, and see that it's not there
246     writer.deleteDocuments(new Term("id", "1"));
247 
248     // NOTE: important to hold ref here so GC doesn't clear
249     // the cache entry!  Else the assert below may sometimes
250     // fail:
251     oldReader = reader;
252     reader = refreshReader(reader);
253 
254     searcher = newSearcher(reader, false);
255 
256     missCount = query.missCount;
257     docs = searcher.search(new ConstantScoreQuery(query), 1);
258     assertEquals("[query + filter] Should *not* find a hit...", 0, docs.totalHits);
259 
260     // cache hit
261     assertEquals(missCount, query.missCount);
262     docs = searcher.search(constantScore, 1);
263     assertEquals("[just filter] Should *not* find a hit...", 0, docs.totalHits);
264 
265     // apply deletes dynamically:
266     query = new CachingWrapperQuery(startQuery, QueryCachingPolicy.ALWAYS_CACHE);
267     writer.addDocument(doc);
268     reader = refreshReader(reader);
269     searcher = newSearcher(reader, false);
270 
271     docs = searcher.search(new ConstantScoreQuery(query), 1);
272     assertEquals("[query + filter] Should find a hit...", 1, docs.totalHits);
273     missCount = query.missCount;
274     assertTrue(missCount > 0);
275     constantScore = new ConstantScoreQuery(query);
276     docs = searcher.search(constantScore, 1);
277     assertEquals("[just filter] Should find a hit...", 1, docs.totalHits);
278     assertEquals(missCount, query.missCount);
279 
280     writer.addDocument(doc);
281 
282     // NOTE: important to hold ref here so GC doesn't clear
283     // the cache entry!  Else the assert below may sometimes
284     // fail:
285     oldReader = reader;
286 
287     reader = refreshReader(reader);
288     searcher = newSearcher(reader, false);
289 
290     docs = searcher.search(new ConstantScoreQuery(query), 1);
291     assertEquals("[query + filter] Should find 2 hits...", 2, docs.totalHits);
292     assertTrue(query.missCount > missCount);
293     missCount = query.missCount;
294 
295     constantScore = new ConstantScoreQuery(query);
296     docs = searcher.search(constantScore, 1);
297     assertEquals("[just filter] Should find a hit...", 2, docs.totalHits);
298     assertEquals(missCount, query.missCount);
299 
300     // now delete the doc, refresh the reader, and see that it's not there
301     writer.deleteDocuments(new Term("id", "1"));
302 
303     reader = refreshReader(reader);
304     searcher = newSearcher(reader, false);
305 
306     docs = searcher.search(new ConstantScoreQuery(query), 1);
307     assertEquals("[query + filter] Should *not* find a hit...", 0, docs.totalHits);
308     // CWF reused the same entry (it dynamically applied the deletes):
309     assertEquals(missCount, query.missCount);
310 
311     docs = searcher.search(constantScore, 1);
312     assertEquals("[just filter] Should *not* find a hit...", 0, docs.totalHits);
313     // CWF reused the same entry (it dynamically applied the deletes):
314     assertEquals(missCount, query.missCount);
315 
316     // NOTE: silliness to make sure JRE does not eliminate
317     // our holding onto oldReader to prevent
318     // CachingWrapperFilter's WeakHashMap from dropping the
319     // entry:
320     assertTrue(oldReader != null);
321 
322     reader.close();
323     writer.close();
324     dir.close();
325   }

327   private static DirectoryReader refreshReader(DirectoryReader reader) throws IOException {
328     DirectoryReader oldReader = reader;
329     reader = DirectoryReader.openIfChanged(reader);
330     if (reader != null) {
331       oldReader.close();
332       return reader;
333     } else {
334       return oldReader;
335     }
336   }
