32  /**
33   * Models a single field somewhere in a hierarchy of fields as part of a pivot facet.  
34   * This pivot field contains {@link PivotFacetValue}s which may each contain a nested
35   * {@link PivotFacetField} child.  This <code>PivotFacetField</code> may itself 
36   * be a child of a {@link PivotFacetValue} parent.
37   *
38   * @see PivotFacetValue
39   * @see PivotFacetFieldValueCollection
40   */
41  @SuppressWarnings("rawtypes")
42  public class PivotFacetField {
43    
44    public final String field;
45  
46    // null if this is a top level pivot, 
47    // otherwise the value of the parent pivot we are nested under
48    public final PivotFacetValue parentValue;
49  
50    public final PivotFacetFieldValueCollection valueCollection;
51    
52    // Facet parameters relating to this field
53    private final int facetFieldLimit;
54    private final int facetFieldMinimumCount;
55    private final int facetFieldOffset;  
56    private final String facetFieldSort;
57  
58    private final Map<Integer, Integer> numberOfValuesContributedByShard = new HashMap<>();
59    private final Map<Integer, Integer> shardLowestCount = new HashMap<>();
60  
61    private boolean needRefinementAtThisLevel = true;
62      
63    private PivotFacetField(ResponseBuilder rb, PivotFacetValue parent, String fieldName) {
64      
65      field = fieldName;
66      parentValue = parent;
67      
68      // facet params
69      SolrParams parameters = rb.req.getParams();
70      facetFieldMinimumCount = parameters.getFieldInt(field, FacetParams.FACET_PIVOT_MINCOUNT, 1);
71      facetFieldOffset = parameters.getFieldInt(field, FacetParams.FACET_OFFSET, 0);
72      facetFieldLimit = parameters.getFieldInt(field, FacetParams.FACET_LIMIT, 100);
73      String defaultSort = (facetFieldLimit > 0) ? FacetParams.FACET_SORT_COUNT : FacetParams.FACET_SORT_INDEX;
74      facetFieldSort = parameters.getFieldParam(field, FacetParams.FACET_SORT, defaultSort);
75  
76      valueCollection = new PivotFacetFieldValueCollection(facetFieldMinimumCount, facetFieldOffset, facetFieldLimit, facetFieldSort);
77      
78      if ( (facetFieldLimit < 0) || 
79           // TODO: possible refinement issue if limit=0 & mincount=0 & missing=true
80           // (ie: we only want the missing count for this field)
81           (facetFieldLimit <= 0 && facetFieldMinimumCount == 0) ||
82           (facetFieldSort.equals(FacetParams.FACET_SORT_INDEX) && facetFieldMinimumCount <= 0) 
83           ) {
84        // in any of these cases, there's no need to refine this level of the pivot
85        needRefinementAtThisLevel = false;
86      }
87    }

89    /** 
90     * A recursive method that walks up the tree of pivot fields/values to build 
91     * a list of String representations of the values that lead down to this 
92     * PivotFacetField.
93     *
94     * @return A mutable List of the pivot values leading down to this pivot field, 
95     *      will never be null but may contain nulls and may be empty if this is a top 
96     *      level pivot field
97     * @see PivotFacetValue#getValuePath
98     */
99    public List<String> getValuePath() {
100     if (null != parentValue) {
101       return parentValue.getValuePath();
102     }
103     return new ArrayList<String>(3);
104   }

106   /**
107    * A recursive method to construct a new <code>PivotFacetField</code> object from 
108    * the contents of the {@link NamedList}s provided by the specified shard, relative 
109    * to a parent value (if this is not the top field in the pivot hierarchy)
110    *
111    * The associated child {@link PivotFacetValue}s will be recursively built as well.
112    *
113    * @see PivotFacetValue#createFromNamedList
114    * @param shardNumber the id of the shard that provided this data
115    * @param rb The response builder of the current request
116    * @param owner the parent value in the current pivot (may be null)
117    * @param pivotValues the data from the specified shard for this pivot field, may be null or empty
118    * @return the new PivotFacetField, null if pivotValues is null or empty.
119    */
120   public static PivotFacetField createFromListOfNamedLists(int shardNumber, ResponseBuilder rb, PivotFacetValue owner, List<NamedList<Object>> pivotValues) {
121     
122     if (null == pivotValues || pivotValues.size() <= 0) return null;
123     
124     NamedList<Object> firstValue = pivotValues.get(0);
125     PivotFacetField createdPivotFacetField 
126       = new PivotFacetField(rb, owner, PivotFacetHelper.getField(firstValue));
127     
128     int lowestCount = Integer.MAX_VALUE;
129     
130     for (NamedList<Object> pivotValue : pivotValues) {
131       
132       lowestCount = Math.min(lowestCount, PivotFacetHelper.getCount(pivotValue));
133       
134       PivotFacetValue newValue = PivotFacetValue.createFromNamedList
135         (shardNumber, rb, createdPivotFacetField, pivotValue);
136       createdPivotFacetField.valueCollection.add(newValue);
137     }
138       
139     createdPivotFacetField.shardLowestCount.put(shardNumber,  lowestCount);
140     createdPivotFacetField.numberOfValuesContributedByShard.put(shardNumber, pivotValues.size());
141 
142     return createdPivotFacetField;
143   }

145   /**
146    * Destructive method that recursively prunes values from the data structure 
147    * based on the counts for those values and the effective sort, mincount, limit, 
148    * and offset being used for each field.
149    * <p>
150    * This method should only be called after all refinement is completed just prior 
151    * calling {@link #convertToListOfNamedLists}
152    * </p>
153    *
154    * @see PivotFacet#getTrimmedPivotsAsListOfNamedLists
155    * @see PivotFacetFieldValueCollection#trim
156    */
157   public void trim() {
158     // SOLR-6331...
159     //
160     // we can probably optimize the memory usage by trimming each level of the pivot once
161     // we know we've fully refined the values at that level 
162     // (ie: fold this logic into refineNextLevelOfFacets)
163     this.valueCollection.trim();
164   }

167    * Recursively sorts the collection of values associated with this field, and 
168    * any sub-pivots those values have.
169    *
170    * @see FacetParams#FACET_SORT
171    * @see PivotFacetFieldValueCollection#sort
172    */
173   public void sort() {
174     this.valueCollection.sort();
175   }

177   /** 
178    * A recursive method for generating <code>NamedLists</code> from this field 
179    * suitable for including in a pivot facet response to the original distributed request.
180    */
181   public List<NamedList<Object>> convertToListOfNamedLists() { 
182     
183     List<NamedList<Object>> convertedPivotList = null;
184     
185     if (valueCollection.size() > 0) {
186       convertedPivotList = new LinkedList<>();
187       for (PivotFacetValue pivot : valueCollection)
188         convertedPivotList.add(pivot.convertToNamedList());
189     }
190   
191     return convertedPivotList;
192   }

194   /** 
195    * A recursive method for determining which {@link PivotFacetValue}s need to be
196    * refined for this pivot.
197    *
198    * @see PivotFacet#queuePivotRefinementRequests
199    */
200   public void queuePivotRefinementRequests(PivotFacet pf) {
201     
202     if (needRefinementAtThisLevel) {
203 
204       if (0 < facetFieldMinimumCount) {
205         // missing is always a candidate for refinement if at least one shard met the minimum
206         PivotFacetValue missing = valueCollection.getMissingValue();
207         if (null != missing) {
208           processDefiniteCandidateElement(pf, valueCollection.getMissingValue());
209         }
210       }
211 
212       if (! valueCollection.getExplicitValuesList().isEmpty()) {
213 
214         if (FacetParams.FACET_SORT_COUNT.equals(facetFieldSort)) {
215           // we only need to things that are currently in our limit,
216           // or might be in our limit if we get increased counts from shards that
217           // didn't include this value the first time
218           final int indexOfCountThreshold 
219             = Math.min(valueCollection.getExplicitValuesListSize(), 
220                        facetFieldOffset + facetFieldLimit) - 1;
221           final int countThreshold = valueCollection.getAt(indexOfCountThreshold).getCount();
222           
223           int positionInResults = 0;
224           
225           for (PivotFacetValue value : valueCollection.getExplicitValuesList()) {
226             if (positionInResults <= indexOfCountThreshold) {
227               // This element is within the top results, so we need to get information 
228               // from all of the shards.
229               processDefiniteCandidateElement(pf, value);
230             } else {
231               // This element is not within the top results, but may still need to be refined.
232               processPossibleCandidateElement(pf, value, countThreshold);
233             }
234             
235             positionInResults++;
236           }
237         } else { // FACET_SORT_INDEX
238           // everything needs refined to see what the per-shard mincount excluded
239           for (PivotFacetValue value : valueCollection.getExplicitValuesList()) {
240             processDefiniteCandidateElement(pf, value);
241           }
242         }
243       }
244 
245       needRefinementAtThisLevel = false;
246     }
247       
248     if ( pf.isRefinementsRequired() ) {
249       // if any refinements are needed, then we need to stop and wait to
250       // see how the picture may change before drilling down to child pivot fields 
251       return;
252     } else {
253       // Since outstanding requests have been filled, then we can drill down 
254       // to the next deeper level and check it.
255       refineNextLevelOfFacets(pf);
256     }
257   }
258       
259   /**
260    * Adds refinement requests for the value for each shard that has not already contributed 
261    * a count for this value.
262    */
263   private void processDefiniteCandidateElement(PivotFacet pf, PivotFacetValue value) {
264     
265     for (int shard = pf.knownShards.nextSetBit(0); 
266          0 <= shard; 
267          shard = pf.knownShards.nextSetBit(shard+1)) {   
268       if ( ! value.shardHasContributed(shard) ) {
269         if ( // if we're doing index order, we need to refine anything  
270              // (mincount may have excluded from a shard)
271             FacetParams.FACET_SORT_INDEX.equals(facetFieldSort)
272             || (// 'missing' value isn't affected by limit, needs refined if shard didn't provide
273                 null == value.getValue() ||
274                 // if we are doing count order, we need to refine if the limit was hit
275                 // (if not, the shard doesn't have the value or it would have returned already)
276                 numberOfValuesContributedByShardWasLimitedByFacetFieldLimit(shard))) {
277           pf.addRefinement(shard, value);
278         }
279       }
280     }  
281   }
282 
283   private boolean numberOfValuesContributedByShardWasLimitedByFacetFieldLimit(int shardNumber) {
284     return facetFieldLimit <= numberOfValuesContributedByShard(shardNumber);
285   }
286   
287   private int numberOfValuesContributedByShard(final int shardNumber) { 
288     return numberOfValuesContributedByShard.containsKey(shardNumber)
289       ? numberOfValuesContributedByShard.get(shardNumber) 
290       : 0;
291   }
292   
293   /** 
294    * Checks the {@link #lowestCountContributedbyShard} for each shard, combined with the 
295    * counts we already know, to see if this value is a viable candidate -- 
296    * <b>Does not make sense when using {@link FacetParams#FACET_SORT_INDEX}</b>
297    *
298    * @see #processDefiniteCandidateElement
299    */
300   private void processPossibleCandidateElement(PivotFacet pf, PivotFacetValue value, 
301                                                final int refinementThreshold) {
302     
303     assert FacetParams.FACET_SORT_COUNT.equals(facetFieldSort)
304       : "Method only makes sense when sorting by count";
305 
306     int maxPossibleCountAfterRefinement = value.getCount();
307     
308     for (int shard = pf.knownShards.nextSetBit(0); 
309          0 <= shard;
310          shard = pf.knownShards.nextSetBit(shard+1)) {
311       if ( ! value.shardHasContributed(shard) ) {
312         maxPossibleCountAfterRefinement += lowestCountContributedbyShard(shard);
313       }
314     }
315     
316     if (refinementThreshold <= maxPossibleCountAfterRefinement) {
317       processDefiniteCandidateElement(pf, value);
318     }
319   }
320    
321   private int lowestCountContributedbyShard(int shardNumber) {
322     return (shardLowestCount.containsKey(shardNumber))
323       ? shardLowestCount.get(shardNumber) 
324       : 0;
325   }
326   
327   private void refineNextLevelOfFacets(PivotFacet pf) {
328 
329     List<PivotFacetValue> explicitValsToRefine 
330       = valueCollection.getNextLevelValuesToRefine();
331     
332     for (PivotFacetValue value : explicitValsToRefine) {
333       if (null != value.getChildPivot()) {
334         value.getChildPivot().queuePivotRefinementRequests(pf);
335       }
336     }
337 
338     PivotFacetValue missing = this.valueCollection.getMissingValue();
339     if(null != missing && null != missing.getChildPivot()) {
340       missing.getChildPivot().queuePivotRefinementRequests(pf);
341     }
342   }
343   
344   private void incrementShardValueCount(int shardNumber) {
345     if (!numberOfValuesContributedByShard.containsKey(shardNumber)) {
346       numberOfValuesContributedByShard.put(shardNumber, 1);
347     } else {
348       numberOfValuesContributedByShard.put(shardNumber, numberOfValuesContributedByShard.get(shardNumber)+1);
349     }
350   }
351   
352   private void contributeValueFromShard(int shardNumber, ResponseBuilder rb, NamedList<Object> shardValue) {
353     
354     incrementShardValueCount(shardNumber);
355 
356     Comparable value = PivotFacetHelper.getValue(shardValue);
357     int count = PivotFacetHelper.getCount(shardValue);
358     
359     // We're changing values so we most mark the collection as dirty
360     valueCollection.markDirty();
361     
362     if ( ( !shardLowestCount.containsKey(shardNumber) )
363          || shardLowestCount.get(shardNumber) > count) {
364       shardLowestCount.put(shardNumber,  count);
365     }
366     
367     PivotFacetValue facetValue = valueCollection.get(value);
368     if (null == facetValue) {
369       // never seen before, we need to create it from scratch
370       facetValue = PivotFacetValue.createFromNamedList(shardNumber, rb, this, shardValue);
371       this.valueCollection.add(facetValue);
372     } else {
373       facetValue.mergeContributionFromShard(shardNumber, rb, shardValue);
374     }
375   }
376   
377   /**
378    * Recursively merges the contributions from the specified shard for each 
379    * {@link PivotFacetValue} represended in the <code>response</code>.
380    * 
381    * @see PivotFacetValue#mergeContributionFromShard
382    * @param shardNumber the id of the shard that provided this data
383    * @param rb The response builder of the current request
384    * @param response the data from the specified shard for this pivot field, may be null
385    */
386   public void contributeFromShard(int shardNumber, ResponseBuilder rb, List<NamedList<Object>> response) {
387     if (null == response) return;
388 
389     for (NamedList<Object> responseValue : response) {
390       contributeValueFromShard(shardNumber, rb, responseValue);
391     }
392   }
393   
394   public String toString(){
395     return String.format(Locale.ROOT, "P:%s F:%s V:%s",
396                          parentValue, field, valueCollection);
397   }
