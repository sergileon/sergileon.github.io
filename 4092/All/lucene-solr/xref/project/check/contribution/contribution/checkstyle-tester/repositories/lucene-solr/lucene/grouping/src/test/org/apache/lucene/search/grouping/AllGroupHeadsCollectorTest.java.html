376   private boolean arrayContains(int[] expected, int[] actual) {
377     Arrays.sort(actual); // in some cases the actual docs aren't sorted by docid. This method expects that.
378     if (expected.length != actual.length) {
379       return false;
380     }
381 
382     for (int e : expected) {
383       boolean found = false;
384       for (int a : actual) {
385         if (e == a) {
386           found = true;
387           break;
388         }
389       }
390 
391       if (!found) {
392         return false;
393       }
394     }
395 
396     return true;
397   }

399   private boolean openBitSetContains(int[] expectedDocs, FixedBitSet actual, int maxDoc) throws IOException {
400     if (expectedDocs.length != actual.cardinality()) {
401       return false;
402     }
403 
404     FixedBitSet expected = new FixedBitSet(maxDoc);
405     for (int expectedDoc : expectedDocs) {
406       expected.set(expectedDoc);
407     }
408 
409     for (int docId = expected.nextSetBit(0); docId != DocIdSetIterator.NO_MORE_DOCS; docId = docId + 1 >= expected.length() ? DocIdSetIterator.NO_MORE_DOCS : expected.nextSetBit(docId + 1)) {
410       if (!actual.get(docId)) {
411         return false;
412       }
413     }
414 
415     return true;
416   }

418   private int[] createExpectedGroupHeads(String searchTerm, GroupDoc[] groupDocs, Sort docSort, boolean sortByScoreOnly, int[] fieldIdToDocID) {
419     Map<BytesRef, List<GroupDoc>> groupHeads = new HashMap<>();
420     for (GroupDoc groupDoc : groupDocs) {
421       if (!groupDoc.content.startsWith(searchTerm)) {
422         continue;
423       }
424 
425       if (!groupHeads.containsKey(groupDoc.group)) {
426         List<GroupDoc> list = new ArrayList<>();
427         list.add(groupDoc);
428         groupHeads.put(groupDoc.group, list);
429         continue;
430       }
431       groupHeads.get(groupDoc.group).add(groupDoc);
432     }
433 
434     int[] allGroupHeads = new int[groupHeads.size()];
435     int i = 0;
436     for (BytesRef groupValue : groupHeads.keySet()) {
437       List<GroupDoc> docs = groupHeads.get(groupValue);
438       Collections.sort(docs, getComparator(docSort, sortByScoreOnly, fieldIdToDocID));
439       allGroupHeads[i++] = docs.get(0).id;
440     }
441 
442     return allGroupHeads;
443   }

445   private Sort getRandomSort(boolean scoreOnly) {
446     final List<SortField> sortFields = new ArrayList<>();
447     if (random().nextInt(7) == 2 || scoreOnly) {
448       sortFields.add(SortField.FIELD_SCORE);
449     } else {
450       if (random().nextBoolean()) {
451         if (random().nextBoolean()) {
452           sortFields.add(new SortField("sort1", SortField.Type.STRING, random().nextBoolean()));
453         } else {
454           sortFields.add(new SortField("sort2", SortField.Type.STRING, random().nextBoolean()));
455         }
456       } else if (random().nextBoolean()) {
457         sortFields.add(new SortField("sort1", SortField.Type.STRING, random().nextBoolean()));
458         sortFields.add(new SortField("sort2", SortField.Type.STRING, random().nextBoolean()));
459       }
460     }
461     // Break ties:
462     if (random().nextBoolean() && !scoreOnly) {
463       sortFields.add(new SortField("sort3", SortField.Type.STRING));
464     } else if (!scoreOnly) {
465       sortFields.add(new SortField("id", SortField.Type.INT));
466     }
467     return new Sort(sortFields.toArray(new SortField[sortFields.size()]));
468   }

470   private Comparator<GroupDoc> getComparator(Sort sort, final boolean sortByScoreOnly, final int[] fieldIdToDocID) {
471     final SortField[] sortFields = sort.getSort();
472     return new Comparator<GroupDoc>() {
473       @Override
474       public int compare(GroupDoc d1, GroupDoc d2) {
475         for (SortField sf : sortFields) {
476           final int cmp;
477           if (sf.getType() == SortField.Type.SCORE) {
478             if (d1.score > d2.score) {
479               cmp = -1;
480             } else if (d1.score < d2.score) {
481               cmp = 1;
482             } else {
483               cmp = sortByScoreOnly ? fieldIdToDocID[d1.id] - fieldIdToDocID[d2.id] : 0;
484             }
485           } else if (sf.getField().equals("sort1")) {
486             cmp = d1.sort1.compareTo(d2.sort1);
487           } else if (sf.getField().equals("sort2")) {
488             cmp = d1.sort2.compareTo(d2.sort2);
489           } else if (sf.getField().equals("sort3")) {
490             cmp = d1.sort3.compareTo(d2.sort3);
491           } else {
492             assertEquals(sf.getField(), "id");
493             cmp = d1.id - d2.id;
494           }
495           if (cmp != 0) {
496             return sf.getReverse() ? -cmp : cmp;
497           }
498         }
499         // Our sort always fully tie breaks:
500         fail();
501         return 0;
502       }
503     };
504   }

506   @SuppressWarnings({"unchecked","rawtypes"})
507   private AbstractAllGroupHeadsCollector<?> createRandomCollector(String groupField, Sort sortWithinGroup) {
508     AbstractAllGroupHeadsCollector<? extends AbstractAllGroupHeadsCollector.GroupHead> collector;
509     if (random().nextBoolean()) {
510       ValueSource vs = new BytesRefFieldSource(groupField);
511       collector =  new FunctionAllGroupHeadsCollector(vs, new HashMap<>(), sortWithinGroup);
512     } else {
513       collector =  TermAllGroupHeadsCollector.create(groupField, sortWithinGroup);
514     }
515 
516     if (VERBOSE) {
517       System.out.println("Selected implementation: " + collector.getClass().getSimpleName());
518     }
519 
520     return collector;
521   }

523   private void addGroupField(Document doc, String groupField, String value, DocValuesType valueType) {
524     Field valuesField = null;
525     switch(valueType) {
526       case BINARY:
527         valuesField = new BinaryDocValuesField(groupField, new BytesRef(value));
528         break;
529       case SORTED:
530         valuesField = new SortedDocValuesField(groupField, new BytesRef(value));
531         break;
532       default:
533         fail("unhandled type");
534     }
535     doc.add(valuesField);
536   }
