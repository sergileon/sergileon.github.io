20  import org.apache.lucene.misc.SweetSpotSimilarity;
21  import org.apache.lucene.search.similarities.DefaultSimilarity;
22  import org.apache.lucene.search.similarities.Similarity;
23  import org.junit.BeforeClass;
24  
25  /**
26   * Tests {@link SweetSpotSimilarityFactory}
27   */
28  public class TestSweetSpotSimilarityFactory extends BaseSimilarityTestCase {
29    @BeforeClass
30    public static void beforeClass() throws Exception {
31      initCore("solrconfig-basic.xml","schema-sweetspot.xml");
32    }
33    
34    /** default parameters */
35    public void testDefaults() throws Exception {
36      SweetSpotSimilarity sim = getSimilarity("text", SweetSpotSimilarity.class);
37  
38      // SSS tf w/defaults should behave just like DS
39      DefaultSimilarity d = new DefaultSimilarity();
40      for (int i = 0; i <=1000; i++) {
41        assertEquals("tf: i="+i, d.tf(i), sim.tf(i), 0.0F);
42      }
43  
44      // default norm sanity check
45      assertEquals("norm 1",  1.00F, sim.computeLengthNorm(1),  0.0F);
46      assertEquals("norm 4",  0.50F, sim.computeLengthNorm(4),  0.0F);
47      assertEquals("norm 16", 0.25F, sim.computeLengthNorm(16), 0.0F);
48    }
49    
50    /** baseline with parameters */
51    public void testBaselineParameters() throws Exception {
52      SweetSpotSimilarity sim = getSimilarity("text_baseline", 
53                                              SweetSpotSimilarity.class);
54      
55      DefaultSimilarity d = new DefaultSimilarity();
56  
57      // constant up to 6
58      for (int i = 1; i <=6; i++) {
59        assertEquals("tf i="+i, 1.5F, sim.tf(i), 0.0F);
60      }
61      // less then default sim above 6
62      for (int i = 6; i <=1000; i++) {
63        assertTrue("tf: i="+i+" : s="+sim.tf(i)+
64                   " < d="+d.tf(i),
65                   sim.tf(i) < d.tf(i));
66      }
67  
68      // norms: plateau from 3-5
69      assertEquals("norm 1 == 7", 
70                   sim.computeLengthNorm(1), sim.computeLengthNorm(7),  0.0F);
71      assertEquals("norm 2 == 6",  
72                   sim.computeLengthNorm(1), sim.computeLengthNorm(7),  0.0F);
73      assertEquals("norm 3",  1.00F, sim.computeLengthNorm(3),  0.0F);
74      assertEquals("norm 4",  1.00F, sim.computeLengthNorm(4),  0.0F);
75      assertEquals("norm 5",  1.00F, sim.computeLengthNorm(5),  0.0F);
76      assertTrue("norm 6 too high: " + sim.computeLengthNorm(6),
77                 sim.computeLengthNorm(6) < 1.0F);
78      assertTrue("norm 7 higher then norm 6", 
79                 sim.computeLengthNorm(7) < sim.computeLengthNorm(6));
80      assertEquals("norm 20", 0.25F, sim.computeLengthNorm(20), 0.0F);
81    }
82  
83    /** hyperbolic with parameters */
84    public void testHyperbolicParameters() throws Exception {
85      SweetSpotSimilarity sim = getSimilarity("text_hyperbolic", 
86                                              SweetSpotSimilarity.class);
87  
88      for (int i = 1; i <=1000; i++) {
89        assertTrue("MIN tf: i="+i+" : s="+sim.tf(i),
90                   3.3F <= sim.tf(i));
91        assertTrue("MAX tf: i="+i+" : s="+sim.tf(i),
92                   sim.tf(i) <= 7.7F);
93      }
94      assertEquals("MID tf", 3.3F+(7.7F - 3.3F)/2.0F, sim.tf(5), 0.00001F);
95  
96      // norms: plateau from 1-5, shallow slope
97      assertEquals("norm 1",  1.00F, sim.computeLengthNorm(1),  0.0F);
98      assertEquals("norm 2",  1.00F, sim.computeLengthNorm(2),  0.0F);
99      assertEquals("norm 3",  1.00F, sim.computeLengthNorm(3),  0.0F);
100     assertEquals("norm 4",  1.00F, sim.computeLengthNorm(4),  0.0F);
101     assertEquals("norm 5",  1.00F, sim.computeLengthNorm(5),  0.0F);
102     assertTrue("norm 6 too high: " + sim.computeLengthNorm(6),
103                sim.computeLengthNorm(6) < 1.0F);
104     assertTrue("norm 7 higher then norm 6", 
105                sim.computeLengthNorm(7) < sim.computeLengthNorm(6));
106     assertTrue("norm 20 not high enough: " + sim.computeLengthNorm(20),
107                0.25F < sim.computeLengthNorm(20));
108   }
