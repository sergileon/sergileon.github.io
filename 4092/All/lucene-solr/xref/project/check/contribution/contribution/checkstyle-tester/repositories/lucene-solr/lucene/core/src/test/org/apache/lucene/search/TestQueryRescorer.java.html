43    private IndexSearcher getSearcher(IndexReader r) {
44      IndexSearcher searcher = newSearcher(r);

52    public void testBasic() throws Exception {
53      Directory dir = newDirectory();
54      RandomIndexWriter w = new RandomIndexWriter(random(), dir);
55  
56      Document doc = new Document();
57      doc.add(newStringField("id", "0", Field.Store.YES));
58      doc.add(newTextField("field", "wizard the the the the the oz", Field.Store.NO));
59      w.addDocument(doc);
60      doc = new Document();
61      doc.add(newStringField("id", "1", Field.Store.YES));
62      // 1 extra token, but wizard and oz are close;
63      doc.add(newTextField("field", "wizard oz the the the the the the", Field.Store.NO));
64      w.addDocument(doc);
65      IndexReader r = w.getReader();
66      w.close();
67  
68      // Do ordinary BooleanQuery:
69      BooleanQuery.Builder bq = new BooleanQuery.Builder();
70      bq.add(new TermQuery(new Term("field", "wizard")), Occur.SHOULD);
71      bq.add(new TermQuery(new Term("field", "oz")), Occur.SHOULD);
72      IndexSearcher searcher = getSearcher(r);
73      searcher.setSimilarity(new DefaultSimilarity());
74  
75      TopDocs hits = searcher.search(bq.build(), 10);
76      assertEquals(2, hits.totalHits);
77      assertEquals("0", searcher.doc(hits.scoreDocs[0].doc).get("id"));
78      assertEquals("1", searcher.doc(hits.scoreDocs[1].doc).get("id"));
79  
80      // Now, resort using PhraseQuery:
81      PhraseQuery pq = new PhraseQuery(5, "field", "wizard", "oz");
82  
83      TopDocs hits2 = QueryRescorer.rescore(searcher, hits, pq, 2.0, 10);
84  
85      // Resorting changed the order:
86      assertEquals(2, hits2.totalHits);
87      assertEquals("1", searcher.doc(hits2.scoreDocs[0].doc).get("id"));
88      assertEquals("0", searcher.doc(hits2.scoreDocs[1].doc).get("id"));
89  
90      // Resort using SpanNearQuery:
91      SpanTermQuery t1 = new SpanTermQuery(new Term("field", "wizard"));
92      SpanTermQuery t2 = new SpanTermQuery(new Term("field", "oz"));
93      SpanNearQuery snq = new SpanNearQuery(new SpanQuery[] {t1, t2}, 0, true);
94  
95      TopDocs hits3 = QueryRescorer.rescore(searcher, hits, snq, 2.0, 10);
96  
97      // Resorting changed the order:
98      assertEquals(2, hits3.totalHits);
99      assertEquals("1", searcher.doc(hits3.scoreDocs[0].doc).get("id"));
100     assertEquals("0", searcher.doc(hits3.scoreDocs[1].doc).get("id"));
101 
102     r.close();
103     dir.close();
104   }

150     Document doc = new Document();

155     doc.add(newStringField("id", "1", Field.Store.YES));
156     // 1 extra token, but wizard and oz are close;
157     doc.add(newTextField("field", "wizard oz the the the the the the", Field.Store.NO));
158     w.addDocument(doc);
159     IndexReader r = w.getReader();
160     w.close();
161 
162     // Do ordinary BooleanQuery:
163     BooleanQuery.Builder bq = new BooleanQuery.Builder();
164     bq.add(new TermQuery(new Term("field", "wizard")), Occur.SHOULD);
165     bq.add(new TermQuery(new Term("field", "oz")), Occur.SHOULD);
166     IndexSearcher searcher = getSearcher(r);
167 
168     TopDocs hits = searcher.search(bq.build(), 10);
169     assertEquals(2, hits.totalHits);
170     assertEquals("0", searcher.doc(hits.scoreDocs[0].doc).get("id"));
171     assertEquals("1", searcher.doc(hits.scoreDocs[1].doc).get("id"));
172 
173     // Now, resort using PhraseQuery, but with an
174     // opposite-world combine:
175     PhraseQuery pq = new PhraseQuery(5, "field", "wizard", "oz");
176     
177     TopDocs hits2 = new QueryRescorer(pq) {
178         @Override
179         protected float combine(float firstPassScore, boolean secondPassMatches, float secondPassScore) {
180           float score = firstPassScore;
181           if (secondPassMatches) {
182             score -= 2.0 * secondPassScore;

184           return score;
185         }
186       }.rescore(searcher, hits, 10);
187 
188     // Resorting didn't change the order:
189     assertEquals(2, hits2.totalHits);
190     assertEquals("0", searcher.doc(hits2.scoreDocs[0].doc).get("id"));
191     assertEquals("1", searcher.doc(hits2.scoreDocs[1].doc).get("id"));

197   public void testExplain() throws Exception {
198     Directory dir = newDirectory();
199     RandomIndexWriter w = new RandomIndexWriter(random(), dir);
200 
201     Document doc = new Document();
202     doc.add(newStringField("id", "0", Field.Store.YES));
203     doc.add(newTextField("field", "wizard the the the the the oz", Field.Store.NO));
204     w.addDocument(doc);
205     doc = new Document();
206     doc.add(newStringField("id", "1", Field.Store.YES));
207     // 1 extra token, but wizard and oz are close;
208     doc.add(newTextField("field", "wizard oz the the the the the the", Field.Store.NO));
209     w.addDocument(doc);
210     IndexReader r = w.getReader();
211     w.close();
212 
213     // Do ordinary BooleanQuery:
214     BooleanQuery.Builder bq = new BooleanQuery.Builder();
215     bq.add(new TermQuery(new Term("field", "wizard")), Occur.SHOULD);
216     bq.add(new TermQuery(new Term("field", "oz")), Occur.SHOULD);
217     IndexSearcher searcher = getSearcher(r);
218 
219     TopDocs hits = searcher.search(bq.build(), 10);
220     assertEquals(2, hits.totalHits);
221     assertEquals("0", searcher.doc(hits.scoreDocs[0].doc).get("id"));
222     assertEquals("1", searcher.doc(hits.scoreDocs[1].doc).get("id"));
223 
224     // Now, resort using PhraseQuery:
225     PhraseQuery pq = new PhraseQuery("field", "wizard", "oz");
226 
227     Rescorer rescorer = new QueryRescorer(pq) {
228         @Override
229         protected float combine(float firstPassScore, boolean secondPassMatches, float secondPassScore) {
230           float score = firstPassScore;
231           if (secondPassMatches) {
232             score += 2.0 * secondPassScore;
233           }
234           return score;
235         }
236       };
237 
238     TopDocs hits2 = rescorer.rescore(searcher, hits, 10);
239 
240     // Resorting changed the order:
241     assertEquals(2, hits2.totalHits);
242     assertEquals("1", searcher.doc(hits2.scoreDocs[0].doc).get("id"));
243     assertEquals("0", searcher.doc(hits2.scoreDocs[1].doc).get("id"));
244 
245     int docID = hits2.scoreDocs[0].doc;
246     Explanation explain = rescorer.explain(searcher,
247                                            searcher.explain(bq.build(), docID),
248                                            docID);
249     String s = explain.toString();
250     assertTrue(s.contains("TestQueryRescorer$"));
251     assertTrue(s.contains("combined first and second pass score"));
252     assertTrue(s.contains("first pass score"));
253     assertTrue(s.contains("= second pass score"));
254     assertEquals(hits2.scoreDocs[0].score, explain.getValue(), 0.0f);
255 
256     docID = hits2.scoreDocs[1].doc;
257     explain = rescorer.explain(searcher,
258                                searcher.explain(bq.build(), docID),
259                                docID);
260     s = explain.toString();
261     assertTrue(s.contains("TestQueryRescorer$"));
262     assertTrue(s.contains("combined first and second pass score"));
263     assertTrue(s.contains("first pass score"));
264     assertTrue(s.contains("no second pass score"));
265     assertFalse(s.contains("= second pass score"));
266     assertEquals(hits2.scoreDocs[1].score, explain.getValue(), 0.0f);
267 
268     r.close();
269     dir.close();
270   }

325   public void testRandom() throws Exception {
326     Directory dir = newDirectory();
327     int numDocs = atLeast(1000);
328     RandomIndexWriter w = new RandomIndexWriter(random(), dir);
329 
330     final int[] idToNum = new int[numDocs];
331     int maxValue = TestUtil.nextInt(random(), 10, 1000000);
332     for(int i=0;i<numDocs;i++) {
333       Document doc = new Document();
334       doc.add(newStringField("id", ""+i, Field.Store.YES));
335       int numTokens = TestUtil.nextInt(random(), 1, 10);
336       StringBuilder b = new StringBuilder();
337       for(int j=0;j<numTokens;j++) {
338         b.append("a ");
339       }
340       doc.add(newTextField("field", b.toString(), Field.Store.NO));
341       idToNum[i] = random().nextInt(maxValue);
342       doc.add(new NumericDocValuesField("num", idToNum[i]));
343       w.addDocument(doc);
344     }
345     final IndexReader r = w.getReader();
346     w.close();
347 
348     IndexSearcher s = newSearcher(r);
349     int numHits = TestUtil.nextInt(random(), 1, numDocs);
350     boolean reverse = random().nextBoolean();
351 
352     //System.out.println("numHits=" + numHits + " reverse=" + reverse);
353     TopDocs hits = s.search(new TermQuery(new Term("field", "a")), numHits);
354 
355     TopDocs hits2 = new QueryRescorer(new FixedScoreQuery(idToNum, reverse)) {
356         @Override
357         protected float combine(float firstPassScore, boolean secondPassMatches, float secondPassScore) {
358           return secondPassScore;
359         }
360       }.rescore(s, hits, numHits);
361 
362     Integer[] expected = new Integer[numHits];
363     for(int i=0;i<numHits;i++) {
364       expected[i] = hits.scoreDocs[i].doc;
365     }
366 
367     final int reverseInt = reverse ? -1 : 1;
368 
369     Arrays.sort(expected,
370                 new Comparator<Integer>() {
371                   @Override
372                   public int compare(Integer a, Integer b) {
373                     try {
374                       int av = idToNum[Integer.parseInt(r.document(a).get("id"))];
375                       int bv = idToNum[Integer.parseInt(r.document(b).get("id"))];
376                       if (av < bv) {
377                         return -reverseInt;
378                       } else if (bv < av) {
379                         return reverseInt;
380                       } else {
381                         // Tie break by docID, ascending
382                         return a - b;
383                       }
384                     } catch (IOException ioe) {
385                       throw new RuntimeException(ioe);
386                     }
387                   }
388                 });
389 
390     boolean fail = false;
391     for(int i=0;i<numHits;i++) {
392       //System.out.println("expected=" + expected[i] + " vs " + hits2.scoreDocs[i].doc + " v=" + idToNum[Integer.parseInt(r.document(expected[i]).get("id"))]);
393       if (expected[i].intValue() != hits2.scoreDocs[i].doc) {
394         //System.out.println("  diff!");
395         fail = true;
396       }
397     }
398     assertFalse(fail);
399 
400     r.close();
401     dir.close();
402   }

404   /** Just assigns score == idToNum[doc("id")] for each doc. */
405   private static class FixedScoreQuery extends Query {
406     private final int[] idToNum;
407     private final boolean reverse;
408 
409     public FixedScoreQuery(int[] idToNum, boolean reverse) {
410       this.idToNum = idToNum;
411       this.reverse = reverse;
412     }
413 
414     @Override
415     public Weight createWeight(IndexSearcher searcher, boolean needsScores) throws IOException {
416 
417       return new Weight(FixedScoreQuery.this) {
418 
419         @Override
420         public void extractTerms(Set<Term> terms) {
421         }
422 
423         @Override
424         public float getValueForNormalization() {
425           return 1.0f;
426         }
427 
428         @Override
429         public void normalize(float queryNorm, float topLevelBoost) {
430         }
431 
432         @Override
433         public Scorer scorer(final LeafReaderContext context) throws IOException {
434 
435           return new Scorer(null) {
436             int docID = -1;
437 
438             @Override
439             public int docID() {
440               return docID;
441             }
442 
443             @Override
444             public int freq() {
445               return 1;
446             }
447 
448             @Override
449             public long cost() {
450               return 1;
451             }
452 
453             @Override
454             public int nextDoc() {
455               docID++;
456               if (docID >= context.reader().maxDoc()) {
457                 return NO_MORE_DOCS;
458               }
459               return docID;
460             }
461 
462             @Override
463             public int advance(int target) {
464               docID = target;
465               return docID;
466             }
467 
468             @Override
469             public float score() throws IOException {
470               int num = idToNum[Integer.parseInt(context.reader().document(docID).get("id"))];
471               if (reverse) {
472                 //System.out.println("score doc=" + docID + " num=" + num);
473                 return num;
474               } else {
475                 //System.out.println("score doc=" + docID + " num=" + -num);
476                 return -num;
477               }
478             }
479           };
480         }
481 
482         @Override
483         public Explanation explain(LeafReaderContext context, int doc) throws IOException {
484           return null;
485         }
486       };
487     }
488 
489     @Override
490     public String toString(String field) {
491       return "FixedScoreQuery " + idToNum.length + " ids; reverse=" + reverse;
492     }
493 
494     @Override
495     public boolean equals(Object o) {
496       if ((o instanceof FixedScoreQuery) == false) {
497         return false;
498       }
499       FixedScoreQuery other = (FixedScoreQuery) o;
500       return super.equals(o) &&
501         reverse == other.reverse &&
502         Arrays.equals(idToNum, other.idToNum);
503     }
504 
505     @Override
506     public Query clone() {
507       return new FixedScoreQuery(idToNum, reverse);
508     }
509 
510     @Override
511     public int hashCode() {
512       int PRIME = 31;
513       int hash = super.hashCode();
514       if (reverse) {
515         hash = PRIME * hash + 3623;
516       }
517       hash = PRIME * hash + Arrays.hashCode(idToNum);
518       return hash;
519     }
520   }
