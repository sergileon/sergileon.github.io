38  public abstract class RangeEndpointCalculator<T extends Comparable<T>> {
39    protected final SchemaField field;
40    protected final RangeFacetRequest request;
41    
42    public RangeEndpointCalculator(final RangeFacetRequest request) {
43      this.field = request.getField();
44      this.request = request;
45    }
46  
47    /**
48     * Formats a Range endpoint for use as a range label name in the response.
49     * Default Impl just uses toString()
50     */
51    public String formatValue(final T val) {
52      return val.toString();
53    }
54    
55    /**
56     * Parses a String param into an Range endpoint value throwing 
57     * a useful exception if not possible
58     */
59    public final T getValue(final String rawval) {
60      try {
61        return parseVal(rawval);
62      } catch (Exception e) {
63        throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Can't parse value "+rawval+" for field: " + field.getName(), e);
64      }
65    }
66    
67    /**
68     * Parses a String param into an Range endpoint. 
69     * Can throw a low level format exception as needed.
70     */
71    protected abstract T parseVal(final String rawval) throws java.text.ParseException;
72  
73    /** 
74     * Parses a String param into a value that represents the gap and 
75     * can be included in the response, throwing 
76     * a useful exception if not possible.
77     *
78     * Note: uses Object as the return type instead of T for things like 
79     * Date where gap is just a DateMathParser string 
80     */
81    public final Object getGap(final String gap) {
82      try {
83        return parseGap(gap);
84      } catch (Exception e) {
85        throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Can't parse gap "+gap+" for field: " + field.getName(), e);
86      }
87    }
88  
89    /**
90     * Parses a String param into a value that represents the gap and 
91     * can be included in the response. 
92     * Can throw a low level format exception as needed.
93     *
94     * Default Impl calls parseVal
95     */
96    protected Object parseGap(final String rawval) throws java.text.ParseException {
97      return parseVal(rawval);
98    }
99  
100   /**
101    * Adds the String gap param to a low Range endpoint value to determine 
102    * the corrisponding high Range endpoint value, throwing 
103    * a useful exception if not possible.
104    */
105   public final T addGap(T value, String gap) {
106     try {
107       return parseAndAddGap(value, gap);
108     } catch (Exception e) {
109       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Can't add gap "+gap+" to value " + value + " for field: " + field.getName(), e);
110     }
111   }
112   
113   /**
114    * Adds the String gap param to a low Range endpoint value to determine 
115    * the corrisponding high Range endpoint value.
116    * Can throw a low level format exception as needed.
117    */
118   protected abstract T parseAndAddGap(T value, String gap) throws java.text.ParseException;

120   public static class FacetRange {
121     public final String name;
122     public final String lower;
123     public final String upper;
124     public final boolean includeLower;
125     public final boolean includeUpper;
126     
127     public FacetRange(String name, String lower, String upper, boolean includeLower, boolean includeUpper) {
128       this.name = name;
129       this.lower = lower;
130       this.upper = upper;
131       this.includeLower = includeLower;
132       this.includeUpper = includeUpper;
133     }
134   }
135   
136   public List<FacetRange> getRanges(){
137 
138     final T start = getValue(request.getStart());
139     T end = getValue(request.getEnd()); // not final, hardend may change this
140     
141     if( end.compareTo(start) < 0 ){
142       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "range facet 'end' comes before 'start': "+end+" < "+start);
143     }
144     
145     // explicitly return the gap.  compute this early so we are more 
146     // likely to catch parse errors before attempting math
147     final String[] gaps = request.getGaps();
148     String gap = gaps[0];
149     
150     final EnumSet<FacetRangeInclude> include = request.getInclude();
151         
152     T low = start;
153     
154     List<FacetRange> ranges = new ArrayList<>();
155     
156     int gapCounter = 0;
157     
158     while (low.compareTo(end) < 0) {
159       if (gapCounter<gaps.length) {
160         gap = gaps[gapCounter++];
161       }
162       T high = addGap(low,gap);
163       if (end.compareTo(high) < 0) {
164         if (request.isHardEnd()){
165           high = end;
166         } else {
167           end = high;
168         }
169       }
170       
171       if (high.compareTo(low) < 0) {
172         throw new SolrException (SolrException.ErrorCode.BAD_REQUEST, "range facet infinite loop (is gap negative? did the math overflow?)");
173       }
174       
175       if (high.compareTo(low) == 0) {
176         throw new SolrException (SolrException.ErrorCode.BAD_REQUEST, "range facet infinite loop: gap is either zero, or too small relative start/end and caused underflow: " + low + " + " + gap + " = " + high );
177       }
178       
179       final boolean includeLower = (include.contains(FacetRangeInclude.ALL) ||
180                                     include.contains(FacetRangeInclude.LOWER) ||
181                                    (include.contains(FacetRangeInclude.EDGE) && 
182                                    0 == low.compareTo(start)));
183       final boolean includeUpper = (include.contains(FacetRangeInclude.ALL) ||
184                                     include.contains(FacetRangeInclude.UPPER) ||
185                                    (include.contains(FacetRangeInclude.EDGE) && 
186                                    0 == high.compareTo(end)));
187       
188       final String lowS = formatValue(low);
189       final String highS = formatValue(high);
190 
191       ranges.add( new FacetRange(lowS,lowS,highS,includeLower,includeUpper) );
192       low = high;
193     }
194     
195     final Set<FacetRangeOther> others = request.getOthers();
196     if (null != others && 0 < others.size() ) {
197       
198       // no matter what other values are listed, we don't do
199       // anything if "none" is specified.
200       if( !others.contains(FacetRangeOther.NONE) ) {
201         
202         boolean all = others.contains(FacetRangeOther.ALL);
203 
204         if (all || others.contains(FacetRangeOther.BEFORE)) {
205           // include upper bound if "outer" or if first gap doesn't already include it
206           ranges.add( new FacetRange(FacetRangeOther.BEFORE.toString(), 
207                                         null, formatValue(start), false, include.contains(FacetRangeInclude.OUTER) || include.contains(FacetRangeInclude.ALL) ||
208                                                             !(include.contains(FacetRangeInclude.LOWER) || include.contains(FacetRangeInclude.EDGE)) ) );
209           
210         }
211         if (all || others.contains(FacetRangeOther.AFTER)) {
212           // include lower bound if "outer" or if last gap doesn't already include it
213           ranges.add( new FacetRange(FacetRangeOther.AFTER.toString(), 
214                                         formatValue(end), null, include.contains(FacetRangeInclude.OUTER) || include.contains(FacetRangeInclude.ALL) ||
215                                                    !(include.contains(FacetRangeInclude.UPPER) || include.contains(FacetRangeInclude.EDGE)), false) );
216         }
217         if (all || others.contains(FacetRangeOther.BETWEEN)) {
218           ranges.add( new FacetRange(FacetRangeOther.BETWEEN.toString(), formatValue(start), formatValue(end),
219                                         include.contains(FacetRangeInclude.LOWER) || include.contains(FacetRangeInclude.EDGE) || include.contains(FacetRangeInclude.ALL),
220                                         include.contains(FacetRangeInclude.UPPER) || include.contains(FacetRangeInclude.EDGE) || include.contains(FacetRangeInclude.ALL)) );
221         }
222       }
223       
224     }
225   
226     return ranges;
227   }

229   public static RangeEndpointCalculator<? extends Comparable<?>> create(RangeFacetRequest request){
230     final SchemaField sf = request.getField();
231     final FieldType ft = sf.getType();
232     final RangeEndpointCalculator<?> calc;
233     if (ft instanceof TrieField) {
234       final TrieField trie = (TrieField)ft;
235       switch (trie.getType()) {
236         case FLOAT:
237           calc = new FloatRangeEndpointCalculator(request);
238           break;
239         case DOUBLE:
240           calc = new DoubleRangeEndpointCalculator(request);
241           break;
242         case INTEGER:
243           calc = new IntegerRangeEndpointCalculator(request);
244           break;
245         case LONG:
246           calc = new LongRangeEndpointCalculator(request);
247           break;
248         case DATE:
249           calc = new DateRangeEndpointCalculator(request, null);
250           break;
251         default:
252           throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Unable to range facet on tried field of unexpected type:" + sf.getName());
253       }
254     } else {
255       throw new SolrException (SolrException.ErrorCode.BAD_REQUEST, "Unable to range facet on field:" + sf);
256     } 
257     return calc;
258   }

260   public static class FloatRangeEndpointCalculator extends RangeEndpointCalculator<Float> {
261   
262     public FloatRangeEndpointCalculator(final RangeFacetRequest request) { super(request); }
263     
264     @Override
265     protected Float parseVal(String rawval) {
266       return Float.valueOf(rawval);
267     }
268     
269     @Override
270     public Float parseAndAddGap(Float value, String gap) {
271       return new Float(value.floatValue() + Float.valueOf(gap).floatValue());
272     }

324   public static class DateRangeEndpointCalculator extends RangeEndpointCalculator<Date> {
325     private final Date now;
326     public DateRangeEndpointCalculator(final RangeFacetRequest request, final Date now) { 
327       super(request); 
328       this.now = now;
329       if (! (field.getType() instanceof TrieDateField) ) {
330         throw new IllegalArgumentException("SchemaField must use field type extending TrieDateField");
331       }
332     }
333     
334     @Override
335     public String formatValue(Date val) {
336       return DateFormatUtil.formatExternal(val);
337     }
338     
339     @Override
340     protected Date parseVal(String rawval) {
341       return DateFormatUtil.parseMath(now, rawval);
342     }
343     
344     @Override
345     protected Object parseGap(final String rawval) {
346       return rawval;
347     }
348     
349     @Override
350     public Date parseAndAddGap(Date value, String gap) throws java.text.ParseException {
351       final DateMathParser dmp = new DateMathParser();
352       dmp.setNow(value);
353       return dmp.parseMath(gap);
354     }
