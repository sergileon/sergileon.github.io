82  /**
83   * A SolrClient implementation that talks directly to a Solr server via HTTP
84   *
85   * There are two ways to use an HttpSolrClient:
86   *
87   * 1) Pass a URL to the constructor that points directly at a particular core
88   * <pre>
89   *   SolrClient client = new HttpSolrClient("http://my-solr-server:8983/solr/core1");
90   *   QueryResponse resp = client.query(new SolrQuery("*:*"));
91   * </pre>
92   * In this case, you can query the given core directly, but you cannot query any other
93   * cores or issue CoreAdmin requests with this client.
94   *
95   * 2) Pass the base URL of the node to the constructor
96   * <pre>
97   *   SolrClient client = new HttpSolrClient("http://my-solr-server:8983/solr");
98   *   QueryResponse resp = client.query("core1", new SolrQuery("*:*"));
99   * </pre>
100  * In this case, you must pass the name of the required core for all queries and updates,
101  * but you may use the same client for all cores, and for CoreAdmin requests.
102  */
103 public class HttpSolrClient extends SolrClient {

169   public HttpSolrClient(String baseURL, HttpClient client, ResponseParser parser) {
170     this.baseUrl = baseURL;
171     if (baseUrl.endsWith("/")) {
172       baseUrl = baseUrl.substring(0, baseUrl.length() - 1);
173     }
174     if (baseUrl.indexOf('?') >= 0) {
175       throw new RuntimeException(
176           "Invalid base url for solrj.  The base URL must not contain parameters: "
177               + baseUrl);
178     }
179     
180     if (client != null) {
181       httpClient = client;
182       internalClient = false;
183     } else {
184       internalClient = true;
185       ModifiableSolrParams params = new ModifiableSolrParams();
186       params.set(HttpClientUtil.PROP_MAX_CONNECTIONS, 128);
187       params.set(HttpClientUtil.PROP_MAX_CONNECTIONS_PER_HOST, 32);
188       params.set(HttpClientUtil.PROP_FOLLOW_REDIRECTS, followRedirects);
189       httpClient = HttpClientUtil.createClient(params);
190     }
191     
192     this.parser = parser;
193   }

225       throws SolrServerException, IOException {
226     ResponseParser responseParser = request.getResponseParser();
227     if (responseParser == null) {
228       responseParser = parser;
229     }
230     return request(request, responseParser, collection);
231   }

237   public NamedList<Object> request(final SolrRequest request, final ResponseParser processor, String collection)
238       throws SolrServerException, IOException {
239     HttpRequestBase method = createMethod(request, collection);
240     setBasicAuthHeader(request, method);
241     return executeMethod(method, processor);
242   }
243 
244   private void setBasicAuthHeader(SolrRequest request, HttpRequestBase method) throws UnsupportedEncodingException {
245     if (request.getBasicAuthUser() != null && request.getBasicAuthPassword() != null) {
246       String userPass = request.getBasicAuthUser() + ":" + request.getBasicAuthPassword();
247       String encoded = Base64.byteArrayToBase64(userPass.getBytes(UTF_8));
248       method.setHeader(new BasicHeader("Authorization", "Basic " + encoded));
249     }
250   }

273    * @lucene.experimental
274    */
275   public HttpUriRequestResponse httpUriRequest(final SolrRequest request, final ResponseParser processor) throws SolrServerException, IOException {
276     HttpUriRequestResponse mrr = new HttpUriRequestResponse();
277     final HttpRequestBase method = createMethod(request, null);
278     ExecutorService pool = ExecutorUtil.newMDCAwareFixedThreadPool(1, new SolrjNamedThreadFactory("httpUriRequest"));
279     try {
280       MDC.put("HttpSolrClient.url", baseUrl);
281       mrr.future = pool.submit(new Callable<NamedList<Object>>(){
282 
283         @Override
284         public NamedList<Object> call() throws Exception {
285           return executeMethod(method, processor);
286         }});
287  
288     } finally {
289       pool.shutdown();
290       MDC.remove("HttpSolrClient.url");
291     }
292     assert method != null;
293     mrr.httpUriRequest = method;
294     return mrr;
295   }

297   protected ModifiableSolrParams calculateQueryParams(Set<String> queryParamNames,
298       ModifiableSolrParams wparams) {
299     ModifiableSolrParams queryModParams = new ModifiableSolrParams();
300     if (queryParamNames != null) {
301       for (String param : queryParamNames) {
302         String[] value = wparams.getParams(param) ;
303         if (value != null) {
304           for (String v : value) {
305             queryModParams.add(param, v);
306           }
307           wparams.remove(param);
308         }
309       }
310     }
311     return queryModParams;
312   }

314   protected HttpRequestBase createMethod(final SolrRequest request, String collection) throws IOException, SolrServerException {
315 
316     SolrParams params = request.getParams();
317     Collection<ContentStream> streams = requestWriter.getContentStreams(request);
318     String path = requestWriter.getPath(request);
319     if (path == null || !path.startsWith("/")) {
320       path = DEFAULT_PATH;
321     }
322     
323     ResponseParser parser = request.getResponseParser();
324     if (parser == null) {
325       parser = this.parser;
326     }
327     
328     // The parser 'wt=' and 'version=' params are used instead of the original
329     // params
330     ModifiableSolrParams wparams = new ModifiableSolrParams(params);
331     if (parser != null) {
332       wparams.set(CommonParams.WT, parser.getWriterType());
333       wparams.set(CommonParams.VERSION, parser.getVersion());
334     }
335     if (invariantParams != null) {
336       wparams.add(invariantParams);
337     }
338 
339     String basePath = baseUrl;
340     if (collection != null)
341       basePath += "/" + collection;
342 
343     if (SolrRequest.METHOD.GET == request.getMethod()) {
344       if (streams != null) {
345         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "GET can't send streams!");
346       }
347       return new HttpGet(basePath + path + wparams.toQueryString());
348     }
349 
350     if (SolrRequest.METHOD.POST == request.getMethod() || SolrRequest.METHOD.PUT == request.getMethod()) {
351 
352       String url = basePath + path;
353       boolean hasNullStreamName = false;
354       if (streams != null) {
355         for (ContentStream cs : streams) {
356           if (cs.getName() == null) {
357             hasNullStreamName = true;
358             break;
359           }
360         }
361       }
362       boolean isMultipart = ((this.useMultiPartPost && SolrRequest.METHOD.POST == request.getMethod())
363           || (streams != null && streams.size() > 1)) && !hasNullStreamName;
364 
365       LinkedList<NameValuePair> postOrPutParams = new LinkedList<>();
366       if (streams == null || isMultipart) {
367         // send server list and request list as query string params
368         ModifiableSolrParams queryParams = calculateQueryParams(this.queryParams, wparams);
369         queryParams.add(calculateQueryParams(request.getQueryParams(), wparams));
370         String fullQueryUrl = url + queryParams.toQueryString();
371         HttpEntityEnclosingRequestBase postOrPut = SolrRequest.METHOD.POST == request.getMethod() ?
372             new HttpPost(fullQueryUrl) : new HttpPut(fullQueryUrl);
373         if (!isMultipart) {
374           postOrPut.addHeader("Content-Type",
375               "application/x-www-form-urlencoded; charset=UTF-8");
376         }
377 
378         List<FormBodyPart> parts = new LinkedList<>();
379         Iterator<String> iter = wparams.getParameterNamesIterator();
380         while (iter.hasNext()) {
381           String p = iter.next();
382           String[] vals = wparams.getParams(p);
383           if (vals != null) {
384             for (String v : vals) {
385               if (isMultipart) {
386                 parts.add(new FormBodyPart(p, new StringBody(v, StandardCharsets.UTF_8)));
387               } else {
388                 postOrPutParams.add(new BasicNameValuePair(p, v));
389               }
390             }
391           }
392         }
393 
394         if (isMultipart && streams != null) {
395           for (ContentStream content : streams) {
396             String contentType = content.getContentType();
397             if (contentType == null) {
398               contentType = BinaryResponseParser.BINARY_CONTENT_TYPE; // default
399             }
400             String name = content.getName();
401             if (name == null) {
402               name = "";
403             }
404             parts.add(new FormBodyPart(name,
405                 new InputStreamBody(
406                     content.getStream(),
407                     contentType,
408                     content.getName())));
409           }
410         }
411 
412         if (parts.size() > 0) {
413           MultipartEntity entity = new MultipartEntity(HttpMultipartMode.STRICT);
414           for (FormBodyPart p : parts) {
415             entity.addPart(p);
416           }
417           postOrPut.setEntity(entity);
418         } else {
419           //not using multipart
420           postOrPut.setEntity(new UrlEncodedFormEntity(postOrPutParams, StandardCharsets.UTF_8));
421         }
422 
423         return postOrPut;
424       }
425       // It is has one stream, it is the post body, put the params in the URL
426       else {
427         String fullQueryUrl = url + wparams.toQueryString();
428         HttpEntityEnclosingRequestBase postOrPut = SolrRequest.METHOD.POST == request.getMethod() ?
429             new HttpPost(fullQueryUrl) : new HttpPut(fullQueryUrl);
430 
431         // Single stream as body
432         // Using a loop just to get the first one
433         final ContentStream[] contentStream = new ContentStream[1];
434         for (ContentStream content : streams) {
435           contentStream[0] = content;
436           break;
437         }
438         if (contentStream[0] instanceof RequestWriter.LazyContentStream) {
439           postOrPut.setEntity(new InputStreamEntity(contentStream[0].getStream(), -1) {
440             @Override
441             public Header getContentType() {
442               return new BasicHeader("Content-Type", contentStream[0].getContentType());
443             }
444 
445             @Override
446             public boolean isRepeatable() {
447               return false;
448             }
449 
450           });
451         } else {
452           postOrPut.setEntity(new InputStreamEntity(contentStream[0].getStream(), -1) {
453             @Override
454             public Header getContentType() {
455               return new BasicHeader("Content-Type", contentStream[0].getContentType());
456             }
457 
458             @Override
459             public boolean isRepeatable() {
460               return false;
461             }
462           });
463         }
464         return postOrPut;
465       }
466     }
467 
468     throw new SolrServerException("Unsupported method: " + request.getMethod());
469 
470   }

472   protected NamedList<Object> executeMethod(HttpRequestBase method, final ResponseParser processor) throws SolrServerException {
473     method.addHeader("User-Agent", AGENT);
474     
475     InputStream respBody = null;
476     boolean shouldClose = true;
477     boolean success = false;
478     try {
479       // Execute the method.
480       final HttpResponse response = httpClient.execute(method);
481       int httpStatus = response.getStatusLine().getStatusCode();
482       
483       // Read the contents
484       respBody = response.getEntity().getContent();
485       Header ctHeader = response.getLastHeader("content-type");
486       String contentType;
487       if (ctHeader != null) {
488         contentType = ctHeader.getValue();
489       } else {
490         contentType = "";
491       }
492       
493       // handle some http level checks before trying to parse the response
494       switch (httpStatus) {
495         case HttpStatus.SC_OK:
496         case HttpStatus.SC_BAD_REQUEST:
497         case HttpStatus.SC_CONFLICT:  // 409
498           break;
499         case HttpStatus.SC_MOVED_PERMANENTLY:
500         case HttpStatus.SC_MOVED_TEMPORARILY:
501           if (!followRedirects) {
502             throw new SolrServerException("Server at " + getBaseURL()
503                 + " sent back a redirect (" + httpStatus + ").");
504           }
505           break;
506         default:
507           if (processor == null || "".equals(contentType)) {
508             throw new RemoteSolrException(baseUrl, httpStatus, "non ok status: " + httpStatus
509                 + ", message:" + response.getStatusLine().getReasonPhrase(),
510                 null);
511           }
512       }
513       if (processor == null || processor instanceof InputStreamResponseParser) {
514         
515         // no processor specified, return raw stream
516         NamedList<Object> rsp = new NamedList<>();
517         rsp.add("stream", respBody);
518         // Only case where stream should not be closed
519         shouldClose = false;
520         success = true;
521         return rsp;
522       }
523       
524       String procCt = processor.getContentType();
525       if (procCt != null) {
526         String procMimeType = ContentType.parse(procCt).getMimeType().trim().toLowerCase(Locale.ROOT);
527         String mimeType = ContentType.parse(contentType).getMimeType().trim().toLowerCase(Locale.ROOT);
528         if (!procMimeType.equals(mimeType)) {
529           // unexpected mime type
530           String msg = "Expected mime type " + procMimeType + " but got " + mimeType + ".";
531           Header encodingHeader = response.getEntity().getContentEncoding();
532           String encoding;
533           if (encodingHeader != null) {
534             encoding = encodingHeader.getValue();
535           } else {
536             encoding = "UTF-8"; // try UTF-8
537           }
538           try {
539             msg = msg + " " + IOUtils.toString(respBody, encoding);
540           } catch (IOException e) {
541             throw new RemoteSolrException(baseUrl, httpStatus, "Could not parse response with encoding " + encoding, e);
542           }
543           throw new RemoteSolrException(baseUrl, httpStatus, msg, null);
544         }
545       }
546       
547       NamedList<Object> rsp = null;
548       String charset = EntityUtils.getContentCharSet(response.getEntity());
549       try {
550         rsp = processor.processResponse(respBody, charset);
551       } catch (Exception e) {
552         throw new RemoteSolrException(baseUrl, httpStatus, e.getMessage(), e);
553       }
554       if (httpStatus != HttpStatus.SC_OK) {
555         NamedList<String> metadata = null;
556         String reason = null;
557         try {
558           NamedList err = (NamedList) rsp.get("error");
559           if (err != null) {
560             reason = (String) err.get("msg");
561             if(reason == null) {
562               reason = (String) err.get("trace");
563             }
564             metadata = (NamedList<String>)err.get("metadata");
565           }
566         } catch (Exception ex) {}
567         if (reason == null) {
568           StringBuilder msg = new StringBuilder();
569           msg.append(response.getStatusLine().getReasonPhrase())
570             .append("\n\n")
571             .append("request: ")
572             .append(method.getURI());
573           reason = java.net.URLDecoder.decode(msg.toString(), UTF_8);
574         }
575         RemoteSolrException rss = new RemoteSolrException(baseUrl, httpStatus, reason, null);
576         if (metadata != null) rss.setMetadata(metadata);
577         throw rss;
578       }
579       success = true;
580       return rsp;
581     } catch (ConnectException e) {
582       throw new SolrServerException("Server refused connection at: "
583           + getBaseURL(), e);
584     } catch (SocketTimeoutException e) {
585       throw new SolrServerException(
586           "Timeout occured while waiting response from server at: "
587               + getBaseURL(), e);
588     } catch (IOException e) {
589       throw new SolrServerException(
590           "IOException occured when talking to server at: " + getBaseURL(), e);
591     } finally {
592       if (respBody != null && shouldClose) {
593         try {
594           respBody.close();
595         } catch (IOException e) {
596           log.error("", e);
597         } finally {
598           if (!success) {
599             method.abort();
600           }
601         }
602       }
603     }
604   }

671   /**
672    * Configure whether the client should follow redirects or not.
673    * <p>
674    * This defaults to false under the assumption that if you are following a
675    * redirect to get to a Solr installation, something is misconfigured
676    * somewhere.
