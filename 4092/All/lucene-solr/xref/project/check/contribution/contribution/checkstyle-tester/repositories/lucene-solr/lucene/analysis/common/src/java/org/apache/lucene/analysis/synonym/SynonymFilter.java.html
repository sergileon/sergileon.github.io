41  /**
42   * Matches single or multi word synonyms in a token stream.
43   * This token stream cannot properly handle position
44   * increments != 1, ie, you should place this filter before
45   * filtering out stop words.
46   * 
47   * <p>Note that with the current implementation, parsing is
48   * greedy, so whenever multiple parses would apply, the rule
49   * starting the earliest and parsing the most tokens wins.
50   * For example if you have these rules:
51   *      
52   * <pre>
53   *   a -&gt; x
54   *   a b -&gt; y
55   *   b c d -&gt; z
56   * </pre>
57   *
58   * Then input <code>a b c d e</code> parses to <code>y b c
59   * d</code>, ie the 2nd rule "wins" because it started
60   * earliest and matched the most input tokens of other rules
61   * starting at that point.
62   *
63   * <p>A future improvement to this filter could allow
64   * non-greedy parsing, such that the 3rd rule would win, and
65   * also separately allow multiple parses, such that all 3
66   * rules would match, perhaps even on a rule by rule
67   * basis.</p>
68   *
69   * <p><b>NOTE</b>: when a match occurs, the output tokens
70   * associated with the matching rule are "stacked" on top of
71   * the input stream (if the rule had
72   * <code>keepOrig=true</code>) and also on top of another
73   * matched rule's output tokens.  This is not a correct
74   * solution, as really the output should be an arbitrary
75   * graph/lattice.  For example, with the above match, you
76   * would expect an exact <code>PhraseQuery</code> <code>"y b
77   * c"</code> to match the parsed tokens, but it will fail to
78   * do so.  This limitation is necessary because Lucene's
79   * TokenStream (and index) cannot yet represent an arbitrary
80   * graph.</p>
81   *
82   * <p><b>NOTE</b>: If multiple incoming tokens arrive on the
83   * same position, only the first token at that position is
84   * used for parsing.  Subsequent tokens simply pass through
85   * and are not parsed.  A future improvement would be to
86   * allow these tokens to also be matched.</p>
87   */

92  // TODO: a more efficient approach would be Aho/Corasick's
93  // algorithm
94  // http://en.wikipedia.org/wiki/Aho%E2%80%93Corasick_string_matching_algorithm
95  // It improves over the current approach here
96  // because it does not fully re-start matching at every
97  // token.  For example if one pattern is "a b c x"
98  // and another is "b c d" and the input is "a b c d", on
99  // trying to parse "a b c x" but failing when you got to x,
100 // rather than starting over again your really should
101 // immediately recognize that "b c d" matches at the next
102 // input.  I suspect this won't matter that much in
103 // practice, but it's possible on some set of synonyms it
104 // will.  We'd have to modify Aho/Corasick to enforce our
105 // conflict resolving (eg greedy matching) because that algo
106 // finds all matches.  This really amounts to adding a .*
107 // closure to the FST and then determinizing it.
108 //
109 // Another possible solution is described at http://www.cis.uni-muenchen.de/people/Schulz/Pub/dictle5.ps

130   // How many future input tokens have already been matched
131   // to a synonym; because the matching is "greedy" we don't
132   // try to do any more matching for such tokens:
133   private int inputSkipCount;
134 
135   // Hold all buffered (read ahead) stacked input tokens for
136   // a future position.  When multiple tokens are at the
137   // same position, we only store (and match against) the
138   // term for the first token at the position, but capture
139   // state for (and enumerate) all other tokens at this
140   // position:
141   private static class PendingInput {
142     final CharsRefBuilder term = new CharsRefBuilder();
143     AttributeSource.State state;
144     boolean keepOrig;
145     boolean matched;
146     boolean consumed = true;
147     int startOffset;
148     int endOffset;
149     
150     public void reset() {
151       state = null;
152       consumed = true;
153       keepOrig = false;
154       matched = false;
155     }
156   };
157 
158   // Rolling buffer, holding pending input tokens we had to
159   // clone because we needed to look ahead, indexed by
160   // position:
161   private final PendingInput[] futureInputs;
162 
163   // Holds pending output synonyms for one future position:
164   private static class PendingOutputs {
165     CharsRefBuilder[] outputs;
166     int[] endOffsets;
167     int[] posLengths;
168     int upto;
169     int count;
170     int posIncr = 1;
171     int lastEndOffset;
172     int lastPosLength;
173 
174     public PendingOutputs() {
175       outputs = new CharsRefBuilder[1];
176       endOffsets = new int[1];
177       posLengths = new int[1];
178     }
179 
180     public void reset() {
181       upto = count = 0;
182       posIncr = 1;
183     }
184 
185     public CharsRef pullNext() {
186       assert upto < count;
187       lastEndOffset = endOffsets[upto];
188       lastPosLength = posLengths[upto];
189       final CharsRefBuilder result = outputs[upto++];
190       posIncr = 0;
191       if (upto == count) {
192         reset();
193       }
194       return result.get();
195     }
196 
197     public int getLastEndOffset() {
198       return lastEndOffset;
199     }
200 
201     public int getLastPosLength() {
202       return lastPosLength;
203     }
204 
205     public void add(char[] output, int offset, int len, int endOffset, int posLength) {
206       if (count == outputs.length) {
207         outputs = Arrays.copyOf(outputs, ArrayUtil.oversize(1+count, RamUsageEstimator.NUM_BYTES_OBJECT_REF));
208       }
209       if (count == endOffsets.length) {
210         final int[] next = new int[ArrayUtil.oversize(1+count, RamUsageEstimator.NUM_BYTES_INT)];
211         System.arraycopy(endOffsets, 0, next, 0, count);
212         endOffsets = next;
213       }
214       if (count == posLengths.length) {
215         final int[] next = new int[ArrayUtil.oversize(1+count, RamUsageEstimator.NUM_BYTES_INT)];
216         System.arraycopy(posLengths, 0, next, 0, count);
217         posLengths = next;
218       }
219       if (outputs[count] == null) {
220         outputs[count] = new CharsRefBuilder();
221       }
222       outputs[count].copyChars(output, offset, len);
223       // endOffset can be -1, in which case we should simply
224       // use the endOffset of the input token, or X >= 0, in
225       // which case we use X as the endOffset for this output
226       endOffsets[count] = endOffset;
227       posLengths[count] = posLength;
228       count++;
229     }
230   };

236   private final PendingOutputs[] futureOutputs;

242   private int nextRead;

244   // True once we've read last token
245   private boolean finished;

255   private final CharsRefBuilder scratchChars = new CharsRefBuilder();

257   /**
258    * @param input input tokenstream
259    * @param synonyms synonym map
260    * @param ignoreCase case-folds input for matching with {@link Character#toLowerCase(int)}.
261    *                   Note, if you set this to true, it's your responsibility to lowercase
262    *                   the input entries when you create the {@link SynonymMap}
263    */
264   public SynonymFilter(TokenStream input, SynonymMap synonyms, boolean ignoreCase) {
265     super(input);
266     this.synonyms = synonyms;
267     this.ignoreCase = ignoreCase;
268     this.fst = synonyms.fst;
269     if (fst == null) {
270       throw new IllegalArgumentException("fst must be non-null");
271     }
272     this.fstReader = fst.getBytesReader();
273 
274     // Must be 1+ so that when roll buffer is at full
275     // lookahead we can distinguish this full buffer from
276     // the empty buffer:
277     rollBufferSize = 1+synonyms.maxHorizontalContext;
278 
279     futureInputs = new PendingInput[rollBufferSize];
280     futureOutputs = new PendingOutputs[rollBufferSize];
281     for(int pos=0;pos<rollBufferSize;pos++) {
282       futureInputs[pos] = new PendingInput();
283       futureOutputs[pos] = new PendingOutputs();
284     }
285 
286     //System.out.println("FSTFilt maxH=" + synonyms.maxHorizontalContext);
287 
288     scratchArc = new FST.Arc<>();
289   }

291   private void capture() {
292     captureCount++;
293     //System.out.println("  capture slot=" + nextWrite);
294     final PendingInput input = futureInputs[nextWrite];
295 
296     input.state = captureState();
297     input.consumed = false;
298     input.term.copyChars(termAtt.buffer(), 0, termAtt.length());
299 
300     nextWrite = rollIncr(nextWrite);
301 
302     // Buffer head should never catch up to tail:
303     assert nextWrite != nextRead;
304   }

306   /*
307    This is the core of this TokenFilter: it locates the
308    synonym matches and buffers up the results into
309    futureInputs/Outputs.
310 
311    NOTE: this calls input.incrementToken and does not
312    capture the state if no further tokens were checked.  So
313    caller must then forward state to our caller, or capture:
314   */
315   private int lastStartOffset;
316   private int lastEndOffset;
317 
318   private void parse() throws IOException {
319     //System.out.println("\nS: parse");
320 
321     assert inputSkipCount == 0;
322 
323     int curNextRead = nextRead;
324 
325     // Holds the longest match we've seen so far:
326     BytesRef matchOutput = null;
327     int matchInputLength = 0;
328     int matchEndOffset = -1;
329 
330     BytesRef pendingOutput = fst.outputs.getNoOutput();
331     fst.getFirstArc(scratchArc);
332 
333     assert scratchArc.output == fst.outputs.getNoOutput();
334 
335     int tokenCount = 0;
336 
337     byToken:
338     while(true) {
339       
340       // Pull next token's chars:
341       final char[] buffer;
342       final int bufferLen;
343       //System.out.println("  cycle nextRead=" + curNextRead + " nextWrite=" + nextWrite);
344 
345       int inputEndOffset = 0;
346 
347       if (curNextRead == nextWrite) {
348 
349         // We used up our lookahead buffer of input tokens
350         // -- pull next real input token:
351 
352         if (finished) {
353           break;
354         } else  {
355           //System.out.println("  input.incrToken");
356           assert futureInputs[nextWrite].consumed;
357           // Not correct: a syn match whose output is longer
358           // than its input can set future inputs keepOrig
359           // to true:
360           //assert !futureInputs[nextWrite].keepOrig;
361           if (input.incrementToken()) {
362             buffer = termAtt.buffer();
363             bufferLen = termAtt.length();
364             final PendingInput input = futureInputs[nextWrite];
365             lastStartOffset = input.startOffset = offsetAtt.startOffset();
366             lastEndOffset = input.endOffset = offsetAtt.endOffset();
367             inputEndOffset = input.endOffset;
368             //System.out.println("  new token=" + new String(buffer, 0, bufferLen));
369             if (nextRead != nextWrite) {
370               capture();
371             } else {
372               input.consumed = false;
373             }
374 
375           } else {
376             // No more input tokens
377             //System.out.println("      set end");
378             finished = true;
379             break;
380           }
381         }
382       } else {
383         // Still in our lookahead
384         buffer = futureInputs[curNextRead].term.chars();
385         bufferLen = futureInputs[curNextRead].term.length();
386         inputEndOffset = futureInputs[curNextRead].endOffset;
387         //System.out.println("  old token=" + new String(buffer, 0, bufferLen));
388       }
389 
390       tokenCount++;
391 
392       // Run each char in this token through the FST:
393       int bufUpto = 0;
394       while(bufUpto < bufferLen) {
395         final int codePoint = Character.codePointAt(buffer, bufUpto, bufferLen);
396         if (fst.findTargetArc(ignoreCase ? Character.toLowerCase(codePoint) : codePoint, scratchArc, scratchArc, fstReader) == null) {
397           //System.out.println("    stop");
398           break byToken;
399         }
400 
401         // Accum the output
402         pendingOutput = fst.outputs.add(pendingOutput, scratchArc.output);
403         //System.out.println("    char=" + buffer[bufUpto] + " output=" + pendingOutput + " arc.output=" + scratchArc.output);
404         bufUpto += Character.charCount(codePoint);
405       }
406 
407       // OK, entire token matched; now see if this is a final
408       // state:
409       if (scratchArc.isFinal()) {
410         matchOutput = fst.outputs.add(pendingOutput, scratchArc.nextFinalOutput);
411         matchInputLength = tokenCount;
412         matchEndOffset = inputEndOffset;
413         //System.out.println("  found matchLength=" + matchInputLength + " output=" + matchOutput);
414       }
415 
416       // See if the FST wants to continue matching (ie, needs to
417       // see the next input token):
418       if (fst.findTargetArc(SynonymMap.WORD_SEPARATOR, scratchArc, scratchArc, fstReader) == null) {
419         // No further rules can match here; we're done
420         // searching for matching rules starting at the
421         // current input position.
422         break;
423       } else {
424         // More matching is possible -- accum the output (if
425         // any) of the WORD_SEP arc:
426         pendingOutput = fst.outputs.add(pendingOutput, scratchArc.output);
427         if (nextRead == nextWrite) {
428           capture();
429         }
430       }
431 
432       curNextRead = rollIncr(curNextRead);
433     }
434 
435     if (nextRead == nextWrite && !finished) {
436       //System.out.println("  skip write slot=" + nextWrite);
437       nextWrite = rollIncr(nextWrite);
438     }
439 
440     if (matchOutput != null) {
441       //System.out.println("  add matchLength=" + matchInputLength + " output=" + matchOutput);
442       inputSkipCount = matchInputLength;
443       addOutput(matchOutput, matchInputLength, matchEndOffset);
444     } else if (nextRead != nextWrite) {
445       // Even though we had no match here, we set to 1
446       // because we need to skip current input token before
447       // trying to match again:
448       inputSkipCount = 1;
449     } else {
450       assert finished;
451     }
452 
453     //System.out.println("  parse done inputSkipCount=" + inputSkipCount + " nextRead=" + nextRead + " nextWrite=" + nextWrite);
454   }
455 
456   // Interleaves all output tokens onto the futureOutputs:
457   private void addOutput(BytesRef bytes, int matchInputLength, int matchEndOffset) {
458     bytesReader.reset(bytes.bytes, bytes.offset, bytes.length);
459 
460     final int code = bytesReader.readVInt();
461     final boolean keepOrig = (code & 0x1) == 0;
462     final int count = code >>> 1;
463     //System.out.println("  addOutput count=" + count + " keepOrig=" + keepOrig);
464     for(int outputIDX=0;outputIDX<count;outputIDX++) {
465       synonyms.words.get(bytesReader.readVInt(),
466                          scratchBytes);
467       //System.out.println("    outIDX=" + outputIDX + " bytes=" + scratchBytes.length);
468       scratchChars.copyUTF8Bytes(scratchBytes);
469       int lastStart = 0;
470       final int chEnd = lastStart + scratchChars.length();
471       int outputUpto = nextRead;
472       for(int chIDX=lastStart;chIDX<=chEnd;chIDX++) {
473         if (chIDX == chEnd || scratchChars.charAt(chIDX) == SynonymMap.WORD_SEPARATOR) {
474           final int outputLen = chIDX - lastStart;
475           // Caller is not allowed to have empty string in
476           // the output:
477           assert outputLen > 0: "output contains empty string: " + scratchChars;
478           final int endOffset;
479           final int posLen;
480           if (chIDX == chEnd && lastStart == 0) {
481             // This rule had a single output token, so, we set
482             // this output's endOffset to the current
483             // endOffset (ie, endOffset of the last input
484             // token it matched):
485             endOffset = matchEndOffset;
486             posLen = keepOrig ? matchInputLength : 1;
487           } else {
488             // This rule has more than one output token; we
489             // can't pick any particular endOffset for this
490             // case, so, we inherit the endOffset for the
491             // input token which this output overlaps:
492             endOffset = -1;
493             posLen = 1;
494           }
495           futureOutputs[outputUpto].add(scratchChars.chars(), lastStart, outputLen, endOffset, posLen);
496           //System.out.println("      " + new String(scratchChars.chars, lastStart, outputLen) + " outputUpto=" + outputUpto);
497           lastStart = 1+chIDX;
498           //System.out.println("  slot=" + outputUpto + " keepOrig=" + keepOrig);
499           outputUpto = rollIncr(outputUpto);
500           assert futureOutputs[outputUpto].posIncr == 1: "outputUpto=" + outputUpto + " vs nextWrite=" + nextWrite;
501         }
502       }
503     }
504 
505     int upto = nextRead;
506     for(int idx=0;idx<matchInputLength;idx++) {
507       futureInputs[upto].keepOrig |= keepOrig;
508       futureInputs[upto].matched = true;
509       upto = rollIncr(upto);
510     }
511   }
512 
513   // ++ mod rollBufferSize
514   private int rollIncr(int count) {
515     count++;
516     if (count == rollBufferSize) {
517       return 0;
518     } else {
519       return count;
520     }
521   }

528   @Override
529   public boolean incrementToken() throws IOException {
530 
531     //System.out.println("\nS: incrToken inputSkipCount=" + inputSkipCount + " nextRead=" + nextRead + " nextWrite=" + nextWrite);
532 
533     while(true) {
534 
535       // First play back any buffered future inputs/outputs
536       // w/o running parsing again:
537       while (inputSkipCount != 0) {
538         
539         // At each position, we first output the original
540         // token
541 
542         // TODO: maybe just a PendingState class, holding
543         // both input & outputs?
544         final PendingInput input = futureInputs[nextRead];
545         final PendingOutputs outputs = futureOutputs[nextRead];
546         
547         //System.out.println("  cycle nextRead=" + nextRead + " nextWrite=" + nextWrite + " inputSkipCount="+ inputSkipCount + " input.keepOrig=" + input.keepOrig + " input.consumed=" + input.consumed + " input.state=" + input.state);
548 
549         if (!input.consumed && (input.keepOrig || !input.matched)) {
550           if (input.state != null) {
551             // Return a previously saved token (because we
552             // had to lookahead):
553             restoreState(input.state);
554           } else {
555             // Pass-through case: return token we just pulled
556             // but didn't capture:
557             assert inputSkipCount == 1: "inputSkipCount=" + inputSkipCount + " nextRead=" + nextRead;
558           }
559           input.reset();
560           if (outputs.count > 0) {
561             outputs.posIncr = 0;
562           } else {
563             nextRead = rollIncr(nextRead);
564             inputSkipCount--;
565           }
566           //System.out.println("  return token=" + termAtt.toString());
567           return true;
568         } else if (outputs.upto < outputs.count) {
569           // Still have pending outputs to replay at this
570           // position
571           input.reset();
572           final int posIncr = outputs.posIncr;
573           final CharsRef output = outputs.pullNext();
574           clearAttributes();
575           termAtt.copyBuffer(output.chars, output.offset, output.length);
576           typeAtt.setType(TYPE_SYNONYM);
577           int endOffset = outputs.getLastEndOffset();
578           if (endOffset == -1) {
579             endOffset = input.endOffset;
580           }
581           offsetAtt.setOffset(input.startOffset, endOffset);
582           posIncrAtt.setPositionIncrement(posIncr);
583           posLenAtt.setPositionLength(outputs.getLastPosLength());
584           if (outputs.count == 0) {
585             // Done with the buffered input and all outputs at
586             // this position
587             nextRead = rollIncr(nextRead);
588             inputSkipCount--;
589           }
590           //System.out.println("  return token=" + termAtt.toString());
591           return true;
592         } else {
593           // Done with the buffered input and all outputs at
594           // this position
595           input.reset();
596           nextRead = rollIncr(nextRead);
597           inputSkipCount--;
598         }
599       }
600 
601       if (finished && nextRead == nextWrite) {
602         // End case: if any output syns went beyond end of
603         // input stream, enumerate them now:
604         final PendingOutputs outputs = futureOutputs[nextRead];
605         if (outputs.upto < outputs.count) {
606           final int posIncr = outputs.posIncr;
607           final CharsRef output = outputs.pullNext();
608           futureInputs[nextRead].reset();
609           if (outputs.count == 0) {
610             nextWrite = nextRead = rollIncr(nextRead);
611           }
612           clearAttributes();
613           // Keep offset from last input token:
614           offsetAtt.setOffset(lastStartOffset, lastEndOffset);
615           termAtt.copyBuffer(output.chars, output.offset, output.length);
616           typeAtt.setType(TYPE_SYNONYM);
617           //System.out.println("  set posIncr=" + outputs.posIncr + " outputs=" + outputs);
618           posIncrAtt.setPositionIncrement(posIncr);
619           //System.out.println("  return token=" + termAtt.toString());
620           return true;
621         } else {
622           return false;
623         }
624       }
625 
626       // Find new synonym matches:
627       parse();
628     }
629   }

631   @Override
632   public void reset() throws IOException {
633     super.reset();
634     captureCount = 0;
635     finished = false;
636     inputSkipCount = 0;
637     nextRead = nextWrite = 0;
638 
639     // In normal usage these resets would not be needed,
640     // since they reset-as-they-are-consumed, but the app
641     // may not consume all input tokens (or we might hit an
642     // exception), in which case we have leftover state
643     // here:
644     for (PendingInput input : futureInputs) {
645       input.reset();
646     }
647     for (PendingOutputs output : futureOutputs) {
648       output.reset();
649     }
650   }
