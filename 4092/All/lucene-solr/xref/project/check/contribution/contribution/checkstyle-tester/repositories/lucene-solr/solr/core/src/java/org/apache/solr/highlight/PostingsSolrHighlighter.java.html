50  /** 
51   * Highlighter impl that uses {@link PostingsHighlighter}
52   * <p>
53   * Example configuration:
54   * <pre class="prettyprint">
55   *   &lt;requestHandler name="standard" class="solr.StandardRequestHandler"&gt;
56   *     &lt;lst name="defaults"&gt;
57   *       &lt;int name="hl.snippets"&gt;1&lt;/int&gt;
58   *       &lt;str name="hl.tag.pre"&gt;&amp;lt;em&amp;gt;&lt;/str&gt;
59   *       &lt;str name="hl.tag.post"&gt;&amp;lt;/em&amp;gt;&lt;/str&gt;
60   *       &lt;str name="hl.tag.ellipsis"&gt;... &lt;/str&gt;
61   *       &lt;bool name="hl.defaultSummary"&gt;true&lt;/bool&gt;
62   *       &lt;str name="hl.encoder"&gt;simple&lt;/str&gt;
63   *       &lt;float name="hl.score.k1"&gt;1.2&lt;/float&gt;
64   *       &lt;float name="hl.score.b"&gt;0.75&lt;/float&gt;
65   *       &lt;float name="hl.score.pivot"&gt;87&lt;/float&gt;
66   *       &lt;str name="hl.bs.language"&gt;&lt;/str&gt;
67   *       &lt;str name="hl.bs.country"&gt;&lt;/str&gt;
68   *       &lt;str name="hl.bs.variant"&gt;&lt;/str&gt;
69   *       &lt;str name="hl.bs.type"&gt;SENTENCE&lt;/str&gt;
70   *       &lt;int name="hl.maxAnalyzedChars"&gt;10000&lt;/int&gt;
71   *       &lt;str name="hl.multiValuedSeparatorChar"&gt; &lt;/str&gt;
72   *       &lt;bool name="hl.highlightMultiTerm"&gt;false&lt;/bool&gt;
73   *     &lt;/lst&gt;
74   *   &lt;/requestHandler&gt;
75   * </pre>
76   * ...
77   * <pre class="prettyprint">
78   *   &lt;searchComponent class="solr.HighlightComponent" name="highlight"&gt;
79   *     &lt;highlighting class="org.apache.solr.highlight.PostingsSolrHighlighter"/&gt;
80   *   &lt;/searchComponent&gt;
81   * </pre>
82   * <p>
83   * Notes:
84   *  <ul>
85   *    <li>fields to highlight must be configured with storeOffsetsWithPositions="true"
86   *    <li>hl.q (string) can specify the query
87   *    <li>hl.fl (string) specifies the field list.
88   *    <li>hl.snippets (int) specifies how many underlying passages form the resulting snippet.
89   *    <li>hl.tag.pre (string) specifies text which appears before a highlighted term.
90   *    <li>hl.tag.post (string) specifies text which appears after a highlighted term.
91   *    <li>hl.tag.ellipsis (string) specifies text which joins non-adjacent passages.
92   *    <li>hl.defaultSummary (bool) specifies if a field should have a default summary.
93   *    <li>hl.encoder (string) can be 'html' (html escapes content) or 'simple' (no escaping).
94   *    <li>hl.score.k1 (float) specifies bm25 scoring parameter 'k1'
95   *    <li>hl.score.b (float) specifies bm25 scoring parameter 'b'
96   *    <li>hl.score.pivot (float) specifies bm25 scoring parameter 'avgdl'
97   *    <li>hl.bs.type (string) specifies how to divide text into passages: [SENTENCE, LINE, WORD, CHAR, WHOLE]
98   *    <li>hl.bs.language (string) specifies language code for BreakIterator. default is empty string (root locale)
99   *    <li>hl.bs.country (string) specifies country code for BreakIterator. default is empty string (root locale)
100  *    <li>hl.bs.variant (string) specifies country code for BreakIterator. default is empty string (root locale)
101  *    <li>hl.maxAnalyzedChars specifies how many characters at most will be processed in a document.
102  *    <li>hl.multiValuedSeparatorChar specifies the logical separator between values for multi-valued fields.
103  *    <li>hl.highlightMultiTerm enables highlighting for range/wildcard/fuzzy/prefix queries.
104  *        NOTE: currently hl.maxAnalyzedChars cannot yet be specified per-field
105  *  </ul>
106  *  
107  * @lucene.experimental 
108  */
109 public class PostingsSolrHighlighter extends SolrHighlighter implements PluginInfoInitialized {
110   
111   @Override
112   public void init(PluginInfo info) {}
113 
114   @Override
115   public NamedList<Object> doHighlighting(DocList docs, Query query, SolrQueryRequest req, String[] defaultFields) throws IOException {
116     final SolrParams params = req.getParams(); 
117     
118     // if highlighting isnt enabled, then why call doHighlighting?
119     if (!isHighlightingEnabled(params))
120       return null;
121 
122     SolrIndexSearcher searcher = req.getSearcher();
123     int[] docIDs = toDocIDs(docs);
124 
125     // fetch the unique keys
126     String[] keys = getUniqueKeys(searcher, docIDs);
127 
128     // query-time parameters
129     String[] fieldNames = getHighlightFields(query, req, defaultFields);
130 
131     int maxPassages[] = new int[fieldNames.length];
132     for (int i = 0; i < fieldNames.length; i++) {
133       maxPassages[i] = params.getFieldInt(fieldNames[i], HighlightParams.SNIPPETS, 1);
134     }
135 
136     PostingsHighlighter highlighter = getHighlighter(req);
137     Map<String,String[]> snippets = highlighter.highlightFields(fieldNames, query, searcher, docIDs, maxPassages);
138     return encodeSnippets(keys, fieldNames, snippets);
139   }
140 
141   /** Creates an instance of the Lucene PostingsHighlighter. Provided for subclass extension so that
142    * a subclass can return a subclass of {@link PostingsSolrHighlighter.SolrExtendedPostingsHighlighter}. */
143   protected PostingsHighlighter getHighlighter(SolrQueryRequest req) {
144     return new SolrExtendedPostingsHighlighter(req);
145   }
146 
147   /** 
148    * Encodes the resulting snippets into a namedlist
149    * @param keys the document unique keys
150    * @param fieldNames field names to highlight in the order
151    * @param snippets map from field name to snippet array for the docs
152    * @return encoded namedlist of summaries
153    */
154   protected NamedList<Object> encodeSnippets(String[] keys, String[] fieldNames, Map<String,String[]> snippets) {
155     NamedList<Object> list = new SimpleOrderedMap<>();
156     for (int i = 0; i < keys.length; i++) {
157       NamedList<Object> summary = new SimpleOrderedMap<>();
158       for (String field : fieldNames) {
159         String snippet = snippets.get(field)[i];
160         // box in an array to match the format of existing highlighters, 
161         // even though it's always one element.
162         if (snippet == null) {
163           summary.add(field, new String[0]);
164         } else {
165           summary.add(field, new String[] { snippet });
166         }
167       }
168       list.add(keys[i], summary);
169     }
170     return list;
171   }
172   
173   /** Converts solr's DocList to the int[] docIDs */
174   protected int[] toDocIDs(DocList docs) {
175     int[] docIDs = new int[docs.size()];
176     DocIterator iterator = docs.iterator();
177     for (int i = 0; i < docIDs.length; i++) {
178       if (!iterator.hasNext()) {
179         throw new AssertionError();
180       }
181       docIDs[i] = iterator.nextDoc();
182     }
183     if (iterator.hasNext()) {
184       throw new AssertionError();
185     }
186     return docIDs;
187   }
188   
189   /** Retrieves the unique keys for the topdocs to key the results */
190   protected String[] getUniqueKeys(SolrIndexSearcher searcher, int[] docIDs) throws IOException {
191     IndexSchema schema = searcher.getSchema();
192     SchemaField keyField = schema.getUniqueKeyField();
193     if (keyField != null) {
194       Set<String> selector = Collections.singleton(keyField.getName());
195       String uniqueKeys[] = new String[docIDs.length];
196       for (int i = 0; i < docIDs.length; i++) {
197         int docid = docIDs[i];
198         Document doc = searcher.doc(docid, selector);
199         String id = schema.printableUniqueKey(doc);
200         uniqueKeys[i] = id;
201       }
202       return uniqueKeys;
203     } else {
204       return new String[docIDs.length];
205     }
206   }

208   /** From {@link #getHighlighter(org.apache.solr.request.SolrQueryRequest)}. */
209   public class SolrExtendedPostingsHighlighter extends PostingsHighlighter {
210     protected final SolrParams params;
211     protected final IndexSchema schema;
212 
213     public SolrExtendedPostingsHighlighter(SolrQueryRequest req) {
214       super(req.getParams().getInt(HighlightParams.MAX_CHARS, PostingsHighlighter.DEFAULT_MAX_LENGTH));
215       this.params = req.getParams();
216       this.schema = req.getSchema();
217     }
218 
219     @Override
220     protected Passage[] getEmptyHighlight(String fieldName, BreakIterator bi, int maxPassages) {
221       boolean defaultSummary = params.getFieldBool(fieldName, HighlightParams.DEFAULT_SUMMARY, true);
222       if (defaultSummary) {
223         return super.getEmptyHighlight(fieldName, bi, maxPassages);
224       } else {
225         //TODO reuse logic of DefaultSolrHighlighter.alternateField
226         return new Passage[0];
227       }
228     }
229 
230     @Override
231     protected PassageFormatter getFormatter(String fieldName) {
232       String preTag = params.getFieldParam(fieldName, HighlightParams.TAG_PRE, "<em>");
233       String postTag = params.getFieldParam(fieldName, HighlightParams.TAG_POST, "</em>");
234       String ellipsis = params.getFieldParam(fieldName, HighlightParams.TAG_ELLIPSIS, "... ");
235       String encoder = params.getFieldParam(fieldName, HighlightParams.ENCODER, "simple");
236       return new DefaultPassageFormatter(preTag, postTag, ellipsis, "html".equals(encoder));
237     }
238 
239     @Override
240     protected PassageScorer getScorer(String fieldName) {
241       float k1 = params.getFieldFloat(fieldName, HighlightParams.SCORE_K1, 1.2f);
242       float b = params.getFieldFloat(fieldName, HighlightParams.SCORE_B, 0.75f);
243       float pivot = params.getFieldFloat(fieldName, HighlightParams.SCORE_PIVOT, 87f);
244       return new PassageScorer(k1, b, pivot);
245     }
246 
247     @Override
248     protected BreakIterator getBreakIterator(String field) {
249       String language = params.getFieldParam(field, HighlightParams.BS_LANGUAGE);
250       String country = params.getFieldParam(field, HighlightParams.BS_COUNTRY);
251       String variant = params.getFieldParam(field, HighlightParams.BS_VARIANT);
252       Locale locale = parseLocale(language, country, variant);
253       String type = params.getFieldParam(field, HighlightParams.BS_TYPE);
254       return parseBreakIterator(type, locale);
255     }
256 
257     @Override
258     protected char getMultiValuedSeparator(String field) {
259       String sep = params.getFieldParam(field, HighlightParams.MULTI_VALUED_SEPARATOR, " ");
260       if (sep.length() != 1) {
261         throw new IllegalArgumentException(HighlightParams.MULTI_VALUED_SEPARATOR + " must be exactly one character.");
262       }
263       return sep.charAt(0);
264     }
265 
266     @Override
267     protected Analyzer getIndexAnalyzer(String field) {
268       if (params.getFieldBool(field, HighlightParams.HIGHLIGHT_MULTI_TERM, false)) {
269         return schema.getIndexAnalyzer();
270       } else {
271         return null;
272       }
273     }
274   }
275 
276   /** parse a break iterator type for the specified locale */
277   protected BreakIterator parseBreakIterator(String type, Locale locale) {
278     if (type == null || "SENTENCE".equals(type)) {
279       return BreakIterator.getSentenceInstance(locale);
280     } else if ("LINE".equals(type)) {
281       return BreakIterator.getLineInstance(locale);
282     } else if ("WORD".equals(type)) {
283       return BreakIterator.getWordInstance(locale);
284     } else if ("CHARACTER".equals(type)) {
285       return BreakIterator.getCharacterInstance(locale);
286     } else if ("WHOLE".equals(type)) {
287       return new WholeBreakIterator();
288     } else {
289       throw new IllegalArgumentException("Unknown " + HighlightParams.BS_TYPE + ": " + type);
290     }
291   }
292   
293   /** parse a locale from a language+country+variant spec */
294   protected Locale parseLocale(String language, String country, String variant) {
295     if (language == null && country == null && variant == null) {
296       return Locale.ROOT;
297     } else if (language != null && country == null && variant != null) {
298       throw new IllegalArgumentException("To specify variant, country is required");
299     } else if (language != null && country != null && variant != null) {
300       return new Locale(language, country, variant);
301     } else if (language != null && country != null) {
302       return new Locale(language, country);
303     } else { 
304       return new Locale(language);
305     }
306   }
