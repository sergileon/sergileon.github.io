227   private synchronized void initializeAuthorizationPlugin(Map<String, Object> authorizationConf) {
228     authorizationConf = Utils.getDeepCopy(authorizationConf, 4);
229     //Initialize the Authorization module
230     SecurityPluginHolder<AuthorizationPlugin> old = authorizationPlugin;
231     SecurityPluginHolder<AuthorizationPlugin> authorizationPlugin = null;
232     if (authorizationConf != null) {
233       String klas = (String) authorizationConf.get("class");
234       if (klas == null) {
235         throw new SolrException(ErrorCode.SERVER_ERROR, "class is required for authorization plugin");
236       }
237       if (old != null && old.getZnodeVersion() == readVersion(authorizationConf)) {
238         return;
239       }
240       log.info("Initializing authorization plugin: " + klas);
241       authorizationPlugin = new SecurityPluginHolder<>(readVersion(authorizationConf),
242           getResourceLoader().newInstance(klas, AuthorizationPlugin.class));
243 
244       // Read and pass the authorization context to the plugin
245       authorizationPlugin.plugin.init(authorizationConf);
246     } else {
247       log.info("Security conf doesn't exist. Skipping setup for authorization module.");
248     }
249     this.authorizationPlugin = authorizationPlugin;
250     if (old != null) {
251       try {
252         old.plugin.close();
253       } catch (Exception e) {
254       }
255     }
256   }

258   private synchronized void initializeAuthenticationPlugin(Map<String, Object> authenticationConfig) {
259     authenticationConfig = Utils.getDeepCopy(authenticationConfig, 4);
260     String pluginClassName = null;
261     if (authenticationConfig != null) {
262       if (authenticationConfig.containsKey("class")) {
263         pluginClassName = String.valueOf(authenticationConfig.get("class"));
264       } else {
265         throw new SolrException(ErrorCode.SERVER_ERROR, "No 'class' specified for authentication in ZK.");
266       }
267     }
268 
269     if (pluginClassName != null) {
270       log.info("Authentication plugin class obtained from ZK: "+pluginClassName);
271     } else if (System.getProperty(AUTHENTICATION_PLUGIN_PROP) != null) {
272       pluginClassName = System.getProperty(AUTHENTICATION_PLUGIN_PROP);
273       log.info("Authentication plugin class obtained from system property '" +
274           AUTHENTICATION_PLUGIN_PROP + "': " + pluginClassName);
275     } else {
276       log.info("No authentication plugin used.");
277     }
278     SecurityPluginHolder<AuthenticationPlugin> old = authenticationPlugin;
279     SecurityPluginHolder<AuthenticationPlugin> authenticationPlugin = null;
280 
281     // Initialize the filter
282     if (pluginClassName != null) {
283       authenticationPlugin = new SecurityPluginHolder<>(readVersion(authenticationConfig),
284           getResourceLoader().newInstance(pluginClassName, AuthenticationPlugin.class));
285     }
286     if (authenticationPlugin != null) {
287       authenticationPlugin.plugin.init(authenticationConfig);
288       addHttpConfigurer(authenticationPlugin.plugin);
289     }
290     this.authenticationPlugin = authenticationPlugin;
291     try {
292       if (old != null) old.plugin.close();
293     } catch (Exception e) {/*do nothing*/ }
294 
295   }

297   private void addHttpConfigurer(Object authcPlugin) {
298     if (authcPlugin instanceof HttpClientInterceptorPlugin) {
299       // Setup HttpClient to use the plugin's configurer for internode communication
300       HttpClientConfigurer configurer = ((HttpClientInterceptorPlugin) authcPlugin).getClientConfigurer();
301       HttpClientUtil.setConfigurer(configurer);
302 
303       // The default http client of the core container's shardHandlerFactory has already been created and
304       // configured using the default httpclient configurer. We need to reconfigure it using the plugin's
305       // http client configurer to set it up for internode communication.
306       log.info("Reconfiguring the shard handler factory and update shard handler.");
307       if (getShardHandlerFactory() instanceof HttpShardHandlerFactory) {
308         ((HttpShardHandlerFactory) getShardHandlerFactory()).reconfigureHttpClient(configurer);
309       }
310       getUpdateShardHandler().reconfigureHttpClient(configurer);
311     } else {
312       if (pkiAuthenticationPlugin != null) {
313         //this happened due to an authc plugin reload. no need to register the pkiAuthc plugin again
314         if(pkiAuthenticationPlugin.isInterceptorRegistered()) return;
315         log.info("PKIAuthenticationPlugin is managing internode requests");
316         addHttpConfigurer(pkiAuthenticationPlugin);
317         pkiAuthenticationPlugin.setInterceptorRegistered();
318       }
319     }
320   }

322   private static int readVersion(Map<String, Object> conf) {
323     if (conf == null) return -1;
324     Map meta = (Map) conf.get("");
325     if (meta == null) return -1;
326     Number v = (Number) meta.get("v");
327     return v == null ? -1 : v.intValue();
328   }

379   /**
380    * Load the cores defined for this CoreContainer
381    */
382   public void load()  {
383     log.info("Loading cores into CoreContainer [instanceDir={}]", loader.getInstancePath());
384 
385     // add the sharedLib to the shared resource loader before initializing cfg based plugins
386     String libDir = cfg.getSharedLibDirectory();
387     if (libDir != null) {
388       Path libPath = loader.getInstancePath().resolve(libDir);
389       try {
390         loader.addToClassLoader(SolrResourceLoader.getURLs(libPath));
391         loader.reloadLuceneSPI();
392       } catch (IOException e) {
393         log.warn("Couldn't add files from {} to classpath: {}", libPath, e.getMessage());
394       }
395     }
396 
397 
398     shardHandlerFactory = ShardHandlerFactory.newInstance(cfg.getShardHandlerFactoryPluginInfo(), loader);
399 
400     updateShardHandler = new UpdateShardHandler(cfg.getUpdateShardHandlerConfig());
401 
402     solrCores.allocateLazyCores(cfg.getTransientCacheSize(), loader);
403 
404     logging = LogWatcher.newRegisteredLogWatcher(cfg.getLogWatcherConfig(), loader);
405 
406     hostName = cfg.getNodeName();
407 
408     zkSys.initZooKeeper(this, solrHome, cfg.getCloudConfig());
409     if(isZooKeeperAware())  pkiAuthenticationPlugin = new PKIAuthenticationPlugin(this, zkSys.getZkController().getNodeName());
410 
411     ZkStateReader.ConfigData securityConfig = isZooKeeperAware() ? getZkController().getZkStateReader().getSecurityProps(false) : new ZkStateReader.ConfigData(EMPTY_MAP, -1);
412     initializeAuthorizationPlugin((Map<String, Object>) securityConfig.data.get("authorization"));
413     initializeAuthenticationPlugin((Map<String, Object>) securityConfig.data.get("authentication"));
414 
415     containerHandlers.put(ZK_PATH, new ZookeeperInfoHandler(this));
416     securityConfHandler = new SecurityConfHandler(this);
417     collectionsHandler = createHandler(cfg.getCollectionsHandlerClass(), CollectionsHandler.class);
418     containerHandlers.put(COLLECTIONS_HANDLER_PATH, collectionsHandler);
419     infoHandler        = createHandler(cfg.getInfoHandlerClass(), InfoHandler.class);
420     containerHandlers.put(INFO_HANDLER_PATH, infoHandler);
421     coreAdminHandler   = createHandler(cfg.getCoreAdminHandlerClass(), CoreAdminHandler.class);
422     containerHandlers.put(CORES_HANDLER_PATH, coreAdminHandler);
423     configSetsHandler = createHandler(cfg.getConfigSetsHandlerClass(), ConfigSetsHandler.class);
424     containerHandlers.put(CONFIGSETS_HANDLER_PATH, configSetsHandler);
425     containerHandlers.put(AUTHZ_PATH, securityConfHandler);
426     containerHandlers.put(AUTHC_PATH, securityConfHandler);
427     if(pkiAuthenticationPlugin != null)
428       containerHandlers.put(PKIAuthenticationPlugin.PATH, pkiAuthenticationPlugin.getRequestHandler());
429 
430     coreConfigService = ConfigSetService.createConfigSetService(cfg, loader, zkSys.zkController);
431 
432     containerProperties.putAll(cfg.getSolrProperties());
433 
434     // setup executor to load cores in parallel
435     // do not limit the size of the executor in zk mode since cores may try and wait for each other.
436     final ExecutorService coreLoadExecutor = ExecutorUtil.newMDCAwareFixedThreadPool(
437         ( zkSys.getZkController() == null ? cfg.getCoreLoadThreadCount() : Integer.MAX_VALUE ),
438         new DefaultSolrThreadFactory("coreLoadExecutor") );
439     final List<Future<SolrCore>> futures = new ArrayList<Future<SolrCore>>();
440     try {
441 
442       List<CoreDescriptor> cds = coresLocator.discover(this);
443       checkForDuplicateCoreNames(cds);
444 
445 
446       for (final CoreDescriptor cd : cds) {
447         if (cd.isTransient() || !cd.isLoadOnStartup()) {
448           solrCores.putDynamicDescriptor(cd.getName(), cd);
449         } else if (asyncSolrCoreLoad) {
450           solrCores.markCoreAsLoading(cd);
451         }
452         if (cd.isLoadOnStartup()) {
453           futures.add(coreLoadExecutor.submit(new Callable<SolrCore>() {
454             @Override
455             public SolrCore call() throws Exception {
456               SolrCore core;
457               try {
458                 if (zkSys.getZkController() != null) {
459                   zkSys.getZkController().throwErrorIfReplicaReplaced(cd);
460                 }
461                 
462                 core = create(cd, false);
463               } finally {
464                 if (asyncSolrCoreLoad) {
465                   solrCores.markCoreAsNotLoading(cd);
466                 }
467               }
468               try {
469                 zkSys.registerInZk(core, true);
470               } catch (RuntimeException e) {
471                 SolrException.log(log, "Error registering SolrCore", e);
472               }
473               return core;
474             }
475           }));
476         }
477       }
478 
479 
480       // Start the background thread
481       backgroundCloser = new CloserThread(this, solrCores, cfg);
482       backgroundCloser.start();
483 
484     } finally {
485       if (asyncSolrCoreLoad && futures != null) {
486         Thread shutdownThread = new Thread() {
487           public void run() {
488             try {
489               for (Future<SolrCore> future : futures) {
490                 try {
491                   future.get();
492                 } catch (InterruptedException e) {
493                   Thread.currentThread().interrupt();
494                 } catch (ExecutionException e) {
495                   log.error("Error waiting for SolrCore to be created", e);
496                 }
497               }
498             } finally {
499               ExecutorUtil.shutdownAndAwaitTermination(coreLoadExecutor);
500             }
501           }
502         };
503         coreContainerWorkExecutor.submit(shutdownThread);
504       } else {
505         ExecutorUtil.shutdownAndAwaitTermination(coreLoadExecutor);
506       }
507     }
508     
509     if (isZooKeeperAware()) {
510       zkSys.getZkController().checkOverseerDesignate();
511     }
512   }

521   private static void checkForDuplicateCoreNames(List<CoreDescriptor> cds) {
522     Map<String, String> addedCores = Maps.newHashMap();
523     for (CoreDescriptor cd : cds) {
524       final String name = cd.getName();
525       if (addedCores.containsKey(name))
526         throw new SolrException(ErrorCode.SERVER_ERROR,
527             String.format(Locale.ROOT, "Found multiple cores with the name [%s], with instancedirs [%s] and [%s]",
528                 name, addedCores.get(name), cd.getInstanceDir()));
529       addedCores.put(name, cd.getInstanceDir());
530     }

539   /**
540    * Stops all cores.
541    */
542   public void shutdown() {
543     log.info("Shutting down CoreContainer instance="
544         + System.identityHashCode(this));
545     
546     isShutDown = true;
547     
548     ExecutorUtil.shutdownAndAwaitTermination(coreContainerWorkExecutor);
549     
550     if (isZooKeeperAware()) {
551       cancelCoreRecoveries();
552       zkSys.publishCoresAsDown(solrCores.getCores());
553     }
554 
555     try {
556       if (coreAdminHandler != null) coreAdminHandler.shutdown();
557     } catch (Exception e) {
558       log.warn("Error shutting down CoreAdminHandler. Continuing to close CoreContainer.", e);
559     }
560 
561     try {
562       // First wake up the closer thread, it'll terminate almost immediately since it checks isShutDown.
563       synchronized (solrCores.getModifyLock()) {
564         solrCores.getModifyLock().notifyAll(); // wake up anyone waiting
565       }
566       if (backgroundCloser != null) { // Doesn't seem right, but tests get in here without initializing the core.
567         try {
568           backgroundCloser.join();
569         } catch (InterruptedException e) {
570           Thread.currentThread().interrupt();
571           if (log.isDebugEnabled()) {
572             log.debug("backgroundCloser thread was interrupted before finishing");
573           }
574         }
575       }
576       // Now clear all the cores that are being operated upon.
577       solrCores.close();
578 
579       // It's still possible that one of the pending dynamic load operation is waiting, so wake it up if so.
580       // Since all the pending operations queues have been drained, there should be nothing to do.
581       synchronized (solrCores.getModifyLock()) {
582         solrCores.getModifyLock().notifyAll(); // wake up the thread
583       }
584 
585     } finally {
586       try {
587         if (shardHandlerFactory != null) {
588           shardHandlerFactory.close();
589         }
590       } finally {
591         try {
592           if (updateShardHandler != null) {
593             updateShardHandler.close();
594           }
595         } finally {
596           // we want to close zk stuff last
597           zkSys.close();
598         }
599       }
600     }
601 
602     // It should be safe to close the authorization plugin at this point.
603     try {
604       if(authorizationPlugin != null) {
605         authorizationPlugin.plugin.close();
606       }
607     } catch (IOException e) {
608       log.warn("Exception while closing authorization plugin.", e);
609     }
610 
611     // It should be safe to close the authentication plugin at this point.
612     try {
613       if(authenticationPlugin != null) {
614         authenticationPlugin.plugin.close();
615         authenticationPlugin = null;
616       }
617     } catch (Exception e) {
618       log.warn("Exception while closing authentication plugin.", e);
619     }
620 
621     org.apache.lucene.util.IOUtils.closeWhileHandlingException(loader); // best effort
622   }

624   public void cancelCoreRecoveries() {
625 
626     List<SolrCore> cores = solrCores.getCores();
627 
628     // we must cancel without holding the cores sync
629     // make sure we wait for any recoveries to stop
630     for (SolrCore core : cores) {
631       try {
632         core.getSolrCoreState().cancelRecovery();
633       } catch (Exception e) {
634         SolrException.log(log, "Error canceling recovery for core", e);
635       }
636     }
637   }

641     try {
642       if(!isShutDown){
643         log.error("CoreContainer was not close prior to finalize(), indicates a bug -- POSSIBLE RESOURCE LEAK!!!  instance=" + System.identityHashCode(this));
644       }
645     } finally {
646       super.finalize();
647     }

654   protected SolrCore registerCore(String name, SolrCore core, boolean registerInZk) {
655     if( core == null ) {
656       throw new RuntimeException( "Can not register a null core." );
657     }
658     if( name == null ||
659         name.indexOf( '/'  ) >= 0 ||
660         name.indexOf( '\\' ) >= 0 ){
661       throw new RuntimeException( "Invalid core name: "+name );
662     }
663     // We can register a core when creating them via the admin UI, so we need to insure that the dynamic descriptors
664     // are up to date
665     CoreDescriptor cd = core.getCoreDescriptor();
666     if ((cd.isTransient() || ! cd.isLoadOnStartup())
667         && solrCores.getDynamicDescriptor(name) == null) {
668       // Store it away for later use. includes non-transient but not
669       // loaded at startup cores.
670       solrCores.putDynamicDescriptor(name, cd);
671     }
672 
673     SolrCore old;
674 
675     if (isShutDown) {
676       core.close();
677       throw new IllegalStateException("This CoreContainer has been close");
678     }
679     if (cd.isTransient()) {
680       old = solrCores.putTransientCore(cfg, name, core, loader);
681     } else {
682       old = solrCores.putCore(name, core);
683     }
684       /*
685       * set both the name of the descriptor and the name of the
686       * core, since the descriptors name is used for persisting.
687       */
688 
689     core.setName(name);
690 
691     coreInitFailures.remove(name);
692 
693     if( old == null || old == core) {
694       log.info( "registering core: "+name );
695       if (registerInZk) {
696         zkSys.registerInZk(core, false);
697       }
698       return null;
699     }
700     else {
701       log.info( "replacing core: "+name );
702       old.close();
703       if (registerInZk) {
704         zkSys.registerInZk(core, false);
705       }
706       return old;
707     }
708   }

719   /**
720    * Creates a new core based on a CoreDescriptor.
721    *
722    * @param dcore        a core descriptor
723    * @param publishState publish core state to the cluster if true
724    *
725    * @return the newly created core
726    */
727   public SolrCore create(CoreDescriptor dcore, boolean publishState) {
728 
729     if (isShutDown) {
730       throw new SolrException(ErrorCode.SERVICE_UNAVAILABLE, "Solr has been shutdown.");
731     }
732 
733     SolrCore core = null;
734     try {
735       MDCLoggingContext.setCore(core);
736       if (zkSys.getZkController() != null) {
737         zkSys.getZkController().preRegister(dcore);
738       }
739 
740       ConfigSet coreConfig = coreConfigService.getConfig(dcore);
741       log.info("Creating SolrCore '{}' using configuration from {}", dcore.getName(), coreConfig.getName());
742       core = new SolrCore(dcore, coreConfig);
743       solrCores.addCreated(core);
744 
745       // always kick off recovery if we are in non-Cloud mode
746       if (!isZooKeeperAware() && core.getUpdateHandler().getUpdateLog() != null) {
747         core.getUpdateHandler().getUpdateLog().recoverFromLog();
748       }
749 
750       registerCore(dcore.getName(), core, publishState);
751 
752       return core;
753     } catch (Exception e) {
754       coreInitFailures.put(dcore.getName(), new CoreLoadFailure(dcore, e));
755       log.error("Error creating core [{}]: {}", dcore.getName(), e.getMessage(), e);
756       final SolrException solrException = new SolrException(ErrorCode.SERVER_ERROR, "Unable to create core [" + dcore.getName() + "]", e);
757       if(core != null && !core.isClosed())
758         IOUtils.closeQuietly(core);
759       throw solrException;
760     } catch (Throwable t) {
761       SolrException e = new SolrException(ErrorCode.SERVER_ERROR, "JVM Error creating core [" + dcore.getName() + "]: " + t.getMessage(), t);
762       log.error("Error creating core [{}]: {}", dcore.getName(), t.getMessage(), t);
763       coreInitFailures.put(dcore.getName(), new CoreLoadFailure(dcore, e));
764       if(core != null && !core.isClosed())
765         IOUtils.closeQuietly(core);
766       throw t;
767     } finally {
768       MDCLoggingContext.clear();
769     }
770 
771   }

827   /**
828    * Recreates a SolrCore.
829    * While the new core is loading, requests will continue to be dispatched to
830    * and processed by the old core
831    * 
832    * @param name the name of the SolrCore to reload
833    */
834   public void reload(String name) {
835 
836     SolrCore core = solrCores.getCoreFromAnyList(name, false);
837     if (core == null)
838       throw new SolrException( SolrException.ErrorCode.BAD_REQUEST, "No such core: " + name );
839 
840     CoreDescriptor cd = core.getCoreDescriptor();
841     try {
842       solrCores.waitAddPendingCoreOps(name);
843       ConfigSet coreConfig = coreConfigService.getConfig(cd);
844       log.info("Reloading SolrCore '{}' using configuration from {}", cd.getName(), coreConfig.getName());
845       SolrCore newCore = core.reload(coreConfig);
846       registerCore(name, newCore, false);
847     }
848     catch (Exception e) {
849       coreInitFailures.put(cd.getName(), new CoreLoadFailure(cd, e));
850       throw new SolrException(ErrorCode.SERVER_ERROR, "Unable to reload core [" + cd.getName() + "]", e);
851     }
852     finally {
853       solrCores.removeFromPendingOps(name);
854     }
855 
856   }

880   /**
881    * Unload a core from this container, optionally removing the core's data and configuration
882    *
883    * @param name the name of the core to unload
884    * @param deleteIndexDir if true, delete the core's index on close
885    * @param deleteDataDir if true, delete the core's data directory on close
886    * @param deleteInstanceDir if true, delete the core's instance directory on close
887    */
888   public void unload(String name, boolean deleteIndexDir, boolean deleteDataDir, boolean deleteInstanceDir) {
889 
890     if (name != null) {
891       // check for core-init errors first
892       CoreLoadFailure loadFailure = coreInitFailures.remove(name);
893       if (loadFailure != null) {
894         // getting the index directory requires opening a DirectoryFactory with a SolrConfig, etc,
895         // which we may not be able to do because of the init error.  So we just go with what we
896         // can glean from the CoreDescriptor - datadir and instancedir
897         SolrCore.deleteUnloadedCore(loadFailure.cd, deleteDataDir, deleteInstanceDir);
898         return;
899       }
900     }
901 
902     CoreDescriptor cd = solrCores.getCoreDescriptor(name);
903     if (cd == null)
904       throw new SolrException(ErrorCode.BAD_REQUEST, "Cannot unload non-existent core [" + name + "]");
905 
906     boolean close = solrCores.isLoadedNotPendingClose(name);
907     SolrCore core = solrCores.remove(name);
908     coresLocator.delete(this, cd);
909 
910     if (core == null) {
911       // transient core
912       SolrCore.deleteUnloadedCore(cd, deleteDataDir, deleteInstanceDir);
913       return;
914     }
915 
916     if (zkSys.getZkController() != null) {
917       // cancel recovery in cloud mode
918       core.getSolrCoreState().cancelRecovery();
919     }
920     
921     core.unloadOnClose(deleteIndexDir, deleteDataDir, deleteInstanceDir);
922     if (close)
923       core.close();
924 
925     if (zkSys.getZkController() != null) {
926       try {
927         zkSys.getZkController().unregister(name, cd);
928       } catch (InterruptedException e) {
929         Thread.currentThread().interrupt();
930         throw new SolrException(ErrorCode.SERVER_ERROR, "Interrupted while unregistering core [" + name + "] from cloud state");
931       } catch (KeeperException e) {
932         throw new SolrException(ErrorCode.SERVER_ERROR, "Error unregistering core [" + name + "] from cloud state", e);
933       }
934     }
935   }

968   /**
969    * Gets a core by name and increase its refcount.
970    *
971    * @see SolrCore#close()
972    * @param name the core name
973    * @return the core if found, null if a SolrCore by this name does not exist
974    * @exception SolrException if a SolrCore with this name failed to be initialized
975    */
976   public SolrCore getCore(String name) {
977 
978     // Do this in two phases since we don't want to lock access to the cores over a load.
979     SolrCore core = solrCores.getCoreFromAnyList(name, true);
980 
981     if (core != null) {
982       return core;
983     }
984 
985     // OK, it's not presently in any list, is it in the list of dynamic cores but not loaded yet? If so, load it.
986     CoreDescriptor desc = solrCores.getDynamicDescriptor(name);
987     if (desc == null) { //Nope, no transient core with this name
988 
989       // if there was an error initalizing this core, throw a 500
990       // error with the details for clients attempting to access it.
991       CoreLoadFailure loadFailure = getCoreInitFailures().get(name);
992       if (null != loadFailure) {
993         throw new SolrException(ErrorCode.SERVER_ERROR, "SolrCore '" + name +
994                                 "' is not available due to init failure: " +
995                                 loadFailure.exception.getMessage(), loadFailure.exception);
996       }
997       // otherwise the user is simply asking for something that doesn't exist.
998       return null;
999     }
1000 
1001     // This will put an entry in pending core ops if the core isn't loaded
1002     core = solrCores.waitAddPendingCoreOps(name);
1003 
1004     if (isShutDown) return null; // We're quitting, so stop. This needs to be after the wait above since we may come off
1005                                  // the wait as a consequence of shutting down.
1006     try {
1007       if (core == null) {
1008         if (zkSys.getZkController() != null) {
1009           zkSys.getZkController().throwErrorIfReplicaReplaced(desc);
1010         }
1011         core = create(desc); // This should throw an error if it fails.
1012       }
1013       core.open();
1014     }
1015     finally {
1016       solrCores.removeFromPendingOps(name);
1017     }
1018 
1019     return core;
1020   }

1147 class CloserThread extends Thread {
1148   CoreContainer container;
1149   SolrCores solrCores;
1150   NodeConfig cfg;
1151 
1152 
1153   CloserThread(CoreContainer container, SolrCores solrCores, NodeConfig cfg) {
1154     this.container = container;
1155     this.solrCores = solrCores;
1156     this.cfg = cfg;
1157   }
1158 
1159   // It's important that this be the _only_ thread removing things from pendingDynamicCloses!
1160   // This is single-threaded, but I tried a multi-threaded approach and didn't see any performance gains, so
1161   // there's no good justification for the complexity. I suspect that the locking on things like DefaultSolrCoreState
1162   // essentially create a single-threaded process anyway.
1163   @Override
1164   public void run() {
1165     while (! container.isShutDown()) {
1166       synchronized (solrCores.getModifyLock()) { // need this so we can wait and be awoken.
1167         try {
1168           solrCores.getModifyLock().wait();
1169         } catch (InterruptedException e) {
1170           // Well, if we've been told to stop, we will. Otherwise, continue on and check to see if there are
1171           // any cores to close.
1172         }
1173       }
1174       for (SolrCore removeMe = solrCores.getCoreToClose();
1175            removeMe != null && !container.isShutDown();
1176            removeMe = solrCores.getCoreToClose()) {
1177         try {
1178           removeMe.close();
1179         } finally {
1180           solrCores.removeFromPendingOps(removeMe.getName());
1181         }
1182       }
1183     }
1184   }
