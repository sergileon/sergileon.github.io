34  /**
35   * Test indexes ~82M docs with 26 terms each, so you get &gt; Integer.MAX_VALUE terms/docs pairs
36   * @lucene.experimental
37   */
38  @SuppressCodecs({ "SimpleText", "Memory", "Direct", "Compressing" })
39  @TimeoutSuite(millis = 4 * TimeUnits.HOUR)
40  public class Test2BPostings extends LuceneTestCase {
41  
42    @Nightly
43    public void test() throws Exception {
44      BaseDirectoryWrapper dir = newFSDirectory(createTempDir("2BPostings"));
45      if (dir instanceof MockDirectoryWrapper) {
46        ((MockDirectoryWrapper)dir).setThrottling(MockDirectoryWrapper.Throttling.NEVER);
47      }
48  
49      IndexWriterConfig iwc = new IndexWriterConfig(new MockAnalyzer(random()))
50          .setMaxBufferedDocs(IndexWriterConfig.DISABLE_AUTO_FLUSH)
51          .setRAMBufferSizeMB(256.0)
52          .setMergeScheduler(new ConcurrentMergeScheduler())
53          .setMergePolicy(newLogMergePolicy(false, 10))
54          .setOpenMode(IndexWriterConfig.OpenMode.CREATE);
55      
56      IndexWriter w = new IndexWriter(dir, iwc);
57  
58      MergePolicy mp = w.getConfig().getMergePolicy();
59      if (mp instanceof LogByteSizeMergePolicy) {
60       // 1 petabyte:
61       ((LogByteSizeMergePolicy) mp).setMaxMergeMB(1024*1024*1024);
62      }
63  
64      Document doc = new Document();
65      FieldType ft = new FieldType(TextField.TYPE_NOT_STORED);
66      ft.setOmitNorms(true);
67      ft.setIndexOptions(IndexOptions.DOCS);
68      Field field = new Field("field", new MyTokenStream(), ft);
69      doc.add(field);
70      
71      final int numDocs = (Integer.MAX_VALUE / 26) + 1;
72      for (int i = 0; i < numDocs; i++) {
73        w.addDocument(doc);
74        if (VERBOSE && i % 100000 == 0) {
75          System.out.println(i + " of " + numDocs + "...");
76        }
77      }
78      w.forceMerge(1);
79      w.close();
80      dir.close();
81    }
82    
83    public static final class MyTokenStream extends TokenStream {
84      private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
85      int index;
86  
87      @Override
88      public boolean incrementToken() {
89        if (index <= 'z') {
90          clearAttributes();
91          termAtt.setLength(1);
92          termAtt.buffer()[0] = (char) index++;
93          return true;
94        }
95        return false;
96      }
97      
98      @Override
99      public void reset() {
100       index = 'a';
101     }
102   }
103 }
