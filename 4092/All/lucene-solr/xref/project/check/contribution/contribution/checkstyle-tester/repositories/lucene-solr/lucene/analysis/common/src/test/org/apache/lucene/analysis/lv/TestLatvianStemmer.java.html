68    /**
69     * decl II nouns with (s,t) -&gt; š and (d,z) -&gt; ž
70     * palatalization will generally conflate to two stems
71     * due to the ambiguity (plural and singular).
72     */
73    public void testNouns2() throws IOException {
74      // decl. II
75      
76      // c -> č palatalization
77      checkOneTerm(a, "lācis",  "lāc"); // nom. sing.
78      checkOneTerm(a, "lāči",   "lāc"); // nom. pl.
79      checkOneTerm(a, "lāča",   "lāc"); // gen. sing.
80      checkOneTerm(a, "lāču",   "lāc"); // gen. pl.
81      checkOneTerm(a, "lācim",  "lāc"); // dat. sing.
82      checkOneTerm(a, "lāčiem", "lāc"); // dat. pl.
83      checkOneTerm(a, "lāci",   "lāc"); // acc. sing.
84      checkOneTerm(a, "lāčus",  "lāc"); // acc. pl.
85      checkOneTerm(a, "lācī",   "lāc"); // loc. sing.
86      checkOneTerm(a, "lāčos",  "lāc"); // loc. pl.
87      checkOneTerm(a, "lāci",   "lāc"); // voc. sing.
88      checkOneTerm(a, "lāči",   "lāc"); // voc. pl.
89      
90      // n -> ņ palatalization
91      checkOneTerm(a, "akmens",   "akmen"); // nom. sing.
92      checkOneTerm(a, "akmeņi",   "akmen"); // nom. pl.
93      checkOneTerm(a, "akmens",   "akmen"); // gen. sing.

95      checkOneTerm(a, "akmenim",  "akmen"); // dat. sing.
96      checkOneTerm(a, "akmeņiem", "akmen"); // dat. pl.
97      checkOneTerm(a, "akmeni",   "akmen"); // acc. sing.
98      checkOneTerm(a, "akmeņus",  "akmen"); // acc. pl.
99      checkOneTerm(a, "akmenī",   "akmen"); // loc. sing.
100     checkOneTerm(a, "akmeņos",  "akmen"); // loc. pl.
101     checkOneTerm(a, "akmens",   "akmen"); // voc. sing.
