41  /**
42   * Selects index terms according to provided pluggable
43   * {@link IndexTermSelector}, and stores them in a prefix trie that's
44   * loaded entirely in RAM stored as an FST.  This terms
45   * index only supports unsigned byte term sort order
46   * (unicode codepoint order when the bytes are UTF8).
47   *
48   * @lucene.experimental */
49  public class VariableGapTermsIndexWriter extends TermsIndexWriterBase {
50    protected IndexOutput out;
51  
52    /** Extension of terms index file */
53    static final String TERMS_INDEX_EXTENSION = "tiv";
54  
55    final static String CODEC_NAME = "VariableGapTermsIndex";
56    final static int VERSION_START = 3;
57    final static int VERSION_CURRENT = VERSION_START;
58  
59    private final List<FSTFieldWriter> fields = new ArrayList<>();
60    
61    @SuppressWarnings("unused") private final FieldInfos fieldInfos; // unread
62    private final IndexTermSelector policy;
63  
64    /** 
65     * Hook for selecting which terms should be placed in the terms index.
66     * <p>
67     * {@link #newField} is called at the start of each new field, and
68     * {@link #isIndexTerm} for each term in that field.
69     * 
70     * @lucene.experimental 
71     */
72    public static abstract class IndexTermSelector {
73      /** 
74       * Called sequentially on every term being written,
75       * returning true if this term should be indexed
76       */
77      public abstract boolean isIndexTerm(BytesRef term, TermStats stats);
78      /**
79       * Called when a new field is started.
80       */
81      public abstract void newField(FieldInfo fieldInfo);
82    }
83  
84    /** Same policy as {@link FixedGapTermsIndexWriter} */
85    public static final class EveryNTermSelector extends IndexTermSelector {
86      private int count;
87      private final int interval;
88  
89      public EveryNTermSelector(int interval) {
90        this.interval = interval;
91        // First term is first indexed term:
92        count = interval;
93      }
94  
95      @Override
96      public boolean isIndexTerm(BytesRef term, TermStats stats) {
97        if (count >= interval) {
98          count = 1;
99          return true;
100       } else {
101         count++;
102         return false;
103       }
104     }
105 
106     @Override
107     public void newField(FieldInfo fieldInfo) {
108       count = interval;
109     }
110   }
111 
112   /** Sets an index term when docFreq &gt;= docFreqThresh, or
113    *  every interval terms.  This should reduce seek time
114    *  to high docFreq terms.  */
115   public static final class EveryNOrDocFreqTermSelector extends IndexTermSelector {
116     private int count;
117     private final int docFreqThresh;
118     private final int interval;
119 
120     public EveryNOrDocFreqTermSelector(int docFreqThresh, int interval) {
121       this.interval = interval;
122       this.docFreqThresh = docFreqThresh;
123 
124       // First term is first indexed term:
125       count = interval;
126     }
127 
128     @Override
129     public boolean isIndexTerm(BytesRef term, TermStats stats) {
130       if (stats.docFreq >= docFreqThresh || count >= interval) {
131         count = 1;
132         return true;
133       } else {
134         count++;
135         return false;
136       }
137     }
138 
139     @Override
140     public void newField(FieldInfo fieldInfo) {
141       count = interval;
142     }
143   }

145   // TODO: it'd be nice to let the FST builder prune based
146   // on term count of each node (the prune1/prune2 that it
147   // accepts), and build the index based on that.  This
148   // should result in a more compact terms index, more like
149   // a prefix trie than the other selectors, because it
150   // only stores enough leading bytes to get down to N
151   // terms that may complete that prefix.  It becomes
152   // "deeper" when terms are dense, and "shallow" when they
153   // are less dense.
154   //
155   // However, it's not easy to make that work this this
156   // API, because that pruning doesn't immediately know on
157   // seeing each term whether that term will be a seek point
158   // or not.  It requires some non-causality in the API, ie
159   // only on seeing some number of future terms will the
160   // builder decide which past terms are seek points.
161   // Somehow the API'd need to be able to return a "I don't
162   // know" value, eg like a Future, which only later on is
163   // flipped (frozen) to true or false.
164   //
165   // We could solve this with a 2-pass approach, where the
166   // first pass would build an FSA (no outputs) solely to
167   // determine which prefixes are the 'leaves' in the
168   // pruning. The 2nd pass would then look at this prefix
169   // trie to mark the seek points and build the FST mapping
170   // to the true output.
171   //
172   // But, one downside to this approach is that it'd result
173   // in uneven index term selection.  EG with prune1=10, the
174   // resulting index terms could be as frequent as every 10
175   // terms or as rare as every <maxArcCount> * 10 (eg 2560),
176   // in the extremes.
177 
178   public VariableGapTermsIndexWriter(SegmentWriteState state, IndexTermSelector policy) throws IOException {
179     final String indexFileName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, TERMS_INDEX_EXTENSION);
180     out = state.directory.createOutput(indexFileName, state.context);
181     boolean success = false;
182     try {
183       fieldInfos = state.fieldInfos;
184       this.policy = policy;
185       CodecUtil.writeIndexHeader(out, CODEC_NAME, VERSION_CURRENT, state.segmentInfo.getId(), state.segmentSuffix);
186       success = true;
187     } finally {
188       if (!success) {
189         IOUtils.closeWhileHandlingException(out);
190       }
191     }
192   }
193 
194   @Override
195   public FieldWriter addField(FieldInfo field, long termsFilePointer) throws IOException {
196     ////System.out.println("VGW: field=" + field.name);
197     policy.newField(field);
198     FSTFieldWriter writer = new FSTFieldWriter(field, termsFilePointer);
199     fields.add(writer);
200     return writer;
201   }

203   /** NOTE: if your codec does not sort in unicode code
204    *  point order, you must override this method, to simply
205    *  return indexedTerm.length. */
206   protected int indexedTermPrefixLength(final BytesRef priorTerm, final BytesRef indexedTerm) {
207     // As long as codec sorts terms in unicode codepoint
208     // order, we can safely strip off the non-distinguishing
209     // suffix to save RAM in the loaded terms index.
210     final int idxTermOffset = indexedTerm.offset;
211     final int priorTermOffset = priorTerm.offset;
212     final int limit = Math.min(priorTerm.length, indexedTerm.length);
213     for(int byteIdx=0;byteIdx<limit;byteIdx++) {
214       if (priorTerm.bytes[priorTermOffset+byteIdx] != indexedTerm.bytes[idxTermOffset+byteIdx]) {
215         return byteIdx+1;
216       }
217     }
218     return Math.min(1+priorTerm.length, indexedTerm.length);
219   }

221   private class FSTFieldWriter extends FieldWriter {
222     private final Builder<Long> fstBuilder;
223     private final PositiveIntOutputs fstOutputs;
224     private final long startTermsFilePointer;
225 
226     final FieldInfo fieldInfo;
227     FST<Long> fst;
228     final long indexStart;
229 
230     private final BytesRefBuilder lastTerm = new BytesRefBuilder();
231     private boolean first = true;
232 
233     public FSTFieldWriter(FieldInfo fieldInfo, long termsFilePointer) throws IOException {
234       this.fieldInfo = fieldInfo;
235       fstOutputs = PositiveIntOutputs.getSingleton();
236       fstBuilder = new Builder<>(FST.INPUT_TYPE.BYTE1, fstOutputs);
237       indexStart = out.getFilePointer();
238       ////System.out.println("VGW: field=" + fieldInfo.name);
239 
240       // Always put empty string in
241       fstBuilder.add(new IntsRef(), termsFilePointer);
242       startTermsFilePointer = termsFilePointer;
243     }
244 
245     @Override
246     public boolean checkIndexTerm(BytesRef text, TermStats stats) throws IOException {
247       //System.out.println("VGW: index term=" + text.utf8ToString());
248       // NOTE: we must force the first term per field to be
249       // indexed, in case policy doesn't:
250       if (policy.isIndexTerm(text, stats) || first) {
251         first = false;
252         //System.out.println("  YES");
253         return true;
254       } else {
255         lastTerm.copyBytes(text);
256         return false;
257       }
258     }
259 
260     private final IntsRefBuilder scratchIntsRef = new IntsRefBuilder();
261 
262     @Override
263     public void add(BytesRef text, TermStats stats, long termsFilePointer) throws IOException {
264       if (text.length == 0) {
265         // We already added empty string in ctor
266         assert termsFilePointer == startTermsFilePointer;
267         return;
268       }
269       final int lengthSave = text.length;
270       text.length = indexedTermPrefixLength(lastTerm.get(), text);
271       try {
272         fstBuilder.add(Util.toIntsRef(text, scratchIntsRef), termsFilePointer);
273       } finally {
274         text.length = lengthSave;
275       }
276       lastTerm.copyBytes(text);
277     }
278 
279     @Override
280     public void finish(long termsFilePointer) throws IOException {
281       fst = fstBuilder.finish();
282       if (fst != null) {
283         fst.save(out);
284       }
285     }
286   }
287 
288   @Override
289   public void close() throws IOException {
290     if (out != null) {
291       try {
292         final long dirStart = out.getFilePointer();
293         final int fieldCount = fields.size();
294         
295         int nonNullFieldCount = 0;
296         for(int i=0;i<fieldCount;i++) {
297           FSTFieldWriter field = fields.get(i);
298           if (field.fst != null) {
299             nonNullFieldCount++;
300           }
301         }
302         
303         out.writeVInt(nonNullFieldCount);
304         for(int i=0;i<fieldCount;i++) {
305           FSTFieldWriter field = fields.get(i);
306           if (field.fst != null) {
307             out.writeVInt(field.fieldInfo.number);
308             out.writeVLong(field.indexStart);
309           }
310         }
311         writeTrailer(dirStart);
312         CodecUtil.writeFooter(out);
313       } finally {
314         out.close();
315         out = null;
316       }
317     }
318   }
319 
320   private void writeTrailer(long dirStart) throws IOException {
321     out.writeLong(dirStart);
322   }
