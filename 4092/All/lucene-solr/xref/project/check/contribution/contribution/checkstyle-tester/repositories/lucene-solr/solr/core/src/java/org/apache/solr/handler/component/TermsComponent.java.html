43  /**
44   * Return TermEnum information, useful for things like auto suggest.
45   * 
46   * <pre class="prettyprint">
47   * &lt;searchComponent name="termsComponent" class="solr.TermsComponent"/&gt;
48   * 
49   * &lt;requestHandler name="/terms" class="solr.SearchHandler"&gt;
50   *   &lt;lst name="defaults"&gt;
51   *     &lt;bool name="terms"&gt;true&lt;/bool&gt;
52   *   &lt;/lst&gt;
53   *   &lt;arr name="components"&gt;
54   *     &lt;str&gt;termsComponent&lt;/str&gt;
55   *   &lt;/arr&gt;
56   * &lt;/requestHandler&gt;</pre>
57   *
58   * @see org.apache.solr.common.params.TermsParams
59   *      See Lucene's TermEnum class
60   *
61   */
62  public class TermsComponent extends SearchComponent {
63    public static final int UNLIMITED_MAX_COUNT = -1;
64    public static final String COMPONENT_NAME = "terms";
65  
66    @Override
67    public void prepare(ResponseBuilder rb) throws IOException {
68      SolrParams params = rb.req.getParams();
69      if (params.getBool(TermsParams.TERMS, false)) {
70        rb.doTerms = true;
71      }
72  
73      // TODO: temporary... this should go in a different component.
74      String shards = params.get(ShardParams.SHARDS);
75      if (shards != null) {
76        rb.isDistrib = true;
77        if (params.get(ShardParams.SHARDS_QT) == null) {
78          throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "No shards.qt parameter specified");
79        }
80        List<String> lst = StrUtils.splitSmart(shards, ",", true);
81        rb.shards = lst.toArray(new String[lst.size()]);
82      }
83    }

85    @Override
86    public void process(ResponseBuilder rb) throws IOException {
87      SolrParams params = rb.req.getParams();
88      if (!params.getBool(TermsParams.TERMS, false)) return;
89  
90      String[] fields = params.getParams(TermsParams.TERMS_FIELD);
91  
92      NamedList<Object> termsResult = new SimpleOrderedMap<>();
93      rb.rsp.add("terms", termsResult);
94  
95      if (fields == null || fields.length==0) return;
96  
97      int limit = params.getInt(TermsParams.TERMS_LIMIT, 10);
98      if (limit < 0) {
99        limit = Integer.MAX_VALUE;
100     }
101 
102     String lowerStr = params.get(TermsParams.TERMS_LOWER);
103     String upperStr = params.get(TermsParams.TERMS_UPPER);
104     boolean upperIncl = params.getBool(TermsParams.TERMS_UPPER_INCLUSIVE, false);
105     boolean lowerIncl = params.getBool(TermsParams.TERMS_LOWER_INCLUSIVE, true);
106     boolean sort = !TermsParams.TERMS_SORT_INDEX.equals(
107         params.get(TermsParams.TERMS_SORT, TermsParams.TERMS_SORT_COUNT));
108     int freqmin = params.getInt(TermsParams.TERMS_MINCOUNT, 1);
109     int freqmax = params.getInt(TermsParams.TERMS_MAXCOUNT, UNLIMITED_MAX_COUNT);
110     if (freqmax<0) {
111       freqmax = Integer.MAX_VALUE;
112     }
113     String prefix = params.get(TermsParams.TERMS_PREFIX_STR);
114     String regexp = params.get(TermsParams.TERMS_REGEXP_STR);
115     Pattern pattern = regexp != null ? Pattern.compile(regexp, resolveRegexpFlags(params)) : null;
116 
117     boolean raw = params.getBool(TermsParams.TERMS_RAW, false);
118 
119 
120     final LeafReader indexReader = rb.req.getSearcher().getLeafReader();
121     Fields lfields = indexReader.fields();
122 
123     for (String field : fields) {
124       NamedList<Integer> fieldTerms = new NamedList<>();
125       termsResult.add(field, fieldTerms);
126 
127       Terms terms = lfields.terms(field);
128       if (terms == null) {
129         // field does not exist
130         continue;
131       }
132 
133       FieldType ft = raw ? null : rb.req.getSchema().getFieldTypeNoEx(field);
134       if (ft==null) ft = new StrField();
135 
136       // prefix must currently be text
137       BytesRef prefixBytes = prefix==null ? null : new BytesRef(prefix);
138 
139       BytesRef upperBytes = null;
140       if (upperStr != null) {
141         BytesRefBuilder b = new BytesRefBuilder();
142         ft.readableToIndexed(upperStr, b);
143         upperBytes = b.get();
144       }
145 
146       BytesRef lowerBytes;
147       if (lowerStr == null) {
148         // If no lower bound was specified, use the prefix
149         lowerBytes = prefixBytes;
150       } else {
151         lowerBytes = new BytesRef();
152         if (raw) {
153           // TODO: how to handle binary? perhaps we don't for "raw"... or if the field exists
154           // perhaps we detect if the FieldType is non-character and expect hex if so?
155           lowerBytes = new BytesRef(lowerStr);
156         } else {
157           BytesRefBuilder b = new BytesRefBuilder();
158           ft.readableToIndexed(lowerStr, b);
159           lowerBytes = b.get();
160         }
161       }
162 
163 
164      TermsEnum termsEnum = terms.iterator();
165      BytesRef term = null;
166 
167       if (lowerBytes != null) {
168         if (termsEnum.seekCeil(lowerBytes) == TermsEnum.SeekStatus.END) {
169           termsEnum = null;
170         } else {
171           term = termsEnum.term();
172           //Only advance the enum if we are excluding the lower bound and the lower Term actually matches
173           if (lowerIncl == false && term.equals(lowerBytes)) {
174             term = termsEnum.next();
175           }
176         }
177       } else {
178         // position termsEnum on first term
179         term = termsEnum.next();
180       }
181 
182       int i = 0;
183       BoundedTreeSet<CountPair<BytesRef, Integer>> queue = (sort ? new BoundedTreeSet<CountPair<BytesRef, Integer>>(limit) : null);
184       CharsRefBuilder external = new CharsRefBuilder();
185       while (term != null && (i<limit || sort)) {
186         boolean externalized = false; // did we fill in "external" yet for this term?
187 
188         // stop if the prefix doesn't match
189         if (prefixBytes != null && !StringHelper.startsWith(term, prefixBytes)) break;
190 
191         if (pattern != null) {
192           // indexed text or external text?
193           // TODO: support "raw" mode?
194           ft.indexedToReadable(term, external);
195           externalized = true;
196           if (!pattern.matcher(external.get()).matches()) {
197             term = termsEnum.next();
198             continue;
199           }
200         }
201 
202         if (upperBytes != null) {
203           int upperCmp = term.compareTo(upperBytes);
204           // if we are past the upper term, or equal to it (when don't include upper) then stop.
205           if (upperCmp>0 || (upperCmp==0 && !upperIncl)) break;
206         }
207 
208         // This is a good term in the range.  Check if mincount/maxcount conditions are satisfied.
209         int docFreq = termsEnum.docFreq();
210         if (docFreq >= freqmin && docFreq <= freqmax) {
211           // add the term to the list
212           if (sort) {
213             queue.add(new CountPair<>(BytesRef.deepCopyOf(term), docFreq));
214           } else {
215 
216             // TODO: handle raw somehow
217             if (!externalized) {
218               ft.indexedToReadable(term, external);
219             }
220             fieldTerms.add(external.toString(), docFreq);
221             i++;
222           }
223         }
224 
225         term = termsEnum.next();
226       }
227 
228       if (sort) {
229         for (CountPair<BytesRef, Integer> item : queue) {
230           if (i >= limit) break;
231           ft.indexedToReadable(item.key, external);          
232           fieldTerms.add(external.toString(), item.val);
233           i++;
234         }
235       }
236     }
237   }

239   int resolveRegexpFlags(SolrParams params) {
240       String[] flagParams = params.getParams(TermsParams.TERMS_REGEXP_FLAG);
241       if (flagParams == null) {
242           return 0;
243       }
244       int flags = 0;
245       for (String flagParam : flagParams) {
246           try {
247             flags |= TermsParams.TermsRegexpFlag.valueOf(flagParam.toUpperCase(Locale.ROOT)).getValue();
248           } catch (IllegalArgumentException iae) {
249               throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Unknown terms regex flag '" + flagParam + "'");
250           }
251       }
252       return flags;
253   }

255   @Override
256   public int distributedProcess(ResponseBuilder rb) throws IOException {
257     if (!rb.doTerms) {
258       return ResponseBuilder.STAGE_DONE;
259     }
260 
261     if (rb.stage == ResponseBuilder.STAGE_EXECUTE_QUERY) {
262       TermsHelper th = rb._termsHelper;
263       if (th == null) {
264         th = rb._termsHelper = new TermsHelper();
265         th.init(rb.req.getParams());
266       }
267       ShardRequest sreq = createShardQuery(rb.req.getParams());
268       rb.addRequest(this, sreq);
269     }
270 
271     if (rb.stage < ResponseBuilder.STAGE_EXECUTE_QUERY) {
272       return ResponseBuilder.STAGE_EXECUTE_QUERY;
273     } else {
274       return ResponseBuilder.STAGE_DONE;
275     }
276   }

278   @Override
279   public void handleResponses(ResponseBuilder rb, ShardRequest sreq) {
280     if (!rb.doTerms || (sreq.purpose & ShardRequest.PURPOSE_GET_TERMS) == 0) {
281       return;
282     }
283     TermsHelper th = rb._termsHelper;
284     if (th != null) {
285       for (ShardResponse srsp : sreq.responses) {
286         @SuppressWarnings("unchecked")
287         NamedList<NamedList<Number>> terms = (NamedList<NamedList<Number>>) srsp.getSolrResponse().getResponse().get("terms");
288         th.parse(terms);
289       }
290     }
291   }
292 
293   @Override
294   public void finishStage(ResponseBuilder rb) {
295     if (!rb.doTerms || rb.stage != ResponseBuilder.STAGE_EXECUTE_QUERY) {
296       return;
297     }
298 
299     TermsHelper ti = rb._termsHelper;
300     NamedList terms = ti.buildResponse();
301 
302     rb.rsp.add("terms", terms);
303     rb._termsHelper = null;
304   }

306   private ShardRequest createShardQuery(SolrParams params) {
307     ShardRequest sreq = new ShardRequest();
308     sreq.purpose = ShardRequest.PURPOSE_GET_TERMS;
309 
310     // base shard request on original parameters
311     sreq.params = new ModifiableSolrParams(params);
312 
313     // remove any limits for shards, we want them to return all possible
314     // responses
315     // we want this so we can calculate the correct counts
316     // dont sort by count to avoid that unnecessary overhead on the shards
317     sreq.params.remove(TermsParams.TERMS_MAXCOUNT);
318     sreq.params.remove(TermsParams.TERMS_MINCOUNT);
319     sreq.params.set(TermsParams.TERMS_LIMIT, -1);
320     sreq.params.set(TermsParams.TERMS_SORT, TermsParams.TERMS_SORT_INDEX);
321 
322     return sreq;
323   }

325   public class TermsHelper {
326     // map to store returned terms
327     private HashMap<String, HashMap<String, TermsResponse.Term>> fieldmap;
328     private SolrParams params;
329 
330     public TermsHelper() {
331       fieldmap = new HashMap<>(5);
332     }
333 
334     public void init(SolrParams params) {
335       this.params = params;
336       String[] fields = params.getParams(TermsParams.TERMS_FIELD);
337       if (fields != null) {
338         for (String field : fields) {
339           // TODO: not sure 128 is the best starting size
340           // It use it because that is what is used for facets
341           fieldmap.put(field, new HashMap<String, TermsResponse.Term>(128));
342         }
343       }
344     }
345 
346     public void parse(NamedList<NamedList<Number>> terms) {
347       // exit if there is no terms
348       if (terms == null) {
349         return;
350       }
351 
352       TermsResponse termsResponse = new TermsResponse(terms);
353       
354       // loop though each field and add each term+freq to map
355       for (String key : fieldmap.keySet()) {
356         HashMap<String, TermsResponse.Term> termmap = fieldmap.get(key);
357         List<TermsResponse.Term> termlist = termsResponse.getTerms(key); 
358 
359         // skip this field if there are no terms
360         if (termlist == null) {
361           continue;
362         }
363 
364         // loop though each term
365         for (TermsResponse.Term tc : termlist) {
366           String term = tc.getTerm();
367           if (termmap.containsKey(term)) {
368             TermsResponse.Term oldtc = termmap.get(term);
369             oldtc.addFrequency(tc.getFrequency());
370             termmap.put(term, oldtc);
371           } else {
372             termmap.put(term, tc);
373           }
374         }
375       }
376     }
377 
378     public NamedList buildResponse() {
379       NamedList<Object> response = new SimpleOrderedMap<>();
380 
381       // determine if we are going index or count sort
382       boolean sort = !TermsParams.TERMS_SORT_INDEX.equals(params.get(
383           TermsParams.TERMS_SORT, TermsParams.TERMS_SORT_COUNT));
384 
385       // init minimum frequency
386       long freqmin = 1;
387       String s = params.get(TermsParams.TERMS_MINCOUNT);
388       if (s != null)  freqmin = Long.parseLong(s);
389 
390       // init maximum frequency, default to max int
391       long freqmax = -1;
392       s = params.get(TermsParams.TERMS_MAXCOUNT);
393       if (s != null)  freqmax = Long.parseLong(s);
394       if (freqmax < 0) {
395         freqmax = Long.MAX_VALUE;
396       }
397 
398       // init limit, default to max int
399       long limit = 10;
400       s = params.get(TermsParams.TERMS_LIMIT);
401       if (s != null)  limit = Long.parseLong(s);
402       if (limit < 0) {
403         limit = Long.MAX_VALUE;
404       }
405 
406       // loop though each field we want terms from
407       for (String key : fieldmap.keySet()) {
408         NamedList<Number> fieldterms = new SimpleOrderedMap<>();
409         TermsResponse.Term[] data = null;
410         if (sort) {
411           data = getCountSorted(fieldmap.get(key));
412         } else {
413           data = getLexSorted(fieldmap.get(key));
414         }
415 
416         // loop though each term until we hit limit
417         int cnt = 0;
418         for (TermsResponse.Term tc : data) {
419           if (tc.getFrequency() >= freqmin && tc.getFrequency() <= freqmax) {
420             fieldterms.add(tc.getTerm(), num(tc.getFrequency()));
421             cnt++;
422           }
423 
424           if (cnt >= limit) {
425             break;
426           }
427         }
428 
429         response.add(key, fieldterms);
430       }
431 
432       return response;
433     }
434 
435     // use <int> tags for smaller facet counts (better back compatibility)
436     private Number num(long val) {
437       if (val < Integer.MAX_VALUE) return (int) val;
438       else return val;
439     }
440 
441     // based on code from facets
442     public TermsResponse.Term[] getLexSorted(HashMap<String, TermsResponse.Term> data) {
443       TermsResponse.Term[] arr = data.values().toArray(new TermsResponse.Term[data.size()]);
444 
445       Arrays.sort(arr, new Comparator<TermsResponse.Term>() {
446         @Override
447         public int compare(TermsResponse.Term o1, TermsResponse.Term o2) {
448           return o1.getTerm().compareTo(o2.getTerm());
449         }
450       });
451 
452       return arr;
453     }
454 
455     // based on code from facets
456     public TermsResponse.Term[] getCountSorted(HashMap<String, TermsResponse.Term> data) {
457       TermsResponse.Term[] arr = data.values().toArray(new TermsResponse.Term[data.size()]);
458 
459       Arrays.sort(arr, new Comparator<TermsResponse.Term>() {
460         @Override
461         public int compare(TermsResponse.Term o1, TermsResponse.Term o2) {
462           long freq1 = o1.getFrequency();
463           long freq2 = o2.getFrequency();
464           
465           if (freq2 < freq1) {
466             return -1;
467           } else if (freq1 < freq2) {
468             return 1;
469           } else {
470             return o1.getTerm().compareTo(o2.getTerm());
471           }
472         }
473       });
474 
475       return arr;
476     }
