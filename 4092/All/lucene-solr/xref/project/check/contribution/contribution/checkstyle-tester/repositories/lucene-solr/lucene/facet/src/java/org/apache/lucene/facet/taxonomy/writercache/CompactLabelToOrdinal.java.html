32  /**
33   * This is a very efficient LabelToOrdinal implementation that uses a
34   * CharBlockArray to store all labels and a configurable number of HashArrays to
35   * reference the labels.
36   * <p>
37   * Since the HashArrays don't handle collisions, a {@link CollisionMap} is used
38   * to store the colliding labels.
39   * <p>
40   * This data structure grows by adding a new HashArray whenever the number of
41   * collisions in the {@link CollisionMap} exceeds {@code loadFactor} * 
42   * {@link #getMaxOrdinal()}. Growing also includes reinserting all colliding
43   * labels into the HashArrays to possibly reduce the number of collisions.
44   * 
45   * For setting the {@code loadFactor} see 
46   * {@link #CompactLabelToOrdinal(int, float, int)}. 
47   * 
48   * <p>
49   * This data structure has a much lower memory footprint (~30%) compared to a
50   * Java HashMap&lt;String, Integer&gt;. It also only uses a small fraction of objects
51   * a HashMap would use, thus limiting the GC overhead. Ingestion speed was also
52   * ~50% faster compared to a HashMap for 3M unique labels.
53   * 
54   * @lucene.experimental
55   */
56  public class CompactLabelToOrdinal extends LabelToOrdinal {
57  
58    /** Default maximum load factor. */
59    public static final float DefaultLoadFactor = 0.15f;
60  
61    static final char TERMINATOR_CHAR = 0xffff;
62    private static final int COLLISION = -5;
63  
64    private HashArray[] hashArrays;
65    private CollisionMap collisionMap;
66    private CharBlockArray labelRepository;
67  
68    private int capacity;
69    private int threshold;
70    private float loadFactor;
71  
72    /** How many labels. */
73    public int sizeOfMap() {
74      return this.collisionMap.size();
75    }
76  
77    private CompactLabelToOrdinal() {
78    }
79  
80    /** Sole constructor. */
81    public CompactLabelToOrdinal(int initialCapacity, float loadFactor,
82                                  int numHashArrays) {
83  
84      this.hashArrays = new HashArray[numHashArrays];
85  
86      this.capacity = determineCapacity((int) Math.pow(2, numHashArrays),
87          initialCapacity);
88      init();
89      this.collisionMap = new CollisionMap(this.labelRepository);
90  
91      this.counter = 0;
92      this.loadFactor = loadFactor;
93  
94      this.threshold = (int) (this.loadFactor * this.capacity);
95    }
96  
97    static int determineCapacity(int minCapacity, int initialCapacity) {
98      int capacity = minCapacity;
99      while (capacity < initialCapacity) {
100       capacity <<= 1;
101     }
102     return capacity;
103   }
104 
105   private void init() {
106     labelRepository = new CharBlockArray();
107     CategoryPathUtils.serialize(new FacetLabel(), labelRepository);
108 
109     int c = this.capacity;
110     for (int i = 0; i < this.hashArrays.length; i++) {
111       this.hashArrays[i] = new HashArray(c);
112       c /= 2;
113     }
114   }
115 
116   @Override
117   public void addLabel(FacetLabel label, int ordinal) {
118     if (collisionMap.size() > threshold) {
119       grow();
120     }
121 
122     int hash = CompactLabelToOrdinal.stringHashCode(label);
123     for (int i = 0; i < this.hashArrays.length; i++) {
124       if (addLabel(this.hashArrays[i], label, hash, ordinal)) {
125         return;
126       }
127     }
128 
129     int prevVal = collisionMap.addLabel(label, hash, ordinal);
130     if (prevVal != ordinal) {
131       throw new IllegalArgumentException("Label already exists: " + label + " prev ordinal " + prevVal);
132     }
133   }
134 
135   @Override
136   public int getOrdinal(FacetLabel label) {
137     if (label == null) {
138       return LabelToOrdinal.INVALID_ORDINAL;
139     }
140 
141     int hash = CompactLabelToOrdinal.stringHashCode(label);
142     for (int i = 0; i < this.hashArrays.length; i++) {
143       int ord = getOrdinal(this.hashArrays[i], label, hash);
144       if (ord != COLLISION) {
145         return ord;
146       }
147     }
148 
149     return this.collisionMap.get(label, hash);
150   }
151 
152   private void grow() {
153     HashArray temp = this.hashArrays[this.hashArrays.length - 1];
154 
155     for (int i = this.hashArrays.length - 1; i > 0; i--) {
156       this.hashArrays[i] = this.hashArrays[i - 1];
157     }
158 
159     this.capacity *= 2;
160     this.hashArrays[0] = new HashArray(this.capacity);
161 
162     for (int i = 1; i < this.hashArrays.length; i++) {
163       int[] sourceOffsetArray = this.hashArrays[i].offsets;
164       int[] sourceCidsArray = this.hashArrays[i].cids;
165 
166       for (int k = 0; k < sourceOffsetArray.length; k++) {
167 
168         for (int j = 0; j < i && sourceOffsetArray[k] != 0; j++) {
169           int[] targetOffsetArray = this.hashArrays[j].offsets;
170           int[] targetCidsArray = this.hashArrays[j].cids;
171 
172           int newIndex = indexFor(stringHashCode(
173               this.labelRepository, sourceOffsetArray[k]),
174               targetOffsetArray.length);
175           if (targetOffsetArray[newIndex] == 0) {
176             targetOffsetArray[newIndex] = sourceOffsetArray[k];
177             targetCidsArray[newIndex] = sourceCidsArray[k];
178             sourceOffsetArray[k] = 0;
179           }
180         }
181       }
182     }
183 
184     for (int i = 0; i < temp.offsets.length; i++) {
185       int offset = temp.offsets[i];
186       if (offset > 0) {
187         int hash = stringHashCode(this.labelRepository, offset);
188         addLabelOffset(hash, temp.cids[i], offset);
189       }
190     }
191 
192     CollisionMap oldCollisionMap = this.collisionMap;
193     this.collisionMap = new CollisionMap(oldCollisionMap.capacity(),
194         this.labelRepository);
195     this.threshold = (int) (this.capacity * this.loadFactor);
196 
197     Iterator<CollisionMap.Entry> it = oldCollisionMap.entryIterator();
198     while (it.hasNext()) {
199       CollisionMap.Entry e = it.next();
200       addLabelOffset(stringHashCode(this.labelRepository, e.offset),
201           e.cid, e.offset);
202     }
203   }
204 
205   private boolean addLabel(HashArray a, FacetLabel label, int hash, int ordinal) {
206     int index = CompactLabelToOrdinal.indexFor(hash, a.offsets.length);
207     int offset = a.offsets[index];
208 
209     if (offset == 0) {
210       a.offsets[index] = this.labelRepository.length();
211       CategoryPathUtils.serialize(label, labelRepository);
212       a.cids[index] = ordinal;
213       return true;
214     }
215 
216     return false;
217   }
218 
219   private void addLabelOffset(int hash, int cid, int knownOffset) {
220     for (int i = 0; i < this.hashArrays.length; i++) {
221       if (addLabelOffsetToHashArray(this.hashArrays[i], hash, cid,
222           knownOffset)) {
223         return;
224       }
225     }
226 
227     this.collisionMap.addLabelOffset(hash, knownOffset, cid);
228 
229     if (this.collisionMap.size() > this.threshold) {
230       grow();
231     }
232   }
233 
234   private boolean addLabelOffsetToHashArray(HashArray a, int hash, int ordinal,
235                                             int knownOffset) {
236 
237     int index = CompactLabelToOrdinal.indexFor(hash, a.offsets.length);
238     int offset = a.offsets[index];
239 
240     if (offset == 0) {
241       a.offsets[index] = knownOffset;
242       a.cids[index] = ordinal;
243       return true;
244     }
245 
246     return false;
247   }
248 
249   private int getOrdinal(HashArray a, FacetLabel label, int hash) {
250     if (label == null) {
251       return LabelToOrdinal.INVALID_ORDINAL;
252     }
253 
254     int index = indexFor(hash, a.offsets.length);
255     int offset = a.offsets[index];
256     if (offset == 0) {
257       return LabelToOrdinal.INVALID_ORDINAL;
258     }
259 
260     if (CategoryPathUtils.equalsToSerialized(label, labelRepository, offset)) {
261       return a.cids[index];
262     }
263 
264     return COLLISION;
265   }
266 
267   /** Returns index for hash code h. */
268   static int indexFor(int h, int length) {
269     return h & (length - 1);
270   }

286   static int stringHashCode(FacetLabel label) {
287     int hash = label.hashCode();
288 
289     hash = hash ^ ((hash >>> 20) ^ (hash >>> 12));
290     hash = hash ^ (hash >>> 7) ^ (hash >>> 4);
291 
292     return hash;
293 
294   }

296   static int stringHashCode(CharBlockArray labelRepository, int offset) {
297     int hash = CategoryPathUtils.hashCodeOfSerialized(labelRepository, offset);
298     hash = hash ^ ((hash >>> 20) ^ (hash >>> 12));
299     hash = hash ^ (hash >>> 7) ^ (hash >>> 4);
300     return hash;
301   }

325   /**
326    * Returns an estimate of the amount of memory used by this table. Called only in
327    * this package. Memory is consumed mainly by three structures: the hash arrays,
328    * label repository and collision map.
329    */
330   int getMemoryUsage() {
331     int memoryUsage = 0;
332     if (this.hashArrays != null) {
333       // HashArray capacity is instance-specific.
334       for (HashArray ha : this.hashArrays) {
335         // Each has 2 capacity-length arrays of ints.
336         memoryUsage += ( ha.capacity * 2 * 4 ) + 4;
337       }
338     }
339     if (this.labelRepository != null) {
340       // All blocks are the same size.
341       int blockSize = this.labelRepository.blockSize;
342       // Each block has room for blockSize UTF-16 chars.
343       int actualBlockSize = ( blockSize * 2 ) + 4;
344       memoryUsage += this.labelRepository.blocks.size() * actualBlockSize; 
345       memoryUsage += 8;   // Two int values for array as a whole.
346     }
347     if (this.collisionMap != null) {
348       memoryUsage += this.collisionMap.getMemoryUsage();
349     }
350     return memoryUsage;
351   }

353   /**
354    * Opens the file and reloads the CompactLabelToOrdinal. The file it expects
355    * is generated from the {@link #flush(Path)} command.
356    */
357   static CompactLabelToOrdinal open(Path file, float loadFactor,
358                                     int numHashArrays) throws IOException {
359     /**
360      * Part of the file is the labelRepository, which needs to be rehashed
361      * and label offsets re-added to the object. I am unsure as to why we
362      * can't just store these off in the file as well, but in keeping with
363      * the spirit of the original code, I did it this way. (ssuppe)
364      */
365     CompactLabelToOrdinal l2o = new CompactLabelToOrdinal();
366     l2o.loadFactor = loadFactor;
367     l2o.hashArrays = new HashArray[numHashArrays];
368 
369     DataInputStream dis = null;
370     try {
371       dis = new DataInputStream(new BufferedInputStream(
372           Files.newInputStream(file)));
373 
374       // TaxiReader needs to load the "counter" or occupancy (L2O) to know
375       // the next unique facet. we used to load the delimiter too, but
376       // never used it.
377       l2o.counter = dis.readInt();
378 
379       l2o.capacity = determineCapacity((int) Math.pow(2,
380           l2o.hashArrays.length), l2o.counter);
381       l2o.init();
382 
383       // now read the chars
384       l2o.labelRepository = CharBlockArray.open(dis);
385 
386       l2o.collisionMap = new CollisionMap(l2o.labelRepository);
387 
388       // Calculate hash on the fly based on how CategoryPath hashes
389       // itself. Maybe in the future we can call some static based methods
390       // in CategoryPath so that this doesn't break again? I don't like
391       // having code in two different places...
392       int cid = 0;
393       // Skip the initial offset, it's the CategoryPath(0,0), which isn't
394       // a hashed value.
395       int offset = 1;
396       int lastStartOffset = offset;
397       // This loop really relies on a well-formed input (assumes pretty blindly
398       // that array offsets will work).  Since the initial file is machine 
399       // generated, I think this should be OK.
400       while (offset < l2o.labelRepository.length()) {
401         // identical code to CategoryPath.hashFromSerialized. since we need to
402         // advance offset, we cannot call the method directly. perhaps if we
403         // could pass a mutable Integer or something...
404         int length = (short) l2o.labelRepository.charAt(offset++);
405         int hash = length;
406         if (length != 0) {
407           for (int i = 0; i < length; i++) {
408             int len = (short) l2o.labelRepository.charAt(offset++);
409             hash = hash * 31 + l2o.labelRepository.subSequence(offset, offset + len).hashCode();
410             offset += len;
411           }
412         }
413         // Now that we've hashed the components of the label, do the
414         // final part of the hash algorithm.
415         hash = hash ^ ((hash >>> 20) ^ (hash >>> 12));
416         hash = hash ^ (hash >>> 7) ^ (hash >>> 4);
417         // Add the label, and let's keep going
418         l2o.addLabelOffset(hash, cid, lastStartOffset);
419         cid++;
420         lastStartOffset = offset;
421       }
422 
423     } catch (ClassNotFoundException cnfe) {
424       throw new IOException("Invalid file format. Cannot deserialize.");
425     } finally {
426       if (dis != null) {
427         dis.close();
428       }
429     }
430 
431     l2o.threshold = (int) (l2o.loadFactor * l2o.capacity);
432     return l2o;
433 
434   }

436   void flush(Path file) throws IOException {
437     OutputStream fos = Files.newOutputStream(file);
438 
439     try {
440       BufferedOutputStream os = new BufferedOutputStream(fos);
441 
442       DataOutputStream dos = new DataOutputStream(os);
443       dos.writeInt(this.counter);
444 
445       // write the labelRepository
446       this.labelRepository.flush(dos);
447 
448       // Closes the data output stream
449       dos.close();
450 
451     } finally {
452       fos.close();
453     }
454   }

458     int[] cids;

462     HashArray(int c) {
463       this.capacity = c;
464       this.offsets = new int[this.capacity];
465       this.cids = new int[this.capacity];
