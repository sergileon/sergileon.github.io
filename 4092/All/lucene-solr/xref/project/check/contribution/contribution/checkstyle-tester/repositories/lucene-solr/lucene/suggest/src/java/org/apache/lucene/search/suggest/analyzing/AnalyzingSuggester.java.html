67  /**
68   * Suggester that first analyzes the surface form, adds the
69   * analyzed form to a weighted FST, and then does the same
70   * thing at lookup time.  This means lookup is based on the
71   * analyzed form while suggestions are still the surface
72   * form(s).
73   *
74   * <p>
75   * This can result in powerful suggester functionality.  For
76   * example, if you use an analyzer removing stop words, 
77   * then the partial text "ghost chr..." could see the
78   * suggestion "The Ghost of Christmas Past". Note that
79   * position increments MUST NOT be preserved for this example
80   * to work, so you should call the constructor with 
81   * <code>preservePositionIncrements</code> parameter set to 
82   * false
83   *
84   * <p>
85   * If SynonymFilter is used to map wifi and wireless network to
86   * hotspot then the partial text "wirele..." could suggest
87   * "wifi router".  Token normalization like stemmers, accent
88   * removal, etc., would allow suggestions to ignore such
89   * variations.
90   *
91   * <p>
92   * When two matching suggestions have the same weight, they
93   * are tie-broken by the analyzed form.  If their analyzed
94   * form is the same then the order is undefined.
95   *
96   * <p>
97   * There are some limitations:
98   * <ul>
99   *
100  *   <li> A lookup from a query like "net" in English won't
101  *        be any different than "net " (ie, user added a
102  *        trailing space) because analyzers don't reflect
103  *        when they've seen a token separator and when they
104  *        haven't.
105  *
106  *   <li> If you're using {@code StopFilter}, and the user will
107  *        type "fast apple", but so far all they've typed is
108  *        "fast a", again because the analyzer doesn't convey whether
109  *        it's seen a token separator after the "a",
110  *        {@code StopFilter} will remove that "a" causing
111  *        far more matches than you'd expect.
112  *
113  *   <li> Lookups with the empty string return no results
114  *        instead of all results.
115  * </ul>
116  * 
117  * @lucene.experimental
118  */
119 public class AnalyzingSuggester extends Lookup {
120

214   /**
215    * Creates a new suggester.
216    * 
217    * @param indexAnalyzer Analyzer that will be used for
218    *   analyzing suggestions while building the index.
219    * @param queryAnalyzer Analyzer that will be used for
220    *   analyzing query text during lookup
221    * @param options see {@link #EXACT_FIRST}, {@link #PRESERVE_SEP}
222    * @param maxSurfaceFormsPerAnalyzedForm Maximum number of
223    *   surface forms to keep for a single analyzed form.
224    *   When there are too many surface forms we discard the
225    *   lowest weighted ones.
226    * @param maxGraphExpansions Maximum number of graph paths
227    *   to expand from the analyzed form.  Set this to -1 for
228    *   no limit.
229    * @param preservePositionIncrements Whether position holes
230    *   should appear in the automata
231    */
232   public AnalyzingSuggester(Analyzer indexAnalyzer, Analyzer queryAnalyzer, int options, int maxSurfaceFormsPerAnalyzedForm, int maxGraphExpansions,
233       boolean preservePositionIncrements) {
234     this.indexAnalyzer = indexAnalyzer;
235     this.queryAnalyzer = queryAnalyzer;
236     if ((options & ~(EXACT_FIRST | PRESERVE_SEP)) != 0) {
237       throw new IllegalArgumentException("options should only contain EXACT_FIRST and PRESERVE_SEP; got " + options);
238     }
239     this.exactFirst = (options & EXACT_FIRST) != 0;
240     this.preserveSep = (options & PRESERVE_SEP) != 0;
241 
242     // NOTE: this is just an implementation limitation; if
243     // somehow this is a problem we could fix it by using
244     // more than one byte to disambiguate ... but 256 seems
245     // like it should be way more then enough.
246     if (maxSurfaceFormsPerAnalyzedForm <= 0 || maxSurfaceFormsPerAnalyzedForm > 256) {
247       throw new IllegalArgumentException("maxSurfaceFormsPerAnalyzedForm must be > 0 and < 256 (got: " + maxSurfaceFormsPerAnalyzedForm + ")");
248     }
249     this.maxSurfaceFormsPerAnalyzedForm = maxSurfaceFormsPerAnalyzedForm;
250 
251     if (maxGraphExpansions < 1 && maxGraphExpansions != -1) {
252       throw new IllegalArgumentException("maxGraphExpansions must -1 (no limit) or > 0 (got: " + maxGraphExpansions + ")");
253     }
254     this.maxGraphExpansions = maxGraphExpansions;
255     this.preservePositionIncrements = preservePositionIncrements;
256   }

273   // Replaces SEP with epsilon or remaps them if
274   // we were asked to preserve them:
275   private Automaton replaceSep(Automaton a) {
276 
277     int numStates = a.getNumStates();
278     Automaton.Builder result = new Automaton.Builder(numStates, a.getNumTransitions());
279     // Copy all states over
280     result.copyStates(a);
281 
282     // Go in reverse topo sort so we know we only have to
283     // make one pass:
284     Transition t = new Transition();
285     int[] topoSortStates = Operations.topoSortStates(a);
286     for(int i=0;i<topoSortStates.length;i++) {
287       int state = topoSortStates[topoSortStates.length-1-i];
288       int count = a.initTransition(state, t);
289       for(int j=0;j<count;j++) {
290         a.getNextTransition(t);
291         if (t.min == TokenStreamToAutomaton.POS_SEP) {
292           assert t.max == TokenStreamToAutomaton.POS_SEP;
293           if (preserveSep) {
294             // Remap to SEP_LABEL:
295             result.addTransition(state, t.dest, SEP_LABEL);
296           } else {
297             result.addEpsilon(state, t.dest);
298           }
299         } else if (t.min == TokenStreamToAutomaton.HOLE) {
300           assert t.max == TokenStreamToAutomaton.HOLE;
301 
302           // Just remove the hole: there will then be two
303           // SEP tokens next to each other, which will only
304           // match another hole at search time.  Note that
305           // it will also match an empty-string token ... if
306           // that's somehow a problem we can always map HOLE
307           // to a dedicated byte (and escape it in the
308           // input).
309           result.addEpsilon(state, t.dest);
310         } else {
311           result.addTransition(state, t.dest, t.min, t.max);
312         }
313       }
314     }
315 
316     return result.finish();
317   }

331   private static class AnalyzingComparator implements Comparator<BytesRef> {
332 
333     private final boolean hasPayloads;
334 
335     public AnalyzingComparator(boolean hasPayloads) {
336       this.hasPayloads = hasPayloads;
337     }
338 
339     private final ByteArrayDataInput readerA = new ByteArrayDataInput();
340     private final ByteArrayDataInput readerB = new ByteArrayDataInput();
341     private final BytesRef scratchA = new BytesRef();
342     private final BytesRef scratchB = new BytesRef();
343 
344     @Override
345     public int compare(BytesRef a, BytesRef b) {
346 
347       // First by analyzed form:
348       readerA.reset(a.bytes, a.offset, a.length);
349       scratchA.length = readerA.readShort();
350       scratchA.bytes = a.bytes;
351       scratchA.offset = readerA.getPosition();
352 
353       readerB.reset(b.bytes, b.offset, b.length);
354       scratchB.bytes = b.bytes;
355       scratchB.length = readerB.readShort();
356       scratchB.offset = readerB.getPosition();
357 
358       int cmp = scratchA.compareTo(scratchB);
359       if (cmp != 0) {
360         return cmp;
361       }
362       readerA.skipBytes(scratchA.length);
363       readerB.skipBytes(scratchB.length);
364 
365       // Next by cost:
366       long aCost = readerA.readInt();
367       long bCost = readerB.readInt();
368       assert decodeWeight(aCost) >= 0;
369       assert decodeWeight(bCost) >= 0;
370       if (aCost < bCost) {
371         return -1;
372       } else if (aCost > bCost) {
373         return 1;
374       }
375 
376       // Finally by surface form:
377       if (hasPayloads) {
378         scratchA.length = readerA.readShort();
379         scratchB.length = readerB.readShort();
380         scratchA.offset = readerA.getPosition();
381         scratchB.offset = readerB.getPosition();
382       } else {
383         scratchA.offset = readerA.getPosition();
384         scratchB.offset = readerB.getPosition();
385         scratchA.length = a.length - scratchA.offset;
386         scratchB.length = b.length - scratchB.offset;
387       }
388    
389       return scratchA.compareTo(scratchB);
390     }
391   }

393   @Override
394   public void build(InputIterator iterator) throws IOException {
395     if (iterator.hasContexts()) {
396       throw new IllegalArgumentException("this suggester doesn't support contexts");
397     }
398     String prefix = getClass().getSimpleName();
399     Path directory = OfflineSorter.getDefaultTempDir();
400     Path tempInput = Files.createTempFile(directory, prefix, ".input");
401     Path tempSorted = Files.createTempFile(directory, prefix, ".sorted");
402 
403     hasPayloads = iterator.hasPayloads();
404 
405     OfflineSorter.ByteSequencesWriter writer = new OfflineSorter.ByteSequencesWriter(tempInput);
406     OfflineSorter.ByteSequencesReader reader = null;
407     BytesRefBuilder scratch = new BytesRefBuilder();
408 
409     TokenStreamToAutomaton ts2a = getTokenStreamToAutomaton();
410 
411     boolean success = false;
412     count = 0;
413     byte buffer[] = new byte[8];
414     try {
415       ByteArrayDataOutput output = new ByteArrayDataOutput(buffer);
416 
417       for (BytesRef surfaceForm; (surfaceForm = iterator.next()) != null;) {
418         LimitedFiniteStringsIterator finiteStrings =
419             new LimitedFiniteStringsIterator(toAutomaton(surfaceForm, ts2a), maxGraphExpansions);
420 
421         for (IntsRef string; (string = finiteStrings.next()) != null; count++) {
422           Util.toBytesRef(string, scratch);
423           
424           // length of the analyzed text (FST input)
425           if (scratch.length() > Short.MAX_VALUE-2) {
426             throw new IllegalArgumentException("cannot handle analyzed forms > " + (Short.MAX_VALUE-2) + " in length (got " + scratch.length() + ")");
427           }
428           short analyzedLength = (short) scratch.length();
429 
430           // compute the required length:
431           // analyzed sequence + weight (4) + surface + analyzedLength (short)
432           int requiredLength = analyzedLength + 4 + surfaceForm.length + 2;
433 
434           BytesRef payload;
435 
436           if (hasPayloads) {
437             if (surfaceForm.length > (Short.MAX_VALUE-2)) {
438               throw new IllegalArgumentException("cannot handle surface form > " + (Short.MAX_VALUE-2) + " in length (got " + surfaceForm.length + ")");
439             }
440             payload = iterator.payload();
441             // payload + surfaceLength (short)
442             requiredLength += payload.length + 2;
443           } else {
444             payload = null;
445           }
446           
447           buffer = ArrayUtil.grow(buffer, requiredLength);
448           
449           output.reset(buffer);
450 
451           output.writeShort(analyzedLength);
452 
453           output.writeBytes(scratch.bytes(), 0, scratch.length());
454 
455           output.writeInt(encodeWeight(iterator.weight()));
456 
457           if (hasPayloads) {
458             for(int i=0;i<surfaceForm.length;i++) {
459               if (surfaceForm.bytes[i] == PAYLOAD_SEP) {
460                 throw new IllegalArgumentException("surface form cannot contain unit separator character U+001F; this character is reserved");
461               }
462             }
463             output.writeShort((short) surfaceForm.length);
464             output.writeBytes(surfaceForm.bytes, surfaceForm.offset, surfaceForm.length);
465             output.writeBytes(payload.bytes, payload.offset, payload.length);
466           } else {
467             output.writeBytes(surfaceForm.bytes, surfaceForm.offset, surfaceForm.length);
468           }
469 
470           assert output.getPosition() == requiredLength: output.getPosition() + " vs " + requiredLength;
471           writer.write(buffer, 0, output.getPosition());
472         }
473 
474         maxAnalyzedPathsForOneInput = Math.max(maxAnalyzedPathsForOneInput, finiteStrings.size());
475       }
476       writer.close();
477 
478       // Sort all input/output pairs (required by FST.Builder):
479       new OfflineSorter(new AnalyzingComparator(hasPayloads)).sort(tempInput, tempSorted);
480 
481       // Free disk space:
482       Files.delete(tempInput);
483 
484       reader = new OfflineSorter.ByteSequencesReader(tempSorted);
485      
486       PairOutputs<Long,BytesRef> outputs = new PairOutputs<>(PositiveIntOutputs.getSingleton(), ByteSequenceOutputs.getSingleton());
487       Builder<Pair<Long,BytesRef>> builder = new Builder<>(FST.INPUT_TYPE.BYTE1, outputs);
488 
489       // Build FST:
490       BytesRefBuilder previousAnalyzed = null;
491       BytesRefBuilder analyzed = new BytesRefBuilder();
492       BytesRef surface = new BytesRef();
493       IntsRefBuilder scratchInts = new IntsRefBuilder();
494       ByteArrayDataInput input = new ByteArrayDataInput();
495 
496       // Used to remove duplicate surface forms (but we
497       // still index the hightest-weight one).  We clear
498       // this when we see a new analyzed form, so it cannot
499       // grow unbounded (at most 256 entries):
500       Set<BytesRef> seenSurfaceForms = new HashSet<>();
501 
502       int dedup = 0;
503       while (reader.read(scratch)) {
504         input.reset(scratch.bytes(), 0, scratch.length());
505         short analyzedLength = input.readShort();
506         analyzed.grow(analyzedLength+2);
507         input.readBytes(analyzed.bytes(), 0, analyzedLength);
508         analyzed.setLength(analyzedLength);
509 
510         long cost = input.readInt();
511 
512         surface.bytes = scratch.bytes();
513         if (hasPayloads) {
514           surface.length = input.readShort();
515           surface.offset = input.getPosition();
516         } else {
517           surface.offset = input.getPosition();
518           surface.length = scratch.length() - surface.offset;
519         }
520         
521         if (previousAnalyzed == null) {
522           previousAnalyzed = new BytesRefBuilder();
523           previousAnalyzed.copyBytes(analyzed.get());
524           seenSurfaceForms.add(BytesRef.deepCopyOf(surface));
525         } else if (analyzed.get().equals(previousAnalyzed.get())) {
526           dedup++;
527           if (dedup >= maxSurfaceFormsPerAnalyzedForm) {
528             // More than maxSurfaceFormsPerAnalyzedForm
529             // dups: skip the rest:
530             continue;
531           }
532           if (seenSurfaceForms.contains(surface)) {
533             continue;
534           }
535           seenSurfaceForms.add(BytesRef.deepCopyOf(surface));
536         } else {
537           dedup = 0;
538           previousAnalyzed.copyBytes(analyzed);
539           seenSurfaceForms.clear();
540           seenSurfaceForms.add(BytesRef.deepCopyOf(surface));
541         }
542 
543         // TODO: I think we can avoid the extra 2 bytes when
544         // there is no dup (dedup==0), but we'd have to fix
545         // the exactFirst logic ... which would be sort of
546         // hairy because we'd need to special case the two
547         // (dup/not dup)...
548 
549         // NOTE: must be byte 0 so we sort before whatever
550         // is next
551         analyzed.append((byte) 0);
552         analyzed.append((byte) dedup);
553 
554         Util.toIntsRef(analyzed.get(), scratchInts);
555         //System.out.println("ADD: " + scratchInts + " -> " + cost + ": " + surface.utf8ToString());
556         if (!hasPayloads) {
557           builder.add(scratchInts.get(), outputs.newPair(cost, BytesRef.deepCopyOf(surface)));
558         } else {
559           int payloadOffset = input.getPosition() + surface.length;
560           int payloadLength = scratch.length() - payloadOffset;
561           BytesRef br = new BytesRef(surface.length + 1 + payloadLength);
562           System.arraycopy(surface.bytes, surface.offset, br.bytes, 0, surface.length);
563           br.bytes[surface.length] = PAYLOAD_SEP;
564           System.arraycopy(scratch.bytes(), payloadOffset, br.bytes, surface.length+1, payloadLength);
565           br.length = br.bytes.length;
566           builder.add(scratchInts.get(), outputs.newPair(cost, br));
567         }
568       }
569       fst = builder.finish();
570 
571       //Util.dotToFile(fst, "/tmp/suggest.dot");
572       
573       success = true;
574     } finally {
575       IOUtils.closeWhileHandlingException(reader, writer);
576       
577       if (success) {
578         IOUtils.deleteFilesIfExist(tempInput, tempSorted);
579       } else {
580         IOUtils.deleteFilesIgnoringExceptions(tempInput, tempSorted);
581       }
582     }
583   }

585   @Override
586   public boolean store(DataOutput output) throws IOException {
587     output.writeVLong(count);
588     if (fst == null) {
589       return false;
590     }
591 
592     fst.save(output);
593     output.writeVInt(maxAnalyzedPathsForOneInput);
594     output.writeByte((byte) (hasPayloads ? 1 : 0));
595     return true;
596   }

598   @Override
599   public boolean load(DataInput input) throws IOException {
600     count = input.readVLong();
601     this.fst = new FST<>(input, new PairOutputs<>(PositiveIntOutputs.getSingleton(), ByteSequenceOutputs.getSingleton()));
602     maxAnalyzedPathsForOneInput = input.readVInt();
603     hasPayloads = input.readByte() == 1;
604     return true;
605   }

607   private LookupResult getLookupResult(Long output1, BytesRef output2, CharsRefBuilder spare) {
608     LookupResult result;
609     if (hasPayloads) {
610       int sepIndex = -1;
611       for(int i=0;i<output2.length;i++) {
612         if (output2.bytes[output2.offset+i] == PAYLOAD_SEP) {
613           sepIndex = i;
614           break;
615         }
616       }
617       assert sepIndex != -1;
618       spare.grow(sepIndex);
619       final int payloadLen = output2.length - sepIndex - 1;
620       spare.copyUTF8Bytes(output2.bytes, output2.offset, sepIndex);
621       BytesRef payload = new BytesRef(payloadLen);
622       System.arraycopy(output2.bytes, sepIndex+1, payload.bytes, 0, payloadLen);
623       payload.length = payloadLen;
624       result = new LookupResult(spare.toString(), decodeWeight(output1), payload);
625     } else {
626       spare.grow(output2.length);
627       spare.copyUTF8Bytes(output2);
628       result = new LookupResult(spare.toString(), decodeWeight(output1));
629     }
630 
631     return result;
632   }

634   private boolean sameSurfaceForm(BytesRef key, BytesRef output2) {
635     if (hasPayloads) {
636       // output2 has at least PAYLOAD_SEP byte:
637       if (key.length >= output2.length) {
638         return false;
639       }
640       for(int i=0;i<key.length;i++) {
641         if (key.bytes[key.offset+i] != output2.bytes[output2.offset+i]) {
642           return false;
643         }
644       }
645       return output2.bytes[output2.offset + key.length] == PAYLOAD_SEP;
646     } else {
647       return key.bytesEquals(output2);
648     }
649   }

651   @Override
652   public List<LookupResult> lookup(final CharSequence key, Set<BytesRef> contexts, boolean onlyMorePopular, int num) {
653     assert num > 0;
654 
655     if (onlyMorePopular) {
656       throw new IllegalArgumentException("this suggester only works with onlyMorePopular=false");
657     }
658     if (contexts != null) {
659       throw new IllegalArgumentException("this suggester doesn't support contexts");
660     }
661     if (fst == null) {
662       return Collections.emptyList();
663     }
664 
665     //System.out.println("lookup key=" + key + " num=" + num);
666     for (int i = 0; i < key.length(); i++) {
667       if (key.charAt(i) == 0x1E) {
668         throw new IllegalArgumentException("lookup key cannot contain HOLE character U+001E; this character is reserved");
669       }
670       if (key.charAt(i) == 0x1F) {
671         throw new IllegalArgumentException("lookup key cannot contain unit separator character U+001F; this character is reserved");
672       }
673     }
674     final BytesRef utf8Key = new BytesRef(key);
675     try {
676       Automaton lookupAutomaton = toLookupAutomaton(key);
677 
678       final CharsRefBuilder spare = new CharsRefBuilder();
679 
680       //System.out.println("  now intersect exactFirst=" + exactFirst);
681     
682       // Intersect automaton w/ suggest wFST and get all
683       // prefix starting nodes & their outputs:
684       //final PathIntersector intersector = getPathIntersector(lookupAutomaton, fst);
685 
686       //System.out.println("  prefixPaths: " + prefixPaths.size());
687 
688       BytesReader bytesReader = fst.getBytesReader();
689 
690       FST.Arc<Pair<Long,BytesRef>> scratchArc = new FST.Arc<>();
691 
692       final List<LookupResult> results = new ArrayList<>();
693 
694       List<FSTUtil.Path<Pair<Long,BytesRef>>> prefixPaths = FSTUtil.intersectPrefixPaths(convertAutomaton(lookupAutomaton), fst);
695 
696       if (exactFirst) {
697 
698         int count = 0;
699         for (FSTUtil.Path<Pair<Long,BytesRef>> path : prefixPaths) {
700           if (fst.findTargetArc(END_BYTE, path.fstNode, scratchArc, bytesReader) != null) {
701             // This node has END_BYTE arc leaving, meaning it's an
702             // "exact" match:
703             count++;
704           }
705         }
706 
707         // Searcher just to find the single exact only
708         // match, if present:
709         Util.TopNSearcher<Pair<Long,BytesRef>> searcher;
710         searcher = new Util.TopNSearcher<>(fst, count * maxSurfaceFormsPerAnalyzedForm, count * maxSurfaceFormsPerAnalyzedForm, weightComparator);
711 
712         // NOTE: we could almost get away with only using
713         // the first start node.  The only catch is if
714         // maxSurfaceFormsPerAnalyzedForm had kicked in and
715         // pruned our exact match from one of these nodes
716         // ...:
717         for (FSTUtil.Path<Pair<Long,BytesRef>> path : prefixPaths) {
718           if (fst.findTargetArc(END_BYTE, path.fstNode, scratchArc, bytesReader) != null) {
719             // This node has END_BYTE arc leaving, meaning it's an
720             // "exact" match:
721             searcher.addStartPaths(scratchArc, fst.outputs.add(path.output, scratchArc.output), false, path.input);
722           }
723         }
724 
725         TopResults<Pair<Long,BytesRef>> completions = searcher.search();
726         assert completions.isComplete;
727 
728         // NOTE: this is rather inefficient: we enumerate
729         // every matching "exactly the same analyzed form"
730         // path, and then do linear scan to see if one of
731         // these exactly matches the input.  It should be
732         // possible (though hairy) to do something similar
733         // to getByOutput, since the surface form is encoded
734         // into the FST output, so we more efficiently hone
735         // in on the exact surface-form match.  Still, I
736         // suspect very little time is spent in this linear
737         // seach: it's bounded by how many prefix start
738         // nodes we have and the
739         // maxSurfaceFormsPerAnalyzedForm:
740         for(Result<Pair<Long,BytesRef>> completion : completions) {
741           BytesRef output2 = completion.output.output2;
742           if (sameSurfaceForm(utf8Key, output2)) {
743             results.add(getLookupResult(completion.output.output1, output2, spare));
744             break;
745           }
746         }
747 
748         if (results.size() == num) {
749           // That was quick:
750           return results;
751         }
752       }
753 
754       Util.TopNSearcher<Pair<Long,BytesRef>> searcher;
755       searcher = new Util.TopNSearcher<Pair<Long,BytesRef>>(fst,
756                                                             num - results.size(),
757                                                             num * maxAnalyzedPathsForOneInput,
758                                                             weightComparator) {
759         private final Set<BytesRef> seen = new HashSet<>();
760 
761         @Override
762         protected boolean acceptResult(IntsRef input, Pair<Long,BytesRef> output) {
763 
764           // Dedup: when the input analyzes to a graph we
765           // can get duplicate surface forms:
766           if (seen.contains(output.output2)) {
767             return false;
768           }
769           seen.add(output.output2);
770           
771           if (!exactFirst) {
772             return true;
773           } else {
774             // In exactFirst mode, don't accept any paths
775             // matching the surface form since that will
776             // create duplicate results:
777             if (sameSurfaceForm(utf8Key, output.output2)) {
778               // We found exact match, which means we should
779               // have already found it in the first search:
780               assert results.size() == 1;
781               return false;
782             } else {
783               return true;
784             }
785           }
786         }
787       };
788 
789       prefixPaths = getFullPrefixPaths(prefixPaths, lookupAutomaton, fst);
790       
791       for (FSTUtil.Path<Pair<Long,BytesRef>> path : prefixPaths) {
792         searcher.addStartPaths(path.fstNode, path.output, true, path.input);
793       }
794 
795       TopResults<Pair<Long,BytesRef>> completions = searcher.search();
796       assert completions.isComplete;
797 
798       for(Result<Pair<Long,BytesRef>> completion : completions) {
799 
800         LookupResult result = getLookupResult(completion.output.output1, completion.output.output2, spare);
801 
802         // TODO: for fuzzy case would be nice to return
803         // how many edits were required
804 
805         //System.out.println("    result=" + result);
806         results.add(result);
807 
808         if (results.size() == num) {
809           // In the exactFirst=true case the search may
810           // produce one extra path
811           break;
812         }
813       }
814 
815       return results;
816     } catch (IOException bogus) {
817       throw new RuntimeException(bogus);
818     }
819   }

834   final Automaton toAutomaton(final BytesRef surfaceForm, final TokenStreamToAutomaton ts2a) throws IOException {
835     // Analyze surface form:
836     Automaton automaton;
837     try (TokenStream ts = indexAnalyzer.tokenStream("", surfaceForm.utf8ToString())) {
838 
839       // Create corresponding automaton: labels are bytes
840       // from each analyzed token, with byte 0 used as
841       // separator between tokens:
842       automaton = ts2a.toAutomaton(ts);
843     }
844 
845     automaton = replaceSep(automaton);
846     automaton = convertAutomaton(automaton);
847 
848     // TODO: LUCENE-5660 re-enable this once we disallow massive suggestion strings
849     // assert SpecialOperations.isFinite(automaton);
850 
851     // Get all paths from the automaton (there can be
852     // more than one path, eg if the analyzer created a
853     // graph using SynFilter or WDF):
854     return automaton;
855   }

857   final Automaton toLookupAutomaton(final CharSequence key) throws IOException {
858     // TODO: is there a Reader from a CharSequence?
859     // Turn tokenstream into automaton:
860     Automaton automaton = null;
861     try (TokenStream ts = queryAnalyzer.tokenStream("", key.toString())) {
862         automaton = getTokenStreamToAutomaton().toAutomaton(ts);
863     }
864 
865     automaton = replaceSep(automaton);
866 
867     // TODO: we can optimize this somewhat by determinizing
868     // while we convert
869     automaton = Operations.determinize(automaton, DEFAULT_MAX_DETERMINIZED_STATES);
870     return automaton;

881   /** cost -&gt; weight */
882   private static int decodeWeight(long encoded) {
883     return (int)(Integer.MAX_VALUE - encoded);
884   }
885   
886   /** weight -&gt; cost */
887   private static int encodeWeight(long value) {
888     if (value < 0 || value > Integer.MAX_VALUE) {
889       throw new UnsupportedOperationException("cannot encode value: " + value);
890     }
891     return Integer.MAX_VALUE - (int)value;
892   }
