31  /**
32   * test demonstrating how overrequesting helps finds top-terms in the "long tail" 
33   * of shards that don't have even distributions of terms (something that can be common
34   * in cases of custom sharding -- even if you don't know that there is a corrolation 
35   * between the property you are sharding on and the property you are faceting on).
36   *
37   * NOTE: This test ignores the control collection (in single node mode, there is no 
38   * need for the overrequesting, all the data is local -- so comparisons with it wouldn't 
39   * be valid in the cases we are testing here)
40   */
41  public class DistributedFacetPivotLongTailTest extends BaseDistributedSearchTestCase {
42    
43    private int docNumber = 0;
44    
45    public int getDocNum() {
46      docNumber++;
47      return docNumber;
48    }
49  
50    @Test
51    @ShardsFixed(num = 3)
52    public void test() throws Exception {
53  
54      final SolrClient shard0 = clients.get(0);
55      final SolrClient shard1 = clients.get(1);
56      final SolrClient shard2 = clients.get(2);
57      
58      // the 5 top foo_s terms have 100 docs each on every shard
59      for (int i = 0; i < 100; i++) {
60        for (int j = 0; j < 5; j++) {
61          shard0.add(sdoc("id", getDocNum(), "foo_s", "aaa"+j, "stat_i", j * 13 - i));
62          shard1.add(sdoc("id", getDocNum(), "foo_s", "aaa"+j, "stat_i", j * 3 + i));
63          shard2.add(sdoc("id", getDocNum(), "foo_s", "aaa"+j, "stat_i", i * 7 + j));
64        }
65      }
66  
67      // 20 foo_s terms that come in "second" with 50 docs each 
68      // on both shard0 & shard1 ("bbb_")
69      for (int i = 0; i < 50; i++) {
70        for (int j = 0; j < 20; j++) {
71          shard0.add(sdoc("id", getDocNum(), "foo_s", "bbb"+j, "stat_i", 0));
72          shard1.add(sdoc("id", getDocNum(), "foo_s", "bbb"+j, "stat_i", 1));
73        }
74        // distracting term appears on only on shard2 50 times
75        shard2.add(sdoc("id", getDocNum(), "foo_s", "junkA"));
76      }
77      // put "bbb0" on shard2 exactly once to sanity check refinement
78      shard2.add(sdoc("id", getDocNum(), "foo_s", "bbb0", "stat_i", -2));
79  
80      // long 'tail' foo_s term appears in 45 docs on every shard
81      // foo_s:tail is the only term with bar_s sub-pivot terms
82      for (int i = 0; i < 45; i++) {
83  
84        // for sub-pivot, shard0 & shard1 have 6 docs each for "tailB"
85        // but the top 5 terms are ccc(0-4) -- 7 on each shard
86        // (4 docs each have junk terms)
87        String sub_term = (i < 35) ? "ccc"+(i % 5) : ((i < 41) ? "tailB" : "junkA");
88        shard0.add(sdoc("id", getDocNum(), "foo_s", "tail", "bar_s", sub_term, "stat_i", i));
89        shard1.add(sdoc("id", getDocNum(), "foo_s", "tail", "bar_s", sub_term, "stat_i", i));
90  
91        // shard2's top 5 sub-pivot terms are junk only it has with 8 docs each
92        // and 5 docs that use "tailB"
93        // NOTE: none of these get stat_i ! !
94        sub_term = (i < 40) ? "junkB"+(i % 5) : "tailB";
95        shard2.add(sdoc("id", getDocNum(), "foo_s", "tail", "bar_s", sub_term));
96      }
97  
98      // really long tail uncommon foo_s terms on shard2
99      for (int i = 0; i < 30; i++) {
100       shard2.add(sdoc("id", getDocNum(), "foo_s", "zzz"+i));
101     }
102 
103     commit();
104 
105     SolrParams req = params( "q", "*:*", 
106                              "distrib", "false",
107                              "facet", "true", 
108                              "facet.limit", "10",
109                              "facet.pivot", "foo_s,bar_s");
110 
111     // sanity check that our expectations about each shard (non-distrib) are correct
112 
113     PivotField pivot = null;
114     List<PivotField> pivots = null;
115     List<PivotField>[] shardPivots = new List[3];
116     shardPivots[0] = shard0.query( req ).getFacetPivot().get("foo_s,bar_s");
117     shardPivots[1] = shard1.query( req ).getFacetPivot().get("foo_s,bar_s");
118     shardPivots[2] = shard2.query( req ).getFacetPivot().get("foo_s,bar_s");
119 
120     // top 5 same on all shards
121     for (int i = 0; i < 3; i++) {
122       assertEquals(10, shardPivots[i].size());
123       for (int j = 0; j < 5; j++) {
124         pivot = shardPivots[i].get(j);
125         assertEquals(pivot.toString(), "aaa"+j, pivot.getValue());
126         assertEquals(pivot.toString(), 100, pivot.getCount());
127       }
128     }
129     // top 6-10 same on shard0 & shard11
130     for (int i = 0; i < 2; i++) {
131       for (int j = 5; j < 10; j++) {
132         pivot = shardPivots[i].get(j);
133         assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("bbb"));
134         assertEquals(pivot.toString(), 50, pivot.getCount());
135       }
136     }
137     // 6-10 on shard2
138     assertEquals("junkA", shardPivots[2].get(5).getValue());
139     assertEquals(50, shardPivots[2].get(5).getCount());
140     assertEquals("tail", shardPivots[2].get(6).getValue());
141     assertEquals(45, shardPivots[2].get(6).getCount());
142     assertEquals("bbb0", shardPivots[2].get(7).getValue());
143     assertEquals(1, shardPivots[2].get(7).getCount());
144     for (int j = 8; j < 10; j++) {
145       pivot = shardPivots[2].get(j);
146       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("zzz"));
147       assertEquals(pivot.toString(), 1, pivot.getCount());
148     }
149     // check sub-shardPivots on "tail" from shard2
150     pivots = shardPivots[2].get(6).getPivot();
151     assertEquals(6, pivots.size());
152     for (int j = 0; j < 5; j++) {
153       pivot = pivots.get(j);
154       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("junkB"));
155       assertEquals(pivot.toString(), 8, pivot.getCount());
156     }
157     pivot = pivots.get(5);
158     assertEquals("tailB", pivot.getValue());
159     assertEquals(5, pivot.getCount());
160 
161     // if we disable overrequesting, we don't find the long tail
162 
163     pivots = queryServer( params( "q", "*:*",
164                                   "shards", getShardsString(),
165                                   FacetParams.FACET_OVERREQUEST_COUNT, "0",
166                                   FacetParams.FACET_OVERREQUEST_RATIO, "0",
167                                   "facet", "true",
168                                   "facet.limit", "6",
169                                   "facet.pivot", "{!stats=sxy}foo_s,bar_s",
170                                   "stats", "true",
171                                   "stats.field", "{!tag=sxy}stat_i")
172                           ).getFacetPivot().get("foo_s,bar_s");
173     assertEquals(6, pivots.size());
174     for (int i = 0; i < 5; i++) {
175       pivot = pivots.get(i);
176       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("aaa"));
177       assertEquals(pivot.toString(), 300, pivot.getCount());
178     }
179     { // even w/o the long tail, we should have still asked shard2 to refine bbb0
180       pivot = pivots.get(5);
181       assertTrue(pivot.toString(), pivot.getValue().equals("bbb0"));
182       assertEquals(pivot.toString(), 101, pivot.getCount());
183       // basic check of refined stats
184       FieldStatsInfo bbb0Stats = pivot.getFieldStatsInfo().get("stat_i");
185       assertEquals("stat_i", bbb0Stats.getName());
186       assertEquals(-2.0, bbb0Stats.getMin());
187       assertEquals(1.0, bbb0Stats.getMax());
188       assertEquals(101, (long) bbb0Stats.getCount());
189       assertEquals(0, (long) bbb0Stats.getMissing());
190       assertEquals(48.0, bbb0Stats.getSum());
191       assertEquals(0.475247524752475, (double) bbb0Stats.getMean(), 0.1E-7);
192       assertEquals(54.0, bbb0Stats.getSumOfSquares(), 0.1E-7);
193       assertEquals(0.55846323792, bbb0Stats.getStddev(), 0.1E-7);
194     }
195 
196 
197     // with default overrequesting, we should find the correct top 6 including 
198     // long tail and top sub-pivots
199     // (even if we disable overrequesting on the sub-pivot)
200     for (ModifiableSolrParams q : new ModifiableSolrParams[] { 
201         params(),
202         params("f.bar_s.facet.overrequest.ratio","0",
203                "f.bar_s.facet.overrequest.count","0")      }) {
204       
205       q.add( params( "q", "*:*",
206                      "shards", getShardsString(),
207                      "facet", "true",
208                      "facet.limit", "6",
209                      "facet.pivot", "foo_s,bar_s" ));
210       pivots = queryServer( q ).getFacetPivot().get("foo_s,bar_s");
211         
212       assertEquals(6, pivots.size());
213       for (int i = 0; i < 5; i++) {
214         pivot = pivots.get(i);
215         assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("aaa"));
216         assertEquals(pivot.toString(), 300, pivot.getCount());
217       }
218       pivot = pivots.get(5);
219       assertEquals(pivot.toString(), "tail", pivot.getValue());
220       assertEquals(pivot.toString(), 135, pivot.getCount());
221       // check the sub pivots
222       pivots = pivot.getPivot();
223       assertEquals(6, pivots.size());
224       pivot = pivots.get(0);
225       assertEquals(pivot.toString(), "tailB", pivot.getValue());
226       assertEquals(pivot.toString(), 17, pivot.getCount());
227       for (int i = 1; i < 6; i++) { // ccc(0-4)
228         pivot = pivots.get(i);
229         assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("ccc"));
230         assertEquals(pivot.toString(), 14, pivot.getCount());
231       }
232     }
233 
234     // if we lower the facet.limit on the sub-pivot, overrequesting should still ensure 
235     // that we get the correct top5 including "tailB"
236 
237     pivots = queryServer( params( "q", "*:*",
238                                   "shards", getShardsString(),
239                                   "facet", "true",
240                                   "facet.limit", "6",
241                                   "f.bar_s.facet.limit", "5",
242                                   "facet.pivot", "foo_s,bar_s" )
243                           ).getFacetPivot().get("foo_s,bar_s");
244     assertEquals(6, pivots.size());
245     for (int i = 0; i < 5; i++) {
246       pivot = pivots.get(i);
247       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("aaa"));
248       assertEquals(pivot.toString(), 300, pivot.getCount());
249     }
250     pivot = pivots.get(5);
251     assertEquals(pivot.toString(), "tail", pivot.getValue());
252     assertEquals(pivot.toString(), 135, pivot.getCount());
253     // check the sub pivots
254     pivots = pivot.getPivot();
255     assertEquals(5, pivots.size());
256     pivot = pivots.get(0);
257     assertEquals(pivot.toString(), "tailB", pivot.getValue());
258     assertEquals(pivot.toString(), 17, pivot.getCount());
259     for (int i = 1; i < 5; i++) { // ccc(0-3)
260       pivot = pivots.get(i);
261       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("ccc"));
262       assertEquals(pivot.toString(), 14, pivot.getCount());
263     }
264 
265     // however with a lower limit and overrequesting disabled, 
266     // we're going to miss out on tailB
267 
268     pivots = queryServer( params( "q", "*:*",
269                                   "shards", getShardsString(),
270                                   "facet", "true",
271                                   "facet.limit", "6",
272                                   "f.bar_s.facet.overrequest.ratio", "0",
273                                   "f.bar_s.facet.overrequest.count", "0",
274                                   "f.bar_s.facet.limit", "5",
275                                   "facet.pivot", "foo_s,bar_s" )
276                           ).getFacetPivot().get("foo_s,bar_s");
277     assertEquals(6, pivots.size());
278     for (int i = 0; i < 5; i++) {
279       pivot = pivots.get(i);
280       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("aaa"));
281       assertEquals(pivot.toString(), 300, pivot.getCount());
282     }
283     pivot = pivots.get(5);
284     assertEquals(pivot.toString(), "tail", pivot.getValue());
285     assertEquals(pivot.toString(), 135, pivot.getCount());
286     // check the sub pivots
287     pivots = pivot.getPivot();
288     assertEquals(5, pivots.size());
289     for (int i = 0; i < 5; i++) { // ccc(0-4)
290       pivot = pivots.get(i);
291       assertTrue(pivot.toString(), pivot.getValue().toString().startsWith("ccc"));
292       assertEquals(pivot.toString(), 14, pivot.getCount());
293     }
294     
295     doTestDeepPivotStats();
296   }

298   public void doTestDeepPivotStats() throws Exception {
299     // Deep checking of some Facet stats - no refinement involved here
300 
301     List<PivotField> pivots = 
302       query("q", "*:*",
303             "shards", getShardsString(),
304             "facet", "true",
305             "rows" , "0",
306             "facet.pivot","{!stats=s1}foo_s,bar_s",
307             "stats", "true",
308             "stats.field", "{!key=avg_price tag=s1}stat_i").getFacetPivot().get("foo_s,bar_s");
309     PivotField aaa0PivotField = pivots.get(0);
310     assertEquals("aaa0", aaa0PivotField.getValue());
311     assertEquals(300, aaa0PivotField.getCount());
312 
313     FieldStatsInfo aaa0StatsInfo = aaa0PivotField.getFieldStatsInfo().get("avg_price");
314     assertEquals("avg_price", aaa0StatsInfo.getName());
315     assertEquals(-99.0, aaa0StatsInfo.getMin());
316     assertEquals(693.0, aaa0StatsInfo.getMax());
317     assertEquals(300, (long) aaa0StatsInfo.getCount());
318     assertEquals(0, (long) aaa0StatsInfo.getMissing());
319     assertEquals(34650.0, aaa0StatsInfo.getSum());
320     assertEquals(1.674585E7, aaa0StatsInfo.getSumOfSquares(), 0.1E-7);
321     assertEquals(115.5, (double) aaa0StatsInfo.getMean(), 0.1E-7);
322     assertEquals(206.4493184076, aaa0StatsInfo.getStddev(), 0.1E-7);
323 
324     PivotField tailPivotField = pivots.get(5);
325     assertEquals("tail", tailPivotField.getValue());
326     assertEquals(135, tailPivotField.getCount());
327 
328     FieldStatsInfo tailPivotFieldStatsInfo = tailPivotField.getFieldStatsInfo().get("avg_price");
329     assertEquals("avg_price", tailPivotFieldStatsInfo.getName());
330     assertEquals(0.0, tailPivotFieldStatsInfo.getMin());
331     assertEquals(44.0, tailPivotFieldStatsInfo.getMax());
332     assertEquals(90, (long) tailPivotFieldStatsInfo.getCount());
333     assertEquals(45, (long) tailPivotFieldStatsInfo.getMissing());
334     assertEquals(1980.0, tailPivotFieldStatsInfo.getSum());
335     assertEquals(22.0, (double) tailPivotFieldStatsInfo.getMean(), 0.1E-7);
336     assertEquals(58740.0, tailPivotFieldStatsInfo.getSumOfSquares(), 0.1E-7);
337     assertEquals(13.0599310011, tailPivotFieldStatsInfo.getStddev(), 0.1E-7);
338 
339     PivotField tailBPivotField = tailPivotField.getPivot().get(0);
340     assertEquals("tailB", tailBPivotField.getValue());
341     assertEquals(17, tailBPivotField.getCount());
342 
343     FieldStatsInfo tailBPivotFieldStatsInfo = tailBPivotField.getFieldStatsInfo().get("avg_price");
344     assertEquals("avg_price", tailBPivotFieldStatsInfo.getName());
345     assertEquals(35.0, tailBPivotFieldStatsInfo.getMin());
346     assertEquals(40.0, tailBPivotFieldStatsInfo.getMax());
347     assertEquals(12, (long) tailBPivotFieldStatsInfo.getCount());
348     assertEquals(5, (long) tailBPivotFieldStatsInfo.getMissing());
349     assertEquals(450.0, tailBPivotFieldStatsInfo.getSum());
350     assertEquals(37.5, (double) tailBPivotFieldStatsInfo.getMean(), 0.1E-7);
351     assertEquals(16910.0, tailBPivotFieldStatsInfo.getSumOfSquares(), 0.1E-7);
352     assertEquals(1.78376517, tailBPivotFieldStatsInfo.getStddev(), 0.1E-7);
353   }
