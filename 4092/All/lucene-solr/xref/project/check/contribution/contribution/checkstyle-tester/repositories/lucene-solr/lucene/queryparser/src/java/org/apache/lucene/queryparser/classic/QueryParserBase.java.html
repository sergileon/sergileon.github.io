110   /** Parses a query string, returning a {@link org.apache.lucene.search.Query}.
111    *  @param query  the query string to be parsed.
112    *  @throws ParseException if the parsing fails
113    */
114   public Query parse(String query) throws ParseException {
115     ReInit(new FastCharStream(new StringReader(query)));
116     try {
117       // TopLevelQuery is a Query followed by the end-of-input (EOF)
118       Query res = TopLevelQuery(field);
119       return res!=null ? res : newBooleanQuery(false).build();
120     }
121     catch (ParseException | TokenMgrError tme) {
122       // rethrow to include the original query:
123       ParseException e = new ParseException("Cannot parse '" +query+ "': " + tme.getMessage());
124       e.initCause(tme);
125       throw e;
126     } catch (BooleanQuery.TooManyClauses tmc) {
127       ParseException e = new ParseException("Cannot parse '" +query+ "': too many boolean clauses");
128       e.initCause(tmc);
129       throw e;
130     }
131   }

357   /**
358    * Returns the date resolution that is used by RangeQueries for the given field.
359    * Returns null, if no default or field specific date resolution has been set
360    * for the given field.
361    *
362    */
363   public DateTools.Resolution getDateResolution(String fieldName) {
364     if (fieldName == null) {
365       throw new IllegalArgumentException("Field cannot be null.");
366     }
367 
368     if (fieldToDateResolution == null) {
369       // no field specific date resolutions set; return default date resolution instead
370       return this.dateResolution;
371     }
372 
373     DateTools.Resolution resolution = fieldToDateResolution.get(fieldName);
374     if (resolution == null) {
375       // no date resolutions set for the given field; return default date resolution instead
376       resolution = this.dateResolution;
377     }
378 
379     return resolution;
380   }

418   protected void addClause(List<BooleanClause> clauses, int conj, int mods, Query q) {
419     boolean required, prohibited;
420 
421     // If this term is introduced by AND, make the preceding term required,
422     // unless it's already prohibited
423     if (clauses.size() > 0 && conj == CONJ_AND) {
424       BooleanClause c = clauses.get(clauses.size()-1);
425       if (!c.isProhibited())
426         clauses.set(clauses.size() - 1, new BooleanClause(c.getQuery(), Occur.MUST));
427     }
428 
429     if (clauses.size() > 0 && operator == AND_OPERATOR && conj == CONJ_OR) {
430       // If this term is introduced by OR, make the preceding term optional,
431       // unless it's prohibited (that means we leave -a OR b but +a OR b-->a OR b)
432       // notice if the input is a OR b, first term is parsed as required; without
433       // this modification a OR b would parsed as +a OR b
434       BooleanClause c = clauses.get(clauses.size()-1);
435       if (!c.isProhibited())
436         clauses.set(clauses.size() - 1, new BooleanClause(c.getQuery(), Occur.SHOULD));
437     }
438 
439     // We might have been passed a null query; the term might have been
440     // filtered away by the analyzer.
441     if (q == null)
442       return;
443 
444     if (operator == OR_OPERATOR) {
445       // We set REQUIRED if we're introduced by AND or +; PROHIBITED if
446       // introduced by NOT or -; make sure not to set both.
447       prohibited = (mods == MOD_NOT);
448       required = (mods == MOD_REQ);
449       if (conj == CONJ_AND && !prohibited) {
450         required = true;
451       }
452     } else {
453       // We set PROHIBITED if we're introduced by NOT or -; We set REQUIRED
454       // if not PROHIBITED and not introduced by OR
455       prohibited = (mods == MOD_NOT);
456       required   = (!prohibited && conj != CONJ_OR);
457     }
458     if (required && !prohibited)
459       clauses.add(newBooleanClause(q, BooleanClause.Occur.MUST));
460     else if (!required && !prohibited)
461       clauses.add(newBooleanClause(q, BooleanClause.Occur.SHOULD));
462     else if (!required && prohibited)
463       clauses.add(newBooleanClause(q, BooleanClause.Occur.MUST_NOT));
464     else
465       throw new RuntimeException("Clause cannot be both required and prohibited");
466   }

485   /**
486    * Base implementation delegates to {@link #getFieldQuery(String,String,boolean)}.
487    * This method may be overridden, for example, to return
488    * a SpanNearQuery instead of a PhraseQuery.
489    *
490    * @exception org.apache.lucene.queryparser.classic.ParseException throw in overridden method to disallow
491    */
492   protected Query getFieldQuery(String field, String queryText, int slop)
493         throws ParseException {
494     Query query = getFieldQuery(field, queryText, true);
495 
496     if (query instanceof PhraseQuery) {
497       PhraseQuery.Builder builder = new PhraseQuery.Builder();
498       builder.setSlop(slop);
499       PhraseQuery pq = (PhraseQuery) query;
500       org.apache.lucene.index.Term[] terms = pq.getTerms();
501       int[] positions = pq.getPositions();
502       for (int i = 0; i < terms.length; ++i) {
503         builder.add(terms[i], positions[i]);
504       }
505       query = builder.build();
506     }
507     if (query instanceof MultiPhraseQuery) {
508       ((MultiPhraseQuery) query).setSlop(slop);
509     }
510 
511     return query;
512   }

514   protected Query getRangeQuery(String field,
515                                 String part1,
516                                 String part2,
517                                 boolean startInclusive,
518                                 boolean endInclusive) throws ParseException
519   {
520     if (lowercaseExpandedTerms) {
521       part1 = part1==null ? null : part1.toLowerCase(locale);
522       part2 = part2==null ? null : part2.toLowerCase(locale);
523     }
524 
525 
526     DateFormat df = DateFormat.getDateInstance(DateFormat.SHORT, locale);
527     df.setLenient(true);
528     DateTools.Resolution resolution = getDateResolution(field);
529     
530     try {
531       part1 = DateTools.dateToString(df.parse(part1), resolution);
532     } catch (Exception e) { }
533 
534     try {
535       Date d2 = df.parse(part2);
536       if (endInclusive) {
537         // The user can only specify the date, not the time, so make sure
538         // the time is set to the latest possible time of that date to really
539         // include all documents:
540         Calendar cal = Calendar.getInstance(timeZone, locale);
541         cal.setTime(d2);
542         cal.set(Calendar.HOUR_OF_DAY, 23);
543         cal.set(Calendar.MINUTE, 59);
544         cal.set(Calendar.SECOND, 59);
545         cal.set(Calendar.MILLISECOND, 999);
546         d2 = cal.getTime();
547       }
548       part2 = DateTools.dateToString(d2, resolution);
549     } catch (Exception e) { }
550 
551     return newRangeQuery(field, part1, part2, startInclusive, endInclusive);
552   }

603   private BytesRef analyzeMultitermTerm(String field, String part) {
604     return analyzeMultitermTerm(field, part, getAnalyzer());
605   }

607   protected BytesRef analyzeMultitermTerm(String field, String part, Analyzer analyzerIn) {
608     if (analyzerIn == null) analyzerIn = getAnalyzer();
609 
610     try (TokenStream source = analyzerIn.tokenStream(field, part)) {
611       source.reset();
612       
613       TermToBytesRefAttribute termAtt = source.getAttribute(TermToBytesRefAttribute.class);
614 
615       if (!source.incrementToken())
616         throw new IllegalArgumentException("analyzer returned no terms for multiTerm term: " + part);
617       BytesRef bytes = BytesRef.deepCopyOf(termAtt.getBytesRef());
618       if (source.incrementToken())
619         throw new IllegalArgumentException("analyzer returned too many terms for multiTerm term: " + part);
620       source.end();
621       return bytes;
622     } catch (IOException e) {
623       throw new RuntimeException("Error analyzing multiTerm term: " + part, e);
624     }
625   }

627   /**
628    * Builds a new {@link TermRangeQuery} instance
629    * @param field Field
630    * @param part1 min
631    * @param part2 max
632    * @param startInclusive true if the start of the range is inclusive
633    * @param endInclusive true if the end of the range is inclusive
634    * @return new {@link TermRangeQuery} instance
635    */
636   protected Query newRangeQuery(String field, String part1, String part2, boolean startInclusive, boolean endInclusive) {
637     final BytesRef start;
638     final BytesRef end;
639      
640     if (part1 == null) {
641       start = null;
642     } else {
643       start = analyzeRangeTerms ? analyzeMultitermTerm(field, part1) : new BytesRef(part1);
644     }
645      
646     if (part2 == null) {
647       end = null;
648     } else {
649       end = analyzeRangeTerms ? analyzeMultitermTerm(field, part2) : new BytesRef(part2);
650     }
651       
652     final TermRangeQuery query = new TermRangeQuery(field, start, end, startInclusive, endInclusive);
653 
654     query.setRewriteMethod(multiTermRewriteMethod);
655     return query;
656   }

698    * Can be overridden by extending classes, to modify query being
699    * returned.

701    * @param clauses List that contains {@link org.apache.lucene.search.BooleanClause} instances
702    *    to join.
703    * @param disableCoord true if coord scoring should be disabled.
704    *
705    * @return Resulting {@link org.apache.lucene.search.Query} object.
706    * @exception org.apache.lucene.queryparser.classic.ParseException throw in overridden method to disallow
707    */
708   protected Query getBooleanQuery(List<BooleanClause> clauses, boolean disableCoord)
709     throws ParseException
710   {
711     if (clauses.size()==0) {
712       return null; // all clause words were filtered away by the analyzer.
713     }
714     BooleanQuery.Builder query = newBooleanQuery(disableCoord);
715     for(final BooleanClause clause: clauses) {
716       query.add(clause);
717     }
718     return query.build();

721   /**
722    * Factory method for generating a query. Called when parser
723    * parses an input term token that contains one or more wildcard
724    * characters (? and *), but is not a prefix term token (one
725    * that has just a single * character at the end)
726    *<p>
727    * Depending on settings, prefix term may be lower-cased
728    * automatically. It will not go through the default Analyzer,
729    * however, since normal Analyzers are unlikely to work properly
730    * with wildcard templates.
731    *<p>
732    * Can be overridden by extending classes, to provide custom handling for
733    * wildcard queries, which may be necessary due to missing analyzer calls.
734    *
735    * @param field Name of the field query will use.
736    * @param termStr Term token that contains one or more wild card
737    *   characters (? or *), but is not simple prefix term
738    *
739    * @return Resulting {@link org.apache.lucene.search.Query} built for the term
740    * @exception org.apache.lucene.queryparser.classic.ParseException throw in overridden method to disallow
741    */
742   protected Query getWildcardQuery(String field, String termStr) throws ParseException
743   {
744     if ("*".equals(field)) {
745       if ("*".equals(termStr)) return newMatchAllDocsQuery();
746     }
747     if (!allowLeadingWildcard && (termStr.startsWith("*") || termStr.startsWith("?")))
748       throw new ParseException("'*' or '?' not allowed as first character in WildcardQuery");
749     if (lowercaseExpandedTerms) {
750       termStr = termStr.toLowerCase(locale);
751     }
752     Term t = new Term(field, termStr);
753     return newWildcardQuery(t);
754   }

912   /**
913    * Returns a String where the escape char has been
914    * removed, or kept only once if there was a double escape.
915    *
916    * Supports escaped unicode characters, e. g. translates
917    * <code>\\u0041</code> to <code>A</code>.
918    *
919    */
920   String discardEscapeChar(String input) throws ParseException {
921     // Create char array to hold unescaped char sequence
922     char[] output = new char[input.length()];
923 
924     // The length of the output can be less than the input
925     // due to discarded escape chars. This variable holds
926     // the actual length of the output
927     int length = 0;
928 
929     // We remember whether the last processed character was
930     // an escape character
931     boolean lastCharWasEscapeChar = false;
932 
933     // The multiplier the current unicode digit must be multiplied with.
934     // E. g. the first digit must be multiplied with 16^3, the second with 16^2...
935     int codePointMultiplier = 0;
936 
937     // Used to calculate the codepoint of the escaped unicode character
938     int codePoint = 0;
939 
940     for (int i = 0; i < input.length(); i++) {
941       char curChar = input.charAt(i);
942       if (codePointMultiplier > 0) {
943         codePoint += hexToInt(curChar) * codePointMultiplier;
944         codePointMultiplier >>>= 4;
945         if (codePointMultiplier == 0) {
946           output[length++] = (char)codePoint;
947           codePoint = 0;
948         }
949       } else if (lastCharWasEscapeChar) {
950         if (curChar == 'u') {
951           // found an escaped unicode character
952           codePointMultiplier = 16 * 16 * 16;
953         } else {
954           // this character was escaped
955           output[length] = curChar;
956           length++;
957         }
958         lastCharWasEscapeChar = false;
959       } else {
960         if (curChar == '\\') {
961           lastCharWasEscapeChar = true;
962         } else {
963           output[length] = curChar;
964           length++;
965         }
966       }
967     }
968 
969     if (codePointMultiplier > 0) {
970       throw new ParseException("Truncated unicode escape sequence.");
971     }
972 
973     if (lastCharWasEscapeChar) {
974       throw new ParseException("Term can not end with escape character.");
975     }
976 
977     return new String(output, 0, length);
978   }

980   /** Returns the numeric value of the hexadecimal character */
981   static final int hexToInt(char c) throws ParseException {
982     if ('0' <= c && c <= '9') {
983       return c - '0';
984     } else if ('a' <= c && c <= 'f'){
985       return c - 'a' + 10;
986     } else if ('A' <= c && c <= 'F') {
987       return c - 'A' + 10;
988     } else {
989       throw new ParseException("Non-hex character in Unicode escape sequence: " + c);
990     }
991   }

993   /**
994    * Returns a String where those characters that QueryParser
995    * expects to be escaped are escaped by a preceding <code>\</code>.
996    */
997   public static String escape(String s) {
998     StringBuilder sb = new StringBuilder();
999     for (int i = 0; i < s.length(); i++) {
1000       char c = s.charAt(i);
1001       // These characters are part of the query syntax and must be escaped
1002       if (c == '\\' || c == '+' || c == '-' || c == '!' || c == '(' || c == ')' || c == ':'
1003         || c == '^' || c == '[' || c == ']' || c == '\"' || c == '{' || c == '}' || c == '~'
1004         || c == '*' || c == '?' || c == '|' || c == '&' || c == '/') {
1005         sb.append('\\');
1006       }
1007       sb.append(c);
1008     }
1009     return sb.toString();
1010   }
