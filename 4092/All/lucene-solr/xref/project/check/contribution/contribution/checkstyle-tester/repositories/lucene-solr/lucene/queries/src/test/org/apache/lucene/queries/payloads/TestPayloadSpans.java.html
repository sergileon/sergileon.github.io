258   public void testShrinkToAfterShortestMatch() throws IOException {
259     Directory directory = newDirectory();
260     RandomIndexWriter writer = new RandomIndexWriter(random(), directory,
261                                                      newIndexWriterConfig(new TestPayloadAnalyzer()));
262 
263     Document doc = new Document();
264     doc.add(new TextField("content", new StringReader("a b c d e f g h i j a k")));
265     writer.addDocument(doc);
266 
267     IndexReader reader = writer.getReader();
268     IndexSearcher is = newSearcher(reader);
269     writer.close();
270 
271     SpanTermQuery stq1 = new SpanTermQuery(new Term("content", "a"));
272     SpanTermQuery stq2 = new SpanTermQuery(new Term("content", "k"));
273     SpanQuery[] sqs = { stq1, stq2 };
274     SpanNearQuery snq = new SpanNearQuery(sqs, 1, true);
275     VerifyingCollector collector = new VerifyingCollector();
276     Spans spans = MultiSpansWrapper.wrap(is.getIndexReader(), snq, SpanWeight.Postings.PAYLOADS);
277 
278     TopDocs topDocs = is.search(snq, 1);
279     Set<String> payloadSet = new HashSet<>();
280     for (int i = 0; i < topDocs.scoreDocs.length; i++) {
281       while (spans.nextDoc() != Spans.NO_MORE_DOCS) {
282         while (spans.nextStartPosition() != Spans.NO_MORE_POSITIONS) {
283           collector.reset();
284           spans.collect(collector);
285           for (final BytesRef payload : collector.payloads) {
286             payloadSet.add(Term.toString(payload));
287           }
288         }
289       }
290     }
291     assertEquals(2, payloadSet.size());
292     assertTrue(payloadSet.contains("a:Noise:10"));
293     assertTrue(payloadSet.contains("k:Noise:11"));
294     reader.close();
295     directory.close();

381   static class VerifyingCollector implements SpanCollector {
382 
383     List<BytesRef> payloads = new ArrayList<>();
384 
385     @Override
386     public void collectLeaf(PostingsEnum postings, int position, Term term) throws IOException {
387       if (postings.getPayload() != null) {
388         payloads.add(BytesRef.deepCopyOf(postings.getPayload()));
389       }
390     }
391 
392     @Override
393     public void reset() {
394       payloads.clear();
395     }
396 
397     public void verify(int expectedLength, int expectedFirstByte) {
398       for (BytesRef payload : payloads) {
399         assertEquals("Incorrect payload length", expectedLength, payload.length);
400         assertEquals("Incorrect first byte", expectedFirstByte, payload.bytes[0]);
401       }
402     }
403   }

405   private void checkSpans(Spans spans, int expectedNumSpans, int expectedNumPayloads,
406                           int expectedPayloadLength, int expectedFirstByte) throws IOException {
407     assertTrue("spans is null and it shouldn't be", spans != null);
408     //each position match should have a span associated with it, since there is just one underlying term query, there should
409     //only be one entry in the span
410     VerifyingCollector collector = new VerifyingCollector();
411     int seen = 0;
412     while (spans.nextDoc() != Spans.NO_MORE_DOCS) {
413       while (spans.nextStartPosition() != Spans.NO_MORE_POSITIONS) {
414         collector.reset();
415         spans.collect(collector);
416         collector.verify(expectedPayloadLength, expectedFirstByte);
417         assertEquals("expectedNumPayloads", expectedNumPayloads, collector.payloads.size());
418         seen++;
419       }
420     }
421     assertEquals("expectedNumSpans", expectedNumSpans, seen);
422   }

446   private void checkSpans(Spans spans, int numSpans, int[] numPayloads) throws IOException {
447     int cnt = 0;
448     VerifyingCollector collector = new VerifyingCollector();
449     while (spans.nextDoc() != Spans.NO_MORE_DOCS) {
450       while (spans.nextStartPosition() != Spans.NO_MORE_POSITIONS) {
451         if(VERBOSE)
452           System.out.println("\nSpans Dump --");
453         collector.reset();
454         spans.collect(collector);
455         assertEquals("payload size", numPayloads[cnt], collector.payloads.size());
456 
457         cnt++;
458       }
459     }
460 
461     assertEquals("expected numSpans", numSpans, cnt);
462   }

466     @Override

469       return new TokenStreamComponents(result, new PayloadFilter(result));

473   final class PayloadFilter extends TokenFilter {
474     Set<String> entities = new HashSet<>();
475     Set<String> nopayload = new HashSet<>();
476     int pos;
477     PayloadAttribute payloadAtt;
478     CharTermAttribute termAtt;
479     PositionIncrementAttribute posIncrAtt;
480 
481     public PayloadFilter(TokenStream input) {
482       super(input);
483       pos = 0;
484       entities.add("xx");
485       entities.add("one");
486       nopayload.add("nopayload");
487       nopayload.add("np");
488       termAtt = addAttribute(CharTermAttribute.class);
489       posIncrAtt = addAttribute(PositionIncrementAttribute.class);
490       payloadAtt = addAttribute(PayloadAttribute.class);
491     }
492 
493     @Override
494     public boolean incrementToken() throws IOException {
495       if (input.incrementToken()) {
496         String token = termAtt.toString();
497 
498         if (!nopayload.contains(token)) {
499           if (entities.contains(token)) {
500             payloadAtt.setPayload(new BytesRef(token + ":Entity:"+ pos ));
501           } else {
502             payloadAtt.setPayload(new BytesRef(token + ":Noise:" + pos ));
503           }
504         }
505         pos += posIncrAtt.getPositionIncrement();
506         return true;
507       }
508       return false;
509     }
510 
511     @Override
512     public void reset() throws IOException {
513       super.reset();
514       this.pos = 0;
515     }
516   }
