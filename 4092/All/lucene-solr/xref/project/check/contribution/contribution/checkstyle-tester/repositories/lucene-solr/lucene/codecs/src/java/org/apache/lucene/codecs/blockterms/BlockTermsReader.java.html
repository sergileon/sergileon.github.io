53   *  doc/freqs/positions reading to an instance of {@link
54   *  PostingsReaderBase}.  This class is reusable for

57   *  make their own terms dict impl).

108   public BlockTermsReader(TermsIndexReaderBase indexReader, PostingsReaderBase postingsReader, SegmentReadState state) throws IOException {
109     
110     this.postingsReader = postingsReader;
111     
112     String filename = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, BlockTermsWriter.TERMS_EXTENSION);
113     in = state.directory.openInput(filename, state.context);
114 
115     boolean success = false;
116     try {
117       CodecUtil.checkIndexHeader(in, BlockTermsWriter.CODEC_NAME, 
118                                        BlockTermsWriter.VERSION_START,
119                                        BlockTermsWriter.VERSION_CURRENT,
120                                        state.segmentInfo.getId(), state.segmentSuffix);
121 
122       // Have PostingsReader init itself
123       postingsReader.init(in, state);
124       
125       // NOTE: data file is too costly to verify checksum against all the bytes on open,
126       // but for now we at least verify proper structure of the checksum footer: which looks
127       // for FOOTER_MAGIC + algorithmID. This is cheap and can detect some forms of corruption
128       // such as file truncation.
129       CodecUtil.retrieveChecksum(in);
130 
131       // Read per-field details
132       seekDir(in);
133 
134       final int numFields = in.readVInt();
135       if (numFields < 0) {
136         throw new CorruptIndexException("invalid number of fields: " + numFields, in);
137       }
138       for(int i=0;i<numFields;i++) {
139         final int field = in.readVInt();
140         final long numTerms = in.readVLong();
141         assert numTerms >= 0;
142         final long termsStartPointer = in.readVLong();
143         final FieldInfo fieldInfo = state.fieldInfos.fieldInfo(field);
144         final long sumTotalTermFreq = fieldInfo.getIndexOptions() == IndexOptions.DOCS ? -1 : in.readVLong();
145         final long sumDocFreq = in.readVLong();
146         final int docCount = in.readVInt();
147         final int longsSize = in.readVInt();
148         if (docCount < 0 || docCount > state.segmentInfo.maxDoc()) { // #docs with field must be <= #docs
149           throw new CorruptIndexException("invalid docCount: " + docCount + " maxDoc: " + state.segmentInfo.maxDoc(), in);
150         }
151         if (sumDocFreq < docCount) {  // #postings must be >= #docs with field
152           throw new CorruptIndexException("invalid sumDocFreq: " + sumDocFreq + " docCount: " + docCount, in);
153         }
154         if (sumTotalTermFreq != -1 && sumTotalTermFreq < sumDocFreq) { // #positions must be >= #postings
155           throw new CorruptIndexException("invalid sumTotalTermFreq: " + sumTotalTermFreq + " sumDocFreq: " + sumDocFreq, in);
156         }
157         FieldReader previous = fields.put(fieldInfo.name, new FieldReader(fieldInfo, numTerms, termsStartPointer, sumTotalTermFreq, sumDocFreq, docCount, longsSize));
158         if (previous != null) {
159           throw new CorruptIndexException("duplicate fields: " + fieldInfo.name, in);
160         }
161       }
162       success = true;
163     } finally {
164       if (!success) {
165         in.close();
166       }
167     }
168 
169     this.indexReader = indexReader;
170   }

172   private void seekDir(IndexInput input) throws IOException {
173     input.seek(input.length() - CodecUtil.footerLength() - 8);
174     long dirOffset = input.readLong();
175     input.seek(dirOffset);
176   }

178   @Override
179   public void close() throws IOException {
180     try {
181       try {
182         if (indexReader != null) {
183           indexReader.close();
184         }
185       } finally {
186         // null so if an app hangs on to us (ie, we are not
187         // GCable, despite being closed) we still free most
188         // ram
189         indexReader = null;
190         if (in != null) {
191           in.close();
192         }
193       }
194     } finally {
195       if (postingsReader != null) {
196         postingsReader.close();
197       }
198     }
199   }

302       /* This is true if indexEnum is "still" seek'd to the index term
303          for the current term. We set it to true on seeking, and then it
304          remains valid until next() is called enough times to load another

337       public SegmentTermsEnum() throws IOException {
338         in = BlockTermsReader.this.in.clone();
339         in.seek(termsStartPointer);
340         indexEnum = indexReader.getFieldEnum(fieldInfo);
341         doOrd = indexReader.supportsOrd();
342         fieldTerm.field = fieldInfo.name;
343         state = postingsReader.newTermState();
344         state.totalTermFreq = -1;
345         state.ord = -1;
346 
347         termSuffixes = new byte[128];
348         docFreqBytes = new byte[64];
349         //System.out.println("BTR.enum init this=" + this + " postingsReader=" + postingsReader);
350         longs = new long[longsSize];
351       }
352 
353       // TODO: we may want an alternate mode here which is
354       // "if you are about to return NOT_FOUND I won't use
355       // the terms data from that"; eg FuzzyTermsEnum will
356       // (usually) just immediately call seek again if we
357       // return NOT_FOUND so it's a waste for us to fill in
358       // the term that was actually NOT_FOUND
359       @Override
360       public SeekStatus seekCeil(final BytesRef target) throws IOException {
361 
362         if (indexEnum == null) {
363           throw new IllegalStateException("terms index was not loaded");
364         }
365    
366         //System.out.println("BTR.seek seg=" + segment + " target=" + fieldInfo.name + ":" + target.utf8ToString() + " " + target + " current=" + term().utf8ToString() + " " + term() + " indexIsCurrent=" + indexIsCurrent + " didIndexNext=" + didIndexNext + " seekPending=" + seekPending + " divisor=" + indexReader.getDivisor() + " this="  + this);
367         if (didIndexNext) {
368           if (nextIndexTerm == null) {
369             //System.out.println("  nextIndexTerm=null");
370           } else {
371             //System.out.println("  nextIndexTerm=" + nextIndexTerm.utf8ToString());
372           }
373         }
374 
375         boolean doSeek = true;
376 
377         // See if we can avoid seeking, because target term
378         // is after current term but before next index term:
379         if (indexIsCurrent) {
380 
381           final int cmp = BytesRef.getUTF8SortedAsUnicodeComparator().compare(term.get(), target);
382 
383           if (cmp == 0) {
384             // Already at the requested term
385             return SeekStatus.FOUND;
386           } else if (cmp < 0) {
387 
388             // Target term is after current term
389             if (!didIndexNext) {
390               if (indexEnum.next() == -1) {
391                 nextIndexTerm = null;
392               } else {
393                 nextIndexTerm = indexEnum.term();
394               }
395               //System.out.println("  now do index next() nextIndexTerm=" + (nextIndexTerm == null ? "null" : nextIndexTerm.utf8ToString()));
396               didIndexNext = true;
397             }
398 
399             if (nextIndexTerm == null || BytesRef.getUTF8SortedAsUnicodeComparator().compare(target, nextIndexTerm) < 0) {
400               // Optimization: requested term is within the
401               // same term block we are now in; skip seeking
402               // (but do scanning):
403               doSeek = false;
404               //System.out.println("  skip seek: nextIndexTerm=" + (nextIndexTerm == null ? "null" : nextIndexTerm.utf8ToString()));
405             }
406           }
407         }
408 
409         if (doSeek) {
410           //System.out.println("  seek");
411 
412           // Ask terms index to find biggest indexed term (=
413           // first term in a block) that's <= our text:
414           in.seek(indexEnum.seek(target));
415           boolean result = nextBlock();
416 
417           // Block must exist since, at least, the indexed term
418           // is in the block:
419           assert result;
420 
421           indexIsCurrent = true;
422           didIndexNext = false;
423 
424           if (doOrd) {
425             state.ord = indexEnum.ord()-1;
426           }
427 
428           term.copyBytes(indexEnum.term());
429           //System.out.println("  seek: term=" + term.utf8ToString());
430         } else {
431           //System.out.println("  skip seek");
432           if (state.termBlockOrd == blockTermCount && !nextBlock()) {
433             indexIsCurrent = false;
434             return SeekStatus.END;
435           }
436         }
437 
438         seekPending = false;
439 
440         int common = 0;
441 
442         // Scan within block.  We could do this by calling
443         // _next() and testing the resulting term, but this
444         // is wasteful.  Instead, we first confirm the
445         // target matches the common prefix of this block,
446         // and then we scan the term bytes directly from the
447         // termSuffixesreader's byte[], saving a copy into
448         // the BytesRef term per term.  Only when we return
449         // do we then copy the bytes into the term.
450 
451         while(true) {
452 
453           // First, see if target term matches common prefix
454           // in this block:
455           if (common < termBlockPrefix) {
456             final int cmp = (term.byteAt(common)&0xFF) - (target.bytes[target.offset + common]&0xFF);
457             if (cmp < 0) {
458 
459               // TODO: maybe we should store common prefix
460               // in block header?  (instead of relying on
461               // last term of previous block)
462 
463               // Target's prefix is after the common block
464               // prefix, so term cannot be in this block
465               // but it could be in next block.  We
466               // must scan to end-of-block to set common
467               // prefix for next block:
468               if (state.termBlockOrd < blockTermCount) {
469                 while(state.termBlockOrd < blockTermCount-1) {
470                   state.termBlockOrd++;
471                   state.ord++;
472                   termSuffixesReader.skipBytes(termSuffixesReader.readVInt());
473                 }
474                 final int suffix = termSuffixesReader.readVInt();
475                 term.setLength(termBlockPrefix + suffix);
476                 term.grow(term.length());
477                 termSuffixesReader.readBytes(term.bytes(), termBlockPrefix, suffix);
478               }
479               state.ord++;
480               
481               if (!nextBlock()) {
482                 indexIsCurrent = false;
483                 return SeekStatus.END;
484               }
485               common = 0;
486 
487             } else if (cmp > 0) {
488               // Target's prefix is before the common prefix
489               // of this block, so we position to start of
490               // block and return NOT_FOUND:
491               assert state.termBlockOrd == 0;
492 
493               final int suffix = termSuffixesReader.readVInt();
494               term.setLength(termBlockPrefix + suffix);
495               term.grow(term.length());
496               termSuffixesReader.readBytes(term.bytes(), termBlockPrefix, suffix);
497               return SeekStatus.NOT_FOUND;
498             } else {
499               common++;
500             }
501 
502             continue;
503           }
504 
505           // Test every term in this block
506           while (true) {
507             state.termBlockOrd++;
508             state.ord++;
509 
510             final int suffix = termSuffixesReader.readVInt();
511             
512             // We know the prefix matches, so just compare the new suffix:
513             final int termLen = termBlockPrefix + suffix;
514             int bytePos = termSuffixesReader.getPosition();
515 
516             boolean next = false;
517             final int limit = target.offset + (termLen < target.length ? termLen : target.length);
518             int targetPos = target.offset + termBlockPrefix;
519             while(targetPos < limit) {
520               final int cmp = (termSuffixes[bytePos++]&0xFF) - (target.bytes[targetPos++]&0xFF);
521               if (cmp < 0) {
522                 // Current term is still before the target;
523                 // keep scanning
524                 next = true;
525                 break;
526               } else if (cmp > 0) {
527                 // Done!  Current term is after target. Stop
528                 // here, fill in real term, return NOT_FOUND.
529                 term.setLength(termBlockPrefix + suffix);
530                 term.grow(term.length());
531                 termSuffixesReader.readBytes(term.bytes(), termBlockPrefix, suffix);
532                 //System.out.println("  NOT_FOUND");
533                 return SeekStatus.NOT_FOUND;
534               }
535             }
536 
537             if (!next && target.length <= termLen) {
538               term.setLength(termBlockPrefix + suffix);
539               term.grow(term.length());
540               termSuffixesReader.readBytes(term.bytes(), termBlockPrefix, suffix);
541 
542               if (target.length == termLen) {
543                 // Done!  Exact match.  Stop here, fill in
544                 // real term, return FOUND.
545                 //System.out.println("  FOUND");
546                 return SeekStatus.FOUND;
547               } else {
548                 //System.out.println("  NOT_FOUND");
549                 return SeekStatus.NOT_FOUND;
550               }
551             }
552 
553             if (state.termBlockOrd == blockTermCount) {
554               // Must pre-fill term for next block's common prefix
555               term.setLength(termBlockPrefix + suffix);
556               term.grow(term.length());
557               termSuffixesReader.readBytes(term.bytes(), termBlockPrefix, suffix);
558               break;
559             } else {
560               termSuffixesReader.skipBytes(suffix);
561             }
562           }
563 
564           // The purpose of the terms dict index is to seek
565           // the enum to the closest index term before the
566           // term we are looking for.  So, we should never
567           // cross another index term (besides the first
568           // one) while we are scanning:
569 
570           assert indexIsCurrent;
571 
572           if (!nextBlock()) {
573             //System.out.println("  END");
574             indexIsCurrent = false;
575             return SeekStatus.END;
576           }
577           common = 0;
578         }
579       }

581       @Override
582       public BytesRef next() throws IOException {
583         //System.out.println("BTR.next() seekPending=" + seekPending + " pendingSeekCount=" + state.termBlockOrd);
584 
585         // If seek was previously called and the term was cached,
586         // usually caller is just going to pull a D/&PEnum or get
587         // docFreq, etc.  But, if they then call next(),
588         // this method catches up all internal state so next()
589         // works properly:
590         if (seekPending) {
591           assert !indexIsCurrent;
592           in.seek(state.blockFilePointer);
593           final int pendingSeekCount = state.termBlockOrd;
594           boolean result = nextBlock();
595 
596           final long savOrd = state.ord;
597 
598           // Block must exist since seek(TermState) was called w/ a
599           // TermState previously returned by this enum when positioned
600           // on a real term:
601           assert result;
602 
603           while(state.termBlockOrd < pendingSeekCount) {
604             BytesRef nextResult = _next();
605             assert nextResult != null;
606           }
607           seekPending = false;
608           state.ord = savOrd;
609         }
610         return _next();
611       }
612 
613       /* Decodes only the term bytes of the next term.  If caller then asks for
614          metadata, ie docFreq, totalTermFreq or pulls a D/&PEnum, we then (lazily)
615          decode all metadata up to the current term. */
616       private BytesRef _next() throws IOException {
617         //System.out.println("BTR._next seg=" + segment + " this=" + this + " termCount=" + state.termBlockOrd + " (vs " + blockTermCount + ")");
618         if (state.termBlockOrd == blockTermCount && !nextBlock()) {
619           //System.out.println("  eof");
620           indexIsCurrent = false;
621           return null;
622         }
623 
624         // TODO: cutover to something better for these ints!  simple64?
625         final int suffix = termSuffixesReader.readVInt();
626         //System.out.println("  suffix=" + suffix);
627 
628         term.setLength(termBlockPrefix + suffix);
629         term.grow(term.length());
630         termSuffixesReader.readBytes(term.bytes(), termBlockPrefix, suffix);
631         state.termBlockOrd++;
632 
633         // NOTE: meaningless in the non-ord case
634         state.ord++;
635 
636         //System.out.println("  return term=" + fieldInfo.name + ":" + term.utf8ToString() + " " + term + " tbOrd=" + state.termBlockOrd);
637         return term.get();
638       }

675       @Override
676       public void seekExact(BytesRef target, TermState otherState) {
677         //System.out.println("BTR.seekExact termState target=" + target.utf8ToString() + " " + target + " this=" + this);
678         assert otherState != null && otherState instanceof BlockTermState;
679         assert !doOrd || ((BlockTermState) otherState).ord < numTerms;
680         state.copyFrom(otherState);
681         seekPending = true;
682         indexIsCurrent = false;
683         term.copyBytes(target);
684       }

688         //System.out.println("BTR.termState this=" + this);

695       @Override
696       public void seekExact(long ord) throws IOException {
697         //System.out.println("BTR.seek by ord ord=" + ord);
698         if (indexEnum == null) {
699           throw new IllegalStateException("terms index was not loaded");
700         }
701 
702         assert ord < numTerms;
703 
704         // TODO: if ord is in same terms block and
705         // after current ord, we should avoid this seek just
706         // like we do in the seek(BytesRef) case
707         in.seek(indexEnum.seek(ord));
708         boolean result = nextBlock();
709 
710         // Block must exist since ord < numTerms:
711         assert result;
712 
713         indexIsCurrent = true;
714         didIndexNext = false;
715         seekPending = false;
716 
717         state.ord = indexEnum.ord()-1;
718         assert state.ord >= -1: "ord=" + state.ord;
719         term.copyBytes(indexEnum.term());
720 
721         // Now, scan:
722         int left = (int) (ord - state.ord);
723         while(left > 0) {
724           final BytesRef term = _next();
725           assert term != null;
726           left--;
727           assert indexIsCurrent;
728         }
729       }

732       public long ord() {
733         if (!doOrd) {
734           throw new UnsupportedOperationException();
735         }
736         return state.ord;
737       }

739       /* Does initial decode of next block of terms; this
740          doesn't actually decode the docFreq, totalTermFreq,
741          postings details (frq/prx offset, etc.) metadata;
742          it just loads them as byte[] blobs which are then      
743          decoded on-demand if the metadata is ever requested
744          for any term in this block.  This enables terms-only
745          intensive consumes (eg certain MTQs, respelling) to
746          not pay the price of decoding metadata they won't
747          use. */
748       private boolean nextBlock() throws IOException {
749 
750         // TODO: we still lazy-decode the byte[] for each
751         // term (the suffix), but, if we decoded
752         // all N terms up front then seeking could do a fast
753         // bsearch w/in the block...
754 
755         //System.out.println("BTR.nextBlock() fp=" + in.getFilePointer() + " this=" + this);
756         state.blockFilePointer = in.getFilePointer();
757         blockTermCount = in.readVInt();
758         //System.out.println("  blockTermCount=" + blockTermCount);
759         if (blockTermCount == 0) {
760           return false;
761         }
762         termBlockPrefix = in.readVInt();
763 
764         // term suffixes:
765         int len = in.readVInt();
766         if (termSuffixes.length < len) {
767           termSuffixes = new byte[ArrayUtil.oversize(len, 1)];
768         }
769         //System.out.println("  termSuffixes len=" + len);
770         in.readBytes(termSuffixes, 0, len);
771         termSuffixesReader.reset(termSuffixes, 0, len);
772 
773         // docFreq, totalTermFreq
774         len = in.readVInt();
775         if (docFreqBytes.length < len) {
776           docFreqBytes = new byte[ArrayUtil.oversize(len, 1)];
777         }
778         //System.out.println("  freq bytes len=" + len);
779         in.readBytes(docFreqBytes, 0, len);
780         freqReader.reset(docFreqBytes, 0, len);
781 
782         // metadata
783         len = in.readVInt();
784         if (bytes == null) {
785           bytes = new byte[ArrayUtil.oversize(len, 1)];
786           bytesReader = new ByteArrayDataInput();
787         } else if (bytes.length < len) {
788           bytes = new byte[ArrayUtil.oversize(len, 1)];
789         }
790         in.readBytes(bytes, 0, len);
791         bytesReader.reset(bytes, 0, len);
792 
793         metaDataUpto = 0;
794         state.termBlockOrd = 0;
795 
796         indexIsCurrent = false;
797         //System.out.println("  indexIsCurrent=" + indexIsCurrent);
798 
799         return true;
800       }

802       private void decodeMetaData() throws IOException {
803         //System.out.println("BTR.decodeMetadata mdUpto=" + metaDataUpto + " vs termCount=" + state.termBlockOrd + " state=" + state);
804         if (!seekPending) {
805           // TODO: cutover to random-access API
806           // here.... really stupid that we have to decode N
807           // wasted term metadata just to get to the N+1th
808           // that we really need...
809 
810           // lazily catch up on metadata decode:
811           final int limit = state.termBlockOrd;
812           boolean absolute = metaDataUpto == 0;
813           // TODO: better API would be "jump straight to term=N"???
814           while (metaDataUpto < limit) {
815             //System.out.println("  decode mdUpto=" + metaDataUpto);
816             // TODO: we could make "tiers" of metadata, ie,
817             // decode docFreq/totalTF but don't decode postings
818             // metadata; this way caller could get
819             // docFreq/totalTF w/o paying decode cost for
820             // postings
821 
822             // TODO: if docFreq were bulk decoded we could
823             // just skipN here:
824 
825             // docFreq, totalTermFreq
826             state.docFreq = freqReader.readVInt();
827             //System.out.println("    dF=" + state.docFreq);
828             if (fieldInfo.getIndexOptions() != IndexOptions.DOCS) {
829               state.totalTermFreq = state.docFreq + freqReader.readVLong();
830               //System.out.println("    totTF=" + state.totalTermFreq);
831             }
832             // metadata
833             for (int i = 0; i < longs.length; i++) {
834               longs[i] = bytesReader.readVLong();
835             }
836             postingsReader.decodeTerm(longs, bytesReader, fieldInfo, state, absolute);
837             metaDataUpto++;
838             absolute = false;
839           }
840         } else {
841           //System.out.println("  skip! seekPending");
842         }
843       }
844     }

849     long ramBytesUsed = BASE_RAM_BYTES_USED;
850     ramBytesUsed += (postingsReader!=null) ? postingsReader.ramBytesUsed() : 0;
851     ramBytesUsed += (indexReader!=null) ? indexReader.ramBytesUsed() : 0;
852     ramBytesUsed += fields.size() * 2L * RamUsageEstimator.NUM_BYTES_OBJECT_REF;
853     for (FieldReader reader : fields.values()) {

855     }
