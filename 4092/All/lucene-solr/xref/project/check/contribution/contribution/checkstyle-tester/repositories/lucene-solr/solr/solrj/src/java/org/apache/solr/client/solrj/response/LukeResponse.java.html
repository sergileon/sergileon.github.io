88      @SuppressWarnings("unchecked")
89      public void read(NamedList<Object> nl) {
90        for (Map.Entry<String, Object> entry : nl) {
91          String key = entry.getKey();
92          if ("fields".equals(key) && entry.getValue() != null) {
93            List<String> theFields = (List<String>) entry.getValue();
94            fields = new ArrayList<>(theFields);
95          } else if ("tokenized".equals(key) == true) {
96            tokenized = Boolean.parseBoolean(entry.getValue().toString());
97          } else if ("analyzer".equals(key) == true) {
98            analyzer = entry.getValue().toString();
99          } else if ("className".equals(key) == true) {
100           className = entry.getValue().toString();
101         }
102       }
103     }

106   public static class FieldInfo implements Serializable {
107     String name;
108     String type;
109     String schema;
110     int docs;
111     int distinct;
112     EnumSet<FieldFlag> flags;
113     boolean cacheableFaceting;
114     NamedList<Integer> topTerms;
115 
116     public FieldInfo(String n) {
117       name = n;
118     }
119 
120     @SuppressWarnings("unchecked")
121     public void read(NamedList<Object> nl) {
122       for (Map.Entry<String, Object> entry : nl) {
123         if ("type".equals(entry.getKey())) {
124           type = (String) entry.getValue();
125         }
126         if ("flags".equals(entry.getKey())) {
127           flags = parseFlags((String) entry.getValue());
128         } else if ("schema".equals(entry.getKey())) {
129           schema = (String) entry.getValue();
130         } else if ("docs".equals(entry.getKey())) {
131           docs = (Integer) entry.getValue();
132         } else if ("distinct".equals(entry.getKey())) {
133           distinct = (Integer) entry.getValue();
134         } else if ("cacheableFaceting".equals(entry.getKey())) {
135           cacheableFaceting = (Boolean) entry.getValue();
136         } else if ("topTerms".equals(entry.getKey())) {
137           topTerms = (NamedList<Integer>) entry.getValue();
138         }
139       }
140     }
141 
142     public static EnumSet<FieldFlag> parseFlags(String flagStr) {
143       EnumSet<FieldFlag> result = EnumSet.noneOf(FieldFlag.class);
144       char[] chars = flagStr.toCharArray();
145       for (int i = 0; i < chars.length; i++) {
146         if (chars[i] != '-') {
147           FieldFlag flag = FieldFlag.getFlag(chars[i]);
148           result.add(flag);
149         }
150       }
151       return result;
152     }

187   private NamedList<Object> indexInfo;
188   private Map<String, FieldInfo> fieldInfo;
189   private Map<String, FieldTypeInfo> fieldTypeInfo;
190 
191   @Override
192   @SuppressWarnings("unchecked")
193   public void setResponse(NamedList<Object> res) {
194     super.setResponse(res);
195 
196     // Parse indexinfo
197     indexInfo = (NamedList<Object>) res.get("index");
198 
199     NamedList<Object> schema = (NamedList<Object>) res.get("schema");
200     NamedList<Object> flds = (NamedList<Object>) res.get("fields");
201     if (flds == null && schema != null ) {
202       flds = (NamedList<Object>) schema.get("fields");
203     }
204     if (flds != null) {
205       fieldInfo = new HashMap<>();
206       for (Map.Entry<String, Object> field : flds) {
207         FieldInfo f = new FieldInfo(field.getKey());
208         f.read((NamedList<Object>) field.getValue());
209         fieldInfo.put(field.getKey(), f);
210       }
211     }
212 
213     if( schema != null ) {
214       NamedList<Object> fldTypes = (NamedList<Object>) schema.get("types");
215       if (fldTypes != null) {
216         fieldTypeInfo = new HashMap<>();
217         for (Map.Entry<String, Object> fieldType : fldTypes) {
218           FieldTypeInfo ft = new FieldTypeInfo(fieldType.getKey());
219           ft.read((NamedList<Object>) fieldType.getValue());
220           fieldTypeInfo.put(fieldType.getKey(), ft);
221         }
222       }
223     }
224   }
