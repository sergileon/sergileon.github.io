51    void makeExternalFile(String field, String contents) {
52      String dir = h.getCore().getDataDir();
53      String filename = dir + "/external_" + field + "." + (start++);
54      try {
55        Writer out = new OutputStreamWriter(new FileOutputStream(filename), StandardCharsets.UTF_8);
56        out.write(contents);
57        out.close();
58      } catch (Exception e) {
59        throw new RuntimeException(e);
60      }
61    }

64    void createIndex(String field, float... values) {
65      // lrf.args.put("version","2.0");
66      for (float val : values) {
67        String s = Float.toString(val);
68  
69        if (field!=null) assertU(adoc("id", s, field, s));
70        else assertU(adoc("id", s));
71  
72        if (random().nextInt(100) < 20) {
73          if (field!=null) assertU(adoc("id", s, field, s));
74          else assertU(adoc("id", s));
75        }
76  
77        if (random().nextInt(100) < 20) {
78          assertU(commit());
79  
80        }
81  
82  
83        // System.out.println("added doc for " + val);
84      }
85      // assertU(optimize()); // squeeze out any possible deleted docs
86      assertU(commit());
87    }

89    // replace \0 with the field name and create a parseable string 
90    public String func(String field, String template) {
91      StringBuilder sb = new StringBuilder("{!func}");
92      for (char ch : template.toCharArray()) {
93        if (ch=='\0') {
94          sb.append(field);
95          continue;
96        }
97        sb.append(ch);
98      }
99      return sb.toString();
100   }

102   void singleTest(String field, String funcTemplate, List<String> args, float... results) {
103     String parseableQuery = func(field, funcTemplate);
104 
105     List<String> nargs = new ArrayList<>(Arrays.asList("q", parseableQuery
106             ,"fl", "*,score"
107             ,"indent","on"
108             ,"rows","100"));
109 
110     if (args != null) {
111       for (String arg : args) {
112         nargs.add(arg.replace("\0",field));
113       }
114     }
115 
116     List<String> tests = new ArrayList<>();
117 
118     // Construct xpaths like the following:
119     // "//doc[./float[@name='foo_f']='10.0' and ./float[@name='score']='10.0']"
120 
121     for (int i=0; i<results.length; i+=2) {
122       String xpath = "//doc[./float[@name='" + "id" + "']='"
123               + results[i] + "' and ./float[@name='score']='"
124               + results[i+1] + "']";
125       tests.add(xpath);
126     }
127 
128     assertQ(req(nargs.toArray(new String[]{}))
129             , tests.toArray(new String[]{})
130     );
131   }

194     singleTest(field,"sum(query({!func v=\0},7.1),query({!func v=\0}))", 10, 20, 100, 200);
195     // test with sub-queries specified by other request args

209   @Test
210   public void testExternalField() throws Exception {
211     String field = "foo_extf";
212 
213     float[] ids = {100,-4,0,10,25,5,77,23,55,-78,-45,-24,63,78,94,22,34,54321,261,-627};
214 
215     createIndex(null,ids);
216 
217     // Unsorted field, largest first
218     makeExternalFile(field, "54321=543210\n0=-999\n25=250");
219     // test identity (straight field value)
220     singleTest(field, "\0", 54321, 543210, 0,-999, 25,250, 100, 1);
221     Object orig = FileFloatSource.onlyForTesting;
222     singleTest(field, "log(\0)");
223     // make sure the values were cached
224     assertTrue(orig == FileFloatSource.onlyForTesting);
225     singleTest(field, "sqrt(\0)");
226     assertTrue(orig == FileFloatSource.onlyForTesting);
227 
228     makeExternalFile(field, "0=1");
229     assertU(h.query("/reloadCache",lrf.makeRequest("","")));
230     singleTest(field, "sqrt(\0)");
231     assertTrue(orig != FileFloatSource.onlyForTesting);
232 
233 
234     Random r = random();
235     for (int i=0; i<10; i++) {   // do more iterations for a thorough test
236       int len = r.nextInt(ids.length+1);
237       boolean sorted = r.nextBoolean();
238       // shuffle ids
239       for (int j=0; j<ids.length; j++) {
240         int other=r.nextInt(ids.length);
241         float v=ids[0];
242         ids[0] = ids[other];
243         ids[other] = v;
244       }
245 
246       if (sorted) {
247         // sort only the first elements
248         Arrays.sort(ids,0,len);
249       }
250 
251       // make random values
252       float[] vals = new float[len];
253       for (int j=0; j<len; j++) {
254         vals[j] = r.nextInt(200)-100;
255       }
256 
257       // make and write the external file
258       StringBuilder sb = new StringBuilder();
259       for (int j=0; j<len; j++) {
260         sb.append("" + ids[j] + "=" + vals[j]+"\n");        
261       }
262       makeExternalFile(field, sb.toString());
263 
264       // make it visible
265       assertU(h.query("/reloadCache",lrf.makeRequest("","")));
266 
267       // test it
268       float[] answers = new float[ids.length*2];
269       for (int j=0; j<len; j++) {
270         answers[j*2] = ids[j];
271         answers[j*2+1] = vals[j];
272       }
273       for (int j=len; j<ids.length; j++) {
274         answers[j*2] = ids[j];
275         answers[j*2+1] = 1;  // the default values
276       }
277 
278       singleTest(field, "\0", answers);
279       // System.out.println("Done test "+i);
280     }  
281   }

307   @Test
308   public void testGeneral() throws Exception {
309     clearIndex();
310     
311     assertU(adoc("id","1", "a_tdt","2009-08-31T12:10:10.123Z", "b_tdt","2009-08-31T12:10:10.124Z"));
312     assertU(adoc("id","2", "a_t","how now brown cow"));
313     assertU(commit()); // create more than one segment
314     assertU(adoc("id","3", "a_t","brown cow"));
315     assertU(adoc("id","4"));
316     assertU(commit()); // create more than one segment
317     assertU(adoc("id","5"));
318     assertU(adoc("id","6", "a_t","cow cow cow cow cow"));
319     assertU(commit());
320 
321     // test relevancy functions
322     assertQ(req("fl","*,score","q", "{!func}numdocs()", "fq","id:6"), "//float[@name='score']='6.0'");
323     assertQ(req("fl","*,score","q", "{!func}maxdoc()", "fq","id:6"), "//float[@name='score']='6.0'");
324     assertQ(req("fl","*,score","q", "{!func}docfreq(a_t,cow)", "fq","id:6"), "//float[@name='score']='3.0'");
325     assertQ(req("fl","*,score","q", "{!func}docfreq('a_t','cow')", "fq","id:6"), "//float[@name='score']='3.0'");
326     assertQ(req("fl","*,score","q", "{!func}docfreq($field,$value)", "fq","id:6", "field","a_t", "value","cow"), "//float[@name='score']='3.0'");
327     assertQ(req("fl","*,score","q", "{!func}termfreq(a_t,cow)", "fq","id:6"), "//float[@name='score']='5.0'");
328 
329     TFIDFSimilarity similarity = new DefaultSimilarity();
330 
331     // make sure it doesn't get a NPE if no terms are present in a field.
332     assertQ(req("fl","*,score","q", "{!func}termfreq(nofield_t,cow)", "fq","id:6"), "//float[@name='score']='0.0'");
333     assertQ(req("fl","*,score","q", "{!func}docfreq(nofield_t,cow)", "fq","id:6"), "//float[@name='score']='0.0'");
334     assertQ(req("fl","*,score","q", "{!func}idf(nofield_t,cow)", "fq","id:6"),
335         "//float[@name='score']='" + similarity.idf(0,6)  + "'");
336      assertQ(req("fl","*,score","q", "{!func}tf(nofield_t,cow)", "fq","id:6"),
337         "//float[@name='score']='" + similarity.tf(0)  + "'");
338 
339     assertQ(req("fl","*,score","q", "{!func}idf(a_t,cow)", "fq","id:6"),
340         "//float[@name='score']='" + similarity.idf(3,6)  + "'");
341     assertQ(req("fl","*,score","q", "{!func}tf(a_t,cow)", "fq","id:6"),
342         "//float[@name='score']='" + similarity.tf(5)  + "'");
343     FieldInvertState state = new FieldInvertState("a_t");
344     state.setBoost(1.0f);
345     state.setLength(4);
346     long norm = similarity.computeNorm(state);
347     float nrm = similarity.decodeNormValue((byte) norm);
348     assertQ(req("fl","*,score","q", "{!func}norm(a_t)", "fq","id:2"),
349         "//float[@name='score']='" + nrm  + "'");  // sqrt(4)==2 and is exactly representable when quantized to a byte
350 
351     // test that ord and rord are working on a global index basis, not just
352     // at the segment level (since Lucene 2.9 has switched to per-segment searching)
353     assertQ(req("fl","*,score","q", "{!func}ord(id)", "fq","id:6"), "//float[@name='score']='5.0'");
354     assertQ(req("fl","*,score","q", "{!func}top(ord(id))", "fq","id:6"), "//float[@name='score']='5.0'");
355     assertQ(req("fl","*,score","q", "{!func}rord(id)", "fq","id:1"),"//float[@name='score']='5.0'");
356     assertQ(req("fl","*,score","q", "{!func}top(rord(id))", "fq","id:1"),"//float[@name='score']='5.0'");
357 
358 
359     // test that we can subtract dates to millisecond precision
360     assertQ(req("fl","*,score","q", "{!func}ms(a_tdt,b_tdt)", "fq","id:1"), "//float[@name='score']='-1.0'");
361     assertQ(req("fl","*,score","q", "{!func}ms(b_tdt,a_tdt)", "fq","id:1"), "//float[@name='score']='1.0'");
362     assertQ(req("fl","*,score","q", "{!func}ms(2009-08-31T12:10:10.125Z,2009-08-31T12:10:10.124Z)", "fq","id:1"), "//float[@name='score']='1.0'");
363     assertQ(req("fl","*,score","q", "{!func}ms(2009-08-31T12:10:10.124Z,a_tdt)", "fq","id:1"), "//float[@name='score']='1.0'");
364     assertQ(req("fl","*,score","q", "{!func}ms(2009-08-31T12:10:10.125Z,b_tdt)", "fq","id:1"), "//float[@name='score']='1.0'");
365 
366     assertQ(req("fl","*,score","q", "{!func}ms(2009-08-31T12:10:10.125Z/SECOND,2009-08-31T12:10:10.124Z/SECOND)", "fq","id:1"), "//float[@name='score']='0.0'");
367 
368     // test that we can specify "NOW"
369     assertQ(req("fl","*,score","q", "{!func}ms(NOW)", "NOW","1000"), "//float[@name='score']='1000.0'");
370 
371 
372     for (int i=100; i<112; i++) {
373       assertU(adoc("id",""+i, "text","batman"));
374     }
375     assertU(commit());
376     assertU(adoc("id","120", "text","batman superman"));   // in a smaller segment
377     assertU(adoc("id","121", "text","superman"));
378     assertU(commit());
379 
380     // superman has a higher df (thus lower idf) in one segment, but reversed in the complete index
381     String q ="{!func}query($qq)";
382     String fq="id:120"; 
383     assertQ(req("fl","*,score","q", q, "qq","text:batman", "fq",fq), "//float[@name='score']<'1.0'");
384     assertQ(req("fl","*,score","q", q, "qq","text:superman", "fq",fq), "//float[@name='score']>'1.0'");
385 
386     // test weighting through a function range query
387     assertQ(req("fl","*,score", "fq",fq,  "q", "{!frange l=1 u=10}query($qq)", "qq","text:superman"), "//*[@numFound='1']");
388 
389     // test weighting through a complex function
390     q ="{!func}sub(div(sum(0.0,product(1,query($qq))),1),0)";
391     assertQ(req("fl","*,score","q", q, "qq","text:batman", "fq",fq), "//float[@name='score']<'1.0'");
392     assertQ(req("fl","*,score","q", q, "qq","text:superman", "fq",fq), "//float[@name='score']>'1.0'");
393 
394 
395     // test full param dereferencing
396     assertQ(req("fl","*,score","q", "{!func}add($v1,$v2)", "v1","add($v3,$v4)", "v2","1", "v3","2", "v4","5"
397         , "fq","id:1"), "//float[@name='score']='8.0'");
398 
399     // test ability to parse multiple values
400     assertQ(req("fl","*,score","q", "{!func}dist(2,vector(1,1),$pt)", "pt","3,1"
401         , "fq","id:1"), "//float[@name='score']='2.0'");
402 
403     // test that extra stuff after a function causes an error
404     try {
405       assertQ(req("fl","*,score","q", "{!func}10 wow dude ignore_exception"));
406       fail();
407     } catch (Exception e) {
408       // OK
409     }
410 
411     // test that sorting by function weights correctly.  superman should sort higher than batman due to idf of the whole index
412 
413     assertQ(req("q", "*:*", "fq","id:120 OR id:121", "sort","{!func v=$sortfunc} desc", "sortfunc","query($qq)", "qq","text:(batman OR superman)")
414            ,"*//doc[1]/float[.='120.0']"
415            ,"*//doc[2]/float[.='121.0']"
416     );

558                 "fl", "good:strdist(x_s, 'toil', edit)",

563             , "count(//float[starts-with(@name,'bad')])=0"
564             );
565 
566     // in a query context, there is always a number...
567     //
568     // if a ValueSource is missing, it is maximally distant from every other
