37  /**
38   * <p>This class offers access to the operations exposed by the Solr Schema API.</p>
39   * <p>Most of the operations of this class offer a very abstract interface avoiding
40   * in this manner eventual changes due to Solr Schema API updates. On the other
41   * hand, the creation of request parameters for creating new fields or field types
42   * can be tedious because it is not strongly typed (the user has to build on his own
43   * a {@link NamedList} argument containing the field/field type properties).</p>
44   * <p>The class does not currently offer explicit support for the Schema API operations exposed
45   * through Managed Resources, but such operations can be built with little effort manually
46   * based on this class within the client applications.</p>
47   * <p>This class is experimental and it is subject to change.</p>
48   *
49   * @see <a href="https://cwiki.apache.org/confluence/display/solr/Schema+API">Solr Schema API</a>
50   * @see <a href="https://cwiki.apache.org/confluence/display/solr/Managed+Resources">Solr managed resources</a>
51   * @since solr 5.3
52   */
53  public class SchemaRequest extends AbstractSchemaRequest<SchemaResponse> {

69    private static NamedList<Object> createAddFieldTypeNamedList(FieldTypeDefinition fieldTypeDefinition) {
70      final NamedList<Object> addFieldTypeNamedList = new NamedList<>();
71      addFieldTypeNamedList.addAll(fieldTypeDefinition.getAttributes());
72      AnalyzerDefinition analyzerDefinition = fieldTypeDefinition.getAnalyzer();
73      if (analyzerDefinition != null) {
74        NamedList<Object> analyzerNamedList = createAnalyzerNamedList(analyzerDefinition);
75        addFieldTypeNamedList.add("analyzer", analyzerNamedList);
76      }
77      AnalyzerDefinition indexAnalyzerDefinition = fieldTypeDefinition.getIndexAnalyzer();
78      if (indexAnalyzerDefinition != null) {
79        NamedList<Object> indexAnalyzerNamedList = createAnalyzerNamedList(indexAnalyzerDefinition);
80        addFieldTypeNamedList.add("indexAnalyzer", indexAnalyzerNamedList);
81      }
82      AnalyzerDefinition queryAnalyzerDefinition = fieldTypeDefinition.getQueryAnalyzer();
83      if (queryAnalyzerDefinition != null) {
84        NamedList<Object> queryAnalyzerNamedList = createAnalyzerNamedList(queryAnalyzerDefinition);
85        addFieldTypeNamedList.add("queryAnalyzer", queryAnalyzerNamedList);
86      }
87      AnalyzerDefinition multiTermAnalyzerDefinition = fieldTypeDefinition.getMultiTermAnalyzer();
88      if (multiTermAnalyzerDefinition != null) {
89        NamedList<Object> multiTermAnalyzerNamedList = createAnalyzerNamedList(multiTermAnalyzerDefinition);
90        addFieldTypeNamedList.add("multiTermAnalyzer", multiTermAnalyzerNamedList);
91      }
92      Map<String, Object> similarityAttributes = fieldTypeDefinition.getSimilarity();
93      if (similarityAttributes != null && !similarityAttributes.isEmpty()) {
94        addFieldTypeNamedList.add("similarity", new NamedList<>(similarityAttributes));
95      }
96  
97      return addFieldTypeNamedList;
98    }

100   private static NamedList<Object> createAnalyzerNamedList(AnalyzerDefinition analyzerDefinition) {
101     NamedList<Object> analyzerNamedList = new NamedList<>();
102     Map<String, Object> analyzerAttributes = analyzerDefinition.getAttributes();
103     if (analyzerAttributes != null)
104       analyzerNamedList.addAll(analyzerAttributes);
105     List<Map<String, Object>> charFiltersAttributes = analyzerDefinition.getCharFilters();
106     if (charFiltersAttributes != null) {
107       List<NamedList<Object>> charFiltersList = new LinkedList<>();
108       for (Map<String, Object> charFilterAttributes : charFiltersAttributes)
109         charFiltersList.add(new NamedList<>(charFilterAttributes));
110       analyzerNamedList.add("charFilters", charFiltersList);
111     }
112     Map<String, Object> tokenizerAttributes = analyzerDefinition.getTokenizer();
113     if (tokenizerAttributes != null) {
114       analyzerNamedList.add("tokenizer", new NamedList<>(tokenizerAttributes));
115     }
116     List<Map<String, Object>> filtersAttributes = analyzerDefinition.getFilters();
117     if (filtersAttributes != null) {
118       List<NamedList<Object>> filtersList = new LinkedList<>();
119       for (Map<String, Object> filterAttributes : filtersAttributes)
120         filtersList.add(new NamedList<>(filterAttributes));
121       analyzerNamedList.add("filters", filtersList);
122     }
123     return analyzerNamedList;
124   }

126   private static NamedList<Object> createAddFieldNamedList(Map<String, Object> fieldAttributes) {
127     final NamedList<Object> addFieldProps = new NamedList<>();
128     if (fieldAttributes != null) addFieldProps.addAll(fieldAttributes);
129     return addFieldProps;
130   }

759   /**
760    * <p>The Schema API offers the possibility to perform one or more add requests in a single command.</p>
761    * <p>The API is transactional and all commands in a single {@link #process(SolrClient, String)} call
762    * either succeed or fail together.</p>
763    */
764   public static class MultiUpdate extends Update {
765     private List<Update> updateSchemaRequests = new LinkedList<>();
766 
767     public MultiUpdate(List<Update> updateSchemaRequests) {
768       this(updateSchemaRequests, null);
769     }
770 
771     public MultiUpdate(List<Update> updateSchemaRequests, SolrParams q) {
772       super(q);
773       if (updateSchemaRequests == null) {
774         throw new IllegalArgumentException("updateSchemaRequests must be non-null");
775       }
776       for (Update updateSchemaRequest : updateSchemaRequests) {
777         if (updateSchemaRequest == null) {
778           throw new IllegalArgumentException("updateSchemaRequests elements must be non-null");
779         }
780         this.updateSchemaRequests.add(updateSchemaRequest);
781       }
782     }
783 
784     @Override
785     protected NamedList<Object> getRequestParameters() {
786       NamedList<Object> multipleRequestsParameters = new NamedList<>();
787       for (Update updateSchemaRequest : updateSchemaRequests) {
788         multipleRequestsParameters.addAll(updateSchemaRequest.getRequestParameters());
789       }
790       return multipleRequestsParameters;
791     }
792   }

794   /**
795    * Simple extension of the noggit JSONWriter used to be write objects
796    * of type {@link NamedList}.
797    * Writing of objects of the type {@link NamedList} is done in very much
798    * the same way as for a map.
799    * <p>
800    * This writer is particularly useful when doing multiple update requests.
801    * In update Schema API there can be done multiple add operations of the same
802    * type (e.g. : add-field-type), they are grouped in an associative array, even though
803    * this can't be done normally in JSON. For such a use-case, the {@link NamedList}
804    * objects are particularly useful because they can group key-value mappings
805    * having the same values for the keys (unlike maps).
806    */
807   private static class SchemaRequestJSONWriter extends JSONWriter {
808     public SchemaRequestJSONWriter(CharArr out, int indentSize) {
809       super(out, indentSize);
810     }
811 
812     public SchemaRequestJSONWriter(CharArr out) {
813       super(out);
814     }
815 
816     public void write(Object o) {
817       if (o instanceof NamedList) {
818         write((NamedList) o);
819       } else super.write(o);
820     }
821 
822     /**
823      * @see #write(Map)
824      */
825     @SuppressWarnings("unchecked")
826     public void write(NamedList<?> val) {
827       this.startObject();
828       int sz = val.size();
829       boolean first = true;
830       Iterator i$ = val.iterator();
831 
832       while (i$.hasNext()) {
833         Map.Entry<String, ?> entry = (Map.Entry<String, ?>) i$.next();
834         if (first) {
835           first = false;
836         } else {
837           this.writeValueSeparator();
838         }
839 
840         if (sz > 1) {
841           this.indent();
842         }
843 
844         this.writeString(entry.getKey());
845         this.writeNameSeparator();
846         this.write(entry.getValue());
847       }
848 
849       this.endObject();
850     }
851   }
