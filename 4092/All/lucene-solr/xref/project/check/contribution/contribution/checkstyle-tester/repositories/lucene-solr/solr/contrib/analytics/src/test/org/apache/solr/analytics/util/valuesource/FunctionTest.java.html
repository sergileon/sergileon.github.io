27    static String fileName = "/analytics/requestFiles/functions.txt";
28  
29    static public final int INT = 71;
30    static public final int LONG = 36;
31    static public final int FLOAT = 93;
32    static public final int DOUBLE = 49;
33    static public final int DATE = 12;
34    static public final int STRING = 28;
35    static public final int NUM_LOOPS = 100;

37    @BeforeClass
38    public static void beforeClass() throws Exception {
39      initCore("solrconfig-basic.xml","schema-analytics.xml");
40      h.update("<delete><query>*:*</query></delete>");
41      
42      for (int j = 0; j < NUM_LOOPS; ++j) {
43        int i = j%INT+1;
44        long l = j%LONG+1;
45        float f = j%FLOAT+1;
46        double d = j%DOUBLE+1;
47        double d0 = j%DOUBLE;
48        String dt = (1800+j%DATE) + "-06-30T23:59:59Z";
49        String s = "str" + (j%STRING);
50  
51        double add_if = (double)i+f;
52        double add_ldf = (double)l+d+f;
53        double mult_if = (double)i*f;
54        double mult_ldf = (double)l*d*f;
55        double div_if = (double)i/f;
56        double div_ld = (double)l/d;
57        double pow_if = Math.pow(i,f);
58        double pow_ld = Math.pow(l,d);
59        double neg_i = (double)i*-1;
60        double neg_l = (double)l*-1;
61        String dm_2y = (1802+j%DATE) + "-06-30T23:59:59Z";
62        String dm_2m = (1800+j%DATE) + "-08-30T23:59:59Z";
63        String concat_first = "this is the first"+s;
64        String concat_second = "this is the second"+s;
65        String rev = new StringBuilder(s).reverse().toString();
66        
67        assertU(adoc(AbstractAnalyticsFacetTest.filter("id", "1000" + j, "int_id", "" + i, "long_ld", "" + l, "float_fd", "" + f, 
68              "double_dd", "" + d,  "date_dtd", dt, "string_sd", s,
69              "add_if_dd", ""+add_if, "add_ldf_dd", ""+add_ldf, "mult_if_dd", ""+mult_if, "mult_ldf_dd", ""+mult_ldf,
70              "div_if_dd", ""+div_if, "div_ld_dd", ""+div_ld, "pow_if_dd", ""+pow_if, "pow_ld_dd", ""+pow_ld,
71              "neg_i_dd", ""+neg_i, "neg_l_dd", ""+neg_l, "const_8_dd", "8", "const_10_dd", "10", "dm_2y_dtd", dm_2y, "dm_2m_dtd", dm_2m,
72              "const_00_dtd", "1800-06-30T23:59:59Z", "const_04_dtd", "1804-06-30T23:59:59Z", "const_first_sd", "this is the first", "const_second_sd", "this is the second",
73              "concat_first_sd", concat_first, "concat_second_sd", concat_second, "rev_sd", rev, "miss_dd", ""+d0 )));
74        
75        
76        if (usually()) {
77          assertU(commit()); // to have several segments
78        }
79      }
80      
81      assertU(commit()); 
82      
83      setResponse(h.query(request(fileToStringArr(FunctionTest.class, fileName))));
84    }

86    @Test
87    public void addTest() throws Exception { 
88      double result = (Double)getStatResult("ar", "sum", VAL_TYPE.DOUBLE);
89      double calculated = (Double)getStatResult("ar", "sumc", VAL_TYPE.DOUBLE);
90      assertEquals(getRawResponse(), result, calculated, 0.0);
91      // TODO checfk why asserted 2times
92      assertEquals(getRawResponse(), result, calculated, 0.0);
93  
94      result = (Double)getStatResult("ar", "mean", VAL_TYPE.DOUBLE);
95      calculated = (Double)getStatResult("ar", "meanc", VAL_TYPE.DOUBLE);
96      assertTrue(result==calculated);
97      assertEquals(getRawResponse(), result, calculated, 0.0);
98    }
