52    public static void createIndex() throws IOException, Exception {
53      int i = 0;
54      List<List<String[]>> blocks = createBlocks();
55      for (List<String[]> block : blocks) {
56        List<XmlDoc> updBlock = new ArrayList<>();
57        
58        for (String[] doc : block) {
59          String[] idDoc = Arrays.copyOf(doc,doc.length+2);
60          idDoc[doc.length]="id";
61          idDoc[doc.length+1]=Integer.toString(i);
62          updBlock.add(doc(idDoc));
63          i++;
64        }
65        //got xmls for every doc. now nest all into the last one
66        XmlDoc parentDoc = updBlock.get(updBlock.size()-1);
67        parentDoc.xml = parentDoc.xml.replace("</doc>", 
68            updBlock.subList(0, updBlock.size()-1).toString().replaceAll("[\\[\\]]","")+"</doc>");
69        assertU(add(parentDoc));
70        
71        if (random().nextBoolean()) {
72          assertU(commit());
73          // force empty segment (actually, this will no longer create an empty segment, only a new segments_n)
74          if (random().nextBoolean()) {
75            assertU(commit());
76          }
77        }
78      }
79      assertU(commit());
80      assertQ(req("q", "*:*"), "//*[@numFound='" + i + "']");
81      /*
82       * dump docs well System.out.println(h.query(req("q","*:*",
83       * "sort","_docid_ asc", "fl",
84       * "parent_s,child_s,parentchild_s,grand_s,grand_child_s,grand_parentchild_s"
85       * , "wt","csv", "rows","1000"))); /
86       */
87    }

90    private static List<List<String[]>> createBlocks() {
91      List<List<String[]>> blocks = new ArrayList<>();
92      for (String parent : abcdef) {
93        List<String[]> block = createChildrenBlock(parent);
94        block.add(new String[] {"parent_s", parent});
95        blocks.add(block);
96      }
97      Collections.shuffle(blocks, random());
98      return blocks;
99    }

101   private static List<String[]> createChildrenBlock(String parent) {
102     List<String[]> block = new ArrayList<>();
103     for (String child : klm) {
104       block
105           .add(new String[] {"child_s", child, "parentchild_s", parent + child});
106     }
107     Collections.shuffle(block, random());
108     addGrandChildren(block);
109     return block;
110   }
111   
112   private static void addGrandChildren(List<String[]> block) {
113     List<String> grandChildren = new ArrayList<>(xyz);
114     // add grandchildren after children
115     for (ListIterator<String[]> iter = block.listIterator(); iter.hasNext();) {
116       String[] child = iter.next();
117       assert child[0]=="child_s" && child[2]=="parentchild_s": Arrays.toString(child);
118       String child_s = child[1];
119       String parentchild_s = child[3];
120       int grandChildPos = 0;
121       boolean lastLoopButStillHasGrCh = !iter.hasNext()
122           && !grandChildren.isEmpty();
123       while (!grandChildren.isEmpty()
124           && ((grandChildPos = random().nextInt(grandChildren.size() * 2)) < grandChildren
125               .size() || lastLoopButStillHasGrCh)) {
126         grandChildPos = grandChildPos >= grandChildren.size() ? 0
127             : grandChildPos;
128         iter.add(new String[] {"grand_s", grandChildren.remove(grandChildPos),
129             "grand_child_s", child_s, "grand_parentchild_s", parentchild_s});
130       }
131     }
132     // and reverse after that
133     Collections.reverse(block);
134   }

278   @Test
279   public void testCacheHit() throws IOException {
280 
281     SolrCache parentFilterCache = (SolrCache) h.getCore().getInfoRegistry()
282         .get("perSegFilter");
283 
284     SolrCache filterCache = (SolrCache) h.getCore().getInfoRegistry()
285         .get("filterCache");
286 
287     NamedList parentsBefore = parentFilterCache.getStatistics();
288 
289     NamedList filtersBefore = filterCache.getStatistics();
290 
291     // it should be weird enough to be uniq
292     String parentFilter = "parent_s:([a TO c] [d TO f])";
293 
294     assertQ("search by parent filter",
295         req("q", "{!parent which=\"" + parentFilter + "\"}"),
296         "//*[@numFound='6']");
297 
298     assertQ("filter by parent filter",
299         req("q", "*:*", "fq", "{!parent which=\"" + parentFilter + "\"}"),
300         "//*[@numFound='6']");
301 
302     assertEquals("didn't hit fqCache yet ", 0L,
303         delta("hits", filterCache.getStatistics(), filtersBefore));
304 
305     assertQ(
306         "filter by join",
307         req("q", "*:*", "fq", "{!parent which=\"" + parentFilter
308             + "\"}child_s:l"), "//*[@numFound='6']");
309 
310     assertEquals("in cache mode every request lookups", 3,
311         delta("lookups", parentFilterCache.getStatistics(), parentsBefore));
312     assertEquals("last two lookups causes hits", 2,
313         delta("hits", parentFilterCache.getStatistics(), parentsBefore));
314     assertEquals("the first lookup gets insert", 1,
315         delta("inserts", parentFilterCache.getStatistics(), parentsBefore));
316 
317 
318     assertEquals("true join query is cached in fqCache", 1L,
319         delta("lookups", filterCache.getStatistics(), filtersBefore));
320   }
321   
322   private long delta(String key, NamedList a, NamedList b) {
323     return (Long) a.get(key) - (Long) b.get(key);
324   }
