41  /**
42   * Analyzer factory construction task.  The name given to the constructed factory may
43   * be given to NewAnalyzerTask, which will call AnalyzerFactory.create().
44   *
45   * Params are in the form argname:argvalue or argname:"argvalue" or argname:'argvalue';
46   * use backslashes to escape '"' or "'" inside a quoted value when it's used as the enclosing
47   * quotation mark,
48   *
49   * Specify params in a comma separated list of the following, in order:
50   * <ol>
51   *   <li>Analyzer args:
52   *     <ul>
53   *       <li><b>Required</b>: <code>name:<i>analyzer-factory-name</i></code></li>
54   *       <li>Optional: <tt>positionIncrementGap:<i>int value</i></tt> (default: 0)</li>
55   *       <li>Optional: <tt>offsetGap:<i>int value</i></tt> (default: 1)</li>
56   *     </ul>
57   *   </li>
58   *   <li>zero or more CharFilterFactory's, followed by</li>
59   *   <li>exactly one TokenizerFactory, followed by</li>
60   *   <li>zero or more TokenFilterFactory's</li>
61   * </ol>
62   *
63   * Each component analysis factory may specify <tt>luceneMatchVersion</tt> (defaults to
64   * {@link Version#LATEST}) and any of the args understood by the specified
65   * *Factory class, in the above-describe param format.
66   * <p>
67   * Example:
68   * <pre>
69   *     -AnalyzerFactory(name:'strip html, fold to ascii, whitespace tokenize, max 10k tokens',
70   *                      positionIncrementGap:100,
71   *                      HTMLStripCharFilter,
72   *                      MappingCharFilter(mapping:'mapping-FoldToASCII.txt'),
73   *                      WhitespaceTokenizer(luceneMatchVersion:LUCENE_5_0_0),
74   *                      TokenLimitFilter(maxTokenCount:10000, consumeAllTokens:false))
75   *     [...]
76   *     -NewAnalyzer('strip html, fold to ascii, whitespace tokenize, max 10k tokens')
77   * </pre>
78   * <p>
79   * AnalyzerFactory will direct analysis component factories to look for resources
80   * under the directory specified in the "work.dir" property.
81   */
82  public class AnalyzerFactoryTask extends PerfTask {
83    private static final String LUCENE_ANALYSIS_PACKAGE_PREFIX = "org.apache.lucene.analysis.";
84    private static final Pattern ANALYSIS_COMPONENT_SUFFIX_PATTERN
85        = Pattern.compile("(?s:(?:(?:Token|Char)?Filter|Tokenizer)(?:Factory)?)$");
86    private static final Pattern TRAILING_DOT_ZERO_PATTERN = Pattern.compile("\\.0$");
87  
88    private enum ArgType {ANALYZER_ARG, ANALYZER_ARG_OR_CHARFILTER_OR_TOKENIZER, TOKENFILTER }
89  
90    String factoryName = null;
91    Integer positionIncrementGap = null;
92    Integer offsetGap = null;
93    private List<CharFilterFactory> charFilterFactories = new ArrayList<>();
94    private TokenizerFactory tokenizerFactory = null;
95    private List<TokenFilterFactory> tokenFilterFactories = new ArrayList<>();
96  
97    public AnalyzerFactoryTask(PerfRunData runData) {
98      super(runData);
99    }
100 
101   @Override
102   public int doLogic() {
103     return 1;
104   }
105 
106   /**
107    * Sets the params.
108    * Analysis component factory names may optionally include the "Factory" suffix.
109    *
110    * @param params analysis pipeline specification: name, (optional) positionIncrementGap,
111    *               (optional) offsetGap, 0+ CharFilterFactory's, 1 TokenizerFactory,
112    *               and 0+ TokenFilterFactory's
113    */
114   @Override
115   @SuppressWarnings("fallthrough")
116   public void setParams(String params) {
117     super.setParams(params);
118     ArgType expectedArgType = ArgType.ANALYZER_ARG;
119 
120     final StreamTokenizer stok = new StreamTokenizer(new StringReader(params));
121     stok.commentChar('#');
122     stok.quoteChar('"');
123     stok.quoteChar('\'');
124     stok.eolIsSignificant(false);
125     stok.ordinaryChar('(');
126     stok.ordinaryChar(')');
127     stok.ordinaryChar(':');
128     stok.ordinaryChar(',');
129     try {
130       while (stok.nextToken() != StreamTokenizer.TT_EOF) {
131         switch (stok.ttype) {
132           case ',': {
133             // Do nothing
134             break;
135           }
136           case StreamTokenizer.TT_WORD: {
137             if (expectedArgType.equals(ArgType.ANALYZER_ARG)) {
138               final String argName = stok.sval;
139               if ( ! argName.equalsIgnoreCase("name")
140                   && ! argName.equalsIgnoreCase("positionIncrementGap")
141                   && ! argName.equalsIgnoreCase("offsetGap")) {
142                 throw new RuntimeException
143                     ("Line #" + lineno(stok) + ": Missing 'name' param to AnalyzerFactory: '" + params + "'");
144               }
145               stok.nextToken();
146               if (stok.ttype != ':') {
147                 throw new RuntimeException
148                     ("Line #" + lineno(stok) + ": Missing ':' after '" + argName + "' param to AnalyzerFactory");
149               }
150 
151               stok.nextToken();
152               String argValue = stok.sval;
153               switch (stok.ttype) {
154                 case StreamTokenizer.TT_NUMBER: {
155                   argValue = Double.toString(stok.nval);
156                   // Drop the ".0" from numbers, for integer arguments
157                   argValue = TRAILING_DOT_ZERO_PATTERN.matcher(argValue).replaceFirst("");
158                   // Intentional fallthrough
159                 }
160                 case '"':
161                 case '\'':
162                 case StreamTokenizer.TT_WORD: {
163                   if (argName.equalsIgnoreCase("name")) {
164                     factoryName = argValue;
165                     expectedArgType = ArgType.ANALYZER_ARG_OR_CHARFILTER_OR_TOKENIZER;
166                   } else {
167                     int intArgValue = 0;
168                     try {
169                       intArgValue = Integer.parseInt(argValue);
170                     } catch (NumberFormatException e) {
171                       throw new RuntimeException
172                           ("Line #" + lineno(stok) + ": Exception parsing " + argName + " value '" + argValue + "'", e);
173                     }
174                     if (argName.equalsIgnoreCase("positionIncrementGap")) {
175                       positionIncrementGap = intArgValue;
176                     } else if (argName.equalsIgnoreCase("offsetGap")) {
177                       offsetGap = intArgValue;
178                     }
179                   }
180                   break;
181                 }
182                 case StreamTokenizer.TT_EOF: {
183                   throw new RuntimeException("Unexpected EOF: " + stok.toString());
184                 }
185                 default: {
186                   throw new RuntimeException
187                       ("Line #" + lineno(stok) + ": Unexpected token: " + stok.toString());
188                 }
189               }
190             } else if (expectedArgType.equals(ArgType.ANALYZER_ARG_OR_CHARFILTER_OR_TOKENIZER)) {
191               final String argName = stok.sval;
192 
193               if (argName.equalsIgnoreCase("positionIncrementGap")
194                   || argName.equalsIgnoreCase("offsetGap")) {
195                 stok.nextToken();
196                 if (stok.ttype != ':') {
197                   throw new RuntimeException
198                       ("Line #" + lineno(stok) + ": Missing ':' after '" + argName + "' param to AnalyzerFactory");
199                 }
200                 stok.nextToken();
201                 int intArgValue = (int)stok.nval;
202                 switch (stok.ttype) {
203                   case '"':
204                   case '\'':
205                   case StreamTokenizer.TT_WORD: {
206                     intArgValue = 0;
207                     try {
208                       intArgValue = Integer.parseInt(stok.sval.trim());
209                     } catch (NumberFormatException e) {
210                       throw new RuntimeException
211                           ("Line #" + lineno(stok) + ": Exception parsing " + argName + " value '" + stok.sval + "'", e);
212                     }
213                     // Intentional fall-through
214                   }
215                   case StreamTokenizer.TT_NUMBER: {
216                     if (argName.equalsIgnoreCase("positionIncrementGap")) {
217                       positionIncrementGap = intArgValue;
218                     } else if (argName.equalsIgnoreCase("offsetGap")) {
219                       offsetGap = intArgValue;
220                     }
221                     break;
222                   }
223                   case StreamTokenizer.TT_EOF: {
224                     throw new RuntimeException("Unexpected EOF: " + stok.toString());
225                   }
226                   default: {
227                     throw new RuntimeException
228                         ("Line #" + lineno(stok) + ": Unexpected token: " + stok.toString());
229                   }
230                 }
231                 break;
232               }
233               try {
234                 final Class<? extends CharFilterFactory> clazz;
235                 clazz = lookupAnalysisClass(argName, CharFilterFactory.class);
236                 createAnalysisPipelineComponent(stok, clazz);
237               } catch (IllegalArgumentException e) {
238                 try {
239                   final Class<? extends TokenizerFactory> clazz;
240                   clazz = lookupAnalysisClass(argName, TokenizerFactory.class);
241                   createAnalysisPipelineComponent(stok, clazz);
242                   expectedArgType = ArgType.TOKENFILTER;
243                 } catch (IllegalArgumentException e2) {
244                   throw new RuntimeException("Line #" + lineno(stok) + ": Can't find class '"
245                                              + argName + "' as CharFilterFactory or TokenizerFactory");
246                 }
247               }
248             } else { // expectedArgType = ArgType.TOKENFILTER
249               final String className = stok.sval;
250               final Class<? extends TokenFilterFactory> clazz;
251               try {
252                 clazz = lookupAnalysisClass(className, TokenFilterFactory.class);
253               } catch (IllegalArgumentException e) {
254                   throw new RuntimeException
255                       ("Line #" + lineno(stok) + ": Can't find class '" + className + "' as TokenFilterFactory");
256               }
257               createAnalysisPipelineComponent(stok, clazz);
258             }
259             break;
260           }
261           default: {
262             throw new RuntimeException("Line #" + lineno(stok) + ": Unexpected token: " + stok.toString());
263           }
264         }
265       }
266     } catch (RuntimeException e) {
267       if (e.getMessage().startsWith("Line #")) {
268         throw e;
269       } else {
270         throw new RuntimeException("Line #" + lineno(stok) + ": ", e);
271       }
272     } catch (Throwable t) {
273       throw new RuntimeException("Line #" + lineno(stok) + ": ", t);
274     }
275 
276     final AnalyzerFactory analyzerFactory = new AnalyzerFactory
277         (charFilterFactories, tokenizerFactory, tokenFilterFactories);
278     analyzerFactory.setPositionIncrementGap(positionIncrementGap);
279     analyzerFactory.setOffsetGap(offsetGap);
280     getRunData().getAnalyzerFactories().put(factoryName, analyzerFactory);
281   }

283   /**
284    * Instantiates the given analysis factory class after pulling params from
285    * the given stream tokenizer, then stores the result in the appropriate
286    * pipeline component list.
287    *
288    * @param stok stream tokenizer from which to draw analysis factory params
289    * @param clazz analysis factory class to instantiate
290    */
291   @SuppressWarnings("fallthrough")
292   private void createAnalysisPipelineComponent
293       (StreamTokenizer stok, Class<? extends AbstractAnalysisFactory> clazz) {
294     Map<String,String> argMap = new HashMap<>();
295     boolean parenthetical = false;
296     try {
297       WHILE_LOOP: while (stok.nextToken() != StreamTokenizer.TT_EOF) {
298         switch (stok.ttype) {
299           case ',': {
300             if (parenthetical) {
301               // Do nothing
302               break;
303             } else {
304               // Finished reading this analysis factory configuration
305               break WHILE_LOOP;
306             }
307           }
308           case '(': {
309             if (parenthetical) {
310               throw new RuntimeException
311                   ("Line #" + lineno(stok) + ": Unexpected opening parenthesis.");
312             }
313             parenthetical = true;
314             break;
315           }
316           case ')': {
317             if (parenthetical) {
318               parenthetical = false;
319             } else {
320               throw new RuntimeException
321                   ("Line #" + lineno(stok) + ": Unexpected closing parenthesis.");
322             }
323             break;
324           }
325           case StreamTokenizer.TT_WORD: {
326             if ( ! parenthetical) {
327               throw new RuntimeException("Line #" + lineno(stok) + ": Unexpected token '" + stok.sval + "'");
328             }
329             String argName = stok.sval;
330             stok.nextToken();
331             if (stok.ttype != ':') {
332               throw new RuntimeException
333                   ("Line #" + lineno(stok) + ": Missing ':' after '" + argName + "' param to " + clazz.getSimpleName());
334             }
335             stok.nextToken();
336             String argValue = stok.sval;
337             switch (stok.ttype) {
338               case StreamTokenizer.TT_NUMBER: {
339                   argValue = Double.toString(stok.nval);
340                   // Drop the ".0" from numbers, for integer arguments
341                   argValue = TRAILING_DOT_ZERO_PATTERN.matcher(argValue).replaceFirst("");
342                   // Intentional fall-through
343               }
344               case '"':
345               case '\'':
346               case StreamTokenizer.TT_WORD: {
347                 argMap.put(argName, argValue);
348                 break;
349               }
350               case StreamTokenizer.TT_EOF: {
351                 throw new RuntimeException("Unexpected EOF: " + stok.toString());
352               }
353               default: {
354                 throw new RuntimeException
355                     ("Line #" + lineno(stok) + ": Unexpected token: " + stok.toString());
356               }
357             }
358           }
359         }
360       }
361       if (!argMap.containsKey("luceneMatchVersion")) {
362         argMap.put("luceneMatchVersion", Version.LATEST.toString());
363       }
364       final AbstractAnalysisFactory instance;
365       try {
366         instance = clazz.getConstructor(Map.class).newInstance(argMap);
367       } catch (Exception e) {
368         throw new RuntimeException("Line #" + lineno(stok) + ": ", e);
369       }
370       if (instance instanceof ResourceLoaderAware) {
371         Path baseDir = Paths.get(getRunData().getConfig().get("work.dir", "work"));
372         if (!Files.isDirectory(baseDir)) {
373           baseDir = Paths.get(".");
374         }
375         ((ResourceLoaderAware)instance).inform(new FilesystemResourceLoader(baseDir));
376       }
377       if (CharFilterFactory.class.isAssignableFrom(clazz)) {
378         charFilterFactories.add((CharFilterFactory)instance);
379       } else if (TokenizerFactory.class.isAssignableFrom(clazz)) {
380         tokenizerFactory = (TokenizerFactory)instance;
381       } else if (TokenFilterFactory.class.isAssignableFrom(clazz)) {
382         tokenFilterFactories.add((TokenFilterFactory)instance);
383       }
384     } catch (RuntimeException e) {
385       if (e.getMessage().startsWith("Line #")) {
386         throw (e);
387       } else {
388         throw new RuntimeException("Line #" + lineno(stok) + ": ", e);
389       }
390     } catch (Throwable t) {
391       throw new RuntimeException("Line #" + lineno(stok) + ": ", t);
392     }
393   }

395   /**
396    * This method looks up a class with its fully qualified name (FQN), or a short-name
397    * class-simplename, or with a package suffix, assuming "org.apache.lucene.analysis."
398    * as the package prefix (e.g. "standard.ClassicTokenizerFactory" -&gt;
399    * "org.apache.lucene.analysis.standard.ClassicTokenizerFactory").
400    *
401    * If className contains a period, the class is first looked up as-is, assuming that it
402    * is an FQN.  If this fails, lookup is retried after prepending the Lucene analysis
403    * package prefix to the class name.
404    *
405    * If className does not contain a period, the analysis SPI *Factory.lookupClass()
406    * methods are used to find the class.
407    *
408    * @param className The name or the short name of the class.
409    * @param expectedType The superclass className is expected to extend
410    * @return the loaded class.
411    * @throws ClassNotFoundException if lookup fails
412    */
413   public <T> Class<? extends T> lookupAnalysisClass(String className, Class<T> expectedType)
414       throws ClassNotFoundException {
415     if (className.contains(".")) {
416       try {
417         // First, try className == FQN
418         return Class.forName(className).asSubclass(expectedType);
419       } catch (ClassNotFoundException e) {
420         try {
421           // Second, retry lookup after prepending the Lucene analysis package prefix
422           return Class.forName(LUCENE_ANALYSIS_PACKAGE_PREFIX + className).asSubclass(expectedType);
423         } catch (ClassNotFoundException e1) {
424           throw new ClassNotFoundException("Can't find class '" + className
425                                            + "' or '" + LUCENE_ANALYSIS_PACKAGE_PREFIX + className + "'");
426         }
427       }
428     }
429     // No dot - use analysis SPI lookup
430     final String analysisComponentName = ANALYSIS_COMPONENT_SUFFIX_PATTERN.matcher(className).replaceFirst("");
431     if (CharFilterFactory.class.isAssignableFrom(expectedType)) {
432       return CharFilterFactory.lookupClass(analysisComponentName).asSubclass(expectedType);
433     } else if (TokenizerFactory.class.isAssignableFrom(expectedType)) {
434       return TokenizerFactory.lookupClass(analysisComponentName).asSubclass(expectedType);
435     } else if (TokenFilterFactory.class.isAssignableFrom(expectedType)) {
436       return TokenFilterFactory.lookupClass(analysisComponentName).asSubclass(expectedType);
437     }
438 
439     throw new ClassNotFoundException("Can't find class '" + className + "'");
440   }
