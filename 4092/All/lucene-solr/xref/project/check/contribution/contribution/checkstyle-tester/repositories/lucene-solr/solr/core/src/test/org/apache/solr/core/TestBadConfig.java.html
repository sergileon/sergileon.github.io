18  package org.apache.solr.core;
19  
20  import javax.script.ScriptEngineManager;
21  
22  import org.junit.Assume;
23  
24  public class TestBadConfig extends AbstractBadConfigTestBase {
25  
26    public void testUnsetSysProperty() throws Exception {
27      assertConfigs("bad_solrconfig.xml","schema.xml","unset.sys.property");
28    }
29  
30    public void testMultipleDirectoryFactories() throws Exception {
31        assertConfigs("bad-solrconfig-multiple-dirfactory.xml", "schema12.xml",
32                      "directoryFactory");
33    }
34    public void testMultipleIndexConfigs() throws Exception {
35        assertConfigs("bad-solrconfig-multiple-indexconfigs.xml", "schema12.xml",
36                      "indexConfig");
37    }
38    public void testMultipleCFS() throws Exception {
39        assertConfigs("bad-solrconfig-multiple-cfs.xml", "schema12.xml",
40                      "useCompoundFile");
41    }
42  
43    public void testUpdateLogButNoVersionField() throws Exception {
44      
45      System.setProperty("enable.update.log", "true");
46      try {
47        assertConfigs("solrconfig.xml", "schema12.xml", "_version_");
48      } finally {
49        System.clearProperty("enable.update.log");
50      }
51    }
52  
53    public void testBogusScriptEngine() throws Exception {
54      // sanity check
55      Assume.assumeTrue(null == (new ScriptEngineManager()).getEngineByName("giberish"));
56                        
57      assertConfigs("bad-solrconfig-bogus-scriptengine-name.xml",
58                    "schema.xml","giberish");
59    }
60  
61    public void testMissingScriptFile() throws Exception {
62      // sanity check
63      Assume.assumeNotNull((new ScriptEngineManager()).getEngineByExtension("js"));
64      assertConfigs("bad-solrconfig-missing-scriptfile.xml",
65                    "schema.xml","a-file-name-that-does-not-exist.js");
66    }
67  
68    public void testInvalidScriptFile() throws Exception {
69      // sanity check
70      Assume.assumeNotNull((new ScriptEngineManager()).getEngineByName("javascript"));
71      assertConfigs("bad-solrconfig-invalid-scriptfile.xml",
72                    "schema.xml","currency.xml");
73    }
74  
75    public void testBogusMergePolicy() throws Exception {
76      assertConfigs("bad-mp-solrconfig.xml", "schema-minimal.xml",
77                    "DummyMergePolicy");
78    }
79  
80    public void testSchemaMutableButNotManaged() throws Exception {
81      assertConfigs("bad-solrconfig-schema-mutable-but-not-managed.xml",
82                    "schema-minimal.xml", "Unexpected arg(s): {mutable=false,managedSchemaResourceName=schema.xml}");
83    }
84  
85    public void testManagedSchemaCannotBeNamedSchemaDotXml() throws Exception {
86      assertConfigs("bad-solrconfig-managed-schema-named-schema.xml.xml",
87                    "schema-minimal.xml", "managedSchemaResourceName can't be 'schema.xml'");
88    }
89    
90    public void testUnknownSchemaAttribute() throws Exception {
91      assertConfigs("bad-solrconfig-unexpected-schema-attribute.xml", "schema-minimal.xml",
92                    "Unexpected arg(s): {bogusParam=bogusValue}");
93    }
94  }
