43  /**
44   * LBHttpSolrClient or "LoadBalanced HttpSolrClient" is a load balancing wrapper around
45   * {@link HttpSolrClient}. This is useful when you
46   * have multiple Solr servers and the requests need to be Load Balanced among them.
47   *
48   * Do <b>NOT</b> use this class for indexing in master/slave scenarios since documents must be sent to the
49   * correct master; no inter-node routing is done.
50   *
51   * In SolrCloud (leader/replica) scenarios, it is usually better to use
52   * {@link CloudSolrClient}, but this class may be used
53   * for updates because the server will forward them to the appropriate leader.
54   *
55   * <p>
56   * It offers automatic failover when a server goes down and it detects when the server comes back up.
57   * <p>
58   * Load balancing is done using a simple round-robin on the list of servers.
59   * <p>
60   * If a request to a server fails by an IOException due to a connection timeout or read timeout then the host is taken
61   * off the list of live servers and moved to a 'dead server list' and the request is resent to the next live server.
62   * This process is continued till it tries all the live servers. If at least one server is alive, the request succeeds,
63   * and if not it fails.
64   * <blockquote><pre>
65   * SolrClient lbHttpSolrClient = new LBHttpSolrClient("http://host1:8080/solr/", "http://host2:8080/solr", "http://host2:8080/solr");
66   * //or if you wish to pass the HttpClient do as follows
67   * httpClient httpClient = new HttpClient();
68   * SolrClient lbHttpSolrClient = new LBHttpSolrClient(httpClient, "http://host1:8080/solr/", "http://host2:8080/solr", "http://host2:8080/solr");
69   * </pre></blockquote>
70   * This detects if a dead server comes alive automatically. The check is done in fixed intervals in a dedicated thread.
71   * This interval can be set using {@link #setAliveCheckInterval} , the default is set to one minute.
72   * <p>
73   * <b>When to use this?</b><br> This can be used as a software load balancer when you do not wish to setup an external
74   * load balancer. Alternatives to this code are to use
75   * a dedicated hardware load balancer or using Apache httpd with mod_proxy_balancer as a load balancer. See <a
76   * href="http://en.wikipedia.org/wiki/Load_balancing_(computing)">Load balancing on Wikipedia</a>
77   *
78   * @since solr 1.4
79   */

218   /** The provided httpClient should use a multi-threaded connection manager */  
219   public LBHttpSolrClient(HttpClient httpClient, ResponseParser parser, String... solrServerUrl) {
220     clientIsInternal = (httpClient == null);
221     this.parser = parser;
222     if (httpClient == null) {
223       ModifiableSolrParams params = new ModifiableSolrParams();
224       if (solrServerUrl.length > 1) {
225         // we prefer retrying another server
226         params.set(HttpClientUtil.PROP_USE_RETRY, false);
227       } else {
228         params.set(HttpClientUtil.PROP_USE_RETRY, true);
229       }
230       this.httpClient = HttpClientUtil.createClient(params);
231     } else {
232       this.httpClient = httpClient;
233     }
234     for (String s : solrServerUrl) {
235       ServerWrapper wrapper = new ServerWrapper(makeSolrClient(s));
236       aliveServers.put(wrapper.getKey(), wrapper);
237     }
238     updateAliveList();
239   }

273   /**
274    * Tries to query a live server from the list provided in Req. Servers in the dead pool are skipped.
275    * If a request fails due to an IOException, the server is moved to the dead pool for a certain period of
276    * time, or until a test request on that server succeeds.
277    *
278    * Servers are queried in the exact order given (except servers currently in the dead pool are skipped).
279    * If no live servers from the provided list remain to be tried, a number of previously skipped dead servers will be tried.
280    * Req.getNumDeadServersToTry() controls how many dead servers will be tried.
281    *
282    * If no live servers are found a SolrServerException is thrown.
283    *
284    * @param req contains both the request as well as the list of servers to query
285    *
286    * @return the result of the request
287    *
288    * @throws IOException If there is a low-level I/O error.
289    */
290   public Rsp request(Req req) throws SolrServerException, IOException {
291     Rsp rsp = new Rsp();
292     Exception ex = null;
293     boolean isUpdate = req.request instanceof IsUpdateRequest;
294     List<ServerWrapper> skipped = null;
295 
296     long timeAllowedNano = getTimeAllowedInNanos(req.getRequest());
297     long timeOutTime = System.nanoTime() + timeAllowedNano;
298     for (String serverStr : req.getServers()) {
299       if(isTimeExceeded(timeAllowedNano, timeOutTime)) {
300         break;
301       }
302       
303       serverStr = normalize(serverStr);
304       // if the server is currently a zombie, just skip to the next one
305       ServerWrapper wrapper = zombieServers.get(serverStr);
306       if (wrapper != null) {
307         // System.out.println("ZOMBIE SERVER QUERIED: " + serverStr);
308         final int numDeadServersToTry = req.getNumDeadServersToTry();
309         if (numDeadServersToTry > 0) {
310           if (skipped == null) {
311             skipped = new ArrayList<>(numDeadServersToTry);
312             skipped.add(wrapper);
313           }
314           else if (skipped.size() < numDeadServersToTry) {
315             skipped.add(wrapper);
316           }
317         }
318         continue;
319       }
320       rsp.server = serverStr;
321       try {
322         MDC.put("LBHttpSolrClient.url", serverStr);
323         HttpSolrClient client = makeSolrClient(serverStr);
324 
325         ex = doRequest(client, req, rsp, isUpdate, false, null);
326         if (ex == null) {
327           return rsp; // SUCCESS
328         }
329       } finally {
330         MDC.remove("LBHttpSolrClient.url");
331       }
332     }
333 
334     // try the servers we previously skipped
335     if (skipped != null) {
336       for (ServerWrapper wrapper : skipped) {
337         if(isTimeExceeded(timeAllowedNano, timeOutTime)) {
338           break;
339         }
340 
341         ex = doRequest(wrapper.client, req, rsp, isUpdate, true, wrapper.getKey());
342         if (ex == null) {
343           return rsp; // SUCCESS
344         }
345       }
346     }
347 
348 
349     if (ex == null) {
350       throw new SolrServerException("No live SolrServers available to handle this request");
351     } else {
352       throw new SolrServerException("No live SolrServers available to handle this request:" + zombieServers.keySet(), ex);
353     }
354 
355   }

357   protected Exception addZombie(HttpSolrClient server, Exception e) {
358 
359     ServerWrapper wrapper;
360 
361     wrapper = new ServerWrapper(server);
362     wrapper.standard = false;
363     zombieServers.put(wrapper.getKey(), wrapper);
364     startAliveCheckExecutor();
365     return e;
366   }

368   protected Exception doRequest(HttpSolrClient client, Req req, Rsp rsp, boolean isUpdate,
369       boolean isZombie, String zombieKey) throws SolrServerException, IOException {
370     Exception ex = null;
371     try {
372       rsp.rsp = client.request(req.getRequest(), (String) null);
373       if (isZombie) {
374         zombieServers.remove(zombieKey);
375       }
376     } catch (SolrException e) {
377       // we retry on 404 or 403 or 503 or 500
378       // unless it's an update - then we only retry on connect exception
379       if (!isUpdate && RETRY_CODES.contains(e.code())) {
380         ex = (!isZombie) ? addZombie(client, e) : e;
381       } else {
382         // Server is alive but the request was likely malformed or invalid
383         if (isZombie) {
384           zombieServers.remove(zombieKey);
385         }
386         throw e;
387       }
388     } catch (SocketException e) {
389       if (!isUpdate || e instanceof ConnectException) {
390         ex = (!isZombie) ? addZombie(client, e) : e;
391       } else {
392         throw e;
393       }
394     } catch (SocketTimeoutException e) {
395       if (!isUpdate) {
396         ex = (!isZombie) ? addZombie(client, e) : e;
397       } else {
398         throw e;
399       }
400     } catch (SolrServerException e) {
401       Throwable rootCause = e.getRootCause();
402       if (!isUpdate && rootCause instanceof IOException) {
403         ex = (!isZombie) ? addZombie(client, e) : e;
404       } else if (isUpdate && rootCause instanceof ConnectException) {
405         ex = (!isZombie) ? addZombie(client, e) : e;
406       } else {
407         throw e;
408       }
409     } catch (Exception e) {
410       throw new SolrServerException(e);
411     }
412 
413     return ex;
414   }

416   private void updateAliveList() {
417     synchronized (aliveServers) {
418       aliveServerList = aliveServers.values().toArray(new ServerWrapper[aliveServers.size()]);

422   private ServerWrapper removeFromAlive(String key) {
423     synchronized (aliveServers) {
424       ServerWrapper wrapper = aliveServers.remove(key);
425       if (wrapper != null)
426         updateAliveList();
427       return wrapper;
428     }
429   }

431   private void addToAlive(ServerWrapper wrapper) {
432     synchronized (aliveServers) {
433       ServerWrapper prev = aliveServers.put(wrapper.getKey(), wrapper);
434       // TODO: warn if there was a previous entry?
435       updateAliveList();
436     }
437   }

444   public String removeSolrServer(String server) {
445     try {
446       server = new URL(server).toExternalForm();
447     } catch (MalformedURLException e) {
448       throw new RuntimeException(e);
449     }
450     if (server.endsWith("/")) {
451       server = server.substring(0, server.length() - 1);
452     }
453 
454     // there is a small race condition here - if the server is in the process of being moved between
455     // lists, we could fail to remove it.
456     removeFromAlive(server);
457     zombieServers.remove(server);
458     return null;
459   }

489   /**
490    * Tries to query a live server. A SolrServerException is thrown if all servers are dead.
491    * If the request failed due to IOException then the live server is moved to dead pool and the request is
492    * retried on another live server.  After live servers are exhausted, any servers previously marked as dead
493    * will be tried before failing the request.
494    *
495    * @param request the SolrRequest.
496    *
497    * @return response
498    *
499    * @throws IOException If there is a low-level I/O error.
500    */
501   @Override
502   public NamedList<Object> request(final SolrRequest request, String collection)
503           throws SolrServerException, IOException {
504     Exception ex = null;
505     ServerWrapper[] serverList = aliveServerList;
506     
507     int maxTries = serverList.length;
508     Map<String,ServerWrapper> justFailed = null;
509 
510     long timeAllowedNano = getTimeAllowedInNanos(request);
511     long timeOutTime = System.nanoTime() + timeAllowedNano;
512     for (int attempts=0; attempts<maxTries; attempts++) {
513       if(isTimeExceeded(timeAllowedNano, timeOutTime)) {
514         break;
515       }
516       
517       int count = counter.incrementAndGet() & Integer.MAX_VALUE;
518       ServerWrapper wrapper = serverList[count % serverList.length];
519 
520       try {
521         return wrapper.client.request(request, collection);
522       } catch (SolrException e) {
523         // Server is alive but the request was malformed or invalid
524         throw e;
525       } catch (SolrServerException e) {
526         if (e.getRootCause() instanceof IOException) {
527           ex = e;
528           moveAliveToDead(wrapper);
529           if (justFailed == null) justFailed = new HashMap<>();
530           justFailed.put(wrapper.getKey(), wrapper);
531         } else {
532           throw e;
533         }
534       } catch (Exception e) {
535         throw new SolrServerException(e);
536       }
537     }
538 
539     // try other standard servers that we didn't try just now
540     for (ServerWrapper wrapper : zombieServers.values()) {
541       if(isTimeExceeded(timeAllowedNano, timeOutTime)) {
542         break;
543       }
544       
545       if (wrapper.standard==false || justFailed!=null && justFailed.containsKey(wrapper.getKey())) continue;
546       try {
547         NamedList<Object> rsp = wrapper.client.request(request, collection);
548         // remove from zombie list *before* adding to alive to avoid a race that could lose a server
549         zombieServers.remove(wrapper.getKey());
550         addToAlive(wrapper);
551         return rsp;
552       } catch (SolrException e) {
553         // Server is alive but the request was malformed or invalid
554         throw e;
555       } catch (SolrServerException e) {
556         if (e.getRootCause() instanceof IOException) {
557           ex = e;
558           // still dead
559         } else {
560           throw e;
561         }
562       } catch (Exception e) {
563         throw new SolrServerException(e);
564       }
565     }
566 
567 
568     if (ex == null) {
569       throw new SolrServerException("No live SolrServers available to handle this request");
570     } else {
571       throw new SolrServerException("No live SolrServers available to handle this request", ex);
572     }
573   }

576    * @return time allowed in nanos, returns -1 if no time_allowed is specified.
577    */
578   private long getTimeAllowedInNanos(final SolrRequest req) {
579     SolrParams reqParams = req.getParams();
580     return reqParams == null ? -1 : 
581       TimeUnit.NANOSECONDS.convert(reqParams.getInt(CommonParams.TIME_ALLOWED, -1), TimeUnit.MILLISECONDS);
582   }

584   private boolean isTimeExceeded(long timeAllowedNano, long timeOutTime) {
585     return timeAllowedNano > 0 && System.nanoTime() > timeOutTime;
586   }

588   /**
589    * Takes up one dead server and check for aliveness. The check is done in a roundrobin. Each server is checked for
590    * aliveness once in 'x' millis where x is decided by the setAliveCheckinterval() or it is defaulted to 1 minute
591    *
592    * @param zombieServer a server in the dead pool
593    */
594   private void checkAZombieServer(ServerWrapper zombieServer) {
595     try {
596       QueryResponse resp = zombieServer.client.query(solrQuery);
597       if (resp.getStatus() == 0) {
598         // server has come back up.
599         // make sure to remove from zombies before adding to alive to avoid a race condition
600         // where another thread could mark it down, move it back to zombie, and then we delete
601         // from zombie and lose it forever.
602         ServerWrapper wrapper = zombieServers.remove(zombieServer.getKey());
603         if (wrapper != null) {
604           wrapper.failedPings = 0;
605           if (wrapper.standard) {
606             addToAlive(wrapper);
607           }
608         } else {
609           // something else already moved the server from zombie to alive
610         }
611       }
612     } catch (Exception e) {
613       //Expected. The server is still down.
614       zombieServer.failedPings++;
615 
616       // If the server doesn't belong in the standard set belonging to this load balancer
617       // then simply drop it after a certain number of failed pings.
618       if (!zombieServer.standard && zombieServer.failedPings >= NONSTANDARD_PING_LIMIT) {
619         zombieServers.remove(zombieServer.getKey());
620       }
621     }
622   }

624   private void moveAliveToDead(ServerWrapper wrapper) {
625     wrapper = removeFromAlive(wrapper.getKey());
626     if (wrapper == null)
627       return;  // another thread already detected the failure and removed it
628     zombieServers.put(wrapper.getKey(), wrapper);
629     startAliveCheckExecutor();
630   }

632   private int interval = CHECK_INTERVAL;
633 
634   /**
635    * LBHttpSolrServer keeps pinging the dead servers at fixed interval to find if it is alive. Use this to set that
636    * interval
637    *
638    * @param interval time in milliseconds
639    */
640   public void setAliveCheckInterval(int interval) {
641     if (interval <= 0) {
642       throw new IllegalArgumentException("Alive check interval must be " +
643               "positive, specified value = " + interval);
644     }
645     this.interval = interval;
646   }
647 
648   private void startAliveCheckExecutor() {
649     // double-checked locking, but it's OK because we don't *do* anything with aliveCheckExecutor
650     // if it's not null.
651     if (aliveCheckExecutor == null) {
652       synchronized (this) {
653         if (aliveCheckExecutor == null) {
654           aliveCheckExecutor = Executors.newSingleThreadScheduledExecutor(
655               new SolrjNamedThreadFactory("aliveCheckExecutor"));
656           aliveCheckExecutor.scheduleAtFixedRate(
657                   getAliveCheckRunner(new WeakReference<>(this)),
658                   this.interval, this.interval, TimeUnit.MILLISECONDS);
659         }
660       }
661     }
662   }

664   private static Runnable getAliveCheckRunner(final WeakReference<LBHttpSolrClient> lbRef) {
665     return new Runnable() {
666       @Override
667       public void run() {
668         LBHttpSolrClient lb = lbRef.get();
669         if (lb != null && lb.zombieServers != null) {
670           for (ServerWrapper zombieServer : lb.zombieServers.values()) {
671             lb.checkAZombieServer(zombieServer);
672           }
673         }
674       }
675     };
676   }
