90      try {
91  
92        DirectUpdateHandler2.commitOnClose = false;
93        final Semaphore logReplay = new Semaphore(0);
94        final Semaphore logReplayFinish = new Semaphore(0);
95  
96        UpdateLog.testing_logReplayHook = new Runnable() {
97          @Override
98          public void run() {
99            try {
100             assertTrue(logReplay.tryAcquire(timeout, TimeUnit.SECONDS));
101           } catch (Exception e) {
102             throw new RuntimeException(e);
103           }
104         }
105       };
106 
107       UpdateLog.testing_logReplayFinishHook = new Runnable() {
108         @Override
109         public void run() {
110           logReplayFinish.release();
111         }
112       };
113 
114 
115       clearIndex();
116       assertU(commit());
117 
118       Deque<Long> versions = new ArrayDeque<>();
119       versions.addFirst(addAndGetVersion(sdoc("id", "A1"), null));
120       versions.addFirst(addAndGetVersion(sdoc("id", "A11"), null));
121       versions.addFirst(addAndGetVersion(sdoc("id", "A12"), null));
122       versions.addFirst(deleteByQueryAndGetVersion("id:A11", null));
123       versions.addFirst(addAndGetVersion(sdoc("id", "A13"), null));
124 
125       assertJQ(req("q","*:*"),"/response/numFound==0");
126 
127       assertJQ(req("qt","/get", "getVersions",""+versions.size()) ,"/versions==" + versions);
128 
129       h.close();
130       createCore();
131       // Solr should kick this off now
132       // h.getCore().getUpdateHandler().getUpdateLog().recoverFromLog();
133 
134       // verify that previous close didn't do a commit
135       // recovery should be blocked by our hook
136       assertJQ(req("q","*:*") ,"/response/numFound==0");
137 
138       // make sure we can still access versions after a restart
139       assertJQ(req("qt","/get", "getVersions",""+versions.size()),"/versions==" + versions);
140 
141       // unblock recovery
142       logReplay.release(1000);
143 
144       // make sure we can still access versions during recovery
145       assertJQ(req("qt","/get", "getVersions",""+versions.size()),"/versions==" + versions);
146 
147       // wait until recovery has finished
148       assertTrue(logReplayFinish.tryAcquire(timeout, TimeUnit.SECONDS));
149 
150       assertJQ(req("q","*:*") ,"/response/numFound==3");
151 
152       // make sure we can still access versions after recovery

193   @Test
194   public void testBuffering() throws Exception {
195 
196     DirectUpdateHandler2.commitOnClose = false;
197     final Semaphore logReplay = new Semaphore(0);
198     final Semaphore logReplayFinish = new Semaphore(0);
199 
200     UpdateLog.testing_logReplayHook = new Runnable() {
201       @Override
202       public void run() {
203         try {
204           assertTrue(logReplay.tryAcquire(timeout, TimeUnit.SECONDS));
205         } catch (Exception e) {
206           throw new RuntimeException(e);
207         }
208       }
209     };
210 
211     UpdateLog.testing_logReplayFinishHook = new Runnable() {
212       @Override
213       public void run() {
214         logReplayFinish.release();
215       }
216     };
217 
218 
219     SolrQueryRequest req = req();
220     UpdateHandler uhandler = req.getCore().getUpdateHandler();
221     UpdateLog ulog = uhandler.getUpdateLog();
222 
223     try {
224       clearIndex();
225       assertU(commit());
226 
227       assertEquals(UpdateLog.State.ACTIVE, ulog.getState());
228       ulog.bufferUpdates();
229       assertEquals(UpdateLog.State.BUFFERING, ulog.getState());
230       Future<UpdateLog.RecoveryInfo> rinfoFuture = ulog.applyBufferedUpdates();
231       assertTrue(rinfoFuture == null);
232       assertEquals(UpdateLog.State.ACTIVE, ulog.getState());
233 
234       ulog.bufferUpdates();
235       assertEquals(UpdateLog.State.BUFFERING, ulog.getState());
236 
237       // simulate updates from a leader
238       updateJ(jsonAdd(sdoc("id","B1", "_version_","1010")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
239       updateJ(jsonAdd(sdoc("id","B11", "_version_","1015")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
240       updateJ(jsonDelQ("id:B1 id:B11 id:B2 id:B3"), params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_","-1017"));
241       updateJ(jsonAdd(sdoc("id","B2", "_version_","1020")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
242       updateJ(jsonAdd(sdoc("id","B3", "_version_","1030")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
243       deleteAndGetVersion("B1", params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_","-2010"));
244 
245       assertJQ(req("qt","/get", "getVersions","6")
246           ,"=={'versions':[-2010,1030,1020,-1017,1015,1010]}"
247       );
248 
249       assertU(commit());
250 
251       assertJQ(req("qt","/get", "getVersions","6")
252           ,"=={'versions':[-2010,1030,1020,-1017,1015,1010]}"
253       );
254 
255       // updates should be buffered, so we should not see any results yet.
256       assertJQ(req("q", "*:*")
257           , "/response/numFound==0"
258       );
259 
260       // real-time get should also not show anything (this could change in the future,
261       // but it's currently used for validating version numbers too, so it would
262       // be bad for updates to be visible if we're just buffering.
263       assertJQ(req("qt","/get", "id","B3")
264           ,"=={'doc':null}"
265       );
266 
267 
268       rinfoFuture = ulog.applyBufferedUpdates();
269       assertTrue(rinfoFuture != null);
270 
271       assertEquals(UpdateLog.State.APPLYING_BUFFERED, ulog.getState());
272 
273       logReplay.release(1000);
274 
275       UpdateLog.RecoveryInfo rinfo = rinfoFuture.get();
276       assertEquals(UpdateLog.State.ACTIVE, ulog.getState());
277 
278 
279       assertJQ(req("qt","/get", "getVersions","6")
280           ,"=={'versions':[-2010,1030,1020,-1017,1015,1010]}"
281       );
282 
283 
284       assertJQ(req("q", "*:*")
285           , "/response/numFound==2"
286       );
287 
288       // move back to recovering
289       ulog.bufferUpdates();
290       assertEquals(UpdateLog.State.BUFFERING, ulog.getState());
291 
292       Long ver = getVer(req("qt","/get", "id","B3"));
293       assertEquals(1030L, ver.longValue());
294 
295       // add a reordered doc that shouldn't overwrite one in the index
296       updateJ(jsonAdd(sdoc("id","B3", "_version_","3")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
297 
298       // reorder two buffered updates
299       updateJ(jsonAdd(sdoc("id","B4", "_version_","1040")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
300       deleteAndGetVersion("B4", params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_","-940"));   // this update should not take affect
301       updateJ(jsonAdd(sdoc("id","B6", "_version_","1060")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
302       updateJ(jsonAdd(sdoc("id","B5", "_version_","1050")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
303       updateJ(jsonAdd(sdoc("id","B8", "_version_","1080")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
304 
305       // test that delete by query is at least buffered along with everything else so it will delete the
306       // currently buffered id:8 (even if it doesn't currently support versioning)
307       updateJ("{\"delete\": { \"query\":\"id:B2 OR id:B8\" }}", params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_","-3000"));
308 
309       assertJQ(req("qt","/get", "getVersions","13")
310           ,"=={'versions':[-3000,1080,1050,1060,-940,1040,3,-2010,1030,1020,-1017,1015,1010]}"  // the "3" appears because versions aren't checked while buffering
311       );
312 
313       logReplay.drainPermits();
314       rinfoFuture = ulog.applyBufferedUpdates();
315       assertTrue(rinfoFuture != null);
316       assertEquals(UpdateLog.State.APPLYING_BUFFERED, ulog.getState());
317 
318       // apply a single update
319       logReplay.release(1);
320 
321       // now add another update
322       updateJ(jsonAdd(sdoc("id","B7", "_version_","1070")), params(DISTRIB_UPDATE_PARAM,FROM_LEADER));
323 
324       // a reordered update that should be dropped
325       deleteAndGetVersion("B5", params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_","-950"));
326 
327       deleteAndGetVersion("B6", params(DISTRIB_UPDATE_PARAM,FROM_LEADER, "_version_","-2060"));
328 
329       logReplay.release(1000);
330       UpdateLog.RecoveryInfo recInfo = rinfoFuture.get();
331 
332       assertJQ(req("q", "*:*", "sort","id asc", "fl","id,_version_")
333           , "/response/docs==["
334                            + "{'id':'B3','_version_':1030}"
335                            + ",{'id':'B4','_version_':1040}"
336                            + ",{'id':'B5','_version_':1050}"
337                            + ",{'id':'B7','_version_':1070}"
338                            +"]"
339       );
340 
341       assertEquals(1, recInfo.deleteByQuery);
342 
343       assertEquals(UpdateLog.State.ACTIVE, ulog.getState()); // leave each test method in a good state
344     } finally {
345       DirectUpdateHandler2.commitOnClose = true;
346       UpdateLog.testing_logReplayHook = null;
347       UpdateLog.testing_logReplayFinishHook = null;
348 
349       req().close();
350     }
351 
352   }

441           + ",{'id':'C105','_version_':205}"
442           + ",{'id':'C106','_version_':206}"
443           +"]"
444       );

1072   // NOTE: replacement must currently be same size
1073   private static void findReplace(byte[] from, byte[] to, byte[] data) {
1074     int idx = -from.length;
1075     for(;;) {
1076       idx = indexOf(from, data, idx + from.length);  // skip over previous match
1077       if (idx < 0) break;
1078       for (int i=0; i<to.length; i++) {
1079         data[idx+i] = to[i];
1080       }
1081     }
1082   }
1083   
1084   private static int indexOf(byte[] target, byte[] data, int start) {
1085     outer: for (int i=start; i<data.length - target.length; i++) {
1086       for (int j=0; j<target.length; j++) {
1087         if (data[i+j] != target[j]) continue outer;
1088       }
1089       return i;
1090     }
1091     return -1;
1092   }

1116   private static Long getVer(SolrQueryRequest req) throws Exception {
1117     String response = JQ(req);
1118     Map rsp = (Map) ObjectBuilder.fromJSON(response);
1119     Map doc = null;
1120     if (rsp.containsKey("doc")) {
1121       doc = (Map)rsp.get("doc");
1122     } else if (rsp.containsKey("docs")) {
1123       List lst = (List)rsp.get("docs");
1124       if (lst.size() > 0) {
1125         doc = (Map)lst.get(0);
1126       }
1127     } else if (rsp.containsKey("response")) {
1128       Map responseMap = (Map)rsp.get("response");
1129       List lst = (List)responseMap.get("docs");
1130       if (lst.size() > 0) {
1131         doc = (Map)lst.get(0);
1132       }
1133     }
1134 
1135     if (doc == null) return null;
1136 
1137     return (Long)doc.get("_version_");
