283   List<String> getAlternatives(String field) {
284     int idx = field.lastIndexOf("_");
285     if (idx<=0 || idx>=field.length()) return Collections.singletonList(field);
286     String suffix = field.substring(idx);
287     String[] alternativeSuffixes = suffixMap.get(suffix);
288     if (alternativeSuffixes == null) return Collections.singletonList(field);
289     String base = field.substring(0, idx);
290     List<String> out = new ArrayList<>(alternativeSuffixes.length);
291     for (String altS : alternativeSuffixes) {
292       out.add( base + altS );
293     }
294     Collections.shuffle(out, random());
295     return out;
296   }
