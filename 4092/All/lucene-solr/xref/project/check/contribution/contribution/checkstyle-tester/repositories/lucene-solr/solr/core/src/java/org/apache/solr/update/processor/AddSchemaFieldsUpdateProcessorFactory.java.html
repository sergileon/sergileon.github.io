54  /**
55   * <p>
56   * This processor will dynamically add fields to the schema if an input document contains
57   * one or more fields that don't match any field or dynamic field in the schema.
58   * </p>
59   * <p>
60   * By default, this processor selects all fields that don't match a schema field or
61   * dynamic field.  The "fieldName" and "fieldRegex" selectors may be specified to further
62   * restrict the selected fields, but the other selectors ("typeName", "typeClass", and
63   * "fieldNameMatchesSchemaField") may not be specified.
64   * </p>
65   * <p>
66   * This processor is configured to map from each field's values' class(es) to the schema
67   * field type that will be used when adding the new field to the schema.  All new fields
68   * are then added to the schema in a single batch.  If schema addition fails for any
69   * field, addition is re-attempted only for those that donâ€™t match any schema
70   * field.  This process is repeated, either until all new fields are successfully added,
71   * or until there are no new fields (presumably because the fields that were new when
72   * this processor started its work were subsequently added by a different update
73   * request, possibly on a different node).
74   * </p>
75   * <p>
76   * This processor takes as configuration a sequence of zero or more "typeMapping"-s from
77   * one or more "valueClass"-s, specified as either an &lt;arr&gt; of &lt;str&gt;, or
78   * multiple &lt;str&gt; with the same name, to an existing schema "fieldType".
79   * </p>
80   * <p>
81   * If more than one "valueClass" is specified in a "typeMapping", field values with any
82   * of the specified "valueClass"-s will be mapped to the specified target "fieldType".
83   * The "typeMapping"-s are attempted in the specified order; if a field value's class
84   * is not specified in a "valueClass", the next "typeMapping" is attempted. If no
85   * "typeMapping" succeeds, then the specified "defaultFieldType" is used. 
86   * </p>
87   * <p>
88   * Example configuration:
89   * </p>
90   * 
91   * <pre class="prettyprint">
92   * &lt;processor class="solr.AddSchemaFieldsUpdateProcessorFactory"&gt;
93   *   &lt;str name="defaultFieldType"&gt;text_general&lt;/str&gt;
94   *   &lt;lst name="typeMapping"&gt;
95   *     &lt;str name="valueClass"&gt;Boolean&lt;/str&gt;
96   *     &lt;str name="fieldType"&gt;boolean&lt;/str&gt;
97   *   &lt;/lst&gt;
98   *   &lt;lst name="typeMapping"&gt;
99   *     &lt;str name="valueClass"&gt;Integer&lt;/str&gt;
100  *     &lt;str name="fieldType"&gt;tint&lt;/str&gt;
101  *   &lt;/lst&gt;
102  *   &lt;lst name="typeMapping"&gt;
103  *     &lt;str name="valueClass"&gt;Float&lt;/str&gt;
104  *     &lt;str name="fieldType"&gt;tfloat&lt;/str&gt;
105  *   &lt;/lst&gt;
106  *   &lt;lst name="typeMapping"&gt;
107  *     &lt;str name="valueClass"&gt;Date&lt;/str&gt;
108  *     &lt;str name="fieldType"&gt;tdate&lt;/str&gt;
109  *   &lt;/lst&gt;
110  *   &lt;lst name="typeMapping"&gt;
111  *     &lt;str name="valueClass"&gt;Long&lt;/str&gt;
112  *     &lt;str name="valueClass"&gt;Integer&lt;/str&gt;
113  *     &lt;str name="fieldType"&gt;tlong&lt;/str&gt;
114  *   &lt;/lst&gt;
115  *   &lt;lst name="typeMapping"&gt;
116  *     &lt;arr name="valueClass"&gt;
117  *       &lt;str&gt;Double&lt;/str&gt;
118  *       &lt;str&gt;Float&lt;/str&gt;
119  *     &lt;/arr&gt;
120  *     &lt;str name="fieldType"&gt;tdouble&lt;/str&gt;
121  *   &lt;/lst&gt;
122  * &lt;/processor&gt;</pre>
123  */
124 public class AddSchemaFieldsUpdateProcessorFactory extends UpdateRequestProcessorFactory implements SolrCoreAware {
125   private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
126 
127   private static final String TYPE_MAPPING_PARAM = "typeMapping";
128   private static final String VALUE_CLASS_PARAM = "valueClass";
129   private static final String FIELD_TYPE_PARAM = "fieldType";
130   private static final String DEFAULT_FIELD_TYPE_PARAM = "defaultFieldType";
131   
132   private List<TypeMapping> typeMappings = Collections.emptyList();
133   private SelectorParams inclusions = new SelectorParams();
134   private Collection<SelectorParams> exclusions = new ArrayList<>();
135   private SolrResourceLoader solrResourceLoader = null;
136   private String defaultFieldType;
137 
138   @Override
139   public UpdateRequestProcessor getInstance(SolrQueryRequest req, 
140                                             SolrQueryResponse rsp, 
141                                             UpdateRequestProcessor next) {
142     return new AddSchemaFieldsUpdateProcessor(next);
143   }
144 
145   @Override
146   public void init(NamedList args) {
147     inclusions = FieldMutatingUpdateProcessorFactory.parseSelectorParams(args);
148     validateSelectorParams(inclusions);
149     inclusions.fieldNameMatchesSchemaField = false;  // Explicitly (non-configurably) require unknown field names
150     exclusions = FieldMutatingUpdateProcessorFactory.parseSelectorExclusionParams(args);
151     for (SelectorParams exclusion : exclusions) {
152       validateSelectorParams(exclusion);
153     }
154     Object defaultFieldTypeParam = args.remove(DEFAULT_FIELD_TYPE_PARAM);
155     if (null == defaultFieldTypeParam) {
156       throw new SolrException(SERVER_ERROR, "Missing required init param '" + DEFAULT_FIELD_TYPE_PARAM + "'");
157     } else {
158       if ( ! (defaultFieldTypeParam instanceof CharSequence)) {
159         throw new SolrException(SERVER_ERROR, "Init param '" + DEFAULT_FIELD_TYPE_PARAM + "' must be a <str>");
160       }
161     }
162     defaultFieldType = defaultFieldTypeParam.toString();
163 
164     typeMappings = parseTypeMappings(args);
165 
166     super.init(args);
167   }
168 
169   @Override
170   public void inform(SolrCore core) {
171     solrResourceLoader = core.getResourceLoader();
172 
173     for (TypeMapping typeMapping : typeMappings) {
174       typeMapping.populateValueClasses(core);
175     }
176   }
177 
178   private static List<TypeMapping> parseTypeMappings(NamedList args) {
179     List<TypeMapping> typeMappings = new ArrayList<>();
180     List<Object> typeMappingsParams = args.getAll(TYPE_MAPPING_PARAM);
181     for (Object typeMappingObj : typeMappingsParams) {
182       if (null == typeMappingObj) {
183         throw new SolrException(SERVER_ERROR, "'" + TYPE_MAPPING_PARAM + "' init param cannot be null");
184       }
185       if ( ! (typeMappingObj instanceof NamedList) ) {
186         throw new SolrException(SERVER_ERROR, "'" + TYPE_MAPPING_PARAM + "' init param must be a <lst>");
187       }
188       NamedList typeMappingNamedList = (NamedList)typeMappingObj;
189 
190       Object fieldTypeObj = typeMappingNamedList.remove(FIELD_TYPE_PARAM);
191       if (null == fieldTypeObj) {
192         throw new SolrException(SERVER_ERROR,
193             "Each '" + TYPE_MAPPING_PARAM + "' <lst/> must contain a '" + FIELD_TYPE_PARAM + "' <str>");
194       }
195       if ( ! (fieldTypeObj instanceof CharSequence)) {
196         throw new SolrException(SERVER_ERROR, "'" + FIELD_TYPE_PARAM + "' init param must be a <str>");
197       }
198       if (null != typeMappingNamedList.get(FIELD_TYPE_PARAM)) {
199         throw new SolrException(SERVER_ERROR,
200             "Each '" + TYPE_MAPPING_PARAM + "' <lst/> may contain only one '" + FIELD_TYPE_PARAM + "' <str>");
201       }
202       String fieldType = fieldTypeObj.toString();
203 
204       Collection<String> valueClasses
205           = typeMappingNamedList.removeConfigArgs(VALUE_CLASS_PARAM);
206       if (valueClasses.isEmpty()) {
207         throw new SolrException(SERVER_ERROR, 
208             "Each '" + TYPE_MAPPING_PARAM + "' <lst/> must contain at least one '" + VALUE_CLASS_PARAM + "' <str>");
209       }
210       typeMappings.add(new TypeMapping(fieldType, valueClasses));
211 
212       if (0 != typeMappingNamedList.size()) {
213         throw new SolrException(SERVER_ERROR, 
214             "Unexpected '" + TYPE_MAPPING_PARAM + "' init sub-param(s): '" + typeMappingNamedList.toString() + "'");
215       }
216       args.remove(TYPE_MAPPING_PARAM);
217     }
218     return typeMappings;
219   }
220 
221   private void validateSelectorParams(SelectorParams params) {
222     if ( ! params.typeName.isEmpty()) {
223       throw new SolrException(SERVER_ERROR, "'typeName' init param is not allowed in this processor");
224     }
225     if ( ! params.typeClass.isEmpty()) {
226       throw new SolrException(SERVER_ERROR, "'typeClass' init param is not allowed in this processor");
227     }
228     if (null != params.fieldNameMatchesSchemaField) {
229       throw new SolrException(SERVER_ERROR, "'fieldNameMatchesSchemaField' init param is not allowed in this processor");
230     }
231   }
232 
233   private static class TypeMapping {
234     public String fieldTypeName;
235     public Collection<String> valueClassNames;
236     public Set<Class<?>> valueClasses;
237 
238     public TypeMapping(String fieldTypeName, Collection<String> valueClassNames) {
239       this.fieldTypeName = fieldTypeName;
240       this.valueClassNames = valueClassNames;
241       // this.valueClasses population is delayed until the schema is available
242     }
243 
244     public void populateValueClasses(SolrCore core) {
245       IndexSchema schema = core.getLatestSchema();
246       ClassLoader loader = core.getResourceLoader().getClassLoader();
247       if (null == schema.getFieldTypeByName(fieldTypeName)) {
248         throw new SolrException(SERVER_ERROR, "fieldType '" + fieldTypeName + "' not found in the schema");
249       }
250       valueClasses = new HashSet<>();
251       for (String valueClassName : valueClassNames) {
252         try {
253           valueClasses.add(loader.loadClass(valueClassName));
254         } catch (ClassNotFoundException e) {
255           throw new SolrException(SERVER_ERROR,
256               "valueClass '" + valueClassName + "' not found for fieldType '" + fieldTypeName + "'");
257         }
258       }
259     }
260   }

262   private class AddSchemaFieldsUpdateProcessor extends UpdateRequestProcessor {
263     public AddSchemaFieldsUpdateProcessor(UpdateRequestProcessor next) {
264       super(next);
265     }
266     
267     @Override
268     public void processAdd(AddUpdateCommand cmd) throws IOException {
269       if ( ! cmd.getReq().getSchema().isMutable()) {
270         final String message = "This IndexSchema is not mutable.";
271         throw new SolrException(BAD_REQUEST, message);
272       }
273       final SolrInputDocument doc = cmd.getSolrInputDocument();
274       final SolrCore core = cmd.getReq().getCore();
275       // use the cmd's schema rather than the latest, because the schema
276       // can be updated during processing.  Using the cmd's schema guarantees
277       // this will be detected and the cmd's schema updated.
278       IndexSchema oldSchema = cmd.getReq().getSchema();
279       for (;;) {
280         List<SchemaField> newFields = new ArrayList<>();
281         // build a selector each time through the loop b/c the schema we are
282         // processing may have changed
283         FieldNameSelector selector = buildSelector(oldSchema);
284         Map<String,List<SolrInputField>> unknownFields = new HashMap<>();
285         getUnknownFields(selector, doc, unknownFields);
286         for (final Map.Entry<String,List<SolrInputField>> entry : unknownFields.entrySet()) {
287           String fieldName = entry.getKey();
288           String fieldTypeName = mapValueClassesToFieldType(entry.getValue());
289           newFields.add(oldSchema.newField(fieldName, fieldTypeName, Collections.<String,Object>emptyMap()));
290         }
291         if (newFields.isEmpty()) {
292           // nothing to do - no fields will be added - exit from the retry loop
293           log.debug("No fields to add to the schema.");
294           break;
295         } else if ( isImmutableConfigSet(core) ) {
296           final String message = "This ConfigSet is immutable.";
297           throw new SolrException(BAD_REQUEST, message);
298         }
299         if (log.isDebugEnabled()) {
300           StringBuilder builder = new StringBuilder();
301           builder.append("Fields to be added to the schema: [");
302           boolean isFirst = true;
303           for (SchemaField field : newFields) {
304             builder.append(isFirst ? "" : ",");
305             isFirst = false;
306             builder.append(field.getName());
307             builder.append("{type=").append(field.getType().getTypeName()).append("}");
308           }
309           builder.append("]");
310           log.debug(builder.toString());
311         }
312         // Need to hold the lock during the entire attempt to ensure that
313         // the schema on the request is the latest
314         synchronized (oldSchema.getSchemaUpdateLock()) {
315           try {
316             IndexSchema newSchema = oldSchema.addFields(newFields);
317             if (null != newSchema) {
318               core.setLatestSchema(newSchema);
319               cmd.getReq().updateSchemaToLatest();
320               log.debug("Successfully added field(s) to the schema.");
321               break; // success - exit from the retry loop
322             } else {
323               throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Failed to add fields.");
324             }
325           } catch (ManagedIndexSchema.FieldExistsException e) {
326             log.error("At least one field to be added already exists in the schema - retrying.");
327             oldSchema = core.getLatestSchema();
328             cmd.getReq().updateSchemaToLatest();
329           } catch (ManagedIndexSchema.SchemaChangedInZkException e) {
330             log.debug("Schema changed while processing request - retrying.");
331             oldSchema = core.getLatestSchema();
332             cmd.getReq().updateSchemaToLatest();
333           }
334         }
335       }
336       super.processAdd(cmd);
337     }

339     /**
340      * Recursively find unknown fields in the given doc and its child documents, if any.
341      */
342     private void getUnknownFields
343     (FieldNameSelector selector, SolrInputDocument doc, Map<String,List<SolrInputField>> unknownFields) {
344       for (final String fieldName : doc.getFieldNames()) {
345         if (selector.shouldMutate(fieldName)) { // returns false if the field already exists in the current schema
346           List<SolrInputField> solrInputFields = unknownFields.get(fieldName);
347           if (null == solrInputFields) {
348             solrInputFields = new ArrayList<>();
349             unknownFields.put(fieldName, solrInputFields);
350           }
351           solrInputFields.add(doc.getField(fieldName));
352         }
353       }
354       List<SolrInputDocument> childDocs = doc.getChildDocuments();
355       if (null != childDocs) {
356         for (SolrInputDocument childDoc : childDocs) {
357           getUnknownFields(selector, childDoc, unknownFields);
358         }
359       }
360     }

362     /**
363      * Maps all given field values' classes to a field type using the configured type mapping rules.
364      * 
365      * @param fields one or more (same-named) field values from one or more documents
366      */
367     private String mapValueClassesToFieldType(List<SolrInputField> fields) {
368       NEXT_TYPE_MAPPING: for (TypeMapping typeMapping : typeMappings) {
369         for (SolrInputField field : fields) {
370           NEXT_FIELD_VALUE: for (Object fieldValue : field.getValues()) {
371             for (Class<?> valueClass : typeMapping.valueClasses) {
372               if (valueClass.isInstance(fieldValue)) {
373                 continue NEXT_FIELD_VALUE;
374               }
375             }
376             // This fieldValue is not an instance of any of the mapped valueClass-s,
377             // so mapping fails - go try the next type mapping.
378             continue NEXT_TYPE_MAPPING;
379           }
380         }
381         // Success! Each of this field's values is an instance of a mapped valueClass
382         return typeMapping.fieldTypeName;
383       }
384       // At least one of this field's values is not an instance of any of the mapped valueClass-s
385       return defaultFieldType;
386     }
387 
388     private FieldNameSelector getDefaultSelector(final IndexSchema schema) {
389       return new FieldNameSelector() {
390         @Override
391         public boolean shouldMutate(final String fieldName) {
392           return null == schema.getFieldTypeNoEx(fieldName);
393         }
394       };
395     }
396 
397     private FieldNameSelector buildSelector(IndexSchema schema) {
398       FieldNameSelector selector = FieldMutatingUpdateProcessor.createFieldNameSelector
399         (solrResourceLoader, schema, inclusions, getDefaultSelector(schema));
400 
401       for (SelectorParams exc : exclusions) {
402         selector = FieldMutatingUpdateProcessor.wrap(selector, FieldMutatingUpdateProcessor.createFieldNameSelector
403           (solrResourceLoader, schema, exc, FieldMutatingUpdateProcessor.SELECT_NO_FIELDS));
404       }
405       return selector;
406     }
407 
408     private boolean isImmutableConfigSet(SolrCore core) {
409       NamedList args = core.getConfigSetProperties();
410       Object immutable = args != null ? args.get(IMMUTABLE_CONFIGSET_ARG) : null;
411       return immutable != null ? Boolean.parseBoolean(immutable.toString()) : false;
412     }
