17  package org.apache.solr.search;
18  
19  import org.apache.lucene.search.Query;
20  import org.apache.solr.common.params.CommonParams;
21  import org.apache.solr.common.params.SolrParams;
22  import org.apache.solr.request.SolrQueryRequest;
23  
24  /**
25   * @see LuceneQParserPlugin
26   */
27  public class LuceneQParser extends QParser {
28    SolrQueryParser lparser;
29  
30    public LuceneQParser(String qstr, SolrParams localParams, SolrParams params, SolrQueryRequest req) {
31      super(qstr, localParams, params, req);
32    }
33  
34  
35    @Override
36    public Query parse() throws SyntaxError {
37      String qstr = getString();
38      if (qstr == null || qstr.length()==0) return null;
39  
40      String defaultField = getParam(CommonParams.DF);
41      if (defaultField==null) {
42        defaultField = getReq().getSchema().getDefaultSearchFieldName();
43      }
44      lparser = new SolrQueryParser(this, defaultField);
45  
46      lparser.setDefaultOperator
47        (QueryParsing.getQueryParserDefaultOperator(getReq().getSchema(),
48                                                    getParam(QueryParsing.OP)));
49  
50      return lparser.parse(qstr);
51    }
52  
53  
54    @Override
55    public String[] getDefaultHighlightFields() {
56      return lparser == null ? new String[]{} : new String[]{lparser.getDefaultField()};
57    }
58  
59  }
