20  import java.io.IOException;
21  
22  import org.apache.lucene.codecs.FieldsConsumer;
23  import org.apache.lucene.codecs.FieldsProducer;
24  import org.apache.lucene.codecs.PostingsFormat;
25  import org.apache.lucene.codecs.PostingsReaderBase;
26  import org.apache.lucene.codecs.PostingsWriterBase;
27  import org.apache.lucene.codecs.blocktree.BlockTreeTermsWriter;
28  import org.apache.lucene.codecs.lucene50.Lucene50PostingsReader;
29  import org.apache.lucene.codecs.lucene50.Lucene50PostingsWriter;
30  import org.apache.lucene.index.SegmentReadState;
31  import org.apache.lucene.index.SegmentWriteState;
32  import org.apache.lucene.util.IOUtils;
33  
34  /** Uses {@link OrdsBlockTreeTermsWriter} with {@link Lucene50PostingsWriter}. */
35  public class BlockTreeOrdsPostingsFormat extends PostingsFormat {
36  
37    private final int minTermBlockSize;
38    private final int maxTermBlockSize;
39  
40    /**
41     * Fixed packed block size, number of integers encoded in 
42     * a single packed block.
43     */
44    // NOTE: must be multiple of 64 because of PackedInts long-aligned encoding/decoding
45    public final static int BLOCK_SIZE = 128;
46  
47    /** Creates {@code Lucene41PostingsFormat} with default
48     *  settings. */
49    public BlockTreeOrdsPostingsFormat() {
50      this(OrdsBlockTreeTermsWriter.DEFAULT_MIN_BLOCK_SIZE, OrdsBlockTreeTermsWriter.DEFAULT_MAX_BLOCK_SIZE);
51    }
52  
53    /** Creates {@code Lucene41PostingsFormat} with custom
54     *  values for {@code minBlockSize} and {@code
55     *  maxBlockSize} passed to block terms dictionary.
56     *  @see OrdsBlockTreeTermsWriter#OrdsBlockTreeTermsWriter(SegmentWriteState,PostingsWriterBase,int,int) */
57    public BlockTreeOrdsPostingsFormat(int minTermBlockSize, int maxTermBlockSize) {
58      super("BlockTreeOrds");
59      this.minTermBlockSize = minTermBlockSize;
60      this.maxTermBlockSize = maxTermBlockSize;
61      BlockTreeTermsWriter.validateSettings(minTermBlockSize, maxTermBlockSize);
62    }
63  
64    @Override
65    public String toString() {
66      return getName() + "(blocksize=" + BLOCK_SIZE + ")";
67    }
68  
69    @Override
70    public FieldsConsumer fieldsConsumer(SegmentWriteState state) throws IOException {
71      PostingsWriterBase postingsWriter = new Lucene50PostingsWriter(state);
72  
73      boolean success = false;
74      try {
75        FieldsConsumer ret = new OrdsBlockTreeTermsWriter(state, 
76                                                          postingsWriter,
77                                                          minTermBlockSize, 
78                                                          maxTermBlockSize);
79        success = true;
80        return ret;
81      } finally {
82        if (!success) {
83          IOUtils.closeWhileHandlingException(postingsWriter);
84        }
85      }
86    }
87  
88    @Override
89    public FieldsProducer fieldsProducer(SegmentReadState state) throws IOException {
90      PostingsReaderBase postingsReader = new Lucene50PostingsReader(state);
91      boolean success = false;
92      try {
93        FieldsProducer ret = new OrdsBlockTreeTermsReader(postingsReader, state);
94        success = true;
95        return ret;
96      } finally {
97        if (!success) {
98          IOUtils.closeWhileHandlingException(postingsReader);
99        }
100     }
101   }
102 }
