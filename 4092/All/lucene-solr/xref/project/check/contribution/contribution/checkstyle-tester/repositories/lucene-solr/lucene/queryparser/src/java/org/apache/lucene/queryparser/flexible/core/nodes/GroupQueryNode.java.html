20  import java.util.ArrayList;
21  import java.util.List;
22  
23  import org.apache.lucene.queryparser.flexible.messages.MessageImpl;
24  import org.apache.lucene.queryparser.flexible.core.QueryNodeError;
25  import org.apache.lucene.queryparser.flexible.core.messages.QueryParserMessages;
26  import org.apache.lucene.queryparser.flexible.core.parser.EscapeQuerySyntax;
27  
28  /**
29   * A {@link GroupQueryNode} represents a location where the original user typed
30   * real parenthesis on the query string. This class is useful for queries like:
31   * a) a AND b OR c b) ( a AND b) OR c
32   * 
33   * Parenthesis might be used to define the boolean operation precedence.
34   */
35  public class GroupQueryNode extends QueryNodeImpl {
36  
37    /**
38     * This QueryNode is used to identify parenthesis on the original query string
39     */
40    public GroupQueryNode(QueryNode query) {
41      if (query == null) {
42        throw new QueryNodeError(new MessageImpl(
43            QueryParserMessages.PARAMETER_VALUE_NOT_SUPPORTED, "query", "null"));
44      }
45  
46      allocate();
47      setLeaf(false);
48      add(query);
49    }
50  
51    public QueryNode getChild() {
52      return getChildren().get(0);
53    }
54  
55    @Override
56    public String toString() {
57      return "<group>" + "\n" + getChild().toString() + "\n</group>";
58    }
59  
60    @Override
61    public CharSequence toQueryString(EscapeQuerySyntax escapeSyntaxParser) {
62      if (getChild() == null)
63        return "";
64  
65      return "( " + getChild().toQueryString(escapeSyntaxParser) + " )";
66    }
67  
68    @Override
69    public QueryNode cloneTree() throws CloneNotSupportedException {
70      GroupQueryNode clone = (GroupQueryNode) super.cloneTree();
71  
72      return clone;
73    }
74  
75    public void setChild(QueryNode child) {
76      List<QueryNode> list = new ArrayList<>();
77      list.add(child);
78      this.set(list);
79    }
80  
81  }
