44  /**
45   * Identifies the language of a set of input fields.
46   * Also supports mapping of field names based
47   * on detected language.
48   * <p>
49   * See <a href="http://wiki.apache.org/solr/LanguageDetection">http://wiki.apache.org/solr/LanguageDetection</a>
50   * @since 3.5
51   * @lucene.experimental
52   */
53  public abstract class LanguageIdentifierUpdateProcessor extends UpdateRequestProcessor implements LangIdParams {
54  
55    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
56  
57    protected boolean enabled;
58  
59    protected String[] inputFields = {};
60    protected String[] mapFields = {};
61    protected Pattern mapPattern;
62    protected String mapReplaceStr;
63    protected String langField;
64    protected String langsField; // MultiValued, contains all languages detected
65    protected String docIdField;
66    protected String fallbackValue;
67    protected String[] fallbackFields = {};
68    protected boolean enableMapping;
69    protected boolean mapKeepOrig;
70    protected boolean overwrite;
71    protected boolean mapOverwrite;
72    protected boolean mapIndividual;
73    protected boolean enforceSchema;
74    protected double threshold;
75    protected HashSet<String> langWhitelist;
76    protected HashSet<String> mapIndividualFieldsSet;
77    protected HashSet<String> allMapFieldsSet;
78    protected HashMap<String,String> lcMap;
79    protected HashMap<String,String> mapLcMap;
80    protected IndexSchema schema;
81    protected int maxFieldValueChars;
82    protected int maxTotalChars;
83  
84    // Regex patterns
85    protected final Pattern tikaSimilarityPattern = Pattern.compile(".*\\((.*?)\\)");
86    protected final Pattern langPattern = Pattern.compile("\\{lang\\}");

88    public LanguageIdentifierUpdateProcessor(SolrQueryRequest req,
89                                             SolrQueryResponse rsp, UpdateRequestProcessor next) {
90      super(next);
91      schema = req.getSchema();
92  
93      initParams(req.getParams());
94    }
95  
96    private void initParams(SolrParams params) {
97      if (params != null) {
98        // Document-centric langId params
99        setEnabled(params.getBool(LANGUAGE_ID, true));
100       if(params.get(FIELDS_PARAM, "").length() > 0) {
101         inputFields = params.get(FIELDS_PARAM, "").split(",");
102       }
103       langField = params.get(LANG_FIELD, DOCID_LANGFIELD_DEFAULT);
104       langsField = params.get(LANGS_FIELD, DOCID_LANGSFIELD_DEFAULT);
105       SchemaField uniqueKeyField = schema.getUniqueKeyField();
106       docIdField = params.get(DOCID_PARAM, uniqueKeyField == null ? DOCID_FIELD_DEFAULT : uniqueKeyField.getName());
107       fallbackValue = params.get(FALLBACK);
108       if(params.get(FALLBACK_FIELDS, "").length() > 0) {
109         fallbackFields = params.get(FALLBACK_FIELDS).split(",");
110       }
111       overwrite = params.getBool(OVERWRITE, false);
112       langWhitelist = new HashSet<>();
113       threshold = params.getDouble(THRESHOLD, DOCID_THRESHOLD_DEFAULT);
114       if(params.get(LANG_WHITELIST, "").length() > 0) {
115         for(String lang : params.get(LANG_WHITELIST, "").split(",")) {
116           langWhitelist.add(lang);
117         }
118       }
119 
120       // Mapping params (field centric)
121       enableMapping = params.getBool(MAP_ENABLE, false);
122       if(params.get(MAP_FL, "").length() > 0) {
123         mapFields = params.get(MAP_FL, "").split(",");
124       } else {
125         mapFields = inputFields;
126       }
127       mapKeepOrig = params.getBool(MAP_KEEP_ORIG, false);
128       mapOverwrite = params.getBool(MAP_OVERWRITE, false);
129       mapIndividual = params.getBool(MAP_INDIVIDUAL, false);
130 
131       // Process individual fields
132       String[] mapIndividualFields = {};
133       if(params.get(MAP_INDIVIDUAL_FL, "").length() > 0) {
134         mapIndividualFields = params.get(MAP_INDIVIDUAL_FL, "").split(",");
135       } else {
136         mapIndividualFields = mapFields;
137       }
138       mapIndividualFieldsSet = new HashSet<>(Arrays.asList(mapIndividualFields));
139       // Compile a union of the lists of fields to map
140       allMapFieldsSet = new HashSet<>(Arrays.asList(mapFields));
141       if(Arrays.equals(mapFields, mapIndividualFields)) {
142         allMapFieldsSet.addAll(mapIndividualFieldsSet);
143       }
144 
145       // Normalize detected langcode onto normalized langcode
146       lcMap = new HashMap<>();
147       if(params.get(LCMAP) != null) {
148         for(String mapping : params.get(LCMAP).split("[, ]")) {
149           String[] keyVal = mapping.split(":");
150           if(keyVal.length == 2) {
151             lcMap.put(keyVal[0], keyVal[1]);
152           } else {
153             log.error("Unsupported format for langid.lcmap: "+mapping+". Skipping this mapping.");
154           }
155         }
156       }
157 
158       // Language Code mapping
159       mapLcMap = new HashMap<>();
160       if(params.get(MAP_LCMAP) != null) {
161         for(String mapping : params.get(MAP_LCMAP).split("[, ]")) {
162           String[] keyVal = mapping.split(":");
163           if(keyVal.length == 2) {
164             mapLcMap.put(keyVal[0], keyVal[1]);
165           } else {
166             log.error("Unsupported format for langid.map.lcmap: "+mapping+". Skipping this mapping.");
167           }
168         }
169       }
170       enforceSchema = params.getBool(ENFORCE_SCHEMA, true);
171 
172       mapPattern = Pattern.compile(params.get(MAP_PATTERN, MAP_PATTERN_DEFAULT));
173       mapReplaceStr = params.get(MAP_REPLACE, MAP_REPLACE_DEFAULT);
174       maxFieldValueChars = params.getInt(MAX_FIELD_VALUE_CHARS, MAX_FIELD_VALUE_CHARS_DEFAULT);
175       maxTotalChars = params.getInt(MAX_TOTAL_CHARS, MAX_TOTAL_CHARS_DEFAULT);
176       if (maxFieldValueChars > maxTotalChars) {
177         if (maxTotalChars == MAX_TOTAL_CHARS_DEFAULT) {
178           // If the user specified only maxFieldValueChars, make maxTotalChars the same as it
179           log.warn(MAX_FIELD_VALUE_CHARS + " (" + maxFieldValueChars + ") is less than " + MAX_TOTAL_CHARS + " ("
180               + maxTotalChars + ").  Setting " + MAX_TOTAL_CHARS + " to " + maxFieldValueChars + ".");
181           maxTotalChars = maxFieldValueChars;
182         } else {
183           // If the user specified maxTotalChars, make maxFieldValueChars the same as it
184           log.warn(MAX_FIELD_VALUE_CHARS + " (" + maxFieldValueChars + ") is less than " + MAX_TOTAL_CHARS + " ("
185               + maxTotalChars + ").  Setting " + MAX_FIELD_VALUE_CHARS + " to " + maxTotalChars + ".");
186           maxFieldValueChars = maxTotalChars;
187         }
188       }
189     }
190     log.debug("LangId configured");
191 
192 
193     if (inputFields.length == 0) {
194       throw new SolrException(ErrorCode.BAD_REQUEST,
195               "Missing or faulty configuration of LanguageIdentifierUpdateProcessor. Input fields must be specified as a comma separated list");
196     }
197 
198   }

200   @Override
201   public void processAdd(AddUpdateCommand cmd) throws IOException {
202     if (isEnabled()) {
203       process(cmd.getSolrInputDocument());
204     } else {
205       log.debug("Processor not enabled, not running");
206     }
207     super.processAdd(cmd);
208   }

210   /**
211    * This is the main, testable process method called from processAdd()
212    * @param doc the SolrInputDocument to work on
213    * @return the modified SolrInputDocument
214    */
215   protected SolrInputDocument process(SolrInputDocument doc) {
216     String docLang = null;
217     HashSet<String> docLangs = new HashSet<>();
218     String fallbackLang = getFallbackLang(doc, fallbackFields, fallbackValue);
219 
220     if(langField == null || !doc.containsKey(langField) || (doc.containsKey(langField) && overwrite)) {
221       List<DetectedLanguage> languagelist = detectLanguage(doc);
222       docLang = resolveLanguage(languagelist, fallbackLang);
223       docLangs.add(docLang);
224       log.debug("Detected main document language from fields "+ Arrays.toString(inputFields) +": "+docLang);
225 
226       if(doc.containsKey(langField) && overwrite) {
227         log.debug("Overwritten old value "+doc.getFieldValue(langField));
228       }
229       if(langField != null && langField.length() != 0) {
230         doc.setField(langField, docLang);
231       }
232     } else {
233       // langField is set, we sanity check it against whitelist and fallback
234       docLang = resolveLanguage((String) doc.getFieldValue(langField), fallbackLang);
235       docLangs.add(docLang);
236       log.debug("Field "+langField+" already contained value "+docLang+", not overwriting.");
237     }
238 
239     if(enableMapping) {
240       for (String fieldName : allMapFieldsSet) {
241         if(doc.containsKey(fieldName)) {
242           String fieldLang;
243           if(mapIndividual && mapIndividualFieldsSet.contains(fieldName)) {
244             List<DetectedLanguage> languagelist = detectLanguage(doc);
245             fieldLang = resolveLanguage(languagelist, docLang);
246             docLangs.add(fieldLang);
247             log.debug("Mapping field "+fieldName+" using individually detected language "+fieldLang);
248           } else {
249             fieldLang = docLang;
250             log.debug("Mapping field "+fieldName+" using document global language "+fieldLang);
251           }
252           String mappedOutputField = getMappedField(fieldName, fieldLang);
253 
254           if (mappedOutputField != null) {
255             log.debug("Mapping field {} to {}", doc.getFieldValue(docIdField), fieldLang);
256             SolrInputField inField = doc.getField(fieldName);
257             doc.setField(mappedOutputField, inField.getValue(), inField.getBoost());
258             if(!mapKeepOrig) {
259               log.debug("Removing old field {}", fieldName);
260               doc.removeField(fieldName);
261             }
262           } else {
263             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Invalid output field mapping for "
264                     + fieldName + " field and language: " + fieldLang);
265           }
266         }
267       }
268     }
269 
270     // Set the languages field to an array of all detected languages
271     if(langsField != null && langsField.length() != 0) {
272       doc.setField(langsField, docLangs.toArray());
273     }
274 
275     return doc;
276   }

278   /**
279    * Decides the fallback language, either from content of fallback field or fallback value
280    * @param doc the Solr document
281    * @param fallbackFields an array of strings with field names containing fallback language codes
282    * @param fallbackValue a language code to use in case no fallbackFields are found
283    */
284   private String getFallbackLang(SolrInputDocument doc, String[] fallbackFields, String fallbackValue) {
285     String lang = null;
286     for(String field : fallbackFields) {
287       if(doc.containsKey(field)) {
288         lang = (String) doc.getFieldValue(field);
289         log.debug("Language fallback to field "+field);
290         break;
291       }
292     }
293     if(lang == null) {
294       log.debug("Language fallback to value "+fallbackValue);
295       lang = fallbackValue;
296     }
297     return lang;
298   }

300   /**
301    * Detects language(s) from a string.
302    * Classes wishing to implement their own language detection module should override this method.
303    * @param content The content to identify
304    * @return List of detected language(s) according to RFC-3066
305    */
306   protected abstract List<DetectedLanguage> detectLanguage(SolrInputDocument content);
307 
308   /**
309    * Chooses a language based on the list of candidates detected
310    * @param language language code as a string
311    * @param fallbackLang the language code to use as a fallback
312    * @return a string of the chosen language
313    */
314   protected String resolveLanguage(String language, String fallbackLang) {
315     List<DetectedLanguage> l = new ArrayList<>();
316     l.add(new DetectedLanguage(language, 1.0));
317     return resolveLanguage(l, fallbackLang);
318   }
319 
320   /**
321    * Chooses a language based on the list of candidates detected
322    * @param languages a List of DetectedLanguages with certainty score
323    * @param fallbackLang the language code to use as a fallback
324    * @return a string of the chosen language
325    */
326   protected String resolveLanguage(List<DetectedLanguage> languages, String fallbackLang) {
327     String langStr;
328     if(languages.size() == 0) {
329       log.debug("No language detected, using fallback {}", fallbackLang);
330       langStr = fallbackLang;
331     } else {
332       DetectedLanguage lang = languages.get(0);
333       String normalizedLang = normalizeLangCode(lang.getLangCode());
334       if(langWhitelist.isEmpty() || langWhitelist.contains(normalizedLang)) {
335         log.debug("Language detected {} with certainty {}", normalizedLang, lang.getCertainty());
336         if(lang.getCertainty() >= threshold) {
337           langStr = normalizedLang;
338         } else {
339           log.debug("Detected language below threshold {}, using fallback {}", threshold, fallbackLang);
340           langStr = fallbackLang;
341         }
342       } else {
343         log.debug("Detected a language not in whitelist ({}), using fallback {}", lang.getLangCode(), fallbackLang);
344         langStr = fallbackLang;
345       }
346     }
347 
348     if(langStr == null || langStr.length() == 0) {
349       log.warn("Language resolved to null or empty string. Fallback not configured?");
350       langStr = "";
351     }
352 
353     return langStr;
354   }

356   /**
357    * Looks up language code in map (langid.lcmap) and returns mapped value
358    * @param langCode the language code string returned from detector
359    * @return the normalized/mapped language code
360    */
361   protected String normalizeLangCode(String langCode) {
362     if (lcMap.containsKey(langCode)) {
363       String lc = lcMap.get(langCode);
364       log.debug("Doing langcode normalization mapping from "+langCode+" to "+lc);
365       return lc;
366     }
367     return langCode;
368   }

370   /**
371    * Returns the name of the field to map the current contents into, so that they are properly analyzed.  For instance
372    * if the currentField is "text" and the code is "en", the new field would by default be "text_en".
373    * This method also performs custom regex pattern replace if configured. If enforceSchema=true
374    * and the resulting field name doesn't exist, then null is returned.
375    *
376    * @param currentField The current field name
377    * @param language the language code
378    * @return The new schema field name, based on pattern and replace, or null if illegal
379    */
380   protected String getMappedField(String currentField, String language) {
381     String lc = mapLcMap.containsKey(language) ? mapLcMap.get(language) : language;
382     String newFieldName = langPattern.matcher(mapPattern.matcher(currentField).replaceFirst(mapReplaceStr)).replaceFirst(lc);
383     if(enforceSchema && schema.getFieldOrNull(newFieldName) == null) {
384       log.warn("Unsuccessful field name mapping from {} to {}, field does not exist and enforceSchema=true; skipping mapping.", currentField, newFieldName);
385       return null;
386     } else {
387       log.debug("Doing mapping from "+currentField+" with language "+language+" to field "+newFieldName);
388     }
389     return newFieldName;
390   }
