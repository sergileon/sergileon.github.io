55  package org.egothor.stemmer;
56  
57  /**
58   * The Optimizer class is a Trie that will be reduced (have empty rows removed).
59   * <p>
60   * This is the result of allowing a joining of rows when there is no collision
61   * between non-<tt>null</tt> values in the rows. Information loss, resulting in
62   * the stemmer not being able to recognize words (as in Optimizer), is
63   * curtailed, allowing the stemmer to recognize words for which the original
64   * trie was built. Use of this class allows the stemmer to be self-teaching.
65   */
66  public class Optimizer2 extends Optimizer {
67    /**
68     * Constructor for the Optimizer2 object.
69     */
70    public Optimizer2() {}
71    
72    /**
73     * Merge the given Cells and return the resulting Cell.
74     * 
75     * @param m the master Cell
76     * @param e the existing Cell
77     * @return the resulting Cell, or <tt>null</tt> if the operation cannot be
78     *         realized
79     */
80    @Override
81    public Cell merge(Cell m, Cell e) {
82      if (m.cmd == e.cmd && m.ref == e.ref && m.skip == e.skip) {
83        Cell c = new Cell(m);
84        c.cnt += e.cnt;
85        return c;
86      } else {
87        return null;
88      }
89    }
90  }
