58    private IndexWriter newWriter(Directory dir, IndexWriterConfig conf)
59        throws IOException {
60      LogDocMergePolicy logByteSizeMergePolicy = new LogDocMergePolicy();
61      logByteSizeMergePolicy.setNoCFSRatio(0.0); // make sure we use plain
62      // files
63      conf.setMergePolicy(logByteSizeMergePolicy);
64  
65      final IndexWriter writer = new IndexWriter(dir, conf);
66      return writer;
67    }

69    private void addDocs(IndexWriter writer, int numDocs) throws IOException {
70      for (int i = 0; i < numDocs; i++) {
71        Document doc = new Document();
72        doc.add(newTextField("content", "aaa", Field.Store.NO));
73        writer.addDocument(doc);
74      }
75    }

98    public void testMergeUnusedPerFieldCodec() throws IOException {
99      Directory dir = newDirectory();
100     IndexWriterConfig iwconf = newIndexWriterConfig(new MockAnalyzer(random()))
101                                  .setOpenMode(OpenMode.CREATE).setCodec(new MockCodec());
102     IndexWriter writer = newWriter(dir, iwconf);
103     addDocs(writer, 10);
104     writer.commit();
105     addDocs3(writer, 10);
106     writer.commit();
107     addDocs2(writer, 10);
108     writer.commit();
109     assertEquals(30, writer.maxDoc());
110     TestUtil.checkIndex(dir);
111     writer.forceMerge(1);
112     assertEquals(30, writer.maxDoc());
113     writer.close();
114     dir.close();
115   }

204     final PostingsFormat luceneDefault = TestUtil.getDefaultPostingsFormat();

215         return luceneDefault;

234   /*
235    * Test per field codec support - adding fields with random codecs
236    */
237   @Test
238   public void testStressPerFieldCodec() throws IOException {
239     Directory dir = newDirectory(random());
240     final int docsPerRound = 97;
241     int numRounds = atLeast(1);
242     for (int i = 0; i < numRounds; i++) {
243       int num = TestUtil.nextInt(random(), 30, 60);
244       IndexWriterConfig config = newIndexWriterConfig(random(),
245           new MockAnalyzer(random()));
246       config.setOpenMode(OpenMode.CREATE_OR_APPEND);
247       IndexWriter writer = newWriter(dir, config);
248       for (int j = 0; j < docsPerRound; j++) {
249         final Document doc = new Document();
250         for (int k = 0; k < num; k++) {
251           FieldType customType = new FieldType(TextField.TYPE_NOT_STORED);
252           customType.setTokenized(random().nextBoolean());
253           customType.setOmitNorms(random().nextBoolean());
254           Field field = newField("" + k, TestUtil
255               .randomRealisticUnicodeString(random(), 128), customType);
256           doc.add(field);
257         }
258         writer.addDocument(doc);
259       }
260       if (random().nextBoolean()) {
261         writer.forceMerge(1);
262       }
263       writer.commit();
264       assertEquals((i + 1) * docsPerRound, writer.maxDoc());
265       writer.close();
266     }
267     dir.close();
268   }

270   public void testSameCodecDifferentInstance() throws Exception {
271     Codec codec = new AssertingCodec() {
272       @Override
273       public PostingsFormat getPostingsFormatForField(String field) {
274         if ("id".equals(field)) {
275           return new MemoryPostingsFormat();
276         } else if ("date".equals(field)) {
277           return new MemoryPostingsFormat();
278         } else {
279           return super.getPostingsFormatForField(field);
280         }
281       }
282     };
283     doTestMixedPostings(codec);
284   }

286   public void testSameCodecDifferentParams() throws Exception {
287     Codec codec = new AssertingCodec() {
288       @Override
289       public PostingsFormat getPostingsFormatForField(String field) {
290         if ("id".equals(field)) {
291           return new LuceneVarGapFixedInterval(1);
292         } else if ("date".equals(field)) {
293           return new LuceneVarGapFixedInterval(2);
294         } else {
295           return super.getPostingsFormatForField(field);
296         }
297       }
298     };
299     doTestMixedPostings(codec);
300   }
301   
302   private void doTestMixedPostings(Codec codec) throws Exception {
303     Directory dir = newDirectory();
304     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
305     iwc.setCodec(codec);
306     RandomIndexWriter iw = new RandomIndexWriter(random(), dir, iwc);
307     Document doc = new Document();
308     FieldType ft = new FieldType(TextField.TYPE_NOT_STORED);
309     // turn on vectors for the checkindex cross-check
310     ft.setStoreTermVectors(true);
311     ft.setStoreTermVectorOffsets(true);
312     ft.setStoreTermVectorPositions(true);
313     Field idField = new Field("id", "", ft);
314     Field dateField = new Field("date", "", ft);
315     doc.add(idField);
316     doc.add(dateField);
317     for (int i = 0; i < 100; i++) {
318       idField.setStringValue(Integer.toString(random().nextInt(50)));
319       dateField.setStringValue(Integer.toString(random().nextInt(100)));
320       iw.addDocument(doc);
321     }
322     iw.close();
323     dir.close(); // checkindex
324   }
