26  /**
27   * GeoShape representing a path across the surface of the globe,
28   * with a specified half-width.  Path is described by a series of points.
29   * Distances are measured from the starting point along the path, and then at right
30   * angles to the path.
31   *
32   * @lucene.experimental
33   */
34  public class GeoPath extends GeoBaseDistanceShape {
35    /** The cutoff angle (width) */
36    protected final double cutoffAngle;
37  
38    /** Sine of cutoff angle */
39    protected final double sinAngle;
40    /** Cosine of cutoff angle */
41    protected final double cosAngle;
42  
43    /** The original list of path points */
44    protected final List<GeoPoint> points = new ArrayList<GeoPoint>();
45    
46    /** A list of SegmentEndpoints */
47    protected List<SegmentEndpoint> endPoints;
48    /** A list of PathSegments */
49    protected List<PathSegment> segments;
50  
51    /** A point on the edge */
52    protected GeoPoint[] edgePoints;
53  
54    /** Set to true if path has been completely constructed */
55    protected boolean isDone = false;
56    
57    /** Constructor.
58     *@param planetModel is the planet model.
59     *@param maxCutoffAngle is the width of the path, measured as an angle.
60     *@param pathPoints are the points in the path.
61     */
62    public GeoPath(final PlanetModel planetModel, final double maxCutoffAngle, final GeoPoint[] pathPoints) {
63      this(planetModel, maxCutoffAngle);
64      Collections.addAll(points, pathPoints);
65      done();
66    }
67    
68    /** Piece-wise constructor.  Use in conjunction with addPoint() and done().
69     *@param planetModel is the planet model.
70     *@param maxCutoffAngle is the width of the path, measured as an angle.
71     */
72    public GeoPath(final PlanetModel planetModel, final double maxCutoffAngle) {
73      super(planetModel);
74      if (maxCutoffAngle <= 0.0 || maxCutoffAngle > Math.PI * 0.5)
75        throw new IllegalArgumentException("Cutoff angle out of bounds");
76      this.cutoffAngle = maxCutoffAngle;
77      this.cosAngle = Math.cos(maxCutoffAngle);
78      this.sinAngle = Math.sin(maxCutoffAngle);
79    }
80  
81    /** Add a point to the path.
82     *@param lat is the latitude of the point.
83     *@param lon is the longitude of the point.
84     */
85    public void addPoint(final double lat, final double lon) {
86      if (isDone)
87        throw new IllegalStateException("Can't call addPoint() if done() already called");
88      points.add(new GeoPoint(planetModel, lat, lon));
89    }
90    
91    /** Complete the path.
92     */
93    public void done() {
94      if (isDone)
95        throw new IllegalStateException("Can't call done() twice");
96      if (points.size() == 0)
97        throw new IllegalArgumentException("Path must have at least one point");
98      isDone = true;
99  
100     endPoints = new ArrayList<>(points.size());
101     segments = new ArrayList<>(points.size());
102     // Compute an offset to use for all segments.  This will be based on the minimum magnitude of
103     // the entire ellipsoid.
104     final double cutoffOffset = this.sinAngle * planetModel.getMinimumMagnitude();
105     
106     // First, build all segments.  We'll then go back and build corresponding segment endpoints.
107     GeoPoint lastPoint = null;
108     for (final GeoPoint end : points) {
109       if (lastPoint != null) {
110         final Plane normalizedConnectingPlane = new Plane(lastPoint, end);
111         if (normalizedConnectingPlane == null) {
112           continue;
113         }
114         segments.add(new PathSegment(planetModel, lastPoint, end, normalizedConnectingPlane, cutoffOffset));
115       }
116       lastPoint = end;
117     }
118     
119     if (segments.size() == 0) {
120       // Simple circle
121       double lat = points.get(0).getLatitude();
122       double lon = points.get(0).getLongitude();
123       // Compute two points on the circle, with the right angle from the center.  We'll use these
124       // to obtain the perpendicular plane to the circle.
125       double upperLat = lat + cutoffAngle;
126       double upperLon = lon;
127       if (upperLat > Math.PI * 0.5) {
128         upperLon += Math.PI;
129         if (upperLon > Math.PI)
130           upperLon -= 2.0 * Math.PI;
131         upperLat = Math.PI - upperLat;
132       }
133       double lowerLat = lat - cutoffAngle;
134       double lowerLon = lon;
135       if (lowerLat < -Math.PI * 0.5) {
136         lowerLon += Math.PI;
137         if (lowerLon > Math.PI)
138           lowerLon -= 2.0 * Math.PI;
139         lowerLat = -Math.PI - lowerLat;
140       }
141       final GeoPoint upperPoint = new GeoPoint(planetModel, upperLat, upperLon);
142       final GeoPoint lowerPoint = new GeoPoint(planetModel, lowerLat, lowerLon);
143       final GeoPoint point = points.get(0);
144       
145       // Construct normal plane
146       final Plane normalPlane = Plane.constructNormalizedZPlane(upperPoint, lowerPoint, point);
147 
148       final SegmentEndpoint onlyEndpoint = new SegmentEndpoint(point, normalPlane, upperPoint, lowerPoint);
149       endPoints.add(onlyEndpoint);
150       this.edgePoints = new GeoPoint[]{onlyEndpoint.circlePlane.getSampleIntersectionPoint(planetModel, normalPlane)};
151       return;
152     }
153     
154     // Create segment endpoints.  Use an appropriate constructor for the start and end of the path.
155     for (int i = 0; i < segments.size(); i++) {
156       final PathSegment currentSegment = segments.get(i);
157       
158       if (i == 0) {
159         // Starting endpoint
160         final SegmentEndpoint startEndpoint = new SegmentEndpoint(currentSegment.start, 
161           currentSegment.startCutoffPlane, currentSegment.ULHC, currentSegment.LLHC);
162         endPoints.add(startEndpoint);
163         this.edgePoints = new GeoPoint[]{currentSegment.ULHC};
164         continue;
165       }
166       
167       // General intersection case
168       final PathSegment prevSegment = segments.get(i-1);
169       // We construct four separate planes, and evaluate which one includes all interior points with least overlap
170       final SidedPlane candidate1 = SidedPlane.constructNormalizedThreePointSidedPlane(currentSegment.start, prevSegment.URHC, currentSegment.ULHC, currentSegment.LLHC);
171       final SidedPlane candidate2 = SidedPlane.constructNormalizedThreePointSidedPlane(currentSegment.start, currentSegment.ULHC, currentSegment.LLHC, prevSegment.LRHC);
172       final SidedPlane candidate3 = SidedPlane.constructNormalizedThreePointSidedPlane(currentSegment.start, currentSegment.LLHC, prevSegment.LRHC, prevSegment.URHC);
173       final SidedPlane candidate4 = SidedPlane.constructNormalizedThreePointSidedPlane(currentSegment.start, prevSegment.LRHC, prevSegment.URHC, currentSegment.ULHC);
174 
175       if (candidate1 == null && candidate2 == null && candidate3 == null && candidate4 == null) {
176         // The planes are identical.  We wouldn't need a circle at all except for the possibility of
177         // backing up, which is hard to detect here.
178         final SegmentEndpoint midEndpoint = new SegmentEndpoint(currentSegment.start, 
179           prevSegment.endCutoffPlane, currentSegment.startCutoffPlane, currentSegment.ULHC, currentSegment.LLHC);
180         //don't need a circle at all.  Special constructor...
181         endPoints.add(midEndpoint);
182       } else {
183         endPoints.add(new SegmentEndpoint(currentSegment.start,
184           prevSegment.endCutoffPlane, currentSegment.startCutoffPlane,
185           prevSegment.URHC, prevSegment.LRHC,
186           currentSegment.ULHC, currentSegment.LLHC,
187           candidate1, candidate2, candidate3, candidate4));
188       }
189     }
190     // Do final endpoint
191     final PathSegment lastSegment = segments.get(segments.size()-1);
192     endPoints.add(new SegmentEndpoint(lastSegment.end,
193       lastSegment.endCutoffPlane, lastSegment.URHC, lastSegment.LRHC));
194 
195   }

197   @Override
198   protected double distance(final DistanceStyle distanceStyle, final double x, final double y, final double z) {
199     // Algorithm:
200     // (1) If the point is within any of the segments along the path, return that value.
201     // (2) If the point is within any of the segment end circles along the path, return that value.
202     double currentDistance = 0.0;
203     for (PathSegment segment : segments) {
204       double distance = segment.pathDistance(planetModel, distanceStyle, x,y,z);
205       if (distance != Double.MAX_VALUE)
206         return currentDistance + distance;
207       currentDistance += segment.fullPathDistance(distanceStyle);
208     }
209 
210     int segmentIndex = 0;
211     currentDistance = 0.0;
212     for (SegmentEndpoint endpoint : endPoints) {
213       double distance = endpoint.pathDistance(distanceStyle, x, y, z);
214       if (distance != Double.MAX_VALUE)
215         return currentDistance + distance;
216       if (segmentIndex < segments.size())
217         currentDistance += segments.get(segmentIndex++).fullPathDistance(distanceStyle);
218     }
219 
220     return Double.MAX_VALUE;
221   }
222 
223   @Override
224   protected double outsideDistance(final DistanceStyle distanceStyle, final double x, final double y, final double z) {
225     double minDistance = Double.MAX_VALUE;
226     for (final SegmentEndpoint endpoint : endPoints) {
227       final double newDistance = endpoint.outsideDistance(distanceStyle, x,y,z);
228       if (newDistance < minDistance)
229         minDistance = newDistance;
230     }
231     for (final PathSegment segment : segments) {
232       final double newDistance = segment.outsideDistance(planetModel, distanceStyle, x, y, z);
233       if (newDistance < minDistance)
234         minDistance = newDistance;
235     }
236     return minDistance;
237   }

239   @Override
240   public boolean isWithin(final double x, final double y, final double z) {
241     for (SegmentEndpoint pathPoint : endPoints) {
242       if (pathPoint.isWithin(x, y, z))
243         return true;
244     }
245     for (PathSegment pathSegment : segments) {
246       if (pathSegment.isWithin(x, y, z))
247         return true;
248     }
249     return false;
250   }

257   @Override
258   public boolean intersects(final Plane plane, final GeoPoint[] notablePoints, final Membership... bounds) {
259     // We look for an intersection with any of the exterior edges of the path.
260     // We also have to look for intersections with the cones described by the endpoints.
261     // Return "true" if any such intersections are found.
262 
263     // For plane intersections, the basic idea is to come up with an equation of the line that is
264     // the intersection (if any).  Then, find the intersections with the unit sphere (if any).  If
265     // any of the intersection points are within the bounds, then we've detected an intersection.
266     // Well, sort of.  We can detect intersections also due to overlap of segments with each other.
267     // But that's an edge case and we won't be optimizing for it.
268     //System.err.println(" Looking for intersection of plane "+plane+" with path "+this);
269     for (final SegmentEndpoint pathPoint : endPoints) {
270       if (pathPoint.intersects(planetModel, plane, notablePoints, bounds)) {
271         return true;
272       }
273     }
274 
275     for (final PathSegment pathSegment : segments) {
276       if (pathSegment.intersects(planetModel, plane, notablePoints, bounds)) {
277         return true;
278       }
279     }
280 
281     return false;
282   }

284   @Override
285   public void getBounds(Bounds bounds) {
286     super.getBounds(bounds);
287     // For building bounds, order matters.  We want to traverse
288     // never more than 180 degrees longitude at a pop or we risk having the
289     // bounds object get itself inverted.  So do the edges first.
290     for (PathSegment pathSegment : segments) {
291       pathSegment.getBounds(planetModel, bounds);
292     }
293     for (SegmentEndpoint pathPoint : endPoints) {
294       pathPoint.getBounds(planetModel, bounds);
295     }
296   }

298   @Override
299   public boolean equals(Object o) {
300     if (!(o instanceof GeoPath))
301       return false;
302     GeoPath p = (GeoPath) o;
303     if (!super.equals(p))
304       return false;
305     if (cutoffAngle != p.cutoffAngle)
306       return false;
307     return points.equals(p.points);
308   }

310   @Override
311   public int hashCode() {
312     int result = super.hashCode();
313     long temp = Double.doubleToLongBits(cutoffAngle);
314     result = 31 * result + (int) (temp ^ (temp >>> 32));
315     result = 31 * result + points.hashCode();
316     return result;
317   }

324   /**
325    * This is precalculated data for segment endpoint.
326    * Note well: This is not necessarily a circle.  There are four cases:
327    * (1) The path consists of a single endpoint.  In this case, we build a simple circle with the proper cutoff offset.
328    * (2) This is the end of a path.  The circle plane must be constructed to go through two supplied points and be perpendicular to a connecting plane.
329    * (2.5) Intersection, but the path on both sides is linear.  We generate a circle, but we use the cutoff planes to limit its influence in the straight line case.
330    * (3) This is an intersection in a path.  We are supplied FOUR planes.  If there are intersections within bounds for both upper and lower, then
331    *    we generate no circle at all.  If there is one intersection only, then we generate a plane that includes that intersection, as well as the remaining
332    *    cutoff plane/edge plane points.
333    */
334   public static class SegmentEndpoint {
335     /** The center point of the endpoint */
336     public final GeoPoint point;
337     /** A plane describing the circle */
338     public final SidedPlane circlePlane;
339     /** Pertinent cutoff planes from adjoining segments */
340     public final Membership[] cutoffPlanes;
341     /** Notable points for this segment endpoint */
342     public final GeoPoint[] notablePoints;
343     /** No notable points from the circle itself */
344     public final static GeoPoint[] circlePoints = new GeoPoint[0];
345     /** Null membership */
346     public final static Membership[] NO_MEMBERSHIP = new Membership[0];
347     
348     /** Base case.  Does nothing at all.
349      */
350     public SegmentEndpoint(final GeoPoint point) {
351       this.point = point;
352       this.circlePlane = null;
353       this.cutoffPlanes = null;
354       this.notablePoints = null;
355     }
356     
357     /** Constructor for case (1).
358      * Generate a simple circle cutoff plane.
359      *@param point is the center point.
360      *@param upperPoint is a point that must be on the circle plane.
361      *@param lowerPoint is another point that must be on the circle plane.
362      */
363     public SegmentEndpoint(final GeoPoint point, final Plane normalPlane, final GeoPoint upperPoint, final GeoPoint lowerPoint) {
364       this.point = point;
365       // Construct a sided plane that goes through the two points and whose normal is in the normalPlane.
366       this.circlePlane = SidedPlane.constructNormalizedPerpendicularSidedPlane(point, normalPlane, upperPoint, lowerPoint);
367       this.cutoffPlanes = NO_MEMBERSHIP;
368       this.notablePoints = circlePoints;
369     }
370     
371     /** Constructor for case (2).
372      * Generate an endpoint, given a single cutoff plane plus upper and lower edge points.
373      *@param point is the center point.
374      *@param cutoffPlane is the plane from the adjoining path segment marking the boundary between this endpoint and that segment.
375      *@param topEdgePoint is a point on the cutoffPlane that should be also on the circle plane.
376      *@param bottomEdgePoint is another point on the cutoffPlane that should be also on the circle plane.
377      */
378     public SegmentEndpoint(final GeoPoint point,
379       final SidedPlane cutoffPlane, final GeoPoint topEdgePoint, final GeoPoint bottomEdgePoint) {
380       this.point = point;
381       this.cutoffPlanes = new Membership[]{new SidedPlane(cutoffPlane)};
382       this.notablePoints = new GeoPoint[]{topEdgePoint, bottomEdgePoint};
383       // To construct the plane, we now just need D, which is simply the negative of the evaluation of the circle normal vector at one of the points.
384       this.circlePlane = SidedPlane.constructNormalizedPerpendicularSidedPlane(point, cutoffPlane, topEdgePoint, bottomEdgePoint);
385     }
386 
387     /** Constructor for case (2.5).
388      * Generate an endpoint, given two cutoff planes plus upper and lower edge points.
389      *@param point is the center.
390      *@param cutoffPlane1 is one adjoining path segment cutoff plane.
391      *@param cutoffPlane2 is another adjoining path segment cutoff plane.
392      *@param topEdgePoint is a point on the cutoffPlane that should be also on the circle plane.
393      *@param bottomEdgePoint is another point on the cutoffPlane that should be also on the circle plane.
394      */
395     public SegmentEndpoint(final GeoPoint point,
396       final SidedPlane cutoffPlane1, final SidedPlane cutoffPlane2, final GeoPoint topEdgePoint, final GeoPoint bottomEdgePoint) {
397       this.point = point;
398       this.cutoffPlanes = new Membership[]{new SidedPlane(cutoffPlane1), new SidedPlane(cutoffPlane2)};
399       this.notablePoints = new GeoPoint[]{topEdgePoint, bottomEdgePoint};
400       // To construct the plane, we now just need D, which is simply the negative of the evaluation of the circle normal vector at one of the points.
401       this.circlePlane = SidedPlane.constructNormalizedPerpendicularSidedPlane(point, cutoffPlane1, topEdgePoint, bottomEdgePoint);
402     }
403     
404     /** Constructor for case (3).
405      * Generate an endpoint for an intersection, given four points.
406      *@param point is the center.
407      *@param prevCutoffPlane is the previous adjoining segment cutoff plane.
408      *@param nextCutoffPlane is the next path segment cutoff plane.
409      *@param notCand2Point is a point NOT on candidate2.
410      *@param notCand1Point is a point NOT on candidate1.
411      *@param notCand3Point is a point NOT on candidate3.
412      *@param notCand4Point is a point NOT on candidate4.
413      *@param candidate1 one of four candidate circle planes.
414      *@param candidate2 one of four candidate circle planes.
415      *@param candidate3 one of four candidate circle planes.
416      *@param candidate4 one of four candidate circle planes.
417      */
418     public SegmentEndpoint(final GeoPoint point,
419       final SidedPlane prevCutoffPlane, final SidedPlane nextCutoffPlane,
420       final GeoPoint notCand2Point, final GeoPoint notCand1Point,
421       final GeoPoint notCand3Point, final GeoPoint notCand4Point,
422       final SidedPlane candidate1, final SidedPlane candidate2, final SidedPlane candidate3, final SidedPlane candidate4) {
423       // Note: What we really need is a single plane that goes through all four points.
424       // Since that's not possible in the ellipsoid case (because three points determine a plane, not four), we
425       // need an approximation that at least creates a boundary that has no interruptions.
426       // There are three obvious choices for the third point: either (a) one of the two remaining points, or (b) the top or bottom edge
427       // intersection point.  (a) has no guarantee of continuity, while (b) is capable of producing something very far from a circle if
428       // the angle between segments is acute.
429       // The solution is to look for the side (top or bottom) that has an intersection within the shape.  We use the two points from
430       // the opposite side to determine the plane, AND we pick the third to be either of the two points on the intersecting side
431       // PROVIDED that the other point is within the final circle we come up with.
432       this.point = point;
433       
434       // We construct four separate planes, and evaluate which one includes all interior points with least overlap
435       // (Constructed beforehand because we need them for degeneracy check)
436 
437       final boolean cand1IsOtherWithin = candidate1!=null?candidate1.isWithin(notCand1Point):false;
438       final boolean cand2IsOtherWithin = candidate2!=null?candidate2.isWithin(notCand2Point):false;
439       final boolean cand3IsOtherWithin = candidate3!=null?candidate3.isWithin(notCand3Point):false;
440       final boolean cand4IsOtherWithin = candidate4!=null?candidate4.isWithin(notCand4Point):false;
441       
442       if (cand1IsOtherWithin && cand2IsOtherWithin && cand3IsOtherWithin && cand4IsOtherWithin) {
443         // The only way we should see both within is if all four points are coplanar.  In that case, we default to the simplest treatment.
444         this.circlePlane = candidate1;  // doesn't matter which
445         this.notablePoints = new GeoPoint[]{notCand2Point, notCand3Point, notCand1Point, notCand4Point};
446         this.cutoffPlanes = new Membership[]{new SidedPlane(prevCutoffPlane), new SidedPlane(nextCutoffPlane)};
447       } else if (cand1IsOtherWithin) {
448         // Use candidate1, and DON'T include prevCutoffPlane in the cutoff planes list
449         this.circlePlane = candidate1;
450         this.notablePoints = new GeoPoint[]{notCand2Point, notCand3Point, notCand4Point};
451         this.cutoffPlanes = new Membership[]{new SidedPlane(nextCutoffPlane)};
452       } else if (cand2IsOtherWithin) {
453         // Use candidate2
454         this.circlePlane = candidate2;
455         this.notablePoints = new GeoPoint[]{notCand3Point, notCand4Point, notCand1Point};
456         this.cutoffPlanes = new Membership[]{new SidedPlane(nextCutoffPlane)};
457       } else if (cand3IsOtherWithin) {
458         this.circlePlane = candidate3;
459         this.notablePoints = new GeoPoint[]{notCand4Point, notCand1Point, notCand2Point};
460         this.cutoffPlanes = new Membership[]{new SidedPlane(prevCutoffPlane)};
461       } else if (cand4IsOtherWithin) {
462         this.circlePlane = candidate4;
463         this.notablePoints = new GeoPoint[]{notCand1Point, notCand2Point, notCand3Point};
464         this.cutoffPlanes = new Membership[]{new SidedPlane(prevCutoffPlane)};
465       } else {
466         // dunno what happened
467         throw new RuntimeException("Couldn't come up with a plane through three points that included the fourth");
468       }
469     }

471     /** Check if point is within this endpoint.
472      *@param point is the point.
473      *@return true of within.
474      */
475     public boolean isWithin(final Vector point) {
476       if (circlePlane == null)
477         return false;
478       if (!circlePlane.isWithin(point))
479         return false;
480       for (final Membership m : cutoffPlanes) {
481         if (!m.isWithin(point)) {
482           return false;
483         }
484       }
485       return true;
486     }
487 
488     /** Check if point is within this endpoint.
489      *@param x is the point x.
490      *@param y is the point y.
491      *@param z is the point z.
492      *@return true of within.
493      */
494     public boolean isWithin(final double x, final double y, final double z) {
495       if (circlePlane == null)
496         return false;
497       if (!circlePlane.isWithin(x, y, z))
498         return false;
499       for (final Membership m : cutoffPlanes) {
500         if (!m.isWithin(x,y,z)) {
501           return false;
502         }
503       }
504       return true;
505     }

507     /** Compute interior path distance.
508      *@param distanceStyle is the distance style.
509      *@param x is the point x.
510      *@param y is the point y.
511      *@param z is the point z.
512      *@return the distance metric.
513      */
514     public double pathDistance(final DistanceStyle distanceStyle, final double x, final double y, final double z) {
515       if (!isWithin(x,y,z))
516         return Double.MAX_VALUE;
517       return distanceStyle.computeDistance(this.point, x, y, z);
518     }

525      *@return the distance metric.
526      */

528       return distanceStyle.computeDistance(this.point, x, y, z);

531     /** Determine if this endpoint intersects a specified plane.
532      *@param planetModel is the planet model.
533      *@param p is the plane.
534      *@param notablePoints are the points associated with the plane.
535      *@param bounds are any bounds which the intersection must lie within.
536      *@return true if there is a matching intersection.
537      */
538     public boolean intersects(final PlanetModel planetModel, final Plane p, final GeoPoint[] notablePoints, final Membership[] bounds) {
539       //System.err.println("  looking for intersection between plane "+p+" and circle "+circlePlane+" on proper side of "+cutoffPlanes+" within "+bounds);
540       if (circlePlane == null)
541         return false;
542       return circlePlane.intersects(planetModel, p, notablePoints, this.notablePoints, bounds, this.cutoffPlanes);
543     }

545     /** Get the bounds for a segment endpoint.
546      *@param planetModel is the planet model.
547      *@param bounds are the bounds to be modified.

549     public void getBounds(final PlanetModel planetModel, Bounds bounds) {
550       bounds.addPoint(point);
551       if (circlePlane == null)
552         return;
553       bounds.addPlane(planetModel, circlePlane);
554     }

575   /**
576    * This is the pre-calculated data for a path segment.
577    */
578   public static class PathSegment {
579     /** Starting point of the segment */
580     public final GeoPoint start;
581     /** End point of the segment */
582     public final GeoPoint end;
583     /** Place to keep any complete segment distances we've calculated so far */
584     public final Map<DistanceStyle,Double> fullDistanceCache = new HashMap<DistanceStyle,Double>();
585     /** Normalized plane connecting the two points and going through world center */
586     public final Plane normalizedConnectingPlane;
587     /** Cutoff plane parallel to connecting plane representing one side of the path segment */
588     public final SidedPlane upperConnectingPlane;
589     /** Cutoff plane parallel to connecting plane representing the other side of the path segment */
590     public final SidedPlane lowerConnectingPlane;
591     /** Plane going through the center and start point, marking the start edge of the segment */
592     public final SidedPlane startCutoffPlane;
593     /** Plane going through the center and end point, marking the end edge of the segment */
594     public final SidedPlane endCutoffPlane;
595     /** Upper right hand corner of segment */
596     public final GeoPoint URHC;
597     /** Lower right hand corner of segment */
598     public final GeoPoint LRHC;
599     /** Upper left hand corner of segment */
600     public final GeoPoint ULHC;
601     /** Lower left hand corner of segment */
602     public final GeoPoint LLHC;
603     /** Notable points for the upper connecting plane */
604     public final GeoPoint[] upperConnectingPlanePoints;
605     /** Notable points for the lower connecting plane */
606     public final GeoPoint[] lowerConnectingPlanePoints;
607     /** Notable points for the start cutoff plane */
608     public final GeoPoint[] startCutoffPlanePoints;
609     /** Notable points for the end cutoff plane */
610     public final GeoPoint[] endCutoffPlanePoints;
611 
612     /** Construct a path segment.
613      *@param planetModel is the planet model.
614      *@param start is the starting point.
615      *@param end is the ending point.
616      *@param normalizedConnectingPlane is the connecting plane.
617      *@param planeBoundingOffset is the linear offset from the connecting plane to either side.
618      */
619     public PathSegment(final PlanetModel planetModel, final GeoPoint start, final GeoPoint end,
620       final Plane normalizedConnectingPlane, final double planeBoundingOffset) {
621       this.start = start;
622       this.end = end;
623       this.normalizedConnectingPlane = normalizedConnectingPlane;
624         
625       // Either start or end should be on the correct side
626       upperConnectingPlane = new SidedPlane(start, normalizedConnectingPlane, -planeBoundingOffset);
627       lowerConnectingPlane = new SidedPlane(start, normalizedConnectingPlane, planeBoundingOffset);
628       // Cutoff planes use opposite endpoints as correct side examples
629       startCutoffPlane = new SidedPlane(end, normalizedConnectingPlane, start);
630       endCutoffPlane = new SidedPlane(start, normalizedConnectingPlane, end);
631       final Membership[] upperSide = new Membership[]{upperConnectingPlane};
632       final Membership[] lowerSide = new Membership[]{lowerConnectingPlane};
633       final Membership[] startSide = new Membership[]{startCutoffPlane};
634       final Membership[] endSide = new Membership[]{endCutoffPlane};
635       GeoPoint[] points;
636       points = upperConnectingPlane.findIntersections(planetModel, startCutoffPlane, lowerSide, endSide);
637       if (points.length == 0) {
638         throw new IllegalArgumentException("Some segment boundary points are off the ellipsoid; path too wide");
639       }
640       this.ULHC = points[0];
641       points = upperConnectingPlane.findIntersections(planetModel, endCutoffPlane, lowerSide, startSide);
642       if (points.length == 0) {
643         throw new IllegalArgumentException("Some segment boundary points are off the ellipsoid; path too wide");
644       }
645       this.URHC = points[0];
646       points = lowerConnectingPlane.findIntersections(planetModel, startCutoffPlane, upperSide, endSide);
647       if (points.length == 0) {
648         throw new IllegalArgumentException("Some segment boundary points are off the ellipsoid; path too wide");
649       }
650       this.LLHC = points[0];
651       points = lowerConnectingPlane.findIntersections(planetModel, endCutoffPlane, upperSide, startSide);
652       if (points.length == 0) {
653         throw new IllegalArgumentException("Some segment boundary points are off the ellipsoid; path too wide");
654       }
655       this.LRHC = points[0];
656       upperConnectingPlanePoints = new GeoPoint[]{ULHC, URHC};
657       lowerConnectingPlanePoints = new GeoPoint[]{LLHC, LRHC};
658       startCutoffPlanePoints = new GeoPoint[]{ULHC, LLHC};
659       endCutoffPlanePoints = new GeoPoint[]{URHC, LRHC};
660     }
661 
662     /** Compute the full distance along this path segment.
663      *@param distanceStyle is the distance style.
664      *@return the distance metric.
665      */
666     public double fullPathDistance(final DistanceStyle distanceStyle) {
667       synchronized (fullDistanceCache) {
668         Double dist = fullDistanceCache.get(distanceStyle);
669         if (dist == null) {
670           dist = new Double(distanceStyle.computeDistance(start, end.x, end.y, end.z));
671           fullDistanceCache.put(distanceStyle, dist);
672         }
673         return dist.doubleValue();
674       }
675     }
676   
677     /** Check if point is within this segment.
678      *@param point is the point.
679      *@return true of within.
680      */
681     public boolean isWithin(final Vector point) {
682       return startCutoffPlane.isWithin(point) &&
683           endCutoffPlane.isWithin(point) &&
684           upperConnectingPlane.isWithin(point) &&
685           lowerConnectingPlane.isWithin(point);
686     }
687 
688     /** Check if point is within this segment.
689      *@param x is the point x.
690      *@param y is the point y.
691      *@param z is the point z.
692      *@return true of within.
693      */
694     public boolean isWithin(final double x, final double y, final double z) {
695       return startCutoffPlane.isWithin(x, y, z) &&
696           endCutoffPlane.isWithin(x, y, z) &&
697           upperConnectingPlane.isWithin(x, y, z) &&
698           lowerConnectingPlane.isWithin(x, y, z);
699     }
700 
701     /** Compute interior path distance.
702      *@param planetModel is the planet model.
703      *@param distanceStyle is the distance style.
704      *@param x is the point x.
705      *@param y is the point y.
706      *@param z is the point z.
707      *@return the distance metric.
708      */
709     public double pathDistance(final PlanetModel planetModel, final DistanceStyle distanceStyle, final double x, final double y, final double z) {
710       if (!isWithin(x,y,z))
711         return Double.MAX_VALUE;
712 
713       // (1) Compute normalizedPerpPlane.  If degenerate, then return point distance from start to point.
714       // Want no allocations or expensive operations!  so we do this the hard way
715       final double perpX = normalizedConnectingPlane.y * z - normalizedConnectingPlane.z * y;
716       final double perpY = normalizedConnectingPlane.z * x - normalizedConnectingPlane.x * z;
717       final double perpZ = normalizedConnectingPlane.x * y - normalizedConnectingPlane.y * x;
718       final double magnitude = Math.sqrt(perpX * perpX + perpY * perpY + perpZ * perpZ);
719       if (Math.abs(magnitude) < Vector.MINIMUM_RESOLUTION)
720         return distanceStyle.computeDistance(start, x,y,z);
721       final double normFactor = 1.0/magnitude;
722       final Plane normalizedPerpPlane = new Plane(perpX * normFactor, perpY * normFactor, perpZ * normFactor, 0.0);
723       
724       // Old computation: too expensive, because it calculates the intersection point twice.
725       //return distanceStyle.computeDistance(planetModel, normalizedConnectingPlane, x, y, z, startCutoffPlane, endCutoffPlane) +
726       //  distanceStyle.computeDistance(planetModel, normalizedPerpPlane, start.x, start.y, start.z, upperConnectingPlane, lowerConnectingPlane);
727 
728       final GeoPoint[] intersectionPoints = normalizedConnectingPlane.findIntersections(planetModel, normalizedPerpPlane);
729       GeoPoint thePoint;
730       if (intersectionPoints.length == 0)
731         throw new RuntimeException("Can't find world intersection for point x="+x+" y="+y+" z="+z);
732       else if (intersectionPoints.length == 1)
733         thePoint = intersectionPoints[0];
734       else {
735         if (startCutoffPlane.isWithin(intersectionPoints[0]) && endCutoffPlane.isWithin(intersectionPoints[0]))
736           thePoint = intersectionPoints[0];
737         else if (startCutoffPlane.isWithin(intersectionPoints[1]) && endCutoffPlane.isWithin(intersectionPoints[1]))
738           thePoint = intersectionPoints[1];
739         else
740           throw new RuntimeException("Can't find world intersection for point x="+x+" y="+y+" z="+z);
741       }
742       return distanceStyle.computeDistance(thePoint, x, y, z) + distanceStyle.computeDistance(start, thePoint.x, thePoint.y, thePoint.z);
743     }
744 
745     /** Compute external distance.
746      *@param planetModel is the planet model.
747      *@param distanceStyle is the distance style.
748      *@param x is the point x.
749      *@param y is the point y.
750      *@param z is the point z.
751      *@return the distance metric.
752      */
753     public double outsideDistance(final PlanetModel planetModel, final DistanceStyle distanceStyle, final double x, final double y, final double z) {
754       final double upperDistance = distanceStyle.computeDistance(planetModel, upperConnectingPlane, x,y,z, lowerConnectingPlane, startCutoffPlane, endCutoffPlane);
755       final double lowerDistance = distanceStyle.computeDistance(planetModel, lowerConnectingPlane, x,y,z, upperConnectingPlane, startCutoffPlane, endCutoffPlane);
756       final double startDistance = distanceStyle.computeDistance(planetModel, startCutoffPlane, x,y,z, endCutoffPlane, lowerConnectingPlane, upperConnectingPlane);
757       final double endDistance = distanceStyle.computeDistance(planetModel, endCutoffPlane, x,y,z, startCutoffPlane, lowerConnectingPlane, upperConnectingPlane);
758       final double ULHCDistance = distanceStyle.computeDistance(ULHC, x,y,z);
759       final double URHCDistance = distanceStyle.computeDistance(URHC, x,y,z);
760       final double LLHCDistance = distanceStyle.computeDistance(LLHC, x,y,z);
761       final double LRHCDistance = distanceStyle.computeDistance(LRHC, x,y,z);
762       return Math.min(
763         Math.min(
764           Math.min(upperDistance,lowerDistance),
765           Math.min(startDistance,endDistance)),
766         Math.min(
767           Math.min(ULHCDistance, URHCDistance),
768           Math.min(LLHCDistance, LRHCDistance)));
769     }
770 
771     /** Determine if this endpoint intersects a specified plane.
772      *@param planetModel is the planet model.
773      *@param p is the plane.
774      *@param notablePoints are the points associated with the plane.
775      *@param bounds are any bounds which the intersection must lie within.
776      *@return true if there is a matching intersection.
777      */
778     public boolean intersects(final PlanetModel planetModel, final Plane p, final GeoPoint[] notablePoints, final Membership[] bounds) {
779       return upperConnectingPlane.intersects(planetModel, p, notablePoints, upperConnectingPlanePoints, bounds, lowerConnectingPlane, startCutoffPlane, endCutoffPlane) ||
780           lowerConnectingPlane.intersects(planetModel, p, notablePoints, lowerConnectingPlanePoints, bounds, upperConnectingPlane, startCutoffPlane, endCutoffPlane);
781     }
782 
783     /** Get the bounds for a segment endpoint.
784      *@param planetModel is the planet model.
785      *@param bounds are the bounds to be modified.
786      */
787     public void getBounds(final PlanetModel planetModel, Bounds bounds) {
788       // We need to do all bounding planes as well as corner points
789       bounds.addPoint(start).addPoint(end).addPoint(ULHC).addPoint(URHC).addPoint(LRHC).addPoint(LLHC);
790       bounds.addPlane(planetModel, upperConnectingPlane, lowerConnectingPlane, startCutoffPlane, endCutoffPlane);
791       bounds.addPlane(planetModel, lowerConnectingPlane, upperConnectingPlane, startCutoffPlane, endCutoffPlane);
792       bounds.addPlane(planetModel, startCutoffPlane, endCutoffPlane, upperConnectingPlane, lowerConnectingPlane);
793       bounds.addPlane(planetModel, endCutoffPlane, startCutoffPlane, upperConnectingPlane, lowerConnectingPlane);
794     }
