93    private void incompleteOrOverlappingCustomRangeTest() throws Exception  {
94      ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
95      final DocRouter router = clusterState.getCollection(AbstractDistribZkTestBase.DEFAULT_COLLECTION).getRouter();
96      Slice shard1 = clusterState.getSlice(AbstractDistribZkTestBase.DEFAULT_COLLECTION, SHARD1);
97      DocRouter.Range shard1Range = shard1.getRange() != null ? shard1.getRange() : router.fullRange();
98  
99      List<DocRouter.Range> subRanges = new ArrayList<>();
100     List<DocRouter.Range> ranges = router.partitionRange(4, shard1Range);
101 
102     // test with only one range
103     subRanges.add(ranges.get(0));
104     try {
105       splitShard(AbstractDistribZkTestBase.DEFAULT_COLLECTION, SHARD1, subRanges, null);
106       fail("Shard splitting with just one custom hash range should not succeed");
107     } catch (HttpSolrClient.RemoteSolrException e) {
108       log.info("Expected exception:", e);
109     }
110     subRanges.clear();
111 
112     // test with ranges with a hole in between them
113     subRanges.add(ranges.get(3)); // order shouldn't matter
114     subRanges.add(ranges.get(0));
115     try {
116       splitShard(AbstractDistribZkTestBase.DEFAULT_COLLECTION, SHARD1, subRanges, null);
117       fail("Shard splitting with missing hashes in between given ranges should not succeed");
118     } catch (HttpSolrClient.RemoteSolrException e) {
119       log.info("Expected exception:", e);
120     }
121     subRanges.clear();
122 
123     // test with overlapping ranges
124     subRanges.add(ranges.get(0));
125     subRanges.add(ranges.get(1));
126     subRanges.add(ranges.get(2));
127     subRanges.add(new DocRouter.Range(ranges.get(3).min - 15, ranges.get(3).max));
128     try {
129       splitShard(AbstractDistribZkTestBase.DEFAULT_COLLECTION, SHARD1, subRanges, null);
130       fail("Shard splitting with overlapping ranges should not succeed");
131     } catch (HttpSolrClient.RemoteSolrException e) {
132       log.info("Expected exception:", e);
133     }
134     subRanges.clear();
135   }

137   private void splitByUniqueKeyTest() throws Exception {
138     ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
139     final DocRouter router = clusterState.getCollection(AbstractDistribZkTestBase.DEFAULT_COLLECTION).getRouter();
140     Slice shard1 = clusterState.getSlice(AbstractDistribZkTestBase.DEFAULT_COLLECTION, SHARD1);
141     DocRouter.Range shard1Range = shard1.getRange() != null ? shard1.getRange() : router.fullRange();
142     List<DocRouter.Range> subRanges = new ArrayList<>();
143     if (usually())  {
144       List<DocRouter.Range> ranges = router.partitionRange(4, shard1Range);
145       // 75% of range goes to shard1_0 and the rest to shard1_1
146       subRanges.add(new DocRouter.Range(ranges.get(0).min, ranges.get(2).max));
147       subRanges.add(ranges.get(3));
148     } else  {
149       subRanges = router.partitionRange(2, shard1Range);
150     }
151     final List<DocRouter.Range> ranges = subRanges;
152     final int[] docCounts = new int[ranges.size()];
153     int numReplicas = shard1.getReplicas().size();
154 
155     del("*:*");
156     for (int id = 0; id <= 100; id++) {
157       String shardKey = "" + (char)('a' + (id % 26)); // See comment in ShardRoutingTest for hash distribution
158       indexAndUpdateCount(router, ranges, docCounts, shardKey + "!" + String.valueOf(id), id);
159     }
160     commit();
161 
162     Thread indexThread = new Thread() {
163       @Override
164       public void run() {
165         Random random = random();
166         int max = atLeast(random, 401);
167         int sleep = atLeast(random, 25);
168         log.info("SHARDSPLITTEST: Going to add " + max + " number of docs at 1 doc per " + sleep + "ms");
169         Set<String> deleted = new HashSet<>();
170         for (int id = 101; id < max; id++) {
171           try {
172             indexAndUpdateCount(router, ranges, docCounts, String.valueOf(id), id);
173             Thread.sleep(sleep);
174             if (usually(random))  {
175               String delId = String.valueOf(random.nextInt(id - 101 + 1) + 101);
176               if (deleted.contains(delId))  continue;
177               try {
178                 deleteAndUpdateCount(router, ranges, docCounts, delId);
179                 deleted.add(delId);
180               } catch (Exception e) {
181                 log.error("Exception while deleting docs", e);
182               }
183             }
184           } catch (Exception e) {
185             log.error("Exception while adding doc id = " + id, e);
186             // do not select this id for deletion ever
187             deleted.add(String.valueOf(id));
188           }
189         }
190       }
191     };
192     indexThread.start();
193 
194     try {
195       for (int i = 0; i < 3; i++) {
196         try {
197           splitShard(AbstractDistribZkTestBase.DEFAULT_COLLECTION, SHARD1, subRanges, null);
198           log.info("Layout after split: \n");
199           printLayout();
200           break;
201         } catch (HttpSolrClient.RemoteSolrException e) {
202           if (e.code() != 500)  {
203             throw e;
204           }
205           log.error("SPLITSHARD failed. " + (i < 2 ? " Retring split" : ""), e);
206           if (i == 2) {
207             fail("SPLITSHARD was not successful even after three tries");
208           }
209         }
210       }
211     } finally {
212       try {
213         indexThread.join();
214       } catch (InterruptedException e) {
215         log.error("Indexing thread interrupted", e);
216       }
217     }
218 
219     waitForRecoveriesToFinish(true);
220     checkDocCountsAndShardStates(docCounts, numReplicas);
221   }

442   protected void checkSubShardConsistency(String shard) throws SolrServerException, IOException {
443     SolrQuery query = new SolrQuery("*:*").setRows(1000).setFields("id", "_version_");
444     query.set("distrib", false);
445 
446     ClusterState clusterState = cloudClient.getZkStateReader().getClusterState();
447     Slice slice = clusterState.getSlice(AbstractDistribZkTestBase.DEFAULT_COLLECTION, shard);
448     long[] numFound = new long[slice.getReplicasMap().size()];
449     int c = 0;
450     for (Replica replica : slice.getReplicas()) {
451       String coreUrl = new ZkCoreNodeProps(replica).getCoreUrl();
452       QueryResponse response;
453       try (HttpSolrClient client = new HttpSolrClient(coreUrl)) {
454         response = client.query(query);
455       }
456       numFound[c++] = response.getResults().getNumFound();
457       log.info("Shard: " + shard + " Replica: {} has {} docs", coreUrl, String.valueOf(response.getResults().getNumFound()));
458       assertTrue("Shard: " + shard + " Replica: " + coreUrl + " has 0 docs", response.getResults().getNumFound() > 0);
459     }
460     for (int i = 0; i < slice.getReplicasMap().size(); i++) {
461       assertEquals(shard + " is not consistent", numFound[0], numFound[i]);
462     }
463   }

465   protected void splitShard(String collection, String shardId, List<DocRouter.Range> subRanges, String splitKey) throws SolrServerException, IOException {
466     ModifiableSolrParams params = new ModifiableSolrParams();
467     params.set("action", CollectionParams.CollectionAction.SPLITSHARD.toString());
468     params.set("collection", collection);
469     if (shardId != null)  {
470       params.set("shard", shardId);
471     }
472     if (subRanges != null)  {
473       StringBuilder ranges = new StringBuilder();
474       for (int i = 0; i < subRanges.size(); i++) {
475         DocRouter.Range subRange = subRanges.get(i);
476         ranges.append(subRange.toString());
477         if (i < subRanges.size() - 1)
478           ranges.append(",");
479       }
480       params.set("ranges", ranges.toString());
481     }
482     if (splitKey != null) {
483       params.set("split.key", splitKey);
484     }
485     SolrRequest request = new QueryRequest(params);
486     request.setPath("/admin/collections");
487 
488     String baseUrl = ((HttpSolrClient) shardToJetty.get(SHARD1).get(0).client.solrClient)
489         .getBaseURL();
490     baseUrl = baseUrl.substring(0, baseUrl.length() - "collection1".length());
491 
492     try (HttpSolrClient baseServer = new HttpSolrClient(baseUrl)) {
493       baseServer.setConnectionTimeout(30000);
494       baseServer.setSoTimeout(60000 * 5);
495       baseServer.request(request);
496     }
497   }

503     if (idx != -1)  {
504       docCounts[idx]++;
505     }

508   protected void deleteAndUpdateCount(DocRouter router, List<DocRouter.Range> ranges, int[] docCounts, String id) throws Exception {
509     controlClient.deleteById(id);
510     cloudClient.deleteById(id);
511 
512     int idx = getHashRangeIdx(router, ranges, id);
513     if (idx != -1)  {
514       docCounts[idx]--;
515     }
516   }

518   public static int getHashRangeIdx(DocRouter router, List<DocRouter.Range> ranges, String id) {
519     int hash = 0;
520     if (router instanceof HashBasedRouter) {
521       HashBasedRouter hashBasedRouter = (HashBasedRouter) router;
522       hash = hashBasedRouter.sliceHash(id, null, null,null);
523     }
524     for (int i = 0; i < ranges.size(); i++) {
525       DocRouter.Range range = ranges.get(i);
526       if (range.includes(hash))
527         return i;
528     }
529     return -1;
530   }

541     Map<String, SolrDocument> shard10Docs = new HashMap<>();
542     Map<String, SolrDocument> shard11Docs = new HashMap<>();
543     for (int i = 0; i < response.getResults().size(); i++) {

547       if (old != null) {

554       String version = idVsVersion.get(value);

561       }
