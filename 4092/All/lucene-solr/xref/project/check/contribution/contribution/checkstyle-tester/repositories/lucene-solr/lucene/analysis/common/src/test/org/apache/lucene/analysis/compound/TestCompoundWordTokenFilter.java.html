221   public void testReset() throws Exception {
222     CharArraySet dict = makeDictionary("Rind", "Fleisch", "Draht", "Schere", "Gesetz",
223         "Aufgabe", "Überwachung");
224 
225     MockTokenizer wsTokenizer = new MockTokenizer(MockTokenizer.WHITESPACE, false);
226     wsTokenizer.setEnableChecks(false); // we will reset in a strange place
227     wsTokenizer.setReader(new StringReader("Rindfleischüberwachungsgesetz"));
228     DictionaryCompoundWordTokenFilter tf = new DictionaryCompoundWordTokenFilter(
229         wsTokenizer, dict,
230         CompoundWordTokenFilterBase.DEFAULT_MIN_WORD_SIZE,
231         CompoundWordTokenFilterBase.DEFAULT_MIN_SUBWORD_SIZE,
232         CompoundWordTokenFilterBase.DEFAULT_MAX_SUBWORD_SIZE, false);
233     
234     CharTermAttribute termAtt = tf.getAttribute(CharTermAttribute.class);
235     tf.reset();
236     assertTrue(tf.incrementToken());
237     assertEquals("Rindfleischüberwachungsgesetz", termAtt.toString());
238     assertTrue(tf.incrementToken());
239     assertEquals("Rind", termAtt.toString());
240     tf.end();
241     tf.close();
242     wsTokenizer.setReader(new StringReader("Rindfleischüberwachungsgesetz"));
243     tf.reset();
244     assertTrue(tf.incrementToken());
245     assertEquals("Rindfleischüberwachungsgesetz", termAtt.toString());
246   }

248   public void testRetainMockAttribute() throws Exception {
249     CharArraySet dict = makeDictionary("abc", "d", "efg");
250     Tokenizer tokenizer = new MockTokenizer(MockTokenizer.WHITESPACE, false);
251     tokenizer.setReader(new StringReader("abcdefg"));
252     TokenStream stream = new MockRetainAttributeFilter(tokenizer);
253     stream = new DictionaryCompoundWordTokenFilter(
254         stream, dict,
255         CompoundWordTokenFilterBase.DEFAULT_MIN_WORD_SIZE,
256         CompoundWordTokenFilterBase.DEFAULT_MIN_SUBWORD_SIZE,
257         CompoundWordTokenFilterBase.DEFAULT_MAX_SUBWORD_SIZE, false);
258     MockRetainAttribute retAtt = stream.addAttribute(MockRetainAttribute.class);
259     stream.reset();
260     while (stream.incrementToken()) {
261       assertTrue("Custom attribute value was lost", retAtt.getRetain());
262     }
263

266   public static interface MockRetainAttribute extends Attribute {
267     void setRetain(boolean attr);
268     boolean getRetain();
269   }
270 
271   public static final class MockRetainAttributeImpl extends AttributeImpl implements MockRetainAttribute {
272     private boolean retain = false;
273     
274     @Override
275     public void clear() {
276       retain = false;
277     }
278     
279     @Override
280     public boolean getRetain() {
281       return retain;
282     }
283     
284     @Override
285     public void setRetain(boolean retain) {
286       this.retain = retain;
287     }
288     
289     @Override
290     public void copyTo(AttributeImpl target) {
291       MockRetainAttribute t = (MockRetainAttribute) target;
292       t.setRetain(retain);
293     }
294 
295     @Override
296     public void reflectWith(AttributeReflector reflector) {
297       reflector.reflect(MockRetainAttribute.class, "retain", retain);
298     }
299   }
300 
301   private static class MockRetainAttributeFilter extends TokenFilter {
302     
303     MockRetainAttribute retainAtt = addAttribute(MockRetainAttribute.class);
304     
305     MockRetainAttributeFilter(TokenStream input) {
306       super(input);
307     }
308     
309     @Override
310     public boolean incrementToken() throws IOException {
311       if (input.incrementToken()){
312         retainAtt.setRetain(true); 
313         return true;
314       } else {
315       return false;
316       }
317     }
318   }

320   // SOLR-2891
321   // *CompoundWordTokenFilter blindly adds term length to offset, but this can take things out of bounds
322   // wrt original text if a previous filter increases the length of the word (in this case ü -> ue)
323   // so in this case we behave like WDF, and preserve any modified offsets
324   public void testInvalidOffsets() throws Exception {
325     final CharArraySet dict = makeDictionary("fall");
326     final NormalizeCharMap.Builder builder = new NormalizeCharMap.Builder();
327     builder.add("ü", "ue");
328     final NormalizeCharMap normMap = builder.build();
329     
330     Analyzer analyzer = new Analyzer() {
331 
332       @Override
333       protected TokenStreamComponents createComponents(String fieldName) {
334         Tokenizer tokenizer = new MockTokenizer(MockTokenizer.WHITESPACE, false);
335         TokenFilter filter = new DictionaryCompoundWordTokenFilter(tokenizer, dict);
336         return new TokenStreamComponents(tokenizer, filter);
337       }
338 
339       @Override
340       protected Reader initReader(String fieldName, Reader reader) {
341         return new MappingCharFilter(normMap, reader);
342       }
343     };
344 
345     assertAnalyzesTo(analyzer, "banküberfall", 
346         new String[] { "bankueberfall", "fall" },
347         new int[] { 0,  0 },
348         new int[] { 12, 12 });
349     analyzer.close();
350   }
