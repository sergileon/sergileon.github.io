47  public class SchemaHandler extends RequestHandlerBase {
48    private static final Logger log = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
49    private boolean isImmutableConfigSet = false;
50  
51    @Override
52    public void init(NamedList args) {
53      super.init(args);
54      Object immutable = args.get(IMMUTABLE_CONFIGSET_ARG);
55      isImmutableConfigSet = immutable  != null ? Boolean.parseBoolean(immutable.toString()) : false;
56    }
57  
58    @Override
59    public void handleRequestBody(SolrQueryRequest req, SolrQueryResponse rsp) throws Exception {
60      SolrConfigHandler.setWt(req, JSON);
61      String httpMethod = (String) req.getContext().get("httpMethod");
62      if ("POST".equals(httpMethod)) {
63        if (isImmutableConfigSet) {
64          rsp.add("errors", "ConfigSet is immutable");
65          return;
66        }
67        if (req.getContentStreams() == null) {
68          rsp.add("errors", "no stream");
69          return;
70        }
71  
72        for (ContentStream stream : req.getContentStreams()) {
73          try {
74            List errs = new SchemaManager(req).performOperations(stream.getReader());
75            if (!errs.isEmpty()) rsp.add("errors", errs);
76          } catch (IOException e) {
77            rsp.add("errors", Collections.singletonList("Error reading input String " + e.getMessage()));
78            rsp.setException(e);
79          }
80          break;
81        }
82      } else {
83        handleGET(req, rsp);
84      }
85    }
86  
87    private void handleGET(SolrQueryRequest req, SolrQueryResponse rsp) {
88      try {
89        String path = (String) req.getContext().get("path");
90        switch (path) {
91          case "/schema":
92            rsp.add(IndexSchema.SCHEMA, req.getSchema().getNamedPropertyValues());
93            break;
94          case "/schema/version":
95            rsp.add(IndexSchema.VERSION, req.getSchema().getVersion());
96            break;
97          case "/schema/uniquekey":
98            rsp.add(IndexSchema.UNIQUE_KEY, req.getSchema().getUniqueKeyField().getName());
99            break;
100         case "/schema/similarity":
101           rsp.add(IndexSchema.SIMILARITY, req.getSchema().getSimilarityFactory().getNamedPropertyValues());
102           break;
103         case "/schema/name": {
104           final String schemaName = req.getSchema().getSchemaName();
105           if (null == schemaName) {
106             String message = "Schema has no name";
107             throw new SolrException(SolrException.ErrorCode.NOT_FOUND, message);
108           }
109           rsp.add(IndexSchema.NAME, schemaName);
110           break;
111         }
112         case "/schema/defaultsearchfield": {
113           final String defaultSearchFieldName = req.getSchema().getDefaultSearchFieldName();
114           if (null == defaultSearchFieldName) {
115             final String message = "undefined " + IndexSchema.DEFAULT_SEARCH_FIELD;
116             throw new SolrException(SolrException.ErrorCode.NOT_FOUND, message);
117           }
118           rsp.add(IndexSchema.DEFAULT_SEARCH_FIELD, defaultSearchFieldName);
119           break;
120         }
121         case "/schema/solrqueryparser": {
122           SimpleOrderedMap<Object> props = new SimpleOrderedMap<>();
123           props.add(IndexSchema.DEFAULT_OPERATOR, req.getSchema().getQueryParserDefaultOperator());
124           rsp.add(IndexSchema.SOLR_QUERY_PARSER, props);
125           break;
126         }
127         case "/schema/zkversion": {
128           int refreshIfBelowVersion = -1;
129           Object refreshParam = req.getParams().get("refreshIfBelowVersion");
130           if (refreshParam != null)
131             refreshIfBelowVersion = (refreshParam instanceof Number) ? ((Number) refreshParam).intValue()
132                 : Integer.parseInt(refreshParam.toString());
133           int zkVersion = -1;
134           IndexSchema schema = req.getSchema();
135           if (schema instanceof ManagedIndexSchema) {
136             ManagedIndexSchema managed = (ManagedIndexSchema) schema;
137             zkVersion = managed.getSchemaZkVersion();
138             if (refreshIfBelowVersion != -1 && zkVersion < refreshIfBelowVersion) {
139               log.info("REFRESHING SCHEMA (refreshIfBelowVersion=" + refreshIfBelowVersion +
140                   ", currentVersion=" + zkVersion + ") before returning version!");
141               ZkSolrResourceLoader zkSolrResourceLoader = (ZkSolrResourceLoader) req.getCore().getResourceLoader();
142               ZkIndexSchemaReader zkIndexSchemaReader = zkSolrResourceLoader.getZkIndexSchemaReader();
143               managed = zkIndexSchemaReader.refreshSchemaFromZk(refreshIfBelowVersion);
144               zkVersion = managed.getSchemaZkVersion();
145             }
146           }
147           rsp.add("zkversion", zkVersion);
148           break;
149         }
150         case "/schema/solrqueryparser/defaultoperator": {
151           rsp.add(IndexSchema.DEFAULT_OPERATOR, req.getSchema().getQueryParserDefaultOperator());
152           break;
153         }
154         default: {
155           throw new SolrException(SolrException.ErrorCode.NOT_FOUND, "No such path " + path);
156         }
157       }
158 
159     } catch (Exception e) {
160       rsp.setException(e);
161     }
162   }
163 
164   private static Set<String> subPaths = new HashSet<>(Arrays.asList(
165       "/version",
166       "/uniquekey",
167       "/name",
168       "/similarity",
169       "/defaultsearchfield",
170       "/solrqueryparser",
171       "/zkversion",
172       "/solrqueryparser/defaultoperator"
173   ));
174 
175   @Override
176   public SolrRequestHandler getSubHandler(String subPath) {
177     if (subPaths.contains(subPath)) return this;
178     return null;
179   }
180 
181   @Override
182   public String getDescription() {
183     return "CRUD operations over the Solr schema";
184   }
