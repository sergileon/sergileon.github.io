22  import org.apache.lucene.analysis.TokenFilter;
23  import org.apache.lucene.analysis.TokenStream;
24  import org.apache.lucene.analysis.standard.StandardTokenizer;
25  import org.apache.lucene.analysis.tokenattributes.CharTermAttribute;
26  import org.apache.lucene.analysis.tokenattributes.OffsetAttribute;
27  import org.apache.lucene.analysis.tokenattributes.PositionIncrementAttribute;

32  /**
33   * Forms bigrams of CJK terms that are generated from StandardTokenizer
34   * or ICUTokenizer.
35   * <p>
36   * CJK types are set by these tokenizers, but you can also use 
37   * {@link #CJKBigramFilter(TokenStream, int)} to explicitly control which
38   * of the CJK scripts are turned into bigrams.
39   * <p>
40   * By default, when a CJK character has no adjacent characters to form
41   * a bigram, it is output in unigram form. If you want to always output
42   * both unigrams and bigrams, set the <code>outputUnigrams</code>
43   * flag in {@link CJKBigramFilter#CJKBigramFilter(TokenStream, int, boolean)}.
44   * This can be used for a combined unigram+bigram approach.
45   * <p>
46   * In all cases, all non-CJK input is passed thru unmodified.
47   */
48  public final class CJKBigramFilter extends TokenFilter {
49    // configuration
50    /** bigram flag for Han Ideographs */
51    public static final int HAN = 1;
52    /** bigram flag for Hiragana */
53    public static final int HIRAGANA = 2;
54    /** bigram flag for Katakana */
55    public static final int KATAKANA = 4;
56    /** bigram flag for Hangul */
57    public static final int HANGUL = 8;
58  
59    /** when we emit a bigram, it's then marked as this type */
60    public static final String DOUBLE_TYPE = "<DOUBLE>";
61    /** when we emit a unigram, it's then marked as this type */
62    public static final String SINGLE_TYPE = "<SINGLE>";
63  
64    // the types from standardtokenizer
65    private static final String HAN_TYPE = StandardTokenizer.TOKEN_TYPES[StandardTokenizer.IDEOGRAPHIC];
66    private static final String HIRAGANA_TYPE = StandardTokenizer.TOKEN_TYPES[StandardTokenizer.HIRAGANA];
67    private static final String KATAKANA_TYPE = StandardTokenizer.TOKEN_TYPES[StandardTokenizer.KATAKANA];
68    private static final String HANGUL_TYPE = StandardTokenizer.TOKEN_TYPES[StandardTokenizer.HANGUL];
69    
70    // sentinel value for ignoring a script 
71    private static final Object NO = new Object();
72  
73    // these are set to either their type or NO if we want to pass them thru
74    private final Object doHan;
75    private final Object doHiragana;
76    private final Object doKatakana;
77    private final Object doHangul;
78    
79    // true if we should output unigram tokens always
80    private final boolean outputUnigrams;
81    private boolean ngramState; // false = output unigram, true = output bigram
82      
83    private final CharTermAttribute termAtt = addAttribute(CharTermAttribute.class);
84    private final TypeAttribute typeAtt = addAttribute(TypeAttribute.class);
85    private final OffsetAttribute offsetAtt = addAttribute(OffsetAttribute.class);
86    private final PositionIncrementAttribute posIncAtt = addAttribute(PositionIncrementAttribute.class);
87    private final PositionLengthAttribute posLengthAtt = addAttribute(PositionLengthAttribute.class);
88    
89    // buffers containing codepoint and offsets in parallel
90    int buffer[] = new int[8];
91    int startOffset[] = new int[8];
92    int endOffset[] = new int[8];
93    // length of valid buffer
94    int bufferLen;
95    // current buffer index
96    int index;
97    
98    // the last end offset, to determine if we should bigram across tokens
99    int lastEndOffset;
100   
101   private boolean exhausted;
102   
103   /** 
104    * Calls {@link CJKBigramFilter#CJKBigramFilter(TokenStream, int)
105    *       CJKBigramFilter(in, HAN | HIRAGANA | KATAKANA | HANGUL)}
106    */
107   public CJKBigramFilter(TokenStream in) {
108     this(in, HAN | HIRAGANA | KATAKANA | HANGUL);
109   }
110   
111   /** 
112    * Calls {@link CJKBigramFilter#CJKBigramFilter(TokenStream, int, boolean)
113    *       CJKBigramFilter(in, flags, false)}
114    */
115   public CJKBigramFilter(TokenStream in, int flags) {
116     this(in, flags, false);
117   }
118   
119   /**
120    * Create a new CJKBigramFilter, specifying which writing systems should be bigrammed,
121    * and whether or not unigrams should also be output.
122    * @param flags OR'ed set from {@link CJKBigramFilter#HAN}, {@link CJKBigramFilter#HIRAGANA}, 
123    *        {@link CJKBigramFilter#KATAKANA}, {@link CJKBigramFilter#HANGUL}
124    * @param outputUnigrams true if unigrams for the selected writing systems should also be output.
125    *        when this is false, this is only done when there are no adjacent characters to form
126    *        a bigram.
127    */
128   public CJKBigramFilter(TokenStream in, int flags, boolean outputUnigrams) {
129     super(in);
130     doHan =      (flags & HAN) == 0      ? NO : HAN_TYPE;
131     doHiragana = (flags & HIRAGANA) == 0 ? NO : HIRAGANA_TYPE;
132     doKatakana = (flags & KATAKANA) == 0 ? NO : KATAKANA_TYPE;
133     doHangul =   (flags & HANGUL) == 0   ? NO : HANGUL_TYPE;
134     this.outputUnigrams = outputUnigrams;
135   }
136   
137   /*
138    * much of this complexity revolves around handling the special case of a 
139    * "lone cjk character" where cjktokenizer would output a unigram. this 
140    * is also the only time we ever have to captureState.
141    */
142   @Override
143   public boolean incrementToken() throws IOException {
144     while (true) {
145       if (hasBufferedBigram()) {
146         
147         // case 1: we have multiple remaining codepoints buffered,
148         // so we can emit a bigram here.
149         
150         if (outputUnigrams) {
151 
152           // when also outputting unigrams, we output the unigram first,
153           // then rewind back to revisit the bigram.
154           // so an input of ABC is A + (rewind)AB + B + (rewind)BC + C
155           // the logic in hasBufferedUnigram ensures we output the C, 
156           // even though it did actually have adjacent CJK characters.
157 
158           if (ngramState) {
159             flushBigram();
160           } else {
161             flushUnigram();
162             index--;
163           }
164           ngramState = !ngramState;
165         } else {
166           flushBigram();
167         }
168         return true;
169       } else if (doNext()) {
170         
171         // case 2: look at the token type. should we form any n-grams?
172         
173         String type = typeAtt.type();
174         if (type == doHan || type == doHiragana || type == doKatakana || type == doHangul) {
175           
176           // acceptable CJK type: we form n-grams from these.
177           // as long as the offsets are aligned, we just add these to our current buffer.
178           // otherwise, we clear the buffer and start over.
179           
180           if (offsetAtt.startOffset() != lastEndOffset) { // unaligned, clear queue
181             if (hasBufferedUnigram()) {
182               
183               // we have a buffered unigram, and we peeked ahead to see if we could form
184               // a bigram, but we can't, because the offsets are unaligned. capture the state 
185               // of this peeked data to be revisited next time thru the loop, and dump our unigram.
186               
187               loneState = captureState();
188               flushUnigram();
189               return true;
190             }
191             index = 0;
192             bufferLen = 0;
193           }
194           refill();
195         } else {
196           
197           // not a CJK type: we just return these as-is.
198           
199           if (hasBufferedUnigram()) {
200             
201             // we have a buffered unigram, and we peeked ahead to see if we could form
202             // a bigram, but we can't, because it's not a CJK type. capture the state 
203             // of this peeked data to be revisited next time thru the loop, and dump our unigram.
204             
205             loneState = captureState();
206             flushUnigram();
207             return true;
208           }
209           return true;
210         }
211       } else {
212         
213         // case 3: we have only zero or 1 codepoints buffered, 
214         // so not enough to form a bigram. But, we also have no
215         // more input. So if we have a buffered codepoint, emit
216         // a unigram, otherwise, it's end of stream.
217         
218         if (hasBufferedUnigram()) {
219           flushUnigram(); // flush our remaining unigram
220           return true;
221         }
222         return false;
223       }
224     }
225   }
226   
227   private State loneState; // rarely used: only for "lone cjk characters", where we emit unigrams
228   
229   /** 
230    * looks at next input token, returning false is none is available 
231    */
232   private boolean doNext() throws IOException {
233     if (loneState != null) {
234       restoreState(loneState);
235       loneState = null;
236       return true;
237     } else {
238       if (exhausted) {
239         return false;
240       } else if (input.incrementToken()) {
241         return true;
242       } else {
243         exhausted = true;
244         return false;
245       }
246     }
247   }
248   
249   /**
250    * refills buffers with new data from the current token.
251    */
252   private void refill() {
253     // compact buffers to keep them smallish if they become large
254     // just a safety check, but technically we only need the last codepoint
255     if (bufferLen > 64) {
256       int last = bufferLen - 1;
257       buffer[0] = buffer[last];
258       startOffset[0] = startOffset[last];
259       endOffset[0] = endOffset[last];
260       bufferLen = 1;
261       index -= last;
262     }
263 
264     char termBuffer[] = termAtt.buffer();
265     int len = termAtt.length();
266     int start = offsetAtt.startOffset();
267     int end = offsetAtt.endOffset();
268     
269     int newSize = bufferLen + len;
270     buffer = ArrayUtil.grow(buffer, newSize);
271     startOffset = ArrayUtil.grow(startOffset, newSize);
272     endOffset = ArrayUtil.grow(endOffset, newSize);
273     lastEndOffset = end;
274 
275     if (end - start != len) {
276       // crazy offsets (modified by synonym or charfilter): just preserve
277       for (int i = 0, cp = 0; i < len; i += Character.charCount(cp)) {
278         cp = buffer[bufferLen] = Character.codePointAt(termBuffer, i, len);
279         startOffset[bufferLen] = start;
280         endOffset[bufferLen] = end;
281         bufferLen++;
282       }
283     } else {
284       // normal offsets
285       for (int i = 0, cp = 0, cpLen = 0; i < len; i += cpLen) {
286         cp = buffer[bufferLen] = Character.codePointAt(termBuffer, i, len);
287         cpLen = Character.charCount(cp);
288         startOffset[bufferLen] = start;
289         start = endOffset[bufferLen] = start + cpLen;
290         bufferLen++;
291       }
292     }
293   }
294 
295   /** 
296    * Flushes a bigram token to output from our buffer 
297    * This is the normal case, e.g. ABC -&gt; AB BC
298    */
299   private void flushBigram() {
300     clearAttributes();
301     char termBuffer[] = termAtt.resizeBuffer(4); // maximum bigram length in code units (2 supplementaries)
302     int len1 = Character.toChars(buffer[index], termBuffer, 0);
303     int len2 = len1 + Character.toChars(buffer[index+1], termBuffer, len1);
304     termAtt.setLength(len2);
305     offsetAtt.setOffset(startOffset[index], endOffset[index+1]);
306     typeAtt.setType(DOUBLE_TYPE);
307     // when outputting unigrams, all bigrams are synonyms that span two unigrams
308     if (outputUnigrams) {
309       posIncAtt.setPositionIncrement(0);
310       posLengthAtt.setPositionLength(2);
311     }
312     index++;
313   }
314   
315   /** 
316    * Flushes a unigram token to output from our buffer.
317    * This happens when we encounter isolated CJK characters, either the whole
318    * CJK string is a single character, or we encounter a CJK character surrounded 
319    * by space, punctuation, english, etc, but not beside any other CJK.
320    */
321   private void flushUnigram() {
322     clearAttributes();
323     char termBuffer[] = termAtt.resizeBuffer(2); // maximum unigram length (2 surrogates)
324     int len = Character.toChars(buffer[index], termBuffer, 0);
325     termAtt.setLength(len);
326     offsetAtt.setOffset(startOffset[index], endOffset[index]);
327     typeAtt.setType(SINGLE_TYPE);
328     index++;
329   }
330   
331   /**
332    * True if we have multiple codepoints sitting in our buffer
333    */
334   private boolean hasBufferedBigram() {
335     return bufferLen - index > 1;
336   }
337 
338   /**
339    * True if we have a single codepoint sitting in our buffer, where its future
340    * (whether it is emitted as unigram or forms a bigram) depends upon not-yet-seen
341    * inputs.
342    */
343   private boolean hasBufferedUnigram() {
344     if (outputUnigrams) {
345       // when outputting unigrams always
346       return bufferLen - index == 1;
347     } else {
348       // otherwise it's only when we have a lone CJK character
349       return bufferLen == 1 && index == 0;
350     }
351   }
352 
353   @Override
354   public void reset() throws IOException {
355     super.reset();
356     bufferLen = 0;
357     index = 0;
358     lastEndOffset = 0;
359     loneState = null;
360     exhausted = false;
361     ngramState = false;
362   }
