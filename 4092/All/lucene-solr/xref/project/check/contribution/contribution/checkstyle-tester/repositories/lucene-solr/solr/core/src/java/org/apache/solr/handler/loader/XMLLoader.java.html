81    public static final int XSLT_CACHE_DEFAULT = 60;

83    int xsltCacheLifetimeSeconds;
84    XMLInputFactory inputFactory;

87    @Override
88    public XMLLoader init(SolrParams args) {
89      // Init StAX parser:
90      inputFactory = XMLInputFactory.newInstance();
91      EmptyEntityResolver.configureXMLInputFactory(inputFactory);
92      inputFactory.setXMLReporter(xmllog);
93      try {
94        // The java 1.6 bundled stax parser (sjsxp) does not currently have a thread-safe
95        // XMLInputFactory, as that implementation tries to cache and reuse the
96        // XMLStreamReader.  Setting the parser-specific "reuse-instance" property to false
97        // prevents this.
98        // All other known open-source stax parsers (and the bea ref impl)
99        // have thread-safe factories.
100       inputFactory.setProperty("reuse-instance", Boolean.FALSE);
101     } catch (IllegalArgumentException ex) {
102       // Other implementations will likely throw this exception since "reuse-instance"
103       // isimplementation specific.
104       log.debug("Unable to set the 'reuse-instance' property for the input chain: " + inputFactory);
105     }
106     
107     // Init SAX parser (for XSL):
108     saxFactory = SAXParserFactory.newInstance();
109     saxFactory.setNamespaceAware(true); // XSL needs this!
110     EmptyEntityResolver.configureSAXParserFactory(saxFactory);
111     
112     xsltCacheLifetimeSeconds = XSLT_CACHE_DEFAULT;
113     if(args != null) {
114       xsltCacheLifetimeSeconds = args.getInt(XSLT_CACHE_PARAM,XSLT_CACHE_DEFAULT);
115       log.info("xsltCacheLifetimeSeconds=" + xsltCacheLifetimeSeconds);
116     }
117     return this;
118   }

125   @Override
126   public void load(SolrQueryRequest req, SolrQueryResponse rsp, ContentStream stream, UpdateRequestProcessor processor) throws Exception {
127     final String charset = ContentStreamBase.getCharsetFromContentType(stream.getContentType());
128     
129     InputStream is = null;
130     XMLStreamReader parser = null;
131 
132     String tr = req.getParams().get(CommonParams.TR,null);
133     if(tr!=null) {
134       final Transformer t = getTransformer(tr,req);
135       final DOMResult result = new DOMResult();
136       
137       // first step: read XML and build DOM using Transformer (this is no overhead, as XSL always produces
138       // an internal result DOM tree, we just access it directly as input for StAX):
139       try {
140         is = stream.getStream();
141         final InputSource isrc = new InputSource(is);
142         isrc.setEncoding(charset);
143         final XMLReader xmlr = saxFactory.newSAXParser().getXMLReader();
144         xmlr.setErrorHandler(xmllog);
145         xmlr.setEntityResolver(EmptyEntityResolver.SAX_INSTANCE);
146         final SAXSource source = new SAXSource(xmlr, isrc);
147         t.transform(source, result);
148       } catch(TransformerException te) {
149         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, te.getMessage(), te);
150       } finally {
151         IOUtils.closeQuietly(is);
152       }
153       // second step: feed the intermediate DOM tree into StAX parser:
154       try {
155         parser = inputFactory.createXMLStreamReader(new DOMSource(result.getNode()));
156         this.processUpdate(req, processor, parser);
157       } catch (XMLStreamException e) {
158         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, e.getMessage(), e);
159       } finally {
160         if (parser != null) parser.close();
161       }
162     }
163     // Normal XML Loader
164     else {
165       try {
166         is = stream.getStream();
167         if (log.isTraceEnabled()) {
168           final byte[] body = IOUtils.toByteArray(is);
169           // TODO: The charset may be wrong, as the real charset is later
170           // determined by the XML parser, the content-type is only used as a hint!
171           log.trace("body", new String(body, (charset == null) ?
172             ContentStreamBase.DEFAULT_CHARSET : charset));
173           IOUtils.closeQuietly(is);
174           is = new ByteArrayInputStream(body);
175         }
176         parser = (charset == null) ?
177           inputFactory.createXMLStreamReader(is) : inputFactory.createXMLStreamReader(is, charset);
178         this.processUpdate(req, processor, parser);
179       } catch (XMLStreamException e) {
180         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, e.getMessage(), e);
181       } finally {
182         if (parser != null) parser.close();
183         IOUtils.closeQuietly(is);
184       }
185     }
186   }

189   /** Get Transformer from request context, or from TransformerProvider.
190    *  This allows either getContentType(...) or write(...) to instantiate the Transformer,
191    *  depending on which one is called first, then the other one reuses the same Transformer
192    */
193   Transformer getTransformer(String xslt, SolrQueryRequest request) throws IOException {
194     // not the cleanest way to achieve this
195     // no need to synchronize access to context, right? 
196     // Nothing else happens with it at the same time
197     final Map<Object,Object> ctx = request.getContext();
198     Transformer result = (Transformer)ctx.get(CONTEXT_TRANSFORMER_KEY);
199     if(result==null) {
200       SolrConfig solrConfig = request.getCore().getSolrConfig();
201       result = TransformerProvider.instance.getTransformer(solrConfig, xslt, xsltCacheLifetimeSeconds);
202       result.setErrorListener(xmllog);
203       ctx.put(CONTEXT_TRANSFORMER_KEY,result);
204     }
205     return result;
206   }

209   /**
210    * @since solr 1.2
211    */
212   void processUpdate(SolrQueryRequest req, UpdateRequestProcessor processor, XMLStreamReader parser)
213           throws XMLStreamException, IOException, FactoryConfigurationError {
214     AddUpdateCommand addCmd = null;
215     SolrParams params = req.getParams();
216     while (true) {
217       int event = parser.next();
218       switch (event) {
219         case XMLStreamConstants.END_DOCUMENT:
220           parser.close();
221           return;
222 
223         case XMLStreamConstants.START_ELEMENT:
224           String currTag = parser.getLocalName();
225           if (currTag.equals(UpdateRequestHandler.ADD)) {
226             log.trace("SolrCore.update(add)");
227 
228             addCmd = new AddUpdateCommand(req);
229 
230             // First look for commitWithin parameter on the request, will be overwritten for individual <add>'s
231             addCmd.commitWithin = params.getInt(UpdateParams.COMMIT_WITHIN, -1);
232             addCmd.overwrite = params.getBool(UpdateParams.OVERWRITE, true);
233             
234             for (int i = 0; i < parser.getAttributeCount(); i++) {
235               String attrName = parser.getAttributeLocalName(i);
236               String attrVal = parser.getAttributeValue(i);
237               if (UpdateRequestHandler.OVERWRITE.equals(attrName)) {
238                 addCmd.overwrite = StrUtils.parseBoolean(attrVal);
239               } else if (UpdateRequestHandler.COMMIT_WITHIN.equals(attrName)) {
240                 addCmd.commitWithin = Integer.parseInt(attrVal);
241               } else {
242                 log.warn("XML element <add> has invalid XML attr: " + attrName);
243               }
244             }
245 
246           } else if ("doc".equals(currTag)) {
247             if(addCmd != null) {
248               log.trace("adding doc...");
249               addCmd.clear();
250               addCmd.solrDoc = readDoc(parser);
251               processor.processAdd(addCmd);
252             } else {
253               throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "Unexpected <doc> tag without an <add> tag surrounding it.");
254             }
255           } else if (UpdateRequestHandler.COMMIT.equals(currTag) || UpdateRequestHandler.OPTIMIZE.equals(currTag)) {
256             log.trace("parsing " + currTag);
257 
258             CommitUpdateCommand cmd = new CommitUpdateCommand(req, UpdateRequestHandler.OPTIMIZE.equals(currTag));
259             ModifiableSolrParams mp = new ModifiableSolrParams();
260             
261             for (int i = 0; i < parser.getAttributeCount(); i++) {
262               String attrName = parser.getAttributeLocalName(i);
263               String attrVal = parser.getAttributeValue(i);
264               mp.set(attrName, attrVal);
265             }
266 
267             RequestHandlerUtils.validateCommitParams(mp);
268             SolrParams p = SolrParams.wrapDefaults(mp, req.getParams());   // default to the normal request params for commit options
269             RequestHandlerUtils.updateCommit(cmd, p);
270 
271             processor.processCommit(cmd);
272           } // end commit
273           else if (UpdateRequestHandler.ROLLBACK.equals(currTag)) {
274             log.trace("parsing rollback");
275 
276             RollbackUpdateCommand cmd = new RollbackUpdateCommand(req);
277 
278             processor.processRollback(cmd);
279           } // end rollback
280           else if (UpdateRequestHandler.DELETE.equals(currTag)) {
281             log.trace("parsing delete");
282             processDelete(req, processor, parser);
283           } // end delete
284           break;
285       }
286     }
287   }

289   /**
290    * @since solr 1.3
291    */
292   void processDelete(SolrQueryRequest req, UpdateRequestProcessor processor, XMLStreamReader parser) throws XMLStreamException, IOException {
293     // Parse the command
294     DeleteUpdateCommand deleteCmd = new DeleteUpdateCommand(req);
295 
296     // First look for commitWithin parameter on the request, will be overwritten for individual <delete>'s
297     SolrParams params = req.getParams();
298     deleteCmd.commitWithin = params.getInt(UpdateParams.COMMIT_WITHIN, -1);
299 
300     for (int i = 0; i < parser.getAttributeCount(); i++) {
301       String attrName = parser.getAttributeLocalName(i);
302       String attrVal = parser.getAttributeValue(i);
303       if ("fromPending".equals(attrName)) {
304         // deprecated
305       } else if ("fromCommitted".equals(attrName)) {
306         // deprecated
307       } else if (UpdateRequestHandler.COMMIT_WITHIN.equals(attrName)) {
308         deleteCmd.commitWithin = Integer.parseInt(attrVal);
309       } else {
310         log.warn("XML element <delete> has invalid XML attr: " + attrName);
311       }
312     }
313 
314     StringBuilder text = new StringBuilder();
315     while (true) {
316       int event = parser.next();
317       switch (event) {
318         case XMLStreamConstants.START_ELEMENT:
319           String mode = parser.getLocalName();
320           if (!("id".equals(mode) || "query".equals(mode))) {
321             String msg = "XML element <delete> has invalid XML child element: " + mode;
322             log.warn(msg);
323             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
324                                     msg);
325           }
326           text.setLength(0);
327           
328           if ("id".equals(mode)) {
329             for (int i = 0; i < parser.getAttributeCount(); i++) {
330               String attrName = parser.getAttributeLocalName(i);
331               String attrVal = parser.getAttributeValue(i);
332               if (UpdateRequestHandler.VERSION.equals(attrName)) {
333                 deleteCmd.setVersion(Long.parseLong(attrVal));
334               }
335               if (UpdateRequest.ROUTE.equals(attrName)) {
336                 deleteCmd.setRoute(attrVal);
337               }
338             }
339           }
340           break;
341 
342         case XMLStreamConstants.END_ELEMENT:
343           String currTag = parser.getLocalName();
344           if ("id".equals(currTag)) {
345             deleteCmd.setId(text.toString());         
346           } else if ("query".equals(currTag)) {
347             deleteCmd.setQuery(text.toString());
348           } else if ("delete".equals(currTag)) {
349             return;
350           } else {
351             String msg = "XML element <delete> has invalid XML (closing) child element: " + currTag;
352             log.warn(msg);
353             throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
354                                     msg);
355           }
356           processor.processDelete(deleteCmd);
357           deleteCmd.clear();
358           break;
359 
360           // Add everything to the text
361         case XMLStreamConstants.SPACE:
362         case XMLStreamConstants.CDATA:
363         case XMLStreamConstants.CHARACTERS:
364           text.append(parser.getText());
365           break;
366       }
367     }
368   }

371   /**
372    * Given the input stream, read a document
373    *
374    * @since solr 1.3
375    */
376   public SolrInputDocument readDoc(XMLStreamReader parser) throws XMLStreamException {
377     SolrInputDocument doc = new SolrInputDocument();
378 
379     String attrName = "";
380     for (int i = 0; i < parser.getAttributeCount(); i++) {
381       attrName = parser.getAttributeLocalName(i);
382       if ("boost".equals(attrName)) {
383         doc.setDocumentBoost(Float.parseFloat(parser.getAttributeValue(i)));
384       } else {
385         log.warn("XML element <doc> has invalid XML attr:" + attrName);
386       }
387     }
388 
389     StringBuilder text = new StringBuilder();
390     String name = null;
391     float boost = 1.0f;
392     boolean isNull = false;
393     String update = null;
394     Collection<SolrInputDocument> subDocs = null;
395     Map<String, Map<String, Object>> updateMap = null;
396     boolean complete = false;
397     while (!complete) {
398       int event = parser.next();
399       switch (event) {
400         // Add everything to the text
401         case XMLStreamConstants.SPACE:
402         case XMLStreamConstants.CDATA:
403         case XMLStreamConstants.CHARACTERS:
404           text.append(parser.getText());
405           break;
406 
407         case XMLStreamConstants.END_ELEMENT:
408           if ("doc".equals(parser.getLocalName())) {
409             if (subDocs != null && !subDocs.isEmpty()) {
410               doc.addChildDocuments(subDocs);
411               subDocs = null;
412             }
413             complete = true;
414             break;
415           } else if ("field".equals(parser.getLocalName())) {
416             // should I warn in some text has been found too
417             Object v = isNull ? null : text.toString();
418             if (update != null) {
419               if (updateMap == null) updateMap = new HashMap<>();
420               Map<String, Object> extendedValues = updateMap.get(name);
421               if (extendedValues == null) {
422                 extendedValues = new HashMap<>(1);
423                 updateMap.put(name, extendedValues);
424               }
425               Object val = extendedValues.get(update);
426               if (val == null) {
427                 extendedValues.put(update, v);
428               } else {
429                 // multiple val are present
430                 if (val instanceof List) {
431                   List list = (List) val;
432                   list.add(v);
433                 } else {
434                   List<Object> values = new ArrayList<>();
435                   values.add(val);
436                   values.add(v);
437                   extendedValues.put(update, values);
438                 }
439               }
440               break;
441             }
442             doc.addField(name, v, boost);
443             boost = 1.0f;
444             // field is over
445             name = null;
446           }
447           break;
448 
449         case XMLStreamConstants.START_ELEMENT:
450           text.setLength(0);
451           String localName = parser.getLocalName();
452           if ("doc".equals(localName)) {
453             if (subDocs == null)
454               subDocs = Lists.newArrayList();
455             subDocs.add(readDoc(parser));
456           }
457           else {
458             if (!"field".equals(localName)) {
459               String msg = "XML element <doc> has invalid XML child element: " + localName;
460               log.warn(msg);
461               throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
462                                       msg);
463             }
464             boost = 1.0f;
465             update = null;
466             isNull = false;
467             String attrVal = "";
468             for (int i = 0; i < parser.getAttributeCount(); i++) {
469               attrName = parser.getAttributeLocalName(i);
470               attrVal = parser.getAttributeValue(i);
471               if (NAME.equals(attrName)) {
472                 name = attrVal;
473               } else if ("boost".equals(attrName)) {
474                 boost = Float.parseFloat(attrVal);
475               } else if ("null".equals(attrName)) {
476                 isNull = StrUtils.parseBoolean(attrVal);
477               } else if ("update".equals(attrName)) {
478                 update = attrVal;
479               } else {
480                 log.warn("XML element <field> has invalid XML attr: " + attrName);
481               }
482             }
483           }
484           break;
485       }
486     }
487 
488     if (updateMap != null)  {
489       for (Map.Entry<String, Map<String, Object>> entry : updateMap.entrySet()) {
490         name = entry.getKey();
491         Map<String, Object> value = entry.getValue();
492         doc.addField(name, value, 1.0f);
493       }
494     }
495 
496     return doc;
497   }
