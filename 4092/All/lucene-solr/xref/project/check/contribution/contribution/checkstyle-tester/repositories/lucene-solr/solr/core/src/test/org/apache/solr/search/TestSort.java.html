102   public void testRandomFieldNameSorts() throws Exception {
103     SolrQueryRequest req = lrf.makeRequest("q", "*:*");
104 
105     final int iters = atLeast(5000);
106 
107     // infinite loop abort when trying to generate a non-blank sort "name"
108     final int nonBlankAttempts = 37;
109 
110     for (int i = 0; i < iters; i++) {
111       final StringBuilder input = new StringBuilder();
112       final String[] names = new String[TestUtil.nextInt(r, 1, 10)];
113       final boolean[] reverse = new boolean[names.length];
114       for (int j = 0; j < names.length; j++) {
115         names[j] = null;
116         for (int k = 0; k < nonBlankAttempts && null == names[j]; k++) {
117           names[j] = TestUtil.randomRealisticUnicodeString(r, 1, 100);
118 
119           // munge anything that might make this a function
120           names[j] = names[j].replaceFirst("\\{","\\{\\{");
121           names[j] = names[j].replaceFirst("\\(","\\(\\(");
122           names[j] = names[j].replaceFirst("(\\\"|\\')","$1$1z");
123           names[j] = names[j].replaceFirst("(\\d)","$1x");
124 
125           // eliminate pesky problem chars
126           names[j] = names[j].replaceAll("\\p{Cntrl}|\\p{javaWhitespace}","");
127           
128           if (0 == names[j].length()) {
129             names[j] = null;
130           }
131         }
132         // with luck this bad, never go to vegas
133         // alternatively: if (null == names[j]) names[j] = "never_go_to_vegas";
134         assertNotNull("Unable to generate a (non-blank) names["+j+"] after "
135                       + nonBlankAttempts + " attempts", names[j]);
136 
137         reverse[j] = r.nextBoolean();
138 
139         input.append(r.nextBoolean() ? " " : "");
140         input.append(names[j]);
141         input.append(" ");
142         input.append(reverse[j] ? "desc," : "asc,");
143       }
144       input.deleteCharAt(input.length()-1);
145       SortField[] sorts = null;
146       List<SchemaField> fields = null;
147       try {
148         SortSpec spec = SortSpecParsing.parseSortSpec(input.toString(), req);
149         sorts = spec.getSort().getSort();
150         fields = spec.getSchemaFields();
151       } catch (RuntimeException e) {
152         throw new RuntimeException("Failed to parse sort: " + input, e);
153       }
154       assertEquals("parsed sorts had unexpected size", 
155                    names.length, sorts.length);
156       assertEquals("parsed sort schema fields had unexpected size", 
157                    names.length, fields.size());
158       for (int j = 0; j < names.length; j++) {
159         assertEquals("sorts["+j+"] had unexpected reverse: " + input,
160                      reverse[j], sorts[j].getReverse());
161 
162         final Type type = sorts[j].getType();
163 
164         if (Type.SCORE.equals(type)) {
165           assertEquals("sorts["+j+"] is (unexpectedly) type score : " + input,
166                        "score", names[j]);
167         } else if (Type.DOC.equals(type)) {
168           assertEquals("sorts["+j+"] is (unexpectedly) type doc : " + input,
169                        "_docid_", names[j]);
170         } else if (Type.CUSTOM.equals(type) || Type.REWRITEABLE.equals(type)) {
171 
172           fail("sorts["+j+"] resulted in a '" + type.toString()
173                + "', either sort parsing code is broken, or func/query " 
174                + "semantics have gotten broader and munging in this test "
175                + "needs improved: " + input);
176 
177         } else {
178           assertEquals("sorts["+j+"] ("+type.toString()+
179                        ") had unexpected field in: " + input,
180                        names[j], sorts[j].getField());
181           assertEquals("fields["+j+"] ("+type.toString()+
182                        ") had unexpected name in: " + input,
183                        names[j], fields.get(j).getName());
184         }
185       }
186     }
187   }

191   public void testSort() throws Exception {
192     Directory dir = new RAMDirectory();
193     Field f = new StringField("f", "0", Field.Store.NO);
194     Field f2 = new StringField("f2", "0", Field.Store.NO);
195 
196     for (int iterCnt = 0; iterCnt<iter; iterCnt++) {
197       IndexWriter iw = new IndexWriter(
198           dir,
199           new IndexWriterConfig(new SimpleAnalyzer()).
200               setOpenMode(IndexWriterConfig.OpenMode.CREATE)
201       );
202       final MyDoc[] mydocs = new MyDoc[ndocs];
203 
204       int v1EmptyPercent = 50;
205       int v2EmptyPercent = 50;
206 
207       int commitCountdown = commitCount;
208       for (int i=0; i< ndocs; i++) {
209         MyDoc mydoc = new MyDoc();
210         mydoc.doc = i;
211         mydocs[i] = mydoc;
212 
213         Document document = new Document();
214         if (r.nextInt(100) < v1EmptyPercent) {
215           mydoc.val = Integer.toString(r.nextInt(maxval));
216           f.setStringValue(mydoc.val);
217           document.add(f);
218         }
219         if (r.nextInt(100) < v2EmptyPercent) {
220           mydoc.val2 = Integer.toString(r.nextInt(maxval));
221           f2.setStringValue(mydoc.val2);
222           document.add(f2);
223         }
224 
225 
226         iw.addDocument(document);
227         if (--commitCountdown <= 0) {
228           commitCountdown = commitCount;
229           iw.commit();
230         }
231       }
232       iw.close();
233 
234       Map<String,UninvertingReader.Type> mapping = new HashMap<>();
235       mapping.put("f", UninvertingReader.Type.SORTED);
236       mapping.put("f2", UninvertingReader.Type.SORTED);
237 
238       DirectoryReader reader = UninvertingReader.wrap(DirectoryReader.open(dir), mapping);
239       IndexSearcher searcher = new IndexSearcher(reader);
240       // System.out.println("segments="+searcher.getIndexReader().getSequentialSubReaders().length);
241       assertTrue(reader.leaves().size() > 1);
242 
243       for (int i=0; i<qiter; i++) {
244         Filter filt = new Filter() {
245           @Override
246           public DocIdSet getDocIdSet(LeafReaderContext context, Bits acceptDocs) {
247             return BitsFilteredDocIdSet.wrap(randSet(context.reader().maxDoc()), acceptDocs);
248           }
249           @Override
250           public String toString(String field) {
251             return "TestSortFilter";
252           }
253         };
254 
255         int top = r.nextInt((ndocs>>3)+1)+1;
256         final boolean luceneSort = r.nextBoolean();
257         final boolean sortMissingLast = !luceneSort && r.nextBoolean();
258         final boolean sortMissingFirst = !luceneSort && !sortMissingLast;
259         final boolean reverse = r.nextBoolean();
260         List<SortField> sfields = new ArrayList<>();
261 
262         final boolean secondary = r.nextBoolean();
263         final boolean luceneSort2 = r.nextBoolean();
264         final boolean sortMissingLast2 = !luceneSort2 && r.nextBoolean();
265         final boolean sortMissingFirst2 = !luceneSort2 && !sortMissingLast2;
266         final boolean reverse2 = r.nextBoolean();
267 
268         if (r.nextBoolean()) sfields.add( new SortField(null, SortField.Type.SCORE));
269         // hit both use-cases of sort-missing-last
270         sfields.add( Sorting.getStringSortField("f", reverse, sortMissingLast, sortMissingFirst) );
271         if (secondary) {
272           sfields.add( Sorting.getStringSortField("f2", reverse2, sortMissingLast2, sortMissingFirst2) );
273         }
274         if (r.nextBoolean()) sfields.add( new SortField(null, SortField.Type.SCORE));
275 
276         Sort sort = new Sort(sfields.toArray(new SortField[sfields.size()]));
277 
278         final String nullRep = luceneSort || sortMissingFirst && !reverse || sortMissingLast && reverse ? "" : "zzz";
279         final String nullRep2 = luceneSort2 || sortMissingFirst2 && !reverse2 || sortMissingLast2 && reverse2 ? "" : "zzz";
280 
281         boolean trackScores = r.nextBoolean();
282         boolean trackMaxScores = r.nextBoolean();
283         boolean scoreInOrder = r.nextBoolean();
284         final TopFieldCollector topCollector = TopFieldCollector.create(sort, top, true, trackScores, trackMaxScores);
285 
286         final List<MyDoc> collectedDocs = new ArrayList<>();
287         // delegate and collect docs ourselves
288         Collector myCollector = new FilterCollector(topCollector) {
289 
290           @Override
291           public LeafCollector getLeafCollector(LeafReaderContext context)
292               throws IOException {
293             final int docBase = context.docBase;
294             return new FilterLeafCollector(super.getLeafCollector(context)) {
295               @Override
296               public void collect(int doc) throws IOException {
297                 super.collect(doc);
298                 collectedDocs.add(mydocs[docBase + doc]);
299               }
300             };
301           }
302 
303         };
304 
305         searcher.search(new FilteredQuery(new MatchAllDocsQuery(), filt), myCollector);
306 
307         Collections.sort(collectedDocs, new Comparator<MyDoc>() {
308           @Override
309           public int compare(MyDoc o1, MyDoc o2) {
310             String v1 = o1.val==null ? nullRep : o1.val;
311             String v2 = o2.val==null ? nullRep : o2.val;
312             int cmp = v1.compareTo(v2);
313             if (reverse) cmp = -cmp;
314             if (cmp != 0) return cmp;
315 
316             if (secondary) {
317                v1 = o1.val2==null ? nullRep2 : o1.val2;
318                v2 = o2.val2==null ? nullRep2 : o2.val2;
319                cmp = v1.compareTo(v2);
320                if (reverse2) cmp = -cmp;
321             }
322 
323             cmp = cmp==0 ? o1.doc-o2.doc : cmp;
324             return cmp;
325           }
326         });
327 
328 
329         TopDocs topDocs = topCollector.topDocs();
330         ScoreDoc[] sdocs = topDocs.scoreDocs;
331         for (int j=0; j<sdocs.length; j++) {
332           int id = sdocs[j].doc;
333           if (id != collectedDocs.get(j).doc) {
334             log.error("Error at pos " + j
335             + "\n\tsortMissingFirst=" + sortMissingFirst + " sortMissingLast=" + sortMissingLast + " reverse=" + reverse
336             + "\n\tEXPECTED=" + collectedDocs 
337             );
338           }
339           assertEquals(id, collectedDocs.get(j).doc);
340         }
341       }
342       reader.close();
343     }
344     dir.close();
345 
346   }

348   public DocIdSet randSet(int sz) {
349     FixedBitSet obs = new FixedBitSet(sz);
350     int n = r.nextInt(sz);
351     for (int i=0; i<n; i++) {
352       obs.set(r.nextInt(sz));
353     }
354     return new BitDocIdSet(obs);
355   }
