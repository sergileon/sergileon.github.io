77    /** Called by {@link org.apache.solr.request.SimpleFacets} to compute heatmap facets. */
78    public static NamedList<Object> getHeatmapForField(String fieldKey, String fieldName, final ResponseBuilder rb, SolrParams params, final DocSet docSet) throws IOException {
79      //get the strategy from the field type
80      final SchemaField schemaField = rb.req.getSchema().getField(fieldName);
81      final FieldType type = schemaField.getType();
82  
83      final PrefixTreeStrategy strategy;
84      final DistanceUnits distanceUnits;
85      // note: the two instanceof conditions is not ideal, versus one. If we start needing to add more then refactor.
86      if ((type instanceof AbstractSpatialPrefixTreeFieldType)) {
87        AbstractSpatialPrefixTreeFieldType rptType = (AbstractSpatialPrefixTreeFieldType) type;
88        strategy = (PrefixTreeStrategy) rptType.getStrategy(fieldName);
89        distanceUnits = rptType.getDistanceUnits();
90      } else if (type instanceof RptWithGeometrySpatialField) {
91        RptWithGeometrySpatialField rptSdvType  = (RptWithGeometrySpatialField) type;
92        strategy = rptSdvType.getStrategy(fieldName).getIndexStrategy();
93        distanceUnits = rptSdvType.getDistanceUnits();
94      } else {
95        //FYI we support the term query one too but few people use that one
96        throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, "heatmap field needs to be of type "
97            + SpatialRecursivePrefixTreeFieldType.class + " or " + RptWithGeometrySpatialField.class);
98      }
99  
100     final SpatialContext ctx = strategy.getSpatialContext();
101 
102     //get the bbox (query Rectangle)
103     String geomStr = params.getFieldParam(fieldKey, FacetParams.FACET_HEATMAP_GEOM);
104     final Shape boundsShape = geomStr == null ? ctx.getWorldBounds() : SpatialUtils.parseGeomSolrException(geomStr, ctx);
105 
106     //get the grid level (possibly indirectly via distErr or distErrPct)
107     final int gridLevel;
108     Integer gridLevelObj = params.getFieldInt(fieldKey, FacetParams.FACET_HEATMAP_LEVEL);
109     final int maxGridLevel = strategy.getGrid().getMaxLevels();
110     if (gridLevelObj != null) {
111       gridLevel = gridLevelObj;
112       if (gridLevel <= 0 || gridLevel > maxGridLevel) {
113         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
114             FacetParams.FACET_HEATMAP_LEVEL +" should be > 0 and <= " + maxGridLevel);
115       }
116     } else {
117       //SpatialArgs has utility methods to resolve a 'distErr' from optionally set distErr & distErrPct. Arguably that
118       // should be refactored to feel less weird than using it like this.
119       SpatialArgs spatialArgs = new SpatialArgs(SpatialOperation.Intersects/*ignored*/,
120           boundsShape == null ? ctx.getWorldBounds() : boundsShape);
121       final Double distErrObj = params.getFieldDouble(fieldKey, FacetParams.FACET_HEATMAP_DIST_ERR);
122       if (distErrObj != null) {
123         // convert distErr units based on configured units
124         spatialArgs.setDistErr(distErrObj * distanceUnits.multiplierFromThisUnitToDegrees());
125       }
126       spatialArgs.setDistErrPct(params.getFieldDouble(fieldKey, FacetParams.FACET_HEATMAP_DIST_ERR_PCT));
127       double distErr = spatialArgs.resolveDistErr(ctx, DEFAULT_DIST_ERR_PCT);
128       if (distErr <= 0) {
129         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
130             FacetParams.FACET_HEATMAP_DIST_ERR_PCT + " or " + FacetParams.FACET_HEATMAP_DIST_ERR
131                 + " should be > 0 or instead provide " + FacetParams.FACET_HEATMAP_LEVEL + "=" + maxGridLevel
132                 + " if you insist on maximum detail");
133       }
134       //The SPT (grid) can lookup a grid level satisfying an error distance constraint
135       gridLevel = strategy.getGrid().getLevelForDistance(distErr);
136     }
137 
138     // Turn docSet into Bits
139     Bits topAcceptDocs;
140     if (docSet instanceof BitDocSet) {
141       BitDocSet set = (BitDocSet) docSet;
142       topAcceptDocs = set.getBits();
143     } else {
144       topAcceptDocs = new Bits() {
145         @Override
146         public boolean get(int index) {
147           return docSet.exists(index);
148         }
149 
150         @Override
151         public int length() {
152           return rb.req.getSearcher().maxDoc();
153         }
154       };
155     }
156 
157     //Compute!
158     final HeatmapFacetCounter.Heatmap heatmap;
159     try {
160       heatmap = HeatmapFacetCounter.calcFacets(
161           strategy,
162           rb.req.getSearcher().getTopReaderContext(),
163           topAcceptDocs,
164           boundsShape,
165           gridLevel,
166           params.getFieldInt(fieldKey, FacetParams.FACET_HEATMAP_MAX_CELLS, 100_000) // will throw if exceeded
167       );
168     } catch (IllegalArgumentException e) {//e.g. too many cells
169       throw new SolrException(SolrException.ErrorCode.BAD_REQUEST, e.toString(), e);
170     }
171 
172     //Populate response
173     NamedList<Object> result = new NamedList<>();
174     result.add("gridLevel", gridLevel);
175     result.add("columns", heatmap.columns);
176     result.add("rows", heatmap.rows);
177     result.add("minX", heatmap.region.getMinX());
178     result.add("maxX", heatmap.region.getMaxX());
179     result.add("minY", heatmap.region.getMinY());
180     result.add("maxY", heatmap.region.getMaxY());
181 
182     boolean hasNonZero = false;
183     for (int count : heatmap.counts) {
184       if (count > 0) {
185         hasNonZero = true;
186         break;
187       }
188     }
189     formatCountsAndAddToNL(fieldKey, rb, params, heatmap.columns, heatmap.rows, hasNonZero ? heatmap.counts : null, result);
190 
191     return result;
192   }

194   private static void formatCountsAndAddToNL(String fieldKey, ResponseBuilder rb, SolrParams params,
195                                              int columns, int rows, int[] counts, NamedList<Object> result) {
196     final String format = params.getFieldParam(fieldKey, FacetParams.FACET_HEATMAP_FORMAT, FORMAT_INTS2D);
197     final Object countsVal;
198     switch (format) {
199       case FORMAT_INTS2D: //A List of List of Integers. Good for small heatmaps and ease of consumption
200         countsVal = counts != null ? asInts2D(columns, rows, counts) : null;
201         break;
202       case FORMAT_PNG: //A PNG graphic; compressed.  Good for large & dense heatmaps; hard to consume.
203         countsVal = counts != null ? asPngBytes(columns, rows, counts, rb) : null;
204         break;
205       //TODO  case skipList: //A sequence of values; negative values are actually how many 0's to insert.
206       //            Good for small or large but sparse heatmaps.
207       //TODO    auto choose png or skipList; use skipList when < ~25% full or <= ~512 cells
208       //  remember to augment error list below when we add more formats.
209       default:
210         throw new SolrException(SolrException.ErrorCode.BAD_REQUEST,
211             "format should be " + FORMAT_INTS2D + " or " + FORMAT_PNG);
212     }
213     result.add("counts_" + format, countsVal);
214   }

216   static List<List<Integer>> asInts2D(final int columns, final int rows, final int[] counts) {
217     //Returns a view versus returning a copy. This saves memory.
218     //The data is oriented naturally for human/developer viewing: one row at a time top-down
219     return new AbstractList<List<Integer>>() {
220       @Override
221       public List<Integer> get(final int rowIdx) {//top-down remember; the heatmap.counts is bottom up
222         //check if all zeroes and return null if so
223         boolean hasNonZero = false;
224         final int y = rows - rowIdx - 1;//flip direction for 'y'
225         for (int c = 0; c < columns; c++) {
226           if (counts[c * rows + y] > 0) {
227             hasNonZero = true;
228             break;
229           }
230         }
231         if (!hasNonZero) {
232           return null;
233         }
234 
235         return new AbstractList<Integer>() {
236           @Override
237           public Integer get(int columnIdx) {
238             return counts[columnIdx * rows + y];
239           }
240 
241           @Override
242           public int size() {
243             return columns;
244           }
245         };
246       }
247 
248       @Override
249       public int size() {
250         return rows;
251       }
252     };
253   }

255   //package access for tests
256   static byte[] asPngBytes(final int columns, final int rows, final int[] counts, ResponseBuilder rb) {
257     long startTimeNano = System.nanoTime();
258       BufferedImage image = PngHelper.newImage(columns, rows);
259       for (int c = 0; c < columns; c++) {
260         for (int r = 0; r < rows; r++) {
261           PngHelper.writeCountAtColumnRow(image, rows, c, r, counts[c * rows + r]);
262         }
263       }
264       byte[] bytes = PngHelper.writeImage(image);
265       long durationMs = TimeUnit.NANOSECONDS.toMillis(System.nanoTime() - startTimeNano);
266       log.debug("heatmap nativeSize={} pngSize={} pngTime={}", (counts.length * 4), bytes.length, durationMs);
267       if (rb != null && rb.isDebugTimings()) {
268         rb.addDebug(durationMs, "timing", "heatmap png generation");
269       }
270       return bytes;
271   }

277   /** Parses request to "HeatmapFacet" instances. */
278   public static LinkedHashMap<String,HeatmapFacet> distribParse(SolrParams params, ResponseBuilder rb) {
279     final LinkedHashMap<String, HeatmapFacet> heatmapFacets = new LinkedHashMap<>();
280     final String[] heatmapFields = params.getParams(FacetParams.FACET_HEATMAP);
281     if (heatmapFields != null) {
282       for (String heatmapField : heatmapFields) {
283         HeatmapFacet facet = new HeatmapFacet(rb, heatmapField);
284         heatmapFacets.put(facet.getKey(), facet);
285       }
286     }
287     return heatmapFacets;
288   }

290   /** Called by FacetComponent's impl of
291    * {@link org.apache.solr.handler.component.SearchComponent#modifyRequest(ResponseBuilder, SearchComponent, ShardRequest)}. */
292   public static void distribModifyRequest(ShardRequest sreq, LinkedHashMap<String, HeatmapFacet> heatmapFacets) {
293     // Set the format to PNG because it's compressed and it's the only format we have code to read at the moment.
294     //  Changing a param is sadly tricky because field-specific params can show up as local-params (highest precedence)
295     //  or as f.key.facet.heatmap.whatever. Ugh. So we re-write the facet.heatmap list with the local-params
296     //  moved out to the "f.key." prefix, but we need to keep the key local-param because that's the only way to
297     //  set an output key. This approach means we only need to know about the parameter we're changing, not of
298     //  all possible heatmap params.
299 
300     //Remove existing heatmap field param vals; we will rewrite
301     sreq.params.remove(FacetParams.FACET_HEATMAP);
302     for (Map.Entry<String, HeatmapFacet> entry : heatmapFacets.entrySet()) {
303       final String key = entry.getKey();
304       final HeatmapFacet facet = entry.getValue();
305       //add heatmap field param
306       if (!key.equals(facet.facetOn)) {
307         sreq.params.add(FacetParams.FACET_HEATMAP,
308             "{!" + CommonParams.OUTPUT_KEY + "=" + QueryParsing.encodeLocalParamVal(key) + "}" + facet.facetOn);
309       } else {
310         sreq.params.add(FacetParams.FACET_HEATMAP, facet.facetOn);
311       }
312       // Turn local-params into top-level f.key.param=value style params
313       if (facet.localParams != null) {
314         final Iterator<String> localNameIter = facet.localParams.getParameterNamesIterator();
315         while (localNameIter.hasNext()) {
316           String pname = localNameIter.next();
317           if (!pname.startsWith(FacetParams.FACET_HEATMAP)) {
318             continue; // could be 'key', or 'v' even
319           }
320           String pval = facet.localParams.get(pname);
321           sreq.params.set("f." + key + "." + pname, pval);
322         }
323       }
324       // Set format to PNG; it's the only one we parse
325       sreq.params.set("f." + key + "." + FacetParams.FACET_HEATMAP_FORMAT, FORMAT_PNG);
326     }
327   }

329   /** Called by FacetComponent.countFacets which is in turn called by FC's impl of
330    * {@link org.apache.solr.handler.component.SearchComponent#handleResponses(ResponseBuilder, ShardRequest)}. */
331   @SuppressWarnings("unchecked")
332   public static void distribHandleResponse(LinkedHashMap<String, HeatmapFacet> heatmapFacets, NamedList srsp_facet_counts) {
333     NamedList<NamedList<Object>> facet_heatmaps = (NamedList<NamedList<Object>>) srsp_facet_counts.get(RESPONSE_KEY);
334     if (facet_heatmaps == null) {
335       return;
336     }
337     // (should the caller handle the above logic?  Arguably yes.)
338     for (Map.Entry<String, NamedList<Object>> entry : facet_heatmaps) {
339       String fieldKey = entry.getKey();
340       NamedList<Object> shardNamedList = entry.getValue();
341       final HeatmapFacet facet = heatmapFacets.get(fieldKey);
342       if (facet == null) {
343         log.error("received heatmap for field/key {} that we weren't expecting", fieldKey);
344         continue;
345       }
346       facet.counts = addPngToIntArray((byte[]) shardNamedList.remove("counts_" + FORMAT_PNG), facet.counts);
347       if (facet.namedList == null) {
348         // First shard
349         facet.namedList = shardNamedList;
350       } else {
351         assert facet.namedList.equals(shardNamedList);
352       }
353     }
354   }

356   //package access for tests
357   static int[] addPngToIntArray(byte[] pngBytes, int[] counts) {
358     if (pngBytes == null) {
359       return counts;
360     }
361     //read PNG
362     final BufferedImage image = PngHelper.readImage(pngBytes);
363     int columns = image.getWidth();
364     int rows = image.getHeight();
365     if (counts == null) {
366       counts = new int[columns * rows];
367     } else {
368       assert counts.length == columns * rows;
369     }
370     for (int c = 0; c < columns; c++) {
371       for (int r = 0; r < rows; r++) {
372         counts[c * rows + r] += PngHelper.getCountAtColumnRow(image, rows, c, r);
373       }
374     }
375     return counts;
376   }

378   /** Called by FacetComponent's impl of
379    * {@link org.apache.solr.handler.component.SearchComponent#finishStage(ResponseBuilder)}. */
380   public static NamedList distribFinish(LinkedHashMap<String, HeatmapFacet> heatmapInfos, ResponseBuilder rb) {
381     NamedList<NamedList<Object>> result = new SimpleOrderedMap<>();
382     for (Map.Entry<String, HeatmapFacet> entry : heatmapInfos.entrySet()) {
383       final HeatmapFacet facet = entry.getValue();
384       final NamedList<Object> namedList = facet.namedList;
385       if (namedList == null) {
386         continue;//should never happen but play it safe
387       }
388       formatCountsAndAddToNL(entry.getKey(), rb, SolrParams.wrapDefaults(facet.localParams, rb.req.getParams()),
389           (int) namedList.get("columns"), (int) namedList.get("rows"), facet.counts, namedList);
390       result.add(entry.getKey(), namedList);
391     }
392     return result;
393   }

415   //package access for tests
416   static class PngHelper {
417 
418     static final ImageReaderSpi imageReaderSpi;//thread-safe
419     static {
420       final Iterator<ImageReader> imageReaders = ImageIO.getImageReadersByFormatName("png");
421       if (!imageReaders.hasNext()) {
422         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Can't find png image reader, neaded for heatmaps!");
423       }
424       ImageReader imageReader = imageReaders.next();
425       imageReaderSpi = imageReader.getOriginatingProvider();
426     }
427 
428     static BufferedImage readImage(final byte[] bytes) {
429       // Wrap ImageInputStream around the bytes.  We could use MemoryCacheImageInputStream but it will
430       // cache the data which is quite unnecessary given we have it all in-memory already.
431       ImageInputStream imageInputStream = new ImageInputStreamImpl() {
432         //TODO re-use this instance; superclass has 8KB buffer.
433 
434         @Override
435         public int read() throws IOException {
436           checkClosed();
437           bitOffset = 0;
438           if (streamPos >= bytes.length) {
439             return -1;
440           } else {
441             return bytes[(int) streamPos++];
442           }
443         }
444 
445         @Override
446         public int read(byte[] b, int off, int len) throws IOException {
447           checkClosed();
448           bitOffset = 0;
449           if (streamPos >= bytes.length) {
450             return -1;
451           } else {
452             int copyLen = Math.min(len, bytes.length - (int)streamPos);
453             System.arraycopy(bytes, (int)streamPos, b, off, copyLen);
454             streamPos += copyLen;
455             return copyLen;
456           }
457         }
458 
459         @Override
460         public long length() {
461           return bytes.length;
462         }
463 
464         @Override
465         public boolean isCached() {
466           return true;
467         }
468 
469         @Override
470         public boolean isCachedMemory() {
471           return true;
472         }
473       };
474       try {
475         //TODO can/should we re-use an imageReader instance on FacetInfo?
476         ImageReader imageReader = imageReaderSpi.createReaderInstance();
477 
478         imageReader.setInput(imageInputStream,
479             false,//forwardOnly
480             true);//ignoreMetadata
481         return imageReader.read(0);//read first & only image
482       } catch (IOException e) {
483         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Problem reading png heatmap: " + e);
484       }
485     }
486 
487     static byte[] writeImage(BufferedImage image) {
488       ByteArrayOutputStream baos = new ByteArrayOutputStream(
489           // initialize to roughly 1/4th the size a native int would take per-pixel
490           image.getWidth() * image.getHeight() + 1024
491       );
492       try {
493         ImageIO.write(image, FORMAT_PNG, baos);
494       } catch (IOException e) {
495         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "While generating PNG: " + e);
496       }
497       //too bad we can't access the raw byte[]; this copies to a new one
498       return baos.toByteArray();
499     }
500 
501     // We abuse the image for storing integers (4 bytes), and so we need a 4-byte ABGR.
502     // first (low) byte is blue, next byte is green, next byte red, and last (high) byte is alpha.
503     static BufferedImage newImage(int columns, int rows) {
504       return new BufferedImage(columns, rows, BufferedImage.TYPE_4BYTE_ABGR);
505     }
506 
507     // 'y' dimension goes top-down, so invert.
508     // Alpha chanel is high byte; 0 means transparent. So XOR those bits with '1' so that we need
509     //  to have counts > 16M before the picture starts to fade
510 
511     static void writeCountAtColumnRow(BufferedImage image, int rows, int c, int r, int val) {
512       image.setRGB(c, rows - 1 - r, val ^ 0xFF_00_00_00);
513     }
514 
515     static int getCountAtColumnRow(BufferedImage image, int rows, int c, int r) {
516       return image.getRGB(c, rows - 1 - r) ^ 0xFF_00_00_00;
517     }
518 
519   }
