44  /**
45   * Tests the {@link SimilarityBase}-based Similarities. Contains unit tests and 
46   * integration tests for all Similarities and correctness tests for a select
47   * few.
48   * <p>This class maintains a list of
49   * {@code SimilarityBase} subclasses. Each test case performs its test on all
50   * items in the list. If a test case fails, the name of the Similarity that
51   * caused the failure is returned as part of the assertion error message.</p>
52   * <p>Unit testing is performed by constructing statistics manually and calling
53   * the {@link SimilarityBase#score(BasicStats, float, float)} method of the
54   * Similarities. The statistics represent corner cases of corpus distributions.
55   * </p>
56   * <p>For the integration tests, a small (8-document) collection is indexed. The
57   * tests verify that for a specific query, all relevant documents are returned
58   * in the correct order. The collection consists of two poems of English poet
59   * <a href="http://en.wikipedia.org/wiki/William_blake">William Blake</a>.</p>
60   * <p>Note: the list of Similarities is maintained by hand. If a new Similarity
61   * is added to the {@code org.apache.lucene.search.similarities} package, the
62   * list should be updated accordingly.</p>
63   * <p>
64   * In the correctness tests, the score is verified against the result of manual
65   * computation. Since it would be impossible to test all Similarities
66   * (e.g. all possible DFR combinations, all parameter values for LM), only 
67   * the best performing setups in the original papers are verified.
68   * </p>
69   */
70  public class TestSimilarityBase extends LuceneTestCase {

167   private BasicStats createStats() {

177   private CollectionStatistics toCollectionStats(BasicStats stats) {
178     return new CollectionStatistics(stats.field, stats.getNumberOfDocuments(), -1, stats.getNumberOfFieldTokens(), -1);
179   }
180   
181   private TermStatistics toTermStats(BasicStats stats) {
182     return new TermStatistics(new BytesRef("spoofyText"), stats.getDocFreq(), stats.getTotalTermFreq());

184   /**
185    * The generic test core called by all unit test methods. It calls the
186    * {@link SimilarityBase#score(BasicStats, float, float)} method of all
187    * Similarities in {@link #sims} and checks if the score is valid; i.e. it
188    * is a finite positive real number.
189    */
190   private void unitTestCore(BasicStats stats, float freq, int docLen) {
191     for (SimilarityBase sim : sims) {
192       BasicStats realStats = (BasicStats) sim.computeWeight(
193           toCollectionStats(stats), 
194           toTermStats(stats));
195       realStats.normalize(1f, stats.getBoost());
196       float score = sim.score(realStats, freq, docLen);
197       float explScore = sim.explain(
198           realStats, 1, Explanation.match(freq, "freq"), docLen).getValue();
199       assertFalse("Score infinite: " + sim.toString(), Float.isInfinite(score));
200       assertFalse("Score NaN: " + sim.toString(), Float.isNaN(score));
201       assertTrue("Score negative: " + sim.toString(), score >= 0);
202       assertEquals("score() and explain() return different values: "
203           + sim.toString(), score, explScore, FLOAT_EPSILON);
204     }
205   }

390   /** Correctness test for the Dirichlet LM model. */
391   public void testLMDirichlet() throws IOException {
392     float p =
393         (FREQ + 2000.0f * (TOTAL_TERM_FREQ + 1) / (NUMBER_OF_FIELD_TOKENS + 1.0f)) /
394         (DOC_LEN + 2000.0f);
395     float a = 2000.0f / (DOC_LEN + 2000.0f);
396     float gold = (float)(
397         Math.log(p / (a * (TOTAL_TERM_FREQ + 1) / (NUMBER_OF_FIELD_TOKENS + 1.0f))) +
398         Math.log(a));
399     correctnessTestCore(new LMDirichletSimilarity(), gold);
400   }

430   /** Correctness test for the PL2 DFR model. */
431   public void testPL2() throws IOException {
432     SimilarityBase sim = new DFRSimilarity(
433         new BasicModelP(), new AfterEffectL(), new NormalizationH2());
434     float tfn = (float)(FREQ * SimilarityBase.log2(
435         1 + AVG_FIELD_LENGTH / DOC_LEN));  // 8.1894750101
436     float l = 1.0f / (tfn + 1.0f);         // 0.108820144666
437     float lambda = (1.0f + TOTAL_TERM_FREQ) / (1f + NUMBER_OF_DOCUMENTS);  // 0.7029703
438     float p = (float)(tfn * SimilarityBase.log2(tfn / lambda) +
439               (lambda + 1 / (12 * tfn) - tfn) * SimilarityBase.log2(Math.E) +
440               0.5 * SimilarityBase.log2(2 * Math.PI * tfn)); // 21.065619
441     float gold = l * p;                    // 2.2923636
442     correctnessTestCore(sim, gold);
443   }

459   /** Correctness test for the BEB1 DFR model. */
460   public void testBEB1() throws IOException {
461     SimilarityBase sim = new DFRSimilarity(
462         new BasicModelBE(), new AfterEffectB(), new NormalizationH1());
463     float tfn = FREQ * AVG_FIELD_LENGTH / DOC_LEN;  // 8.75
464     float b = (TOTAL_TERM_FREQ + 1 + 1) / ((DOC_FREQ + 1) * (tfn + 1));  // 0.67132866
465     double f = TOTAL_TERM_FREQ + 1 + tfn;
466     double n = f + NUMBER_OF_DOCUMENTS;
467     double n1 = n + f - 1;        // 258.5
468     double m1 = n + f - tfn - 2;  // 248.75
469     double n2 = f;                                      // 79.75
470     double m2 = f - tfn;                                // 71.0
471     float be = (float)(-SimilarityBase.log2(n - 1) -
472                SimilarityBase.log2(Math.E) +                   // -8.924494472554715
473                ((m1 + 0.5f) * SimilarityBase.log2(n1 / m1) +
474                 (n1 - m1) * SimilarityBase.log2(n1)) -         // 91.9620374903885
475                ((m2 + 0.5f) * SimilarityBase.log2(n2 / m2) +
476                 (n2 - m2) * SimilarityBase.log2(n2)));         // 67.26544321004599
477                // 15.7720995
478     float gold = b * be;                                       // 10.588263
479     correctnessTestCore(sim, gold);
480   }

482   /** Correctness test for the D DFR model (basic model only). */
483   public void testD() throws IOException {
484     SimilarityBase sim = new DFRSimilarity(new BasicModelD(), new AfterEffect.NoAfterEffect(), new Normalization.NoNormalization());
485     double totalTermFreqNorm = TOTAL_TERM_FREQ + FREQ + 1;
486     double p = 1.0 / (NUMBER_OF_DOCUMENTS + 1);                // 0.009900990099009901
487     double phi = FREQ / totalTermFreqNorm;                       // 0.08974358974358974
488     double D = phi * SimilarityBase.log2(phi / p) +            // 0.17498542370019005
489               (1 - phi) * SimilarityBase.log2((1 - phi) / (1 - p));
490     float gold = (float)(totalTermFreqNorm * D + 0.5 * SimilarityBase.log2(
491                  1 + 2 * Math.PI * FREQ * (1 - phi)));         // 16.328257
492     correctnessTestCore(sim, gold);
493   }

517   /**
518    * The generic test core called by all correctness test methods. It calls the
519    * {@link SimilarityBase#score(BasicStats, float, float)} method of all
520    * Similarities in {@link #sims} and compares the score against the manually
521    * computed {@code gold}.
522    */
523   private void correctnessTestCore(SimilarityBase sim, float gold) {
524     BasicStats stats = createStats();
525     BasicStats realStats = (BasicStats) sim.computeWeight(
526         toCollectionStats(stats), 
527         toTermStats(stats));
528     realStats.normalize(1f, stats.getBoost());
529     float score = sim.score(realStats, FREQ, DOC_LEN);
530     assertEquals(
531         sim.toString() + " score not correct.", gold, score, FLOAT_EPSILON);
532   }
