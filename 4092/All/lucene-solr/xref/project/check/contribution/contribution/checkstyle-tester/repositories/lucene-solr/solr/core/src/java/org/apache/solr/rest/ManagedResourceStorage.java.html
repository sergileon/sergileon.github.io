54  /**
55   * Abstract base class that provides most of the functionality needed
56   * to store arbitrary data for managed resources. Concrete implementations
57   * need to decide the underlying format that data is stored in, such as JSON.
58   * 
59   * The underlying storage I/O layer will be determined by the environment
60   * Solr is running in, e.g. in cloud mode, data will be stored and loaded
61   * from ZooKeeper.
62   */
63  public abstract class ManagedResourceStorage {
64    
65    /**
66     * Hides the underlying storage implementation for data being managed
67     * by a ManagedResource. For instance, a ManagedResource may use JSON as
68     * the data format and an instance of this class to persist and load 
69     * the JSON bytes to/from some backing store, such as ZooKeeper.
70     */
71    public static interface StorageIO {
72      String getInfo();
73      void configure(SolrResourceLoader loader, NamedList<String> initArgs) throws SolrException;
74      boolean exists(String storedResourceId) throws IOException;
75      InputStream openInputStream(String storedResourceId) throws IOException;  
76      OutputStream openOutputStream(String storedResourceId) throws IOException;
77      boolean delete(String storedResourceId) throws IOException;
78    }
79    
80    public static final String STORAGE_IO_CLASS_INIT_ARG = "storageIO"; 
81    public static final String STORAGE_DIR_INIT_ARG = "storageDir";
82    
83    /**
84     * Creates a new StorageIO instance for a Solr core, taking into account
85     * whether the core is running in cloud mode as well as initArgs. 
86     */
87    public static StorageIO newStorageIO(String collection, SolrResourceLoader resourceLoader, NamedList<String> initArgs) {
88      StorageIO storageIO = null;
89  
90      SolrZkClient zkClient = null;
91      String zkConfigName = null;
92      if (resourceLoader instanceof ZkSolrResourceLoader) {
93        zkClient = ((ZkSolrResourceLoader)resourceLoader).getZkController().getZkClient();
94        try {
95          zkConfigName = ((ZkSolrResourceLoader)resourceLoader).getZkController().
96              getZkStateReader().readConfigName(collection);
97        } catch (Exception e) {
98          log.error("Failed to get config name for collection {} due to: {}", 
99              collection, e.toString());
100       } 
101       if (zkConfigName == null) {
102         throw new SolrException(ErrorCode.SERVER_ERROR, 
103             "Could not find config name for collection:" + collection);
104       }
105     }
106     
107     if (initArgs.get(STORAGE_IO_CLASS_INIT_ARG) != null) {
108       storageIO = resourceLoader.newInstance(initArgs.get(STORAGE_IO_CLASS_INIT_ARG), StorageIO.class); 
109     } else {
110       if (zkClient != null) {
111         String znodeBase = "/configs/"+zkConfigName;
112         log.info("Setting up ZooKeeper-based storage for the RestManager with znodeBase: "+znodeBase);      
113         storageIO = new ManagedResourceStorage.ZooKeeperStorageIO(zkClient, znodeBase);
114       } else {
115         storageIO = new FileStorageIO();        
116       }
117     }
118     
119     if (storageIO instanceof FileStorageIO) {
120       // using local fs, if storageDir is not set in the solrconfig.xml, assume the configDir for the core
121       if (initArgs.get(STORAGE_DIR_INIT_ARG) == null) {
122         File configDir = new File(resourceLoader.getConfigDir());
123         boolean hasAccess = false;
124         try {
125           hasAccess = configDir.isDirectory() && configDir.canWrite();
126         } catch (java.security.AccessControlException ace) {}
127         
128         if (hasAccess) {
129           initArgs.add(STORAGE_DIR_INIT_ARG, configDir.getAbsolutePath());
130         } else {
131           // most likely this is because of a unit test 
132           // that doesn't have write-access to the config dir
133           // while this failover approach is not ideal, it's better
134           // than causing the core to fail esp. if managed resources aren't being used
135           log.warn("Cannot write to config directory "+configDir.getAbsolutePath()+
136               "; switching to use InMemory storage instead.");
137           storageIO = new ManagedResourceStorage.InMemoryStorageIO();
138         }
139       }       
140     }
141     
142     storageIO.configure(resourceLoader, initArgs);     
143     
144     return storageIO;
145   }

147   /**
148    * Local file-based storage implementation.
149    */
150   public static class FileStorageIO implements StorageIO {
151 
152     private String storageDir;
153     
154     @Override
155     public void configure(SolrResourceLoader loader, NamedList<String> initArgs) throws SolrException {
156       String storageDirArg = initArgs.get(STORAGE_DIR_INIT_ARG);
157       
158       if (storageDirArg == null || storageDirArg.trim().length() == 0)
159         throw new IllegalArgumentException("Required configuration parameter '"+
160            STORAGE_DIR_INIT_ARG+"' not provided!");
161       
162       File dir = new File(storageDirArg);
163       if (!dir.isDirectory())
164         dir.mkdirs();
165 
166       storageDir = dir.getAbsolutePath();      
167       log.info("File-based storage initialized to use dir: "+storageDir);
168     }
169     
170     @Override
171     public boolean exists(String storedResourceId) throws IOException {
172       return (new File(storageDir, storedResourceId)).exists();
173     }    
174     
175     @Override
176     public InputStream openInputStream(String storedResourceId) throws IOException {
177       return new FileInputStream(storageDir+"/"+storedResourceId);
178     }
179 
180     @Override
181     public OutputStream openOutputStream(String storedResourceId) throws IOException {
182       return new FileOutputStream(storageDir+"/"+storedResourceId);
183     }
184 
185     @Override
186     public boolean delete(String storedResourceId) throws IOException {
187       File storedFile = new File(storageDir, storedResourceId);
188       return deleteIfFile(storedFile);
189     }
190     
191     // TODO: this interface should probably be changed, this simulates the old behavior,
192     // only throw security exception, just return false otherwise
193     private boolean deleteIfFile(File f) {
194       if (!f.isFile()) {
195         return false;
196       }
197       try {
198         Files.delete(f.toPath());
199         return true;
200       } catch (IOException cause) {
201         return false;
202       }
203     }
204 
205     @Override
206     public String getInfo() {
207       return "file:dir="+storageDir;
208     }
209   } // end FileStorageIO

211   /**
212    * ZooKeeper based storage implementation that uses the SolrZkClient provided
213    * by the CoreContainer.
214    */
215   public static class ZooKeeperStorageIO implements StorageIO {
216     
217     protected SolrZkClient zkClient;
218     protected String znodeBase;
219     protected boolean retryOnConnLoss = true;
220     
221     public ZooKeeperStorageIO(SolrZkClient zkClient, String znodeBase) {
222       this.zkClient = zkClient;  
223       this.znodeBase = znodeBase;
224     }
225 
226     @Override
227     public void configure(SolrResourceLoader loader, NamedList<String> initArgs) throws SolrException {
228       // validate connectivity and the configured znode base
229       try {
230         if (!zkClient.exists(znodeBase, retryOnConnLoss)) {
231           zkClient.makePath(znodeBase, retryOnConnLoss);
232         }
233       } catch (Exception exc) {
234         String errMsg = String.format
235             (Locale.ROOT, "Failed to verify znode at %s due to: %s", znodeBase, exc.toString());
236         log.error(errMsg, exc);
237         throw new SolrException(ErrorCode.SERVER_ERROR, errMsg, exc);
238       }
239       
240       log.info("Configured ZooKeeperStorageIO with znodeBase: "+znodeBase);      
241     }    
242     
243     @Override
244     public boolean exists(String storedResourceId) throws IOException {
245       final String znodePath = getZnodeForResource(storedResourceId);
246       try {
247         return zkClient.exists(znodePath, retryOnConnLoss);
248       } catch (Exception e) {
249         if (e instanceof IOException) {
250           throw (IOException)e;
251         } else {
252           throw new IOException("Failed to read data at "+znodePath, e);
253         }
254       }
255     }    
256     
257     @Override
258     public InputStream openInputStream(String storedResourceId) throws IOException {
259       final String znodePath = getZnodeForResource(storedResourceId);
260       byte[] znodeData = null;
261       try {
262         if (zkClient.exists(znodePath, retryOnConnLoss)) {
263           znodeData = zkClient.getData(znodePath, null, null, retryOnConnLoss);
264         }
265       } catch (Exception e) {
266         if (e instanceof IOException) {
267           throw (IOException)e;
268         } else {
269           throw new IOException("Failed to read data at "+znodePath, e);
270         }
271       }
272       
273       if (znodeData != null) {
274         log.info("Read {} bytes from znode {}", znodeData.length, znodePath);
275       } else {
276         znodeData = new byte[0];
277         log.info("No data found for znode {}", znodePath);
278       }
279       
280       return new ByteArrayInputStream(znodeData);
281     }
282 
283     @Override
284     public OutputStream openOutputStream(String storedResourceId) throws IOException {
285       final String znodePath = getZnodeForResource(storedResourceId);
286       final boolean retryOnConnLoss = this.retryOnConnLoss;
287       ByteArrayOutputStream baos = new ByteArrayOutputStream() {
288         @Override
289         public void close() {
290           byte[] znodeData = toByteArray();
291           try {
292             if (zkClient.exists(znodePath, retryOnConnLoss)) {
293               zkClient.setData(znodePath, znodeData, retryOnConnLoss);
294               log.info("Wrote {} bytes to existing znode {}", znodeData.length, znodePath);
295             } else {
296               zkClient.makePath(znodePath, znodeData, retryOnConnLoss);
297               log.info("Wrote {} bytes to new znode {}", znodeData.length, znodePath);
298             }
299           } catch (Exception e) {
300             // have to throw a runtimer here as we're in close, 
301             // which doesn't throw IOException
302             if (e instanceof RuntimeException) {
303               throw (RuntimeException)e;              
304             } else {
305               throw new ResourceException(Status.SERVER_ERROR_INTERNAL, 
306                   "Failed to save data to ZooKeeper znode: "+znodePath+" due to: "+e, e);
307             }
308           }
309         }
310       };
311       return baos;
312     }
313 
314     /**
315      * Returns the Znode for the given storedResourceId by combining it
316      * with the znode base.
317      */
318     protected String getZnodeForResource(String storedResourceId) {
319       return String.format(Locale.ROOT, "%s/%s", znodeBase, storedResourceId);
320     }
321 
322     @Override
323     public boolean delete(String storedResourceId) throws IOException {
324       boolean wasDeleted = false;
325       final String znodePath = getZnodeForResource(storedResourceId);
326       
327       // this might be overkill for a delete operation
328       try {
329         if (zkClient.exists(znodePath, retryOnConnLoss)) {
330           log.info("Attempting to delete znode {}", znodePath);
331           zkClient.delete(znodePath, -1, retryOnConnLoss);
332           wasDeleted = zkClient.exists(znodePath, retryOnConnLoss);
333           
334           if (wasDeleted) {
335             log.info("Deleted znode {}", znodePath);
336           } else {
337             log.warn("Failed to delete znode {}", znodePath);
338           }
339         } else {
340           log.warn("Znode {} does not exist; delete operation ignored.", znodePath);
341         }
342       } catch (Exception e) {
343         if (e instanceof IOException) {
344           throw (IOException)e;
345         } else {
346           throw new IOException("Failed to read data at "+znodePath, e);
347         }
348       }
349       
350       return wasDeleted;
351     }
352 
353     @Override
354     public String getInfo() {
355       return "ZooKeeperStorageIO:path="+znodeBase;
356     }
357   } // end ZooKeeperStorageIO

359   /**
360    * Memory-backed storage IO; not really intended for storage large amounts
361    * of data in production, but useful for testing and other transient workloads.
362    */
363   public static class InMemoryStorageIO implements StorageIO {
364     
365     Map<String,BytesRef> storage = new HashMap<>();
366     
367     @Override
368     public void configure(SolrResourceLoader loader, NamedList<String> initArgs)
369         throws SolrException {}
370 
371     @Override
372     public boolean exists(String storedResourceId) throws IOException {
373       return storage.containsKey(storedResourceId);
374     }
375     
376     @Override
377     public InputStream openInputStream(String storedResourceId)
378         throws IOException {
379       
380       BytesRef storedVal = storage.get(storedResourceId);
381       if (storedVal == null)
382         throw new FileNotFoundException(storedResourceId);
383       
384       return new ByteArrayInputStream(storedVal.bytes, storedVal.offset, storedVal.length);            
385     }
386 
387     @Override
388     public OutputStream openOutputStream(final String storedResourceId)
389         throws IOException {
390       ByteArrayOutputStream boas = new ByteArrayOutputStream() {
391         @Override
392         public void close() {
393           storage.put(storedResourceId, new BytesRef(toByteArray()));
394         }
395       };
396       return boas;
397     }
398 
399     @Override
400     public boolean delete(String storedResourceId) throws IOException {
401       return (storage.remove(storedResourceId) != null);
402     }
403 
404     @Override
405     public String getInfo() {
406       return "InMemoryStorage";
407     }
408   } // end InMemoryStorageIO class

410   /**
411    * Default storage implementation that uses JSON as the storage format for managed data.
412    */
413   public static class JsonStorage extends ManagedResourceStorage {
414     
415     public JsonStorage(StorageIO storageIO, SolrResourceLoader loader) {
416       super(storageIO, loader);
417     }
418 
419     /**
420      * Determines the relative path (from the storage root) for the given resource.
421      * In this case, it returns a file named with the .json extension.
422      */
423     @Override
424     public String getStoredResourceId(String resourceId) {
425       return resourceId.replace('/','_')+".json";
426     }  
427       
428     @Override
429     protected Object parseText(Reader reader, String resourceId) throws IOException {
430       return ObjectBuilder.getVal(new JSONParser(reader));    
431     }
432 
433     @Override
434     public void store(String resourceId, Object toStore) throws IOException {
435       String json = JSONUtil.toJSON(toStore);
436       String storedResourceId = getStoredResourceId(resourceId);
437       OutputStreamWriter writer = null;
438       try {
439         writer = new OutputStreamWriter(storageIO.openOutputStream(storedResourceId), UTF_8);
440         writer.write(json);
441         writer.flush();
442       } finally {
443         if (writer != null) {
444           try {
445             writer.close();
446           } catch (Exception ignore){}
447         }
448       }    
449       log.info("Saved JSON object to path {} using {}", 
450           storedResourceId, storageIO.getInfo());
451     }
452   } // end JsonStorage

476   /**
477    * Gets the unique identifier for a stored resource, typically based
478    * on the resourceId and some storage-specific information, such as
479    * file extension and storage root directory.
480    */
481   public abstract String getStoredResourceId(String resourceId);
482    
483   /**
484    * Loads a resource from storage; the default implementation makes
485    * the assumption that the data is stored as UTF-8 encoded text, 
486    * such as JSON. This method should be overridden if that assumption
487    * is invalid. 
488    */
489   public Object load(String resourceId) throws IOException {
490     String storedResourceId = getStoredResourceId(resourceId);
491     
492     log.info("Reading {} using {}", storedResourceId, storageIO.getInfo());
493     
494     InputStream inputStream = storageIO.openInputStream(storedResourceId);
495     if (inputStream == null) {
496       return null;
497     }
498     Object parsed = null;
499     InputStreamReader reader = null;
500     try {
501       reader = new InputStreamReader(inputStream, UTF_8);
502       parsed = parseText(reader, resourceId);
503     } finally {
504       if (reader != null) {
505         try {
506           reader.close();
507         } catch (Exception ignore){}
508       }
509     }
510     
511     String objectType = (parsed != null) ? parsed.getClass().getSimpleName() : "null"; 
512     log.info(String.format(Locale.ROOT, "Loaded %s at path %s using %s",
513                                         objectType, storedResourceId, storageIO.getInfo()));
514     
515     return parsed;
516   }
517 
518   /**
519    * Called by {@link ManagedResourceStorage#load(String)} to convert the
520    * serialized resource into its in-memory representation.
521    */
522   protected Object parseText(Reader reader, String resourceId) throws IOException {
523     // no-op: base classes should override this if they deal with text.
524     return null;
525   }
526 
527   /** Persists the given toStore object with the given resourceId. */
528   public abstract void store(String resourceId, Object toStore) throws IOException;
529 
530   /** Removes the given resourceId's persisted representation. */
531   public boolean delete(String resourceId) throws IOException {
532     return storageIO.delete(getStoredResourceId(resourceId));
533   }
