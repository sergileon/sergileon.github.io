106   @Test
107   public void testSpelling() throws Exception {
108     IndexBasedSpellChecker checker = new IndexBasedSpellChecker();
109 
110     NamedList spellchecker = new NamedList();
111     spellchecker.add("classname", IndexBasedSpellChecker.class.getName());
112 
113     File indexDir = createTempDir().toFile();
114 
115     spellchecker.add(AbstractLuceneSpellChecker.INDEX_DIR, indexDir.getAbsolutePath());
116     spellchecker.add(AbstractLuceneSpellChecker.FIELD, "title");
117     spellchecker.add(AbstractLuceneSpellChecker.SPELLCHECKER_ARG_NAME, spellchecker);
118     SolrCore core = h.getCore();
119 
120     String dictName = checker.init(spellchecker, core);
121     assertTrue(dictName + " is not equal to " + SolrSpellChecker.DEFAULT_DICTIONARY_NAME,
122             dictName.equals(SolrSpellChecker.DEFAULT_DICTIONARY_NAME) == true);
123     RefCounted<SolrIndexSearcher> holder = core.getSearcher();
124     SolrIndexSearcher searcher = holder.get();
125     try {
126     checker.build(core, searcher);
127 
128     IndexReader reader = searcher.getIndexReader();
129     Collection<Token> tokens = queryConverter.convert("documemt");
130     SpellingOptions spellOpts = new SpellingOptions(tokens, reader);
131     SpellingResult result = checker.getSuggestions(spellOpts);
132     assertTrue("result is null and it shouldn't be", result != null);
133     //should be lowercased, b/c we are using a lowercasing analyzer
134     Map<String, Integer> suggestions = result.get(spellOpts.tokens.iterator().next());
135     assertTrue("documemt is null and it shouldn't be", suggestions != null);
136     assertTrue("documemt Size: " + suggestions.size() + " is not: " + 1, suggestions.size() == 1);
137     Map.Entry<String, Integer> entry = suggestions.entrySet().iterator().next();
138     assertTrue(entry.getKey() + " is not equal to " + "document", entry.getKey().equals("document") == true);
139     assertTrue(entry.getValue() + " does not equal: " + SpellingResult.NO_FREQUENCY_INFO, entry.getValue() == SpellingResult.NO_FREQUENCY_INFO);
140 
141     //test something not in the spell checker
142     spellOpts.tokens = queryConverter.convert("super");
143     result = checker.getSuggestions(spellOpts);
144     assertTrue("result is null and it shouldn't be", result != null);
145     suggestions = result.get(spellOpts.tokens.iterator().next());
146     assertTrue("suggestions size should be 0", suggestions.size()==0);
147 
148     //test something that is spelled correctly
149     spellOpts.tokens = queryConverter.convert("document");
150     result = checker.getSuggestions(spellOpts);
151     assertTrue("result is null and it shouldn't be", result != null);
152     suggestions = result.get(spellOpts.tokens.iterator().next());
153     assertTrue("suggestions is null and it shouldn't be", suggestions == null);
154 
155     //Has multiple possibilities, but the exact exists, so that should be returned
156     spellOpts.tokens = queryConverter.convert("red");
157     spellOpts.count = 2;
158     result = checker.getSuggestions(spellOpts);
159     assertNotNull(result);
160     suggestions = result.get(spellOpts.tokens.iterator().next());
161     assertTrue("suggestions is not null and it should be", suggestions == null);
162 
163     //Try out something which should have multiple suggestions
164     spellOpts.tokens = queryConverter.convert("bug");
165     result = checker.getSuggestions(spellOpts);
166     assertNotNull(result);
167     suggestions = result.get(spellOpts.tokens.iterator().next());
168     assertNotNull(suggestions);
169     assertTrue("suggestions Size: " + suggestions.size() + " is not: " + 2, suggestions.size() == 2);
170 
171     entry = suggestions.entrySet().iterator().next();
172     assertTrue(entry.getKey() + " is equal to " + "bug and it shouldn't be", entry.getKey().equals("bug") == false);
173     assertTrue(entry.getValue() + " does not equal: " + SpellingResult.NO_FREQUENCY_INFO, entry.getValue() == SpellingResult.NO_FREQUENCY_INFO);
174 
175     entry = suggestions.entrySet().iterator().next();
176     assertTrue(entry.getKey() + " is equal to " + "bug and it shouldn't be", entry.getKey().equals("bug") == false);
177     assertTrue(entry.getValue() + " does not equal: " + SpellingResult.NO_FREQUENCY_INFO, entry.getValue() == SpellingResult.NO_FREQUENCY_INFO);
178     } finally {
179       holder.decref();
180     }
181   }

269   @Test
270   public void testAlternateLocation() throws Exception {
271     String[] ALT_DOCS = new String[]{
272             "jumpin jack flash",
273             "Sargent Peppers Lonely Hearts Club Band",
274             "Born to Run",
275             "Thunder Road",
276             "Londons Burning",
277             "A Horse with No Name",
278             "Sweet Caroline"
279     };
280 
281     IndexBasedSpellChecker checker = new IndexBasedSpellChecker();
282     NamedList spellchecker = new NamedList();
283     spellchecker.add("classname", IndexBasedSpellChecker.class.getName());
284     
285     File tmpDir = createTempDir().toFile();
286     File indexDir = new File(tmpDir, "spellingIdx");
287     //create a standalone index
288     File altIndexDir = new File(tmpDir, "alternateIdx" + new Date().getTime());
289     Directory dir = newFSDirectory(altIndexDir.toPath());
290     IndexWriter iw = new IndexWriter(
291         dir,
292         new IndexWriterConfig(new WhitespaceAnalyzer())
293     );
294     for (int i = 0; i < ALT_DOCS.length; i++) {
295       Document doc = new Document();
296       doc.add(new TextField("title", ALT_DOCS[i], Field.Store.YES));
297       iw.addDocument(doc);
298     }
299     iw.forceMerge(1);
300     iw.close();
301     dir.close();
302     indexDir.mkdirs();
303     spellchecker.add(AbstractLuceneSpellChecker.INDEX_DIR, indexDir.getAbsolutePath());
304     spellchecker.add(AbstractLuceneSpellChecker.LOCATION, altIndexDir.getAbsolutePath());
305     spellchecker.add(AbstractLuceneSpellChecker.FIELD, "title");
306     spellchecker.add(AbstractLuceneSpellChecker.SPELLCHECKER_ARG_NAME, spellchecker);
307     SolrCore core = h.getCore();
308     String dictName = checker.init(spellchecker, core);
309     assertTrue(dictName + " is not equal to " + SolrSpellChecker.DEFAULT_DICTIONARY_NAME,
310             dictName.equals(SolrSpellChecker.DEFAULT_DICTIONARY_NAME) == true);
311     RefCounted<SolrIndexSearcher> holder = core.getSearcher();
312     SolrIndexSearcher searcher = holder.get();

314     checker.build(core, searcher);
315 
316     IndexReader reader = searcher.getIndexReader();

319     SpellingResult result = checker.getSuggestions(spellOpts);
320     assertTrue("result is null and it shouldn't be", result != null);
321     //should be lowercased, b/c we are using a lowercasing analyzer
322     Map<String, Integer> suggestions = result.get(spellOpts.tokens.iterator().next());
323     assertTrue("flesh is null and it shouldn't be", suggestions != null);
324     assertTrue("flesh Size: " + suggestions.size() + " is not: " + 1, suggestions.size() == 1);
325     Map.Entry<String, Integer> entry = suggestions.entrySet().iterator().next();
326     assertTrue(entry.getKey() + " is not equal to " + "flash", entry.getKey().equals("flash") == true);
327     assertTrue(entry.getValue() + " does not equal: " + 1, entry.getValue() == 1);
328 
329     //test something not in the spell checker

338     assertTrue("result is null and it shouldn't be", result != null);
339     suggestions = result.get(spellOpts.tokens.iterator().next());
