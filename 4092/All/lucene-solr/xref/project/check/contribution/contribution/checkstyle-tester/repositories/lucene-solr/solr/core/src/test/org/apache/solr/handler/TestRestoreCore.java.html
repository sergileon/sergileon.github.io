62    private static JettySolrRunner createJetty(TestReplicationHandler.SolrInstance instance) throws Exception {
63      FileUtils.copyFile(new File(SolrTestCaseJ4.TEST_HOME(), "solr.xml"), new File(instance.getHomeDir(), "solr.xml"));

65      jetty.setDataDir(instance.getDataDir());
66      jetty.start();
67      return jetty;

73        HttpSolrClient client = new HttpSolrClient(buildUrl(port, context) + "/" + DEFAULT_TEST_CORENAME);
74        client.setConnectionTimeout(15000);
75        client.setSoTimeout(60000);
76        client.setDefaultMaxConnectionsPerHost(100);
77        client.setMaxTotalConnections(100);

81        throw new RuntimeException(ex);

110   @Test
111   public void testSimpleRestore() throws Exception {
112 
113     int nDocs = TestReplicationHandlerBackup.indexDocs(masterClient);
114 
115     String snapshotName;
116     String location;
117     String params = "";
118 
119     //Use the default backup location or an externally provided location.
120     if (random().nextBoolean()) {
121       location = createTempDir().toFile().getAbsolutePath();
122       params += "&location=" + URLEncoder.encode(location, "UTF-8");
123     }
124 
125     //named snapshot vs default snapshot name
126     if (random().nextBoolean()) {
127       snapshotName = TestUtil.randomSimpleString(random(), 1, 5);
128       params += "&name=" + snapshotName;
129     }
130 
131     TestReplicationHandlerBackup.runBackupCommand(masterJetty, ReplicationHandler.CMD_BACKUP, params);
132 
133     CheckBackupStatus checkBackupStatus = new CheckBackupStatus((HttpSolrClient) masterClient, null);
134     while (!checkBackupStatus.success) {
135       checkBackupStatus.fetchStatus();
136       Thread.sleep(1000);
137     }
138 
139     //Modify existing index before we call restore.
140 
141     //Delete a few docs
142     int numDeletes = TestUtil.nextInt(random(), 1, nDocs);
143     for(int i=0; i<numDeletes; i++) {
144       masterClient.deleteByQuery("id:" + i);
145     }
146     masterClient.commit();
147 
148     //Add a few more
149     int moreAdds = TestUtil.nextInt(random(), 1, 100);
150     for (int i=0; i<moreAdds; i++) {
151       SolrInputDocument doc = new SolrInputDocument();
152       doc.addField("id", i + nDocs);
153       doc.addField("name", "name = " + (i + nDocs));
154       masterClient.add(doc);
155     }
156     //Purposely not calling commit once in a while. There can be some docs which are not committed
157     if (usually()) {
158       masterClient.commit();
159     }
160 
161     TestReplicationHandlerBackup.runBackupCommand(masterJetty, ReplicationHandler.CMD_RESTORE, params);
162 
163     while (!fetchRestoreStatus()) {
164       Thread.sleep(1000);
165     }
166 
167     //See if restore was successful by checking if all the docs are present again
168     verifyDocs(nDocs);
169   }

171   @Test
172   public void testFailedRestore() throws Exception {
173     int nDocs = TestReplicationHandlerBackup.indexDocs(masterClient);
174 
175     String location = createTempDir().toFile().getAbsolutePath();
176     String snapshotName = TestUtil.randomSimpleString(random(), 1, 5);
177     String params = "&name=" + snapshotName + "&location=" + URLEncoder.encode(location, "UTF-8");
178 
179     TestReplicationHandlerBackup.runBackupCommand(masterJetty, ReplicationHandler.CMD_BACKUP, params);
180 
181     CheckBackupStatus checkBackupStatus = new CheckBackupStatus((HttpSolrClient) masterClient, null);
182     while (!checkBackupStatus.success) {
183       checkBackupStatus.fetchStatus();
184       Thread.sleep(1000);
185     }
186 
187     //Remove the segments_n file so that the backup index is corrupted.
188     //Restore should fail and it should automatically rollback to the original index.
189     Path restoreIndexPath = Paths.get(location).resolve("snapshot." + snapshotName);
190     try (DirectoryStream<Path> stream = Files.newDirectoryStream(restoreIndexPath, IndexFileNames.SEGMENTS + "*")) {
191       Path segmentFileName = stream.iterator().next();
192       Files.delete(segmentFileName);
193     }
194 
195     TestReplicationHandlerBackup.runBackupCommand(masterJetty, ReplicationHandler.CMD_RESTORE, params);
196 
197     try {
198       while (!fetchRestoreStatus()) {
199         Thread.sleep(1000);
200       }
201       fail("Should have thrown an error because restore could not have been successful");
202     } catch (AssertionError e) {
203       //supposed to happen
204     }
205 
206     verifyDocs(nDocs);
207 
208     //make sure we can write to the index again
209     nDocs = TestReplicationHandlerBackup.indexDocs(masterClient);
210     verifyDocs(nDocs);
211 
212   }

214   private void verifyDocs(int nDocs) throws SolrServerException, IOException {
215     ModifiableSolrParams queryParams = new ModifiableSolrParams();
216     queryParams.set("q", "*:*");
217     QueryResponse response = masterClient.query(queryParams);
218 
219     assertEquals(0, response.getStatus());
220     assertEquals(nDocs, response.getResults().getNumFound());
221   }

223   private boolean fetchRestoreStatus() throws IOException {
224     String masterUrl = buildUrl(masterJetty.getLocalPort(), context) + "/" + DEFAULT_TEST_CORENAME +
225         "/replication?command=" + ReplicationHandler.CMD_RESTORE_STATUS;
226     final Pattern pException = Pattern.compile("<str name=\"exception\">(.*?)</str>");
227 
228     InputStream stream = null;
229     try {
230       URL url = new URL(masterUrl);
231       stream = url.openStream();
232       String response = IOUtils.toString(stream, "UTF-8");
233       Matcher matcher = pException.matcher(response);
234       if(matcher.find()) {
235         fail("Failed to complete restore action with exception " + matcher.group(1));
236       }
237       if(response.contains("<str name=\"status\">success</str>")) {
238         return true;
239       } else if (response.contains("<str name=\"status\">failed</str>")){
240         fail("Restore Failed");
241       }
242       stream.close();
243     } finally {
244       IOUtils.closeQuietly(stream);
245     }
246     return false;
247   }
