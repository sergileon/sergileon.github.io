19  import org.apache.solr.util.RestTestBase;
20  import org.eclipse.jetty.servlet.ServletHolder;
21  import org.junit.BeforeClass;
22  import org.junit.Test;
23  import org.restlet.ext.servlet.ServerServlet;
24  
25  import java.util.SortedMap;
26  import java.util.TreeMap;
27  
28  
29  public class TestClassNameShortening extends RestTestBase {
30  
31    @BeforeClass
32    public static void init() throws Exception {
33      final SortedMap<ServletHolder,String> extraServlets = new TreeMap<>();
34      final ServletHolder solrRestApi = new ServletHolder("SolrSchemaRestApi", ServerServlet.class);
35      solrRestApi.setInitParameter("org.restlet.application", "org.apache.solr.rest.SolrSchemaRestApi");
36      extraServlets.put(solrRestApi, "/schema/*");  // '/schema/*' matches '/schema', '/schema/', and '/schema/whatever...'
37  
38      createJettyAndHarness(TEST_HOME(), "solrconfig-minimal.xml", "schema-class-name-shortening-on-serialization.xml", 
39                            "/solr", true, extraServlets);
40    }
41  
42    @Test
43    public void testClassNamesNotShortened() throws Exception {
44      assertQ("/schema/fieldtypes/fullClassNames?indent=on&wt=xml&showDefaults=true",
45              "count(/response/lst[@name='fieldType']) = 1",
46              "/response/lst[@name='fieldType']/str[@name='class'] = 'org.apache.solr.schema.TextField'",
47              "//arr[@name='charFilters']/lst/str[@name='class'] = 'org.apache.solr.analysis.MockCharFilterFactory'",
48              "//lst[@name='tokenizer']/str[@name='class'] = 'org.apache.solr.analysis.MockTokenizerFactory'",
49              "//arr[@name='filters']/lst/str[@name='class'] = 'org.apache.solr.analysis.MockTokenFilterFactory'",
50              "/response/lst[@name='fieldType']/lst[@name='similarity']/str[@name='class'] = 'org.apache.lucene.misc.SweetSpotSimilarity'");
51    }
52  
53    @Test
54    public void testShortenedGlobalSimilarityStaysShortened() throws Exception {
55      assertQ("/schema/similarity?indent=on&wt=xml",
56              "count(/response/lst[@name='similarity']) = 1",
57              "/response/lst[@name='similarity']/str[@name='class'][.='solr.SchemaSimilarityFactory']");
58    }
59  
60    @Test
61    public void testShortenedClassNamesStayShortened() throws Exception {
62      assertQ("/schema/fieldtypes/shortenedClassNames?indent=on&wt=xml&showDefaults=true",
63              "count(/response/lst[@name='fieldType']) = 1",
64              "/response/lst[@name='fieldType']/str[@name='class'] = 'solr.TextField'",
65              "//arr[@name='charFilters']/lst/str[@name='class'] = 'solr.MockCharFilterFactory'",
66              "//lst[@name='tokenizer']/str[@name='class'] = 'solr.MockTokenizerFactory'",
67              "//arr[@name='filters']/lst/str[@name='class'] = 'solr.MockTokenFilterFactory'",
68              "/response/lst[@name='fieldType']/lst[@name='similarity']/str[@name='class'] = 'solr.SweetSpotSimilarityFactory'");
69    }
70  }
