160   private static Set<String> difFiles(String[] files1, String[] files2) {
161     Set<String> set1 = new HashSet<>();
162     Set<String> set2 = new HashSet<>();
163     Set<String> extra = new HashSet<>();
164     
165     for (int x=0; x < files1.length; x++) {
166       set1.add(files1[x]);
167     }
168     for (int x=0; x < files2.length; x++) {
169       set2.add(files2[x]);
170     }
171     Iterator<String> i1 = set1.iterator();
172     while (i1.hasNext()) {
173       String o = i1.next();
174       if (!set2.contains(o)) {
175         extra.add(o);
176       }
177     }
178     Iterator<String> i2 = set2.iterator();
179     while (i2.hasNext()) {
180       String o = i2.next();
181       if (!set1.contains(o)) {
182         extra.add(o);
183       }
184     }
185     return extra;
186   }

188   private String asString(String[] l) {
189     String s = "";
190     for(int i=0;i<l.length;i++) {
191       if (i > 0) {
192         s += "\n    ";
193       }
194       s += l[i];
195     }
196     return s;
197   }

199   public void copyFile(Directory dir, String src, String dest) throws IOException {
200     IndexInput in = dir.openInput(src, newIOContext(random()));
201     IndexOutput out = dir.createOutput(dest, newIOContext(random()));
202     byte[] b = new byte[1024];
203     long remainder = in.length();
204     while(remainder > 0) {
205       int len = (int) Math.min(b.length, remainder);
206       in.readBytes(b, 0, len);
207       out.writeBytes(b, len);
208       remainder -= len;
209     }
210     in.close();
211     out.close();
212   }

222   public void testVirusScannerDoesntCorruptIndex() throws IOException {
223     MockDirectoryWrapper dir = newMockDirectory();
224     dir.setPreventDoubleWrite(false); // we arent trying to test this
225     dir.setEnableVirusScanner(false); // we have our own to make test reproduce always
226     
227     // add empty commit
228     new IndexWriter(dir, new IndexWriterConfig(null)).close();
229     // add a trash unreferenced file
230     dir.createOutput("_0.si", IOContext.DEFAULT).close();
231 
232     // start virus scanner
233     final AtomicBoolean stopScanning = new AtomicBoolean();
234     dir.failOn(new MockDirectoryWrapper.Failure() {
235       @Override
236       public void eval(MockDirectoryWrapper dir) throws IOException {
237         if (stopScanning.get()) {
238           return;
239         }
240         for (StackTraceElement f : new Exception().getStackTrace()) {
241           if ("deleteFile".equals(f.getMethodName())) {
242             throw new IOException("temporarily cannot delete file");
243           }
244         }
245       }
246     });
247     
248     IndexWriter iw = new IndexWriter(dir, new IndexWriterConfig(null));
249     iw.addDocument(new Document());
250     // stop virus scanner
251     stopScanning.set(true);
252     iw.commit();
253     iw.close();
254     dir.close();
255   }

405   // IFD's inflater is "raw" and expects to only see codec files, 
406   // and rightfully so, it filters them out.
407   static void inflateGens(SegmentInfos sis, Collection<String> files, InfoStream stream) {
408     List<String> filtered = new ArrayList<>();
409     for (String file : files) {
410       if (IndexFileNames.CODEC_FILE_PATTERN.matcher(file).matches() ||
411           file.startsWith(IndexFileNames.SEGMENTS) ||
412           file.startsWith(IndexFileNames.PENDING_SEGMENTS)) {
413         filtered.add(file);
414       }
415     }
416     IndexFileDeleter.inflateGens(sis, filtered, stream);
417   }

419   // LUCENE-5919
420   public void testExcInDecRef() throws Throwable {
421     MockDirectoryWrapper dir = newMockDirectory();
422 
423     // disable slow things: we don't rely upon sleeps here.
424     dir.setThrottling(MockDirectoryWrapper.Throttling.NEVER);
425     dir.setUseSlowOpenClosers(false);
426 
427     final AtomicBoolean doFailExc = new AtomicBoolean();
428 
429     dir.failOn(new MockDirectoryWrapper.Failure() {
430         @Override
431         public void eval(MockDirectoryWrapper dir) throws IOException {
432           if (doFailExc.get() && random().nextInt(4) == 1) {
433             Exception e = new Exception();
434             StackTraceElement stack[] = e.getStackTrace();
435             for (int i = 0; i < stack.length; i++) {
436               if (stack[i].getClassName().equals(IndexFileDeleter.class.getName()) && stack[i].getMethodName().equals("decRef")) {
437                 throw new RuntimeException("fake fail");
438               }
439             }
440           }
441         }
442       });
443 
444     IndexWriterConfig iwc = newIndexWriterConfig(new MockAnalyzer(random()));
445     //iwc.setMergeScheduler(new SerialMergeScheduler());
446     MergeScheduler ms = iwc.getMergeScheduler();
447     if (ms instanceof ConcurrentMergeScheduler) {
448       final ConcurrentMergeScheduler suppressFakeFail = new ConcurrentMergeScheduler() {
449           @Override
450           protected void handleMergeException(Directory dir, Throwable exc) {
451             // suppress only FakeIOException:
452             if (exc instanceof RuntimeException && exc.getMessage().equals("fake fail")) {
453               // ok to ignore
454             } else if ((exc instanceof AlreadyClosedException || exc instanceof IllegalStateException) 
455                         && exc.getCause() != null && "fake fail".equals(exc.getCause().getMessage())) {
456               // also ok to ignore
457             } else {
458               super.handleMergeException(dir, exc);
459             }
460           }
461         };
462       final ConcurrentMergeScheduler cms = (ConcurrentMergeScheduler) ms;
463       suppressFakeFail.setMaxMergesAndThreads(cms.getMaxMergeCount(), cms.getMaxThreadCount());
464       iwc.setMergeScheduler(suppressFakeFail);
465     }
466 
467     RandomIndexWriter w = new RandomIndexWriter(random(), dir, iwc);
468 
469     // Since we hit exc during merging, a partial
470     // forceMerge can easily return when there are still
471     // too many segments in the index:
472     w.setDoRandomForceMergeAssert(false);
473 
474     doFailExc.set(true);
475     int ITERS = atLeast(1000);
476     for(int iter=0;iter<ITERS;iter++) {
477       try {
478         if (random().nextInt(10) == 5) {
479           w.commit();
480         } else if (random().nextInt(10) == 7) {
481           w.getReader().close();
482         } else {
483           Document doc = new Document();
484           doc.add(newTextField("field", "some text", Field.Store.NO));
485           w.addDocument(doc);
486         }
487       } catch (Throwable t) {
488         if (t.toString().contains("fake fail") || (t.getCause() != null && t.getCause().toString().contains("fake fail"))) {
489           // ok
490         } else {
491           throw t;
492         }
493       }
494     }
495 
496     doFailExc.set(false);
497     w.close();
498     dir.close();
499   }
