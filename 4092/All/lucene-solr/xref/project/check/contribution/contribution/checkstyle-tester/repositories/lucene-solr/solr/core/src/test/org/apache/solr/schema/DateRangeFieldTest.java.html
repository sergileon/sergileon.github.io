20  import org.apache.solr.SolrTestCaseJ4;
21  import org.junit.BeforeClass;
22  import org.junit.Test;

24  public class DateRangeFieldTest extends SolrTestCaseJ4 {
25  
26    @BeforeClass
27    public static void beforeClass() throws Exception {
28      initCore("solrconfig.xml", "schema.xml");
29    }
30  
31    public void test() {
32      assertU(delQ("*:*"));
33      assertU(adoc("id", "0", "dateRange", "[* TO *]"));
34      assertU(adoc("id", "1", "dateRange", "2014-05-21T12:00:00.000Z"));
35      assertU(adoc("id", "2", "dateRange", "[2000 TO 2014-05-21]"));
36      assertU(adoc("id", "3", "dateRange", "2020-05-21T12:00:00.000Z/DAY"));//DateMath syntax
37      assertU(commit());
38  
39      //ensure stored value resolves datemath
40      assertQ(req("q", "id:1", "fl", "dateRange"), "//result/doc/arr[@name='dateRange']/str[.='2014-05-21T12:00:00Z']");//no 000 ms
41      assertQ(req("q", "id:2", "fl", "dateRange"), "//result/doc/arr[@name='dateRange']/str[.='[2000 TO 2014-05-21]']");//a range; same
42      assertQ(req("q", "id:3", "fl", "dateRange"), "//result/doc/arr[@name='dateRange']/str[.='2020-05-21T00:00:00Z']");//resolve datemath
43  
44      String[] commonParams = {"q", "{!field f=dateRange op=$op v=$qq}", "sort", "id asc"};
45      assertQ(req(commonParams, "qq", "[* TO *]"), xpathMatches(0, 1, 2, 3));
46      assertQ(req(commonParams, "qq", "2012"), xpathMatches(0, 2));
47      assertQ(req(commonParams, "qq", "2013", "op", "Contains"), xpathMatches(0, 2));
48      assertQ(req(commonParams, "qq", "2014", "op", "Contains"), xpathMatches(0));
49      assertQ(req(commonParams, "qq", "[1999 TO 2001]", "op", "IsWithin"), xpathMatches());
50      assertQ(req(commonParams, "qq", "2014-05", "op", "IsWithin"), xpathMatches(1));
51  
52      assertQ(req("q", "dateRange:[1998 TO 2000}"), xpathMatches(0));//exclusive end, so we barely miss one doc
53  
54      //show without local-params
55      assertQ(req("q", "dateRange:\"2014-05-21T12:00:00.000Z\""), xpathMatches(0, 1, 2));
56      assertQ(req("q", "dateRange:[1999 TO 2001]"), xpathMatches(0, 2));
57    }
58  
59    @Test
60    public void testMultiValuedDateRanges() {
61      assertU(delQ("*:*"));
62      assertU(adoc("id", "0", "dateRange", "[2000 TO 2010]", "dateRange", "[2011 TO 2014]"));
63      assertU(adoc("id", "1", "dateRange", "[2000-01 TO 2010-10]", "dateRange", "[2010-11 TO 2014-12]"));
64      assertU(adoc("id", "2", "dateRange", "[2000-01-01 TO 2010-08-01]", "dateRange", "[2010-08-01 TO 2014-12-01]"));
65      assertU(adoc("id", "3", "dateRange", "[1990 TO 1995]", "dateRange", "[1997 TO 1999]"));
66      assertU(commit());
67  
68      String[] commonParams = {"q", "{!field f=dateRange op=$op v=$qq}", "sort", "id asc"};
69  
70      assertQ(req(commonParams, "qq", "[2000 TO 2014]", "op", "IsWithin"), xpathMatches(0, 1, 2));
71      assertQ(req(commonParams, "qq", "[2000 TO 2013]", "op", "IsWithin"), xpathMatches());
72      assertQ(req(commonParams, "qq", "[2000 TO 2014]", "op", "Contains"), xpathMatches(0, 1));
73      assertQ(req(commonParams, "qq", "[2000 TO 2015]", "op", "Contains"), xpathMatches());
74  
75      assertQ(req(commonParams, "qq", "[2000-01 TO 2014-12]", "op", "IsWithin"), xpathMatches(0, 1, 2));
76      assertQ(req(commonParams, "qq", "[2000 TO 2014-11]", "op", "IsWithin"), xpathMatches());
77      assertQ(req(commonParams, "qq", "[2000-01 TO 2014-12]", "op", "Contains"), xpathMatches(0, 1));
78  
79      assertQ(req(commonParams, "qq", "[2000-01-01 TO 2014-12-31]", "op", "IsWithin"), xpathMatches(0, 1, 2));
80      assertQ(req(commonParams, "qq", "[2000-01-01 TO 2014-12-01]", "op", "Contains"), xpathMatches(0, 1, 2));
81      assertQ(req(commonParams, "qq", "[2000 TO 2000]", "op", "Contains"), xpathMatches(0, 1, 2));
82  
83      assertQ(req(commonParams, "qq", "[2000 TO 2000]", "op", "Contains"), xpathMatches(0, 1, 2));
84  
85      assertQ(req(commonParams, "qq", "[1996-01-01 TO 1996-12-31]", "op", "Contains"), xpathMatches());
86    }
87  
88    private String[] xpathMatches(int... docIds) {
89      String[] tests = new String[docIds != null ? docIds.length + 1 : 1];
90      tests[0] = "*[count(//doc)=" + (tests.length-1) + "]";
91      if (docIds != null && docIds.length > 0) {
92        int i = 1;
93        for (int docId : docIds) {
94          tests[i++] = "//result/doc/int[@name='id'][.='" + docId + "']";
95        }
96      }
97      return tests;
98    }
