32  /**
33   * {@link DocumentsWriterDeleteQueue} is a non-blocking linked pending deletes
34   * queue. In contrast to other queue implementation we only maintain the
35   * tail of the queue. A delete queue is always used in a context of a set of
36   * DWPTs and a global delete pool. Each of the DWPT and the global pool need to
37   * maintain their 'own' head of the queue (as a DeleteSlice instance per DWPT).
38   * The difference between the DWPT and the global pool is that the DWPT starts
39   * maintaining a head once it has added its first document since for its segments
40   * private deletes only the deletes after that document are relevant. The global
41   * pool instead starts maintaining the head once this instance is created by
42   * taking the sentinel instance as its initial head.
43   * <p>
44   * Since each {@link DeleteSlice} maintains its own head and the list is only
45   * single linked the garbage collector takes care of pruning the list for us.
46   * All nodes in the list that are still relevant should be either directly or
47   * indirectly referenced by one of the DWPT's private {@link DeleteSlice} or by
48   * the global {@link BufferedUpdates} slice.
49   * <p>
50   * Each DWPT as well as the global delete pool maintain their private
51   * DeleteSlice instance. In the DWPT case updating a slice is equivalent to
52   * atomically finishing the document. The slice update guarantees a "happens
53   * before" relationship to all other updates in the same indexing session. When a
54   * DWPT updates a document it:
55   * 
56   * <ol>
57   * <li>consumes a document and finishes its processing</li>
58   * <li>updates its private {@link DeleteSlice} either by calling
59   * {@link #updateSlice(DeleteSlice)} or {@link #add(Term, DeleteSlice)} (if the
60   * document has a delTerm)</li>
61   * <li>applies all deletes in the slice to its private {@link BufferedUpdates}
62   * and resets it</li>
63   * <li>increments its internal document id</li>
64   * </ol>
65   * 
66   * The DWPT also doesn't apply its current documents delete term until it has
67   * updated its delete slice which ensures the consistency of the update. If the
68   * update fails before the DeleteSlice could have been updated the deleteTerm
69   * will also not be added to its private deletes neither to the global deletes.
70   * 
71   */
72  final class DocumentsWriterDeleteQueue implements Accountable {
73  
74    private volatile Node<?> tail;
75    
76    @SuppressWarnings("rawtypes")
77    private static final AtomicReferenceFieldUpdater<DocumentsWriterDeleteQueue,Node> tailUpdater = AtomicReferenceFieldUpdater
78        .newUpdater(DocumentsWriterDeleteQueue.class, Node.class, "tail");
79  
80    private final DeleteSlice globalSlice;
81    private final BufferedUpdates globalBufferedUpdates;
82    
83    // only acquired to update the global deletes, pkg-private for access by tests:
84    final ReentrantLock globalBufferLock = new ReentrantLock();
85  
86    final long generation;
87    
88    DocumentsWriterDeleteQueue() {
89      this(0);
90    }
91    
92    DocumentsWriterDeleteQueue(long generation) {
93      this(new BufferedUpdates(), generation);
94    }
95  
96    DocumentsWriterDeleteQueue(BufferedUpdates globalBufferedUpdates, long generation) {
97      this.globalBufferedUpdates = globalBufferedUpdates;
98      this.generation = generation;
99      /*
100      * we use a sentinel instance as our initial tail. No slice will ever try to
101      * apply this tail since the head is always omitted.
102      */
103     tail = new Node<>(null); // sentinel
104     globalSlice = new DeleteSlice(tail);
105   }
106 
107   void addDelete(Query... queries) {
108     add(new QueryArrayNode(queries));
109     tryApplyGlobalSlice();
110   }
111 
112   void addDelete(Term... terms) {
113     add(new TermArrayNode(terms));
114     tryApplyGlobalSlice();
115   }
116 
117   void addDocValuesUpdates(DocValuesUpdate... updates) {
118     add(new DocValuesUpdatesNode(updates));
119     tryApplyGlobalSlice();
120   }

122   /**
123    * invariant for document update
124    */
125   void add(Term term, DeleteSlice slice) {
126     final TermNode termNode = new TermNode(term);
127 //    System.out.println(Thread.currentThread().getName() + ": push " + termNode + " this=" + this);
128     add(termNode);
129     /*
130      * this is an update request where the term is the updated documents
131      * delTerm. in that case we need to guarantee that this insert is atomic
132      * with regards to the given delete slice. This means if two threads try to
133      * update the same document with in turn the same delTerm one of them must
134      * win. By taking the node we have created for our del term as the new tail
135      * it is guaranteed that if another thread adds the same right after us we
136      * will apply this delete next time we update our slice and one of the two
137      * competing updates wins!
138      */
139     slice.sliceTail = termNode;
140     assert slice.sliceHead != slice.sliceTail : "slice head and tail must differ after add";
141     tryApplyGlobalSlice(); // TODO doing this each time is not necessary maybe
142     // we can do it just every n times or so?
143   }
144 
145   void add(Node<?> item) {
146     /*
147      * this non-blocking / 'wait-free' linked list add was inspired by Apache
148      * Harmony's ConcurrentLinkedQueue Implementation.
149      */
150     while (true) {
151       final Node<?> currentTail = this.tail;
152       final Node<?> tailNext = currentTail.next;
153       if (tail == currentTail) {
154         if (tailNext != null) {
155           /*
156            * we are in intermediate state here. the tails next pointer has been
157            * advanced but the tail itself might not be updated yet. help to
158            * advance the tail and try again updating it.
159            */
160           tailUpdater.compareAndSet(this, currentTail, tailNext); // can fail
161         } else {
162           /*
163            * we are in quiescent state and can try to insert the item to the
164            * current tail if we fail to insert we just retry the operation since
165            * somebody else has already added its item
166            */
167           if (currentTail.casNext(null, item)) {
168             /*
169              * now that we are done we need to advance the tail while another
170              * thread could have advanced it already so we can ignore the return
171              * type of this CAS call
172              */
173             tailUpdater.compareAndSet(this, currentTail, item);
174             return;
175           }
176         }
177       }
178     }
179   }
180 
181   boolean anyChanges() {
182     globalBufferLock.lock();
183     try {
184       /*
185        * check if all items in the global slice were applied 
186        * and if the global slice is up-to-date
187        * and if globalBufferedUpdates has changes
188        */
189       return globalBufferedUpdates.any() || !globalSlice.isEmpty() || globalSlice.sliceTail != tail
190           || tail.next != null;
191     } finally {
192       globalBufferLock.unlock();
193     }
194   }
195 
196   void tryApplyGlobalSlice() {
197     if (globalBufferLock.tryLock()) {
198       /*
199        * The global buffer must be locked but we don't need to update them if
200        * there is an update going on right now. It is sufficient to apply the
201        * deletes that have been added after the current in-flight global slices
202        * tail the next time we can get the lock!
203        */
204       try {
205         if (updateSlice(globalSlice)) {
206 //          System.out.println(Thread.currentThread() + ": apply globalSlice");
207           globalSlice.apply(globalBufferedUpdates, BufferedUpdates.MAX_INT);
208         }
209       } finally {
210         globalBufferLock.unlock();
211       }
212     }
213   }
214 
215   FrozenBufferedUpdates freezeGlobalBuffer(DeleteSlice callerSlice) {
216     globalBufferLock.lock();
217     /*
218      * Here we freeze the global buffer so we need to lock it, apply all
219      * deletes in the queue and reset the global slice to let the GC prune the
220      * queue.
221      */
222     final Node<?> currentTail = tail; // take the current tail make this local any
223     // Changes after this call are applied later
224     // and not relevant here
225     if (callerSlice != null) {
226       // Update the callers slices so we are on the same page
227       callerSlice.sliceTail = currentTail;
228     }
229     try {
230       if (globalSlice.sliceTail != currentTail) {
231         globalSlice.sliceTail = currentTail;
232         globalSlice.apply(globalBufferedUpdates, BufferedUpdates.MAX_INT);
233       }
234 
235 //      System.out.println(Thread.currentThread().getName() + ": now freeze global buffer " + globalBufferedDeletes);
236       final FrozenBufferedUpdates packet = new FrozenBufferedUpdates(
237           globalBufferedUpdates, false);
238       globalBufferedUpdates.clear();
239       return packet;
240     } finally {
241       globalBufferLock.unlock();
242     }
243   }
244 
245   DeleteSlice newSlice() {
246     return new DeleteSlice(tail);
247   }
248 
249   boolean updateSlice(DeleteSlice slice) {
250     if (slice.sliceTail != tail) { // If we are the same just
251       slice.sliceTail = tail;
252       return true;
253     }
254     return false;
255   }

257   static class DeleteSlice {
258     // No need to be volatile, slices are thread captive (only accessed by one thread)!
259     Node<?> sliceHead; // we don't apply this one
260     Node<?> sliceTail;
261 
262     DeleteSlice(Node<?> currentTail) {
263       assert currentTail != null;
264       /*
265        * Initially this is a 0 length slice pointing to the 'current' tail of
266        * the queue. Once we update the slice we only need to assign the tail and
267        * have a new slice
268        */
269       sliceHead = sliceTail = currentTail;
270     }
271 
272     void apply(BufferedUpdates del, int docIDUpto) {
273       if (sliceHead == sliceTail) {
274         // 0 length slice
275         return;
276       }
277       /*
278        * When we apply a slice we take the head and get its next as our first
279        * item to apply and continue until we applied the tail. If the head and
280        * tail in this slice are not equal then there will be at least one more
281        * non-null node in the slice!
282        */
283       Node<?> current = sliceHead;
284       do {
285         current = current.next;
286         assert current != null : "slice property violated between the head on the tail must not be a null node";
287         current.apply(del, docIDUpto);
288 //        System.out.println(Thread.currentThread().getName() + ": pull " + current + " docIDUpto=" + docIDUpto);
289       } while (current != sliceTail);
290       reset();
291     }
292 
293     void reset() {
294       // Reset to a 0 length slice
295       sliceHead = sliceTail;
296     }
297 
298     /**
299      * Returns <code>true</code> iff the given item is identical to the item
300      * hold by the slices tail, otherwise <code>false</code>.
301      */
302     boolean isTailItem(Object item) {
303       return sliceTail.item == item;
304     }
305 
306     boolean isEmpty() {
307       return sliceHead == sliceTail;
308     }
309   }
310 
311   public int numGlobalTermDeletes() {
312     return globalBufferedUpdates.numTermDeletes.get();
313   }
314 
315   void clear() {
316     globalBufferLock.lock();
317     try {
318       final Node<?> currentTail = tail;
319       globalSlice.sliceHead = globalSlice.sliceTail = currentTail;
320       globalBufferedUpdates.clear();
321     } finally {
322       globalBufferLock.unlock();
323     }
324   }

326   private static class Node<T> {
327     volatile Node<?> next;
328     final T item;
329 
330     Node(T item) {
331       this.item = item;
332     }
333 
334     @SuppressWarnings("rawtypes")
335     static final AtomicReferenceFieldUpdater<Node,Node> nextUpdater = AtomicReferenceFieldUpdater
336         .newUpdater(Node.class, Node.class, "next");
337 
338     void apply(BufferedUpdates bufferedDeletes, int docIDUpto) {
339       throw new IllegalStateException("sentinel item must never be applied");
340     }
341 
342     boolean casNext(Node<?> cmp, Node<?> val) {
343       return nextUpdater.compareAndSet(this, cmp, val);
344     }
345   }

347   private static final class TermNode extends Node<Term> {
348 
349     TermNode(Term term) {
350       super(term);
351     }
352 
353     @Override
354     void apply(BufferedUpdates bufferedDeletes, int docIDUpto) {
355       bufferedDeletes.addTerm(item, docIDUpto);
356     }
357 
358     @Override
359     public String toString() {
360       return "del=" + item;
361     }
362   }

364   private static final class QueryArrayNode extends Node<Query[]> {
365     QueryArrayNode(Query[] query) {
366       super(query);
367     }
368 
369     @Override
370     void apply(BufferedUpdates bufferedUpdates, int docIDUpto) {
371       for (Query query : item) {
372         bufferedUpdates.addQuery(query, docIDUpto);  
373       }
374     }
375   }

377   private static final class TermArrayNode extends Node<Term[]> {
378     TermArrayNode(Term[] term) {
379       super(term);
380     }
381 
382     @Override
383     void apply(BufferedUpdates bufferedUpdates, int docIDUpto) {
384       for (Term term : item) {
385         bufferedUpdates.addTerm(term, docIDUpto);  
386       }
387     }

395   private static final class DocValuesUpdatesNode extends Node<DocValuesUpdate[]> {
396 
397     DocValuesUpdatesNode(DocValuesUpdate... updates) {
398       super(updates);
399     }
400 
401     @Override
402     void apply(BufferedUpdates bufferedUpdates, int docIDUpto) {
403       for (DocValuesUpdate update : item) {
404         switch (update.type) {
405           case NUMERIC:
406             bufferedUpdates.addNumericUpdate(new NumericDocValuesUpdate(update.term, update.field, (Long) update.value), docIDUpto);
407             break;
408           case BINARY:
409             bufferedUpdates.addBinaryUpdate(new BinaryDocValuesUpdate(update.term, update.field, (BytesRef) update.value), docIDUpto);
410             break;
411           default:
412             throw new IllegalArgumentException(update.type + " DocValues updates not supported yet!");
413         }
414       }
415     }
416 
417     @Override
418     public String toString() {
419       StringBuilder sb = new StringBuilder();
420       sb.append("docValuesUpdates: ");
421       if (item.length > 0) {
422         sb.append("term=").append(item[0].term).append("; updates: [");
423         for (DocValuesUpdate update : item) {
424           sb.append(update.field).append(':').append(update.value).append(',');
425         }
426         sb.setCharAt(sb.length()-1, ']');
427       }
428       return sb.toString();
429     }
430   }

432   private boolean forceApplyGlobalSlice() {
433     globalBufferLock.lock();
434     final Node<?> currentTail = tail;
435     try {
436       if (globalSlice.sliceTail != currentTail) {
437         globalSlice.sliceTail = currentTail;
438         globalSlice.apply(globalBufferedUpdates, BufferedUpdates.MAX_INT);
439       }
440       return globalBufferedUpdates.any();
441     } finally {
442       globalBufferLock.unlock();
443     }
444   }
445 
446   public int getBufferedUpdatesTermsSize() {
447     globalBufferLock.lock();
448     try {
449       forceApplyGlobalSlice();
450       return globalBufferedUpdates.terms.size();
451     } finally {
452       globalBufferLock.unlock();
453     }
454   }
