38  /**
39   * Utility class for reading and writing versioned headers.
40   * <p>
41   * Writing codec headers is useful to ensure that a file is in 
42   * the format you think it is.
43   * 
44   * @lucene.experimental
45   */
46  
47  public final class CodecUtil {
48    private CodecUtil() {} // no instance
49  
50    /**
51     * Constant to identify the start of a codec header.
52     */
53    public final static int CODEC_MAGIC = 0x3fd76c17;
54    /**
55     * Constant to identify the start of a codec footer.
56     */
57    public final static int FOOTER_MAGIC = ~CODEC_MAGIC;
58  
59    /**
60     * Writes a codec header, which records both a string to
61     * identify the file and a version number. This header can
62     * be parsed and validated with 
63     * {@link #checkHeader(DataInput, String, int, int) checkHeader()}.
64     * <p>
65     * CodecHeader --&gt; Magic,CodecName,Version
66     * <ul>
67     *    <li>Magic --&gt; {@link DataOutput#writeInt Uint32}. This
68     *        identifies the start of the header. It is always {@value #CODEC_MAGIC}.
69     *    <li>CodecName --&gt; {@link DataOutput#writeString String}. This
70     *        is a string to identify this file.
71     *    <li>Version --&gt; {@link DataOutput#writeInt Uint32}. Records
72     *        the version of the file.
73     * </ul>
74     * <p>
75     * Note that the length of a codec header depends only upon the
76     * name of the codec, so this length can be computed at any time
77     * with {@link #headerLength(String)}.
78     * 
79     * @param out Output stream
80     * @param codec String to identify this file. It should be simple ASCII, 
81     *              less than 128 characters in length.
82     * @param version Version number
83     * @throws IOException If there is an I/O error writing to the underlying medium.
84     * @throws IllegalArgumentException If the codec name is not simple ASCII, or is more than 127 characters in length
85     */
86    public static void writeHeader(DataOutput out, String codec, int version) throws IOException {
87      BytesRef bytes = new BytesRef(codec);
88      if (bytes.length != codec.length() || bytes.length >= 128) {
89        throw new IllegalArgumentException("codec must be simple ASCII, less than 128 characters in length [got " + codec + "]");
90      }
91      out.writeInt(CODEC_MAGIC);
92      out.writeString(codec);
93      out.writeInt(version);
94    }

96    /**
97     * Writes a codec header for an index file, which records both a string to
98     * identify the format of the file, a version number, and data to identify
99     * the file instance (ID and auxiliary suffix such as generation).
100    * <p>
101    * This header can be parsed and validated with 
102    * {@link #checkIndexHeader(DataInput, String, int, int, byte[], String) checkIndexHeader()}.
103    * <p>
104    * IndexHeader --&gt; CodecHeader,ObjectID,ObjectSuffix
105    * <ul>
106    *    <li>CodecHeader   --&gt; {@link #writeHeader}
107    *    <li>ObjectID     --&gt; {@link DataOutput#writeByte byte}<sup>16</sup>
108    *    <li>ObjectSuffix --&gt; SuffixLength,SuffixBytes
109    *    <li>SuffixLength  --&gt; {@link DataOutput#writeByte byte}
110    *    <li>SuffixBytes   --&gt; {@link DataOutput#writeByte byte}<sup>SuffixLength</sup>
111    * </ul>
112    * <p>
113    * Note that the length of an index header depends only upon the
114    * name of the codec and suffix, so this length can be computed at any time
115    * with {@link #indexHeaderLength(String,String)}.
116    * 
117    * @param out Output stream
118    * @param codec String to identify the format of this file. It should be simple ASCII, 
119    *              less than 128 characters in length.
120    * @param id Unique identifier for this particular file instance.
121    * @param suffix auxiliary suffix information for the file. It should be simple ASCII,
122    *              less than 256 characters in length.
123    * @param version Version number
124    * @throws IOException If there is an I/O error writing to the underlying medium.
125    * @throws IllegalArgumentException If the codec name is not simple ASCII, or 
126    *         is more than 127 characters in length, or if id is invalid,
127    *         or if the suffix is not simple ASCII, or more than 255 characters
128    *         in length.
129    */
130   public static void writeIndexHeader(DataOutput out, String codec, int version, byte[] id, String suffix) throws IOException {
131     if (id.length != StringHelper.ID_LENGTH) {
132       throw new IllegalArgumentException("Invalid id: " + StringHelper.idToString(id));
133     }
134     writeHeader(out, codec, version);
135     out.writeBytes(id, 0, id.length);
136     BytesRef suffixBytes = new BytesRef(suffix);
137     if (suffixBytes.length != suffix.length() || suffixBytes.length >= 256) {
138       throw new IllegalArgumentException("suffix must be simple ASCII, less than 256 characters in length [got " + suffix + "]");
139     }
140     out.writeByte((byte) suffixBytes.length);
141     out.writeBytes(suffixBytes.bytes, suffixBytes.offset, suffixBytes.length);
142   }

144   /**
145    * Computes the length of a codec header.
146    * 
147    * @param codec Codec name.
148    * @return length of the entire codec header.
149    * @see #writeHeader(DataOutput, String, int)
150    */
151   public static int headerLength(String codec) {
152     return 9+codec.length();
153   }
154   
155   /**
156    * Computes the length of an index header.
157    * 
158    * @param codec Codec name.
159    * @return length of the entire index header.
160    * @see #writeIndexHeader(DataOutput, String, int, byte[], String)
161    */
162   public static int indexHeaderLength(String codec, String suffix) {
163     return headerLength(codec) + StringHelper.ID_LENGTH + 1 + suffix.length();
164   }

166   /**
167    * Reads and validates a header previously written with 
168    * {@link #writeHeader(DataOutput, String, int)}.
169    * <p>
170    * When reading a file, supply the expected <code>codec</code> and
171    * an expected version range (<code>minVersion to maxVersion</code>).
172    * 
173    * @param in Input stream, positioned at the point where the
174    *        header was previously written. Typically this is located
175    *        at the beginning of the file.
176    * @param codec The expected codec name.
177    * @param minVersion The minimum supported expected version number.
178    * @param maxVersion The maximum supported expected version number.
179    * @return The actual version found, when a valid header is found 
180    *         that matches <code>codec</code>, with an actual version 
181    *         where {@code minVersion <= actual <= maxVersion}.
182    *         Otherwise an exception is thrown.
183    * @throws CorruptIndexException If the first four bytes are not
184    *         {@link #CODEC_MAGIC}, or if the actual codec found is
185    *         not <code>codec</code>.
186    * @throws IndexFormatTooOldException If the actual version is less 
187    *         than <code>minVersion</code>.
188    * @throws IndexFormatTooNewException If the actual version is greater 
189    *         than <code>maxVersion</code>.
190    * @throws IOException If there is an I/O error reading from the underlying medium.
191    * @see #writeHeader(DataOutput, String, int)
192    */
193   public static int checkHeader(DataInput in, String codec, int minVersion, int maxVersion) throws IOException {
194     // Safety to guard against reading a bogus string:
195     final int actualHeader = in.readInt();
196     if (actualHeader != CODEC_MAGIC) {
197       throw new CorruptIndexException("codec header mismatch: actual header=" + actualHeader + " vs expected header=" + CODEC_MAGIC, in);
198     }
199     return checkHeaderNoMagic(in, codec, minVersion, maxVersion);
200   }
201 
202   /** Like {@link
203    *  #checkHeader(DataInput,String,int,int)} except this
204    *  version assumes the first int has already been read
205    *  and validated from the input. */
206   public static int checkHeaderNoMagic(DataInput in, String codec, int minVersion, int maxVersion) throws IOException {
207     final String actualCodec = in.readString();
208     if (!actualCodec.equals(codec)) {
209       throw new CorruptIndexException("codec mismatch: actual codec=" + actualCodec + " vs expected codec=" + codec, in);
210     }
211 
212     final int actualVersion = in.readInt();
213     if (actualVersion < minVersion) {
214       throw new IndexFormatTooOldException(in, actualVersion, minVersion, maxVersion);
215     }
216     if (actualVersion > maxVersion) {
217       throw new IndexFormatTooNewException(in, actualVersion, minVersion, maxVersion);
218     }
219 
220     return actualVersion;
221   }

223   /**
224    * Reads and validates a header previously written with 
225    * {@link #writeIndexHeader(DataOutput, String, int, byte[], String)}.
226    * <p>
227    * When reading a file, supply the expected <code>codec</code>,
228    * expected version range (<code>minVersion to maxVersion</code>),
229    * and object ID and suffix.
230    * 
231    * @param in Input stream, positioned at the point where the
232    *        header was previously written. Typically this is located
233    *        at the beginning of the file.
234    * @param codec The expected codec name.
235    * @param minVersion The minimum supported expected version number.
236    * @param maxVersion The maximum supported expected version number.
237    * @param expectedID The expected object identifier for this file.
238    * @param expectedSuffix The expected auxiliary suffix for this file.
239    * @return The actual version found, when a valid header is found 
240    *         that matches <code>codec</code>, with an actual version 
241    *         where {@code minVersion <= actual <= maxVersion}, 
242    *         and matching <code>expectedID</code> and <code>expectedSuffix</code>
243    *         Otherwise an exception is thrown.
244    * @throws CorruptIndexException If the first four bytes are not
245    *         {@link #CODEC_MAGIC}, or if the actual codec found is
246    *         not <code>codec</code>, or if the <code>expectedID</code>
247    *         or <code>expectedSuffix</code> do not match.
248    * @throws IndexFormatTooOldException If the actual version is less 
249    *         than <code>minVersion</code>.
250    * @throws IndexFormatTooNewException If the actual version is greater 
251    *         than <code>maxVersion</code>.
252    * @throws IOException If there is an I/O error reading from the underlying medium.
253    * @see #writeIndexHeader(DataOutput, String, int, byte[],String)
254    */
255   public static int checkIndexHeader(DataInput in, String codec, int minVersion, int maxVersion, byte[] expectedID, String expectedSuffix) throws IOException {
256     int version = checkHeader(in, codec, minVersion, maxVersion);
257     checkIndexHeaderID(in, expectedID);
258     checkIndexHeaderSuffix(in, expectedSuffix);
259     return version;
260   }
261   
262   /** Expert: just reads and verifies the object ID of an index header */
263   public static byte[] checkIndexHeaderID(DataInput in, byte[] expectedID) throws IOException {
264     byte id[] = new byte[StringHelper.ID_LENGTH];
265     in.readBytes(id, 0, id.length);
266     if (!Arrays.equals(id, expectedID)) {
267       throw new CorruptIndexException("file mismatch, expected id=" + StringHelper.idToString(expectedID) 
268                                                          + ", got=" + StringHelper.idToString(id), in);
269     }
270     return id;
271   }
272   
273   /** Expert: just reads and verifies the suffix of an index header */
274   public static String checkIndexHeaderSuffix(DataInput in, String expectedSuffix) throws IOException {
275     int suffixLength = in.readByte() & 0xFF;
276     byte suffixBytes[] = new byte[suffixLength];
277     in.readBytes(suffixBytes, 0, suffixBytes.length);
278     String suffix = new String(suffixBytes, 0, suffixBytes.length, StandardCharsets.UTF_8);
279     if (!suffix.equals(expectedSuffix)) {
280       throw new CorruptIndexException("file mismatch, expected suffix=" + expectedSuffix
281                                                              + ", got=" + suffix, in);
282     }
283     return suffix;
284   }

286   /**
287    * Writes a codec footer, which records both a checksum
288    * algorithm ID and a checksum. This footer can
289    * be parsed and validated with 
290    * {@link #checkFooter(ChecksumIndexInput) checkFooter()}.
291    * <p>
292    * CodecFooter --&gt; Magic,AlgorithmID,Checksum
293    * <ul>
294    *    <li>Magic --&gt; {@link DataOutput#writeInt Uint32}. This
295    *        identifies the start of the footer. It is always {@value #FOOTER_MAGIC}.
296    *    <li>AlgorithmID --&gt; {@link DataOutput#writeInt Uint32}. This
297    *        indicates the checksum algorithm used. Currently this is always 0,
298    *        for zlib-crc32.
299    *    <li>Checksum --&gt; {@link DataOutput#writeLong Uint64}. The
300    *        actual checksum value for all previous bytes in the stream, including
301    *        the bytes from Magic and AlgorithmID.
302    * </ul>
303    * 
304    * @param out Output stream
305    * @throws IOException If there is an I/O error writing to the underlying medium.
306    */
307   public static void writeFooter(IndexOutput out) throws IOException {
308     out.writeInt(FOOTER_MAGIC);
309     out.writeInt(0);
310     writeCRC(out);
311   }

313   /**
314    * Computes the length of a codec footer.
315    * 
316    * @return length of the entire codec footer.
317    * @see #writeFooter(IndexOutput)
318    */
319   public static int footerLength() {
320     return 16;
321   }

323   /** 
324    * Validates the codec footer previously written by {@link #writeFooter}. 
325    * @return actual checksum value
326    * @throws IOException if the footer is invalid, if the checksum does not match, 
327    *                     or if {@code in} is not properly positioned before the footer
328    *                     at the end of the stream.
329    */
330   public static long checkFooter(ChecksumIndexInput in) throws IOException {
331     validateFooter(in);
332     long actualChecksum = in.getChecksum();
333     long expectedChecksum = readCRC(in);
334     if (expectedChecksum != actualChecksum) {
335       throw new CorruptIndexException("checksum failed (hardware problem?) : expected=" + Long.toHexString(expectedChecksum) +  
336                                                        " actual=" + Long.toHexString(actualChecksum), in);
337     }
338     return actualChecksum;
339   }

341   /** 
342    * Validates the codec footer previously written by {@link #writeFooter}, optionally
343    * passing an unexpected exception that has already occurred.
344    * <p>
345    * When a {@code priorException} is provided, this method will add a suppressed exception 
346    * indicating whether the checksum for the stream passes, fails, or cannot be computed, and 
347    * rethrow it. Otherwise it behaves the same as {@link #checkFooter(ChecksumIndexInput)}.
348    * <p>
349    * Example usage:
350    * <pre class="prettyprint">
351    * try (ChecksumIndexInput input = ...) {
352    *   Throwable priorE = null;
353    *   try {
354    *     // ... read a bunch of stuff ... 
355    *   } catch (Throwable exception) {
356    *     priorE = exception;
357    *   } finally {
358    *     CodecUtil.checkFooter(input, priorE);
359    *   }
360    * }
361    * </pre>
362    */
363   public static void checkFooter(ChecksumIndexInput in, Throwable priorException) throws IOException {
364     if (priorException == null) {
365       checkFooter(in);
366     } else {
367       try {
368         long remaining = in.length() - in.getFilePointer();
369         if (remaining < footerLength()) {
370           // corruption caused us to read into the checksum footer already: we can't proceed
371           priorException.addSuppressed(new CorruptIndexException("checksum status indeterminate: remaining=" + remaining +
372                                                                  ", please run checkindex for more details", in));
373         } else {
374           // otherwise, skip any unread bytes.
375           in.skipBytes(remaining - footerLength());
376           
377           // now check the footer
378           try {
379             long checksum = checkFooter(in);
380             priorException.addSuppressed(new CorruptIndexException("checksum passed (" + Long.toHexString(checksum) + 
381                                                                    "). possibly transient resource issue, or a Lucene or JVM bug", in));
382           } catch (CorruptIndexException t) {
383             priorException.addSuppressed(t);
384           }
385         }
386       } catch (Throwable t) {
387         // catch-all for things that shouldn't go wrong (e.g. OOM during readInt) but could...
388         priorException.addSuppressed(new CorruptIndexException("checksum status indeterminate: unexpected exception", in, t));
389       }
390       IOUtils.reThrow(priorException);
391     }
392   }

394   /** 
395    * Returns (but does not validate) the checksum previously written by {@link #checkFooter}.
396    * @return actual checksum value
397    * @throws IOException if the footer is invalid
398    */
399   public static long retrieveChecksum(IndexInput in) throws IOException {
400     in.seek(in.length() - footerLength());
401     validateFooter(in);
402     return readCRC(in);
403   }
404   
405   private static void validateFooter(IndexInput in) throws IOException {
406     long remaining = in.length() - in.getFilePointer();
407     long expected = footerLength();
408     if (remaining < expected) {
409       throw new CorruptIndexException("misplaced codec footer (file truncated?): remaining=" + remaining + ", expected=" + expected, in);
410     } else if (remaining > expected) {
411       throw new CorruptIndexException("misplaced codec footer (file extended?): remaining=" + remaining + ", expected=" + expected, in);
412     }
413     
414     final int magic = in.readInt();
415     if (magic != FOOTER_MAGIC) {
416       throw new CorruptIndexException("codec footer mismatch (file truncated?): actual footer=" + magic + " vs expected footer=" + FOOTER_MAGIC, in);
417     }
418     
419     final int algorithmID = in.readInt();
420     if (algorithmID != 0) {
421       throw new CorruptIndexException("codec footer mismatch: unknown algorithmID: " + algorithmID, in);
422     }
423   }

425   /**
426    * Checks that the stream is positioned at the end, and throws exception
427    * if it is not. 
428    * @deprecated Use {@link #checkFooter} instead, this should only used for files without checksums 
429    */
430   @Deprecated
431   public static void checkEOF(IndexInput in) throws IOException {
432     if (in.getFilePointer() != in.length()) {
433       throw new CorruptIndexException("did not read all bytes from file: read " + in.getFilePointer() + " vs size " + in.length(), in);
434     }
435   }

437   /** 
438    * Clones the provided input, reads all bytes from the file, and calls {@link #checkFooter} 
439    * <p>
440    * Note that this method may be slow, as it must process the entire file.
441    * If you just need to extract the checksum value, call {@link #retrieveChecksum}.
442    */
443   public static long checksumEntireFile(IndexInput input) throws IOException {
444     IndexInput clone = input.clone();
445     clone.seek(0);
446     ChecksumIndexInput in = new BufferedChecksumIndexInput(clone);
447     assert in.getFilePointer() == 0;
448     in.seek(in.length() - footerLength());
449     return checkFooter(in);
450   }

452   /**
453    * Reads CRC32 value as a 64-bit long from the input.
454    * @throws CorruptIndexException if CRC is formatted incorrectly (wrong bits set)
455    * @throws IOException if an i/o error occurs
456    */
457   public static long readCRC(IndexInput input) throws IOException {
458     long value = input.readLong();
459     if ((value & 0xFFFFFFFF00000000L) != 0) {
460       throw new CorruptIndexException("Illegal CRC-32 checksum: " + value, input);
461     }
462     return value;
463   }
464   
465   /**
466    * Writes CRC32 value as a 64-bit long to the output.
467    * @throws IllegalStateException if CRC is formatted incorrectly (wrong bits set)
468    * @throws IOException if an i/o error occurs
469    */
470   public static void writeCRC(IndexOutput output) throws IOException {
471     long value = output.getChecksum();
472     if ((value & 0xFFFFFFFF00000000L) != 0) {
473       throw new IllegalStateException("Illegal CRC-32 checksum: " + value + " (resource=" + output + ")");
474     }
475     output.writeLong(value);
476   }
