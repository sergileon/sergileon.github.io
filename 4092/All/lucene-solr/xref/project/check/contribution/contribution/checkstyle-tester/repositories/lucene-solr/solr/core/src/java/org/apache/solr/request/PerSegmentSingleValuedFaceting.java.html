51  class PerSegmentSingleValuedFaceting {
52  
53    // input params
54    SolrIndexSearcher searcher;
55    DocSet docs;
56    String fieldName;
57    int offset;
58    int limit;
59    int mincount;
60    boolean missing;
61    String sort;
62    String prefix;
63    String contains;
64    boolean ignoreCase;
65  
66    Filter baseSet;
67  
68    int nThreads;
69  
70    public PerSegmentSingleValuedFaceting(SolrIndexSearcher searcher, DocSet docs, String fieldName, int offset, int limit, int mincount, boolean missing, String sort, String prefix, String contains, boolean ignoreCase) {
71      this.searcher = searcher;
72      this.docs = docs;
73      this.fieldName = fieldName;
74      this.offset = offset;
75      this.limit = limit;
76      this.mincount = mincount;
77      this.missing = missing;
78      this.sort = sort;
79      this.prefix = prefix;
80      this.contains = contains;
81      this.ignoreCase = ignoreCase;
82    }
83  
84    public void setNumThreads(int threads) {
85      nThreads = threads;
86    }
87  
88  
89    NamedList<Integer> getFacetCounts(Executor executor) throws IOException {
90  
91      CompletionService<SegFacet> completionService = new ExecutorCompletionService<>(executor);
92  
93      // reuse the translation logic to go from top level set to per-segment set
94      baseSet = docs.getTopFilter();
95  
96      final List<LeafReaderContext> leaves = searcher.getTopReaderContext().leaves();
97      // The list of pending tasks that aren't immediately submitted
98      // TODO: Is there a completion service, or a delegating executor that can
99      // limit the number of concurrent tasks submitted to a bigger executor?
100     LinkedList<Callable<SegFacet>> pending = new LinkedList<>();
101 
102     int threads = nThreads <= 0 ? Integer.MAX_VALUE : nThreads;
103 
104     for (final LeafReaderContext leave : leaves) {
105       final SegFacet segFacet = new SegFacet(leave);
106 
107       Callable<SegFacet> task = new Callable<SegFacet>() {
108         @Override
109         public SegFacet call() throws Exception {
110           segFacet.countTerms();
111           return segFacet;
112         }
113       };
114 
115       // TODO: if limiting threads, submit by largest segment first?
116 
117       if (--threads >= 0) {
118         completionService.submit(task);
119       } else {
120         pending.add(task);
121       }
122     }
123 
124 
125     // now merge the per-segment results
126     PriorityQueue<SegFacet> queue = new PriorityQueue<SegFacet>(leaves.size()) {
127       @Override
128       protected boolean lessThan(SegFacet a, SegFacet b) {
129         return a.tempBR.compareTo(b.tempBR) < 0;
130       }
131     };
132 
133 
134     boolean hasMissingCount=false;
135     int missingCount=0;
136     for (int i=0, c=leaves.size(); i<c; i++) {
137       SegFacet seg = null;
138 
139       try {
140         Future<SegFacet> future = completionService.take();        
141         seg = future.get();
142         if (!pending.isEmpty()) {
143           completionService.submit(pending.removeFirst());
144         }
145       } catch (InterruptedException e) {
146         Thread.currentThread().interrupt();
147         throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, e);
148       } catch (ExecutionException e) {
149         Throwable cause = e.getCause();
150         if (cause instanceof RuntimeException) {
151           throw (RuntimeException)cause;
152         } else {
153           throw new SolrException(SolrException.ErrorCode.SERVER_ERROR, "Error in per-segment faceting on field: " + fieldName, cause);
154         }
155       }
156 
157 
158       if (seg.startTermIndex < seg.endTermIndex) {
159         if (seg.startTermIndex==-1) {
160           hasMissingCount=true;
161           missingCount += seg.counts[0];
162           seg.pos = 0;
163         } else {
164           seg.pos = seg.startTermIndex;
165         }
166         if (seg.pos < seg.endTermIndex) {
167           seg.tenum = seg.si.termsEnum();
168           seg.tenum.seekExact(seg.pos);
169           seg.tempBR = seg.tenum.term();
170           queue.add(seg);
171         }
172       }
173     }
174 
175     FacetCollector collector;
176     if (sort.equals(FacetParams.FACET_SORT_COUNT) || sort.equals(FacetParams.FACET_SORT_COUNT_LEGACY)) {
177       collector = new CountSortedFacetCollector(offset, limit, mincount);
178     } else {
179       collector = new IndexSortedFacetCollector(offset, limit, mincount);
180     }
181 
182     BytesRefBuilder val = new BytesRefBuilder();
183 
184     while (queue.size() > 0) {
185       SegFacet seg = queue.top();
186       
187       // if facet.contains specified, only actually collect the count if substring contained
188       boolean collect = contains == null || SimpleFacets.contains(seg.tempBR.utf8ToString(), contains, ignoreCase);
189       
190       // we will normally end up advancing the term enum for this segment
191       // while still using "val", so we need to make a copy since the BytesRef
192       // may be shared across calls.
193       if (collect) {
194         val.copyBytes(seg.tempBR);
195       }
196       
197       int count = 0;
198 
199       do {
200         if (collect) {
201           count += seg.counts[seg.pos - seg.startTermIndex];
202         }
203 
204         // TODO: OPTIMIZATION...
205         // if mincount>0 then seg.pos++ can skip ahead to the next non-zero entry.
206         seg.pos++;
207         if (seg.pos >= seg.endTermIndex) {
208           queue.pop();
209           seg = queue.top();
210         } else {
211           seg.tempBR = seg.tenum.next();
212           seg = queue.updateTop();
213         }
214       } while (seg != null && val.get().compareTo(seg.tempBR) == 0);
215 
216       if (collect) {
217         boolean stop = collector.collect(val.get(), count);
218         if (stop) break;
219       }
220     }
221 
222     NamedList<Integer> res = collector.getFacetCounts();
223 
224     // convert labels to readable form    
225     FieldType ft = searcher.getSchema().getFieldType(fieldName);
226     int sz = res.size();
227     for (int i=0; i<sz; i++) {
228       res.setName(i, ft.indexedToReadable(res.getName(i)));
229     }
230 
231     if (missing) {
232       if (!hasMissingCount) {
233         missingCount = SimpleFacets.getFieldMissingCount(searcher,docs,fieldName);
234       }
235       res.add(null, missingCount);
236     }
237 
238     return res;
239   }
240 
241   class SegFacet {
242     LeafReaderContext context;
243     SegFacet(LeafReaderContext context) {
244       this.context = context;
245     }
246     
247     SortedDocValues si;
248     int startTermIndex;
249     int endTermIndex;
250     int[] counts;
251 
252     int pos; // only used when merging
253     TermsEnum tenum; // only used when merging
254 
255     BytesRef tempBR = new BytesRef();
256 
257     void countTerms() throws IOException {
258       si = DocValues.getSorted(context.reader(), fieldName);
259       // SolrCore.log.info("reader= " + reader + "  FC=" + System.identityHashCode(si));
260 
261       if (prefix!=null) {
262         BytesRefBuilder prefixRef = new BytesRefBuilder();
263         prefixRef.copyChars(prefix);
264         startTermIndex = si.lookupTerm(prefixRef.get());
265         if (startTermIndex<0) startTermIndex=-startTermIndex-1;
266         prefixRef.append(UnicodeUtil.BIG_TERM);
267         // TODO: we could constrain the lower endpoint if we had a binarySearch method that allowed passing start/end
268         endTermIndex = si.lookupTerm(prefixRef.get());
269         assert endTermIndex < 0;
270         endTermIndex = -endTermIndex-1;
271       } else {
272         startTermIndex=-1;
273         endTermIndex=si.getValueCount();
274       }
275       final int nTerms=endTermIndex-startTermIndex;
276       if (nTerms == 0) return;
277 
278       // count collection array only needs to be as big as the number of terms we are
279       // going to collect counts for.
280       final int[] counts = this.counts = new int[nTerms];
281       DocIdSet idSet = baseSet.getDocIdSet(context, null);  // this set only includes live docs
282       DocIdSetIterator iter = idSet.iterator();
283 
284       if (prefix==null) {
285         // specialized version when collecting counts for all terms
286         int doc;
287         while ((doc = iter.nextDoc()) < DocIdSetIterator.NO_MORE_DOCS) {
288           counts[1+si.getOrd(doc)]++;
289         }
290       } else {
291         // version that adjusts term numbers because we aren't collecting the full range
292         int doc;
293         while ((doc = iter.nextDoc()) < DocIdSetIterator.NO_MORE_DOCS) {
294           int term = si.getOrd(doc);
295           int arrIdx = term-startTermIndex;
296           if (arrIdx>=0 && arrIdx<nTerms) counts[arrIdx]++;
297         }
298       }
299     }
300   }
301 
302 }

306 abstract class FacetCollector {
307   /*** return true to stop collection */
308   public abstract boolean collect(BytesRef term, int count);
309   public abstract NamedList<Integer> getFacetCounts();
310 }
311 
312 
313 // This collector expects facets to be collected in index order
314 class CountSortedFacetCollector extends FacetCollector {
315   private final CharsRefBuilder spare = new CharsRefBuilder();
316 
317   final int offset;
318   final int limit;
319   final int maxsize;
320   final BoundedTreeSet<SimpleFacets.CountPair<String,Integer>> queue;
321 
322   int min;  // the smallest value in the top 'N' values
323 
324   public CountSortedFacetCollector(int offset, int limit, int mincount) {
325     this.offset = offset;
326     this.limit = limit;
327     maxsize = limit>0 ? offset+limit : Integer.MAX_VALUE-1;
328     queue = new BoundedTreeSet<>(maxsize);
329     min=mincount-1;  // the smallest value in the top 'N' values
330   }
331 
332   @Override
333   public boolean collect(BytesRef term, int count) {
334     if (count > min) {
335       // NOTE: we use c>min rather than c>=min as an optimization because we are going in
336       // index order, so we already know that the keys are ordered.  This can be very
337       // important if a lot of the counts are repeated (like zero counts would be).
338       spare.copyUTF8Bytes(term);
339       queue.add(new SimpleFacets.CountPair<>(spare.toString(), count));
340       if (queue.size()>=maxsize) min=queue.last().val;
341     }
342     return false;
343   }
344 
345   @Override
346   public NamedList<Integer> getFacetCounts() {
347     NamedList<Integer> res = new NamedList<>();
348     int off=offset;
349     int lim=limit>=0 ? limit : Integer.MAX_VALUE;
350      // now select the right page from the results
351      for (SimpleFacets.CountPair<String,Integer> p : queue) {
352        if (--off>=0) continue;
353        if (--lim<0) break;
354        res.add(p.key, p.val);
355      }
356     return res;
357   }
358 }

360 // This collector expects facets to be collected in index order
361 class IndexSortedFacetCollector extends FacetCollector {
362   private final CharsRefBuilder spare = new CharsRefBuilder();
363 
364   int offset;
365   int limit;
366   final int mincount;
367   final NamedList<Integer> res = new NamedList<>();
368 
369   public IndexSortedFacetCollector(int offset, int limit, int mincount) {
370     this.offset = offset;
371     this.limit = limit>0 ? limit : Integer.MAX_VALUE;
372     this.mincount = mincount;
373   }
374 
375   @Override
376   public boolean collect(BytesRef term, int count) {
377     if (count < mincount) {
378       return false;
379     }
380 
381     if (offset > 0) {
382       offset--;
383       return false;
384     }
385 
386     if (limit > 0) {
387       spare.copyUTF8Bytes(term);
388       res.add(spare.toString(), count);
389       limit--;
390     }
391 
392     return limit <= 0;
393   }
394 
395   @Override
396   public NamedList<Integer> getFacetCounts() {
397     return res;
398   }
