68    /** expert: Creates a new writer */
69    public Lucene54DocValuesConsumer(SegmentWriteState state, String dataCodec, String dataExtension, String metaCodec, String metaExtension) throws IOException {
70      boolean success = false;
71      try {
72        String dataName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, dataExtension);
73        data = state.directory.createOutput(dataName, state.context);
74        CodecUtil.writeIndexHeader(data, dataCodec, Lucene54DocValuesFormat.VERSION_CURRENT, state.segmentInfo.getId(), state.segmentSuffix);
75        String metaName = IndexFileNames.segmentFileName(state.segmentInfo.name, state.segmentSuffix, metaExtension);
76        meta = state.directory.createOutput(metaName, state.context);
77        CodecUtil.writeIndexHeader(meta, metaCodec, Lucene54DocValuesFormat.VERSION_CURRENT, state.segmentInfo.getId(), state.segmentSuffix);
78        maxDoc = state.segmentInfo.maxDoc();
79        success = true;
80      } finally {
81        if (!success) {
82          IOUtils.closeWhileHandlingException(this);
83        }
84      }
85    }

92    void addNumericField(FieldInfo field, final Iterable<Number> values, NumberType numberType) throws IOException {
93      long count = 0;
94      long minValue = Long.MAX_VALUE;
95      long maxValue = Long.MIN_VALUE;
96      long gcd = 0;
97      long missingCount = 0;
98      long zeroCount = 0;
99      // TODO: more efficient?
100     HashSet<Long> uniqueValues = null;
101     long missingOrdCount = 0;
102     if (numberType == NumberType.VALUE) {
103       uniqueValues = new HashSet<>();
104 
105       for (Number nv : values) {
106         final long v;
107         if (nv == null) {
108           v = 0;
109           missingCount++;
110           zeroCount++;
111         } else {
112           v = nv.longValue();
113           if (v == 0) {
114             zeroCount++;
115           }
116         }
117 
118         if (gcd != 1) {
119           if (v < Long.MIN_VALUE / 2 || v > Long.MAX_VALUE / 2) {
120             // in that case v - minValue might overflow and make the GCD computation return
121             // wrong results. Since these extreme values are unlikely, we just discard
122             // GCD computation for them
123             gcd = 1;
124           } else if (count != 0) { // minValue needs to be set first
125             gcd = MathUtil.gcd(gcd, v - minValue);
126           }
127         }
128 
129         minValue = Math.min(minValue, v);
130         maxValue = Math.max(maxValue, v);
131 
132         if (uniqueValues != null) {
133           if (uniqueValues.add(v)) {
134             if (uniqueValues.size() > 256) {
135               uniqueValues = null;
136             }
137           }
138         }
139 
140         ++count;
141       }
142     } else {
143       for (Number nv : values) {
144         long v = nv.longValue();
145         if (v == -1L) {
146           missingOrdCount++;
147         }
148         minValue = Math.min(minValue, v);
149         maxValue = Math.max(maxValue, v);
150         ++count;
151       }
152     }
153     
154     final long delta = maxValue - minValue;
155     final int deltaBitsRequired = DirectWriter.unsignedBitsRequired(delta);
156     final int tableBitsRequired = uniqueValues == null
157         ? Integer.MAX_VALUE
158         : DirectWriter.bitsRequired(uniqueValues.size() - 1);
159 
160     final boolean sparse; // 1% of docs or less have a value
161     switch (numberType) {
162       case VALUE:
163         sparse = (double) missingCount / count >= 0.99;
164         break;
165       case ORDINAL:
166         sparse = (double) missingOrdCount / count >= 0.99;
167         break;
168       default:
169         throw new AssertionError();
170     }
171 
172     final int format;
173     if (uniqueValues != null 
174         && count <= Integer.MAX_VALUE
175         && (uniqueValues.size() == 1
176            || (uniqueValues.size() == 2 && missingCount > 0 && zeroCount == missingCount))) {
177       // either one unique value C or two unique values: "missing" and C
178       format = CONST_COMPRESSED;
179     } else if (sparse && count >= 1024) {
180       // require at least 1024 docs to avoid flipping back and forth when doing NRT search
181       format = SPARSE_COMPRESSED;
182     } else if (uniqueValues != null && tableBitsRequired < deltaBitsRequired) {
183       format = TABLE_COMPRESSED;
184     } else if (gcd != 0 && gcd != 1) {
185       final long gcdDelta = (maxValue - minValue) / gcd;
186       final long gcdBitsRequired = DirectWriter.unsignedBitsRequired(gcdDelta);
187       format = gcdBitsRequired < deltaBitsRequired ? GCD_COMPRESSED : DELTA_COMPRESSED;
188     } else {
189       format = DELTA_COMPRESSED;
190     }
191     meta.writeVInt(field.number);
192     meta.writeByte(Lucene54DocValuesFormat.NUMERIC);
193     meta.writeVInt(format);
194     if (format == SPARSE_COMPRESSED) {
195       meta.writeLong(data.getFilePointer());
196       final long numDocsWithValue;
197       switch (numberType) {
198         case VALUE:
199           numDocsWithValue = count - missingCount;
200           break;
201         case ORDINAL:
202           numDocsWithValue = count - missingOrdCount;
203           break;
204         default:
205           throw new AssertionError();
206       }
207       final long maxDoc = writeSparseMissingBitset(values, numberType, numDocsWithValue);
208       assert maxDoc == count;
209     } else if (missingCount == 0) {
210       meta.writeLong(ALL_LIVE);
211     } else if (missingCount == count) {
212       meta.writeLong(ALL_MISSING);
213     } else {
214       meta.writeLong(data.getFilePointer());
215       writeMissingBitset(values);
216     }
217     meta.writeLong(data.getFilePointer());
218     meta.writeVLong(count);
219 
220     switch (format) {
221       case CONST_COMPRESSED:
222         // write the constant (nonzero value in the n=2 case, singleton value otherwise)
223         meta.writeLong(minValue < 0 ? Collections.min(uniqueValues) : Collections.max(uniqueValues));
224         break;
225       case GCD_COMPRESSED:
226         meta.writeLong(minValue);
227         meta.writeLong(gcd);
228         final long maxDelta = (maxValue - minValue) / gcd;
229         final int bits = DirectWriter.unsignedBitsRequired(maxDelta);
230         meta.writeVInt(bits);
231         final DirectWriter quotientWriter = DirectWriter.getInstance(data, count, bits);
232         for (Number nv : values) {
233           long value = nv == null ? 0 : nv.longValue();
234           quotientWriter.add((value - minValue) / gcd);
235         }
236         quotientWriter.finish();
237         break;
238       case DELTA_COMPRESSED:
239         final long minDelta = delta < 0 ? 0 : minValue;
240         meta.writeLong(minDelta);
241         meta.writeVInt(deltaBitsRequired);
242         final DirectWriter writer = DirectWriter.getInstance(data, count, deltaBitsRequired);
243         for (Number nv : values) {
244           long v = nv == null ? 0 : nv.longValue();
245           writer.add(v - minDelta);
246         }
247         writer.finish();
248         break;
249       case TABLE_COMPRESSED:
250         final Long[] decode = uniqueValues.toArray(new Long[uniqueValues.size()]);
251         Arrays.sort(decode);
252         final HashMap<Long,Integer> encode = new HashMap<>();
253         meta.writeVInt(decode.length);
254         for (int i = 0; i < decode.length; i++) {
255           meta.writeLong(decode[i]);
256           encode.put(decode[i], i);
257         }
258         meta.writeVInt(tableBitsRequired);
259         final DirectWriter ordsWriter = DirectWriter.getInstance(data, count, tableBitsRequired);
260         for (Number nv : values) {
261           ordsWriter.add(encode.get(nv == null ? 0 : nv.longValue()));
262         }
263         ordsWriter.finish();
264         break;
265       case SPARSE_COMPRESSED:
266         final Iterable<Number> filteredMissingValues;
267         switch (numberType) {
268           case VALUE:
269             meta.writeByte((byte) 0);
270             filteredMissingValues = new Iterable<Number>() {
271               @Override
272               public Iterator<Number> iterator() {
273                 return filter(values.iterator(),
274                     new Predicate<Number>() {
275                       boolean apply(Number input) {
276                         return input != null;
277                       }
278                     });
279               }
280             };
281             break;
282           case ORDINAL:
283             meta.writeByte((byte) 1);
284             filteredMissingValues = new Iterable<Number>() {
285               @Override
286               public Iterator<Number> iterator() {
287                 return filter(values.iterator(),
288                     new Predicate<Number>() {
289                       boolean apply(Number input) {
290                         return input.longValue() != -1L;
291                       }
292                     });
293               }
294             };
295             break;
296           default:
297             throw new AssertionError();
298         }
299         // Write non-missing values as a numeric field
300         addNumericField(field, filteredMissingValues, numberType);
301         break;
302       default:
303         throw new AssertionError();
304     }
305     meta.writeLong(data.getFilePointer());
306   }

308   private static abstract class Predicate<T> {
309     abstract boolean apply(T input);
310   }
311 
312   // filter the given iterator by the given predicate
313   private static <T> Iterator<T> filter(final Iterator<? extends T> iterator,
314       final Predicate<? super T> predicate) {
315     return new Iterator<T>() {
316 
317       boolean hasNext = false;
318       T next;
319 
320       @Override
321       public boolean hasNext() {
322         if (hasNext) {
323           return true;
324         }
325         while (iterator.hasNext()) {
326           next = iterator.next();
327           if (predicate.apply(next)) {
328             return hasNext = true;
329           }
330         }
331         next = null;
332         return hasNext = false;
333       }
334 
335       @Override
336       public T next() {
337         if (!hasNext()) {
338           throw new NoSuchElementException();
339         }
340         final T next = this.next;
341         hasNext = false;
342         this.next = null;
343         return next;
344       }
345 
346       @Override
347       public void remove() {
348         throw new UnsupportedOperationException();
349       }
350       
351     };
352   }

354   // TODO: in some cases representing missing with minValue-1 wouldn't take up additional space and so on,
355   // but this is very simple, and algorithms only check this for values of 0 anyway (doesnt slow down normal decode)
356   void writeMissingBitset(Iterable<?> values) throws IOException {
357     byte bits = 0;
358     int count = 0;
359     for (Object v : values) {
360       if (count == 8) {
361         data.writeByte(bits);
362         count = 0;
363         bits = 0;
364       }
365       if (v != null) {
366         bits |= 1 << (count & 7);
367       }
368       count++;
369     }
370     if (count > 0) {
371       data.writeByte(bits);
372     }
373   }

375   long writeSparseMissingBitset(Iterable<Number> values, NumberType numberType, long numDocsWithValue) throws IOException {
376     meta.writeVLong(numDocsWithValue);
377 
378     // Write doc IDs that have a value
379     meta.writeVInt(DIRECT_MONOTONIC_BLOCK_SHIFT);
380     final DirectMonotonicWriter docIdsWriter = DirectMonotonicWriter.getInstance(meta, data, numDocsWithValue, DIRECT_MONOTONIC_BLOCK_SHIFT);
381     long docID = 0;
382     for (Number nv : values) {
383       switch (numberType) {
384         case VALUE:
385           if (nv != null) {
386             docIdsWriter.add(docID);
387           }
388           break;
389         case ORDINAL:
390           if (nv.longValue() != -1L) {
391             docIdsWriter.add(docID);
392           }
393           break;
394         default:
395           throw new AssertionError();
396       }
397       docID++;
398     }
399     docIdsWriter.finish();
400     return docID;
401   }

403   @Override
404   public void addBinaryField(FieldInfo field, Iterable<BytesRef> values) throws IOException {
405     // write the byte[] data
406     meta.writeVInt(field.number);
407     meta.writeByte(Lucene54DocValuesFormat.BINARY);
408     int minLength = Integer.MAX_VALUE;
409     int maxLength = Integer.MIN_VALUE;
410     final long startFP = data.getFilePointer();
411     long count = 0;
412     long missingCount = 0;
413     for(BytesRef v : values) {
414       final int length;
415       if (v == null) {
416         length = 0;
417         missingCount++;
418       } else {
419         length = v.length;
420       }
421       minLength = Math.min(minLength, length);
422       maxLength = Math.max(maxLength, length);
423       if (v != null) {
424         data.writeBytes(v.bytes, v.offset, v.length);
425       }
426       count++;
427     }
428     meta.writeVInt(minLength == maxLength ? BINARY_FIXED_UNCOMPRESSED : BINARY_VARIABLE_UNCOMPRESSED);
429     if (missingCount == 0) {
430       meta.writeLong(ALL_LIVE);
431     } else if (missingCount == count) {
432       meta.writeLong(ALL_MISSING);
433     } else {
434       meta.writeLong(data.getFilePointer());
435       writeMissingBitset(values);
436     }
437     meta.writeVInt(minLength);
438     meta.writeVInt(maxLength);
439     meta.writeVLong(count);
440     meta.writeLong(startFP);
441     
442     // if minLength == maxLength, it's a fixed-length byte[], we are done (the addresses are implicit)
443     // otherwise, we need to record the length fields...
444     if (minLength != maxLength) {
445       meta.writeLong(data.getFilePointer());
446       meta.writeVInt(DIRECT_MONOTONIC_BLOCK_SHIFT);
447 
448       final DirectMonotonicWriter writer = DirectMonotonicWriter.getInstance(meta, data, count + 1, DIRECT_MONOTONIC_BLOCK_SHIFT);
449       long addr = 0;
450       writer.add(addr);
451       for (BytesRef v : values) {
452         if (v != null) {
453           addr += v.length;
454         }
455         writer.add(addr);
456       }
457       writer.finish();
458       meta.writeLong(data.getFilePointer());
459     }
460   }

462   /** expert: writes a value dictionary for a sorted/sortedset field */
463   private void addTermsDict(FieldInfo field, final Iterable<BytesRef> values) throws IOException {
464     // first check if it's a "fixed-length" terms dict
465     int minLength = Integer.MAX_VALUE;
466     int maxLength = Integer.MIN_VALUE;
467     long numValues = 0;
468     for (BytesRef v : values) {
469       minLength = Math.min(minLength, v.length);
470       maxLength = Math.max(maxLength, v.length);
471       numValues++;
472     }
473     if (minLength == maxLength) {
474       // no index needed: direct addressing by mult
475       addBinaryField(field, values);
476     } else if (numValues < REVERSE_INTERVAL_COUNT) {
477       // low cardinality: waste a few KB of ram, but can't really use fancy index etc
478       addBinaryField(field, values);
479     } else {
480       assert numValues > 0; // we don't have to handle the empty case
481       // header
482       meta.writeVInt(field.number);
483       meta.writeByte(Lucene54DocValuesFormat.BINARY);
484       meta.writeVInt(BINARY_PREFIX_COMPRESSED);
485       meta.writeLong(-1L);
486       // now write the bytes: sharing prefixes within a block
487       final long startFP = data.getFilePointer();
488       // currently, we have to store the delta from expected for every 1/nth term
489       // we could avoid this, but it's not much and less overall RAM than the previous approach!
490       RAMOutputStream addressBuffer = new RAMOutputStream();
491       MonotonicBlockPackedWriter termAddresses = new MonotonicBlockPackedWriter(addressBuffer, MONOTONIC_BLOCK_SIZE);
492       // buffers up 16 terms
493       RAMOutputStream bytesBuffer = new RAMOutputStream();
494       // buffers up block header
495       RAMOutputStream headerBuffer = new RAMOutputStream();
496       BytesRefBuilder lastTerm = new BytesRefBuilder();
497       lastTerm.grow(maxLength);
498       long count = 0;
499       int suffixDeltas[] = new int[INTERVAL_COUNT];
500       for (BytesRef v : values) {
501         int termPosition = (int) (count & INTERVAL_MASK);
502         if (termPosition == 0) {
503           termAddresses.add(data.getFilePointer() - startFP);
504           // abs-encode first term
505           headerBuffer.writeVInt(v.length);
506           headerBuffer.writeBytes(v.bytes, v.offset, v.length);
507           lastTerm.copyBytes(v);
508         } else {
509           // prefix-code: we only share at most 255 characters, to encode the length as a single
510           // byte and have random access. Larger terms just get less compression.
511           int sharedPrefix = Math.min(255, StringHelper.bytesDifference(lastTerm.get(), v));
512           bytesBuffer.writeByte((byte) sharedPrefix);
513           bytesBuffer.writeBytes(v.bytes, v.offset + sharedPrefix, v.length - sharedPrefix);
514           // we can encode one smaller, because terms are unique.
515           suffixDeltas[termPosition] = v.length - sharedPrefix - 1;
516         }
517         
518         count++;
519         // flush block
520         if ((count & INTERVAL_MASK) == 0) {
521           flushTermsDictBlock(headerBuffer, bytesBuffer, suffixDeltas);
522         }
523       }
524       // flush trailing crap
525       int leftover = (int) (count & INTERVAL_MASK);
526       if (leftover > 0) {
527         Arrays.fill(suffixDeltas, leftover, suffixDeltas.length, 0);
528         flushTermsDictBlock(headerBuffer, bytesBuffer, suffixDeltas);
529       }
530       final long indexStartFP = data.getFilePointer();
531       // write addresses of indexed terms
532       termAddresses.finish();
533       addressBuffer.writeTo(data);
534       addressBuffer = null;
535       termAddresses = null;
536       meta.writeVInt(minLength);
537       meta.writeVInt(maxLength);
538       meta.writeVLong(count);
539       meta.writeLong(startFP);
540       meta.writeLong(indexStartFP);
541       meta.writeVInt(PackedInts.VERSION_CURRENT);
542       meta.writeVInt(MONOTONIC_BLOCK_SIZE);
543       addReverseTermIndex(field, values, maxLength);
544     }
545   }

547   // writes term dictionary "block"
548   // first term is absolute encoded as vint length + bytes.
549   // lengths of subsequent N terms are encoded as either N bytes or N shorts.
550   // in the double-byte case, the first byte is indicated with -1.
551   // subsequent terms are encoded as byte suffixLength + bytes.
552   private void flushTermsDictBlock(RAMOutputStream headerBuffer, RAMOutputStream bytesBuffer, int suffixDeltas[]) throws IOException {
553     boolean twoByte = false;
554     for (int i = 1; i < suffixDeltas.length; i++) {
555       if (suffixDeltas[i] > 254) {
556         twoByte = true;
557       }
558     }
559     if (twoByte) {
560       headerBuffer.writeByte((byte)255);
561       for (int i = 1; i < suffixDeltas.length; i++) {
562         headerBuffer.writeShort((short) suffixDeltas[i]);
563       }
564     } else {
565       for (int i = 1; i < suffixDeltas.length; i++) {
566         headerBuffer.writeByte((byte) suffixDeltas[i]);
567       }
568     }
569     headerBuffer.writeTo(data);
570     headerBuffer.reset();
571     bytesBuffer.writeTo(data);
572     bytesBuffer.reset();
573   }

575   // writes reverse term index: used for binary searching a term into a range of 64 blocks
576   // for every 64 blocks (1024 terms) we store a term, trimming any suffix unnecessary for comparison
577   // terms are written as a contiguous byte[], but never spanning 2^15 byte boundaries.
578   private void addReverseTermIndex(FieldInfo field, final Iterable<BytesRef> values, int maxLength) throws IOException {
579     long count = 0;
580     BytesRefBuilder priorTerm = new BytesRefBuilder();
581     priorTerm.grow(maxLength);
582     BytesRef indexTerm = new BytesRef();
583     long startFP = data.getFilePointer();
584     PagedBytes pagedBytes = new PagedBytes(15);
585     MonotonicBlockPackedWriter addresses = new MonotonicBlockPackedWriter(data, MONOTONIC_BLOCK_SIZE);
586     
587     for (BytesRef b : values) {
588       int termPosition = (int) (count & REVERSE_INTERVAL_MASK);
589       if (termPosition == 0) {
590         int len = StringHelper.sortKeyLength(priorTerm.get(), b);
591         indexTerm.bytes = b.bytes;
592         indexTerm.offset = b.offset;
593         indexTerm.length = len;
594         addresses.add(pagedBytes.copyUsingLengthPrefix(indexTerm));
595       } else if (termPosition == REVERSE_INTERVAL_MASK) {
596         priorTerm.copyBytes(b);
597       }
598       count++;
599     }
600     addresses.finish();
601     long numBytes = pagedBytes.getPointer();
602     pagedBytes.freeze(true);
603     PagedBytesDataInput in = pagedBytes.getDataInput();
604     meta.writeLong(startFP);
605     data.writeVLong(numBytes);
606     data.copyBytes(in, numBytes);
607   }

617   @Override
618   public void addSortedNumericField(FieldInfo field, final Iterable<Number> docToValueCount, final Iterable<Number> values) throws IOException {
619     meta.writeVInt(field.number);
620     meta.writeByte(Lucene54DocValuesFormat.SORTED_NUMERIC);
621     if (isSingleValued(docToValueCount)) {
622       meta.writeVInt(SORTED_SINGLE_VALUED);
623       // The field is single-valued, we can encode it as NUMERIC
624       addNumericField(field, singletonView(docToValueCount, values, null));
625     } else {
626       final SortedSet<LongsRef> uniqueValueSets = uniqueValueSets(docToValueCount, values);
627       if (uniqueValueSets != null) {
628         meta.writeVInt(SORTED_SET_TABLE);
629 
630         // write the set_id -> values mapping
631         writeDictionary(uniqueValueSets);
632 
633         // write the doc -> set_id as a numeric field
634         addNumericField(field, docToSetId(uniqueValueSets, docToValueCount, values), NumberType.ORDINAL);
635       } else {
636         meta.writeVInt(SORTED_WITH_ADDRESSES);
637         // write the stream of values as a numeric field
638         addNumericField(field, values, NumberType.VALUE);
639         // write the doc -> ord count as a absolute index to the stream
640         addOrdIndex(field, docToValueCount);
641       }
642     }
643   }

645   @Override
646   public void addSortedSetField(FieldInfo field, Iterable<BytesRef> values, final Iterable<Number> docToOrdCount, final Iterable<Number> ords) throws IOException {
647     meta.writeVInt(field.number);
648     meta.writeByte(Lucene54DocValuesFormat.SORTED_SET);
649 
650     if (isSingleValued(docToOrdCount)) {
651       meta.writeVInt(SORTED_SINGLE_VALUED);
652       // The field is single-valued, we can encode it as SORTED
653       addSortedField(field, values, singletonView(docToOrdCount, ords, -1L));
654     } else {
655       final SortedSet<LongsRef> uniqueValueSets = uniqueValueSets(docToOrdCount, ords);
656       if (uniqueValueSets != null) {
657         meta.writeVInt(SORTED_SET_TABLE);
658 
659         // write the set_id -> ords mapping
660         writeDictionary(uniqueValueSets);
661 
662         // write the ord -> byte[] as a binary field
663         addTermsDict(field, values);
664 
665         // write the doc -> set_id as a numeric field
666         addNumericField(field, docToSetId(uniqueValueSets, docToOrdCount, ords), NumberType.ORDINAL);
667       } else {
668         meta.writeVInt(SORTED_WITH_ADDRESSES);
669 
670         // write the ord -> byte[] as a binary field
671         addTermsDict(field, values);
672 
673         // write the stream of ords as a numeric field
674         // NOTE: we could return an iterator that delta-encodes these within a doc
675         addNumericField(field, ords, NumberType.ORDINAL);
676 
677         // write the doc -> ord count as a absolute index to the stream
678         addOrdIndex(field, docToOrdCount);
679       }
680     }
681   }
682 
683   private SortedSet<LongsRef> uniqueValueSets(Iterable<Number> docToValueCount, Iterable<Number> values) {
684     Set<LongsRef> uniqueValueSet = new HashSet<>();
685     LongsRef docValues = new LongsRef(256);
686 
687     Iterator<Number> valueCountIterator = docToValueCount.iterator();
688     Iterator<Number> valueIterator = values.iterator();
689     int totalDictSize = 0;
690     while (valueCountIterator.hasNext()) {
691       docValues.length = valueCountIterator.next().intValue();
692       if (docValues.length > 256) {
693         return null;
694       }
695       for (int i = 0; i < docValues.length; ++i) {
696         docValues.longs[i] = valueIterator.next().longValue();
697       }
698       if (uniqueValueSet.contains(docValues)) {
699         continue;
700       }
701       totalDictSize += docValues.length;
702       if (totalDictSize > 256) {
703         return null;
704       }
705       uniqueValueSet.add(new LongsRef(Arrays.copyOf(docValues.longs, docValues.length), 0, docValues.length));
706     }
707     assert valueIterator.hasNext() == false;
708     return new TreeSet<>(uniqueValueSet);
709   }
710 
711   private void writeDictionary(SortedSet<LongsRef> uniqueValueSets) throws IOException {
712     int lengthSum = 0;
713     for (LongsRef longs : uniqueValueSets) {
714       lengthSum += longs.length;
715     }
716 
717     meta.writeInt(lengthSum);
718     for (LongsRef valueSet : uniqueValueSets) {
719       for (int  i = 0; i < valueSet.length; ++i) {
720         meta.writeLong(valueSet.longs[valueSet.offset + i]);
721       }
722     }
723 
724     meta.writeInt(uniqueValueSets.size());
725     for (LongsRef valueSet : uniqueValueSets) {
726       meta.writeInt(valueSet.length);
727     }
728   }

730   private Iterable<Number> docToSetId(SortedSet<LongsRef> uniqueValueSets, final Iterable<Number> docToValueCount, final Iterable<Number> values) {
731     final Map<LongsRef, Integer> setIds = new HashMap<>();
732     int i = 0;
733     for (LongsRef set : uniqueValueSets) {
734       setIds.put(set, i++);
735     }
736     assert i == uniqueValueSets.size();
737 
738     return new Iterable<Number>() {
739 
740       @Override
741       public Iterator<Number> iterator() {
742         final Iterator<Number> valueCountIterator = docToValueCount.iterator();
743         final Iterator<Number> valueIterator = values.iterator();
744         final LongsRef docValues = new LongsRef(256);
745         return new Iterator<Number>() {
746 
747           @Override
748           public boolean hasNext() {
749             return valueCountIterator.hasNext();
750           }
751 
752           @Override
753           public Number next() {
754             docValues.length = valueCountIterator.next().intValue();
755             for (int i = 0; i < docValues.length; ++i) {
756               docValues.longs[i] = valueIterator.next().longValue();
757             }
758             final Integer id = setIds.get(docValues);
759             assert id != null;
760             return id;
761           }
762 
763           @Override
764           public void remove() {
765             throw new UnsupportedOperationException();
766           }
767         };
768 
769       }
770     };
771   }

773   // writes addressing information as MONOTONIC_COMPRESSED integer
774   private void addOrdIndex(FieldInfo field, Iterable<Number> values) throws IOException {
775     meta.writeVInt(field.number);
776     meta.writeByte(Lucene54DocValuesFormat.NUMERIC);
777     meta.writeVInt(MONOTONIC_COMPRESSED);
778     meta.writeLong(-1L);
779     meta.writeLong(data.getFilePointer());
780     meta.writeVLong(maxDoc);
781     meta.writeVInt(DIRECT_MONOTONIC_BLOCK_SHIFT);
782 
783     final DirectMonotonicWriter writer = DirectMonotonicWriter.getInstance(meta, data, maxDoc + 1, DIRECT_MONOTONIC_BLOCK_SHIFT);
784     long addr = 0;
785     writer.add(addr);
786     for (Number v : values) {
787       addr += v.longValue();
788       writer.add(addr);
789     }
790     writer.finish();
791     meta.writeLong(data.getFilePointer());
792   }

794   @Override
795   public void close() throws IOException {
796     boolean success = false;
797     try {
798       if (meta != null) {
799         meta.writeVInt(-1); // write EOF marker
800         CodecUtil.writeFooter(meta); // write checksum
801       }
802       if (data != null) {
803         CodecUtil.writeFooter(data); // write checksum
804       }
805       success = true;
806     } finally {
807       if (success) {
808         IOUtils.close(data, meta);
809       } else {
810         IOUtils.closeWhileHandlingException(data, meta);
811       }
812       meta = data = null;
813     }
814   }
