148   @Test
149   public void testPutFile() throws Exception {
150     // test put file
151     String[] args = new String[] {"-zkhost", zkServer.getZkAddress(), "-cmd",
152         "putfile", "/solr.xml", SOLR_HOME + File.separator + "solr-stress-new.xml"};
153     ZkCLI.main(args);
154 
155     String fromZk = new String(zkClient.getData("/solr.xml", null, null, true), StandardCharsets.UTF_8);
156     File locFile = new File(SOLR_HOME + File.separator + "solr-stress-new.xml");
157     InputStream is = new FileInputStream(locFile);
158     String fromLoc;
159     try {
160       fromLoc = new String(IOUtils.toByteArray(is), StandardCharsets.UTF_8);
161     } finally {
162       IOUtils.closeQuietly(is);
163     }
164     assertEquals("Should get back what we put in ZK", fromZk, fromLoc);
165   }

190   @Test
191   public void testUpConfigLinkConfigClearZk() throws Exception {
192     File tmpDir = createTempDir().toFile();
193     
194     // test upconfig
195     String confsetname = "confsetone";
196     String[] args = new String[] {
197         "-zkhost",
198         zkServer.getZkAddress(),
199         "-cmd",
200         "upconfig",
201         "-confdir",
202         ExternalPaths.TECHPRODUCTS_CONFIGSET, "-confname", confsetname};
203     ZkCLI.main(args);
204     
205     assertTrue(zkClient.exists(ZkConfigManager.CONFIGS_ZKNODE + "/" + confsetname, true));
206 
207     // print help
208     // ZkCLI.main(new String[0]);
209     
210     // test linkconfig
211     args = new String[] {"-zkhost", zkServer.getZkAddress(), "-cmd",
212         "linkconfig", "-collection", "collection1", "-confname", confsetname};
213     ZkCLI.main(args);
214     
215     ZkNodeProps collectionProps = ZkNodeProps.load(zkClient.getData(ZkStateReader.COLLECTIONS_ZKNODE + "/collection1", null, null, true));
216     assertTrue(collectionProps.containsKey("configName"));
217     assertEquals(confsetname, collectionProps.getStr("configName"));
218     
219     // test down config
220     File confDir = new File(tmpDir,
221         "solrtest-confdropspot-" + this.getClass().getName() + "-" + System.nanoTime());
222     assertFalse(confDir.exists());
223 
224     args = new String[] {"-zkhost", zkServer.getZkAddress(), "-cmd",
225         "downconfig", "-confdir", confDir.getAbsolutePath(), "-confname", confsetname};
226     ZkCLI.main(args);
227     
228     File[] files = confDir.listFiles();
229     List<String> zkFiles = zkClient.getChildren(ZkConfigManager.CONFIGS_ZKNODE + "/" + confsetname, null, true);
230     assertEquals(files.length, zkFiles.size());
231     
232     File sourceConfDir = new File(ExternalPaths.TECHPRODUCTS_CONFIGSET);
233     // filter out all directories starting with . (e.g. .svn)
234     Collection<File> sourceFiles = FileUtils.listFiles(sourceConfDir, TrueFileFilter.INSTANCE, new RegexFileFilter("[^\\.].*"));
235     for (File sourceFile :sourceFiles){
236         int indexOfRelativePath = sourceFile.getAbsolutePath().lastIndexOf("sample_techproducts_configs" + File.separator + "conf");
237         String relativePathofFile = sourceFile.getAbsolutePath().substring(indexOfRelativePath + 33, sourceFile.getAbsolutePath().length());
238         File downloadedFile = new File(confDir,relativePathofFile);
239         assertTrue(downloadedFile.getAbsolutePath() + " does not exist source:" + sourceFile.getAbsolutePath(), downloadedFile.exists());
240         assertTrue("Content didn't change",FileUtils.contentEquals(sourceFile,downloadedFile));
241     }
242     
243    
244     // test reset zk
245     args = new String[] {"-zkhost", zkServer.getZkAddress(), "-cmd",
246         "clear", "/"};
247     ZkCLI.main(args);
248 
249     assertEquals(0, zkClient.getChildren("/", null, true).size());
250   }

301   @Test
302   public void testSetClusterProperty() throws Exception {
303     ZkStateReader reader = new ZkStateReader(zkClient);
304     try {
305       // add property urlScheme=http
306       String[] args = new String[] {"-zkhost", zkServer.getZkAddress(),
307           "-cmd", "CLUSTERPROP", "-name", "urlScheme", "-val", "http"};
308       ZkCLI.main(args);
309       assertEquals("http", reader.getClusterProps().get("urlScheme"));
310       
311       // remove it again
312       args = new String[] {"-zkhost", zkServer.getZkAddress(),
313           "-cmd", "CLUSTERPROP", "-name", "urlScheme"};
314       ZkCLI.main(args);
315       assertNull(reader.getClusterProps().get("urlScheme"));
316     } finally {
317       reader.close();
318     }
319   }
