56    /**
57     * Similarity to eliminate tf, idf and lengthNorm effects to isolate test
58     * case.
59     * 
60     * <p>
61     * same as TestRankingSimilarity in TestRanking.zip from
62     * http://issues.apache.org/jira/browse/LUCENE-323
63     * </p>
64     */
65    private static class TestSimilarity extends DefaultSimilarity {
66      
67      public TestSimilarity() {}
68      
69      @Override
70      public float tf(float freq) {
71        if (freq > 0.0f) return 1.0f;
72        else return 0.0f;
73      }
74      
75      @Override
76      public float lengthNorm(FieldInvertState state) {
77        // Disable length norm
78        return state.getBoost();
79      }
80      
81      @Override
82      public float idf(long docFreq, long numDocs) {
83        return 1.0f;
84      }
85    }

173   public void testSkipToFirsttimeMiss() throws IOException {
174     final DisjunctionMaxQuery dq = new DisjunctionMaxQuery(
175         Arrays.asList(tq("id", "d1"), tq("dek", "DOES_NOT_EXIST")), 0.0f);
176 
177     QueryUtils.check(random(), dq, s);
178     assertTrue(s.getTopReaderContext() instanceof LeafReaderContext);
179     final Weight dw = s.createNormalizedWeight(dq, true);
180     LeafReaderContext context = (LeafReaderContext)s.getTopReaderContext();
181     final Scorer ds = dw.scorer(context);
182     final boolean skipOk = ds.advance(3) != DocIdSetIterator.NO_MORE_DOCS;
183     if (skipOk) {
184       fail("firsttime skipTo found a match? ... "
185           + r.document(ds.docID()).get("id"));
186     }
187   }

415           + " >? " + score2, score1 > score2);

417       assertEquals("wrong fourth", "d1", doc3);
418       assertTrue("d1 does not have worse score then d3: " + score2 + " >? "

481     int hits = 0;
482     Directory directory = newDirectory();
483     Analyzer indexerAnalyzer = new MockAnalyzer(random());
484 
485     IndexWriterConfig config = new IndexWriterConfig(indexerAnalyzer);
486     IndexWriter writer = new IndexWriter(directory, config);

490     writer.addDocument(d);
491     writer.close();
492 
493     IndexReader indexReader = DirectoryReader.open(directory);
494     IndexSearcher searcher = newSearcher(indexReader);
495 
496     DisjunctionMaxQuery query = new DisjunctionMaxQuery(
497         Arrays.<Query>asList(
498             new SpanTermQuery(new Term(FIELD, "clockwork")),
499             new SpanTermQuery(new Term(FIELD, "clckwork"))),
500         1.0f);
501     TopScoreDocCollector collector = TopScoreDocCollector.create(1000);
502     searcher.search(query, collector);
503     hits = collector.topDocs().scoreDocs.length;
504     for (ScoreDoc scoreDoc : collector.topDocs().scoreDocs){
505       System.out.println(scoreDoc.doc);
506     }
507     indexReader.close();
508     assertEquals(hits, 1);
509     directory.close();

523   protected void printHits(String test, ScoreDoc[] h, IndexSearcher searcher)
524       throws Exception {
525     
526     System.err.println("------- " + test + " -------");
527     
528     DecimalFormat f = new DecimalFormat("0.000000000", DecimalFormatSymbols.getInstance(Locale.ROOT));
529     
530     for (int i = 0; i < h.length; i++) {
531       Document d = searcher.doc(h[i].doc);
532       float score = h[i].score;
533       System.err
534           .println("#" + i + ": " + f.format(score) + " - " + d.get("id"));
535     }
