76      listCollection();

108   void rebalanceLeaderTest() throws InterruptedException, IOException, SolrServerException, KeeperException {
109     recordInitialState();
110     for (int idx = 0; idx < reps; ++idx) {
111       issueCommands();
112       checkConsistency();
113     }
114   }

116   // After we've called the rebalance command, we want to insure that:
117   // 1> all replicas appear once and only once in the respective leader election queue
118   // 2> All the replicas we _think_ are leaders are in the 0th position in the leader election queue.
119   // 3> The node that ZooKeeper thinks is the leader is the one we think should be the leader.
120   void checkConsistency() throws InterruptedException, KeeperException {
121     TimeOut timeout = new TimeOut(timeoutMs, TimeUnit.MILLISECONDS);
122     boolean checkAppearOnce = false;
123     boolean checkElectionZero = false;
124     boolean checkZkLeadersAgree = false;
125     while (!timeout.hasTimedOut()) {
126       checkAppearOnce = checkAppearOnce();
127       checkElectionZero = checkElectionZero();
128       checkZkLeadersAgree = checkZkLeadersAgree();
129       if (checkAppearOnce && checkElectionZero && checkZkLeadersAgree) {
130         return;
131       }
132       Thread.sleep(1000);
133     }
134 
135     fail("Checking the rebalance leader command failed, checkAppearOnce=" + checkAppearOnce + " checkElectionZero="
136         + checkElectionZero + " checkZkLeadersAgree=" + checkZkLeadersAgree);
137   }

140   // Do all the nodes appear exactly once in the leader election queue and vice-versa?
141   Boolean checkAppearOnce() throws KeeperException, InterruptedException {
142 
143     for (Map.Entry<String, List<Replica>> ent : initial.entrySet()) {
144       List<String> leaderQueue = cloudClient.getZkStateReader().getZkClient().getChildren("/collections/" + COLLECTION_NAME +
145           "/leader_elect/" + ent.getKey() + "/election", null, true);
146 
147       if (leaderQueue.size() != ent.getValue().size()) {
148         return false;
149       }
150       // Check that each election node has a corresponding replica.
151       for (String electionNode : leaderQueue) {
152         if (checkReplicaName(LeaderElector.getNodeName(electionNode), ent.getValue())) {
153           continue;
154         }
155         return false;
156       }
157       // Check that each replica has an election node.
158       for (Replica rep : ent.getValue()) {
159         if (checkElectionNode(rep.getName(), leaderQueue)) {
160           continue;
161         }
162         return false;
163       }
164     }
165     return true;
166   }
167 
168   // Check that the given name is in the leader election queue
169   Boolean checkElectionNode(String repName, List<String> leaderQueue) {
170     for (String electionNode : leaderQueue) {
171       if (repName.equals(LeaderElector.getNodeName(electionNode))) {
172         return true;
173       }
174     }
175     return false;
176   }

178   // Check that the name passed in corresponds to a replica.
179   Boolean checkReplicaName(String toCheck, List<Replica> replicas) {
180     for (Replica rep : replicas) {
181       if (toCheck.equals(rep.getName())) {
182         return true;
183       }
184     }
185     return false;
186   }

188   // Get the shard leader election from ZK and sort it. The node may not actually be there, so retry
189   List<String> getOverseerSort(String key) {
190     List<String> ret = null;
191     try {
192       ret = OverseerCollectionConfigSetProcessor.getSortedElectionNodes(cloudClient.getZkStateReader().getZkClient(),
193           "/collections/" + COLLECTION_NAME + "/leader_elect/" + key + "/election");
194       return ret;
195     } catch (KeeperException e) {
196       cloudClient.connect();
197     } catch (InterruptedException e) {
198       return null;
199     }
200     return null;
201   }

203   // Is every node we think is the leader in the zeroth position in the leader election queue?
204   Boolean checkElectionZero() {
205     for (Map.Entry<String, Replica> ent : expected.entrySet()) {
206 
207       List<String> leaderQueue = getOverseerSort(ent.getKey());
208       if (leaderQueue == null) return false;
209 
210       String electName = LeaderElector.getNodeName(leaderQueue.get(0));
211       String coreName = ent.getValue().getName();
212       if (electName.equals(coreName) == false) {
213         return false;
214       }
215     }
216     return true;
217   }

219   // Do who we _think_ should be the leader agree with the leader nodes?
220   Boolean checkZkLeadersAgree() throws KeeperException, InterruptedException {
221     for (Map.Entry<String,Replica> ent : expected.entrySet()) {
222       
223       String path = "/collections/" + COLLECTION_NAME + "/leaders/" + ent.getKey() + "/leader";
224       byte[] data = getZkData(cloudClient, path);
225       if (data == null) {
226         log.warn("path to check not found {}", path);
227         return false;
228       }
229       
230       String repCore = null;
231       String zkCore = null;
232       
233       Map m = (Map) Utils.fromJSON(data);
234       zkCore = (String) m.get("core");
235       repCore = ent.getValue().getStr("core");
236       if (zkCore.equals(repCore) == false) {
237         log.warn("leader in zk does not match what we expect: {} != {}", zkCore, repCore);
238         return false;
239       }
240       
241     }
242     return true;
243   }

245   byte[] getZkData(CloudSolrClient client, String path) {
246     org.apache.zookeeper.data.Stat stat = new org.apache.zookeeper.data.Stat();
247     try {
248       byte[] data = client.getZkStateReader().getZkClient().getData(path, null, stat, true);
249       if (data != null) {
250         return data;
251       }
252     } catch (KeeperException.NoNodeException e) {
253       try {
254         Thread.sleep(1000);
255       } catch (InterruptedException e1) {
256         return null;
257       }
258     } catch (InterruptedException | KeeperException e) {
259       return null;
260     }
261     return null;
262   }

264   // It's OK not to check the return here since the subsequent tests will fail.
265   void issueCommands() throws IOException, SolrServerException, KeeperException, InterruptedException {
266 
267     // Find a replica to make the preferredLeader. NOTE: may be one that's _already_ leader!
268     expected.clear();
269     for (Map.Entry<String, List<Replica>> ent : initial.entrySet()) {
270       List<Replica> replicas = ent.getValue();
271       Replica rep = replicas.get(Math.abs(random().nextInt()) % replicas.size());
272       expected.put(ent.getKey(), rep);
273       issuePreferred(ent.getKey(), rep);
274     }
275 
276     if (waitForAllPreferreds() == false) {
277       fail("Waited for timeout for preferredLeader assignments to be made and they werent.");
278     }
279     //fillExpectedWithCurrent();
280     // Now rebalance the leaders
281     ModifiableSolrParams params = new ModifiableSolrParams();
282     params.set("action", CollectionParams.CollectionAction.REBALANCELEADERS.toString());
283 
284     // Insure we get error returns when omitting required parameters
285     params.set("collection", COLLECTION_NAME);
286     params.set("maxAtOnce", "10");
287     SolrRequest request = new QueryRequest(params);
288     request.setPath("/admin/collections");
289     cloudClient.request(request);
290   }

292   void issuePreferred(String slice, Replica rep) throws IOException, SolrServerException, InterruptedException {
293     ModifiableSolrParams params = new ModifiableSolrParams();
294     params.set("action", CollectionParams.CollectionAction.ADDREPLICAPROP.toString());
295 
296     // Insure we get error returns when omitting required parameters
297 
298     params.set("collection", COLLECTION_NAME);
299     params.set("shard", slice);
300     params.set("replica", rep.getName());
301     params.set("property", "preferredLeader");
302     params.set("property.value", "true");
303 
304     SolrRequest request = new QueryRequest(params);
305     request.setPath("/admin/collections");
306     cloudClient.request(request);
307   }

309   boolean waitForAllPreferreds() throws KeeperException, InterruptedException {
310     boolean goAgain = true;
311     TimeOut timeout = new TimeOut(timeoutMs, TimeUnit.MILLISECONDS);
312     while (! timeout.hasTimedOut()) {
313       goAgain = false;
314       cloudClient.getZkStateReader().updateClusterState();
315       Map<String, Slice> slices = cloudClient.getZkStateReader().getClusterState().getCollection(COLLECTION_NAME).getSlicesMap();
316 
317       for (Map.Entry<String, Replica> ent : expected.entrySet()) {
318         Replica me = slices.get(ent.getKey()).getReplica(ent.getValue().getName());
319         if (me.getBool("property.preferredleader", false) == false) {
320           goAgain = true;
321           break;
322         }
323       }
324       if (goAgain) {
325         Thread.sleep(250);
326       } else {
327         return true;
328       }
329     }
330     return false;
331   }
