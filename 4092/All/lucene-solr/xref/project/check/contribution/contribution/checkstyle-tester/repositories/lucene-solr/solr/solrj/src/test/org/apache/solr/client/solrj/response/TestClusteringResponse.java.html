19  import java.io.InputStream;
20  import java.io.InputStreamReader;
21  import java.io.Reader;
22  import java.nio.charset.StandardCharsets;
23  import java.util.LinkedList;
24  import java.util.List;
25  
26  import org.apache.solr.SolrJettyTestBase;
27  import org.apache.solr.client.solrj.impl.XMLResponseParser;
28  import org.apache.solr.common.util.NamedList;
29  import org.apache.solr.core.SolrResourceLoader;
30  import org.junit.Assert;
31  import org.junit.Test;
32  
33  /**
34   * Test for ClusteringComponent's response in Solrj
35   */
36  public class TestClusteringResponse extends SolrJettyTestBase {
37  
38    @Test
39    public void testClusteringResponse() throws Exception {
40      XMLResponseParser parser = new XMLResponseParser();
41      /*Load a simple XML with the clustering response encoded in an XML format*/
42      InputStream is = new SolrResourceLoader().openResource("solrj/sampleClusteringResponse.xml");
43      assertNotNull(is);
44      Reader in = new InputStreamReader(is, StandardCharsets.UTF_8);
45      NamedList<Object> response = parser.processResponse(in);
46      in.close();
47  
48      QueryResponse qr = new QueryResponse(response, null);
49      ClusteringResponse clusteringResponse = qr.getClusteringResponse();
50      List<Cluster> clusters = clusteringResponse.getClusters();
51      Assert.assertEquals(4, clusters.size());
52  
53      //First Cluster
54      Cluster cluster1 = clusters.get(0);
55      List<String> expectedLabel1 = new LinkedList<String>();
56      expectedLabel1.add("label1");
57      List<String> expectedDocs1 = new LinkedList<String>();
58      expectedDocs1.add("id1");
59      expectedDocs1.add("id2");
60      expectedDocs1.add("id3");
61      Assert.assertEquals(expectedLabel1, cluster1.getLabels());
62      Assert.assertEquals(expectedDocs1, cluster1.getDocs());
63      Assert.assertEquals(expectedLabel1, cluster1.getLabels());
64      Assert.assertEquals(0.6, cluster1.getScore(), 0);
65      //Second Cluster
66      Cluster cluster2 = clusters.get(1);
67      List<String> expectedLabel2 = new LinkedList<String>();
68      expectedLabel2.add("label2");
69      List<String> expectedDocs2 = new LinkedList<String>();
70      expectedDocs2.add("id5");
71      expectedDocs2.add("id6");
72      Assert.assertEquals(expectedLabel2, cluster2.getLabels());
73      Assert.assertEquals(expectedDocs2, cluster2.getDocs());
74      Assert.assertEquals(expectedLabel2, cluster2.getLabels());
75      Assert.assertEquals(0.93, cluster2.getScore(), 0);
76      //Third Cluster
77      Cluster cluster3 = clusters.get(2);
78      List<String> expectedLabel3 = new LinkedList<String>();
79      expectedLabel3.add("label3");
80      List<String> expectedDocs3 = new LinkedList<String>();
81      expectedDocs3.add("id7");
82      expectedDocs3.add("id8");
83      Assert.assertEquals(expectedLabel3, cluster3.getLabels());
84      Assert.assertEquals(expectedDocs3, cluster3.getDocs());
85      Assert.assertEquals(expectedLabel3, cluster3.getLabels());
86      Assert.assertEquals(1.26, cluster3.getScore(), 0);
87      //Fourth Cluster
88      Cluster cluster4 = clusters.get(3);
89      List<String> expectedLabel4 = new LinkedList<String>();
90      expectedLabel4.add("label4");
91      List<String> expectedDocs4 = new LinkedList<String>();
92      expectedDocs4.add("id9");
93      Assert.assertEquals(expectedLabel4, cluster4.getLabels());
94      Assert.assertEquals(expectedDocs4, cluster4.getDocs());
95      Assert.assertEquals(expectedLabel4, cluster4.getLabels());
96      Assert.assertEquals(0.0, cluster4.getScore(), 0);
97  
98    }
