48  /**
49   * Implementation of a Ternary Search Trie, a data structure for storing
50   * <code>String</code> objects that combines the compact size of a binary search
51   * tree with the speed of a digital search trie, and is therefore ideal for
52   * practical use in sorting and searching data.
53   * 
54   * <p>
55   * This data structure is faster than hashing for many typical search problems,
56   * and supports a broader range of useful problems and operations. Ternary
57   * searches are faster than hashing and more powerful, too.
58   * </p>
59   * 
60   * <p>
61   * The theory of ternary search trees was described at a symposium in 1997 (see
62   * "Fast Algorithms for Sorting and Searching Strings," by J.L. Bentley and R.
63   * Sedgewick, Proceedings of the 8th Annual ACM-SIAM Symposium on Discrete
64   * Algorithms, January 1997). Algorithms in C, Third Edition, by Robert
65   * Sedgewick (Addison-Wesley, 1998) provides yet another view of ternary search
66   * trees.
67   * </p>
68   *
69   * @deprecated Migrate to one of the newer suggesters which are much more RAM efficient.
70   */
71  @Deprecated
72  public class JaspellTernarySearchTrie implements Accountable {

221   /**
222    * Constructs a Ternary Search Trie and loads data from a <code>File</code>
223    * into the Trie. The file is a normal text document, where each line is of
224    * the form "word TAB float".
225    * 
226    *@param file
227    *          The <code>File</code> with the data to load into the Trie.
228    *@param compression
229    *          If true, the file is compressed with the GZIP algorithm, and if
230    *          false, the file is a normal text document.
231    *@exception IOException
232    *              A problem occured while reading the data.
233    */
234   public JaspellTernarySearchTrie(Path file, boolean compression)
235           throws IOException {
236     this();
237     BufferedReader in;
238     if (compression)
239       in = new BufferedReader(IOUtils.getDecodingReader(new GZIPInputStream(
240               Files.newInputStream(file)), StandardCharsets.UTF_8));
241     else in = Files.newBufferedReader(file, StandardCharsets.UTF_8);
242     String word;
243     int pos;
244     Float occur, one = new Float(1);
245     while ((word = in.readLine()) != null) {
246       pos = word.indexOf("\t");
247       occur = one;
248       if (pos != -1) {
249         occur = Float.parseFloat(word.substring(pos + 1).trim());
250         word = word.substring(0, pos);
251       }
252       String key = word.toLowerCase(locale);
253       if (rootNode == null) {
254         rootNode = new TSTNode(key.charAt(0), null);
255       }
256       TSTNode node = null;
257       if (key.length() > 0 && rootNode != null) {
258         TSTNode currentNode = rootNode;
259         int charIndex = 0;
260         while (true) {
261           if (currentNode == null) break;
262           int charComp = compareCharsAlphabetically(key.charAt(charIndex),
263                   currentNode.splitchar);
264           if (charComp == 0) {
265             charIndex++;
266             if (charIndex == key.length()) {
267               node = currentNode;
268               break;
269             }
270             currentNode = currentNode.relatives[TSTNode.EQKID];
271           } else if (charComp < 0) {
272             currentNode = currentNode.relatives[TSTNode.LOKID];
273           } else {
274             currentNode = currentNode.relatives[TSTNode.HIKID];
275           }
276         }
277         Float occur2 = null;
278         if (node != null) occur2 = ((Float) (node.data));
279         if (occur2 != null) {
280           occur += occur2.floatValue();
281         }
282         currentNode = getOrCreateNode(word.trim().toLowerCase(locale));
283         currentNode.data = occur;
284       }
285     }
286     in.close();
287   }

308   /**
309    * Recursively visits each node to be deleted.
310    * 
311    * To delete a node, first set its data to null, then pass it into this
312    * method, then pass the node returned by this method into this method (make
313    * sure you don't delete the data of any of the nodes returned from this
314    * method!) and continue in this fashion until the node returned by this
315    * method is <code>null</code>.
316    * 
317    * The TSTNode instance returned by this method will be next node to be
318    * operated on by <code>deleteNodeRecursion</code> (This emulates recursive
319    * method call while avoiding the JVM overhead normally associated with a
320    * recursive method.)
321    * 
322    *@param currentNode
323    *          The node to delete.
324    *@return The next node to be called in deleteNodeRecursion.
325    */
326   private TSTNode deleteNodeRecursion(TSTNode currentNode) {
327     if (currentNode == null) {
328       return null;
329     }
330     if (currentNode.relatives[TSTNode.EQKID] != null
331             || currentNode.data != null) {
332       return null;
333     }
334     // can't delete this node if it has a non-null eq kid or data
335     TSTNode currentParent = currentNode.relatives[TSTNode.PARENT];
336     boolean lokidNull = currentNode.relatives[TSTNode.LOKID] == null;
337     boolean hikidNull = currentNode.relatives[TSTNode.HIKID] == null;
338     int childType;
339     if (currentParent.relatives[TSTNode.LOKID] == currentNode) {
340       childType = TSTNode.LOKID;
341     } else if (currentParent.relatives[TSTNode.EQKID] == currentNode) {
342       childType = TSTNode.EQKID;
343     } else if (currentParent.relatives[TSTNode.HIKID] == currentNode) {
344       childType = TSTNode.HIKID;
345     } else {
346       rootNode = null;
347       return null;
348     }
349     if (lokidNull && hikidNull) {
350       currentParent.relatives[childType] = null;
351       return currentParent;
352     }
353     if (lokidNull) {
354       currentParent.relatives[childType] = currentNode.relatives[TSTNode.HIKID];
355       currentNode.relatives[TSTNode.HIKID].relatives[TSTNode.PARENT] = currentParent;
356       return currentParent;
357     }
358     if (hikidNull) {
359       currentParent.relatives[childType] = currentNode.relatives[TSTNode.LOKID];
360       currentNode.relatives[TSTNode.LOKID].relatives[TSTNode.PARENT] = currentParent;
361       return currentParent;
362     }
363     int deltaHi = currentNode.relatives[TSTNode.HIKID].splitchar
364             - currentNode.splitchar;
365     int deltaLo = currentNode.splitchar
366             - currentNode.relatives[TSTNode.LOKID].splitchar;
367     int movingKid;
368     TSTNode targetNode;
369     if (deltaHi == deltaLo) {
370       if (Math.random() < 0.5) {
371         deltaHi++;
372       } else {
373         deltaLo++;
374       }
375     }
376     if (deltaHi > deltaLo) {
377       movingKid = TSTNode.HIKID;
378       targetNode = currentNode.relatives[TSTNode.LOKID];
379     } else {
380       movingKid = TSTNode.LOKID;
381       targetNode = currentNode.relatives[TSTNode.HIKID];
382     }
383     while (targetNode.relatives[movingKid] != null) {
384       targetNode = targetNode.relatives[movingKid];
385     }
386     targetNode.relatives[movingKid] = currentNode.relatives[movingKid];
387     currentParent.relatives[childType] = targetNode;
388     targetNode.relatives[TSTNode.PARENT] = currentParent;
389     if (!lokidNull) {
390       currentNode.relatives[TSTNode.LOKID] = null;
391     }
392     if (!hikidNull) {
393       currentNode.relatives[TSTNode.HIKID] = null;
394     }
395     return currentParent;
396   }

414    * Retrieve the <code>Float</code> indexed by key, increment it by one unit
415    * and store the new <code>Float</code>.

422     String key2 = key.trim().toLowerCase(locale);
423     TSTNode node = getNode(key2);
424     if (node == null) {
425       return null;
426     }
427     Float aux = (Float) (node.data);
428     if (aux == null) {
429       aux = new Float(1);
430     } else {
431       aux = new Float(aux.intValue() + 1);
432     }
433     put(key2, aux);
434     return aux;

437   /**
438    * Returns the key that indexes the node argument.
439    * 
440    *@param node
441    *          The node whose index is to be calculated.
442    *@return The <code>String</code> that indexes the node argument.
443    */
444   protected String getKey(TSTNode node) {
445     StringBuilder getKeyBuffer = new StringBuilder();
446     getKeyBuffer.setLength(0);
447     getKeyBuffer.append("" + node.splitchar);
448     TSTNode currentNode;
449     TSTNode lastNode;
450     currentNode = node.relatives[TSTNode.PARENT];
451     lastNode = node;
452     while (currentNode != null) {
453       if (currentNode.relatives[TSTNode.EQKID] == lastNode) {
454         getKeyBuffer.append("" + currentNode.splitchar);
455       }
456       lastNode = currentNode;
457       currentNode = currentNode.relatives[TSTNode.PARENT];
458     }
459     getKeyBuffer.reverse();
460     return getKeyBuffer.toString();
461   }

476   /**
477    * Returns the node indexed by key, or <code>null</code> if that node doesn't
478    * exist. The search begins at root node.
479    * 
480    *@param key
481    *          A <code>String</code> that indexes the node that is returned.
482    *@param startNode
483    *          The top node defining the subtrie to be searched.
484    *@return The node object indexed by key. This object is an instance of an
485    *         inner class named <code>TernarySearchTrie.TSTNode</code>.
486    */
487   protected TSTNode getNode(CharSequence key, TSTNode startNode) {
488     if (key == null || startNode == null || key.length() == 0) {
489       return null;
490     }
491     TSTNode currentNode = startNode;
492     int charIndex = 0;
493     while (true) {
494       if (currentNode == null) {
495         return null;
496       }
497       int charComp = compareCharsAlphabetically(key.charAt(charIndex),
498               currentNode.splitchar);
499       if (charComp == 0) {
500         charIndex++;
501         if (charIndex == key.length()) {
502           return currentNode;
503         }
504         currentNode = currentNode.relatives[TSTNode.EQKID];
505       } else if (charComp < 0) {
506         currentNode = currentNode.relatives[TSTNode.LOKID];
507       } else {
508         currentNode = currentNode.relatives[TSTNode.HIKID];
509       }
510     }
511   }

513   /**
514    * Returns the node indexed by key, creating that node if it doesn't exist,
515    * and creating any required intermediate nodes if they don't exist.
516    * 
517    *@param key
518    *          A <code>String</code> that indexes the node that is returned.
519    *@return The node object indexed by key. This object is an instance of an
520    *         inner class named <code>TernarySearchTrie.TSTNode</code>.
521    *@exception NullPointerException
522    *              If the key is <code>null</code>.
523    *@exception IllegalArgumentException
524    *              If the key is an empty <code>String</code>.
525    */
526   protected TSTNode getOrCreateNode(CharSequence key) throws NullPointerException,
527           IllegalArgumentException {
528     if (key == null) {
529       throw new NullPointerException(
530               "attempt to get or create node with null key");
531     }
532     if (key.length() == 0) {
533       throw new IllegalArgumentException(
534               "attempt to get or create node with key of zero length");
535     }
536     if (rootNode == null) {
537       rootNode = new TSTNode(key.charAt(0), null);
538     }
539     TSTNode currentNode = rootNode;
540     int charIndex = 0;
541     while (true) {
542       int charComp = compareCharsAlphabetically(key.charAt(charIndex),
543               currentNode.splitchar);
544       if (charComp == 0) {
545         charIndex++;
546         if (charIndex == key.length()) {
547           return currentNode;
548         }
549         if (currentNode.relatives[TSTNode.EQKID] == null) {
550           currentNode.relatives[TSTNode.EQKID] = new TSTNode(key
551                   .charAt(charIndex), currentNode);
552         }
553         currentNode = currentNode.relatives[TSTNode.EQKID];
554       } else if (charComp < 0) {
555         if (currentNode.relatives[TSTNode.LOKID] == null) {
556           currentNode.relatives[TSTNode.LOKID] = new TSTNode(key
557                   .charAt(charIndex), currentNode);
558         }
559         currentNode = currentNode.relatives[TSTNode.LOKID];
560       } else {
561         if (currentNode.relatives[TSTNode.HIKID] == null) {
562           currentNode.relatives[TSTNode.HIKID] = new TSTNode(key
563                   .charAt(charIndex), currentNode);
564         }
565         currentNode = currentNode.relatives[TSTNode.HIKID];
566       }
567     }
568   }

609   /**
610    * Recursivelly vists the nodes in order to find the ones that almost match a
611    * given key.
612    * 
613    *@param currentNode
614    *          The current node.
615    *@param charIndex
616    *          The current char.
617    *@param d
618    *          The number of differences so far.
619    *@param matchAlmostNumReturnValues
620    *          The maximum number of values in the result <code>List</code>.
621    *@param matchAlmostResult2
622    *          The results so far.
623    *@param upTo
624    *          If true all keys having up to and including matchAlmostDiff
625    *          mismatched letters will be included in the result (including a key
626    *          that is exactly the same as the target string) otherwise keys will
627    *          be included in the result only if they have exactly
628    *          matchAlmostDiff number of mismatched letters.
629    *@param matchAlmostKey
630    *          The key being searched.
631    *@return A <code>List</code> with the results.
632    */
633   private List<String> matchAlmostRecursion(TSTNode currentNode, int charIndex,
634           int d, CharSequence matchAlmostKey, int matchAlmostNumReturnValues,
635           List<String> matchAlmostResult2, boolean upTo) {
636     if ((currentNode == null)
637             || (matchAlmostNumReturnValues != -1 && matchAlmostResult2.size() >= matchAlmostNumReturnValues)
638             || (d < 0) || (charIndex >= matchAlmostKey.length())) {
639       return matchAlmostResult2;
640     }
641     int charComp = compareCharsAlphabetically(matchAlmostKey.charAt(charIndex),
642             currentNode.splitchar);
643     List<String> matchAlmostResult = matchAlmostResult2;
644     if ((d > 0) || (charComp < 0)) {
645       matchAlmostResult = matchAlmostRecursion(
646               currentNode.relatives[TSTNode.LOKID], charIndex, d,
647               matchAlmostKey, matchAlmostNumReturnValues, matchAlmostResult,
648               upTo);
649     }
650     int nextD = (charComp == 0) ? d : d - 1;
651     boolean cond = (upTo) ? (nextD >= 0) : (nextD == 0);
652     if ((matchAlmostKey.length() == charIndex + 1) && cond
653             && (currentNode.data != null)) {
654       matchAlmostResult.add(getKey(currentNode));
655     }
656     matchAlmostResult = matchAlmostRecursion(
657             currentNode.relatives[TSTNode.EQKID], charIndex + 1, nextD,
658             matchAlmostKey, matchAlmostNumReturnValues, matchAlmostResult, upTo);
659     if ((d > 0) || (charComp > 0)) {
660       matchAlmostResult = matchAlmostRecursion(
661               currentNode.relatives[TSTNode.HIKID], charIndex, d,
662               matchAlmostKey, matchAlmostNumReturnValues, matchAlmostResult,
663               upTo);
664     }
665     return matchAlmostResult;
666   }

683    * Returns an alphabetical <code>List</code> of all keys in the trie that

686    * 
687    *@param prefix
688    *          Each key returned from this method will begin with the characters
689    *          in prefix.
690    *@param numReturnValues
691    *          The maximum number of values returned from this method.
692    *@return A <code>List</code> with the results
693    */
694   public List<String> matchPrefix(CharSequence prefix, int numReturnValues) {
695     Vector<String> sortKeysResult = new Vector<>();
696     TSTNode startNode = getNode(prefix);
697     if (startNode == null) {
698       return sortKeysResult;
699     }
700     if (startNode.data != null) {
701       sortKeysResult.addElement(getKey(startNode));
702     }
703     return sortKeysRecursion(startNode.relatives[TSTNode.EQKID],
704             ((numReturnValues < 0) ? -1 : numReturnValues), sortKeysResult);
705   }

762   /**
763    * Recursivelly visists each node to calculate the number of nodes.
764    * 
765    *@param currentNode
766    *          The current node.
767    *@param checkData
768    *          If true we check the data to be different of <code>null</code>.
769    *@param numNodes2
770    *          The number of nodes so far.
771    *@return The number of nodes accounted.
772    */
773   private int recursiveNodeCalculator(TSTNode currentNode, boolean checkData,
774           int numNodes2) {
775     if (currentNode == null) {
776       return numNodes2;
777     }
778     int numNodes = recursiveNodeCalculator(
779             currentNode.relatives[TSTNode.LOKID], checkData, numNodes2);
780     numNodes = recursiveNodeCalculator(currentNode.relatives[TSTNode.EQKID],
781             checkData, numNodes);
782     numNodes = recursiveNodeCalculator(currentNode.relatives[TSTNode.HIKID],
783             checkData, numNodes);
784     if (checkData) {
785       if (currentNode.data != null) {
786         numNodes++;
787       }
788     } else {
789       numNodes++;
790     }
791     return numNodes;
792   }

844   /**
845    * Returns keys sorted in alphabetical order. This includes the start Node and

847    * <p>

859             : numReturnValues), new Vector<String>());

862   /**
863    * Returns keys sorted in alphabetical order. This includes the current Node
864    * and all nodes connected to the current Node.
865    * <p>
866    * Sorted keys will be appended to the end of the resulting <code>List</code>.
867    * The result may be empty when this method is invoked, but may not be
868    * <code>null</code>.
869    * 
870    *@param currentNode
871    *          The current node.
872    *@param sortKeysNumReturnValues
873    *          The maximum number of values in the result.
874    *@param sortKeysResult2
875    *          The results so far.
876    *@return A <code>List</code> with the results.
877    */
878   private List<String> sortKeysRecursion(TSTNode currentNode,
879           int sortKeysNumReturnValues, List<String> sortKeysResult2) {
880     if (currentNode == null) {
881       return sortKeysResult2;
882     }
883     List<String> sortKeysResult = sortKeysRecursion(
884             currentNode.relatives[TSTNode.LOKID], sortKeysNumReturnValues,
885             sortKeysResult2);
886     if (sortKeysNumReturnValues != -1
887             && sortKeysResult.size() >= sortKeysNumReturnValues) {
888       return sortKeysResult;
889     }
890     if (currentNode.data != null) {
891       sortKeysResult.add(getKey(currentNode));
892     }
893     sortKeysResult = sortKeysRecursion(currentNode.relatives[TSTNode.EQKID],
894             sortKeysNumReturnValues, sortKeysResult);
895     return sortKeysRecursion(currentNode.relatives[TSTNode.HIKID],
896             sortKeysNumReturnValues, sortKeysResult);
897   }
