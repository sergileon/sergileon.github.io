69  /**
70   * This strategy is used with the {@link Builder} AST transform to create a builder helper class
71   * for the fluent and type-safe creation of instances of a specified class.
72   *
73   * It is modelled roughly on the design outlined here:
74   * http://michid.wordpress.com/2008/08/13/type-safe-builder-pattern-in-java/
75   *
76   * You define classes which use the type-safe initializer pattern as follows:
77   * <pre>
78   * import groovy.transform.builder.*
79   * import groovy.transform.*
80   *
81   * {@code @ToString}
82   * {@code @Builder}(builderStrategy=InitializerStrategy) class Person {
83   *     String firstName
84   *     String lastName
85   *     int age
86   * }
87   * </pre>
88   * While it isn't required to do so, the benefit of this builder strategy comes in conjunction with static type-checking or static compilation. Typical usage is as follows:
89   * <pre>
90   * {@code @CompileStatic}
91   * def main() {
92   *     println new Person(Person.createInitializer().firstName("John").lastName("Smith").age(21))
93   * }
94   * </pre>
95   * which prints:
96   * <pre>
97   * Person(John, Smith, 21)
98   * </pre>
99   * If you don't initialise some of the properties, your code won't compile, e.g. if the method body above was changed to this:
100  * <pre>
101  * println new Person(Person.createInitializer().firstName("John").lastName("Smith"))
102  * </pre>
103  * then the following compile-time error would result:
104  * <pre>
105  * [Static type checking] - Cannot find matching method Person#<init>(Person$PersonInitializer <groovy.transform.builder.InitializerStrategy$SET, groovy.transform.builder.InitializerStrategy$SET, groovy.transform.builder.InitializerStrategy$UNSET>). Please check if the declared type is right and if the method exists.
106  * </pre>
107  * The message is a little cryptic, but it is basically the static compiler telling us that the third parameter, {@code age} in our case, is unset.
108  *
109  * You can also add this annotation to your predefined constructors. These will be made private and an initializer will be set up
110  * to call your constructor. Any parameters to your constructor become the properties expected by the initializer.
111  * If you use such a builder on a constructor as well as on the class or on more than one constructor, then it is up to you
112  * to define unique values for 'builderClassName' and 'builderMethodName' for each annotation.
113  */
114 public class InitializerStrategy extends BuilderASTTransformation.AbstractBuilderStrategy {
115 
116     /**
117      * Internal phantom type used by the {@code InitializerStrategy} to indicate that a property has been set. It is used in conjunction with the generated parameterized type helper class.
118      */
119     public abstract static class SET {
120     }

122     /**
123      * Internal phantom type used by the {@code InitializerStrategy} to indicate that a property remains unset. It is used in conjunction with the generated parameterized type helper class.
124      */
125     public abstract static class UNSET {
126     }

128     private static final int PUBLIC_STATIC = ACC_PUBLIC | ACC_STATIC;
129     private static final Expression DEFAULT_INITIAL_VALUE = null;
130 
131     public void build(BuilderASTTransformation transform, AnnotatedNode annotatedNode, AnnotationNode anno) {
132         if (unsupportedAttribute(transform, anno, "forClass")) return;
133         if (annotatedNode instanceof ClassNode) {
134             createBuilderForAnnotatedClass(transform, (ClassNode) annotatedNode, anno);
135         } else if (annotatedNode instanceof MethodNode) {
136             createBuilderForAnnotatedMethod(transform, (MethodNode) annotatedNode, anno);
137         }
138     }
139 
140     private void createBuilderForAnnotatedClass(BuilderASTTransformation transform, ClassNode buildee, AnnotationNode anno) {
141         List<String> excludes = new ArrayList<String>();
142         List<String> includes = new ArrayList<String>();
143         if (!getIncludeExclude(transform, anno, buildee, excludes, includes)) return;
144         List<FieldNode> fields = getInstancePropertyFields(buildee);
145         List<FieldNode> filteredFields = filterFields(fields, includes, excludes);
146         if (filteredFields.isEmpty()) {
147             transform.addError("Error during " + BuilderASTTransformation.MY_TYPE_NAME +
148                     " processing: at least one property is required for this strategy", anno);
149         }
150         ClassNode builder = createInnerHelperClass(buildee, getBuilderClassName(buildee, anno), filteredFields.size());
151         addFields(buildee, filteredFields, builder);
152 
153         buildCommon(buildee, anno, filteredFields, builder);
154         createBuildeeConstructors(transform, buildee, builder, filteredFields, true);
155     }
156 
157     private void createBuilderForAnnotatedMethod(BuilderASTTransformation transform, MethodNode mNode, AnnotationNode anno) {
158         if (transform.getMemberValue(anno, "includes") != null || transform.getMemberValue(anno, "includes") != null) {
159             transform.addError("Error during " + BuilderASTTransformation.MY_TYPE_NAME +
160                     " processing: includes/excludes only allowed on classes", anno);
161         }
162         if (mNode instanceof ConstructorNode) {
163             mNode.setModifiers(ACC_PRIVATE | ACC_SYNTHETIC);
164         } else {
165             if ((mNode.getModifiers() & ACC_STATIC) == 0) {
166                 transform.addError("Error during " + BuilderASTTransformation.MY_TYPE_NAME +
167                         " processing: method builders only allowed on static methods", anno);
168             }
169             mNode.setModifiers(ACC_PRIVATE | ACC_SYNTHETIC | ACC_STATIC);
170         }
171         ClassNode buildee = mNode.getDeclaringClass();
172         Parameter[] parameters = mNode.getParameters();
173         if (parameters.length == 0) {
174             transform.addError("Error during " + BuilderASTTransformation.MY_TYPE_NAME +
175                     " processing: at least one parameter is required for this strategy", anno);
176         }
177         ClassNode builder = createInnerHelperClass(buildee, getBuilderClassName(buildee, anno), parameters.length);
178         List<FieldNode> convertedFields = convertParamsToFields(builder, parameters);
179 
180         buildCommon(buildee, anno, convertedFields, builder);
181         if (mNode instanceof ConstructorNode) {
182             createBuildeeConstructors(transform, buildee, builder, convertedFields, false);
183         } else {
184             createBuildeeMethods(buildee, mNode, builder, convertedFields);
185         }
186     }
187 
188     private String getBuilderClassName(ClassNode buildee, AnnotationNode anno) {
189         return getMemberStringValue(anno, "builderClassName", buildee.getNameWithoutPackage() + "Initializer");
190     }
191 
192     private void addFields(ClassNode buildee, List<FieldNode> filteredFields, ClassNode builder) {
193         for (FieldNode filteredField : filteredFields) {
194             builder.addField(createFieldCopy(buildee, filteredField));
195         }
196     }
197 
198     private void buildCommon(ClassNode buildee, AnnotationNode anno, List<FieldNode> fieldNodes, ClassNode builder) {
199         String prefix = getMemberStringValue(anno, "prefix", "");
200         String buildMethodName = getMemberStringValue(anno, "buildMethodName", "create");
201         createBuilderConstructors(builder, buildee, fieldNodes);
202         buildee.getModule().addClass(builder);
203         String builderMethodName = getMemberStringValue(anno, "builderMethodName", "createInitializer");
204         buildee.addMethod(createBuilderMethod(buildMethodName, builder, fieldNodes.size(), builderMethodName));
205         for (int i = 0; i < fieldNodes.size(); i++) {
206             builder.addMethod(createBuilderMethodForField(builder, fieldNodes, prefix, i));
207         }
208         builder.addMethod(createBuildMethod(builder, buildMethodName, fieldNodes));
209     }
210 
211     private List<FieldNode> convertParamsToFields(ClassNode builder, Parameter[] parameters) {
212         List<FieldNode> fieldNodes = new ArrayList<FieldNode>();
213         for(Parameter parameter: parameters) {
214             Map<String,ClassNode> genericsSpec = createGenericsSpec(builder);
215             ClassNode correctedType = correctToGenericsSpecRecurse(genericsSpec, parameter.getType());
216             FieldNode fieldNode = new FieldNode(parameter.getName(), parameter.getModifiers(), correctedType, builder, DEFAULT_INITIAL_VALUE);
217             fieldNodes.add(fieldNode);
218             builder.addField(fieldNode);
219         }
220         return fieldNodes;
221     }
222 
223     private ClassNode createInnerHelperClass(ClassNode buildee, String builderClassName, int fieldsSize) {
224         final String fullName = buildee.getName() + "$" + builderClassName;
225         ClassNode builder = new InnerClassNode(buildee, fullName, PUBLIC_STATIC, OBJECT_TYPE);
226         GenericsType[] gtypes = new GenericsType[fieldsSize];
227         for (int i = 0; i < gtypes.length; i++) {
228             gtypes[i] = makePlaceholder(i);
229         }
230         builder.setGenericsTypes(gtypes);
231         return builder;
232     }
233 
234     private static MethodNode createBuilderMethod(String buildMethodName, ClassNode builder, int numFields, String builderMethodName) {
235         final BlockStatement body = new BlockStatement();
236         body.addStatement(returnS(callX(builder, buildMethodName)));
237         ClassNode returnType = makeClassSafeWithGenerics(builder, unsetGenTypes(numFields));
238         return new MethodNode(builderMethodName, PUBLIC_STATIC, returnType, NO_PARAMS, NO_EXCEPTIONS, body);
239     }
240 
241     private static GenericsType[] unsetGenTypes(int numFields) {
242         GenericsType[] gtypes = new GenericsType[numFields];
243         for (int i = 0; i < gtypes.length; i++) {
244             gtypes[i] = new GenericsType(ClassHelper.make(UNSET.class));
245         }
246         return gtypes;
247     }
248 
249     private static GenericsType[] setGenTypes(int numFields) {
250         GenericsType[] gtypes = new GenericsType[numFields];
251         for (int i = 0; i < gtypes.length; i++) {
252             gtypes[i] = new GenericsType(ClassHelper.make(SET.class));
253         }
254         return gtypes;
255     }
256 
257     private static void createBuilderConstructors(ClassNode builder, ClassNode buildee, List<FieldNode> fields) {
258         builder.addConstructor(ACC_PRIVATE, NO_PARAMS, NO_EXCEPTIONS, block(ctorSuperS()));
259         final BlockStatement body = new BlockStatement();
260         body.addStatement(ctorSuperS());
261         initializeFields(fields, body);
262         builder.addConstructor(ACC_PRIVATE, getParams(fields, buildee), NO_EXCEPTIONS, body);
263     }
264 
265     private static void createBuildeeConstructors(BuilderASTTransformation transform, ClassNode buildee, ClassNode builder, List<FieldNode> fields, boolean needsConstructor) {
266         ConstructorNode initializer = createInitializerConstructor(buildee, builder, fields);
267         if (transform.hasAnnotation(buildee, ImmutableASTTransformation.MY_TYPE)) {
268             initializer.putNodeMetaData(ImmutableASTTransformation.IMMUTABLE_SAFE_FLAG, Boolean.TRUE);
269         } else if (needsConstructor) {
270             final BlockStatement body = new BlockStatement();
271             body.addStatement(ctorSuperS());
272             initializeFields(fields, body);
273             buildee.addConstructor(ACC_PRIVATE | ACC_SYNTHETIC, getParams(fields, buildee), NO_EXCEPTIONS, body);
274         }
275     }
276 
277     private static void createBuildeeMethods(ClassNode buildee, MethodNode mNode, ClassNode builder, List<FieldNode> fields) {
278         ClassNode paramType = makeClassSafeWithGenerics(builder, setGenTypes(fields.size()));
279         List<Expression> argsList = new ArrayList<Expression>();
280         Parameter initParam = param(paramType, "initializer");
281         for (FieldNode fieldNode : fields) {
282             argsList.add(propX(varX(initParam), fieldNode.getName()));
283         }
284         String newName = "$" + mNode.getName(); // can't have private and public methods of the same name, so rename original
285         buildee.addMethod(mNode.getName(), PUBLIC_STATIC, mNode.getReturnType(), params(param(paramType, "initializer")), NO_EXCEPTIONS,
286                 block(stmt(callX(buildee, newName, args(argsList)))));
287         renameMethod(buildee, mNode, newName);
288     }
289 
290     // no rename so delete and add
291     private static void renameMethod(ClassNode buildee, MethodNode mNode, String newName) {
292         buildee.addMethod(newName, mNode.getModifiers(), mNode.getReturnType(), mNode.getParameters(), mNode.getExceptions(), mNode.getCode());
293         buildee.removeMethod(mNode);
294     }
295 
296     private static Parameter[] getParams(List<FieldNode> fields, ClassNode cNode) {
297         Parameter[] parameters = new Parameter[fields.size()];
298         for (int i = 0; i < parameters.length; i++) {
299             FieldNode fNode = fields.get(i);
300             Map<String,ClassNode> genericsSpec = createGenericsSpec(fNode.getDeclaringClass());
301             extractSuperClassGenerics(fNode.getType(), cNode, genericsSpec);
302             ClassNode correctedType = correctToGenericsSpecRecurse(genericsSpec, fNode.getType());
303             parameters[i] = new Parameter(correctedType, fNode.getName());
304         }
305         return parameters;
306     }
307 
308     private static ConstructorNode createInitializerConstructor(ClassNode buildee, ClassNode builder, List<FieldNode> fields) {
309         ClassNode paramType = makeClassSafeWithGenerics(builder, setGenTypes(fields.size()));
310         List<Expression> argsList = new ArrayList<Expression>();
311         Parameter initParam = param(paramType, "initializer");
312         for (FieldNode fieldNode : fields) {
313             argsList.add(propX(varX(initParam), fieldNode.getName()));
314         }
315         return buildee.addConstructor(ACC_PUBLIC, params(param(paramType, "initializer")), NO_EXCEPTIONS, block(ctorThisS(args(argsList))));
316     }
317 
318     private static MethodNode createBuildMethod(ClassNode builder, String buildMethodName, List<FieldNode> fields) {
319         ClassNode returnType = makeClassSafeWithGenerics(builder, unsetGenTypes(fields.size()));
320         return new MethodNode(buildMethodName, PUBLIC_STATIC, returnType, NO_PARAMS, NO_EXCEPTIONS, block(returnS(ctorX(returnType))));
321     }
322 
323     private MethodNode createBuilderMethodForField(ClassNode builder, List<FieldNode> fields, String prefix, int fieldPos) {
324         String fieldName = fields.get(fieldPos).getName();
325         String setterName = getSetterName(prefix, fieldName);
326         GenericsType[] gtypes = new GenericsType[fields.size()];
327         List<Expression> argList = new ArrayList<Expression>();
328         for (int i = 0; i < fields.size(); i++) {
329             gtypes[i] = i == fieldPos ? new GenericsType(ClassHelper.make(SET.class)) : makePlaceholder(i);
330             argList.add(i == fieldPos ? propX(varX("this"), constX(fieldName)) : varX(fields.get(i).getName()));
331         }
332         ClassNode returnType = makeClassSafeWithGenerics(builder, gtypes);
333         FieldNode fNode = fields.get(fieldPos);
334         Map<String,ClassNode> genericsSpec = createGenericsSpec(fNode.getDeclaringClass());
335         extractSuperClassGenerics(fNode.getType(), builder, genericsSpec);
336         ClassNode correctedType = correctToGenericsSpecRecurse(genericsSpec, fNode.getType());
337         return new MethodNode(setterName, ACC_PUBLIC, returnType, params(param(correctedType, fieldName)), NO_EXCEPTIONS, block(
338                 stmt(assignX(propX(varX("this"), constX(fieldName)), varX(fieldName, correctedType))),
339                 returnS(ctorX(returnType, args(argList)))
340         ));
341     }
342 
343     private GenericsType makePlaceholder(int i) {
344         ClassNode type = ClassHelper.makeWithoutCaching("T" + i);
345         type.setRedirect(OBJECT_TYPE);
346         type.setGenericsPlaceHolder(true);
347         return new GenericsType(type);
348     }
349 
350     private static FieldNode createFieldCopy(ClassNode buildee, FieldNode fNode) {
351         Map<String,ClassNode> genericsSpec = createGenericsSpec(fNode.getDeclaringClass());
352         extractSuperClassGenerics(fNode.getType(), buildee, genericsSpec);
353         ClassNode correctedType = correctToGenericsSpecRecurse(genericsSpec, fNode.getType());
354         return new FieldNode(fNode.getName(), fNode.getModifiers(), correctedType, buildee, DEFAULT_INITIAL_VALUE);
355     }
356 
357     private static List<FieldNode> filterFields(List<FieldNode> fieldNodes, List<String> includes, List<String> excludes) {
358         List<FieldNode> fields = new ArrayList<FieldNode>();
359         for (FieldNode fNode : fieldNodes) {
360             if (AbstractASTTransformation.shouldSkip(fNode.getName(), excludes, includes)) continue;
361             fields.add(fNode);
362         }
363         return fields;
364     }
365 
366     private static void initializeFields(List<FieldNode> fields, BlockStatement body) {
367         for (FieldNode field : fields) {
368             body.addStatement(stmt(assignX(propX(varX("this"), field.getName()), varX(param(field.getType(), field.getName())))));
369         }
370     }
