48      public GenericsType(ClassNode type, ClassNode[] upperBounds, ClassNode lowerBound) {
49          this.type = type;
50          this.name = type.isGenericsPlaceHolder() ? type.getUnresolvedName() : type.getName();
51          this.upperBounds = upperBounds;
52          this.lowerBound = lowerBound;
53          placeholder = type.isGenericsPlaceHolder();
54          resolved = false;
55      }
56  
57      public GenericsType(ClassNode basicType) {
58          this(basicType, null, null);
59      }

62          return type;
63      }

66          this.type = type;
67      }

69      public String toString() {
70          Set<String> visited = new HashSet<String>();
71          return toString(visited);
72      }
73  
74      private String toString(Set<String> visited) {
75          if (placeholder) visited.add(name);
76          String ret = wildcard?"?":((type == null || placeholder) ? name : genericsBounds(type, visited));
77          if (upperBounds != null) {
78              if (placeholder && upperBounds.length==1 && !upperBounds[0].isGenericsPlaceHolder() && upperBounds[0].getName().equals("java.lang.Object")) {
79                  // T extends Object should just be printed as T
80              } else {
81                  ret += " extends ";
82                  for (int i = 0; i < upperBounds.length; i++) {
83                      ret += genericsBounds(upperBounds[i], visited);
84                      if (i + 1 < upperBounds.length) ret += " & ";
85                  }
86              }
87          } else if (lowerBound != null) {
88              ret += " super " + genericsBounds(lowerBound, visited);
89          }
90          return ret;
91      }

93      private String nameOf(ClassNode theType) {
94          StringBuilder ret = new StringBuilder();
95          if (theType.isArray()) {
96              ret.append(nameOf(theType.getComponentType()));
97              ret.append("[]");
98          } else {
99              ret.append(theType.getName());
100         }
101         return ret.toString();
102     }
103 
104     private String genericsBounds(ClassNode theType, Set<String> visited) {
105 
106         StringBuilder ret = new StringBuilder();
107 
108         if (theType.isArray()) {
109             ret.append(nameOf(theType));
110         } else if (theType.redirect() instanceof InnerClassNode) {
111             InnerClassNode innerClassNode = (InnerClassNode) theType.redirect();
112             String parentClassNodeName = innerClassNode.getOuterClass().getName();
113             if (Modifier.isStatic(innerClassNode.getModifiers()) || innerClassNode.isInterface()) {
114                 ret.append(innerClassNode.getOuterClass().getName());
115             } else {
116                 ret.append(genericsBounds(innerClassNode.getOuterClass(), new HashSet<String>()));
117             }
118             ret.append(".");
119             String typeName = theType.getName();
120             ret.append(typeName.substring(parentClassNodeName.length() + 1));
121         } else {
122             ret.append(theType.getName());
123         }
124 
125         GenericsType[] genericsTypes = theType.getGenericsTypes();
126         if (genericsTypes == null || genericsTypes.length == 0)
127             return ret.toString();
128 
129         // TODO instead of catching Object<T> here stop it from being placed into type in first place
130         if (genericsTypes.length == 1 && genericsTypes[0].isPlaceholder() && theType.getName().equals("java.lang.Object")) {
131             return genericsTypes[0].getName();
132         }
133 
134         ret.append("<");
135         for (int i = 0; i < genericsTypes.length; i++) {
136             if (i != 0) ret.append(", ");
137 
138             GenericsType type = genericsTypes[i];
139             if (type.isPlaceholder() && visited.contains(type.getName())) {
140                 ret.append(type.getName());
141             }
142             else {
143                 ret.append(type.toString(visited));
144             }
145         }
146         ret.append(">");
147 
148         return ret.toString();
149     }

192     /**
193      * Tells if the provided class node is compatible with this generic type definition
194      * @param classNode the class node to be checked
195      * @return true if the class node is compatible with this generics type definition
196      */
197     public boolean isCompatibleWith(ClassNode classNode) {
198         return new GenericsTypeMatcher().matches(classNode);
199     }

201     /**
202      * Implements generics type comparison.
203      */
204     private class GenericsTypeMatcher {
205 
206         public boolean implementsInterfaceOrIsSubclassOf(ClassNode type, ClassNode superOrInterface) {
207             boolean result = type.equals(superOrInterface)
208                     || type.isDerivedFrom(superOrInterface)
209                     || type.implementsInterface(superOrInterface);
210             if (result) {
211                 return true;
212             }
213             if (GROOVY_OBJECT_TYPE.equals(superOrInterface) && type.getCompileUnit()!=null) {
214                 // type is being compiled so it will implement GroovyObject later
215                 return true;
216             }
217             if (superOrInterface instanceof WideningCategories.LowestUpperBoundClassNode) {
218                 WideningCategories.LowestUpperBoundClassNode cn = (WideningCategories.LowestUpperBoundClassNode) superOrInterface;
219                 result = implementsInterfaceOrIsSubclassOf(type, cn.getSuperClass());
220                 if (result) {
221                     for (ClassNode interfaceNode : cn.getInterfaces()) {
222                         result = implementsInterfaceOrIsSubclassOf(type,interfaceNode);
223                         if (!result) break;
224                     }
225                 }
226                 if (result) return true;
227             }
228             if (type.isArray() && superOrInterface.isArray()) {
229                 return implementsInterfaceOrIsSubclassOf(type.getComponentType(), superOrInterface.getComponentType());
230             }
231             return false;
232         }

234         /**
235          * Compares this generics type with the one represented by the provided class node. If the provided
236          * classnode is compatible with the generics specification, returns true. Otherwise, returns false.
237          * The check is complete, meaning that we also check "nested" generics.
238          * @param classNode the classnode to be checked
239          * @return true iff the classnode is compatible with this generics specification
240          */
241         public boolean matches(ClassNode classNode) {
242             GenericsType[] genericsTypes = classNode.getGenericsTypes();
243             // diamond always matches
244             if (genericsTypes!=null && genericsTypes.length==0) return true;
245             if (classNode.isGenericsPlaceHolder()) {
246                 // if the classnode we compare to is a generics placeholder (like <E>) then we
247                 // only need to check that the names are equal
248                 if (genericsTypes==null) return true;
249                 if (isWildcard()) {
250                     if (lowerBound!=null) return genericsTypes[0].getName().equals(lowerBound.getUnresolvedName());
251                     if (upperBounds!=null) {
252                         for (ClassNode upperBound : upperBounds) {
253                             String name = upperBound.getGenericsTypes()[0].getName();
254                             if (genericsTypes[0].getName().equals(name)) return true;
255                         }
256                         return false;
257                     }
258                 }
259                 return genericsTypes[0].getName().equals(name);
260             }
261             if (wildcard || placeholder) {
262                 // if the current generics spec is a wildcard spec or a placeholder spec
263                 // then we must check upper and lower bounds
264                 if (upperBounds != null) {
265                     // check that the provided classnode is a subclass of all provided upper bounds
266                     boolean upIsOk = true;
267                     for (int i = 0, upperBoundsLength = upperBounds.length; i < upperBoundsLength && upIsOk; i++) {
268                         final ClassNode upperBound = upperBounds[i];
269                         upIsOk = implementsInterfaceOrIsSubclassOf(classNode, upperBound);
270                     }
271                     // if the provided classnode is a subclass of the upper bound
272                     // then check that the generic types supplied by the class node are compatible with
273                     // this generics specification
274                     // for example, we could have the spec saying List<String> but provided classnode
275                     // saying List<Integer>
276                     upIsOk = upIsOk && checkGenerics(classNode);
277                     return upIsOk;
278                 }
279                 if (lowerBound != null) {
280                     // if a lower bound is declared, then we must perform the same checks that for an upper bound
281                     // but with reversed arguments
282                     return implementsInterfaceOrIsSubclassOf(lowerBound, classNode) && checkGenerics(classNode);
283                 }
284             }
285             // if this is not a generics placeholder, first compare that types represent the same type
286             if ((type!=null && !type.equals(classNode))) {
287                 return false;
288             }
289             // last, we could have the spec saying List<String> and a classnode saying List<Integer> so
290             // we must check that generics are compatible.
291             // The null check is normally not required but done to prevent from NPEs
292             return type == null || compareGenericsWithBound(classNode, type);
293         }
294 
295         /**
296          * Iterates over each generics bound of this generics specification, and checks
297          * that the generics defined by the bound are compatible with the generics specified
298          * by the type.
299          * @param classNode the classnode the bounds should be compared with
300          * @return true if generics from bounds are compatible
301          */
302         private boolean checkGenerics(final ClassNode classNode) {
303             if (upperBounds!=null) {
304                 for (ClassNode upperBound : upperBounds) {
305                     if (!compareGenericsWithBound(classNode, upperBound)) return false;
306                 }
307             }
308             if (lowerBound!=null) {
309                 if (!lowerBound.redirect().isUsingGenerics()) {
310                     if (!compareGenericsWithBound(classNode, lowerBound)) return false;
311                 }
312             }
313             return true;
314         }
315 
316         /**
317          * Given a parameterized type (List&lt;String&gt; for example), checks that its
318          * generic types are compatible with those from a bound.
319          * @param classNode the classnode from which we will compare generics types
320          * @param bound the bound to which the types will be compared
321          * @return true if generics are compatible
322          */
323         private boolean compareGenericsWithBound(final ClassNode classNode, final ClassNode bound) {
324             if (classNode==null) return false;
325             if (!bound.isUsingGenerics() || (classNode.getGenericsTypes()==null && classNode.redirect().getGenericsTypes()!=null)) {
326                 // if the bound is not using generics, there's nothing to compare with
327                 return true;
328             }
329             if (!classNode.equals(bound)) {
330                  // the class nodes are on different types
331                 // in this situation, we must choose the correct execution path : either the bound
332                 // is an interface and we must find the implementing interface from the classnode
333                 // to compare their parameterized generics, or the bound is a regular class and we
334                 // must compare the bound with a superclass
335                 if (bound.isInterface()) {
336                     Set<ClassNode> interfaces = classNode.getAllInterfaces();
337                     // iterate over all interfaces to check if any corresponds to the bound we are
338                     // comparing to
339                     for (ClassNode anInterface : interfaces) {
340                         if (anInterface.equals(bound)) {
341                             // when we obtain an interface, the types represented by the interface
342                             // class node are not parameterized. This means that we must create a
343                             // new class node with the parameterized types that the current class node
344                             // has defined.
345                             ClassNode node = GenericsUtils.parameterizeType(classNode, anInterface);
346                             return compareGenericsWithBound(node, bound);
347                         }
348                     }
349                 }
350                 if (bound instanceof WideningCategories.LowestUpperBoundClassNode) {
351                     // another special case here, where the bound is a "virtual" type
352                     // we must then check the superclass and the interfaces
353                     boolean success = compareGenericsWithBound(classNode, bound.getSuperClass());
354                     if (success) {
355                         ClassNode[] interfaces = bound.getInterfaces();
356                         for (ClassNode anInterface : interfaces) {
357                             success &= compareGenericsWithBound(classNode, anInterface);
358                             if (!success) break;
359                         }
360                         if (success) return true;
361                     }
362                 }
363                 return compareGenericsWithBound(getParameterizedSuperClass(classNode), bound);
364             }
365             GenericsType[] cnTypes = classNode.getGenericsTypes();
366             if (cnTypes==null && classNode.isRedirectNode()) cnTypes=classNode.redirect().getGenericsTypes();
367             if (cnTypes==null) {
368                 // may happen if generic type is Foo<T extends Foo> and classnode is Foo -> Foo
369                 return true;
370             }
371             GenericsType[] redirectBoundGenericTypes = bound.redirect().getGenericsTypes();
372             Map<String, GenericsType> classNodePlaceholders = GenericsUtils.extractPlaceholders(classNode);
373             Map<String, GenericsType> boundPlaceHolders = GenericsUtils.extractPlaceholders(bound);
374             boolean match = true;
375             for (int i = 0; redirectBoundGenericTypes!=null && i < redirectBoundGenericTypes.length && match; i++) {
376                 GenericsType redirectBoundType = redirectBoundGenericTypes[i];
377                 GenericsType classNodeType = cnTypes[i];
378                 // The following code has been commented out because it causes GROOVY-5415
379                 // However, commenting doesn't make any test fail, which is curious...
380                 if (classNodeType.isPlaceholder()) {
381                     String name = classNodeType.getName();
382                     if (redirectBoundType.isPlaceholder()) {
383                         match = name.equals(redirectBoundType.getName());
384                         if (!match) {
385                             GenericsType genericsType = boundPlaceHolders.get(redirectBoundType.getName());
386                             match = false;
387                             if (genericsType!=null) {
388                                 if (genericsType.isPlaceholder()) {
389                                     match = true;
390                                 } else if (genericsType.isWildcard()) {
391                                     if (genericsType.getUpperBounds()!=null) {
392                                         for (ClassNode up : genericsType.getUpperBounds()) {
393                                             match |= redirectBoundType.isCompatibleWith(up);
394                                         }
395                                         if (genericsType.getLowerBound()!=null) {
396                                             match |= redirectBoundType.isCompatibleWith(genericsType.getLowerBound());
397                                         }
398                                     }
399                                 }
400                             }
401                         }
402                     } else {
403                         if (classNodePlaceholders.containsKey(name)) classNodeType=classNodePlaceholders.get(name);
404                         match = classNodeType.isCompatibleWith(redirectBoundType.getType());
405                     }
406                 } else {
407                     if (redirectBoundType.isPlaceholder()) {
408                         if (classNodeType.isPlaceholder()) {
409                             match = classNodeType.getName().equals(redirectBoundType.getName());
410                         } else {
411                             String name = redirectBoundType.getName();
412                             if (boundPlaceHolders.containsKey(name)) {
413                                 redirectBoundType = boundPlaceHolders.get(name);
414                                 boolean wildcard = redirectBoundType.isWildcard();
415                                 boolean placeholder = redirectBoundType.isPlaceholder();
416                                 if (placeholder || wildcard) {
417                                     // placeholder aliases, like Map<U,V> -> Map<K,V>
418 //                                    redirectBoundType = classNodePlaceholders.get(name);
419                                     if (wildcard) {
420                                         // ex: Comparable<Integer> <=> Comparable<? super T>
421                                         if (redirectBoundType.lowerBound!=null) {
422                                             GenericsType gt = new GenericsType(redirectBoundType.lowerBound);
423                                             if (gt.isPlaceholder()) {
424                                                 // check for recursive generic typedef, like in
425                                                 // <T extends Comparable<? super T>>
426                                                 if (classNodePlaceholders.containsKey(gt.getName())) {
427                                                     gt = classNodePlaceholders.get(gt.getName());
428                                                 }
429                                             }
430                                             match = implementsInterfaceOrIsSubclassOf(gt.getType(), classNodeType.getType());
431                                         }
432                                         if (match && redirectBoundType.upperBounds!=null) {
433                                             for (ClassNode upperBound : redirectBoundType.upperBounds) {
434                                                 GenericsType gt = new GenericsType(upperBound);
435                                                 if (gt.isPlaceholder()) {
436                                                     // check for recursive generic typedef, like in
437                                                     // <T extends Comparable<? super T>>
438                                                     if (classNodePlaceholders.containsKey(gt.getName())) {
439                                                         gt = classNodePlaceholders.get(gt.getName());
440                                                     }
441                                                 }
442                                                 match = match &&
443                                                         (implementsInterfaceOrIsSubclassOf(classNodeType.getType(), gt.getType())
444                                                          || classNodeType.isCompatibleWith(gt.getType())); // workaround for GROOVY-6095
445                                                 if (!match) break;
446                                             }
447                                         }
448                                         return match;
449                                     } else {
450                                         redirectBoundType = classNodePlaceholders.get(name);
451                                     }
452 
453                                 }
454                             }
455                             match = redirectBoundType.isCompatibleWith(classNodeType.getType());
456                         }
457                     } else {
458                         // todo: the check for isWildcard should be replaced with a more complete check
459                         match = redirectBoundType.isWildcard() || classNodeType.isCompatibleWith(redirectBoundType.getType());
460                     }
461                 }
462             }
463             if (!match) return false;
464             return true;
465         }

468     /**
469      * If you have a class which extends a class using generics, returns the superclass with parameterized types. For
470      * example, if you have:
471      * <code>class MyList&lt;T&gt; extends LinkedList&lt;T&gt;
472      * def list = new MyList&lt;String&gt;
473      * </code>
474      * then the parameterized superclass for MyList&lt;String&gt; is LinkedList&lt;String&gt;
475      * @param classNode the class for which we want to return the parameterized superclass
476      * @return the parameterized superclass
477      */
478     private static ClassNode getParameterizedSuperClass(ClassNode classNode) {
479         if (ClassHelper.OBJECT_TYPE.equals(classNode)) return null;
480         ClassNode superClass = classNode.getUnresolvedSuperClass();
481         if (superClass==null) {
482             return ClassHelper.OBJECT_TYPE;
483         }
484         if (!classNode.isUsingGenerics() || !superClass.isUsingGenerics()) return superClass;
485         GenericsType[] genericsTypes = classNode.getGenericsTypes();
486         GenericsType[] redirectGenericTypes = classNode.redirect().getGenericsTypes();
487         superClass = superClass.getPlainNodeReference();
488         if (genericsTypes==null || redirectGenericTypes==null || superClass.getGenericsTypes()==null) return superClass;
489         for (int i = 0, genericsTypesLength = genericsTypes.length; i < genericsTypesLength; i++) {
490             if (redirectGenericTypes[i].isPlaceholder()) {
491                 final GenericsType genericsType = genericsTypes[i];
492                 GenericsType[] superGenericTypes = superClass.getGenericsTypes();
493                 for (int j = 0, superGenericTypesLength = superGenericTypes.length; j < superGenericTypesLength; j++) {
494                     final GenericsType superGenericType = superGenericTypes[j];
495                     if (superGenericType.isPlaceholder() && superGenericType.getName().equals(redirectGenericTypes[i].getName())) {
496                         superGenericTypes[j] = genericsType;
497                     }
498                 }
499             }
500         }
501         return superClass;
502     }
