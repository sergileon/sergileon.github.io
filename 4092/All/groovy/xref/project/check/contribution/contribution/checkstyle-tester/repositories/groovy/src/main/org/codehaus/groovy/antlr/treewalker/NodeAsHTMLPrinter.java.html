27  /**
28   * A visitor that prints a html tags of each node to the supplied PrintStream
29   *
30   * @author <a href="mailto:groovy@ross-rayner.com">Jeremy Rayner</a>
31   */
32  
33  public class NodeAsHTMLPrinter extends VisitorAdapter {
34      private final String[] tokenNames;
35      private final PrintStream out;
36      private final Stack stack;
37  
38      /**
39       * A visitor that prints a html tags, for each node, to the supplied PrintStream.
40       * @param out supplied PrintStream to output nodes to
41       * @param tokenNames an array of token names to use
42       */
43      public NodeAsHTMLPrinter(PrintStream out,String[] tokenNames) {
44          this.tokenNames = tokenNames;
45          this.out = out;
46          this.stack = new Stack();
47      }
48  
49      public void setUp() {
50          out.println("<html><head></head><body><pre>");
51      }
52  
53      public void visitDefault(GroovySourceAST t,int visit) {
54          if (visit == OPENING_VISIT) {
55              out.print("<code title=" + quote(tokenNames[t.getType()]) + "><font color='#" + colour(t) + "'>");
56          } else if (visit == CLOSING_VISIT) {
57              out.print("</font></code>");
58          }
59      }
60  
61      private String quote(String tokenName) {
62          if (tokenName.length() > 0 && tokenName.charAt(0) != '\'')
63              return "'" + tokenName + "'";
64          else
65              return "\"" + tokenName + "\"";
66      }
67  
68      public void tearDown() {
69          out.println("</pre></body></html>");
70      }

310     public void push(GroovySourceAST t) {
