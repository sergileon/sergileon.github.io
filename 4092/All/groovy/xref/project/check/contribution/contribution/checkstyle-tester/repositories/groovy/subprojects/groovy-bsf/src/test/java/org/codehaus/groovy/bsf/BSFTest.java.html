19  package org.codehaus.groovy.bsf;
20  
21  import junit.framework.TestCase;
22  import org.apache.bsf.BSFEngine;
23  import org.apache.bsf.BSFException;
24  import org.apache.bsf.BSFManager;
25  import org.codehaus.groovy.runtime.ResourceGroovyMethods;
26  
27  import java.io.File;
28  import java.util.List;
29  import java.util.Vector;
30  
31  /**
32   * Tests the BSF integration
33   *
34   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>
35   * @author Paul King
36   */
37  public class BSFTest extends TestCase {
38      private static final Class ENGINE = GroovyEngine.class;
39  
40      protected BSFManager manager;
41  
42      protected void setUp() throws Exception {
43          BSFManager.registerScriptingEngine("groovy", ENGINE.getName(), new String[]{"groovy", "gy"});
44          manager = new BSFManager();
45      }
46  
47      public void testScriptDefPrefixRemoval() throws Exception {
48          manager.exec("groovy","scriptdef_foo", 0, 0, "assert bsf != null && this.class.name.equals('foo')");
49          manager.exec("groovy","scriptdef_", 0, 0, "assert bsf != null && this.class.name.equals('_')");
50      }
51  
52      public void testInvalidName() throws Exception {
53          manager.exec("groovy", null, 0, 0, "assert bsf != null");
54          manager.exec("groovy", "", 0, 0, "assert bsf != null");
55          manager.exec("groovy", "-", 0, 0, "assert bsf != null");
56      }
57  
58      public void testCompileErrorWithExec() throws Exception {
59          try {
60              manager.exec("groovy", "dummy", 0, 0, "assert assert");
61              fail("Should have caught compile exception");
62          } catch (BSFException e) {
63              assertTrue("e.getMessage() should contain CompilationError: " + e.getMessage(),
64                      e.getMessage().contains("CompilationError"));
65          }
66      }
67  
68      public void testCompileErrorWithEval() throws Exception {
69          try {
70              manager.eval("groovy", "dummy", 0, 0, "assert assert");
71              fail("Should have caught compile exception");
72          } catch (BSFException e) {
73              assertTrue("e.getMessage() should contain CompilationError: " + e.getMessage(),
74                      e.getMessage().contains("CompilationError"));
75          }
76      }
77  
78      public void testExec() throws Exception {
79          manager.exec("groovy", "Test1.groovy", 0, 0, "assert bsf != null , 'should have a bsf variable'");
80      }
81  
82      public void testApplyWithClosure() throws Exception {
83          Vector ignoreParamNames = null;
84          Vector ignoreArgs = null;
85          Integer actual = (Integer) manager.apply("groovy", "applyTest", 0, 0,
86                  "251", ignoreParamNames, ignoreArgs);
87          assertEquals(251, actual.intValue());
88      }
89  
90      public void testApply() throws Exception {
91          Vector ignoreParamNames = null;
92          Vector<Integer> args = new Vector<Integer>();
93          args.add(2);
94          args.add(5);
95          args.add(1);
96          Integer actual = (Integer) manager.apply("groovy", "applyTest", 0, 0,
97                  "def summer = { a, b, c -> a * 100 + b * 10 + c }", ignoreParamNames, args);
98          assertEquals(251, actual.intValue());
99      }
100 
101     public void testBracketName() throws Exception {
102         manager.exec("groovy", "Test1<groovy>", 0, 0, "assert bsf != null , 'should have a bsf variable'");
103     }
104 
105     public void testEval() throws Exception {
106         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0, "return [1, 2, 3]");
107         assertTrue("Should return a list: " + answer, answer instanceof List);
108         List list = (List) answer;
109         assertEquals("List should be of right size", 3, list.size());
110     }
111 
112     public void testTwoEvalsWithSameName() throws Exception {
113         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0, "return 'cheese'");
114         assertEquals("cheese", answer);
115         answer = manager.eval("groovy", "Test1.groovy", 0, 0, "return 'gromit'");
116         assertEquals("gromit", answer);
117     }
118 
119     public void testExecBug() throws Exception {
120         for (int i = 0; i < 10; i++) {
121             manager.exec("groovy", "Test1.groovy", 0, 0, "assert true");
122             manager.exec("groovy", "Test1.groovy", 0, 0, "assert true");
123         }
124     }
125 
126     public void testBsfVariables() throws Exception {
127         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0,
128                 "assert this.bsf != null\n  return this.bsf");
129         assertTrue("Should have an answer", answer != null);
130     }
131 
132     public void testNotFoundVariables() throws Exception {
133         manager.registerBean("x", 4);
134         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0,
135                 "def valueOfX = this.bsf.lookupBean('y'); assert valueOfX == null");
136         assertNull("Undeclared beans should yield null", answer);
137     }
138 
139     public void testRegisteredVariables() throws Exception {
140         manager.registerBean("x", 4);
141         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0,
142                 "def valueOfX = this.bsf.lookupBean('x'); assert valueOfX == 4; valueOfX + 1");
143         assertEquals("Incorrect return", 5, answer);
144     }
145 
146     public void testUnregisteredVariables() throws Exception {
147         manager.registerBean("x", 4);
148         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0,
149                 "def valueOfX = this.bsf.lookupBean('x'); assert valueOfX == 4; valueOfX + 1");
150         assertEquals("Incorrect return", 5, answer);
151         manager.unregisterBean("x");
152         // have to lookup registered beans
153         answer = manager.eval("groovy", "Test1.groovy", 0, 0,
154                 "def valueOfX = this.bsf.lookupBean('x'); assert valueOfX == null");
155         assertNull("Unregistered beans should yield null", answer);
156     }
157 
158     public void testDeclaredVariables() throws Exception {
159         manager.declareBean("xyz", 4, Integer.class);
160         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0, "xyz + 1");
161         assertEquals("Incorrect return", 5, answer);
162     }
163 
164     public void testUndeclaredVariables() throws Exception {
165         manager.declareBean("abc", 4, Integer.class);
166         // declared beans should just be available
167         Object answer = manager.eval("groovy", "Test1.groovy", 0, 0, "abc + 1");
168         assertEquals("Incorrect return", 5, answer);
169         manager.undeclareBean("abc");
170         answer = manager.eval("groovy", "Test1.groovy", 0, 0, "abc");
171         assertNull("Undeclared beans should yield null", answer);
172     }
173 
174     public void testCall() throws Exception {
175         BSFEngine bsfEngine = manager.loadScriptingEngine("groovy");
176         manager.declareBean("myvar", "hello", String.class);
177         Object myvar = manager.lookupBean("myvar");
178         String result = (String) bsfEngine.call(myvar, "reverse", new Object[]{});
179         assertEquals("olleh", result);
180     }
181 
182     public void testExecFile() throws Exception {
183         execScript("src/test/resources/groovy/script/MapFromList.groovy");
184     }
185 
186     protected void execScript(String fileName) throws Exception {
187         manager.exec("groovy", fileName, 0, 0, ResourceGroovyMethods.getText(new File(fileName)));
188     }
189 }
