176     private String[] sortWithMemberComparator(String[] details1, String[] details2) {
177         List details = new ArrayList();
178         details.add(details1);
179         details.add(details2);
180         Inspector.sort(details);
181         return (String[]) details.get(0);
182     }

226     private List getNamesFor(Object[] metaMethods) {
227         List result = new ArrayList();
228         for (int i = 0; i < metaMethods.length; i++) {
229             String[] strings = (String[]) metaMethods[i];
230             result.add(strings[Inspector.MEMBER_NAME_IDX]);
231         }
232         return result;
233     }
234 
235     private void assertNameEquals(String[] names, Object[] metaMethods) {
236         Set metaSet = new HashSet();
237         for (int i = 0; i < metaMethods.length; i++) {
238             String[] strings = (String[]) metaMethods[i];
239             metaSet.add(strings[Inspector.MEMBER_NAME_IDX]);
240         }
241         Set nameSet = new HashSet(Arrays.asList(names));
242         assertEquals(nameSet, metaSet);
243     }

245     private void assertContains(Object[] candidates, String[] sample) {
246         String sampleBuffer = concat(sample);
247         for (int i = 0; i < candidates.length; i++) {
248             String[] entry = (String[]) candidates[i];
249             if (sampleBuffer.equals(concat(entry))) return;
250         }
251         fail("should have found sample: " + sampleBuffer);
252     }
253 
254     private void assertUnique(Collection sortedMembers) {
255         if (sortedMembers.size() < 2) return;
256         Comparator comp = new Inspector.MemberComparator();
257         Iterator iter = sortedMembers.iterator();
258         Object last = iter.next();
259         while (iter.hasNext()) {
260             Object element = iter.next();
261             if (0 == comp.compare(last, element)) {
262                 fail("found duplication for element " + element);
263             }
264             last = element;
265         }
266     }
267 
268     private String concat(String[] details) {
269         StringBuilder detailBuffer = new StringBuilder();
270         for (int i = 0; i < details.length; i++) {
271             detailBuffer.append(details[i]);
272             detailBuffer.append(" ");
273         }
274         return detailBuffer.toString();
275     }
