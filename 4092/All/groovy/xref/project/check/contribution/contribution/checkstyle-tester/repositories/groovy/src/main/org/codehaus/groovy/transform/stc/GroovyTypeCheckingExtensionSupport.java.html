95      /**
96       * Builds a type checking extension relying on a Groovy script (type checking DSL).
97       *
98       * @param typeCheckingVisitor the type checking visitor
99       * @param scriptPath the path to the type checking script (in classpath)
100      * @param compilationUnit

102     public GroovyTypeCheckingExtensionSupport(
103             final StaticTypeCheckingVisitor typeCheckingVisitor,
104             final String scriptPath, final CompilationUnit compilationUnit) {
105         super(typeCheckingVisitor);
106         this.scriptPath = scriptPath;
107         this.compilationUnit = compilationUnit;
108     }

114     @Override
115     public void setup() {
116         CompilerConfiguration config = new CompilerConfiguration();
117         config.setScriptBaseClass("org.codehaus.groovy.transform.stc.GroovyTypeCheckingExtensionSupport.TypeCheckingDSL");
118         ImportCustomizer ic = new ImportCustomizer();
119         ic.addStarImports("org.codehaus.groovy.ast.expr");
120         ic.addStaticStars("org.codehaus.groovy.ast.ClassHelper");
121         ic.addStaticStars("org.codehaus.groovy.transform.stc.StaticTypeCheckingSupport");
122         config.addCompilationCustomizers(ic);
123         final GroovyClassLoader transformLoader = compilationUnit!=null?compilationUnit.getTransformLoader():typeCheckingVisitor.getSourceUnit().getClassLoader();
124 
125         // since Groovy 2.2, it is possible to use FQCN for type checking extension scripts
126         TypeCheckingDSL script = null;
127         try {
128             Class<?> clazz = transformLoader.loadClass(scriptPath, false, true);
129             if (TypeCheckingDSL.class.isAssignableFrom(clazz)) {
130                 script = (TypeCheckingDSL) clazz.newInstance();
131             } else if (TypeCheckingExtension.class.isAssignableFrom(clazz)) {
132                 // since 2.4, we can also register precompiled type checking extensions which are not scripts
133                 try {
134                     Constructor<?> declaredConstructor = clazz.getDeclaredConstructor(StaticTypeCheckingVisitor.class);
135                     TypeCheckingExtension extension = (TypeCheckingExtension) declaredConstructor.newInstance(typeCheckingVisitor);
136                     typeCheckingVisitor.addTypeCheckingExtension(extension);
137                     extension.setup();
138                     return;
139                 } catch (InstantiationException e) {
140                     addLoadingError(config);
141                 } catch (IllegalAccessException e) {
142                     addLoadingError(config);
143                 } catch (NoSuchMethodException e) {
144                     context.getErrorCollector().addFatalError(
145                             new SimpleMessage("Static type checking extension '" + scriptPath + "' could not be loaded because it doesn't have a constructor accepting StaticTypeCheckingVisitor.",
146                                     config.getDebug(), typeCheckingVisitor.getSourceUnit())
147                     );
148                 } catch (InvocationTargetException e) {
149                     addLoadingError(config);
150                 }
151             }
152         } catch (ClassNotFoundException e) {
153             // silent
154         } catch (InstantiationException e) {
155             addLoadingError(config);
156         } catch (IllegalAccessException e) {
157             addLoadingError(config);
158         }
159         if (script==null) {
160             ClassLoader cl = typeCheckingVisitor.getSourceUnit().getClassLoader();
161             // cast to prevent incorrect @since 1.7 warning
162             InputStream is = ((ClassLoader)transformLoader).getResourceAsStream(scriptPath);
163             if (is == null) {
164                 // fallback to the source unit classloader
165                 is = cl.getResourceAsStream(scriptPath);
166             }
167             if (is == null) {
168                 // fallback to the compiler classloader
169                 cl = GroovyTypeCheckingExtensionSupport.class.getClassLoader();
170                 is = cl.getResourceAsStream(scriptPath);
171             }
172             if (is == null) {
173                 // if the input stream is still null, we've not found the extension
174                 context.getErrorCollector().addFatalError(
175                         new SimpleMessage("Static type checking extension '" + scriptPath + "' was not found on the classpath.",
176                                 config.getDebug(), typeCheckingVisitor.getSourceUnit()));
177             }
178             try {
179                 GroovyShell shell = new GroovyShell(transformLoader, new Binding(), config);
180                 script = (TypeCheckingDSL) shell.parse(
181                         new InputStreamReader(is, typeCheckingVisitor.getSourceUnit().getConfiguration().getSourceEncoding())
182                 );
183             } catch (CompilationFailedException e) {
184                 throw new GroovyBugError("An unexpected error was thrown during custom type checking", e);
185             } catch (UnsupportedEncodingException e) {
186                 throw new GroovyBugError("Unsupported encoding found in compiler configuration", e);
187             }
188         }
189         if (script!=null) {
190             script.extension = this;
191             script.run();
192             List<Closure> list = eventHandlers.get("setup");
193             if (list != null) {
194                 for (Closure closure : list) {
195                     safeCall(closure);
196                 }
197             }
198         }
199     }

201     private void addLoadingError(final CompilerConfiguration config) {
202         context.getErrorCollector().addFatalError(
203                 new SimpleMessage("Static type checking extension '" + scriptPath + "' could not be loaded.",
204                         config.getDebug(), typeCheckingVisitor.getSourceUnit())
205         );
206     }

208     @Override
209     public void finish() {
210         List<Closure> list = eventHandlers.get("finish");
211         if (list != null) {
212             for (Closure closure : list) {
213                 safeCall(closure);
214             }
215         }

354     @Override
355     @SuppressWarnings("unchecked")
356     public List<MethodNode> handleMissingMethod(final ClassNode receiver, final String name, final ArgumentListExpression argumentList, final ClassNode[] argumentTypes, final MethodCall call) {
357         List<Closure> onMethodSelection = eventHandlers.get("handleMissingMethod");
358         List<MethodNode> methodList = new LinkedList<MethodNode>();
359         if (onMethodSelection != null) {
360             for (Closure closure : onMethodSelection) {
361                 Object result = safeCall(closure, receiver, name, argumentList, argumentTypes, call);
362                 if (result != null) {
363                     if (result instanceof MethodNode) {
364                         methodList.add((MethodNode) result);
365                     } else if (result instanceof Collection) {
366                         methodList.addAll((Collection<? extends MethodNode>) result);
367                     } else {
368                         throw new GroovyBugError("Type checking extension returned unexpected method list: " + result);
369                     }
370                 }
371             }
372         }
373         return methodList;
374     }
375 
376     @Override
377     @SuppressWarnings("unchecked")
378     public List<MethodNode> handleAmbiguousMethods(final List<MethodNode> nodes, final Expression origin) {
379         List<Closure> onMethodSelection = eventHandlers.get("handleAmbiguousMethods");
380         List<MethodNode> methodList = nodes;
381         if (onMethodSelection != null) {
382             Iterator<Closure> iterator = onMethodSelection.iterator();
383             while (methodList.size()>1 && iterator.hasNext() ) {
384                 final Closure closure = iterator.next();
385                 Object result = safeCall(closure, methodList, origin);
386                 if (result != null) {
387                     if (result instanceof MethodNode) {
388                         methodList = Collections.singletonList((MethodNode) result);
389                     } else if (result instanceof Collection) {
390                         methodList = new LinkedList<MethodNode>((Collection<? extends MethodNode>) result);
391                     } else {
392                         throw new GroovyBugError("Type checking extension returned unexpected method list: " + result);
393                     }
394                 }
395             }
396         }
397         return methodList;
398     }

408     public abstract static class TypeCheckingDSL extends Script {
409         private GroovyTypeCheckingExtensionSupport extension;
410 
411         @Override
412         public Object getProperty(final String property) {
413             try {
414                 return InvokerHelper.getProperty(extension, property);
415             } catch (Exception e) {
416                 return super.getProperty(property);
417             }
418         }
419 
420         @Override
421         public void setProperty(final String property, final Object newValue) {
422             try {
423                 InvokerHelper.setProperty(extension, property, newValue);
424             } catch (Exception e) {
425                 super.setProperty(property, newValue);
426             }
427         }
428 
429         @Override
430         public Object invokeMethod(final String name, final Object args) {
431             if (name.startsWith("is") && name.endsWith("Expression") && args instanceof Object[] && ((Object[]) args).length == 1) {
432                 String type = name.substring(2);
433                 Object target = ((Object[]) args)[0];
434                 if (target==null) return false;
435                 try {
436                     Class typeClass = Class.forName("org.codehaus.groovy.ast.expr."+type);
437                     return typeClass.isAssignableFrom(target.getClass());
438                 } catch (ClassNotFoundException e) {
439                     return false;
440                 }
441             }
442             if (args instanceof Object[] && ((Object[]) args).length == 1 && ((Object[]) args)[0] instanceof Closure) {
443                 Object[] argsArray = (Object[]) args;
444                 String methodName = METHOD_ALIASES.get(name);
445                 if (methodName == null) {
446                     return InvokerHelper.invokeMethod(extension, name, args);
447                 }
448                 List<Closure> closures = extension.eventHandlers.get(methodName);
449                 if (closures == null) {
450                     closures = new LinkedList<Closure>();
451                     extension.eventHandlers.put(methodName, closures);
452                 }
453                 closures.add((Closure) argsArray[0]);
454                 return null;
455             } else {
456                 return InvokerHelper.invokeMethod(extension, name, args);
457             }
458         }
