46  /**
47   * A specialized version of the multi type binary expression dispatcher which is aware of static compilation.
48   * It is able to generate optimized bytecode for some operations using JVM instructions when available.
49   *
50   * @author Cedric Champeau
51   * @author Jochen Theodorou
52   */
53  public class StaticTypesBinaryExpressionMultiTypeDispatcher extends BinaryExpressionMultiTypeDispatcher implements Opcodes {

55      private final AtomicInteger labelCounter = new AtomicInteger();

58      public StaticTypesBinaryExpressionMultiTypeDispatcher(WriterController wc) {

62      private int incValue(Token token) {
63          switch (token.getType()) {
64              case Types.PLUS_PLUS:
65                  return 1;
66              case Types.MINUS_MINUS:
67                  return -1;
68          }
69          return 0;
70      }

72      @Override
73      protected void writePostOrPrefixMethod(int op, String method, Expression expression, Expression orig) {
74          MethodNode mn = orig.getNodeMetaData(StaticTypesMarker.DIRECT_METHOD_CALL_TARGET);
75          WriterController controller = getController();
76          OperandStack operandStack = controller.getOperandStack();
77          if (mn!=null) {
78              operandStack.pop();
79              MethodCallExpression call = new MethodCallExpression(
80                      expression,
81                      method,
82                      ArgumentListExpression.EMPTY_ARGUMENTS
83              );
84              call.setMethodTarget(mn);
85              call.visit(controller.getAcg());
86              return;
87          }
88  
89          ClassNode top = operandStack.getTopOperand();
90          if (ClassHelper.isPrimitiveType(top) && (ClassHelper.isNumberType(top)||char_TYPE.equals(top))) {
91              MethodVisitor mv = controller.getMethodVisitor();
92              if (WideningCategories.isIntCategory(top) || char_TYPE.equals(top)) {
93                  mv.visitInsn(ICONST_1);
94              } else if (long_TYPE.equals(top)) {
95                  mv.visitInsn(LCONST_1);
96              } else if (float_TYPE.equals(top)) {
97                  mv.visitInsn(FCONST_1);
98              } else if (double_TYPE.equals(top)) {
99                  mv.visitInsn(DCONST_1);
100             }
101             if ("next".equals(method)) {
102                 if (WideningCategories.isIntCategory(top) || char_TYPE.equals(top)) {
103                     mv.visitInsn(IADD);
104                 } else if (long_TYPE.equals(top)) {
105                     mv.visitInsn(LADD);
106                 } else if (float_TYPE.equals(top)) {
107                     mv.visitInsn(FADD);
108                 } else if (double_TYPE.equals(top)) {
109                     mv.visitInsn(DADD);
110                 }
111             } else {
112                 if (WideningCategories.isIntCategory(top) || char_TYPE.equals(top)) {
113                     mv.visitInsn(ISUB);
114                 } else if (long_TYPE.equals(top)) {
115                     mv.visitInsn(LSUB);
116                 } else if (float_TYPE.equals(top)) {
117                     mv.visitInsn(FSUB);
118                 } else if (double_TYPE.equals(top)) {
119                     mv.visitInsn(DSUB);
120                 }
121             }
122             return;
123         }
124         super.writePostOrPrefixMethod(op, method, expression, orig);
125     }

127     @Override
128     public void evaluateEqual(final BinaryExpression expression, final boolean defineVariable) {
129         if (!defineVariable) {
130             Expression leftExpression = expression.getLeftExpression();
131             if (leftExpression instanceof PropertyExpression) {
132                 PropertyExpression pexp = (PropertyExpression) leftExpression;
133                 if (makeSetProperty(
134                         pexp.getObjectExpression(),
135                         pexp.getProperty(),
136                         expression.getRightExpression(),
137                         pexp.isSafe(),
138                         pexp.isSpreadSafe(),
139                         pexp.isImplicitThis(),
140                         pexp instanceof AttributeExpression)) return;
141             }
142         }
143         // GROOVY-5620: Spread safe/Null safe operator on LHS is not supported
144         if (expression.getLeftExpression() instanceof PropertyExpression
145                 && ((PropertyExpression) expression.getLeftExpression()).isSpreadSafe()
146                 && StaticTypeCheckingSupport.isAssignment(expression.getOperation().getType())) {
147             // rewrite it so that it can be statically compiled
148             transformSpreadOnLHS(expression);
149             return;
150         }
151         super.evaluateEqual(expression, defineVariable);
152     }

154     private void transformSpreadOnLHS(BinaryExpression origin) {
155         PropertyExpression spreadExpression = (PropertyExpression) origin.getLeftExpression();
156         Expression value = origin.getRightExpression();
157         WriterController controller = getController();
158         MethodVisitor mv = controller.getMethodVisitor();
159         CompileStack compileStack = controller.getCompileStack();
160         TypeChooser typeChooser = controller.getTypeChooser();
161         OperandStack operandStack = controller.getOperandStack();
162         ClassNode classNode = controller.getClassNode();
163         int counter = labelCounter.incrementAndGet();
164         Expression receiver = spreadExpression.getObjectExpression();
165 
166         // create an empty arraylist
167         VariableExpression result = new VariableExpression(
168                 this.getClass().getSimpleName()+"$spreadresult" + counter,
169                 ARRAYLIST_CLASSNODE
170         );
171         ConstructorCallExpression cce = new ConstructorCallExpression(ARRAYLIST_CLASSNODE, ArgumentListExpression.EMPTY_ARGUMENTS);
172         cce.setNodeMetaData(StaticTypesMarker.DIRECT_METHOD_CALL_TARGET, ARRAYLIST_CONSTRUCTOR);
173         DeclarationExpression declr = new DeclarationExpression(
174                 result,
175                 Token.newSymbol("=", spreadExpression.getLineNumber(), spreadExpression.getColumnNumber()),
176                 cce
177         );
178         declr.visit(controller.getAcg());
179         // if (receiver != null)
180         receiver.visit(controller.getAcg());
181         Label ifnull = compileStack.createLocalLabel("ifnull_" + counter);
182         mv.visitJumpInsn(IFNULL, ifnull);
183         operandStack.remove(1); // receiver consumed by if()
184         Label nonull = compileStack.createLocalLabel("nonull_" + counter);
185         mv.visitLabel(nonull);
186         ClassNode componentType = StaticTypeCheckingVisitor.inferLoopElementType(typeChooser.resolveType(receiver, classNode));
187         Parameter iterator = new Parameter(componentType, "for$it$" + counter);
188         VariableExpression iteratorAsVar = new VariableExpression(iterator);
189         PropertyExpression pexp = spreadExpression instanceof AttributeExpression?
190                 new AttributeExpression(iteratorAsVar, spreadExpression.getProperty(), true):
191                 new PropertyExpression(iteratorAsVar, spreadExpression.getProperty(), true);
192         pexp.setImplicitThis(spreadExpression.isImplicitThis());
193         pexp.setSourcePosition(spreadExpression);
194         BinaryExpression assignment = new BinaryExpression(
195                 pexp,
196                 origin.getOperation(),
197                 value
198         );
199         MethodCallExpression add = new MethodCallExpression(
200                 result,
201                 "add",
202                 assignment
203         );
204         add.setMethodTarget(ARRAYLIST_ADD_METHOD);
205         // for (e in receiver) { result.add(e?.method(arguments) }
206         ForStatement stmt = new ForStatement(
207                 iterator,
208                 receiver,
209                 new ExpressionStatement(add)
210         );
211         stmt.visit(controller.getAcg());
212         // else { empty list }
213         mv.visitLabel(ifnull);
214 
215         // end of if/else
216         // return result list
217         result.visit(controller.getAcg());
218 
219     }

221     private boolean makeSetProperty(final Expression receiver, final Expression message, final Expression arguments, final boolean safe, final boolean spreadSafe, final boolean implicitThis, final boolean isAttribute) {
222         WriterController controller = getController();
223         TypeChooser typeChooser = controller.getTypeChooser();
224         ClassNode receiverType = typeChooser.resolveType(receiver, controller.getClassNode());
225         String property = message.getText();
226         boolean isThisExpression = receiver instanceof VariableExpression && ((VariableExpression) receiver).isThisExpression();
227         if (isAttribute
228                 || (isThisExpression &&
229                     receiverType.getDeclaredField(property)!=null)) {
230             ClassNode current = receiverType;
231             FieldNode fn = null;
232             while (fn==null && current!=null) {
233                 fn = current.getDeclaredField(property);
234                 if (fn==null){
235                     current = current.getSuperClass();
236                 }
237             }
238             if (fn!=null && receiverType!=current && !fn.isPublic()) {
239                 // check that direct access is allowed
240                 if (!fn.isProtected()) {
241                     return false;
242                 }
243                 String pkg1 = receiverType.getPackageName();
244                 String pkg2 = current.getPackageName();
245                 if (pkg1!=pkg2 && !pkg1.equals(pkg2)) {
246                     return false;
247                 }
248                 OperandStack operandStack = controller.getOperandStack();
249                 MethodVisitor mv = controller.getMethodVisitor();
250                 if (!fn.isStatic()) {
251                     receiver.visit(controller.getAcg());
252                 }
253                 arguments.visit(controller.getAcg());
254                 operandStack.doGroovyCast(fn.getOriginType());
255                 mv.visitFieldInsn(fn.isStatic() ? PUTSTATIC : PUTFIELD,
256                         BytecodeHelper.getClassInternalName(fn.getOwner()),
257                         property,
258                         BytecodeHelper.getTypeDescription(fn.getOriginType()));
259                 operandStack.remove(fn.isStatic()?1:2);
260                 return true;
261             }
262         }
263         if (!isAttribute) {
264             String setter = "set" + MetaClassHelper.capitalize(property);
265             MethodNode setterMethod = receiverType.getSetterMethod(setter, false);
266             ClassNode declaringClass = setterMethod!=null?setterMethod.getDeclaringClass():null;
267             if (isThisExpression && declaringClass!=null && declaringClass.equals(controller.getClassNode())) {
268                 // this.x = ... shouldn't use a setter if in the same class
269                 setterMethod = null;
270             } else if (setterMethod == null) {
271                 PropertyNode propertyNode = receiverType.getProperty(property);
272                 if (propertyNode != null) {
273                     int mods = propertyNode.getModifiers();
274                     if (!Modifier.isFinal(mods)) {
275                         setterMethod = new MethodNode(
276                                 setter,
277                                 ACC_PUBLIC,
278                                 ClassHelper.VOID_TYPE,
279                                 new Parameter[]{new Parameter(propertyNode.getOriginType(), "value")},
280                                 ClassNode.EMPTY_ARRAY,
281                                 EmptyStatement.INSTANCE
282                         );
283                         setterMethod.setDeclaringClass(receiverType);
284                     }
285                 }
286             }
287             if (setterMethod != null) {
288                 Expression call = StaticPropertyAccessHelper.transformToSetterCall(
289                         receiver,
290                         setterMethod,
291                         arguments,
292                         implicitThis,
293                         safe,
294                         spreadSafe,
295                         true, // to be replaced with a proper test whether a return value should be used or not
296                         message
297                 );
298                 call.visit(controller.getAcg());
299                 return true;
300             }
301         }
302         return false;
303     }

305     protected void assignToArray(Expression parent, Expression receiver, Expression index, Expression rhsValueLoader) {
306         ClassNode current = getController().getClassNode();
307         ClassNode arrayType = getController().getTypeChooser().resolveType(receiver, current);
308         ClassNode arrayComponentType = arrayType.getComponentType();
309         int operationType = getOperandType(arrayComponentType);
310         BinaryExpressionWriter bew = binExpWriter[operationType];
311         AsmClassGenerator acg = getController().getAcg();
312 
313         if (bew.arraySet(true) && arrayType.isArray()) {
314             super.assignToArray(parent, receiver, index, rhsValueLoader);
315         } else {
316             /******
317             / This code path is needed because ACG creates array access expressions
318             *******/
319 
320             WriterController controller = getController();
321             StaticTypeCheckingVisitor visitor = new StaticCompilationVisitor(controller.getSourceUnit(), controller.getClassNode());
322             // let's replace this assignment to a subscript operator with a
323             // method call
324             // e.g. x[5] = 10
325             // -> (x, [], 5), =, 10
326             // -> methodCall(x, "putAt", [5, 10])
327             ArgumentListExpression ae = new ArgumentListExpression(index, rhsValueLoader);
328             if (rhsValueLoader instanceof VariableSlotLoader && parent instanceof BinaryExpression) {
329                 // GROOVY-6061
330                 Expression right = ((BinaryExpression) parent).getRightExpression();
331                 rhsValueLoader.putNodeMetaData(StaticTypesMarker.INFERRED_TYPE,
332                         controller.getTypeChooser().resolveType(parent, controller.getClassNode()));
333             }
334             MethodCallExpression mce = new MethodCallExpression(
335                     receiver,
336                     "putAt",
337                     ae
338             );
339             mce.setSourcePosition(parent);
340             visitor.visitMethodCallExpression(mce);
341             OperandStack operandStack = controller.getOperandStack();
342             int height = operandStack.getStackLength();
343             mce.visit(controller.getAcg());
344             operandStack.pop();
345             operandStack.remove(operandStack.getStackLength()-height);
346             // return value of assignment
347             rhsValueLoader.visit(controller.getAcg());
348         }
349     }
