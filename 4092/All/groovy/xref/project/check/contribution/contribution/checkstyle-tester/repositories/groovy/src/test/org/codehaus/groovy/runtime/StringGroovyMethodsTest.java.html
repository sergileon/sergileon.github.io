19  package org.codehaus.groovy.runtime;
20  
21  import groovy.util.GroovyTestCase;
22  
23  import java.math.BigDecimal;
24  import java.math.BigInteger;
25  
26  /**
27   * @author James Strachan
28   * @author Marc Guillemot
29   * @author Brad Long
30   */
31  public class StringGroovyMethodsTest extends GroovyTestCase {
32  
33      public void testIncrementString() throws Exception {
34          String original = "z";
35          String answer = StringGroovyMethods.next(original);
36  
37          assertEquals("{", answer);
38          assertTrue(answer.compareTo(original) > 0);
39      }
40  
41      public void testDecrementString() throws Exception {
42          String original = "a";
43          String answer = StringGroovyMethods.previous(original);
44  
45          assertEquals("`", answer);
46          assertTrue(ScriptBytecodeAdapter.compareLessThan(answer, original));
47      }
48      
49      public void testToMethods() throws Exception {
50          assertEquals(StringGroovyMethods.toInteger("1"), new Integer(1));
51          assertEquals(StringGroovyMethods.toLong("1"), new Long(1));
52          assertEquals(StringGroovyMethods.toFloat("1"), new Float(1));
53          assertEquals(StringGroovyMethods.toDouble("1"), new Double(1));
54          assertEquals(StringGroovyMethods.toBigInteger("1"), new BigInteger("1"));
55          assertEquals(StringGroovyMethods.toBigDecimal("1"), new BigDecimal("1"));
56          assertEquals(StringGroovyMethods.toBoolean("True"), Boolean.TRUE);
57          assertEquals(StringGroovyMethods.toBoolean("Y"), Boolean.TRUE);
58          assertEquals(StringGroovyMethods.toBoolean(" y "), Boolean.TRUE);
59          assertEquals(StringGroovyMethods.toBoolean("1"), Boolean.TRUE);
60          assertEquals(StringGroovyMethods.toBoolean("false"), Boolean.FALSE);
61          assertEquals(StringGroovyMethods.toBoolean("n"), Boolean.FALSE);
62          assertEquals(StringGroovyMethods.toBoolean("0"), Boolean.FALSE);
63      }
64  
65      public void testIsMethods() throws Exception {
66          String intStr = "123";
67          String floatStr = "1.23E-1";
68          String nonNumberStr = "ONE";
69  
70          assertTrue(StringGroovyMethods.isInteger(intStr));
71          assertFalse(StringGroovyMethods.isInteger(floatStr));
72          assertFalse(StringGroovyMethods.isInteger(nonNumberStr));
73          assertTrue(StringGroovyMethods.isLong(intStr));
74          assertFalse(StringGroovyMethods.isLong(floatStr));
75          assertFalse(StringGroovyMethods.isLong(nonNumberStr));
76  
77          assertTrue(StringGroovyMethods.isFloat(intStr));
78          assertTrue(StringGroovyMethods.isFloat(floatStr));
79          assertFalse(StringGroovyMethods.isFloat(nonNumberStr));
80          assertTrue(StringGroovyMethods.isDouble(intStr));
81          assertTrue(StringGroovyMethods.isDouble(floatStr));
82          assertFalse(StringGroovyMethods.isDouble(nonNumberStr));
83  
84          assertTrue(StringGroovyMethods.isBigInteger(intStr));
85          assertFalse(StringGroovyMethods.isBigInteger(floatStr));
86          assertFalse(StringGroovyMethods.isBigInteger(nonNumberStr));
87          assertTrue(StringGroovyMethods.isBigDecimal(intStr));
88          assertTrue(StringGroovyMethods.isBigDecimal(floatStr));
89          assertFalse(StringGroovyMethods.isBigDecimal(nonNumberStr));
90          assertTrue(StringGroovyMethods.isNumber(intStr));
91          assertTrue(StringGroovyMethods.isNumber(floatStr));
92          assertFalse(StringGroovyMethods.isNumber(nonNumberStr));
93      }
94  }
