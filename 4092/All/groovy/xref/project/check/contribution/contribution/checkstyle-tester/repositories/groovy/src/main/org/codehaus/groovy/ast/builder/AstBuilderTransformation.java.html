37  /**
38   * Transformation to capture ASTBuilder from code statements.
39   * <p>
40   * The AstBuilder "from code" approach is used with a single Closure
41   * parameter. This transformation converts the ClosureExpression back
42   * into source code and rewrites the AST so that the "from string"
43   * builder is invoked on the source. In order for this to work, the
44   * closure source must be given a goto label. It is the "from string"
45   * approach's responsibility to remove the BlockStatement created
46   * by the label.
47   *
48   * @author Hamlet D'Arcy
49   */
50  
51  @GroovyASTTransformation(phase = CompilePhase.SEMANTIC_ANALYSIS)
52  public class AstBuilderTransformation implements ASTTransformation {
53  
54      public void visit(ASTNode[] nodes, SourceUnit sourceUnit) {
55  
56          // todo : are there other import types that can be specified?
57          AstBuilderInvocationTrap transformer = new AstBuilderInvocationTrap(
58                  sourceUnit.getAST().getImports(),
59                  sourceUnit.getAST().getStarImports(),
60                  sourceUnit.getSource(),
61                  sourceUnit
62          );
63          if (nodes != null) {
64              for (ASTNode it : nodes) {
65                  if (!(it instanceof AnnotationNode) && !(it instanceof ClassNode)) {
66                      it.visit(transformer);
67                  }
68              }
69          }
70          if (sourceUnit.getAST() != null) {
71              sourceUnit.getAST().visit(transformer);
72              if (sourceUnit.getAST().getStatementBlock() != null) {
73                  sourceUnit.getAST().getStatementBlock().visit(transformer);
74              }
75              if (sourceUnit.getAST().getClasses() != null) {
76                  for (ClassNode classNode : sourceUnit.getAST().getClasses()) {
77                      if (classNode.getMethods() != null) {
78                          for (MethodNode node : classNode.getMethods()) {
79                              if (node != null && node.getCode() != null) {
80                                  node.getCode().visit(transformer);
81                              }
82                          }
83                      }
84  
85                      try {
86                          if (classNode.getDeclaredConstructors() != null) {
87                              for (MethodNode node : classNode.getDeclaredConstructors()) {
88                                  if (node != null && node.getCode() != null) {
89                                      node.getCode().visit(transformer);
90                                  }
91                              }
92                          }
93                      } catch (MissingPropertyException ignored) {
94                          // todo: inner class nodes don't have a constructors field available
95                      }
96  
97                      // all properties are also always fields
98                      if (classNode.getFields() != null) {
99                          for (FieldNode node : classNode.getFields()) {
100                             if (node.getInitialValueExpression() != null) {
101                                 node.getInitialValueExpression().visit(transformer);
102                             }
103                         }
104                     }
105 
106                     try {
107                         if (classNode.getObjectInitializerStatements() != null) {
108                             for (Statement node : classNode.getObjectInitializerStatements()) {
109                                 if (node != null) {
110                                     node.visit(transformer);
111                                 }
112                             }
113                         }
114                     } catch (MissingPropertyException ignored) {
115                         // todo: inner class nodes don't have a objectInitializers field available
116                     }
117 
118                     // todo: is there anything to do with the module ???
119                 }
120             }
121             if (sourceUnit.getAST().getMethods() != null) {
122                 for (MethodNode node : sourceUnit.getAST().getMethods()) {
123                     if (node != null) {
124                         if (node.getParameters() != null) {
125                             for (Parameter parameter : node.getParameters()) {
126                                 if (parameter != null && parameter.getInitialExpression() != null) {
127                                     parameter.getInitialExpression().visit(transformer);
128                                 }
129                             }
130                         }
131                         if (node.getCode() != null) {
132                             node.getCode().visit(transformer);
133                         }
134                     }
135                 }
136             }
137         }
138     }
139 
140     /**
141      * This class traps invocations of AstBuilder.build(CompilePhase, boolean, Closure) and converts
142      * the contents of the closure into expressions by reading the source of the Closure and sending
143      * that as a String to AstBuilder.build(String, CompilePhase, boolean) at runtime.
144      */
145     private static class AstBuilderInvocationTrap extends CodeVisitorSupport {
146 
147         private final List<String> factoryTargets = new ArrayList<String>();
148         private final ReaderSource source;
149         private final SourceUnit sourceUnit;
150 
151         /**
152          * Creates the trap and captures all the ways in which a class may be referenced via imports.
153          *
154          * @param imports        all the imports from the source
155          * @param importPackages all the imported packages from the source
156          * @param source         the reader source that contains source for the SourceUnit
157          * @param sourceUnit     the source unit being compiled. Used for error messages.
158          */
159         AstBuilderInvocationTrap(List<ImportNode> imports, List<ImportNode> importPackages, ReaderSource source, SourceUnit sourceUnit) {
160             if (source == null) throw new IllegalArgumentException("Null: source");
161             if (sourceUnit == null) throw new IllegalArgumentException("Null: sourceUnit");
162             this.source = source;
163             this.sourceUnit = sourceUnit;
164 
165             // factory type may be references as fully qualified, an import, or an alias
166             factoryTargets.add("org.codehaus.groovy.ast.builder.AstBuilder");//default package
167 
168             if (imports != null) {
169                 for (ImportNode importStatement : imports) {
170                     if ("org.codehaus.groovy.ast.builder.AstBuilder".equals(importStatement.getType().getName())) {
171                         factoryTargets.add(importStatement.getAlias());
172                     }
173                 }
174             }
175 
176             if (importPackages != null) {
177                 for (ImportNode importPackage : importPackages) {
178                     if ("org.codehaus.groovy.ast.builder.".equals(importPackage.getPackageName())) {
179                         factoryTargets.add("AstBuilder");
180                         break;
181                     }
182                 }
183             }
184         }
185 
186         /**
187          * Reports an error back to the source unit.
188          *
189          * @param msg  the error message
190          * @param expr the expression that caused the error message.
191          */
192         private void addError(String msg, ASTNode expr) {
193             sourceUnit.getErrorCollector().addErrorAndContinue(
194                     new SyntaxErrorMessage(new SyntaxException(msg + '\n', expr.getLineNumber(), expr.getColumnNumber(), expr.getLastLineNumber(), expr.getLastColumnNumber()), sourceUnit)
195             );
196         }
197 
198 
199         /**
200          * Attempts to find AstBuilder 'from code' invocations. When found, converts them into calls
201          * to the 'from string' approach.
202          *
203          * @param call the method call expression that may or may not be an AstBuilder 'from code' invocation.
204          */
205         public void visitMethodCallExpression(MethodCallExpression call) {
206 
207             if (isBuildInvocation(call)) {
208 
209                 ClosureExpression closureExpression = getClosureArgument(call);
210                 List<Expression> otherArgs = getNonClosureArguments(call);
211                 String source = convertClosureToSource(closureExpression);
212 
213                 // parameter order is build(CompilePhase, boolean, String)
214                 otherArgs.add(new ConstantExpression(source));
215                 call.setArguments(new ArgumentListExpression(otherArgs));
216                 call.setMethod(new ConstantExpression("buildFromBlock"));
217                 call.setSpreadSafe(false);
218                 call.setSafe(false);
219                 call.setImplicitThis(false);
220             } else {
221                 // continue normal tree walking
222                 call.getObjectExpression().visit(this);
223                 call.getMethod().visit(this);
224                 call.getArguments().visit(this);
225             }
226         }
227 
228         private List<Expression> getNonClosureArguments(MethodCallExpression call) {
229             List<Expression> result = new ArrayList<Expression>();
230             if (call.getArguments() instanceof TupleExpression) {
231                 for (ASTNode node : ((TupleExpression) call.getArguments()).getExpressions()) {
232                     if (!(node instanceof ClosureExpression)) {
233                         result.add((Expression) node);
234                     }
235                 }
236             }
237             return result;
238         }
239 
240         private ClosureExpression getClosureArgument(MethodCallExpression call) {
241 
242             if (call.getArguments() instanceof TupleExpression) {
243                 for (ASTNode node : ((TupleExpression) call.getArguments()).getExpressions()) {
244                     if (node instanceof ClosureExpression) {
245                         return (ClosureExpression) node;
246                     }
247                 }
248             }
249             return null;
250         }
251 
252         /**
253          * Looks for method calls on the AstBuilder class called build that take
254          * a Closure as parameter. This is all needed b/c build is overloaded.
255          *
256          * @param call the method call expression, may not be null
257          */
258         private boolean isBuildInvocation(MethodCallExpression call) {
259             if (call == null) throw new IllegalArgumentException("Null: call");
260 
261             // is method name correct?
262             if (call.getMethod() instanceof ConstantExpression && "buildFromCode".equals(((ConstantExpression) call.getMethod()).getValue())) {
263 
264                 // is method object correct type?
265                 if (call.getObjectExpression() != null && call.getObjectExpression().getType() != null) {
266                     String name = call.getObjectExpression().getType().getName();
267                     if (name != null && !"".equals(name) && factoryTargets.contains(name)) {
268 
269                         // is one of the arguments a closure?
270                         if (call.getArguments() != null && call.getArguments() instanceof TupleExpression) {
271                             if (((TupleExpression) call.getArguments()).getExpressions() != null) {
272                                 for (ASTNode node : ((TupleExpression) call.getArguments()).getExpressions()) {
273                                     if (node instanceof ClosureExpression) {
274                                         return true;
275                                     }
276                                 }
277                             }
278                         }
279                     }
280                 }
281             }
282             return false;
283         }
284 
285         /**
286          * Converts a ClosureExpression into the String source.
287          *
288          * @param expression a closure
289          * @return the source the closure was created from
290          */
291         private String convertClosureToSource(ClosureExpression expression) {
292             try {
293                 return ClosureUtils.convertClosureToSource(source, expression);
294             } catch(Exception e) {
295                 addError(e.getMessage(), expression);
296             }
297             return null;
298         }
299     }
