270     /**
271      * Load the file and then execute it
272      */
273     public void execute() throws BuildException {
274         log.debug("execute()");
275 
276         command = command.trim();
277 
278         if (srcFile == null && command.length() == 0 && filesets.isEmpty()) {
279             throw new BuildException("Source file does not exist!", getLocation());
280         }
281 
282         if (srcFile != null && !srcFile.exists()) {
283             throw new BuildException("Source file does not exist!", getLocation());
284         }
285 
286         // TODO: any of this used?
287         // deal with the filesets
288         for (int i = 0; i < filesets.size(); i++) {
289             FileSet fs = filesets.elementAt(i);
290             DirectoryScanner ds = fs.getDirectoryScanner(getProject());
291             File srcDir = fs.getDir(getProject());
292             String[] srcFiles = ds.getIncludedFiles();
293         }
294 
295         try {
296             PrintStream out = System.out;
297             try {
298                 if (output != null) {
299                     log.verbose("Opening PrintStream to output file " + output);
300                     out = new PrintStream(
301                             new BufferedOutputStream(
302                                     new FileOutputStream(output.getAbsolutePath(), append)));
303                 }
304 
305                 // if there are no groovy statements between the enclosing Groovy tags
306                 // then read groovy statements in from a text file using the src attribute
307                 if (command == null || command.trim().length() == 0) {
308                     createClasspath().add(new Path(getProject(), srcFile.getParentFile().getCanonicalPath()));
309                     command = getText(new BufferedReader(new FileReader(srcFile)));
310                 }
311 
312                 if (command != null) {
313                     execGroovy(command, out);
314                 } else {
315                     throw new BuildException("Source file does not exist!", getLocation());
316                 }
317 
318             } finally {
319                 if (out != null && out != System.out) {
320                     out.close();
321                 }
322             }
323         } catch (IOException e) {
324             throw new BuildException(e, getLocation());
325         }
326 
327         log.verbose("statements executed successfully");
328     }

330     private static String getText(BufferedReader reader) throws IOException {
331         StringBuilder answer = new StringBuilder();
332         // reading the content of the file within a char buffer allow to keep the correct line endings
333         char[] charBuffer = new char[4096];
334         int nbCharRead = 0;
335         while ((nbCharRead = reader.read(charBuffer)) != -1) {
336             // appends buffer
337             answer.append(charBuffer, 0, nbCharRead);
338         }
339         reader.close();
340         return answer.toString();
341     }

347     /**
348      * Read in lines and execute them.
349      *
350      * @param reader the reader from which to get the groovy source to exec
351      * @param out    the outputstream to use
352      * @throws java.io.IOException if something goes wrong
353      */
354     protected void runStatements(Reader reader, PrintStream out)
355             throws IOException {
356         log.debug("runStatements()");
357         StringBuilder txt = new StringBuilder();
358         String line = "";
359         BufferedReader in = new BufferedReader(reader);
360 
361         while ((line = in.readLine()) != null) {
362             line = getProject().replaceProperties(line);
363             if (line.indexOf("--") >= 0) {
364                 txt.append("\n");
365             }
366         }
367         // Catch any statements not followed by ;
368         if (!txt.toString().equals("")) {
369             execGroovy(txt.toString(), out);
370         }
371     }

373     /**
374      * Exec the statement.
375      *
376      * @param txt the groovy source to exec
377      * @param out not used?
378      */
379     protected void execGroovy(final String txt, final PrintStream out) {
380         log.debug("execGroovy()");
381 
382         // Check and ignore empty statements
383         if ("".equals(txt.trim())) {
384             return;
385         }
386 
387         log.verbose("Script: " + txt);
388         if (classpath != null) {
389             log.debug("Explicit Classpath: " + classpath.toString());
390         }
391 
392         if (fork) {
393             log.debug("Using fork mode");
394             try {
395                 createClasspathParts();
396                 createNewArgs(txt);
397                 super.setFork(fork);
398                 super.setClassname(useGroovyShell ? "groovy.lang.GroovyShell" : "org.codehaus.groovy.ant.Groovy");
399                 configureCompiler();
400                 super.execute();
401             } catch (Exception e) {
402                 StringWriter writer = new StringWriter();
403                 new ErrorReporter(e, false).write(new PrintWriter(writer));
404                 String message = writer.toString();
405                 throw new BuildException("Script Failed: " + message, e, getLocation());
406             }
407             return;
408         }
409 
410         Object mavenPom = null;
411         final Project project = getProject();
412         final ClassLoader baseClassLoader;
413         ClassLoader savedLoader = null;
414         final Thread thread = Thread.currentThread();
415         boolean maven = "org.apache.commons.grant.GrantProject".equals(project.getClass().getName());
416         // treat the case Ant is run through Maven, and
417         if (maven) {
418             if (contextClassLoader) {
419                 throw new BuildException("Using setContextClassLoader not permitted when using Maven.", getLocation());
420             }
421             try {
422                 final Object propsHandler = project.getClass().getMethod("getPropsHandler").invoke(project);
423                 final Field contextField = propsHandler.getClass().getDeclaredField("context");
424                 contextField.setAccessible(true);
425                 final Object context = contextField.get(propsHandler);
426                 mavenPom = InvokerHelper.invokeMethod(context, "getProject", EMPTY_OBJECT_ARRAY);
427             }
428             catch (Exception e) {
429                 throw new BuildException("Impossible to retrieve Maven's Ant project: " + e.getMessage(), getLocation());
430             }
431             // load groovy into "root.maven" classloader instead of "root" so that
432             // groovy script can access Maven classes
433             baseClassLoader = mavenPom.getClass().getClassLoader();
434         } else {
435             baseClassLoader = GroovyShell.class.getClassLoader();
436         }
437         if (contextClassLoader || maven) {
438             savedLoader = thread.getContextClassLoader();
439             thread.setContextClassLoader(GroovyShell.class.getClassLoader());
440         }
441 
442         final String scriptName = computeScriptName();
443         final GroovyClassLoader classLoader = new GroovyClassLoader(baseClassLoader);
444         addClassPathes(classLoader);
445         configureCompiler();
446         final GroovyShell groovy = new GroovyShell(classLoader, new Binding(), configuration);
447         try {
448             parseAndRunScript(groovy, txt, mavenPom, scriptName, null, new AntBuilder(this));
449         } finally {
450             groovy.resetLoadedClasses();
451             groovy.getClassLoader().clearCache();
452             if (contextClassLoader || maven) thread.setContextClassLoader(savedLoader);
453         }
454     }

456     private void configureCompiler() {
457         if (scriptBaseClass!=null) {
458             configuration.setScriptBaseClass(scriptBaseClass);
459         }
460         if (indy) {
461             configuration.getOptimizationOptions().put("indy", Boolean.TRUE);
462             configuration.getOptimizationOptions().put("int", Boolean.FALSE);
463         }
464         if (configscript!=null) {
465             Binding binding = new Binding();
466             binding.setVariable("configuration", configuration);
467 
468             CompilerConfiguration configuratorConfig = new CompilerConfiguration();
469             ImportCustomizer customizer = new ImportCustomizer();
470             customizer.addStaticStars("org.codehaus.groovy.control.customizers.builder.CompilerCustomizationBuilder");
471             configuratorConfig.addCompilationCustomizers(customizer);
472 
473             GroovyShell shell = new GroovyShell(binding, configuratorConfig);
474             File confSrc = new File(configscript);
475             try {
476                 shell.evaluate(confSrc);
477             } catch (IOException e) {
478                 throw new BuildException("Unable to configure compiler using configuration file: "+confSrc, e);
479             }
480         }
481     }

483     private void parseAndRunScript(GroovyShell shell, String txt, Object mavenPom, String scriptName, File scriptFile, AntBuilder builder) {
484         try {
485             final Script script;
486             if (scriptFile != null) {
487                 script = shell.parse(scriptFile);
488             } else {
489                 script = shell.parse(txt, scriptName);
490             }
491             final Project project = getProject();
492             script.setProperty("ant", builder);
493             script.setProperty("project", project);
494             script.setProperty("properties", new AntProjectPropertiesDelegate(project));
495             script.setProperty("target", getOwningTarget());
496             script.setProperty("task", this);
497             script.setProperty("args", cmdline.getCommandline());
498             if (mavenPom != null) {
499                 script.setProperty("pom", mavenPom);
500             }
501             script.run();
502         }
503         catch (final MissingMethodException mme) {
504             // not a script, try running through run method but properties will not be available
505             if (scriptFile != null) {
506                 try {
507                     shell.run(scriptFile, cmdline.getCommandline());
508                 } catch (IOException e) {
509                     processError(e);
510                 }
511             } else {
512                 shell.run(txt, scriptName, cmdline.getCommandline());
513             }
514         }
515         catch (final CompilationFailedException e) {
516             processError(e);
517         } catch (IOException e) {
518             processError(e);
519         }
520     }

522     private void processError(Exception e) {
523         StringWriter writer = new StringWriter();
524         new ErrorReporter(e, false).write(new PrintWriter(writer));
525         String message = writer.toString();
526         throw new BuildException("Script Failed: " + message, e, getLocation());
527     }

529     public static void main(String[] args) {
530         final GroovyShell shell = new GroovyShell(new Binding());
531         final Groovy groovy = new Groovy();
532         for (int i = 1; i < args.length; i++) {
533             final Commandline.Argument argument = groovy.createArg();
534             argument.setValue(args[i]);
535         }
536         final AntBuilder builder = new AntBuilder();
537         groovy.setProject(builder.getProject());
538         groovy.parseAndRunScript(shell, null, null, null, new File(args[0]), builder);
539     }

541     private void createClasspathParts() {
542         Path path;
543         if (classpath != null) {
544             path = super.createClasspath();
545             path.setPath(classpath.toString());
546         }
547 
548         if (includeAntRuntime) {
549             path = super.createClasspath();
550             path.setPath(System.getProperty("java.class.path"));
551         }
552         String groovyHome = null;
553         final String[] strings = getSysProperties().getVariables();
554         if (strings != null) {
555             for (String prop : strings) {
556                 if (prop.startsWith("-Dgroovy.home=")) {
557                     groovyHome = prop.substring("-Dgroovy.home=".length());
558                 }
559             }
560         }
561         if (groovyHome == null) {
562             groovyHome = System.getProperty("groovy.home");
563         }
564         if (groovyHome == null) {
565             groovyHome = System.getenv("GROOVY_HOME");
566         }
567         if (groovyHome == null) {
568             throw new IllegalStateException("Neither ${groovy.home} nor GROOVY_HOME defined.");
569         }
570         File jarDir = new File(groovyHome, "embeddable");
571         if (!jarDir.exists()) {
572             throw new IllegalStateException("GROOVY_HOME incorrectly defined. No embeddable directory found in: " + groovyHome);
573         }
574         final File[] files = jarDir.listFiles();
575         for (File file : files) {
576             try {
577                 log.debug("Adding jar to classpath: " + file.getCanonicalPath());
578             } catch (IOException e) {
579                 // ignore
580             }
581             path = super.createClasspath();
582             path.setLocation(file);
583         }
584     }

586     private void createNewArgs(String txt) throws IOException {
587         final String[] args = cmdline.getCommandline();
588         // Temporary file - delete on exit, create (assured unique name).
589         final File tempFile = FileUtils.getFileUtils().createTempFile(PREFIX, SUFFIX, null, true, true);
590         final String[] commandline = new String[args.length + 1];
591         ResourceGroovyMethods.write(tempFile, txt);
592         commandline[0] = tempFile.getCanonicalPath();
593         System.arraycopy(args, 0, commandline, 1, args.length);
594         super.clearArgs();
595         for (String arg : commandline) {
596             final Commandline.Argument argument = super.createArg();
597             argument.setValue(arg);
598         }
599     }

601     /**
602      * Try to build a script name for the script of the groovy task to have an helpful value in stack traces in case of exception
603      *
604      * @return the name to use when compiling the script
605      */
606     private String computeScriptName() {
607         if (srcFile != null) {
608             return srcFile.getAbsolutePath();
609         } else {
610             String name = PREFIX;
611             if (getLocation().getFileName().length() > 0)
612                 name += getLocation().getFileName().replaceAll("[^\\w_\\.]", "_").replaceAll("[\\.]", "_dot_");
613             else
614                 name += SUFFIX;
615 
616             return name;
617         }
618     }
