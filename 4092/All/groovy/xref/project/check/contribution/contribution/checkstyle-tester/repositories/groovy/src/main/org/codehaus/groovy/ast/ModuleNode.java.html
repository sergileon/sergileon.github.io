91      public List<ClassNode> getClasses() {
92          if (createClassForStatements && (!statementBlock.isEmpty() || !methods.isEmpty() || isPackageInfo())) {
93              ClassNode mainClass = createStatementsClass();
94              mainClassName = mainClass.getName(); 
95              createClassForStatements = false;
96              classes.add(0, mainClass);
97              mainClass.setModule(this);
98              addToCompileUnit(mainClass);
99          }
100         return classes;
101     }

235     public ClassNode getScriptClassDummy() {
236         if (scriptDummy!=null) {
237             setScriptBaseClassFromConfig(scriptDummy);
238             return scriptDummy;
239         }
240         
241         String name = getPackageName();
242         if (name == null) {
243             name = "";
244         }
245         // now let's use the file name to determine the class name
246         if (getDescription() == null) {
247             throw new RuntimeException("Cannot generate main(String[]) class for statements when we have no file description");
248         }
249         name += GeneratorContext.encodeAsValidClassName(extractClassFromFileDescription());
250 
251         ClassNode classNode;
252         if (isPackageInfo()) {
253             classNode = new ClassNode(name, ACC_ABSTRACT | ACC_INTERFACE, ClassHelper.OBJECT_TYPE);
254         } else {
255             classNode = new ClassNode(name, ACC_PUBLIC, ClassHelper.SCRIPT_TYPE);
256             setScriptBaseClassFromConfig(classNode);
257             classNode.setScript(true);
258             classNode.setScriptBody(true);
259         }
260 
261         scriptDummy = classNode;
262         return classNode;
263     }
264 
265     private void setScriptBaseClassFromConfig(ClassNode cn) {
266         String baseClassName = null;
267         if (unit != null) {
268             baseClassName = unit.getConfig().getScriptBaseClass();
269         } else if (context != null) {
270             baseClassName = context.getConfiguration().getScriptBaseClass();
271         }
272         if (baseClassName != null) {
273             if (!cn.getSuperClass().getName().equals(baseClassName)) {
274                 cn.setSuperClass(ClassHelper.make(baseClassName));
275                 AnnotationNode annotationNode = new AnnotationNode(BaseScriptASTTransformation.MY_TYPE);
276                 cn.addAnnotation(annotationNode);
277             }
278         }
279     }

281     protected ClassNode createStatementsClass() {
282         ClassNode classNode = getScriptClassDummy();
283         if (classNode.getName().endsWith("package-info")) {
284             return classNode;
285         }
286         
287         handleMainMethodIfPresent(methods);
288 
289         // return new Foo(new ShellContext(args)).run()
290         classNode.addMethod(
291             new MethodNode(
292                 "main",
293                 ACC_PUBLIC | ACC_STATIC,
294                 ClassHelper.VOID_TYPE,
295                 new Parameter[] { new Parameter(ClassHelper.STRING_TYPE.makeArray(), "args")},
296                 ClassNode.EMPTY_ARRAY,
297                 new ExpressionStatement(
298                     new MethodCallExpression(
299                         new ClassExpression(ClassHelper.make(InvokerHelper.class)),
300                         "runScript",
301                         new ArgumentListExpression(
302                                 new ClassExpression(classNode),
303                                 new VariableExpression("args"))))));
304 
305         MethodNode methodNode = new MethodNode("run", ACC_PUBLIC, ClassHelper.OBJECT_TYPE, Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, statementBlock);
306         methodNode.setIsScriptBody();
307         classNode.addMethod(methodNode);
308 
309         classNode.addConstructor(ACC_PUBLIC, Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, new BlockStatement());
310 
311         Statement stmt;
312         // A script's contextual constructor should call it's super class' contextual constructor, if it has one.
313         // In practice this will always be true because currently this visitor is run before the AST transformations
314         // (like @BaseScript) that could change this.  But this is cautious and anticipates possible compiler changes.
315         if (classNode.getSuperClass().getDeclaredConstructor(SCRIPT_CONTEXT_CTOR) != null) {
316             stmt = new ExpressionStatement(
317                     new ConstructorCallExpression(ClassNode.SUPER,
318                             new ArgumentListExpression(
319                                     new VariableExpression("context"))));
320         } else {
321             // Fallback for non-standard base "script" classes with no context (Binding) constructor.
322             stmt = new ExpressionStatement(
323                     new MethodCallExpression(
324                             new VariableExpression("super"),
325                             "setBinding",
326                             new ArgumentListExpression(
327                                     new VariableExpression("context"))));
328         }
329 
330         classNode.addConstructor(
331             ACC_PUBLIC,
332             new Parameter[] { new Parameter(ClassHelper.make(Binding.class), "context")},
333             ClassNode.EMPTY_ARRAY,
334             stmt);
335 
336         for (MethodNode node : methods) {
337             int modifiers = node.getModifiers();
338             if ((modifiers & ACC_ABSTRACT) != 0) {
339                 throw new RuntimeException(
340                     "Cannot use abstract methods in a script, they are only available inside classes. Method: "
341                         + node.getName());
342             }
343             // br: the old logic seems to add static to all def f().... in a script, which makes enclosing
344             // inner classes (including closures) in a def function difficult. Comment it out.
345             node.setModifiers(modifiers /*| ACC_STATIC*/);
346 
347             classNode.addMethod(node);
348         }
349         return classNode;
350     }

352     /*
353      * If a main method is provided by user, account for it under run() as scripts generate their own 'main' so they can run.  
354      */
355     private void handleMainMethodIfPresent(List methods) {
356         boolean found = false;
357         for (Iterator iter = methods.iterator(); iter.hasNext();) {
358             MethodNode node = (MethodNode) iter.next();
359             if(node.getName().equals("main")) {
360                 if (node.isStatic() && node.getParameters().length == 1) {
361                     boolean retTypeMatches, argTypeMatches;
362                     ClassNode argType = node.getParameters()[0].getType();
363                     ClassNode retType = node.getReturnType();
364 
365                     argTypeMatches = (argType.equals(ClassHelper.OBJECT_TYPE) || argType.getName().contains("String[]"));
366                     retTypeMatches = (retType == ClassHelper.VOID_TYPE || retType == ClassHelper.OBJECT_TYPE);
367                     
368                     if(retTypeMatches && argTypeMatches) {
369                         if(found) {
370                             throw new RuntimeException("Repetitive main method found.");
371                         } else {
372                             found = true;
373                         }
374                         // if script has both loose statements as well as main(), then main() is ignored
375                         if(statementBlock.isEmpty()) {
376                             addStatement(node.getCode());
377                         }
378                         iter.remove();
379                     }
380                 }
381             }
382         }
383     }

385     protected String extractClassFromFileDescription() {
386         String answer = getDescription();
387         try {
388             URI uri = new URI(answer);
389             String path = uri.getPath();
390             String schemeSpecific = uri.getSchemeSpecificPart();
391             if (path!=null) {
392                 answer = path;
393             } else if (schemeSpecific!=null) {
394                 answer = schemeSpecific;
395             }
396         } catch (URISyntaxException e) {}
397         // let's strip off everything after the last '.'
398         int slashIdx = answer.lastIndexOf('/');
399         int separatorIdx = answer.lastIndexOf(File.separatorChar);
400         int dotIdx = answer.lastIndexOf('.');
401         if (dotIdx > 0 && dotIdx > Math.max(slashIdx, separatorIdx)) {
402             answer = answer.substring(0, dotIdx);
403         }
404         // new let's strip everything up to and including the path separators
405         if (slashIdx >= 0) {
406             answer = answer.substring(slashIdx + 1);
407         }
408         // recalculate in case we have already done some stripping
409         separatorIdx = answer.lastIndexOf(File.separatorChar);
410         if (separatorIdx >= 0) {
411             answer = answer.substring(separatorIdx + 1);
412         }
413         return answer;
414     }

418     }

420     public void sortClasses(){
421         if (isEmpty()) return;
422         List<ClassNode> classes = getClasses();
423         LinkedList<ClassNode> sorted = new LinkedList<ClassNode>();
424         int level=1;
425         while (!classes.isEmpty()) {
426             for (Iterator<ClassNode> cni = classes.iterator(); cni.hasNext();) {
427                 ClassNode cn = cni.next();
428                 ClassNode sn = cn;
429                 for (int i=0; sn!=null && i<level; i++) sn = sn.getSuperClass();
430                 if (sn!=null && sn.isPrimaryClassNode()) continue;
431                 cni.remove();
432                 sorted.addLast(cn);
433             }
434             level++;
435         }
436         this.classes = sorted;
437     }
