37          Map<String, TriggerBinding> result = new HashMap<String, TriggerBinding>();
38          result.put(JTable.class.getName() + "#elements",
39                  new TriggerBinding() {
40                      public FullBinding createBinding(SourceBinding source, TargetBinding target) {
41                          return new JTableElementsBinding((PropertyBinding) source, target);
42                      }
43                  });
44          result.put(JTable.class.getName() + "#selectedElement",
45                  new TriggerBinding() {
46                      public FullBinding createBinding(SourceBinding source, TargetBinding target) {
47                          return new JTableSelectedElementBinding((PropertyBinding) source, target, "selectedElement");
48                      }
49                  });
50          result.put(JTable.class.getName() + "#selectedElements",
51                  new TriggerBinding() {
52                      public FullBinding createBinding(SourceBinding source, TargetBinding target) {
53                          return new JTableSelectedElementBinding((PropertyBinding) source, target, "selectedElements");
54                      }
55                  });
56          return result;
57      }

60  class JTableElementsBinding extends AbstractSyntheticBinding implements TableModelListener, PropertyChangeListener {
61      JTable boundTable;
62  
63      public JTableElementsBinding(PropertyBinding propertyBinding, TargetBinding target) {
64          super(propertyBinding, target, JTable.class, "elements");
65      }
66  
67      protected void syntheticBind() {
68          boundTable = (JTable) ((PropertyBinding)sourceBinding).getBean();
69          boundTable.addPropertyChangeListener("model", this);
70          boundTable.getModel().addTableModelListener(this);
71      }
72  
73      protected void syntheticUnbind() {
74          boundTable.removePropertyChangeListener("model", this);
75          boundTable.getModel().removeTableModelListener(this);
76      }
77  
78      public void tableChanged(TableModelEvent e) {
79          update();
80      }
81  
82      public void propertyChange(PropertyChangeEvent event) {
83          update();
84          ((TableModel) event.getOldValue()).removeTableModelListener(this);
85          ((TableModel) event.getNewValue()).addTableModelListener(this);
86      }
87  }

89  class JTableSelectedElementBinding extends AbstractSyntheticBinding implements PropertyChangeListener, ListSelectionListener {
90      JTable boundTable;
91  
92      protected JTableSelectedElementBinding(PropertyBinding source, TargetBinding target, String propertyName) {
93          super(source, target, JTable.class, propertyName);
94      }
95  
96      public synchronized void syntheticBind() {
97          boundTable = (JTable) ((PropertyBinding)sourceBinding).getBean();
98          boundTable.addPropertyChangeListener("selectionModel", this);
99          boundTable.getSelectionModel().addListSelectionListener(this);
100     }
101 
102     public synchronized void syntheticUnbind() {
103         boundTable.removePropertyChangeListener("selectionModel", this);
104         boundTable.getSelectionModel().removeListSelectionListener(this);
105         boundTable = null;
106     }
107 
108     public void propertyChange(PropertyChangeEvent event) {
109         update();
110         ((ListSelectionModel) event.getOldValue()).removeListSelectionListener(this);
111         ((ListSelectionModel) event.getNewValue()).addListSelectionListener(this);
112     }
113 
114     public void valueChanged(ListSelectionEvent e) {
115         update();
116     }
