40  /**
41   * <p>All templates compiled through {@link groovy.text.markup.MarkupTemplateEngine} extend this abstract class,
42   * which provides a number of utility methods to generate markup. An instance of this class can be obtained
43   * after calling {@link groovy.text.Template#make()} or {@link groovy.text.Template#make(java.util.Map)})} on
44   * a template generated by {@link groovy.text.markup.MarkupTemplateEngine#createTemplate(java.io.Reader)}.</p>
45   *
46   * <p>It is advised to use a distinct template instance for each thread (or more simply, each rendered document)
47   * for thread safety and avoiding mixing models.</p>
48   *
49   * <p>For the application needs, it is possible to provide more helper methods by extending this class and
50   * configuring the base template class using the {@link groovy.text.markup.TemplateConfiguration#setBaseTemplateClass(Class)}
51   * method.</p>
52   *
53   * @author Cedric Champeau
54   */
55  public abstract class BaseTemplate implements Writable {
56      private static final Map EMPTY_MODEL = Collections.emptyMap();
57  
58      private final Map model;
59      private final Map<String,String> modelTypes;
60      private final MarkupTemplateEngine engine;
61      private final TemplateConfiguration configuration;
62      private final Map<String, Template> cachedFragments;
63  
64      private Writer out;
65      private boolean doWriteIndent;
66  
67      public BaseTemplate(final MarkupTemplateEngine templateEngine, final Map model, final Map<String,String> modelTypes, final TemplateConfiguration configuration) {
68          this.model = model==null?EMPTY_MODEL:model;
69          this.engine = templateEngine;
70          this.configuration = configuration;
71          this.modelTypes = modelTypes;
72          this.cachedFragments = new LinkedHashMap<String, Template>();
73      }
74  
75      public Map getModel() {
76          return model;
77      }

81      /**
82       * Renders the object provided as parameter using its {@link Object#toString()} method,
83       * The contents is rendered as is, unescaped. This means that depending on what the
84       * {@link Object#toString()} method call returns, you might create invalid markup.
85       * @param obj the object to be rendered unescaped
86       * @return this template instance
87       * @throws IOException
88       */
89      public BaseTemplate yieldUnescaped(Object obj) throws IOException {
90          writeIndent();
91          out.write(obj.toString());
92          return this;
93      }

95      /**
96       * Renders the object provided as parameter using its {@link Object#toString()} method,
97       * The contents is rendered after being escaped for XML, enforcing valid XML output.
98       * @param obj the object to be rendered
99       * @return this template instance
100      * @throws IOException
101      */
102     public BaseTemplate yield(Object obj) throws IOException {
103         writeIndent();
104         out.write(escapeXml(obj.toString()));
105         return this;
106     }

108     public String stringOf(Closure cl) throws IOException {
109         Writer old = out;
110         StringWriter stringWriter = new StringWriter(32);
111         out = stringWriter;
112         Object result = cl.call();
113         if (result!=null && result!=this) {
114             stringWriter.append(result.toString());
115         }
116         out = old;
117         return stringWriter.toString();
118     }

121     /**
122      * Renders the supplied object using its {@link Object#toString} method inside a
123      * comment markup block (&lt;!-- ... --&gt;). The object is rendered as is, unescaped.
124      * @param cs the object to be rendered inside an XML comment block.
125      * @return this template instance.
126      * @throws IOException
127      */
128     public BaseTemplate comment(Object cs) throws IOException {
129         writeIndent();
130         out.write("<!--");
131         out.write(cs.toString());
132         out.write("-->");
133         return this;
134     }

136     /**
137      * Renders an XML declaration header. If the declaration encoding is set in the
138      * {@link TemplateConfiguration#getDeclarationEncoding() template configuration},
139      * then the encoding is rendered into the declaration.
140      * @return this template instance
141      * @throws IOException
142      */
143     public BaseTemplate xmlDeclaration() throws IOException {
144         out.write("<?xml ");
145         writeAttribute("version", "1.0");
146         if (configuration.getDeclarationEncoding() != null) {
147             writeAttribute(" encoding", configuration.getDeclarationEncoding());
148         }
149         out.write("?>");
150         out.write(configuration.getNewLineString());
151         return this;
152     }

154     /**
155      * <p>Renders processing instructions. The supplied map contains all elements to be
156      * rendered as processing instructions. The key is the name of the element, the value
157      * is either a map of attributes, or an object to be rendered directly. For example:</p>
158      * <code>
159      *     pi("xml-stylesheet":[href:"mystyle.css", type:"text/css"])
160      * </code>
161      *
162      * <p>will be rendered as:</p>
163      *
164      * <pre>
165      *     &lt;?xml-stylesheet href='mystyle.css' type='text/css'?&gt;
166      * </pre>
167      *
168      * @param attrs the attributes to render
169      * @return this template instance
170      * @throws IOException
171      */
172     public BaseTemplate pi(Map<?, ?> attrs) throws IOException {
173         for (Map.Entry<?, ?> entry : attrs.entrySet()) {
174             Object target = entry.getKey();
175             Object instruction = entry.getValue();
176             out.write("<?");
177             if (instruction instanceof Map) {
178                 out.write(target.toString());
179                 writeAttributes((Map) instruction);
180             } else {
181                 out.write(target.toString());
182                 out.write(" ");
183                 out.write(instruction.toString());
184             }
185             out.write("?>");
186             out.write(configuration.getNewLineString());
187         }
188         return this;
189     }
190 
191     private void writeAttribute(String attName, String value) throws IOException {
192         out.write(attName);
193         out.write("=");
194         writeQt();
195         out.write(escapeQuotes(value));
196         writeQt();
197     }
198 
199     private void writeQt() throws IOException {
200         if (configuration.isUseDoubleQuotes()) {
201             out.write('"');
202         } else {
203             out.write('\'');
204         }
205     }
206 
207     private void writeIndent() throws IOException {
208         if (out instanceof DelegatingIndentWriter && doWriteIndent) {
209             ((DelegatingIndentWriter)out).writeIndent();
210             doWriteIndent = false;
211         }
212     }
213 
214     private String escapeQuotes(String str) {
215         String quote = configuration.isUseDoubleQuotes() ? "\"" : "'";
216         String escape = configuration.isUseDoubleQuotes() ? "&quote;" : "&apos;";
217         return str.replace(quote, escape);
218     }

220     /**
221      * This is the main method responsible for writing a tag and its attributes.
222      * The arguments may be:
223      * <ul>
224      *     <li>a closure</li> in which case the closure is rendered inside the tag body
225      *     <li>a string</li>, in which case the string is rendered as the tag body
226      *     <li>a map of attributes</li> in which case the attributes are rendered inside the opening tag
227      * </ul>
228      * <p>or a combination of (attributes,string), (attributes,closure)</p>
229      * @param tagName the name of the tag
230      * @param args tag generation arguments
231      * @return this template instance
232      * @throws IOException
233      */
234     public Object methodMissing(String tagName, Object args) throws IOException {
235         Object o = model.get(tagName);
236         if (o instanceof Closure) {
237             if (args instanceof Object[]) {
238                 yieldUnescaped(((Closure) o).call((Object[])args));
239                 return this;
240             }
241             yieldUnescaped(((Closure) o).call(args));
242             return this;
243         } else if (args instanceof Object[]) {
244             final Writer wrt = out;
245             TagData tagData = new TagData(args).invoke();
246             Object body = tagData.getBody();
247             writeIndent();
248             wrt.write('<');
249             wrt.write(tagName);
250             writeAttributes(tagData.getAttributes());
251             if (body != null) {
252                 wrt.write('>');
253                 writeBody(body);
254                 writeIndent();
255                 wrt.write("</");
256                 wrt.write(tagName);
257                 wrt.write('>');
258             } else {
259                 if (configuration.isExpandEmptyElements()) {
260                     wrt.write("></");
261                     wrt.write(tagName);
262                     wrt.write('>');
263                 } else {
264                     wrt.write("/>");
265                 }
266             }
267         }
268         return this;
269     }
270 
271     private void writeBody(final Object body) throws IOException {
272         boolean indent = out instanceof DelegatingIndentWriter;
273         if (body instanceof Closure) {
274             if (indent) {
275                 ((DelegatingIndentWriter)(out)).next();
276             }
277             ((Closure) body).call();
278             if (indent) {
279                 ((DelegatingIndentWriter)(out)).previous();
280             }
281         } else {
282             out.write(body.toString());
283         }
284     }
285 
286     private void writeAttributes(final Map<?, ?> attributes) throws IOException {
287         if (attributes == null) {
288             return;
289         }
290         final Writer wrt = out;
291         for (Map.Entry entry : attributes.entrySet()) {
292             wrt.write(' ');
293             String attName = entry.getKey().toString();
294             String value = entry.getValue() == null ? "" : entry.getValue().toString();
295             writeAttribute(attName, value);
296         }
297     }

334      * Escapes the string representation of the supplied object if it derives from {@link java.lang.CharSequence},
335      * otherwise returns the object itself.
336      * @param contents an object to be escaped for XML
337      * @return  an escaped string, or the object itself

340         if (contents instanceof CharSequence) {
341             return escapeXml(contents.toString());

365     /**
366      * Renders an embedded template as a fragment. Fragments are cached in a template, meaning that
367      * if you use the same fragment in a template, it will only be compiled once, but once <b>per template
368      * instance</b>. This is less performant than using {@link #layout(java.util.Map, String)}.
369      *
370      * @param model model to be passed to the template
371      * @param templateText template body
372      * @return this template instance
373      * @throws IOException
374      * @throws ClassNotFoundException
375      */
376     public Object fragment(Map model, String templateText) throws IOException, ClassNotFoundException {
377         Template template = cachedFragments.get(templateText);
378         if (template==null) {
379             template = engine.createTemplate(new StringReader(templateText));
380             cachedFragments.put(templateText, template);
381         }
382         template.make(model).writeTo(out);
383         return this;
384     }

386     /**
387      * Imports a template and renders it using the specified model, allowing fine grained composition
388      * of templates and layouting. This works similarily to a template include but allows a distinct
389      * model to be used. This version doesn't inherit the model from the parent. If you need model
390      * inheritance, see {@link #layout(java.util.Map, String, boolean)}.
391      * @param model model to be passed to the template
392      * @param templateName the name of the template to be used as a layout
393      * @return this template instance
394      * @throws IOException
395      * @throws ClassNotFoundException
396      */
397     public Object layout(Map model, String templateName) throws IOException, ClassNotFoundException {
398         return layout(model, templateName, false);
399     }

401     /**
402      * Imports a template and renders it using the specified model, allowing fine grained composition of templates and
403      * layouting. This works similarily to a template include but allows a distinct model to be used. If the layout
404      * inherits from the parent model, a new model is created, with the values from the parent model, eventually
405      * overriden with those provided specifically for this layout.
406      *
407      * @param model        model to be passed to the template
408      * @param templateName the name of the template to be used as a layout
409      * @param inheritModel a boolean indicating if we should inherit the parent model
410      * @return this template instance
411      * @throws IOException
412      * @throws ClassNotFoundException
413      */
414     public Object layout(Map model, String templateName, boolean inheritModel) throws IOException, ClassNotFoundException {
415         Map submodel = inheritModel ? forkModel(model) : model;
416         URL resource = engine.resolveTemplate(templateName);
417         engine.createTypeCheckedModelTemplate(resource, modelTypes).make(submodel).writeTo(out);
418         return this;
419     }
420 
421     @SuppressWarnings("unchecked")
422     private Map forkModel(Map m) {
423         Map result = new HashMap();
424         result.putAll(model);
425         result.putAll(m);
426         return result;
427     }

429     /**
430      * Wraps a closure so that it can be used as a prototype for inclusion in layouts. This is useful when
431      * you want to use a closure in a model, but that you don't want to render the result of the closure but instead
432      * call it as if it was a specification of a template fragment.
433      * @param cl the fragment to be wrapped
434      * @return a wrapped closure returning an empty string
435      */
436     public Closure contents(final Closure cl) {
437         return new Closure(cl.getOwner(), cl.getThisObject()) {
438             @Override
439             public Object call() {
440                 cl.call();
441                 return "";
442             }
443 
444             @Override
445             public Object call(final Object... args) {
446                 cl.call(args);
447                 return "";
448             }
449 
450             @Override
451             public Object call(final Object arguments) {
452                 cl.call(arguments);
453                 return "";
454             }
455         };
456     }

458     /**
459      * Main method used to render a template.
460      * @param out the Writer to which this Writable should output its data.
461      * @return a writer instance
462      * @throws IOException
463      */
464     public Writer writeTo(final Writer out) throws IOException {
465         if (this.out!=null) {
466             // StackOverflow prevention
467             return NullWriter.DEFAULT;
468         }
469         try {
470             this.out = createWriter(out);
471             run();
472             return out;
473         } finally {
474             if (this.out!=null) {
475                 this.out.flush();
476             }
477             this.out = null;
478         }
479     }
480 
481     private Writer createWriter(final Writer out) {
482         return configuration.isAutoIndent() && !(out instanceof DelegatingIndentWriter)?new DelegatingIndentWriter(out, configuration.getAutoIndentString()):out;
483     }

485     private class TagData {
486         private final Object[] array;
487         private Map attributes;
488         private Object body;
489 
490         public TagData(final Object args) {
491             this.array = (Object[])args;
492         }
493 
494         public Map getAttributes() {
495             return attributes;
496         }
497 
498         public Object getBody() {
499             return body;
500         }
501 
502         public TagData invoke() {
503             attributes = null;
504             body = null;
505             for (Object o : array) {
506                 if (o instanceof Map) {
507                     attributes = (Map) o;
508                 } else {
509                     body = o;
510                 }
511             }
512             return this;
513         }
514     }
515 
516     public String toString() {
517         StringWriter wrt = new StringWriter(512);
518         try {
519             writeTo(wrt);
520         } catch (IOException e) {
521             ExceptionUtils.sneakyThrow(e);
522         }
523         return wrt.toString();
524     }
