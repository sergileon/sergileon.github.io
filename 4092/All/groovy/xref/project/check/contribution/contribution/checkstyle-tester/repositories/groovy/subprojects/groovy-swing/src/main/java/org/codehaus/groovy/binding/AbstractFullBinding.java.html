19  package org.codehaus.groovy.binding;
20  
21  import groovy.lang.Closure;
22  
23  /**
24   * @author <a href="mailto:shemnon@yahoo.com">Danno Ferrin</a>
25   * @since Groovy 1.1
26   */
27  public abstract class AbstractFullBinding  implements FullBinding {
28  
29      protected SourceBinding sourceBinding;
30      protected TargetBinding targetBinding;
31      protected Closure validator;
32      protected Closure converter;
33      protected Closure reverseConverter;
34  
35      private void fireBinding() {
36          if ((sourceBinding == null) || (targetBinding == null)) {
37              // should we throw invalid binding exception?  or fail quietly?
38              return;
39          }
40          Object result = sourceBinding.getSourceValue();
41          if (getValidator() != null) {
42              Object validation = getValidator().call(result);
43              if ((validation == null)
44                  || ((validation instanceof Boolean) && !(Boolean) validation))
45              {
46                  // should we throw a validation failed exception?  or fail quietly?
47                  return;
48              }
49          }
50          if (getConverter() != null) {
51              result = getConverter().call(result);
52          }
53          targetBinding.updateTargetValue(result);
54      }
55  
56      public void update() {
57          fireBinding();
58      }
59  
60      private void fireReverseBinding() {
61          if (!(sourceBinding instanceof TargetBinding) || !(targetBinding instanceof SourceBinding)) {
62              throw new RuntimeException("Binding Instance is not reversable");
63          }
64          Object result = ((SourceBinding)targetBinding).getSourceValue();
65          if (getReverseConverter() != null) {
66              result = getReverseConverter().call(result);
67          }
68          ((TargetBinding)sourceBinding).updateTargetValue(result);
69      }
70  
71      public void reverseUpdate() {
72          fireReverseBinding();
73      }
74  
75      public SourceBinding getSourceBinding() {
76          return sourceBinding;
77      }
78  
79      public void setSourceBinding(SourceBinding sourceBinding) {
80          this.sourceBinding = sourceBinding;
81      }
82  
83      public TargetBinding getTargetBinding() {
84          return targetBinding;
85      }
86  
87      public void setTargetBinding(TargetBinding targetBinding) {
88          this.targetBinding = targetBinding;
89      }
90  
91      public Closure getValidator() {
92          return validator;
93      }
94  
95      public void setValidator(Closure validator) {
96          this.validator = validator;
97      }
98  
99      public Closure getConverter() {
100         return converter;
101     }
102 
103     public void setConverter(Closure converter) {
104         this.converter = converter;
105     }
106 
107     public Closure getReverseConverter() {
108         return reverseConverter;
109     }
110 
111     public void setReverseConverter(Closure reverseConverter) {
112         this.reverseConverter = reverseConverter;
113     }
114 }
