60  /**
61   * Handles generation of code for the {@code @Vetoable} annotation, and {@code @Bindable}
62   * if also present.
63   * <p>
64   * Generally, it adds (if needed) a VetoableChangeSupport field and
65   * the needed add/removeVetoableChangeListener methods to support the
66   * listeners.
67   * <p>
68   * It also generates the setter and wires the setter through the
69   * VetoableChangeSupport.
70   * <p>
71   * If a {@link Bindable} annotation is detected it also adds support similar
72   * to what {@link BindableASTTransformation} would do.
73   *
74   * @author Danno Ferrin (shemnon)
75   * @author Chris Reeves
76   */
77  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
78  public class VetoableASTTransformation extends BindableASTTransformation {
79  
80      protected static ClassNode constrainedClassNode = ClassHelper.make(Vetoable.class);
81  
82      /**
83       * Convenience method to see if an annotated node is {@code @Vetoable}.
84       *
85       * @param node the node to check
86       * @return true if the node is constrained
87       */
88      public static boolean hasVetoableAnnotation(AnnotatedNode node) {
89          for (AnnotationNode annotation : node.getAnnotations()) {
90              if (constrainedClassNode.equals(annotation.getClassNode())) {
91                  return true;
92              }
93          }
94          return false;
95      }
96  
97      /**
98       * Handles the bulk of the processing, mostly delegating to other methods.
99       *
100      * @param nodes   the AST nodes
101      * @param source  the source unit for the nodes
102      */
103     public void visit(ASTNode[] nodes, SourceUnit source) {
104         if (!(nodes[0] instanceof AnnotationNode) || !(nodes[1] instanceof AnnotatedNode)) {
105             throw new RuntimeException("Internal error: wrong types: $node.class / $parent.class");
106         }
107         AnnotationNode node = (AnnotationNode) nodes[0];
108 
109         if (nodes[1] instanceof ClassNode) {
110             addListenerToClass(source, (ClassNode) nodes[1]);
111         } else {
112             if ((((FieldNode)nodes[1]).getModifiers() & Opcodes.ACC_FINAL) != 0) {
113                 source.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(
114                         new SyntaxException("@groovy.beans.Vetoable cannot annotate a final property.",
115                                 node.getLineNumber(), node.getColumnNumber(), node.getLastLineNumber(), node.getLastColumnNumber()),
116                         source));
117             }
118 
119             addListenerToProperty(source, node, (AnnotatedNode) nodes[1]);
120         }
121     }
122 
123     private void addListenerToProperty(SourceUnit source, AnnotationNode node, AnnotatedNode parent) {
124         ClassNode declaringClass = parent.getDeclaringClass();
125         FieldNode field = ((FieldNode) parent);
126         String fieldName = field.getName();
127         for (PropertyNode propertyNode : declaringClass.getProperties()) {
128             boolean bindable = BindableASTTransformation.hasBindableAnnotation(parent)
129                     || BindableASTTransformation.hasBindableAnnotation(parent.getDeclaringClass());
130 
131             if (propertyNode.getName().equals(fieldName)) {
132                 if (field.isStatic()) {
133                     //noinspection ThrowableInstanceNeverThrown
134                     source.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(
135                             new SyntaxException("@groovy.beans.Vetoable cannot annotate a static property.",
136                                     node.getLineNumber(), node.getColumnNumber(), node.getLastLineNumber(), node.getLastColumnNumber()),
137                             source));
138                 } else {
139                     createListenerSetter(source, bindable, declaringClass, propertyNode);
140                 }
141                 return;
142             }
143         }
144         //noinspection ThrowableInstanceNeverThrown
145         source.getErrorCollector().addErrorAndContinue(new SyntaxErrorMessage(
146                 new SyntaxException("@groovy.beans.Vetoable must be on a property, not a field.  Try removing the private, protected, or public modifier.",
147                         node.getLineNumber(), node.getColumnNumber(), node.getLastLineNumber(), node.getLastColumnNumber()),
148                 source));
149     }

152     private void addListenerToClass(SourceUnit source, ClassNode classNode) {
153         boolean bindable = BindableASTTransformation.hasBindableAnnotation(classNode);
154         for (PropertyNode propertyNode : classNode.getProperties()) {
155             if (!hasVetoableAnnotation(propertyNode.getField())
156                 && !propertyNode.getField().isFinal()
157                 && !propertyNode.getField().isStatic())
158             {
159                 createListenerSetter(source,
160                         bindable || BindableASTTransformation.hasBindableAnnotation(propertyNode.getField()),
161                     classNode, propertyNode);
162             }
163         }
164     }

166     /**
167      * Wrap an existing setter.
168      */
169     private void wrapSetterMethod(ClassNode classNode, boolean bindable, String propertyName) {
170         String getterName = "get" + MetaClassHelper.capitalize(propertyName);
171         MethodNode setter = classNode.getSetterMethod("set" + MetaClassHelper.capitalize(propertyName));
172 
173         if (setter != null) {
174             // Get the existing code block
175             Statement code = setter.getCode();
176 
177             Expression oldValue = varX("$oldValue");
178             Expression newValue = varX("$newValue");
179             Expression proposedValue = varX(setter.getParameters()[0].getName());
180             BlockStatement block = new BlockStatement();
181 
182             // create a local variable to hold the old value from the getter
183             block.addStatement(declS(oldValue, callThisX(getterName)));
184 
185             // add the fireVetoableChange method call
186             block.addStatement(stmt(callThisX("fireVetoableChange", args(
187                     constX(propertyName), oldValue, proposedValue))));
188 
189             // call the existing block, which will presumably set the value properly
190             block.addStatement(code);
191 
192             if (bindable) {
193                 // get the new value to emit in the event
194                 block.addStatement(declS(newValue, callThisX(getterName)));
195 
196                 // add the firePropertyChange method call
197                 block.addStatement(stmt(callThisX("firePropertyChange", args(constX(propertyName), oldValue, newValue))));
198             }
199 
200             // replace the existing code block with our new one
201             setter.setCode(block);
202         }
203     }

205     private void createListenerSetter(SourceUnit source, boolean bindable, ClassNode declaringClass, PropertyNode propertyNode) {
206         if (bindable && needsPropertyChangeSupport(declaringClass, source)) {
207             addPropertyChangeSupport(declaringClass);
208         }
209         if (needsVetoableChangeSupport(declaringClass, source)) {
210             addVetoableChangeSupport(declaringClass);
211         }
212         String setterName = "set" + MetaClassHelper.capitalize(propertyNode.getName());
213         if (declaringClass.getMethods(setterName).isEmpty()) {
214             Expression fieldExpression = fieldX(propertyNode.getField());
215             BlockStatement setterBlock = new BlockStatement();
216             setterBlock.addStatement(createConstrainedStatement(propertyNode, fieldExpression));
217             if (bindable) {
218                 setterBlock.addStatement(createBindableStatement(propertyNode, fieldExpression));
219             } else {
220                 setterBlock.addStatement(createSetStatement(fieldExpression));
221             }
222 
223             // create method void <setter>(<type> fieldName)
224             createSetterMethod(declaringClass, propertyNode, setterName, setterBlock);
225         } else {
226             wrapSetterMethod(declaringClass, bindable, propertyNode.getName());
227         }
228     }

230     /**
231      * Creates a statement body similar to:
232      * <code>this.fireVetoableChange("field", field, field = value)</code>
233      *
234      * @param propertyNode           the field node for the property
235      * @param fieldExpression a field expression for setting the property value
236      * @return the created statement
237      */
238     protected Statement createConstrainedStatement(PropertyNode propertyNode, Expression fieldExpression) {
239         return stmt(callThisX("fireVetoableChange", args(constX(propertyNode.getName()), fieldExpression, varX("value"))));
240     }
241 
242     /**
243      * Creates a statement body similar to:
244      * <code>field = value</code>.
245      * <p>
246      * Used when the field is not also {@code @Bindable}.
247      *
248      * @param fieldExpression a field expression for setting the property value
249      * @return the created statement
250      */
251     protected Statement createSetStatement(Expression fieldExpression) {
252         return assignS(fieldExpression, varX("value"));
253     }

255     /**
256      * Snoops through the declaring class and all parents looking for a field
257      * of type VetoableChangeSupport.  Remembers the field and returns false
258      * if found otherwise returns true to indicate that such support should
259      * be added.
260      *
261      * @param declaringClass the class to search
262      * @return true if vetoable change support should be added
263      */
264     protected boolean needsVetoableChangeSupport(ClassNode declaringClass, SourceUnit sourceUnit) {
265         boolean foundAdd = false, foundRemove = false, foundFire = false;
266         ClassNode consideredClass = declaringClass;
267         while (consideredClass!= null) {
268             for (MethodNode method : consideredClass.getMethods()) {
269                 // just check length, MOP will match it up
270                 foundAdd = foundAdd || method.getName().equals("addVetoableChangeListener") && method.getParameters().length == 1;
271                 foundRemove = foundRemove || method.getName().equals("removeVetoableChangeListener") && method.getParameters().length == 1;
272                 foundFire = foundFire || method.getName().equals("fireVetoableChange") && method.getParameters().length == 3;
273                 if (foundAdd && foundRemove && foundFire) {
274                     return false;
275                 }
276             }
277             consideredClass = consideredClass.getSuperClass();
278         }
279         // check if a super class has @Vetoable annotations
280         consideredClass = declaringClass.getSuperClass();
281         while (consideredClass!=null) {
282             if (hasVetoableAnnotation(consideredClass)) return false;
283             for (FieldNode field : consideredClass.getFields()) {
284                 if (hasVetoableAnnotation(field)) return false;
285             }
286             consideredClass = consideredClass.getSuperClass();
287         }
288         if (foundAdd || foundRemove || foundFire) {
289             sourceUnit.getErrorCollector().addErrorAndContinue(
290                 new SimpleMessage("@Vetoable cannot be processed on "
291                     + declaringClass.getName()
292                     + " because some but not all of addVetoableChangeListener, removeVetoableChange, and fireVetoableChange were declared in the current or super classes.",
293                 sourceUnit)
294             );
295             return false;
296         }
297         return true;
298     }

300     /**
301      * Creates a setter method with the given body.
302      * <p>
303      * This differs from normal setters in that we need to add a declared
304      * exception java.beans.PropertyVetoException
305      *
306      * @param declaringClass the class to which we will add the setter
307      * @param propertyNode          the field to back the setter
308      * @param setterName     the name of the setter
309      * @param setterBlock    the statement representing the setter block
310      */
311     protected void createSetterMethod(ClassNode declaringClass, PropertyNode propertyNode, String setterName, Statement setterBlock) {
312         ClassNode[] exceptions = {ClassHelper.make(PropertyVetoException.class)};
313         MethodNode setter = new MethodNode(
314                 setterName,
315                 propertyNode.getModifiers(),
316                 ClassHelper.VOID_TYPE,
317                 params(param(propertyNode.getType(), "value")),
318                 exceptions,
319                 setterBlock);
320         setter.setSynthetic(true);
321         // add it to the class
322         declaringClass.addMethod(setter);
323     }

325     /**
326      * Adds the necessary field and methods to support vetoable change support.
327      * <p>
328      * Adds a new field:
329      * <code>"protected final java.beans.VetoableChangeSupport this$vetoableChangeSupport = new java.beans.VetoableChangeSupport(this)"</code>
330      * <p>
331      * Also adds support methods:
332      * <code>public void addVetoableChangeListener(java.beans.VetoableChangeListener)</code>
333      * <code>public void addVetoableChangeListener(String, java.beans.VetoableChangeListener)</code>
334      * <code>public void removeVetoableChangeListener(java.beans.VetoableChangeListener)</code>
335      * <code>public void removeVetoableChangeListener(String, java.beans.VetoableChangeListener)</code>
336      * <code>public java.beans.VetoableChangeListener[] getVetoableChangeListeners()</code>
337      *
338      * @param declaringClass the class to which we add the support field and methods
339      */
340     protected void addVetoableChangeSupport(ClassNode declaringClass) {
341         ClassNode vcsClassNode = ClassHelper.make(VetoableChangeSupport.class);
342         ClassNode vclClassNode = ClassHelper.make(VetoableChangeListener.class);
343 
344         // add field:
345         // protected static VetoableChangeSupport this$vetoableChangeSupport = new java.beans.VetoableChangeSupport(this)
346         FieldNode vcsField = declaringClass.addField(
347                 "this$vetoableChangeSupport",
348                 ACC_FINAL | ACC_PRIVATE | ACC_SYNTHETIC,
349                 vcsClassNode,
350                 ctorX(vcsClassNode, args(varX("this"))));
351 
352         // add method:
353         // void addVetoableChangeListener(listener) {
354         //     this$vetoableChangeSupport.addVetoableChangeListener(listener)
355         //  }
356         declaringClass.addMethod(
357                 new MethodNode(
358                         "addVetoableChangeListener",
359                         ACC_PUBLIC,
360                         ClassHelper.VOID_TYPE,
361                         params(param(vclClassNode, "listener")),
362                         ClassNode.EMPTY_ARRAY,
363                         stmt(callX(fieldX(vcsField), "addVetoableChangeListener", args(varX("listener", vclClassNode))))));
364 
365         // add method:
366         // void addVetoableChangeListener(name, listener) {
367         //     this$vetoableChangeSupport.addVetoableChangeListener(name, listener)
368         //  }
369         declaringClass.addMethod(
370                 new MethodNode(
371                         "addVetoableChangeListener",
372                         ACC_PUBLIC,
373                         ClassHelper.VOID_TYPE,
374                         params(param(ClassHelper.STRING_TYPE, "name"), param(vclClassNode, "listener")),
375                         ClassNode.EMPTY_ARRAY,
376                         stmt(callX(fieldX(vcsField), "addVetoableChangeListener", args(varX("name", ClassHelper.STRING_TYPE), varX("listener", vclClassNode))))));
377 
378         // add method:
379         // boolean removeVetoableChangeListener(listener) {
380         //    return this$vetoableChangeSupport.removeVetoableChangeListener(listener);
381         // }
382         declaringClass.addMethod(
383                 new MethodNode(
384                         "removeVetoableChangeListener",
385                         ACC_PUBLIC,
386                         ClassHelper.VOID_TYPE,
387                         params(param(vclClassNode, "listener")),
388                         ClassNode.EMPTY_ARRAY,
389                         stmt(callX(fieldX(vcsField), "removeVetoableChangeListener", args(varX("listener", vclClassNode))))));
390 
391         // add method: void removeVetoableChangeListener(name, listener)
392         declaringClass.addMethod(
393                 new MethodNode(
394                         "removeVetoableChangeListener",
395                         ACC_PUBLIC,
396                         ClassHelper.VOID_TYPE,
397                         params(param(ClassHelper.STRING_TYPE, "name"), param(vclClassNode, "listener")),
398                         ClassNode.EMPTY_ARRAY,
399                         stmt(callX(fieldX(vcsField), "removeVetoableChangeListener", args(varX("name", ClassHelper.STRING_TYPE), varX("listener", vclClassNode))))));
400 
401         // add method:
402         // void fireVetoableChange(String name, Object oldValue, Object newValue)
403         //    throws PropertyVetoException
404         // {
405         //     this$vetoableChangeSupport.fireVetoableChange(name, oldValue, newValue)
406         //  }
407         declaringClass.addMethod(
408                 new MethodNode(
409                         "fireVetoableChange",
410                         ACC_PUBLIC,
411                         ClassHelper.VOID_TYPE,
412                         params(param(ClassHelper.STRING_TYPE, "name"), param(ClassHelper.OBJECT_TYPE, "oldValue"), param(ClassHelper.OBJECT_TYPE, "newValue")),
413                         new ClassNode[] {ClassHelper.make(PropertyVetoException.class)},
414                         stmt(callX(fieldX(vcsField), "fireVetoableChange", args(varX("name", ClassHelper.STRING_TYPE), varX("oldValue"), varX("newValue"))))));
415 
416         // add method:
417         // VetoableChangeListener[] getVetoableChangeListeners() {
418         //   return this$vetoableChangeSupport.getVetoableChangeListeners
419         // }
420         declaringClass.addMethod(
421                 new MethodNode(
422                         "getVetoableChangeListeners",
423                         ACC_PUBLIC,
424                         vclClassNode.makeArray(),
425                         Parameter.EMPTY_ARRAY,
426                         ClassNode.EMPTY_ARRAY,
427                         returnS(callX(fieldX(vcsField), "getVetoableChangeListeners"))));
428 
429         // add method:
430         // VetoableChangeListener[] getVetoableChangeListeners(String name) {
431         //   return this$vetoableChangeSupport.getVetoableChangeListeners(name)
432         // }
433         declaringClass.addMethod(
434                 new MethodNode(
435                         "getVetoableChangeListeners",
436                         ACC_PUBLIC,
437                         vclClassNode.makeArray(),
438                         params(param(ClassHelper.STRING_TYPE, "name")),
439                         ClassNode.EMPTY_ARRAY,
440                         returnS(callX(fieldX(vcsField), "getVetoableChangeListeners", args(varX("name", ClassHelper.STRING_TYPE))))));
441     }
