91      /**
92       * Returns the Selector
93       */
94      public static Selector getSelector(MutableCallSite callSite, Class sender, String methodName, int callID, boolean safeNavigation, boolean thisCall, boolean spreadCall, Object[] arguments) {
95          CALL_TYPES callType = CALL_TYPES.values()[callID];
96          switch (callType) {
97              case INIT: return new InitSelector(callSite, sender, methodName, callType, safeNavigation, thisCall, spreadCall, arguments);
98              case METHOD: return new MethodSelector(callSite, sender, methodName, callType, safeNavigation, thisCall, spreadCall, arguments);
99              case GET: 
100                 return new PropertySelector(callSite, sender, methodName, callType, safeNavigation, thisCall, spreadCall, arguments);
101             case SET:
102                 throw new GroovyBugError("your call tried to do a property set, which is not supported.");
103             case CAST:  return new CastSelector(callSite, arguments);
104             default: throw new GroovyBugError("unexpected call type");
105         }
106     }

109     /**
110      * Helper method to transform the given arguments, consisting of the receiver 
111      * and the actual arguments in an Object[], into a new Object[] consisting
112      * of the receiver and the arguments directly. Before the size of args was 
113      * always 2, the returned Object[] will have a size of 1+n, where n is the
114      * number arguments.
115      */
116     private static Object[] spread(Object[] args, boolean spreadCall) {
117         if (!spreadCall) return args;
118         Object[] normalArguments = (Object[]) args[1];
119         Object[] ret = new Object[normalArguments.length+1];
120         ret[0] = args[0];
121         System.arraycopy(normalArguments, 0, ret, 1, ret.length-1);
122         return ret;
123     }

125     private static class CastSelector extends MethodSelector {
126         private Class<?> staticSourceType, staticTargetType;
127 
128         public CastSelector(MutableCallSite callSite, Object[] arguments) {
129             super(callSite, Selector.class, "", CALL_TYPES.CAST, false, false, false, arguments);
130             this.staticSourceType = callSite.type().parameterType(0);
131             this.staticTargetType = callSite.type().returnType();
132         }
133 
134         @Override
135         public void setCallSiteTarget() {
136             // targetTypes String, Enum and Class are handled 
137             // by the compiler already
138             // Boolean / boolean
139             handleBoolean();
140             handleNullWithoutBoolean();
141 
142             // !! from here on args[0] is always not null !!
143             handleInstanceCase();
144 
145             // targetType is abstract Collection fitting for HashSet or ArrayList
146             // and object is Collection or array
147             handleCollections();
148             handleSAM();
149 
150             // will handle :
151             //      * collection case where argument is an array
152             //      * array transformation (staticTargetType.isArray())
153             //      * constructor invocation
154             //      * final GroovyCastException
155             castToTypeFallBack();
156 
157             if (!handle.type().equals(callSite.type())) castAndSetGuards();
158         }
159 
160         private void castAndSetGuards() {
161             handle =  MethodHandles.explicitCastArguments(handle,targetType);
162             setGuards(args[0]);
163             doCallSiteTargetSet();
164         }
165 
166         private void handleNullWithoutBoolean() {
167             if (handle!=null || args[0]!=null) return;
168 
169             if (staticTargetType.isPrimitive()) {
170                 handle = MethodHandles.insertArguments(GROOVY_CAST_EXCEPTION,1,staticTargetType);
171                 // need to call here here because we used the static target type
172                 // it won't be done otherwise because handle.type() == callSite.type()
173                 castAndSetGuards(); 
174             } else {
175                handle = MethodHandles.identity(staticSourceType);
176             }
177         }
178 
179         private void handleInstanceCase() {
180             if (handle!=null) return;
181 
182             if (staticTargetType.isAssignableFrom(args[0].getClass())) {
183                 handle = MethodHandles.identity(staticSourceType);
184             }
185         }
186 
187         private static boolean isAbstractClassOf(Class toTest, Class givenOnCallSite) {
188             if (!toTest.isAssignableFrom(givenOnCallSite)) return false;
189             if (givenOnCallSite.isInterface()) return true;
190             return Modifier.isAbstract(givenOnCallSite.getModifiers());
191         }
192 
193         private void handleCollections() {
194             if (handle!=null) return;
195 
196             if (!(args[0] instanceof Collection)) return;
197             if (isAbstractClassOf(HashSet.class, staticTargetType)) {
198                 handle = HASHSET_CONSTRUCTOR;
199             } else if (isAbstractClassOf(ArrayList.class, staticTargetType)) {
200                 handle = ARRAYLIST_CONSTRUCTOR;
201             }
202         }
203 
204         private void handleSAM() {
205             if (handle!=null) return;
206 
207             if (!(args[0] instanceof Closure)) return;
208             Method m = CachedSAMClass.getSAMMethod(staticTargetType);
209             if (m==null) return;
210             //TODO: optimize: add guard based on type Closure
211             handle = MethodHandles.insertArguments(SAM_CONVERSION, 1, m, staticTargetType, staticTargetType.isInterface());
212         }
213 
214         private void castToTypeFallBack() {
215             if (handle!=null) return;
216 
217             // generic fallback to castToType
218             handle = MethodHandles.insertArguments(DTT_CAST_TO_TYPE, 1, staticTargetType);
219         }
220 
221         private void handleBoolean() {
222             if (handle!=null) return;
223 
224             // boolean->boolean, Boolean->boolean, boolean->Boolean
225             // is handled by compiler
226             // that leaves (T)Z and (T)Boolean, where T is the static type
227             // but runtime type of T might be Boolean
228 
229             boolean primitive = staticTargetType==boolean.class;
230             if (!primitive && staticTargetType!=Boolean.class) return;
231             if (args[0]==null) {
232                 if (primitive) {
233                     handle = MethodHandles.constant(boolean.class, false);
234                     handle = MethodHandles.dropArguments(handle, 0, staticSourceType);
235                 } else {
236                     handle = BOOLEAN_IDENTITY;
237                 }
238             } else if (args[0] instanceof Boolean) {
239                 // give value through or unbox
240                 handle = BOOLEAN_IDENTITY;
241             } else { 
242                 //call asBoolean
243                 name = "asBoolean";
244                 super.setCallSiteTarget();
245                 return;
246             }
247         }
248     }

265         /**
266          * this method chooses a property from the meta class.
267          */
268         @Override
269         public void chooseMeta(MetaClassImpl mci) {
270             Object receiver = getCorrectedReceiver();
271             if (receiver instanceof GroovyObject) {
272                 Class aClass = receiver.getClass();
273                 Method reflectionMethod = null;
274                 try {
275                     reflectionMethod = aClass.getMethod("getProperty", String.class);
276                     if (!reflectionMethod.isSynthetic()) {
277                         handle = MethodHandles.insertArguments(GROOVY_OBJECT_GET_PROPERTY, 1, name);
278                         return;
279                     }
280                 } catch (ReflectiveOperationException e)  {}
281             } else if (receiver instanceof Class) {
282                 handle = MOP_GET;
283                 handle = MethodHandles.insertArguments(handle, 2, name);
284                 handle = MethodHandles.insertArguments(handle, 0, this.mc);
285                 return;
286             }
287 
288             if (method!=null || mci==null) return;
289             Class chosenSender = this.sender;
290             if (mci.getTheClass()!= chosenSender && GroovyCategorySupport.hasCategoryInCurrentThread()) {
291                 chosenSender = mci.getTheClass();
292             }
293             MetaProperty res = mci.getEffectiveGetMetaProperty(chosenSender, receiver, name, false);
294             if (res instanceof MethodMetaProperty) {
295                 MethodMetaProperty mmp = (MethodMetaProperty) res;
296                 method = mmp.getMetaMethod();
297                 insertName = true;
298             } else if (res instanceof CachedField) {
299                 CachedField cf = (CachedField) res;
300                 Field f = cf.field;
301                 try {
302                     handle = LOOKUP.unreflectGetter(f);
303                     if (Modifier.isStatic(f.getModifiers())) {
304                         // normally we would do the following
305                         // handle = MethodHandles.dropArguments(handle,0,Class.class);
306                         // but because there is a bug in invokedynamic in all jdk7 versions 
307                         // maybe use Unsafe.ensureClassInitialized
308                         handle = META_PROPERTY_GETTER.bindTo(res);
309                     }
310                 } catch (IllegalAccessException iae) {
311                     throw new GroovyBugError(iae);
312                 }
313             } else {
314                 handle = META_PROPERTY_GETTER.bindTo(res);
315             } 
316         }

378             Object[] newArgs = removeRealReceiver(args);
379             method = mci.retrieveConstructor(newArgs);
380             if (method instanceof MetaConstructor) {
381                 MetaConstructor mcon = (MetaConstructor) method;

384                     beanConstructor = true;

389         /**
390          * Adds {@link MetaConstructor} handling.
391          */
392         @Override
393         public void setHandleForMetaMethod() {
394             if (method==null) return;
395             if (method instanceof MetaConstructor) {
396                 if (LOG_ENABLED) LOG.info("meta method is MetaConstructor instance");
397                 MetaConstructor mc = (MetaConstructor) method;
398                 isVargs = mc.isVargsMethod();
399                 Constructor con = mc.getCachedConstrcutor().cachedConstructor;
400                 try {
401                     handle = LOOKUP.unreflectConstructor(con);
402                     if (LOG_ENABLED) LOG.info("successfully unreflected constructor");
403                 } catch (IllegalAccessException e) {
404                     throw new GroovyBugError(e);
405                 }
406             } else {
407                 super.setHandleForMetaMethod();
408             }
409             if (beanConstructor) {
410                 // we have handle that takes no arguments to create the bean, 
411                 // we have to use its return value to call #setBeanProperties with it
412                 // and the meta class.
413 
414                 // to do this we first bind the values to #setBeanProperties
415                 MethodHandle con = BEAN_CONSTRUCTOR_PROPERTY_SETTER.bindTo(mc);
416                 // inner class case
417                 MethodType foldTargetType = MethodType.methodType(Object.class);
418                 if (args.length==3) {
419                     con = MethodHandles.dropArguments(con, 1, targetType.parameterType(1));
420                     foldTargetType = foldTargetType.insertParameterTypes(0, targetType.parameterType(1));
421                 }
422                 handle = MethodHandles.foldArguments(con, handle.asType(foldTargetType));
423             }
424             if (method instanceof MetaConstructor) {
425                 handle = MethodHandles.dropArguments(handle, 0, Class.class);
426             }
427         }

464     /**
465      * Method invocation based {@link Selector}.
466      * This Selector is called for method invocations and is base for cosntructor
467      * calls as well as getProperty calls.
468      * @author <a href="mailto:blackdrag@gmx.org">Jochen "blackdrag" Theodorou</a>
469      */
470     private static class MethodSelector extends Selector {
471         protected MetaClass mc;
472         private boolean isCategoryMethod;
473         public MethodSelector(MutableCallSite callSite, Class sender, String methodName, CALL_TYPES callType, Boolean safeNavigation, Boolean thisCall, Boolean spreadCall, Object[] arguments) {
474             this.callType = callType;
475             this.targetType = callSite.type();
476             this.name = methodName;
477             this.originalArguments = arguments;
478             this.args = spread(arguments, spreadCall);
479             this.callSite = callSite;
480             this.sender = sender;
481             this.safeNavigationOrig = safeNavigation;
482             this.safeNavigation = safeNavigation && arguments[0]==null;
483             this.thisCall = thisCall;
484             this.spread = spreadCall;
485             this.cache = !spread;
486 
487             if (LOG_ENABLED) {
488                 String msg =
489                     "----------------------------------------------------"+
490                     "\n\t\tinvocation of method '"+methodName+"'"+
491                     "\n\t\tinvocation type: "+callType+
492                     "\n\t\tsender: "+sender+
493                     "\n\t\ttargetType: "+targetType+
494                     "\n\t\tsafe navigation: "+safeNavigation+
495                     "\n\t\tthisCall: "+thisCall+
496                     "\n\t\tspreadCall: "+spreadCall+
497                     "\n\t\twith "+arguments.length+" arguments";
498                 for (int i=0; i<arguments.length; i++) {
499                     msg += "\n\t\t\targument["+i+"] = "+arguments[i];
500                 }
501                 LOG.info(msg);
502             }
503         }

505         /**
506          * Sets the null constant for safe navigation.
507          * In case of foo?.bar() and foo being null, we don't call the method,
508          * instead we simply return null. This produces a handle, which will 
509          * return the constant.
510          */
511         public boolean setNullForSafeNavigation() {
512             if (!safeNavigation) return false;
513             handle = MethodHandles.dropArguments(NULL_REF,0,targetType.parameterArray());
514             if (LOG_ENABLED) LOG.info("set null returning handle for safe navigation");
515             return true;
516         }

518         /**
519          * Gives the meta class to an Object.
520          */
521         public void getMetaClass() {
522             Object receiver = args[0];
523             if (receiver == null) {
524                 mc = NullObject.getNullObject().getMetaClass();
525             } else if (receiver instanceof GroovyObject) {
526                 mc = ((GroovyObject) receiver).getMetaClass();
527             } else if (receiver instanceof Class) {
528                 Class c = (Class) receiver;
529                 ClassLoader cl = c.getClassLoader();
530                 try {
531                     Class.forName(c.getName(), true, cl);
532                 } catch (ClassNotFoundException e) {}
533                 mc = GroovySystem.getMetaClassRegistry().getMetaClass(c);
534                 this.cache &= !ClassInfo.getClassInfo(c).hasPerInstanceMetaClasses();
535             } else {
536                 mc = ((MetaClassRegistryImpl) GroovySystem.getMetaClassRegistry()).getMetaClass(receiver);
537                 this.cache &= !ClassInfo.getClassInfo(receiver.getClass()).hasPerInstanceMetaClasses();
538             }
539             mc.initialize();
540         }

542         /**
543          * Uses the meta class to get a meta method for a method call.
544          * There will be no meta method selected, if the meta class is no MetaClassImpl
545          * or the meta class is an AdaptingMetaClass.
546          */
547         public void chooseMeta(MetaClassImpl mci) {
548             if (mci==null) return;
549             Object receiver = getCorrectedReceiver();
550             Object[] newArgs = removeRealReceiver(args);
551             if (receiver instanceof Class) {
552                 if (LOG_ENABLED) LOG.info("receiver is a class");
553                 if (!mci.hasCustomStaticInvokeMethod()) method = mci.retrieveStaticMethod(name, newArgs);
554             } else {
555                 String changedName = name;
556                 if (receiver instanceof GeneratedClosure && changedName.equals("call")) {changedName = "doCall";}
557                 if (!mci.hasCustomInvokeMethod()) method = mci.getMethodWithCaching(selectionBase, changedName, newArgs, false);
558             }
559             if (LOG_ENABLED) LOG.info("retrieved method from meta class: "+method);
560         }

562         /**
563          * Creates a MethodHandle using a before selected MetaMethod.
564          * If the MetaMethod has reflective information available, then
565          * we will use that information to create the target MethodHandle. 
566          * If that is not the case we will produce a handle, which will use the
567          * MetaMethod itself for invocation.
568          */
569         public void setHandleForMetaMethod() {
570             MetaMethod metaMethod = method;
571             isCategoryMethod = method instanceof CategoryMethod;
572 
573             if (
574                     metaMethod instanceof NumberNumberMetaMethod ||
575                     (method instanceof GeneratedMetaMethod && (name.equals("next") || name.equals("previous")))
576             ) {
577                 if (LOG_ENABLED) LOG.info("meta method is number method");
578                 if (IndyMath.chooseMathMethod(this, metaMethod)) {
579                     catchException = false;
580                     if (LOG_ENABLED) LOG.info("indy math successfull");
581                     return;
582                 }
583             }
584 
585             boolean isCategoryTypeMethod = metaMethod instanceof NewInstanceMetaMethod;
586             if (LOG_ENABLED) LOG.info("meta method is category type method: "+isCategoryTypeMethod);
587             boolean isStaticCategoryTypeMethod = metaMethod instanceof NewStaticMetaMethod;
588             if (LOG_ENABLED) LOG.info("meta method is static category type method: "+isCategoryTypeMethod);
589 
590             if (metaMethod instanceof ReflectionMetaMethod) {
591                 if (LOG_ENABLED) LOG.info("meta method is reflective method");
592                 ReflectionMetaMethod rmm = (ReflectionMetaMethod) metaMethod;
593                 metaMethod = rmm.getCachedMethod();
594             }
595 
596             if (metaMethod instanceof CachedMethod) {
597                 if (LOG_ENABLED) LOG.info("meta method is CachedMethod instance");
598                 CachedMethod cm = (CachedMethod) metaMethod;
599                 isVargs = cm.isVargsMethod();
600                 try {
601                     Method m = cm.getCachedMethod();
602                     handle = correctClassForNameAndUnReflectOtherwise(m);
603                     if (LOG_ENABLED) LOG.info("successfully unreflected method");
604                     if (isStaticCategoryTypeMethod) {
605                         handle = MethodHandles.insertArguments(handle, 0, new Object[]{null});
606                         handle = MethodHandles.dropArguments(handle, 0, targetType.parameterType(0));
607                     } else if (!isCategoryTypeMethod && isStatic(m)) {
608                         // we drop the receiver, which might be a Class (invocation on Class)
609                         // or it might be an object (static method invocation on instance)
610                         // Object.class handles both cases at once
611                         handle = MethodHandles.dropArguments(handle, 0, Object.class);
612                     } 
613                 } catch (IllegalAccessException e) {
614                     throw new GroovyBugError(e);
615                 }
616             } else if (method != null) {
617                 if (LOG_ENABLED) LOG.info("meta method is dgm helper");
618                 // generic meta method invocation path
619                 handle = META_METHOD_INVOKER;
620                 handle = handle.bindTo(method);
621                 if (spread) {
622                     args = originalArguments;
623                     skipSpreadCollector = true;
624                 } else {
625                     // wrap arguments from call site in Object[]
626                     handle = handle.asCollector(Object[].class, targetType.parameterCount()-1);
627                 }

630             }

684          * Corrects method argument wrapping.
685          * In cases in which we want to force a certain method selection
686          * we use Wrapper classes to transport the static type information.
687          * This method will be used to undo the wrapping.
688          */
689         public void correctWrapping() {
690             if (useMetaClass) return;
691             Class[] pt = handle.type().parameterArray();
692             if (currentType!=null) pt = currentType.parameterArray();
693             for (int i=1; i<args.length; i++) {
694                 if (args[i] instanceof Wrapper) {
695                     Class type = pt[i];
696                     MethodType mt = MethodType.methodType(type, Wrapper.class);
697                     handle = MethodHandles.filterArguments(handle, i, UNWRAP_METHOD.asType(mt));
698                     if (LOG_ENABLED) LOG.info("added filter for Wrapper for argument at pos "+i);
699                 }

703         /**
704          * Handles cases in which we have to correct the length of arguments
705          * using the parameters. This might be needed for vargs and for one 
706          * parameter calls without arguments (null is used then).  
707          */
708         public void correctParameterLength() {
709             if (handle==null) return;
710 
711             Class[] params = handle.type().parameterArray();
712             if (currentType!=null) params = currentType.parameterArray();
713             if (!isVargs) {
714                 if (spread && useMetaClass) return;
715                 if (params.length==2 && args.length==1) {
716                     //TODO: this Object[] can be constant
717                     handle = MethodHandles.insertArguments(handle, 1, new Object[]{null});
718                 }
719                 return;
720             }
721 
722             Class lastParam = params[params.length-1];
723             Object lastArg = unwrapIfWrapped(args[args.length-1]);
724             if (params.length == args.length) {
725                 // may need rewrap
726                 if (lastArg == null) return;
727                 if (lastParam.isInstance(lastArg)) return;
728                 if (lastArg.getClass().isArray()) return;
729                 // arg is not null and not assignment compatible
730                 // so we really need to rewrap
731                 handle = handle.asCollector(lastParam, 1);
732             } else if (params.length > args.length) {
733                 // we depend on the method selection having done a good 
734                 // job before already, so the only case for this here is, that
735                 // we have no argument for the array, meaning params.length is
736                 // args.length+1. In that case we have to fill in an empty array
737                 handle = MethodHandles.insertArguments(handle, params.length-1, Array.newInstance(lastParam.getComponentType(), 0));
738                 if (LOG_ENABLED) LOG.info("added empty array for missing vargs part");
739             } else { //params.length < args.length
740                 // we depend on the method selection having done a good 
741                 // job before already, so the only case for this here is, that
742                 // all trailing arguments belong into the vargs array
743                 handle = handle.asCollector(
744                         lastParam,
745                         args.length - params.length + 1);
746                 if (LOG_ENABLED) LOG.info("changed surplus arguments to be collected for vargs call");
747             }
748         }

750         /**
751          * There are some conversions we have to do explicitly.
752          * These are GString to String, Number to Byte and Number to BigInteger
753          * conversions.
754          */
755         public void correctCoerce() {
756             if (useMetaClass) return;
757 
758             Class[] parameters = handle.type().parameterArray();
759             if (currentType!=null) parameters = currentType.parameterArray();
760             if (args.length != parameters.length) {
761                 throw new GroovyBugError("At this point argument array length and parameter array length should be the same");
762             }
763             for (int i=0; i<args.length; i++) {
764                 if (parameters[i]==Object.class) continue; 
765                 Object arg = unwrapIfWrapped(args[i]);
766                 // we have to handle here different cases in which we do no
767                 // transformations. We depend on our method selection to have
768                 // selected only a compatible method, that means for a null
769                 // argument we don't have to do anything. Same of course is if
770                 // the argument is an instance of the parameter type. We also
771                 // exclude boxing, since the MethodHandles will do that part
772                 // already for us. Another case is the conversion of a primitive
773                 // to another primitive or of the wrappers, or a combination of 
774                 // these. This is also handled already. What is left is the 
775                 // GString conversion and the number conversions.
776 
777                 if (arg==null) continue;
778                 Class got = arg.getClass();
779 
780                 // equal class, nothing to do
781                 if (got==parameters[i]) continue;
782 
783                 Class wrappedPara = TypeHelper.getWrapperClass(parameters[i]);
784                 // equal class with one maybe a primitive, the later explicitCastArguments will solve this case
785                 if (wrappedPara==TypeHelper.getWrapperClass(got)) continue;
786 
787                 // equal in terms of an assignment in Java. That means according to Java widening rules, or
788                 // a subclass, interface, superclass relation, this case then handles also 
789                 // primitive to primitive conversion. Those case are also solved by explicitCastArguments.
790                 if (parameters[i].isAssignableFrom(got)) continue;
791 
792                 // to aid explicitCastArguments we convert to the wrapper type to let is only unbox
793                 handle = TypeTransformers.addTransformer(handle, i, arg, wrappedPara);
794                 if (LOG_ENABLED) LOG.info("added transformer at pos "+i+" for type "+got+" to type "+wrappedPara);
795             }
796         }

816          * Adds the standard exception handler.  
817          */
818         public void addExceptionHandler() {
819             //TODO: if we would know exactly which paths require the exceptions
820             //      and which paths not, we can sometimes save this guard 
821             if (handle==null || catchException==false) return;

832         /**
833          * Sets all argument and receiver guards.
834          */
835         public void setGuards (Object receiver) {
836             if (handle==null) return;
837             if (!cache) return;
838 
839             MethodHandle fallback = makeFallBack(callSite, sender, name, callType.ordinal(), targetType, safeNavigationOrig, thisCall, spread);
840 
841             // special guards for receiver
842             if (receiver instanceof GroovyObject) {
843                 GroovyObject go = (GroovyObject) receiver;
844                 MetaClass mc = (MetaClass) go.getMetaClass();
845                 MethodHandle test = SAME_MC.bindTo(mc); 
846                 // drop dummy receiver
847                 test = test.asType(MethodType.methodType(boolean.class,targetType.parameterType(0)));
848                 handle = MethodHandles.guardWithTest(test, handle, fallback);
849                 if (LOG_ENABLED) LOG.info("added meta class equality check");
850             } else if (receiver instanceof Class) {
851                 MethodHandle test = EQUALS.bindTo(receiver);
852                 test = test.asType(MethodType.methodType(boolean.class,targetType.parameterType(0)));
853                 handle = MethodHandles.guardWithTest(test, handle, fallback);
854                 if (LOG_ENABLED) LOG.info("added class equality check");
855             }
856 
857             if (!useMetaClass && isCategoryMethod) {
858                 // category method needs Thread check
859                 // cases:
860                 // (1) method is a category method
861                 //     We need to check if the category in the current thread is still active.
862                 //     Since we invalidate on leaving the category checking for it being
863                 //     active directly is good enough.
864                 // (2) method is in use scope, but not from category
865                 //     Since entering/leaving a category will invalidate, there is no need for any special check
866                 // (3) method is not in use scope /and not from category
867                 //     Since entering/leaving a category will invalidate, there is no need for any special check
868                 if (method instanceof NewInstanceMetaMethod) {
869                     handle = MethodHandles.guardWithTest(HAS_CATEGORY_IN_CURRENT_THREAD_GUARD, handle, fallback);
870                     if (LOG_ENABLED) LOG.info("added category-in-current-thread-guard for category method");
871                 }
872             }
873 
874             // handle constant meta class and category changes
875             handle = switchPoint.guardWithTest(handle, fallback);
876             if (LOG_ENABLED) LOG.info("added switch point guard");
877 
878             // guards for receiver and parameter
879             Class[] pt = handle.type().parameterArray();
880             for (int i=0; i<args.length; i++) {
881                 Object arg = args[i];
882                 MethodHandle test = null;
883                 if (arg==null) {
884                     test = IS_NULL.asType(MethodType.methodType(boolean.class, pt[i]));
885                     if (LOG_ENABLED) LOG.info("added null argument check at pos "+i);
886                 } else { 
887                     Class argClass = arg.getClass();
888                     if (pt[i].isPrimitive()) continue;
889                     //if (Modifier.isFinal(argClass.getModifiers()) && TypeHelper.argumentClassIsParameterClass(argClass,pt[i])) continue;
890                     test = SAME_CLASS.
891                                 bindTo(argClass).
892                                 asType(MethodType.methodType(boolean.class, pt[i]));
893                     if (LOG_ENABLED) LOG.info("added same class check at pos "+i);
894                 }
895                 Class[] drops = new Class[i];
896                 for (int j=0; j<drops.length; j++) drops[j] = pt[j];
897                 test = MethodHandles.dropArguments(test, 0, drops);
898                 handle = MethodHandles.guardWithTest(test, handle, fallback);
899             }
900         }

939         /**
940          * setting a call site target consists of the following steps:
941          * # get the meta class
942          * # select a method/constructor/property from it, if it is a MetaClassImpl
943          * # make a handle out of the selection
944          * # if nothing could be selected select a path through the given MetaClass or the GroovyObject
945          * # apply transformations for vargs, implicit null argument, coercion, wrapping, null receiver and spreading
946          */
947         @Override
948         public void setCallSiteTarget() {
949             if (!setNullForSafeNavigation() && !setInterceptor()) {
950                 getMetaClass();
951                 if (LOG_ENABLED) LOG.info("meta class is "+mc);
952                 setSelectionBase();
953                 MetaClassImpl mci = getMetaClassImpl(mc, callType != CALL_TYPES.GET);
954                 chooseMeta(mci);
955                 setHandleForMetaMethod();
956                 setMetaClassCallHandleIfNedded(mci!=null);
957                 correctParameterLength();
958                 correctCoerce();
959                 correctWrapping();
960                 correctNullReceiver();
961                 correctSpreading();
962 
963                 if (LOG_ENABLED) LOG.info("casting explicit from "+handle.type()+" to "+targetType);
964                 handle =  MethodHandles.explicitCastArguments(handle,targetType);
965 
966                 addExceptionHandler();
967             }
968             setGuards(args[0]);
969             doCallSiteTargetSet();
970         }

1004     /**
1005      * Returns the MetaClassImpl if the given MetaClass is one of
1006      * MetaClassImpl, AdaptingMetaClass or ClosureMetaClass. If
1007      * none of these cases matches, this method returns null.
1008      */
1009     private static MetaClassImpl getMetaClassImpl(MetaClass mc, boolean includeEMC) {
1010         Class mcc = mc.getClass();
1011         boolean valid = mcc == MetaClassImpl.class ||
1012                          mcc == AdaptingMetaClass.class ||
1013                          mcc == ClosureMetaClass.class ||
1014                          (includeEMC && mcc == ExpandoMetaClass.class);
1015         if (!valid) {
1016             if (LOG_ENABLED) LOG.info("meta class is neither MetaClassImpl, nor AdoptingMetaClass, nor ClosureMetaClass, normal method selection path disabled.");
1017             return null;
1018         }
1019         if (LOG_ENABLED) LOG.info("meta class is a recognized MetaClassImpl");
1020         return (MetaClassImpl) mc;
1021     }
