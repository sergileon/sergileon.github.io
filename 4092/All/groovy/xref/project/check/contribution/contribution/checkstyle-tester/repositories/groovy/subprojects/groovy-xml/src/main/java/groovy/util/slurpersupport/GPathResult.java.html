53  public abstract class GPathResult extends GroovyObjectSupport implements Writable, Buildable, Iterable {
54      protected final GPathResult parent;
55      protected final String name;
56      protected final String namespacePrefix;
57      protected final Map namespaceMap = new HashMap();
58      protected final Map<String, String> namespaceTagHints;

60      /**
61       * Creates a new GPathResult named <code>name</code> with the parent <code>parent</code>,
62       * the namespacePrefix <code>namespacePrefix</code> and the namespaceTagHints specified in
63       * the <code>namespaceTagHints</code> Map.
64       *
65       * @param parent the GPathResult prior to the application of the expression creating this GPathResult
66       * @param name if the GPathResult corresponds to something with a name, e.g. a node
67       * @param namespacePrefix the namespace prefix if any
68       * @param namespaceTagHints the known tag to namespace mappings
69       */
70      public GPathResult(final GPathResult parent, final String name, final String namespacePrefix, final Map<String, String> namespaceTagHints) {
71          if (parent == null) {
72              // we are the top of the tree
73              this.parent = this;
74              this.namespaceMap.put("xml", "http://www.w3.org/XML/1998/namespace");  // The XML namespace is always defined
75          } else {
76              this.parent = parent;
77              this.namespaceMap.putAll(parent.namespaceMap);
78          }
79          this.name = name;
80          this.namespacePrefix = namespacePrefix;
81          this.namespaceTagHints = namespaceTagHints;
82  
83          setMetaClass(getMetaClass()); // wrap the standard MetaClass with the delegate
84      }
85  
86      /**
87       * Replaces the MetaClass of this GPathResult.
88       *
89       * @param metaClass the new MetaClass
90       */
91      @Override
92      public void setMetaClass(final MetaClass metaClass) {
93          final MetaClass newMetaClass = new DelegatingMetaClass(metaClass) {
94              @Override
95              public Object getAttribute(final Object object, final String attribute) {
96                  return GPathResult.this.getProperty("@" + attribute);
97              }
98  
99              @Override
100             public void setAttribute(final Object object, final String attribute, final Object newValue) {
101                 GPathResult.this.setProperty("@" + attribute, newValue);
102             }
103         };
104         super.setMetaClass(newMetaClass);
105     }

107     /**
108      * Returns the specified Property of this GPathResult.
109      * <p>
110      * Realizes the follow shortcuts:
111      * <ul>
112      * <li><code>'..'</code> for <code>parent()</code>
113      * <li><code>'*'</code> for <code>children()</code>
114      * <li><code>'**'</code> for <code>depthFirst()</code>
115      * <li><code>'@'</code> for attribute access
116      * </ul>
117      * @param property the Property to fetch
118      */
119     public Object getProperty(final String property) {
120         if ("..".equals(property)) {
121             return parent();
122         } else if ("*".equals(property)) {
123             return children();
124         } else if ("**".equals(property)) {
125             return depthFirst();
126         } else if (property.startsWith("@")) {
127             if (property.contains(":") && !this.namespaceTagHints.isEmpty()) {
128                 final int i = property.indexOf(":");
129                 return new Attributes(this, "@" + property.substring(i + 1), property.substring(1, i), this.namespaceTagHints);
130             } else {
131                 return new Attributes(this, property, this.namespaceTagHints);
132             }
133         } else {
134             if (property.contains(":") && !this.namespaceTagHints.isEmpty()) {
135                 final int i = property.indexOf(":");
136                 return new NodeChildren(this, property.substring(i + 1), property.substring(0, i), this.namespaceTagHints);
137             } else {
138                 return new NodeChildren(this, property, this.namespaceTagHints);
139             }
140         }
141     }

143     /**
144      * Replaces the specified property of this GPathResult with a new value.
145      *
146      * @param property the property of this GPathResult to replace
147      * @param newValue the new value of the property
148      */
149     public void setProperty(final String property, final Object newValue) {
150         if (property.startsWith("@")) {
151             if (newValue instanceof String || newValue instanceof GString) {
152                 final Iterator iter = iterator();
153 
154                 while (iter.hasNext()) {
155                     final NodeChild child = (NodeChild) iter.next();
156 
157                     child.attributes().put(property.substring(1), newValue);
158                 }
159             }
160         } else {
161             final GPathResult result = new NodeChildren(this, property, this.namespaceTagHints);
162 
163             if (newValue instanceof Map) {
164                 for (Object o : ((Map) newValue).entrySet()) {
165                     final Map.Entry entry = (Map.Entry) o;
166                     result.setProperty("@" + entry.getKey(), entry.getValue());
167                 }
168             } else {
169                 if (newValue instanceof Closure) {
170                     result.replaceNode((Closure) newValue);
171                 } else {
172                     result.replaceBody(newValue);
173                 }
174             }
175         }
176     }

178     /**
179      * Overloads the left shift operator to provide an easy way to
180      * lazily append Objects to this GPathResult.
181      *
182      * @param newValue the Object to append
183      * @return <code>this</code>
184      */
185     public Object leftShift(final Object newValue) {
186         appendNode(newValue);
187         return this;
188     }

190     /**
191      * Lazily adds the specified Object to this GPathResult.
192      *
193      * @param newValue the Object to add
194      * @return <code>this</code>
195      */
196     public Object plus(final Object newValue) {
197         this.replaceNode(new Closure(this) {
198             public void doCall(Object[] args) {
199                 final GroovyObject delegate = (GroovyObject) getDelegate();
200                 delegate.getProperty("mkp");
201                 delegate.invokeMethod("yield", args);
202                 delegate.getProperty("mkp");
203                 delegate.invokeMethod("yield", new Object[]{newValue});
204             }
205         });
206 
207         return this;
208     }

375     /**
376      * Supports the subscript operator for a GPathResult.
377      * <pre>
378      * def text = """
379      * &lt;characterList&gt;
380      *   &lt;character/&gt;
381      *   &lt;character&gt;
382      *     &lt;name>Gromit&lt;/name&gt;
383      *   &lt;/character&gt;
384      * &lt;/characterList&gt;"""
385      *
386      * GPathResult characterList = new XmlSlurper().parseText(text)
387      *
388      * assert characterList.character[1].name == 'Gromit'
389      * </pre>
390      * @param index an index
391      * @return the value at the given index
392      */
393     public Object getAt(final int index) {
394         if (index < 0) {
395             // calculate whole list in this case
396             // recommend avoiding -ve's as this is obviously not as efficient
397             List list = list();
398             int adjustedIndex = index + list.size();
399             if (adjustedIndex >= 0 && adjustedIndex < list.size())
400                 return list.get(adjustedIndex);
401         } else {
402             final Iterator iter = iterator();
403             int count = 0;
404 
405             while (iter.hasNext()) {
406                 if (count++ == index) {
407                     return iter.next();
408                 } else {
409                     iter.next();
410                 }
411             }
412         }
413 
414         return new NoChildren(this, this.name, this.namespaceTagHints);
415     }

417     /**
418      * Supports the range subscript operator for a GPathResult.
419      * <pre>
420      * def text = """
421      * &lt;characterList>
422      *   &lt;character&gt;Wallace&lt;/character&gt;
423      *   &lt;character&gt;Gromit&lt;/character&gt;
424      *   &lt;character&gt;Shaun&lt;/character&gt;
425      * &lt;/characterList&gt;"""
426      *
427      * GPathResult characterList = new XmlSlurper().parseText(text)
428      *
429      * assert characterList.character[1..2].join(',') == 'Gromit,Shaun'
430      * </pre>
431      * @param range a Range indicating the items to get
432      * @return a new list based on range borders
433      */
434     public Object getAt(final IntRange range) {
435         return DefaultGroovyMethods.getAt(list(), range);
436     }

444         final GPathResult result = (GPathResult)getAt(index);

448             result.replaceBody(newValue);

452     /**
453      * Provides an Iterator over all the nodes of this GPathResult using a depth-first traversal.
454      *
455      * @return the <code>Iterator</code> of (depth-first) ordered GPathResults
456      */
457     public Iterator depthFirst() {
458         return new Iterator() {
459             private final List list = new LinkedList();
460             private final Stack stack = new Stack();
461             private Iterator iter = iterator();
462             private GPathResult next = getNextByDepth();
463 
464             public boolean hasNext() {
465                 return this.next != null;
466             }
467 
468             public Object next() {
469                 try {
470                     return this.next;
471                 } finally {
472                     this.next = getNextByDepth();
473                 }
474             }
475 
476             public void remove() {
477                 throw new UnsupportedOperationException();
478             }
479 
480             private GPathResult getNextByDepth() {
481                 while (this.iter.hasNext()) {
482                     final GPathResult node = (GPathResult) this.iter.next();
483                     this.list.add(node);
484                     this.stack.push(this.iter);
485                     this.iter = node.children().iterator();
486                 }
487 
488                 if (this.list.isEmpty()) {
489                     return null;
490                 } else {
491                     GPathResult result = (GPathResult) this.list.get(0);
492                     this.list.remove(0);
493                     this.iter = (Iterator) this.stack.pop();
494                     return result;
495                 }
496             }
497         };
498     }

500     /**
501      * Provides an Iterator over all the nodes of this GPathResult using a breadth-first traversal.
502      *
503      * @return the <code>Iterator</code> of (breadth-first) ordered GPathResults
504      */
505     public Iterator breadthFirst() {
506         return new Iterator() {
507             private final List list = new LinkedList();
508             private Iterator iter = iterator();
509             private GPathResult next = getNextByBreadth();
510 
511             public boolean hasNext() {
512                 return this.next != null;
513             }
514 
515             public Object next() {
516                 try {
517                     return this.next;
518                 } finally {
519                     this.next = getNextByBreadth();
520                 }
521             }
522 
523             public void remove() {
524                 throw new UnsupportedOperationException();
525             }
526 
527             private GPathResult getNextByBreadth() {
528                 List children = new ArrayList();
529                 while (this.iter.hasNext() || !children.isEmpty()) {
530                     if (this.iter.hasNext()) {
531                         final GPathResult node = (GPathResult) this.iter.next();
532                         this.list.add(node);
533                         this.list.add(this.iter);
534                         children.add(node.children());
535                     } else {
536                         List nextLevel = new ArrayList();
537                         for (Object child : children) {
538                             GPathResult next = (GPathResult) child;
539                             Iterator iterator = next.iterator();
540                             while (iterator.hasNext()) {
541                                 nextLevel.add(iterator.next());
542                             }
543                         }
544                         this.iter = nextLevel.iterator();
545                         children = new ArrayList();
546                     }
547                 }
548                 if (this.list.isEmpty()) {
549                     return null;
550                 } else {
551                     GPathResult result = (GPathResult) this.list.get(0);
552                     this.list.remove(0);
553                     this.iter = (Iterator) this.list.get(0);
554                     this.list.remove(0);
555                     return result;
556                 }
557             }
558         };
559     }

561     /**
562      * Creates a list of objects representing this GPathResult.
563      *
564      * @return a list representing of this GPathResult
565      */
566     public List list() {
567         final Iterator iter = nodeIterator();
568         final List result = new LinkedList();
569         while (iter.hasNext()) {
570             result.add(new NodeChild((Node) iter.next(), this.parent, this.namespacePrefix, this.namespaceTagHints));
571         }
572         return result;
573     }

584     /**
585      * Creates a Closure representing the body of this GPathResult.
586      *
587      * @return the body of this GPathResult, converted to a <code>Closure</code>
588      */
589     public Closure getBody() {
590         return new Closure(this.parent,this) {
591             public void doCall(Object[] args) {
592                 final GroovyObject delegate = (GroovyObject)getDelegate();
593                 final GPathResult thisObject = (GPathResult)getThisObject();
594 
595                 Node node = (Node)thisObject.getAt(0);
596                 List children = node.children();
597 
598                 for (Object child : children) {
599                     delegate.getProperty("mkp");
600                     if (child instanceof Node) {
601                         delegate.invokeMethod("yield", new Object[]{new NodeChild((Node) child, thisObject, "*", null)});
602                     } else {
603                         delegate.invokeMethod("yield", new Object[]{child});
604                     }
605                 }                
606             }
607         };
608     }
