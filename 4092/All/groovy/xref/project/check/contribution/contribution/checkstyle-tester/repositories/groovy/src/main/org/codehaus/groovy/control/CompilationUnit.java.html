126     /**
127      * Initializes the CompilationUnit with a CodeSource for controlling
128      * security stuff, a class loader for loading classes, and a class
129      * loader for loading AST transformations.
130      * <b>Note</b> The transform loader must be
131      * able to load compiler classes. That means CompilationUnit.class.classLoader
132      * must be at last a parent to transformLoader. The other loader has no such constraint.
133      *
134      * @param transformLoader - the loader for transforms
135      * @param loader          - loader used to resolve classes against during compilation
136      * @param security        - security setting for the compilation
137      * @param configuration   - compilation configuration
138      */
139     public CompilationUnit(CompilerConfiguration configuration, CodeSource security,
140                            GroovyClassLoader loader, GroovyClassLoader transformLoader) {
141         super(configuration, loader, null);
142 
143         this.astTransformationsContext = new ASTTransformationsContext(this, transformLoader);
144         this.names = new ArrayList<String>();
145         this.queuedSources = new LinkedList<SourceUnit>();
146         this.sources = new HashMap<String, SourceUnit>();
147         this.summariesBySourceName = new HashMap();
148         this.summariesByPublicClassName = new HashMap();
149         this.classSourcesByPublicClassName = new HashMap();
150 
151         this.ast = new CompileUnit(this.classLoader, security, this.configuration);
152         this.generatedClasses = new ArrayList<GroovyClass>();
153 
154         this.verifier = new Verifier();
155         this.resolveVisitor = new ResolveVisitor(this);
156         this.staticImportVisitor = new StaticImportVisitor();
157         this.optimizer = new OptimizerVisitor(this);
158 
159         phaseOperations = new LinkedList[Phases.ALL + 1];
160         newPhaseOperations = new LinkedList[Phases.ALL + 1];
161         for (int i = 0; i < phaseOperations.length; i++) {
162             phaseOperations[i] = new LinkedList();
163             newPhaseOperations[i] = new LinkedList();
164         }
165         addPhaseOperation(new SourceUnitOperation() {
166             public void call(SourceUnit source) throws CompilationFailedException {
167                 source.parse();
168             }
169         }, Phases.PARSING);
170         addPhaseOperation(convert, Phases.CONVERSION);
171         addPhaseOperation(new PrimaryClassNodeOperation() {
172             public void call(SourceUnit source, GeneratorContext context,
173                              ClassNode classNode) throws CompilationFailedException {
174                 EnumVisitor ev = new EnumVisitor(CompilationUnit.this, source);
175                 ev.visitClass(classNode);
176             }
177         }, Phases.CONVERSION);
178         addPhaseOperation(resolve, Phases.SEMANTIC_ANALYSIS);
179         addPhaseOperation(staticImport, Phases.SEMANTIC_ANALYSIS);
180         addPhaseOperation(new PrimaryClassNodeOperation() {
181             @Override
182             public void call(SourceUnit source, GeneratorContext context,
183                              ClassNode classNode) throws CompilationFailedException {
184                 InnerClassVisitor iv = new InnerClassVisitor(CompilationUnit.this, source);
185                 iv.visitClass(classNode);
186             }
187         }, Phases.SEMANTIC_ANALYSIS);
188         addPhaseOperation(new PrimaryClassNodeOperation() {
189             public void call(SourceUnit source, GeneratorContext context,
190                              ClassNode classNode) throws CompilationFailedException {
191                 TraitComposer.doExtendTraits(classNode, source, CompilationUnit.this);
192             }
193         }, Phases.CANONICALIZATION);
194         addPhaseOperation(compileCompleteCheck, Phases.CANONICALIZATION);
195         addPhaseOperation(classgen, Phases.CLASS_GENERATION);
196         addPhaseOperation(output);
197 
198         addPhaseOperation(new PrimaryClassNodeOperation() {
199             @Override
200             public void call(SourceUnit source, GeneratorContext context,
201                              ClassNode classNode) throws CompilationFailedException {
202                 AnnotationCollectorTransform.ClassChanger actt = new AnnotationCollectorTransform.ClassChanger();
203                 actt.transformClass(classNode);
204             }
205         }, Phases.SEMANTIC_ANALYSIS);
206         ASTTransformationVisitor.addPhaseOperations(this);
207         addPhaseOperation(new PrimaryClassNodeOperation() {
208             @Override
209             public void call(SourceUnit source, GeneratorContext context,
210                              ClassNode classNode) throws CompilationFailedException {
211                 StaticVerifier sv = new StaticVerifier();
212                 sv.visitClass(classNode, source);
213             }
214         }, Phases.SEMANTIC_ANALYSIS);
215         addPhaseOperation(new PrimaryClassNodeOperation() {
216             @Override
217             public void call(SourceUnit source, GeneratorContext context,
218                              ClassNode classNode) throws CompilationFailedException {
219                 InnerClassCompletionVisitor iv = new InnerClassCompletionVisitor(CompilationUnit.this, source);
220                 iv.visitClass(classNode);
221             }
222         }, Phases.CANONICALIZATION);
223         addPhaseOperation(new PrimaryClassNodeOperation() {
224             public void call(SourceUnit source, GeneratorContext context,
225                              ClassNode classNode) throws CompilationFailedException {
226                 EnumCompletionVisitor ecv = new EnumCompletionVisitor(CompilationUnit.this, source);
227                 ecv.visitClass(classNode);
228             }
229         }, Phases.CANONICALIZATION);
230 
231         // apply configuration customizers if any
232         if (configuration != null) {
233             final List<CompilationCustomizer> customizers = configuration.getCompilationCustomizers();
234             for (CompilationCustomizer customizer : customizers) {
235                 if (customizer instanceof CompilationUnitAware) {
236                     ((CompilationUnitAware) customizer).setCompilationUnit(this);
237                 }
238                 addPhaseOperation(customizer, customizer.getPhase().getPhaseNumber());
239             }
240         }
241         this.classgenCallback = null;
242         this.classNodeResolver = new ClassNodeResolver();
243     }

420      * Adds a SourceUnit to the unit.
421      */
422     public SourceUnit addSource(SourceUnit source) {
423         String name = source.getName();
424         source.setClassLoader(this.classLoader);
425         for (SourceUnit su : queuedSources) {
426             if (name.equals(su.getName())) return su;
427         }
428         queuedSources.add(source);
429         return source;
430     }

528     /**
529      * Compiles the compilation unit from sources.
530      */
531     public void compile(int throughPhase) throws CompilationFailedException {
532         //
533         // To support delta compilations, we always restart
534         // the compiler.  The individual passes are responsible
535         // for not reprocessing old code.
536         gotoPhase(Phases.INITIALIZATION);
537         throughPhase = Math.min(throughPhase, Phases.ALL);
538 
539         while (throughPhase >= phase && phase <= Phases.ALL) {
540 
541             if (phase == Phases.SEMANTIC_ANALYSIS) {
542                 doPhaseOperation(resolve);
543                 if (dequeued()) continue;
544             }
545 
546             processPhaseOperations(phase);
547             // Grab processing may have brought in new AST transforms into various phases, process them as well
548             processNewPhaseOperations(phase);
549 
550             if (progressCallback != null) progressCallback.call(this, phase);
551             completePhase();
552             applyToSourceUnits(mark);
553 
554             if (dequeued()) continue;
555 
556             gotoPhase(phase + 1);
557 
558             if (phase == Phases.CLASS_GENERATION) {
559                 sortClasses();
560             }
561         }
562 
563         errorCollector.failIfErrors();
564     }
565 
566     private void processPhaseOperations(int ph) {
567         LinkedList ops = phaseOperations[ph];
568         for (Object next : ops) {
569             doPhaseOperation(next);
570         }
571     }
572 
573     private void processNewPhaseOperations(int currPhase) {
574         recordPhaseOpsInAllOtherPhases(currPhase);
575         LinkedList currentPhaseNewOps = newPhaseOperations[currPhase];
576         while (!currentPhaseNewOps.isEmpty()) {
577             Object operation = currentPhaseNewOps.removeFirst();
578             // push this operation to master list and then process it.
579             phaseOperations[currPhase].add(operation);
580             doPhaseOperation(operation);
581             // if this operation has brought in more phase ops for ast transforms, keep recording them
582             // in master list of other phases and keep processing them for this phase.
583             recordPhaseOpsInAllOtherPhases(currPhase);
584             currentPhaseNewOps = newPhaseOperations[currPhase];
585         }
586 
587     }
588 
589     private void doPhaseOperation(Object operation) {
590         if (operation instanceof PrimaryClassNodeOperation) {
591             applyToPrimaryClassNodes((PrimaryClassNodeOperation) operation);
592         } else if (operation instanceof SourceUnitOperation) {
593             applyToSourceUnits((SourceUnitOperation) operation);
594         } else {
595             applyToGeneratedGroovyClasses((GroovyClassOperation) operation);
596         }
597     }
598 
599     private void recordPhaseOpsInAllOtherPhases(int currPhase) {
600         // apart from current phase, push new operations for every other phase in the master phase ops list
601         for (int ph = Phases.INITIALIZATION; ph <= Phases.ALL; ph++) {
602             if (ph != currPhase && !newPhaseOperations[ph].isEmpty()) {
603                 phaseOperations[ph].addAll(newPhaseOperations[ph]);
604                 newPhaseOperations[ph].clear();
605             }
606         }
607     }
608 
609     private void sortClasses() throws CompilationFailedException {
610         for (ModuleNode module : this.ast.getModules()) {
611             module.sortClasses();
612         }
613     }

616     /**
617      * Dequeues any source units add through addSource and resets the compiler phase
618      * to initialization.
619      * <p>
620      * Note: this does not mean a file is recompiled. If a SourceUnit has already passed
621      * a phase it is skipped until a higher phase is reached.
622      *
623      * @return true if there was a queued source
624      * @throws CompilationFailedException
625      */
626     protected boolean dequeued() throws CompilationFailedException {
627         boolean dequeue = !queuedSources.isEmpty();
628         while (!queuedSources.isEmpty()) {
629             SourceUnit su = queuedSources.removeFirst();
630             String name = su.getName();
631             names.add(name);
632             sources.put(name, su);
633         }
634         if (dequeue) {
635             gotoPhase(Phases.INITIALIZATION);
636         }
637         return dequeue;
638     }

641      * Resolves all types
642      */
643     private final SourceUnitOperation resolve = new SourceUnitOperation() {
644         public void call(SourceUnit source) throws CompilationFailedException {
645             List<ClassNode> classes = source.ast.getClasses();
646             for (ClassNode node : classes) {
647                 VariableScopeVisitor scopeVisitor = new VariableScopeVisitor(source);
648                 scopeVisitor.visitClass(node);
649 
650                 resolveVisitor.setClassNodeResolver(classNodeResolver);
651                 resolveVisitor.startResolving(node, source);
652             }
653 
654         }
655     };

664      * Runs convert() on a single SourceUnit.

666     private SourceUnitOperation convert = new SourceUnitOperation() {
667         public void call(SourceUnit source) throws CompilationFailedException {
668             source.convert();
669             CompilationUnit.this.ast.addModule(source.getAST());

678     private GroovyClassOperation output = new GroovyClassOperation() {
679         public void call(GroovyClass gclass) throws CompilationFailedException {
680             String name = gclass.getName().replace('.', File.separatorChar) + ".class";
681             File path = new File(configuration.getTargetDirectory(), name);
682 
683             //
684             // Ensure the path is ready for the file
685             //
686             File directory = path.getParentFile();
687             if (directory != null && !directory.exists()) {
688                 directory.mkdirs();
689             }
690 
691             //
692             // Create the file and write out the data
693             //
694             byte[] bytes = gclass.getBytes();
695 
696             FileOutputStream stream = null;
697             try {
698                 stream = new FileOutputStream(path);
699                 stream.write(bytes, 0, bytes.length);
700             } catch (IOException e) {
701                 getErrorCollector().addError(Message.create(e.getMessage(), CompilationUnit.this));
702             } finally {
703                 if (stream != null) {
704                     try {
705                         stream.close();
706                     } catch (Exception e) {
707                         // Ignore
708                     }
709                 }
710             }
711         }
712     };

714     /* checks if all needed classes are compiled before generating the bytecode */
715     private SourceUnitOperation compileCompleteCheck = new SourceUnitOperation() {
716         public void call(SourceUnit source) throws CompilationFailedException {
717             List<ClassNode> classes = source.ast.getClasses();
718             for (ClassNode node : classes) {
719                 CompileUnit cu = node.getCompileUnit();
720                 for (Iterator iter = cu.iterateClassNodeToCompile(); iter.hasNext();) {
721                     String name = (String) iter.next();
722                     SourceUnit su = ast.getScriptSourceLocation(name);
723                     List<ClassNode> classesInSourceUnit = su.ast.getClasses();
724                     StringBuilder message = new StringBuilder();
725                     message
726                             .append("Compilation incomplete: expected to find the class ")
727                             .append(name)
728                             .append(" in ")
729                             .append(su.getName());
730                     if (classesInSourceUnit.isEmpty()) {
731                         message.append(", but the file seems not to contain any classes");
732                     } else {
733                         message.append(", but the file contains the classes: ");
734                         boolean first = true;
735                         for (ClassNode cn : classesInSourceUnit) {
736                             if (!first) {
737                                 message.append(", ");
738                             } else {
739                                 first = false;
740                             }
741                             message.append(cn.getName());
742                         }
743                     }
744 
745                     getErrorCollector().addErrorAndContinue(
746                             new SimpleMessage(message.toString(), CompilationUnit.this)
747                     );
748                     iter.remove();
749                 }
750             }
751         }
752     };

755     /**
756      * Runs classgen() on a single ClassNode.
757      */
758     private PrimaryClassNodeOperation classgen = new PrimaryClassNodeOperation() {
759         public boolean needSortedInput() {
760             return true;
761         }
762 
763         public void call(SourceUnit source, GeneratorContext context, ClassNode classNode) throws CompilationFailedException {
764 
765             optimizer.visitClass(classNode, source); // GROOVY-4272: repositioned it here from staticImport
766 
767             if(!classNode.isSynthetic()) {
768                 GenericsVisitor genericsVisitor = new GenericsVisitor(source);
769                 genericsVisitor.visitClass(classNode);
770             }
771 
772             //
773             // Run the Verifier on the outer class
774             //
775             try {
776                 verifier.visitClass(classNode);
777             } catch (GroovyRuntimeException rpe) {
778                 ASTNode node = rpe.getNode();
779                 getErrorCollector().addError(
780                         new SyntaxException(rpe.getMessage(), node.getLineNumber(), node.getColumnNumber(), node.getLastLineNumber(), node.getLastColumnNumber()),
781                         source
782                 );
783             }
784 
785             LabelVerifier lv = new LabelVerifier(source);
786             lv.visitClass(classNode);
787 
788             ClassCompletionVerifier completionVerifier = new ClassCompletionVerifier(source);
789             completionVerifier.visitClass(classNode);
790 
791             ExtendedVerifier xverifier = new ExtendedVerifier(source);
792             xverifier.visitClass(classNode);
793 
794             // because the class may be generated even if a error was found
795             // and that class may have an invalid format we fail here if needed
796             getErrorCollector().failIfErrors();
797 
798             //
799             // Prep the generator machinery
800             //
801             ClassVisitor visitor = createClassVisitor();
802             
803             String sourceName = (source == null ? classNode.getModule().getDescription() : source.getName());
804             // only show the file name and its extension like javac does in its stacktraces rather than the full path
805             // also takes care of both \ and / depending on the host compiling environment
806             if (sourceName != null)
807                 sourceName = sourceName.substring(Math.max(sourceName.lastIndexOf('\\'), sourceName.lastIndexOf('/')) + 1);
808             AsmClassGenerator generator = new AsmClassGenerator(source, context, visitor, sourceName);
809 
810             //
811             // Run the generation and create the class (if required)
812             //
813             generator.visitClass(classNode);
814 
815             byte[] bytes = ((ClassWriter) visitor).toByteArray();
816             generatedClasses.add(new GroovyClass(classNode.getName(), bytes));
817 
818             //
819             // Handle any callback that's been set
820             //
821             if (CompilationUnit.this.classgenCallback != null) {
822                 classgenCallback.call(visitor, classNode);
823             }
824 
825             //
826             // Recurse for inner classes
827             //
828             LinkedList innerClasses = generator.getInnerClasses();
829             while (!innerClasses.isEmpty()) {
830                 classgen.call(source, context, (ClassNode) innerClasses.removeFirst());
831             }
832         }
833     };

835     protected ClassVisitor createClassVisitor() {
836         CompilerConfiguration config = getConfiguration();
837         int computeMaxStackAndFrames = ClassWriter.COMPUTE_MAXS;
838         if (CompilerConfiguration.isPostJDK7(config.getTargetBytecode())
839                 || Boolean.TRUE.equals(config.getOptimizationOptions().get("indy"))) {
840             computeMaxStackAndFrames += ClassWriter.COMPUTE_FRAMES;
841         }
842         return new ClassWriter(computeMaxStackAndFrames) {
843             private ClassNode getClassNode(String name) {
844                 // try classes under compilation
845                 CompileUnit cu = getAST();
846                 ClassNode cn = cu.getClass(name);
847                 if (cn!=null) return cn;
848                 // try inner classes
849                 cn = cu.getGeneratedInnerClass(name);
850                 if (cn!=null) return cn;
851                 // try class loader classes
852                 try {
853                     cn = ClassHelper.make(
854                             cu.getClassLoader().loadClass(name,false,true),
855                             false);
856                 } catch (Exception e) {
857                     throw new GroovyBugError(e);
858                 }
859                 return cn;
860             }
861             private ClassNode getCommonSuperClassNode(ClassNode c, ClassNode d) {
862                 // adapted from ClassWriter code
863                 if (c.isDerivedFrom(d)) return d;
864                 if (d.isDerivedFrom(c)) return c;
865                 if (c.isInterface() || d.isInterface()) return ClassHelper.OBJECT_TYPE;
866                 do {
867                     c = c.getSuperClass();
868                 } while (c!=null && !d.isDerivedFrom(c));
869                 if (c==null) return ClassHelper.OBJECT_TYPE;
870                 return c;
871             }
872             @Override
873             protected String getCommonSuperClass(String arg1, String arg2) {
874                 ClassNode a = getClassNode(arg1.replace('/', '.')); 
875                 ClassNode b = getClassNode(arg2.replace('/', '.'));
876                 return getCommonSuperClassNode(a,b).getName().replace('.','/');
877             }
878 
879         };
880     }

968     private int getSuperClassCount(ClassNode element) {
969         int count = 0;
970         while (element != null) {
971             count++;
972             element = element.getSuperClass();
973         }
974         return count;
975     }

977     private int getSuperInterfaceCount(ClassNode element) {
978         int count = 1;
979         ClassNode[] interfaces = element.getInterfaces();
980         for (ClassNode anInterface : interfaces) {
981             count = Math.max(count, getSuperInterfaceCount(anInterface) + 1);
982         }
983         return count;
984     }

986     private List getPrimaryClassNodes(boolean sort) {
987         List<ClassNode> unsorted = new ArrayList<ClassNode>();
988         for (ModuleNode module : this.ast.getModules()) {
989             for (ClassNode classNode : module.getClasses()) {
990                 unsorted.add(classNode);
991             }
992         }
993 
994         if (!sort) return unsorted;
995 
996         int[] indexClass = new int[unsorted.size()];
997         int[] indexInterface = new int[unsorted.size()];
998         {
999             int i = 0;
1000             for (Iterator<ClassNode> iter = unsorted.iterator(); iter.hasNext(); i++) {
1001                 ClassNode element = iter.next();
1002                 if (element.isInterface()) {
1003                     indexInterface[i] = getSuperInterfaceCount(element);
1004                     indexClass[i] = -1;
1005                 } else {
1006                     indexClass[i] = getSuperClassCount(element);
1007                     indexInterface[i] = -1;
1008                 }
1009             }
1010         }
1011 
1012         List<ClassNode> sorted = getSorted(indexInterface, unsorted);
1013         sorted.addAll(getSorted(indexClass, unsorted));
1014         return sorted;
1015     }
1016 
1017     private List<ClassNode> getSorted(int[] index, List<ClassNode> unsorted) {
1018         List<ClassNode> sorted = new ArrayList<ClassNode>(unsorted.size());
1019         for (int i = 0; i < unsorted.size(); i++) {
1020             int min = -1;
1021             for (int j = 0; j < unsorted.size(); j++) {
1022                 if (index[j] == -1) continue;
1023                 if (min == -1) {
1024                     min = j;
1025                 } else if (index[j] < index[min]) {
1026                     min = j;
1027                 }
1028             }
1029             if (min == -1) break;
1030             sorted.add(unsorted.get(min));
1031             index[min] = -1;
1032         }
1033         return sorted;
1034     }

1036     /**
1037      * A loop driver for applying operations to all primary ClassNodes in
1038      * our AST.  Automatically skips units that have already been processed
1039      * through the current phase.
1040      */
1041     public void applyToPrimaryClassNodes(PrimaryClassNodeOperation body) throws CompilationFailedException {
1042         Iterator classNodes = getPrimaryClassNodes(body.needSortedInput()).iterator();
1043         while (classNodes.hasNext()) {
1044             SourceUnit context = null;
1045             try {
1046                 ClassNode classNode = (ClassNode) classNodes.next();
1047                 context = classNode.getModule().getContext();
1048                 if (context == null || context.phase < phase || (context.phase == phase && !context.phaseComplete)) {
1049                     int offset = 1;
1050                     Iterator<InnerClassNode> iterator = classNode.getInnerClasses();
1051                     while (iterator.hasNext()) {
1052                         iterator.next();
1053                         offset++;
1054                     }
1055                     body.call(context, new GeneratorContext(this.ast, offset), classNode);
1056                 }
1057             } catch (CompilationFailedException e) {
1058                 // fall through, getErrorReporter().failIfErrors() will trigger
1059             } catch (NullPointerException npe) {
1060                 GroovyBugError gbe = new GroovyBugError("unexpected NullpointerException", npe);
1061                 changeBugText(gbe, context);
1062                 throw gbe;
1063             } catch (GroovyBugError e) {
1064                 changeBugText(e, context);
1065                 throw e;
1066             } catch (NoClassDefFoundError e) {
1067                 // effort to get more logging in case a dependency of a class is loaded
1068                 // although it shouldn't have
1069                 convertUncaughtExceptionToCompilationError(e);
1070             } catch (Exception e) {
1071                 convertUncaughtExceptionToCompilationError(e);
1072             }
1073         }
1074 
1075         getErrorCollector().failIfErrors();
1076     }
1077 
1078     private void convertUncaughtExceptionToCompilationError(final Throwable e) {
1079         // check the exception for a nested compilation exception
1080         ErrorCollector nestedCollector = null;
1081         for (Throwable next = e.getCause(); next != e && next != null; next = next.getCause()) {
1082             if (!(next instanceof MultipleCompilationErrorsException)) continue;
1083             MultipleCompilationErrorsException mcee = (MultipleCompilationErrorsException) next;
1084             nestedCollector = mcee.collector;
1085             break;
1086         }
1087 
1088         if (nestedCollector != null) {
1089             getErrorCollector().addCollectorContents(nestedCollector);
1090         } else {
1091             Exception err = e instanceof Exception?((Exception)e):new RuntimeException(e);
1092             getErrorCollector().addError(new ExceptionMessage(err, configuration.getDebug(), this));
1093         }
1094     }

1096     public void applyToGeneratedGroovyClasses(GroovyClassOperation body) throws CompilationFailedException {
1097         if (this.phase != Phases.OUTPUT && !(this.phase == Phases.CLASS_GENERATION && this.phaseComplete)) {
1098             throw new GroovyBugError("CompilationUnit not ready for output(). Current phase=" + getPhaseDescription());
1099         }
1100 
1101         for (GroovyClass gclass : this.generatedClasses) {
1102             //
1103             // Get the class and calculate its filesystem name
1104             //
1105             try {
1106                 body.call(gclass);
1107             } catch (CompilationFailedException e) {
1108                 // fall through, getErrorReporter().failIfErrors() will trigger
1109             } catch (NullPointerException npe) {
1110                 throw npe;
1111             } catch (GroovyBugError e) {
1112                 changeBugText(e, null);
1113                 throw e;
1114             } catch (Exception e) {
1115                 throw new GroovyBugError(e);
1116             }
1117         }

1120     }
