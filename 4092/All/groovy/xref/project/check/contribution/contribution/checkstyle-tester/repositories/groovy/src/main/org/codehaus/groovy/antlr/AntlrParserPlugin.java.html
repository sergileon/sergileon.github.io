109     public /*final*/ Reduction parseCST(final SourceUnit sourceUnit, Reader reader) throws CompilationFailedException {
110         final SourceBuffer sourceBuffer = new SourceBuffer();
111         transformCSTIntoAST(sourceUnit, reader, sourceBuffer);
112         processAST();
113         return outputAST(sourceUnit, sourceBuffer);
114     }

116     protected void transformCSTIntoAST(SourceUnit sourceUnit, Reader reader, SourceBuffer sourceBuffer) throws CompilationFailedException {
117         ast = null;
118 
119         setController(sourceUnit);
120 
121         // TODO find a way to inject any GroovyLexer/GroovyRecognizer
122 
123         UnicodeEscapingReader unicodeReader = new UnicodeEscapingReader(reader, sourceBuffer);
124         UnicodeLexerSharedInputState inputState = new UnicodeLexerSharedInputState(unicodeReader);
125         GroovyLexer lexer = new GroovyLexer(inputState);
126         unicodeReader.setLexer(lexer);
127         GroovyRecognizer parser = GroovyRecognizer.make(lexer);
128         parser.setSourceBuffer(sourceBuffer);
129         tokenNames = parser.getTokenNames();
130         parser.setFilename(sourceUnit.getName());
131 
132         // start parsing at the compilationUnit rule
133         try {
134             parser.compilationUnit();
135         }
136         catch (TokenStreamRecognitionException tsre) {
137             RecognitionException e = tsre.recog;
138             SyntaxException se = new SyntaxException(e.getMessage(), e, e.getLine(), e.getColumn());
139             se.setFatal(true);
140             sourceUnit.addError(se);
141         }
142         catch (RecognitionException e) {
143             SyntaxException se = new SyntaxException(e.getMessage(), e, e.getLine(), e.getColumn());
144             se.setFatal(true);
145             sourceUnit.addError(se);
146         }
147         catch (TokenStreamException e) {
148             sourceUnit.addException(e);
149         }
150 
151         ast = parser.getAST();
152     }

155         AntlrASTProcessor snippets = new AntlrASTProcessSnippets();
156         ast = snippets.process(ast);
157     }

170     private void outputASTInVariousFormsIfNeeded(SourceUnit sourceUnit, SourceBuffer sourceBuffer) {
171         // straight xstream output of AST
172         String formatProp = System.getProperty("ANTLR.AST".toLowerCase()); // uppercase to hide from jarjar
173 
174         if ("xml".equals(formatProp)) {
175             saveAsXML(sourceUnit.getName(), ast);
176         }
177 
178         // 'pretty printer' output of AST
179         if ("groovy".equals(formatProp)) {
180             try {
181                 PrintStream out = new PrintStream(new FileOutputStream(sourceUnit.getName() + ".pretty.groovy"));
182                 Visitor visitor = new SourcePrinter(out, tokenNames);
183                 AntlrASTProcessor treewalker = new SourceCodeTraversal(visitor);
184                 treewalker.process(ast);
185             } catch (FileNotFoundException e) {
186                 System.out.println("Cannot create " + sourceUnit.getName() + ".pretty.groovy");
187             }
188         }
189 
190         // output AST in format suitable for opening in http://freemind.sourceforge.net
191         // which is a really nice way of seeing the AST, folding nodes etc
192         if ("mindmap".equals(formatProp)) {
193             try {
194                 PrintStream out = new PrintStream(new FileOutputStream(sourceUnit.getName() + ".mm"));
195                 Visitor visitor = new MindMapPrinter(out, tokenNames);
196                 AntlrASTProcessor treewalker = new PreOrderTraversal(visitor);
197                 treewalker.process(ast);
198             } catch (FileNotFoundException e) {
199                 System.out.println("Cannot create " + sourceUnit.getName() + ".mm");
200             }
201         }
202 
203         // include original line/col info and source code on the mindmap output
204         if ("extendedMindmap".equals(formatProp)) {
205             try {
206                 PrintStream out = new PrintStream(new FileOutputStream(sourceUnit.getName() + ".mm"));
207                 Visitor visitor = new MindMapPrinter(out, tokenNames, sourceBuffer);
208                 AntlrASTProcessor treewalker = new PreOrderTraversal(visitor);
209                 treewalker.process(ast);
210             } catch (FileNotFoundException e) {
211                 System.out.println("Cannot create " + sourceUnit.getName() + ".mm");
212             }
213         }
214 
215         // html output of AST
216         if ("html".equals(formatProp)) {
217             try {
218                 PrintStream out = new PrintStream(new FileOutputStream(sourceUnit.getName() + ".html"));
219                 List<VisitorAdapter> v = new ArrayList<VisitorAdapter>();
220                 v.add(new NodeAsHTMLPrinter(out, tokenNames));
221                 v.add(new SourcePrinter(out, tokenNames));
222                 Visitor visitors = new CompositeVisitor(v);
223                 AntlrASTProcessor treewalker = new SourceCodeTraversal(visitors);
224                 treewalker.process(ast);
225             } catch (FileNotFoundException e) {
226                 System.out.println("Cannot create " + sourceUnit.getName() + ".html");
227             }
228         }
229     }

243     public ModuleNode buildAST(SourceUnit sourceUnit, ClassLoader classLoader, Reduction cst) throws ParserException {
244         setClassLoader(classLoader);
245         makeModule();
246         try {
247             convertGroovy(ast);
248             if (output.getStatementBlock().isEmpty() && output.getMethods().isEmpty() && output.getClasses().isEmpty()) {
249                 output.addStatement(ReturnStatement.RETURN_NULL_OR_VOID);
250             }
251 
252             // set the script source position
253 
254             ClassNode scriptClassNode = output.getScriptClassDummy();
255             if (scriptClassNode != null) {
256                 List<Statement> statements = output.getStatementBlock().getStatements();
257                 if (statements.size() > 0) {
258                     Statement firstStatement = statements.get(0);
259                     Statement lastStatement = statements.get(statements.size() - 1);
260 
261                     scriptClassNode.setSourcePosition(firstStatement);
262                     scriptClassNode.setLastColumnNumber(lastStatement.getLastColumnNumber());
263                     scriptClassNode.setLastLineNumber(lastStatement.getLastLineNumber());
264                 }
265             }
266         }
267         catch (ASTRuntimeException e) {
268             throw new ASTParserException(e.getMessage() + ". File: " + sourceUnit.getName(), e);
269         }
270         return output;
271     }

276     protected void convertGroovy(AST node) {
277         while (node != null) {
278             int type = node.getType();
279             switch (type) {
280                 case PACKAGE_DEF:
281                     packageDef(node);
282                     break;
283 
284                 case STATIC_IMPORT:
285                 case IMPORT:
286                     importDef(node);
287                     break;
288 
289                 case TRAIT_DEF:
290                 case CLASS_DEF:
291                     classDef(node);
292                     break;
293 
294                 case INTERFACE_DEF:
295                     interfaceDef(node);
296                     break;
297 
298                 case METHOD_DEF:
299                     methodDef(node);
300                     break;
301 
302                 case ENUM_DEF:
303                     enumDef(node);
304                     break;
305 
306                 case ANNOTATION_DEF:
307                     annotationDef(node);
308                     break;
309 
310                 default: {
311                     Statement statement = statement(node);
312                     output.addStatement(statement);
313                 }
314             }
315             node = node.getNextSibling();
316         }
317     }

322     protected void packageDef(AST packageDef) {
323         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
324         AST node = packageDef.getFirstChild();
325         if (isType(ANNOTATIONS, node)) {
326             processAnnotations(annotations, node);
327             node = node.getNextSibling();
328         }
329         String name = qualifiedName(node);
330         // TODO should we check package node doesn't already exist? conflict?
331         PackageNode packageNode = setPackage(name, annotations);
332         configureAST(packageNode, packageDef);
333     }

335     protected void importDef(AST importNode) {
336         try {
337             // GROOVY-6094
338             output.putNodeMetaData(ImportNode.class, ImportNode.class);
339 
340             boolean isStatic = importNode.getType() == STATIC_IMPORT;
341             List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
342 
343             AST node = importNode.getFirstChild();
344             if (isType(ANNOTATIONS, node)) {
345                 processAnnotations(annotations, node);
346                 node = node.getNextSibling();
347             }
348 
349             String alias = null;
350             if (isType(LITERAL_as, node)) {
351                 //import is like "import Foo as Bar"
352                 node = node.getFirstChild();
353                 AST aliasNode = node.getNextSibling();
354                 alias = identifier(aliasNode);
355             }
356 
357             if (node.getNumberOfChildren() == 0) {
358                 String name = identifier(node);
359                 // import is like  "import Foo"
360                 ClassNode type = ClassHelper.make(name);
361                 configureAST(type, importNode);
362                 addImport(type, name, alias, annotations);
363                 return;
364             }
365 
366             AST packageNode = node.getFirstChild();
367             String packageName = qualifiedName(packageNode);
368             AST nameNode = packageNode.getNextSibling();
369             if (isType(STAR, nameNode)) {
370                 if (isStatic) {
371                     // import is like "import static foo.Bar.*"
372                     // packageName is actually a className in this case
373                     ClassNode type = ClassHelper.make(packageName);
374                     configureAST(type, importNode);
375                     addStaticStarImport(type, packageName, annotations);
376                 } else {
377                     // import is like "import foo.*"
378                     addStarImport(packageName, annotations);
379                 }
380 
381                 if (alias != null) throw new GroovyBugError(
382                         "imports like 'import foo.* as Bar' are not " +
383                                 "supported and should be caught by the grammar");
384             } else {
385                 String name = identifier(nameNode);
386                 if (isStatic) {
387                     // import is like "import static foo.Bar.method"
388                     // packageName is really class name in this case
389                     ClassNode type = ClassHelper.make(packageName);
390                     configureAST(type, importNode);
391                     addStaticImport(type, name, alias, annotations);
392                 } else {
393                     // import is like "import foo.Bar"
394                     ClassNode type = ClassHelper.make(packageName + "." + name);
395                     configureAST(type, importNode);
396                     addImport(type, name, alias, annotations);
397                 }
398             }
399         } finally {
400             // we're using node metadata here in order to fix GROOVY-6094
401             // without breaking external APIs
402             Object node = output.getNodeMetaData(ImportNode.class);
403             if (node!=null && node!=ImportNode.class) {
404                 configureAST((ImportNode)node, importNode);
405             }
406             output.removeNodeMetaData(ImportNode.class);
407         }
408     }

411         AST child = node.getFirstChild();
412         while (child != null) {
413             if (isType(ANNOTATION, child))
414                 annotations.add(annotation(child));
415             child = child.getNextSibling();
416         }

419     protected void annotationDef(AST classDef) {
420         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
421         AST node = classDef.getFirstChild();
422         int modifiers = Opcodes.ACC_PUBLIC;
423         if (isType(MODIFIERS, node)) {
424             modifiers = modifiers(node, annotations, modifiers);
425             checkNoInvalidModifier(classDef, "Annotation Definition", modifiers, Opcodes.ACC_SYNCHRONIZED, "synchronized");
426             node = node.getNextSibling();
427         }
428         modifiers |= Opcodes.ACC_ABSTRACT | Opcodes.ACC_INTERFACE | Opcodes.ACC_ANNOTATION;
429 
430         String name = identifier(node);
431         node = node.getNextSibling();
432         ClassNode superClass = ClassHelper.OBJECT_TYPE;
433 
434         GenericsType[] genericsType = null;
435         if (isType(TYPE_PARAMETERS, node)) {
436             genericsType = makeGenericsType(node);
437             node = node.getNextSibling();
438         }
439 
440         ClassNode[] interfaces = ClassNode.EMPTY_ARRAY;
441         if (isType(EXTENDS_CLAUSE, node)) {
442             interfaces = interfaces(node);
443             node = node.getNextSibling();
444         }
445 
446         boolean syntheticPublic = ((modifiers & Opcodes.ACC_SYNTHETIC) != 0);
447         modifiers &= ~Opcodes.ACC_SYNTHETIC;
448         classNode = new ClassNode(dot(getPackageName(), name), modifiers, superClass, interfaces, null);
449         classNode.setSyntheticPublic(syntheticPublic);
450         classNode.addAnnotations(annotations);
451         classNode.setGenericsTypes(genericsType);
452         classNode.addInterface(ClassHelper.Annotation_TYPE);
453         configureAST(classNode, classDef);
454 
455         assertNodeType(OBJBLOCK, node);
456         objectBlock(node);
457         output.addClass(classNode);
458         classNode = null;
459     }

493         }

496         boolean syntheticPublic = ((modifiers & Opcodes.ACC_SYNTHETIC) != 0);

513         objectBlock(node);
514         output.addClass(classNode);

518     }

532     protected Expression anonymousInnerClassDef(AST node) {
533         ClassNode oldNode = classNode;
534         ClassNode outerClass = getClassOrScript(oldNode);
535         String fullName = outerClass.getName() + '$' + innerClassCounter;
536         innerClassCounter++;
537         if (enumConstantBeingDef) {
538             classNode = new EnumConstantClassNode(outerClass, fullName, Opcodes.ACC_PUBLIC, ClassHelper.OBJECT_TYPE);
539         } else {
540             classNode = new InnerClassNode(outerClass, fullName, Opcodes.ACC_PUBLIC, ClassHelper.OBJECT_TYPE);
541         }
542         ((InnerClassNode) classNode).setAnonymous(true);
543         classNode.setEnclosingMethod(methodNode);
544 
545         assertNodeType(OBJBLOCK, node);
546         objectBlock(node);
547         output.addClass(classNode);
548         AnonymousInnerClassCarrier ret = new AnonymousInnerClassCarrier();
549         ret.innerClass = classNode;
550         classNode = oldNode;
551 
552         return ret;
553     }

555     protected void innerClassDef(AST classDef) {
556         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
557 
558         if (isType(TRAIT_DEF, classDef)) {
559             annotations.add(new AnnotationNode(ClassHelper.make("groovy.transform.Trait")));
560         }
561 
562         AST node = classDef.getFirstChild();
563         int modifiers = Opcodes.ACC_PUBLIC;
564         if (isType(MODIFIERS, node)) {
565             modifiers = modifiers(node, annotations, modifiers);
566             checkNoInvalidModifier(classDef, "Class", modifiers, Opcodes.ACC_SYNCHRONIZED, "synchronized");
567             node = node.getNextSibling();
568         }
569 
570         String name = identifier(node);
571         node = node.getNextSibling();
572 
573         GenericsType[] genericsType = null;
574         if (isType(TYPE_PARAMETERS, node)) {
575             genericsType = makeGenericsType(node);
576             node = node.getNextSibling();
577         }
578 
579         ClassNode superClass = null;
580         if (isType(EXTENDS_CLAUSE, node)) {
581             superClass = makeTypeWithArguments(node);
582             node = node.getNextSibling();
583         }
584 
585         ClassNode[] interfaces = ClassNode.EMPTY_ARRAY;
586         if (isType(IMPLEMENTS_CLAUSE, node)) {
587             interfaces = interfaces(node);
588             node = node.getNextSibling();
589         }
590 
591         // TODO read mixins
592         MixinNode[] mixins = {};
593         ClassNode outerClass = classNode;
594         boolean syntheticPublic = ((modifiers & Opcodes.ACC_SYNTHETIC) != 0);
595         modifiers &= ~Opcodes.ACC_SYNTHETIC;
596         if (classNode != null) {
597             name = classNode.getNameWithoutPackage() + "$" + name;
598             String fullName = dot(classNode.getPackageName(), name);
599             classNode = new InnerClassNode(classNode, fullName, modifiers, superClass, interfaces, mixins);
600         } else {
601             classNode = new ClassNode(dot(getPackageName(), name), modifiers, superClass, interfaces, mixins);
602         }
603         classNode.addAnnotations(annotations);
604         classNode.setGenericsTypes(genericsType);
605         classNode.setSyntheticPublic(syntheticPublic);
606         configureAST(classNode, classDef);
607 
608         // we put the class already in output to avoid the most inner classes
609         // will be used as first class later in the loader. The first class
610         // there determines what GCL#parseClass for example will return, so we
611         // have here to ensure it won't be the inner class
612         output.addClass(classNode);
613 
614         int oldClassCount = innerClassCounter;
615 
616         assertNodeType(OBJBLOCK, node);
617         objectBlock(node);
618 
619         classNode = outerClass;
620         innerClassCounter = oldClassCount;
621     }

623     protected void objectBlock(AST objectBlock) {
624         for (AST node = objectBlock.getFirstChild(); node != null; node = node.getNextSibling()) {
625             int type = node.getType();
626             switch (type) {
627                 case OBJBLOCK:
628                     objectBlock(node);
629                     break;
630 
631                 case ANNOTATION_FIELD_DEF:
632                 case METHOD_DEF:
633                     methodDef(node);
634                     break;
635 
636                 case CTOR_IDENT:
637                     constructorDef(node);
638                     break;
639 
640                 case VARIABLE_DEF:
641                     fieldDef(node);
642                     break;
643 
644                 case STATIC_INIT:
645                     staticInit(node);
646                     break;
647 
648                 case INSTANCE_INIT:
649                     objectInit(node);
650                     break;
651 
652                 case ENUM_DEF:
653                     enumDef(node);
654                     break;
655 
656                 case ENUM_CONSTANT_DEF:
657                     enumConstantDef(node);
658                     break;
659 
660                 case TRAIT_DEF:
661                 case CLASS_DEF:
662                     innerClassDef(node);
663                     break;
664 
665                 case INTERFACE_DEF:
666                     innerInterfaceDef(node);
667                     break;
668 
669                 default:
670                     unknownAST(node);
671             }
672         }

676         assertNodeType(ENUM_DEF, enumNode);
677         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();

679         AST node = enumNode.getFirstChild();
680         int modifiers = Opcodes.ACC_PUBLIC;

687         node = node.getNextSibling();

699         classNode = enumClass;

701         assertNodeType(OBJBLOCK, node);
702         objectBlock(node);

708     protected void enumConstantDef(AST node) {
709         enumConstantBeingDef = true;
710         assertNodeType(ENUM_CONSTANT_DEF, node);
711         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
712         AST element = node.getFirstChild();
713         if (isType(ANNOTATIONS, element)) {
714             processAnnotations(annotations, element);
715             element = element.getNextSibling();
716         }
717         String identifier = identifier(element);
718         Expression init = null;
719         element = element.getNextSibling();
720 
721         if (element != null) {
722             init = expression(element);
723             ClassNode innerClass;
724             if (element.getNextSibling() == null) {
725                 innerClass = getAnonymousInnerClassNode(init);
726                 if (innerClass != null) {
727                     init = null;
728                 }
729             } else {
730                 element = element.getNextSibling();
731                 Expression next = expression(element);
732                 innerClass = getAnonymousInnerClassNode(next);
733             }
734 
735             if (innerClass != null) {
736                 // we have to handle an enum constant with a class overriding
737                 // a method in which case we need to configure the inner class
738                 innerClass.setSuperClass(classNode.getPlainNodeReference());
739                 innerClass.setModifiers(classNode.getModifiers() | Opcodes.ACC_FINAL);
740                 // we use a ClassExpression for transportation to EnumVisitor
741                 Expression inner = new ClassExpression(innerClass);
742                 if (init == null) {
743                     ListExpression le = new ListExpression();
744                     le.addExpression(inner);
745                     init = le;
746                 } else {
747                     if (init instanceof ListExpression) {
748                         ((ListExpression) init).addExpression(inner);
749                     } else {
750                         ListExpression le = new ListExpression();
751                         le.addExpression(init);
752                         le.addExpression(inner);
753                         init = le;
754                     }
755                 }
756                 // and remove the final modifier from classNode to allow the sub class
757                 classNode.setModifiers(classNode.getModifiers() & ~Opcodes.ACC_FINAL);
758             } else if (isType(ELIST, element)) {
759                 if (init instanceof ListExpression && !((ListExpression) init).isWrapped()) {
760                     ListExpression le = new ListExpression();
761                     le.addExpression(init);
762                     init = le;
763                 }
764             }
765         }
766         FieldNode enumField = EnumHelper.addEnumConstant(classNode, identifier, init);
767         enumField.addAnnotations(annotations);
768         configureAST(enumField, node);
769         enumConstantBeingDef = false;
770     }

786     protected void methodDef(AST methodDef) {
787         MethodNode oldNode = methodNode;
788         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
789         AST node = methodDef.getFirstChild();
790 
791         GenericsType[] generics = null;
792         if (isType(TYPE_PARAMETERS, node)) {
793             generics = makeGenericsType(node);
794             node = node.getNextSibling();
795         }
796 
797         int modifiers = Opcodes.ACC_PUBLIC;
798         if (isType(MODIFIERS, node)) {
799             modifiers = modifiers(node, annotations, modifiers);
800             checkNoInvalidModifier(methodDef, "Method", modifiers, Opcodes.ACC_VOLATILE, "volatile");
801             node = node.getNextSibling();
802         }
803 
804         if (isAnInterface()) {
805             modifiers |= Opcodes.ACC_ABSTRACT;
806         }
807 
808         ClassNode returnType = null;
809         if (isType(TYPE, node)) {
810             returnType = makeTypeWithArguments(node);
811             node = node.getNextSibling();
812         }
813 
814         String name = identifier(node);
815         if (classNode != null && !classNode.isAnnotationDefinition()) {
816             if (classNode.getNameWithoutPackage().equals(name)) {
817                 if (isAnInterface()) {
818                     throw new ASTRuntimeException(methodDef, "Constructor not permitted within an interface.");
819                 }
820                 throw new ASTRuntimeException(methodDef, "Invalid constructor format. Remove '" + returnType.getName() +
821                         "' as the return type if you want a constructor, or use a different name if you want a method.");
822             }
823         }
824         node = node.getNextSibling();
825 
826         Parameter[] parameters = Parameter.EMPTY_ARRAY;
827         ClassNode[] exceptions = ClassNode.EMPTY_ARRAY;
828 
829         if (classNode == null || !classNode.isAnnotationDefinition()) {
830 
831             assertNodeType(PARAMETERS, node);
832             parameters = parameters(node);
833             if (parameters == null) parameters = Parameter.EMPTY_ARRAY;
834             node = node.getNextSibling();
835 
836             if (isType(LITERAL_throws, node)) {
837                 AST throwsNode = node.getFirstChild();
838                 List<ClassNode> exceptionList = new ArrayList<ClassNode>();
839                 throwsList(throwsNode, exceptionList);
840                 exceptions = exceptionList.toArray(exceptions);
841                 node = node.getNextSibling();
842             }
843         }
844 
845         boolean hasAnnotationDefault = false;
846         Statement code = null;
847         boolean syntheticPublic = ((modifiers & Opcodes.ACC_SYNTHETIC) != 0);
848         modifiers &= ~Opcodes.ACC_SYNTHETIC;
849         methodNode = new MethodNode(name, modifiers, returnType, parameters, exceptions, code);
850         if ((modifiers & Opcodes.ACC_ABSTRACT) == 0) {
851             if (node == null) {
852                 throw new ASTRuntimeException(methodDef, "You defined a method without body. Try adding a body, or declare it abstract.");
853             }
854             assertNodeType(SLIST, node);
855             code = statementList(node);
856         } else if (node != null && classNode.isAnnotationDefinition()) {
857             code = statement(node);
858             hasAnnotationDefault = true;
859         } else if ((modifiers & Opcodes.ACC_ABSTRACT) > 0) {
860             if (node != null) {
861                 throw new ASTRuntimeException(methodDef, "Abstract methods do not define a body.");
862             }
863         }
864         methodNode.setCode(code);
865         methodNode.addAnnotations(annotations);
866         methodNode.setGenericsTypes(generics);
867         methodNode.setAnnotationDefault(hasAnnotationDefault);
868         methodNode.setSyntheticPublic(syntheticPublic);
869         configureAST(methodNode, methodDef);
870 
871         if (classNode != null) {
872             classNode.addMethod(methodNode);
873         } else {
874             output.addMethod(methodNode);
875         }
876         methodNode = oldNode;
877     }

940     protected void fieldDef(AST fieldDef) {
941         List<AnnotationNode> annotations = new ArrayList<AnnotationNode>();
942         AST node = fieldDef.getFirstChild();
943 
944         int modifiers = 0;
945         if (isType(MODIFIERS, node)) {
946             modifiers = modifiers(node, annotations, modifiers);
947             node = node.getNextSibling();
948         }
949 
950         if (classNode.isInterface()) {
951             modifiers |= Opcodes.ACC_STATIC | Opcodes.ACC_FINAL;
952             if ((modifiers & (Opcodes.ACC_PRIVATE | Opcodes.ACC_PROTECTED)) == 0) {
953                 modifiers |= Opcodes.ACC_PUBLIC;
954             }
955         }
956 
957         ClassNode type = null;
958         if (isType(TYPE, node)) {
959             type = makeTypeWithArguments(node);
960             node = node.getNextSibling();
961         }
962 
963         String name = identifier(node);
964         node = node.getNextSibling();
965 
966         Expression initialValue = null;
967         if (node != null) {
968             assertNodeType(ASSIGN, node);
969             initialValue = expression(node.getFirstChild());
970         }
971 
972         if (classNode.isInterface() && initialValue == null && type != null) {
973             if (type == ClassHelper.int_TYPE) {
974                 initialValue = new ConstantExpression(0);
975             } else if (type == ClassHelper.long_TYPE) {
976                 initialValue = new ConstantExpression(0L);
977             } else if (type == ClassHelper.double_TYPE) {
978                 initialValue = new ConstantExpression(0.0);
979             } else if (type == ClassHelper.float_TYPE) {
980                 initialValue = new ConstantExpression(0.0F);
981             } else if (type == ClassHelper.boolean_TYPE) {
982                 initialValue = ConstantExpression.FALSE;
983             } else if (type == ClassHelper.short_TYPE) {
984                 initialValue = new ConstantExpression((short) 0);
985             } else if (type == ClassHelper.byte_TYPE) {
986                 initialValue = new ConstantExpression((byte) 0);
987             } else if (type == ClassHelper.char_TYPE) {
988                 initialValue = new ConstantExpression((char) 0);
989             }
990         }
991 
992 
993         FieldNode fieldNode = new FieldNode(name, modifiers, type, classNode, initialValue);
994         fieldNode.addAnnotations(annotations);
995         configureAST(fieldNode, fieldDef);
996 
997         if (!hasVisibility(modifiers)) {
998             // let's set the modifiers on the field
999             int fieldModifiers = 0;
1000             int flags = Opcodes.ACC_STATIC | Opcodes.ACC_TRANSIENT | Opcodes.ACC_VOLATILE | Opcodes.ACC_FINAL;
1001 
1002             if (!hasVisibility(modifiers)) {
1003                 modifiers |= Opcodes.ACC_PUBLIC;
1004                 fieldModifiers |= Opcodes.ACC_PRIVATE;
1005             }
1006 
1007             // let's pass along any other modifiers we need
1008             fieldModifiers |= (modifiers & flags);
1009             fieldNode.setModifiers(fieldModifiers);
1010             fieldNode.setSynthetic(true);
1011 
1012             // in the case that there is already a field, we would
1013             // like to use that field, instead of the default field
1014             // for the property
1015             FieldNode storedNode = classNode.getDeclaredField(fieldNode.getName());
1016             if (storedNode != null && !classNode.hasProperty(name)) {
1017                 fieldNode = storedNode;
1018                 // we remove it here, because addProperty will add it
1019                 // again and we want to avoid it showing up multiple
1020                 // times in the fields list.
1021                 classNode.getFields().remove(storedNode);
1022             }
1023 
1024             PropertyNode propertyNode = new PropertyNode(fieldNode, modifiers, null, null);
1025             configureAST(propertyNode, fieldDef);
1026             classNode.addProperty(propertyNode);
1027         } else {
1028             fieldNode.setModifiers(modifiers);
1029             // if there is a property of that name, then a field of that
1030             // name already exists, which means this new field here should
1031             // be used instead of the field the property originally has.
1032             PropertyNode pn = classNode.getProperty(name);
1033             if (pn != null && pn.getField().isSynthetic()) {
1034                 classNode.getFields().remove(pn.getField());
1035                 pn.setField(fieldNode);
1036             }
1037             classNode.addField(fieldNode);
1038         }
1039     }

1133     protected int modifiers(AST modifierNode, List<AnnotationNode> annotations, int defaultModifiers) {
1134         assertNodeType(MODIFIERS, modifierNode);
1135 
1136         boolean access = false;
1137         int answer = 0;
1138 
1139         for (AST node = modifierNode.getFirstChild(); node != null; node = node.getNextSibling()) {
1140             int type = node.getType();
1141             switch (type) {
1142                 case STATIC_IMPORT:
1143                     // ignore
1144                     break;
1145 
1146                 // annotations
1147                 case ANNOTATION:
1148                     annotations.add(annotation(node));
1149                     break;
1150 
1151                 // core access scope modifiers
1152                 case LITERAL_private:
1153                     answer = setModifierBit(node, answer, Opcodes.ACC_PRIVATE);
1154                     access = setAccessTrue(node, access);
1155                     break;
1156 
1157                 case LITERAL_protected:
1158                     answer = setModifierBit(node, answer, Opcodes.ACC_PROTECTED);
1159                     access = setAccessTrue(node, access);
1160                     break;
1161 
1162                 case LITERAL_public:
1163                     answer = setModifierBit(node, answer, Opcodes.ACC_PUBLIC);
1164                     access = setAccessTrue(node, access);
1165                     break;
1166 
1167                 // other modifiers
1168                 case ABSTRACT:
1169                     answer = setModifierBit(node, answer, Opcodes.ACC_ABSTRACT);
1170                     break;
1171 
1172                 case FINAL:
1173                     answer = setModifierBit(node, answer, Opcodes.ACC_FINAL);
1174                     break;
1175 
1176                 case LITERAL_native:
1177                     answer = setModifierBit(node, answer, Opcodes.ACC_NATIVE);
1178                     break;
1179 
1180                 case LITERAL_static:
1181                     answer = setModifierBit(node, answer, Opcodes.ACC_STATIC);
1182                     break;
1183 
1184                 case STRICTFP:
1185                     answer = setModifierBit(node, answer, Opcodes.ACC_STRICT);
1186                     break;
1187 
1188                 case LITERAL_synchronized:
1189                     answer = setModifierBit(node, answer, Opcodes.ACC_SYNCHRONIZED);
1190                     break;
1191 
1192                 case LITERAL_transient:
1193                     answer = setModifierBit(node, answer, Opcodes.ACC_TRANSIENT);
1194                     break;
1195 
1196                 case LITERAL_volatile:
1197                     answer = setModifierBit(node, answer, Opcodes.ACC_VOLATILE);
1198                     break;
1199 
1200                 default:
1201                     unknownAST(node);
1202             }
1203         }
1204         if (!access) {
1205             answer |= defaultModifiers;
1206             // ACC_SYNTHETIC isn't used here, use it as a special flag
1207             if (defaultModifiers == Opcodes.ACC_PUBLIC) answer |= Opcodes.ACC_SYNTHETIC;
1208         }
1209         return answer;
1210     }

1566     protected Statement caseStatements(AST node, List cases) {
1567         List<Expression> expressions = new LinkedList<Expression>();
1568         Statement statement = EmptyStatement.INSTANCE;
1569         Statement defaultStatement = EmptyStatement.INSTANCE;
1570         AST nextSibling = node;
1571         do {
1572             Expression expression = expression(nextSibling.getFirstChild());
1573             expressions.add(expression);
1574             nextSibling = nextSibling.getNextSibling();
1575         } while (isType(LITERAL_case, nextSibling));
1576         if (nextSibling != null) {
1577             if (isType(LITERAL_default, nextSibling)) {
1578                 defaultStatement = statement(nextSibling.getNextSibling());
1579                 statement = EmptyStatement.INSTANCE;
1580             } else {
1581                 statement = statement(nextSibling);
1582             }
1583         }

1585         while (iterator.hasNext()) {
1586             Expression expr = (Expression) iterator.next();
1587             Statement stmt;
1588             if (iterator.hasNext()) {
1589                 stmt = new CaseStatement(expr, EmptyStatement.INSTANCE);
1590             } else {
1591                 stmt = new CaseStatement(expr, statement);
1592             }
1593             configureAST(stmt, node);
1594             cases.add(stmt);
1595         }
1596         return defaultStatement;

1652         AST node = catchNode.getFirstChild();

1675                     configureAST(answer, catchNode);

3044             answer = answer.makeArray();
