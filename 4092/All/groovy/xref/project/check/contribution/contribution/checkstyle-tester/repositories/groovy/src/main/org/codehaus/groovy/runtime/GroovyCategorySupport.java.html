42      private static int categoriesInUse = 0;
43      private static AtomicInteger atomicCategoryUsageCounter = new AtomicInteger();

45      public static class CategoryMethodList extends ArrayList<CategoryMethod> {
46          public final int level;
47          final CategoryMethodList previous;
48          final AtomicInteger usage;
49  
50          public CategoryMethodList(String name, int level, CategoryMethodList previous) {
51              this.level = level;
52              this.previous = previous;
53              if (previous != null) {
54                  addAll(previous);
55                  usage = previous.usage;
56              }
57              else {
58                  usage = getCategoryNameUsage (name);
59              }
60          }
61  
62          public boolean add(CategoryMethod o) {
63              usage.incrementAndGet();
64              return super.add(o);
65          }
66      }

68      public static class ThreadCategoryInfo extends HashMap<String, CategoryMethodList>{
69          int level;
70  
71          private Map<String, String> propertyGetterMap;
72          private Map<String, String> propertySetterMap;
73  
74          private void newScope () {
75              atomicCategoryUsageCounter.incrementAndGet();
76              categoriesInUse = atomicCategoryUsageCounter.get();
77              DefaultMetaClassInfo.setCategoryUsed(true);
78              VMPluginFactory.getPlugin().invalidateCallSites();
79              level++;
80          }
81  
82          private void endScope () {
83              for (Iterator<Map.Entry<String, CategoryMethodList>> it = entrySet().iterator(); it.hasNext(); ) {
84                  final Map.Entry<String, CategoryMethodList> e = it.next();
85                  final CategoryMethodList list = e.getValue();
86                  if (list.level == level) {
87                      final CategoryMethodList prev = list.previous;
88                      if (prev == null) {
89                        it.remove();
90                        list.usage.addAndGet(-list.size());
91                      }
92                      else {
93                        e.setValue(prev);
94                        list.usage.addAndGet(prev.size()-list.size());
95                      }
96                  }
97              }
98              level--;
99              atomicCategoryUsageCounter.getAndDecrement();
100             categoriesInUse = atomicCategoryUsageCounter.get();
101             VMPluginFactory.getPlugin().invalidateCallSites();
102             if (categoriesInUse==0) DefaultMetaClassInfo.setCategoryUsed(false);
103             if (level == 0) {
104                 THREAD_INFO.remove();
105             }
106         }
107 
108         private <T> T use(Class categoryClass, Closure<T> closure) {
109             newScope();
110             try {
111                 use(categoryClass);
112                 return closure.call();
113             } finally {
114                 endScope();
115             }
116         }
117 
118         public <T> T use(List<Class> categoryClasses, Closure<T> closure) {
119             newScope();
120             try {
121                 for (Class categoryClass : categoryClasses) {
122                     use(categoryClass);
123                 }
124                 return closure.call();
125             } finally {
126                 endScope();
127             }
128         }
129 
130         private void applyUse(CachedClass cachedClass) {
131             CachedMethod[] methods = cachedClass.getMethods();
132             for (CachedMethod cachedMethod : methods) {
133                 if (cachedMethod.isStatic() && cachedMethod.isPublic()) {
134                     CachedClass[] paramTypes = cachedMethod.getParameterTypes();
135                     if (paramTypes.length > 0) {
136                         CachedClass metaClass = paramTypes[0];
137                         CategoryMethod mmethod = new CategoryMethod(cachedMethod, metaClass.getTheClass());
138                         final String name = cachedMethod.getName();
139                         CategoryMethodList list = get(name);
140                         if (list == null || list.level != level) {
141                             list = new CategoryMethodList(name, level, list);
142                             put(name, list);
143                         }
144                         list.add(mmethod);
145                         Collections.sort(list);
146                         cachePropertyAccessor(mmethod);
147                     }
148                 }
149             }
150         }
151 
152         private void cachePropertyAccessor(CategoryMethod method) {
153              String name = method.getName();
154              int parameterLength = method.getParameterTypes().length;
155 
156              if (name.startsWith("get") && name.length() > 3 && parameterLength == 0) {
157                  propertyGetterMap = putPropertyAccessor(3, name, propertyGetterMap);
158              }
159              else if (name.startsWith("set") && name.length() > 3 && parameterLength == 1) {
160                  propertySetterMap = putPropertyAccessor(3, name, propertySetterMap);
161              }
162         }
163 
164         // Precondition: accessorName.length() > prefixLength
165         private Map<String, String> putPropertyAccessor(int prefixLength, String accessorName, Map<String, String> map) {
166             if (map == null) {
167               map = new HashMap<String, String>();
168             }
169             String property = accessorName.substring(prefixLength, prefixLength+1).toLowerCase() + accessorName.substring(prefixLength+1);
170             map.put(property, accessorName);
171             return map;
172         }
173 
174         private void use(Class categoryClass) {
175             CachedClass cachedClass = ReflectionCache.getCachedClass(categoryClass);
176             LinkedList<CachedClass> classStack = new LinkedList<CachedClass>();
177             for (CachedClass superClass = cachedClass; superClass.getTheClass()!=Object.class; superClass = superClass.getCachedSuperClass()) {
178                 classStack.add(superClass);
179             }
180 
181             while (!classStack.isEmpty()) {
182                 CachedClass klazz = classStack.removeLast();
183                 applyUse(klazz);
184             }
185         }

187         public CategoryMethodList getCategoryMethods(String name) {
188             return level == 0 ? null : get(name);
189         }

203     public static class CategoryMethod extends NewInstanceMetaMethod implements Comparable {
204         private final Class metaClass;
205 
206         public CategoryMethod(CachedMethod metaMethod, Class metaClass) {
207             super(metaMethod);
208             this.metaClass = metaClass;
209         }
210 
211         public boolean isCacheable() { return false; }
212 
213         /**
214          * Sort by most specific to least specific.
215          *
216          * @param o the object to compare against
217          */
218         public int compareTo(Object o) {
219             CategoryMethod thatMethod = (CategoryMethod) o;
220             Class thisClass = metaClass;
221             Class thatClass = thatMethod.metaClass;
222             if (thisClass == thatClass) return 0;
223             if (isChildOfParent(thisClass, thatClass)) return -1;
224             if (isChildOfParent(thatClass, thisClass)) return 1;
225             return 0;
226         }
227 
228         private boolean isChildOfParent(Class candidateChild, Class candidateParent) {
229             Class loop = candidateChild;
230             while(loop != null && loop != Object.class) {
231                 loop = loop.getSuperclass();
232                 if (loop == candidateParent) {
233                     return true;
234                 }
235             }
236             return false;
237         }
238     }

297     private static class MyThreadLocal extends ThreadLocal<SoftReference> {
298 
299         ConcurrentHashMap<String,AtomicInteger> usage = new ConcurrentHashMap<String,AtomicInteger> ();
300 
301         public ThreadCategoryInfo getInfo() {
302             final SoftReference reference = get();
303             ThreadCategoryInfo tcinfo;
304             if (reference != null) {
305                 tcinfo = (ThreadCategoryInfo) reference.get();
306                 if( tcinfo == null ) {
307                     tcinfo = new ThreadCategoryInfo();
308                     set(new SoftReference(tcinfo));
309                 }
310             }
311             else {
312                 tcinfo = new ThreadCategoryInfo();
313                 set(new SoftReference(tcinfo));
314             }
315             return tcinfo;
316         }
317 
318         public ThreadCategoryInfo getInfoNullable() {
319             final SoftReference reference = get();
320             return reference == null ? null : (ThreadCategoryInfo) reference.get();
321         }
322 
323         public AtomicInteger getUsage (String name) {
324             AtomicInteger u = usage.get(name);
325             if (u != null) {
326                 return u;
327             }
328 
329             final AtomicInteger ai = new AtomicInteger();
330             final AtomicInteger prev = usage.putIfAbsent(name, ai);
331             return prev == null ? ai : prev;
332         }
