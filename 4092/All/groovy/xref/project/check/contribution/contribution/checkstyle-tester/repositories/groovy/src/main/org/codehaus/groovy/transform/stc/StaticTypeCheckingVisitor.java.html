233     @Override
234     public void visitClass(final ClassNode node) {
235         if (shouldSkipClassNode(node)) return;
236         if (extension.beforeVisitClass(node)) {
237             extension.afterVisitClass(node);
238             return;
239         }
240         Object type = node.getNodeMetaData(StaticTypesMarker.INFERRED_TYPE);
241         if (type != null) {
242             // transformation has already been run on this class node
243             // so we'll use a silent collector in order not to duplicate errors
244             typeCheckingContext.pushErrorCollector();
245         }
246         typeCheckingContext.pushEnclosingClassNode(node);
247         Set<MethodNode> oldVisitedMethod = typeCheckingContext.alreadyVisitedMethods;
248         typeCheckingContext.alreadyVisitedMethods = new LinkedHashSet<MethodNode>();
249         super.visitClass(node);
250         Iterator<InnerClassNode> innerClasses = node.getInnerClasses();
251         while (innerClasses.hasNext()) {
252             InnerClassNode innerClassNode = innerClasses.next();
253             visitClass(innerClassNode);
254         }
255         typeCheckingContext.alreadyVisitedMethods = oldVisitedMethod;
256         node.putNodeMetaData(StaticTypesMarker.INFERRED_TYPE, node);
257         // mark all methods as visited. We can't do this in visitMethod because the type checker
258         // works in a two pass sequence and we don't want to skip the second pass
259         for (MethodNode methodNode : node.getMethods()) {
260             methodNode.putNodeMetaData(StaticTypeCheckingVisitor.class, Boolean.TRUE);
261         }
262         for (ConstructorNode constructorNode : node.getDeclaredConstructors()) {
263             constructorNode.putNodeMetaData(StaticTypeCheckingVisitor.class, Boolean.TRUE);
264         }
265         extension.afterVisitClass(node);
266     }

286     public boolean isSkipMode(final AnnotatedNode node) {
287         if (node == null) return false;
288         for (ClassNode tca : getTypeCheckingAnnotations()) {
289             List<AnnotationNode> annotations = node.getAnnotations(tca);
290             if (annotations != null) {
291                 for (AnnotationNode annotation : annotations) {
292                     Expression value = annotation.getMember("value");
293                     if (value != null) {
294                         if (value instanceof ConstantExpression) {
295                             ConstantExpression ce = (ConstantExpression) value;
296                             if (TypeCheckingMode.SKIP.toString().equals(ce.getValue().toString())) return true;
297                         } else if (value instanceof PropertyExpression) {
298                             PropertyExpression pe = (PropertyExpression) value;
299                             if (TypeCheckingMode.SKIP.toString().equals(pe.getPropertyAsString())) return true;
300                         }
301                     }
302                 }
303             }
304         }
305         if (node instanceof MethodNode) {
306             return isSkipMode(node.getDeclaringClass());
307         }
308         if (isSkippedInnerClass(node)) return true;
309         return false;
310     }

314      * @param node
315      * @return true if the inner class node should be skipped

317     protected boolean isSkippedInnerClass(AnnotatedNode node) {
318         if (!(node instanceof InnerClassNode)) return false;
319         MethodNode enclosingMethod = ((InnerClassNode) node).getEnclosingMethod();
320         return enclosingMethod != null && isSkipMode(enclosingMethod);

343     /**
344      * Given a field node, checks if we are calling a private field from an inner class.
345      */
346     private void checkOrMarkPrivateAccess(Expression source, FieldNode fn) {
347         if (fn!=null && Modifier.isPrivate(fn.getModifiers()) &&
348             (fn.getDeclaringClass() != typeCheckingContext.getEnclosingClassNode() || typeCheckingContext.getEnclosingClosure()!=null) &&
349             fn.getDeclaringClass().getModule() == typeCheckingContext.getEnclosingClassNode().getModule()) {
350             addPrivateFieldOrMethodAccess(source, fn.getDeclaringClass(), StaticTypesMarker.PV_FIELDS_ACCESS, fn);
351         }
352     }

354     /**
355      * Given a method node, checks if we are calling a private method from an inner class.
356      */
357     private void checkOrMarkPrivateAccess(Expression source, MethodNode mn) {
358         if (mn==null) {
359             return;
360         }
361         ClassNode declaringClass = mn.getDeclaringClass();
362         ClassNode enclosingClassNode = typeCheckingContext.getEnclosingClassNode();
363         if (declaringClass != enclosingClassNode || typeCheckingContext.getEnclosingClosure() != null) {
364             int mods = mn.getModifiers();
365             boolean sameModule = declaringClass.getModule() == enclosingClassNode.getModule();
366             String packageName = declaringClass.getPackageName();
367             if (packageName==null) {
368                 packageName = "";
369             }
370             if ((Modifier.isPrivate(mods) && sameModule)
371                     || (Modifier.isProtected(mods) && !packageName.equals(enclosingClassNode.getPackageName()))) {
372                 addPrivateFieldOrMethodAccess(source, sameModule? declaringClass : enclosingClassNode, StaticTypesMarker.PV_METHODS_ACCESS, mn);
373             }
374         }
375     }

377     private void checkSuperCallFromClosure(Expression call, MethodNode directCallTarget) {
378         if (call instanceof MethodCallExpression && typeCheckingContext.getEnclosingClosure() != null) {
379             Expression objectExpression = ((MethodCallExpression)call).getObjectExpression();
380             if (objectExpression instanceof VariableExpression) {
381                 VariableExpression var = (VariableExpression) objectExpression;
382                 if (var.isSuperExpression()) {
383                     ClassNode current = typeCheckingContext.getEnclosingClassNode();
384                     LinkedList<MethodNode> list = current.getNodeMetaData(StaticTypesMarker.SUPER_MOP_METHOD_REQUIRED);
385                     if (list == null) {
386                         list = new LinkedList<MethodNode>();
387                         current.putNodeMetaData(StaticTypesMarker.SUPER_MOP_METHOD_REQUIRED, list);
388                     }
389                     list.add(directCallTarget);
390                     call.putNodeMetaData(StaticTypesMarker.SUPER_MOP_METHOD_REQUIRED, current);
391                 }
392             }
393         }

409     private boolean storeTypeForThis(VariableExpression vexp) {
410         if (vexp == VariableExpression.THIS_EXPRESSION) return true;
411         if (!vexp.isThisExpression()) return false;
412         ClassNode enclosingClassNode = typeCheckingContext.getEnclosingClassNode();
413         storeType(vexp, makeType(enclosingClassNode, typeCheckingContext.isInStaticContext));
414         return true;
415     }

417     private boolean storeTypeForSuper(VariableExpression vexp) {
418         if (vexp == VariableExpression.SUPER_EXPRESSION) return true;
419         if (!vexp.isSuperExpression()) return false;
420         ClassNode superClassNode = typeCheckingContext.getEnclosingClassNode().getSuperClass();
421         storeType(vexp, makeType(superClassNode, typeCheckingContext.isInStaticContext));
422         return true;
423     }

425     @Override
426     public void visitVariableExpression(VariableExpression vexp) {
427         super.visitVariableExpression(vexp);
428 
429         if (storeTypeForThis(vexp)) return;
430         if (storeTypeForSuper(vexp)) return;
431         if (vexp.getAccessedVariable() instanceof PropertyNode) {
432             // we must be careful, because the property node may be of a wrong type:
433             // if a class contains a getter and a setter of different types or
434             // overloaded setters, the type of the property node is arbitrary!
435             if (tryVariableExpressionAsProperty(vexp, vexp.getName())) {
436                 BinaryExpression enclosingBinaryExpression = typeCheckingContext.getEnclosingBinaryExpression();
437                 if (enclosingBinaryExpression != null) {
438                     Expression leftExpression = enclosingBinaryExpression.getLeftExpression();
439                     Expression rightExpression = enclosingBinaryExpression.getRightExpression();
440                     SetterInfo setterInfo = removeSetterInfo(leftExpression);
441                     if (setterInfo != null) {
442                         if (!ensureValidSetter(vexp, leftExpression, rightExpression, setterInfo)) {
443                             return;
444                         }
445 
446                     }
447                 }
448             }
449         }
450 
451         TypeCheckingContext.EnclosingClosure enclosingClosure = typeCheckingContext.getEnclosingClosure();
452         if (enclosingClosure != null) {
453             String name = vexp.getName();
454             if (name.equals("owner") || name.equals("thisObject")) {
455                 storeType(vexp, typeCheckingContext.getEnclosingClassNode());
456                 return;
457             } else if ("delegate".equals(name)) {
458                 DelegationMetadata md = getDelegationMetadata(enclosingClosure.getClosureExpression());
459                 ClassNode type = typeCheckingContext.getEnclosingClassNode();
460                 if (md!=null) type = md.getType();
461                 storeType(vexp, type);
462                 return;
463             }
464         }
465 
466         if (! (vexp.getAccessedVariable() instanceof DynamicVariable)) return;
467 
468         // a dynamic variable is either an undeclared variable
469         // or a member of a class used in a 'with'
470         DynamicVariable dyn = (DynamicVariable) vexp.getAccessedVariable();
471         // first, we must check the 'with' context
472         String dynName = dyn.getName();
473         if (tryVariableExpressionAsProperty(vexp, dynName)) return;
474 
475         if (!extension.handleUnresolvedVariableExpression(vexp)) {
476             addStaticTypeError("The variable [" + vexp.getName() + "] is undeclared.", vexp);
477         }
478     }

480     private boolean tryVariableExpressionAsProperty(final VariableExpression vexp, final String dynName) {
481         VariableExpression implicitThis = new VariableExpression("this");
482         PropertyExpression pe = new PropertyExpression(implicitThis, dynName);
483         pe.setImplicitThis(true);
484         if (visitPropertyExpressionSilent(pe, vexp)) {
485             ClassNode previousIt = vexp.getNodeMetaData(StaticTypesMarker.INFERRED_TYPE);
486             vexp.copyNodeMetaData(implicitThis);
487             vexp.putNodeMetaData(StaticTypesMarker.INFERRED_TYPE, previousIt);
488             storeType(vexp, getType(pe));
489             Object val = pe.getNodeMetaData(StaticTypesMarker.READONLY_PROPERTY);
490             if (val!=null) vexp.putNodeMetaData(StaticTypesMarker.READONLY_PROPERTY,val);
491             val = pe.getNodeMetaData(StaticTypesMarker.IMPLICIT_RECEIVER);
492             if (val!=null) vexp.putNodeMetaData(StaticTypesMarker.IMPLICIT_RECEIVER,val);
493             return true;
494         }
495         return false;
496     }

529     public void visitRangeExpression(final RangeExpression expression) {
530         super.visitRangeExpression(expression);
531         ClassNode fromType = getWrapper(getType(expression.getFrom()));
532         ClassNode toType = getWrapper(getType(expression.getTo()));
533         if (Integer_TYPE.equals(fromType) && Integer_TYPE.equals(toType)) {
534             storeType(expression, ClassHelper.make(IntRange.class));
535         } else {
536             ClassNode rangeType = ClassHelper.make(Range.class).getPlainNodeReference();
537             rangeType.setGenericsTypes(new GenericsType[] { new GenericsType(WideningCategories.lowestUpperBound(fromType, toType))});
538             storeType(expression,  rangeType);
539         }

542     @Override
543     public void visitBinaryExpression(BinaryExpression expression) {
544         BinaryExpression enclosingBinaryExpression = typeCheckingContext.getEnclosingBinaryExpression();
545         typeCheckingContext.pushEnclosingBinaryExpression(expression);
546         try {
547             final Expression leftExpression = expression.getLeftExpression();
548             final Expression rightExpression = expression.getRightExpression();
549             int op = expression.getOperation().getType();
550             leftExpression.visit(this);
551             SetterInfo setterInfo = removeSetterInfo(leftExpression);
552             if (setterInfo != null) {
553                 if (ensureValidSetter(expression, leftExpression, rightExpression, setterInfo)) {
554                     return;
555                 }
556 
557             } else {
558                 rightExpression.visit(this);
559             }
560             ClassNode lType = getType(leftExpression);
561             ClassNode rType = getType(rightExpression);
562             if (isNullConstant(rightExpression)) {
563                 if (!isPrimitiveType(lType))
564                     rType = UNKNOWN_PARAMETER_TYPE; // primitive types should be ignored as they will result in another failure
565             }
566             BinaryExpression reversedBinaryExpression = new BinaryExpression(rightExpression, expression.getOperation(), leftExpression);
567             ClassNode resultType = op==KEYWORD_IN
568                     ?getResultType(rType,op,lType,reversedBinaryExpression)
569                     :getResultType(lType, op, rType, expression);
570             if (op==KEYWORD_IN) {
571                 // in case of the "in" operator, the receiver and the arguments are reversed
572                 // so we use the reversedExpression and get the target method from it
573                 storeTargetMethod(expression, (MethodNode) reversedBinaryExpression.getNodeMetaData(StaticTypesMarker.DIRECT_METHOD_CALL_TARGET));
574             } else if (op == LEFT_SQUARE_BRACKET
575                     && leftExpression instanceof VariableExpression
576                     && leftExpression.getNodeMetaData(StaticTypesMarker.INFERRED_TYPE)==null) {
577                 storeType(leftExpression, lType);
578             }
579             if (resultType == null) {
580                 resultType = lType;
581             }
582 
583             // if left expression is a closure shared variable, a second pass should be done
584             if (leftExpression instanceof VariableExpression) {
585                 VariableExpression leftVar = (VariableExpression) leftExpression;
586                 if (leftVar.isClosureSharedVariable()) {
587                     // if left expression is a closure shared variable, we should check it twice
588                     // see GROOVY-5874
589                     typeCheckingContext.secondPassExpressions.add(new SecondPassExpression<Void>(expression));
590                 }
591             }
592 
593             if (lType.isUsingGenerics() && missesGenericsTypes(resultType) && isAssignment(op)) {
594                 // unchecked assignment
595                 // examples:
596                 // List<A> list = new LinkedList()
597                 // List<A> list = []
598                 // Iterable<A> list = new LinkedList()
599 
600                 // in that case, the inferred type of the binary expression is the type of the RHS
601                 // "completed" with generics type information available in the LHS
602                 ClassNode completedType = GenericsUtils.parameterizeType(lType, resultType.getPlainNodeReference());
603 
604                 resultType = completedType;
605 
606             }
607             if (isArrayOp(op) &&
608                     enclosingBinaryExpression != null
609                     && enclosingBinaryExpression.getLeftExpression() == expression
610                     && isAssignment(enclosingBinaryExpression.getOperation().getType())
611                     && !lType.isArray()) {
612                 // left hand side of an assignment : map['foo'] = ...
613                 Expression enclosingBE_rightExpr = enclosingBinaryExpression.getRightExpression();
614                 if (!(enclosingBE_rightExpr instanceof ClosureExpression)) {
615                     enclosingBE_rightExpr.visit(this);
616                 }
617                 ClassNode[] arguments = {rType, getType(enclosingBE_rightExpr)};
618                 List<MethodNode> nodes = findMethod(lType.redirect(), "putAt", arguments);
619                 if (nodes.size() == 1) {
620                     typeCheckMethodsWithGenericsOrFail(lType, arguments, nodes.get(0), enclosingBE_rightExpr);
621                 } else if (nodes.isEmpty()) {
622                     addNoMatchingMethodError(lType, "putAt", arguments, enclosingBinaryExpression);
623                 }
624             }
625             boolean isEmptyDeclaration = expression instanceof DeclarationExpression && rightExpression instanceof EmptyExpression;
626             if (!isEmptyDeclaration && isAssignment(op)) {
627                 if (rightExpression instanceof ConstructorCallExpression) {
628                     inferDiamondType((ConstructorCallExpression) rightExpression, lType);
629                 }
630 
631                 ClassNode originType = getOriginalDeclarationType(leftExpression);
632                 typeCheckAssignment(expression, leftExpression, originType, rightExpression, resultType);
633                 // if assignment succeeds but result type is not a subtype of original type, then we are in a special cast handling
634                 // and we must update the result type
635                 if (!implementsInterfaceOrIsSubclassOf(getWrapper(resultType), getWrapper(originType))) {
636                     resultType = originType;
637                 } else if (lType.isUsingGenerics() && !lType.isEnum() && hasRHSIncompleteGenericTypeInfo(resultType)) {
638                     // for example, LHS is List<ConcreteClass> and RHS is List<T> where T is a placeholder
639                     resultType = lType;
640                 }
641 
642                 // make sure we keep primitive types
643                 if (isPrimitiveType(originType) && resultType.equals(getWrapper(originType))) {
644                     resultType = originType;
645                 }
646 
647                 // if we are in an if/else branch, keep track of assignment
648                 if (typeCheckingContext.ifElseForWhileAssignmentTracker != null && leftExpression instanceof VariableExpression
649                         && !isNullConstant(rightExpression)) {
650                     Variable accessedVariable = ((VariableExpression) leftExpression).getAccessedVariable();
651                     if (accessedVariable instanceof VariableExpression) {
652                         VariableExpression var = (VariableExpression) accessedVariable;
653                         List<ClassNode> types = typeCheckingContext.ifElseForWhileAssignmentTracker.get(var);
654                         if (types == null) {
655                             types = new LinkedList<ClassNode>();
656                             ClassNode type = var.getNodeMetaData(StaticTypesMarker.INFERRED_TYPE);
657                             types.add(type);
658                             typeCheckingContext.ifElseForWhileAssignmentTracker.put(var, types);
659                         }
660                         types.add(resultType);
661                     }
662                 }
663                 storeType(leftExpression, resultType);
664 
665                 // if right expression is a ClosureExpression, store parameter type information
666                 if (leftExpression instanceof VariableExpression) {
667                     if (rightExpression instanceof ClosureExpression) {
668                         Parameter[] parameters = ((ClosureExpression) rightExpression).getParameters();
669                         leftExpression.putNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS, parameters);
670                     } else if (rightExpression instanceof VariableExpression &&
671                             ((VariableExpression)rightExpression).getAccessedVariable() instanceof Expression &&
672                             ((Expression)((VariableExpression)rightExpression).getAccessedVariable()).getNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS)!=null) {
673                         Variable targetVariable = findTargetVariable((VariableExpression)leftExpression);
674                         if (targetVariable instanceof ASTNode) {
675                             ((ASTNode)targetVariable).putNodeMetaData(
676                                 StaticTypesMarker.CLOSURE_ARGUMENTS,
677                                 ((Expression)((VariableExpression)rightExpression).getAccessedVariable()).getNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS));
678                         }
679                     }
680                 }
681 
682 
683             } else if (op == KEYWORD_INSTANCEOF) {
684                 pushInstanceOfTypeInfo(leftExpression, rightExpression);
685             }
686             if (!isEmptyDeclaration) {
687                 storeType(expression, resultType);
688             }
689         } finally {
690             typeCheckingContext.popEnclosingBinaryExpression();
691         }
692     }

694     /**
695      * Given a binary expression corresponding to an assignment, will check that the type of the RHS matches one
696      * of the possible setters and if not, throw a type checking error.
697      * @param expression the assignment expression
698      * @param leftExpression left expression of the assignment
699      * @param rightExpression right expression of the assignment
700      * @param setterInfo possible setters
701      * @return true if type checking passed
702      */
703     private boolean ensureValidSetter(final Expression expression, final Expression leftExpression, final Expression rightExpression, final SetterInfo setterInfo) {
704         // for expressions like foo = { ... }
705         // we know that the RHS type is a closure
706         // but we must check if the binary expression is an assignment
707         // because we need to check if a setter uses @DelegatesTo
708         VariableExpression ve = new VariableExpression("%", setterInfo.receiverType);
709         MethodCallExpression call = new MethodCallExpression(
710                 ve,
711                 setterInfo.name,
712                 rightExpression
713         );
714         call.setImplicitThis(false);
715         visitMethodCallExpression(call);
716         MethodNode directSetterCandidate = call.getNodeMetaData(StaticTypesMarker.DIRECT_METHOD_CALL_TARGET);
717         if (directSetterCandidate==null) {
718             // this may happen if there's a setter of type boolean/String/Class, and that we are using the property
719             // notation AND that the RHS is not a boolean/String/Class
720             for (MethodNode setter : setterInfo.setters) {
721                 ClassNode type = getWrapper(setter.getParameters()[0].getOriginType());
722                 if (Boolean_TYPE.equals(type) || STRING_TYPE.equals(type) || CLASS_Type.equals(type)) {
723                     call = new MethodCallExpression(
724                             ve,
725                             setterInfo.name,
726                             new CastExpression(type,rightExpression)
727                     );
728                     call.setImplicitThis(false);
729                     visitMethodCallExpression(call);
730                     directSetterCandidate = call.getNodeMetaData(StaticTypesMarker.DIRECT_METHOD_CALL_TARGET);
731                     if (directSetterCandidate!=null) {
732                         break;
733                     }
734                 }
735             }
736         }
737         if (directSetterCandidate != null) {
738             for (MethodNode setter : setterInfo.setters) {
739                 if (setter == directSetterCandidate) {
740                     leftExpression.putNodeMetaData(StaticTypesMarker.DIRECT_METHOD_CALL_TARGET, directSetterCandidate);
741                     storeType(leftExpression, getType(rightExpression));
742                     break;
743                 }
744             }
745         } else {
746             ClassNode firstSetterType = setterInfo.setters.iterator().next().getParameters()[0].getOriginType();
747             addAssignmentError(firstSetterType, getType(rightExpression), expression);
748             return true;
749         }
750         return false;
751     }

753     protected ClassNode getOriginalDeclarationType(Expression lhs) {
754         if (lhs instanceof VariableExpression) {
755             Variable var = findTargetVariable((VariableExpression) lhs);
756             if (var instanceof PropertyNode) {
757                 // Do NOT trust the type of the property node!
758                 return getType(lhs);
759             }
760             if (var instanceof DynamicVariable) return getType(lhs);
761             return var.getOriginType();
762         }
763         if (lhs instanceof FieldExpression) {
764             return ((FieldExpression) lhs).getField().getOriginType();
765         }
766         return getType(lhs);
767     }

769     protected void inferDiamondType(final ConstructorCallExpression cce, final ClassNode lType) {
770         // check if constructor call expression makes use of the diamond operator
771         ClassNode node = cce.getType();
772         if (node.isUsingGenerics() && node instanceof InnerClassNode && ((InnerClassNode) node).isAnonymous()) {
773             ClassNode[] interfaces = node.getInterfaces();
774             node = interfaces != null && interfaces.length == 1 ? interfaces[0] : node.getUnresolvedSuperClass(false);
775             if ((node.getGenericsTypes() == null || node.getGenericsTypes().length == 0) && lType.isUsingGenerics()) {
776                 // InterfaceA<Foo> obj = new InterfaceA<>() { ... }
777                 // InterfaceA<Foo> obj = new ClassA<>() { ... }
778                 // ClassA<Foo> obj = new ClassA<>() { ... }
779                 addStaticTypeError("Cannot use diamond <> with anonymous inner classes", cce);
780             }
781         } else if (node.isUsingGenerics() && node.getGenericsTypes() != null && node.getGenericsTypes().length == 0) {
782             ArgumentListExpression argumentListExpression = InvocationWriter.makeArgumentList(cce.getArguments());
783             if (argumentListExpression.getExpressions().isEmpty()) {
784                 GenericsType[] genericsTypes = lType.getGenericsTypes();
785                 if (lType.getGenericsTypes()==null) {
786                     // case of def foo = new HashMap<>()
787                     genericsTypes = node.redirect().getGenericsTypes();
788                 }
789                 GenericsType[] copy = new GenericsType[genericsTypes.length];
790                 for (int i = 0; i < genericsTypes.length; i++) {
791                     GenericsType genericsType = genericsTypes[i];
792                     copy[i] = new GenericsType(
793                             wrapTypeIfNecessary(genericsType.getType()),
794                             genericsType.getUpperBounds(),
795                             genericsType.getLowerBound()
796                     );
797                 }
798                 node.setGenericsTypes(copy);
799             } else {
800                 ClassNode type = getType(argumentListExpression.getExpression(0));
801                 if (type.isUsingGenerics()) {
802                     GenericsType[] genericsTypes = type.getGenericsTypes();
803                     GenericsType[] copy = new GenericsType[genericsTypes.length];
804                     for (int i = 0; i < genericsTypes.length; i++) {
805                         GenericsType genericsType = genericsTypes[i];
806                         copy[i] = new GenericsType(
807                                 wrapTypeIfNecessary(genericsType.getType()),
808                                 genericsType.getUpperBounds(),
809                                 genericsType.getLowerBound()
810                         );
811                     }
812                     node.setGenericsTypes(copy);
813                 }
814             }
815             // store inferred type on CCE
816             storeType(cce, node);
817         }
818     }

837     private boolean typeCheckMultipleAssignmentAndContinue(Expression leftExpression, Expression rightExpression) {
838         // multiple assignment check
839         if (!(leftExpression instanceof TupleExpression)) return true;
840 
841         if (!(rightExpression instanceof ListExpression)) {
842             addStaticTypeError("Multiple assignments without list expressions on the right hand side are unsupported in static type checking mode", rightExpression);
843             return false;
844         }
845 
846         TupleExpression tuple = (TupleExpression) leftExpression;
847         ListExpression list = (ListExpression) rightExpression;
848         List<Expression> listExpressions = list.getExpressions();
849         List<Expression> tupleExpressions = tuple.getExpressions();
850         if (listExpressions.size() < tupleExpressions.size()) {
851             addStaticTypeError("Incorrect number of values. Expected:" + tupleExpressions.size() + " Was:" + listExpressions.size(), list);
852             return false;
853         }
854         for (int i = 0, tupleExpressionsSize = tupleExpressions.size(); i < tupleExpressionsSize; i++) {
855             Expression tupleExpression = tupleExpressions.get(i);
856             Expression listExpression = listExpressions.get(i);
857             ClassNode elemType = getType(listExpression);
858             ClassNode tupleType = getType(tupleExpression);
859             if (!isAssignableTo(elemType, tupleType)) {
860                 addStaticTypeError("Cannot assign value of type " + elemType.toString(false) + " to variable of type " + tupleType.toString(false), rightExpression);
861                 return false; // avoids too many errors
862             }
863         }
864 
865         return true;
866     }

894     private void addPrecisionErrors(ClassNode leftRedirect, ClassNode lhsType, ClassNode inferredrhsType, Expression rightExpression) {
895         if (isNumberType(leftRedirect) && isNumberType(inferredrhsType)) {
896             if (checkPossibleLooseOfPrecision(leftRedirect, inferredrhsType, rightExpression)) {
897                 addStaticTypeError("Possible loss of precision from " + inferredrhsType + " to " + leftRedirect, rightExpression);
898                 return;
899             }
900         }
901         // if left type is array, we should check the right component types
902         if (!lhsType.isArray()) return;
903         ClassNode leftComponentType = lhsType.getComponentType();
904         ClassNode rightRedirect = rightExpression.getType().redirect();
905         if (rightRedirect.isArray()) {
906             ClassNode rightComponentType = rightRedirect.getComponentType();
907             if (!checkCompatibleAssignmentTypes(leftComponentType, rightComponentType)) {
908                 addStaticTypeError("Cannot assign value of type " + rightComponentType.toString(false) + " into array of type " + lhsType.toString(false), rightExpression);
909             }
910         } else if (rightExpression instanceof ListExpression) {
911             for (Expression element : ((ListExpression) rightExpression).getExpressions()) {
912                 ClassNode rightComponentType = element.getType().redirect();
913                 if (!checkCompatibleAssignmentTypes(leftComponentType, rightComponentType)
914                         && !(isNullConstant(element) && !isPrimitiveType(leftComponentType))) {
915                     addStaticTypeError("Cannot assign value of type " + rightComponentType.toString(false) + " into array of type " + lhsType.toString(false), rightExpression);
916                 }
917             }
918         }
919     }

938                 && !isWildcardLeftHandSide(leftExpressionType)) {

987     protected void typeCheckAssignment(
988             final BinaryExpression assignmentExpression,
989             final Expression leftExpression,
990             final ClassNode leftExpressionType,
991             final Expression rightExpression,
992             final ClassNode inferredRightExpressionType)
993     {
994 
995         if (!typeCheckMultipleAssignmentAndContinue(leftExpression, rightExpression)) return;
996 
997         //TODO: need errors for write-only too!
998         if (addedReadOnlyPropertyError(leftExpression)) return;
999 
1000         ClassNode leftRedirect = leftExpressionType.redirect();
1001         ClassNode wrappedRHS = adjustTypeForSpreading(inferredRightExpressionType, leftExpression);
1002 
1003         // check types are compatible for assignment
1004         boolean compatible = checkCompatibleAssignmentTypes(leftRedirect, wrappedRHS, rightExpression);
1005 
1006 
1007         if (!compatible) {
1008             if (!extension.handleIncompatibleAssignment(leftExpressionType, inferredRightExpressionType, assignmentExpression)) {
1009                 addAssignmentError(leftExpressionType, inferredRightExpressionType, assignmentExpression.getRightExpression());
1010             }
1011         } else {
1012             addPrecisionErrors(leftRedirect, leftExpressionType, inferredRightExpressionType, rightExpression);
1013             addListAssignmentConstructorErrors(leftRedirect, leftExpressionType, inferredRightExpressionType, rightExpression, assignmentExpression);
1014             addMapAssignmentConstructorErrors(leftRedirect, leftExpression, rightExpression);
1015             if (hasGStringStringError(leftExpressionType, wrappedRHS, rightExpression)) return;
1016             checkTypeGenerics(leftExpressionType, wrappedRHS, rightExpression);
1017         }
1018     }

1020     protected void checkGroovyConstructorMap(final Expression receiver, final ClassNode receiverType, final MapExpression mapExpression) {
1021         // workaround for map-style checks putting setter info on wrong AST nodes
1022         typeCheckingContext.pushEnclosingBinaryExpression(null);
1023         for (MapEntryExpression entryExpression : mapExpression.getMapEntryExpressions()) {
1024             Expression keyExpr = entryExpression.getKeyExpression();
1025             if (!(keyExpr instanceof ConstantExpression)) {
1026                 addStaticTypeError("Dynamic keys in map-style constructors are unsupported in static type checking", keyExpr);
1027             } else {
1028                 AtomicReference<ClassNode> lookup = new AtomicReference<ClassNode>();
1029                 PropertyExpression pexp = new PropertyExpression(new VariableExpression("_", receiverType), keyExpr.getText());
1030                 boolean hasProperty = existsProperty(pexp, false, new PropertyLookupVisitor(lookup));
1031                 if (!hasProperty) {
1032                     addStaticTypeError("No such property: " + keyExpr.getText() +
1033                             " for class: " + receiverType.getName(), receiver);
1034                 } else {
1035                     ClassNode valueType = getType(entryExpression.getValueExpression());
1036                     MethodNode setter = receiverType.getSetterMethod("set" + MetaClassHelper.capitalize(pexp.getPropertyAsString()), false);
1037                     ClassNode toBeAssignedTo = setter == null ? lookup.get() : setter.getParameters()[0].getType();
1038                     if (!isAssignableTo(valueType, toBeAssignedTo)
1039                             && !extension.handleIncompatibleAssignment(toBeAssignedTo, valueType, entryExpression)) {
1040                         addAssignmentError(toBeAssignedTo, valueType, entryExpression);
1041                     }
1042                 }
1043             }
1044         }
1045         typeCheckingContext.popEnclosingBinaryExpression();
1046     }

1074     /**
1075      * Checks that a constructor style expression is valid regarding the number of arguments and the argument types.
1076      *
1077      * @param node      the class node for which we will try to find a matching constructor
1078      * @param arguments the constructor arguments
1079      */
1080     protected MethodNode checkGroovyStyleConstructor(final ClassNode node, final ClassNode[] arguments, final ASTNode source) {
1081         if (node.equals(ClassHelper.OBJECT_TYPE) || node.equals(ClassHelper.DYNAMIC_TYPE)) {
1082             // in that case, we are facing a list constructor assigned to a def or object
1083             return null;
1084         }
1085         List<ConstructorNode> constructors = node.getDeclaredConstructors();
1086         if (constructors.isEmpty() && arguments.length == 0) {
1087             return null;
1088         }
1089         List<MethodNode> constructorList = findMethod(node, "<init>", arguments);
1090         if (constructorList.isEmpty()) {
1091             if (isBeingCompiled(node) && arguments.length==1 && LINKEDHASHMAP_CLASSNODE.equals(arguments[0])) {

1402      * to filter out illegal access to instance members from a static context.
1403      *
1404      * Return null if the given member is not static, but we want to access in
1405      * a static way (staticOnly=true). If we want to access in a non-static way

1407      * non-static members is allowed.

1411         if (member == null) return null;

1414         if (member instanceof Variable) {
1415             Variable v = (Variable) member;
1416             isStatic = Modifier.isStatic(v.getModifiers());

1421             }

1423         } else {
1424             MethodNode mn = (MethodNode) member;
1425             isStatic = mn.isStatic();
1426         }
1427         if (staticOnly && !isStatic) return null;
1428         return member;

1710     private void visitPrefixOrPostifExpression(final Expression origin, final Expression innerExpression, final int operationType) {
1711         boolean isPostfix = origin instanceof PostfixExpression;

1713         String name = operationType == PLUS_PLUS ? "next" : operationType == MINUS_MINUS ? "previous" : null;
1714         if (isPrimitiveType(exprType) || isPrimitiveType(getUnwrapper(exprType))) {

1718                     if (node != null) {

1720                         storeType(origin,
1721                                 isPostfix?exprType:getMathWideningClassNode(exprType));
1722                         return;
1723                     }
1724                 }
1725                 storeType(origin, exprType);
1726                 return;

2323     private void inferSAMType(Parameter param, ClassNode receiver, MethodNode methodWithSAMParameter, ArgumentListExpression originalMethodCallArguments, ClosureExpression openBlock) {
2324         // In a method call with SAM coercion the inference is to be
2325         // understood as a two phase process. We have the normal method call
2326         // to the target method with the closure argument and we have the 
2327         // SAM method that will be called inside the normal target method. 
2328         // To infer correctly we have to "simulate" this process. We know the
2329         // call to the closure will be done through the SAM type, so the SAM
2330         // type generics deliver information about the Closure. At the same
2331         // time the SAM class is used in the target method parameter, 
2332         // providing a connection from the SAM type and the target method
2333         // declaration class.
2334 
2335         // First we try to get as much information about the declaration
2336         // class through the receiver
2337         Map<String,GenericsType> targetMethodDeclarationClassConnections = new HashMap<String,GenericsType>();
2338         extractGenericsConnections(targetMethodDeclarationClassConnections, receiver, receiver.redirect());
2339         // then we use the method with the SAM parameter to get more information about the declaration
2340         Parameter[] parametersOfMethodContainingSAM = methodWithSAMParameter.getParameters();
2341         for (int i=0; i<parametersOfMethodContainingSAM.length; i++) {
2342             Expression callArg = originalMethodCallArguments.getExpression(i);
2343             // we look at the closure later in detail, so skip it here
2344             if (callArg==openBlock) continue;
2345             ClassNode parameterType = parametersOfMethodContainingSAM[i].getType();
2346             extractGenericsConnections(targetMethodDeclarationClassConnections, getType(callArg), parameterType);
2347         }
2348 
2349         // To make a connection to the SAM class we use that new information
2350         // to replace the generics in the SAM type parameter of the target
2351         // method and than that to make the connections to the SAM type generics
2352         ClassNode paramTypeWithReceiverInformation = applyGenericsContext(targetMethodDeclarationClassConnections, param.getOriginType());
2353         Map<String,GenericsType> SAMTypeConnections = new HashMap<String,GenericsType>();
2354         ClassNode classForSAM = paramTypeWithReceiverInformation.redirect();
2355         extractGenericsConnections(SAMTypeConnections, paramTypeWithReceiverInformation, classForSAM);
2356 
2357         // should the open block provide final information we apply that
2358         // to the corresponding parameters of the SAM type method
2359         MethodNode methodForSAM = findSAM(classForSAM);
2360         ClassNode[] parameterTypesForSAM = extractTypesFromParameters(methodForSAM.getParameters());
2361         ClassNode[] blockParameterTypes = (ClassNode[]) openBlock.getNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS);
2362         if (blockParameterTypes==null) {
2363             Parameter[] p = openBlock.getParameters();
2364             if (p.length==0 && parameterTypesForSAM.length!=0) {
2365                 // implicit it
2366                 blockParameterTypes = parameterTypesForSAM;
2367             } else {
2368                 blockParameterTypes = extractTypesFromParameters(p);
2369             }
2370         }
2371         for (int i=0; i<blockParameterTypes.length; i++) { //TODO: equal length guaranteed?
2372             extractGenericsConnections(SAMTypeConnections, blockParameterTypes[i], parameterTypesForSAM[i]);
2373         }
2374 
2375         // and finally we apply the generics information to the parameters and 
2376         // store the type of parameter and block type as meta information
2377         for (int i=0; i<blockParameterTypes.length; i++) { //TODO: equal length guaranteed?
2378             ClassNode resolvedParameter =
2379                     applyGenericsContext(SAMTypeConnections, parameterTypesForSAM[i]);
2380             blockParameterTypes[i] = resolvedParameter;
2381         }
2382         openBlock.putNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS, blockParameterTypes);
2383     }

2413     private void doInferClosureParameterTypes(final ClassNode receiver, final Expression arguments, final ClosureExpression expression, final MethodNode selectedMethod, final Expression hintClass, final Expression options) {
2414         List<ClassNode[]> closureSignatures = getSignaturesFromHint(expression, selectedMethod, hintClass, options);
2415         List<ClassNode[]> candidates = new LinkedList<ClassNode[]>();
2416         for (ClassNode[] signature : closureSignatures) {
2417             // in order to compute the inferred types of the closure parameters, we're using the following trick:
2418             // 1. create a dummy MethodNode for which the return type is a class node for which the generic types are the types returned by the hint
2419             // 2. call inferReturnTypeGenerics
2420             // 3. fetch inferred types from the result of inferReturnTypeGenerics
2421             // In practice, it could be done differently but it has the main advantage of reusing
2422             // existing code, hence reducing the amount of code to debug in case of failure.
2423             ClassNode[] inferred = resolveGenericsFromTypeHint(receiver, arguments, selectedMethod, signature);
2424             Parameter[] closureParams = expression.getParameters();
2425             if (signature.length == closureParams.length // same number of arguments
2426                     || (signature.length == 1 && closureParams.length == 0) // implicit it
2427                     || (closureParams.length > signature.length && inferred[inferred.length - 1].isArray())) { // vargs
2428                 candidates.add(inferred);
2429             }
2430         }
2431         Parameter[] closureParams = expression.getParameters();
2432         if (candidates.size()>1) {
2433             Iterator<ClassNode[]> candIt = candidates.iterator();
2434             while (candIt.hasNext()) {
2435                 ClassNode[] inferred = candIt.next();
2436                 final int length = closureParams.length;
2437                 for (int i = 0; i < length; i++) {
2438                     Parameter closureParam = closureParams[i];
2439                     final ClassNode originType = closureParam.getOriginType();
2440                     ClassNode inferredType;
2441                     if (i<inferred.length-1 || inferred.length==closureParams.length) {
2442                         inferredType = inferred[i];
2443                     } else { // vargs?
2444                         ClassNode lastArgInferred = inferred[inferred.length-1];
2445                         if (lastArgInferred.isArray()) {
2446                             inferredType = lastArgInferred.getComponentType();
2447                         } else {
2448                             candIt.remove();
2449                             continue;
2450                         }
2451                     }
2452                     if (!typeCheckMethodArgumentWithGenerics(originType, inferredType, i== length -1)) {
2453                         candIt.remove();
2454                     }
2455                 }
2456             }
2457             if (candidates.size()>1) {
2458                 addError("Ambiguous prototypes for closure. More than one target method matches. Please use explicit argument types.", expression);
2459             }
2460         }
2461         if (candidates.size()==1) {
2462             ClassNode[] inferred = candidates.get(0);
2463             if (closureParams.length==0 && inferred.length==1) {
2464                 expression.putNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS, inferred);
2465             } else {
2466                 final int length = closureParams.length;
2467                 for (int i = 0; i < length; i++) {
2468                     Parameter closureParam = closureParams[i];
2469                     final ClassNode originType = closureParam.getOriginType();
2470                     ClassNode inferredType = OBJECT_TYPE;
2471                     if (i<inferred.length-1 || inferred.length==closureParams.length) {
2472                         inferredType = inferred[i];
2473                     } else { // vargs?
2474                             ClassNode lastArgInferred = inferred[inferred.length-1];
2475                             if (lastArgInferred.isArray()) {
2476                                 inferredType = lastArgInferred.getComponentType();
2477                             } else {
2478                                 addError("Incorrect number of parameters. Expected "+inferred.length+" but found "+closureParams.length, expression);
2479                             }
2480                     }
2481                     boolean lastArg = i == length - 1;
2482                     if (lastArg && inferredType.isArray()) {
2483                         if (inferredType.getComponentType().equals(originType)) {
2484                             inferredType = originType;
2485                         }
2486                     } else if (!typeCheckMethodArgumentWithGenerics(originType, inferredType, lastArg)) {
2487                         addError("Expected parameter of type "+ inferredType.toString(false)+" but got "+originType.toString(false), closureParam.getType());
2488                     }
2489                     typeCheckingContext.controlStructureVariables.put(closureParam, inferredType);
2490                 }
2491             }
2492         }
2493     }

2676                               final Collection<Receiver<String>> owners,

2679             receivers.addAll(owners);

3267         boolean sourceIsNull = isNullConstant(source);

3269         if (targetType.isArray() && expressionType.isArray()) {

3272                 && source instanceof ConstantExpression && source.getText().length() == 1) {
3273             // ex: (char) 'c'
3274         } else if (targetType.equals(Character_TYPE) && (expressionType == STRING_TYPE || sourceIsNull)
3275                 && (sourceIsNull || source instanceof ConstantExpression && source.getText().length() == 1)) {
3276             // ex : (Character) 'c'
3277         } else if (isNumberCategory(getWrapper(targetType)) && (isNumberCategory(getWrapper(expressionType)) || char_TYPE == expressionType)) {
3278             // ex: short s = (short) 0
3279         } else if (sourceIsNull && !isPrimitiveType(targetType)) {
3280             // ex: (Date)null

3282             // char c = (char) ...
3283         } else if (sourceIsNull && isPrimitiveType(targetType) && !boolean_TYPE.equals(targetType)) {

3289         }

3344     protected void storeType(Expression exp, ClassNode cn) {
3345         if (exp instanceof VariableExpression && ((VariableExpression) exp).isClosureSharedVariable() && isPrimitiveType(cn)) {
3346             cn = getWrapper(cn);
3347         } else if (exp instanceof MethodCallExpression && ((MethodCallExpression) exp).isSafe() && isPrimitiveType(cn)) {
3348             cn = getWrapper(cn);
3349         }
3350         if (cn == UNKNOWN_PARAMETER_TYPE) {
3351             // this can happen for example when "null" is used in an assignment or a method parameter.
3352             // In that case, instead of storing the virtual type, we must "reset" type information
3353             // by determining the declaration type of the expression
3354             storeType(exp, getOriginalDeclarationType(exp));
3355             return;
3356         }
3357         ClassNode oldValue = (ClassNode) exp.putNodeMetaData(StaticTypesMarker.INFERRED_TYPE, cn);
3358         if (oldValue != null) {
3359             // this may happen when a variable declaration type is wider than the subsequent assignment values
3360             // for example :
3361             // def o = 1 // first, an int
3362             // o = 'String' // then a string
3363             // o = new Object() // and eventually an object !
3364             // in that case, the INFERRED_TYPE corresponds to the current inferred type, while
3365             // DECLARATION_INFERRED_TYPE is the type which should be used for the initial type declaration
3366             ClassNode oldDIT = (ClassNode) exp.getNodeMetaData(StaticTypesMarker.DECLARATION_INFERRED_TYPE);
3367             if (oldDIT != null) {
3368                 exp.putNodeMetaData(StaticTypesMarker.DECLARATION_INFERRED_TYPE, cn==null?oldDIT : lowestUpperBound(oldDIT, cn));
3369             } else {
3370                 exp.putNodeMetaData(StaticTypesMarker.DECLARATION_INFERRED_TYPE, cn==null?null : lowestUpperBound(oldValue, cn));
3371             }
3372         }
3373         if (exp instanceof VariableExpression) {
3374             VariableExpression var = (VariableExpression) exp;
3375             final Variable accessedVariable = var.getAccessedVariable();
3376             if (accessedVariable != null && accessedVariable != exp && accessedVariable instanceof VariableExpression) {
3377                 storeType((Expression) accessedVariable, cn);
3378             }
3379             if (var.isClosureSharedVariable() && cn!=null) {

3381                 if (assignedTypes == null) {

3384                 }
3385                 assignedTypes.add(cn);
3386             }
3387             if (!typeCheckingContext.temporaryIfBranchTypeInformation.empty()) {
3388                 List<ClassNode> temporaryTypesForExpression = getTemporaryTypesForExpression(exp);
3389                 if (temporaryTypesForExpression != null && !temporaryTypesForExpression.isEmpty()) {
3390                     // a type inference has been made on a variable which type was defined in an instanceof block
3391                     // we erase available information with the new type
3392                     temporaryTypesForExpression.clear();
3393                 }
3394             }

3482                 return boolean_TYPE;

3484         }

3501             if (DIVIDE == op) {

3503             }
3504             return leftRedirect;
3505         } else if (isOperationInGroup(op)) {
3506             if (isNumberCategory(getWrapper(leftRedirect)) && isNumberCategory(getWrapper(rightRedirect))) {
3507                 return getGroupOperationResultType(leftRedirect, rightRedirect);

3749             ClassNode receiver, String name, ClassNode... args) {
3750         if (isPrimitiveType(receiver)) receiver = getWrapper(receiver);
3751         List<MethodNode> methods;
3752         if (!receiver.isInterface() && "<init>".equals(name)) {
3753             methods = addGeneratedMethods(receiver,new ArrayList<MethodNode>(receiver.getDeclaredConstructors()));
3754             if (methods.isEmpty()) {
3755                 MethodNode node = new ConstructorNode(Opcodes.ACC_PUBLIC, Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, GENERATED_EMPTY_STATEMENT);
3756                 node.setDeclaringClass(receiver);
3757                 return Collections.singletonList(node);
3758             }
3759         } else {
3760             methods = findMethodsWithGenerated(receiver,name);
3761             if (receiver.isInterface()) {
3762                 collectAllInterfaceMethodsByName(receiver, name, methods);
3763                 methods.addAll(OBJECT_TYPE.getMethods(name));
3764             }
3765             if (typeCheckingContext.getEnclosingClosure() == null) {
3766                 // not in a closure
3767                 ClassNode parent = receiver;
3768                 while (parent instanceof InnerClassNode && !parent.isStaticClass()) {
3769                     parent = parent.getOuterClass();
3770                     methods.addAll(findMethodsWithGenerated(parent,name));
3771                 }
3772             }
3773             if (methods.isEmpty()) {
3774                 addArrayMethods(methods, receiver, name, args);
3775             }
3776             if (methods.isEmpty() && (args == null || args.length == 0)) {
3777                 // check if it's a property
3778                 String pname = extractPropertyNameFromMethodName("get", name);
3779                 if (pname==null) {
3780                     pname = extractPropertyNameFromMethodName("is", name);
3781                 }
3782                 if (pname != null) {
3783                     // we don't use property exists there because findMethod is called on super clases recursively
3784                     PropertyNode property = null;
3785                     ClassNode curNode = receiver;
3786                     while (property == null && curNode != null) {
3787                         property = curNode.getProperty(pname);

4341            if (expression instanceof ClosureExpression) {

4346                    closureParamTypes = expression.getNodeMetaData(StaticTypesMarker.CLOSURE_ARGUMENTS);
4347                }

4350                    final Parameter parameter = parameters[i];

4352                        genericsToConnect.add(new ClassNode[]{closureParamTypes[i], parameter.getOriginType()});
4353                    }
4354                }
4355                for (ClassNode[] classNodes : genericsToConnect) {
4356                    ClassNode found = classNodes[0];
4357                    ClassNode expected = classNodes[1];
4358                    if (!isAssignableTo(found, expected)) {
4359                        // probably facing a type mismatch
4360                        continue;
4361                    }
4362                    ClassNode generifiedType = GenericsUtils.parameterizeType(found, expected);
4363                    while (expected.isArray()) {
4364                        expected = expected.getComponentType();
4365                        generifiedType = generifiedType.getComponentType();
4366                    }
4367                    if (expected.isGenericsPlaceHolder()) {
4368                        placeholders.put(expected.getGenericsTypes()[0].getName(), new GenericsType(generifiedType));
4369                    } else {
4370                        GenericsType[] expectedGenericsTypes = expected.getGenericsTypes();
4371                        GenericsType[] foundGenericsTypes = generifiedType.getGenericsTypes();
4372 
4373                        for (int i = 0; i < expectedGenericsTypes.length; i++) {
4374                            final GenericsType type = expectedGenericsTypes[i];
4375                            if (type.isPlaceholder()) {
4376                                String name = type.getName();
4377                                placeholders.put(name, foundGenericsTypes[i]);
4378                            }
4379                        }
4380                    }
4381                }
4382            }
4383         }
4384         ClassNode result = applyGenericsContext(placeholders,samType.redirect());
4385         return result;
4386     }

4428     private static Map<String, GenericsType> extractPlaceHolders(MethodNode method, ClassNode receiver, ClassNode declaringClass) {
4429         if (declaringClass.equals(OBJECT_TYPE)) {
4430             Map<String, GenericsType> resolvedPlaceholders = new HashMap<String, GenericsType>();
4431             if (method!=null) addMethodLevelDeclaredGenerics(method, resolvedPlaceholders);
4432             return resolvedPlaceholders;
4433         }
4434 
4435         Map<String, GenericsType> resolvedPlaceholders = null;
4436         if (isPrimitiveType(receiver) && !isPrimitiveType(declaringClass)) {
4437             receiver = getWrapper(receiver);
4438         }
4439         final List<ClassNode> queue;
4440         if (receiver instanceof UnionTypeClassNode) {
4441             queue = Arrays.asList(((UnionTypeClassNode) receiver).getDelegates());
4442         } else {
4443             queue = Collections.singletonList(receiver);
4444         }
4445         for (ClassNode item : queue) {
4446             ClassNode current = item;
4447             while (current!=null) {
4448                 boolean continueLoop = true;
4449                 //extract the place holders
4450                 Map<String, GenericsType> currentPlaceHolders = new HashMap<String, GenericsType>();
4451                 if (isGenericsPlaceHolderOrArrayOf(declaringClass) || declaringClass.equals(current)) {
4452                     extractGenericsConnections(currentPlaceHolders, current, declaringClass);
4453                     if (method!=null) addMethodLevelDeclaredGenerics(method, currentPlaceHolders);
4454                     continueLoop = false;
4455                 } else {
4456                     GenericsUtils.extractPlaceholders(current, currentPlaceHolders);
4457                 }
4458 
4459                 if (resolvedPlaceholders!=null) {
4460                     // merge maps
4461                     Set<Map.Entry<String,GenericsType>> entries = currentPlaceHolders.entrySet();
4462                     for (Map.Entry<String,GenericsType> entry : entries) {
4463                         GenericsType gt = entry.getValue();
4464                         if (!gt.isPlaceholder()) continue;
4465                         GenericsType referenced = resolvedPlaceholders.get(gt.getName());
4466                         if (referenced==null) continue;
4467                         entry.setValue(referenced);
4468                     }
4469                 }
4470                 resolvedPlaceholders = currentPlaceHolders;
4471 
4472                 // we are done if we are now in the declaring class
4473                 if (!continueLoop) break;
4474 
4475                 current = getNextSuperClass(current, declaringClass);
4476                 if (current==null && CLASS_Type.equals(declaringClass)) {
4477                     // this can happen if the receiver is Class<Foo>, then
4478                     // the actual receiver is Foo and declaringClass is Class
4479                     current = declaringClass;
4480                 }
4481             }
4482         }
4483         if (resolvedPlaceholders==null) {
4484             String descriptor = "<>";
4485             if (method!=null) descriptor = method.getTypeDescriptor();
4486             throw new GroovyBugError(
4487                     "Declaring class for method call to '" +
4488                             descriptor + "' declared in " + declaringClass.getName() +
4489                             " was not matched with found receiver "+ receiver.getName() + "." +
4490                             " This should not have happened!");
4491         }
4492         return resolvedPlaceholders;
4493     }
