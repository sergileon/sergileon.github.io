19  package groovy.swing.impl;
20  
21  import javax.swing.*;
22  import java.util.Collection;
23  import java.util.Iterator;
24  import java.util.List;
25  import java.util.ListIterator;
26  
27  /**
28   * A {@code ListModel} implementation that's backed by a live {@code List}.
29   *
30   * @author Andres Almiray
31   */
32  public class ListWrapperListModel<E> extends AbstractListModel {
33      private final List<E> delegate;
34  
35      public ListWrapperListModel(List<E> delegateList) {
36          this.delegate = delegateList;
37      }
38  
39      public List<E> getDelegateList() {
40          return delegate;
41      }
42  
43      public int getSize() {
44          return delegate.size();
45      }
46  
47      public Object getElementAt(int i) {
48          return delegate.get(i);
49      }
50  
51      public E set(int i, E e) {
52          E element = delegate.set(i, e);
53          fireContentsChanged(this, i, i);
54          return element;
55      }
56  
57      public void clear() {
58          int i = delegate.size() - 1;
59          delegate.clear();
60          if (i >= 0) {
61              fireIntervalRemoved(this, 0, i);
62          }
63      }
64  
65      public int lastIndexOf(Object o) {
66          return delegate.lastIndexOf(o);
67      }
68  
69      public boolean contains(Object o) {
70          return delegate.contains(o);
71      }
72  
73      public ListIterator<E> listIterator() {
74          return delegate.listIterator();
75      }
76  
77      public boolean isEmpty() {
78          return delegate.isEmpty();
79      }
80  
81      public int indexOf(Object o) {
82          return delegate.indexOf(o);
83      }
84  
85      public void add(int i, E e) {
86          int index = delegate.size();
87          delegate.add(i, e);
88          fireIntervalAdded(this, index, index);
89      }
90  
91      public Iterator<E> iterator() {
92          return delegate.iterator();
93      }
94  
95      public boolean addAll(Collection<? extends E> es) {
96          int i = delegate.size();
97          boolean added = delegate.addAll(es);
98          if (added) fireIntervalAdded(this, i, i + es.size());
99          return added;
100     }
101 
102     public E remove(int i) {
103         E element = delegate.remove(i);
104         fireIntervalRemoved(this, i, i);
105         return element;
106     }
107 
108     public boolean addAll(int i, Collection<? extends E> es) {
109         boolean added = delegate.addAll(i, es);
110         if (added) fireIntervalAdded(this, i, i + es.size());
111         return added;
112     }
113 
114     public ListIterator<E> listIterator(int i) {
115         return delegate.listIterator(i);
116     }
117 
118     public boolean containsAll(Collection<?> objects) {
119         return delegate.containsAll(objects);
120     }
121 
122     public boolean remove(Object o) {
123         int i = indexOf(o);
124         boolean rv = delegate.remove(o);
125         if (i >= 0) {
126             fireIntervalRemoved(this, i, i);
127         }
128         return rv;
129     }
130 
131     public boolean add(E e) {
132         int i = delegate.size();
133         boolean added = delegate.add(e);
134         if (added) fireIntervalAdded(this, i, i);
135         return added;
136     }
137 
138     public E get(int i) {
139         return delegate.get(i);
140     }
141 
142     public <T> T[] toArray(T[] ts) {
143         return delegate.toArray(ts);
144     }
145 
146     public Object[] toArray() {
147         return delegate.toArray();
148     }
149 
150     public void removeRange(int fromIndex, int toIndex) {
151         if (fromIndex > toIndex) {
152             throw new IllegalArgumentException("fromIndex must be <= toIndex");
153         }
154         for (int i = toIndex; i >= fromIndex; i--) {
155             delegate.remove(i);
156         }
157         fireIntervalRemoved(this, fromIndex, toIndex);
158     }
159 }
