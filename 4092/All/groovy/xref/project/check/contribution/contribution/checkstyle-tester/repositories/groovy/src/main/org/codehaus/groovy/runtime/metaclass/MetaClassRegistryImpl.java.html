88      public MetaClassRegistryImpl(final int loadDefault, final boolean useAccessible) {
89          this.useAccessible = useAccessible;
90  
91          if (loadDefault == LOAD_DEFAULT) {
92              final Map<CachedClass, List<MetaMethod>> map = new HashMap<CachedClass, List<MetaMethod>>();
93  
94              // let's register the default methods
95              registerMethods(null, true, true, map);
96              final Class[] additionals = DefaultGroovyMethods.additionals;
97              for (int i = 0; i != additionals.length; ++i) {
98                  createMetaMethodFromClass(map, additionals[i]);
99              }
100 
101             Class[] pluginDGMs = VMPluginFactory.getPlugin().getPluginDefaultGroovyMethods();
102             for (Class plugin : pluginDGMs) {
103                 registerMethods(plugin, false, true, map);
104             }
105             registerMethods(DefaultGroovyStaticMethods.class, false, false, map);
106             Class[] staticPluginDGMs = VMPluginFactory.getPlugin().getPluginStaticGroovyMethods();
107             for (Class plugin : staticPluginDGMs) {
108                 registerMethods(plugin, false, false, map);
109             }
110 
111             ExtensionModuleScanner scanner = new ExtensionModuleScanner(new DefaultModuleListener(map), this.getClass().getClassLoader());
112             scanner.scanClasspathModules();
113 
114             refreshMopMethods(map);
115 
116         }
117 
118         installMetaClassCreationHandle();
119 
120         final MetaClass emcMetaClass = metaClassCreationHandle.create(ExpandoMetaClass.class, this);
121         emcMetaClass.initialize();
122         ClassInfo.getClassInfo(ExpandoMetaClass.class).setStrongMetaClass(emcMetaClass);
123         
124 
125         addNonRemovableMetaClassRegistryChangeEventListener(new MetaClassRegistryChangeEventListener(){
126             public void updateConstantMetaClass(MetaClassRegistryChangeEvent cmcu) {
127                 synchronized (metaClassInfo) {
128                    metaClassInfo.add(cmcu.getNewMetaClass());
129                    DefaultMetaClassInfo.getNewConstantMetaClassVersioning();
130                    Class c = cmcu.getClassToUpdate();
131                    DefaultMetaClassInfo.setPrimitiveMeta(c, cmcu.getNewMetaClass()==null);
132                    Field sdyn;
133                    try {
134                        sdyn = c.getDeclaredField(Verifier.STATIC_METACLASS_BOOL);
135                        sdyn.setBoolean(null, cmcu.getNewMetaClass()!=null);
136                    } catch (Throwable e) {
137                        //DO NOTHING
138                    }
139 
140                 }
141             }
142         });
143    }

145     private void refreshMopMethods(final Map<CachedClass, List<MetaMethod>> map) {
146         for (Map.Entry<CachedClass, List<MetaMethod>> e : map.entrySet()) {
147             CachedClass cls = e.getKey();
148             cls.setNewMopMethods(e.getValue());
149         }
150     }

161     /**
162      * Looks for a class called 'groovy.runtime.metaclass.CustomMetaClassCreationHandle' and if it exists uses it as the MetaClassCreationHandle
163      * otherwise uses the default
164      *
165      * @see groovy.lang.MetaClassRegistry.MetaClassCreationHandle
166      */
167     private void installMetaClassCreationHandle() {
168            try {
169                final Class customMetaClassHandle = Class.forName("groovy.runtime.metaclass.CustomMetaClassCreationHandle");
170                final Constructor customMetaClassHandleConstructor = customMetaClassHandle.getConstructor(new Class[]{});
171                  this.metaClassCreationHandle = (MetaClassCreationHandle)customMetaClassHandleConstructor.newInstance();
172            } catch (final ClassNotFoundException e) {
173                this.metaClassCreationHandle = new MetaClassCreationHandle();
174            } catch (final Exception e) {
175                throw new GroovyRuntimeException("Could not instantiate custom Metaclass creation handle: "+ e, e);
176            }
177     }

179     private void registerMethods(final Class theClass, final boolean useMethodWrapper, final boolean useInstanceMethods, Map<CachedClass, List<MetaMethod>> map) {
180         if (useMethodWrapper) {
181             // Here we instantiate objects representing MetaMethods for DGM methods.
182             // Calls for such meta methods done without reflection, so more effectively.
183 
184             try {
185                 List<GeneratedMetaMethod.DgmMethodRecord> records = GeneratedMetaMethod.DgmMethodRecord.loadDgmInfo();
186 
187                 for (GeneratedMetaMethod.DgmMethodRecord record : records) {
188                     Class[] newParams = new Class[record.parameters.length - 1];
189                     System.arraycopy(record.parameters, 1, newParams, 0, record.parameters.length-1);
190 
191                     MetaMethod method = new GeneratedMetaMethod.Proxy(
192                             record.className,
193                             record.methodName,
194                             ReflectionCache.getCachedClass(record.parameters[0]),
195                             record.returnType,
196                             newParams
197                     );
198                     final CachedClass declClass = method.getDeclaringClass();
199                     List<MetaMethod> arr = map.get(declClass);
200                     if (arr == null) {
201                         arr = new ArrayList<MetaMethod>(4);
202                         map.put(declClass, arr);
203                     }
204                     arr.add(method);
205                     instanceMethods.add(method);
206                 }
207             } catch (Throwable e) {
208                 e.printStackTrace();
209                 // we print the error, but we don't stop with an exception here
210                 // since it is more comfortable this way for development
211             }
212         } else {
213             CachedMethod[] methods = ReflectionCache.getCachedClass(theClass).getMethods();
214 
215             for (CachedMethod method : methods) {
216                 final int mod = method.getModifiers();
217                 if (Modifier.isStatic(mod) && Modifier.isPublic(mod) && method.getCachedMethod().getAnnotation(Deprecated.class) == null) {
218                     CachedClass[] paramTypes = method.getParameterTypes();
219                     if (paramTypes.length > 0) {
220                         List<MetaMethod> arr = map.get(paramTypes[0]);
221                         if (arr == null) {
222                             arr = new ArrayList<MetaMethod>(4);
223                             map.put(paramTypes[0], arr);
224                         }
225                         if (useInstanceMethods) {
226                             final NewInstanceMetaMethod metaMethod = new NewInstanceMetaMethod(method);
227                             arr.add(metaMethod);
228                             instanceMethods.add(metaMethod);
229                         } else {
230                             final NewStaticMetaMethod metaMethod = new NewStaticMetaMethod(method);
231                             arr.add(metaMethod);
232                             staticMethods.add(metaMethod);
233                         }
234                     }
235                 }
236             }
237         }
238     }

240     private void createMetaMethodFromClass(Map<CachedClass, List<MetaMethod>> map, Class aClass) {
241         try {
242             MetaMethod method = (MetaMethod) aClass.newInstance();
243             final CachedClass declClass = method.getDeclaringClass();
244             List<MetaMethod> arr = map.get(declClass);
245             if (arr == null) {
246                 arr = new ArrayList<MetaMethod>(4);
247                 map.put(declClass, arr);
248             }
249             arr.add(method);
250             instanceMethods.add(method);
251         } catch (InstantiationException e) { /* ignore */
252         } catch (IllegalAccessException e) { /* ignore */
253         }
254     }

264     /**
265      * if oldMc is null, newMc will replace whatever meta class was used before.
266      * if oldMc is not null, then newMc will be used only if he stored mc is
267      * the same as oldMc
268      */
269     private void setMetaClass(Class theClass, MetaClass oldMc, MetaClass newMc) {
270         final ClassInfo info = ClassInfo.getClassInfo(theClass);
271         
272         MetaClass mc = null;
273         info.lock();
274         try {
275             mc = info.getStrongMetaClass();
276             info.setStrongMetaClass(newMc);
277         } finally {
278             info.unlock();
279         }
280         if ((oldMc == null && mc != newMc) || (oldMc != null && mc != newMc && mc != oldMc)) {
281             fireConstantMetaClassUpdate(null, theClass, mc, newMc);
282         }
283     }

285     public void removeMetaClass(Class theClass) {
286         setMetaClass(theClass, null, null);

300     public void setMetaClass(Object obj, MetaClass theMetaClass) {
301         Class theClass = obj.getClass ();
302         final ClassInfo info = ClassInfo.getClassInfo(theClass);
303         MetaClass oldMC = null;
304         info.lock();
305         try {
306             oldMC = info.getPerInstanceMetaClass(obj);
307             info.setPerInstanceMetaClass(obj, theMetaClass);
308         }
309         finally {
310             info.unlock();
311         }
312         
313         fireConstantMetaClassUpdate(obj, theClass, oldMC, theMetaClass);
314     }

379     /**
380      * Causes the execution of all registered listeners. This method is used mostly
381      * internal to kick of the listener notification. It can also be used by subclasses
382      * to achieve the same.
383      *
384      * @param obj object instance if the MetaClass change is on a per-instance metaclass (or null if global)
385      * @param c the class
386      * @param oldMC the old MetaClass
387      * @param newMc the new MetaClass
388      */
389     protected void fireConstantMetaClassUpdate(Object obj, Class c, final MetaClass oldMC, MetaClass newMc) {
390         MetaClassRegistryChangeEventListener[]  listener = getMetaClassRegistryChangeEventListeners();
391         MetaClassRegistryChangeEvent cmcu = new MetaClassRegistryChangeEvent(this, obj, c, oldMC, newMc);
392         for (int i = 0; i<listener.length; i++) {
393             listener[i].updateConstantMetaClass(cmcu);
394         }
395     }

397     /**
398      * Gets an array of of all registered ConstantMetaClassListener instances.
399      */
400     public MetaClassRegistryChangeEventListener[] getMetaClassRegistryChangeEventListeners() {
401         synchronized (changeListenerList) {
402             ArrayList<MetaClassRegistryChangeEventListener> ret = new ArrayList(changeListenerList.size()+nonRemoveableChangeListenerList.size());
403             ret.addAll(nonRemoveableChangeListenerList);
404             ret.addAll(changeListenerList);
405             return (MetaClassRegistryChangeEventListener[]) ret.toArray(new MetaClassRegistryChangeEventListener[ret.size()]);
406         }
407     }

437     /**
438      * Returns an iterator to iterate over all constant meta classes.
439      * This iterator can be seen as making a snapshot of the current state
440      * of the registry. The snapshot will include all meta classes that has
441      * been used unless they are already collected. Collected meta classes 
442      * will be skipped automatically, so you can expect that each element
443      * of the iteration is not null. Calling this method is thread safe, the
444      * usage of the iterator is not.
445      *  
446      * @return the iterator.
447      */    
448     public Iterator iterator() {
449         final MetaClass[] refs;
450         synchronized (metaClassInfo) {
451             refs = (MetaClass[]) metaClassInfo.toArray(new MetaClass[0]);
452         }
453         
454         return new Iterator() {
455             // index in the ref array
456             private int index = 0;
457             // the current meta class
458             private MetaClass currentMeta;
459             // used to ensure that hasNext has been called
460             private boolean hasNextCalled = false;
461             // the cached hasNext call value
462             private boolean hasNext = false;
463 
464             public boolean hasNext() {
465                 if (hasNextCalled) return hasNext;
466                 hasNextCalled = true;
467                 if(index < refs.length) {
468                     hasNext = true;
469                     currentMeta = refs[index];
470                     index++;
471                 } else {
472                     hasNext = false;
473                 }
474                 return hasNext;
475             }
476             
477             private void ensureNext() {
478                 // we ensure that hasNext has been called before 
479                 // next is called
480                 hasNext();
481                 hasNextCalled = false;
482             }
483             
484             public Object next() {
485                 ensureNext();
486                 return currentMeta;
487             }
488             
489             public void remove() {
490                 ensureNext();
491                 setMetaClass(currentMeta.getTheClass(), currentMeta, null);
492                 currentMeta = null;
493             }
494         };
495     }

497     private class DefaultModuleListener implements ExtensionModuleScanner.ExtensionModuleListener {
498         private final Map<CachedClass, List<MetaMethod>> map;
499 
500         public DefaultModuleListener(final Map<CachedClass, List<MetaMethod>> map) {
501             this.map = map;
502         }
503 
504         public void onModule(final ExtensionModule module) {
505             if (moduleRegistry.hasModule(module.getName())) {
506             ExtensionModule loadedModule = moduleRegistry.getModule(module.getName());
507             if (loadedModule.getVersion().equals(module.getVersion())) {
508                 // already registered
509                 return;
510             } else {
511                 throw new GroovyRuntimeException("Conflicting module versions. Module ["+module.getName()+" is loaded in version "+
512                         loadedModule.getVersion()+" and you are trying to load version "+module.getVersion());
513             }
514         }
515             moduleRegistry.addModule(module);
516             // register MetaMethods
517             List<MetaMethod> metaMethods = module.getMetaMethods();
518             for (MetaMethod metaMethod : metaMethods) {
519                 CachedClass cachedClass = metaMethod.getDeclaringClass();
520                 List<MetaMethod> methods = map.get(cachedClass);
521                 if (methods==null) {
522                     methods = new ArrayList<MetaMethod>(4);
523                     map.put(cachedClass, methods);
524                 }
525                 methods.add(metaMethod);
526                 if (metaMethod.isStatic()) {
527                     staticMethods.add(metaMethod);
528                 } else {
529                     instanceMethods.add(metaMethod);
530                 }
531             }
532         }
533     }
