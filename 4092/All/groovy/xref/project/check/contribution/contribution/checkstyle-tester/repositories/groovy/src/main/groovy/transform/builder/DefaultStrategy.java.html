64  /**
65   * This strategy is used with the {@link Builder} AST transform to create a builder helper class
66   * for the fluent creation of instances of a specified class.&nbsp;It can be used at the class,
67   * static method or constructor levels.
68   *
69   * You use it as follows:
70   * <pre>
71   * import groovy.transform.builder.*
72   *
73   * {@code @Builder}
74   * class Person {
75   *     String firstName
76   *     String lastName
77   *     int age
78   * }
79   * def person = Person.builder().firstName("Robert").lastName("Lewandowski").age(21)
80   * assert person.firstName == "Robert"
81   * assert person.lastName == "Lewandowski"
82   * assert person.age == 21
83   * </pre>
84   * The {@code prefix} annotation attribute can be used to create setters with a different naming convention. The default is the
85   * empty string but you could change that to "set" as follows:
86   * <pre>
87   * {@code @Builder}(prefix='set')
88   * class Person {
89   *     String firstName
90   *     String lastName
91   *     int age
92   * }
93   * def p2 = Person.builder().setFirstName("Robert").setLastName("Lewandowski").setAge(21)
94   * </pre>
95   * or using a prefix of 'with' would result in usage like this:
96   * <pre>
97   * def p3 = Person.builder().withFirstName("Robert").withLastName("Lewandowski").withAge(21)
98   * </pre>
99   *
100  * You can also use the {@code @Builder} annotation in combination with this strategy on one or more constructor or
101  * static method instead of or in addition to using it at the class level. An example with a constructor follows:
102  * <pre>
103  * import groovy.transform.ToString
104  * import groovy.transform.builder.Builder
105  *
106  * {@code @ToString}
107  * class Person {
108  *     String first, last
109  *     int born
110  *
111  *     {@code @Builder}
112  *     Person(String roleName) {
113  *         if (roleName == 'Jack Sparrow') {
114  *             first = 'Johnny'; last = 'Depp'; born = 1963
115  *         }
116  *     }
117  * }
118  * assert Person.builder().roleName("Jack Sparrow").build().toString() == 'Person(Johnny, Depp, 1963)'
119  * </pre>
120  * In this case, the parameter(s) for the constructor or static method become the properties available
121  * in the builder. For the case of a static method, the return type of the static method becomes the
122  * class of the instance being created. For static factory methods, this is normally the class containing the
123  * static method but in general it can be any class.
124  *
125  * Note: if using more than one {@code @Builder} annotation, which is only possible when using static method
126  * or constructor variants, it is up to you to ensure that any generated helper classes or builder methods
127  * have unique names. E.g.&nbsp;we can modify the previous example to have three builders. At least two of the builders
128  * in our case will need to set the 'builderClassName' and 'builderMethodName' annotation attributes to ensure
129  * we have unique names. This is shown in the following example:
130  * <pre>
131  * import groovy.transform.builder.*
132  * import groovy.transform.*
133  *
134  * {@code @ToString}
135  * {@code @Builder}
136  * class Person {
137  *     String first, last
138  *     int born
139  *
140  *     Person(){} // required to retain no-arg constructor
141  *
142  *     {@code @Builder}(builderClassName='MovieBuilder', builderMethodName='byRoleBuilder')
143  *     Person(String roleName) {
144  *         if (roleName == 'Jack Sparrow') {
145  *             this.first = 'Johnny'; this.last = 'Depp'; this.born = 1963
146  *         }
147  *     }
148  *
149  *     {@code @Builder}(builderClassName='SplitBuilder', builderMethodName='splitBuilder')
150  *     static Person split(String name, int year) {
151  *         def parts = name.split(' ')
152  *         new Person(first: parts[0], last: parts[1], born: year)
153  *     }
154  * }
155  *
156  * assert Person.splitBuilder().name("Johnny Depp").year(1963).build().toString() == 'Person(Johnny, Depp, 1963)'
157  * assert Person.byRoleBuilder().roleName("Jack Sparrow").build().toString() == 'Person(Johnny, Depp, 1963)'
158  * assert Person.builder().first("Johnny").last('Depp').born(1963).build().toString() == 'Person(Johnny, Depp, 1963)'
159  * </pre>
160  *
161  * The 'forClass' annotation attribute for the {@code @Builder} transform isn't applicable for this strategy.
162  *
163  * @author Paul King
164  */
165 public class DefaultStrategy extends BuilderASTTransformation.AbstractBuilderStrategy {
166     private static final Expression DEFAULT_INITIAL_VALUE = null;
167     private static final int PUBLIC_STATIC = ACC_PUBLIC | ACC_STATIC;
168 
169     public void build(BuilderASTTransformation transform, AnnotatedNode annotatedNode, AnnotationNode anno) {
170         if (unsupportedAttribute(transform, anno, "forClass")) return;
171         if (annotatedNode instanceof ClassNode) {
172             buildClass(transform, (ClassNode) annotatedNode, anno);
173         } else if (annotatedNode instanceof MethodNode) {
174             buildMethod(transform, (MethodNode) annotatedNode, anno);
175         }
176     }
177 
178     public void buildMethod(BuilderASTTransformation transform, MethodNode mNode, AnnotationNode anno) {
179         if (transform.getMemberValue(anno, "includes") != null || transform.getMemberValue(anno, "includes") != null) {
180             transform.addError("Error during " + BuilderASTTransformation.MY_TYPE_NAME +
181                     " processing: includes/excludes only allowed on classes", anno);
182         }
183         ClassNode buildee = mNode.getDeclaringClass();
184         ClassNode builder = createBuilder(anno, buildee);
185         createBuilderFactoryMethod(anno, buildee, builder);
186         for (Parameter parameter : mNode.getParameters()) {
187             builder.addField(createFieldCopy(buildee, parameter));
188             builder.addMethod(createBuilderMethodForProp(builder, new PropertyInfo(parameter.getName(), parameter.getType()), getPrefix(anno)));
189         }
190         builder.addMethod(createBuildMethodForMethod(anno, buildee, mNode, mNode.getParameters()));
191     }
192 
193     public void buildClass(BuilderASTTransformation transform, ClassNode buildee, AnnotationNode anno) {
194         List<String> excludes = new ArrayList<String>();
195         List<String> includes = new ArrayList<String>();
196         if (!getIncludeExclude(transform, anno, buildee, excludes, includes)) return;
197         ClassNode builder = createBuilder(anno, buildee);
198         createBuilderFactoryMethod(anno, buildee, builder);
199         List<FieldNode> fields = getInstancePropertyFields(buildee);
200         List<FieldNode> filteredFields = selectFieldsFromExistingClass(fields, includes, excludes);
201         for (FieldNode fieldNode : filteredFields) {
202             ClassNode correctedType = getCorrectedType(buildee, fieldNode);
203             String fieldName = fieldNode.getName();
204             builder.addField(createFieldCopy(buildee, fieldName, correctedType));
205             builder.addMethod(createBuilderMethodForProp(builder, new PropertyInfo(fieldName, correctedType), getPrefix(anno)));
206         }
207         builder.addMethod(createBuildMethod(anno, buildee, filteredFields));
208     }
209 
210     private ClassNode getCorrectedType(ClassNode buildee, FieldNode fieldNode) {
211         Map<String,ClassNode> genericsSpec = createGenericsSpec(fieldNode.getDeclaringClass());
212         extractSuperClassGenerics(fieldNode.getType(), buildee, genericsSpec);
213         return correctToGenericsSpecRecurse(genericsSpec, fieldNode.getType());
214     }
215 
216     private void createBuilderFactoryMethod(AnnotationNode anno, ClassNode buildee, ClassNode builder) {
217         buildee.getModule().addClass(builder);
218         buildee.addMethod(createBuilderMethod(anno, builder));
219     }
220 
221     private ClassNode createBuilder(AnnotationNode anno, ClassNode buildee) {
222         return new InnerClassNode(buildee, getFullName(anno, buildee), PUBLIC_STATIC, OBJECT_TYPE);
223     }
224 
225     private String getFullName(AnnotationNode anno, ClassNode buildee) {
226         String builderClassName = getMemberStringValue(anno, "builderClassName", buildee.getNameWithoutPackage() + "Builder");
227         return buildee.getName() + "$" + builderClassName;
228     }
229 
230     private String getPrefix(AnnotationNode anno) {
231         return getMemberStringValue(anno, "prefix", "");
232     }
233 
234     private MethodNode createBuildMethodForMethod(AnnotationNode anno, ClassNode buildee, MethodNode mNode, Parameter[] params) {
235         String buildMethodName = getMemberStringValue(anno, "buildMethodName", "build");
236         final BlockStatement body = new BlockStatement();
237         ClassNode returnType;
238         if (mNode instanceof ConstructorNode) {
239             returnType = newClass(buildee);
240             body.addStatement(returnS(ctorX(newClass(mNode.getDeclaringClass()), args(params))));
241         } else {
242             body.addStatement(returnS(callX(newClass(mNode.getDeclaringClass()), mNode.getName(), args(params))));
243             returnType = newClass(mNode.getReturnType());
244         }
245         return new MethodNode(buildMethodName, ACC_PUBLIC, returnType, NO_PARAMS, NO_EXCEPTIONS, body);
246     }
247 
248     private static MethodNode createBuilderMethod(AnnotationNode anno, ClassNode builder) {
249         String builderMethodName = getMemberStringValue(anno, "builderMethodName", "builder");
250         final BlockStatement body = new BlockStatement();
251         body.addStatement(returnS(ctorX(builder)));
252         return new MethodNode(builderMethodName, PUBLIC_STATIC, builder, NO_PARAMS, NO_EXCEPTIONS, body);
253     }
254 
255     private static MethodNode createBuildMethod(AnnotationNode anno, ClassNode buildee, List<FieldNode> fields) {
256         String buildMethodName = getMemberStringValue(anno, "buildMethodName", "build");
257         final BlockStatement body = new BlockStatement();
258         body.addStatement(returnS(initializeInstance(buildee, fields, body)));
259         return new MethodNode(buildMethodName, ACC_PUBLIC, newClass(buildee), NO_PARAMS, NO_EXCEPTIONS, body);
260     }
261 
262     private MethodNode createBuilderMethodForProp(ClassNode builder, PropertyInfo pinfo, String prefix) {
263         ClassNode fieldType = pinfo.getType();
264         String fieldName = pinfo.getName();
265         String setterName = getSetterName(prefix, fieldName);
266         return new MethodNode(setterName, ACC_PUBLIC, newClass(builder), params(param(fieldType, fieldName)), NO_EXCEPTIONS, block(
267                 stmt(assignX(propX(varX("this"), constX(fieldName)), varX(fieldName, fieldType))),
268                 returnS(varX("this", builder))
269         ));
270     }
271 
272     private static FieldNode createFieldCopy(ClassNode buildee, Parameter param) {
273         Map<String,ClassNode> genericsSpec = createGenericsSpec(buildee);
274         extractSuperClassGenerics(param.getType(), buildee, genericsSpec);
275         ClassNode correctedParamType = correctToGenericsSpecRecurse(genericsSpec, param.getType());
276         return new FieldNode(param.getName(), ACC_PRIVATE, correctedParamType, buildee, param.getInitialExpression());
277     }
278 
279     private static FieldNode createFieldCopy(ClassNode buildee, String fieldName, ClassNode fieldType) {
280         return new FieldNode(fieldName, ACC_PRIVATE, fieldType, buildee, DEFAULT_INITIAL_VALUE);
281     }
282 
283     private static List<FieldNode> selectFieldsFromExistingClass(List<FieldNode> fieldNodes, List<String> includes, List<String> excludes) {
284         List<FieldNode> fields = new ArrayList<FieldNode>();
285         for (FieldNode fNode : fieldNodes) {
286             if (shouldSkip(fNode.getName(), excludes, includes)) continue;
287             fields.add(fNode);
288         }
289         return fields;
290     }
291 
292     private static Expression initializeInstance(ClassNode buildee, List<FieldNode> fields, BlockStatement body) {
293         Expression instance = varX("_the" + buildee.getNameWithoutPackage(), buildee);
294         body.addStatement(declS(instance, ctorX(buildee)));
295         for (FieldNode field : fields) {
296             body.addStatement(stmt(assignX(propX(instance, field.getName()), varX(field))));
297         }
298         return instance;
299     }
