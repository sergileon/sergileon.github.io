65  /**
66   * Compiles Groovy source files using Ant.
67   * <p>
68   * Typically involves using Ant from the command-line and an Ant build file such as:
69   * <pre>
70   * &lt;?xml version="1.0"?&gt;
71   * &lt;project name="MyGroovyBuild" default="compile"&gt;
72   *   &lt;property name="groovy.home" value="/Path/To/Groovy"/&gt;
73   *   &lt;property name="groovy.version" value="X.Y.Z"/&gt;
74   *   &lt;path id="groovy.classpath"&gt;
75   *     &lt;fileset dir="${groovy.home}/embeddable"&gt;
76   *       &lt;include name="groovy-all-${groovy.version}.jar" /&gt;
77   *     &lt;/fileset&gt;
78   *   &lt;/path&gt;
79   *   &lt;taskdef name="groovyc" classname="org.codehaus.groovy.ant.Groovyc" classpathref="groovy.classpath"/&gt;
80   *
81   *   &lt;target name="compile" description="compile groovy sources"&gt;
82   *     &lt;groovyc srcdir="src" listfiles="true" classpathref="groovy.classpath"/&gt;
83   *   &lt;/target&gt;
84   * &lt;/project&gt;
85   * </pre>
86   * <p>
87   * This task can take the following arguments:
88   * <ul>
89   * <li>srcdir</li>
90   * <li>scriptExtension</li>
91   * <li>targetBytecode</li>
92   * <li>destdir</li>
93   * <li>sourcepath</li>
94   * <li>sourcepathRef</li>
95   * <li>classpath</li>
96   * <li>classpathRef</li>
97   * <li>listfiles</li>
98   * <li>failonerror</li>
99   * <li>proceed</li>
100  * <li>memoryInitialSize</li>
101  * <li>memoryMaximumSize</li>
102  * <li>encoding</li>
103  * <li>verbose</li>
104  * <li>includeantruntime</li>
105  * <li>includejavaruntime</li>
106  * <li>fork</li>
107  * <li>javaHome</li>
108  * <li>executable</li>
109  * <li>updatedProperty</li>
110  * <li>errorProperty</li>
111  * <li>includeDestClasses</li>
112  * <li>jointCompilationOptions</li>
113  * <li>stacktrace</li>
114  * <li>indy</li>
115  * <li>scriptBaseClass</li>
116  * <li>stubdir</li>
117  * <li>keepStubs</li>
118  * <li>forceLookupUnnamedFiles</li>
119  * <li>configscript</li>
120  * </ul>
121  * And these nested tasks:
122  * <ul>
123  * <li>javac</li>
124  * </ul>
125  * Of these arguments, the <b>srcdir</b> and <b>destdir</b> are required.
126  * <p>
127  * <p>When this task executes, it will recursively scan srcdir and destdir looking for Groovy source files
128  * to compile. This task makes its compile decision based on timestamp.
129  * <p>
130  * A more elaborate build file showing joint compilation:
131  * <pre>
132  * &lt;?xml version="1.0"?&gt;
133  * &lt;project name="MyJointBuild" default="compile"&gt;
134  *   &lt;property name="groovy.home" value="/Path/To/Groovy"/&gt;
135  *   &lt;property name="groovy.version" value="X.Y.Z"/&gt;
136  *
137  *   &lt;path id="groovy.classpath"&gt;
138  *     &lt;fileset dir="${groovy.home}/embeddable"&gt;
139  *       &lt;include name="groovy-all-${groovy.version}.jar" /&gt;
140  *     &lt;/fileset&gt;
141  *   &lt;/path&gt;
142  *
143  *   &lt;target name="clean" description="remove all built files"&gt;
144  *     &lt;delete dir="classes" /&gt;
145  *   &lt;/target&gt;
146  *
147  *   &lt;target name="compile" depends="init" description="compile java and groovy sources"&gt;
148  *     &lt;mkdir dir="classes" /&gt;
149  *     &lt;groovyc destdir="classes" srcdir="src" listfiles="true" keepStubs="true" stubdir="stubs"&gt;
150  *       &lt;javac debug="on" deprecation="true"/&gt;
151  *       &lt;classpath&gt;
152  *         &lt;fileset dir="classes"/&gt;
153  *         &lt;path refid="groovy.classpath"/&gt;
154  *       &lt;/classpath&gt;
155  *     &lt;/groovyc&gt;
156  *   &lt;/target&gt;
157  *
158  *   &lt;target name="init"&gt;
159  *     &lt;taskdef name="groovyc" classname="org.codehaus.groovy.ant.Groovyc" classpathref="groovy.classpath"/&gt;
160  *   &lt;/target&gt;
161  * &lt;/project&gt;
162  * </pre>
163  * <p>
164  * Based heavily on the Javac implementation in Ant.
165  * <p>
166  * Can also be used from {@link groovy.util.AntBuilder} to allow the build file to be scripted in Groovy.
167  *
168  * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>
169  * @author Hein Meling
170  * @author <a href="mailto:russel.winder@concertant.com">Russel Winder</a>
171  * @author Danno Ferrin
172  * @author Paul King
173  */

262     /**
263      * Set the extension to use when searching for Groovy source files.
264      * Accepts extensions in the form *.groovy, .groovy or groovy
265      *
266      * @param scriptExtension the extension of Groovy source files
267      */
268     public void setScriptExtension(String scriptExtension) {
269         if (scriptExtension.startsWith("*.")) {
270             this.scriptExtension = scriptExtension;
271         } else if (scriptExtension.startsWith(".")) {
272             this.scriptExtension = "*" + scriptExtension;
273         } else {
274             this.scriptExtension = "*." + scriptExtension;
275         }
276     }

806     /**
807      * Executes the task.
808      *
809      * @throws BuildException if an error occurs
810      */
811     public void execute() throws BuildException {
812         checkParameters();
813         resetFileLists();
814         loadRegisteredScriptExtensions();
815 
816         if (javac != null) jointCompilation = true;
817 
818         // scan source directories and dest directory to build up
819         // compile lists
820         String[] list = src.list();
821         for (String filename : list) {
822             File file = getProject().resolveFile(filename);
823             if (!file.exists()) {
824                 throw new BuildException("srcdir \"" + file.getPath() + "\" does not exist!", getLocation());
825             }
826             DirectoryScanner ds = this.getDirectoryScanner(file);
827             String[] files = ds.getIncludedFiles();
828             scanDir(file, destDir != null ? destDir : file, files);
829         }
830 
831         compile();
832         if (updatedProperty != null
833                 && taskSuccess
834                 && compileList.length != 0) {
835             getProject().setNewProperty(updatedProperty, "true");
836         }
837     }

847     /**
848      * Scans the directory looking for source files to be compiled.
849      * The results are returned in the class variable compileList
850      *
851      * @param srcDir  The source directory
852      * @param destDir The destination directory
853      * @param files   An array of filenames
854      */
855     protected void scanDir(File srcDir, File destDir, String[] files) {
856         GlobPatternMapper m = new GlobPatternMapper();
857         SourceFileScanner sfs = new SourceFileScanner(this);
858         File[] newFiles;
859         for (String extension : getScriptExtensions()) {
860             m.setFrom("*." + extension);
861             m.setTo("*.class");
862             newFiles = sfs.restrictAsFiles(files, srcDir, destDir, m);
863             addToCompileList(newFiles);
864         }
865 
866         if (jointCompilation) {
867             m.setFrom("*.java");
868             m.setTo("*.class");
869             newFiles = sfs.restrictAsFiles(files, srcDir, destDir, m);
870             addToCompileList(newFiles);
871         }
872     }

874     protected void addToCompileList(File[] newFiles) {
875         if (newFiles.length > 0) {
876             File[] newCompileList = new File[compileList.length + newFiles.length];
877             System.arraycopy(compileList, 0, newCompileList, 0, compileList.length);
878             System.arraycopy(newFiles, 0, newCompileList, compileList.length, newFiles.length);
879             compileList = newCompileList;
880         }
881     }

920     private List<String> extractJointOptions(Path classpath) {
921         List<String> jointOptions = new ArrayList<String>();
922         if (!jointCompilation) return jointOptions;
923 
924         // extract joint options, some get pushed up...
925         RuntimeConfigurable rc = javac.getRuntimeConfigurableWrapper();
926         for (Object o1 : rc.getAttributeMap().entrySet()) {
927             final Map.Entry e = (Map.Entry) o1;
928             final String key = e.getKey().toString();
929             final String value = getProject().replaceProperties(e.getValue().toString());
930             if (key.contains("debug")) {
931                 String level = "";
932                 if (javac.getDebugLevel() != null) {
933                     level = ":" + javac.getDebugLevel();
934                 }
935                 jointOptions.add("-Fg" + level);
936             } else if (key.contains("debugLevel")) {
937                 // ignore, taken care of in debug
938             } else if ((key.contains("nowarn"))
939                     || (key.contains("verbose"))
940                     || (key.contains("deprecation"))) {
941                 // false is default, so something to do only in true case
942                 if ("on".equalsIgnoreCase(value) || "true".equalsIgnoreCase(value) || "yes".equalsIgnoreCase(value))
943                     jointOptions.add("-F" + key);
944             } else if (key.contains("classpath")) {
945                 classpath.add(javac.getClasspath());
946             } else if ((key.contains("depend"))
947                     || (key.contains("extdirs"))
948                     || (key.contains("encoding"))
949                     || (key.contains("source"))
950                     || (key.contains("target"))
951                     || (key.contains("verbose"))) { // already handling verbose but pass on too
952                 jointOptions.add("-J" + key + "=" + value);
953             } else {
954                 log.warn("The option " + key + " cannot be set on the contained <javac> element. The option will be ignored");
955             }
956             // TODO includes? excludes?
957         }
958 
959         // ant's <javac> supports nested <compilerarg value=""> elements (there can be multiple of them)
960         // for additional options to be passed to javac.
961         Enumeration children = rc.getChildren();
962         while (children.hasMoreElements()) {
963             RuntimeConfigurable childrc = (RuntimeConfigurable) children.nextElement();
964             if (childrc.getElementTag().equals("compilerarg")) {
965                 for (Object o : childrc.getAttributeMap().entrySet()) {
966                     final Map.Entry e = (Map.Entry) o;
967                     final String key = e.getKey().toString();
968                     if (key.equals("value")) {
969                         final String value = getProject().replaceProperties(e.getValue().toString());
970                         StringTokenizer st = new StringTokenizer(value, " ");
971                         while (st.hasMoreTokens()) {
972                             String optionStr = st.nextToken();
973                             String replaced = optionStr.replace("-X", "-FX");
974                             if (optionStr.equals(replaced)) {
975                                 replaced = optionStr.replace("-W", "-FW"); // GROOVY-5063
976                             }
977                             jointOptions.add(replaced);
978                         }
979                     }
980                 }
981             }
982         }
983 
984         return jointOptions;
985     }

987     private void doForkCommandLineList(List<String> commandLineList, Path classpath, String separator) {
988         if (!fork) return;
989 
990         if (includeAntRuntime) {
991             classpath.addExisting((new Path(getProject())).concatSystemClasspath("last"));
992         }
993         if (includeJavaRuntime) {
994             classpath.addJavaRuntime();
995         }
996 
997         if (forkedExecutable != null && !forkedExecutable.equals("")) {
998             commandLineList.add(forkedExecutable);
999         } else {
1000             String javaHome;
1001             if (forkJavaHome != null) {
1002                 javaHome = forkJavaHome.getPath();
1003             } else {
1004                 javaHome = System.getProperty("java.home");
1005             }
1006             commandLineList.add(javaHome + separator + "bin" + separator + "java");
1007         }
1008         commandLineList.add("-classpath");
1009         commandLineList.add(classpath.toString());
1010 
1011         final String fileEncodingProp = System.getProperty("file.encoding");
1012         if ((fileEncodingProp != null) && !fileEncodingProp.equals("")) {
1013             commandLineList.add("-Dfile.encoding=" + fileEncodingProp);
1014         }
1015         if (targetBytecode != null) {
1016             commandLineList.add("-Dgroovy.target.bytecode=" + targetBytecode);
1017         }
1018 
1019         if ((memoryInitialSize != null) && !memoryInitialSize.equals("")) {
1020             commandLineList.add("-Xms" + memoryInitialSize);
1021         }
1022         if ((memoryMaximumSize != null) && !memoryMaximumSize.equals("")) {
1023             commandLineList.add("-Xmx" + memoryMaximumSize);
1024         }
1025         if (!"*.groovy".equals(getScriptExtension())) {
1026             String tmpExtension = getScriptExtension();
1027             if (tmpExtension.startsWith("*."))
1028                 tmpExtension = tmpExtension.substring(1);
1029             commandLineList.add("-Dgroovy.default.scriptExtension=" + tmpExtension);
1030         }
1031         commandLineList.add(FileSystemCompilerFacade.class.getName());
1032         if (forceLookupUnnamedFiles) {
1033             commandLineList.add("--forceLookupUnnamedFiles");
1034         }
1035     }

1068     private void addSourceFiles(List<String> commandLineList) {
1069         // check to see if an external file is needed
1070         int count = 0;
1071         if (fork) {
1072             for (File srcFile : compileList) {
1073                 count += srcFile.getPath().length();
1074             }
1075             for (Object commandLineArg : commandLineList) {
1076                 count += commandLineArg.toString().length();
1077             }
1078             count += compileList.length;
1079             count += commandLineList.size();
1080         }
1081         // 32767 is the command line length limit on Windows
1082         if (fork && (count > 32767)) {
1083             try {
1084                 File tempFile = File.createTempFile("groovyc-files-", ".txt");
1085                 temporaryFiles.add(tempFile);
1086                 PrintWriter pw = new PrintWriter(new FileWriter(tempFile));
1087                 for (File srcFile : compileList) {
1088                     pw.println(srcFile.getPath());
1089                 }
1090                 pw.close();
1091                 commandLineList.add("@" + tempFile.getPath());
1092             } catch (IOException e) {
1093                 log.error("Error creating file list", e);
1094             }
1095         } else {
1096             for (File srcFile : compileList) {
1097                 commandLineList.add(srcFile.getPath());
1098             }
1099         }
1100     }

1137     private void runCompiler(String[] commandLine) {
1138         // hand crank it so we can add our own compiler configuration
1139         try {
1140             Options options = FileSystemCompiler.createCompilationOptions();
1141 
1142             CommandLineParser cliParser = new GroovyInternalPosixParser();
1143 
1144             CommandLine cli;
1145             cli = cliParser.parse(options, commandLine);
1146 
1147             configuration = FileSystemCompiler.generateCompilerConfigurationFromOptions(cli);
1148             configuration.setScriptExtensions(getScriptExtensions());
1149             String tmpExtension = getScriptExtension();
1150             if (tmpExtension.startsWith("*."))
1151                 tmpExtension = tmpExtension.substring(1);
1152             configuration.setDefaultScriptExtension(tmpExtension);
1153 
1154             // Load the file name list
1155             String[] filenames = FileSystemCompiler.generateFileNamesFromOptions(cli);
1156             boolean fileNameErrors = filenames == null;
1157 
1158             fileNameErrors = fileNameErrors && !FileSystemCompiler.validateFiles(filenames);
1159 
1160             if (targetBytecode != null) {
1161                 configuration.setTargetBytecode(targetBytecode);
1162             }
1163 
1164             if (!fileNameErrors) {
1165                 FileSystemCompiler.doCompilation(configuration, makeCompileUnit(), filenames, forceLookupUnnamedFiles);
1166             }
1167 
1168         } catch (Exception re) {
1169             Throwable t = re;
1170             if ((re.getClass() == RuntimeException.class) && (re.getCause() != null)) {
1171                 // unwrap to the real exception
1172                 t = re.getCause();
1173             }
1174             StringWriter writer = new StringWriter();
1175             new ErrorReporter(t, false).write(new PrintWriter(writer));
1176             String message = writer.toString();
1177 
1178             taskSuccess = false;
1179             if (errorProperty != null) {
1180                 getProject().setNewProperty(errorProperty, "true");
1181             }
1182 
1183             if (failOnError) {
1184                 log.error(message);
1185                 throw new BuildException("Compilation Failed", t, getLocation());
1186             } else {
1187                 log.error(message);
1188             }
1189         }
1190     }

1192     protected void compile() {
1193         if (compileList.length == 0) return;
1194 
1195         try {
1196             log.info("Compiling " + compileList.length + " source file"
1197                     + (compileList.length == 1 ? "" : "s")
1198                     + (destDir != null ? " to " + destDir : ""));
1199 
1200             listFiles();
1201             Path classpath = getClasspath() != null ? getClasspath() : new Path(getProject());
1202             List<String> jointOptions = extractJointOptions(classpath);
1203 
1204             String separator = System.getProperty("file.separator");
1205             List<String> commandLineList = new ArrayList<String>();
1206 
1207             doForkCommandLineList(commandLineList, classpath, separator);
1208             doNormalCommandLineList(commandLineList, jointOptions, classpath);
1209             addSourceFiles(commandLineList);
1210 
1211             String[] commandLine = makeCommandLine(commandLineList);
1212 
1213             if (fork) {
1214                 runForked(commandLine);
1215             } else {
1216                 runCompiler(commandLine);
1217             }
1218         } finally {
1219             for (File temporaryFile : temporaryFiles) {
1220                 try {
1221                     FileSystemCompiler.deleteRecursive(temporaryFile);
1222                 } catch (Throwable t) {
1223                     System.err.println("error: could not delete temp files - " + temporaryFile.getPath());
1224                 }
1225             }
1226         }
1227     }

1229     protected CompilationUnit makeCompileUnit() {
1230         Map<String, Object> options = configuration.getJointCompilationOptions();
1231         if (options != null) {
1232             if (keepStubs) {
1233                 options.put("keepStubs", Boolean.TRUE);
1234             }
1235             if (stubDir != null) {
1236                 options.put("stubDir", stubDir);
1237             } else {
1238                 try {
1239                     File tempStubDir = DefaultGroovyStaticMethods.createTempDir(null, "groovy-generated-", "-java-source");
1240                     temporaryFiles.add(tempStubDir);
1241                     options.put("stubDir", tempStubDir);
1242                 } catch (IOException ioe) {
1243                     throw new BuildException(ioe);
1244                 }
1245             }
1246             return new JavaAwareCompilationUnit(configuration, buildClassLoaderFor());
1247         } else {
1248             return new CompilationUnit(configuration, null, buildClassLoaderFor());
1249         }

1252     protected GroovyClassLoader buildClassLoaderFor() {
1253         // GROOVY-5044
1254         if (!fork && !getIncludeantruntime()) {
1255             throw new IllegalArgumentException("The includeAntRuntime=false option is not compatible with fork=false");
1256         }
1257         ClassLoader parent = getIncludeantruntime()
1258                 ? getClass().getClassLoader()
1259                 : new AntClassLoader(new RootLoader(EMPTY_URL_ARRAY, null), getProject(), getClasspath());
1260         if (parent instanceof AntClassLoader) {
1261             AntClassLoader antLoader = (AntClassLoader) parent;
1262             String[] pathElm = antLoader.getClasspath().split(File.pathSeparator);
1263             List<String> classpath = configuration.getClasspath();
1264             /*
1265              * Iterate over the classpath provided to groovyc, and add any missing path
1266              * entries to the AntClassLoader.  This is a workaround, since for some reason
1267              * 'directory' classpath entries were not added to the AntClassLoader' classpath.
1268              */
1269             for (String cpEntry : classpath) {
1270                 boolean found = false;
1271                 for (String path : pathElm) {
1272                     if (cpEntry.equals(path)) {
1273                         found = true;
1274                         break;
1275                     }
1276                 }
1277                 /*
1278                  * fix for GROOVY-2284
1279                  * seems like AntClassLoader doesn't check if the file
1280                  * may not exist in the classpath yet
1281                  */
1282                 if (!found && new File(cpEntry).exists()) {
1283                 	try {
1284                 		antLoader.addPathElement(cpEntry);
1285                 	}
1286                 	catch(BuildException e) {
1287                 		log.warn("The classpath entry " + cpEntry + " is not a valid Java resource");
1288                 	}
1289                 }
1290             }
1291         }
1292 
1293         GroovyClassLoader loader = new GroovyClassLoader(parent, configuration);
1294         if (!forceLookupUnnamedFiles) {
1295             // in normal case we don't need to do script lookups
1296             loader.setResourceLoader(new GroovyResourceLoader() {
1297                 public URL loadGroovySource(String filename) throws MalformedURLException {
1298                     return null;
1299                 }
1300             });
1301         }
1302         return loader;
1303     }

1309     private void loadRegisteredScriptExtensions() {
1310         if (scriptExtensions.isEmpty()) {
1311 
1312             scriptExtensions.add(getScriptExtension().substring(2)); // first extension will be the one set explicitly on <groovyc>

1317             for (String file : pe) {
1318                 loader.addClasspath(file);
1319             }
1320             scriptExtensions.addAll(SourceExtensionHandler.getRegisteredExtensions(loader));
1321         }
