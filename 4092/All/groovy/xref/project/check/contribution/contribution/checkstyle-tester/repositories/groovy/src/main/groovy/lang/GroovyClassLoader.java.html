48  /**
49   * A ClassLoader which can load Groovy classes. The loaded classes are cached,
50   * classes from other classloaders should not be cached. To be able to load a
51   * script that was asked for earlier but was created later it is essential not
52   * to keep anything like a "class not found" information for that class name.
53   * This includes possible parent loaders. Classes that are not cached are always
54   * reloaded.
55   *
56   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>

61   * @author <a href="mailto:blackdrag@gmx.org">Jochen Theodorou</a>

159     /**
160      * Loads the given class node returning the implementation Class.
161      * <p>
162      * WARNING: this compilation is not synchronized
163      *
164      * @param classNode
165      * @return a class
166      */
167     public Class defineClass(ClassNode classNode, String file, String newCodeBase) {
168         CodeSource codeSource = null;
169         try {
170             codeSource = new CodeSource(new URL("file", "", newCodeBase), (java.security.cert.Certificate[]) null);
171         } catch (MalformedURLException e) {
172             //swallow
173         }
174 
175         CompilationUnit unit = createCompilationUnit(config, codeSource);
176         ClassCollector collector = createCollector(unit, classNode.getModule().getContext());
177         try {
178             unit.addClassNode(classNode);
179             unit.setClassgenCallback(collector);
180             unit.compile(Phases.CLASS_GENERATION);
181             definePackage(collector.generatedClass.getName());
182             return collector.generatedClass;
183         } catch (CompilationFailedException e) {
184             throw new RuntimeException(e);
185         }
186     }

231     /**
232      * @deprecated Prefer using methods taking a Reader rather than an InputStream to avoid wrong encoding issues.
233      */
234     public Class parseClass(final InputStream in, final String fileName) throws CompilationFailedException {
235         // For generic input streams, provide a catch-all codebase of GroovyScript
236         // Security for these classes can be administered via policy grants with
237         // a codebase of file:groovy.script
238         GroovyCodeSource gcs = AccessController.doPrivileged(new PrivilegedAction<GroovyCodeSource>() {
239             public GroovyCodeSource run() {
240                 try {
241                     String scriptText = config.getSourceEncoding() != null ?
242                             IOGroovyMethods.getText(in, config.getSourceEncoding()) :
243                             IOGroovyMethods.getText(in);
244                     return new GroovyCodeSource(scriptText, fileName, "/groovy/script");
245                 } catch (IOException e) {
246                     throw new RuntimeException("Impossible to read the content of the input stream for file named: " + fileName, e);
247                 }
248             }
249         });
250         return parseClass(gcs);
251     }

257     /**
258      * Parses the given code source into a Java class. If there is a class file
259      * for the given code source, then no parsing is done, instead the cached class is returned.
260      *
261      * @param shouldCacheSource if true then the generated class will be stored in the source cache
262      * @return the main class defined in the given script
263      */
264     public Class parseClass(GroovyCodeSource codeSource, boolean shouldCacheSource) throws CompilationFailedException {
265         synchronized (sourceCache) {
266             Class answer = sourceCache.get(codeSource.getName());
267             if (answer != null) return answer;
268             answer = doParseClass(codeSource);
269             if (shouldCacheSource) sourceCache.put(codeSource.getName(), answer);
270             return answer;
271         }
272     }
273 
274     private Class doParseClass(GroovyCodeSource codeSource) {
275         validate(codeSource);
276         Class answer;  // Was neither already loaded nor compiling, so compile and add to cache.
277         CompilationUnit unit = createCompilationUnit(config, codeSource.getCodeSource());
278         if (recompile!=null && recompile || recompile==null && config.getRecompileGroovySource()) {
279             unit.addFirstPhaseOperation(TimestampAdder.INSTANCE, CompilePhase.CLASS_GENERATION.getPhaseNumber());
280         }
281         SourceUnit su = null;
282         File file = codeSource.getFile();
283         if (file != null) {
284             su = unit.addSource(file);
285         } else {
286             URL url = codeSource.getURL();
287             if (url != null) {
288                 su = unit.addSource(url);
289             } else {
290                 su = unit.addSource(codeSource.getName(), codeSource.getScriptText());
291             }
292         }
293 
294         ClassCollector collector = createCollector(unit, su);
295         unit.setClassgenCallback(collector);
296         int goalPhase = Phases.CLASS_GENERATION;
297         if (config != null && config.getTargetDirectory() != null) goalPhase = Phases.OUTPUT;
298         unit.compile(goalPhase);
299 
300         answer = collector.generatedClass;
301         String mainClass = su.getAST().getMainClassName();
302         for (Object o : collector.getLoadedClasses()) {
303             Class clazz = (Class) o;
304             String clazzName = clazz.getName();
305             definePackage(clazzName);
306             setClassCacheEntry(clazz);
307             if (clazzName.equals(mainClass)) answer = clazz;
308         }
309         return answer;
310     }
311 
312     private void validate(GroovyCodeSource codeSource) {
313         if (codeSource.getFile() == null) {
314             if (codeSource.getScriptText() == null) {
315                 throw new IllegalArgumentException("Script text to compile cannot be null!");
316             }
317         }
318     }

320     private void definePackage(String className) {
321         int i = className.lastIndexOf('.');
322         if (i != -1) {
323             String pkgName = className.substring(0, i);
324             java.lang.Package pkg = getPackage(pkgName);
325             if (pkg == null) {
326                 definePackage(pkgName, null, null, null, null, null, null, null);
327             }
328         }
329     }

347     protected PermissionCollection getPermissions(CodeSource codeSource) {
348         PermissionCollection perms;
349         try {
350             try {
351                 perms = super.getPermissions(codeSource);
352             } catch (SecurityException e) {
353                 // We lied about our CodeSource and that makes URLClassLoader unhappy.
354                 perms = new Permissions();
355             }
356 
357             ProtectionDomain myDomain = AccessController.doPrivileged(new PrivilegedAction<ProtectionDomain>() {
358                 public ProtectionDomain run() {
359                     return getClass().getProtectionDomain();
360                 }
361             });
362             PermissionCollection myPerms = myDomain.getPermissions();
363             if (myPerms != null) {
364                 for (Enumeration<Permission> elements = myPerms.elements(); elements.hasMoreElements();) {
365                     perms.add(elements.nextElement());
366                 }
367             }
368         } catch (Throwable e) {
369             // We lied about our CodeSource and that makes URLClassLoader unhappy.
370             perms = new Permissions();
371         }
372         perms.setReadOnly();
373         return perms;
374     }

475     public static class ClassCollector extends CompilationUnit.ClassgenCallback {
476         private Class generatedClass;
477         private final GroovyClassLoader cl;
478         private final SourceUnit su;
479         private final CompilationUnit unit;
480         private final Collection<Class> loadedClasses;
481 
482         protected ClassCollector(InnerLoader cl, CompilationUnit unit, SourceUnit su) {
483             this.cl = cl;
484             this.unit = unit;
485             this.loadedClasses = new ArrayList<Class>();
486             this.su = su;
487         }
488 
489         public GroovyClassLoader getDefiningClassLoader() {
490             return cl;
491         }
492 
493         protected Class createClass(byte[] code, ClassNode classNode) {
494             BytecodeProcessor bytecodePostprocessor = unit.getConfiguration().getBytecodePostprocessor();
495             byte[] fcode = code;
496             if (bytecodePostprocessor!=null) {
497                 fcode = bytecodePostprocessor.processBytecode(classNode.getName(), fcode);
498             }
499             GroovyClassLoader cl = getDefiningClassLoader();
500             Class theClass = cl.defineClass(classNode.getName(), fcode, 0, fcode.length, unit.getAST().getCodeSource());
501             this.loadedClasses.add(theClass);
502 
503             if (generatedClass == null) {
504                 ModuleNode mn = classNode.getModule();
505                 SourceUnit msu = null;
506                 if (mn != null) msu = mn.getContext();
507                 ClassNode main = null;
508                 if (mn != null) main = (ClassNode) mn.getClasses().get(0);
509                 if (msu == su && main == classNode) generatedClass = theClass;
510             }
511 
512             return theClass;
513         }
514 
515         protected Class onClassNode(ClassWriter classWriter, ClassNode classNode) {
516             byte[] code = classWriter.toByteArray();
517             return createClass(code, classNode);
518         }
519 
520         public void call(ClassVisitor classWriter, ClassNode classNode) {
521             onClassNode((ClassWriter) classWriter, classNode);
522         }
523 
524         public Collection getLoadedClasses() {
525             return this.loadedClasses;
526         }
527     }

603     /**
604      * Indicates if a class is recompilable. Recompilable means, that the classloader
605      * will try to locate a groovy source file for this class and then compile it again,
606      * adding the resulting class as entry to the cache. Giving null as class is like a
607      * recompilation, so the method should always return true here. Only classes that are
608      * implementing GroovyObject are compilable and only if the timestamp in the class
609      * is lower than Long.MAX_VALUE.
610      * <p>
611      * NOTE: First the parent loaders will be asked and only if they don't return a
612      * class the recompilation will happen. Recompilation also only happen if the source
613      * file is newer.
614      *
615      * @param cls the class to be tested. If null the method should return true
616      * @return true if the class should be compiled again
617      * @see #isSourceNewer(URL, Class)
618      */
619     protected boolean isRecompilable(Class cls) {
620         if (cls == null) return true;
621         if (cls.getClassLoader() == this) return false;
622         if (recompile == null && !config.getRecompileGroovySource()) return false;
623         if (recompile != null && !recompile) return false;
624         if (!GroovyObject.class.isAssignableFrom(cls)) return false;
625         long timestamp = getTimeStamp(cls);
626         if (timestamp == Long.MAX_VALUE) return false;
627         return true;
628     }

654     /**
655      * loads a class from a file or a parent classloader.
656      *
657      * @param name                  of the class to be loaded
658      * @param lookupScriptFiles     if false no lookup at files is done at all
659      * @param preferClassOverScript if true the file lookup is only done if there is no class
660      * @param resolve               see {@link java.lang.ClassLoader#loadClass(java.lang.String, boolean)}
661      * @return the class found or the class created from a file lookup
662      * @throws ClassNotFoundException     if the class could not be found
663      * @throws CompilationFailedException if the source file could not be compiled
664      */
665     public Class loadClass(final String name, boolean lookupScriptFiles, boolean preferClassOverScript, boolean resolve)
666             throws ClassNotFoundException, CompilationFailedException {
667         // look into cache
668         Class cls = getClassCacheEntry(name);
669 
670         // enable recompilation?
671         boolean recompile = isRecompilable(cls);
672         if (!recompile) return cls;
673 
674         // try parent loader
675         ClassNotFoundException last = null;
676         try {
677             Class parentClassLoaderClass = super.loadClass(name, resolve);
678             // always return if the parent loader was successful
679             if (cls != parentClassLoaderClass) return parentClassLoaderClass;
680         } catch (ClassNotFoundException cnfe) {
681             last = cnfe;
682         } catch (NoClassDefFoundError ncdfe) {
683             if (ncdfe.getMessage().indexOf("wrong name") > 0) {
684                 last = new ClassNotFoundException(name);
685             } else {
686                 throw ncdfe;
687             }
688         }
689 
690         // check security manager
691         SecurityManager sm = System.getSecurityManager();
692         if (sm != null) {
693             String className = name.replace('/', '.');
694             int i = className.lastIndexOf('.');
695             // no checks on the sun.reflect classes for reflection speed-up
696             // in particular ConstructorAccessorImpl, MethodAccessorImpl, FieldAccessorImpl and SerializationConstructorAccessorImpl
697             // which are generated at runtime by the JDK
698             if (i != -1 && !className.startsWith("sun.reflect.")) {
699                 sm.checkPackageAccess(className.substring(0, i));
700             }
701         }
702 
703         // prefer class if no recompilation
704         if (cls != null && preferClassOverScript) return cls;
705 
706         // at this point the loading from a parent loader failed
707         // and we want to recompile if needed.
708         if (lookupScriptFiles) {
709             // try groovy file
710             try {
711                 // check if recompilation already happened.
712                 final Class classCacheEntry = getClassCacheEntry(name);
713                 if (classCacheEntry != cls) return classCacheEntry;
714                 URL source = resourceLoader.loadGroovySource(name);
715                 // if recompilation fails, we want cls==null
716                 Class oldClass = cls;
717                 cls = null;
718                 cls = recompile(source, name, oldClass);
719             } catch (IOException ioe) {
720                 last = new ClassNotFoundException("IOException while opening groovy source: " + name, ioe);
721             } finally {
722                 if (cls == null) {
723                     removeClassCacheEntry(name);
724                 } else {
725                     setClassCacheEntry(cls);
726                 }
727             }
728         }
729 
730         if (cls == null) {
731             // no class found, there should have been an exception before now
732             if (last == null) throw new AssertionError(true);
733             throw last;
734         }
735         return cls;
736     }

738     /**
739      * (Re)Compiles the given source.
740      * This method starts the compilation of a given source, if
741      * the source has changed since the class was created. For
742      * this isSourceNewer is called.
743      *
744      * @param source    the source pointer for the compilation
745      * @param className the name of the class to be generated
746      * @param oldClass  a possible former class
747      * @return the old class if the source wasn't new enough, the new class else
748      * @throws CompilationFailedException if the compilation failed
749      * @throws IOException                if the source is not readable
750      * @see #isSourceNewer(URL, Class)
751      */
752     protected Class recompile(URL source, String className, Class oldClass) throws CompilationFailedException, IOException {
753         if (source != null) {
754             // found a source, compile it if newer
755             if ((oldClass != null && isSourceNewer(source, oldClass)) || (oldClass == null)) {
756                 synchronized (sourceCache) {
757                     String name = source.toExternalForm();
758                     sourceCache.remove(name);
759                     if (isFile(source)) {
760                         try {
761                             return parseClass(new GroovyCodeSource(new File(source.toURI()), config.getSourceEncoding()));
762                         } catch (URISyntaxException e) {
763                           // do nothing and fall back to the other version
764                         }
765                     } 
766                     return parseClass(source.openStream(), name);
767                 }
768             }
769         }
770         return oldClass;
771     }

803     /**
804      * This method will take a file name and try to "decode" any URL encoded characters.  For example
805      * if the file name contains any spaces this method call will take the resulting %20 encoded values
806      * and convert them to spaces.
807      * <p>
808      * This method was added specifically to fix defect:  Groovy-1787.  The defect involved a situation
809      * where two scripts were sitting in a directory with spaces in its name.  The code would fail
810      * when the class loader tried to resolve the file name and would choke on the URLEncoded space values.
811      */
812     private String decodeFileName(String fileName) {
813         String decodedFile = fileName;
814         try {
815             decodedFile = URLDecoder.decode(fileName, "UTF-8");
816         } catch (UnsupportedEncodingException e) {
817             System.err.println("Encountered an invalid encoding scheme when trying to use URLDecoder.decode() inside of the GroovyClassLoader.decodeFileName() method.  Returning the unencoded URL.");
818             System.err.println("Please note that if you encounter this error and you have spaces in your directory you will run into issues.  Refer to GROOVY-1787 for description of this bug.");
819         }
820 
821         return decodedFile;
822     }

825         return ret != null && ret.getProtocol().equals("file");

828     private File getFileForUrl(URL ret, String filename) {
829         String fileWithoutPackage = filename;
830         if (fileWithoutPackage.indexOf('/') != -1) {
831             int index = fileWithoutPackage.lastIndexOf('/');
832             fileWithoutPackage = fileWithoutPackage.substring(index + 1);
833         }
834         return fileReallyExists(ret, fileWithoutPackage);
835     }

837     private File fileReallyExists(URL ret, String fileWithoutPackage) {
838         File path;
839         try {
840             /* fix for GROOVY-5809 */ 
841             path = new File(ret.toURI());
842         } catch(URISyntaxException e) {
843             path = new File(decodeFileName(ret.getFile()));
844         }
845         path = path.getParentFile();
846         if (path.exists() && path.isDirectory()) {
847             File file = new File(path, fileWithoutPackage);
848             if (file.exists()) {
849                 // file.exists() might be case insensitive. Let's do
850                 // case sensitive match for the filename
851                 File parent = file.getParentFile();
852                 for (String child : parent.list()) {
853                     if (child.equals(fileWithoutPackage)) return file;
854                 }
855             }
856         }
857         //file does not exist!
858         return null;
859     }

861     private URL getSourceFile(String name, String extension) {
862         String filename = name.replace('.', '/') + "." + extension;
863         URL ret = getResource(filename);
864         if (isFile(ret) && getFileForUrl(ret, filename) == null) return null;
865         return ret;

868     /**
869      * Decides if the given source is newer than a class.
870      *
871      * @param source the source we may want to compile
872      * @param cls    the former class
873      * @return true if the source is newer, false else
874      * @throws IOException if it is not possible to open an
875      *                     connection for the given source
876      * @see #getTimeStamp(Class)
877      */
878     protected boolean isSourceNewer(URL source, Class cls) throws IOException {
879         long lastMod;
880 
881         // Special handling for file:// protocol, as getLastModified() often reports
882         // incorrect results (-1)
883         if (isFile(source)) {
884             // Coerce the file URL to a File
885             // See ClassNodeResolver.isSourceNewer for another method that replaces '|' with ':'.
886             // WTF: Why is this done and where is it documented?
887             String path = source.getPath().replace('/', File.separatorChar).replace('|', ':');
888             File file = new File(path);
889             lastMod = file.lastModified();
890         } else {
891             URLConnection conn = source.openConnection();
892             lastMod = conn.getLastModified();
893             conn.getInputStream().close();
894         }
895         long classTime = getTimeStamp(cls);
896         return classTime + config.getMinimumRecompilationInterval() < lastMod;
897     }

899     /**
900      * adds a classpath to this classloader.
901      *
902      * @param path is a jar file or a directory.
903      * @see #addURL(URL)
904      */
905     public void addClasspath(final String path) {
906         AccessController.doPrivileged(new PrivilegedAction<Void>() {
907             public Void run() {
908 
909                 URI newURI;
910                 try {
911                     newURI = new URI(path);
912                     // check if we can create a URL from that URI
913                     newURI.toURL();
914                 } catch (URISyntaxException e) {
915                     // the URI has a false format, so lets try it with files ...
916                     newURI=new File(path).toURI();
917                 } catch (MalformedURLException e) {
918                     // the URL has a false format, so lets try it with files ...
919                     newURI=new File(path).toURI();
920                 } catch (IllegalArgumentException e) {
921                     // the URL is not absolute, so lets try it with files ...
922                     newURI=new File(path).toURI();
923                 }
924 
925                 URL[] urls = getURLs();
926                 for (URL url : urls) {
927                     // Do not use URL.equals.  It uses the network to resolve names and compares ip addresses!
928                     // That is a violation of RFC and just plain evil.
929                     // http://michaelscharf.blogspot.com/2006/11/javaneturlequals-and-hashcode-make.html
930                     // http://docs.oracle.com/javase/7/docs/api/java/net/URL.html#equals(java.lang.Object)
931                     // "Since hosts comparison requires name resolution, this operation is a blocking operation.
932                     // Note: The defined behavior for equals is known to be inconsistent with virtual hosting in HTTP."
933                     try {
934                         if (newURI.equals(url.toURI())) return null;
935                     } catch (URISyntaxException e) {
936                         // fail fast! if we got a malformed URI the Classloader has to tell it
937                         throw new RuntimeException( e );
938                     }
939                 }
940                 try {
941                     addURL(newURI.toURL());
942                 } catch (MalformedURLException e) {
943                     // fail fast! if we got a malformed URL the Classloader has to tell it
944                     throw new RuntimeException( e );
945                 }
946                 return null;
947             }
948         });
949     }

984         protected void addTimeStamp(ClassNode node) {
985             if (node.getDeclaredField(Verifier.__TIMESTAMP) == null) { // in case if verifier visited the call already
986                 FieldNode timeTagField = new FieldNode(
987                         Verifier.__TIMESTAMP,
988                         ACC_PUBLIC | ACC_STATIC | ACC_SYNTHETIC,
989                         ClassHelper.long_TYPE,
990                         //"",
991                         node,

993                 // alternatively, FieldNode timeTagField = SourceUnit.createFieldNode("public static final long __timeStamp = " + System.currentTimeMillis() + "L");
994                 timeTagField.setSynthetic(true);
995                 node.addField(timeTagField);
996 
997                 timeTagField = new FieldNode(
998                         Verifier.__TIMESTAMP__ + String.valueOf(System.currentTimeMillis()),
999                         ACC_PUBLIC | ACC_STATIC | ACC_SYNTHETIC,
1000                         ClassHelper.long_TYPE,
1001                         //"",
1002                         node,
1003                         new ConstantExpression((long) 0));
1004                 // alternatively, FieldNode timeTagField = SourceUnit.createFieldNode("public static final long __timeStamp = " + System.currentTimeMillis() + "L");
1005                 timeTagField.setSynthetic(true);

1007             }
