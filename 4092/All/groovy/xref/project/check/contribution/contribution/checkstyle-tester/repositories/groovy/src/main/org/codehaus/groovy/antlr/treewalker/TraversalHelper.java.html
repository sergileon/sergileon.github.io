296     protected void accept_v_FirstChildsFirstChild_v_Child2_Child3_v_Child4_v___v_LastChild(GroovySourceAST t) {
297         openingVisit(t);
298         GroovySourceAST expr2 = t.childAt(0);
299         skip(expr2);
300         accept(expr2.childAt(0));
301         closingVisit(t);
302 
303         GroovySourceAST sibling = (GroovySourceAST) expr2.getNextSibling();
304         boolean firstSList = true;
305         while (sibling != null) {
306             if (!firstSList) {
307                 subsequentVisit(t);
308             }
309             firstSList = false;
310             accept(sibling);
311             sibling = (GroovySourceAST) sibling.getNextSibling();
312         }
313     }
