46  /**
47   * This visitor is responsible for amending the AST with static compilation metadata or transform the AST so that
48   * a class or a method can be statically compiled. It may also throw errors specific to static compilation which
49   * are not considered as an error at the type check pass. For example, usage of spread operator is not allowed
50   * in statically compiled portions of code, while it may be statically checked.
51   *
52   * Static compilation relies on static type checking, which explains why this visitor extends the type checker
53   * visitor.
54   *
55   * @author Cedric Champeau
56   */
57  public class StaticCompilationVisitor extends StaticTypeCheckingVisitor {
58      private static final ClassNode TYPECHECKED_CLASSNODE = ClassHelper.make(TypeChecked.class);
59      private static final ClassNode COMPILESTATIC_CLASSNODE = ClassHelper.make(CompileStatic.class);
60      private static final ClassNode[] TYPECHECKED_ANNOTATIONS = {TYPECHECKED_CLASSNODE, COMPILESTATIC_CLASSNODE};
61

84      public static boolean isStaticallyCompiled(AnnotatedNode node) {
85          if (node.getNodeMetaData(STATIC_COMPILE_NODE)!=null) return (Boolean)node.getNodeMetaData(STATIC_COMPILE_NODE);
86          if (node instanceof MethodNode) {
87              return isStaticallyCompiled(node.getDeclaringClass());
88          }
89          if (node instanceof InnerClassNode) {
90              return isStaticallyCompiled(((InnerClassNode)node).getOuterClass());
91          }
92          return false;
93      }

95      private void addPrivateFieldAndMethodAccessors(ClassNode node) {
96          addPrivateBridgeMethods(node);
97          addPrivateFieldsAccessors(node);
98          Iterator<InnerClassNode> it = node.getInnerClasses();
99          while (it.hasNext()) {
100             addPrivateFieldAndMethodAccessors(it.next());
101         }
102     }

104     @Override
105     public void visitClass(final ClassNode node) {
106         boolean skip = shouldSkipClassNode(node);
107         if (!skip && !anyMethodSkip(node)) {
108             node.putNodeMetaData(MopWriter.Factory.class, StaticCompilationMopWriter.FACTORY);
109         }
110         ClassNode oldCN = classNode;
111         classNode = node;
112         Iterator<InnerClassNode> innerClasses = classNode.getInnerClasses();
113         while (innerClasses.hasNext()) {
114             InnerClassNode innerClassNode = innerClasses.next();
115             boolean innerStaticCompile = !(skip || isSkippedInnerClass(innerClassNode));
116             innerClassNode.putNodeMetaData(STATIC_COMPILE_NODE, innerStaticCompile);
117             innerClassNode.putNodeMetaData(WriterControllerFactory.class, node.getNodeMetaData(WriterControllerFactory.class));
118             if (innerStaticCompile && !anyMethodSkip(innerClassNode)) {
119                 innerClassNode.putNodeMetaData(MopWriter.Factory.class, StaticCompilationMopWriter.FACTORY);
120             }
121         }
122         super.visitClass(node);
123         addPrivateFieldAndMethodAccessors(node);
124         classNode = oldCN;
125     }
126 
127     private boolean anyMethodSkip(final ClassNode node) {
128         for (MethodNode methodNode : node.getMethods()) {
129             if (isSkipMode(methodNode)) return true;
130         }
131         return false;
132     }

134     /**
135      * If we are in a constructor, that is static compiled, but in a class, that
136      * is not, it may happen that init code from object initializers, fields
137      * or properties is added into the constructor code. The backend assumes
138      * a purely static contructor, so it may fail if it encounters dynamic
139      * code here. Thus we make this kind of code fail
140      */
141     private void checkForConstructorWithCSButClassWithout(MethodNode node) {
142         if (!(node instanceof ConstructorNode)) return;
143         Object meta = node.getNodeMetaData(STATIC_COMPILE_NODE);
144         if (!Boolean.TRUE.equals(meta)) return;
145         ClassNode clz = typeCheckingContext.getEnclosingClassNode();
146         meta = clz.getNodeMetaData(STATIC_COMPILE_NODE);
147         if (Boolean.TRUE.equals(meta)) return;
148         if (    clz.getObjectInitializerStatements().isEmpty() &&
149                 clz.getFields().isEmpty() &&
150                 clz.getProperties().isEmpty())
151         {
152             return;
153         }
154 
155         addStaticTypeError("Cannot statically compile constructor implicitly including non static elements from object initializers, properties or fields.",node);
156     }
157 
158     @Override
159     public void visitMethod(final MethodNode node) {
160         if (isSkipMode(node)) {
161             node.putNodeMetaData(STATIC_COMPILE_NODE, false);
162         }
163         super.visitMethod(node);
164         checkForConstructorWithCSButClassWithout(node);
165     }

167     /**
168      * Adds special accessors for private constants so that inner classes can retrieve them.
169      */
170     @SuppressWarnings("unchecked")
171     private void addPrivateFieldsAccessors(ClassNode node) {
172         Set<ASTNode> accessedFields = (Set<ASTNode>) node.getNodeMetaData(StaticTypesMarker.PV_FIELDS_ACCESS);
173         if (accessedFields==null) return;
174         Map<String, MethodNode> privateConstantAccessors = (Map<String, MethodNode>) node.getNodeMetaData(PRIVATE_FIELDS_ACCESSORS);
175         if (privateConstantAccessors!=null) {
176             // already added
177             return;
178         }
179         int acc = -1;
180         privateConstantAccessors = new HashMap<String, MethodNode>();
181         final int access = Opcodes.ACC_STATIC | Opcodes.ACC_PUBLIC | Opcodes.ACC_SYNTHETIC;
182         for (FieldNode fieldNode : node.getFields()) {
183             if (accessedFields.contains(fieldNode)) {
184 
185                 acc++;
186                 Parameter param = new Parameter(node.getPlainNodeReference(), "$that");
187                 Expression receiver = fieldNode.isStatic()?new ClassExpression(node):new VariableExpression(param);
188                 Statement stmt = new ExpressionStatement(new PropertyExpression(
189                         receiver,
190                         fieldNode.getName()
191                 ));
192                 MethodNode accessor = node.addMethod("pfaccess$"+acc, access, fieldNode.getOriginType(), new Parameter[]{param}, ClassNode.EMPTY_ARRAY, stmt);
193                 privateConstantAccessors.put(fieldNode.getName(), accessor);
194             }
195         }
196         node.setNodeMetaData(PRIVATE_FIELDS_ACCESSORS, privateConstantAccessors);
197     }

199     /**
200      * This method is used to add "bridge" methods for private methods of an inner/outer
201      * class, so that the outer class is capable of calling them. It does basically
202      * the same job as access$000 like methods in Java.
203      *
204      * @param node an inner/outer class node for which to generate bridge methods
205      */
206     @SuppressWarnings("unchecked")
207     private void addPrivateBridgeMethods(final ClassNode node) {
208         Set<ASTNode> accessedMethods = (Set<ASTNode>) node.getNodeMetaData(StaticTypesMarker.PV_METHODS_ACCESS);
209         if (accessedMethods==null) return;
210         List<MethodNode> methods = new ArrayList<MethodNode>(node.getAllDeclaredMethods());
211         Map<MethodNode, MethodNode> privateBridgeMethods = (Map<MethodNode, MethodNode>) node.getNodeMetaData(PRIVATE_BRIDGE_METHODS);
212         if (privateBridgeMethods!=null) {
213             // private bridge methods already added
214             return;
215         }
216         privateBridgeMethods = new HashMap<MethodNode, MethodNode>();
217         int i=-1;
218         final int access = Opcodes.ACC_STATIC | Opcodes.ACC_PUBLIC | Opcodes.ACC_SYNTHETIC;
219         for (MethodNode method : methods) {
220             if (accessedMethods.contains(method)) {
221                 List<String> methodSpecificGenerics = methodSpecificGenerics(method);
222                 i++;
223                 ClassNode declaringClass = method.getDeclaringClass();
224                 Map<String,ClassNode> genericsSpec = createGenericsSpec(node);
225                 genericsSpec = addMethodGenerics(method, genericsSpec);
226                 extractSuperClassGenerics(node, declaringClass, genericsSpec);
227                 Parameter[] methodParameters = method.getParameters();
228                 Parameter[] newParams = new Parameter[methodParameters.length+1];
229                 for (int j = 1; j < newParams.length; j++) {
230                     Parameter orig = methodParameters[j-1];
231                     newParams[j] = new Parameter(
232                             correctToGenericsSpecRecurse(genericsSpec, orig.getOriginType(), methodSpecificGenerics),
233                             orig.getName()
234                     );
235                 }
236                 newParams[0] = new Parameter(node.getPlainNodeReference(), "$that");
237                 Expression arguments;
238                 if (method.getParameters()==null || method.getParameters().length==0) {
239                     arguments = ArgumentListExpression.EMPTY_ARGUMENTS;
240                 } else {
241                     List<Expression> args = new LinkedList<Expression>();
242                     for (Parameter parameter : methodParameters) {
243                         args.add(new VariableExpression(parameter));
244                     }
245                     arguments = new ArgumentListExpression(args);
246                 }
247                 Expression receiver = method.isStatic()?new ClassExpression(node):new VariableExpression(newParams[0]);
248                 MethodCallExpression mce = new MethodCallExpression(receiver, method.getName(), arguments);
249                 mce.setMethodTarget(method);
250 
251                 ExpressionStatement returnStatement = new ExpressionStatement(mce);
252                 MethodNode bridge = node.addMethod(
253                         "access$"+i, access,
254                         correctToGenericsSpecRecurse(genericsSpec, method.getReturnType(), methodSpecificGenerics),
255                         newParams,
256                         method.getExceptions(),
257                         returnStatement);
258                 GenericsType[] origGenericsTypes = method.getGenericsTypes();
259                 if (origGenericsTypes !=null) {
260                     bridge.setGenericsTypes(applyGenericsContextToPlaceHolders(genericsSpec,origGenericsTypes));
261                 }
262                 privateBridgeMethods.put(method, bridge);
263                 bridge.addAnnotation(new AnnotationNode(COMPILESTATIC_CLASSNODE));
264             }
265         }
266         if (!privateBridgeMethods.isEmpty()) {
267             node.setNodeMetaData(PRIVATE_BRIDGE_METHODS, privateBridgeMethods);
268         }
269     }

271     private static List<String> methodSpecificGenerics(final MethodNode method) {
272         List<String> genericTypeTokens = new ArrayList<String>();
273         GenericsType[] candidateGenericsTypes = method.getGenericsTypes();
274         if (candidateGenericsTypes != null) {
275             for (GenericsType gt : candidateGenericsTypes) {
276                 genericTypeTokens.add(gt.getName());
277             }
278         }
279         return genericTypeTokens;
280     }

282     private void memorizeInitialExpressions(final MethodNode node) {
283         // add node metadata for default parameters because they are erased by the Verifier
284         if (node.getParameters()!=null) {
285             for (Parameter parameter : node.getParameters()) {
286                 parameter.putNodeMetaData(StaticTypesMarker.INITIAL_EXPRESSION, parameter.getInitialExpression());
287             }
288         }
289     }

295     @Override
296     public void visitMethodCallExpression(final MethodCallExpression call) {
297         super.visitMethodCallExpression(call);
298 
299         MethodNode target = (MethodNode) call.getNodeMetaData(DIRECT_METHOD_CALL_TARGET);
300         if (target!=null) {
301             call.setMethodTarget(target);
302             memorizeInitialExpressions(target);
303         }
304 
305         if (call.getMethodTarget()==null && call.getLineNumber()>0) {
306             addError("Target method for method call expression hasn't been set", call);
307         }
308 
309     }

311     @Override
312     public void visitConstructorCallExpression(final ConstructorCallExpression call) {
313         super.visitConstructorCallExpression(call);
314 
315         MethodNode target = (MethodNode) call.getNodeMetaData(DIRECT_METHOD_CALL_TARGET);
316         if (target==null && call.getLineNumber()>0) {
317             addError("Target constructor for constructor call expression hasn't been set", call);
318         } else {
319             if (target==null) {
320                 // try to find a target
321                 ArgumentListExpression argumentListExpression = InvocationWriter.makeArgumentList(call.getArguments());
322                 List<Expression> expressions = argumentListExpression.getExpressions();
323                 ClassNode[] args = new ClassNode[expressions.size()];
324                 for (int i = 0; i < args.length; i++) {
325                     args[i] = typeChooser.resolveType(expressions.get(i), classNode);
326                 }
327                 MethodNode constructor = findMethodOrFail(call, call.isSuperCall() ? classNode.getSuperClass() : classNode, "<init>", args);
328                 call.putNodeMetaData(DIRECT_METHOD_CALL_TARGET, constructor);
329                 target = constructor;
330             }
331         }
332         if (target!=null) {
333             memorizeInitialExpressions(target);
334         }
335     }

337     @Override
338     public void visitForLoop(final ForStatement forLoop) {
339         super.visitForLoop(forLoop);
340         Expression collectionExpression = forLoop.getCollectionExpression();
341         if (!(collectionExpression instanceof ClosureListExpression)) {
342             final ClassNode collectionType = getType(forLoop.getCollectionExpression());
343             ClassNode componentType = inferLoopElementType(collectionType);
344             forLoop.getVariable().setType(componentType);
345             forLoop.getVariable().setOriginType(componentType);
346         }
347     }

349     @Override
350     protected MethodNode findMethodOrFail(final Expression expr, final ClassNode receiver, final String name, final ClassNode... args) {
351         MethodNode methodNode = super.findMethodOrFail(expr, receiver, name, args);
352         if (expr instanceof BinaryExpression && methodNode!=null) {
353             expr.putNodeMetaData(BINARY_EXP_TARGET, new Object[] {methodNode, name});
354         }
355         return methodNode;
356     }

358     @Override
359     protected boolean existsProperty(final PropertyExpression pexp, final boolean checkForReadOnly, final ClassCodeVisitorSupport visitor) {
360         Expression objectExpression = pexp.getObjectExpression();
361         ClassNode objectExpressionType = getType(objectExpression);
362         final Reference<ClassNode> rType = new Reference<ClassNode>(objectExpressionType);
363         ClassCodeVisitorSupport receiverMemoizer = new ClassCodeVisitorSupport() {
364             @Override
365             protected SourceUnit getSourceUnit() {
366                 return null;
367             }
368 
369             public void visitField(final FieldNode node) {
370                 if (visitor!=null) visitor.visitField(node);
371                 ClassNode declaringClass = node.getDeclaringClass();
372                 if (declaringClass!=null) {
373                     if (StaticTypeCheckingSupport.implementsInterfaceOrIsSubclassOf(declaringClass, ClassHelper.LIST_TYPE)) {
374                         boolean spread = declaringClass.getDeclaredField(node.getName()) != node;
375                         pexp.setSpreadSafe(spread);
376                     }
377                     rType.set(declaringClass);
378                 }
379             }
380 
381             public void visitMethod(final MethodNode node) {
382                 if (visitor!=null) visitor.visitMethod(node);
383                 ClassNode declaringClass = node.getDeclaringClass();
384                 if (declaringClass!=null){
385                     if (StaticTypeCheckingSupport.implementsInterfaceOrIsSubclassOf(declaringClass, ClassHelper.LIST_TYPE)) {
386                         List<MethodNode> properties = declaringClass.getDeclaredMethods(node.getName());
387                         boolean spread = true;
388                         for (MethodNode mn : properties) {
389                             if (node==mn) {
390                                 spread = false;
391                                 break;
392                             }
393                         }
394                         // it's no real property but a property of the component
395                         pexp.setSpreadSafe(spread);
396                     }
397                     rType.set(declaringClass);
398                 }
399             }
400 
401             @Override
402             public void visitProperty(final PropertyNode node) {
403                 if (visitor!=null) visitor.visitProperty(node);
404                 ClassNode declaringClass = node.getDeclaringClass();
405                 if (declaringClass!=null) {
406                     if (StaticTypeCheckingSupport.implementsInterfaceOrIsSubclassOf(declaringClass, ClassHelper.LIST_TYPE)) {
407                         List<PropertyNode> properties = declaringClass.getProperties();
408                         boolean spread = true;
409                         for (PropertyNode propertyNode : properties) {
410                             if (propertyNode==node) {
411                                 spread = false;
412                                 break;
413                             }
414                         }
415                         // it's no real property but a property of the component
416                         pexp.setSpreadSafe(spread);
417                     }
418                     rType.set(declaringClass);
419                 }
420             }
421         };
422         boolean exists = super.existsProperty(pexp, checkForReadOnly, receiverMemoizer);
423         if (exists) {
424             if (objectExpression.getNodeMetaData(StaticCompilationMetadataKeys.PROPERTY_OWNER)==null) {
425                 objectExpression.putNodeMetaData(StaticCompilationMetadataKeys.PROPERTY_OWNER, rType.get());
426             }
427             if (StaticTypeCheckingSupport.implementsInterfaceOrIsSubclassOf(objectExpressionType, ClassHelper.LIST_TYPE)) {
428                 objectExpression.putNodeMetaData(COMPONENT_TYPE, inferComponentType(objectExpressionType, ClassHelper.int_TYPE));
429             }
430         }
431         return exists;
432     }

434     @Override
435     public void visitPropertyExpression(final PropertyExpression pexp) {
436         super.visitPropertyExpression(pexp);
437         Object dynamic = pexp.getNodeMetaData(StaticTypesMarker.DYNAMIC_RESOLUTION);
438         if (dynamic !=null) {
439             pexp.getObjectExpression().putNodeMetaData(StaticCompilationMetadataKeys.RECEIVER_OF_DYNAMIC_PROPERTY, dynamic);
440         }
