19  package org.codehaus.groovy.ast.expr;
20  
21  import java.util.ArrayList;
22  import java.util.List;
23  
24  import org.codehaus.groovy.ast.ClassHelper;
25  import org.codehaus.groovy.ast.ClassNode;
26  import org.codehaus.groovy.ast.AnnotatedNode;
27  import org.codehaus.groovy.GroovyBugError;
28  
29  /**
30   * Represents a base class for expressions which evaluate as an object
31   * 
32   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>
33   */
34  public abstract class Expression extends AnnotatedNode {
35  
36      private ClassNode type=ClassHelper.DYNAMIC_TYPE;
37      
38      /**
39       * Return a copy of the expression calling the transformer on any nested expressions 
40       * @param transformer
41       */
42      public abstract Expression transformExpression(ExpressionTransformer transformer);
43  
44      /**
45       * Transforms the list of expressions
46       * @return a new list of transformed expressions
47       */
48      protected List<Expression> transformExpressions(List<? extends Expression> expressions, ExpressionTransformer transformer) {
49          List<Expression> list = new ArrayList<Expression>(expressions.size());
50          for (Expression expr : expressions ) {
51              list.add(transformer.transform(expr));
52          }
53          return list;
54      }
55  
56      /**
57       * Transforms the list of expressions, and checks that all transformed expressions have the given type.
58       *
59       * @return a new list of transformed expressions
60       */
61      protected <T extends Expression> List<T> transformExpressions(List<? extends Expression> expressions,
62              ExpressionTransformer transformer, Class<T> transformedType) {
63          List<T> list = new ArrayList<T>(expressions.size());
64          for (Expression expr : expressions) {
65              Expression transformed = transformer.transform(expr);
66              if (!transformedType.isInstance(transformed))
67                  throw new GroovyBugError(String.format("Transformed expression should have type %s but has type %s",
68                      transformedType, transformed.getClass()));
69              list.add(transformedType.cast(transformed));
70          }
71          return list;
72      }
73      
74      public ClassNode getType() {
75          return type;
76      }
77      
78      public void setType(ClassNode t) {
79          type=t;
80      }
81  }
