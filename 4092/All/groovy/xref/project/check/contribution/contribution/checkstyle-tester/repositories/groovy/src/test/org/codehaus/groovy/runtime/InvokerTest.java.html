34  public class InvokerTest extends GroovyTestCase {
35  
36      public void testAsCollectionWithArray() {
37          Object[] array = {"A", "B", "C"};
38          assertAsCollection(array, 3);
39      }
40  
41      public void testAsCollectionWithMap() {
42          Map map = new HashMap();
43          map.put("A", "abc");
44          map.put("B", "def");
45          map.put("C", "xyz");
46          assertAsCollection(map, 3);
47      }
48  
49      public void testAsCollectionWithList() {
50          List list = new ArrayList();
51          list.add("A");
52          list.add("B");
53          list.add("C");
54          assertAsCollection(list, 3);
55      }
56  
57      public void testInvokerException() throws Throwable {
58          try {
59              throw new GroovyRuntimeException("message", new NullPointerException());
60          }
61          catch (GroovyRuntimeException e) {
62              // worked
63              assertEquals("message", e.getMessage());
64              assertTrue(e.getCause() instanceof NullPointerException);
65          }
66      }
67  
68      public void testAsBoolean() {
69          assertAsBoolean(true, Boolean.TRUE);
70          assertAsBoolean(true, "true");
71          assertAsBoolean(true, "TRUE");
72          assertAsBoolean(true, "false");
73          assertAsBoolean(false, Boolean.FALSE);
74          assertAsBoolean(false, (String) null);
75          assertAsBoolean(false, "");
76          GString emptyGString = new GString(new Object[]{""}) {
77              public String[] getStrings() {
78                  return new String[]{""};
79              }
80          };
81          assertAsBoolean(false, emptyGString);
82          GString nonEmptyGString = new GString(new Object[]{"x"}) {
83              public String[] getStrings() {
84                  return new String[]{"x"};
85              }
86          };
87          assertAsBoolean(true, nonEmptyGString);
88          assertAsBoolean(true, new Integer(1234));
89          assertAsBoolean(false, new Integer(0));
90          assertAsBoolean(true, new Float(0.3f));
91          assertAsBoolean(true, new Double(3.0f));
92          assertAsBoolean(false, new Float(0.0f));
93          assertAsBoolean(true, new Character((char) 1));
94          assertAsBoolean(false, new Character((char) 0));
95          assertAsBoolean(false, Collections.EMPTY_LIST);
96          assertAsBoolean(true, Arrays.asList(new Integer[]{new Integer(1)}));
97      }
98  
99      public void testLessThan() {
100         assertTrue(ScriptBytecodeAdapter.compareLessThan(new Integer(1), new Integer(2)));
101         assertTrue(ScriptBytecodeAdapter.compareLessThanEqual(new Integer(2), new Integer(2)));
102     }
103 
104     public void testGreaterThan() {
105         assertTrue(ScriptBytecodeAdapter.compareGreaterThan(new Integer(3), new Integer(2)));
106         assertTrue(ScriptBytecodeAdapter.compareGreaterThanEqual(new Integer(2), new Integer(2)));
107     }
108 
109     public void testCompareTo() {
110         assertTrue(DefaultTypeTransformation.compareEqual("x", new Integer('x')));
111     }

113     // Implementation methods
114     //-------------------------------------------------------------------------
115 
116     /**
117      * Asserts the asBoolean method returns the given flag
118      */
119     protected void assertAsBoolean(boolean expected, Object value) {
120         boolean answer = DefaultTypeTransformation.castToBoolean(value);
121         assertEquals("value: " + value + " asBoolean()", expected, answer);
122     }
123 
124     /**
125      * Asserts that the given object can be converted into a collection and iterator
126      * of the given size
127      */
128     protected void assertAsCollection(Object collectionObject, int count) {
129         Collection collection = DefaultTypeTransformation.asCollection(collectionObject);
130         assertTrue("Collection is not null", collection != null);
131         assertEquals("Collection size", count, collection.size());
132 
133         assertIterator("collections iterator", collection.iterator(), count);
134         assertIterator("InvokerHelper.asIterator", InvokerHelper.asIterator(collectionObject), count);
135         assertIterator("InvokerHelper.asIterator(InvokerHelper.asCollection)", InvokerHelper.asIterator(collection), count);
136         assertIterator("InvokerHelper.asIterator(InvokerHelper.asIterator)", InvokerHelper.asIterator(InvokerHelper.asIterator(collectionObject)), count);
137     }
138 
139     /**
140      * Asserts that the iterator is valid and of the right size
141      */
142     protected void assertIterator(String message, Iterator iterator, int count) {
143         for (int i = 0; i < count; i++) {
144             assertTrue(message + ": should have item: " + i, iterator.hasNext());
145             assertTrue(message + ": item: " + i + " should not be null", iterator.next() != null);
146         }
147 
148         assertFalse(
149                 message + ": should not have item after iterating through: " + count + " items",
150                 iterator.hasNext());
151     }
