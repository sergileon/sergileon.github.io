65  /**
66   * Handles generation of code for the <code>@Delegate</code> annotation
67   *
68   * @author Alex Tkachman
69   * @author Guillaume Laforge
70   * @author Paul King
71   * @author Andre Steingress
72   */
73  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
74  public class DelegateASTTransformation extends AbstractASTTransformation {
75  
76      private static final Class MY_CLASS = Delegate.class;
77      private static final ClassNode MY_TYPE = make(MY_CLASS);
78      private static final String MY_TYPE_NAME = "@" + MY_TYPE.getNameWithoutPackage();
79      private static final ClassNode DEPRECATED_TYPE = make(Deprecated.class);
80      private static final ClassNode GROOVYOBJECT_TYPE = make(GroovyObject.class);
81      private static final ClassNode LAZY_TYPE = make(Lazy.class);
82  
83      private static final String MEMBER_DEPRECATED = "deprecated";
84      private static final String MEMBER_INTERFACES = "interfaces";
85      private static final String MEMBER_INCLUDES = "includes";
86      private static final String MEMBER_EXCLUDES = "excludes";
87      private static final String MEMBER_INCLUDE_TYPES = "includeTypes";
88      private static final String MEMBER_EXCLUDE_TYPES = "excludeTypes";
89      private static final String MEMBER_PARAMETER_ANNOTATIONS = "parameterAnnotations";
90      private static final String MEMBER_METHOD_ANNOTATIONS = "methodAnnotations";
91  
92      public void visit(ASTNode[] nodes, SourceUnit source) {
93          init(nodes, source);
94  
95          AnnotatedNode parent = (AnnotatedNode) nodes[1];
96          AnnotationNode node = (AnnotationNode) nodes[0];
97  
98          if (parent instanceof FieldNode) {
99              FieldNode fieldNode = (FieldNode) parent;
100             final ClassNode type = fieldNode.getType();
101             final ClassNode owner = fieldNode.getOwner();
102             if (type.equals(ClassHelper.OBJECT_TYPE) || type.equals(GROOVYOBJECT_TYPE)) {
103                 addError(MY_TYPE_NAME + " field '" + fieldNode.getName() + "' has an inappropriate type: " + type.getName() +
104                         ". Please add an explicit type but not java.lang.Object or groovy.lang.GroovyObject.", parent);
105                 return;
106             }
107             if (type.equals(owner)) {
108                 addError(MY_TYPE_NAME + " field '" + fieldNode.getName() + "' has an inappropriate type: " + type.getName() +
109                         ". Delegation to own type not supported. Please use a different type.", parent);
110                 return;
111             }
112             final List<MethodNode> fieldMethods = getAllMethods(type);
113             for (ClassNode next : type.getAllInterfaces()) {
114                 fieldMethods.addAll(getAllMethods(next));
115             }
116 
117             final boolean skipInterfaces = memberHasValue(node, MEMBER_INTERFACES, false);
118             final boolean includeDeprecated = memberHasValue(node, MEMBER_DEPRECATED, true) || (type.isInterface() && !skipInterfaces);
119             List<String> excludes = getMemberList(node, MEMBER_EXCLUDES);
120             List<String> includes = getMemberList(node, MEMBER_INCLUDES);
121             List<ClassNode> excludeTypes = getClassList(node, MEMBER_EXCLUDE_TYPES);
122             List<ClassNode> includeTypes = getClassList(node, MEMBER_INCLUDE_TYPES);
123             checkIncludeExclude(node, excludes, includes, excludeTypes, includeTypes, MY_TYPE_NAME);
124 
125             final List<MethodNode> ownerMethods = getAllMethods(owner);
126             for (MethodNode mn : fieldMethods) {
127                 addDelegateMethod(node, fieldNode, owner, ownerMethods, mn, includeDeprecated, includes, excludes, includeTypes, excludeTypes);
128             }
129 
130             for (PropertyNode prop : getAllProperties(type)) {
131                 if (prop.isStatic() || !prop.isPublic())
132                     continue;
133                 String name = prop.getName();
134                 addGetterIfNeeded(fieldNode, owner, prop, name, includes, excludes);
135                 addSetterIfNeeded(fieldNode, owner, prop, name, includes, excludes);
136             }
137 
138             if (skipInterfaces) return;
139 
140             final Set<ClassNode> allInterfaces = getInterfacesAndSuperInterfaces(type);
141             final Set<ClassNode> ownerIfaces = owner.getAllInterfaces();
142             Map<String,ClassNode> genericsSpec = createGenericsSpec(fieldNode.getDeclaringClass());
143             genericsSpec = createGenericsSpec(fieldNode.getType(), genericsSpec);
144             for (ClassNode iface : allInterfaces) {
145                 if (Modifier.isPublic(iface.getModifiers()) && !ownerIfaces.contains(iface)) {
146                     final ClassNode[] ifaces = owner.getInterfaces();
147                     final ClassNode[] newIfaces = new ClassNode[ifaces.length + 1];
148                     for (int i = 0; i < ifaces.length; i++) {
149                         newIfaces[i] = correctToGenericsSpecRecurse(genericsSpec, ifaces[i]);
150                     }
151                     newIfaces[ifaces.length] = correctToGenericsSpecRecurse(genericsSpec, iface);
152                     owner.setInterfaces(newIfaces);
153                 }
154             }
155         }
156     }
157 
158     private void addSetterIfNeeded(FieldNode fieldNode, ClassNode owner, PropertyNode prop, String name, List<String> includes, List<String> excludes) {
159         String setterName = "set" + Verifier.capitalize(name);
160         if ((prop.getModifiers() & ACC_FINAL) == 0
161                 && owner.getSetterMethod(setterName) == null
162                 && !shouldSkipPropertyMethod(name, setterName, excludes, includes)) {
163             owner.addMethod(setterName,
164                     ACC_PUBLIC,
165                     ClassHelper.VOID_TYPE,
166                     params(new Parameter(GenericsUtils.nonGeneric(prop.getType()), "value")),
167                     null,
168                     assignS(propX(varX(fieldNode), name), varX("value"))
169             );
170         }
171     }
172 
173     private void addGetterIfNeeded(FieldNode fieldNode, ClassNode owner, PropertyNode prop, String name, List<String> includes, List<String> excludes) {
174         String getterName = "get" + Verifier.capitalize(name);
175         if (owner.getGetterMethod(getterName) == null
176                 && !shouldSkipPropertyMethod(name, getterName, excludes, includes)) {
177             owner.addMethod(getterName,
178                     ACC_PUBLIC,
179                     GenericsUtils.nonGeneric(prop.getType()),
180                     Parameter.EMPTY_ARRAY,
181                     null,
182                     returnS(propX(varX(fieldNode), name)));
183         }
184     }
185     
186     private boolean shouldSkipPropertyMethod(String propertyName, String methodName, List<String> excludes, List<String> includes) {
187         return (deemedInternalName(propertyName)
188                     || excludes != null && (excludes.contains(propertyName) || excludes.contains(methodName)) 
189                     || (includes != null && !includes.isEmpty() && !includes.contains(propertyName) && !includes.contains(methodName)));
190     }
191 
192     private void addDelegateMethod(AnnotationNode node, FieldNode fieldNode, ClassNode owner, List<MethodNode> ownMethods, MethodNode candidate, boolean includeDeprecated, List<String> includes, List<String> excludes, List<ClassNode> includeTypes, List<ClassNode> excludeTypes) {
193         if (!candidate.isPublic() || candidate.isStatic() || 0 != (candidate.getModifiers () & ACC_SYNTHETIC))
194             return;
195 
196         if (!candidate.getAnnotations(DEPRECATED_TYPE).isEmpty() && !includeDeprecated)
197             return;
198 
199         if (shouldSkip(candidate.getName(), excludes, includes)) return;
200 
201         Map<String,ClassNode> genericsSpec = createGenericsSpec(fieldNode.getDeclaringClass());
202         genericsSpec = addMethodGenerics(candidate, genericsSpec);
203         extractSuperClassGenerics(fieldNode.getType(), candidate.getDeclaringClass(), genericsSpec);
204 
205         if (!excludeTypes.isEmpty() || !includeTypes.isEmpty()) {
206             MethodNode correctedMethodNode = correctToGenericsSpec(genericsSpec, candidate);
207             boolean checkReturn = fieldNode.getType().getMethods().contains(candidate);
208             if (shouldSkipOnDescriptor(checkReturn, genericsSpec, correctedMethodNode, excludeTypes, includeTypes)) return;
209         }
210 
211         // ignore methods from GroovyObject
212         for (MethodNode mn : GROOVYOBJECT_TYPE.getMethods()) {
213             if (mn.getTypeDescriptor().equals(candidate.getTypeDescriptor())) {
214                 return;
215             }
216         }
217 
218         // ignore methods already in owner
219         for (MethodNode mn : owner.getMethods()) {
220             if (mn.getTypeDescriptor().equals(candidate.getTypeDescriptor())) {
221                 return;
222             }
223         }
224 
225         // give precedence to methods of self (but not abstract or static superclass methods)
226         // also allows abstract or static self methods to be selected for overriding but they are ignored later
227         MethodNode existingNode = null;
228         for (MethodNode mn : ownMethods) {
229             if (mn.getTypeDescriptor().equals(candidate.getTypeDescriptor()) && !mn.isAbstract() && !mn.isStatic()) {
230                 existingNode = mn;
231                 break;
232             }
233         }
234         if (existingNode == null || existingNode.getCode() == null) {
235 
236             final ArgumentListExpression args = new ArgumentListExpression();
237             final Parameter[] params = candidate.getParameters();
238             final Parameter[] newParams = new Parameter[params.length];
239             List<String> currentMethodGenPlaceholders = genericPlaceholderNames(candidate);
240             for (int i = 0; i < newParams.length; i++) {
241                 ClassNode newParamType = correctToGenericsSpecRecurse(genericsSpec, params[i].getType(), currentMethodGenPlaceholders);
242                 Parameter newParam = new Parameter(newParamType, getParamName(params, i, fieldNode.getName()));
243                 newParam.setInitialExpression(params[i].getInitialExpression());
244 
245                 if (memberHasValue(node, MEMBER_PARAMETER_ANNOTATIONS, true)) {
246                     newParam.addAnnotations(copyAnnotatedNodeAnnotations(params[i], MY_TYPE_NAME));
247                 }
248 
249                 newParams[i] = newParam;
250                 args.addExpression(varX(newParam));
251             }
252             boolean alsoLazy = !fieldNode.getAnnotations(LAZY_TYPE).isEmpty();
253             // addMethod will ignore attempts to override abstract or static methods with same signature on self
254             MethodCallExpression mce = callX(
255                     alsoLazy ? propX(varX("this"), fieldNode.getName().substring(1)) :
256                     varX(fieldNode.getName(), correctToGenericsSpecRecurse(genericsSpec, fieldNode.getType())),
257                     candidate.getName(),
258                     args);
259             mce.setSourcePosition(fieldNode);
260             ClassNode returnType = correctToGenericsSpecRecurse(genericsSpec, candidate.getReturnType(), currentMethodGenPlaceholders);
261             MethodNode newMethod = owner.addMethod(candidate.getName(),
262                     candidate.getModifiers() & (~ACC_ABSTRACT) & (~ACC_NATIVE),
263                     returnType,
264                     newParams,
265                     candidate.getExceptions(),
266                     stmt(mce));
267             newMethod.setGenericsTypes(candidate.getGenericsTypes());
268 
269             if (memberHasValue(node, MEMBER_METHOD_ANNOTATIONS, true)) {
270                 newMethod.addAnnotations(copyAnnotatedNodeAnnotations(candidate, MY_TYPE_NAME));
271             }
272         }
273     }
274 
275     private List<String> genericPlaceholderNames(MethodNode candidate) {
276         GenericsType[] candidateGenericsTypes = candidate.getGenericsTypes();
277         List<String> names = new ArrayList<String>();
278         if (candidateGenericsTypes != null) {
279             for (GenericsType gt : candidateGenericsTypes) {
280                 names.add(gt.getName());
281             }
282         }
283         return names;
284     }
285 
286     private String getParamName(Parameter[] params, int i, String fieldName) {
287         String name = params[i].getName();
288         while(name.equals(fieldName) || clashesWithOtherParams(name, params, i)) {
289             name = "_" + name;
290         }
291         return name;
292     }
293 
294     private boolean clashesWithOtherParams(String name, Parameter[] params, int i) {
295         for (int j = 0; j < params.length; j++) {
296             if (i == j) continue;
297             if (params[j].getName().equals(name)) return true;
298         }
299         return false;
300     }
