40  /**
41   * A base class dealing with common HTTP servlet API housekeeping aspects.
42   * <p>
43   * <h4>Resource name mangling (pattern replacement)</h4>
44   * <p>
45   * Also implements Groovy's {@link groovy.util.ResourceConnector} in a dynamic
46   * manner. It allows you to modify the resource name that is searched for with a
47   * <i>replace all</i> operation. See {@link java.util.regex.Pattern} and
48   * {@link java.util.regex.Matcher} for details.
49   * The servlet init parameter names are:
50   * <pre>
51   * {@value #INIT_PARAM_RESOURCE_NAME_REGEX} = empty - defaults to null
52   * resource.name.replacement = empty - defaults to null
53   * resource.name.replace.all = true (default) | false means replaceFirst()
54   * </pre>
55   * Note: If you specify a regex, you have to specify a replacement string too!
56   * Otherwise an exception gets raised.
57   * <p>
58   * <h4>Logging and bug-hunting options</h4>
59   * <p>
60   * This implementation provides a verbosity flag switching log statements.
61   * The servlet init parameter name is:
62   * <pre>
63   * verbose = false(default) | true
64   * </pre>
65   * <p>
66   * In order to support class-loading-troubles-debugging with Tomcat 4 or
67   * higher, you can log the class loader responsible for loading some classes.
68   * See <a href="https://issues.apache.org/jira/browse/GROOVY-861">GROOVY-861</a> for details.
69   * The servlet init parameter name is:
70   * <pre>
71   * log.GROOVY861 = false(default) | true
72   * </pre>
73   * <p>
74   * If you experience class-loading-troubles with Tomcat 4 (or higher) or any
75   * other servlet container using custom class loader setups, you can fallback
76   * to use (slower) reflection in Groovy's MetaClass implementation. Please
77   * contact the dev team with your problem! Thanks.
78   * The servlet init parameter name is:
79   * <pre>
80   * reflection = false(default) | true
81   * </pre>
82   *
83   * @author Christian Stein
84   * @author Roshan Dawrani (roshandawrani)
85   */

173     protected void generateNamePrefixOnce () {
174         URI uri = null;
175 
176         String realPath = servletContext.getRealPath("/");
177         if (realPath != null) { uri = new File(realPath).toURI();}//prevent NPE if in .war
178 
179         try {
180             URL res = servletContext.getResource("/");
181             if (res != null) { uri = res.toURI(); }
182         } catch (MalformedURLException ignore) {
183         } catch (URISyntaxException ignore) {
184         }
185 
186         if (uri != null) {
187             try {
188                 namePrefix = uri.toURL().toExternalForm();
189                 return;
190             } catch (MalformedURLException e) {
191                 log("generateNamePrefixOnce [ERROR] Malformed URL for base path / == '"+ uri +'\'', e);
192             }
193         }
194 
195         namePrefix = "";
196     }
197 
198     protected String removeNamePrefix(String name) throws ResourceException {
199         if (namePrefix == null) {
200             generateNamePrefixOnce();
201         }
202         if (name.startsWith(namePrefix)) {//usualy name has text
203             return name.substring(namePrefix.length());
204         }
205         return name;
206     }

208     /**
209      * Interface method for ResourceContainer. This is used by the GroovyScriptEngine.
210      */
211     public URLConnection getResourceConnection (String name) throws ResourceException {
212         name = removeNamePrefix(name).replace('\\', '/');
213 
214         //remove the leading / as we are trying with a leading / now
215         if (name.startsWith("WEB-INF/groovy/")) {
216             name = name.substring(15);//just for uniformity
217         } else if (name.startsWith("/")) {
218             name = name.substring(1);
219         }
220 
221         /*
222         * Try to locate the resource and return an opened connection to it.
223         */
224         try {
225             URL url = servletContext.getResource('/' + name);
226             if (url == null) {
227                 url = servletContext.getResource("/WEB-INF/groovy/" + name);
228             }
229             if (url == null) {
230                 throw new ResourceException("Resource \"" + name + "\" not found!");
231             }
232             return url.openConnection();
233         } catch (IOException e) {
234             throw new ResourceException("Problems getting resource named \"" + name + "\"!", e);
235         }
236     }

238     /**
239      * Returns the include-aware uri of the script or template file.
240      *
241      * @param request the http request to analyze
242      * @return the include-aware uri either parsed from request attributes or
243      *         hints provided by the servlet container
244      */
245     protected String getScriptUri(HttpServletRequest request) {
246         /*
247          * Log some debug information for https://issues.apache.org/jira/browse/GROOVY-861
248          */
249         if (logGROOVY861) {
250             log("Logging request class and its class loader:");
251             log(" c = request.getClass() :\"" + request.getClass() + "\"");
252             log(" l = c.getClassLoader() :\"" + request.getClass().getClassLoader() + "\"");
253             log(" l.getClass()           :\"" + request.getClass().getClassLoader().getClass() + "\"");
254             /*
255              * Keep logging, if we're verbose. Else turn it off.
256              */
257             logGROOVY861 = verbose;
258         }
259 
260         //
261         // NOTE: This piece of code is heavily inspired by Apaches Jasper2!
262         // 
263         // http://cvs.apache.org/viewcvs.cgi/jakarta-tomcat-jasper/jasper2/ \
264         //        src/share/org/apache/jasper/servlet/JspServlet.java?view=markup
265         //
266         // Why doesn't it use request.getRequestURI() or INC_REQUEST_URI?
267         //
268 
269         String uri = null;
270         String info = null;
271 
272         //
273         // Check to see if the requested script/template source file has been the
274         // target of a RequestDispatcher.include().
275         //
276         uri = (String) request.getAttribute(INC_SERVLET_PATH);
277         if (uri != null) {
278             //
279             // Requested script/template file has been target of 
280             // RequestDispatcher.include(). Its path is assembled from the relevant
281             // javax.servlet.include.* request attributes and returned!
282             //
283             info = (String) request.getAttribute(INC_PATH_INFO);
284             if (info != null) {
285                 uri += info;
286             }
287             return applyResourceNameMatcher(uri);
288         }
289 
290         //
291         // Requested script/template file has not been the target of a 
292         // RequestDispatcher.include(). Reconstruct its path from the request's
293         // getServletPath() and getPathInfo() results.
294         //
295         uri = request.getServletPath();
296         info = request.getPathInfo();
297         if (info != null) {
298             uri += info;
299         }
300 
301         /*
302          * TODO : Enable auto ".groovy" extension replacing here!
303          * http://cvs.groovy.codehaus.org/viewrep/groovy/groovy/groovy-core/src/main/groovy/servlet/GroovyServlet.java?r=1.10#l259
304          */
305 
306         return applyResourceNameMatcher(uri);
307     }
308 
309     protected String applyResourceNameMatcher (String uri) {
310         if (resourceNamePattern != null) {// mangle resource name with the compiled pattern.
311             Matcher matcher = resourceNamePattern.matcher(uri);
312 
313             String replaced;
314             if (resourceNameReplaceAll) {
315                 replaced = matcher.replaceAll(resourceNameReplacement);
316             } else {
317                 replaced = matcher.replaceFirst(resourceNameReplacement);
318             }
319             if (!uri.equals(replaced)) {
320                 if (verbose) {
321                     log("Replaced resource name \"" + uri + "\" with \"" + replaced + "\".");
322                 }
323                 return replaced;
324             }
325         }
326         return uri;
327     }

329     /**
330      * Parses the http request for the real script or template source file.
331      * 
332      * @param request
333      *            the http request to analyze
334      * @return a file object using an absolute file path name, or <code>null</code> if the
335      *         servlet container cannot translate the virtual path to a real
336      *         path for any reason (such as when the content is being made
337      *         available from a .war archive).
338      */
339     protected File getScriptUriAsFile(HttpServletRequest request) {
340         String uri = getScriptUri(request);
341         String real = servletContext.getRealPath(uri);
342         if (real == null) {
343             return null;
344         }
345         return new File(real).getAbsoluteFile();
346     }

348     /**
349      * Overrides the generic init method to set some debug flags.
350      *
351      * @param config the servlet configuration provided by the container
352      * @throws ServletException if init() method defined in super class
353      *                          javax.servlet.GenericServlet throws it
354      */
355     public void init(ServletConfig config) throws ServletException {
356         /*
357          * Never forget super.init()!
358          */
359         super.init(config);
360 
361         /*
362          * Grab the servlet context.
363          */
364         this.servletContext = config.getServletContext();
365 
366         // Get verbosity hint.
367         String value = config.getInitParameter("verbose");
368         if (value != null) {
369             this.verbose = Boolean.valueOf(value);
370         }
371 
372         // get encoding
373         value = config.getInitParameter("encoding");
374         if (value != null) {
375             this.encoding = value;
376         }
377 
378         // And now the real init work...
379         if (verbose) {
380             log("Parsing init parameters...");
381         }
382 
383         String regex = config.getInitParameter(INIT_PARAM_RESOURCE_NAME_REGEX);
384         if (regex != null) {
385             String replacement = config.getInitParameter("resource.name.replacement");
386             if (replacement == null) {
387                 Exception npex = new NullPointerException("resource.name.replacement");
388                 String message = "Init-param 'resource.name.replacement' not specified!";
389                 log(message, npex);
390                 throw new ServletException(message, npex);
391             } else if ("EMPTY_STRING".equals(replacement)) {//<param-value></param-value> is prohibited
392                 replacement = "";
393             }
394             int flags = 0; // TODO : Parse pattern compile flags (literal names).
395             String flagsStr = config.getInitParameter(INIT_PARAM_RESOURCE_NAME_REGEX_FLAGS);
396             if (flagsStr != null && flagsStr.length() > 0) {
397               flags = Integer.decode(flagsStr.trim());//throws NumberFormatException
398             }
399             resourceNamePattern = Pattern.compile(regex, flags);
400             this.resourceNameReplacement = replacement;
401             String all = config.getInitParameter("resource.name.replace.all");
402             if (all != null) {
403                 this.resourceNameReplaceAll = Boolean.valueOf(all.trim());
404             }
405         }
406 
407         value = config.getInitParameter("logGROOVY861");
408         if (value != null) {
409             this.logGROOVY861 = Boolean.valueOf(value);
410             // nothing else to do here
411         }
412 
413         /*
414          * If verbose, log the parameter values.
415          */
416         if (verbose) {
417             log("(Abstract) init done. Listing some parameter name/value pairs:");
418             log("verbose = " + verbose); // this *is* verbose! ;)
419             log("reflection = " + reflection);
420             log("logGROOVY861 = " + logGROOVY861);
421             log(INIT_PARAM_RESOURCE_NAME_REGEX + " = " + resourceNamePattern);//toString == pattern
422             log("resource.name.replacement = " + resourceNameReplacement);
423         }
424     }

426     /**
427      * Override this method to set your variables to the Groovy binding.
428      * <p>
429      * All variables bound the binding are passed to the template source text,
430      * e.g. the HTML file, when the template is merged.
431      * <p>
432      * The binding provided by TemplateServlet does already include some default
433      * variables. As of this writing, they are (copied from
434      * {@link groovy.servlet.ServletBinding}):
435      * <ul>
436      * <li><tt>"request"</tt> : HttpServletRequest </li>
437      * <li><tt>"response"</tt> : HttpServletResponse </li>
438      * <li><tt>"context"</tt> : ServletContext </li>
439      * <li><tt>"application"</tt> : ServletContext </li>
440      * <li><tt>"session"</tt> : request.getSession(<b>false</b>) </li>
441      * </ul>
442      * <p>
443      * And via implicit hard-coded keywords:
444      * <ul>
445      * <li><tt>"out"</tt> : response.getWriter() </li>
446      * <li><tt>"sout"</tt> : response.getOutputStream() </li>
447      * <li><tt>"html"</tt> : new MarkupBuilder(response.getWriter()) </li>
448      * </ul>
449      * <p>
450      * The binding also provides convenient methods:
451      * <ul>
452      * <li><tt>"forward(String path)"</tt> : request.getRequestDispatcher(path).forward(request, response);</li>
453      * <li><tt>"include(String path)"</tt> : request.getRequestDispatcher(path).include(request, response);</li>
454      * <li><tt>"redirect(String location)"</tt> : response.sendRedirect(location);</li>
455      * </ul>
456      * <p>
457      * <p>Example binding all servlet context variables:
458      * <pre><code>
459      * class MyServlet extends TemplateServlet {
460      *
461      *   protected void setVariables(ServletBinding binding) {
462      *     // Bind a simple variable
463      *     binding.setVariable("answer", new Long(42));
464      *
465      *     // Bind all servlet context attributes...
466      *     ServletContext context = (ServletContext) binding.getVariable("context");
467      *     Enumeration enumeration = context.getAttributeNames();
468      *     while (enumeration.hasMoreElements()) {
469      *       String name = (String) enumeration.nextElement();
470      *       binding.setVariable(name, context.getAttribute(name));
471      *     }
472      *   }
473      * }
474      * <code></pre>
475      *
476      * @param binding to be modified
477      */
478     protected void setVariables(ServletBinding binding) {
479         // empty
480     }
