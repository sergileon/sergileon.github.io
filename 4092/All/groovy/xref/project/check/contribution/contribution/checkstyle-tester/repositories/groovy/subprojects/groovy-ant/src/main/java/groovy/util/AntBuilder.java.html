44  /**

49   * <a href="http://ant.apache.org/manual/install#optionalTasks">optional tasks</a>
50   * you will need to add one or more additional jars from the ant distribution to
51   * your classpath - see the <a href="http://ant.apache.org/manual/install.html#librarydependencies">library
52   * dependencies</a> for more details.

54   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>

91      public AntBuilder(final Project project, final Target owningTarget) {
92          this.project = project;
93  
94          /*
95           * GROOVY-4524: The following is not needed anymore as an ant Project already by default has inputhandler
96           * set to DefaultInputHandler. And if it is again set here, it mistakenly overrides the custom input handler
97           * if set using -inputhandler switch. 
98           */
99          //this.project.setInputHandler(new DefaultInputHandler());
100 
101         collectorTarget = owningTarget;
102         antXmlContext = new AntXMLContext(project);
103         collectorTarget.setProject(project);
104         antXmlContext.setCurrentTarget(collectorTarget);
105         antXmlContext.setLocator(new AntBuilderLocator());
106         antXmlContext.setCurrentTargets(new HashMap<String, Target>());
107 
108         implicitTarget = new Target();
109         implicitTarget.setProject(project);
110         implicitTarget.setName("");
111         antXmlContext.setImplicitTarget(implicitTarget);
112 
113         // FileScanner is a Groovy utility
114         project.addDataTypeDefinition("fileScanner", FileScanner.class);
115     }

117     public AntBuilder(final Task parentTask) {
118         this(parentTask.getProject(), parentTask.getOwningTarget());
119 
120         // define "owning" task as wrapper to avoid having tasks added to the target
121         // but it needs to be an UnknownElement and no access is available from
122         // task to its original UnknownElement 
123         final UnknownElement ue = new UnknownElement(parentTask.getTaskName());
124         ue.setProject(parentTask.getProject());
125         ue.setTaskType(parentTask.getTaskType());
126         ue.setTaskName(parentTask.getTaskName());
127         ue.setLocation(parentTask.getLocation());
128         ue.setOwningTarget(parentTask.getOwningTarget());
129         ue.setRuntimeConfigurableWrapper(parentTask.getRuntimeConfigurableWrapper());
130         parentTask.getRuntimeConfigurableWrapper().setProxy(ue);
131         antXmlContext.pushWrapper(parentTask.getRuntimeConfigurableWrapper());
132     }

176      */
177     protected static Project createProject() {
178         final Project project = new Project();
179 
180         final ProjectHelper helper = ProjectHelper.getProjectHelper();
181         project.addReference(ProjectHelper.PROJECTHELPER_REFERENCE, helper);
182         helper.getImportStack().addElement("AntBuilder"); // import checks that stack is not empty 
183 
184         final BuildLogger logger = new NoBannerLogger();
185 
186         logger.setMessageOutputLevel(org.apache.tools.ant.Project.MSG_INFO);
187         logger.setOutputPrintStream(System.out);
188         logger.setErrorPrintStream(System.err);
189 
190         project.addBuildListener(logger);
191 
192         project.init();
193         project.getBaseDir();
194         return project;
195     }

200     /**
201      * We don't want to return the node as created in {@link #createNode(Object, Map, Object)}
202      * but the one made ready by {@link #nodeCompleted(Object, Object)}
203      *

205      */
206     protected Object doInvokeMethod(String methodName, Object name, Object args) {

208 
209 
210         // return the completed node
211         return lastCompletedNode;
212     }

214     /**
215      * Determines, when the ANT Task that is represented by the "node" should perform.
216      * Node must be an ANT Task or no "perform" is called.
217      * If node is an ANT Task, it performs right after complete construction.
218      * If node is nested in a TaskContainer, calling "perform" is delegated to that
219      * TaskContainer.
220      *
221      * @param parent note: null when node is root
222      * @param node   the node that now has all its children applied
223      */
224     protected void nodeCompleted(final Object parent, final Object node) {
225         if (parent == null) insideTask = false;
226         antElementHandler.onEndElement(null, null, antXmlContext);
227 
228         lastCompletedNode = node;
229         if (parent != null && !(parent instanceof Target)) {
230             log.finest("parent is not null: no perform on nodeCompleted");
231             return; // parent will care about when children perform
232         }
233         if (definingTarget != null && definingTarget == parent && node instanceof Task) return; // inside defineTarget
234         if (definingTarget == node) {
235             definingTarget = null;
236         }
237 
238         // as in Target.execute()
239         if (node instanceof Task) {
240             Task task = (Task) node;
241             final String taskName = task.getTaskName();
242 
243             if ("antcall".equals(taskName) && parent == null) {
244                 throw new BuildException("antcall not supported within AntBuilder, consider using 'ant.project.executeTarget('targetName')' instead.");
245             }
246 
247             if (saveStreams) {
248                 // save original streams
249                 synchronized (AntBuilder.class) {
250                     int currentStreamCount = streamCount++;
251                     if (currentStreamCount == 0) {
252                         // we are first, save the streams
253                         savedProjectInputStream = project.getDefaultInputStream();
254                         savedIn = System.in;
255                         savedErr = System.err;
256                         savedOut = System.out;
257 
258                         if (!(savedIn instanceof DemuxInputStream)) {
259                             project.setDefaultInputStream(savedIn);
260                             demuxInputStream = new DemuxInputStream(project);
261                             System.setIn(demuxInputStream);
262                         }
263                         demuxOutputStream = new DemuxOutputStream(project, false);
264                         System.setOut(new PrintStream(demuxOutputStream));
265                         demuxErrorStream = new DemuxOutputStream(project, true);
266                         System.setErr(new PrintStream(demuxErrorStream));
267                     }
268                 }
269             }
270 
271             try {
272                 lastCompletedNode = performTask(task);
273             } finally {
274                 if (saveStreams) {
275                     synchronized (AntBuilder.class) {
276                         int currentStreamCount = --streamCount;
277                         if (currentStreamCount == 0) {
278                             // last to leave, turn out the lights: restore original streams
279                             project.setDefaultInputStream(savedProjectInputStream);
280                             System.setOut(savedOut);
281                             System.setErr(savedErr);
282                             if (demuxInputStream != null) {
283                                 System.setIn(savedIn);
284                                 DefaultGroovyMethodsSupport.closeQuietly(demuxInputStream);
285                                 demuxInputStream = null;
286                             }
287                             DefaultGroovyMethodsSupport.closeQuietly(demuxOutputStream);
288                             DefaultGroovyMethodsSupport.closeQuietly(demuxErrorStream);
289                             demuxOutputStream = null;
290                             demuxErrorStream = null;
291                         }
292                     }
293                 }
294             }
295 
296             // restore dummy collector target
297             if ("import".equals(taskName)) {
298                 antXmlContext.setCurrentTarget(collectorTarget);
299             }
300         } else if (node instanceof Target) {
301             // restore dummy collector target
302             antXmlContext.setCurrentTarget(collectorTarget);
303         } else {
304             final RuntimeConfigurable r = (RuntimeConfigurable) node;
305             r.maybeConfigure(project);
306         }
307     }

309     // Copied from org.apache.tools.ant.Task, since we need to get a real thing before it gets nulled in DispatchUtils.execute
310 
311     private Object performTask(Task task) {
312 
313         Throwable reason = null;
314         try {
315             // Have to call fireTestStared/fireTestFinished via reflection as they unfortunately have protected access in Project
316             final Method fireTaskStarted = Project.class.getDeclaredMethod("fireTaskStarted", Task.class);
317             fireTaskStarted.setAccessible(true);
318             fireTaskStarted.invoke(project, task);
319 
320             Object realThing;
321             realThing = task;
322             task.maybeConfigure();
323             if (task instanceof UnknownElement) {
324                 realThing = ((UnknownElement) task).getRealThing();
325             }
326 
327             DispatchUtils.execute(task);
328 
329             return realThing != null ? realThing : task;
330         }
331         catch (BuildException ex) {
332             if (ex.getLocation() == Location.UNKNOWN_LOCATION) {
333                 ex.setLocation(task.getLocation());
334             }
335             reason = ex;
336             throw ex;
337         }
338         catch (Exception ex) {
339             reason = ex;
340             BuildException be = new BuildException(ex);
341             be.setLocation(task.getLocation());
342             throw be;
343         }
344         catch (Error ex) {
345             reason = ex;
346             throw ex;
347         }
348         finally {
349             try {
350                 final Method fireTaskFinished = Project.class.getDeclaredMethod("fireTaskFinished", Task.class, Throwable.class);
351                 fireTaskFinished.setAccessible(true);
352                 fireTaskFinished.invoke(project, task, reason);
353             }
354             catch (Exception e) {
355                 BuildException be = new BuildException(e);
356                 be.setLocation(task.getLocation());
357                 throw be;
358             }
359         }
360     }

378     /**
379      * Builds an {@link Attributes} from a {@link Map}
380      *
381      * @param attributes the attributes to wrap
382      * @return the wrapped attributes
383      */
384     protected static Attributes buildAttributes(final Map attributes) {
385         final AttributesImpl attr = new AttributesImpl();
386         for (Object o : attributes.entrySet()) {
387             final Map.Entry entry = (Map.Entry) o;
388             final String attributeName = (String) entry.getKey();
389             final String attributeValue = String.valueOf(entry.getValue());
390             attr.addAttribute(null, attributeName, attributeName, "CDATA", attributeValue);
391         }
392         return attr;
393     }

395     protected Object createNode(final Object name, final Map attributes) {
396 
397         final Attributes attrs = buildAttributes(attributes);
398         String tagName = name.toString();
399         String ns = "";
400 
401         if (name instanceof QName) {
402             QName q = (QName) name;
403             tagName = q.getLocalPart();
404             ns = q.getNamespaceURI();
405         }
406 
407         // import can be used only as top level element
408         if ("import".equals(name)) {
409             antXmlContext.setCurrentTarget(implicitTarget);
410         } else if ("target".equals(name) && !insideTask) {
411             return onStartTarget(attrs, tagName, ns);
412         } else if ("defineTarget".equals(name) && !insideTask) {
413             return onDefineTarget(attrs, "target", ns);
414         }
415 
416         try {
417             antElementHandler.onStartElement(ns, tagName, tagName, attrs, antXmlContext);
418         }
419         catch (final SAXParseException e) {
420             log.log(Level.SEVERE, "Caught: " + e, e);
421         }
422 
423         insideTask = true;
424         final RuntimeConfigurable wrapper = antXmlContext.getWrapperStack().lastElement();
425         return wrapper.getProxy();
426     }

428     private Target onDefineTarget(final Attributes attrs, String tagName, String ns) {
429         final Target target = new Target();
430         target.setProject(project);
431         target.setLocation(new Location(antXmlContext.getLocator()));
432         try {
433             antTargetHandler.onStartElement(ns, tagName, tagName, attrs, antXmlContext);
434             final Target newTarget = getProject().getTargets().get(attrs.getValue("name"));
435             antXmlContext.setCurrentTarget(newTarget);
436             definingTarget = newTarget;
437             return newTarget;
438         }
439         catch (final SAXParseException e) {
440             log.log(Level.SEVERE, "Caught: " + e, e);
441         }
442         return null;
443     }
444 
445     private Target onStartTarget(final Attributes attrs, String tagName, String ns) {
446         final Target target = new Target();
447         target.setProject(project);
448         target.setLocation(new Location(antXmlContext.getLocator()));
449         try {
450             antTargetHandler.onStartElement(ns, tagName, tagName, attrs, antXmlContext);
451             final Target newTarget = getProject().getTargets().get(attrs.getValue("name"));
452 
453             // execute dependencies (if any)
454             final Vector<Target> targets = new Vector<Target>();
455             for (final Enumeration<String> deps = newTarget.getDependencies(); deps.hasMoreElements();) {
456                 final String targetName = deps.nextElement();
457                 targets.add(project.getTargets().get(targetName));
458             }
459             getProject().executeSortedTargets(targets);
460 
461             antXmlContext.setCurrentTarget(newTarget);
462             return newTarget;
463         }
464         catch (final SAXParseException e) {
465             log.log(Level.SEVERE, "Caught: " + e, e);
466         }
467         return null;
468     }

470     protected void setText(Object task, String text) {
471         final char[] characters = text.toCharArray();
472         try {
473             antElementHandler.characters(characters, 0, characters.length, antXmlContext);
474         }
475         catch (final SAXParseException e) {
476             log.log(Level.WARNING, "SetText failed: " + task + ". Reason: " + e, e);
477         }
478     }
