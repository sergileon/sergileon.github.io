48  public class EnumVisitor extends ClassCodeVisitorSupport {
49      // some constants for modifiers
50      private static final int FS = Opcodes.ACC_FINAL | Opcodes.ACC_STATIC;
51      private static final int PS = Opcodes.ACC_PUBLIC | Opcodes.ACC_STATIC;
52      private static final int PUBLIC_FS = Opcodes.ACC_PUBLIC | FS;
53      private static final int PRIVATE_FS = Opcodes.ACC_PRIVATE | FS;
54  
55      private final SourceUnit sourceUnit;
56  
57  
58      public EnumVisitor(CompilationUnit cu, SourceUnit su) {
59          sourceUnit = su;
60      }
61  
62      public void visitClass(ClassNode node) {
63          if (!node.isEnum()) return;
64          completeEnum(node);
65      }
66  
67      protected SourceUnit getSourceUnit() {
68          return sourceUnit;
69      }
70  
71      private void completeEnum(ClassNode enumClass) {
72          boolean isAic = isAnonymousInnerClass(enumClass);
73          // create MIN_VALUE and MAX_VALUE fields
74          FieldNode minValue = null, maxValue = null, values = null;
75  
76          if (!isAic) {
77              ClassNode enumRef = enumClass.getPlainNodeReference();
78  
79              // create values field
80              values = new FieldNode("$VALUES", PRIVATE_FS | Opcodes.ACC_SYNTHETIC, enumRef.makeArray(), enumClass, null);
81              values.setSynthetic(true);
82  
83              addMethods(enumClass, values);
84              checkForAbstractMethods(enumClass);
85  
86              // create MIN_VALUE and MAX_VALUE fields
87              minValue = new FieldNode("MIN_VALUE", PUBLIC_FS, enumRef, enumClass, null);
88              maxValue = new FieldNode("MAX_VALUE", PUBLIC_FS, enumRef, enumClass, null);
89          }
90          addInit(enumClass, minValue, maxValue, values, isAic);
91      }
92  
93      private void checkForAbstractMethods(ClassNode enumClass) {
94          List<MethodNode> methods = enumClass.getMethods();
95          for (MethodNode m : methods) {
96              if (m.isAbstract()) {
97                  // make the class abstract also see Effective Java p.152
98                  enumClass.setModifiers(enumClass.getModifiers() | Opcodes.ACC_ABSTRACT);
99                  break;
100             }
101         }
102     }
103 
104     private void addMethods(ClassNode enumClass, FieldNode values) {
105         List<MethodNode> methods = enumClass.getMethods();
106 
107         boolean hasNext = false;
108         boolean hasPrevious = false;
109         for (MethodNode m : methods) {
110             if (m.getName().equals("next") && m.getParameters().length == 0) hasNext = true;
111             if (m.getName().equals("previous") && m.getParameters().length == 0) hasPrevious = true;
112             if (hasNext && hasPrevious) break;
113         }
114 
115         ClassNode enumRef = enumClass.getPlainNodeReference();
116 
117         {
118             // create values() method
119             MethodNode valuesMethod = new MethodNode("values", PUBLIC_FS, enumRef.makeArray(), Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, null);
120             valuesMethod.setSynthetic(true);
121             BlockStatement code = new BlockStatement();
122             MethodCallExpression cloneCall = new MethodCallExpression(new FieldExpression(values), "clone", MethodCallExpression.NO_ARGUMENTS);
123             cloneCall.setMethodTarget(values.getType().getMethod("clone", Parameter.EMPTY_ARRAY));
124             code.addStatement(new ReturnStatement(cloneCall));
125             valuesMethod.setCode(code);
126             enumClass.addMethod(valuesMethod);
127         }
128 
129         if (!hasNext) {
130             // create next() method, code:
131             //     Day next() {
132             //        int ordinal = ordinal().next()
133             //        if (ordinal >= values().size()) ordinal = 0
134             //        return values()[ordinal]
135             //     }
136             Token assign = Token.newSymbol(Types.ASSIGN, -1, -1);
137             Token ge = Token.newSymbol(Types.COMPARE_GREATER_THAN_EQUAL, -1, -1);
138             MethodNode nextMethod = new MethodNode("next", Opcodes.ACC_PUBLIC | Opcodes.ACC_SYNTHETIC, enumRef, Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, null);
139             nextMethod.setSynthetic(true);
140             BlockStatement code = new BlockStatement();
141             BlockStatement ifStatement = new BlockStatement();
142             ifStatement.addStatement(
143                     new ExpressionStatement(
144                             new BinaryExpression(new VariableExpression("ordinal"), assign, new ConstantExpression(0))
145                     )
146             );
147 
148             code.addStatement(
149                     new ExpressionStatement(
150                             new DeclarationExpression(
151                                     new VariableExpression("ordinal"),
152                                     assign,
153                                     new MethodCallExpression(
154                                             new MethodCallExpression(
155                                                     VariableExpression.THIS_EXPRESSION,
156                                                     "ordinal",
157                                                     MethodCallExpression.NO_ARGUMENTS),
158                                             "next",
159                                             MethodCallExpression.NO_ARGUMENTS
160                                     )
161                             )
162                     )
163             );
164             code.addStatement(
165                     new IfStatement(
166                             new BooleanExpression(new BinaryExpression(
167                                     new VariableExpression("ordinal"),
168                                     ge,
169                                     new MethodCallExpression(
170                                             new FieldExpression(values),
171                                             "size",
172                                             MethodCallExpression.NO_ARGUMENTS
173                                     )
174                             )),
175                             ifStatement,
176                             EmptyStatement.INSTANCE
177                     )
178             );
179             code.addStatement(
180                     new ReturnStatement(
181                             new MethodCallExpression(new FieldExpression(values), "getAt", new VariableExpression("ordinal"))
182                     )
183             );
184             nextMethod.setCode(code);
185             enumClass.addMethod(nextMethod);
186         }
187 
188         if (!hasPrevious) {
189             // create previous() method, code:
190             //    Day previous() {
191             //        int ordinal = ordinal().previous()
192             //        if (ordinal < 0) ordinal = values().size() - 1
193             //        return values()[ordinal]
194             //    }
195             Token assign = Token.newSymbol(Types.ASSIGN, -1, -1);
196             Token lt = Token.newSymbol(Types.COMPARE_LESS_THAN, -1, -1);
197             MethodNode nextMethod = new MethodNode("previous", Opcodes.ACC_PUBLIC | Opcodes.ACC_SYNTHETIC, enumRef, Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, null);
198             nextMethod.setSynthetic(true);
199             BlockStatement code = new BlockStatement();
200             BlockStatement ifStatement = new BlockStatement();
201             ifStatement.addStatement(
202                     new ExpressionStatement(
203                             new BinaryExpression(new VariableExpression("ordinal"), assign,
204                                     new MethodCallExpression(
205                                             new MethodCallExpression(
206                                                     new FieldExpression(values),
207                                                     "size",
208                                                     MethodCallExpression.NO_ARGUMENTS
209                                             ),
210                                             "minus",
211                                             new ConstantExpression(1)
212                                     )
213                             )
214                     )
215             );
216 
217             code.addStatement(
218                     new ExpressionStatement(
219                             new DeclarationExpression(
220                                     new VariableExpression("ordinal"),
221                                     assign,
222                                     new MethodCallExpression(
223                                             new MethodCallExpression(
224                                                     VariableExpression.THIS_EXPRESSION,
225                                                     "ordinal",
226                                                     MethodCallExpression.NO_ARGUMENTS),
227                                             "previous",
228                                             MethodCallExpression.NO_ARGUMENTS
229                                     )
230                             )
231                     )
232             );
233             code.addStatement(
234                     new IfStatement(
235                             new BooleanExpression(new BinaryExpression(
236                                     new VariableExpression("ordinal"),
237                                     lt,
238                                     new ConstantExpression(0)
239                             )),
240                             ifStatement,
241                             EmptyStatement.INSTANCE
242                     )
243             );
244             code.addStatement(
245                     new ReturnStatement(
246                             new MethodCallExpression(new FieldExpression(values), "getAt", new VariableExpression("ordinal"))
247                     )
248             );
249             nextMethod.setCode(code);
250             enumClass.addMethod(nextMethod);
251         }
252 
253         {
254             // create valueOf
255             Parameter stringParameter = new Parameter(ClassHelper.STRING_TYPE, "name");
256             MethodNode valueOfMethod = new MethodNode("valueOf", PS, enumRef, new Parameter[]{stringParameter}, ClassNode.EMPTY_ARRAY, null);
257             ArgumentListExpression callArguments = new ArgumentListExpression();
258             callArguments.addExpression(new ClassExpression(enumClass));
259             callArguments.addExpression(new VariableExpression("name"));
260 
261             BlockStatement code = new BlockStatement();
262             code.addStatement(
263                     new ReturnStatement(
264                             new MethodCallExpression(new ClassExpression(ClassHelper.Enum_Type), "valueOf", callArguments)
265                     )
266             );
267             valueOfMethod.setCode(code);
268             valueOfMethod.setSynthetic(true);
269             enumClass.addMethod(valueOfMethod);
270         }
271     }

273     private void addInit(ClassNode enumClass, FieldNode minValue,
274                          FieldNode maxValue, FieldNode values,
275                          boolean isAic) {
276         // constructor helper
277         // This method is used instead of calling the constructor as
278         // calling the constructor may require a table with MetaClass
279         // selecting the constructor for each enum value. So instead we
280         // use this method to have a central point for constructor selection
281         // and only one table. The whole construction is needed because 
282         // Reflection forbids access to the enum constructor.
283         // code:
284         // def $INIT(Object[] para) {
285         //  return this(*para)
286         // }
287         ClassNode enumRef = enumClass.getPlainNodeReference();
288         Parameter[] parameter = new Parameter[]{new Parameter(ClassHelper.OBJECT_TYPE.makeArray(), "para")};
289         MethodNode initMethod = new MethodNode("$INIT", PUBLIC_FS | Opcodes.ACC_SYNTHETIC, enumRef, parameter, ClassNode.EMPTY_ARRAY, null);
290         initMethod.setSynthetic(true);
291         ConstructorCallExpression cce = new ConstructorCallExpression(
292                 ClassNode.THIS,
293                 new ArgumentListExpression(
294                         new SpreadExpression(new VariableExpression("para"))
295                 )
296         );
297         BlockStatement code = new BlockStatement();
298         code.addStatement(new ReturnStatement(cce));
299         initMethod.setCode(code);
300         enumClass.addMethod(initMethod);
301 
302         // static init
303         List<FieldNode> fields = enumClass.getFields();
304         List<Expression> arrayInit = new ArrayList<Expression>();
305         int value = -1;
306         Token assign = Token.newSymbol(Types.ASSIGN, -1, -1);
307         List<Statement> block = new ArrayList<Statement>();
308         FieldNode tempMin = null;
309         FieldNode tempMax = null;
310         for (FieldNode field : fields) {
311             if ((field.getModifiers() & Opcodes.ACC_ENUM) == 0) continue;
312             value++;
313             if (tempMin == null) tempMin = field;
314             tempMax = field;
315 
316             ClassNode enumBase = enumClass;
317             ArgumentListExpression args = new ArgumentListExpression();
318             args.addExpression(new ConstantExpression(field.getName()));
319             args.addExpression(new ConstantExpression(value));
320             if (field.getInitialExpression() == null) {
321                 if ((enumClass.getModifiers() & Opcodes.ACC_ABSTRACT) != 0) {
322                     addError(field, "The enum constant " + field.getName() + " must override abstract methods from " + enumBase.getName() + ".");
323                     continue;
324                 }
325             } else {
326                 ListExpression oldArgs = (ListExpression) field.getInitialExpression();
327                 List<MapEntryExpression> savedMapEntries = new ArrayList<MapEntryExpression>();
328                 for (Expression exp : oldArgs.getExpressions()) {
329                     if (exp instanceof MapEntryExpression) {
330                         savedMapEntries.add((MapEntryExpression) exp);
331                         continue;
332                     }
333 
334                     InnerClassNode inner = null;
335                     if (exp instanceof ClassExpression) {
336                         ClassExpression clazzExp = (ClassExpression) exp;
337                         ClassNode ref = clazzExp.getType();
338                         if (ref instanceof EnumConstantClassNode) {
339                             inner = (InnerClassNode) ref;
340                         }
341                     }
342                     if (inner != null) {
343                         List<MethodNode> baseMethods = enumBase.getMethods();
344                         for (MethodNode methodNode : baseMethods) {
345                             if (!methodNode.isAbstract()) continue;
346                             MethodNode enumConstMethod = inner.getMethod(methodNode.getName(), methodNode.getParameters());
347                             if (enumConstMethod == null || (enumConstMethod.getModifiers() & Opcodes.ACC_ABSTRACT) != 0) {
348                                 addError(field, "Can't have an abstract method in enum constant " + field.getName() + ". Implement method '" + methodNode.getTypeDescriptor() + "'.");
349                             }
350                         }
351                         if (inner.getVariableScope() == null) {
352                             enumBase = inner;
353                             /*
354                              * GROOVY-3985: Remove the final modifier from $INIT method in this case
355                              * so that subclasses of enum generated for enum constants (aic) can provide
356                              * their own $INIT method
357                              */
358                             initMethod.setModifiers(initMethod.getModifiers() & ~Opcodes.ACC_FINAL);
359                             continue;
360                         }
361                     }
362                     args.addExpression(exp);
363                 }
364                 if (savedMapEntries.size() > 0) {
365                     args.getExpressions().add(2, new MapExpression(savedMapEntries));
366                 }
367             }
368             field.setInitialValueExpression(null);
369             block.add(
370                     new ExpressionStatement(
371                             new BinaryExpression(
372                                     new FieldExpression(field),
373                                     assign,
374                                     new StaticMethodCallExpression(enumBase, "$INIT", args)
375                             )
376                     )
377             );
378             arrayInit.add(new FieldExpression(field));
379         }
380 
381         if (!isAic) {
382             if (tempMin != null) {
383                 block.add(
384                         new ExpressionStatement(
385                                 new BinaryExpression(
386                                         new FieldExpression(minValue),
387                                         assign,
388                                         new FieldExpression(tempMin)
389                                 )
390                         )
391                 );
392                 block.add(
393                         new ExpressionStatement(
394                                 new BinaryExpression(
395                                         new FieldExpression(maxValue),
396                                         assign,
397                                         new FieldExpression(tempMax)
398                                 )
399                         )
400                 );
401                 enumClass.addField(minValue);
402                 enumClass.addField(maxValue);
403             }
404 
405             block.add(
406                     new ExpressionStatement(
407                             new BinaryExpression(new FieldExpression(values), assign, new ArrayExpression(enumClass, arrayInit))
408                     )
409             );
410             enumClass.addField(values);
411         }
412         enumClass.addStaticInitializerStatements(block, true);
413     }

415     private void addError(AnnotatedNode exp, String msg) {
416         sourceUnit.getErrorCollector().addErrorAndContinue(
417                 new SyntaxErrorMessage(
418                         new SyntaxException(msg + '\n', exp.getLineNumber(), exp.getColumnNumber(), exp.getLastLineNumber(), exp.getLastColumnNumber()), sourceUnit)
419         );
420     }
421 
422     private boolean isAnonymousInnerClass(ClassNode enumClass) {
423         if (!(enumClass instanceof EnumConstantClassNode)) return false;
424         InnerClassNode ic = (InnerClassNode) enumClass;
425         return ic.getVariableScope() == null;
426     }
