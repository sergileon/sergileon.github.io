19  package org.codehaus.groovy.classgen;
20  
21  import org.codehaus.groovy.ast.ClassHelper;
22  import org.codehaus.groovy.ast.ClassNode;
23  import org.codehaus.groovy.ast.PropertyNode;
24  import org.codehaus.groovy.runtime.DummyBean;
25  
26  import java.lang.reflect.Modifier;
27  
28  /**
29   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>
30   */
31  public class PropertyTest extends TestSupport {
32  
33      public void testFields() throws Exception {
34          ClassNode classNode = new ClassNode("Foo", ACC_PUBLIC, ClassHelper.OBJECT_TYPE);
35          classNode.addField("x", ACC_PUBLIC, ClassHelper.OBJECT_TYPE, null);
36          classNode.addField("y", ACC_PUBLIC, ClassHelper.Integer_TYPE, null);
37          classNode.addField("z", ACC_PRIVATE, ClassHelper.STRING_TYPE, null);
38  
39          Class fooClass = loadClass(classNode);
40          assertTrue("Loaded a new class", fooClass != null);
41  
42          assertField(fooClass, "x", Modifier.PUBLIC, ClassHelper.OBJECT_TYPE);
43          assertField(fooClass, "y", Modifier.PUBLIC, ClassHelper.Integer_TYPE);
44          assertField(fooClass, "z", Modifier.PRIVATE, ClassHelper.STRING_TYPE);
45      }
46  
47      public void testProperties() throws Exception {
48          ClassNode classNode = new ClassNode("Foo", ACC_PUBLIC + ACC_SUPER, ClassHelper.OBJECT_TYPE);
49          classNode.addProperty(new PropertyNode("bar", ACC_PUBLIC, ClassHelper.STRING_TYPE, classNode, null, null, null));
50  
51          Class fooClass = loadClass(classNode);
52          assertTrue("Loaded a new class", fooClass != null);
53  
54          Object bean = fooClass.newInstance();
55          assertTrue("Managed to create bean", bean != null);
56  
57          assertField(fooClass, "bar", 0, ClassHelper.STRING_TYPE);
58  
59          assertGetProperty(bean, "bar", null);
60          assertSetProperty(bean, "bar", "newValue");
61      }
62  
63      public void testInheritedProperties() throws Exception {
64          ClassNode classNode = new ClassNode("Foo", ACC_PUBLIC + ACC_SUPER, ClassHelper.make(DummyBean.class));
65          classNode.addProperty(new PropertyNode("bar", ACC_PUBLIC, ClassHelper.STRING_TYPE, classNode, null, null, null));
66  
67          Class fooClass = loadClass(classNode);
68          assertTrue("Loaded a new class", fooClass != null);
69  
70          Object bean = fooClass.newInstance();
71          assertTrue("Managed to create bean", bean != null);
72  
73          assertField(fooClass, "bar", 0, ClassHelper.STRING_TYPE);
74  
75          assertGetProperty(bean, "name", "James");
76          assertSetProperty(bean, "name", "Bob");
77  
78          assertGetProperty(bean, "bar", null);
79          assertSetProperty(bean, "bar", "newValue");
80      }
81  }
