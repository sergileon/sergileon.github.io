36  /**
37   * This class handles the invocation of the ASTAnnotationTransformation
38   * when it is encountered by a tree walk.  One instance of each exists
39   * for each phase of the compilation it applies to.  Before invocation the
40   * <p>
41   * {@link org.codehaus.groovy.transform.ASTTransformationCollectorCodeVisitor} will add a list
42   * of annotations that this visitor should be concerned about.  All other
43   * annotations are ignored, whether or not they are GroovyASTTransformation
44   * annotated or not.
45   * <p>
46   * A Two-pass method is used. First all candidate annotations are added to a
47   * list then the transformations are called on those collected annotations.
48   * This is done to avoid concurrent modification exceptions during the AST tree
49   * walk and allows the transformations to alter any portion of the AST tree.
50   * Hence annotations that are added in this phase will not be processed as
51   * transformations.  They will only be handled in later phases (and then only
52   * if the type was in the AST prior to any AST transformations being run
53   * against it).
54   *
55   * @author Danno Ferrin (shemnon)
56   */
57  public final class ASTTransformationVisitor extends ClassCodeVisitorSupport {
58  
59      private final ASTTransformationsContext context;
60      private final CompilePhase phase;
61      private SourceUnit source;
62      private List<ASTNode[]> targetNodes;
63      private Map<ASTNode, List<ASTTransformation>> transforms;
64  
65      private ASTTransformationVisitor(final CompilePhase phase, final ASTTransformationsContext context) {
66          this.phase = phase;
67          this.context = context;
68      }
69  
70      protected SourceUnit getSourceUnit() {
71          return source;
72      }
73  
74      /**
75       * Main loop entry.
76       * <p>
77       * First, it delegates to the super visitClass so we can collect the
78       * relevant annotations in an AST tree walk.
79       * <p>
80       * Second, it calls the visit method on the transformation for each relevant
81       * annotation found.
82       *
83       * @param classNode the class to visit
84       */
85      public void visitClass(ClassNode classNode) {
86          // only descend if we have annotations to look for
87          Map<Class<? extends ASTTransformation>, Set<ASTNode>> baseTransforms = classNode.getTransforms(phase);
88          if (!baseTransforms.isEmpty()) {
89              final Map<Class<? extends ASTTransformation>, ASTTransformation> transformInstances = new HashMap<Class<? extends ASTTransformation>, ASTTransformation>();
90              for (Class<? extends ASTTransformation> transformClass : baseTransforms.keySet()) {
91                  try {
92                      transformInstances.put(transformClass, transformClass.newInstance());
93                  } catch (InstantiationException e) {
94                      source.getErrorCollector().addError(
95                              new SimpleMessage(
96                                      "Could not instantiate Transformation Processor " + transformClass
97                                      , //+ " declared by " + annotation.getClassNode().getName(),
98                                      source));
99                  } catch (IllegalAccessException e) {
100                     source.getErrorCollector().addError(
101                             new SimpleMessage(
102                                     "Could not instantiate Transformation Processor " + transformClass
103                                     , //+ " declared by " + annotation.getClassNode().getName(),
104                                     source));
105                 }
106             }
107 
108 
109 
110             // invert the map, is now one to many
111             transforms = new HashMap<ASTNode, List<ASTTransformation>>();
112             for (Map.Entry<Class<? extends ASTTransformation>, Set<ASTNode>> entry : baseTransforms.entrySet()) {
113                 for (ASTNode node : entry.getValue()) {
114                     List<ASTTransformation> list = transforms.get(node);
115                     if (list == null)  {
116                         list = new ArrayList<ASTTransformation>();
117                         transforms.put(node, list);
118                     }
119                     list.add(transformInstances.get(entry.getKey()));
120                 }
121             }
122 
123             targetNodes = new LinkedList<ASTNode[]>();
124 
125             // first pass, collect nodes
126             super.visitClass(classNode);
127 
128             // second pass, call visit on all of the collected nodes
129             for (ASTNode[] node : targetNodes) {
130                 for (ASTTransformation snt : transforms.get(node[0])) {
131                     if (snt instanceof CompilationUnitAware) {
132                         ((CompilationUnitAware)snt).setCompilationUnit(context.getCompilationUnit());
133                     }
134                     snt.visit(node, source);
135                 }
136             }
137         }
138     }
139 
140     /**
141      * Adds the annotation to the internal target list if a match is found.
142      *
143      * @param node the node to be processed
144      */
145     public void visitAnnotations(AnnotatedNode node) {
146         super.visitAnnotations(node);
147         for (AnnotationNode annotation : node.getAnnotations()) {
148             if (transforms.containsKey(annotation)) {
149                 targetNodes.add(new ASTNode[]{annotation, node});
150             }
151         }
152     }

154     public static void addPhaseOperations(final CompilationUnit compilationUnit) {
155         final ASTTransformationsContext context = compilationUnit.getASTTransformationsContext();
156         addGlobalTransforms(context);
157 
158         compilationUnit.addPhaseOperation(new CompilationUnit.PrimaryClassNodeOperation() {
159             public void call(SourceUnit source, GeneratorContext context, ClassNode classNode) throws CompilationFailedException {
160                 ASTTransformationCollectorCodeVisitor collector = 
161                     new ASTTransformationCollectorCodeVisitor(source, compilationUnit.getTransformLoader());
162                 collector.visitClass(classNode);
163             }
164         }, Phases.SEMANTIC_ANALYSIS);
165         for (CompilePhase phase : CompilePhase.values()) {
166             final ASTTransformationVisitor visitor = new ASTTransformationVisitor(phase, context);
167             switch (phase) {
168                 case INITIALIZATION:
169                 case PARSING:
170                 case CONVERSION:
171                     // with transform detection alone these phases are inaccessible, so don't add it
172                     break;
173 
174                 default:
175                     compilationUnit.addPhaseOperation(new CompilationUnit.PrimaryClassNodeOperation() {
176                         public void call(SourceUnit source, GeneratorContext context, ClassNode classNode) throws CompilationFailedException {
177                             visitor.source = source;
178                             visitor.visitClass(classNode);
179                         }
180                     }, phase.getPhaseNumber());
181                     break;
182 
183             }
184         }
185     }

187     public static void addGlobalTransformsAfterGrab(ASTTransformationsContext context) {
188         doAddGlobalTransforms(context, false);
189     }

191     public static void addGlobalTransforms(ASTTransformationsContext context) {
192         doAddGlobalTransforms(context, true);
193     }

195     private static void doAddGlobalTransforms(ASTTransformationsContext context, boolean isFirstScan) {
196         final CompilationUnit compilationUnit = context.getCompilationUnit();
197         GroovyClassLoader transformLoader = compilationUnit.getTransformLoader();
198         Map<String, URL> transformNames = new LinkedHashMap<String, URL>();
199         try {
200             Enumeration<URL> globalServices = transformLoader.getResources("META-INF/services/org.codehaus.groovy.transform.ASTTransformation");
201             while (globalServices.hasMoreElements()) {
202                 URL service = globalServices.nextElement();
203                 String className;
204                 BufferedReader svcIn = null;
205                 try {
206                     svcIn = new BufferedReader(new InputStreamReader(service.openStream(), "UTF-8"));
207                     try {
208                         className = svcIn.readLine();
209                     } catch (IOException ioe) {
210                         compilationUnit.getErrorCollector().addError(new SimpleMessage(
211                                 "IOException reading the service definition at "
212                                         + service.toExternalForm() + " because of exception " + ioe.toString(), null));
213                         continue;
214                     }
215                     Set<String> disabledGlobalTransforms = compilationUnit.getConfiguration().getDisabledGlobalASTTransformations();
216                     if (disabledGlobalTransforms==null) disabledGlobalTransforms=Collections.emptySet();
217                     while (className != null) {
218                         if (!className.startsWith("#") && className.length() > 0) {
219                             if (!disabledGlobalTransforms.contains(className)) {
220                                 if (transformNames.containsKey(className)) {
221                                     if (!service.equals(transformNames.get(className))) {
222                                         compilationUnit.getErrorCollector().addWarning(
223                                                 WarningMessage.POSSIBLE_ERRORS,
224                                                 "The global transform for class " + className + " is defined in both "
225                                                         + transformNames.get(className).toExternalForm()
226                                                         + " and "
227                                                         + service.toExternalForm()
228                                                         + " - the former definition will be used and the latter ignored.",
229                                                 null,
230                                                 null);
231                                     }
232 
233                                 } else {
234                                     transformNames.put(className, service);
235                                 }
236                             }
237                         }
238                         try {
239                             className = svcIn.readLine();
240                         } catch (IOException ioe) {
241                             compilationUnit.getErrorCollector().addError(new SimpleMessage(
242                                     "IOException reading the service definition at "
243                                             + service.toExternalForm() + " because of exception " + ioe.toString(), null));
244                             //noinspection UnnecessaryContinue
245                             continue;
246                         }
247                     }
248                 } finally {
249                     if (svcIn != null)
250                         svcIn.close();
251                 }
252             }
253         } catch (IOException e) {
254             //FIXME the warning message will NPE with what I have :(
255             compilationUnit.getErrorCollector().addError(new SimpleMessage(
256                 "IO Exception attempting to load global transforms:" + e.getMessage(),
257                 null));
258         }
259         try {
260             Class.forName("java.lang.annotation.Annotation"); // test for 1.5 JVM
261         } catch (Exception e) {
262             // we failed, notify the user
263             StringBuilder sb = new StringBuilder();
264             sb.append("Global ASTTransformations are not enabled in retro builds of groovy.\n");
265             sb.append("The following transformations will be ignored:");
266             for (Map.Entry<String, URL> entry : transformNames.entrySet()) {
267                 sb.append('\t');
268                 sb.append(entry.getKey());
269                 sb.append('\n');
270             }
271             compilationUnit.getErrorCollector().addWarning(new WarningMessage(
272                 WarningMessage.POSSIBLE_ERRORS, sb.toString(), null, null));
273             return;
274         }
275         
276         // record the transforms found in the first scan, so that in the 2nd scan, phase operations 
277         // can be added for only for new transforms that have come in 
278         if(isFirstScan) {
279             for (Map.Entry<String, URL> entry : transformNames.entrySet()) {
280                 context.getGlobalTransformNames().add(entry.getKey());
281             }
282             addPhaseOperationsForGlobalTransforms(context.getCompilationUnit(), transformNames, isFirstScan);
283         } else {
284             Iterator<Map.Entry<String, URL>> it = transformNames.entrySet().iterator();
285             while(it.hasNext()) {
286                 Map.Entry<String, URL> entry = it.next();
287                 if(!context.getGlobalTransformNames().add(entry.getKey())) {
288                     // phase operations for this transform class have already been added before, so remove from current scan cycle
289                     it.remove(); 
290                 }
291             }
292             addPhaseOperationsForGlobalTransforms(context.getCompilationUnit(), transformNames, isFirstScan);
293         }
294     }

296     private static void addPhaseOperationsForGlobalTransforms(CompilationUnit compilationUnit,
297             Map<String, URL> transformNames, boolean isFirstScan) {
298         GroovyClassLoader transformLoader = compilationUnit.getTransformLoader();
299         for (Map.Entry<String, URL> entry : transformNames.entrySet()) {
300             try {
301                 Class gTransClass = transformLoader.loadClass(entry.getKey(), false, true, false);
302                 //no inspection unchecked
303                 GroovyASTTransformation transformAnnotation = (GroovyASTTransformation) gTransClass.getAnnotation(GroovyASTTransformation.class);
304                 if (transformAnnotation == null) {
305                     compilationUnit.getErrorCollector().addWarning(new WarningMessage(
306                         WarningMessage.POSSIBLE_ERRORS,
307                         "Transform Class " + entry.getKey() + " is specified as a global transform in " + entry.getValue().toExternalForm()
308                         + " but it is not annotated by " + GroovyASTTransformation.class.getName()
309                         + " the global transform associated with it may fail and cause the compilation to fail.",
310                         null,
311                         null));
312                     continue;
313                 }
314                 if (ASTTransformation.class.isAssignableFrom(gTransClass)) {
315                     final ASTTransformation instance = (ASTTransformation)gTransClass.newInstance();
316                     if (instance instanceof CompilationUnitAware) {
317                         ((CompilationUnitAware)instance).setCompilationUnit(compilationUnit);
318                     }
319                     CompilationUnit.SourceUnitOperation suOp = new CompilationUnit.SourceUnitOperation() {
320                         public void call(SourceUnit source) throws CompilationFailedException {
321                             instance.visit(new ASTNode[] {source.getAST()}, source);
322                         }
323                     }; 
324                     if(isFirstScan) {
325                         compilationUnit.addPhaseOperation(suOp, transformAnnotation.phase().getPhaseNumber());
326                     } else {
327                         compilationUnit.addNewPhaseOperation(suOp, transformAnnotation.phase().getPhaseNumber());
328                     }
329                 } else {
330                     compilationUnit.getErrorCollector().addError(new SimpleMessage(
331                         "Transform Class " + entry.getKey() + " specified at "
332                         + entry.getValue().toExternalForm() + " is not an ASTTransformation.", null));
333                 }
334             } catch (Exception e) {
335                 compilationUnit.getErrorCollector().addError(new SimpleMessage(
336                     "Could not instantiate global transform class " + entry.getKey() + " specified at "
337                     + entry.getValue().toExternalForm() + "  because of exception " + e.toString(), null));
338             }
339         }
340     }
