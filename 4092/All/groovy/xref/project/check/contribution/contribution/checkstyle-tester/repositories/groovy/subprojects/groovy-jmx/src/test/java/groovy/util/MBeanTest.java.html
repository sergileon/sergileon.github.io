19  package groovy.util;
20  
21  import javax.management.MBeanServer;
22  import javax.management.MBeanServerFactory;
23  import javax.management.ObjectName;
24  
25  import static org.codehaus.groovy.runtime.DefaultGroovyMethods.join;
26  import static org.codehaus.groovy.runtime.DefaultGroovyMethods.sort;
27  
28  /**
29   * Tests using the GroovyObject API from Java to access MBeans via
30   * the normal properties API (to simulate normal Groovy property access)
31   *
32   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>
33   */
34  public class MBeanTest extends GroovyTestCase {
35  
36      public void testGetProperty() throws Exception {
37          MBeanServer mbeanServer = MBeanServerFactory.createMBeanServer();
38          ObjectName name = new ObjectName("groovy.test:role=TestMBean,type=Dummy");
39          // use Class.forName instead of new Dummy() to allow separate compilation
40          mbeanServer.registerMBean(Class.forName("groovy.util.Dummy").newInstance(), name);
41  
42          assertEquals("JMX value of Name", "James", mbeanServer.getAttribute(name, "Name"));
43  
44          GroovyMBean object = new GroovyMBean(mbeanServer, name);
45  
46          Object value = object.getProperty("Name");
47          assertEquals("Name property", "James", value);
48  
49          object.setProperty("Name", "Bob");
50          assertEquals("Name property", "Bob", object.getProperty("Name"));
51  
52          // now let's look up the name via JMX to check
53          assertEquals("JMX value of Name", "Bob", mbeanServer.getAttribute(name, "Name"));
54  
55          assertEquals("Location : London|Name : Bob|Size : 12", join(sort(object.listAttributeValues()), "|"));
56          assertEquals("start|stop", join(sort(object.listOperationNames()), "|"));
57          assertEquals("void start()", join(sort(object.describeOperation("start")), "|"));
58          assertEquals("(rw) java.lang.String Location", object.describeAttribute("Location"));
59      }
60  }
