46  public class InnerClassCompletionVisitor extends InnerClassVisitorHelper implements Opcodes {

48      private final SourceUnit sourceUnit;
49      private ClassNode classNode;
50      private FieldNode thisField = null;

52      private static final String
53              CLOSURE_INTERNAL_NAME   = BytecodeHelper.getClassInternalName(CLOSURE_TYPE),
54              CLOSURE_DESCRIPTOR      = BytecodeHelper.getTypeDescription(CLOSURE_TYPE);
55  
56      public InnerClassCompletionVisitor(CompilationUnit cu, SourceUnit su) {
57          sourceUnit = su;
58      }
59  
60      @Override
61      protected SourceUnit getSourceUnit() {
62          return sourceUnit;
63      }
64  
65      @Override
66      public void visitClass(ClassNode node) {
67          this.classNode = node;
68          thisField = null;
69          InnerClassNode innerClass = null;
70          if (!node.isEnum() && !node.isInterface() && node instanceof InnerClassNode) {
71              innerClass = (InnerClassNode) node;
72              thisField = innerClass.getField("this$0");
73              if (innerClass.getVariableScope() == null && innerClass.getDeclaredConstructors().isEmpty()) {
74                  // add dummy constructor
75                  innerClass.addConstructor(ACC_PUBLIC, Parameter.EMPTY_ARRAY, null, null);
76              }
77          }
78          if (node.isEnum() || node.isInterface()) return;
79          // use Iterator.hasNext() to check for available inner classes
80          if (node.getInnerClasses().hasNext()) addDispatcherMethods(node);
81          if (innerClass == null) return;
82          super.visitClass(node);
83          addDefaultMethods(innerClass);
84      }
85  
86      @Override
87      public void visitConstructor(ConstructorNode node) {
88          addThisReference(node);
89          super.visitConstructor(node);
90      }

92      private String getTypeDescriptor(ClassNode node, boolean isStatic) {
93          return BytecodeHelper.getTypeDescription(getClassNode(node, isStatic));
94      }
95  
96      private String getInternalName(ClassNode node, boolean isStatic) {
97          return BytecodeHelper.getClassInternalName(getClassNode(node, isStatic));
98      }

100     private void addDispatcherMethods(ClassNode classNode) {
101         final int objectDistance = getObjectDistance(classNode);
102 
103         // since we added an anonymous inner class we should also
104         // add the dispatcher methods
105 
106         // add method dispatcher
107         Parameter[] parameters = new Parameter[]{
108                 new Parameter(ClassHelper.STRING_TYPE, "name"),
109                 new Parameter(ClassHelper.OBJECT_TYPE, "args")
110         };
111         MethodNode method = classNode.addSyntheticMethod(
112                 "this$dist$invoke$" + objectDistance,
113                 ACC_PUBLIC + ACC_SYNTHETIC,
114                 ClassHelper.OBJECT_TYPE,
115                 parameters,
116                 ClassNode.EMPTY_ARRAY,
117                 null
118         );
119 
120         BlockStatement block = new BlockStatement();
121         setMethodDispatcherCode(block, VariableExpression.THIS_EXPRESSION, parameters);
122         method.setCode(block);
123 
124         // add property setter
125         parameters = new Parameter[]{
126                 new Parameter(ClassHelper.STRING_TYPE, "name"),
127                 new Parameter(ClassHelper.OBJECT_TYPE, "value")
128         };
129         method = classNode.addSyntheticMethod(
130                 "this$dist$set$" + objectDistance,
131                 ACC_PUBLIC + ACC_SYNTHETIC,
132                 ClassHelper.VOID_TYPE,
133                 parameters,
134                 ClassNode.EMPTY_ARRAY,
135                 null
136         );
137 
138         block = new BlockStatement();
139         setPropertySetterDispatcher(block, VariableExpression.THIS_EXPRESSION, parameters);
140         method.setCode(block);
141 
142         // add property getter
143         parameters = new Parameter[]{
144                 new Parameter(ClassHelper.STRING_TYPE, "name")
145         };
146         method = classNode.addSyntheticMethod(
147                 "this$dist$get$" + objectDistance,
148                 ACC_PUBLIC + ACC_SYNTHETIC,
149                 ClassHelper.OBJECT_TYPE,
150                 parameters,
151                 ClassNode.EMPTY_ARRAY,
152                 null
153         );
154 
155         block = new BlockStatement();
156         setPropertyGetterDispatcher(block, VariableExpression.THIS_EXPRESSION, parameters);
157         method.setCode(block);
158     }

160     private void getThis(MethodVisitor mv, String classInternalName, String outerClassDescriptor, String innerClassInternalName) {
161         mv.visitVarInsn(ALOAD, 0);
162         if (CLOSURE_TYPE.equals(thisField.getType())) {
163             mv.visitFieldInsn(GETFIELD, classInternalName, "this$0", CLOSURE_DESCRIPTOR);
164             mv.visitMethodInsn(INVOKEVIRTUAL, CLOSURE_INTERNAL_NAME, "getThisObject", "()Ljava/lang/Object;", false);
165             mv.visitTypeInsn(CHECKCAST, innerClassInternalName);
166         } else {
167             mv.visitFieldInsn(GETFIELD, classInternalName, "this$0", outerClassDescriptor);
168         }
169     }

171     private void addDefaultMethods(InnerClassNode node) {
172         final boolean isStatic = isStatic(node);
173 
174         ClassNode outerClass = node.getOuterClass();
175         final String classInternalName = org.codehaus.groovy.classgen.asm.BytecodeHelper.getClassInternalName(node);
176         final String outerClassInternalName = getInternalName(outerClass, isStatic);
177         final String outerClassDescriptor = getTypeDescriptor(outerClass, isStatic);
178         final int objectDistance = getObjectDistance(outerClass);
179 
180         // add method dispatcher
181         Parameter[] parameters = new Parameter[]{
182                 new Parameter(ClassHelper.STRING_TYPE, "name"),
183                 new Parameter(ClassHelper.OBJECT_TYPE, "args")
184         };
185 
186         String methodName = "methodMissing";
187         if (isStatic)
188             addCompilationErrorOnCustomMethodNode(node, methodName, parameters);
189 
190         MethodNode method = node.addSyntheticMethod(
191                 methodName,
192                 Opcodes.ACC_PUBLIC,
193                 ClassHelper.OBJECT_TYPE,
194                 parameters,
195                 ClassNode.EMPTY_ARRAY,
196                 null
197         );
198 
199         BlockStatement block = new BlockStatement();
200         if (isStatic) {
201             setMethodDispatcherCode(block, new ClassExpression(outerClass), parameters);
202         } else {
203             block.addStatement(
204                     new BytecodeSequence(new BytecodeInstruction() {
205                         public void visit(MethodVisitor mv) {
206                             getThis(mv,classInternalName,outerClassDescriptor,outerClassInternalName);
207                             mv.visitVarInsn(ALOAD, 1);
208                             mv.visitVarInsn(ALOAD, 2);
209                             mv.visitMethodInsn(INVOKEVIRTUAL, outerClassInternalName, "this$dist$invoke$" + objectDistance, "(Ljava/lang/String;Ljava/lang/Object;)Ljava/lang/Object;", false);
210                             mv.visitInsn(ARETURN);
211                         }
212                     })
213             );
214         }
215         method.setCode(block);
216 
217         // add property getter dispatcher
218         parameters = new Parameter[]{
219                 new Parameter(ClassHelper.STRING_TYPE, "name"),
220                 new Parameter(ClassHelper.OBJECT_TYPE, "val")
221         };
222 
223         methodName = "propertyMissing";
224         if (isStatic)
225             addCompilationErrorOnCustomMethodNode(node, methodName, parameters);
226 
227         method = node.addSyntheticMethod(
228                 methodName,
229                 Opcodes.ACC_PUBLIC,
230                 ClassHelper.VOID_TYPE,
231                 parameters,
232                 ClassNode.EMPTY_ARRAY,
233                 null
234         );
235 
236         block = new BlockStatement();
237         if (isStatic) {
238             setPropertySetterDispatcher(block, new ClassExpression(node.getOuterClass()), parameters);
239         } else {
240             block.addStatement(
241                     new BytecodeSequence(new BytecodeInstruction() {
242                         public void visit(MethodVisitor mv) {
243                             getThis(mv,classInternalName,outerClassDescriptor,outerClassInternalName);
244                             mv.visitVarInsn(ALOAD, 1);
245                             mv.visitVarInsn(ALOAD, 2);
246                             mv.visitMethodInsn(INVOKEVIRTUAL, outerClassInternalName, "this$dist$set$" + objectDistance, "(Ljava/lang/String;Ljava/lang/Object;)V", false);
247                             mv.visitInsn(RETURN);
248                         }
249                     })
250             );
251         }
252         method.setCode(block);
253 
254         // add property setter dispatcher
255         parameters = new Parameter[]{
256                 new Parameter(ClassHelper.STRING_TYPE, "name")
257         };
258 
259         methodName = "propertyMissing";
260         if (isStatic)
261             addCompilationErrorOnCustomMethodNode(node, methodName, parameters);
262 
263         method = node.addSyntheticMethod(
264                 methodName,
265                 Opcodes.ACC_PUBLIC,
266                 ClassHelper.OBJECT_TYPE,
267                 parameters,
268                 ClassNode.EMPTY_ARRAY,
269                 null
270         );
271 
272         block = new BlockStatement();
273         if (isStatic) {
274             setPropertyGetterDispatcher(block, new ClassExpression(node.getOuterClass()), parameters);
275         } else {
276             block.addStatement(
277                     new BytecodeSequence(new BytecodeInstruction() {
278                         public void visit(MethodVisitor mv) {
279                             getThis(mv,classInternalName,outerClassDescriptor,outerClassInternalName);
280                             mv.visitVarInsn(ALOAD, 1);
281                             mv.visitMethodInsn(INVOKEVIRTUAL, outerClassInternalName, "this$dist$get$" + objectDistance, "(Ljava/lang/String;)Ljava/lang/Object;", false);
282                             mv.visitInsn(ARETURN);
283                         }
284                     })
285             );
286         }
287         method.setCode(block);
288     }

290     /**
291      * Adds a compilation error if a {@link MethodNode} with the given <tt>methodName</tt> and
292      * <tt>parameters</tt> exists in the {@link InnerClassNode}.
293      */
294     private void addCompilationErrorOnCustomMethodNode(InnerClassNode node, String methodName, Parameter[] parameters) {
295         MethodNode existingMethodNode = node.getMethod(methodName, parameters);
296         // if there is a user-defined methodNode, add compiler error msg and continue
297         if (existingMethodNode != null && !existingMethodNode.isSynthetic())  {
298             addError("\"" +methodName + "\" implementations are not supported on static inner classes as " +
299                     "a synthetic version of \"" + methodName + "\" is added during compilation for the purpose " +
300                     "of outer class delegation.",
301                     existingMethodNode);
302         }
303     }

305     private boolean shouldHandleImplicitThisForInnerClass(ClassNode cn) {
306         if (cn.isEnum() || cn.isInterface()) return false;
307         if ((cn.getModifiers() & Opcodes.ACC_STATIC) != 0) return false;
308         if (!(cn instanceof InnerClassNode)) return false;
309         InnerClassNode innerClass = (InnerClassNode) cn;
310         // scope != null means aic, we don't handle that here
311         if (innerClass.getVariableScope() != null) return false;
312         // static inner classes don't need this$0
313         return (innerClass.getModifiers() & ACC_STATIC) == 0;
314     }

316     private void addThisReference(ConstructorNode node) {
317         if (!shouldHandleImplicitThisForInnerClass(classNode)) return;
318         Statement code = node.getCode();
319 
320         // add "this$0" field init
321 
322         //add this parameter to node
323         Parameter[] params = node.getParameters();
324         Parameter[] newParams = new Parameter[params.length + 1];
325         System.arraycopy(params, 0, newParams, 1, params.length);
326         String name = getUniqueName(params, node);
327 
328         Parameter thisPara = new Parameter(classNode.getOuterClass().getPlainNodeReference(), name);
329         newParams[0] = thisPara;
330         node.setParameters(newParams);
331 
332         BlockStatement block = null;
333         if (code == null) {
334             block = new BlockStatement();
335         } else if (!(code instanceof BlockStatement)) {
336             block = new BlockStatement();
337             block.addStatement(code);
338         } else {
339             block = (BlockStatement) code;
340         }
341         BlockStatement newCode = new BlockStatement();
342         addFieldInit(thisPara, thisField, newCode);
343         ConstructorCallExpression cce = getFirstIfSpecialConstructorCall(block);
344         if (cce == null) {
345             cce = new ConstructorCallExpression(ClassNode.SUPER, new TupleExpression());
346             block.getStatements().add(0, new ExpressionStatement(cce));
347         }
348         if (shouldImplicitlyPassThisPara(cce)) {
349             // add thisPara to this(...)
350             TupleExpression args = (TupleExpression) cce.getArguments();
351             List<Expression> expressions = args.getExpressions();
352             VariableExpression ve = new VariableExpression(thisPara.getName());
353             ve.setAccessedVariable(thisPara);
354             expressions.add(0, ve);
355         }
356         if (cce.isSuperCall()) {
357             // we have a call to super here, so we need to add
358             // our code after that
359             block.getStatements().add(1, newCode);
360         }
361         node.setCode(block);
362     }
363 
364     private boolean shouldImplicitlyPassThisPara(ConstructorCallExpression cce) {
365         boolean pass = false;
366         ClassNode superCN = classNode.getSuperClass();
367         if (cce.isThisCall()) {
368             pass = true;
369         } else if (cce.isSuperCall()) {
370             // if the super class is another non-static inner class in the same outer class hierarchy, implicit this
371             // needs to be passed
372             if (!superCN.isEnum() && !superCN.isInterface() && superCN instanceof InnerClassNode) {
373                 InnerClassNode superInnerCN = (InnerClassNode) superCN;
374                 if (!isStatic(superInnerCN) && classNode.getOuterClass().isDerivedFrom(superCN.getOuterClass())) {
375                     pass = true;
376                 }
377             }
378         }
379         return pass;
380     }

382     private String getUniqueName(Parameter[] params, ConstructorNode node) {
383         String namePrefix = "$p";
384         outer:
385         for (int i = 0; i < 100; i++) {
386             namePrefix = namePrefix + "$";
387             for (Parameter p : params) {
388                 if (p.getName().equals(namePrefix)) continue outer;
389             }
390             return namePrefix;
391         }
392         addError("unable to find a unique prefix name for synthetic this reference in inner class constructor", node);
393         return namePrefix;
394     }
395 
396     private ConstructorCallExpression getFirstIfSpecialConstructorCall(BlockStatement code) {
397         if (code == null) return null;
398 
399         final List<Statement> statementList = code.getStatements();
400         if (statementList.isEmpty()) return null;
401 
402         final Statement statement = statementList.get(0);
403         if (!(statement instanceof ExpressionStatement)) return null;
404 
405         Expression expression = ((ExpressionStatement) statement).getExpression();
406         if (!(expression instanceof ConstructorCallExpression)) return null;
407         ConstructorCallExpression cce = (ConstructorCallExpression) expression;
408         if (cce.isSpecialCall()) return cce;
409         return null;
410     }
