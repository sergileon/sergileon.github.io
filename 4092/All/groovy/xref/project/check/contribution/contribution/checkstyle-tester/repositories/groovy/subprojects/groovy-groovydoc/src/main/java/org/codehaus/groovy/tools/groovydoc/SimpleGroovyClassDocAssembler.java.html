54      public SimpleGroovyClassDocAssembler(String packagePath, String file, SourceBuffer sourceBuffer, List<LinkArgument> links, Properties properties, boolean isGroovy) {
55          this.sourceBuffer = sourceBuffer;
56          this.packagePath = packagePath;
57          this.links = links;
58          this.properties = properties;
59          this.isGroovy = isGroovy;
60  
61          stack = new Stack<GroovySourceAST>();
62          classDocs = new HashMap<String, GroovyClassDoc>();
63          className = file;
64          if (file != null) {
65              // todo: replace this simple idea of default class name
66              int idx = file.lastIndexOf(".");
67              className = file.substring(0, idx);
68          }
69  
70          deferSetup = packagePath.equals("DefaultPackage");
71          importedClassesAndPackages = new ArrayList<String>();
72          aliases = new HashMap<String, String>();
73          if (!deferSetup) setUpImports(packagePath, links, isGroovy, className);
74          lastLineCol = new LineColumn(1, 1);
75      }

77      private void setUpImports(String packagePath, List<LinkArgument> links, boolean isGroovy, String className) {
78          importedClassesAndPackages.add(packagePath + "/*");  // everything in this package
79          if (isGroovy) {
80              for (String pkg : ResolveVisitor.DEFAULT_IMPORTS) {
81                  importedClassesAndPackages.add(pkg.replace('.', '/') + "*");
82              }
83          } else {
84              importedClassesAndPackages.add("java/lang/*");
85          }
86          SimpleGroovyClassDoc currentClassDoc = new SimpleGroovyClassDoc(importedClassesAndPackages, aliases, className, links);
87          currentClassDoc.setFullPathName(packagePath + FS + className);
88          currentClassDoc.setGroovy(isGroovy);
89          classDocs.put(currentClassDoc.getFullPathName(), currentClassDoc);
90      }

121     @Override
122     public void visitClassDef(GroovySourceAST t, int visit) {
123         if (visit == OPENING_VISIT) {
124             SimpleGroovyClassDoc parent = getCurrentClassDoc();
125             String shortName = getIdentFor(t);
126             String className = shortName;
127             if (parent != null && isNested() && !insideAnonymousInnerClass()) {
128                 className = parent.name() + "." + className;
129             } else {
130                 foundClasses = new HashMap<String, SimpleGroovyClassDoc>();
131             }
132             SimpleGroovyClassDoc current = (SimpleGroovyClassDoc) classDocs.get(packagePath + FS + className);
133             if (current == null) {
134                 current = new SimpleGroovyClassDoc(importedClassesAndPackages, aliases, className, links);
135                 current.setGroovy(isGroovy);
136             }
137             current.setRawCommentText(getJavaDocCommentsBeforeNode(t));
138             current.setFullPathName(packagePath + FS + current.name());
139             current.setTokenType(t.getType());
140             current.setNameWithTypeArgs(getIdentPlusTypeArgsFor(t));
141             processAnnotations(t, current);
142             processModifiers(t, current);
143             classDocs.put(current.getFullPathName(), current);
144             foundClasses.put(shortName, current);
145             if (parent != null) {
146                 parent.addNested(current);
147                 current.setOuter(parent);
148             }
149         }
150     }

161     public void visitImport(GroovySourceAST t, int visit) {
162         if (visit == OPENING_VISIT) {
163             String importTextWithSlashesInsteadOfDots = extractImportPath(t);
164 
165             GroovySourceAST child = t.childOfType(LITERAL_as);
166             if (child != null)  {
167                 String alias = child.childOfType(DOT).getNextSibling().getText();
168 
169                 child = child.childOfType(DOT);
170                 importTextWithSlashesInsteadOfDots = recurseDownImportBranch(child);
171 
172                 aliases.put(alias, importTextWithSlashesInsteadOfDots);
173             }
174 
175             importedClassesAndPackages.add(importTextWithSlashesInsteadOfDots);
176         }
177     }

189     @Override
190     public void visitExtendsClause(GroovySourceAST t, int visit) {
191         SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
192         if (visit == OPENING_VISIT) {
193             for (GroovySourceAST superClassNode : findTypeNames(t)) {
194                 String superClassName = extractName(superClassNode);
195                 if (currentClassDoc.isInterface()) {
196                     currentClassDoc.addInterfaceName(superClassName);
197                 } else {
198                     currentClassDoc.setSuperClassName(superClassName);
199                 }
200             }
201         }
202     }

205     public void visitImplementsClause(GroovySourceAST t, int visit) {
206         if (visit == OPENING_VISIT) {
207             for (GroovySourceAST classNode : findTypeNames(t)) {
208                 getCurrentClassDoc().addInterfaceName(extractName(classNode));
209             }
210         }
211     }

213     private List<GroovySourceAST> findTypeNames(GroovySourceAST t) {
214         List<GroovySourceAST> types = new ArrayList<GroovySourceAST>();
215         for (AST child = t.getFirstChild(); child != null; child = child.getNextSibling()) {
216             GroovySourceAST groovySourceAST = (GroovySourceAST) child;
217             if (groovySourceAST.getType() == TYPE) {
218                 types.add((GroovySourceAST) groovySourceAST.getFirstChild());
219             } else {
220                 types.add(groovySourceAST);
221             }
222         }
223         return types;
224     }

227     public void visitCtorIdent(GroovySourceAST t, int visit) {
228         if (visit == OPENING_VISIT && !insideEnum && !insideAnonymousInnerClass()) {
229             SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
230             SimpleGroovyConstructorDoc currentConstructorDoc = new SimpleGroovyConstructorDoc(currentClassDoc.name(), currentClassDoc);

232             processModifiers(t, currentConstructorDoc);
233             addParametersTo(t, currentConstructorDoc);
234             processAnnotations(t, currentConstructorDoc);
235             currentClassDoc.add(currentConstructorDoc);
236         }

239     @Override
240     public void visitMethodDef(GroovySourceAST t, int visit) {
241         if (visit == OPENING_VISIT && !insideEnum && !insideAnonymousInnerClass()) {
242             SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
243             if (currentClassDoc == null) {
244                 // assume we have a script
245                 if ("true".equals(properties.getProperty("processScripts", "true"))) {
246                     currentClassDoc = new SimpleGroovyClassDoc(importedClassesAndPackages, aliases, className, links);
247                     currentClassDoc.setFullPathName(packagePath + FS + className);
248                     currentClassDoc.setPublic(true);
249                     currentClassDoc.setScript(true);
250                     currentClassDoc.setGroovy(isGroovy);
251                     currentClassDoc.setSuperClassName("groovy/lang/Script");
252                     if ("true".equals(properties.getProperty("includeMainForScripts", "true"))) {
253                         currentClassDoc.add(createMainMethod(currentClassDoc));
254                     }
255                     classDocs.put(currentClassDoc.getFullPathName(), currentClassDoc);
256                     if (foundClasses == null) {
257                         foundClasses = new HashMap<String, SimpleGroovyClassDoc>();
258                     }
259                     foundClasses.put(className, currentClassDoc);
260                 } else {
261                     return;
262                 }
263             }
264             SimpleGroovyMethodDoc currentMethodDoc = createMethod(t, currentClassDoc);
265             currentClassDoc.add(currentMethodDoc);
266         }
267     }

269     private SimpleGroovyMethodDoc createMethod(GroovySourceAST t, SimpleGroovyClassDoc currentClassDoc) {
270         String methodName = getIdentFor(t);
271         SimpleGroovyMethodDoc currentMethodDoc = new SimpleGroovyMethodDoc(methodName, currentClassDoc);
272         currentMethodDoc.setRawCommentText(getJavaDocCommentsBeforeNode(t));
273         processModifiers(t, currentMethodDoc);
274         currentMethodDoc.setReturnType(new SimpleGroovyType(getTypeOrDefault(t)));
275         addParametersTo(t, currentMethodDoc);
276         processAnnotations(t, currentMethodDoc);
277         return currentMethodDoc;
278     }

280     private GroovyMethodDoc createMainMethod(SimpleGroovyClassDoc currentClassDoc) {
281         SimpleGroovyMethodDoc mainMethod = new SimpleGroovyMethodDoc("main", currentClassDoc);
282         mainMethod.setPublic(true);
283         mainMethod.setStatic(true);
284         mainMethod.setCommentText("Implicit main method for Groovy Scripts");
285         mainMethod.setFirstSentenceCommentText(mainMethod.commentText());
286         SimpleGroovyParameter args = new SimpleGroovyParameter("args");
287         GroovyType argsType = new SimpleGroovyType("java.lang.String[]");
288         args.setType(argsType);
289         mainMethod.add(args);
290         GroovyType returnType = new SimpleGroovyType("void");
291         mainMethod.setReturnType(returnType);
292         return mainMethod;
293     }

295     @Override
296     public void visitAnnotationFieldDef(GroovySourceAST t, int visit) {
297         if (isGroovy && visit == OPENING_VISIT) {
298             // TODO shouldn't really be treating annotation fields as methods - remove this hack
299             SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
300             SimpleGroovyMethodDoc currentMethodDoc = createMethod(t, currentClassDoc);
301             String defaultText = getDefaultValue(t);
302             if (defaultText != null) {
303                 String orig = currentMethodDoc.getRawCommentText();
304                 currentMethodDoc.setRawCommentText(orig + "\n* @default " + defaultText);
305             }
306             currentClassDoc.add(currentMethodDoc);
307         } else if (visit == OPENING_VISIT) {
308 //        if (visit == OPENING_VISIT) {
309             visitVariableDef(t, visit);
310             String defaultText = getDefaultValue(t);
311             if (isGroovy) {
312                 currentFieldDoc.setPublic(true);
313             }
314             if (defaultText != null) {
315                 currentFieldDoc.setConstantValueExpression(defaultText);
316                 String orig = currentFieldDoc.getRawCommentText();
317                 currentFieldDoc.setRawCommentText(orig + "\n* @default " + defaultText);
318             }
319         }
320     }

322     @Override
323     public void visitEnumConstantDef(GroovySourceAST t, int visit) {
324         if (visit == OPENING_VISIT) {
325             SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
326             insideEnum = true;
327             String enumConstantName = getIdentFor(t);
328             SimpleGroovyFieldDoc currentEnumConstantDoc = new SimpleGroovyFieldDoc(enumConstantName, currentClassDoc);
329             currentEnumConstantDoc.setRawCommentText(getJavaDocCommentsBeforeNode(t));
330             processModifiers(t, currentEnumConstantDoc);
331             String typeName = getTypeNodeAsText(t.childOfType(TYPE), currentClassDoc.getTypeDescription());
332             currentEnumConstantDoc.setType(new SimpleGroovyType(typeName));
333             currentClassDoc.addEnumConstant(currentEnumConstantDoc);
334         } else if (visit == CLOSING_VISIT) {
335             insideEnum = false;
336         }

339     @Override
340     public void visitVariableDef(GroovySourceAST t, int visit) {
341         if (visit == OPENING_VISIT && !insideAnonymousInnerClass() && isFieldDefinition()) {
342             SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
343             if (currentClassDoc != null) {
344                 String fieldName = getIdentFor(t);
345                 currentFieldDoc = new SimpleGroovyFieldDoc(fieldName, currentClassDoc);
346                 currentFieldDoc.setRawCommentText(getJavaDocCommentsBeforeNode(t));
347                 boolean isProp = processModifiers(t, currentFieldDoc);
348                 currentFieldDoc.setType(new SimpleGroovyType(getTypeOrDefault(t)));
349                 processAnnotations(t, currentFieldDoc);
350                 if (isProp) {
351                     currentClassDoc.addProperty(currentFieldDoc);
352                 } else {
353                     currentClassDoc.add(currentFieldDoc);
354                 }
355             }
356         }
357     }

369     private void gobbleComments(GroovySourceAST t, int visit) {
370         if (visit == OPENING_VISIT) {
371             SimpleGroovyClassDoc currentClassDoc = getCurrentClassDoc();
372             if (currentClassDoc == null || currentClassDoc.isScript()) {
373                 if (t.getLine() > lastLineCol.getLine() ||
374                         (t.getLine() == lastLineCol.getLine() && t.getColumn() > lastLineCol.getColumn())) {
375                     getJavaDocCommentsBeforeNode(t);
376                     // not normally set for non-major types but appropriate for a script
377                     lastLineCol = new LineColumn(t.getLine(), t.getColumn());
378                 }
379             }
380         }
381     }

383     // Step through ClassDocs and tie up loose ends
384     private void postProcessClassDocs() {
385         for (GroovyClassDoc groovyClassDoc : classDocs.values()) {
386             SimpleGroovyClassDoc classDoc = (SimpleGroovyClassDoc) groovyClassDoc;
387 
388             // potentially add default constructor to class docs (but not interfaces)
389             if (classDoc.isClass()) {
390                 GroovyConstructorDoc[] constructors = classDoc.constructors();
391                 if (constructors != null && constructors.length == 0) { // add default constructor to doc
392                     // name of class for the constructor
393                     GroovyConstructorDoc constructorDoc = new SimpleGroovyConstructorDoc(classDoc.name(), classDoc);
394                     // don't forget to tell the class about this default constructor.
395                     classDoc.add(constructorDoc);
396                 }
397             }
398         }
399     }

406         if (node == null) return false;

411     private void adjustForAutomaticEnumMethods(SimpleGroovyClassDoc currentClassDoc) {
412         SimpleGroovyMethodDoc valueOf = new SimpleGroovyMethodDoc("valueOf", currentClassDoc);
413         valueOf.setRawCommentText("Returns the enum constant of this type with the specified name.");
414         SimpleGroovyParameter parameter = new SimpleGroovyParameter("name");
415         parameter.setTypeName("String");
416         valueOf.add(parameter);
417         valueOf.setReturnType(new SimpleGroovyType(currentClassDoc.name()));
418         currentClassDoc.add(valueOf);
419 
420         SimpleGroovyMethodDoc values = new SimpleGroovyMethodDoc("values", currentClassDoc);
421         values.setRawCommentText("Returns an array containing the constants of this enum type, in the order they are declared.");
422         values.setReturnType(new SimpleGroovyType(currentClassDoc.name() + "[]"));
423         currentClassDoc.add(values);
424     }

438     private String recurseDownImportBranch(GroovySourceAST t) {
439         if (t != null) {
440             if (t.getType() == DOT) {
441                 GroovySourceAST firstChild = (GroovySourceAST) t.getFirstChild();
442                 GroovySourceAST secondChild = (GroovySourceAST) firstChild.getNextSibling();
443                 return (recurseDownImportBranch(firstChild) + "/" + recurseDownImportBranch(secondChild));
444             }
445             if (t.getType() == IDENT) {
446                 return t.getText();
447             }
448             if (t.getType() == STAR) {
449                 return t.getText();
450             }
451         }
452         return "";
453     }

483     private String getDefaultValue(GroovySourceAST t) {
484         GroovySourceAST child = (GroovySourceAST) t.getFirstChild();
485         if (t.getNumberOfChildren() != 4) return null;
486         for (int i = 1; i < t.getNumberOfChildren(); i++) {
487             child = (GroovySourceAST) child.getNextSibling();
488         }
489         GroovySourceAST nodeToProcess = child;
490         if (child.getType() != ANNOTATION_ARRAY_INIT && child.getNumberOfChildren() > 0) {
491             nodeToProcess = (GroovySourceAST) child.getFirstChild();
492         }
493         return getChildTextFromSource(nodeToProcess, ";");
494     }

496     private String getChildTextFromSource(GroovySourceAST child) {
497         return sourceBuffer.getSnippet(
498                 new LineColumn(child.getLine(), child.getColumn()),
499                 new LineColumn(child.getLineLast(), child.getColumnLast()));

502     private String getChildTextFromSource(GroovySourceAST child, String tokens) {
503         String text = sourceBuffer.getSnippet(
504                 new LineColumn(child.getLine(), child.getColumn()),
505                 new LineColumn(child.getLine() + 1, 0));
506         StringTokenizer st = new StringTokenizer(text, tokens);
507         return st.nextToken();

520     // return true if a property is found
521     private boolean processModifiers(GroovySourceAST t, SimpleGroovyAbstractableElementDoc memberOrClass) {
522         GroovySourceAST modifiers = t.childOfType(MODIFIERS);
523         boolean hasNonPublicVisibility = false;
524         boolean hasPublicVisibility = false;
525         if (modifiers != null) {
526             AST currentModifier = modifiers.getFirstChild();
527             while (currentModifier != null) {
528                 int type = currentModifier.getType();
529                 switch (type) {
530                     case LITERAL_public:
531                         memberOrClass.setPublic(true);
532                         hasPublicVisibility = true;
533                         break;
534                     case LITERAL_protected:
535                         memberOrClass.setProtected(true);
536                         hasNonPublicVisibility = true;
537                         break;
538                     case LITERAL_private:
539                         memberOrClass.setPrivate(true);
540                         hasNonPublicVisibility = true;
541                         break;
542                     case LITERAL_static:
543                         memberOrClass.setStatic(true);
544                         break;
545                     case FINAL:
546                         memberOrClass.setFinal(true);
547                         break;
548                     case ABSTRACT:
549                         memberOrClass.setAbstract(true);
550                         break;
551                 }
552                 currentModifier = currentModifier.getNextSibling();
553             }
554             if (!hasNonPublicVisibility && isGroovy && !(memberOrClass instanceof GroovyFieldDoc)) {
555                 // in groovy, methods and classes are assumed public, unless informed otherwise
556                 if (isPackageScope(modifiers)) {
557                     memberOrClass.setPackagePrivate(true);
558                     hasNonPublicVisibility = true;
559                 } else {
560                     memberOrClass.setPublic(true);
561                 }
562             } else if (!hasNonPublicVisibility && !hasPublicVisibility && !isGroovy) {
563                 if (insideInterface(memberOrClass) || insideAnnotationDef(memberOrClass)) {
564                     memberOrClass.setPublic(true);
565                 } else {
566                     memberOrClass.setPackagePrivate(true);
567                 }
568             }
569             if (memberOrClass instanceof GroovyFieldDoc && isGroovy && !hasNonPublicVisibility & !hasPublicVisibility) {
570                 if (isPackageScope(modifiers)) {
571                     memberOrClass.setPackagePrivate(true);
572                     hasNonPublicVisibility = true;
573                 }
574             }
575             if (memberOrClass instanceof GroovyFieldDoc && !hasNonPublicVisibility && !hasPublicVisibility && isGroovy) return true;
576         } else if (isGroovy && !(memberOrClass instanceof GroovyFieldDoc)) {
577             // in groovy, methods and classes are assumed public, unless informed otherwise
578             memberOrClass.setPublic(true);
579         } else if (!isGroovy) {
580             if (insideInterface(memberOrClass) || insideAnnotationDef(memberOrClass)) {
581                 memberOrClass.setPublic(true);
582             } else {
583                 memberOrClass.setPackagePrivate(true);
584             }
585         }
586         return memberOrClass instanceof GroovyFieldDoc && isGroovy && !hasNonPublicVisibility & !hasPublicVisibility;
587     }

589     private boolean isPackageScope(GroovySourceAST modifiers) {
590         List<String> names = getAnnotationNames(modifiers);
591         return names.contains("groovy/transform/PackageScope") || names.contains("PackageScope");
592     }

594     private List<String> getAnnotationNames(GroovySourceAST modifiers) {
595         List<String> annotationNames = new ArrayList<String>();
596         List<GroovySourceAST> annotations = modifiers.childrenOfType(ANNOTATION);
597         for (GroovySourceAST annotation : annotations) {
598             annotationNames.add(buildName((GroovySourceAST) annotation.getFirstChild()));
599         }
600         return annotationNames;
601     }

606         return current.isInterface();

609     private boolean insideAnnotationDef(SimpleGroovyAbstractableElementDoc memberOrClass) {
610         SimpleGroovyClassDoc current = getCurrentClassDoc();
611         if (current == null || current == memberOrClass) return false;
612         return current.isAnnotationType();
613     }

617     private String getJavaDocCommentsBeforeNode(GroovySourceAST t) {
618         String result = "";
619         LineColumn thisLineCol = new LineColumn(t.getLine(), t.getColumn());
620         String text = sourceBuffer.getSnippet(lastLineCol, thisLineCol);
621         if (text != null) {
622             Matcher m = PREV_JAVADOC_COMMENT_PATTERN.matcher(text);

627         if (isMajorType(t)) {
628             lastLineCol = thisLineCol;
629         }
630         return result;

649     private String extractName(GroovySourceAST typeNode) {
650         String typeName = buildName(typeNode);
651         if (!typeName.contains("/")) {
652             String slashName = "/" + typeName;
653             // Groovy currently resolves this to last found so traverse in reverse order
654             for (int i = importedClassesAndPackages.size() - 1; i >= 0; i--) {
655                 String name = importedClassesAndPackages.get(i);
656                 if (name.endsWith(slashName)) {
657                     typeName = name;
658                     break;
659                 }
660             }
661         }
662         return typeName;
663     }

665     private String buildName(GroovySourceAST t) {
666         if (t != null) {
667             if (t.getType() == DOT) {
668                 GroovySourceAST firstChild = (GroovySourceAST) t.getFirstChild();
669                 GroovySourceAST secondChild = (GroovySourceAST) firstChild.getNextSibling();
670                 return (buildName(firstChild) + "/" + buildName(secondChild));
671             }
672             if (t.getType() == IDENT) {
673                 return t.getText();
674             }
675         }
676         return "";
677     }

684     private String getTypeNodeAsText(GroovySourceAST typeNode, String defaultText) {
685         // TODO refactor to retain richer type information rather than converting to String
686         if (typeNode == null) {
687             return defaultText;
688         }
689         if (typeNode.getType() == TYPE) {
690             return getAsText(typeNode, defaultText);
691         } else if (typeNode.getType() == TYPE_ARGUMENT) {
692             return getTypeNodeAsText((GroovySourceAST) typeNode.getFirstChild(), defaultText);
693         } else if (typeNode.getType() == WILDCARD_TYPE) {
694             AST next = typeNode.getNextSibling();
695             if (next == null && typeNode.getFirstChild() != null) {
696                 // Java2Groovy produces a slightly different tree structure (TODO fix converter or java.g instead?)
697                 next = typeNode.getFirstChild();
698             }
699             if (next == null) return "?";
700             String boundType = getTypeNodeAsText((GroovySourceAST) next.getFirstChild(), defaultText);
701             if (next.getType() == TYPE_UPPER_BOUNDS) return "? extends " + boundType;
702             if (next.getType() == TYPE_LOWER_BOUNDS) return "? super " + boundType;
703         } else if (typeNode.getType() == IDENT) {
704             return getAsTextCurrent(typeNode, defaultText);
705         }
706         return defaultText;
707     }

709     private String getAsText(GroovySourceAST typeNode, String defaultText) {
710         GroovySourceAST child = (GroovySourceAST) typeNode.getFirstChild();
711         return getAsTextCurrent(child, defaultText);
712     }

714     private String getAsTextCurrent(GroovySourceAST node, String defaultText) {
715         if (node == null) return defaultText;
716         switch (node.getType()) {
717             // literals
718             case LITERAL_boolean:
719                 return "boolean";
720             case LITERAL_byte:
721                 return "byte";
722             case LITERAL_char:
723                 return "char";
724             // note: LITERAL_def never created
725             case LITERAL_double:
726                 return "double";
727             case LITERAL_float:
728                 return "float";
729             case LITERAL_int:
730                 return "int";
731             case LITERAL_long:
732                 return "long";
733             case LITERAL_short:
734                 return "short";
735             case LITERAL_void:
736                 return "void";
737             case ARRAY_DECLARATOR:
738                 String componentType = getAsText(node, defaultText);
739                 if (!componentType.equals("def")) return componentType + "[]";
740                 return "java/lang/Object[]";
741             // identifiers
742             case IDENT:
743                 StringBuilder ident = new StringBuilder();
744                 ident.append(node.getText());
745                 GroovySourceAST identChild = (GroovySourceAST) node.getFirstChild();
746                 getTypeArguments(identChild, ident, defaultText);
747                 return ident.toString();
748             case DOT:
749                 StringBuilder dot = new StringBuilder();
750                 GroovySourceAST dotChild = (GroovySourceAST) node.getFirstChild();
751                 while (dotChild != null) {
752                     if (dotChild.getType() == IDENT) {
753                         if (dot.length() > 0) dot.append("/");
754                         dot.append(getAsTextCurrent(dotChild, defaultText));
755                     } else if (dotChild.getType() == DOT) {
756                         if (dot.length() > 0) dot.append("/");
757                         dot.append(getAsTextCurrent(dotChild, defaultText));
758                     } else if (dotChild.getType() == TYPE_ARGUMENTS) {
759                         getTypeArguments(dotChild, dot, defaultText);
760                     }
761                     dotChild = (GroovySourceAST) dotChild.getNextSibling();
762                 }
763                 return dot.toString();
764         }
765         return defaultText;
766     }

768     private void getTypeArguments(GroovySourceAST child, StringBuilder result, String defaultText) {
769         if (child != null && child.getType() == TYPE_ARGUMENTS && child.getNumberOfChildren() > 0) {
770             result.append("<");
771             GroovySourceAST typeArgumentsNext = (GroovySourceAST) child.getFirstChild();
772             List<String> typeArgumentParts = new ArrayList<String>();
773             while (typeArgumentsNext != null) {
774                 if (typeArgumentsNext.getType() == TYPE_ARGUMENT && typeArgumentsNext.getNumberOfChildren() > 0) {
775                     typeArgumentParts.add(getTypeNodeAsText((GroovySourceAST) typeArgumentsNext.getFirstChild(), defaultText));
776                 }
777                 typeArgumentsNext = (GroovySourceAST) typeArgumentsNext.getNextSibling();
778             }
779             result.append(DefaultGroovyMethods.join(typeArgumentParts, ", "));
780             result.append(">");
781         }
782     }

784     private void getTypeParameters(GroovySourceAST child, StringBuilder result, String defaultText) {
785         if (child != null && child.getType() == TYPE_PARAMETERS && child.getNumberOfChildren() > 0) {
786             result.append("<");
787             GroovySourceAST typeParametersNext = (GroovySourceAST) child.getFirstChild();
788             List<String> typeParameterParts = new ArrayList<String>();
789             while (typeParametersNext != null) {
790                 if (typeParametersNext.getType() == TYPE_PARAMETER && typeParametersNext.getNumberOfChildren() > 0) {
791                     typeParameterParts.add(getTypeNodeAsText((GroovySourceAST) typeParametersNext.getFirstChild(), defaultText));
792                 }
793                 typeParametersNext = (GroovySourceAST) typeParametersNext.getNextSibling();
794             }
795             result.append(DefaultGroovyMethods.join(typeParameterParts, ", "));
796             result.append(">");
797         }
798     }

800     private void addParametersTo(GroovySourceAST t, SimpleGroovyExecutableMemberDoc executableMemberDoc) {
801         // parameters
802         GroovySourceAST parametersNode = t.childOfType(PARAMETERS);
803         if (parametersNode != null && parametersNode.getNumberOfChildren() > 0) {
804             GroovySourceAST currentNode = (GroovySourceAST) parametersNode.getFirstChild();
805             while (currentNode != null) {
806                 String parameterTypeName = getTypeOrDefault(currentNode);
807                 String parameterName = getText(currentNode.childOfType(IDENT));
808                 SimpleGroovyParameter parameter = new SimpleGroovyParameter(parameterName);
809                 parameter.setVararg(currentNode.getType() == VARIABLE_PARAMETER_DEF);
810                 parameter.setTypeName(parameterTypeName);
811                 GroovySourceAST modifiers = currentNode.childOfType(MODIFIERS);
812                 if (modifiers != null) {
813                     List<GroovySourceAST> annotations = modifiers.childrenOfType(ANNOTATION);
814                     for (GroovySourceAST a : annotations) {
815                         addAnnotationRef(parameter, a);
816                     }
817                 }
818                 executableMemberDoc.add(parameter);
819                 if (currentNode.getNumberOfChildren() == 4) {
820                     handleDefaultValue(currentNode, parameter);
821                 }
822                 currentNode = (GroovySourceAST) currentNode.getNextSibling();
823             }
824         }
825     }

827     private void handleDefaultValue(GroovySourceAST currentNode, SimpleGroovyParameter parameter) {
828         GroovySourceAST paramPart = (GroovySourceAST) currentNode.getFirstChild();
829         for (int i = 1; i < currentNode.getNumberOfChildren(); i++) {
830             paramPart = (GroovySourceAST) paramPart.getNextSibling();
831         }
832         GroovySourceAST nodeToProcess = paramPart;
833         if (paramPart.getNumberOfChildren() > 0) {
834             nodeToProcess = (GroovySourceAST) paramPart.getFirstChild();
835         }
836         // hack warning!
837         // TODO handle , and ) when they occur within Strings
838         parameter.setDefaultValue(getChildTextFromSource(nodeToProcess, ",)"));
839     }

852     private GroovySourceAST getParentNode() {
853         GroovySourceAST parentNode = null;
854         GroovySourceAST currentNode = stack.pop();
855         if (!stack.empty()) {
856             parentNode = stack.peek();
857         }
858         stack.push(currentNode);

862     private GroovySourceAST getGrandParentNode() {
863         GroovySourceAST grandParentNode = null;
864         GroovySourceAST parentNode;
865         GroovySourceAST currentNode = stack.pop();
866         if (!stack.empty()) {
867             parentNode = stack.pop();
868             if (!stack.empty()) {
869                 grandParentNode = stack.peek();
870             }
871             stack.push(parentNode);
872         }
873         stack.push(currentNode);
874         return grandParentNode;
875     }

883     private SimpleGroovyClassDoc getCurrentClassDoc() {
884         if (stack.isEmpty()) return null;
885         GroovySourceAST node = getParentNode();
886         if (isTopLevelConstruct(node)) return foundClasses.get(getIdentFor(node));
887         GroovySourceAST saved = stack.pop();
888         SimpleGroovyClassDoc result = getCurrentClassDoc();
889         stack.push(saved);
890         return result;
891     }
