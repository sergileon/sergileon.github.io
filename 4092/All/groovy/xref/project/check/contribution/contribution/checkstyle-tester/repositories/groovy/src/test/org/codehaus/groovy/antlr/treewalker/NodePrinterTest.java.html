19  package org.codehaus.groovy.antlr.treewalker;
20  
21  import junit.framework.TestCase;
22  import org.custommonkey.xmlunit.XMLUnit;
23  
24  /**
25   * Testcases for the antlr AST visitor that prints groovy source code nodes as Pseudo XML.
26   */
27  public class NodePrinterTest extends TestCase {
28  
29      protected void setUp() throws Exception {
30          XMLUnit.setIgnoreWhitespace(true);
31      }
32  
33      public void testAbstract() throws Exception {
34          assertPseudoXmlEquals("<MODIFIERS><\"public\"></\"public\"></MODIFIERS><\"abstract\"></\"abstract\"><CLASS_DEF></CLASS_DEF><an identifier></an identifier><EXTENDS_CLAUSE></EXTENDS_CLAUSE><IMPLEMENTS_CLAUSE></IMPLEMENTS_CLAUSE><OBJBLOCK></OBJBLOCK>",
35                  nodify("public abstract class Foo{}"));
36      }
37  
38      public void testArrayDeclarator() throws Exception {
39          assertPseudoXmlEquals(
40                  "<VARIABLE_DEF><TYPE><ARRAY_DECLARATOR><\"int\"></\"int\"></ARRAY_DECLARATOR></TYPE><an identifier></an identifier><'='><\"new\"><\"int\"></\"int\"></\"new\"><ARRAY_DECLARATOR><a numeric literal></a numeric literal></ARRAY_DECLARATOR></\"new\"></'='></VARIABLE_DEF>",
41                  nodify("int[] primes = new int[5]"));
42      }
43  
44      public void testRegexMatch() throws Exception {
45          assertPseudoXmlEquals(
46                  "<\"if\"><an identifier></an identifier><'==~'></'==~'><a string literal></a string literal></\"if\"><SLIST></SLIST>",
47                  nodify("if (foo==~\"bar\"){}"));
48      }
49  
50      private void assertPseudoXmlEquals(String expected, String actual) throws Exception {
51          assertEquals(expected, actual);
52      }
53  
54      private String nodify(String input) throws Exception {
55          TraversalTestHelper traverser = new TraversalTestHelper();
56          return traverser.traverse(input, NodePrinter.class);
57      }
58  
59  }
