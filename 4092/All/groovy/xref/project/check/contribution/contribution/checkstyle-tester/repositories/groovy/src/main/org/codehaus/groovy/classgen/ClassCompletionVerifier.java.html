60      public ClassNode getClassNode() {
61          return currentClass;

64      public void visitClass(ClassNode node) {
65          ClassNode oldClass = currentClass;
66          currentClass = node;
67          checkImplementsAndExtends(node);
68          if (source != null && !source.getErrorCollector().hasErrors()) {
69              checkClassForIncorrectModifiers(node);
70              checkInterfaceMethodVisibility(node);
71              checkClassForOverwritingFinal(node);
72              checkMethodsForIncorrectModifiers(node);
73              checkMethodsForWeakerAccess(node);
74              checkMethodsForOverridingFinal(node);
75              checkNoAbstractMethodsNonabstractClass(node);
76              checkClassExtendsAllSelfTypes(node);
77              checkNoStaticMethodWithSameSignatureAsNonStatic(node);
78              checkGenericsUsage(node, node.getUnresolvedInterfaces());
79              checkGenericsUsage(node, node.getUnresolvedSuperClass());
80          }
81          super.visitClass(node);
82          currentClass = oldClass;
83      }

85      private void checkNoStaticMethodWithSameSignatureAsNonStatic(final ClassNode node) {
86          Map<String, MethodNode> result = new HashMap<String, MethodNode>();
87          // add in unimplemented abstract methods from the interfaces
88          for (ClassNode iface : node.getInterfaces()) {
89              Map<String, MethodNode> ifaceMethodsMap = iface.getDeclaredMethodsMap();
90              for (String methSig : ifaceMethodsMap.keySet()) {
91                  if (!result.containsKey(methSig)) {
92                      MethodNode methNode = ifaceMethodsMap.get(methSig);
93                      result.put(methSig, methNode);
94                  }
95              }
96          }
97          for (MethodNode methodNode : node.getMethods()) {
98              MethodNode mn = result.get(methodNode.getTypeDescriptor());
99              if (mn!=null && methodNode.isStatic() && !methodNode.isStaticConstructor()) {
100                 ClassNode declaringClass = mn.getDeclaringClass();
101                 ClassNode cn = declaringClass.getOuterClass();
102                 if (cn==null && declaringClass.isResolved()) {
103                     // in case of a precompiled class, the outerclass is unknown
104                     Class typeClass = declaringClass.getTypeClass();
105                     typeClass = typeClass.getEnclosingClass();
106                     if (typeClass!=null) {
107                         cn = ClassHelper.make(typeClass);
108                     }
109                 }
110                 if (cn==null || !Traits.isTrait(cn)) {
111                     addError("Method '" + mn.getName() + "' is already defined in " + getDescription(node) + ". You cannot have " +
112                             "both a static and a non static method with the same signature", methodNode);
113                 }
114             }
115             result.put(methodNode.getTypeDescriptor(), methodNode);
116         }
117     }

119     private void checkInterfaceMethodVisibility(ClassNode node) {
120         if (!node.isInterface()) return;
121         for (MethodNode method : node.getMethods()) {
122             if (method.isPrivate()) {
123                 addError("Method '" + method.getName() + "' is private but should be public in " + getDescription(currentClass) + ".", method);
124             } else if (method.isProtected()) {
125                 addError("Method '" + method.getName() + "' is protected but should be public in " + getDescription(currentClass) + ".", method);
126             }
127         }
128     }

130     private void checkNoAbstractMethodsNonabstractClass(ClassNode node) {
131         if (isAbstract(node.getModifiers())) return;
132         List<MethodNode> abstractMethods = node.getAbstractMethods();
133         if (abstractMethods == null) return;
134         for (MethodNode method : abstractMethods) {
135             MethodNode sameArgsMethod = node.getMethod(method.getName(), method.getParameters());
136             if (sameArgsMethod==null || method.getReturnType().equals(sameArgsMethod.getReturnType())) {
137                 addError("Can't have an abstract method in a non-abstract class." +
138                         " The " + getDescription(node) + " must be declared abstract or" +
139                         " the " + getDescription(method) + " must be implemented.", node);
140             } else {
141                 addError("Abstract "+getDescription(method)+" is not implemented but a " +
142                                 "method of the same name but different return type is defined: "+
143                                 (sameArgsMethod.isStatic()?"static ":"")+
144                                 getDescription(sameArgsMethod), method
145                 );
146             }
147         }
148     }

150     private void checkClassExtendsAllSelfTypes(ClassNode node) {
151         int modifiers = node.getModifiers();
152         if (!isInterface(modifiers)) {
153             for (ClassNode anInterface : GeneralUtils.getInterfacesAndSuperInterfaces(node)) {
154                 if (Traits.isTrait(anInterface)) {
155                     LinkedHashSet<ClassNode> selfTypes = new LinkedHashSet<ClassNode>();
156                     for (ClassNode type : Traits.collectSelfTypes(anInterface, selfTypes, true, false)) {
157                         if (type.isInterface() && !node.implementsInterface(type)) {
158                             addError(getDescription(node)
159                                     + " implements " + getDescription(anInterface)
160                                     + " but does not implement self type " + getDescription(type),
161                                     anInterface);
162                         } else if (!type.isInterface() && !node.isDerivedFrom(type)) {
163                             addError(getDescription(node)
164                                             + " implements " + getDescription(anInterface)
165                                             + " but does not extend self type " + getDescription(type),
166                                     anInterface);
167                         }
168                     }
169                 }
170             }
171         }
172     }

174     private void checkClassForIncorrectModifiers(ClassNode node) {
175         checkClassForAbstractAndFinal(node);
176         checkClassForOtherModifiers(node);
177     }
178 
179     private void checkClassForAbstractAndFinal(ClassNode node) {
180         if (!isAbstract(node.getModifiers())) return;
181         if (!isFinal(node.getModifiers())) return;
182         if (node.isInterface()) {
183             addError("The " + getDescription(node) + " must not be final. It is by definition abstract.", node);
184         } else {
185             addError("The " + getDescription(node) + " must not be both final and abstract.", node);
186         }
187     }
188 
189     private void checkClassForOtherModifiers(ClassNode node) {
190         checkClassForModifier(node, isTransient(node.getModifiers()), "transient");
191         checkClassForModifier(node, isVolatile(node.getModifiers()), "volatile");
192         checkClassForModifier(node, isNative(node.getModifiers()), "native");
193         if (!(node instanceof InnerClassNode)) {
194             checkClassForModifier(node, isStatic(node.getModifiers()), "static");
195             checkClassForModifier(node, isPrivate(node.getModifiers()), "private");
196         }
197         // don't check synchronized here as it overlaps with ACC_SUPER
198     }

200     private void checkMethodForModifier(MethodNode node, boolean condition, String modifierName) {
201         if (!condition) return;
202         addError("The " + getDescription(node) + " has an incorrect modifier " + modifierName + ".", node);
203     }

205     private void checkClassForModifier(ClassNode node, boolean condition, String modifierName) {
206         if (!condition) return;
207         addError("The " + getDescription(node) + " has an incorrect modifier " + modifierName + ".", node);
208     }

210     private String getDescription(ClassNode node) {
211         return (node.isInterface() ? (Traits.isTrait(node)?"trait":"interface") : "class") + " '" + node.getName() + "'";

222     private void checkAbstractDeclaration(MethodNode methodNode) {
223         if (!methodNode.isAbstract()) return;
224         if (isAbstract(currentClass.getModifiers())) return;
225         addError("Can't have an abstract method in a non-abstract class." +
226                 " The " + getDescription(currentClass) + " must be declared abstract or the method '" +
227                 methodNode.getTypeDescriptor() + "' must not be abstract.", methodNode);
228     }

230     private void checkClassForOverwritingFinal(ClassNode cn) {
231         ClassNode superCN = cn.getSuperClass();
232         if (superCN == null) return;
233         if (!isFinal(superCN.getModifiers())) return;
234         StringBuilder msg = new StringBuilder();
235         msg.append("You are not allowed to overwrite the final ");
236         msg.append(getDescription(superCN));
237         msg.append(".");
238         addError(msg.toString(), cn);
239     }

241     private void checkImplementsAndExtends(ClassNode node) {
242         ClassNode cn = node.getSuperClass();
243         if (cn.isInterface() && !node.isInterface()) {
244             addError("You are not allowed to extend the " + getDescription(cn) + ", use implements instead.", node);
245         }
246         for (ClassNode anInterface : node.getInterfaces()) {
247             cn = anInterface;
248             if (!cn.isInterface()) {
249                 addError("You are not allowed to implement the " + getDescription(cn) + ", use extends instead.", node);
250             }
251         }
252     }
253 
254     private void checkMethodsForIncorrectModifiers(ClassNode cn) {
255         if (!cn.isInterface()) return;
256         for (MethodNode method : cn.getMethods()) {
257             if (method.isFinal()) {
258                 addError("The " + getDescription(method) + " from " + getDescription(cn) +
259                         " must not be final. It is by definition abstract.", method);
260             }
261             if (method.isStatic() && !isConstructor(method)) {
262                 addError("The " + getDescription(method) + " from " + getDescription(cn) +
263                         " must not be static. Only fields may be static in an interface.", method);
264             }
265         }
266     }
267 
268     private void checkMethodsForWeakerAccess(ClassNode cn) {
269         for (MethodNode method : cn.getMethods()) {
270             checkMethodForWeakerAccessPrivileges(method, cn);
271         }
272     }

274     private boolean isConstructor(MethodNode method) {
275         return method.getName().equals("<clinit>");
276     }
277 
278     private void checkMethodsForOverridingFinal(ClassNode cn) {
279         for (MethodNode method : cn.getMethods()) {
280             Parameter[] params = method.getParameters();
281             for (MethodNode superMethod : cn.getSuperClass().getMethods(method.getName())) {
282                 Parameter[] superParams = superMethod.getParameters();
283                 if (!hasEqualParameterTypes(params, superParams)) continue;
284                 if (!superMethod.isFinal()) break;
285                 addInvalidUseOfFinalError(method, params, superMethod.getDeclaringClass());
286                 return;
287             }
288         }
289     }
290 
291     private void addInvalidUseOfFinalError(MethodNode method, Parameter[] parameters, ClassNode superCN) {
292         StringBuilder msg = new StringBuilder();
293         msg.append("You are not allowed to override the final method ").append(method.getName());
294         msg.append("(");
295         boolean needsComma = false;
296         for (Parameter parameter : parameters) {
297             if (needsComma) {
298                 msg.append(",");
299             } else {
300                 needsComma = true;
301             }
302             msg.append(parameter.getType());
303         }
304         msg.append(") from ").append(getDescription(superCN));
305         msg.append(".");
306         addError(msg.toString(), method);
307     }
308 
309     private void addWeakerAccessError(ClassNode cn, MethodNode method, Parameter[] parameters, MethodNode superMethod) {
310         StringBuilder msg = new StringBuilder();
311         msg.append(method.getName());
312         msg.append("(");
313         boolean needsComma = false;
314         for (Parameter parameter : parameters) {
315             if (needsComma) {
316                 msg.append(",");
317             } else {
318                 needsComma = true;
319             }
320             msg.append(parameter.getType());
321         }
322         msg.append(") in ");
323         msg.append(cn.getName());
324         msg.append(" cannot override ");
325         msg.append(superMethod.getName());
326         msg.append(" in ");
327         msg.append(superMethod.getDeclaringClass().getName());
328         msg.append("; attempting to assign weaker access privileges; was ");
329         msg.append(superMethod.isPublic() ? "public" : "protected");
330         addError(msg.toString(), method);
331     }

333     private boolean hasEqualParameterTypes(Parameter[] first, Parameter[] second) {
334         if (first.length != second.length) return false;
335         for (int i = 0; i < first.length; i++) {
336             String ft = first[i].getType().getName();
337             String st = second[i].getType().getName();
338             if (ft.equals(st)) continue;
339             return false;
340         }
341         return true;
342     }

348     public void visitMethod(MethodNode node) {
349         inConstructor = false;
350         inStaticConstructor = node.isStaticConstructor();
351         checkAbstractDeclaration(node);
352         checkRepetitiveMethod(node);
353         checkOverloadingPrivateAndPublic(node);
354         checkMethodModifiers(node);
355         checkGenericsUsage(node, node.getParameters());
356         checkGenericsUsage(node, node.getReturnType());
357         super.visitMethod(node);
358     }
359 
360     private void checkMethodModifiers(MethodNode node) {
361         // don't check volatile here as it overlaps with ACC_BRIDGE
362         // additional modifiers not allowed for interfaces
363         if ((this.currentClass.getModifiers() & ACC_INTERFACE) != 0) {
364             checkMethodForModifier(node, isStrict(node.getModifiers()), "strictfp");
365             checkMethodForModifier(node, isSynchronized(node.getModifiers()), "synchronized");
366             checkMethodForModifier(node, isNative(node.getModifiers()), "native");
367         }
368     }
369 
370     private void checkMethodForWeakerAccessPrivileges(MethodNode mn, ClassNode cn) {
371         if (mn.isPublic()) return;
372         Parameter[] params = mn.getParameters();
373         for (MethodNode superMethod : cn.getSuperClass().getMethods(mn.getName())) {
374             Parameter[] superParams = superMethod.getParameters();
375             if (!hasEqualParameterTypes(params, superParams)) continue;
376             if ((mn.isPrivate() && !superMethod.isPrivate()) ||
377                     (mn.isProtected() && superMethod.isPublic())) {
378                 addWeakerAccessError(cn, mn, params, superMethod);
379                 return;
380             }
381         }
382     }

384     private void checkOverloadingPrivateAndPublic(MethodNode node) {
385         if (isConstructor(node)) return;
386         boolean hasPrivate = node.isPrivate();
387         boolean hasPublic = node.isPublic();
388         for (MethodNode method : currentClass.getMethods(node.getName())) {
389             if (method == node) continue;
390             if (!method.getDeclaringClass().equals(node.getDeclaringClass())) continue;
391             if (method.isPublic() || method.isProtected()) {
392                 hasPublic = true;
393             } else {
394                 hasPrivate = true;
395             }
396             if (hasPrivate && hasPublic) break;
397         }
398         if (hasPrivate && hasPublic) {
399             addError("Mixing private and public/protected methods of the same name causes multimethods to be disabled and is forbidden to avoid surprising behaviour. Renaming the private methods will solve the problem.", node);
400         }
401     }

403     private void checkRepetitiveMethod(MethodNode node) {
404         if (isConstructor(node)) return;
405         for (MethodNode method : currentClass.getMethods(node.getName())) {
406             if (method == node) continue;
407             if (!method.getDeclaringClass().equals(node.getDeclaringClass())) continue;
408             Parameter[] p1 = node.getParameters();
409             Parameter[] p2 = method.getParameters();
410             if (p1.length != p2.length) continue;
411             addErrorIfParamsAndReturnTypeEqual(p2, p1, node, method);
412         }
413     }
414 
415     private void addErrorIfParamsAndReturnTypeEqual(Parameter[] p2, Parameter[] p1,
416                                                     MethodNode node, MethodNode element) {
417         boolean isEqual = true;
418         for (int i = 0; i < p2.length; i++) {
419             isEqual &= p1[i].getType().equals(p2[i].getType());
420             if (!isEqual) break;
421         }
422         isEqual &= node.getReturnType().equals(element.getReturnType());
423         if (isEqual) {
424             addError("Repetitive method name/signature for " + getDescription(node) +
425                     " in " + getDescription(currentClass) + ".", node);
426         }
427     }

429     public void visitField(FieldNode node) {
430         if (currentClass.getDeclaredField(node.getName()) != node) {
431             addError("The " + getDescription(node) + " is declared multiple times.", node);
432         }
433         checkInterfaceFieldModifiers(node);
434         checkGenericsUsage(node, node.getType());
435         super.visitField(node);
436     }

438     public void visitProperty(PropertyNode node) {
439         checkDuplicateProperties(node);
440         checkGenericsUsage(node, node.getType());
441         super.visitProperty(node);
442     }

444     private void checkDuplicateProperties(PropertyNode node) {
445         ClassNode cn = node.getDeclaringClass();
446         String name = node.getName();
447         String getterName = "get" + MetaClassHelper.capitalize(name);
448         if (Character.isUpperCase(name.charAt(0))) {
449             for (PropertyNode propNode : cn.getProperties()) {
450                 String otherName = propNode.getField().getName();
451                 String otherGetterName = "get" + MetaClassHelper.capitalize(otherName);
452                 if (node != propNode && getterName.equals(otherGetterName)) {
453                     String msg = "The field " + name + " and " + otherName + " on the class " +
454                             cn.getName() + " will result in duplicate JavaBean properties, which is not allowed";
455                     addError(msg, node);
456                 }
457             }
458         }
459     }

461     private void checkInterfaceFieldModifiers(FieldNode node) {
462         if (!currentClass.isInterface()) return;
463         if ((node.getModifiers() & (ACC_PUBLIC | ACC_STATIC | ACC_FINAL)) == 0 ||
464                 (node.getModifiers() & (ACC_PRIVATE | ACC_PROTECTED)) != 0) {
465             addError("The " + getDescription(node) + " is not 'public static final' but is defined in " +
466                     getDescription(currentClass) + ".", node);
467         }
468     }

470     public void visitBinaryExpression(BinaryExpression expression) {
471         if (expression.getOperation().getType() == Types.LEFT_SQUARE_BRACKET &&
472                 expression.getRightExpression() instanceof MapEntryExpression) {
473             addError("You tried to use a map entry for an index operation, this is not allowed. " +
474                     "Maybe something should be set in parentheses or a comma is missing?",
475                     expression.getRightExpression());
476         }
477         super.visitBinaryExpression(expression);
478 
479         switch (expression.getOperation().getType()) {
480             case Types.EQUAL: // = assignment
481             case Types.BITWISE_AND_EQUAL:
482             case Types.BITWISE_OR_EQUAL:
483             case Types.BITWISE_XOR_EQUAL:
484             case Types.PLUS_EQUAL:
485             case Types.MINUS_EQUAL:
486             case Types.MULTIPLY_EQUAL:
487             case Types.DIVIDE_EQUAL:
488             case Types.INTDIV_EQUAL:
489             case Types.MOD_EQUAL:
490             case Types.POWER_EQUAL:
491             case Types.LEFT_SHIFT_EQUAL:
492             case Types.RIGHT_SHIFT_EQUAL:
493             case Types.RIGHT_SHIFT_UNSIGNED_EQUAL:
494                 checkFinalFieldAccess(expression.getLeftExpression());
495                 break;
496             default:
497                 break;
498         }
499     }

501     private void checkFinalFieldAccess(Expression expression) {
502         if (!(expression instanceof VariableExpression) && !(expression instanceof PropertyExpression)) return;
503         Variable v = null;
504         if (expression instanceof VariableExpression) {
505             VariableExpression ve = (VariableExpression) expression;
506             v = ve.getAccessedVariable();
507         } else {
508             PropertyExpression propExp = ((PropertyExpression) expression);
509             Expression objectExpression = propExp.getObjectExpression();
510             if (objectExpression instanceof VariableExpression) {
511                 VariableExpression varExp = (VariableExpression) objectExpression;
512                 if (varExp.isThisExpression()) {
513                     v = currentClass.getDeclaredField(propExp.getPropertyAsString());
514                 }
515             }
516         }
517         if (v instanceof FieldNode) {
518             FieldNode fn = (FieldNode) v;
519 
520             /*
521              *  if it is static final but not accessed inside a static constructor, or,
522              *  if it is an instance final but not accessed inside a instance constructor, it is an error
523              */
524             boolean isFinal = fn.isFinal();
525             boolean isStatic = fn.isStatic();
526             boolean error = isFinal && ((isStatic && !inStaticConstructor) || (!isStatic && !inConstructor));
527 
528             if (error) addError("cannot modify" + (isStatic ? " static" : "") + " final field '" + fn.getName() +
529                     "' outside of " + (isStatic ? "static initialization block." : "constructor."), expression);
530         }
531     }

533     public void visitConstructor(ConstructorNode node) {
534         inConstructor = true;
535         inStaticConstructor = node.isStaticConstructor();
536         checkGenericsUsage(node, node.getParameters());
537         super.visitConstructor(node);
538     }

540     public void visitCatchStatement(CatchStatement cs) {
541         if (!(cs.getExceptionType().isDerivedFrom(ClassHelper.make(Throwable.class)))) {
542             addError("Catch statement parameter type is not a subclass of Throwable.", cs);
543         }
544         super.visitCatchStatement(cs);
545     }

547     public void visitMethodCallExpression(MethodCallExpression mce) {
548         super.visitMethodCallExpression(mce);
549         Expression aexp = mce.getArguments();
550         if (aexp instanceof TupleExpression) {
551             TupleExpression arguments = (TupleExpression) aexp;
552             for (Expression e : arguments.getExpressions()) {
553                 checkForInvalidDeclaration(e);
554             }
555         } else {
556             checkForInvalidDeclaration(aexp);
557         }
558     }

560     @Override
561     public void visitDeclarationExpression(DeclarationExpression expression) {
562         super.visitDeclarationExpression(expression);
563         if (expression.isMultipleAssignmentDeclaration()) return;
564         checkInvalidDeclarationModifier(expression, ACC_ABSTRACT, "abstract");
565         checkInvalidDeclarationModifier(expression, ACC_NATIVE, "native");
566         checkInvalidDeclarationModifier(expression, ACC_PRIVATE, "private");
567         checkInvalidDeclarationModifier(expression, ACC_PROTECTED, "protected");
568         checkInvalidDeclarationModifier(expression, ACC_PUBLIC, "public");
569         checkInvalidDeclarationModifier(expression, ACC_STATIC, "static");
570         checkInvalidDeclarationModifier(expression, ACC_STRICT, "strictfp");
571         checkInvalidDeclarationModifier(expression, ACC_SYNCHRONIZED, "synchronized");
572         checkInvalidDeclarationModifier(expression, ACC_TRANSIENT, "transient");
573         checkInvalidDeclarationModifier(expression, ACC_VOLATILE, "volatile");
574     }
575 
576     private void checkInvalidDeclarationModifier(DeclarationExpression expression, int modifier, String modName) {
577         if ((expression.getVariableExpression().getModifiers() & modifier) != 0) {
578             addError("Modifier '" + modName + "' not allowed here.", expression);
579         }
580     }

582     private void checkForInvalidDeclaration(Expression exp) {
583         if (!(exp instanceof DeclarationExpression)) return;
584         addError("Invalid use of declaration inside method call.", exp);
585     }

587     public void visitConstantExpression(ConstantExpression expression) {
588         super.visitConstantExpression(expression);
589         checkStringExceedingMaximumLength(expression);
590     }

592     public void visitGStringExpression(GStringExpression expression) {
593         super.visitGStringExpression(expression);
594         for (ConstantExpression ce : expression.getStrings()) {
595             checkStringExceedingMaximumLength(ce);
596         }
597     }
598 
599     private void checkStringExceedingMaximumLength(ConstantExpression expression) {
600         Object value = expression.getValue();
601         if (value instanceof String) {
602             String s = (String) value;
603             if (s.length() > 65535) {
604                 addError("String too long. The given string is " + s.length() + " Unicode code units long, but only a maximum of 65535 is allowed.", expression);
605             }
606         }
607     }

609     private void checkGenericsUsage(ASTNode ref, ClassNode[] nodes) {
610         for (ClassNode node : nodes) {
611             checkGenericsUsage(ref, node);
612         }
613     }

615     private void checkGenericsUsage(ASTNode ref, Parameter[] params) {
616         for (Parameter p : params) {
617             checkGenericsUsage(ref, p.getType());
618         }
619     }

621     private void checkGenericsUsage(ASTNode ref, ClassNode node) {
622         if (node.isArray()) {
623             checkGenericsUsage(ref, node.getComponentType());
624         } else if (!node.isRedirectNode() && node.isUsingGenerics()) {
625             addError(   
626                     "A transform used a generics containing ClassNode "+ node + " " +
627                     "for "+getRefDescriptor(ref) + 
628                     "directly. You are not supposed to do this. " +
629                     "Please create a new ClassNode referring to the old ClassNode " +
630                     "and use the new ClassNode instead of the old one. Otherwise " +
631                     "the compiler will create wrong descriptors and a potential " +
632                     "NullPointerException in TypeResolver in the OpenJDK. If this is " +
633                     "not your own doing, please report this bug to the writer of the " +
634                     "transform.",
635                     ref);
636         }
637     }
638 
639     private String getRefDescriptor(ASTNode ref) {
640         if (ref instanceof FieldNode) {
641             FieldNode f = (FieldNode) ref;
642             return "the field "+f.getName()+" ";
643         } else if (ref instanceof PropertyNode) {
644             PropertyNode p = (PropertyNode) ref;
645             return "the property "+p.getName()+" ";
646         } else if (ref instanceof ConstructorNode) {
647             return "the constructor "+ref.getText()+" ";
648         } else if (ref instanceof MethodNode) {
649             return "the method "+ref.getText()+" ";
650         } else if (ref instanceof ClassNode) {
651             return "the super class "+ref+" ";
652         }
653         return "<unknown with class "+ref.getClass()+"> ";
654     }
