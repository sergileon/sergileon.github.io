63      /**
64       * Given a parameterized type and a generic type information, aligns actual type parameters. For example, if a
65       * class uses generic type <pre>&lt;T,U,V&gt;</pre> (redirectGenericTypes), is used with actual type parameters
66       * <pre>&lt;java.lang.String, U,V&gt;</pre>, then a class or interface using generic types <pre>&lt;T,V&gt;</pre>
67       * will be aligned to <pre>&lt;java.lang.String,V&gt;</pre>
68       * @param redirectGenericTypes the type arguments or the redirect class node
69       * @param parameterizedTypes the actual type arguments used on this class node
70       * @param alignmentTarget the generic type arguments to which we want to align to
71       * @return aligned type arguments
72       * @deprecated You shouldn't call this method because it is inherently unreliable
73       */
74      @Deprecated
75      public static GenericsType[] alignGenericTypes(final GenericsType[] redirectGenericTypes, final GenericsType[] parameterizedTypes, final GenericsType[] alignmentTarget) {
76          if (alignmentTarget==null) return EMPTY_GENERICS_ARRAY;
77          if (parameterizedTypes==null || parameterizedTypes.length==0) return alignmentTarget;
78          GenericsType[] generics = new GenericsType[alignmentTarget.length];
79          for (int i = 0, scgtLength = alignmentTarget.length; i < scgtLength; i++) {
80              final GenericsType currentTarget = alignmentTarget[i];
81              GenericsType match = null;
82              if (redirectGenericTypes!=null) {
83                  for (int j = 0; j < redirectGenericTypes.length && match == null; j++) {
84                      GenericsType redirectGenericType = redirectGenericTypes[j];
85                      if (redirectGenericType.isCompatibleWith(currentTarget.getType())) {
86                          if (currentTarget.isPlaceholder() && redirectGenericType.isPlaceholder() && !currentTarget.getName().equals(redirectGenericType.getName())) {
87                              // check if there's a potential better match
88                              boolean skip = false;
89                              for (int k=j+1; k<redirectGenericTypes.length && !skip; k++) {
90                                  GenericsType ogt = redirectGenericTypes[k];
91                                  if (ogt.isPlaceholder() && ogt.isCompatibleWith(currentTarget.getType()) && ogt.getName().equals(currentTarget.getName())) {
92                                      skip = true;
93                                  }
94                              }
95                              if (skip) continue;
96                          }
97                          match = parameterizedTypes[j];
98                          if (currentTarget.isWildcard()) {
99                              // if alignment target is a wildcard type
100                             // then we must make best effort to return a parameterized
101                             // wildcard
102                             ClassNode lower = currentTarget.getLowerBound()!=null?match.getType():null;
103                             ClassNode[] currentUpper = currentTarget.getUpperBounds();
104                             ClassNode[] upper = currentUpper !=null?new ClassNode[currentUpper.length]:null;
105                             if (upper!=null) {
106                                 for (int k = 0; k < upper.length; k++) {
107                                     upper[k] = currentUpper[k].isGenericsPlaceHolder()?match.getType():currentUpper[k];
108                                 }
109                             }
110                             match = new GenericsType(ClassHelper.makeWithoutCaching("?"), upper, lower);
111                             match.setWildcard(true);
112                         }
113                     }
114                 }
115             }
116             if (match == null) {
117                 match = currentTarget;
118             }
119             generics[i]=match;
120         }
121         return generics;
122     }

143     /**
144      * For a given classnode, fills in the supplied map with the parameterized
145      * types it defines.
146      * @param node
147      * @param map
148      */
149     public static void extractPlaceholders(ClassNode node, Map<String, GenericsType> map) {
150         if (node == null) return;
151 
152         if (node.isArray()) {
153             extractPlaceholders(node.getComponentType(), map);
154             return;
155         }
156 
157         if (!node.isUsingGenerics() || !node.isRedirectNode()) return;
158         GenericsType[] parameterized = node.getGenericsTypes();
159         if (parameterized == null || parameterized.length == 0) return;
160         GenericsType[] redirectGenericsTypes = node.redirect().getGenericsTypes();
161         if (redirectGenericsTypes==null) redirectGenericsTypes = parameterized;
162         for (int i = 0; i < redirectGenericsTypes.length; i++) {
163             GenericsType redirectType = redirectGenericsTypes[i];
164             if (redirectType.isPlaceholder()) {
165                 String name = redirectType.getName();
166                 if (!map.containsKey(name)) {
167                     GenericsType value = parameterized[i];
168                     map.put(name, value);
169                     if (value.isWildcard()) {
170                         ClassNode lowerBound = value.getLowerBound();
171                         if (lowerBound!=null) {
172                             extractPlaceholders(lowerBound, map);
173                         }
174                         ClassNode[] upperBounds = value.getUpperBounds();
175                         if (upperBounds!=null) {
176                             for (ClassNode upperBound : upperBounds) {
177                                 extractPlaceholders(upperBound, map);
178                             }
179                         }
180                     } else if (!value.isPlaceholder()) {
181                         extractPlaceholders(value.getType(), map);
182                     }
183                 }
184             }
185         }
186     }

202     /**
203      * Interface class nodes retrieved from {@link org.codehaus.groovy.ast.ClassNode#getInterfaces()}
204      * or {@link org.codehaus.groovy.ast.ClassNode#getAllInterfaces()} are returned with generic type
205      * arguments. This method allows returning a parameterized interface given the parameterized class
206      * node which implements this interface.
207      * @param hint the class node where generics types are parameterized
208      * @param target the interface we want to parameterize generics types
209      * @return a parameterized interface class node
210      */
211     public static ClassNode parameterizeType(final ClassNode hint, final ClassNode target) {
212         if (hint.isArray()) {
213             if (target.isArray()) {
214                 return parameterizeType(hint.getComponentType(), target.getComponentType()).makeArray();
215             }
216             return target;
217         }
218         if (!target.equals(hint) && StaticTypeCheckingSupport.implementsInterfaceOrIsSubclassOf(target, hint)) {
219             ClassNode nextSuperClass = ClassHelper.getNextSuperClass(target, hint);
220             if (!hint.equals(nextSuperClass)) {
221                 Map<String, ClassNode> genericsSpec = createGenericsSpec(hint);
222                 extractSuperClassGenerics(hint, nextSuperClass, genericsSpec);
223                 ClassNode result = correctToGenericsSpecRecurse(genericsSpec, nextSuperClass);
224                 return parameterizeType(result, target);
225             }
226         }
227         Map<String, ClassNode> genericsSpec = createGenericsSpec(hint);
228         ClassNode targetRedirect = target.redirect();
229         genericsSpec = createGenericsSpec(targetRedirect, genericsSpec);
230         extractSuperClassGenerics(hint, targetRedirect, genericsSpec);
231         return correctToGenericsSpecRecurse(genericsSpec, targetRedirect);
232 
233     }

284     public static MethodNode correctToGenericsSpec(Map<String,ClassNode> genericsSpec, MethodNode mn) {
285         ClassNode correctedType = correctToGenericsSpecRecurse(genericsSpec, mn.getReturnType());
286         Parameter[] origParameters = mn.getParameters();
287         Parameter[] newParameters = new Parameter[origParameters.length];
288         for (int i = 0; i < origParameters.length; i++) {
289             Parameter origParameter = origParameters[i];
290             newParameters[i] = new Parameter(correctToGenericsSpecRecurse(genericsSpec, origParameter.getType()), origParameter.getName(), origParameter.getInitialExpression());
291         }
292         return new MethodNode(mn.getName(), mn.getModifiers(), correctedType, newParameters, mn.getExceptions(), mn.getCode());
293     }

302     public static ClassNode[] correctToGenericsSpecRecurse(Map<String,ClassNode> genericsSpec, ClassNode[] types) {
303         if (types==null || types.length==1) return types;
304         ClassNode[] newTypes = new ClassNode[types.length];
305         boolean modified = false;
306         for (int i=0; i<types.length; i++) {
307             newTypes[i] = correctToGenericsSpecRecurse(genericsSpec, types[i], new ArrayList<String>());
308             modified = modified || (types[i]!=newTypes[i]);
309         }
310         if (!modified) return types;
311         return newTypes;
312     }

314     public static ClassNode correctToGenericsSpecRecurse(Map<String,ClassNode> genericsSpec, ClassNode type, List<String> exclusions) {
315         if (type.isArray()) {
316             return correctToGenericsSpecRecurse(genericsSpec, type.getComponentType(), exclusions).makeArray();
317         }
318         if (type.isGenericsPlaceHolder() && !exclusions.contains(type.getUnresolvedName())) {
319             String name = type.getGenericsTypes()[0].getName();
320             type = genericsSpec.get(name);
321             if (type != null && type.isGenericsPlaceHolder() && type.getGenericsTypes() == null) {
322                 ClassNode placeholder = ClassHelper.makeWithoutCaching(type.getUnresolvedName());
323                 placeholder.setGenericsPlaceHolder(true);
324                 type = makeClassSafeWithGenerics(type, new GenericsType(placeholder));
325             }
326         }
327         if (type == null) type = ClassHelper.OBJECT_TYPE;
328         GenericsType[] oldgTypes = type.getGenericsTypes();
329         GenericsType[] newgTypes = GenericsType.EMPTY_ARRAY;
330         if (oldgTypes != null) {
331             newgTypes = new GenericsType[oldgTypes.length];
332             for (int i = 0; i < newgTypes.length; i++) {
333                 GenericsType oldgType = oldgTypes[i];
334                 if (oldgType.isPlaceholder() ) {
335                     if (genericsSpec.get(oldgType.getName())!=null) {
336                         newgTypes[i] = new GenericsType(genericsSpec.get(oldgType.getName()));
337                     } else {
338                         newgTypes[i] = new GenericsType(ClassHelper.OBJECT_TYPE);
339                     }
340                 } else if (oldgType.isWildcard()) {
341                     ClassNode oldLower = oldgType.getLowerBound();
342                     ClassNode lower = oldLower!=null?correctToGenericsSpecRecurse(genericsSpec, oldLower, exclusions):null;
343                     ClassNode[] oldUpper = oldgType.getUpperBounds();
344                     ClassNode[] upper = null;
345                     if (oldUpper!=null) {
346                         upper = new ClassNode[oldUpper.length];
347                         for (int j = 0; j < oldUpper.length; j++) {
348                             upper[j] = correctToGenericsSpecRecurse(genericsSpec,oldUpper[j], exclusions);
349                         }
350                     }
351                     GenericsType fixed = new GenericsType(oldgType.getType(), upper, lower);
352                     fixed.setName(oldgType.getName());
353                     fixed.setWildcard(true);
354                     newgTypes[i] = fixed;
355                 } else {
356                     newgTypes[i] = new GenericsType(correctToGenericsSpecRecurse(genericsSpec,correctToGenericsSpec(genericsSpec, oldgType), exclusions));
357                 }
358             }
359         }
360         return makeClassSafeWithGenerics(type, newgTypes);
361     }

390     public static Map<String,ClassNode> createGenericsSpec(ClassNode current, Map<String,ClassNode> oldSpec) {
391         Map<String,ClassNode> ret = new HashMap<String,ClassNode>(oldSpec);
392         // ret contains the type specs, what we now need is the type spec for the
393         // current class. To get that we first apply the type parameters to the
394         // current class and then use the type names of the current class to reset
395         // the map. Example:
396         //   class A<V,W,X>{}
397         //   class B<T extends Number> extends A<T,Long,String> {}
398         // first we have:    T->Number
399         // we apply it to A<T,Long,String> -> A<Number,Long,String>
400         // resulting in:     V->Number,W->Long,X->String
401 
402         GenericsType[] sgts = current.getGenericsTypes();
403         if (sgts != null) {
404             ClassNode[] spec = new ClassNode[sgts.length];
405             for (int i = 0; i < spec.length; i++) {
406                 spec[i] = correctToGenericsSpec(ret, sgts[i]);
407             }
408             GenericsType[] newGts = current.redirect().getGenericsTypes();
409             if (newGts == null) return ret;
410             ret.clear();
411             for (int i = 0; i < spec.length; i++) {
412                 ret.put(newGts[i].getName(), spec[i]);
413             }
414         }
415         return ret;
416     }

418     public static Map<String,ClassNode> addMethodGenerics(MethodNode current, Map<String,ClassNode> oldSpec) {
419         Map<String,ClassNode> ret = new HashMap<String,ClassNode>(oldSpec);
420         // ret starts with the original type specs, now add gts for the current method if any
421         GenericsType[] sgts = current.getGenericsTypes();
422         if (sgts != null) {
423             for (GenericsType sgt : sgts) {
424                 ret.put(sgt.getName(), sgt.getType());
425             }
426         }
427         return ret;

430     public static void extractSuperClassGenerics(ClassNode type, ClassNode target, Map<String,ClassNode> spec) {
431         // TODO: this method is very similar to StaticTypesCheckingSupport#extractGenericsConnections,
432         // but operates on ClassNodes instead of GenericsType
433         if (target==null || type==target) return;
434         if (type.isArray() && target.isArray()) {
435             extractSuperClassGenerics(type.getComponentType(), target.getComponentType(), spec);
436         } else if (target.isGenericsPlaceHolder() || type.equals(target) || !StaticTypeCheckingSupport.implementsInterfaceOrIsSubclassOf(type, target)) {
437             // structural match route
438             if (target.isGenericsPlaceHolder()) {
439                 spec.put(target.getGenericsTypes()[0].getName(),type);
440             } else {
441                 extractSuperClassGenerics(type.getGenericsTypes(), target.getGenericsTypes(), spec);
442             }
443         } else {
444             // have first to find matching super class or interface
445             Map <String,ClassNode> genSpec = createGenericsSpec(type);
446             ClassNode superClass = ClassHelper.getNextSuperClass(type,target);
447             if (superClass!=null){
448                 ClassNode corrected = GenericsUtils.correctToGenericsSpecRecurse(genSpec, superClass);
449                 extractSuperClassGenerics(corrected, target, spec);
450             } else {
451                 // if we reach here, we have an unhandled case 
452                 throw new GroovyBugError("The type "+type+" seems not to normally extend "+target+". Sorry, I cannot handle this.");
453             }
454         }
455     }

457     private static void extractSuperClassGenerics(GenericsType[] usage, GenericsType[] declaration, Map<String, ClassNode> spec) {
458         // if declaration does not provide generics, there is no connection to make 
459         if (usage==null || declaration==null || declaration.length==0) return;
460         if (usage.length!=declaration.length) return;
461 
462         // both have generics
463         for (int i=0; i<usage.length; i++) {
464             GenericsType ui = usage[i];
465             GenericsType di = declaration[i];
466             if (di.isPlaceholder()) {
467                 spec.put(di.getName(), ui.getType());
468             } else if (di.isWildcard()){
469                 if (ui.isWildcard()) {
470                     extractSuperClassGenerics(ui.getLowerBound(), di.getLowerBound(), spec);
471                     extractSuperClassGenerics(ui.getUpperBounds(), di.getUpperBounds(), spec);
472                 } else {
473                     ClassNode cu = ui.getType();
474                     extractSuperClassGenerics(cu, di.getLowerBound(), spec);
475                     ClassNode[] upperBounds = di.getUpperBounds();
476                     if (upperBounds!=null) {
477                         for (ClassNode cn : upperBounds) {
478                             extractSuperClassGenerics(cu, cn, spec);
479                         }
480                     }
481                 }
482             } else {
483                 extractSuperClassGenerics(ui.getType(), di.getType(), spec);
484             }
485         }
486     }

488     private static void extractSuperClassGenerics(ClassNode[] usage, ClassNode[] declaration, Map<String, ClassNode> spec) {
489         if (usage==null || declaration==null || declaration.length==0) return;
490         // both have generics
491         for (int i=0; i<usage.length; i++) {
492             ClassNode ui = usage[i];
493             ClassNode di = declaration[i];
494             if (di.isGenericsPlaceHolder()) {
495                 spec.put(di.getGenericsTypes()[0].getName(), di);
496             } else if (di.isUsingGenerics()){
497                 extractSuperClassGenerics(ui.getGenericsTypes(), di.getGenericsTypes(), spec);
498             }
499         }
500     }

502     public static ClassNode[] parseClassNodesFromString(
503             final String option,
504             final SourceUnit sourceUnit,
505             final CompilationUnit compilationUnit,
506             final MethodNode mn,
507             final ASTNode usage) {
508         GroovyLexer lexer = new GroovyLexer(new StringReader("DummyNode<" + option + ">"));
509         final GroovyRecognizer rn = GroovyRecognizer.make(lexer);
510         try {
511             rn.classOrInterfaceType(true);
512             final AtomicReference<ClassNode> ref = new AtomicReference<ClassNode>();
513             AntlrParserPlugin plugin = new AntlrParserPlugin() {
514                 @Override
515                 public ModuleNode buildAST(final SourceUnit sourceUnit, final ClassLoader classLoader, final Reduction cst) throws ParserException {
516                     ref.set(makeTypeWithArguments(rn.getAST()));
517                     return null;
518                 }
519             };
520             plugin.buildAST(null, null, null);
521             ClassNode parsedNode = ref.get();
522             // the returned node is DummyNode<Param1, Param2, Param3, ...)
523             GenericsType[] parsedNodeGenericsTypes = parsedNode.getGenericsTypes();
524             if (parsedNodeGenericsTypes == null) {
525                 return null;
526             }
527             ClassNode[] signature = new ClassNode[parsedNodeGenericsTypes.length];
528             for (int i = 0; i < parsedNodeGenericsTypes.length; i++) {
529                 final GenericsType genericsType = parsedNodeGenericsTypes[i];
530                 signature[i] = resolveClassNode(sourceUnit, compilationUnit, mn, usage, genericsType.getType());
531             }
532             return signature;
533         } catch (RecognitionException e) {
534             sourceUnit.addError(new IncorrectTypeHintException(mn, e, usage.getLineNumber(), usage.getColumnNumber()));
535         } catch (TokenStreamException e) {
536             sourceUnit.addError(new IncorrectTypeHintException(mn, e, usage.getLineNumber(), usage.getColumnNumber()));
537         } catch (ParserException e) {
538             sourceUnit.addError(new IncorrectTypeHintException(mn, e, usage.getLineNumber(), usage.getColumnNumber()));
539         }
540         return null;
541     }

543     private static ClassNode resolveClassNode(final SourceUnit sourceUnit, final CompilationUnit compilationUnit, final MethodNode mn, final ASTNode usage, final ClassNode parsedNode) {
544         ClassNode dummyClass = new ClassNode("dummy",0, ClassHelper.OBJECT_TYPE);
545         dummyClass.setModule(new ModuleNode(sourceUnit));
546         dummyClass.setGenericsTypes(mn.getDeclaringClass().getGenericsTypes());
547         MethodNode dummyMN = new MethodNode(
548                 "dummy",
549                 0,
550                 parsedNode,
551                 Parameter.EMPTY_ARRAY,
552                 ClassNode.EMPTY_ARRAY,
553                 EmptyStatement.INSTANCE
554         );
555         dummyMN.setGenericsTypes(mn.getGenericsTypes());
556         dummyClass.addMethod(dummyMN);
557         ResolveVisitor visitor = new ResolveVisitor(compilationUnit) {
558             @Override
559             protected void addError(final String msg, final ASTNode expr) {
560                 sourceUnit.addError(new IncorrectTypeHintException(mn, msg, usage.getLineNumber(), usage.getColumnNumber()));
561             }
562         };
563         visitor.startResolving(dummyClass, sourceUnit);
564         return dummyMN.getReturnType();
565     }

567     /**
568      * transforms generics types from an old context to a new context using the given spec. This method assumes
569      * all generics types will be placeholders. WARNING: The resulting generics types may or may not be placeholders
570      * after the transformation.
571      * @param genericsSpec the generics context information spec
572      * @param oldPlaceHolders the old placeholders
573      * @return the new generics types
574      */
575     public static GenericsType[] applyGenericsContextToPlaceHolders(Map<String, ClassNode> genericsSpec, GenericsType[] oldPlaceHolders) {
576         if (oldPlaceHolders==null || oldPlaceHolders.length==0) return oldPlaceHolders;
577         if (genericsSpec.isEmpty()) return oldPlaceHolders;
578         GenericsType[] newTypes = new GenericsType[oldPlaceHolders.length];
579         for (int i=0; i<oldPlaceHolders.length; i++) {
580             GenericsType old = oldPlaceHolders[i];
581             if (!old.isPlaceholder()) throw new GroovyBugError("Given generics type "+old+" must be a placeholder!");
582             ClassNode fromSpec = genericsSpec.get(old.getName());
583             if (fromSpec!=null) {
584                 if (fromSpec.isGenericsPlaceHolder()) {
585                     ClassNode[] upper = new ClassNode[]{fromSpec.redirect()};
586                     newTypes[i] = new GenericsType(fromSpec, upper, null);
587                 } else {
588                     newTypes[i] = new GenericsType(fromSpec);
589                 }
590             } else {
591                 ClassNode[] upper = old.getUpperBounds();
592                 ClassNode[] newUpper = upper;
593                 if (upper!=null && upper.length>0) {
594                     ClassNode[] upperCorrected = new ClassNode[upper.length];
595                     for (int j=0;j<upper.length;j++) {
596                         upperCorrected[i] = correctToGenericsSpecRecurse(genericsSpec,upper[j]);
597                     }
598                     upper = upperCorrected;
599                 }
600                 ClassNode lower = old.getLowerBound();
601                 ClassNode newLower = correctToGenericsSpecRecurse(genericsSpec,lower);
602                 if (lower==newLower && upper==newUpper) {
603                     newTypes[i] = oldPlaceHolders[i];
604                 } else {
605                     ClassNode newPlaceHolder = ClassHelper.make(old.getName());
606                     GenericsType gt = new GenericsType(newPlaceHolder, newUpper, newLower);
607                     gt.setPlaceholder(true);
608                     newTypes[i] = gt;
609                 }
610             }
611         }
612         return newTypes;
613     }
