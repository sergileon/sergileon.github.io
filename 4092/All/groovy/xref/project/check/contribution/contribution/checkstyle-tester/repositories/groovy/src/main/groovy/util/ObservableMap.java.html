28  /**
29   * Map decorator that will trigger PropertyChangeEvents when a value changes.<br>
30   * An optional Closure may be specified and will work as a filter, if it returns
31   * true the property will trigger an event (if the value indeed changed),
32   * otherwise it won't. The Closure may receive 1 or 2 parameters, the single one
33   * being the value, the other one both the key and value, for example:
34   * <pre>
35   * // skip all properties whose value is a closure
36   * def map = new ObservableMap( {!(it instanceof Closure)} )
37   *
38   * // skip all properties whose name matches a regex
39   * def map = new ObservableMap( { name, value -&gt; !(name =~ /[A-Z+]/) } )
40   * </pre>
41   * The current implementation will trigger specialized events in the following scenarios,
42   * you need not register a different listener as those events extend from PropertyChangeEvent
43   * <ul>
44   * <li>ObservableMap.PropertyAddedEvent - a new property is added to the map</li>
45   * <li>ObservableMap.PropertyRemovedEvent - a property is removed from the map</li>
46   * <li>ObservableMap.PropertyUpdatedEvent - a property changes value (same as regular PropertyChangeEvent)</li>
47   * <li>ObservableMap.PropertyClearedEvent - all properties have been removed from the map</li>
48   * <li>ObservableMap.MultiPropertyEvent - triggered by calling map.putAll(), contains Added|Updated events</li>
49   * </ul>
50   * <p>
51   * <strong>Bound properties</strong>
52   * <ul>
53   * <li><tt>content</tt> - read-only.</li>
54   * <li><tt>size</tt> - read-only.</li>
55   * </ul>
56   *
57   * @author <a href="mailto:aalmiray@users.sourceforge.net">Andres Almiray</a>
58   */
59  public class ObservableMap implements Map {

132     public void clear() {
133         int oldSize = size();
134         Map values = new HashMap();
135         if (!delegate.isEmpty()) {
136             values.putAll(delegate);
137         }
138         delegate.clear();
139         firePropertyClearedEvent(values);
140         fireSizeChangedEvent(oldSize, size());
141     }

175     public Object put(Object key, Object value) {
176         int oldSize = size();
177         Object oldValue = null;
178         boolean newKey = !delegate.containsKey(key);
179         if (test != null) {
180             oldValue = delegate.put(key, value);
181             Object result = null;
182             if (test.getMaximumNumberOfParameters() == 2) {
183                 result = test.call(new Object[]{key, value});
184             } else {
185                 result = test.call(value);
186             }
187             if (result != null && result instanceof Boolean && (Boolean) result) {
188                 if (newKey) {
189                     firePropertyAddedEvent(key, value);
190                     fireSizeChangedEvent(oldSize, size());
191                 } else if (oldValue != value) {
192                     firePropertyUpdatedEvent(key, oldValue, value);
193                 }
194             }
195         } else {
196             oldValue = delegate.put(key, value);
197             if (newKey) {
198                 firePropertyAddedEvent(key, value);
199                 fireSizeChangedEvent(oldSize, size());
200             } else if (oldValue != value) {
201                 firePropertyUpdatedEvent(key, oldValue, value);
202             }
203         }
204         return oldValue;
205     }
206 
207     public void putAll(Map map) {
208         int oldSize = size();
209         if (map != null) {
210             List<PropertyEvent> events = new ArrayList<PropertyEvent>();
211             for (Object o : map.entrySet()) {
212                 Entry entry = (Entry) o;
213 
214                 String key = String.valueOf(entry.getKey());
215                 Object newValue = entry.getValue();
216                 Object oldValue = null;
217 
218                 boolean newKey = !delegate.containsKey(key);
219                 if (test != null) {
220                     oldValue = delegate.put(key, newValue);
221                     Object result = null;
222                     if (test.getMaximumNumberOfParameters() == 2) {
223                         result = test.call(new Object[]{key, newValue});
224                     } else {
225                         result = test.call(newValue);
226                     }
227                     if (result != null && result instanceof Boolean && (Boolean) result) {
228                         if (newKey) {
229                             events.add(new PropertyAddedEvent(this, key, newValue));
230                         } else if (oldValue != newValue) {
231                             events.add(new PropertyUpdatedEvent(this, key, oldValue, newValue));
232                         }
233                     }
234                 } else {
235                     oldValue = delegate.put(key, newValue);
236                     if (newKey) {
237                         events.add(new PropertyAddedEvent(this, key, newValue));
238                     } else if (oldValue != newValue) {
239                         events.add(new PropertyUpdatedEvent(this, key, oldValue, newValue));
240                     }
241                 }
242             }
243             if (events.size() > 0) {
244                 fireMultiPropertyEvent(events);
245                 fireSizeChangedEvent(oldSize, size());
246             }
247         }
248     }
249 
250     public Object remove(Object key) {
251         int oldSize = size();
252         Object result = delegate.remove(key);
253         if (key != null) {
254             firePropertyRemovedEvent(key, result);
255             fireSizeChangedEvent(oldSize, size());
256         }
257         return result;
258     }

302     public enum ChangeType {
303         ADDED, UPDATED, REMOVED, CLEARED, MULTI, NONE;
304 
305         public static final Object oldValue = new Object();
306         public static final Object newValue = new Object();
307 
308         public static ChangeType resolve(int ordinal) {
309             switch (ordinal) {
310                 case 0:
311                     return ADDED;
312                 case 2:
313                     return REMOVED;
314                 case 3:
315                     return CLEARED;
316                 case 4:
317                     return MULTI;
318                 case 5:
319                     return NONE;
320                 case 1:
321                 default:
322                     return UPDATED;
323             }
324         }
325     }
326 
327     public abstract static class PropertyEvent extends PropertyChangeEvent {
328         private ChangeType type;
329 
330         public PropertyEvent(Object source, String propertyName, Object oldValue, Object newValue, ChangeType type) {
331             super(source, propertyName, oldValue, newValue);
332             this.type = type;
333         }
334 
335         public int getType() {
336             return type.ordinal();
337         }
338 
339         public ChangeType getChangeType() {
340             return type;
341         }
342 
343         public String getTypeAsString() {
344             return type.name().toUpperCase();
345         }
346     }

348     public static class PropertyAddedEvent extends PropertyEvent {
349         public PropertyAddedEvent(Object source, String propertyName, Object newValue) {
350             super(source, propertyName, null, newValue, ChangeType.ADDED);
351         }
352     }

354     public static class PropertyUpdatedEvent extends PropertyEvent {

356             super(source, propertyName, oldValue, newValue, ChangeType.UPDATED);
357         }
358     }

360     public static class PropertyRemovedEvent extends PropertyEvent {
361         public PropertyRemovedEvent(Object source, String propertyName, Object oldValue) {
362             super(source, propertyName, oldValue, null, ChangeType.REMOVED);
363         }
364     }
365 
366     public static class PropertyClearedEvent extends PropertyEvent {
367         private Map values = new HashMap();
368 
369         public PropertyClearedEvent(Object source, Map values) {
370             super(source, ObservableMap.CLEARED_PROPERTY, values, null, ChangeType.CLEARED);
371             if (values != null) {
372                 this.values.putAll(values);
373             }
374         }
375 
376         public Map getValues() {
377             return Collections.unmodifiableMap(values);
378         }
379     }
380 
381     public static class MultiPropertyEvent extends PropertyEvent {
382         public static final String MULTI_PROPERTY = "groovy_util_ObservableMap_MultiPropertyEvent_MULTI";
383         private static final PropertyEvent[] EMPTY_PROPERTY_EVENTS = new PropertyEvent[0];
384 
385         private PropertyEvent[] events = EMPTY_PROPERTY_EVENTS;
386 
387         public MultiPropertyEvent(Object source, PropertyEvent[] events) {
388             super(source, MULTI_PROPERTY, ChangeType.oldValue, ChangeType.newValue, ChangeType.MULTI);
389             if (events != null && events.length > 0) {
390                 this.events = new PropertyEvent[events.length];
391                 System.arraycopy(events, 0, this.events, 0, events.length);
392             }
393         }
394 
395         public PropertyEvent[] getEvents() {
396             PropertyEvent[] copy = new PropertyEvent[events.length];
397             System.arraycopy(events, 0, copy, 0, events.length);
398             return copy;
399         }
