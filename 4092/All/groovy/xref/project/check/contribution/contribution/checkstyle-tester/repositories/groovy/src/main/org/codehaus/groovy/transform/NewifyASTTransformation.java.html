51  /**
52   * Handles generation of code for the @Newify annotation.
53   *
54   * @author Paul King
55   */
56  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
57  public class NewifyASTTransformation extends ClassCodeExpressionTransformer implements ASTTransformation {
58      private static final ClassNode MY_TYPE = make(Newify.class);
59      private static final String MY_NAME = MY_TYPE.getNameWithoutPackage();
60      private static final String BASE_BAD_PARAM_ERROR = "Error during @" + MY_NAME +
61              " processing. Annotation parameter must be a class or list of classes but found ";
62      private SourceUnit source;
63      private ListExpression classesToNewify;
64      private DeclarationExpression candidate;
65      private boolean auto;
66  
67      public void visit(ASTNode[] nodes, SourceUnit source) {
68          this.source = source;
69          if (nodes.length != 2 || !(nodes[0] instanceof AnnotationNode) || !(nodes[1] instanceof AnnotatedNode)) {
70              internalError("Expecting [AnnotationNode, AnnotatedClass] but got: " + Arrays.asList(nodes));
71          }
72  
73          AnnotatedNode parent = (AnnotatedNode) nodes[1];
74          AnnotationNode node = (AnnotationNode) nodes[0];
75          if (!MY_TYPE.equals(node.getClassNode())) {
76              internalError("Transformation called from wrong annotation: " + node.getClassNode().getName());
77          }
78  
79          boolean autoFlag = determineAutoFlag(node.getMember("auto"));
80          Expression value = node.getMember("value");
81  
82          if (parent instanceof ClassNode) {
83              newifyClass((ClassNode) parent, autoFlag, determineClasses(value, false));
84          } else if (parent instanceof MethodNode || parent instanceof FieldNode) {
85              newifyMethodOrField(parent, autoFlag, determineClasses(value, false));
86          } else if (parent instanceof DeclarationExpression) {
87              newifyDeclaration((DeclarationExpression) parent, autoFlag, determineClasses(value, true));
88          }
89      }
90  
91      private void newifyDeclaration(DeclarationExpression de, boolean autoFlag, ListExpression list) {
92          ClassNode cNode = de.getDeclaringClass();
93          candidate = de;
94          final ListExpression oldClassesToNewify = classesToNewify;
95          final boolean oldAuto = auto;
96          classesToNewify = list;
97          auto = autoFlag;
98          super.visitClass(cNode);
99          classesToNewify = oldClassesToNewify;
100         auto = oldAuto;
101     }
102 
103     private boolean determineAutoFlag(Expression autoExpr) {
104         return !(autoExpr instanceof ConstantExpression && ((ConstantExpression) autoExpr).getValue().equals(false));
105     }
106 
107     /** allow non-strict mode in scripts because parsing not complete at that point */
108     private ListExpression determineClasses(Expression expr, boolean searchSourceUnit) {
109         ListExpression list = new ListExpression();
110         if (expr instanceof ClassExpression) {
111             list.addExpression(expr);
112         } else if (expr instanceof VariableExpression && searchSourceUnit) {
113             VariableExpression ve = (VariableExpression) expr;
114             ClassNode fromSourceUnit = getSourceUnitClass(ve);
115             if (fromSourceUnit != null) {
116                 ClassExpression found = classX(fromSourceUnit);
117                 found.setSourcePosition(ve);
118                 list.addExpression(found);
119             } else {
120                 addError(BASE_BAD_PARAM_ERROR + "an unresolvable reference to '" + ve.getName() + "'.", expr);
121             }
122         } else if (expr instanceof ListExpression) {
123             list = (ListExpression) expr;
124             final List<Expression> expressions = list.getExpressions();
125             for (int i = 0; i < expressions.size(); i++) {
126                 Expression next = expressions.get(i);
127                 if (next instanceof VariableExpression && searchSourceUnit) {
128                     VariableExpression ve = (VariableExpression) next;
129                     ClassNode fromSourceUnit = getSourceUnitClass(ve);
130                     if (fromSourceUnit != null) {
131                         ClassExpression found = classX(fromSourceUnit);
132                         found.setSourcePosition(ve);
133                         expressions.set(i, found);
134                     } else {
135                         addError(BASE_BAD_PARAM_ERROR + "a list containing an unresolvable reference to '" + ve.getName() + "'.", next);
136                     }
137                 } else if (!(next instanceof ClassExpression)) {
138                     addError(BASE_BAD_PARAM_ERROR + "a list containing type: " + next.getType().getName() + ".", next);
139                 }
140             }
141             checkDuplicateNameClashes(list);
142         } else if (expr != null) {
143             addError(BASE_BAD_PARAM_ERROR + "a type: " + expr.getType().getName() + ".", expr);
144         }
145         return list;
146     }
147 
148     private ClassNode getSourceUnitClass(VariableExpression ve) {
149         List<ClassNode> classes = source.getAST().getClasses();
150         for (ClassNode classNode : classes) {
151             if (classNode.getNameWithoutPackage().equals(ve.getName())) return classNode;
152         }
153         return null;
154     }
155 
156     public Expression transform(Expression expr) {
157         if (expr == null) return null;
158         if (expr instanceof MethodCallExpression && candidate == null) {
159             MethodCallExpression mce = (MethodCallExpression) expr;
160             Expression args = transform(mce.getArguments());
161             if (isNewifyCandidate(mce)) {
162                 Expression transformed = transformMethodCall(mce, args);
163                 transformed.setSourcePosition(mce);
164                 return transformed;
165             }
166             Expression method = transform(mce.getMethod());
167             Expression object = transform(mce.getObjectExpression());
168             MethodCallExpression transformed = callX(object, method, args);
169             transformed.setSourcePosition(mce);
170             return transformed;
171         } else if (expr instanceof ClosureExpression) {
172             ClosureExpression ce = (ClosureExpression) expr;
173             ce.getCode().visit(this);
174         } else if (expr instanceof ConstructorCallExpression) {
175             ConstructorCallExpression cce = (ConstructorCallExpression) expr;
176             if (cce.isUsingAnonymousInnerClass()) {
177                 cce.getType().visitContents(this);
178             }
179         } else if (expr instanceof DeclarationExpression) {
180             DeclarationExpression de = (DeclarationExpression) expr;
181             if (de == candidate || auto) {
182                 candidate = null;
183                 Expression left = de.getLeftExpression();
184                 Expression right = transform(de.getRightExpression());
185                 DeclarationExpression newDecl = new DeclarationExpression(left, de.getOperation(), right);
186                 newDecl.addAnnotations(de.getAnnotations());
187                 return newDecl;
188             }
189             return de;
190         }
191         return expr.transformExpression(this);
192     }
193 
194     private void newifyClass(ClassNode cNode, boolean autoFlag, ListExpression list) {
195         String cName = cNode.getName();
196         if (cNode.isInterface()) {
197             addError("Error processing interface '" + cName + "'. @"
198                     + MY_NAME + " not allowed for interfaces.", cNode);
199         }
200         final ListExpression oldClassesToNewify = classesToNewify;
201         final boolean oldAuto = auto;
202         classesToNewify = list;
203         auto = autoFlag;
204         super.visitClass(cNode);
205         classesToNewify = oldClassesToNewify;
206         auto = oldAuto;
207     }
208 
209     private void newifyMethodOrField(AnnotatedNode parent, boolean autoFlag, ListExpression list) {
210         final ListExpression oldClassesToNewify = classesToNewify;
211         final boolean oldAuto = auto;
212         checkClassLevelClashes(list);
213         checkAutoClash(autoFlag, parent);
214         classesToNewify = list;
215         auto = autoFlag;
216         if (parent instanceof FieldNode) {
217             super.visitField((FieldNode) parent);
218         } else {
219             super.visitMethod((MethodNode) parent);
220         }
221         classesToNewify = oldClassesToNewify;
222         auto = oldAuto;
223     }
224 
225     private void checkDuplicateNameClashes(ListExpression list) {
226         final Set<String> seen = new HashSet<String>();
227         @SuppressWarnings("unchecked")
228         final List<ClassExpression> classes = (List)list.getExpressions();
229         for (ClassExpression ce : classes) {
230             final String name = ce.getType().getNameWithoutPackage();
231             if (seen.contains(name)) {
232                 addError("Duplicate name '" + name + "' found during @" + MY_NAME + " processing.", ce);
233             }
234             seen.add(name);
235         }
236     }
237 
238     private void checkAutoClash(boolean autoFlag, AnnotatedNode parent) {
239         if (auto && !autoFlag) {
240             addError("Error during @" + MY_NAME + " processing. The 'auto' flag can't be false at " +
241                     "method/constructor/field level if it is true at the class level.", parent);
242         }
243     }
244 
245     private void checkClassLevelClashes(ListExpression list) {
246         @SuppressWarnings("unchecked")
247         final List<ClassExpression> classes = (List)list.getExpressions();
248         for (ClassExpression ce : classes) {
249             final String name = ce.getType().getNameWithoutPackage();
250             if (findClassWithMatchingBasename(name)) {
251                 addError("Error during @" + MY_NAME + " processing. Class '" + name + "' can't appear at " +
252                         "method/constructor/field level if it already appears at the class level.", ce);
253             }
254         }
255     }
256 
257     private boolean findClassWithMatchingBasename(String nameWithoutPackage) {
258         if (classesToNewify == null) return false;
259         @SuppressWarnings("unchecked")
260         final List<ClassExpression> classes = (List)classesToNewify.getExpressions();
261         for (ClassExpression ce : classes) {
262             if (ce.getType().getNameWithoutPackage().equals(nameWithoutPackage)) {
263                 return true;
264             }
265         }
266         return false;
267     }
268 
269     private boolean isNewifyCandidate(MethodCallExpression mce) {
270         return mce.getObjectExpression() == VariableExpression.THIS_EXPRESSION
271                 || (auto && isNewMethodStyle(mce));
272     }
273 
274     private boolean isNewMethodStyle(MethodCallExpression mce) {
275         final Expression obj = mce.getObjectExpression();
276         final Expression meth = mce.getMethod();
277         return (obj instanceof ClassExpression && meth instanceof ConstantExpression
278                 && ((ConstantExpression) meth).getValue().equals("new"));
279     }
280 
281     private Expression transformMethodCall(MethodCallExpression mce, Expression args) {
282         ClassNode classType;
283         if (isNewMethodStyle(mce)) {
284             classType = mce.getObjectExpression().getType();
285         } else {
286             classType = findMatchingCandidateClass(mce);
287         }
288         if (classType != null) {
289             return new ConstructorCallExpression(classType, args);
290         }
291         // set the args as they might have gotten Newify transformed GROOVY-3491
292         mce.setArguments(args);
293         return mce;
294     }
295 
296     private ClassNode findMatchingCandidateClass(MethodCallExpression mce) {
297         if (classesToNewify == null) return null;
298         @SuppressWarnings("unchecked")
299         List<ClassExpression> classes = (List)classesToNewify.getExpressions();
300         for (ClassExpression ce : classes) {
301             final ClassNode type = ce.getType();
302             if (type.getNameWithoutPackage().equals(mce.getMethodAsString())) {
303                 return type;
304             }
305         }
306         return null;
307     }
308 
309     private void internalError(String message) {
310         throw new GroovyBugError("Internal error: " + message);
311     }
312 
313     protected SourceUnit getSourceUnit() {
314         return source;
315     }
