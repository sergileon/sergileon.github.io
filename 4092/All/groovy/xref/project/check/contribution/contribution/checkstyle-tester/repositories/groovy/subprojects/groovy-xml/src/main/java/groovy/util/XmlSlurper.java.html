56  /**
57   * Parse XML into a document tree that may be traversed similar to XPath
58   * expressions.  For example:
59   * <pre>
60   * def rootNode = new XmlSlurper().parseText(
61   *    '&lt;root&gt;&lt;one a1="uno!"/&gt;&lt;two&gt;Some text!&lt;/two&gt;&lt;/root&gt;' )
62   *
63   * assert rootNode.name() == 'root'
64   * assert rootNode.one[0].@a1 == 'uno!'
65   * assert rootNode.two.text() == 'Some text!'
66   * rootNode.children().each { assert it.name() in ['one','two'] }
67   * </pre>
68   * <p>
69   * Note that in some cases, a 'selector' expression may not resolve to a
70   * single node.  For example:
71   * <pre>
72   * def rootNode = new XmlSlurper().parseText(
73   *    '''&lt;root&gt;
74   *         &lt;a&gt;one!&lt;/a&gt;
75   *         &lt;a&gt;two!&lt;/a&gt;
76   *       &lt;/root&gt;''' )
77   *
78   * assert rootNode.a.size() == 2
79   * rootNode.a.each { assert it.text() in ['one!','two!'] }
80   * </pre>
81   *
82   * @author John Wilson
83   * @see GPathResult
84   */
85  public class XmlSlurper extends DefaultHandler {
86      private final XMLReader reader;
87      private Node currentNode = null;
88      private final Stack<Node> stack = new Stack<Node>();
89      private final StringBuffer charBuffer = new StringBuffer();
90      private final Map<String, String> namespaceTagHints = new Hashtable<String, String>();
91      private boolean keepIgnorableWhitespace = false;
92      private boolean namespaceAware = false;

106      * 
107      * @param validating <code>true</code> if the parser should validate documents as they are parsed; false otherwise.
108      * @param namespaceAware <code>true</code> if the parser should provide support for XML namespaces; <code>false</code> otherwise.

111      * @throws SAXException for SAX errors.

114         this(validating, namespaceAware, false);

117     /**
118      * Creates a <code>XmlSlurper</code>.
119      * 
120      * @param validating <code>true</code> if the parser should validate documents as they are parsed; false otherwise.
121      * @param namespaceAware <code>true</code> if the parser should provide support for XML namespaces; <code>false</code> otherwise.
122      * @param allowDocTypeDeclaration <code>true</code> if the parser should provide support for DOCTYPE declarations; <code>false</code> otherwise.
123      *
124      * @throws ParserConfigurationException if no parser which satisfies the requested configuration can be created.
125      * @throws SAXException for SAX errors.
126      */
127     public XmlSlurper(final boolean validating, final boolean namespaceAware, boolean allowDocTypeDeclaration) throws ParserConfigurationException, SAXException {
128         SAXParserFactory factory = FactorySupport.createSaxParserFactory();
129         factory.setNamespaceAware(namespaceAware);
130         this.namespaceAware = namespaceAware;
131         factory.setValidating(validating);
132         setQuietly(factory, XMLConstants.FEATURE_SECURE_PROCESSING, true);
133         setQuietly(factory, "http://apache.org/xml/features/disallow-doctype-decl", !allowDocTypeDeclaration);
134         reader = factory.newSAXParser().getXMLReader();
135     }

137     public XmlSlurper(final XMLReader reader) {
138         this.reader = reader;
139     }
140 
141     public XmlSlurper(final SAXParser parser) throws SAXException {
142         this(parser.getXMLReader());
143     }

145     private void setQuietly(SAXParserFactory factory, String feature, boolean value) {
146         try {
147             factory.setFeature(feature, value);
148         }
149         catch (ParserConfigurationException ignored) { }
150         catch (SAXNotRecognizedException ignored) { }
151         catch (SAXNotSupportedException ignored) { }
152     }

179     /**
180      * @return The GPathResult instance created by consuming a stream of SAX events
181      *         Note if one of the parse methods has been called then this returns null
182      *         Note if this is called more than once all calls after the first will return null
183      */
184     public GPathResult getDocument() {
185         try {
186             // xml namespace is always defined
187             if (namespaceAware) {
188                 namespaceTagHints.put("xml", "http://www.w3.org/XML/1998/namespace");
189             }
190             return new NodeChild(currentNode, null, namespaceTagHints);
191         } finally {
192             currentNode = null;
193         }
194     }

196     /**
197      * Parse the content of the specified input source into a GPathResult object
198      *
199      * @param input the InputSource to parse
200      * @return An object which supports GPath expressions
201      * @throws SAXException Any SAX exception, possibly wrapping another exception.
202      * @throws IOException An IO exception from the parser, possibly from a byte stream
203      *         or character stream supplied by the application.
204      */
205     public GPathResult parse(final InputSource input) throws IOException, SAXException {
206         reader.setContentHandler(this);
207         reader.parse(input);
208         return getDocument();
209     }

211     /**
212      * Parses the content of the given file as XML turning it into a GPathResult object
213      *
214      * @param file the File to parse
215      * @return An object which supports GPath expressions
216      * @throws SAXException Any SAX exception, possibly wrapping another exception.
217      * @throws IOException An IO exception from the parser, possibly from a byte stream
218      *         or character stream supplied by the application.
219      */
220     public GPathResult parse(final File file) throws IOException, SAXException {
221         final FileInputStream fis = new FileInputStream(file);
222         final InputSource input = new InputSource(fis);
223         input.setSystemId("file://" + file.getAbsolutePath());
224         try {
225             return parse(input);
226         } finally {
227             fis.close();
228         }
229     }

390         if (namespaceAware) namespaceTagHints.put(tag, uri);
391     }

393     /* (non-Javadoc)
394     * @see org.xml.sax.ContentHandler#startElement(java.lang.String, java.lang.String, java.lang.String, org.xml.sax.Attributes)
395     */
396     public void startElement(final String namespaceURI, final String localName, final String qName, final Attributes atts) throws SAXException {
397         addCdata();
398 
399         final Map<String, String> attributes = new NamespaceAwareHashMap();
400         final Map<String, String> attributeNamespaces = new HashMap<String, String>();
401 
402         for (int i = atts.getLength() - 1; i != -1; i--) {
403             if (atts.getURI(i).length() == 0) {
404                 attributes.put(atts.getQName(i), atts.getValue(i));
405             } else {
406                 String key = new QName(atts.getURI(i), atts.getLocalName(i)).toString();
407                 attributes.put(key, atts.getValue(i));
408                 attributeNamespaces.put(key, atts.getURI(i));
409             }
410         }
411 
412         final Node newElement;
413 
414         if (namespaceURI.length() == 0) {
415             newElement = new Node(currentNode, qName, attributes, attributeNamespaces, namespaceURI);
416         } else {
417             newElement = new Node(currentNode, localName, attributes, attributeNamespaces, namespaceURI);
418         }
419 
420         if (currentNode != null) {
421             currentNode.addChild(newElement);
422         }
423 
424         stack.push(currentNode);
425         currentNode = newElement;
426     }

428     public void ignorableWhitespace(char buffer[], int start, int len) throws SAXException {
429         if (keepIgnorableWhitespace) characters(buffer, start, len);
430     }

433     * @see org.xml.sax.ContentHandler#characters(char[], int, int)
434     */
435     public void characters(final char[] ch, final int start, final int length) throws SAXException {
436         charBuffer.append(ch, start, length);
437     }

439     /* (non-Javadoc)
440     * @see org.xml.sax.ContentHandler#endElement(java.lang.String, java.lang.String, java.lang.String)
441     */
442     public void endElement(final String namespaceURI, final String localName, final String qName) throws SAXException {
443         addCdata();
444         Node oldCurrentNode = stack.pop();
445         if (oldCurrentNode != null) {
446             currentNode = oldCurrentNode;
447         }
448     }

453     public void endDocument() throws SAXException {
454     }

456     private void addCdata() {
457         if (charBuffer.length() != 0) {
458             //
459             // This element is preceded by CDATA if keepIgnorableWhitespace is false (the default setting) and
460             // it's not whitespace add it to the body
461             // Note that, according to the XML spec, we should preserve the CDATA if it's all whitespace
462             // but for the sort of work I'm doing ignoring the whitespace is preferable
463             //
464             final String cdata = charBuffer.toString();
465             charBuffer.setLength(0);
466             if (keepIgnorableWhitespace || cdata.trim().length() != 0) {
467                 currentNode.addChild(cdata);
468             }
469         }
470     }
