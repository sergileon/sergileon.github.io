19  package org.codehaus.groovy.classgen;
20  
21  import org.codehaus.groovy.ast.*;
22  import org.codehaus.groovy.ast.expr.*;
23  import org.codehaus.groovy.ast.stmt.BlockStatement;
24  import org.codehaus.groovy.ast.stmt.ExpressionStatement;
25  import org.codehaus.groovy.ast.stmt.ForStatement;
26  import org.codehaus.groovy.ast.stmt.Statement;
27  import org.codehaus.groovy.runtime.InvokerHelper;
28  import org.codehaus.groovy.runtime.InvokerInvocationException;
29  import org.codehaus.groovy.syntax.Token;
30  
31  /**
32   * @author <a href="mailto:james@coredevelopers.net">James Strachan</a>
33   */
34  public class TupleListTest extends TestSupport {
35  
36      public void testIterateOverList() throws Exception {
37          ListExpression listExpression = new ListExpression();
38          listExpression.addExpression(new ConstantExpression("a"));
39          listExpression.addExpression(new ConstantExpression("b"));
40          listExpression.addExpression(new ConstantExpression("c"));
41          listExpression.addExpression(new ConstantExpression("a"));
42          listExpression.addExpression(new ConstantExpression("b"));
43          listExpression.addExpression(new ConstantExpression("c"));
44          assertIterate("iterateOverList", listExpression);
45      }
46  
47      public void testIterateOverMap() throws Exception {
48          MapExpression mapExpression = new MapExpression();
49          mapExpression.addMapEntryExpression(new ConstantExpression("a"), new ConstantExpression("x"));
50          mapExpression.addMapEntryExpression(new ConstantExpression("b"), new ConstantExpression("y"));
51          mapExpression.addMapEntryExpression(new ConstantExpression("c"), new ConstantExpression("z"));
52          assertIterate("iterateOverMap", mapExpression);
53      }
54  
55      protected void assertIterate(String methodName, Expression listExpression) throws Exception {
56          ClassNode classNode = new ClassNode("Foo", ACC_PUBLIC, ClassHelper.OBJECT_TYPE);
57          classNode.addConstructor(new ConstructorNode(ACC_PUBLIC, null));
58          classNode.addProperty(new PropertyNode("bar", ACC_PUBLIC, ClassHelper.STRING_TYPE, classNode, null, null, null));
59  
60          Statement loopStatement = createPrintlnStatement(new VariableExpression("i"));
61  
62          BlockStatement block = new BlockStatement();
63          block.addStatement(new ExpressionStatement(new DeclarationExpression(new VariableExpression("list"), Token.newSymbol("=", 0, 0), listExpression)));
64          block.addStatement(new ForStatement(new Parameter(ClassHelper.DYNAMIC_TYPE, "i"), new VariableExpression("list"), loopStatement));
65          classNode.addMethod(new MethodNode(methodName, ACC_PUBLIC, ClassHelper.VOID_TYPE, Parameter.EMPTY_ARRAY, ClassNode.EMPTY_ARRAY, block));
66  
67          Class fooClass = loadClass(classNode);
68          assertTrue("Loaded a new class", fooClass != null);
69  
70          Object bean = fooClass.newInstance();
71          assertTrue("Managed to create bean", bean != null);
72  
73          System.out.println("################ Now about to invoke method");
74  
75          try {
76              InvokerHelper.invokeMethod(bean, methodName, null);
77          }
78          catch (InvokerInvocationException e) {
79              System.out.println("Caught: " + e.getCause());
80              e.getCause().printStackTrace();
81              fail("Should not have thrown an exception");
82          }
83          System.out.println("################ Done");
84      }
85  
86  }
