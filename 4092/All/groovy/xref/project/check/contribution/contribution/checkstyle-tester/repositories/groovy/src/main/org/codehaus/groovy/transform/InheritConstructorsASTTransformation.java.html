19  package org.codehaus.groovy.transform;
20  
21  import groovy.transform.InheritConstructors;
22  import org.codehaus.groovy.ast.ASTNode;
23  import org.codehaus.groovy.ast.AnnotatedNode;
24  import org.codehaus.groovy.ast.AnnotationNode;
25  import org.codehaus.groovy.ast.ClassNode;
26  import org.codehaus.groovy.ast.ConstructorNode;
27  import org.codehaus.groovy.ast.Parameter;
28  import org.codehaus.groovy.ast.expr.Expression;
29  import org.codehaus.groovy.control.CompilePhase;
30  import org.codehaus.groovy.control.SourceUnit;
31  
32  import java.util.ArrayList;
33  import java.util.List;
34  import java.util.Map;
35  
36  import static org.codehaus.groovy.ast.ClassHelper.make;
37  import static org.codehaus.groovy.ast.tools.GeneralUtils.args;
38  import static org.codehaus.groovy.ast.tools.GeneralUtils.block;
39  import static org.codehaus.groovy.ast.tools.GeneralUtils.ctorSuperS;
40  import static org.codehaus.groovy.ast.tools.GeneralUtils.param;
41  import static org.codehaus.groovy.ast.tools.GeneralUtils.varX;
42  import static org.codehaus.groovy.ast.tools.GenericsUtils.correctToGenericsSpecRecurse;
43  import static org.codehaus.groovy.ast.tools.GenericsUtils.createGenericsSpec;
44  import static org.codehaus.groovy.ast.tools.GenericsUtils.extractSuperClassGenerics;
45  
46  /**
47   * Handles generation of code for the {@code @}InheritConstructors annotation.
48   *
49   * @author Paul King
50   */
51  @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
52  public class InheritConstructorsASTTransformation extends AbstractASTTransformation {
53  
54      private static final Class MY_CLASS = InheritConstructors.class;
55      private static final ClassNode MY_TYPE = make(MY_CLASS);
56      private static final String MY_TYPE_NAME = "@" + MY_TYPE.getNameWithoutPackage();
57  
58      public void visit(ASTNode[] nodes, SourceUnit source) {
59          init(nodes, source);
60          AnnotatedNode parent = (AnnotatedNode) nodes[1];
61          AnnotationNode node = (AnnotationNode) nodes[0];
62          if (!MY_TYPE.equals(node.getClassNode())) return;
63  
64          if (parent instanceof ClassNode) {
65              processClass((ClassNode) parent, node);
66          }
67      }
68  
69      private void processClass(ClassNode cNode, AnnotationNode node) {
70          if (cNode.isInterface()) {
71              addError("Error processing interface '" + cNode.getName() +
72                      "'. " + MY_TYPE_NAME + " only allowed for classes.", cNode);
73              return;
74          }
75          boolean copyConstructorAnnotations = memberHasValue(node, "constructorAnnotations", true);
76          boolean copyParameterAnnotations = memberHasValue(node, "parameterAnnotations", true);
77          ClassNode sNode = cNode.getSuperClass();
78          List<AnnotationNode> superAnnotations = sNode.getAnnotations(MY_TYPE);
79          if (superAnnotations.size() == 1) {
80              // We need @InheritConstructors from parent classes processed first
81              // so force that order here. The transformation is benign on an already
82              // processed node so processing twice in any order won't matter bar
83              // a very small time penalty.
84              processClass(sNode, node);
85          }
86          for (ConstructorNode cn : sNode.getDeclaredConstructors()) {
87              addConstructorUnlessAlreadyExisting(cNode, cn, copyConstructorAnnotations, copyParameterAnnotations);
88          }
89      }
90  
91      private void addConstructorUnlessAlreadyExisting(ClassNode classNode, ConstructorNode consNode, boolean copyConstructorAnnotations, boolean copyParameterAnnotations) {
92          Parameter[] origParams = consNode.getParameters();
93          if (consNode.isPrivate()) return;
94          Parameter[] params = new Parameter[origParams.length];
95          Map<String, ClassNode> genericsSpec = createGenericsSpec(classNode);
96          extractSuperClassGenerics(classNode, classNode.getSuperClass(), genericsSpec);
97          List<Expression> theArgs = buildParams(origParams, params, genericsSpec, copyParameterAnnotations);
98          if (isExisting(classNode, params)) return;
99          ConstructorNode added = classNode.addConstructor(consNode.getModifiers(), params, consNode.getExceptions(), block(ctorSuperS(args(theArgs))));
100         if (copyConstructorAnnotations) {
101             added.addAnnotations(copyAnnotatedNodeAnnotations(consNode, MY_TYPE_NAME));
102         }
103     }
104 
105     private List<Expression> buildParams(Parameter[] origParams, Parameter[] params, Map<String, ClassNode> genericsSpec, boolean copyParameterAnnotations) {
106         List<Expression> theArgs = new ArrayList<Expression>();
107         for (int i = 0; i < origParams.length; i++) {
108             Parameter p = origParams[i];
109             ClassNode newType = correctToGenericsSpecRecurse(genericsSpec, p.getType());
110             params[i] = p.hasInitialExpression() ? param(newType, p.getName(), p.getInitialExpression()) : param(newType, p.getName());
111             if (copyParameterAnnotations) {
112                 params[i].addAnnotations(copyAnnotatedNodeAnnotations(origParams[i], MY_TYPE_NAME));
113             }
114             theArgs.add(varX(p.getName(), newType));
115         }
116         return theArgs;
117     }
118 
119     private boolean isExisting(ClassNode classNode, Parameter[] params) {
120         for (ConstructorNode consNode : classNode.getDeclaredConstructors()) {
121             if (matchingTypes(params, consNode.getParameters())) {
122                 return true;
123             }
124         }
125         return false;
126     }
127 
128     private boolean matchingTypes(Parameter[] params, Parameter[] existingParams) {
129         if (params.length != existingParams.length) return false;
130         for (int i = 0; i < params.length; i++) {
131             if (!params[i].getType().equals(existingParams[i].getType())) {
132                 return false;
133             }
134         }
135         return true;
136     }
137 }
