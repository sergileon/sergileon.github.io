32  public class ApicallASM implements IASM, Opcodes {
33      private List<HookingSet> target = HookingSet.getHookingMethodSet(Configure.getInstance().hook_apicall_patterns);
34      private Map<String, HookingSet> reserved = new HashMap<String, HookingSet>();
35  
36      public ApicallASM() {
37          AsmUtil.add(reserved, "sun/net/www/protocol/http/HttpURLConnection", "getInputStream()Ljava/io/InputStream;");
38          AsmUtil.add(reserved, "sun/net/www/protocol/http/HttpURLConnection", "connect()V");
39          AsmUtil.add(reserved, "org/apache/commons/httpclient/HttpClient", "executeMethod("
40                  + "Lorg/apache/commons/httpclient/HostConfiguration;" + "Lorg/apache/commons/httpclient/HttpMethod;"
41                  + "Lorg/apache/commons/httpclient/HttpState;" + ")I");
42          AsmUtil.add(reserved, "org/apache/http/impl/client/InternalHttpClient", "doExecute");
43          AsmUtil.add(reserved, "sun/net/www/http/HttpClient", "parseHTTP");
44          AsmUtil.add(reserved, "org/apache/http/impl/client/AbstractHttpClient",//
45                  "execute(Lorg/apache/http/HttpHost;" + "Lorg/apache/http/HttpRequest;"
46                          + "Lorg/apache/http/protocol/HttpContext;)Lorg/apache/http/HttpResponse;");
47          // JCO CLIENT 추가..
48          AsmUtil.add(reserved, "com/sap/mw/jco/JCO$Client", "execute(Ljava/lang/String;" + //
49                  "Lcom/sap/mw/jco/JCO$ParameterList;" + //
50                  "Lcom/sap/mw/jco/JCO$ParameterList;" + //
51                  "Lcom/sap/mw/jco/JCO$ParameterList;" + //
52                  "Lcom/sap/mw/jco/JCO$ParameterList;" + //
53                  "Ljava/lang/String;Ljava/lang/String;I)V");
54          AsmUtil.add(reserved, "io/reactivex/netty/protocol/http/client/HttpClientImpl", "submit");
55      }
56  
57      public ClassVisitor transform(ClassVisitor cv, String className, ClassDesc classDesc) {
58          if (Configure.getInstance()._hook_methods_enabled == false) {
59              return cv;
60          }
61          HookingSet mset = reserved.get(className);
62          if (mset != null)
63              return new ApicallExtCV(cv, mset, className);
64          for (int i = 0; i < target.size(); i++) {
65              mset = target.get(i);
66              if (mset.classMatch.include(className)) {
67                  return new ApicallExtCV(cv, mset, className);
68              }
69          }
70          return cv;
71      }
72  }

74  class ApicallExtCV extends ClassVisitor implements Opcodes {
75      public String className;
76      private HookingSet mset;
77  
78      public ApicallExtCV(ClassVisitor cv, HookingSet mset, String className) {
79          super(ASM4, cv);
80          this.mset = mset;
81          this.className = className;
82      }
83  
84      @Override
85      public MethodVisitor visitMethod(int access, String methodName, String desc, String signature, String[] exceptions) {
86          MethodVisitor mv = super.visitMethod(access, methodName, desc, signature, exceptions);
87          if (mv == null || mset.isA(methodName, desc) == false) {
88              return mv;
89          }
90          if (AsmUtil.isSpecial(methodName)) {
91              return mv;
92          }
93          Logger.println("apicall: " + className + "." + methodName + desc);
94          return new ApicallExtMV(access, desc, mv, Type.getArgumentTypes(desc), (access & ACC_STATIC) != 0, className,
95                  methodName, desc);
96      }
97  }

99  class ApicallExtMV extends LocalVariablesSorter implements Opcodes {
100     private static final String TRACESUBCALL = TraceApiCall.class.getName().replace('.', '/');
101     private final static String START_METHOD = "startApicall";
102     private static final String START_SIGNATURE = "(Ljava/lang/String;Ljava/lang/String;Ljava/lang/String;Ljava/lang/Object;[Ljava/lang/Object;)Ljava/lang/Object;";
103     private final static String END_METHOD = "endApicall";
104     private static final String END_SIGNATURE = "(Ljava/lang/Object;Ljava/lang/Object;Ljava/lang/Throwable;)V";
105     private Label startFinally = new Label();
106 
107     public ApicallExtMV(int access, String desc, MethodVisitor mv, Type[] paramTypes, boolean isStatic,
108                         String classname, String methodname, String methoddesc) {
109         super(ASM4, access, desc, mv);
110         this.paramTypes = paramTypes;
111         this.returnType = Type.getReturnType(desc);
112         this.isStatic = isStatic;
113         this.className = classname;
114         this.methodName = methodname;
115         this.methodDesc = methoddesc;
116     }
117 
118     private Type[] paramTypes;
119     private Type returnType;
120     private boolean isStatic;
121     private String className;
122     private String methodName;
123     private String methodDesc;
124     private int statIdx;
125 
126     public void visitCode() {
127         int sidx = isStatic ? 0 : 1;
128         int arrVarIdx = newLocal(Type.getType("[Ljava/lang/Object;"));
129         AsmUtil.PUSH(mv, paramTypes.length);
130         mv.visitTypeInsn(Opcodes.ANEWARRAY, "java/lang/Object");
131         mv.visitVarInsn(Opcodes.ASTORE, arrVarIdx);
132         for (int i = 0; i < paramTypes.length; i++) {
133             Type tp = paramTypes[i];
134             mv.visitVarInsn(Opcodes.ALOAD, arrVarIdx);
135             AsmUtil.PUSH(mv, i);
136             switch (tp.getSort()) {
137                 case Type.BOOLEAN:
138                     mv.visitVarInsn(Opcodes.ILOAD, sidx);
139                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Boolean", "valueOf", "(Z)Ljava/lang/Boolean;",
140                             false);
141                     break;
142                 case Type.BYTE:
143                     mv.visitVarInsn(Opcodes.ILOAD, sidx);
144                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Byte", "valueOf", "(B)Ljava/lang/Byte;", false);
145                     break;
146                 case Type.CHAR:
147                     mv.visitVarInsn(Opcodes.ILOAD, sidx);
148                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Character", "valueOf", "(C)Ljava/lang/Character;",
149                             false);
150                     break;
151                 case Type.SHORT:
152                     mv.visitVarInsn(Opcodes.ILOAD, sidx);
153                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Short", "valueOf", "(S)Ljava/lang/Short;", false);
154                     break;
155                 case Type.INT:
156                     mv.visitVarInsn(Opcodes.ILOAD, sidx);
157                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Integer", "valueOf", "(I)Ljava/lang/Integer;",
158                             false);
159                     break;
160                 case Type.LONG:
161                     mv.visitVarInsn(Opcodes.LLOAD, sidx);
162                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Long", "valueOf", "(J)Ljava/lang/Long;", false);
163                     break;
164                 case Type.FLOAT:
165                     mv.visitVarInsn(Opcodes.FLOAD, sidx);
166                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Float", "valueOf", "(F)Ljava/lang/Float;", false);
167                     break;
168                 case Type.DOUBLE:
169                     mv.visitVarInsn(Opcodes.DLOAD, sidx);
170                     mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Double", "valueOf", "(D)Ljava/lang/Double;", false);
171                     break;
172                 default:
173                     mv.visitVarInsn(Opcodes.ALOAD, sidx);
174             }
175             mv.visitInsn(Opcodes.AASTORE);
176             sidx += tp.getSize();
177         }
178         AsmUtil.PUSH(mv, className);
179         AsmUtil.PUSH(mv, methodName);
180         AsmUtil.PUSH(mv, methodDesc);
181         if (isStatic) {
182             AsmUtil.PUSHNULL(mv);
183         } else {
184             mv.visitVarInsn(Opcodes.ALOAD, 0);
185         }
186         mv.visitVarInsn(Opcodes.ALOAD, arrVarIdx);
187         mv.visitMethodInsn(Opcodes.INVOKESTATIC, TRACESUBCALL, START_METHOD, START_SIGNATURE, false);
188         statIdx = newLocal(Type.getType(Object.class));
189         mv.visitVarInsn(Opcodes.ASTORE, statIdx);
190         mv.visitLabel(startFinally);
191         mv.visitCode();
192     }

194     @Override
195     public void visitInsn(int opcode) {
196         if ((opcode >= IRETURN && opcode <= RETURN)) {
197             capReturn();
198         }
199         mv.visitInsn(opcode);
200     }
201 
202     private void capReturn() {
203         Type tp = returnType;
204         if (tp == null || tp.equals(Type.VOID_TYPE)) {
205             mv.visitVarInsn(Opcodes.ALOAD, statIdx);
206             mv.visitInsn(Opcodes.ACONST_NULL);
207             mv.visitInsn(Opcodes.ACONST_NULL);
208             mv.visitMethodInsn(Opcodes.INVOKESTATIC, TRACESUBCALL, END_METHOD, END_SIGNATURE, false);
209             return;
210         }
211         int i = newLocal(tp);
212         switch (tp.getSort()) {
213             case Type.BOOLEAN:
214                 mv.visitVarInsn(Opcodes.ISTORE, i);
215                 mv.visitVarInsn(Opcodes.ILOAD, i);
216                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
217                 mv.visitVarInsn(Opcodes.ILOAD, i);
218                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Boolean", "valueOf", "(Z)Ljava/lang/Boolean;", false);
219                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
220                 break;
221             case Type.BYTE:
222                 mv.visitVarInsn(Opcodes.ISTORE, i);
223                 mv.visitVarInsn(Opcodes.ILOAD, i);
224                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
225                 mv.visitVarInsn(Opcodes.ILOAD, i);
226                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Byte", "valueOf", "(B)Ljava/lang/Byte;", false);
227                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
228                 break;
229             case Type.CHAR:
230                 mv.visitVarInsn(Opcodes.ISTORE, i);
231                 mv.visitVarInsn(Opcodes.ILOAD, i);
232                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
233                 mv.visitVarInsn(Opcodes.ILOAD, i);
234                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Character", "valueOf", "(C)Ljava/lang/Character;",
235                         false);
236                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
237                 break;
238             case Type.SHORT:
239                 mv.visitVarInsn(Opcodes.ISTORE, i);
240                 mv.visitVarInsn(Opcodes.ILOAD, i);
241                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
242                 mv.visitVarInsn(Opcodes.ILOAD, i);
243                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Short", "valueOf", "(S)Ljava/lang/Short;", false);
244                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
245                 break;
246             case Type.INT:
247                 mv.visitVarInsn(Opcodes.ISTORE, i);
248                 mv.visitVarInsn(Opcodes.ILOAD, i);
249                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
250                 mv.visitVarInsn(Opcodes.ILOAD, i);
251                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Integer", "valueOf", "(I)Ljava/lang/Integer;", false);
252                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
253                 break;
254             case Type.LONG:
255                 mv.visitVarInsn(Opcodes.LSTORE, i);
256                 mv.visitVarInsn(Opcodes.LLOAD, i);
257                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
258                 mv.visitVarInsn(Opcodes.LLOAD, i);
259                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Long", "valueOf", "(J)Ljava/lang/Long;", false);
260                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
261                 break;
262             case Type.FLOAT:
263                 mv.visitVarInsn(Opcodes.FSTORE, i);
264                 mv.visitVarInsn(Opcodes.FLOAD, i);
265                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
266                 mv.visitVarInsn(Opcodes.FLOAD, i);
267                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Float", "valueOf", "(F)Ljava/lang/Float;", false);
268                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
269                 break;
270             case Type.DOUBLE:
271                 mv.visitVarInsn(Opcodes.DSTORE, i);
272                 mv.visitVarInsn(Opcodes.DLOAD, i);
273                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
274                 mv.visitVarInsn(Opcodes.DLOAD, i);
275                 mv.visitMethodInsn(Opcodes.INVOKESTATIC, "java/lang/Double", "valueOf", "(D)Ljava/lang/Double;", false);
276                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
277                 break;
278             default:
279                 mv.visitVarInsn(Opcodes.ASTORE, i);
280                 mv.visitVarInsn(Opcodes.ALOAD, i);
281                 mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
282                 mv.visitVarInsn(Opcodes.ALOAD, i);// return
283                 mv.visitInsn(Opcodes.ACONST_NULL);// throwable
284         }
285         mv.visitMethodInsn(Opcodes.INVOKESTATIC, TRACESUBCALL, END_METHOD, END_SIGNATURE, false);
286     }
287 
288     @Override
289     public void visitMaxs(int maxStack, int maxLocals) {
290         Label endFinally = new Label();
291         mv.visitTryCatchBlock(startFinally, endFinally, endFinally, null);
292         mv.visitLabel(endFinally);
293         mv.visitInsn(DUP);
294         int errIdx = newLocal(Type.getType(Throwable.class));
295         mv.visitVarInsn(Opcodes.ASTORE, errIdx);
296         mv.visitVarInsn(Opcodes.ALOAD, statIdx);// stat
297         mv.visitInsn(Opcodes.ACONST_NULL);// return
298         mv.visitVarInsn(Opcodes.ALOAD, errIdx);// throwable
299         mv.visitMethodInsn(Opcodes.INVOKESTATIC, TRACESUBCALL, END_METHOD, END_SIGNATURE, false);
300         mv.visitInsn(ATHROW);
301         mv.visitMaxs(maxStack + 8, maxLocals + 2);
302     }
