258 	@Override
259 	protected void paintClientArea(final Graphics graphics) {
260 		// Don't do anything when hidden
261 		if (!isVisible())
262 			return;
263 
264 		super.paintClientArea(graphics);
265 
266 //		graphics.pushState();
267 		graphics.setFont(titleFont);
268 		final Dimension titleSize = FigureUtilities.getTextExtents(title, titleFont);
269 		if(isHorizontal()){
270 			if(getTickLablesSide() == LabelSide.Primary)
271 				graphics.drawText(title,
272 						bounds.x + bounds.width/2 - titleSize.width/2,
273 						bounds.y + bounds.height - titleSize.height);
274 			else
275 				graphics.drawText(title,
276 						bounds.x + bounds.width/2 - titleSize.width/2,
277 						bounds.y);
278 		}else{
279 		    final int w = titleSize.height;
280 		    final int h = titleSize.width +1;
281 
282 			if(getTickLablesSide() == LabelSide.Primary){
283 				GraphicsUtil.drawVerticalText(graphics, title,
284 							bounds.x, bounds.y + bounds.height/2 - h/2, false);
285 			}else {
286 				GraphicsUtil.drawVerticalText(graphics, title,
287 								bounds.x + bounds.width - w, bounds.y + bounds.height/2 - h/2, true);
288 			}
289 		}
290 
291 //		graphics.popState();
292 
293 		// Show the start/end cursor or the 'rubberband' of a zoom operation?
294 		if(armed && end != null && start != null){
295 			switch (zoomType) {
296 			case RUBBERBAND_ZOOM:
297 			case HORIZONTAL_ZOOM:
298 			case VERTICAL_ZOOM:
299 				graphics.setLineStyle(SWTConstants.LINE_DOT);
300 				graphics.setLineWidth(1);
301 				graphics.setForegroundColor(revertBackColor);
302 				graphics.drawRectangle(start.x, start.y, end.x - start.x-1, end.y - start.y-1);
303 				break;
304 
305 			default:
306 				break;
307 			}
308 		}
309 	}

311 	/** @return Range that reflects the minimum and maximum value of all
312 	 *          traces on this axis.
313 	 *          Returns <code>null</code> if there is no trace data.
314 	 */
315     public Range getTraceDataRange()
316     {
317         double low = Double.POSITIVE_INFINITY;
318         double high = Double.NEGATIVE_INFINITY;
319         for (Trace trace : traceList)
320         {
321             if (trace.getDataProvider() == null || !trace.isVisible())
322                 continue;
323             final Range range;
324             if (isHorizontal())
325                 range = trace.getDataProvider().getXDataMinMax();
326             else
327                 range = trace.getDataProvider().getYDataMinMax();
328             if (range == null)
329                 continue;
330             if (Double.isInfinite(range.getLower())
331                     || Double.isInfinite(range.getUpper())
332                     || Double.isNaN(range.getLower())
333                     || Double.isNaN(range.getUpper()))
334                 continue;
335             if (low > range.getLower())
336                 low = range.getLower();
337             if (high < range.getUpper())
338                 high = range.getUpper();
339         }
340         if (Double.isInfinite(low) || Double.isInfinite(high))
341             return null;
342         return new Range(low, high);
343     }

345 	/** Perform an auto-scale:
346 	 *  Axis limits are set to the value range of the traces on this axis.
347 	 *  Includes some optimization:
348 	 *  Axis range is set a little wider than exact trace data range.
349 	 *  When auto-scale would only perform a minor axis adjustment,
350 	 *  axis is left unchanged.
351 	 *
352 	 *  @param force If true, the axis will be auto-scaled by force regardless the autoScale field.
353 	 *  Otherwise, it will use the autoScale field to judge whether an auto-scale will be performed.
354 	 *  @return true if the axis is repainted due to range change.
355 	 *
356 	 *  @see #autoScaleThreshold
357 	 */
358 	public boolean performAutoScale(final boolean force){
359 	    // Anything to do? Autoscale not enabled nor forced?
360 		if (traceList.size() <= 0  ||  !(force || autoScale))
361 		    return false;
362 		if(!force && xyGraph.getZoomType() != ZoomType.NONE)
363 			return false;
364 
365 	    // Get range of data in all traces
366         final Range range = getTraceDataRange();
367         if (range == null)
368             return false;
369 		double tempMin = range.getLower();
370 		double tempMax = range.getUpper();
371 
372 		// Get current axis range, determine how 'different' they are
373 		double max = getRange().getUpper();
374 		double min = getRange().getLower();
375 
376 		if (isLogScaleEnabled())
377 		{   // Transition into log space
378 		    tempMin = Log10.log10(tempMin);
379 		    tempMax = Log10.log10(tempMax);
380 		    max = Log10.log10(max);
381 		    min = Log10.log10(min);
382 		}
383 //Give extra space in plot area
384 //		double extraSpace = (tempMax-tempMin)*0.05;
385 //		double tempMin2 = tempMin -extraSpace;
386 //		double tempMax2 = tempMax + extraSpace;
387 		
388 		final double thr = (max - min)*autoScaleThreshold;
389 
390 		//if both the changes are lower than threshold, return
391 		if(((tempMin - min)>=0 && (tempMin - min)<thr)
392 				&& ((max - tempMax)>=0 && (max - tempMax)<thr)){
393 			return false;
394 		}else { //expand more space than needed
395 			if((tempMin - min)<0)
396 				tempMin -= thr;
397 			if((tempMax - max) > 0)
398 				tempMax += thr;
399 		}
400 	
401 //		if(tempMin>tempMin2)
402 //			tempMin=tempMin2;
403 //		if(tempMax<tempMax2)
404 //			tempMax = tempMax2;
405 		// Any change at all?
406 		if((Double.doubleToLongBits(tempMin) == Double.doubleToLongBits(min)
407 				&& Double.doubleToLongBits(tempMax) == Double.doubleToLongBits(max)) ||
408 				Double.isInfinite(tempMin) || Double.isInfinite(tempMax) ||
409 				Double.isNaN(tempMin) || Double.isNaN(tempMax))
410 			return false;
411 
412         if (isLogScaleEnabled())
413         {   // Revert from log space
414             tempMin = Log10.pow10(tempMin);
415             tempMax = Log10.pow10(tempMax);
416         }
417 
418 		// Update axis
419 		setRange(tempMin, tempMax, true);
420 		repaint();
421 		return true;
422 	}

626 	/** @param zoom Zoom Type
627 	 *  @return <code>true</code> if the zoom type is applicable to this axis
628  	 */
629 	private boolean isValidZoomType(final ZoomType zoom)
630 	{
631 	    return zoom == ZoomType.PANNING   ||
632 	           zoom == ZoomType.RUBBERBAND_ZOOM ||
633                zoom == ZoomType.ZOOM_IN   ||
634                zoom == ZoomType.ZOOM_OUT  ||
635                (isHorizontal() &&
636                 (zoom == ZoomType.HORIZONTAL_ZOOM ||
637                  zoom == ZoomType.ZOOM_IN_HORIZONTALLY ||
638                  zoom == ZoomType.ZOOM_OUT_HORIZONTALLY)
639                ) ||
640                (!isHorizontal() &&
641                 (zoom == ZoomType.VERTICAL_ZOOM ||
642                  zoom == ZoomType.ZOOM_OUT_VERTICALLY ||
643                  zoom == ZoomType.ZOOM_IN_VERTICALLY)
644                );
645 	}

709 	private void pan()
710 	{
711 		if(isHorizontal())
712 		    pan(startRange,
713 		        getPositionValue(start.x, false), getPositionValue(end.x, false));
714 		else
715             pan(startRange,
716                 getPositionValue(start.y, false), getPositionValue(end.y, false));

719 	/** Pan the axis
720 	 *  @param temp Original axis range before the panning started
721 	 *  @param t1 Start of the panning move
722 	 *  @param t2 End of the panning move
723 	 */
724 	void pan(final Range temp, double t1, double t2)
725     {
726         if (isLogScaleEnabled())
727         {
728             final double m = Math.log10(t2) - Math.log10(t1);
729             t1 = Math.pow(10,Math.log10(temp.getLower()) - m);
730             t2 = Math.pow(10,Math.log10(temp.getUpper()) - m);
731         }
732         else
733         {
734             final double m = t2-t1;
735             t1 = temp.getLower() - m;
736             t2 = temp.getUpper() - m;
737         }
738         setRange(t1, t2);
739     }

741     /** Zoom axis
742 	 *  @param center Axis position at the 'center' of the zoom
743 	 *  @param factor Zoom factor. Positive to zoom 'in', negative 'out'.
744 	 */
745 	void zoomInOut(final double center, final double factor)
746     {
747 	    final double t1, t2;
748 	    if (isLogScaleEnabled())
749 	    {
750 	        final double l = Math.log10(getRange().getUpper()) -
751                     Math.log10(getRange().getLower());
752 	        final double r1 = (Math.log10(center) - Math.log10(getRange().getLower()))/l;
753 	        final double r2 = (Math.log10(getRange().getUpper()) - Math.log10(center))/l;
754             t1 = Math.pow(10, Math.log10(getRange().getLower()) + r1 * factor * l);
755             t2 = Math.pow(10, Math.log10(getRange().getUpper()) - r2 * factor * l);
756         }else{
757             final double l = getRange().getUpper() - getRange().getLower();
758             final double r1 = (center - getRange().getLower())/l;
759             final double r2 = (getRange().getUpper() - center)/l;
760             t1 = getRange().getLower() + r1 * factor * l;
761             t2 = getRange().getUpper() - r2 * factor * l;
762         }
763         setRange(t1, t2, true);
764     }

801 	/** Listener to mouse events, performs panning and some zooms
802 	 *  Is very similar to the PlotMouseListener, but unclear
803 	 *  how easy/useful it would be to base them on the same code.
804 	 */
805 	class AxisMouseListener extends MouseMotionListener.Stub implements MouseListener
806 	{
807 		private SaveStateCommand command;
808 
809         public void mousePressed(final MouseEvent me)
810         {
811             // Only react to 'main' mouse button, only react to 'real' zoom
812             if (me.button != 1  ||  !isValidZoomType(zoomType))
813                 return;
814             armed = true;
815             // get start position
816             switch (zoomType)
817             {
818             case RUBBERBAND_ZOOM:
819             	if(isHorizontal())
820             		start = new Point(me.getLocation().x, bounds.y);
821             	else
822             		start = new Point(bounds.x, me.getLocation().y);
823                 end = null;
824                 break;
825             case HORIZONTAL_ZOOM:
826                 start = new Point(me.getLocation().x, bounds.y);
827                 end = null;
828                 break;
829             case VERTICAL_ZOOM:
830                 start = new Point(bounds.x, me.getLocation().y);
831                 end = null;
832                 break;
833             case PANNING:
834                 setCursor(grabbing);
835                 start = me.getLocation();
836                 end = null;
837                 startRange = getRange();
838                 break;
839             case ZOOM_IN:
840             case ZOOM_IN_HORIZONTALLY:
841             case ZOOM_IN_VERTICALLY:
842             case ZOOM_OUT:
843             case ZOOM_OUT_HORIZONTALLY:
844             case ZOOM_OUT_VERTICALLY:
845                 start = me.getLocation();
846                 end = new Point();
847                 // Start timer that will zoom while mouse button is pressed
848                 Display.getCurrent().timerExec(ZOOM_SPEED, new Runnable()
849                 {
850                     public void run()
851                     {
852                         if (!armed)
853                             return;
854                         performInOutZoom();
855                         Display.getCurrent().timerExec(ZOOM_SPEED, this);
856                     }
857                 });
858                 break;
859             default:
860                 break;
861         	}
862 
863             //add command for undo operation
864             command = new AxisPanOrZoomCommand(zoomType.getDescription(), Axis.this);
865             me.consume();
866         }

868         public void mouseDoubleClicked(final MouseEvent me) { /* Ignored */ }
869 
870         @Override
871 		public void mouseDragged(final MouseEvent me)
872 		{
873 			if (! armed)
874 				return;
875 			switch (zoomType)
876 			{
877             case RUBBERBAND_ZOOM:
878                 // Treat rubberband zoom on axis like horiz/vert. zoom
879                 if (isHorizontal())
880                     end = new Point(me.getLocation().x, bounds.y + bounds.height);
881                 else
882                     end = new Point(bounds.x + bounds.width, me.getLocation().y);
883                 break;
884 			case HORIZONTAL_ZOOM:
885                 end = new Point(me.getLocation().x, bounds.y + bounds.height);
886                 break;
887             case VERTICAL_ZOOM:
888                 end = new Point(bounds.x + bounds.width, me.getLocation().y);
889                 break;
890             case PANNING:
891                 end = me.getLocation();
892                 pan();
893                 break;
894             default:
895                 break;
896 			}
897 			Axis.this.repaint();
898 		}

900         @Override
901 		public void mouseExited(final MouseEvent me)
902 		{
903             // Treat like releasing the button to stop zoomIn/Out timer
904 		    switch (zoomType)
905             {
906             case ZOOM_IN:
907             case ZOOM_IN_HORIZONTALLY:
908             case ZOOM_IN_VERTICALLY:
909             case ZOOM_OUT:
910             case ZOOM_OUT_HORIZONTALLY:
911             case ZOOM_OUT_VERTICALLY:
912                 mouseReleased(me);
913             default:
914             }
915 		}
916 
917         public void mouseReleased(final MouseEvent me)
918 		{
919 		    if (! armed)
920 		        return;
921             armed = false;
922             if (zoomType == ZoomType.PANNING)
923                 setCursor(zoomType.getCursor());
924 			if (end == null || start == null || command == null)
925 				return;
926 
927 			switch (zoomType)
928 			{
929             case RUBBERBAND_ZOOM:
930             case HORIZONTAL_ZOOM:
931             case VERTICAL_ZOOM:
932                 performStartEndZoom();
933                 break;
934             case PANNING:
935                 pan();
936                 break;
937             case ZOOM_IN:
938             case ZOOM_IN_HORIZONTALLY:
939             case ZOOM_IN_VERTICALLY:
940             case ZOOM_OUT:
941             case ZOOM_OUT_HORIZONTALLY:
942             case ZOOM_OUT_VERTICALLY:
943                 performInOutZoom();
944                 break;
945             default:
946                 break;
947 			}
948 			command.saveState();
949 			xyGraph.getOperationsManager().addCommand(command);
950 			command = null;
951             start = null;
952             end = null;
953 		}

955 		/** Perform the zoom to mouse start/end */
956         private void performStartEndZoom()
957         {
958             final double t1 = getPositionValue(isHorizontal() ? start.x : start.y, false);
959             final double t2 = getPositionValue(isHorizontal() ? end.x   : end.y,   false);
960             setRange(t1, t2, true);
961         }
962 
963 		/** Perform the in or out zoom according to zoomType */
964         private void performInOutZoom()
965         {
966             final int pixel_pos = isHorizontal() ? start.x : start.y;
967             final double center = getPositionValue(pixel_pos, false);
968             switch (zoomType)
969             {
970             case ZOOM_IN:              zoomInOut(center, ZOOM_RATIO); break;
971             case ZOOM_IN_HORIZONTALLY: zoomInOut(center, ZOOM_RATIO); break;
972             case ZOOM_IN_VERTICALLY:   zoomInOut(center, ZOOM_RATIO); break;
973             case ZOOM_OUT:             zoomInOut(center, -ZOOM_RATIO); break;
974             case ZOOM_OUT_HORIZONTALLY:zoomInOut(center, -ZOOM_RATIO); break;
975             case ZOOM_OUT_VERTICALLY:  zoomInOut(center, -ZOOM_RATIO); break;
976             default:                   // NOP
977             }
978         }
