109 	public void createPartControl(Composite parent) {
110 		counterEngine = ServerManager.getInstance().getServer(serverId).getCounterEngine();
111 		if (this.objHash == 0) {
112 			this.setPartName("Active Service List[" + counterEngine.getDisplayNameObjectType(objType) + "]");
113 		} else {

116 		initialLayout(parent);
117 		IToolBarManager man = getViewSite().getActionBars().getToolBarManager();
118 		man.add(new Action("reload", ImageUtil.getImageDescriptor(Images.refresh)) {
119 			public void run() {
120 				manulRefresh = true;
121 				thread.interrupt();
122 			}
123 		});
124 		man.add(new Separator());

126 	    Action actAutoRefresh = new Action("Auto Refresh in 5 sec.", IAction.AS_CHECK_BOX){ 
127 	        public void run() {
128 	        	autoRefresh = isChecked();
129 	        	if (autoRefresh) {
130 	        		thread.interrupt();
131 	        	}
132 	        }
133 	    };  
134 	    actAutoRefresh.setImageDescriptor(ImageUtil.getImageDescriptor(Images.refresh_auto));
135 	    man.add(actAutoRefresh);
136 	    
137 		thread = new RefreshThread(this, 5000);
138 		thread.start();
139 	}

156 	private void createTableViewer(Composite composite) {
157 		tableViewer = new TableViewer(composite, SWT.MULTI  | SWT.FULL_SELECTION | SWT.BORDER);
158 		tableColumnLayout = new TableColumnLayout();
159 		composite.setLayout(tableColumnLayout);
160 		createColumns();
161 		final Table table = tableViewer.getTable();
162 	    table.setHeaderVisible(true);
163 	    table.setLinesVisible(true);
164 	    tableViewer.setContentProvider(new ArrayContentProvider());
165 	    tableViewer.setLabelProvider(new LabelProvider());
166 	    tableViewer.setComparator(new TableLabelSorter(tableViewer));
167 	    GridData gridData = new GridData(GridData.FILL, GridData.FILL, true, true);
168 	    tableViewer.getControl().setLayoutData(gridData);
169 	    tableViewer.addDoubleClickListener(new IDoubleClickListener() {
170 			public void doubleClick(DoubleClickEvent event) {
171 				StructuredSelection sel = (StructuredSelection) event.getSelection();
172 				Object o = sel.getFirstElement();
173 				if (o instanceof ThreadData) {
174 					ThreadData data = (ThreadData) o;
175 					try {
176 						IWorkbenchWindow win = PlatformUI.getWorkbench().getActiveWorkbenchWindow();
177 						ObjectThreadDetailView view = (ObjectThreadDetailView) win.getActivePage().showView(ObjectThreadDetailView.ID, serverId + "&" + data.objHash, IWorkbenchPage.VIEW_ACTIVATE);
178 						view.setInput(data.id);
179 					} catch (Exception d) {
180 					}
181 				}
182 			}
183 		});
184 	}

196 	private void load() {
197 		List<Pack> packList = AgentDataProxy.getActiveThreadList(objType, objHash, serverId);
198 		final ArrayList<ThreadData> datas = new ArrayList<ThreadData>();
199 		final DecimalValue count = new DecimalValue();
200 		final StringBuilder error = new StringBuilder();
201 		for (Pack pack : packList) {
202 			MapPack mpack = (MapPack) pack;
203 			boolean complete = mpack.getBoolean("complete");
204 			int objHash = mpack.getInt("objHash");
205 			if (complete == false) {
206 				String objName = TextProxy.object.getText(objHash);
207 				if (objName != null) {
208 					error.append(objName + " ");
209 				}
210 			}
211 			ListValue idLv = mpack.getList("id");
212 			ListValue nameLv = mpack.getList("name");
213 			ListValue statLv = mpack.getList("stat");
214 			ListValue cpuLv = mpack.getList("cpu");
215 			ListValue txidLv = mpack.getList("txid");
216 			ListValue elapsedLv = mpack.getList("elapsed");
217 			ListValue serviceLv = mpack.getList("service");
218 			ListValue sqlLv = mpack.getList("sql");
219 			ListValue subcallLv = mpack.getList("subcall");
220 			
221 			int size = idLv.size();
222 			count.value = count.value + size;
223 			
224 			for (int i = 0; i < size; i++) {
225 				ThreadData data = new ThreadData();
226 				data.id = idLv.getLong(i);
227 				data.objHash = objHash;
228 				data.name = nameLv.getString(i);
229 				data.state = statLv.getString(i);
230 				data.cpu = cpuLv.getLong(i);
231 				data.txid = txidLv.getString(i);
232 				data.elapsed = elapsedLv.getLong(i);
233 				data.serviceName = serviceLv.getString(i);
234 				String sql = sqlLv.getString(i);
235 				if (StringUtil.isNotEmpty(sql)) {
236 					data.note = sql;
237 				} else {
238 					data.note = subcallLv.getString(i);
239 				}
240 				datas.add(data);
241 			}
242 		}
243 		if (error.length() > 0) {
244 			error.append("may be not loaded.");
245 		}
246 		ExUtil.exec(tableViewer.getTable(), new Runnable() {
247 			public void run() {
248 				ObjectActiveServiceListView.this.setContentDescription("Count = " + count.value);
249 				if (error.length() > 0) {
250 					GridData gr = (GridData) errorLbl.getLayoutData();
251 					gr.exclude = false;
252 					errorLbl.setVisible(true);
253 					errorLbl.setText(error.toString());
254 				} else {
255 					GridData gr = (GridData) errorLbl.getLayoutData();
256 					gr.exclude = true;
257 					errorLbl.setVisible(false);
258 					errorLbl.setText("");
259 				}
260 				errorLbl.getParent().layout(false);
261 				tableViewer.setInput(datas);
262 			}
263 		});
264 	}

266 	private void createColumns() {
267 		for (ColumnEnum column : ColumnEnum.values()) {
268 			createTableViewerColumn(column.getTitle(), column.getWidth(), column.getAlignment(), column.isResizable(), column.isMoveable(), column.isNumber());
269 		}
270 	}
271 	
272 	private TableViewerColumn createTableViewerColumn(String title, int width, int alignment,  boolean resizable, boolean moveable, final boolean isNumber) {
273 		final TableViewerColumn viewerColumn = new TableViewerColumn(tableViewer, SWT.NONE);
274 		final TableColumn column = viewerColumn.getColumn();
275 		column.setText(title);
276 		column.setAlignment(alignment);
277 		column.setMoveable(moveable);
278 		tableColumnLayout.setColumnData(column, new ColumnWeightData(width, width, resizable));
279 		column.setData("isNumber", isNumber);
280 		column.addSelectionListener(new SelectionAdapter() {
281 			public void widgetSelected(SelectionEvent e) {
282 				TableLabelSorter sorter = (TableLabelSorter) tableViewer.getComparator();
283 				TableColumn selectedColumn = (TableColumn) e.widget;
284 				sorter.setColumn(selectedColumn);
285 			}
286 		});
287 		return viewerColumn;
288 	}

313 		public Color getForeground(Object element) {
314 			if (element instanceof ThreadData) {
315 				ThreadData t = (ThreadData) element;
316 				if (t.elapsed > 8000) {
317 					return ColorUtil.getInstance().getColor(SWT.COLOR_RED);
318 				} else if (t.elapsed > 3000) {
319 					return ColorUtil.getInstance().getColor(SWT.COLOR_MAGENTA);
320 				} else {
321 					return ColorUtil.getInstance().getColor(SWT.COLOR_BLUE);
322 				}
323 			}
324 			return null;
325 		}
326 
327 		public String getColumnText(Object element, int columnIndex) {
328 			if (element instanceof ThreadData) {
329 				if (columnIndex == ColumnEnum.NO.getIndex()) {
330 					return FormatUtil.print(((ThreadData) element).id, "000");
331 				} else if (columnIndex == ColumnEnum.OBJNAME.getIndex()) {
332 					return TextProxy.object.getText(((ThreadData) element).objHash);
333 				} else if (columnIndex == ColumnEnum.NAME.getIndex()) {
334 					return ((ThreadData) element).name;
335 				} else if (columnIndex == ColumnEnum.STATE.getIndex()) {
336 					return ((ThreadData) element).state;
337 				} else if (columnIndex == ColumnEnum.CPU.getIndex()) {
338 					return FormatUtil.print(((ThreadData) element).cpu, "#,##0");
339 				} else if (columnIndex == ColumnEnum.ELAPSED.getIndex()) {
340 					return FormatUtil.print(((ThreadData) element).elapsed, "#,##0");
341 				} else if (columnIndex == ColumnEnum.TXID.getIndex()) {
342 					return ((ThreadData) element).txid;
343 				} else if (columnIndex == ColumnEnum.SERVICE.getIndex()) {
344 					return ((ThreadData) element).serviceName;
345 				} else if (columnIndex == ColumnEnum.NOTE.getIndex()) {
346 					return ((ThreadData) element).note;
347 				}
348 			}
349 			return null;
350 		}
351 	}
