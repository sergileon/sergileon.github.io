122 	public void load() {
123 		final String filter = filterText.getText();
124 		ExUtil.asyncRun(new Runnable() {
125 			public void run() {
126 				TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
127 				MapPack out = null;
128 				try {
129 					MapPack param = new MapPack();
130 					param.put("objHash", objHash);
131 					if (StringUtil.isNotEmpty(filter)) {
132 						param.put("filter", filter);
133 					}
134 					param.put("page", currentPage);
135 					out = (MapPack) tcp.getSingle(RequestCmd.OBJECT_CLASS_LIST, param);
136 					if (out == null) {
137 						return;
138 					}
139 				} catch (Throwable t) {
140 					ConsoleProxy.errorSafe(t.toString());
141 				} finally {
142 					TcpProxy.putTcpProxy(tcp);
143 				}
144 				
145 				currentPage = out.getInt("page");
146 				totalPage = out.getInt("totalPage");
147 				
148 				ListValue indexLv = out.getList("index");
149 				if (indexLv == null) {
150 					return;
151 				}
152 				ListValue nameLv = out.getList("name");
153 				ListValue typeLv = out.getList("type");
154 				ListValue superClassLv = out.getList("superClass");
155 				ListValue interfaceLv = out.getList("interfaces");
156 				ListValue resourceLv = out.getList("resource");
157 				
158 				int count = indexLv.size();
159 				final List<ClassData> classDataList = new ArrayList<ClassData>(count);
160 				
161 				for (int i = 0; i < count; i++) {
162 					ClassData data = new ClassData();
163 					data.index = indexLv.getLong(i);
164 					data.type = typeLv.getString(i);
165 					data.name = nameLv.getString(i);
166 					data.superClass = superClassLv.getString(i);
167 					data.interfaces = interfaceLv.getString(i);
168 					data.resources = resourceLv.getString(i);
169 					classDataList.add(data);
170 				}
171 				ExUtil.exec(tableViewer.getTable(), new Runnable() {
172 					public void run() {
173 						validatePageButton();
174 						tableViewer.setInput(classDataList);
175 					}
176 				});
177 			}
178 		});
179 		
180 	}

220 	    Server server = ServerManager.getInstance().getServer(serverId);
221 	    if (server.isAllowAction(GroupPolicyConstants.ALLOW_EXPORTCLASS)) {
222 		    manager.add(new Action("&Export Class") {
223 		    	public void run() {
224 		    		StructuredSelection selection = (StructuredSelection) tableViewer.getSelection();
225 					ClassData data = (ClassData) selection.getFirstElement();
226 					final String className = data.name;
227 					if (StringUtil.isEmpty(className)) {
228 						return;
229 					}
230 					ExUtil.asyncRun(new Runnable() {
231 						public void run() {
232 							TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
233 							MapPack p = null;
234 							try {
235 								MapPack param = new MapPack();
236 								param.put("objHash", objHash);
237 								param.put("class", className);
238 								p = (MapPack) tcp.getSingle(RequestCmd.OBJECT_LOAD_CLASS_BY_STREAM, param);
239 							} catch (Exception e) {
240 								ConsoleProxy.errorSafe(e.getMessage());
241 							} finally {
242 								TcpProxy.putTcpProxy(tcp);
243 							}
244 							if (p != null) {
245 								String error = CastUtil.cString(p.get("error"));
246 								if (StringUtil.isNotEmpty(error)) {
247 									ConsoleProxy.errorSafe(error);
248 								}
249 								Value v = p.get("class");
250 								if (v != null) {
251 									final BlobValue bv = (BlobValue) v;
252 									ExUtil.exec(tableViewer.getTable(), new Runnable() {
253 										public void run() {
254 											saveClassFile(className, bv);
255 										}
256 									});
257 								}
258 							}
259 						}
260 					});
261 				}
262 		    });
263 		    manager.add(new Action("&Export Jar") {
264 		    	public void run() {
265 		    		StructuredSelection selection = (StructuredSelection) tableViewer.getSelection();
266 					final ClassData data = (ClassData) selection.getFirstElement();
267 					final String resource = data.resources;
268 					if (StringUtil.isEmpty(resource)) {
269 						return;
270 					}
271 					ExUtil.asyncRun(new Runnable() {
272 						public void run() {
273 							TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
274 							MapPack p = null;
275 							try {
276 								MapPack param = new MapPack();
277 								param.put("objHash", objHash);
278 								param.put("resource", resource);
279 								p = (MapPack) tcp.getSingle(RequestCmd.OBJECT_CHECK_RESOURCE_FILE, param);
280 							} catch (Exception e) {
281 								ConsoleProxy.errorSafe(e.getMessage());
282 							} finally {
283 								TcpProxy.putTcpProxy(tcp);
284 							}
285 							if (p != null) {
286 								String error = p.getText("error");
287 								if (StringUtil.isNotEmpty(error)) {
288 									ConsoleProxy.errorSafe(error);
289 								} else {
290 									final String name = p.getText("name");
291 									final long size = p.getLong("size");
292 									ExUtil.exec(tableViewer.getTable(), new Runnable() {
293 										public void run() {
294 											if(MessageDialog.openQuestion(tableViewer.getTable().getShell(), data.name
295 													, name + "(" + ScouterUtil.humanReadableByteCount(size, true) +") will be downloaded.\nContinue?")) {
296 												Shell shell = PlatformUI.getWorkbench().getActiveWorkbenchWindow()	.getShell();
297 												FileDialog dialog = new FileDialog(shell, SWT.SAVE);
298 												dialog.setOverwrite(true);
299 												dialog.setFileName(name);
300 												dialog.setFilterExtensions(new String[] { "*.jar", "*.*" });
301 												dialog.setFilterNames(new String[] { "Jar File(*.jar)", "All Files" });
302 												String fileSelected = dialog.open();
303 												if (fileSelected != null) {
304 													new DownloadJarFileJob(name, resource, fileSelected).schedule();
305 												}
306 											}
307 										}
308 									});
309 								}
310 							}
311 						}
312 					});
313 				}
314 		    });
315 	    }

352 	public void saveClassFile(String className, final BlobValue bv) {
353 		Shell shell = PlatformUI.getWorkbench().getActiveWorkbenchWindow()	.getShell();
354 		FileDialog dialog = new FileDialog(shell, SWT.SAVE);
355 		dialog.setOverwrite(true);
356 		dialog.setFileName(className + ".class");
357 		dialog.setFilterExtensions(new String[] { "*.class", "*.*" });
358 		dialog.setFilterNames(new String[] { "Class File(*.class)", "All Files" });
359 		final String fileSelected = dialog.open();
360 		if (fileSelected != null) {
361 			ExUtil.asyncRun("Decompile-" + className + TimeUtil.getCurrentTime(serverId), new Runnable() {
362 				public void run() {
363 					FileUtil.save(fileSelected, bv.value);
364 					ConsoleProxy.infoSafe(fileSelected + " saved.");
365 				}
366 			});
367 		}

372 		parentGroup.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));

382 		applyButton.setLayoutData(new GridData(SWT.RIGHT, SWT.CENTER, false, false));
383 		applyButton.setImage(Images.filter);
384 		applyButton.setText("Apply Filter");

390 						load();
391 					}

397 			public void keyTraversed(TraverseEvent e) {

405 		leftButton.setLayoutData(new GridData(SWT.RIGHT, SWT.CENTER, false, false));

407 		leftButton.addListener(SWT.Selection, new Listener() {
408 			public void handleEvent(Event event) {
409 				if (currentPage > 1) {
410 					currentPage--;
411 					ExUtil.exec(new Runnable() {
412 						public void run() {
413 							load();
414 						}
415 					});
416 				}
417 			}
418 		});

420         rightButton = new Button(parentGroup, SWT.PUSH);
421         rightButton.setLayoutData(new GridData(SWT.RIGHT, SWT.CENTER, false, false));

423         rightButton.addListener(SWT.Selection, new Listener() {
424

429 						public void run() {
430 							load();
431 						}
432 					});
433 				}
434 			}
435 			
436 		});
437         validatePageButton();
438 	}

440 	private void createColumns() {
441 		for (ClassColumnEnum column : ClassColumnEnum.values()) {
442 			TableViewerColumn c = createTableViewerColumn(column.getTitle(), column.getWidth(), column.getAlignment(), column.isResizable(), column.isMoveable(), column.isNumber());
443 			ColumnLabelProvider labelProvider = null;
444 			switch (column) {
445 			case NO:
446 				labelProvider = new ColumnLabelProvider() {
447 					@Override
448 					public String getText(Object element) {
449 						if (element instanceof ClassData) {
450 							return String.valueOf(((ClassData)element).index);
451 						}
452 						return null;
453 					}
454 				};
455 				break;
456 			case TYPE:
457 				labelProvider = new ColumnLabelProvider() {
458 					@Override
459 					public String getText(Object element) {
460 						if (element instanceof ClassData) {
461 							return ((ClassData) element).type;
462 						}
463 						return null;
464 					}
465 				};
466 				break;
467 			case NAME:
468 				labelProvider = new ColumnLabelProvider() {
469 					@Override
470 					public String getText(Object element) {
471 						if (element instanceof ClassData) {
472 							return ((ClassData) element).name;
473 						}
474 						return null;
475 					}
476 				};
477 				break;
478 			case SUPERCLASS:
479 				labelProvider = new ColumnLabelProvider() {
480 					@Override
481 					public String getText(Object element) {
482 						if (element instanceof ClassData) {
483 							return ((ClassData) element).superClass;
484 						}
485 						return null;
486 					}
487 				};
488 				break;
489 			case INTERFACES:
490 				labelProvider = new ColumnLabelProvider() {
491 					@Override
492 					public String getText(Object element) {
493 						if (element instanceof ClassData) {
494 							return ((ClassData) element).interfaces;
495 						}
496 						return null;
497 					}
498 				};
499 				break;
500 			case RESOURCE:
501 				labelProvider = new ColumnLabelProvider() {
502 					@Override
503 					public String getText(Object element) {
504 						if (element instanceof ClassData) {
505 							return ((ClassData) element).resources;
506 						}
507 						return null;
508 					}
509 				};
510 				break;
511 			}
512 			if (labelProvider != null) {
513 				c.setLabelProvider(labelProvider);
514 			}
515 		}
516 	}

518 	private TableViewerColumn createTableViewerColumn(String title, int width, int alignment,  boolean resizable, boolean moveable, final boolean isNumber) {
519 		final TableViewerColumn viewerColumn = new TableViewerColumn(tableViewer, SWT.NONE);
520 		final TableColumn column = viewerColumn.getColumn();
521 		column.setText(title);
522 		column.setAlignment(alignment);
523 		column.setMoveable(moveable);
524 		tableColumnLayout.setColumnData(column, new ColumnWeightData(width, width, resizable));
525 		column.setData("isNumber", isNumber);
526 		column.addSelectionListener(new SelectionAdapter() {
527 			public void widgetSelected(SelectionEvent e) {
528 				ColumnLabelSorter sorter = (ColumnLabelSorter) tableViewer.getComparator();
529 				TableColumn selectedColumn = (TableColumn) e.widget;
530 				sorter.setColumn(selectedColumn);
531 			}
532 		});
533 		return viewerColumn;
534 	}

539 	private void validatePageButton() {
540 		if (this.currentPage <= 1) {
541 			this.leftButton.setEnabled(false);
542 		} else {
543 			this.leftButton.setEnabled(true);
544 		}
545 		
546 		if (this.currentPage == this.totalPage) {
547 			this.rightButton.setEnabled(false);
548 		} else {
549 			this.rightButton.setEnabled(true);
550 		}
551 	}

553 	private void selectionCopyToClipboard() {
554 		if (tableViewer != null) {
555 			TableItem[] items = tableViewer.getTable().getSelection();
556 			if (items != null && items.length > 0) {
557 				StringBuffer sb = new StringBuffer();
558 				for (int i = 0; i < items.length; i++) {
559 					ClassData data = (ClassData) items[i].getData();
560 					sb.append(data.toString());
561 				}
562 				clipboard.setContents(new Object[] {sb.toString()}, new Transfer[] {TextTransfer.getInstance()});
563 			}
564 		}
565 	}

637 		}

639 		protected IStatus run(IProgressMonitor monitor) {
640 			monitor.beginTask("Redefine class", IProgressMonitor.UNKNOWN);
641 			TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
642 			MapPack p = null;
643 			try {
644 				MapPack param = new MapPack();
645 				param.put("objHash", objHash);
646 				param.put("class", classLv);
647 				p = (MapPack) tcp.getSingle(RequestCmd.REDEFINE_CLASSES, param);
648 			} catch(Exception e) {
649 				ConsoleProxy.errorSafe(e.getMessage());
650 			} finally {
651 				TcpProxy.putTcpProxy(tcp);
652 			}
653 			if (p != null) {
654 				boolean success = p.getBoolean("success");
655 				if (success) {
656 					ConsoleProxy.infoSafe("Redefine complete");
657 				} else {
658 					String error = p.getText("error");
659 					if (StringUtil.isNotEmpty(error)) {
660 						ConsoleProxy.errorSafe(error);
661 					} else {
662 						ConsoleProxy.errorSafe("Redefine failed.");
663 					}
664 				}
665 			}
666 			return Status.CANCEL_STATUS;
667 		}

680 		}

682 		protected IStatus run(IProgressMonitor monitor) {
683 			monitor.beginTask("Downloading resource", IProgressMonitor.UNKNOWN);
684 			TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
685 			MapPack p = null;
686 			try {
687 				MapPack param = new MapPack();
688 				param.put("objHash", objHash);
689 				param.put("resource", resource);
690 				p = (MapPack) tcp.getSingle(RequestCmd.OBJECT_DOWNLOAD_JAR, param);
691 			} catch(Exception e) {
692 				ConsoleProxy.errorSafe(e.getMessage());
693 			} finally {
694 				TcpProxy.putTcpProxy(tcp);
695 			}
696 			if (p != null) {
697 				String error = p.getText("error");
698 				if (StringUtil.isNotEmpty(error)) {
699 					ConsoleProxy.errorSafe(error);
700 				} else {
701 					Value v = p.get("jar");
702 					if (v != null) {
703 						BlobValue bv = (BlobValue) v;
704 						FileUtil.save(saveFile, bv.value);
705 						return Status.OK_STATUS;
706 					}
707 				}
708 			}
709 			return Status.CANCEL_STATUS;
