118     static class JsrHook extends ReturnHook {
119         ArrayList jsrList;
120         CodeGen cgen;
121         int var;
122 
123         JsrHook(CodeGen gen) {
124             super(gen);
125             jsrList = new ArrayList();
126             cgen = gen;
127             var = -1;
128         }
129 
130         private int getVar(int size) {
131             if (var < 0) {
132                 var = cgen.getMaxLocals();
133                 cgen.incMaxLocals(size);
134             }
135 
136             return var;
137         }
138 
139         private void jsrJmp(Bytecode b) {
140             b.addOpcode(Opcode.GOTO);
141             jsrList.add(new int[] {b.currentPc(), var});
142             b.addIndex(0);
143         }
144 
145         protected boolean doit(Bytecode b, int opcode) {
146             switch (opcode) {
147             case Opcode.RETURN :
148                 jsrJmp(b);
149                 break;
150             case ARETURN :
151                 b.addAstore(getVar(1));
152                 jsrJmp(b);
153                 b.addAload(var);
154                 break;
155             case IRETURN :
156                 b.addIstore(getVar(1));
157                 jsrJmp(b);
158                 b.addIload(var);
159                 break;
160             case LRETURN :
161                 b.addLstore(getVar(2));
162                 jsrJmp(b);
163                 b.addLload(var);
164                 break;
165             case DRETURN :
166                 b.addDstore(getVar(2));
167                 jsrJmp(b);
168                 b.addDload(var);
169                 break;
170             case FRETURN :
171                 b.addFstore(getVar(1));
172                 jsrJmp(b);
173                 b.addFload(var);
174                 break;
175             default :
176                 throw new RuntimeException("fatal");
177             }
178 
179             return false;
180         }
181     }

183     static class JsrHook2 extends ReturnHook {
184         int var;
185         int target;
186 
187         JsrHook2(CodeGen gen, int[] retTarget) {
188             super(gen);
189             target = retTarget[0];
190             var = retTarget[1];
191         }
192 
193         protected boolean doit(Bytecode b, int opcode) {
194             switch (opcode) {
195             case Opcode.RETURN :
196                 break;
197             case ARETURN :
198                 b.addAstore(var);
199                 break;
200             case IRETURN :
201                 b.addIstore(var);
202                 break;
203             case LRETURN :
204                 b.addLstore(var);
205                 break;
206             case DRETURN :
207                 b.addDstore(var);
208                 break;
209             case FRETURN :
210                 b.addFstore(var);
211                 break;
212             default :
213                 throw new RuntimeException("fatal");
214             }
215 
216             b.addOpcode(Opcode.GOTO);
217             b.addIndex(target - b.currentPc() + 3);
218             return true;
219         }
220     }

222     protected void atTryStmnt(Stmnt st) throws CompileError {
223         Bytecode bc = bytecode;
224         Stmnt body = (Stmnt)st.getLeft();
225         if (body == null)
226             return;
227 
228         ASTList catchList = (ASTList)st.getRight().getLeft();
229         Stmnt finallyBlock = (Stmnt)st.getRight().getRight().getLeft();
230         ArrayList gotoList = new ArrayList(); 
231 
232         JsrHook jsrHook = null;
233         if (finallyBlock != null)
234             jsrHook = new JsrHook(this);
235 
236         int start = bc.currentPc();
237         body.accept(this);
238         int end = bc.currentPc();
239         if (start == end)
240             throw new CompileError("empty try block");
241 
242         boolean tryNotReturn = !hasReturned;
243         if (tryNotReturn) {
244             bc.addOpcode(Opcode.GOTO);
245             gotoList.add(new Integer(bc.currentPc()));
246             bc.addIndex(0);   // correct later
247         }
248 
249         int var = getMaxLocals();
250         incMaxLocals(1);
251         while (catchList != null) {
252             // catch clause
253             Pair p = (Pair)catchList.head();
254             catchList = catchList.tail();
255             Declarator decl = (Declarator)p.getLeft();
256             Stmnt block = (Stmnt)p.getRight();
257 
258             decl.setLocalVar(var);
259 
260             CtClass type = resolver.lookupClassByJvmName(decl.getClassName());
261             decl.setClassName(MemberResolver.javaToJvmName(type.getName()));
262             bc.addExceptionHandler(start, end, bc.currentPc(), type);
263             bc.growStack(1);
264             bc.addAstore(var);
265             hasReturned = false;
266             if (block != null)
267                 block.accept(this);
268 
269             if (!hasReturned) {
270                 bc.addOpcode(Opcode.GOTO);
271                 gotoList.add(new Integer(bc.currentPc()));
272                 bc.addIndex(0);   // correct later
273                 tryNotReturn = true;
274             }
275         }
276 
277         if (finallyBlock != null) {
278             jsrHook.remove(this);
279             // catch (any) clause
280             int pcAnyCatch = bc.currentPc();
281             bc.addExceptionHandler(start, pcAnyCatch, pcAnyCatch, 0);
282             bc.growStack(1);
283             bc.addAstore(var);
284             hasReturned = false;
285             finallyBlock.accept(this);
286             if (!hasReturned) {
287                 bc.addAload(var);
288                 bc.addOpcode(ATHROW);
289             }
290 
291             addFinally(jsrHook.jsrList, finallyBlock);
292         }
293 
294         int pcEnd = bc.currentPc();
295         patchGoto(gotoList, pcEnd);
296         hasReturned = !tryNotReturn;
297         if (finallyBlock != null) {
298             if (tryNotReturn)
299                 finallyBlock.accept(this);
300         }
301     }

303     /**
304      * Adds a finally clause for earch return statement.
305      */
306     private void addFinally(ArrayList returnList, Stmnt finallyBlock)
307         throws CompileError
308     {
309         Bytecode bc = bytecode;
310         int n = returnList.size();
311         for (int i = 0; i < n; ++i) {
312             final int[] ret = (int[])returnList.get(i);
313             int pc = ret[0];
314             bc.write16bit(pc, bc.currentPc() - pc + 1);
315             ReturnHook hook = new JsrHook2(this, ret);
316             finallyBlock.accept(this);
317             hook.remove(this);
318             if (!hasReturned) {
319                 bc.addOpcode(Opcode.GOTO);
320                 bc.addIndex(pc + 3 - bc.currentPc());
321             }
322         }
323     }

325     public void atNewExpr(NewExpr expr) throws CompileError {
326         if (expr.isArray())
327             atNewArrayExpr(expr);
328         else {
329             CtClass clazz = resolver.lookupClassByName(expr.getClassName());
330             String cname = clazz.getName();
331             ASTList args = expr.getArguments();
332             bytecode.addNew(cname);
333             bytecode.addOpcode(DUP);
334 
335             atMethodCallCore(clazz, MethodInfo.nameInit, args,
336                              false, true, -1, null);
337 
338             exprType = CLASS;
339             arrayDim = 0;
340             className = MemberResolver.javaToJvmName(cname);
341         }
342     }

344     public void atNewArrayExpr(NewExpr expr) throws CompileError {
345         int type = expr.getArrayType();
346         ASTList size = expr.getArraySize();
347         ASTList classname = expr.getClassName();
348         ArrayInit init = expr.getInitializer();
349         if (size.length() > 1) {
350             if (init != null)
351                 throw new CompileError(
352                         "sorry, multi-dimensional array initializer " +
353                         "for new is not supported");
354 
355             atMultiNewArray(type, classname, size);
356             return;
357         }
358 
359         ASTree sizeExpr = size.head();
360         atNewArrayExpr2(type, sizeExpr, Declarator.astToClassName(classname, '/'), init);
361     }

363     private void atNewArrayExpr2(int type, ASTree sizeExpr,
364                         String jvmClassname, ArrayInit init) throws CompileError {
365         if (init == null)
366             if (sizeExpr == null)
367                 throw new CompileError("no array size");
368             else
369                 sizeExpr.accept(this);
370         else
371             if (sizeExpr == null) {
372                 int s = init.length();
373                 bytecode.addIconst(s);
374             }
375             else
376                 throw new CompileError("unnecessary array size specified for new");
377 
378         String elementClass;
379         if (type == CLASS) {
380             elementClass = resolveClassName(jvmClassname);
381             bytecode.addAnewarray(MemberResolver.jvmToJavaName(elementClass));
382         }
383         else {
384             elementClass = null;
385             int atype = 0;
386             switch (type) {
387             case BOOLEAN :
388                 atype = T_BOOLEAN;
389                 break;
390             case CHAR :
391                 atype = T_CHAR;
392                 break;
393             case FLOAT :
394                 atype = T_FLOAT;
395                 break;
396             case DOUBLE :
397                 atype = T_DOUBLE;
398                 break;
399             case BYTE :
400                 atype = T_BYTE;
401                 break;
402             case SHORT :
403                 atype = T_SHORT;
404                 break;
405             case INT :
406                 atype = T_INT;
407                 break;
408             case LONG :
409                 atype = T_LONG;
410                 break;
411             default :
412                 badNewExpr();
413                 break;
414             }
415 
416             bytecode.addOpcode(NEWARRAY);
417             bytecode.add(atype);
418         }
419 
420         if (init != null) {
421             int s = init.length();
422             ASTList list = init;
423             for (int i = 0; i < s; i++) {
424                 bytecode.addOpcode(DUP);
425                 bytecode.addIconst(i);
426                 list.head().accept(this);
427                 if (!isRefType(type))
428                     atNumCastExpr(exprType, type);
429 
430                 bytecode.addOpcode(getArrayWriteOp(type, 0));
431                 list = list.tail();
432             }
433         }
434 
435         exprType = type;
436         arrayDim = 1;
437         className = elementClass;
438     }

453     protected void atMultiNewArray(int type, ASTList classname, ASTList size)
454         throws CompileError
455     {
456         int count, dim;
457         dim = size.length();
458         for (count = 0; size != null; size = size.tail()) {
459             ASTree s = size.head();
460             if (s == null)
461                 break;          // int[][][] a = new int[3][4][];
462 
463             ++count;
464             s.accept(this);
465             if (exprType != INT)
466                 throw new CompileError("bad type for array size");
467         }
468 
469         String desc;
470         exprType = type;
471         arrayDim = dim;
472         if (type == CLASS) {
473             className = resolveClassName(classname);
474             desc = toJvmArrayName(className, dim);
475         }
476         else
477             desc = toJvmTypeName(type, dim);
478 
479         bytecode.addMultiNewarray(desc, count);
480     }

482     public void atCallExpr(CallExpr expr) throws CompileError {
483         String mname = null;
484         CtClass targetClass = null;
485         ASTree method = expr.oprand1();
486         ASTList args = (ASTList)expr.oprand2();
487         boolean isStatic = false;
488         boolean isSpecial = false;
489         int aload0pos = -1;
490 
491         MemberResolver.Method cached = expr.getMethod();
492         if (method instanceof Member) {
493             mname = ((Member)method).get();
494             targetClass = thisClass;
495             if (inStaticMethod || (cached != null && cached.isStatic()))
496                 isStatic = true;            // should be static
497             else {
498                 aload0pos = bytecode.currentPc();
499                 bytecode.addAload(0);       // this
500             }
501         }
502         else if (method instanceof Keyword) {   // constructor
503             isSpecial = true;
504             mname = MethodInfo.nameInit;        // <init>
505             targetClass = thisClass;
506             if (inStaticMethod)
507                 throw new CompileError("a constructor cannot be static");
508             else
509                 bytecode.addAload(0);   // this
510 
511             if (((Keyword)method).get() == SUPER)
512                 targetClass = MemberResolver.getSuperclass(targetClass);
513         }
514         else if (method instanceof Expr) {
515             Expr e = (Expr)method;
516             mname = ((Symbol)e.oprand2()).get();
517             int op = e.getOperator();
518             if (op == MEMBER) {                 // static method
519                 targetClass
520                     = resolver.lookupClass(((Symbol)e.oprand1()).get(), false);
521                 isStatic = true;
522             }
523             else if (op == '.') {
524                 ASTree target = e.oprand1();
525                 String classFollowedByDotSuper = TypeChecker.isDotSuper(target);
526                 if (classFollowedByDotSuper != null) {
527                     isSpecial = true;
528                     targetClass = MemberResolver.getSuperInterface(thisClass,
529                                                         classFollowedByDotSuper);
530                     if (inStaticMethod || (cached != null && cached.isStatic()))
531                         isStatic = true;            // should be static
532                     else {
533                         aload0pos = bytecode.currentPc();
534                         bytecode.addAload(0);       // this
535                     }
536                 }
537                 else {
538                     if (target instanceof Keyword)
539                         if (((Keyword)target).get() == SUPER)
540                             isSpecial = true;
541 
542                     try {
543                         target.accept(this);
544                     }
545                     catch (NoFieldException nfe) {
546                         if (nfe.getExpr() != target)
547                             throw nfe;
548 
549                         // it should be a static method.
550                         exprType = CLASS;
551                         arrayDim = 0;
552                         className = nfe.getField(); // JVM-internal
553                         isStatic = true;
554                     }
555 
556                     if (arrayDim > 0)
557                         targetClass = resolver.lookupClass(javaLangObject, true);
558                     else if (exprType == CLASS /* && arrayDim == 0 */)
559                         targetClass = resolver.lookupClassByJvmName(className);
560                     else
561                         badMethod();
562                 }
563             }
564             else
565                 badMethod();
566         }
567         else
568             fatal();
569 
570         atMethodCallCore(targetClass, mname, args, isStatic, isSpecial,
571                          aload0pos, cached);
572     }

578     /*
579      * atMethodCallCore() is also called by doit() in NewExpr.ProceedForNew
580      *
581      * @param targetClass       the class at which method lookup starts.
582      * @param found         not null if the method look has been already done.
583      */
584     public void atMethodCallCore(CtClass targetClass, String mname,
585                         ASTList args, boolean isStatic, boolean isSpecial,
586                         int aload0pos, MemberResolver.Method found)
587         throws CompileError
588     {
589         int nargs = getMethodArgsLength(args);
590         int[] types = new int[nargs];
591         int[] dims = new int[nargs];
592         String[] cnames = new String[nargs];
593 
594         if (!isStatic && found != null && found.isStatic()) {
595             bytecode.addOpcode(POP);
596             isStatic = true;
597         }
598 
599         int stack = bytecode.getStackDepth();
600 
601         // generate code for evaluating arguments.
602         atMethodArgs(args, types, dims, cnames);
603 
604         if (found == null)
605             found = resolver.lookupMethod(targetClass, thisClass, thisMethod,
606                                           mname, types, dims, cnames);
607 
608         if (found == null) {
609             String msg;
610             if (mname.equals(MethodInfo.nameInit))
611                 msg = "constructor not found";
612             else
613                 msg = "Method " + mname + " not found in "
614                     + targetClass.getName();
615 
616             throw new CompileError(msg);
617         }
618 
619         atMethodCallCore2(targetClass, mname, isStatic, isSpecial,
620                           aload0pos, found);
621     }

623     private void atMethodCallCore2(CtClass targetClass, String mname,
624                                    boolean isStatic, boolean isSpecial,
625                                    int aload0pos,
626                                    MemberResolver.Method found)
627         throws CompileError
628     {
629         CtClass declClass = found.declaring;
630         MethodInfo minfo = found.info;
631         String desc = minfo.getDescriptor();
632         int acc = minfo.getAccessFlags();
633 
634         if (mname.equals(MethodInfo.nameInit)) {
635             isSpecial = true;
636             if (declClass != targetClass)
637                 throw new CompileError("no such constructor: " + targetClass.getName());
638 
639             if (declClass != thisClass && AccessFlag.isPrivate(acc)) {
640                 desc = getAccessibleConstructor(desc, declClass, minfo);
641                 bytecode.addOpcode(Opcode.ACONST_NULL); // the last parameter
642             }
643         }
644         else if (AccessFlag.isPrivate(acc))
645             if (declClass == thisClass)
646                 isSpecial = true;
647             else {
648                 isSpecial = false;
649                 isStatic = true;
650                 String origDesc = desc;
651                 if ((acc & AccessFlag.STATIC) == 0)
652                     desc = Descriptor.insertParameter(declClass.getName(),
653                                                       origDesc);
654 
655                 acc = AccessFlag.setPackage(acc) | AccessFlag.STATIC;
656                 mname = getAccessiblePrivate(mname, origDesc, desc,
657                                              minfo, declClass);
658             }
659 
660         boolean popTarget = false;
661         if ((acc & AccessFlag.STATIC) != 0) {
662             if (!isStatic) {
663                 /* this method is static but the target object is
664                    on stack.  It must be popped out.  If aload0pos >= 0,
665                    then the target object was pushed by aload_0.  It is
666                    overwritten by NOP.
667                 */
668                 isStatic = true;
669                 if (aload0pos >= 0)
670                     bytecode.write(aload0pos, NOP);
671                 else
672                     popTarget = true;
673             }
674 
675             bytecode.addInvokestatic(declClass, mname, desc);
676         }
677         else if (isSpecial)    // if (isSpecial && notStatic(acc))
678             bytecode.addInvokespecial(targetClass, mname, desc);
679         else {
680             if (!Modifier.isPublic(declClass.getModifiers())
681                 || declClass.isInterface() != targetClass.isInterface())
682                 declClass = targetClass;
683 
684             if (declClass.isInterface()) {
685                 int nargs = Descriptor.paramSize(desc) + 1;
686                 bytecode.addInvokeinterface(declClass, mname, desc, nargs);
687             }
688             else
689                 if (isStatic)
690                     throw new CompileError(mname + " is not static");
691                 else
692                     bytecode.addInvokevirtual(declClass, mname, desc);
693         }
694 
695         setReturnType(desc, isStatic, popTarget);
696     }

698     /*
699      * Finds (or adds if necessary) a hidden accessor if the method
700      * is in an enclosing class.
701      *
702      * @param desc          the descriptor of the method.
703      * @param declClass     the class declaring the method.
704      */
705     protected String getAccessiblePrivate(String methodName, String desc,
706                                           String newDesc, MethodInfo minfo,
707                                           CtClass declClass)
708         throws CompileError
709     {
710         if (isEnclosing(declClass, thisClass)) {
711             AccessorMaker maker = declClass.getAccessorMaker();
712             if (maker != null)
713                 return maker.getMethodAccessor(methodName, desc, newDesc,
714                                                minfo);
715         }
716 
717         throw new CompileError("Method " + methodName
718                                + " is private");
719     }

721     /*
722      * Finds (or adds if necessary) a hidden constructor if the given
723      * constructor is in an enclosing class.
724      *
725      * @param desc          the descriptor of the constructor.
726      * @param declClass     the class declaring the constructor.
727      * @param minfo         the method info of the constructor.
728      * @return the descriptor of the hidden constructor.
729      */
730     protected String getAccessibleConstructor(String desc, CtClass declClass,
731                                               MethodInfo minfo)
732         throws CompileError
733     {
734         if (isEnclosing(declClass, thisClass)) {
735             AccessorMaker maker = declClass.getAccessorMaker();
736             if (maker != null)
737                 return maker.getConstructor(declClass, desc, minfo);
738         }
739 
740         throw new CompileError("the called constructor is private in "
741                                + declClass.getName());
742     }

744     private boolean isEnclosing(CtClass outer, CtClass inner) {
745         try {
746             while (inner != null) {
747                 inner = inner.getDeclaringClass();
748                 if (inner == outer)
749                     return true;
750             }
751         }
752         catch (NotFoundException e) {}
753         return false;   
754     }

774     void setReturnType(String desc, boolean isStatic, boolean popTarget)
775         throws CompileError
776     {
777         int i = desc.indexOf(')');
778         if (i < 0)
779             badMethod();
780 
781         char c = desc.charAt(++i);
782         int dim = 0;
783         while (c == '[') {
784             ++dim;
785             c = desc.charAt(++i);
786         }
787 
788         arrayDim = dim;
789         if (c == 'L') {
790             int j = desc.indexOf(';', i + 1);
791             if (j < 0)
792                 badMethod();
793 
794             exprType = CLASS;
795             className = desc.substring(i + 1, j);
796         }
797         else {
798             exprType = MemberResolver.descToType(c);
799             className = null;
800         }
801 
802         int etype = exprType;
803         if (isStatic) {
804             if (popTarget) {
805                 if (is2word(etype, dim)) {
806                     bytecode.addOpcode(DUP2_X1);
807                     bytecode.addOpcode(POP2);
808                     bytecode.addOpcode(POP);
809                 }
810                 else if (etype == VOID)
811                     bytecode.addOpcode(POP);
812                 else {
813                     bytecode.addOpcode(SWAP);
814                     bytecode.addOpcode(POP);
815                 }
816             }
817         }
818     }

820     protected void atFieldAssign(Expr expr, int op, ASTree left,
821                         ASTree right, boolean doDup) throws CompileError
822     {
823         CtField f = fieldAccess(left, false);
824         boolean is_static = resultStatic;
825         if (op != '=' && !is_static)
826             bytecode.addOpcode(DUP);
827 
828         int fi;
829         if (op == '=') {
830             FieldInfo finfo = f.getFieldInfo2();
831             setFieldType(finfo);
832             AccessorMaker maker = isAccessibleField(f, finfo);            
833             if (maker == null)
834                 fi = addFieldrefInfo(f, finfo);
835             else
836                 fi = 0;
837         }
838         else
839             fi = atFieldRead(f, is_static);
840 
841         int fType = exprType;
842         int fDim = arrayDim;
843         String cname = className;
844 
845         atAssignCore(expr, op, right, fType, fDim, cname);
846 
847         boolean is2w = is2word(fType, fDim);
848         if (doDup) {
849             int dup_code;
850             if (is_static)
851                 dup_code = (is2w ? DUP2 : DUP);
852             else
853                 dup_code = (is2w ? DUP2_X1 : DUP_X1);
854 
855             bytecode.addOpcode(dup_code);
856         }
857 
858         atFieldAssignCore(f, is_static, fi, is2w);
859 
860         exprType = fType;
861         arrayDim = fDim;
862         className = cname;
863     }

865     /* If fi == 0, the field must be a private field in an enclosing class.
866      */
867     private void atFieldAssignCore(CtField f, boolean is_static, int fi,
868                                    boolean is2byte) throws CompileError {
869         if (fi != 0) {
870             if (is_static) {
871                bytecode.add(PUTSTATIC);
872                bytecode.growStack(is2byte ? -2 : -1);
873             }
874             else {
875                 bytecode.add(PUTFIELD);
876                 bytecode.growStack(is2byte ? -3 : -2);
877             }
878         
879             bytecode.addIndex(fi);
880         }
881         else {
882             CtClass declClass = f.getDeclaringClass();
883             AccessorMaker maker = declClass.getAccessorMaker();
884             // make should be non null.
885             FieldInfo finfo = f.getFieldInfo2();
886             MethodInfo minfo = maker.getFieldSetter(finfo, is_static);
887             bytecode.addInvokestatic(declClass, minfo.getName(),
888                                      minfo.getDescriptor());
889         }
890     }

898     protected void atFieldRead(ASTree expr) throws CompileError
899     {
900         CtField f = fieldAccess(expr, true);
901         if (f == null) {
902             atArrayLength(expr);
903             return;
904         }
905 
906         boolean is_static = resultStatic;
907         ASTree cexpr = TypeChecker.getConstantFieldValue(f);
908         if (cexpr == null)
909             atFieldRead(f, is_static);
910         else {
911             cexpr.accept(this);
912             setFieldType(f.getFieldInfo2());
913         }
914     }

925     /**
926      * Generates bytecode for reading a field value.
927      * It returns a fieldref_info index or zero if the field is a private
928      * one declared in an enclosing class. 
929      */
930     private int atFieldRead(CtField f, boolean isStatic) throws CompileError {
931         FieldInfo finfo = f.getFieldInfo2();
932         boolean is2byte = setFieldType(finfo);
933         AccessorMaker maker = isAccessibleField(f, finfo);
934         if (maker != null) {
935             MethodInfo minfo = maker.getFieldGetter(finfo, isStatic);
936             bytecode.addInvokestatic(f.getDeclaringClass(), minfo.getName(),
937                                      minfo.getDescriptor());
938             return 0;
939         }
940         else {
941             int fi = addFieldrefInfo(f, finfo);
942             if (isStatic) {
943                 bytecode.add(GETSTATIC);
944                 bytecode.growStack(is2byte ? 2 : 1);
945             }
946             else {
947                 bytecode.add(GETFIELD);
948                 bytecode.growStack(is2byte ? 1 : 0);
949             }
950 
951             bytecode.addIndex(fi);
952             return fi;
953         }
954     }

956     /**
957      * Returns null if the field is accessible.  Otherwise, it throws
958      * an exception or it returns AccessorMaker if the field is a private
959      * one declared in an enclosing class.
960      */
961     private AccessorMaker isAccessibleField(CtField f, FieldInfo finfo)
962         throws CompileError
963     {
964         if (AccessFlag.isPrivate(finfo.getAccessFlags())
965             && f.getDeclaringClass() != thisClass) {
966             CtClass declClass = f.getDeclaringClass(); 
967             if (isEnclosing(declClass, thisClass)) {
968                 AccessorMaker maker = declClass.getAccessorMaker();
969                 if (maker != null)
970                     return maker;
971                 else
972                     throw new CompileError("fatal error.  bug?");
973             }
974             else
975                 throw new CompileError("Field " + f.getName() + " in "
976                                        + declClass.getName() + " is private.");
977         }
978 
979         return null;    // accessible field
980     }

982     /**
983      * Sets exprType, arrayDim, and className.
984      *
985      * @return true if the field type is long or double. 
986      */
987     private boolean setFieldType(FieldInfo finfo) throws CompileError {
988         String type = finfo.getDescriptor();
989 
990         int i = 0;
991         int dim = 0;
992         char c = type.charAt(i);
993         while (c == '[') {
994             ++dim;
995             c = type.charAt(++i);
996         }
997 
998         arrayDim = dim;
999         exprType = MemberResolver.descToType(c);
1000 
1001         if (c == 'L')
1002             className = type.substring(i + 1, type.indexOf(';', i + 1));
1003         else
1004             className = null;
1005 
1006         boolean is2byte = dim == 0 && (c == 'J' || c == 'D');
1007         return is2byte;
1008     }

1049     /* This method also returns a value in resultStatic.
1050      *
1051      * @param acceptLength      true if array length is acceptable
1052      */
1053     protected CtField fieldAccess(ASTree expr, boolean acceptLength)
1054             throws CompileError
1055     {
1056         if (expr instanceof Member) {
1057             String name = ((Member)expr).get();
1058             CtField f = null;
1059             try {
1060                 f = thisClass.getField(name);
1061             }
1062             catch (NotFoundException e) {
1063                 // EXPR might be part of a static member access?
1064                 throw new NoFieldException(name, expr);
1065             }
1066 
1067             boolean is_static = Modifier.isStatic(f.getModifiers());
1068             if (!is_static)
1069                 if (inStaticMethod)
1070                     throw new CompileError(
1071                                 "not available in a static method: " + name);
1072                 else
1073                     bytecode.addAload(0);       // this
1074 
1075             resultStatic = is_static;
1076             return f;
1077         }
1078         else if (expr instanceof Expr) {
1079             Expr e = (Expr)expr;
1080             int op = e.getOperator();
1081             if (op == MEMBER) {
1082                 /* static member by # (extension by Javassist)
1083                  * For example, if int.class is parsed, the resulting tree
1084                  * is (# "java.lang.Integer" "TYPE"). 
1085                  */
1086                 CtField f = resolver.lookupField(((Symbol)e.oprand1()).get(),
1087                                          (Symbol)e.oprand2());
1088                 resultStatic = true;
1089                 return f;
1090             }
1091             else if (op == '.') {
1092                 CtField f = null;
1093                 try {
1094                     e.oprand1().accept(this);
1095                     /* Don't call lookupFieldByJvmName2().
1096                      * The left operand of . is not a class name but
1097                      * a normal expression.
1098                      */
1099                     if (exprType == CLASS && arrayDim == 0)
1100                         f = resolver.lookupFieldByJvmName(className,
1101                                                     (Symbol)e.oprand2());
1102                     else if (acceptLength && arrayDim > 0
1103                              && ((Symbol)e.oprand2()).get().equals("length"))
1104                         return null;    // expr is an array length.
1105                     else
1106                         badLvalue();
1107 
1108                     boolean is_static = Modifier.isStatic(f.getModifiers());
1109                     if (is_static)
1110                         bytecode.addOpcode(POP);
1111 
1112                     resultStatic = is_static;
1113                     return f;
1114                 }
1115                 catch (NoFieldException nfe) {
1116                     if (nfe.getExpr() != e.oprand1())
1117                         throw nfe;
1118 
1119                     /* EXPR should be a static field.
1120                      * If EXPR might be part of a qualified class name,
1121                      * lookupFieldByJvmName2() throws NoFieldException.
1122                      */
1123                     Symbol fname = (Symbol)e.oprand2();
1124                     String cname = nfe.getField();
1125                     f = resolver.lookupFieldByJvmName2(cname, fname, expr);
1126                     resultStatic = true;
1127                     return f;
1128                 }
1129             }
1130             else
1131                 badLvalue();
1132         }
1133         else
1134             badLvalue();
1135 
1136         resultStatic = false;
1137         return null;    // never reach
1138     }

1141         throw new CompileError("bad l-value");
1142     }
