42  /**
43   * Instanceof operator.
44   */
45  public class Instanceof extends Expr {
46      /**
47       * Undocumented constructor.  Do not use; internal-use only.
48       */
49      protected Instanceof(int pos, CodeIterator i, CtClass declaring,
50                           MethodInfo m) {
51          super(pos, i, declaring, m);
52      }
53  
54      /**
55       * Returns the method or constructor containing the instanceof
56       * expression represented by this object.
57       */
58      public CtBehavior where() { return super.where(); }
59  
60      /**
61       * Returns the line number of the source line containing the
62       * instanceof expression.
63       *
64       * @return -1       if this information is not available.
65       */
66      public int getLineNumber() {
67          return super.getLineNumber();
68      }
69  
70      /**
71       * Returns the source file containing the
72       * instanceof expression.
73       *
74       * @return null     if this information is not available.
75       */
76      public String getFileName() {
77          return super.getFileName();
78      }
79  
80      /**
81       * Returns the <code>CtClass</code> object representing
82       * the type name on the right hand side
83       * of the instanceof operator.
84       */
85      public CtClass getType() throws NotFoundException {
86          ConstPool cp = getConstPool();
87          int pos = currentPos;
88          int index = iterator.u16bitAt(pos + 1);
89          String name = cp.getClassInfo(index);
90          return thisClass.getClassPool().getCtClass(name);
91      }
92  
93      /**
94       * Returns the list of exceptions that the expression may throw.
95       * This list includes both the exceptions that the try-catch statements
96       * including the expression can catch and the exceptions that
97       * the throws declaration allows the method to throw.
98       */
99      public CtClass[] mayThrow() {
100         return super.mayThrow();
101     }
102 
103     /**
104      * Replaces the instanceof operator with the bytecode derived from
105      * the given source text.
106      *
107      * <p>$0 is available but the value is <code>null</code>.
108      *
109      * @param statement         a Java statement except try-catch.
110      */
111     public void replace(String statement) throws CannotCompileException {
112         thisClass.getClassFile();   // to call checkModify().
113         ConstPool constPool = getConstPool();
114         int pos = currentPos;
115         int index = iterator.u16bitAt(pos + 1);
116 
117         Javac jc = new Javac(thisClass);
118         ClassPool cp = thisClass.getClassPool();
119         CodeAttribute ca = iterator.get();
120 
121         try {
122             CtClass[] params
123                 = new CtClass[] { cp.get(javaLangObject) };
124             CtClass retType = CtClass.booleanType;
125 
126             int paramVar = ca.getMaxLocals();
127             jc.recordParams(javaLangObject, params, true, paramVar,
128                             withinStatic());
129             int retVar = jc.recordReturnType(retType, true);
130             jc.recordProceed(new ProceedForInstanceof(index));
131 
132             // because $type is not the return type...
133             jc.recordType(getType());
134 
135             /* Is $_ included in the source code?
136              */
137             checkResultValue(retType, statement);
138 
139             Bytecode bytecode = jc.getBytecode();
140             storeStack(params, true, paramVar, bytecode);
141             jc.recordLocalVariables(ca, pos);
142 
143             bytecode.addConstZero(retType);
144             bytecode.addStore(retVar, retType);     // initialize $_
145 
146             jc.compileStmnt(statement);
147             bytecode.addLoad(retVar, retType);
148 
149             replace0(pos, bytecode, 3);
150         }
151         catch (CompileError e) { throw new CannotCompileException(e); }
152         catch (NotFoundException e) { throw new CannotCompileException(e); }
153         catch (BadBytecode e) {
154             throw new CannotCompileException("broken method");
155         }
156     }
157 
158     /* boolean $proceed(Object obj)
159      */
160     static class ProceedForInstanceof implements ProceedHandler {
161         int index;
162 
163         ProceedForInstanceof(int i) {
164             index = i;
165         }
166 
167         public void doit(JvstCodeGen gen, Bytecode bytecode, ASTList args)
168             throws CompileError
169         {
170             if (gen.getMethodArgsLength(args) != 1)
171                 throw new CompileError(Javac.proceedName
172                         + "() cannot take more than one parameter "
173                         + "for instanceof");
174 
175             gen.atMethodArgs(args, new int[1], new int[1], new String[1]);
176             bytecode.addOpcode(Opcode.INSTANCEOF);
177             bytecode.addIndex(index);
178             gen.setType(CtClass.booleanType);
179         }
180 
181         public void setReturnType(JvstTypeChecker c, ASTList args)
182             throws CompileError
183         {
184             c.atMethodArgs(args, new int[1], new int[1], new String[1]);
185             c.setType(CtClass.booleanType);
186         }
187     }
188 }
