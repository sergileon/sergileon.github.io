8   package org.csstudio.swt.xygraph.toolbar;
9   
10  import org.csstudio.swt.xygraph.figures.Annotation;
11  import org.csstudio.swt.xygraph.figures.XYGraph;
12  import org.eclipse.jface.dialogs.Dialog;
13  import org.eclipse.swt.SWT;
14  import org.eclipse.swt.layout.GridData;
15  import org.eclipse.swt.layout.GridLayout;
16  import org.eclipse.swt.widgets.Combo;
17  import org.eclipse.swt.widgets.Composite;
18  import org.eclipse.swt.widgets.Control;
19  import org.eclipse.swt.widgets.Label;
20  import org.eclipse.swt.widgets.Shell;
21  
22  /**The dialog for removing annotation.
23   * @author Xihui Chen
24   * @author Kay Kasemir layout tweaks
25   */
26  public class RemoveAnnotationDialog extends Dialog {
27  	private XYGraph xyGraph;
28  	private Combo annotationsCombo;
29  	private Annotation removedAnnotation;
30  	
31  	protected RemoveAnnotationDialog(Shell parentShell, XYGraph xyGraph) {
32  		super(parentShell);	
33  		this.xyGraph = xyGraph;
34          // Allow resize
35          setShellStyle(getShellStyle() | SWT.RESIZE);
36  	}
37  	
38  	@Override
39  	protected void configureShell(Shell newShell) {
40  		super.configureShell(newShell);
41  		newShell.setText("Remove Annotation");
42  	}
43  	
44  	@Override
45  	protected Control createDialogArea(Composite parent) {
46  		final Composite parent_composite = (Composite) super.createDialogArea(parent);
47          final Composite composite = new Composite(parent_composite, SWT.NONE);
48          composite.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true));
49  		composite.setLayout(new GridLayout(1, false));
50          final Label removeLabel = new Label(composite, SWT.None);
51          removeLabel.setLayoutData(new GridData());
52          if(xyGraph.getPlotArea().getAnnotationList().size() > 0){        	
53  	        removeLabel.setText("Select the annotation to be removed: ");        
54  	        annotationsCombo = new Combo(composite, SWT.DROP_DOWN);
55  	        annotationsCombo.setLayoutData(new GridData(SWT.FILL, 0, true, false));
56  	        for(Annotation annotation : xyGraph.getPlotArea().getAnnotationList())
57  	        	annotationsCombo.add(annotation.getName());
58  	        annotationsCombo.select(0);
59          }else{
60          	removeLabel.setText("There is no annotation on the graph."); 
61          }
62          
63  		return parent_composite;
64  	}
65  	 
66  	@Override
67  	protected void okPressed() {
68  		if(annotationsCombo != null)
69  			removedAnnotation = xyGraph.getPlotArea().getAnnotationList().get(
70  				annotationsCombo.getSelectionIndex());
71  		super.okPressed();
72  	}
73  	 
74  	/**
75  	 * @return the annotation to be removed.
76  	 */
77  	public Annotation getAnnotation() {
78  		return removedAnnotation;
79  	}
80  }
