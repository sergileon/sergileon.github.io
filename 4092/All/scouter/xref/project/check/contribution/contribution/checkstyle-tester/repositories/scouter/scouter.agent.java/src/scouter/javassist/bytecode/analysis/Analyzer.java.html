42  /**
43   * A data-flow analyzer that determines the type state of the stack and local
44   * variable table at every reachable instruction in a method. During analysis,
45   * bytecode verification is performed in a similar manner to that described
46   * in the JVM specification.
47   *
48   * <p>Example:</p>
49   *
50   * <pre>
51   * // Method to analyze
52   * public Object doSomething(int x) {
53   *     Number n;
54   *     if (x &lt; 5) {
55   *        n = new Double(0);
56   *     } else {
57   *        n = new Long(0);
58   *     }
59   *
60   *     return n;
61   * }
62   *
63   * // Which compiles to:
64   * // 0:   iload_1
65   * // 1:   iconst_5
66   * // 2:   if_icmpge   17
67   * // 5:   new #18; //class java/lang/Double
68   * // 8:   dup
69   * // 9:   dconst_0
70   * // 10:  invokespecial   #44; //Method java/lang/Double."&lt;init&gt;":(D)V
71   * // 13:  astore_2
72   * // 14:  goto    26
73   * // 17:  new #16; //class java/lang/Long
74   * // 20:  dup
75   * // 21:  lconst_1
76   * // 22:  invokespecial   #47; //Method java/lang/Long."&lt;init&gt;":(J)V
77   * // 25:  astore_2
78   * // 26:  aload_2
79   * // 27:  areturn
80   *
81   * public void analyzeIt(CtClass clazz, MethodInfo method) {
82   *     Analyzer analyzer = new Analyzer();
83   *     Frame[] frames = analyzer.analyze(clazz, method);
84   *     frames[0].getLocal(0).getCtClass(); // returns clazz;
85   *     frames[0].getLocal(1).getCtClass(); // returns java.lang.String
86   *     frames[1].peek(); // returns Type.INTEGER
87   *     frames[27].peek().getCtClass(); // returns java.lang.Number
88   * }
89   * </pre>
90   *
91   * @see FramePrinter
92   * @author Jason T. Greene
93   */
94  public class Analyzer implements Opcode {
95      private final SubroutineScanner scanner = new SubroutineScanner();
96      private CtClass clazz;
97      private ExceptionInfo[] exceptions;
98      private Frame[] frames;
99      private Subroutine[] subroutines;
100 
101     private static class ExceptionInfo {
102         private int end;
103         private int handler;
104         private int start;
105         private Type type;
106 
107         private ExceptionInfo(int start, int end, int handler, Type type) {
108             this.start = start;
109             this.end = end;
110             this.handler = handler;
111             this.type = type;
112         }
113     }

115     /**
116      * Performs data-flow analysis on a method and returns an array, indexed by
117      * instruction position, containing the starting frame state of all reachable
118      * instructions. Non-reachable code, and illegal code offsets are represented
119      * as a null in the frame state array. This can be used to detect dead code.
120      *
121      * If the method does not contain code (it is either native or abstract), null
122      * is returned.
123      *
124      * @param clazz the declaring class of the method
125      * @param method the method to analyze
126      * @return an array, indexed by instruction position, of the starting frame state,
127      *         or null if this method doesn't have code
128      * @throws BadBytecode if the bytecode does not comply with the JVM specification
129      */
130     public Frame[] analyze(CtClass clazz, MethodInfo method) throws BadBytecode {
131         this.clazz = clazz;
132         CodeAttribute codeAttribute = method.getCodeAttribute();
133         // Native or Abstract
134         if (codeAttribute == null)
135             return null;
136 
137         int maxLocals = codeAttribute.getMaxLocals();
138         int maxStack = codeAttribute.getMaxStack();
139         int codeLength = codeAttribute.getCodeLength();
140 
141         CodeIterator iter = codeAttribute.iterator();
142         IntQueue queue = new IntQueue();
143 
144         exceptions = buildExceptionInfo(method);
145         subroutines = scanner.scan(method);
146 
147         Executor executor = new Executor(clazz.getClassPool(), method.getConstPool());
148         frames = new Frame[codeLength];
149         frames[iter.lookAhead()] = firstFrame(method, maxLocals, maxStack);
150         queue.add(iter.next());
151         while (!queue.isEmpty()) {
152             analyzeNextEntry(method, iter, queue, executor);
153         }
154 
155         return frames;
156     }

158     /**
159      * Performs data-flow analysis on a method and returns an array, indexed by
160      * instruction position, containing the starting frame state of all reachable
161      * instructions. Non-reachable code, and illegal code offsets are represented
162      * as a null in the frame state array. This can be used to detect dead code.
163      *
164      * If the method does not contain code (it is either native or abstract), null
165      * is returned.
166      *
167      * @param method the method to analyze
168      * @return an array, indexed by instruction position, of the starting frame state,
169      *         or null if this method doesn't have code
170      * @throws BadBytecode if the bytecode does not comply with the JVM specification
171      */
172     public Frame[] analyze(CtMethod method) throws BadBytecode {
173         return analyze(method.getDeclaringClass(), method.getMethodInfo2());
174     }
175 
176     private void analyzeNextEntry(MethodInfo method, CodeIterator iter,
177             IntQueue queue, Executor executor) throws BadBytecode {
178         int pos = queue.take();
179         iter.move(pos);
180         iter.next();
181 
182         Frame frame = frames[pos].copy();
183         Subroutine subroutine = subroutines[pos];
184 
185         try {
186             executor.execute(method, pos, iter, frame, subroutine);
187         } catch (RuntimeException e) {
188             throw new BadBytecode(e.getMessage() + "[pos = " + pos + "]", e);
189         }
190 
191         int opcode = iter.byteAt(pos);
192 
193         if (opcode == TABLESWITCH) {
194             mergeTableSwitch(queue, pos, iter, frame);
195         } else if (opcode == LOOKUPSWITCH) {
196             mergeLookupSwitch(queue, pos, iter, frame);
197         } else if (opcode == RET) {
198             mergeRet(queue, iter, pos, frame, subroutine);
199         } else if (Util.isJumpInstruction(opcode)) {
200             int target = Util.getJumpTarget(pos, iter);
201 
202             if (Util.isJsr(opcode)) {
203                 // Merge the state before the jsr into the next instruction
204                 mergeJsr(queue, frames[pos], subroutines[target], pos, lookAhead(iter, pos));
205             } else if (! Util.isGoto(opcode)) {
206                 merge(queue, frame, lookAhead(iter, pos));
207             }
208 
209             merge(queue, frame, target);
210         } else if (opcode != ATHROW && ! Util.isReturn(opcode)) {
211             // Can advance to next instruction
212             merge(queue, frame, lookAhead(iter, pos));
213         }
214 
215         // Merge all exceptions that are reachable from this instruction.
216         // The redundancy is intentional, since the state must be based
217         // on the current instruction frame.
218         mergeExceptionHandlers(queue, method, pos, frame);
219     }

221     private ExceptionInfo[] buildExceptionInfo(MethodInfo method) {
222         ConstPool constPool = method.getConstPool();
223         ClassPool classes = clazz.getClassPool();
224 
225         ExceptionTable table = method.getCodeAttribute().getExceptionTable();
226         ExceptionInfo[] exceptions = new ExceptionInfo[table.size()];
227         for (int i = 0; i < table.size(); i++) {
228             int index = table.catchType(i);
229             Type type;
230             try {
231                 type = index == 0 ? Type.THROWABLE : Type.get(classes.get(constPool.getClassInfo(index)));
232             } catch (NotFoundException e) {
233                 throw new IllegalStateException(e.getMessage());
234             }
235 
236             exceptions[i] = new ExceptionInfo(table.startPc(i), table.endPc(i), table.handlerPc(i), type);
237         }
238 
239         return exceptions;
240     }

242     private Frame firstFrame(MethodInfo method, int maxLocals, int maxStack) {
243         int pos = 0;
244 
245         Frame first = new Frame(maxLocals, maxStack);
246         if ((method.getAccessFlags() & AccessFlag.STATIC) == 0) {
247             first.setLocal(pos++, Type.get(clazz));
248         }
249 
250         CtClass[] parameters;
251         try {
252             parameters = Descriptor.getParameterTypes(method.getDescriptor(), clazz.getClassPool());
253         } catch (NotFoundException e) {
254             throw new RuntimeException(e);
255         }
256 
257         for (int i = 0; i < parameters.length; i++) {
258             Type type = zeroExtend(Type.get(parameters[i]));
259             first.setLocal(pos++, type);
260             if (type.getSize() == 2)
261                 first.setLocal(pos++, Type.TOP);
262         }
263 
264         return first;
265     }

267     private int getNext(CodeIterator iter, int of, int restore) throws BadBytecode {
268         iter.move(of);
269         iter.next();
270         int next = iter.lookAhead();
271         iter.move(restore);
272         iter.next();
273 
274         return next;
275     }
276 
277     private int lookAhead(CodeIterator iter, int pos) throws BadBytecode {
278         if (! iter.hasNext())
279             throw new BadBytecode("Execution falls off end! [pos = " + pos + "]");
280 
281         return iter.lookAhead();
282     }

285     private void merge(IntQueue queue, Frame frame, int target) {
286         Frame old = frames[target];
287         boolean changed;
288 
289         if (old == null) {
290             frames[target] = frame.copy();
291             changed = true;
292         } else {
293             changed = old.merge(frame);
294         }
295 
296         if (changed) {
297             queue.add(target);
298         }
299     }
300 
301     private void mergeExceptionHandlers(IntQueue queue, MethodInfo method, int pos, Frame frame) {
302         for (int i = 0; i < exceptions.length; i++) {
303             ExceptionInfo exception = exceptions[i];
304 
305             // Start is inclusive, while end is exclusive!
306             if (pos >= exception.start && pos < exception.end) {
307                 Frame newFrame = frame.copy();
308                 newFrame.clearStack();
309                 newFrame.push(exception.type);
310                 merge(queue, newFrame, exception.handler);
311             }
312         }
313     }

315     private void mergeJsr(IntQueue queue, Frame frame, Subroutine sub, int pos, int next) throws BadBytecode {
316         if (sub == null)
317             throw new BadBytecode("No subroutine at jsr target! [pos = " + pos + "]");
318 
319         Frame old = frames[next];
320         boolean changed = false;
321 
322         if (old == null) {
323             old = frames[next] = frame.copy();
324             changed = true;
325         } else {
326             for (int i = 0; i < frame.localsLength(); i++) {
327                 // Skip everything accessed by a subroutine, mergeRet must handle this
328                 if (!sub.isAccessed(i)) {
329                     Type oldType = old.getLocal(i);
330                     Type newType = frame.getLocal(i);
331                     if (oldType == null) {
332                         old.setLocal(i, newType);
333                         changed = true;
334                         continue;
335                     }
336 
337                     newType = oldType.merge(newType);
338                     // Always set the type, in case a multi-type switched to a standard type.
339                     old.setLocal(i, newType);
340                     if (!newType.equals(oldType) || newType.popChanged())
341                         changed = true;
342                 }
343             }
344         }
345 
346         if (! old.isJsrMerged()) {
347             old.setJsrMerged(true);
348             changed = true;
349         }
350 
351         if (changed && old.isRetMerged())
352             queue.add(next);
353 
354     }

356     private void mergeLookupSwitch(IntQueue queue, int pos, CodeIterator iter, Frame frame) throws BadBytecode {
357         int index = (pos & ~3) + 4;
358         // default
359         merge(queue, frame, pos + iter.s32bitAt(index));
360         int npairs = iter.s32bitAt(index += 4);
361         int end = npairs * 8 + (index += 4);
362 
363         // skip "match"
364         for (index += 4; index < end; index += 8) {
365             int target = iter.s32bitAt(index) + pos;
366             merge(queue, frame, target);
367         }
368     }
369 
370     private void mergeRet(IntQueue queue, CodeIterator iter, int pos, Frame frame, Subroutine subroutine) throws BadBytecode {
371         if (subroutine == null)
372             throw new BadBytecode("Ret on no subroutine! [pos = " + pos + "]");
373 
374         Iterator callerIter = subroutine.callers().iterator();
375         while (callerIter.hasNext()) {
376             int caller = ((Integer) callerIter.next()).intValue();
377             int returnLoc = getNext(iter, caller, pos);
378             boolean changed = false;
379 
380             Frame old = frames[returnLoc];
381             if (old == null) {
382                 old = frames[returnLoc] = frame.copyStack();
383                 changed = true;
384             } else {
385                 changed = old.mergeStack(frame);
386             }
387 
388             for (Iterator i = subroutine.accessed().iterator(); i.hasNext(); ) {
389                 int index = ((Integer)i.next()).intValue();
390                 Type oldType = old.getLocal(index);
391                 Type newType = frame.getLocal(index);
392                 if (oldType != newType) {
393                     old.setLocal(index, newType);
394                     changed = true;
395                 }
396             }
397 
398             if (! old.isRetMerged()) {
399                 old.setRetMerged(true);
400                 changed = true;
401             }
402 
403             if (changed && old.isJsrMerged())
404                 queue.add(returnLoc);
405         }
406     }

409     private void mergeTableSwitch(IntQueue queue, int pos, CodeIterator iter, Frame frame) throws BadBytecode {
410         // Skip 4 byte alignment padding
411         int index = (pos & ~3) + 4;
412         // default
413         merge(queue, frame, pos + iter.s32bitAt(index));
414         int low = iter.s32bitAt(index += 4);
415         int high = iter.s32bitAt(index += 4);
416         int end = (high - low + 1) * 4 + (index += 4);
417 
418         // Offset table
419         for (; index < end; index += 4) {
420             int target = iter.s32bitAt(index) + pos;
421             merge(queue, frame, target);
422         }
423     }

425     private Type zeroExtend(Type type) {
426         if (type == Type.SHORT || type == Type.BYTE || type == Type.CHAR || type == Type.BOOLEAN)
427             return  Type.INTEGER;
428 
429         return type;
430     }
