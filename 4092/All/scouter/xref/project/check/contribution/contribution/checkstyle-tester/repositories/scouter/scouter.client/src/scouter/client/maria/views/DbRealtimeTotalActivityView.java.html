242 	public void refresh() {
243 		TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
244 		Value v = null;
245 		try {
246 			MapPack param = new MapPack();
247 			ListValue objHashLv = AgentModelThread.getInstance().getLiveObjHashLV(serverId, CounterConstants.MARIA_PLUGIN);
248 			if (objHashLv.size() > 0) {
249 				param.put("objHash", objHashLv);
250 				v = tcp.getSingleValue(RequestCmd.DB_REALTIME_ACTIVITY, param);
251 			}
252 		} catch (Exception e) {
253 			ConsoleProxy.errorSafe(e.toString());
254 		} finally {
255 			TcpProxy.putTcpProxy(tcp);
256 		}
257 		if (v == null) {
258 			ExUtil.exec(canvas, new Runnable() {
259 				public void run() {
260 					setTitleImage(Images.inactive);
261 					long now = TimeUtil.getCurrentTime(serverId);
262 					long stime = now - TIME_RANGE;
263 					xyGraph.primaryXAxis.setRange(stime, now + 1);
264 				}
265 			});
266 		} else {
267 			MapValue mv = (MapValue) v;
268 			final DecimalValue callValue = new DecimalValue(mv.getLong("call"));
269 			final DecimalValue selectValue = new DecimalValue(mv.getLong("select"));
270 			final DecimalValue insertValue = new DecimalValue(mv.getLong("insert"));
271 			final DecimalValue updateValue = new DecimalValue(mv.getLong("update"));
272 			final DecimalValue deleteValue = new DecimalValue(mv.getLong("delete"));
273 			
274 			ExUtil.exec(canvas, new Runnable() {
275 				public void run() {
276 					setTitleImage(Images.active);
277 					long now = TimeUtil.getCurrentTime(serverId) / REFRESH_INTERVAL * REFRESH_INTERVAL;
278 					long stime = now - TIME_RANGE;
279 					xyGraph.primaryXAxis.setRange(stime, now + 1);
280 					ValueLog valueLog = new ValueLog();
281 					valueLog.time = now;
282 					valueLog.delete = deleteValue.value;
283 					valueLog.update = updateValue.value;
284 					valueLog.insert = insertValue.value;
285 					valueLog.select = selectValue.value;
286 					valueLog.call = callValue.value;
287 					valueLogs.add(valueLog);
288 					if (valueLogs.size() > BUFFER_SIZE) {
289 						valueLogs.removeFirst();
290 					}
291 					if (isStackView) {
292 						updateValue.value += deleteValue.value;
293 						insertValue.value += updateValue.value;
294 						selectValue.value += insertValue.value;
295 						callValue.value += selectValue.value;
296 					}
297 					((CircularBufferDataProvider)callTrace.getDataProvider()).addSample(new Sample(now, callValue.value));
298 					((CircularBufferDataProvider)selectTrace.getDataProvider()).addSample(new Sample(now, selectValue.value));
299 					((CircularBufferDataProvider)insertTrace.getDataProvider()).addSample(new Sample(now, insertValue.value));
300 					((CircularBufferDataProvider)updateTrace.getDataProvider()).addSample(new Sample(now, updateValue.value));
301 					((CircularBufferDataProvider)deleteTrace.getDataProvider()).addSample(new Sample(now, deleteValue.value));
302 					xyGraph.primaryYAxis.setRange(0, getMaxYValue());
303 				}
304 				
305 				private double getMaxYValue() {
306 					if (isStackView) {
307 						return ChartUtil.getMax(((CircularBufferDataProvider)callTrace.getDataProvider()).iterator());
308 					}
309 					double value = ChartUtil.getMax(((CircularBufferDataProvider)callTrace.getDataProvider()).iterator());
310 					value = Math.max(value, ChartUtil.getMax(((CircularBufferDataProvider)selectTrace.getDataProvider()).iterator()));
311 					value = Math.max(value, ChartUtil.getMax(((CircularBufferDataProvider)insertTrace.getDataProvider()).iterator()));
312 					value = Math.max(value, ChartUtil.getMax(((CircularBufferDataProvider)updateTrace.getDataProvider()).iterator()));
313 					value = Math.max(value, ChartUtil.getMax(((CircularBufferDataProvider)deleteTrace.getDataProvider()).iterator()));
314 					return value;
315 				}
316 			});
317 		}
318 	}

320 	protected void changeMode() {
321 		CircularBufferDataProvider delProvider = (CircularBufferDataProvider)deleteTrace.getDataProvider();
322 		CircularBufferDataProvider upProvider = (CircularBufferDataProvider)updateTrace.getDataProvider();
323 		CircularBufferDataProvider inProvider = (CircularBufferDataProvider)insertTrace.getDataProvider();
324 		CircularBufferDataProvider selProvider = (CircularBufferDataProvider)selectTrace.getDataProvider();
325 		CircularBufferDataProvider callProvider = (CircularBufferDataProvider)callTrace.getDataProvider();
326 		delProvider.clearTrace();
327 		upProvider.clearTrace();
328 		inProvider.clearTrace();
329 		selProvider.clearTrace();
330 		callProvider.clearTrace();
331 		int size = valueLogs.size();
332 		if (size > 0) {
333 			ENTRY<ValueLog> entry = valueLogs.getFirst();
334 			if (isStackView) {
335 				do {
336 					ValueLog log = entry.item;
337 					double x = log.time;
338 					double delValue = log.delete;
339 					double upValue = delValue + log.update;
340 					double inValue = upValue + log.insert;
341 					double selValue = inValue + log.select;
342 					double callValue = selValue + log.call;
343 					callProvider.addSample(new Sample(x, callValue));
344 					selProvider.addSample(new Sample(x, selValue));
345 					inProvider.addSample(new Sample(x, inValue));
346 					upProvider.addSample(new Sample(x, upValue));
347 					delProvider.addSample(new Sample(x, delValue));
348 				} while ((entry = entry.next) != null);
349 			} else {
350 				do {
351 					ValueLog log = entry.item;
352 					double x = log.time;
353 					double delValue = log.delete;
354 					double upValue = log.update;
355 					double inValue = log.insert;
356 					double selValue = log.select;
357 					double callValue = log.call;
358 					callProvider.addSample(new Sample(x, callValue));
359 					selProvider.addSample(new Sample(x, selValue));
360 					inProvider.addSample(new Sample(x, inValue));
361 					upProvider.addSample(new Sample(x, upValue));
362 					delProvider.addSample(new Sample(x, delValue));
363 				} while ((entry = entry.next) != null);
364 			}
365 		}
366 		if (isStackView) {
367 			deleteTrace.setTraceType(TraceType.AREA);
368 			updateTrace.setTraceType(TraceType.AREA);
369 			insertTrace.setTraceType(TraceType.AREA);
370 			selectTrace.setTraceType(TraceType.AREA);
371 			callTrace.setTraceType(TraceType.AREA);
372 		} else {
373 			deleteTrace.setTraceType(TraceType.SOLID_LINE);
374 			updateTrace.setTraceType(TraceType.SOLID_LINE);
375 			insertTrace.setTraceType(TraceType.SOLID_LINE);
376 			selectTrace.setTraceType(TraceType.SOLID_LINE);
377 			callTrace.setTraceType(TraceType.SOLID_LINE);
378 		}
379 	}
