18  package scouter.client.group.view;
19  
20  import java.util.HashMap;
21  import java.util.Iterator;
22  import java.util.Map;
23  
24  import org.eclipse.swt.widgets.Composite;
25  import org.eclipse.ui.IViewSite;
26  import org.eclipse.ui.PartInitException;
27  
28  import scouter.client.net.TcpProxy;
29  import scouter.client.util.TimeUtil;
30  import scouter.client.util.ScouterUtil;
31  import scouter.client.views.ServiceGroupCommonView;
32  import scouter.lang.pack.MapPack;
33  import scouter.lang.value.ListValue;
34  import scouter.net.RequestCmd;
35  import scouter.util.CastUtil;
36  
37  public class ServiceGroupGroupView extends ServiceGroupCommonView {
38  	
39  	public final static String ID = ServiceGroupGroupView.class.getName();
40  	
41  	String grpName;
42  	private Map<Integer, ListValue> serverObjMap = new HashMap<Integer, ListValue>();
43  	
44  	public void init(IViewSite site) throws PartInitException {
45  		super.init(site);
46  		String secId = site.getSecondaryId();
47  		grpName = secId;
48  	}
49  
50  	public void createPartControl(Composite parent) {
51  		this.setPartName(grpName + " - Service[Throughput]");
52  		super.createPartControl(parent);
53  	}
54  
55  	@Override
56  	public MapPack fetch() {
57  		ScouterUtil.collectGroupObjcts(grpName, serverObjMap);
58  		HashMap<String, PerfStat> valueMap = new HashMap<String, PerfStat>();
59  		Iterator<Integer> itr = serverObjMap.keySet().iterator();
60  		while (itr.hasNext()) {
61  			int serverId = itr.next();
62  			TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
63  			try {
64  				MapPack param = new MapPack();
65  				param.put("objHash", serverObjMap.get(serverId));
66  				MapPack p = (MapPack) tcp.getSingle(RequestCmd.REALTIME_SERVICE_GROUP, param);
67  				if (p != null) {
68  					ListValue nameLv = p.getList("name");
69  					ListValue countLv = p.getList("count");
70  					ListValue elapsedLv = p.getList("elapsed");
71  					ListValue errorLv = p.getList("error");
72  					for (int i = 0, max = (nameLv == null ? 0 : nameLv.size()) ; i < max; i++) {
73  						String name = nameLv.getString(i);
74  						PerfStat perf = valueMap.get(name);
75  						if (perf == null) {
76  							perf = new PerfStat();
77  							valueMap.put(name, perf);
78  						}
79  						perf.count += CastUtil.cint(countLv.get(i));
80  						perf.elapsed += CastUtil.clong(elapsedLv.get(i));
81  						perf.error += CastUtil.cint(errorLv.get(i));
82  					}
83  				}
84  			} catch (Throwable th) {
85  				th.printStackTrace();
86  			} finally {
87  				TcpProxy.putTcpProxy(tcp);
88  			}
89  		}
90  		MapPack m = null;
91  		if (valueMap.size() > 0) {
92  			m = new MapPack();
93  			ListValue nameLv = m.newList("name");
94  			ListValue countLv = m.newList("count");
95  			ListValue elapsedLv = m.newList("elapsed");
96  			ListValue errorLv = m.newList("error");
97  			Iterator<String> itrr = valueMap.keySet().iterator();
98  			while (itrr.hasNext()) {
99  				String name = itrr.next();
100 				PerfStat perf = valueMap.get(name);
101 				nameLv.add(name);
102 				countLv.add(perf.count);
103 				elapsedLv.add(perf.elapsed);
104 				errorLv.add(perf.error);
105 			}
106 			long time = TimeUtil.getCurrentTime();
107 			m.put("time", time);
108 			fiveMinMap.put(time, m);
109 		}
110 		return m;
111 	}
112 	
113 	public static class PerfStat {
114 		public int count;
115 		public int error;
116 		public long elapsed;
117 
118 		public void add(PerfStat o) {
119 			this.count += o.count;
120 			this.error += o.error;
121 			this.elapsed += o.elapsed;
122 		}
123 	}
124 	
125 	@Override
126 	public void setTitleName(MODE mode) {
127 		switch (mode) {
128 			case THROUGHPUT:
129 				this.setPartName(grpName + " - Service[Throughput]");
130 				break;
131 			case ELASPED:
132 				this.setPartName(grpName + " - Service[Elapsed Time]");
133 				break;
134 		}		
135 	}
136 }
