8   package org.csstudio.swt.xygraph;
9   
10  
11  /** Access to preference settings.
12   * 
13   *  See preferences.ini for details on the available settings
14   *  @author Kay Kasemir
15   */
16  @SuppressWarnings("nls")
17  public class Preferences
18  {
19      public static final String PROHIBIT_ADVANCED_GRAPHICS = "prohibit_advanced_graphics"; //$NON-NLS-1$
20  	// useAdvancedGraphics() is called from many drawing operations, so
21      // only determine it once
22      private static boolean use_advanced_graphics = true;
23  
24    
25      public static boolean useAdvancedGraphics()
26      {
27      	if(use_advanced_graphics){
28      		String value = System.getProperty(PROHIBIT_ADVANCED_GRAPHICS); //$NON-NLS-1$
29      		if(value == null || !value.equals("true")) //$NON-NLS-1$
30      			return true;
31      		return  false;
32      	}
33          return false;
34      }
35  }
