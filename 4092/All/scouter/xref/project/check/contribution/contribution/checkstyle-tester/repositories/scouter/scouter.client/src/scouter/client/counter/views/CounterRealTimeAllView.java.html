287 	public void refresh() {
288 		if (manualRefresh) {
289 			manualRefresh = false;
290 			traces.clear();
291 			datas.clear();
292 			getPrevAllPerf();
293 		}
294 		final HashMap<Integer, Value> values = new HashMap<Integer, Value>();
295 		TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
296 		try {
297 			MapPack param = new MapPack();
298 
299 			param.put("objType", objType);
300 			param.put("counter", counter);
301 
302 			MapPack out = (MapPack) tcp.getSingle(RequestCmd.COUNTER_REAL_TIME_ALL, param);
303 			isActive = false;
304 			if (out != null) {
305 				ListValue objHash = out.getList("objHash");
306 				ListValue v = out.getList("value");
307 				for (int i = 0; i < objHash.size(); i++) {
308 					values.put(CastUtil.cint(objHash.get(i)), v.get(i));
309 					isActive = true;
310 				}
311 			}
312 		} catch (Throwable t) {
313 			ConsoleProxy.errorSafe(t.toString());
314 		} finally {
315 			TcpProxy.putTcpProxy(tcp);
316 		}
317 
318 		ExUtil.exec(canvas, new Runnable() {
319 			public void run() {
320 
321 				if (isActive) {
322 					setActive();
323 				} else {
324 					setInactive();
325 				}
326 
327 				long now = TimeUtil.getCurrentTime(serverId);
328 				xyGraph.primaryXAxis.setRange(now - DateUtil.MILLIS_PER_MINUTE * 5, now + 1);
329 				Iterator<Integer> itr = values.keySet().iterator();
330 				while (itr.hasNext()) {
331 					int objHash = itr.next();
332 					Value value = values.get(objHash);
333 					if (value != null && value.getValueType() != ValueEnum.NULL) {
334 						CircularBufferDataProvider data = getDataProvider(objHash);
335 						data.addSample(new Sample(now, CastUtil.cdouble(value)));
336 					}
337 
338 				}
339 				if (CounterUtil.isPercentValue(objType, counter)) {
340 					xyGraph.primaryYAxis.setRange(0, 100);
341 				} else {
342 					double max = getMaxValue();
343 					xyGraph.primaryYAxis.setRange(0, max);
344 				}
345 				CounterRealTimeAllView.this.redraw();
346 			}
347 		});
348 
349 		checkSettingChange();
350 	}

352 	private void checkSettingChange() {
353 		ExUtil.exec(canvas, new Runnable() {
354 			public void run() {
355 				int width = PManager.getInstance().getInt(PreferenceConstants.P_CHART_LINE_WIDTH);
356 				synchronized (traces) {
357 					for (Trace t : traces) {
358 						if (selectedMode == false && t.getLineWidth() != width) {
359 							t.setLineWidth(width);
360 						}
361 						int objHash = HashUtil.hash(t.getName());
362 						AgentObject agent = AgentModelThread.getInstance().getAgentObject(objHash);
363 						if (agent == null || agent.isAlive() == false || agent.getColor() == null) {
364 							return;
365 						}
366 						if (t.getTraceColor() != agent.getColor()) {
367 							t.setTraceColor(agent.getColor());
368 						}
369 					}
370 				}
371 			}
372 		});
373 	}

375 	private void getPrevAllPerf() {
376 		final ArrayList<MapPack> values = new ArrayList<MapPack>();
377 		TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
378 		try {
379 			MapPack param = new MapPack();
380 			long etime = TimeUtil.getCurrentTime(serverId);
381 			long stime = etime - DateUtil.MILLIS_PER_MINUTE * 5;
382 			param.put("stime", stime);
383 			param.put("etime", etime);
384 			param.put("objType", objType);
385 			param.put("counter", counter);
386 			tcp.process(RequestCmd.COUNTER_PAST_TIME_ALL, param,
387 					new INetReader() {
388 						public void process(DataInputX in) throws IOException {
389 							MapPack mpack = (MapPack) in.readPack();
390 							values.add(mpack);
391 						};
392 					});
393 		} catch (Throwable t) {
394 			ConsoleProxy.errorSafe(t.toString());
395 		} finally {
396 			TcpProxy.putTcpProxy(tcp);
397 		}
398 
399 		ExUtil.exec(this.canvas, new Runnable() {
400 			public void run() {
401 				for (MapPack mpack : values) {
402 					int objHash = mpack.getInt("objHash");
403 					ListValue time = mpack.getList("time");
404 					ListValue value = mpack.getList("value");
405 					if (time == null || time.size() < 1) {
406 						continue;
407 					}
408 					CircularBufferDataProvider provider = getDataProvider(objHash);
409 					provider.clearTrace();
410 					for (int i = 0; time != null && i < time.size(); i++) {
411 						long x = time.getLong(i);
412 						double y = value.getDouble(i);
413 						provider.addSample(new Sample(x, y));
414 					}
415 				}
416 			}
417 		});
418 	}

420 	private CircularBufferDataProvider getDataProvider(int objHash) {
421 		CircularBufferDataProvider data = datas.get(objHash);
422 		if (data == null) {
423 			data = new CircularBufferDataProvider(true);
424 			datas.put(objHash, data);
425 
426 			data.setBufferSize(155);
427 			data.setCurrentXDataArray(new double[] {});
428 			data.setCurrentYDataArray(new double[] {});
429 			String name = StringUtil.trimToEmpty(TextProxy.object.getLoadText(
430 					DateUtil.yyyymmdd(TimeUtil.getCurrentTime(serverId)), objHash,
431 					serverId));
432 			final Trace trace = new Trace(name, xyGraph.primaryXAxis, xyGraph.primaryYAxis, data);
433 			trace.setPointStyle(PointStyle.NONE);
434 
435 			trace.setLineWidth(PManager.getInstance().getInt(PreferenceConstants.P_CHART_LINE_WIDTH));
436 			trace.setTraceType(TraceType.SOLID_LINE);
437 			trace.setTraceColor(AgentColorManager.getInstance().assignColor(objType, objHash));
438 			
439 			xyGraph.addTrace(trace);
440 			traces.add(trace);
441 		}
442 		return data;
443 	}

445 	private double getMaxValue() {
446 		Iterator<Integer> objHashs = datas.keySet().iterator();
447 		double max = 0.0;
448 		Range xRange = xyGraph.primaryXAxis.getRange();
449 		double lower = xRange.getLower();
450 		double upper = xRange.getUpper();
451 		while (objHashs.hasNext()) {
452 			int objHash = objHashs.next();
453 			CircularBufferDataProvider data = datas.get(objHash);
454 			if (data != null) {
455 				for (int inx = 0; inx < data.getSize(); inx++) {
456 					Sample sample = (Sample) data.getSample(inx);
457 					double x = sample.getXValue();
458 					if(x < lower || x > upper) {
459 						continue;
460 					}
461 					double y = sample.getYValue();
462 					if (y > max) {
463 						max = y;
464 					}
465 				}
466 			}
467 		}
468 		return ChartUtil.getMaxValue(max);
469 	}

496 	public void onSelectObject(int objHash, final String objName, String objType) {
497 		if (this.objType.equals(objType) == false) {
498 			return;
499 		}
500 		ExUtil.exec(canvas, new Runnable() {
501 			public void run() {
502 				int width = PManager.getInstance().getInt(PreferenceConstants.P_CHART_LINE_WIDTH);
503 				for (Trace t : traces) {
504 					if (t.getName().equals(objName)) {
505 						t.setLineWidth(width + 2);
506 						selectedMode = true;
507 						break;
508 					}
509 				}
510 			}
511 		});
512 	}

518 				for (Trace t : traces) {
519 					t.setLineWidth(width);
520 				}
521 				selectedMode = false;
522 			}

524 	}

536 				}

538 					public void run() {
539 						redraw();
540 					}
541 				});
542 			}
543 		});
