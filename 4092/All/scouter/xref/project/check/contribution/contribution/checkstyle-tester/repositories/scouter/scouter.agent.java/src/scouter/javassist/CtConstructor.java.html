40  /**
41   * An instance of CtConstructor represents a constructor.
42   * It may represent a static constructor
43   * (class initializer).  To distinguish a constructor and a class
44   * initializer, call <code>isClassInitializer()</code>.
45   *
46   * <p>See the super class <code>CtBehavior</code> as well since
47   * a number of useful methods are in <code>CtBehavior</code>.
48   *
49   * @see CtClass#getDeclaredConstructors()
50   * @see CtClass#getClassInitializer()
51   * @see CtNewConstructor
52   */
53  public final class CtConstructor extends CtBehavior {
54      protected CtConstructor(MethodInfo minfo, CtClass declaring) {
55          super(declaring, minfo);
56      }
57  
58      /**
59       * Creates a constructor with no constructor body.
60       * The created constructor
61       * must be added to a class with <code>CtClass.addConstructor()</code>.
62       *
63       * <p>The created constructor does not include a constructor body,
64       * which must be specified with <code>setBody()</code>.
65       *
66       * @param declaring         the class to which the created method is added.
67       * @param parameters        a list of the parameter types
68       *
69       * @see CtClass#addConstructor(CtConstructor)
70       * @see CtConstructor#setBody(String)
71       * @see CtConstructor#setBody(CtConstructor,ClassMap)
72       */
73      public CtConstructor(CtClass[] parameters, CtClass declaring) {
74          this((MethodInfo)null, declaring);
75          ConstPool cp = declaring.getClassFile2().getConstPool();
76          String desc = Descriptor.ofConstructor(parameters);
77          methodInfo = new MethodInfo(cp, "<init>", desc);
78          setModifiers(Modifier.PUBLIC);
79      }

81      /**
82       * Creates a copy of a <code>CtConstructor</code> object.
83       * The created constructor must be
84       * added to a class with <code>CtClass.addConstructor()</code>.
85       *
86       * <p>All occurrences of class names in the created constructor
87       * are replaced with names specified by
88       * <code>map</code> if <code>map</code> is not <code>null</code>.
89       *
90       * <p>By default, all the occurrences of the names of the class
91       * declaring <code>src</code> and the superclass are replaced
92       * with the name of the class and the superclass that
93       * the created constructor is added to.
94       * This is done whichever <code>map</code> is null or not.
95       * To prevent this replacement, call <code>ClassMap.fix()</code>
96       * or <code>put()</code> to explicitly specify replacement.
97       *
98       * <p><b>Note:</b> if the <code>.class</code> notation (for example,
99       * <code>String.class</code>) is included in an expression, the
100      * Javac compiler may produce a helper method.
101      * Since this constructor never
102      * copies this helper method, the programmers have the responsiblity of
103      * copying it.  Otherwise, use <code>Class.forName()</code> in the
104      * expression.
105      *
106      * @param src       the source method.
107      * @param declaring    the class to which the created method is added.
108      * @param map       the hashtable associating original class names
109      *                  with substituted names.
110      *                  It can be <code>null</code>.
111      *
112      * @see CtClass#addConstructor(CtConstructor)
113      * @see ClassMap#fix(String)
114      */
115     public CtConstructor(CtConstructor src, CtClass declaring, ClassMap map)
116         throws CannotCompileException
117     {
118         this((MethodInfo)null, declaring);
119         copy(src, true, map);
120     }

136     /**
137      * Returns the constructor name followed by parameter types
138      * such as <code>javassist.CtConstructor(CtClass[],CtClass)</code>.
139      *
140      * @since 3.5
141      */
142     public String getLongName() {
143         return getDeclaringClass().getName()
144                + (isConstructor() ? Descriptor.toString(getSignature())
145                                   : ("." + MethodInfo.nameClinit + "()"));
146     }

148     /**
149      * Obtains the name of this constructor.
150      * It is the same as the simple name of the class declaring this
151      * constructor.  If this object represents a class initializer,
152      * then this method returns <code>"&lt;clinit&gt;"</code>.
153      */
154     public String getName() {
155         if (methodInfo.isStaticInitializer())
156             return MethodInfo.nameClinit;
157         else
158             return declaringClass.getSimpleName();
159     }

161     /**
162      * Returns true if the constructor (or static initializer)
163      * is the default one.  This method returns true if the constructor
164      * takes some arguments but it does not perform anything except
165      * calling <code>super()</code> (the no-argument constructor of
166      * the super class).
167      */
168     public boolean isEmpty() {
169         CodeAttribute ca = getMethodInfo2().getCodeAttribute();
170         if (ca == null)
171             return false;       // native or abstract??
172                                 // they are not allowed, though.
173 
174         ConstPool cp = ca.getConstPool();
175         CodeIterator it = ca.iterator();
176         try {
177             int pos, desc;
178             int op0 = it.byteAt(it.next());
179             return op0 == Opcode.RETURN     // empty static initializer
180                 || (op0 == Opcode.ALOAD_0
181                     && it.byteAt(pos = it.next()) == Opcode.INVOKESPECIAL
182                     && (desc = cp.isConstructor(getSuperclassName(),
183                                                 it.u16bitAt(pos + 1))) != 0
184                     && "()V".equals(cp.getUtf8Info(desc))
185                     && it.byteAt(it.next()) == Opcode.RETURN
186                     && !it.hasNext());
187         }
188         catch (BadBytecode e) {}
189         return false;
190     }
191 
192     private String getSuperclassName() {
193         ClassFile cf = declaringClass.getClassFile2();
194         return cf.getSuperclass();
195     }

197     /**
198      * Returns true if this constructor calls a constructor
199      * of the super class.  This method returns false if it
200      * calls another constructor of this class by <code>this()</code>. 
201      */
202     public boolean callsSuper() throws CannotCompileException {
203         CodeAttribute codeAttr = methodInfo.getCodeAttribute();
204         if (codeAttr != null) {
205             CodeIterator it = codeAttr.iterator();
206             try {
207                 int index = it.skipSuperConstructor();
208                 return index >= 0;
209             }
210             catch (BadBytecode e) {
211                 throw new CannotCompileException(e);
212             }
213         }
214 
215         return false;
216     }

218     /**
219      * Sets a constructor body.
220      *
221      * @param src       the source code representing the constructor body.
222      *                  It must be a single statement or block.
223      *                  If it is <code>null</code>, the substituted
224      *                  constructor body does nothing except calling
225      *                  <code>super()</code>.
226      */
227     public void setBody(String src) throws CannotCompileException {
228         if (src == null)
229             if (isClassInitializer())
230                 src = ";";
231             else
232                 src = "super();";
233 
234         super.setBody(src);
235     }

238      * Copies a constructor body from another constructor.
239      *
240      * <p>All occurrences of the class names in the copied body
241      * are replaced with the names specified by
242      * <code>map</code> if <code>map</code> is not <code>null</code>.
243      *
244      * @param src       the method that the body is copied from.
245      * @param map       the hashtable associating original class names
246      *                  with substituted names.
247      *                  It can be <code>null</code>.
248      */
249     public void setBody(CtConstructor src, ClassMap map)
250         throws CannotCompileException
251     {
252         setBody0(src.declaringClass, src.methodInfo,
253                  declaringClass, methodInfo, map);
254     }

256     /**
257      * Inserts bytecode just after another constructor in the super class
258      * or this class is called.
259      * It does not work if this object represents a class initializer.
260      *
261      * @param src       the source code representing the inserted bytecode.
262      *                  It must be a single statement or block.
263      */
264     public void insertBeforeBody(String src) throws CannotCompileException {
265         CtClass cc = declaringClass;
266         cc.checkModify();
267         if (isClassInitializer())
268             throw new CannotCompileException("class initializer");
269 
270         CodeAttribute ca = methodInfo.getCodeAttribute();
271         CodeIterator iterator = ca.iterator();
272         Bytecode b = new Bytecode(methodInfo.getConstPool(),
273                                   ca.getMaxStack(), ca.getMaxLocals());
274         b.setStackDepth(ca.getMaxStack());
275         Javac jv = new Javac(b, cc);
276         try {
277             jv.recordParams(getParameterTypes(), false);
278             jv.compileStmnt(src);
279             ca.setMaxStack(b.getMaxStack());
280             ca.setMaxLocals(b.getMaxLocals());
281             iterator.skipConstructor();
282             int pos = iterator.insertEx(b.get());
283             iterator.insert(b.getExceptionTable(), pos);
284             methodInfo.rebuildStackMapIf6(cc.getClassPool(), cc.getClassFile2());
285         }
286         catch (NotFoundException e) {
287             throw new CannotCompileException(e);
288         }
289         catch (CompileError e) {
290             throw new CannotCompileException(e);
291         }
292         catch (BadBytecode e) {
293             throw new CannotCompileException(e);
294         }
295     }

297     /* This method is called by addCatch() in CtBehavior.
298      * super() and this() must not be in a try statement.
299      */
300     int getStartPosOfBody(CodeAttribute ca) throws CannotCompileException {
301         CodeIterator ci = ca.iterator();
302         try {
303             ci.skipConstructor();
304             return ci.next();
305         }
306         catch (BadBytecode e) {
307             throw new CannotCompileException(e);
308         }

311     /**
312      * Makes a copy of this constructor and converts it into a method.
313      * The signature of the mehtod is the same as the that of this constructor.
314      * The return type is <code>void</code>.  The resulting method must be
315      * appended to the class specified by <code>declaring</code>.
316      * If this constructor is a static initializer, the resulting method takes
317      * no parameter.
318      *
319      * <p>An occurrence of another constructor call <code>this()</code>
320      * or a super constructor call <code>super()</code> is
321      * eliminated from the resulting method. 
322      *
323      * <p>The immediate super class of the class declaring this constructor
324      * must be also a super class of the class declaring the resulting method.
325      * If the constructor accesses a field, the class declaring the resulting method
326      * must also declare a field with the same name and type.
327      *
328      * @param name              the name of the resulting method.
329      * @param declaring         the class declaring the resulting method.
330      */
331     public CtMethod toMethod(String name, CtClass declaring)
332         throws CannotCompileException
333     {
334         return toMethod(name, declaring, null);
335     }

337     /**
338      * Makes a copy of this constructor and converts it into a method.
339      * The signature of the method is the same as the that of this constructor.
340      * The return type is <code>void</code>.  The resulting method must be
341      * appended to the class specified by <code>declaring</code>.
342      * If this constructor is a static initializer, the resulting method takes
343      * no parameter.
344      *
345      * <p>An occurrence of another constructor call <code>this()</code>
346      * or a super constructor call <code>super()</code> is
347      * eliminated from the resulting method. 
348      *
349      * <p>The immediate super class of the class declaring this constructor
350      * must be also a super class of the class declaring the resulting method
351      * (this is obviously true if the second parameter <code>declaring</code> is
352      * the same as the class declaring this constructor).
353      * If the constructor accesses a field, the class declaring the resulting method
354      * must also declare a field with the same name and type.
355      *
356      * @param name              the name of the resulting method.
357      * @param declaring         the class declaring the resulting method.
358      *                          It is normally the same as the class declaring this
359      *                          constructor.
360      * @param map       the hash table associating original class names
361      *                  with substituted names.  The original class names will be
362      *                  replaced while making a copy.
363      *                  <code>map</code> can be <code>null</code>.
364      */
365     public CtMethod toMethod(String name, CtClass declaring, ClassMap map)
366         throws CannotCompileException
367     {
368         CtMethod method = new CtMethod(null, declaring);
369         method.copy(this, false, map);
370         if (isConstructor()) {
371             MethodInfo minfo = method.getMethodInfo2();
372             CodeAttribute ca = minfo.getCodeAttribute();
373             if (ca != null) {
374                 removeConsCall(ca);
375                 try {
376                     methodInfo.rebuildStackMapIf6(declaring.getClassPool(),
377                                                   declaring.getClassFile2());
378                 }
379                 catch (BadBytecode e) {
380                     throw new CannotCompileException(e);
381                 }
382             }
383         }
384 
385         method.setName(name);
386         return method;
387     }
388 
389     private static void removeConsCall(CodeAttribute ca)
390         throws CannotCompileException
391     {
392         CodeIterator iterator = ca.iterator();
393         try {
394             int pos = iterator.skipConstructor();
395             if (pos >= 0) {
396                 int mref = iterator.u16bitAt(pos + 1);
397                 String desc = ca.getConstPool().getMethodrefType(mref);
398                 int num = Descriptor.numOfParameters(desc) + 1;
399                 if (num > 3)
400                     pos = iterator.insertGapAt(pos, num - 3, false).position;
401 
402                 iterator.writeByte(Opcode.POP, pos++);  // this
403                 iterator.writeByte(Opcode.NOP, pos);
404                 iterator.writeByte(Opcode.NOP, pos + 1);
405                 Descriptor.Iterator it = new Descriptor.Iterator(desc);
406                 while (true) {
407                     it.next();
408                     if (it.isParameter())
409                         iterator.writeByte(it.is2byte() ? Opcode.POP2 : Opcode.POP,
410                                            pos++);
411                     else
412                         break;
413                 }
414             }
415         }
416         catch (BadBytecode e) {
417             throw new CannotCompileException(e);
418         }
419     }
