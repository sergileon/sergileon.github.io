36  /**
37   * <code>stack_map</code> attribute.
38   *
39   * <p>This is an entry in the attributes table of a Code attribute.
40   * It was introduced by J2SE 6 for the verification by
41   * typechecking.
42   *
43   * @see StackMap
44   * @since 3.4
45   */
46  public class StackMapTable extends AttributeInfo {
47      /**
48       * The name of this attribute <code>"StackMapTable"</code>.
49       */
50      public static final String tag = "StackMapTable";
51  
52      /**
53       * Constructs a <code>stack_map</code> attribute.
54       */
55      StackMapTable(ConstPool cp, byte[] newInfo) {
56          super(cp, tag, newInfo);
57      }
58  
59      StackMapTable(ConstPool cp, int name_id, DataInputStream in)
60          throws IOException
61      {
62          super(cp, name_id, in);
63      }
64  
65      /**
66       * Makes a copy.
67       *
68       * @exception RuntimeCopyException  if a <code>BadBytecode</code>
69       *                          exception is thrown while copying,
70       *                          it is converted into
71       *                          <code>RuntimeCopyException</code>.
72       *
73       */
74      public AttributeInfo copy(ConstPool newCp, Map classnames)
75          throws RuntimeCopyException
76      {
77          try {
78              return new StackMapTable(newCp,
79                              new Copier(this.constPool, info, newCp, classnames).doit());
80          }
81          catch (BadBytecode e) {
82              throw new RuntimeCopyException("bad bytecode. fatal?"); 
83          }
84      }

91          /**

93           */
94          public RuntimeCopyException(String s) {
95              super(s);
96          }
97      }

148     /**
149      * A code walker for a StackMapTable attribute.
150      */
151     public static class Walker {
152         byte[] info;
153         int numOfEntries;
154 
155         /**
156          * Constructs a walker.
157          *
158          * @param smt       the StackMapTable that this walker
159          *                  walks around.
160          */
161         public Walker(StackMapTable smt) {
162             this(smt.get());
163         }
164 
165         /**
166          * Constructs a walker.
167          *
168          * @param data      the <code>info</code> field of the
169          *                  <code>attribute_info</code> structure.
170          *                  It can be obtained by <code>get()</code>
171          *                  in the <code>AttributeInfo</code> class.
172          */
173         public Walker(byte[] data) {
174             info = data;
175             numOfEntries = ByteArray.readU16bit(data, 0);
176         }
177 
178         /**
179          * Returns the number of the entries.
180          */
181         public final int size() { return numOfEntries; }
182 
183         /**
184          * Visits each entry of the stack map frames. 
185          */
186         public void parse() throws BadBytecode {
187             int n = numOfEntries;
188             int pos = 2;
189             for (int i = 0; i < n; i++)
190                 pos = stackMapFrames(pos, i);
191         }
192 
193         /**
194          * Invoked when the next entry of the stack map frames is visited.
195          *
196          * @param pos       the position of the frame in the <code>info</code>
197          *                  field of <code>attribute_info</code> structure.
198          * @param nth       the frame is the N-th
199          *                  (0, 1st, 2nd, 3rd, 4th, ...) entry. 
200          * @return          the position of the next frame.
201          */
202         int stackMapFrames(int pos, int nth) throws BadBytecode {
203             int type = info[pos] & 0xff;
204             if (type < 64) {
205                 sameFrame(pos, type);
206                 pos++;
207             }
208             else if (type < 128)
209                 pos = sameLocals(pos, type);
210             else if (type < 247)
211                 throw new BadBytecode("bad frame_type in StackMapTable");
212             else if (type == 247)   // SAME_LOCALS_1_STACK_ITEM_EXTENDED
213                 pos = sameLocals(pos, type);
214             else if (type < 251) {
215                 int offset = ByteArray.readU16bit(info, pos + 1);
216                 chopFrame(pos, offset, 251 - type);
217                 pos += 3;
218             }
219             else if (type == 251) { // SAME_FRAME_EXTENDED
220                 int offset = ByteArray.readU16bit(info, pos + 1);
221                 sameFrame(pos, offset);
222                 pos += 3;
223             }
224             else if (type < 255)
225                 pos = appendFrame(pos, type);
226             else    // FULL_FRAME
227                 pos = fullFrame(pos);
228 
229             return pos;
230         }
231 
232         /**
233          * Invoked if the visited frame is a <code>same_frame</code> or
234          * a <code>same_frame_extended</code>.
235          *
236          * @param pos       the position of this frame in the <code>info</code>
237          *                  field of <code>attribute_info</code> structure.
238          * @param offsetDelta
239          */
240         public void sameFrame(int pos, int offsetDelta) throws BadBytecode {}
241 
242         private int sameLocals(int pos, int type) throws BadBytecode {
243             int top = pos;
244             int offset;
245             if (type < 128)
246                 offset = type - 64;
247             else { // type == 247
248                 offset = ByteArray.readU16bit(info, pos + 1);
249                 pos += 2;
250             }
251 
252             int tag = info[pos + 1] & 0xff;
253             int data = 0;
254             if (tag == OBJECT || tag == UNINIT) {
255                 data = ByteArray.readU16bit(info, pos + 2);
256                 objectOrUninitialized(tag, data, pos + 2);
257                 pos += 2;
258             }
259 
260             sameLocals(top, offset, tag, data);
261             return pos + 2;
262         }
263 
264         /**
265          * Invoked if the visited frame is a <code>same_locals_1_stack_item_frame</code>
266          * or a <code>same_locals_1_stack_item_frame_extended</code>.
267          *
268          * @param pos               the position.
269          * @param offsetDelta
270          * @param stackTag          <code>stack[0].tag</code>.
271          * @param stackData         <code>stack[0].cpool_index</code>
272          *                          if the tag is <code>OBJECT</code>,
273          *                          or <code>stack[0].offset</code>
274          *                          if the tag is <code>UNINIT</code>.
275          */
276         public void sameLocals(int pos, int offsetDelta, int stackTag, int stackData)
277             throws BadBytecode {}
278 
279         /**
280          * Invoked if the visited frame is a <code>chop_frame</code>.
281          * 
282          * @param pos               the position.
283          * @param offsetDelta
284          * @param k                 the <code>k</code> last locals are absent. 
285          */
286         public void chopFrame(int pos, int offsetDelta, int k) throws BadBytecode {}
287 
288         private int appendFrame(int pos, int type) throws BadBytecode {
289             int k = type - 251;
290             int offset = ByteArray.readU16bit(info, pos + 1);
291             int[] tags = new int[k];
292             int[] data = new int[k];
293             int p = pos + 3;
294             for (int i = 0; i < k; i++) {
295                 int tag = info[p] & 0xff;
296                 tags[i] = tag;
297                 if (tag == OBJECT || tag == UNINIT) {
298                     data[i] = ByteArray.readU16bit(info, p + 1);
299                     objectOrUninitialized(tag, data[i], p + 1);
300                     p += 3;
301                 }
302                 else {
303                     data[i] = 0;
304                     p++;
305                 }
306             }
307 
308             appendFrame(pos, offset, tags, data);
309             return p;
310         }
311 
312         /**
313          * Invoked if the visited frame is a <code>append_frame</code>.
314          *
315          * @param pos           the position.
316          * @param offsetDelta
317          * @param tags          <code>locals[i].tag</code>.
318          * @param data          <code>locals[i].cpool_index</code>
319          *                      or <code>locals[i].offset</code>.
320          */
321         public void appendFrame(int pos, int offsetDelta, int[] tags, int[] data)
322             throws BadBytecode {} 
323 
324         private int fullFrame(int pos) throws BadBytecode {
325             int offset = ByteArray.readU16bit(info, pos + 1);
326             int numOfLocals = ByteArray.readU16bit(info, pos + 3);
327             int[] localsTags = new int[numOfLocals];
328             int[] localsData = new int[numOfLocals];
329             int p = verifyTypeInfo(pos + 5, numOfLocals, localsTags, localsData);
330             int numOfItems = ByteArray.readU16bit(info, p);
331             int[] itemsTags = new int[numOfItems];
332             int[] itemsData = new int[numOfItems];
333             p = verifyTypeInfo(p + 2, numOfItems, itemsTags, itemsData);
334             fullFrame(pos, offset, localsTags, localsData, itemsTags, itemsData);
335             return p;
336         }
337 
338         /**
339          * Invoked if the visited frame is <code>full_frame</code>.
340          *
341          * @param pos               the position.
342          * @param offsetDelta
343          * @param localTags         <code>locals[i].tag</code>
344          * @param localData         <code>locals[i].cpool_index</code>
345          *                          or <code>locals[i].offset</code>
346          * @param stackTags         <code>stack[i].tag</code>
347          * @param stackData         <code>stack[i].cpool_index</code>
348          *                          or <code>stack[i].offset</code>
349          */
350         public void fullFrame(int pos, int offsetDelta, int[] localTags, int[] localData,
351                               int[] stackTags, int[] stackData)
352             throws BadBytecode {}
353 
354         private int verifyTypeInfo(int pos, int n, int[] tags, int[] data) {
355             for (int i = 0; i < n; i++) {
356                 int tag = info[pos++] & 0xff;
357                 tags[i] = tag;
358                 if (tag == OBJECT || tag == UNINIT) {
359                     data[i] = ByteArray.readU16bit(info, pos);
360                     objectOrUninitialized(tag, data[i], pos);
361                     pos += 2;
362                 }
363             }
364 
365             return pos;
366         }
367 
368         /**
369          * Invoked if <code>Object_variable_info</code>
370          * or <code>Uninitialized_variable_info</code> is visited.
371          *
372          * @param tag		<code>OBJECT</code> or <code>UNINIT</code>.
373          * @param data		the value of <code>cpool_index</code> or <code>offset</code>.
374          * @param pos		the position of <code>cpool_index</code> or <code>offset</code>.
375          */
376         public void objectOrUninitialized(int tag, int data, int pos) {}
377     }

379     static class SimpleCopy extends Walker {
380         private Writer writer;
381 
382         public SimpleCopy(byte[] data) {
383             super(data);
384             writer = new Writer(data.length);
385         }
386 
387         public byte[] doit() throws BadBytecode {
388             parse();
389             return writer.toByteArray();
390         }
391 
392         public void sameFrame(int pos, int offsetDelta) {
393             writer.sameFrame(offsetDelta);
394         }
395 
396         public void sameLocals(int pos, int offsetDelta, int stackTag, int stackData) {
397             writer.sameLocals(offsetDelta, stackTag, copyData(stackTag, stackData));
398         }
399 
400         public void chopFrame(int pos, int offsetDelta, int k) {
401             writer.chopFrame(offsetDelta, k);
402         }
403 
404         public void appendFrame(int pos, int offsetDelta, int[] tags, int[] data) {
405             writer.appendFrame(offsetDelta, tags, copyData(tags, data));
406         }
407 
408         public void fullFrame(int pos, int offsetDelta, int[] localTags, int[] localData,
409                               int[] stackTags, int[] stackData) {
410             writer.fullFrame(offsetDelta, localTags, copyData(localTags, localData),
411                              stackTags, copyData(stackTags, stackData));
412         }
413 
414         protected int copyData(int tag, int data) {
415             return data;
416         }
417 
418         protected int[] copyData(int[] tags, int[] data) {
419             return data;
420         }
421     }

423     static class Copier extends SimpleCopy {
424         private ConstPool srcPool, destPool;
425         private Map classnames;
426 
427         public Copier(ConstPool src, byte[] data, ConstPool dest, Map names) {
428             super(data);
429             srcPool = src;
430             destPool = dest;
431             classnames = names;
432         }
433 
434         protected int copyData(int tag, int data) {
435             if (tag == OBJECT)
436                 return srcPool.copy(data, destPool, classnames); 
437             else
438                 return data;
439         }
440 
441         protected int[] copyData(int[] tags, int[] data) {
442             int[] newData = new int[data.length];
443             for (int i = 0; i < data.length; i++)
444                 if (tags[i] == OBJECT)
445                     newData[i] = srcPool.copy(data[i], destPool, classnames);
446                 else
447                     newData[i] = data[i];
448 
449             return newData;
450         }
451     }

453     /**
454      * Updates this stack map table when a new local variable is inserted
455      * for a new parameter.
456      *
457      * @param index          the index of the added local variable.
458      * @param tag            the type tag of that local variable. 
459      * @param classInfo      the index of the <code>CONSTANT_Class_info</code> structure
460      *                       in a constant pool table.  This should be zero unless the tag
461      *                       is <code>ITEM_Object</code>.
462      *
463      * @see scouter.javassist.CtBehavior#addParameter(scouter.javassist.CtClass)
464      * @see #typeTagOf(char)
465      * @see ConstPool
466      */
467     public void insertLocal(int index, int tag, int classInfo)
468         throws BadBytecode
469     {
470         byte[] data = new InsertLocal(this.get(), index, tag, classInfo).doit();
471         this.set(data);
472     }

474     /**
475      * Returns the tag of the type specified by the
476      * descriptor.  This method returns <code>INTEGER</code>
477      * unless the descriptor is either D (double), F (float),
478      * J (long), L (class type), or [ (array).
479      *
480      * @param descriptor        the type descriptor.
481      * @see Descriptor
482      */
483     public static int typeTagOf(char descriptor) {
484         switch (descriptor) {
485         case 'D' :
486             return DOUBLE;
487         case 'F' :
488             return FLOAT;
489         case 'J' :
490             return LONG;
491         case 'L' :
492         case '[' :
493             return OBJECT;
494         // case 'V' :
495         default :
496             return INTEGER;
497         }
498     }

500     /* This implementation assumes that a local variable initially
501      * holding a parameter value is never changed to be a different
502      * type.
503      * 
504      */
505     static class InsertLocal extends SimpleCopy {
506         private int varIndex;
507         private int varTag, varData;
508 
509         public InsertLocal(byte[] data, int varIndex, int varTag, int varData) {
510             super(data);
511             this.varIndex = varIndex;
512             this.varTag = varTag;
513             this.varData = varData;
514         }
515 
516         public void fullFrame(int pos, int offsetDelta, int[] localTags, int[] localData,
517                               int[] stackTags, int[] stackData) {
518             int len = localTags.length;
519             if (len < varIndex) {
520                 super.fullFrame(pos, offsetDelta, localTags, localData, stackTags, stackData);
521                 return;
522             }
523 
524             int typeSize = (varTag == LONG || varTag == DOUBLE) ? 2 : 1;
525             int[] localTags2 = new int[len + typeSize];
526             int[] localData2 = new int[len + typeSize];
527             int index = varIndex;
528             int j = 0;
529             for (int i = 0; i < len; i++) {
530                 if (j == index)
531                     j += typeSize;
532 
533                 localTags2[j] = localTags[i];
534                 localData2[j++] = localData[i];
535             }
536 
537             localTags2[index] = varTag;
538             localData2[index] = varData;
539             if (typeSize > 1) {
540                 localTags2[index + 1] = TOP;
541                 localData2[index + 1] = 0;
542             }
543 
544             super.fullFrame(pos, offsetDelta, localTags2, localData2, stackTags, stackData);
545         }
546     }

548     /**
549      * A writer of stack map tables.
550      */
551     public static class Writer {
552         ByteArrayOutputStream output;
553         int numOfEntries;
554 
555         /**
556          * Constructs a writer.
557          * @param size      the initial buffer size.
558          */
559         public Writer(int size) {
560             output = new ByteArrayOutputStream(size);
561             numOfEntries = 0;
562             output.write(0);        // u2 number_of_entries
563             output.write(0);
564         }
565 
566         /**
567          * Returns the stack map table written out.
568          */
569         public byte[] toByteArray() {
570             byte[] b = output.toByteArray();
571             ByteArray.write16bit(numOfEntries, b, 0);
572             return b;
573         }
574 
575         /**
576          * Constructs and a return a stack map table containing
577          * the written stack map entries.
578          *
579          * @param cp        the constant pool used to write
580          *                  the stack map entries.
581          */
582         public StackMapTable toStackMapTable(ConstPool cp) {
583             return new StackMapTable(cp, toByteArray());
584         }
585 
586         /**
587          * Writes a <code>same_frame</code> or a <code>same_frame_extended</code>.
588          */
589         public void sameFrame(int offsetDelta) {
590             numOfEntries++;
591             if (offsetDelta < 64)
592                 output.write(offsetDelta);
593             else {
594                 output.write(251);  // SAME_FRAME_EXTENDED
595                 write16(offsetDelta);
596             }
597         }
598 
599         /**
600          * Writes a <code>same_locals_1_stack_item</code>
601          * or a <code>same_locals_1_stack_item_extended</code>.
602          *
603          * @param tag           <code>stack[0].tag</code>.
604          * @param data          <code>stack[0].cpool_index</code>
605          *                      if the tag is <code>OBJECT</code>,
606          *                      or <code>stack[0].offset</code>
607          *                      if the tag is <code>UNINIT</code>.
608          *                      Otherwise, this parameter is not used.
609          */
610         public void sameLocals(int offsetDelta, int tag, int data) {
611             numOfEntries++;
612             if (offsetDelta < 64)
613                 output.write(offsetDelta + 64);
614             else {
615                 output.write(247);  // SAME_LOCALS_1_STACK_ITEM_EXTENDED
616                 write16(offsetDelta);
617             }
618 
619             writeTypeInfo(tag, data);
620         }
621 
622         /**
623          * Writes a <code>chop_frame</code>.
624          *
625          * @param k                 the number of absent locals. 1, 2, or 3.
626          */
627         public void chopFrame(int offsetDelta, int k) {
628             numOfEntries++;
629             output.write(251 - k);
630             write16(offsetDelta);
631         }
632 
633         /**
634          * Writes a <code>append_frame</code>.  The number of the appended
635          * locals is specified by the length of <code>tags</code>.
636          *
637          * @param tags           <code>locals[].tag</code>.
638          *                      The length of this array must be
639          *                      either 1, 2, or 3.
640          * @param data          <code>locals[].cpool_index</code>
641          *                      if the tag is <code>OBJECT</code>,
642          *                      or <code>locals[].offset</code>
643          *                      if the tag is <code>UNINIT</code>.
644          *                      Otherwise, this parameter is not used.
645          */
646         public void appendFrame(int offsetDelta, int[] tags, int[] data) {
647             numOfEntries++;
648             int k = tags.length;    // k is 1, 2, or 3
649             output.write(k + 251);
650             write16(offsetDelta);
651             for (int i = 0; i < k; i++)
652                 writeTypeInfo(tags[i], data[i]);
653         }
654 
655         /**
656          * Writes a <code>full_frame</code>.
657          * <code>number_of_locals</code> and <code>number_of_stack_items</code>
658          * are specified by the the length of <code>localTags</code> and
659          * <code>stackTags</code>.
660          *
661          * @param localTags     <code>locals[].tag</code>.
662          * @param localData     <code>locals[].cpool_index</code>
663          *                      if the tag is <code>OBJECT</code>,
664          *                      or <code>locals[].offset</code>
665          *                      if the tag is <code>UNINIT</code>.
666          *                      Otherwise, this parameter is not used.
667          * @param stackTags     <code>stack[].tag</code>.
668          * @param stackData     <code>stack[].cpool_index</code>
669          *                      if the tag is <code>OBJECT</code>,
670          *                      or <code>stack[].offset</code>
671          *                      if the tag is <code>UNINIT</code>.
672          *                      Otherwise, this parameter is not used.
673          */
674         public void fullFrame(int offsetDelta, int[] localTags, int[] localData,
675                               int[] stackTags, int[] stackData) {
676             numOfEntries++;
677             output.write(255);      // FULL_FRAME
678             write16(offsetDelta);
679             int n = localTags.length;
680             write16(n);
681             for (int i = 0; i < n; i++)
682                 writeTypeInfo(localTags[i], localData[i]);
683 
684             n = stackTags.length;
685             write16(n);
686             for (int i = 0; i < n; i++)
687                 writeTypeInfo(stackTags[i], stackData[i]);
688         }
689 
690         private void writeTypeInfo(int tag, int data) {
691             output.write(tag);
692             if (tag == OBJECT || tag == UNINIT)
693                 write16(data);
694         }
695 
696         private void write16(int value) {
697             output.write((value >>> 8) & 0xff);
698             output.write(value & 0xff);
699         }
700     }

836     static class Shifter extends Walker {
837         private StackMapTable stackMap;
838         int where, gap;
839         int position;
840         byte[] updatedInfo;
841         boolean exclusive;
842 
843         public Shifter(StackMapTable smt, int where, int gap, boolean exclusive) {
844             super(smt);
845             stackMap = smt;
846             this.where = where;
847             this.gap = gap;
848             this.position = 0;
849             this.updatedInfo = null;
850             this.exclusive = exclusive;
851         }
852 
853         public void doit() throws BadBytecode {
854             parse();
855             if (updatedInfo != null)
856                 stackMap.set(updatedInfo);
857         }
858 
859         public void sameFrame(int pos, int offsetDelta) {
860             update(pos, offsetDelta, 0, 251);
861         }
862 
863         public void sameLocals(int pos, int offsetDelta, int stackTag, int stackData) {
864             update(pos, offsetDelta, 64, 247);
865         }
866 
867         void update(int pos, int offsetDelta, int base, int entry) {
868             int oldPos = position;
869             position = oldPos + offsetDelta + (oldPos == 0 ? 0 : 1);
870             boolean match;
871             if (exclusive)
872                 match = oldPos < where  && where <= position;
873             else
874                 match = oldPos <= where  && where < position;
875 
876             if (match) {
877                 int newDelta = offsetDelta + gap;
878                 position += gap;
879                 if (newDelta < 64)
880                     info[pos] = (byte)(newDelta + base);
881                 else if (offsetDelta < 64) {
882                     byte[] newinfo = insertGap(info, pos, 2);
883                     newinfo[pos] = (byte)entry;
884                     ByteArray.write16bit(newDelta, newinfo, pos + 1);
885                     updatedInfo = newinfo;
886                 }
887                 else
888                     ByteArray.write16bit(newDelta, info, pos + 1);
889             }
890         }
891 
892         static byte[] insertGap(byte[] info, int where, int gap) {
893             int len = info.length;
894             byte[] newinfo = new byte[len + gap];
895             for (int i = 0; i < len; i++)
896                 newinfo[i + (i < where ? 0 : gap)] = info[i];
897 
898             return newinfo;
899         }
900 
901         public void chopFrame(int pos, int offsetDelta, int k) {
902             update(pos, offsetDelta);
903         }
904 
905         public void appendFrame(int pos, int offsetDelta, int[] tags, int[] data) {
906             update(pos, offsetDelta);
907         }
908 
909         public void fullFrame(int pos, int offsetDelta, int[] localTags, int[] localData,
910                               int[] stackTags, int[] stackData) {
911             update(pos, offsetDelta);
912         }
913 
914         void update(int pos, int offsetDelta) {
915             int oldPos = position;
916             position = oldPos + offsetDelta + (oldPos == 0 ? 0 : 1);
917             boolean match;
918             if (exclusive)
919                 match = oldPos < where  && where <= position;
920             else
921                 match = oldPos <= where  && where < position;
922 
923             if (match) {
924                 int newDelta = offsetDelta + gap;
925                 ByteArray.write16bit(newDelta, info, pos + 1);
926                 position += gap;
927             }
928         }
929     }

938     static class SwitchShifter extends Shifter {
939         SwitchShifter(StackMapTable smt, int where, int gap) {
940             super(smt, where, gap, false);
941         }
942 
943         void update(int pos, int offsetDelta, int base, int entry) {
944             int oldPos = position;
945             position = oldPos + offsetDelta + (oldPos == 0 ? 0 : 1);
946             int newDelta = offsetDelta;
947             if (where == position)
948                 newDelta = offsetDelta - gap;
949             else if (where == oldPos)
950                 newDelta = offsetDelta + gap;
951             else
952                 return;
953 
954             if (offsetDelta < 64)
955                 if (newDelta < 64)
956                     info[pos] = (byte)(newDelta + base);
957                 else {
958                     byte[] newinfo = insertGap(info, pos, 2);
959                     newinfo[pos] = (byte)entry;
960                     ByteArray.write16bit(newDelta, newinfo, pos + 1);
961                     updatedInfo = newinfo;
962                 }
963             else
964                 if (newDelta < 64) {
965                     byte[] newinfo = deleteGap(info, pos, 2);
966                     newinfo[pos] = (byte)(newDelta + base);
967                     updatedInfo = newinfo;
968                 }
969                 else
970                     ByteArray.write16bit(newDelta, info, pos + 1);
971         }
972 
973         static byte[] deleteGap(byte[] info, int where, int gap) {
974             where += gap;
975             int len = info.length;
976             byte[] newinfo = new byte[len - gap];
977             for (int i = 0; i < len; i++)
978                 newinfo[i - (i < where ? 0 : gap)] = info[i];
979 
980             return newinfo;
981         }
982 
983         void update(int pos, int offsetDelta) {
984             int oldPos = position;
985             position = oldPos + offsetDelta + (oldPos == 0 ? 0 : 1);
986             int newDelta = offsetDelta;
987             if (where == position)
988                 newDelta = offsetDelta - gap;
989             else if (where == oldPos)
990                 newDelta = offsetDelta + gap;
991             else
992                 return;
993 
994             ByteArray.write16bit(newDelta, info, pos + 1);
995         }
996     }
