76      /*
77       * Converts an array of tuples of exprType, arrayDim, and className
78       * into a String object.
79       */
80      protected static String argTypesToString(int[] types, int[] dims,
81                                               String[] cnames) {
82          StringBuffer sbuf = new StringBuffer();
83          sbuf.append('(');
84          int n = types.length;
85          if (n > 0) {
86              int i = 0;
87              while (true) {
88                  typeToString(sbuf, types[i], dims[i], cnames[i]);
89                  if (++i < n)
90                      sbuf.append(',');
91                  else
92                      break;
93              }
94          }
95  
96          sbuf.append(')');
97          return sbuf.toString();
98      }

100     /*
101      * Converts a tuple of exprType, arrayDim, and className
102      * into a String object.
103      */
104     protected static StringBuffer typeToString(StringBuffer sbuf,
105                                         int type, int dim, String cname) {
106         String s;
107         if (type == CLASS)
108             s = MemberResolver.jvmToJavaName(cname);
109         else if (type == NULL)
110             s = "Object";
111         else
112             try {
113                 s = MemberResolver.getTypeName(type);
114             }
115             catch (CompileError e) {
116                 s = "?";
117             }
118 
119         sbuf.append(s);
120         while (dim-- > 0)
121             sbuf.append("[]");
122 
123         return sbuf;
124     }

168     public void atNewExpr(NewExpr expr) throws CompileError {
169         if (expr.isArray())
170             atNewArrayExpr(expr);
171         else {
172             CtClass clazz = resolver.lookupClassByName(expr.getClassName());
173             String cname = clazz.getName();
174             ASTList args = expr.getArguments();
175             atMethodCallCore(clazz, MethodInfo.nameInit, args);
176             exprType = CLASS;
177             arrayDim = 0;
178             className = MemberResolver.javaToJvmName(cname);
179         }
180     }

182     public void atNewArrayExpr(NewExpr expr) throws CompileError {
183         int type = expr.getArrayType();
184         ASTList size = expr.getArraySize();
185         ASTList classname = expr.getClassName();
186         ASTree init = expr.getInitializer();
187         if (init != null)
188             init.accept(this);
189 
190         if (size.length() > 1)
191             atMultiNewArray(type, classname, size);
192         else {
193             ASTree sizeExpr = size.head();
194             if (sizeExpr != null)
195                 sizeExpr.accept(this);
196 
197             exprType = type;
198             arrayDim = 1;
199             if (type == CLASS)
200                 className = resolveClassName(classname);
201             else
202                 className = null;
203         }
204     }

206     public void atArrayInit(ArrayInit init) throws CompileError {
207         ASTList list = init;
208         while (list != null) {
209             ASTree h = list.head();
210             list = list.tail();
211             if (h != null)
212                 h.accept(this);
213         }
214     }

216     protected void atMultiNewArray(int type, ASTList classname, ASTList size)
217         throws CompileError
218     {
219         int count, dim;
220         dim = size.length();
221         for (count = 0; size != null; size = size.tail()) {
222             ASTree s = size.head();
223             if (s == null)
224                 break;          // int[][][] a = new int[3][4][];
225 
226             ++count;
227             s.accept(this);
228         }
229 
230         exprType = type;
231         arrayDim = dim;
232         if (type == CLASS)
233             className = resolveClassName(classname);
234         else
235             className = null;
236     }

238     public void atAssignExpr(AssignExpr expr) throws CompileError {
239         // =, %=, &=, *=, /=, +=, -=, ^=, |=, <<=, >>=, >>>=
240         int op = expr.getOperator();
241         ASTree left = expr.oprand1();
242         ASTree right = expr.oprand2();
243         if (left instanceof Variable)
244             atVariableAssign(expr, op, (Variable)left,
245                              ((Variable)left).getDeclarator(),
246                              right);
247         else {
248             if (left instanceof Expr) {
249                 Expr e = (Expr)left;
250                 if (e.getOperator() == ARRAY) {
251                     atArrayAssign(expr, op, (Expr)left, right);
252                     return;
253                 }
254             }
255 
256             atFieldAssign(expr, op, left, right);
257         }
258     }

260     /* op is either =, %=, &=, *=, /=, +=, -=, ^=, |=, <<=, >>=, or >>>=.
261      *
262      * expr and var can be null.
263      */
264     private void atVariableAssign(Expr expr, int op, Variable var,
265                                   Declarator d, ASTree right)
266         throws CompileError
267     {
268         int varType = d.getType();
269         int varArray = d.getArrayDim();
270         String varClass = d.getClassName();
271 
272         if (op != '=')
273             atVariable(var);
274 
275         right.accept(this);
276         exprType = varType;
277         arrayDim = varArray;
278         className = varClass;
279     }

288         right.accept(this);
289         exprType = aType;

308     public void atCondExpr(CondExpr expr) throws CompileError {
309         booleanExpr(expr.condExpr());
310         expr.thenExpr().accept(this);
311         int type1 = exprType;
312         int dim1 = arrayDim;
313         String cname1 = className;
314         expr.elseExpr().accept(this);
315 
316         if (dim1 == 0 && dim1 == arrayDim)
317             if (CodeGen.rightIsStrong(type1, exprType))
318                 expr.setThen(new CastExpr(exprType, 0, expr.thenExpr()));
319             else if (CodeGen.rightIsStrong(exprType, type1)) {
320                 expr.setElse(new CastExpr(type1, 0, expr.elseExpr()));
321                 exprType = type1;
322             }
323     }

325     /*
326      * If atBinExpr() substitutes a new expression for the original
327      * binary-operator expression, it changes the operator name to '+'
328      * (if the original is not '+') and sets the new expression to the
329      * left-hand-side expression and null to the right-hand-side expression. 
330      */
331     public void atBinExpr(BinExpr expr) throws CompileError {
332         int token = expr.getOperator();
333         int k = CodeGen.lookupBinOp(token);
334         if (k >= 0) {
335             /* arithmetic operators: +, -, *, /, %, |, ^, &, <<, >>, >>>
336              */
337             if (token == '+') {
338                 Expr e = atPlusExpr(expr);
339                 if (e != null) {
340                     /* String concatenation has been translated into
341                      * an expression using StringBuffer.
342                      */
343                     e = CallExpr.makeCall(Expr.make('.', e,
344                                             new Member("toString")), null);
345                     expr.setOprand1(e);
346                     expr.setOprand2(null);    // <---- look at this!
347                     className = jvmJavaLangString;
348                 }
349             }
350             else {
351                 ASTree left = expr.oprand1();
352                 ASTree right = expr.oprand2();
353                 left.accept(this);
354                 int type1 = exprType;
355                 right.accept(this);
356                 if (!isConstant(expr, token, left, right))
357                     computeBinExprType(expr, token, type1);
358             }
359         }
360         else {
361             /* equation: &&, ||, ==, !=, <=, >=, <, >
362             */
363             booleanExpr(expr);
364         }
365     }

367     /* EXPR must be a + expression.
368      * atPlusExpr() returns non-null if the given expression is string
369      * concatenation.  The returned value is "new StringBuffer().append..".
370      */
371     private Expr atPlusExpr(BinExpr expr) throws CompileError {
372         ASTree left = expr.oprand1();
373         ASTree right = expr.oprand2();
374         if (right == null) {
375             // this expression has been already type-checked.
376             // see atBinExpr() above.
377             left.accept(this);
378             return null;
379         }
380 
381         if (isPlusExpr(left)) {
382             Expr newExpr = atPlusExpr((BinExpr)left);
383             if (newExpr != null) {
384                 right.accept(this);
385                 exprType = CLASS;
386                 arrayDim = 0;
387                 className = "java/lang/StringBuffer";
388                 return makeAppendCall(newExpr, right);
389             }
390         }
391         else
392             left.accept(this);
393 
394         int type1 = exprType;
395         int dim1 = arrayDim;
396         String cname = className;
397         right.accept(this);
398 
399         if (isConstant(expr, '+', left, right))
400             return null;
401 
402         if ((type1 == CLASS && dim1 == 0 && jvmJavaLangString.equals(cname))
403             || (exprType == CLASS && arrayDim == 0
404                 && jvmJavaLangString.equals(className))) {
405             ASTList sbufClass = ASTList.make(new Symbol("java"),
406                             new Symbol("lang"), new Symbol("StringBuffer"));
407             ASTree e = new NewExpr(sbufClass, null);
408             exprType = CLASS;
409             arrayDim = 0;
410             className = "java/lang/StringBuffer";
411             return makeAppendCall(makeAppendCall(e, left), right);
412         }
413         else {
414             computeBinExprType(expr, '+', type1);
415             return null;
416         }
417     }

419     private boolean isConstant(BinExpr expr, int op, ASTree left,
420                                ASTree right) throws CompileError
421     {
422         left = stripPlusExpr(left);
423         right = stripPlusExpr(right);
424         ASTree newExpr = null;
425         if (left instanceof StringL && right instanceof StringL && op == '+')
426             newExpr = new StringL(((StringL)left).get()
427                                   + ((StringL)right).get());
428         else if (left instanceof IntConst)
429             newExpr = ((IntConst)left).compute(op, right);
430         else if (left instanceof DoubleConst)
431             newExpr = ((DoubleConst)left).compute(op, right);
432 
433         if (newExpr == null)
434             return false;       // not a constant expression
435         else {
436             expr.setOperator('+');
437             expr.setOprand1(newExpr);
438             expr.setOprand2(null);
439             newExpr.accept(this);   // for setting exprType, arrayDim, ...
440             return true;
441         }
442     }

444     /* CodeGen.atSwitchStmnt() also calls stripPlusExpr().
445      */
446     static ASTree stripPlusExpr(ASTree expr) {
447         if (expr instanceof BinExpr) {
448             BinExpr e = (BinExpr)expr;
449             if (e.getOperator() == '+' && e.oprand2() == null)
450                 return e.getLeft();
451         }
452         else if (expr instanceof Expr) {    // note: BinExpr extends Expr.
453             Expr e = (Expr)expr;
454             int op = e.getOperator();
455             if (op == MEMBER) {
456                 ASTree cexpr = getConstantFieldValue((Member)e.oprand2());
457                 if (cexpr != null)
458                     return cexpr;
459             }
460             else if (op == '+' && e.getRight() == null)
461                 return e.getLeft();
462         }
463         else if (expr instanceof Member) {
464             ASTree cexpr = getConstantFieldValue((Member)expr);
465             if (cexpr != null)
466                 return cexpr;
467         }
468 
469         return expr;
470     }

480     public static ASTree getConstantFieldValue(CtField f) {
481         if (f == null)
482             return null;
483 
484         Object value = f.getConstantValue();
485         if (value == null)
486             return null;
487 
488         if (value instanceof String)
489             return new StringL((String)value);
490         else if (value instanceof Double || value instanceof Float) {
491             int token = (value instanceof Double)
492                         ? DoubleConstant : FloatConstant;
493             return new DoubleConst(((Number)value).doubleValue(), token);
494         }
495         else if (value instanceof Number) {
496             int token = (value instanceof Long) ? LongConstant : IntConstant; 
497             return new IntConst(((Number)value).longValue(), token);
498         }
499         else if (value instanceof Boolean)
500             return new Keyword(((Boolean)value).booleanValue()
501                                ? TokenId.TRUE : TokenId.FALSE);
502         else
503             return null;
504     }

506     private static boolean isPlusExpr(ASTree expr) {
507         if (expr instanceof BinExpr) {
508             BinExpr bexpr = (BinExpr)expr;
509             int token = bexpr.getOperator();
510             return token == '+';
511         }
512 
513         return false;
514     }

521     private void computeBinExprType(BinExpr expr, int token, int type1)
522         throws CompileError
523     {
524         // arrayDim should be 0.
525         int type2 = exprType;
526         if (token == LSHIFT || token == RSHIFT || token == ARSHIFT)
527             exprType = type1;
528         else
529             insertCast(expr, type1, type2);
530 
531         if (CodeGen.isP_INT(exprType) && exprType != BOOLEAN)
532             exprType = INT;         // type1 may be BYTE, ...
533     }

535     private void booleanExpr(ASTree expr)
536         throws CompileError
537     {
538         int op = CodeGen.getCompOperator(expr);
539         if (op == EQ) {         // ==, !=, ...
540             BinExpr bexpr = (BinExpr)expr;
541             bexpr.oprand1().accept(this);
542             int type1 = exprType;
543             int dim1 = arrayDim;
544             bexpr.oprand2().accept(this);
545             if (dim1 == 0 && arrayDim == 0)
546                 insertCast(bexpr, type1, exprType);
547         }
548         else if (op == '!')
549             ((Expr)expr).oprand1().accept(this);
550         else if (op == ANDAND || op == OROR) {
551             BinExpr bexpr = (BinExpr)expr;
552             bexpr.oprand1().accept(this);
553             bexpr.oprand2().accept(this);
554         }
555         else                // others
556             expr.accept(this);
557 
558         exprType = BOOLEAN;
559         arrayDim = 0;
560     }

562     private void insertCast(BinExpr expr, int type1, int type2)
563         throws CompileError
564     {
565         if (CodeGen.rightIsStrong(type1, type2))
566             expr.setLeft(new CastExpr(type2, 0, expr.oprand1()));
567         else
568             exprType = type1;
569     }

585     public void atExpr(Expr expr) throws CompileError {
586         // array access, member access,
587         // (unary) +, (unary) -, ++, --, !, ~
588 
589         int token = expr.getOperator();
590         ASTree oprand = expr.oprand1();
591         if (token == '.') {
592             String member = ((Symbol)expr.oprand2()).get();
593             if (member.equals("length"))
594                 try {
595                     atArrayLength(expr);
596                 }
597                 catch (NoFieldException nfe) {
598                     // length might be a class or package name.
599                     atFieldRead(expr);
600                 }
601             else if (member.equals("class"))                
602                 atClassObject(expr);  // .class
603             else
604                 atFieldRead(expr);
605         }
606         else if (token == MEMBER) {     // field read
607             String member = ((Symbol)expr.oprand2()).get();
608             if (member.equals("class"))                
609                 atClassObject(expr);  // .class
610             else
611                 atFieldRead(expr);
612         }
613         else if (token == ARRAY)
614             atArrayRead(oprand, expr.oprand2());
615         else if (token == PLUSPLUS || token == MINUSMINUS)
616             atPlusPlus(token, oprand, expr);
617         else if (token == '!')
618             booleanExpr(expr);
619         else if (token == CALL)              // method call
620             fatal();
621         else {
622             oprand.accept(this);
623             if (!isConstant(expr, token, oprand))
624                 if (token == '-' || token == '~')
625                     if (CodeGen.isP_INT(exprType))
626                         exprType = INT;         // type may be BYTE, ...
627         }
628     }

630     private boolean isConstant(Expr expr, int op, ASTree oprand) {
631         oprand = stripPlusExpr(oprand);
632         if (oprand instanceof IntConst) {
633             IntConst c = (IntConst)oprand;
634             long v = c.get();
635             if (op == '-')
636                 v = -v;
637             else if (op == '~')
638                 v = ~v;
639             else
640                 return false;
641 
642             c.set(v);
643         }
644         else if (oprand instanceof DoubleConst) {
645             DoubleConst c = (DoubleConst)oprand;
646             if (op == '-')
647                 c.set(-c.get());
648             else
649                 return false;
650         }
651         else
652             return false;
653 
654         expr.setOperator('+');
655         return true;
656     }

658     public void atCallExpr(CallExpr expr) throws CompileError {
659         String mname = null;
660         CtClass targetClass = null;
661         ASTree method = expr.oprand1();
662         ASTList args = (ASTList)expr.oprand2();
663 
664         if (method instanceof Member) {
665             mname = ((Member)method).get();
666             targetClass = thisClass;
667         }
668         else if (method instanceof Keyword) {   // constructor
669             mname = MethodInfo.nameInit;        // <init>
670             if (((Keyword)method).get() == SUPER)
671                 targetClass = MemberResolver.getSuperclass(thisClass);
672             else
673                 targetClass = thisClass;
674         }
675         else if (method instanceof Expr) {
676             Expr e = (Expr)method;
677             mname = ((Symbol)e.oprand2()).get();
678             int op = e.getOperator();
679             if (op == MEMBER)                // static method
680                 targetClass
681                         = resolver.lookupClass(((Symbol)e.oprand1()).get(),
682                                                false);
683             else if (op == '.') {
684                 ASTree target = e.oprand1();
685                 String classFollowedByDotSuper = isDotSuper(target);
686                 if (classFollowedByDotSuper != null)
687                     targetClass = MemberResolver.getSuperInterface(thisClass,
688                                                         classFollowedByDotSuper);
689                 else {
690                     try {
691                         target.accept(this);
692                     }
693                     catch (NoFieldException nfe) {
694                         if (nfe.getExpr() != target)
695                             throw nfe;
696 
697                         // it should be a static method.
698                         exprType = CLASS;
699                         arrayDim = 0;
700                         className = nfe.getField(); // JVM-internal
701                         e.setOperator(MEMBER);
702                         e.setOprand1(new Symbol(MemberResolver.jvmToJavaName(
703                                                                 className)));
704                     }
705 
706                     if (arrayDim > 0)
707                         targetClass = resolver.lookupClass(javaLangObject, true);
708                     else if (exprType == CLASS /* && arrayDim == 0 */)
709                         targetClass = resolver.lookupClassByJvmName(className);
710                     else
711                         badMethod();
712                 }
713             }
714             else
715                 badMethod();
716         }
717         else
718             fatal();
719 
720         MemberResolver.Method minfo
721                 = atMethodCallCore(targetClass, mname, args);
722         expr.setMethod(minfo);
723     }

730      * Returns non-null if target is something like Foo.super
731      * for accessing the default method in an interface.
732      * Otherwise, null.
733      *
734      * @return the class name followed by {@code .super} or null.
735      */
736     static String isDotSuper(ASTree target) {
737         if (target instanceof Expr) {
738             Expr e = (Expr)target;
739             if (e.getOperator() == '.') {
740                 ASTree right = e.oprand2();
741                 if (right instanceof Keyword && ((Keyword)right).get() == SUPER)
742                     return ((Symbol)e.oprand1()).get();
743             }
744         }
745 
746         return null;
747     }

750      * @return  a pair of the class declaring the invoked method
751      *          and the MethodInfo of that method.  Never null.

753     public MemberResolver.Method atMethodCallCore(CtClass targetClass,
754                                                   String mname, ASTList args)
755         throws CompileError
756     {
757         int nargs = getMethodArgsLength(args);
758         int[] types = new int[nargs];
759         int[] dims = new int[nargs];
760         String[] cnames = new String[nargs];
761         atMethodArgs(args, types, dims, cnames);
762 
763         MemberResolver.Method found
764             = resolver.lookupMethod(targetClass, thisClass, thisMethod,
765                                     mname, types, dims, cnames);
766         if (found == null) {
767             String clazz = targetClass.getName();
768             String signature = argTypesToString(types, dims, cnames); 
769             String msg;
770             if (mname.equals(MethodInfo.nameInit))
771                 msg = "cannot find constructor " + clazz + signature;
772             else
773                 msg = mname + signature +  " not found in " + clazz;
774 
775             throw new CompileError(msg);
776         }
777 
778         String desc = found.info.getDescriptor();
779         setReturnType(desc);
780         return found;
781     }

787     public void atMethodArgs(ASTList args, int[] types, int[] dims,
788                              String[] cnames) throws CompileError {
789         int i = 0;
790         while (args != null) {
791             ASTree a = args.head();
792             a.accept(this);
793             types[i] = exprType;
794             dims[i] = arrayDim;

796             ++i;
797             args = args.tail();
798         }

801     void setReturnType(String desc) throws CompileError {
802         int i = desc.indexOf(')');
803         if (i < 0)
804             badMethod();
805 
806         char c = desc.charAt(++i);
807         int dim = 0;
808         while (c == '[') {
809             ++dim;
810             c = desc.charAt(++i);
811         }
812 
813         arrayDim = dim;
814         if (c == 'L') {
815             int j = desc.indexOf(';', i + 1);
816             if (j < 0)
817                 badMethod();
818 
819             exprType = CLASS;
820             className = desc.substring(i + 1, j);
821         }
822         else {
823             exprType = MemberResolver.descToType(c);
824             className = null;
825         }
826     }

832     private void atFieldRead(CtField f) throws CompileError {
833         FieldInfo finfo = f.getFieldInfo2();
834         String type = finfo.getDescriptor();
835 
836         int i = 0;
837         int dim = 0;

853     /* if EXPR is to access a static field, fieldAccess() translates EXPR
854      * into an expression using '#' (MEMBER).  For example, it translates
855      * java.lang.Integer.TYPE into java.lang.Integer#TYPE.  This translation
856      * speeds up type resolution by MemberCodeGen.
857      */
858     protected CtField fieldAccess(ASTree expr) throws CompileError {
859         if (expr instanceof Member) {
860             Member mem = (Member)expr;
861             String name = mem.get();
862             try {
863                 CtField f = thisClass.getField(name);
864                 if (Modifier.isStatic(f.getModifiers()))
865                     mem.setField(f);
866 
867                 return f;
868             }
869             catch (NotFoundException e) {
870                 // EXPR might be part of a static member access?
871                 throw new NoFieldException(name, expr);
872             }
873         }
874         else if (expr instanceof Expr) {
875             Expr e = (Expr)expr;
876             int op = e.getOperator();
877             if (op == MEMBER) {
878                 Member mem = (Member)e.oprand2();
879                 CtField f
880                     = resolver.lookupField(((Symbol)e.oprand1()).get(), mem);
881                 mem.setField(f);
882                 return f;
883             }
884             else if (op == '.') {
885                 try {
886                     e.oprand1().accept(this);
887                 }
888                 catch (NoFieldException nfe) {
889                     if (nfe.getExpr() != e.oprand1())
890                         throw nfe;
891 
892                     /* EXPR should be a static field.
893                      * If EXPR might be part of a qualified class name,
894                      * lookupFieldByJvmName2() throws NoFieldException.
895                      */
896                     return fieldAccess2(e, nfe.getField());
897                 }
898 
899                 CompileError err = null;
900                 try {
901                     if (exprType == CLASS && arrayDim == 0)
902                         return resolver.lookupFieldByJvmName(className,
903                                                     (Symbol)e.oprand2());
904                 }
905                 catch (CompileError ce) {
906                     err = ce;
907                 }
908 
909                 /* If a filed name is the same name as a package's,
910                  * a static member of a class in that package is not
911                  * visible.  For example,
912                  *
913                  * class Foo {
914                  *   int javassist;
915                  * }
916                  *
917                  * It is impossible to add the following method:
918                  *
919                  * String m() { return javassist.CtClass.intType.toString(); }
920                  *
921                  * because javassist is a field name.  However, this is
922                  * often inconvenient, this compiler allows it.  The following
923                  * code is for that.
924                  */
925                 ASTree oprnd1 = e.oprand1(); 
926                 if (oprnd1 instanceof Symbol)
927                     return fieldAccess2(e, ((Symbol)oprnd1).get());
928 
929                 if (err != null)
930                     throw err;
931             }
932         }
933 
934         throw new CompileError("bad filed access");
935     }

937     private CtField fieldAccess2(Expr e, String jvmClassName) throws CompileError {
938         Member fname = (Member)e.oprand2();
939         CtField f = resolver.lookupFieldByJvmName2(jvmClassName, fname, e);
940         e.setOperator(MEMBER);
941         e.setOprand1(new Symbol(MemberResolver.jvmToJavaName(jvmClassName)));
942         fname.setField(f);
943         return f;
944     }

974     private void atPlusPlus(int token, ASTree oprand, Expr expr)
975         throws CompileError
976     {
977         boolean isPost = oprand == null;        // ++i or i++?
978         if (isPost)
979             oprand = expr.oprand2();
980 
981         if (oprand instanceof Variable) {
982             Declarator d = ((Variable)oprand).getDeclarator();
983             exprType = d.getType();
984             arrayDim = d.getArrayDim();
985         }
986         else {
987             if (oprand instanceof Expr) {
988                 Expr e = (Expr)oprand;
989                 if (e.getOperator() == ARRAY) {
990                     atArrayRead(e.oprand1(), e.oprand2());
991                     // arrayDim should be 0.
992                     int t = exprType;
993                     if (t == INT || t == BYTE || t == CHAR || t == SHORT)
994                         exprType = INT;
995 
996                     return;
997                 }
998             }
999 
1000             atFieldPlusPlus(oprand);
1001         }
1002     }

1024     public void atKeyword(Keyword k) throws CompileError {
1025         arrayDim = 0;
1026         int token = k.get();
1027         switch (token) {
1028         case TRUE :
1029         case FALSE :
1030             exprType = BOOLEAN;
1031             break;
1032         case NULL :
1033             exprType = NULL;
1034             break;
1035         case THIS :
1036         case SUPER :
1037             exprType = CLASS;
1038             if (token == THIS)
1039                 className = getThisName();
1040             else
1041                 className = getSuperName();             
1042             break;
1043         default :
1044             fatal();
1045         }
1046     }
