109 	public void createPartControl(final Composite parent) {
110 		initializeColoring();
111 		SashForm sashForm = new SashForm(parent, SWT.VERTICAL);
112 		sashForm.SASH_WIDTH = 1;
113 		
114 		Composite composite = new Composite(sashForm, SWT.NONE);

116 		
117 		Composite upperComp = new Composite(composite, SWT.NONE);
118 		upperComp.setLayoutData(new GridData(SWT.FILL, SWT.CENTER, true, false));
119 		upperComp.setLayout(UIUtil.formLayout(3, 3));
120 		
121 		if (ServerManager.getInstance().getServer(serverId).isAllowAction(GroupPolicyConstants.ALLOW_KILLTRANSACTION)) {
122 			stopBtn = new Button(upperComp, SWT.PUSH);
123 			stopBtn.setText("&Stop");

125 			stopBtn.setLayoutData(UIUtil.formData(null, -1, null, -1, 100, -5, null, -1, 100));
126 			stopBtn.addSelectionListener(new SelectionAdapter() {

128 					if (MessageDialog.openConfirm(parent.getShell(), "Stop Thread", "This thread will be terminated. Continue?")) {
129 						controlThread("stop");
130 					}
131 				}
132 			});
133 			
134 			interruptBtn = new Button(upperComp, SWT.PUSH);
135 			interruptBtn.setText("&Interrupt");
136 			interruptBtn.setLayoutData(UIUtil.formData(null, -1, null, -1, stopBtn, -5, null, -1, 100));
137 			interruptBtn.addSelectionListener(new SelectionAdapter() {
138 				public void widgetSelected(SelectionEvent e) {
139 					if (MessageDialog.openConfirm(parent.getShell(), "Interrupt Thread", "This thread will be interrupted. Continue?")) {
140 						controlThread("interrupt");
141 					}
142 				}
143 			});
144 		}
145 		
146 		table = build(composite);
147 		table.addMouseListener(new MouseAdapter() {
148 			public void mouseDoubleClick(MouseEvent e) {
149 				TableItem[] items = table.getSelection();
150 				if (items == null || items.length < 1) {
151 					return;
152 				}
153 				String key = items[0].getText(0);
154 				String content = items[0].getText(1);
155 				new EditableMessageDialog().show(key, content);
156 			}
157 		});
158 		table.addKeyListener(new KeyAdapter() {
159 			public void keyPressed(KeyEvent e) {
160 				if (e.keyCode == SWT.F5) {
161 					reload();
162 				}
163 			}
164 		});
165 
166 		stacktrace = new StyledText(sashForm, SWT.MULTI | SWT.V_SCROLL | SWT.H_SCROLL | SWT.BORDER);
167 
168 		stacktrace.setText("");
169 		stacktrace.setFont(new Font(null, "verdana", 10, 0));
170 		stacktrace.setMargins(10, 10, 10, 10);
171 	
172 		listener = new CustomLineStyleListener(false, defaultHighlightings, false);
173 		stacktrace.addLineStyleListener(listener);
174 		
175 		IToolBarManager man = getViewSite().getActionBars().getToolBarManager();
176 		man.add(new Action("reload",ImageUtil.getImageDescriptor(Images.refresh)) {
177 			public void run() {
178 				reload();
179 			}
180 		});
181 	}

207 	protected void reload() {
208 		ExUtil.asyncRun(new Runnable() {
209 			public void run() {
210 				final MapPack mp = AgentDataProxy.getThreadDetail(objHash, threadid, serverId);
211 				setThreadDetailContens(mp);

216 	private void setThreadDetailContens (final MapPack pack) {
217 		ExUtil.exec(table, new Runnable() {
218 			public void run() {
219 				table.removeAll();
220 				stacktrace.setText("");
221 				int time = CastUtil.cint(pack.get("Service Elapsed"));
222 				boolean serviceThread = false;
223 				String[] names = scouter.util.SortUtil.sort_string(pack.keys(), pack.size());
224 				for (int i = 0, j = 0; i < names.length; i++) {
225 					String key = names[i];
226 					Value value = pack.get(key);
227 					if ("Stack Trace".equals(key)) {
228 						stacktrace.setText(CastUtil.cString(value));
229 						continue;
230 					}
231 					String text = null;
232 					TableItem ti = new TableItem(table, SWT.NONE, j++);
233 					if (value instanceof TextValue) {
234 						text = CastUtil.cString(value);
235 						ti.setText(0, key);
236 						ti.setText(1, text);
237 					} else {
238 						if (value instanceof DecimalValue) {
239 							text = FormatUtil.print(value, "#,##0");
240 						} else if (value instanceof DoubleValue || value instanceof FloatValue) {
241 							text = FormatUtil.print(value, "#,##0.0##");
242 						}
243 						ti.setText(new String[] { key, text });
244 					}
245 					if (key.startsWith("Service")) {
246 						if (time > 8000) {
247 							ti.setForeground(ColorUtil.getInstance().getColor(SWT.COLOR_RED));
248 						} else if (time > 3000) {
249 							ti.setForeground(ColorUtil.getInstance().getColor(SWT.COLOR_MAGENTA));
250 						} else {
251 							ti.setForeground(ColorUtil.getInstance().getColor(SWT.COLOR_BLUE));
252 						}
253 						serviceThread = true;
254 					}
255 				}
256 				if (stopBtn != null && interruptBtn != null) {
257 					stopBtn.setEnabled(serviceThread);
258 					interruptBtn.setEnabled(serviceThread);
259 				}
260 				sortTable();
261 			}
262 		});
263 	}
264 	
265 	private void controlThread(final String action) {
266 		ExUtil.asyncRun(new Runnable() {
267 			public void run() {
268 				TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
269 				MapPack p = null;
270 				try {
271 					MapPack param = new MapPack();
272 					param.put("objHash", objHash);
273 					param.put("id", threadid);
274 					param.put("action", action);
275 					p = (MapPack) tcp.getSingle(RequestCmd.OBJECT_THREAD_CONTROL, param);
276 				} catch (Exception e) {
277 					e.printStackTrace();
278 				} finally {
279 					TcpProxy.putTcpProxy(tcp);
280 				}
281 				if (p != null) {
282 					setThreadDetailContens(p);
283 				}
284 			}
285 		});
286 	}

310 	public void sortTable(){
311 		int col_count = table.getColumnCount();
312 		TableItem[] items = table.getItems();
313 		if (isNum) {
314 			new SortUtil(asc).sort_num(items, col_idx, col_count);
315 		} else {
316 			new SortUtil(asc).sort_str(items, col_idx, col_count);
317 		}
