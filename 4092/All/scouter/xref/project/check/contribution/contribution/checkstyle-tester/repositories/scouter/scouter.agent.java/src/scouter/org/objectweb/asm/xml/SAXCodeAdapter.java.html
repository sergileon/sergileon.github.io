46  /**
47   * A {@link MethodVisitor} that generates SAX 2.0 events from the visited
48   * method.
49   * 
50   * @see scouter.org.objectweb.asm.xml.SAXClassAdapter
51   * @see scouter.org.objectweb.asm.xml.Processor
52   * 
53   * @author Eugene Kuleshov
54   */
55  public final class SAXCodeAdapter extends MethodVisitor {
56  
57      static final String[] TYPES = { "top", "int", "float", "double", "long",
58              "null", "uninitializedThis" };
59  
60      SAXAdapter sa;
61  
62      int access;
63  
64      private final Map<Label, String> labelNames;
65  
66      /**
67       * Constructs a new {@link SAXCodeAdapter SAXCodeAdapter} object.
68       * 
69       * @param sa
70       *            content handler that will be used to send SAX 2.0 events.
71       */
72      public SAXCodeAdapter(final SAXAdapter sa, final int access) {
73          super(Opcodes.ASM5);
74          this.sa = sa;
75          this.access = access;
76          this.labelNames = new HashMap<Label, String>();
77      }
78  
79      @Override
80      public void visitParameter(String name, int access) {
81          AttributesImpl attrs = new AttributesImpl();
82          if (name != null) {
83              attrs.addAttribute("", "name", "name", "", name);
84          }
85          StringBuffer sb = new StringBuffer();
86          SAXClassAdapter.appendAccess(access, sb);
87          attrs.addAttribute("", "access", "access", "", sb.toString());
88          sa.addElement("parameter", attrs);
89      }
90  
91      @Override
92      public final void visitCode() {
93          if ((access & (Opcodes.ACC_ABSTRACT | Opcodes.ACC_INTERFACE | Opcodes.ACC_NATIVE)) == 0) {
94              sa.addStart("code", new AttributesImpl());
95          }
96      }
97  
98      @Override
99      public void visitFrame(final int type, final int nLocal,
100             final Object[] local, final int nStack, final Object[] stack) {
101         AttributesImpl attrs = new AttributesImpl();
102         switch (type) {
103         case Opcodes.F_NEW:
104         case Opcodes.F_FULL:
105             if (type == Opcodes.F_NEW) {
106                 attrs.addAttribute("", "type", "type", "", "NEW");
107             } else {
108                 attrs.addAttribute("", "type", "type", "", "FULL");
109             }
110             sa.addStart("frame", attrs);
111             appendFrameTypes(true, nLocal, local);
112             appendFrameTypes(false, nStack, stack);
113             break;
114         case Opcodes.F_APPEND:
115             attrs.addAttribute("", "type", "type", "", "APPEND");
116             sa.addStart("frame", attrs);
117             appendFrameTypes(true, nLocal, local);
118             break;
119         case Opcodes.F_CHOP:
120             attrs.addAttribute("", "type", "type", "", "CHOP");
121             attrs.addAttribute("", "count", "count", "",
122                     Integer.toString(nLocal));
123             sa.addStart("frame", attrs);
124             break;
125         case Opcodes.F_SAME:
126             attrs.addAttribute("", "type", "type", "", "SAME");
127             sa.addStart("frame", attrs);
128             break;
129         case Opcodes.F_SAME1:
130             attrs.addAttribute("", "type", "type", "", "SAME1");
131             sa.addStart("frame", attrs);
132             appendFrameTypes(false, 1, stack);
133             break;
134         }
135         sa.addEnd("frame");
136     }
137 
138     private void appendFrameTypes(final boolean local, final int n,
139             final Object[] types) {
140         for (int i = 0; i < n; ++i) {
141             Object type = types[i];
142             AttributesImpl attrs = new AttributesImpl();
143             if (type instanceof String) {
144                 attrs.addAttribute("", "type", "type", "", (String) type);
145             } else if (type instanceof Integer) {
146                 attrs.addAttribute("", "type", "type", "",
147                         TYPES[((Integer) type).intValue()]);
148             } else {
149                 attrs.addAttribute("", "type", "type", "", "uninitialized");
150                 attrs.addAttribute("", "label", "label", "",
151                         getLabel((Label) type));
152             }
153             sa.addElement(local ? "local" : "stack", attrs);
154         }
155     }

193     @Override
194     public final void visitMethodInsn(final int opcode, final String owner,
195             final String name, final String desc, final boolean itf) {
196         AttributesImpl attrs = new AttributesImpl();
197         attrs.addAttribute("", "owner", "owner", "", owner);
198         attrs.addAttribute("", "name", "name", "", name);
199         attrs.addAttribute("", "desc", "desc", "", desc);

204     @Override
205     public void visitInvokeDynamicInsn(String name, String desc, Handle bsm,
206             Object... bsmArgs) {
207         AttributesImpl attrs = new AttributesImpl();
208         attrs.addAttribute("", "name", "name", "", name);
209         attrs.addAttribute("", "desc", "desc", "", desc);
210         attrs.addAttribute("", "bsm", "bsm", "",
211                 SAXClassAdapter.encode(bsm.toString()));
212         sa.addStart("INVOKEDYNAMIC", attrs);
213         for (int i = 0; i < bsmArgs.length; i++) {
214             sa.addElement("bsmArg", getConstantAttribute(bsmArgs[i]));
215         }
216         sa.addEnd("INVOKEDYNAMIC");
217     }
218

220     public final void visitJumpInsn(final int opcode, final Label label) {
221         AttributesImpl attrs = new AttributesImpl();
222         attrs.addAttribute("", "label", "label", "", getLabel(label));
223         sa.addElement(Printer.OPCODES[opcode], attrs);
224     }

238     private static AttributesImpl getConstantAttribute(final Object cst) {
239         AttributesImpl attrs = new AttributesImpl();
240         attrs.addAttribute("", "cst", "cst", "",
241                 SAXClassAdapter.encode(cst.toString()));
242         attrs.addAttribute("", "desc", "desc", "",
243                 Type.getDescriptor(cst.getClass()));
244         return attrs;
245     }

250         attrs.addAttribute("", "var", "var", "", Integer.toString(var));
251         attrs.addAttribute("", "inc", "inc", "", Integer.toString(increment));
252         sa.addElement(Printer.OPCODES[Opcodes.IINC], attrs);

255     @Override
256     public final void visitTableSwitchInsn(final int min, final int max,
257             final Label dflt, final Label... labels) {
258         AttributesImpl attrs = new AttributesImpl();
259         attrs.addAttribute("", "min", "min", "", Integer.toString(min));
260         attrs.addAttribute("", "max", "max", "", Integer.toString(max));
261         attrs.addAttribute("", "dflt", "dflt", "", getLabel(dflt));
262         String o = Printer.OPCODES[Opcodes.TABLESWITCH];
263         sa.addStart(o, attrs);
264         for (int i = 0; i < labels.length; i++) {
265             AttributesImpl att2 = new AttributesImpl();
266             att2.addAttribute("", "name", "name", "", getLabel(labels[i]));
267             sa.addElement("label", att2);
268         }
269         sa.addEnd(o);
270     }
271 
272     @Override
273     public final void visitLookupSwitchInsn(final Label dflt, final int[] keys,
274             final Label[] labels) {
275         AttributesImpl att = new AttributesImpl();
276         att.addAttribute("", "dflt", "dflt", "", getLabel(dflt));
277         String o = Printer.OPCODES[Opcodes.LOOKUPSWITCH];
278         sa.addStart(o, att);
279         for (int i = 0; i < labels.length; i++) {
280             AttributesImpl att2 = new AttributesImpl();
281             att2.addAttribute("", "name", "name", "", getLabel(labels[i]));
282             att2.addAttribute("", "key", "key", "", Integer.toString(keys[i]));
283             sa.addElement("label", att2);
284         }
285         sa.addEnd(o);
286     }

296     @Override
297     public final void visitTryCatchBlock(final Label start, final Label end,
298             final Label handler, final String type) {
299         AttributesImpl attrs = new AttributesImpl();
300         attrs.addAttribute("", "start", "start", "", getLabel(start));
301         attrs.addAttribute("", "end", "end", "", getLabel(end));
302         attrs.addAttribute("", "handler", "handler", "", getLabel(handler));
303         if (type != null) {
304             attrs.addAttribute("", "type", "type", "", type);
305         }
306         sa.addElement("TryCatch", attrs);
307     }

321     @Override
322     public void visitLocalVariable(final String name, final String desc,
323             final String signature, final Label start, final Label end,
324             final int index) {
325         AttributesImpl attrs = new AttributesImpl();
326         attrs.addAttribute("", "name", "name", "", name);
327         attrs.addAttribute("", "desc", "desc", "", desc);
328         if (signature != null) {
329             attrs.addAttribute("", "signature", "signature", "",
330                     SAXClassAdapter.encode(signature));
331         }
332         attrs.addAttribute("", "start", "start", "", getLabel(start));
333         attrs.addAttribute("", "end", "end", "", getLabel(end));
334         attrs.addAttribute("", "var", "var", "", Integer.toString(index));
335         sa.addElement("LocalVar", attrs);
336     }

385 
386     @Override
387     public AnnotationVisitor visitLocalVariableAnnotation(int typeRef,
388             TypePath typePath, Label[] start, Label[] end, int[] index,
389             String desc, boolean visible) {
390         String[] s = new String[start.length];
391         String[] e = new String[end.length];
392         for (int i = 0; i < s.length; ++i) {
393             s[i] = getLabel(start[i]);
394         }
395         for (int i = 0; i < e.length; ++i) {
396             e[i] = getLabel(end[i]);
397         }
398         return new SAXAnnotationAdapter(sa, "localVariableAnnotation",
399                 visible ? 1 : -1, null, desc, typeRef, typePath, s, e, index);
400     }
401 
402     @Override
403     public void visitEnd() {
404         sa.addEnd("method");
405     }
406 
407     private final String getLabel(final Label label) {
408         String name = labelNames.get(label);
409         if (name == null) {
410             name = Integer.toString(labelNames.size());
411             labelNames.put(label, name);
412         }
413         return name;
414     }
