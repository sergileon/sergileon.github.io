39  /**
40   * A web server for running sample programs.
41   *
42   * <p>This enables a Java program to instrument class files loaded by
43   * web browsers for applets.  Since the (standard) security manager
44   * does not allow an applet to create and use a class loader,
45   * instrumenting class files must be done by this web server.
46   *
47   * <p><b>Note:</b> although this class is included in the Javassist API,
48   * it is provided as a sample implementation of the web server using
49   * Javassist.  Especially, there might be security flaws in this server.
50   * Please use this with YOUR OWN RISK.
51   */
52  public class Webserver {
53      private ServerSocket socket;
54      private ClassPool classPool;
55      protected Translator translator;
56  
57      private final static byte[] endofline = { 0x0d, 0x0a };
58  
59      private final static int typeHtml = 1;
60      private final static int typeClass = 2;
61      private final static int typeGif = 3;
62      private final static int typeJpeg = 4;
63      private final static int typeText = 5;
64  
65      /**
66       * If this field is not null, the class files taken from
67       * <code>ClassPool</code> are written out under the directory
68       * specified by this field.  The directory name must not end
69       * with a directory separator.
70       */
71      public String debugDir = null;
72  
73      /**
74       * The top directory of html (and .gif, .class, ...) files.
75       * It must end with the directory separator such as "/".
76       * (For portability, "/" should be used as the directory separator.
77       * Javassist automatically translates "/" into a platform-dependent
78       * character.)
79       * If this field is null, the top directory is the current one where
80       * the JVM is running.
81       *
82       * <p>If the given URL indicates a class file and the class file
83       * is not found under the directory specified by this variable,
84       * then <code>Class.getResourceAsStream()</code> is called
85       * for searching the Java class paths.
86       */
87      public String htmlfileBase = null;
88  
89      /**
90       * Starts a web server.
91       * The port number is specified by the first argument.
92       */
93      public static void main(String[] args) throws IOException {
94          if (args.length == 1) {
95              Webserver web = new Webserver(args[0]);
96              web.run();
97          }
98          else
99              System.err.println(
100                         "Usage: java javassist.tools.web.Webserver <port number>");
101     }

189     /**
190      * Begins the HTTP service.
191      */
192     public void run() {
193         System.err.println("ready to service...");
194         for (;;)
195             try {
196                 ServiceThread th = new ServiceThread(this, socket.accept());
197                 th.start();
198             }
199             catch (IOException e) {
200                 logging(e.toString());
201             }
202     }

204     final void process(Socket clnt) throws IOException {
205         InputStream in = new BufferedInputStream(clnt.getInputStream());
206         String cmd = readLine(in);
207         logging(clnt.getInetAddress().getHostName(),
208                 new Date().toString(), cmd);
209         while (skipLine(in) > 0){
210         }
211 
212         OutputStream out = new BufferedOutputStream(clnt.getOutputStream());
213         try {
214             doReply(in, out, cmd);
215         }
216         catch (BadHttpRequest e) {
217             replyError(out, e);
218         }
219 
220         out.flush();
221         in.close();
222         out.close();
223         clnt.close();
224     }
225 
226     private String readLine(InputStream in) throws IOException {
227         StringBuffer buf = new StringBuffer();
228         int c;
229         while ((c = in.read()) >= 0 && c != 0x0d)
230             buf.append((char)c);
231 
232         in.read();      /* skip 0x0a (LF) */
233         return buf.toString();
234     }
235 
236     private int skipLine(InputStream in) throws IOException {
237         int c;
238         int len = 0;
239         while ((c = in.read()) >= 0 && c != 0x0d)
240             ++len;
241 
242         in.read();      /* skip 0x0a (LF) */
243         return len;
244     }
245 
246     /**
247      * Proceses a HTTP request from a client.
248      *
249      * @param out       the output stream to a client
250      * @param cmd       the command received from a client
251      */
252     public void doReply(InputStream in, OutputStream out, String cmd)
253         throws IOException, BadHttpRequest
254     {
255         int len;
256         int fileType;
257         String filename, urlName;
258 
259         if (cmd.startsWith("GET /"))
260             filename = urlName = cmd.substring(5, cmd.indexOf(' ', 5));
261         else
262             throw new BadHttpRequest();
263 
264         if (filename.endsWith(".class"))
265             fileType = typeClass;
266         else if (filename.endsWith(".html") || filename.endsWith(".htm"))
267             fileType = typeHtml;
268         else if (filename.endsWith(".gif"))
269             fileType = typeGif;
270         else if (filename.endsWith(".jpg"))
271             fileType = typeJpeg;
272         else
273             fileType = typeText;        // or textUnknown
274 
275         len = filename.length();
276         if (fileType == typeClass
277             && letUsersSendClassfile(out, filename, len))
278             return;
279 
280         checkFilename(filename, len);
281         if (htmlfileBase != null)
282             filename = htmlfileBase + filename;
283 
284         if (File.separatorChar != '/')
285             filename = filename.replace('/', File.separatorChar);
286 
287         File file = new File(filename);
288         if (file.canRead()) {
289             sendHeader(out, file.length(), fileType);
290             FileInputStream fin = new FileInputStream(file);
291             byte[] filebuffer = new byte[4096];
292             for (;;) {
293                 len = fin.read(filebuffer);
294                 if (len <= 0)
295                     break;
296                 else
297                     out.write(filebuffer, 0, len);
298             }
299 
300             fin.close();
301             return;
302         }
303 
304         // If the file is not found under the html-file directory,
305         // then Class.getResourceAsStream() is tried.
306 
307         if (fileType == typeClass) {
308             InputStream fin
309                 = getClass().getResourceAsStream("/" + urlName);
310             if (fin != null) {
311                 ByteArrayOutputStream barray = new ByteArrayOutputStream();
312                 byte[] filebuffer = new byte[4096];
313                 for (;;) {
314                     len = fin.read(filebuffer);
315                     if (len <= 0)
316                         break;
317                     else
318                         barray.write(filebuffer, 0, len);
319                 }
320 
321                 byte[] classfile = barray.toByteArray();
322                 sendHeader(out, classfile.length, typeClass);
323                 out.write(classfile);
324                 fin.close();
325                 return;
326             }
327         }
328 
329         throw new BadHttpRequest();
330     }
331 
332     private void checkFilename(String filename, int len)
333         throws BadHttpRequest
334     {
335         for (int i = 0; i < len; ++i) {
336             char c = filename.charAt(i);
337             if (!Character.isJavaIdentifierPart(c) && c != '.' && c != '/')
338                 throw new BadHttpRequest();
339         }
340 
341         if (filename.indexOf("..") >= 0)
342             throw new BadHttpRequest();
343     }
344 
345     private boolean letUsersSendClassfile(OutputStream out,
346                                           String filename, int length)
347         throws IOException, BadHttpRequest
348     {
349         if (classPool == null)
350             return false;
351 
352         byte[] classfile;
353         String classname
354             = filename.substring(0, length - 6).replace('/', '.');
355         try {
356             if (translator != null)
357                 translator.onLoad(classPool, classname);
358 
359             CtClass c = classPool.get(classname);
360             classfile = c.toBytecode();
361             if (debugDir != null)
362                 c.writeFile(debugDir);
363         }
364         catch (Exception e) {
365             throw new BadHttpRequest(e);
366         }
367 
368         sendHeader(out, classfile.length, typeClass);
369         out.write(classfile);
370         return true;
371     }
372 
373     private void sendHeader(OutputStream out, long dataLength, int filetype)
374         throws IOException
375     {
376         out.write("HTTP/1.0 200 OK".getBytes());
377         out.write(endofline);
378         out.write("Content-Length: ".getBytes());
379         out.write(Long.toString(dataLength).getBytes());
380         out.write(endofline);
381         if (filetype == typeClass)
382             out.write("Content-Type: application/octet-stream".getBytes());
383         else if (filetype == typeHtml)
384             out.write("Content-Type: text/html".getBytes());
385         else if (filetype == typeGif)
386             out.write("Content-Type: image/gif".getBytes());
387         else if (filetype == typeJpeg)
388             out.write("Content-Type: image/jpg".getBytes());
389         else if (filetype == typeText)
390             out.write("Content-Type: text/plain".getBytes());
391 
392         out.write(endofline);
393         out.write(endofline);
394     }
395 
396     private void replyError(OutputStream out, BadHttpRequest e)
397         throws IOException
398     {
399         logging2("bad request: " + e.toString());
400         out.write("HTTP/1.0 400 Bad Request".getBytes());
401         out.write(endofline);
402         out.write(endofline);
403         out.write("<H1>Bad Request</H1>".getBytes());
404     }
405 }

407 class ServiceThread extends Thread {
408     Webserver web;
409     Socket sock;
410 
411     public ServiceThread(Webserver w, Socket s) {
412         web = w;
413         sock = s;
414     }
415 
416     public void run() {
417         try {
418             web.process(sock);
419         }
420         catch (IOException e) {
421         }
422     }
