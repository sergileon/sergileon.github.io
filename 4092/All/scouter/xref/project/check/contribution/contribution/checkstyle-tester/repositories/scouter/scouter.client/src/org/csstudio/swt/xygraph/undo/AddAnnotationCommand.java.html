8   package org.csstudio.swt.xygraph.undo;
9   
10  import org.csstudio.swt.xygraph.Messages;
11  import org.csstudio.swt.xygraph.figures.Annotation;
12  import org.csstudio.swt.xygraph.figures.XYGraph;
13  
14  /**The undoable command to add an annotation.
15   * @author Xihui Chen
16   *
17   */
18  public class AddAnnotationCommand implements IUndoableCommand {
19  	
20  	private XYGraph xyGraph;
21  	private Annotation annotation;
22  	
23  	public AddAnnotationCommand(XYGraph xyGraph, Annotation annotation) {
24  		this.xyGraph = xyGraph;
25  		this.annotation = annotation;
26  	}
27  
28  	public void redo() {
29  		xyGraph.addAnnotation(annotation);
30  	}
31  
32  	public void undo() {
33  		xyGraph.removeAnnotation(annotation);
34  	}
35  	
36  	@Override
37  	public String toString() {
38  		return Messages.Annotation_Add;
39  	}
40  
41  }
