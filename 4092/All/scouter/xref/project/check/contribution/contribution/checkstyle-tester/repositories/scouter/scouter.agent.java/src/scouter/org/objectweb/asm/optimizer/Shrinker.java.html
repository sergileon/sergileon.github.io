53  /**
54   * A class file shrinker utility.
55   * 
56   * @author Eric Bruneton
57   * @author Eugene Kuleshov
58   */
59  public class Shrinker {
60  
61      static final HashMap<String, String> MAPPING = new HashMap<String, String>();
62  
63      public static void main(final String[] args) throws IOException {
64          Properties properties = new Properties();
65          int n = args.length - 1;
66          for (int i = 0; i < n - 1; ++i) {
67              properties.load(new FileInputStream(args[i]));
68          }
69  
70          for (Map.Entry<Object, Object> entry : properties.entrySet()) {
71              MAPPING.put((String) entry.getKey(), (String) entry.getValue());
72          }
73  
74          final Set<String> unused = new HashSet<String>(MAPPING.keySet());
75  
76          File f = new File(args[n - 1]);
77          File d = new File(args[n]);
78  
79          optimize(f, d, new SimpleRemapper(MAPPING) {
80              @Override
81              public String map(String key) {
82                  String s = super.map(key);
83                  if (s != null) {
84                      unused.remove(key);
85                  }
86                  return s;
87              }
88          });
89  
90          Iterator<String> i = unused.iterator();
91          while (i.hasNext()) {
92              String s = i.next();
93              if (!s.endsWith("/remove")) {
94                  System.out.println("INFO: unused mapping " + s);
95              }
96          }
97      }
98  
99      static void optimize(final File f, final File d, final Remapper remapper)
100             throws IOException {
101         if (f.isDirectory()) {
102             File[] files = f.listFiles();
103             for (int i = 0; i < files.length; ++i) {
104                 optimize(files[i], d, remapper);
105             }
106         } else if (f.getName().endsWith(".class")) {
107             ConstantPool cp = new ConstantPool();
108             ClassReader cr = new ClassReader(new FileInputStream(f));
109             ClassWriter cw = new ClassWriter(0);
110             ClassConstantsCollector ccc = new ClassConstantsCollector(cw, cp);
111             ClassOptimizer co = new ClassOptimizer(ccc, remapper);
112             cr.accept(co, ClassReader.SKIP_DEBUG);
113 
114             Set<Constant> constants = new TreeSet<Constant>(
115                     new ConstantComparator());
116             constants.addAll(cp.values());
117 
118             cr = new ClassReader(cw.toByteArray());
119             cw = new ClassWriter(0);
120             Iterator<Constant> i = constants.iterator();
121             while (i.hasNext()) {
122                 Constant c = i.next();
123                 c.write(cw);
124             }
125             cr.accept(cw, ClassReader.SKIP_DEBUG);
126 
127             if (MAPPING.get(cr.getClassName() + "/remove") != null) {
128                 return;
129             }
130             String n = remapper.mapType(cr.getClassName());
131             File g = new File(d, n + ".class");
132             if (!g.exists() || g.lastModified() < f.lastModified()) {
133                 if (!g.getParentFile().exists() && !g.getParentFile().mkdirs()) {
134                     throw new IOException("Cannot create directory "
135                             + g.getParentFile());
136                 }
137                 OutputStream os = new FileOutputStream(g);
138                 try {
139                     os.write(cw.toByteArray());
140                 } finally {
141                     os.close();
142                 }
143             }
144         }
145     }
146 
147     static class ConstantComparator implements Comparator<Constant> {
148 
149         public int compare(final Constant c1, final Constant c2) {
150             int d = getSort(c1) - getSort(c2);
151             if (d == 0) {
152                 switch (c1.type) {
153                 case 'I':
154                     return new Integer(c1.intVal).compareTo(new Integer(
155                             c2.intVal));
156                 case 'J':
157                     return new Long(c1.longVal).compareTo(new Long(c2.longVal));
158                 case 'F':
159                     return new Float(c1.floatVal).compareTo(new Float(
160                             c2.floatVal));
161                 case 'D':
162                     return new Double(c1.doubleVal).compareTo(new Double(
163                             c2.doubleVal));
164                 case 's':
165                 case 'S':
166                 case 'C':
167                 case 't':
168                     return c1.strVal1.compareTo(c2.strVal1);
169                 case 'T':
170                     d = c1.strVal1.compareTo(c2.strVal1);
171                     if (d == 0) {
172                         d = c1.strVal2.compareTo(c2.strVal2);
173                     }
174                     break;
175                 case 'y':
176                     d = c1.strVal1.compareTo(c2.strVal1);
177                     if (d == 0) {
178                         d = c1.strVal2.compareTo(c2.strVal2);
179                         if (d == 0) {
180                             Handle bsm1 = (Handle) c1.objVal3;
181                             Handle bsm2 = (Handle) c2.objVal3;
182                             d = compareHandle(bsm1, bsm2);
183                             if (d == 0) {
184                                 d = compareObjects(c1.objVals, c2.objVals);
185                             }
186                         }
187                     }
188                     break;
189 
190                 default:
191                     d = c1.strVal1.compareTo(c2.strVal1);
192                     if (d == 0) {
193                         d = c1.strVal2.compareTo(c2.strVal2);
194                         if (d == 0) {
195                             d = ((String) c1.objVal3)
196                                     .compareTo((String) c2.objVal3);
197                         }
198                     }
199                 }
200             }
201             return d;
202         }
203 
204         private static int compareHandle(Handle h1, Handle h2) {
205             int d = h1.getTag() - h2.getTag();
206             if (d == 0) {
207                 d = h1.getOwner().compareTo(h2.getOwner());
208                 if (d == 0) {
209                     d = h1.getName().compareTo(h2.getName());
210                     if (d == 0) {
211                         d = h1.getDesc().compareTo(h2.getDesc());
212                     }
213                 }
214             }
215             return d;
216         }
217 
218         private static int compareType(Type mtype1, Type mtype2) {
219             return mtype1.getDescriptor().compareTo(mtype2.getDescriptor());
220         }
221 
222         private static int compareObjects(Object[] objVals1, Object[] objVals2) {
223             int length = objVals1.length;
224             int d = length - objVals2.length;
225             if (d == 0) {
226                 for (int i = 0; i < length; i++) {
227                     Object objVal1 = objVals1[i];
228                     Object objVal2 = objVals2[i];
229                     d = objVal1.getClass().getName()
230                             .compareTo(objVal2.getClass().getName());
231                     if (d == 0) {
232                         if (objVal1 instanceof Type) {
233                             d = compareType((Type) objVal1, (Type) objVal2);
234                         } else if (objVal1 instanceof Handle) {
235                             d = compareHandle((Handle) objVal1,
236                                     (Handle) objVal2);
237                         } else {
238                             d = ((Comparable) objVal1).compareTo(objVal2);
239                         }
240                     }
241 
242                     if (d != 0) {
243                         return d;
244                     }
245                 }
246             }
247             return 0;
248         }
249 
250         private static int getSort(final Constant c) {
251             switch (c.type) {
252             case 'I':
253                 return 0;
254             case 'J':
255                 return 1;
256             case 'F':
257                 return 2;
258             case 'D':
259                 return 3;
260             case 's':
261                 return 4;
262             case 'S':
263                 return 5;
264             case 'C':
265                 return 6;
266             case 'T':
267                 return 7;
268             case 'G':
269                 return 8;
270             case 'M':
271                 return 9;
272             case 'N':
273                 return 10;
274             case 'y':
275                 return 11;
276             case 't':
277                 return 12;
278             default:
279                 return 100 + c.type - 'h';
280             }
281         }
282     }
