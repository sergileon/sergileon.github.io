285     public void atMethodDecl(MethodDecl method) throws CompileError {
286         ASTList mods = method.getModifiers();
287         setMaxLocals(1);
288         while (mods != null) {
289             Keyword k = (Keyword)mods.head();
290             mods = mods.tail();
291             if (k.get() == STATIC) {
292                 setMaxLocals(0);
293                 inStaticMethod = true;
294             }
295         }
296             
297         ASTList params = method.getParams();
298         while (params != null) {
299             atDeclarator((Declarator)params.head());
300             params = params.tail();
301         }
302 
303         Stmnt s = method.getBody();
304         atMethodBody(s, method.isConstructor(),
305                      method.getReturn().getType() == VOID);
306     }

308     /**
309      * @param isCons	true if super() must be called.
310      *			false if the method is a class initializer.
311      */
312     public void atMethodBody(Stmnt s, boolean isCons, boolean isVoid)
313         throws CompileError
314     {
315         if (s == null)
316             return;
317 
318         if (isCons && needsSuperCall(s))
319             insertDefaultSuperCall();
320 
321         hasReturned = false;
322         s.accept(this);
323         if (!hasReturned)
324             if (isVoid) {
325                 bytecode.addOpcode(Opcode.RETURN);
326                 hasReturned = true;
327             }
328             else
329                 throw new CompileError("no return statement");
330     }

332     private boolean needsSuperCall(Stmnt body) throws CompileError {
333         if (body.getOperator() == BLOCK)
334             body = (Stmnt)body.head();
335 
336         if (body != null && body.getOperator() == EXPR) {
337             ASTree expr = body.head();
338             if (expr != null && expr instanceof Expr
339                 && ((Expr)expr).getOperator() == CALL) {
340                 ASTree target = ((Expr)expr).head();
341                 if (target instanceof Keyword) {
342                     int token = ((Keyword)target).get();
343                     return token != THIS && token != SUPER;
344                 }
345             }
346         }
347 
348         return true;
349     }

353     public void atStmnt(Stmnt st) throws CompileError {
354         if (st == null)
355             return;     // empty
356 
357         int op = st.getOperator();
358         if (op == EXPR) {
359             ASTree expr = st.getLeft();
360             doTypeCheck(expr);
361             if (expr instanceof AssignExpr)
362                 atAssignExpr((AssignExpr)expr, false);
363             else if (isPlusPlusExpr(expr)) {
364                 Expr e = (Expr)expr;
365                 atPlusPlus(e.getOperator(), e.oprand1(), e, false);
366             }
367             else {
368                 expr.accept(this);
369                 if (is2word(exprType, arrayDim))
370                     bytecode.addOpcode(POP2);
371                 else if (exprType != VOID)
372                     bytecode.addOpcode(POP);
373             }
374         }
375         else if (op == DECL || op == BLOCK) {
376             ASTList list = st;
377             while (list != null) {
378                 ASTree h = list.head();
379                 list = list.tail();
380                 if (h != null)
381                     h.accept(this);
382             }
383         }
384         else if (op == IF)
385             atIfStmnt(st);
386         else if (op == WHILE || op == DO)
387             atWhileStmnt(st, op == WHILE);
388         else if (op == FOR)
389             atForStmnt(st);
390         else if (op == BREAK || op == CONTINUE)
391             atBreakStmnt(st, op == BREAK);
392         else if (op == TokenId.RETURN)
393             atReturnStmnt(st);
394         else if (op == THROW)
395             atThrowStmnt(st);
396         else if (op == TRY)
397             atTryStmnt(st);
398         else if (op == SWITCH)
399             atSwitchStmnt(st);
400         else if (op == SYNCHRONIZED)
401             atSyncStmnt(st);
402         else {
403             // LABEL, SWITCH label stament might be null?.
404             hasReturned = false;
405             throw new CompileError(
406                 "sorry, not supported statement: TokenId " + op);
407         }
408     }

410     private void atIfStmnt(Stmnt st) throws CompileError {
411         ASTree expr = st.head();
412         Stmnt thenp = (Stmnt)st.tail().head();
413         Stmnt elsep = (Stmnt)st.tail().tail().head();
414         if (compileBooleanExpr(false, expr)) {
415             hasReturned = false;
416             if (elsep != null)
417                 elsep.accept(this);
418 
419             return;
420         }
421 
422         int pc = bytecode.currentPc();
423         int pc2 = 0;
424         bytecode.addIndex(0);   // correct later
425 
426         hasReturned = false;
427         if (thenp != null)
428             thenp.accept(this);
429 
430         boolean thenHasReturned = hasReturned;
431         hasReturned = false;
432 
433         if (elsep != null && !thenHasReturned) {
434             bytecode.addOpcode(Opcode.GOTO);
435             pc2 = bytecode.currentPc();
436             bytecode.addIndex(0);
437         }
438 
439         bytecode.write16bit(pc, bytecode.currentPc() - pc + 1);
440         if (elsep != null) {
441             elsep.accept(this);
442             if (!thenHasReturned)
443                 bytecode.write16bit(pc2, bytecode.currentPc() - pc2 + 1);
444 
445             hasReturned = thenHasReturned && hasReturned;
446         }
447     }

449     private void atWhileStmnt(Stmnt st, boolean notDo) throws CompileError {
450         ArrayList prevBreakList = breakList;
451         ArrayList prevContList = continueList;
452         breakList = new ArrayList();
453         continueList = new ArrayList();
454 
455         ASTree expr = st.head();
456         Stmnt body = (Stmnt)st.tail();
457 
458         int pc = 0;
459         if (notDo) {
460             bytecode.addOpcode(Opcode.GOTO);
461             pc = bytecode.currentPc();
462             bytecode.addIndex(0);
463         }
464 
465         int pc2 = bytecode.currentPc();
466         if (body != null)
467             body.accept(this);
468 
469         int pc3 = bytecode.currentPc();
470         if (notDo)
471             bytecode.write16bit(pc, pc3 - pc + 1);
472 
473         boolean alwaysBranch = compileBooleanExpr(true, expr);
474         if (alwaysBranch) {
475             bytecode.addOpcode(Opcode.GOTO);
476             alwaysBranch = breakList.size() == 0;
477         }
478 
479         bytecode.addIndex(pc2 - bytecode.currentPc() + 1);
480         patchGoto(breakList, bytecode.currentPc());
481         patchGoto(continueList, pc3);
482         continueList = prevContList;
483         breakList = prevBreakList;
484         hasReturned = alwaysBranch;
485     }

487     protected void patchGoto(ArrayList list, int targetPc) {
488         int n = list.size();
489         for (int i = 0; i < n; ++i) {
490             int pc = ((Integer)list.get(i)).intValue();
491             bytecode.write16bit(pc, targetPc - pc + 1);
492         }
493     }

495     private void atForStmnt(Stmnt st) throws CompileError {
496         ArrayList prevBreakList = breakList;
497         ArrayList prevContList = continueList;
498         breakList = new ArrayList();
499         continueList = new ArrayList();
500 
501         Stmnt init = (Stmnt)st.head();
502         ASTList p = st.tail();
503         ASTree expr = p.head();
504         p = p.tail();
505         Stmnt update = (Stmnt)p.head();
506         Stmnt body = (Stmnt)p.tail();
507 
508         if (init != null)
509             init.accept(this);
510 
511         int pc = bytecode.currentPc();
512         int pc2 = 0;
513         if (expr != null) {
514             if (compileBooleanExpr(false, expr)) {
515                 // in case of "for (...; false; ...)"
516                 continueList = prevContList;
517                 breakList = prevBreakList;
518                 hasReturned = false;
519                 return;
520             }
521 
522             pc2 = bytecode.currentPc();
523             bytecode.addIndex(0);
524         }
525 
526         if (body != null)
527             body.accept(this);
528 
529         int pc3 = bytecode.currentPc();
530         if (update != null)
531             update.accept(this);
532 
533         bytecode.addOpcode(Opcode.GOTO);
534         bytecode.addIndex(pc - bytecode.currentPc() + 1);
535 
536         int pc4 = bytecode.currentPc();
537         if (expr != null)
538             bytecode.write16bit(pc2, pc4 - pc2 + 1);
539 
540         patchGoto(breakList, pc4);
541         patchGoto(continueList, pc3);
542         continueList = prevContList;
543         breakList = prevBreakList;
544         hasReturned = false;
545     }

547     private void atSwitchStmnt(Stmnt st) throws CompileError {
548         compileExpr(st.head());
549 
550         ArrayList prevBreakList = breakList;
551         breakList = new ArrayList();
552         int opcodePc = bytecode.currentPc();
553         bytecode.addOpcode(LOOKUPSWITCH);
554         int npads = 3 - (opcodePc & 3);
555         while (npads-- > 0)
556             bytecode.add(0);
557 
558         Stmnt body = (Stmnt)st.tail();
559         int npairs = 0;
560         for (ASTList list = body; list != null; list = list.tail())
561             if (((Stmnt)list.head()).getOperator() == CASE)
562                 ++npairs;
563 
564         // opcodePc2 is the position at which the default jump offset is.
565         int opcodePc2 = bytecode.currentPc();
566         bytecode.addGap(4);
567         bytecode.add32bit(npairs);
568         bytecode.addGap(npairs * 8);
569 
570         long[] pairs = new long[npairs];
571         int ipairs = 0;
572         int defaultPc = -1;
573         for (ASTList list = body; list != null; list = list.tail()) {
574             Stmnt label = (Stmnt)list.head();
575             int op = label.getOperator();
576             if (op == DEFAULT)
577                 defaultPc = bytecode.currentPc();
578             else if (op != CASE)
579                 fatal();
580             else {
581                 pairs[ipairs++]
582                     = ((long)computeLabel(label.head()) << 32) + 
583                       ((long)(bytecode.currentPc() - opcodePc) & 0xffffffff);
584             }
585 
586             hasReturned = false;
587             ((Stmnt)label.tail()).accept(this);
588         }
589 
590         Arrays.sort(pairs);
591         int pc = opcodePc2 + 8;
592         for (int i = 0; i < npairs; ++i) {
593             bytecode.write32bit(pc, (int)(pairs[i] >>> 32));
594             bytecode.write32bit(pc + 4, (int)pairs[i]);
595             pc += 8;
596         } 
597 
598         if (defaultPc < 0 || breakList.size() > 0)
599             hasReturned = false;
600 
601         int endPc = bytecode.currentPc();
602         if (defaultPc < 0)
603             defaultPc = endPc;
604 
605         bytecode.write32bit(opcodePc2, defaultPc - opcodePc);
606 
607         patchGoto(breakList, endPc);
608         breakList = prevBreakList;
609     }

611     private int computeLabel(ASTree expr) throws CompileError {
612         doTypeCheck(expr);
613         expr = TypeChecker.stripPlusExpr(expr);
614         if (expr instanceof IntConst)
615             return (int)((IntConst)expr).get();
616         else
617             throw new CompileError("bad case label");
618     }

620     private void atBreakStmnt(Stmnt st, boolean notCont)
621         throws CompileError
622     {
623         if (st.head() != null)
624             throw new CompileError(
625                         "sorry, not support labeled break or continue");
626 
627         bytecode.addOpcode(Opcode.GOTO);
628         Integer pc = new Integer(bytecode.currentPc());
629         bytecode.addIndex(0);
630         if (notCont)
631             breakList.add(pc);
632         else
633             continueList.add(pc);
634     }

640     protected final void atReturnStmnt2(ASTree result) throws CompileError {
641         int op;
642         if (result == null)
643             op = Opcode.RETURN;
644         else {
645             compileExpr(result);
646             if (arrayDim > 0)
647                 op = ARETURN;
648             else {
649                 int type = exprType;
650                 if (type == DOUBLE)
651                     op = DRETURN;
652                 else if (type == FLOAT)
653                     op = FRETURN;
654                 else if (type == LONG)
655                     op = LRETURN;
656                 else if (isRefType(type))
657                     op = ARETURN;
658                 else
659                     op = IRETURN;
660             }
661         }
662 
663         for (ReturnHook har = returnHooks; har != null; har = har.next)
664             if (har.doit(bytecode, op)) {
665                 hasReturned = true;
666                 return;
667             }
668 
669         bytecode.addOpcode(op);
670         hasReturned = true;
671     }

689     private void atSyncStmnt(Stmnt st) throws CompileError {
690         int nbreaks = getListSize(breakList);
691         int ncontinues = getListSize(continueList);
692 
693         compileExpr(st.head());
694         if (exprType != CLASS && arrayDim == 0)
695             throw new CompileError("bad type expr for synchronized block");
696 
697         Bytecode bc = bytecode;
698         final int var = bc.getMaxLocals();
699         bc.incMaxLocals(1);
700         bc.addOpcode(DUP);
701         bc.addAstore(var);
702         bc.addOpcode(MONITORENTER);
703 
704         ReturnHook rh = new ReturnHook(this) {
705             protected boolean doit(Bytecode b, int opcode) {
706                 b.addAload(var);
707                 b.addOpcode(MONITOREXIT);
708                 return false;
709             }
710         };
711 
712         int pc = bc.currentPc();
713         Stmnt body = (Stmnt)st.tail();
714         if (body != null)
715             body.accept(this);
716 
717         int pc2 = bc.currentPc();
718         int pc3 = 0;
719         if (!hasReturned) {
720             rh.doit(bc, 0);     // the 2nd arg is ignored.
721             bc.addOpcode(Opcode.GOTO);
722             pc3 = bc.currentPc();
723             bc.addIndex(0);
724         }
725 
726         if (pc < pc2) {         // if the body is not empty
727             int pc4 = bc.currentPc();
728             rh.doit(bc, 0);         // the 2nd arg is ignored.
729             bc.addOpcode(ATHROW);
730             bc.addExceptionHandler(pc, pc2, pc4, 0);
731         }
732 
733         if (!hasReturned)
734             bc.write16bit(pc3, bc.currentPc() - pc3 + 1);
735 
736         rh.remove(this);
737 
738         if (getListSize(breakList) != nbreaks
739             || getListSize(continueList) != ncontinues)
740             throw new CompileError(
741                 "sorry, cannot break/continue in synchronized block");
742     }

998         int[] code = binOp;

1001             if (code[k] == token)

1007     public void atBinExpr(BinExpr expr) throws CompileError {
1008         int token = expr.getOperator();
1009 
1010         /* arithmetic operators: +, -, *, /, %, |, ^, &, <<, >>, >>>
1011          */
1012         int k = lookupBinOp(token);
1013         if (k >= 0) {
1014             expr.oprand1().accept(this);
1015             ASTree right = expr.oprand2();
1016             if (right == null)
1017                 return;     // see TypeChecker.atBinExpr().
1018 
1019             int type1 = exprType;
1020             int dim1 = arrayDim;
1021             String cname1 = className;
1022             right.accept(this);
1023             if (dim1 != arrayDim)
1024                 throw new CompileError("incompatible array types");
1025 
1026             if (token == '+' && dim1 == 0
1027                 && (type1 == CLASS || exprType == CLASS))
1028                 atStringConcatExpr(expr, type1, dim1, cname1);
1029             else
1030                 atArithBinExpr(expr, token, k, type1);
1031         }
1032         else {
1033             /* equation: &&, ||, ==, !=, <=, >=, <, >
1034             */
1035             if (!booleanExpr(true, expr)) {
1036                 bytecode.addIndex(7);
1037                 bytecode.addIconst(0);  // false
1038                 bytecode.addOpcode(Opcode.GOTO);
1039                 bytecode.addIndex(4);
1040             }
1041 
1042             bytecode.addIconst(1);  // true
1043         }
1044     }

1046     /* arrayDim values of the two oprands must be equal.
1047      * If an oprand type is not a numeric type, this method
1048      * throws an exception.
1049      */
1050     private void atArithBinExpr(Expr expr, int token,
1051                                 int index, int type1) throws CompileError
1052     {
1053         if (arrayDim != 0)
1054             badTypes(expr);
1055 
1056         int type2 = exprType;
1057         if (token == LSHIFT || token == RSHIFT || token == ARSHIFT)
1058             if (type2 == INT || type2 == SHORT
1059                 || type2 == CHAR || type2 == BYTE)
1060                 exprType = type1;
1061             else
1062                 badTypes(expr);
1063         else
1064             convertOprandTypes(type1, type2, expr);
1065 
1066         int p = typePrecedence(exprType);
1067         if (p >= 0) {
1068             int op = binOp[index + p + 1];
1069             if (op != NOP) {
1070                 if (p == P_INT && exprType != BOOLEAN)
1071                     exprType = INT;     // type1 may be BYTE, ...
1072 
1073                 bytecode.addOpcode(op);
1074                 return;
1075             }
1076         }
1077 
1078         badTypes(expr);
1079     }

1142     /* Produces the opcode to branch if the condition is true.
1143      * The oprand (branch offset) is not produced.
1144      *
1145      * @return	true if the compiled code is GOTO (always branch).
1146      * 			GOTO is not produced.
1147      */
1148     private boolean booleanExpr(boolean branchIf, ASTree expr)
1149         throws CompileError
1150     {
1151         boolean isAndAnd;
1152         int op = getCompOperator(expr);
1153         if (op == EQ) {         // ==, !=, ...
1154             BinExpr bexpr = (BinExpr)expr;
1155             int type1 = compileOprands(bexpr);
1156             // here, arrayDim might represent the array dim. of the left oprand
1157             // if the right oprand is NULL.
1158             compareExpr(branchIf, bexpr.getOperator(), type1, bexpr);
1159         }
1160         else if (op == '!')
1161             return booleanExpr(!branchIf, ((Expr)expr).oprand1());
1162         else if ((isAndAnd = (op == ANDAND)) || op == OROR) {
1163             BinExpr bexpr = (BinExpr)expr;
1164             if (booleanExpr(!isAndAnd, bexpr.oprand1())) {
1165                 exprType = BOOLEAN;
1166                 arrayDim = 0;
1167                 return true;
1168             }
1169             else {
1170             	int pc = bytecode.currentPc();
1171             	bytecode.addIndex(0);       // correct later
1172             	if (booleanExpr(isAndAnd, bexpr.oprand2()))
1173             		bytecode.addOpcode(Opcode.GOTO);
1174 
1175             	bytecode.write16bit(pc, bytecode.currentPc() - pc + 3);
1176             	if (branchIf != isAndAnd) {
1177             		bytecode.addIndex(6);   // skip GOTO instruction
1178             		bytecode.addOpcode(Opcode.GOTO);
1179             	}
1180             }
1181         }
1182         else if (isAlwaysBranch(expr, branchIf)) {
1183         	// Opcode.GOTO is not added here.  The caller must add it.
1184             exprType = BOOLEAN;
1185             arrayDim = 0;
1186             return true;	// always branch
1187         }
1188         else {                          // others
1189             expr.accept(this);
1190             if (exprType != BOOLEAN || arrayDim != 0)
1191                 throw new CompileError("boolean expr is required");
1192 
1193             bytecode.addOpcode(branchIf ? IFNE : IFEQ);
1194         }
1195 
1196         exprType = BOOLEAN;
1197         arrayDim = 0;
1198         return false;
1199     }

1204             return branchIf ? t == TRUE : t == FALSE;
1205         }

1210     static int getCompOperator(ASTree expr) throws CompileError {
1211         if (expr instanceof Expr) {
1212             Expr bexpr = (Expr)expr;
1213             int token = bexpr.getOperator();

1215                 return '!';
1216             else if ((bexpr instanceof BinExpr)
1217                      && token != OROR && token != ANDAND
1218                      && token != '&' && token != '|')

1221                 return token;

1258     /* Produces the opcode to branch if the condition is true.
1259      * The oprands are not produced.
1260      *
1261      * Parameter expr - compare expression ==, !=, <=, >=, <, >
1262      */
1263     private void compareExpr(boolean branchIf,
1264                              int token, int type1, BinExpr expr)
1265         throws CompileError
1266     {
1267         if (arrayDim == 0)
1268             convertOprandTypes(type1, exprType, expr);
1269 
1270         int p = typePrecedence(exprType);
1271         if (p == P_OTHER || arrayDim > 0)
1272             if (token == EQ)
1273                 bytecode.addOpcode(branchIf ? IF_ACMPEQ : IF_ACMPNE);
1274             else if (token == NEQ)
1275                 bytecode.addOpcode(branchIf ? IF_ACMPNE : IF_ACMPEQ);
1276             else
1277                 badTypes(expr);
1278         else
1279             if (p == P_INT) {
1280                 int op[] = ifOp;
1281                 for (int i = 0; i < op.length; i += 3)
1282                     if (op[i] == token) {
1283                         bytecode.addOpcode(op[i + (branchIf ? 1 : 2)]);
1284                         return;
1285                     }
1286 
1287                 badTypes(expr);
1288             }
1289             else {
1290                 if (p == P_DOUBLE)
1291                     if (token == '<' || token == LE)
1292                         bytecode.addOpcode(DCMPG);
1293                     else
1294                         bytecode.addOpcode(DCMPL);
1295                 else if (p == P_FLOAT)
1296                     if (token == '<' || token == LE)
1297                         bytecode.addOpcode(FCMPG);
1298                     else
1299                         bytecode.addOpcode(FCMPL);
1300                 else if (p == P_LONG)
1301                     bytecode.addOpcode(LCMP); // 1: >, 0: =, -1: <
1302                 else
1303                     fatal();
1304 
1305                 int[] op = ifOp2;
1306                 for (int i = 0; i < op.length; i += 3)
1307                     if (op[i] == token) {
1308                         bytecode.addOpcode(op[i + (branchIf ? 1 : 2)]);
1309                         return;
1310                     }
1311 
1312                 badTypes(expr);
1313             }
1314     }

1364     /* do implicit type conversion.
1365      * arrayDim values of the two oprands must be zero.
1366      */
1367     private void convertOprandTypes(int type1, int type2, Expr expr)
1368         throws CompileError
1369     {
1370         boolean rightStrong;
1371         int type1_p = typePrecedence(type1);
1372         int type2_p = typePrecedence(type2);
1373 
1374         if (type2_p < 0 && type1_p < 0) // not primitive types
1375             return;
1376 
1377         if (type2_p < 0 || type1_p < 0) // either is not a primitive type
1378             badTypes(expr);
1379 
1380         int op, result_type;
1381         if (type1_p <= type2_p) {
1382             rightStrong = false;
1383             exprType = type1;
1384             op = castOp[type2_p * 4 + type1_p];
1385             result_type = type1_p;
1386         }
1387         else {
1388             rightStrong = true;
1389             op = castOp[type1_p * 4 + type2_p];
1390             result_type = type2_p;
1391         }
1392 
1393         if (rightStrong) {
1394             if (result_type == P_DOUBLE || result_type == P_LONG) {
1395                 if (type1_p == P_DOUBLE || type1_p == P_LONG)
1396                     bytecode.addOpcode(DUP2_X2);
1397                 else
1398                     bytecode.addOpcode(DUP2_X1);
1399 
1400                 bytecode.addOpcode(POP2);
1401                 bytecode.addOpcode(op);

1470     void atNumCastExpr(int srcType, int destType)
1471         throws CompileError
1472     {
1473         if (srcType == destType)
1474             return;
1475         
1476         int op, op2;
1477         int stype = typePrecedence(srcType);
1478         int dtype = typePrecedence(destType);
1479         if (0 <= stype && stype < 3)
1480             op = castOp[stype * 4 + dtype];
1481         else
1482             op = NOP;
1483 
1484         if (destType == DOUBLE)
1485             op2 = I2D;
1486         else if (destType == FLOAT)
1487             op2 = I2F;
1488         else if (destType == LONG)
1489             op2 = I2L;
1490         else if (destType == SHORT)
1491             op2 = I2S;
1492         else if (destType == CHAR)
1493             op2 = I2C;
1494         else if (destType == BYTE)
1495             op2 = I2B;
1496         else
1497             op2 = NOP;
1498 
1499         if (op != NOP)
1500             bytecode.addOpcode(op);
1501 
1502         if (op == NOP || op == L2I || op == F2I || op == D2I)
1503             if (op2 != NOP)
1504                 bytecode.addOpcode(op2);
1505     }

1507     public void atExpr(Expr expr) throws CompileError {
1508         // array access, member access,
1509         // (unary) +, (unary) -, ++, --, !, ~
1510 
1511         int token = expr.getOperator();
1512         ASTree oprand = expr.oprand1();
1513         if (token == '.') {
1514             String member = ((Symbol)expr.oprand2()).get();
1515             if (member.equals("class"))                
1516                 atClassObject(expr);  // .class
1517             else
1518                 atFieldRead(expr);
1519         }
1520         else if (token == MEMBER) {     // field read
1521             /* MEMBER ('#') is an extension by Javassist.
1522              * The compiler internally uses # for compiling .class
1523              * expressions such as "int.class".
1524              */
1525             atFieldRead(expr);
1526         }
1527         else if (token == ARRAY)
1528             atArrayRead(oprand, expr.oprand2());
1529         else if (token == PLUSPLUS || token == MINUSMINUS)
1530             atPlusPlus(token, oprand, expr, true);
1531         else if (token == '!') {
1532             if (!booleanExpr(false, expr)) {
1533                 bytecode.addIndex(7);
1534                 bytecode.addIconst(1);
1535                 bytecode.addOpcode(Opcode.GOTO);
1536                 bytecode.addIndex(4);
1537             }
1538 
1539             bytecode.addIconst(0);
1540         }
1541         else if (token == CALL)         // method call
1542             fatal();
1543         else {
1544             expr.oprand1().accept(this);
1545             int type = typePrecedence(exprType);
1546             if (arrayDim > 0)
1547                 badType(expr);
1548 
1549             if (token == '-') {
1550                 if (type == P_DOUBLE)
1551                     bytecode.addOpcode(DNEG);
1552                 else if (type == P_FLOAT)
1553                     bytecode.addOpcode(FNEG);
1554                 else if (type == P_LONG)
1555                     bytecode.addOpcode(LNEG);
1556                 else if (type == P_INT) {
1557                     bytecode.addOpcode(INEG);
1558                     exprType = INT;     // type may be BYTE, ...
1559                 }
1560                 else
1561                     badType(expr);
1562             }
1563             else if (token == '~') {
1564                 if (type == P_INT) {
1565                     bytecode.addIconst(-1);
1566                     bytecode.addOpcode(IXOR);
1567                     exprType = INT;     // type may be BYTE. ...
1568                 }
1569                 else if (type == P_LONG) {
1570                     bytecode.addLconst(-1);
1571                     bytecode.addOpcode(LXOR);
1572                 }
1573                 else
1574                     badType(expr);
1575 
1576             }
1577             else if (token == '+') {
1578                 if (type == P_OTHER)
1579                     badType(expr);
1580 
1581                 // do nothing. ignore.
1582             }
1583             else
1584                 fatal();
1585         }
1586     }

1596     public void atClassObject(Expr expr) throws CompileError {
1597         ASTree op1 = expr.oprand1();
1598         if (!(op1 instanceof Symbol))
1599             throw new CompileError("fatal error: badly parsed .class expr");
1600 
1601         String cname = ((Symbol)op1).get();
1602         if (cname.startsWith("[")) {
1603             int i = cname.indexOf("[L");
1604             if (i >= 0) {
1605                 String name = cname.substring(i + 2, cname.length() - 1);
1606                 String name2 = resolveClassName(name);
1607                 if (!name.equals(name2)) {
1608                     /* For example, to obtain String[].class,
1609                      * "[Ljava.lang.String;" (not "[Ljava/lang/String"!)
1610                      * must be passed to Class.forName().
1611                      */
1612                     name2 = MemberResolver.jvmToJavaName(name2);
1613                     StringBuffer sbuf = new StringBuffer();
1614                     while (i-- >= 0)
1615                         sbuf.append('[');
1616 
1617                     sbuf.append('L').append(name2).append(';');
1618                     cname = sbuf.toString();
1619                 }
1620             }
1621         }
1622         else {
1623             cname = resolveClassName(MemberResolver.javaToJvmName(cname));
1624             cname = MemberResolver.jvmToJavaName(cname);
1625         }
1626 
1627         atClassObject2(cname);
1628         exprType = CLASS;
1629         arrayDim = 0;
1630         className = "java/lang/Class";
1631     }

1749     private void atPlusPlus(int token, ASTree oprand, Expr expr,
1750                             boolean doDup) throws CompileError
1751     {
1752         boolean isPost = oprand == null;        // ++i or i++?
1753         if (isPost)
1754             oprand = expr.oprand2();
1755 
1756         if (oprand instanceof Variable) {
1757             Declarator d = ((Variable)oprand).getDeclarator();
1758             int t = exprType = d.getType();
1759             arrayDim = d.getArrayDim();
1760             int var = getLocalVar(d);
1761             if (arrayDim > 0)
1762                 badType(expr);
1763 
1764             if (t == DOUBLE) {
1765                 bytecode.addDload(var);
1766                 if (doDup && isPost)
1767                     bytecode.addOpcode(DUP2);
1768 
1769                 bytecode.addDconst(1.0);
1770                 bytecode.addOpcode(token == PLUSPLUS ? DADD : DSUB);
1771                 if (doDup && !isPost)
1772                     bytecode.addOpcode(DUP2);
1773 
1774                 bytecode.addDstore(var);
1775             }
1776             else if (t == LONG) {
1777                 bytecode.addLload(var);
1778                 if (doDup && isPost)
1779                     bytecode.addOpcode(DUP2);
1780 
1781                 bytecode.addLconst((long)1);
1782                 bytecode.addOpcode(token == PLUSPLUS ? LADD : LSUB);
1783                 if (doDup && !isPost)
1784                     bytecode.addOpcode(DUP2);
1785 
1786                 bytecode.addLstore(var);
1787             }
1788             else if (t == FLOAT) {
1789                 bytecode.addFload(var);
1790                 if (doDup && isPost)
1791                     bytecode.addOpcode(DUP);
1792 
1793                 bytecode.addFconst(1.0f);
1794                 bytecode.addOpcode(token == PLUSPLUS ? FADD : FSUB);
1795                 if (doDup && !isPost)
1796                     bytecode.addOpcode(DUP);
1797 
1798                 bytecode.addFstore(var);
1799             }
1800             else if (t == BYTE || t == CHAR || t == SHORT || t == INT) {
1801                 if (doDup && isPost)
1802                     bytecode.addIload(var);
1803 
1804                 int delta = token == PLUSPLUS ? 1 : -1;
1805                 if (var > 0xff) {
1806                     bytecode.addOpcode(WIDE);
1807                     bytecode.addOpcode(IINC);
1808                     bytecode.addIndex(var);
1809                     bytecode.addIndex(delta);
1810                 }
1811                 else {
1812                     bytecode.addOpcode(IINC);
1813                     bytecode.add(var);
1814                     bytecode.add(delta);
1815                 }
1816 
1817                 if (doDup && !isPost)
1818                     bytecode.addIload(var);
1819             }
1820             else
1821                 badType(expr);
1822         }
1823         else {
1824             if (oprand instanceof Expr) {
1825                 Expr e = (Expr)oprand;
1826                 if (e.getOperator() == ARRAY) {
1827                     atArrayPlusPlus(token, isPost, e, doDup);
1828                     return;
1829                 }
1830             }
1831 
1832             atFieldPlusPlus(token, isPost, oprand, expr, doDup);
1833         }
1834     }
