270 	private void createColumns() {
271 		for (AlertColumnEnum column : AlertColumnEnum.values()) {
272 			TableViewerColumn c = createTableViewerColumn(column.getTitle(), column.getWeight(), column.getAlignment(), column.isResizable(), column.isMoveable(), column.isNumber());
273 			ColumnLabelProvider labelProvider = null;
274 			switch (column) {
275 			case TIME :
276 				labelProvider = new ColumnLabelProvider() {
277 					public String getText(Object element) {
278 						if (element instanceof AlertData) {
279 							return DateUtil.format(((AlertData) element).time, "yyyy-MM-dd HH:mm:ss");
280 						}
281 						return null;
282 					}
283 				};
284 				break;
285 			case LEVEL :
286 				labelProvider = new ColumnLabelProvider() {
287 					public String getText(Object element) {
288 						if (element instanceof AlertData) {
289 							return ((AlertData) element).level;
290 						}
291 						return null;
292 					}
293 				};
294 				break;
295 			case OBJECT :
296 				labelProvider = new ColumnLabelProvider() {
297 					public String getText(Object element) {
298 						if (element instanceof AlertData) {
299 							return ((AlertData) element).object;
300 						}
301 						return null;
302 					}
303 				};
304 				break;
305 			case TITLE :
306 				labelProvider = new ColumnLabelProvider() {
307 					public String getText(Object element) {
308 						if (element instanceof AlertData) {
309 							return ((AlertData) element).title;
310 						}
311 						return null;
312 					}
313 				};
314 				break;
315 			case MESSAGE :
316 				labelProvider = new ColumnLabelProvider() {
317 					public String getText(Object element) {
318 						if (element instanceof AlertData) {
319 							return ((AlertData) element).message;
320 						}
321 						return null;
322 					}
323 				};
324 				break;
325 			case TAGS :
326 				labelProvider = new ColumnLabelProvider() {
327 					public String getText(Object element) {
328 						if (element instanceof AlertData) {
329 							return ((AlertData) element).tags.toString();
330 						}
331 						return null;
332 					}
333 				};
334 				break;
335 			}
336 			if (labelProvider != null) {
337 				c.setLabelProvider(labelProvider);
338 			}
339 		}
340 	}

342 	private TableViewerColumn createTableViewerColumn(String title, int weight, int alignment,  boolean resizable, boolean moveable, final boolean isNumber) {
343 		final TableViewerColumn viewerColumn = new TableViewerColumn(viewer, SWT.NONE);
344 		final TableColumn column = viewerColumn.getColumn();
345 		column.setText(title);
346 		column.setAlignment(alignment);
347 		column.setMoveable(moveable);
348 		tableColumnLayout.setColumnData(column, new ColumnWeightData(weight, 10, resizable));
349 		column.setData("isNumber", isNumber);
350 		column.addSelectionListener(new SelectionAdapter() {
351 			public void widgetSelected(SelectionEvent e) {
352 				ColumnLabelSorter sorter = (ColumnLabelSorter) viewer.getComparator();
353 				TableColumn selectedColumn = (TableColumn) e.widget;
354 				sorter.setColumn(selectedColumn);
355 			}
356 		});
357 		return viewerColumn;
358 	}

364 	public void setInput(long stime, long etime) {
365 		onPressedOk(DateUtil.yyyymmdd(stime));
366 		Calendar calendar = Calendar.getInstance();
367 		calendar.setTime(new Date(stime));
368 		int hours = calendar.get(Calendar.HOUR_OF_DAY);
369 		int minutes = calendar.get(Calendar.MINUTE);
370 		int seconds = calendar.get(Calendar.SECOND);
371 		fromTime.setTime(hours, minutes, seconds);
372 		calendar = Calendar.getInstance();
373 		calendar.setTime(new Date(etime));
374 		hours = calendar.get(Calendar.HOUR_OF_DAY);
375 		minutes = calendar.get(Calendar.MINUTE);
376 		seconds = calendar.get(Calendar.SECOND);
377 		toTime.setTime(hours, minutes, seconds);
378 	}

380 	private void load() {
381 		int fromHour = fromTime.getHours();
382 		int fromMin = fromTime.getMinutes();
383 		int toHour = toTime.getHours();
384 		int toMin = toTime.getMinutes();
385 		final long stime = DateUtil.getTime(yyyymmdd + (fromHour < 10 ? "0" + fromHour : fromHour) +  (fromMin < 10 ? "0" + fromMin : fromMin), "yyyyMMddHHmm");
386 		final long etime = DateUtil.getTime(yyyymmdd + (toHour < 10 ? "0" + toHour : toHour) +  (toMin < 10 ? "0" + toMin : toMin), "yyyyMMddHHmm");
387 		final String count = maxCountText.getText();
388 		final String level = levelCombo.getText();
389 		final String object = objText.getText();
390 		final String key = keyText.getText();
391 		ExUtil.asyncRun(new Runnable() {
392 			public void run() {
393 				TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
394 				final List<AlertData> dataList = new ArrayList<AlertData>();
395 				try {
396 					MapPack param = new MapPack();
397 					param.put("date", yyyymmdd);
398 					param.put("stime", stime);
399 					param.put("etime", etime);
400 					if (StringUtil.isNotEmpty(count)) {
401 						param.put("count", Long.valueOf(count));
402 					}
403 					if (StringUtil.isNotEmpty(level)  && "ALL".equalsIgnoreCase(level) == false) {
404 						param.put("level", level);
405 					}
406 					if (StringUtil.isNotEmpty(object)) {
407 						param.put("object", object);
408 					}
409 					if (StringUtil.isNotEmpty(key)) {
410 						param.put("key", key);
411 					}
412 					List<Pack> packList = tcp.process(RequestCmd.ALERT_LOAD_TIME, param);
413 					for (Pack pack : packList) {
414 						if (pack instanceof AlertPack) {
415 							AlertPack alertPack = (AlertPack) pack;
416 							AlertData data = new AlertData();
417 							data.time = alertPack.time;
418 							data.level = AlertLevel.getName(alertPack.level);
419 							data.object = TextProxy.object.getLoadText(yyyymmdd, alertPack.objHash, serverId);
420 							data.title = alertPack.title;
421 							data.message = alertPack.message;
422 							data.objType = alertPack.objType;
423 							data.tags = alertPack.tags;
424 							dataList.add(data);
425 						}
426 					}
427 				} catch (Exception e) {
428 					e.printStackTrace();
429 					ConsoleProxy.errorSafe(e.getMessage());
430 				} finally {
431 					TcpProxy.putTcpProxy(tcp);
432 				}
433 				ExUtil.exec(viewer.getTable(), new Runnable() {
434 					public void run() {
435 						viewer.setInput(dataList);
436 					}
437 				});
438 			}
439 		});
440 	}

500 	static class AlertData {
501 		public long time;
502 		public String level;
503 		public String object;
504 		public String title;
505 		public String message;
506 		public MapValue tags;
507 		public String objType;
508 		
509 		public AlertPack toPack() {
510 			AlertPack pack = new AlertPack();
511 			pack.level = AlertLevel.getValue(level);
512 			pack.message = message;
513 			pack.objHash = object == null ? 0 : HashUtil.hash(object);
514 			pack.objType = objType;
515 			pack.tags = tags;
516 			pack.time = time;
517 			pack.title = title;
518 			return pack;
519 		}
520 
521 		public String toString() {
522 			return "AlertData [time=" + time + ", level=" + level + ", object="
523 					+ object + ", title=" + title + ", message=" + message
524 					+ ", tags=" + tags + ", objType=" + objType + "]";
525 		}
