38  /**The dialog for configuring XYGraph properties.
39   * @author Xihui Chen
40   *
41   */
42  public class XYGraphConfigDialog extends Dialog {
43  	
44  	private GraphConfigPage graphConfigPage;
45  	private List<AnnotationConfigPage> annotationConfigPageList;
46  	private List<AxisConfigPage> axisConfigPageList;
47  	private List<TraceConfigPage> traceConfigPageList;
48  	private XYGraph xyGraph;
49  	private XYGraphConfigCommand command;
50  	private boolean changed = false;
51  	
52  	protected XYGraphConfigDialog(Shell parentShell, XYGraph xyGraph) {
53  		super(parentShell);	
54  		this.xyGraph = xyGraph;
55  		graphConfigPage = new GraphConfigPage(xyGraph);
56  		annotationConfigPageList = new ArrayList<AnnotationConfigPage>();
57  		axisConfigPageList = new ArrayList<AxisConfigPage>();
58  		traceConfigPageList = new ArrayList<TraceConfigPage>();
59  		command = new XYGraphConfigCommand(xyGraph);
60  		command.savePreviousStates();
61          // Allow resize
62          setShellStyle(getShellStyle() | SWT.RESIZE);
63  	}
64  	
65  	@Override
66  	protected void configureShell(Shell newShell) {
67  		super.configureShell(newShell);
68  		newShell.setText("Configure Graph Settings");
69  	}
70  	
71  	@Override
72  	protected Control createDialogArea(Composite parent) {
73  		final Composite parent_composite = (Composite) super.createDialogArea(parent);
74  		parent_composite.setLayout(new FillLayout());
75          final TabFolder tabFolder = new TabFolder(parent_composite, SWT.NONE);   
76          
77          Composite graphTabComposite = new Composite(tabFolder, SWT.NONE);        
78          graphConfigPage.createPage(graphTabComposite);
79          
80          TabItem graphConfigTab = new TabItem(tabFolder, SWT.NONE);
81          graphConfigTab.setText("Graph");
82          graphConfigTab.setToolTipText("Configure General Graph Settings");
83          graphConfigTab.setControl(graphTabComposite);
84          
85          
86          //Axis Configure Page
87          Composite axisTabComposite = new Composite(tabFolder, SWT.NONE);
88      	axisTabComposite.setLayout(new GridLayout(1, false));        	
89      	TabItem axisConfigTab = new TabItem(tabFolder, SWT.NONE);
90  	    axisConfigTab.setText("Axes");
91  	    axisConfigTab.setToolTipText("Configure Axes Settings");
92  	    axisConfigTab.setControl(axisTabComposite);
93  	    
94      	Group axisSelectGroup = new Group(axisTabComposite, SWT.NONE);
95      	axisSelectGroup.setLayoutData(new GridData(
96      			SWT.FILL, SWT.FILL,true, false));
97      	axisSelectGroup.setText("Select Axis");
98      	axisSelectGroup.setLayout(new GridLayout(1, false));    	        
99      	final Combo axisCombo = new Combo(axisSelectGroup, SWT.DROP_DOWN);
100     	axisCombo.setLayoutData(new GridData(SWT.FILL, 0, true, false));
101 	    for(Axis axis : xyGraph.getAxisList())
102 	        axisCombo.add(axis.getTitle() + (axis.isHorizontal() ? "(X-Axis)" : "(Y-Axis)"));	   
103 	    axisCombo.select(0);
104     	
105 	    final Composite axisConfigComposite = new Composite(axisTabComposite, SWT.NONE);
106 	    axisConfigComposite.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));
107 	    final StackLayout axisStackLayout = new StackLayout();
108     	axisConfigComposite.setLayout(axisStackLayout);        	
109 	    for(Axis axis : xyGraph.getAxisList()){
110 	        Group axisConfigGroup = new Group(axisConfigComposite, SWT.NONE); 	        	
111 	        axisConfigGroup.setText("Change Settings");
112 	        axisConfigGroup.setLayoutData(new GridData(
113         			SWT.FILL, SWT.FILL,true, true));
114 	        AxisConfigPage axisConfigPage = 
115 	        	new AxisConfigPage(xyGraph, axis);
116 	        axisConfigPageList.add(axisConfigPage);
117 	        axisConfigPage.createPage(axisConfigGroup);   	        
118 	    } 	        
119 	    axisStackLayout.topControl = axisConfigPageList.get(0).getComposite();
120 	    axisCombo.addSelectionListener(new SelectionAdapter(){
121 	        @Override
122     		public void widgetSelected(SelectionEvent e) {
123     			axisStackLayout.topControl = axisConfigPageList.get(
124     					axisCombo.getSelectionIndex()).getComposite();
125     			axisConfigComposite.layout(true, true);
126     		}
127     	}); 
128         
129 	   //Trace Configure Page     
130 	   if(xyGraph.getPlotArea().getTraceList().size() > 0){
131 	        Composite traceTabComposite = new Composite(tabFolder, SWT.NONE);
132 	    	traceTabComposite.setLayout(new GridLayout(1, false));        	
133 	    	TabItem traceConfigTab = new TabItem(tabFolder, SWT.NONE);
134 		    traceConfigTab.setText("Traces");
135 		    traceConfigTab.setToolTipText("Configure Traces Settings");
136 		    traceConfigTab.setControl(traceTabComposite);
137 		    
138 	    	Group traceSelectGroup = new Group(traceTabComposite, SWT.NONE);
139 	    	traceSelectGroup.setLayoutData(new GridData(
140 	    			SWT.FILL, SWT.FILL,true, false));
141 	    	traceSelectGroup.setText("Select Trace");
142 	    	traceSelectGroup.setLayout(new GridLayout(1, false));    	        
143 	    	final Combo traceCombo = new Combo(traceSelectGroup, SWT.DROP_DOWN);
144 	    	traceCombo.setLayoutData(new GridData(SWT.FILL, 0, true, false));
145 		    for(Trace trace : xyGraph.getPlotArea().getTraceList())
146 		        traceCombo.add(trace.getName());	   
147 		    traceCombo.select(0);
148 	    	
149 		    final Composite traceConfigComposite = new Composite(traceTabComposite, SWT.NONE);
150 		    traceConfigComposite.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));
151 		    final StackLayout traceStackLayout = new StackLayout();
152 	    	traceConfigComposite.setLayout(traceStackLayout);        	
153 		    for(Trace trace : xyGraph.getPlotArea().getTraceList()){
154 		        Group traceConfigGroup = new Group(traceConfigComposite, SWT.NONE); 	        	
155 		        traceConfigGroup.setText("Change Settings");
156 		        traceConfigGroup.setLayoutData(new GridData(
157 	        			SWT.FILL, SWT.FILL,true, true));
158 		        TraceConfigPage traceConfigPage = 
159 		        	new TraceConfigPage(xyGraph, trace);
160 		        traceConfigPageList.add(traceConfigPage);
161 		        traceConfigPage.createPage(traceConfigGroup);   	        
162 		    } 	        
163 		        traceStackLayout.topControl = traceConfigPageList.get(0).getComposite();
164 		        traceCombo.addSelectionListener(new SelectionAdapter(){
165 	    		@Override
166 	    		public void widgetSelected(SelectionEvent e) {
167 	    			traceStackLayout.topControl = traceConfigPageList.get(
168 	    					traceCombo.getSelectionIndex()).getComposite();
169 	    			traceConfigComposite.layout(true, true);
170 	    		}
171 	    	}); 
172 	   }
173 	        
174         
175         //Annotation Configure Page
176         if(xyGraph.getPlotArea().getAnnotationList().size()>0){
177         	Composite annoTabComposite = new Composite(tabFolder, SWT.NONE);
178         	annoTabComposite.setLayout(new GridLayout(1, false));        	
179         	TabItem annoConfigTab = new TabItem(tabFolder, SWT.NONE);
180 		    annoConfigTab.setText("Annotations");
181 		    annoConfigTab.setToolTipText("Configure Annotation Settings");
182 		    annoConfigTab.setControl(annoTabComposite);
183 		    
184         	Group annoSelectGroup = new Group(annoTabComposite, SWT.NONE);
185         	annoSelectGroup.setLayoutData(new GridData(
186         			SWT.FILL, SWT.FILL,true, false));
187         	annoSelectGroup.setText("Select Annotation");
188         	annoSelectGroup.setLayout(new GridLayout(1, false));    	        
189         	final Combo annotationsCombo = new Combo(annoSelectGroup, SWT.DROP_DOWN);
190         	annotationsCombo.setLayoutData(new GridData(SWT.FILL, 0, true, false));
191  	        for(Annotation annotation : xyGraph.getPlotArea().getAnnotationList())
192  	        	annotationsCombo.add(annotation.getName());
193  	        annotationsCombo.select(0);
194         	
195  	        final Composite annoConfigComposite = new Composite(annoTabComposite, SWT.NONE);
196  	        annoConfigComposite.setLayoutData(new GridData(
197  	        		SWT.FILL, SWT.FILL, true, false));
198  	        final StackLayout stackLayout = new StackLayout();
199         	annoConfigComposite.setLayout(stackLayout);        	
200  	        for(Annotation annotation : xyGraph.getPlotArea().getAnnotationList()){
201  	        	Group annoConfigGroup = new Group(annoConfigComposite, SWT.NONE); 	        	
202 		        annoConfigGroup.setText("Change Settings");
203 		        annoConfigGroup.setLayoutData(new GridData(
204 	        			SWT.FILL, SWT.FILL,true, true));
205 		        AnnotationConfigPage annotationConfigPage = 
206 		        	new AnnotationConfigPage(xyGraph, annotation);
207 		        annotationConfigPageList.add(annotationConfigPage);
208 		        annotationConfigPage.createPage(annoConfigGroup);   	        
209  	        } 	        
210  	        stackLayout.topControl = annotationConfigPageList.get(0).getComposite();
211  	        annotationsCombo.addSelectionListener(new SelectionAdapter(){
212         		@Override
213         		public void widgetSelected(SelectionEvent e) {
214         			stackLayout.topControl = annotationConfigPageList.get(
215         					annotationsCombo.getSelectionIndex()).getComposite();
216         			annoConfigComposite.layout(true, true);
217         		}
218         	}); 	       
219         }
220         
221 		return parent_composite;
222 	}
223 	
224 	@Override
225 	protected void createButtonsForButtonBar(Composite parent) {
226 		((GridLayout) parent.getLayout()).numColumns++;
227 		Button button = new Button(parent, SWT.PUSH);
228 		button.setText("Apply");
229 		GridData data = new GridData(GridData.HORIZONTAL_ALIGN_FILL);
230 		int widthHint = convertHorizontalDLUsToPixels(IDialogConstants.BUTTON_WIDTH);
231 		Point minSize = button.computeSize(SWT.DEFAULT, SWT.DEFAULT, true);
232 		data.widthHint = Math.max(widthHint, minSize.x);
233 		button.setLayoutData(data);
234 		button.addSelectionListener(new SelectionAdapter(){
235 			@Override
236 			public void widgetSelected(SelectionEvent e) {
237 				applyChanges();
238 			}
239 		});		
240 		super.createButtonsForButtonBar(parent);
241 		Shell shell = parent.getShell();
242 		if (shell != null) {
243 			shell.setDefaultButton(button);
244 		}
245 	}
246 	
247 	
248 	@Override
249 	protected void okPressed() {	
250 		applyChanges();
251 		command.saveAfterStates();
252 		xyGraph.getOperationsManager().addCommand(command);
253 		super.okPressed();
254 	}
255 	
256 	private void applyChanges(){	
257 		changed = true;
258 		graphConfigPage.applyChanges();
259 		for(AxisConfigPage axisConfigPage : axisConfigPageList)
260 			axisConfigPage.applyChanges();
261 		for(TraceConfigPage traceConfigPage : traceConfigPageList)
262 			traceConfigPage.applyChanges();
263 		for(AnnotationConfigPage annotationConfigPage : annotationConfigPageList)
264 			annotationConfigPage.applyChanges();	
265 		
266 		//Add L.PHILIPPE
267 		xyGraph.fireConfigChanged();
268 	}
269 	
270 	@Override
271 	protected void cancelPressed() {
272 		if(changed){
273 			command.saveAfterStates();
274 			xyGraph.getOperationsManager().addCommand(command);
275 		}			
276 		super.cancelPressed();
277 	}
