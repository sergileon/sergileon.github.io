1   package scouter.client.popup;
2   
3   import org.csstudio.swt.xygraph.figures.Axis;
4   import org.eclipse.jface.dialogs.Dialog;
5   import org.eclipse.jface.dialogs.MessageDialog;
6   import org.eclipse.swt.SWT;
7   import org.eclipse.swt.events.SelectionAdapter;
8   import org.eclipse.swt.events.SelectionEvent;
9   import org.eclipse.swt.layout.FillLayout;
10  import org.eclipse.swt.layout.GridData;
11  import org.eclipse.swt.layout.GridLayout;
12  import org.eclipse.swt.widgets.Button;
13  import org.eclipse.swt.widgets.Composite;
14  import org.eclipse.swt.widgets.Control;
15  import org.eclipse.swt.widgets.Group;
16  import org.eclipse.swt.widgets.Label;
17  import org.eclipse.swt.widgets.Shell;
18  import org.eclipse.swt.widgets.Text;
19  
20  import scouter.util.CastUtil;
21  
22  public class SetRangeDialog extends Dialog {
23  	
24  	Button autoRange;
25  	Axis axis;
26  	Text maxTxt;
27  	Text minTxt;
28  
29  	public SetRangeDialog(Shell parentShell, Axis axis) {
30  		super(parentShell);
31  		this.axis = axis;
32  	}
33  
34  	protected Control createDialogArea(Composite parent) {
35  		Composite comp =  (Composite) super.createDialogArea(parent);
36  		FillLayout fillLayout = new FillLayout();
37  		fillLayout.marginWidth = 5;
38  		comp.setLayout(fillLayout);
39  		Group container = new Group(comp, SWT.NONE);
40  		GridLayout layout = new GridLayout();
41  		layout.numColumns = 2;
42  		layout.makeColumnsEqualWidth = false;
43  		layout.marginWidth = 5;
44  		layout.marginHeight = 5;
45  		container.setLayout(layout);
46  		autoRange = new Button(container, SWT.CHECK);
47  		autoRange.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false, 2, 1));
48  		autoRange.setText("Auto Range");
49  		autoRange.addSelectionListener(new SelectionAdapter() {
50  			public void widgetSelected(SelectionEvent e) {
51  				if (autoRange.getSelection()) {
52  					maxTxt.setEnabled(false);
53  					minTxt.setEnabled(false);
54  				} else {
55  					maxTxt.setEnabled(true);
56  					minTxt.setEnabled(true);
57  				}
58  			}
59  		});
60  		Label maxLbl = new Label(container, SWT.NONE);
61  		maxLbl.setLayoutData(new GridData(SWT.FILL, SWT.FILL, false, false));
62  		maxLbl.setAlignment(SWT.RIGHT);
63  		maxLbl.setText("Max : ");
64  		maxTxt = new Text(container, SWT.BORDER);
65  		maxTxt.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));
66  		Label minLbl = new Label(container, SWT.NONE);
67  		minLbl.setLayoutData(new GridData(SWT.FILL, SWT.FILL, false, false));
68  		minLbl.setAlignment(SWT.RIGHT);
69  		minLbl.setText("Min : ");
70  		minTxt = new Text(container, SWT.BORDER);
71  		minTxt.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));
72  		
73  		if (axis.isAutoScale()) {
74  			autoRange.setSelection(true);
75  			maxTxt.setText(CastUtil.cString(axis.getRange().getUpper()));
76  			minTxt.setText(CastUtil.cString(axis.getRange().getLower()));
77  			maxTxt.setEnabled(false);
78  			minTxt.setEnabled(false);
79  		} else {
80  			maxTxt.setText(CastUtil.cString(axis.getRange().getUpper()));
81  			minTxt.setText(CastUtil.cString(axis.getRange().getLower()));
82  		}
83  		return container;
84  	}
85  	
86  	protected void configureShell(Shell newShell) {
87  		super.configureShell(newShell);
88  		newShell.setText("Set Range");
89  	}
90  
91  	protected boolean isResizable() {
92  		return false;
93  	}
94  
95  	protected void okPressed() {
96  		if (autoRange.getSelection()) {
97  			axis.setAutoScale(true);
98  			super.okPressed();
99  		} else {
100 			try {
101 				double max = CastUtil.cdouble(maxTxt.getText().trim());
102 				double min = CastUtil.cdouble(minTxt.getText().trim());
103 				if (max > min) {
104 					axis.setRangeDirect(min, max);
105 				} else {
106 					axis.setRangeDirect(max, min);
107 				}
108 				axis.setAutoScale(false);
109 				super.okPressed();
110 			} catch (Exception e) {
111 				MessageDialog.openError(getShell(), "Error", e.toString());
112 			}
113 		}
114 	}
115 }
