31  public class Lex implements TokenId {
32      private int lastChar;
33      private StringBuffer textBuffer;
34      private Token currentToken;
35      private Token lookAheadTokens;
36  
37      private String input;
38      private int position, maxlen, lineNumber;
39  
40      /**
41       * Constructs a lexical analyzer.
42       */
43      public Lex(String s) {
44          lastChar = -1;
45          textBuffer = new StringBuffer();
46          currentToken = new Token();
47          lookAheadTokens = null;
48  
49          input = s;
50          position = 0;
51          maxlen = s.length();
52          lineNumber = 0;
53      }
54  
55      public int get() {
56          if (lookAheadTokens == null)
57              return get(currentToken);
58          else {
59              Token t;
60              currentToken = t = lookAheadTokens;
61              lookAheadTokens = lookAheadTokens.next;
62              return t.tokenId;
63          }
64      }
65  
66      /**
67       * Looks at the next token.
68       */
69      public int lookAhead() {
70          return lookAhead(0);
71      }
72  
73      public int lookAhead(int i) {
74          Token tk = lookAheadTokens;
75          if (tk == null) {
76              lookAheadTokens = tk = currentToken;  // reuse an object!
77              tk.next = null;
78              get(tk);
79          }
80  
81          for (; i-- > 0; tk = tk.next)
82              if (tk.next == null) {
83                  Token tk2;
84                  tk.next = tk2 = new Token();
85                  get(tk2);
86              }
87  
88          currentToken = tk;
89          return tk.tokenId;
90      }
91  
92      public String getString() {
93          return currentToken.textValue;
94      }
95  
96      public long getLong() {
97          return currentToken.longValue;
98      }
99  
100     public double getDouble() {
101         return currentToken.doubleValue;
102     }

104     private int get(Token token) {
105         int t;
106         do {
107             t = readLine(token);
108         } while (t == '\n');
109         token.tokenId = t;
110         return t;
111     }
112 
113     private int readLine(Token token) {
114         int c = getNextNonWhiteChar();
115         if(c < 0)
116             return c;
117         else if(c == '\n') {
118             ++lineNumber;
119             return '\n';
120         }
121         else if (c == '\'')
122             return readCharConst(token);
123         else if (c == '"')
124             return readStringL(token);
125         else if ('0' <= c && c <= '9')
126             return readNumber(c, token);
127         else if(c == '.'){
128             c = getc();
129             if ('0' <= c && c <= '9') {
130                 StringBuffer tbuf = textBuffer;
131                 tbuf.setLength(0);
132                 tbuf.append('.');
133                 return readDouble(tbuf, c, token);
134             }
135             else{
136                 ungetc(c);
137                 return readSeparator('.');
138             }
139         }
140         else if (Character.isJavaIdentifierStart((char)c))
141             return readIdentifier(c, token);
142         else
143             return readSeparator(c);
144     }
145 
146     private int getNextNonWhiteChar() {
147         int c;
148         do {
149             c = getc();
150             if (c == '/') {
151                 c = getc();
152                 if (c == '/')
153                     do {
154                         c = getc();
155                     } while (c != '\n' && c != '\r' && c != -1);
156                 else if (c == '*')
157                     while (true) {
158                         c = getc();
159                         if (c == -1)
160                             break;
161                         else if (c == '*')
162                             if ((c = getc()) == '/') {
163                                 c = ' ';
164                                 break;
165                             }
166                             else
167                                 ungetc(c);
168                     }
169                 else {
170                     ungetc(c);
171                     c = '/';
172                 }
173             }
174         } while(isBlank(c));
175         return c;
176     }
177 
178     private int readCharConst(Token token) {
179         int c;
180         int value = 0;
181         while ((c = getc()) != '\'')
182             if (c == '\\')
183                 value = readEscapeChar();
184             else if (c < 0x20) {
185                 if (c == '\n')
186                     ++lineNumber;
187 
188                 return BadToken;
189             }
190             else
191                 value = c;
192 
193         token.longValue = value;
194         return CharConstant;
195     }
196 
197     private int readEscapeChar() {
198         int c = getc();
199         if (c == 'n')
200             c = '\n';
201         else if (c == 't')
202             c = '\t';
203         else if (c == 'r')
204             c = '\r';
205         else if (c == 'f')
206             c = '\f';
207         else if (c == '\n')
208             ++lineNumber;
209 
210         return c;
211     }
212 
213     private int readStringL(Token token) {
214         int c;
215         StringBuffer tbuf = textBuffer;
216         tbuf.setLength(0);
217         for (;;) {
218             while ((c = getc()) != '"') {
219                 if (c == '\\')
220                     c = readEscapeChar();
221                 else if (c == '\n' || c < 0) {
222                     ++lineNumber;
223                     return BadToken;
224                 }
225 
226                 tbuf.append((char)c);
227             }
228 
229             for (;;) {
230                 c = getc();
231                 if (c == '\n')
232                     ++lineNumber;
233                 else if (!isBlank(c))
234                     break;
235             }
236 
237             if (c != '"') {
238                 ungetc(c);
239                 break;
240             }
241         }
242 
243         token.textValue = tbuf.toString();
244         return StringL;
245     }
246 
247     private int readNumber(int c, Token token) {
248         long value = 0;
249         int c2 = getc();
250         if (c == '0')
251             if (c2 == 'X' || c2 == 'x')
252                 for (;;) {
253                     c = getc();
254                     if ('0' <= c && c <= '9')
255                         value = value * 16 + (long)(c - '0');
256                     else if ('A' <= c && c <= 'F')
257                         value = value * 16 + (long)(c - 'A' + 10);
258                     else if ('a' <= c && c <= 'f')
259                         value = value * 16 + (long)(c - 'a' + 10);
260                     else {
261                         token.longValue = value;
262                         if (c == 'L' || c == 'l')
263                             return LongConstant;
264                         else {
265                             ungetc(c);
266                             return IntConstant;
267                         }
268                     }
269                 }
270             else if ('0' <= c2 && c2 <= '7') {
271                 value = c2 - '0';
272                 for (;;) {
273                     c = getc();
274                     if ('0' <= c && c <= '7')
275                         value = value * 8 + (long)(c - '0');
276                     else {
277                         token.longValue = value;
278                         if (c == 'L' || c == 'l')
279                             return LongConstant;
280                         else {
281                             ungetc(c);
282                             return IntConstant;
283                         }
284                     }
285                 }
286             }
287 
288         value = c - '0';
289         while ('0' <= c2 && c2 <= '9') {
290             value = value * 10 + c2 - '0';
291             c2 = getc();
292         }
293 
294         token.longValue = value;
295         if (c2 == 'F' || c2 == 'f') {
296             token.doubleValue = (double)value;
297             return FloatConstant;
298         }
299         else if (c2 == 'E' || c2 == 'e'
300                  || c2 == 'D' || c2 == 'd' || c2 == '.') {
301             StringBuffer tbuf = textBuffer;
302             tbuf.setLength(0);
303             tbuf.append(value);
304             return readDouble(tbuf, c2, token);
305         }
306         else if (c2 == 'L' || c2 == 'l')
307             return LongConstant;
308         else {
309             ungetc(c2);
310             return IntConstant;
311         }
312     }

314     private int readDouble(StringBuffer sbuf, int c, Token token) {
315         if (c != 'E' && c != 'e' && c != 'D' && c != 'd') {
316             sbuf.append((char)c);
317             for (;;) {
318                 c = getc();
319                 if ('0' <= c && c <= '9')
320                     sbuf.append((char)c);
321                 else
322                     break;
323             }
324         }
325 
326         if (c == 'E' || c == 'e') {
327             sbuf.append((char)c);
328             c = getc();
329             if (c == '+' || c == '-') {
330                 sbuf.append((char)c);
331                 c = getc();
332             }
333 
334             while ('0' <= c && c <= '9') {
335                 sbuf.append((char)c);
336                 c = getc();
337             }
338         }
339 
340         try {
341             token.doubleValue = Double.parseDouble(sbuf.toString());
342         }
343         catch (NumberFormatException e) {
344             return BadToken;
345         }
346 
347         if (c == 'F' || c == 'f')
348             return FloatConstant;
349         else {
350             if (c != 'D' && c != 'd')
351                 ungetc(c);
352 
353             return DoubleConstant;
354         }
355     }

357     // !"#$%&'(    )*+,-./0    12345678    9:;<=>?
358     private static final int[] equalOps
359         =  { NEQ, 0, 0, 0, MOD_E, AND_E, 0, 0,
360              0, MUL_E, PLUS_E, 0, MINUS_E, 0, DIV_E, 0,
361              0, 0, 0, 0, 0, 0, 0, 0,
362              0, 0, 0, LE, EQ, GE, 0 };
363 
364     private int readSeparator(int c) {
365         int c2, c3;
366         if ('!' <= c && c <= '?') {
367             int t = equalOps[c - '!'];
368             if (t == 0) 
369                 return c;
370             else {
371                 c2 = getc();
372                 if (c == c2)
373                     switch (c) {
374                     case '=' :
375                         return EQ;
376                     case '+' :
377                         return PLUSPLUS;
378                     case '-' :
379                         return MINUSMINUS;
380                     case '&' :
381                         return ANDAND;
382                     case '<' :
383                         c3 = getc();
384                         if (c3 == '=')
385                             return LSHIFT_E;
386                         else {
387                             ungetc(c3);
388                             return LSHIFT;
389                         }
390                     case '>' :
391                         c3 = getc();
392                         if (c3 == '=')
393                             return RSHIFT_E;
394                         else if (c3 == '>') {
395                             c3 = getc();
396                             if (c3 == '=')
397                                 return ARSHIFT_E;
398                             else {
399                                 ungetc(c3);
400                                 return ARSHIFT;
401                             }
402                         }
403                         else {
404                             ungetc(c3);
405                             return RSHIFT;
406                         }
407                     default :
408                         break;
409                     }
410                 else if (c2 == '=')
411                     return t;
412             }
413         }
414         else if (c == '^') {
415             c2 = getc();
416             if (c2 == '=')
417                 return EXOR_E;
418         }
419         else if (c == '|') {
420             c2 = getc();
421             if (c2 == '=')
422                 return OR_E;
423             else if (c2 == '|')
424                 return OROR;
425         }
426         else
427             return c;
428 
429         ungetc(c2);
430         return c;
431     }

433     private int readIdentifier(int c, Token token) {
434         StringBuffer tbuf = textBuffer;
435         tbuf.setLength(0);
436 
437         do {
438             tbuf.append((char)c);
439             c = getc();
440         } while (Character.isJavaIdentifierPart((char)c));
441 
442         ungetc(c);
443 
444         String name = tbuf.toString();
445         int t = ktable.lookup(name);
446         if (t >= 0)
447             return t;
448         else {
449             /* tbuf.toString() is executed quickly since it does not
450              * need memory copy.  Using a hand-written extensible
451              * byte-array class instead of StringBuffer is not a good idea
452              * for execution speed.  Converting a byte array to a String
453              * object is very slow.  Using an extensible char array
454              * might be OK.
455              */
456             token.textValue = name;
457             return Identifier;
458         }
459     }

517     private static boolean isBlank(int c) {
518         return c == ' ' || c == '\t' || c == '\f' || c == '\r'
519             || c == '\n';
520     }
521 
522     private static boolean isDigit(int c) {
523         return '0' <= c && c <= '9';
524     }
525 
526     private void ungetc(int c) {
527         lastChar = c;
528     }
529 
530     public String getTextAround() {
531         int begin = position - 10;
532         if (begin < 0)
533             begin = 0;
534 
535         int end = position + 10;
536         if (end > maxlen)
537             end = maxlen;
538 
539         return input.substring(begin, end);
540     }
541 
542     private int getc() {
543         if (lastChar < 0)
544             if (position < maxlen)
545                 return input.charAt(position++);
546             else
547                 return -1;
548         else {
549             int c = lastChar;
550             lastChar = -1;
551             return c;
552         }
553     }
