83  	protected void makeActions(IWorkbenchWindow window) {
84  
85  		Server server = ServerManager.getInstance().getDefaultServer();
86  		int serverId = 0;
87  		if(server != null){
88  			serverId = server.getId();
89  		}
90  		
91  		exitAction = ActionFactory.QUIT.create(window);
92  		exitAction.setImageDescriptor(ImageUtil.getImageDescriptor(Images.MENU_EXIT));
93  		register(exitAction);
94  		
95  		aboutAction = ActionFactory.ABOUT.create(window);
96  		register(aboutAction);
97  		
98  		preferencesAction = ActionFactory.PREFERENCES.create(window);
99  		preferencesAction.setImageDescriptor(ImageUtil.getImageDescriptor(Images.preference));
100 		register(preferencesAction);
101 		
102 		register(addServerAction = new AddServerAction(window, "Add Server", Images.add));
103 		register(openServerManagerAction = new OpenServerManagerAction());
104 
105 		register(openClientThreadListAction = new OpenClientThreadListAction(window, "Client Thread List", Images.thread));
106 		register(openWorkspaceExplorerAction = new OpenWorkspaceExplorerAction(window, "Workspace Explorer", Images.explorer, serverId));
107 		register(openGroupNavigationAction = new OpenGroupNavigationAction(window));
108 		register(openClientEnvAction = new OpenClientEnvViewAction(window));
109 		register(openRestart = new RestartAction(window, "Restart"));
110 
111 		perspective = ContributionItemFactory.PERSPECTIVES_SHORTLIST.create(window);
112 		register(openAlertRealtimeAction = new OpenAlertRealTimeAction(window, MenuStr.ALERT_REAL, Images.alert));
113 		register(openConsole = new OpenConsoleAction(window, "Console"));
114 		register(openObjDashboard = new OpenObjectDashboardAction(window, "Object Dashboard"));
115 	}

121 	protected void fillMenuBar(IMenuManager menuBar) {
122 		MenuManager fileMenu = new MenuManager("&File", IWorkbenchActionConstants.M_FILE);
123 		MenuManager serverMenu = new MenuManager("&Server", "scouter.client.menu.server");
124 		MenuManager viewMenu = new MenuManager("&Views", "scouter.client.menu.views");
125 		MenuManager windowMenu = new MenuManager("&Window", IWorkbenchActionConstants.M_WINDOW);
126 		MenuManager reportMenu = new MenuManager("&Report", "scouter.client.menu.report");
127 
128 		menuBar.add(fileMenu);
129 		menuBar.add(serverMenu);
130 		menuBar.add(viewMenu);
131 		menuBar.add(windowMenu);
132 		menuBar.add(reportMenu);
133 
134 		viewMenu.add(openObjDashboard);
135 		viewMenu.add(openConsole);
136 		viewMenu.add(openAlertRealtimeAction);
137 		viewMenu.add(openGroupNavigationAction);
138 
139 		MenuManager perspectiveMenu = new MenuManager("&Perspective", "scouter.client.menu.perspective");
140 		perspectiveMenu.add(perspective);
141 
142 		windowMenu.add(perspectiveMenu);
143 		windowMenu.add(preferencesAction);
144 		
145 		serverMenu.add(addServerAction);
146 		serverMenu.add(new Separator());
147 		serverMenu.add(openServerManagerAction);
148 
149 		fileMenu.add(new Separator());
150 		fileMenu.add(openClientThreadListAction);
151 		fileMenu.add(openClientEnvAction);
152 		fileMenu.add(new Separator());
153 		fileMenu.add(openWorkspaceExplorerAction);
154 		fileMenu.add(new Separator());
155 		fileMenu.add(new GroupMarker(IWorkbenchActionConstants.MB_ADDITIONS));
156 		fileMenu.add(new Separator());
157 		fileMenu.add(new GroupMarker(IWorkbenchActionConstants.FILE_END));
158 
159 		fileMenu.add(openRestart);
160 		fileMenu.add(exitAction);
161 	}
