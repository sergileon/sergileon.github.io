8   package org.csstudio.swt.xygraph.undo;
9   
10  /** Abstract base for an undo-able command that saves some state
11   *  @author Kay Kasemir
12   */
13  abstract public class SaveStateCommand implements IUndoableCommand
14  {
15      /** Name of the command, shown in undo/redo GUI */
16      final private String name;
17      
18      /** Initialize
19       *  Derived class should save the 'original' state.
20       *   */
21      public SaveStateCommand(final String name)
22      {
23          this.name = name;
24      }
25  
26      /** Derived class should implement this to save the 'final' state.
27       *  Might be called multiple times to save intermediate states,
28       *  and the state saved on the last call would be the 'redo' state.
29       */
30      abstract public void saveState();
31          
32      /** @return Name of the command, shown in undo/redo GUI */
33      @Override
34      public String toString()
35      {
36          return name;
37      }
38  }
