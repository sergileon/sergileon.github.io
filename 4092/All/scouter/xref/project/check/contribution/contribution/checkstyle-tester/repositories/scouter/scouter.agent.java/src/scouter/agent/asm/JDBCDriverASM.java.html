17  package scouter.agent.asm;
18  import scouter.agent.ClassDesc;
19  import scouter.agent.Configure;
20  import scouter.agent.Logger;
21  import scouter.agent.asm.util.AsmUtil;
22  import scouter.agent.asm.util.HookingSet;
23  import scouter.agent.trace.TraceSQL;
24  import scouter.org.objectweb.asm.*;
25  import scouter.org.objectweb.asm.commons.LocalVariablesSorter;
26  
27  import java.util.Map;
28  public class JDBCDriverASM implements IASM, Opcodes {
29  	//user can define driver.connect()
30  	private Map<String, HookingSet> reserved =HookingSet.getHookingSet(Configure.getInstance().hook_jdbc_wrapping_driver_patterns);
31  	public JDBCDriverASM() {
32  		AsmUtil.add(reserved, "com/ibm/db2/jcc/DB2Driver",	"connect(Ljava/lang/String;Ljava/util/Properties;)Ljava/sql/Connection;");
33  	}
34  
35  	public ClassVisitor transform(ClassVisitor cv, String className, ClassDesc classDesc) {
36  		if (Configure.getInstance()._hook_dbconn_enabled == false) {
37  			return cv;
38  		}
39  		HookingSet mset = reserved.get(className);
40  		if (mset != null){
41  			return new JDBCDriverCV(cv, mset, className);
42  		}
43  
44  		return cv;
45  	}
46  }
47  class JDBCDriverCV extends ClassVisitor implements Opcodes {
48  	public String className;
49  	private HookingSet mset;
50  	public JDBCDriverCV(ClassVisitor cv, HookingSet mset, String className) {
51  		super(ASM4, cv);
52  		this.mset = mset;
53  		this.className = className;
54  	}
55  	@Override
56  	public MethodVisitor visitMethod(int access, String name, String desc, String signature, String[] exceptions) {
57  		MethodVisitor mv = super.visitMethod(access, name, desc, signature, exceptions);
58  		if (mv == null || mset.isA(name, desc) == false) {
59  			return mv;
60  		}
61  		if (AsmUtil.isSpecial(name)) {
62  			return mv;
63  		}
64  		String fullname = AsmUtil.makeMethodFullName(className, name, desc);
65         
66  		Logger.println("A105", "jdbc db2 driver  loaded: " + fullname);
67  		return new JDBCDriverMV(access, desc, mv, fullname);
68  	}
69  }
70  // ///////////////////////////////////////////////////////////////////////////
71  class JDBCDriverMV extends LocalVariablesSorter implements Opcodes {
72  	private static final String TRACE_SQL = TraceSQL.class.getName().replace('.', '/');
73  	private final static String CONNECT_METHOD = "driverConnect";
74  	private static final String CONNECT_SIGNATURE = "(Ljava/sql/Connection;Ljava/lang/String;)Ljava/sql/Connection;";
75  	private static final String ERR_SIGNATURE = "(Ljava/lang/String;Ljava/lang/Throwable;)V";
76  	
77  	private Label startFinally = new Label();
78  	public JDBCDriverMV(int access, String desc, MethodVisitor mv, String fullname) {
79  		super(ASM4,access, desc, mv);
80  		this.strArgIdx = AsmUtil.getStringIdx(access, desc);
81  	}
82  	private int strArgIdx;
83  	@Override
84  	public void visitCode() {
85  		mv.visitLabel(startFinally);
86  		mv.visitCode();
87  	}
88  	@Override
89  	public void visitInsn(int opcode) {
90  		if ((opcode >= IRETURN && opcode <= RETURN)) {
91  			mv.visitVarInsn(Opcodes.ALOAD, strArgIdx);
92  			mv.visitMethodInsn(Opcodes.INVOKESTATIC, TRACE_SQL, CONNECT_METHOD, CONNECT_SIGNATURE,false);
93  		}
94  		mv.visitInsn(opcode);
95  	}
96  	@Override
97  	public void visitMaxs(int maxStack, int maxLocals) {
98  		Label endFinally = new Label();
99  		mv.visitTryCatchBlock(startFinally, endFinally, endFinally, null);
100 		mv.visitLabel(endFinally);
101 		mv.visitInsn(DUP);
102 		
103 		int errIdx = newLocal(Type.getType(Throwable.class));
104 		mv.visitVarInsn(Opcodes.ASTORE, errIdx);
105 		mv.visitVarInsn(Opcodes.ALOAD, strArgIdx);
106 		mv.visitVarInsn(Opcodes.ALOAD, errIdx);
107 		mv.visitMethodInsn(Opcodes.INVOKESTATIC, TRACE_SQL, CONNECT_METHOD, ERR_SIGNATURE,false);
108 		mv.visitInsn(ATHROW);
109 		mv.visitMaxs(maxStack + 8, maxLocals + 2);
110 	}
111 }
