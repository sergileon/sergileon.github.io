50  public class GroupAssignmentDialog implements IAddGroup {

63  		this.display = display;
64  		this.serverId = serverId;
65  		this.objHash = objHash;
66  		this.objName = objName;
67  		this.objType = objType;
68  		this.callback = callback;
69  	}

71  	public void show() {
72  		final Shell dialog = new Shell(display, SWT.DIALOG_TRIM | SWT.APPLICATION_MODAL);
73  		UIUtil.setDialogDefaultFunctions(dialog);
74  		dialog.setLayout(new GridLayout(2, false));
75  		dialog.setText("Group Assignment");
76  		CLabel title = new CLabel(dialog, SWT.NONE);
77  		GridData gr = new GridData(SWT.LEFT, SWT.CENTER, true, true, 2, 1);
78  		title.setLayoutData(gr);
79  		title.setFont(new Font(null, "Arial", 10, SWT.BOLD));
80  		title.setImage(Images.getObjectIcon(objType, true, serverId));
81  		title.setText(objName);
82  		
83  		groupTable = new Table(dialog,  SWT.CHECK | SWT.BORDER);
84  		gr = new GridData(SWT.FILL, SWT.FILL, true, true);
85  		gr.widthHint = 250;
86  		gr.heightHint = 350;
87  		groupTable.setLayoutData(gr);
88  		
89  		Composite buttonComp = new Composite(dialog, SWT.NONE);
90  		gr = new GridData(SWT.FILL, SWT.FILL, false, true);
91  		buttonComp.setLayoutData(gr);
92  		buttonComp.setLayout(UIUtil.formLayout(3, 3));
93  		
94  		Button selectAllBtn = new Button(buttonComp, SWT.PUSH);
95  		selectAllBtn.setLayoutData(UIUtil.formData(null, -1, 0, 5, null, -1, null, -1, 100));
96  		selectAllBtn.setText("&Select All");
97  		selectAllBtn.addListener(SWT.Selection, new Listener() {
98  			public void handleEvent(Event event) {
99  				TableItem[] items = groupTable.getItems();
100 				for (TableItem item : items) {
101 					item.setChecked(true);
102 				}
103 			}
104 		});
105 		
106 		Button deselectAllBtn = new Button(buttonComp, SWT.PUSH);
107 		deselectAllBtn.setLayoutData(UIUtil.formData(null, -1, selectAllBtn, 5, null, -1, null, -1, 100));
108 		deselectAllBtn.setText("&Deselect All");
109 		deselectAllBtn.addListener(SWT.Selection, new Listener() {
110 			public void handleEvent(Event event) {
111 				TableItem[] items = groupTable.getItems();
112 				for (TableItem item : items) {
113 					item.setChecked(false);
114 				}
115 			}
116 		});
117 		
118 		Button newBtn = new Button(buttonComp, SWT.PUSH);
119 		newBtn.setLayoutData(UIUtil.formData(null, -1, deselectAllBtn, 10, null, -1, null, -1, 100));
120 		newBtn.setText("&New");
121 		newBtn.addListener(SWT.Selection, new Listener() {
122 			public void handleEvent(Event event) {
123 				AgentObject agent = AgentModelThread.getInstance().getAgentObject(objHash);
124 				if (agent == null) {
125 					return;
126 				}
127 				Server server = ServerManager.getInstance().getServer(agent.getServerId());
128 				if (server == null) {
129 					server = ServerManager.getInstance().getDefaultServer();
130 				}
131 				String displayObjType = server.getCounterEngine().getDisplayNameObjectType(objType);
132 				AddGroupDialog addDialog = new AddGroupDialog(display, GroupAssignmentDialog.this, objType, displayObjType, server.getId());
133 				addDialog.show();
134 			}
135 		});
136 		
137 		Composite bottomComp = new Composite(dialog, SWT.NONE);
138 		bottomComp.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, true, 2, 1));
139 		bottomComp.setLayout(UIUtil.formLayout(3, 3));
140 		final Button cancelBtn = new Button(bottomComp, SWT.PUSH);
141 		cancelBtn.setLayoutData(UIUtil.formData(null, -1, null, -1, 100, -5, null, -1, 100));
142 		cancelBtn.setText("&Cancel");
143 		cancelBtn.addListener(SWT.Selection, new Listener(){
144 			public void handleEvent(Event event) {
145 				dialog.close();
146 			}
147 		});
148 		
149 		final Button okBtn = new Button(bottomComp, SWT.PUSH);
150 		okBtn.setLayoutData(UIUtil.formData(null, -1, null, -1, cancelBtn, -5, null, -1, 100));
151 		okBtn.setText("&Ok");
152 		okBtn.addListener(SWT.Selection, new Listener(){
153 			public void handleEvent(Event event) {
154 				TableItem[] items = groupTable.getItems();
155 				List<String> checked = new ArrayList<String>();
156 				for (TableItem item : items) {
157 					if (item.getChecked()) {
158 						checked.add(item.getText());
159 					}
160 				}
161 				callback.endGroupAssignment(objHash, checked.toArray(new String[checked.size()]));
162 				dialog.close();
163 			}
164 		});
165 		
166 		Set<String> groupSet = GroupManager.getInstance().getGroupsByType(objType);
167 		Set<String> assginedGroups = GroupManager.getInstance().getObjGroups(objHash);
168 		for (String group : groupSet) {
169 			TableItem item = new TableItem(groupTable, SWT.NONE);
170 			item.setText(group);
171 			item.setChecked(assginedGroups.contains(group));
172 		}
173 		
174 		sortTable(groupTable);
175 		
176 		dialog.pack();
177 		dialog.open();
178 	}
179 
180 	private void sortTable(Table table) {
181 		TableItem[] items = table.getItems();
182 		Collator collator = Collator.getInstance(Locale.getDefault());
183 		for (int i = 1; i < items.length; i++) {
184 			String value1 = items[i].getText(0);
185 			for (int j = 0; j < i; j++) {
186 				String value2 = items[j].getText(0);
187 				if (collator.compare(value1, value2) < 0) {
188 					String text = items[i].getText(0);
189 					boolean checked = items[i].getChecked();
190 					items[i].dispose();
191 					TableItem item = new TableItem(table, SWT.NONE, j);
192 					item.setText(text);
193 					item.setChecked(checked);
194 					items = table.getItems();
195 					break;
196 				}
197 			}
198 		}
199 	}

201 	public boolean addedGroup(String objType, String name) {
202 		if (GroupManager.getInstance().addGroup(objType, name)) {
203 			TableItem item = new TableItem(groupTable, SWT.NONE);
204 			item.setText(name);
205 			sortTable(groupTable);
206 			return true;
207 		}
208 		return false;
209 	}
