92  	public PlotArea(final XYGraph xyGraph) {
93  		this.xyGraph = xyGraph;
94  		setBackgroundColor(XYGraphMediaFactory.getInstance().getColor(255, 255,
95  				255));
96  		setForegroundColor(XYGraphMediaFactory.getInstance().getColor(0, 0, 0));
97  		setOpaque(true);
98  		RGB backRGB = getBackgroundColor().getRGB();
99  		revertBackColor = XYGraphMediaFactory.getInstance().getColor(
100 				255 - backRGB.red, 255 - backRGB.green, 255 - backRGB.blue);
101 		PlotMouseListener zoomer = new PlotMouseListener();
102 		addMouseListener(zoomer);
103 		addMouseMotionListener(zoomer);
104 		grabbing = XYGraphMediaFactory.getCursor(CURSOR_TYPE.GRABBING);
105 		zoomType = ZoomType.NONE;
106 	}

203 					.removeDataProviderListener(annotation);
204 		if (result) {

214 	@Override
215 	protected void layout() {
216 		final Rectangle clientArea = getClientArea();
217 		for (Trace trace : traceList) {
218 			if (trace != null && trace.isVisible())
219 				// Shrink will make the trace has no intersection with axes,
220 				// which will make it only repaints the trace area.
221 				trace.setBounds(clientArea);// .getCopy().shrink(1, 1));
222 		}
223 		for (Grid grid : gridList) {
224 			if (grid != null && grid.isVisible())
225 				grid.setBounds(clientArea);
226 		}
227 
228 		for (Annotation annotation : annotationList) {
229 			if (annotation != null && annotation.isVisible())
230 				annotation.setBounds(clientArea);// .getCopy().shrink(1, 1));
231 		}
232 		super.layout();
233 	}

235 	@Override
236 	protected void paintClientArea(final Graphics graphics) {
237 		super.paintClientArea(graphics);
238 		if (showBorder) {
239 			graphics.setLineWidth(2);
240 			graphics.drawLine(bounds.x, bounds.y, bounds.x + bounds.width,
241 					bounds.y);
242 			graphics.drawLine(bounds.x + bounds.width, bounds.y, bounds.x
243 					+ bounds.width, bounds.y + bounds.height);
244 		}
245 		// Show the start/end cursor or the 'rubberband' of a zoom operation?
246 		if (armed && end != null && start != null) {
247 			switch (zoomType) {
248 			case RUBBERBAND_ZOOM:
249 			case HORIZONTAL_ZOOM:
250 			case VERTICAL_ZOOM:
251 				graphics.setLineStyle(SWTConstants.LINE_DOT);
252 				graphics.setLineWidth(1);
253 				graphics.setForegroundColor(revertBackColor);
254 				graphics.drawRectangle(start.x, start.y, end.x - start.x, end.y
255 						- start.y);
256 				break;
257 
258 			default:
259 				break;
260 			}
261 		}
262 	}

293 	/**
294 	 * Zoom 'in' or 'out' by a fixed factor
295 	 * 
296 	 * @param horizontally
297 	 *            along x axes?
298 	 * @param vertically
299 	 *            along y axes?
300 	 * @param factor
301 	 *            Zoom factor. Positive to zoom 'in', negative 'out'.
302 	 */
303 	private void zoomInOut(final boolean horizontally,
304 			final boolean vertically, final double factor) {
305 		if (horizontally)
306 			for (Axis axis : xyGraph.getXAxisList()) {
307 				final double center = axis.getPositionValue(start.x, false);
308 				axis.zoomInOut(center, factor);
309 			}
310 		if (vertically)
311 			for (Axis axis : xyGraph.getYAxisList()) {
312 				final double center = axis.getPositionValue(start.y, false);
313 				axis.zoomInOut(center, factor);
314 			}
315 	}

331 	/**
332 	 * Listener to mouse events, performs panning and some zooms Is very similar
333 	 * to the Axis.AxisMouseListener, but unclear how easy/useful it would be to
334 	 * base them on the same code.
335 	 */
336 	class PlotMouseListener extends MouseMotionListener.Stub implements
337 			MouseListener {
338 		final private List<Range> xAxisStartRangeList = new ArrayList<Range>();
339 		final private List<Range> yAxisStartRangeList = new ArrayList<Range>();
340 
341 		private SaveStateCommand command;
342 
343 		public void mousePressed(final MouseEvent me) {
344 			// Only react to 'main' mouse button, only react to 'real' zoom
345 			if (me.button != 1 || zoomType == ZoomType.NONE)
346 				return;
347 			armed = true;
348 			// get start position
349 			switch (zoomType) {
350 			case RUBBERBAND_ZOOM:
351 				start = me.getLocation();
352 				end = null;
353 				break;
354 			case HORIZONTAL_ZOOM:
355 				start = new Point(me.getLocation().x, bounds.y);
356 				end = null;
357 				break;
358 			case VERTICAL_ZOOM:
359 				start = new Point(bounds.x, me.getLocation().y);
360 				end = null;
361 				break;
362 			case PANNING:
363 				setCursor(grabbing);
364 				start = me.getLocation();
365 				end = null;
366 				xAxisStartRangeList.clear();
367 				yAxisStartRangeList.clear();
368 				for (Axis axis : xyGraph.getXAxisList())
369 					xAxisStartRangeList.add(axis.getRange());
370 				for (Axis axis : xyGraph.getYAxisList())
371 					yAxisStartRangeList.add(axis.getRange());
372 				break;
373 			case ZOOM_IN:
374 			case ZOOM_IN_HORIZONTALLY:
375 			case ZOOM_IN_VERTICALLY:
376 			case ZOOM_OUT:
377 			case ZOOM_OUT_HORIZONTALLY:
378 			case ZOOM_OUT_VERTICALLY:
379 				start = me.getLocation();
380 				end = new Point();
381 				// Start timer that will zoom while mouse button is pressed
382 				Display.getCurrent().timerExec(Axis.ZOOM_SPEED, new Runnable() {
383 					public void run() {
384 						if (!armed)
385 							return;
386 						performInOutZoom();
387 						Display.getCurrent().timerExec(Axis.ZOOM_SPEED, this);
388 					}
389 				});
390 				break;
391 			default:
392 				break;
393 			}
394 
395 			// add command for undo operation
396 			command = new ZoomCommand(zoomType.getDescription(),
397 					xyGraph.getXAxisList(), xyGraph.getYAxisList());
398 			me.consume();
399 		}

404 		@Override
405 		public void mouseDragged(final MouseEvent me) {
406 			if (!armed)
407 				return;
408 			switch (zoomType) {
409 			case RUBBERBAND_ZOOM:
410 				end = me.getLocation();
411 				break;
412 			case HORIZONTAL_ZOOM:
413 				if (getCursor() != zoomType.getCursor()) {
414 					setCursor(zoomType.getCursor());  // scouter.porject 20150902
415 				}
416 				end = new Point(me.getLocation().x, bounds.y + bounds.height);
417 				break;
418 			case VERTICAL_ZOOM:
419 				end = new Point(bounds.x + bounds.width, me.getLocation().y);
420 				break;
421 			case PANNING:
422 				end = me.getLocation();
423 				pan();
424 				break;
425 			default:
426 				break;
427 			}
428 			PlotArea.this.repaint();
429 		}
430 
431 		@Override
432 		public void mouseExited(final MouseEvent me) {
433 			// Treat like releasing the button to stop zoomIn/Out timer
434 			switch (zoomType) {
435 			case ZOOM_IN:
436 			case ZOOM_IN_HORIZONTALLY:
437 			case ZOOM_IN_VERTICALLY:
438 			case ZOOM_OUT:
439 			case ZOOM_OUT_HORIZONTALLY:
440 			case ZOOM_OUT_VERTICALLY:
441 				mouseReleased(me);
442 			default:
443 			}
444 		}
445 
446 		public void mouseReleased(final MouseEvent me) {
447 			if (!armed)
448 				return;
449 			armed = false;
450 			if (zoomType == ZoomType.PANNING)
451 				setCursor(zoomType.getCursor());
452 			if (end == null || start == null)
453 				return;
454 
455 			switch (zoomType) {
456 			case RUBBERBAND_ZOOM:
457 				for (Axis axis : xyGraph.getXAxisList()) {
458 					final double t1 = axis.getPositionValue(start.x, false);
459 					final double t2 = axis.getPositionValue(end.x, false);
460 					axis.setRange(t1, t2, true);
461 				}
462 				for (Axis axis : xyGraph.getYAxisList()) {
463 					final double t1 = axis.getPositionValue(start.y, false);
464 					final double t2 = axis.getPositionValue(end.y, false);
465 					axis.setRange(t1, t2, true);
466 				}
467 				break;
468 			case HORIZONTAL_ZOOM:
469 				setCursor(ZoomType.NONE.getCursor()); // scouter.porject 20150902
470 				for (Axis axis : xyGraph.getXAxisList()) {
471 					final double t1 = axis.getPositionValue(start.x, false);
472 					final double t2 = axis.getPositionValue(end.x, false);
473 
474 					// scouter.porject 20150707
475 					changeSupport.firePropertyChange("horizontal_range", axis.getRange(), new Range(t1, t2));
476 					if (horizontalZoomEnabled) {
477 						axis.setRange(t1, t2, true);
478 					}
479 				}
480 				break;
481 			case VERTICAL_ZOOM:
482 				for (Axis axis : xyGraph.getYAxisList()) {
483 					final double t1 = axis.getPositionValue(start.y, false);
484 					final double t2 = axis.getPositionValue(end.y, false);
485 					axis.setRange(t1, t2, true);
486 				}
487 				break;
488 			case PANNING:
489 				pan();
490 				break;
491 			case ZOOM_IN:
492 			case ZOOM_IN_HORIZONTALLY:
493 			case ZOOM_IN_VERTICALLY:
494 			case ZOOM_OUT:
495 			case ZOOM_OUT_HORIZONTALLY:
496 			case ZOOM_OUT_VERTICALLY:
497 				performInOutZoom();
498 				break;
499 			default:
500 				break;
501 			}
502 
503 			if (zoomType != ZoomType.NONE && command != null) {
504 				command.saveState();
505 				xyGraph.getOperationsManager().addCommand(command);
506 				command = null;
507 			}
508 			start = null;
509 			end = null;
510 			PlotArea.this.repaint();
511 		}
512 
513 		/** Pan axis according to start/end from mouse listener */
514 		private void pan() {
515 			List<Axis> axes = xyGraph.getXAxisList();
516 			for (int i = 0; i < axes.size(); ++i) {
517 				final Axis axis = axes.get(i);
518 				axis.pan(xAxisStartRangeList.get(i),
519 						axis.getPositionValue(start.x, false),
520 						axis.getPositionValue(end.x, false));
521 			}
522 			axes = xyGraph.getYAxisList();
523 			for (int i = 0; i < axes.size(); ++i) {
524 				final Axis axis = axes.get(i);
525 				axis.pan(yAxisStartRangeList.get(i),
526 						axis.getPositionValue(start.y, false),
527 						axis.getPositionValue(end.y, false));
528 			}
529 		}
530 
531 		/** Perform the in or out zoom according to zoomType */
532 		private void performInOutZoom() {
533 			switch (zoomType) {
534 			case ZOOM_IN:
535 				zoomInOut(true, true, Axis.ZOOM_RATIO);
536 				break;
537 			case ZOOM_IN_HORIZONTALLY:
538 				zoomInOut(true, false, Axis.ZOOM_RATIO);
539 				break;
540 			case ZOOM_IN_VERTICALLY:
541 				zoomInOut(false, true, Axis.ZOOM_RATIO);
542 				break;
543 			case ZOOM_OUT:
544 				zoomInOut(true, true, -Axis.ZOOM_RATIO);
545 				break;
546 			case ZOOM_OUT_HORIZONTALLY:
547 				zoomInOut(true, false, -Axis.ZOOM_RATIO);
548 				break;
549 			case ZOOM_OUT_VERTICALLY:
550 				zoomInOut(false, true, -Axis.ZOOM_RATIO);
551 				break;
552 			default: // NOP
553 			}
554 		}
