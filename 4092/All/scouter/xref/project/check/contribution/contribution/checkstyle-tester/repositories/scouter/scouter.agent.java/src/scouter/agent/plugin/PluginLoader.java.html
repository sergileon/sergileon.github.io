37  	private static PluginLoader instance;
38  	public synchronized static PluginLoader getInstance() {
39  		if (instance == null) {
40  			instance = new PluginLoader();
41  			instance.setDaemon(true);
42  			instance.setName(ThreadUtil.getName(PluginLoader.class));
43  			instance.start();
44  		}
45  		return instance;
46  	}
47  	public void run() {
48  		while (true) {
49  			try {
50  				File root = Configure.getInstance().plugin_dir;
51  				reloadIfModified(root);
52  			} catch (Throwable t) {
53  				Logger.println("A160", t.toString());
54  			}
55  			ThreadUtil.sleep(5000);
56  		}
57  	}
58  	private void reloadIfModified(File root) {
59  		File script = new File(root, "service.plug");
60  		if (script.canRead() == false) {
61  			PluginAppServiceTrace.plugIn = null;
62  		} else {
63  			if (PluginAppServiceTrace.plugIn == null
64  					|| PluginAppServiceTrace.plugIn.lastModified != script.lastModified()) {
65  				PluginAppServiceTrace.plugIn = createAppService(script);
66  			}
67  		}
68  		script = new File(root, "httpservice.plug");
69  		if (script.canRead() == false) {
70  			PluginHttpServiceTrace.plugIn = null;
71  		} else {
72  			if (PluginHttpServiceTrace.plugIn == null
73  					|| PluginHttpServiceTrace.plugIn.lastModified != script.lastModified()) {
74  				PluginHttpServiceTrace.plugIn = createHttpService(script);
75  			}
76  		}
77  		script = new File(root, "capture.plug");
78  		if (script.canRead() == false) {
79  			PluginCaptureTrace.plugIn = null;
80  		} else {
81  			if (PluginCaptureTrace.plugIn == null || PluginCaptureTrace.plugIn.lastModified != script.lastModified()) {
82  				PluginCaptureTrace.plugIn = createICaptureTrace(script);
83  			}
84  		}
85  		script = new File(root, "jdbcpool.plug");
86  		if (script.canRead() == false) {
87  			PluginJdbcPoolTrace.plugIn = null;
88  		} else {
89  			if (PluginJdbcPoolTrace.plugIn == null
90  					|| PluginJdbcPoolTrace.plugIn.lastModified != script.lastModified()) {
91  				PluginJdbcPoolTrace.plugIn = createIJdbcPool(script);
92  				if (PluginJdbcPoolTrace.plugIn != null) {
93  					TraceSQL.clearUrlMap();
94  				}
95  			}
96  		}
97  		script = new File(root, "httpcall.plug");
98  		if (script.canRead() == false) {
99  			PluginHttpCallTrace.plugIn = null;
100 		} else {
101 			if (PluginHttpCallTrace.plugIn == null
102 					|| PluginHttpCallTrace.plugIn.lastModified != script.lastModified()) {
103 				PluginHttpCallTrace.plugIn = createIHttpCall(script);
104 			}
105 		}
106 	}

108 	private AbstractHttpService createHttpService(File script) {
109 		if (IHttpServiceCompile == script.lastModified())
110 			return null;
111 		IHttpServiceCompile = script.lastModified();
112 		try {
113 			HashMap<String, StringBuffer> bodyTable = loadFileText(script);
114 			String superName = AbstractHttpService.class.getName();
115 			String className = "scouter.agent.plugin.impl.HttpServiceImpl";
116 			String METHOD_START = "start";
117 			String METHOD_END = "end";
118 			String METHOD_REJECT = "reject";
119 			String SIGNATURE = nativeName(WrContext.class) + nativeName(WrRequest.class) + nativeName(WrResponse.class);
120 			String METHOD_P1 = WrContext.class.getName();
121 			String METHOD_P2 = WrRequest.class.getName();
122 			String METHOD_P3 = WrResponse.class.getName();
123 			if (bodyTable.containsKey(METHOD_START) == false)
124 				throw new CannotCompileException("no method body: " + METHOD_START);
125 			if (bodyTable.containsKey(METHOD_END) == false)
126 				throw new CannotCompileException("no method body: " + METHOD_END);
127 			if (bodyTable.containsKey(METHOD_REJECT) == false)
128 				throw new CannotCompileException("no method body: " + METHOD_REJECT);
129 			ClassPool cp = ClassPool.getDefault();
130 			String jar = FileUtil.getJarFileName(PluginLoader.class);
131 			if (jar != null) {
132 				cp.appendClassPath(jar);
133 			}
134 			CtClass cc = cp.get(superName);
135 			CtClass impl = null;
136 			CtMethod method_start = null;
137 			CtMethod method_end = null;
138 			CtMethod method_reject = null;
139 			StringBuffer sb = null;
140 			try {
141 				impl = cp.get(className);
142 				impl.defrost();
143 				// START METHOD
144 				method_start = impl.getMethod(METHOD_START, "(" + SIGNATURE + ")V");
145 				// END METHOD
146 				method_end = impl.getMethod(METHOD_END, "(" + SIGNATURE + ")V");
147 				// REJECT METHOD
148 				method_reject = impl.getMethod(METHOD_REJECT, "(" + SIGNATURE + ")Z");
149 			} catch (scouter.javassist.NotFoundException e) {
150 				impl = cp.makeClass(className, cc);
151 				StringBuffer sb1 = new StringBuffer();
152 				sb1.append(METHOD_P1).append(" p1").append(",");
153 				sb1.append(METHOD_P2).append(" p2").append(",");
154 				sb1.append(METHOD_P3).append(" p3");
155 				// START METHOD
156 				sb = new StringBuffer();
157 				sb.append("public void ").append(METHOD_START).append("(").append(sb1).append("){}");
158 				method_start = CtNewMethod.make(sb.toString(), impl);
159 				impl.addMethod(method_start);
160 				// END METHOD
161 				sb = new StringBuffer();
162 				sb.append("public void ").append(METHOD_END).append("(").append(sb1).append("){}");
163 				method_end = CtNewMethod.make(sb.toString(), impl);
164 				impl.addMethod(method_end);
165 				// REJECT METHOD
166 				sb = new StringBuffer();
167 				sb.append("public boolean ").append(METHOD_REJECT).append("(").append(sb1).append("){return false;}");
168 				method_reject = CtNewMethod.make(sb.toString(), impl);
169 				impl.addMethod(method_reject);
170 			}
171 			StringBuffer bodyPrefix = new StringBuffer();
172 			bodyPrefix.append("{");
173 			bodyPrefix.append(METHOD_P1).append(" $ctx=$1;");
174 			bodyPrefix.append(METHOD_P2).append(" $req=$2;");
175 			bodyPrefix.append(METHOD_P3).append(" $res=$3;");
176 			method_start.setBody(
177 					new StringBuffer().append(bodyPrefix).append(bodyTable.get(METHOD_START)).append("\n}").toString());
178 			method_end.setBody(
179 					new StringBuffer().append(bodyPrefix).append(bodyTable.get(METHOD_END)).append("\n}").toString());
180 			method_reject.setBody(
181 					new StringBuffer().append(bodyPrefix).append(bodyTable.get(METHOD_REJECT)).append("\n}").toString());
182 			Class c = impl.toClass(new URLClassLoader(new URL[0], this.getClass().getClassLoader()), null);
183 			AbstractHttpService plugin = (AbstractHttpService) c.newInstance();
184 			plugin.lastModified = script.lastModified();
185 			Logger.println("PLUG-IN : " + AbstractHttpService.class.getName() + " loaded #"
186 					+ Hexa32.toString32(plugin.hashCode()));
187 			return plugin;
188 		} catch (scouter.javassist.CannotCompileException ee) {
189 			Logger.println(ee.getMessage());
190 		} catch (Throwable e) {
191 			Logger.println("A161", e);
192 		}
193 		return null;
194 	}
195 	private HashMap<String, StringBuffer> loadFileText(File script) {
196 		StringBuffer sb = new StringBuffer();
197 		HashMap<String, StringBuffer> result = new HashMap<String, StringBuffer>();
198 		String txt = new String(FileUtil.readAll(script));
199 		try {
200 			BufferedReader r = new BufferedReader(new StringReader(txt));
201 			while (true) {
202 				String line = StringUtil.trim(r.readLine());
203 				if (line == null)
204 					break;
205 				if (line.startsWith("[") && line.endsWith("]")) {
206 					sb = new StringBuffer();
207 					result.put(line.substring(1, line.length() - 1), sb);
208 				} else {
209 					sb.append(line).append("\n");
210 				}
211 			}
212 		} catch (Exception e) {
213 			e.printStackTrace();
214 		}
215 		return result;
216 	}
