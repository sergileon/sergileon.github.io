30  /**
31   * Simple utility class for printing the bytecode instructions of a method.
32   *
33   * @author Jason T. Greene
34   */
35  public class InstructionPrinter implements Opcode {
36  
37      private final static String opcodes[] = Mnemonic.OPCODE;
38      private final PrintStream stream;
39  
40      /**
41       * Constructs a <code>InstructionPrinter</code> object.
42       */
43      public InstructionPrinter(PrintStream stream) {
44          this.stream = stream;
45      }
46  
47      /**
48       * Prints the bytecode instructions of a given method.
49       */
50      public static void print(CtMethod method, PrintStream stream) {
51          (new InstructionPrinter(stream)).print(method);
52      }
53  
54      /**
55       * Prints the bytecode instructions of a given method.
56       */
57      public void print(CtMethod method) {
58          MethodInfo info = method.getMethodInfo2();
59          ConstPool pool = info.getConstPool();
60          CodeAttribute code = info.getCodeAttribute();
61          if (code == null)
62              return;
63  
64          CodeIterator iterator = code.iterator();
65          while (iterator.hasNext()) {
66              int pos;
67              try {
68                  pos = iterator.next();
69              } catch (BadBytecode e) {
70                  throw new RuntimeException(e);
71              }
72  
73              stream.println(pos + ": " + instructionString(iterator, pos, pool));
74          }
75      }
76  
77      /**
78       * Gets a string representation of the bytecode instruction at the specified
79       * position. 
80       */
81      public static String instructionString(CodeIterator iter, int pos, ConstPool pool) {
82          int opcode = iter.byteAt(pos);
83  
84          if (opcode > opcodes.length || opcode < 0)
85              throw new IllegalArgumentException("Invalid opcode, opcode: " + opcode + " pos: "+ pos);
86  
87          String opstring = opcodes[opcode];
88          switch (opcode) {
89              case BIPUSH:
90                  return opstring + " " + iter.byteAt(pos + 1);
91              case SIPUSH:
92                  return opstring + " " + iter.s16bitAt(pos + 1);
93              case LDC:
94                  return opstring + " " + ldc(pool, iter.byteAt(pos + 1));
95              case LDC_W :
96              case LDC2_W :
97                  return opstring + " " + ldc(pool, iter.u16bitAt(pos + 1));
98              case ILOAD:
99              case LLOAD:
100             case FLOAD:
101             case DLOAD:
102             case ALOAD:
103             case ISTORE:
104             case LSTORE:
105             case FSTORE:
106             case DSTORE:
107             case ASTORE:
108                 return opstring + " " + iter.byteAt(pos + 1);
109             case IFEQ:
110             case IFGE:
111             case IFGT:
112             case IFLE:
113             case IFLT:
114             case IFNE:
115             case IFNONNULL:
116             case IFNULL:
117             case IF_ACMPEQ:
118             case IF_ACMPNE:
119             case IF_ICMPEQ:
120             case IF_ICMPGE:
121             case IF_ICMPGT:
122             case IF_ICMPLE:
123             case IF_ICMPLT:
124             case IF_ICMPNE:
125                 return opstring + " " + (iter.s16bitAt(pos + 1) + pos);
126             case IINC:
127                 return opstring + " " + iter.byteAt(pos + 1) + ", " + iter.signedByteAt(pos + 2);
128             case GOTO:
129             case JSR:
130                 return opstring + " " + (iter.s16bitAt(pos + 1) + pos);
131             case RET:
132                 return opstring + " " + iter.byteAt(pos + 1);
133             case TABLESWITCH:
134                 return tableSwitch(iter, pos);
135             case LOOKUPSWITCH:
136                 return lookupSwitch(iter, pos);
137             case GETSTATIC:
138             case PUTSTATIC:
139             case GETFIELD:
140             case PUTFIELD:
141                 return opstring + " " + fieldInfo(pool, iter.u16bitAt(pos + 1));
142             case INVOKEVIRTUAL:
143             case INVOKESPECIAL:
144             case INVOKESTATIC:
145                 return opstring + " " + methodInfo(pool, iter.u16bitAt(pos + 1));
146             case INVOKEINTERFACE:
147                 return opstring + " " + interfaceMethodInfo(pool, iter.u16bitAt(pos + 1));
148             case INVOKEDYNAMIC:
149                 return opstring + " " + iter.u16bitAt(pos + 1);
150             case NEW:
151                 return opstring + " " + classInfo(pool, iter.u16bitAt(pos + 1));
152             case NEWARRAY:
153                 return opstring + " " + arrayInfo(iter.byteAt(pos + 1));
154             case ANEWARRAY:
155             case CHECKCAST:
156                 return opstring + " " + classInfo(pool, iter.u16bitAt(pos + 1));
157             case WIDE:
158                 return wide(iter, pos);
159             case MULTIANEWARRAY:
160                 return opstring + " " + classInfo(pool, iter.u16bitAt(pos + 1));
161             case GOTO_W:
162             case JSR_W:
163                 return opstring + " " + (iter.s32bitAt(pos + 1)+ pos);
164             default:
165                 return opstring;
166         }
167     }

170     private static String wide(CodeIterator iter, int pos) {
171         int opcode = iter.byteAt(pos + 1);
172         int index = iter.u16bitAt(pos + 2);
173         switch (opcode) {
174             case ILOAD:
175             case LLOAD:
176             case FLOAD:
177             case DLOAD:
178             case ALOAD:
179             case ISTORE:
180             case LSTORE:
181             case FSTORE:
182             case DSTORE:
183             case ASTORE:
184             case IINC:
185             case RET:
186                 return opcodes[opcode] + " " + index;
187             default:
188                 throw new RuntimeException("Invalid WIDE operand");
189         }
190     }

193     private static String arrayInfo(int type) {
194         switch (type) {
195             case T_BOOLEAN:
196                 return "boolean";
197             case T_CHAR:
198                 return "char";
199             case T_BYTE:
200                 return "byte";
201             case T_SHORT:
202                 return "short";
203             case T_INT:
204                 return "int";
205             case T_LONG:
206                 return "long";
207             case T_FLOAT:
208                 return "float";
209             case T_DOUBLE:
210                 return "double";
211             default:
212                 throw new RuntimeException("Invalid array type");
213         }
214     }

217     private static String classInfo(ConstPool pool, int index) {
218         return "#" + index + " = Class " + pool.getClassInfo(index);
219     }

222     private static String interfaceMethodInfo(ConstPool pool, int index) {
223         return "#" + index + " = Method "
224                 + pool.getInterfaceMethodrefClassName(index) + "."
225                 + pool.getInterfaceMethodrefName(index) + "("
226                 + pool.getInterfaceMethodrefType(index) + ")";
227     }

229     private static String methodInfo(ConstPool pool, int index) {
230         return "#" + index + " = Method "
231                 + pool.getMethodrefClassName(index) + "."
232                 + pool.getMethodrefName(index) + "("
233                 + pool.getMethodrefType(index) + ")";
234     }

237     private static String fieldInfo(ConstPool pool, int index) {
238         return "#" + index + " = Field "
239             + pool.getFieldrefClassName(index) + "."
240             + pool.getFieldrefName(index) + "("
241             + pool.getFieldrefType(index) + ")";
242     }

245     private static String lookupSwitch(CodeIterator iter, int pos) {
246         StringBuffer buffer = new StringBuffer("lookupswitch {\n");
247         int index = (pos & ~3) + 4;
248         // default
249         buffer.append("\t\tdefault: ").append(pos + iter.s32bitAt(index)).append("\n");
250         int npairs = iter.s32bitAt(index += 4);
251         int end = npairs * 8 + (index += 4);
252 
253         for (; index < end; index += 8) {
254             int match = iter.s32bitAt(index);
255             int target = iter.s32bitAt(index + 4) + pos;
256             buffer.append("\t\t").append(match).append(": ").append(target).append("\n");
257         }
258 
259         buffer.setCharAt(buffer.length() - 1, '}');
260         return buffer.toString();
261     }
262 
263 
264     private static String tableSwitch(CodeIterator iter, int pos) {
265         StringBuffer buffer = new StringBuffer("tableswitch {\n");
266         int index = (pos & ~3) + 4;
267         // default
268         buffer.append("\t\tdefault: ").append(pos + iter.s32bitAt(index)).append("\n");
269         int low = iter.s32bitAt(index += 4);
270         int high = iter.s32bitAt(index += 4);
271         int end = (high - low + 1) * 4 + (index += 4);
272 
273         // Offset table
274         for (int key = low; index < end; index += 4, key++) {
275             int target = iter.s32bitAt(index) + pos;
276             buffer.append("\t\t").append(key).append(": ").append(target).append("\n");
277         }
278 
279         buffer.setCharAt(buffer.length() - 1, '}');
280         return buffer.toString();
281     }

284     private static String ldc(ConstPool pool, int index) {
285         int tag = pool.getTag(index);
286         switch (tag) {
287             case ConstPool.CONST_String:
288                 return "#" + index + " = \"" + pool.getStringInfo(index) + "\"";
289             case ConstPool.CONST_Integer:
290                 return "#" + index + " = int " + pool.getIntegerInfo(index);
291             case ConstPool.CONST_Float:
292                 return "#" + index + " = float " + pool.getFloatInfo(index);
293             case ConstPool.CONST_Long:
294                 return "#" + index + " = long " + pool.getLongInfo(index);
295             case ConstPool.CONST_Double:
296                 return "#" + index + " = int " + pool.getDoubleInfo(index);
297             case ConstPool.CONST_Class:
298                 return classInfo(pool, index);
299             default:
300                 throw new RuntimeException("bad LDC: " + tag);
301         }
302     }
