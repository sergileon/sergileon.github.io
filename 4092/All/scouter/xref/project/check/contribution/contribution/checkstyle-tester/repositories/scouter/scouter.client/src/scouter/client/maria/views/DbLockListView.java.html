66  	public void init(IViewSite site) throws PartInitException {
67  		super.init(site);
68  		String secId = site.getSecondaryId();
69  		String[] ids = secId.split("&");
70  		serverId = CastUtil.cint(ids[0]);
71  		objHash = CastUtil.cint(ids[1]);
72  	}

74  	@Override
75  	public void createPartControl(Composite parent) {
76  		this.setPartName("Lock List[" + TextProxy.object.getText(objHash) + "]");
77  		columnLayout = new TreeColumnLayout();
78  		parent.setLayout(columnLayout);
79  		tree = new Tree(parent, SWT.FULL_SELECTION | SWT.BORDER | SWT.H_SCROLL | SWT.V_SCROLL);
80  		tree.setHeaderVisible(true);
81  		tree.setLinesVisible(true);
82  		viewer = new TreeViewer(tree);
83  		createColumns();
84  		viewer.setContentProvider(new LockContentProvider());
85  		viewer.setLabelProvider(new LockLabelProvider());
86  		viewer.setInput(root);
87  	}

89  	public void setInput(long time){
90  		this.time = time;
91  		load();
92  	}

94  	private void load() {
95  		root.clear();
96  		tree.removeAll();
97  		ExUtil.asyncRun(new Runnable() {
98  			public void run() {
99  				TcpProxy tcpProxy = TcpProxy.getTcpProxy(serverId);
100 				Pack p = null;
101 				try {
102 					MapPack param = new MapPack();
103 					param.put("key", StatusConstants.LOCK_INFO);
104 					param.put("objHash", objHash);
105 					param.put("time", time);
106 					p = tcpProxy.getSingle(RequestCmd.STATUS_AROUND_VALUE, param);
107 				} catch (Throwable th){
108 					th.printStackTrace();
109 				} finally {
110 					TcpProxy.putTcpProxy(tcpProxy);
111 				}
112 				if (p != null) {
113 					String date = DateUtil.yyyymmdd(TimeUtil.getCurrentTime(serverId));
114 					StatusPack sp = (StatusPack) p;
115 					ListValue waitStartedLv = sp.data.getList("WAIT_STARTED");
116 					ListValue waitingPidLv = sp.data.getList("WAITING_PID");
117 					ListValue lockedTableLv = sp.data.getList("LOCKED_TABLE");
118 					ListValue lockedIndexLv = sp.data.getList("LOCKED_INDEX");
119 					ListValue lockedTypeLv = sp.data.getList("LOCKED_TYPE");
120 					ListValue blockingPidLv = sp.data.getList("BLOCKING_PID");
121 					ListValue waitingQueryLv = sp.data.getList("WAITING_QUERY");
122 					ListValue waitingLockModeLv = sp.data.getList("WAITING_LOCK_MODE");
123 					ListValue blockingQueryLv = sp.data.getList("BLOCKING_QUERY");
124 					ListValue blockingTrxStartedLv = sp.data.getList("BLOCKING_TRX_STARTED");
125 					ListValue blockingLockModeLv = sp.data.getList("BLOCKING_LOCK_MODE");
126 					for (int i = 0; i < waitStartedLv.size(); i++) {
127 						long blockPid = blockingPidLv.getLong(i);
128 						LockObject blockObject = root.get(blockPid);
129 						if (blockObject == null) {
130 							blockObject = new LockObject(blockPid);
131 							blockObject.startTime = blockingTrxStartedLv.getLong(i);
132 							blockObject.sql = TextProxy.maria.getLoadText(date, blockingQueryLv.getInt(i), serverId);
133 							blockObject.type = lockedTypeLv.getString(i);
134 							blockObject.waitingLockMode = waitingLockModeLv.getString(i);
135 							blockObject.blockingLockMode = blockingLockModeLv.getString(i);
136 							blockObject.index = lockedIndexLv.getString(i);
137 							blockObject.table = lockedTableLv.getString(i);
138 							root.put(blockPid, blockObject);
139 						}
140 						
141 						long waitPid = waitingPidLv.getLong(i);
142 						LockObject waitObject = new LockObject(waitPid);
143 						waitObject.startTime = waitStartedLv.getLong(i);
144 						waitObject.sql = TextProxy.maria.getLoadText(date, waitingQueryLv.getInt(i), serverId);
145 						waitObject.type = lockedTypeLv.getString(i);
146 						waitObject.waitingLockMode = waitingLockModeLv.getString(i);
147 						waitObject.blockingLockMode = blockingLockModeLv.getString(i);
148 						waitObject.index = lockedIndexLv.getString(i);
149 						waitObject.table = lockedTableLv.getString(i);
150 						waitObject.parent = blockObject;
151 						blockObject.addChild(waitObject);
152 					}
153 				}
154 				ExUtil.exec(tree, new Runnable() {
155 					public void run() {
156 						viewer.refresh();
157 						viewer.expandAll();
158 					}
159 				});
160 			}
161 		});
162 	}

168 	static class LockObject {
169 		long pid;
170 		long startTime;
171 		String sql;
172 		String type;
173 		String waitingLockMode;
174 		String blockingLockMode;
175 		String index;
176 		String table;
177 		LockObject parent;
178 		List<LockObject> childList;
179 		
180 		LockObject(long id) {
181 			this.pid = id;
182 		}
183 		
184 		public void addChild(LockObject child) {
185 			if (childList == null) {
186 				childList = new ArrayList<LockObject>();
187 			}
188 			childList.add(child);
189 		}
190 	}

214 			return null;
215 		}

222 		}

225 			if (element instanceof LockObject) {
226 				return ((LockObject)element).childList != null;
227 			}
228 			return false;
229 		}

233 	class LockLabelProvider implements ITableLabelProvider {
234 
235 		public void addListener(ILabelProviderListener listener) {
236 		}
237 
238 		public void dispose() {
239 		}
240 
241 		public boolean isLabelProperty(Object element, String property) {
242 			return false;
243 		}
244 
245 		public void removeListener(ILabelProviderListener listener) {
246 		}
247 
248 		public Image getColumnImage(Object element, int columnIndex) {
249 			return null;
250 		}
251 
252 		public String getColumnText(Object element, int columnIndex) {
253 			if (element instanceof LockObject) {
254 				LockObject model = (LockObject) element;
255 				LockTableSchema column = columnList.get(columnIndex);
256 				switch (column) {
257 					case PID:
258 						return CastUtil.cString(model.pid);
259 					case START:
260 						return DateUtil.format(model.startTime, "HH:mm:ss");
261 					case SQL:
262 						return model.sql;
263 					case TYPE:
264 						return model.type;
265 					case MODE:
266 						if (model.parent == null) {
267 							return model.blockingLockMode;
268 						} else {
269 							return model.waitingLockMode;
270 						}
271 					case INDEX:
272 						return model.index;
273 					case TABLE:
274 						return model.table;
275 				}
276 			}
277 			return null;
278 		}
279 		
280 	}

282 	ArrayList<LockTableSchema> columnList = new ArrayList<LockTableSchema>();
283 
284 	private void createColumns() {
285 		columnList.clear();
286 		for (LockTableSchema column : LockTableSchema.values()) {
287 			createTreeViewerColumn(column.getTitle(), column.getWidth(), column.getAlignment(), true, true, column.isNumber());
288 			columnList.add(column);
289 		}
290 	}
291 	
292 	private TreeViewerColumn createTreeViewerColumn(String title, int width, int alignment,  boolean resizable, boolean moveable, final boolean isNumber) {
293 		final TreeViewerColumn viewerColumn = new TreeViewerColumn(viewer, SWT.NONE);
294 		final TreeColumn column = viewerColumn.getColumn();
295 		column.setText(title);
296 		column.setAlignment(alignment);
297 		column.setMoveable(moveable);
298 		columnLayout.setColumnData(column, new ColumnPixelData(width, resizable));
299 		return viewerColumn;
300 	}

302 	enum LockTableSchema {
303 		PID("PID", 100, SWT.LEFT, true),
304 	    START("Start", 100, SWT.CENTER, false),
305 	    SQL("SQL", 200, SWT.LEFT, false),
306 	    TYPE("Type", 100, SWT.LEFT, false),
307 	    MODE("Mode", 100, SWT.LEFT, false),
308 	    INDEX("Index", 100, SWT.LEFT, true),
309 	    TABLE("Table", 100, SWT.LEFT,  false);
310 
311 	    private final String title;
312 	    private final int width;
313 	    private final int alignment;
314 	    private final boolean isNumber;
315 
316 	    private LockTableSchema(String text, int width, int alignment, boolean isNumber) {
317 	        this.title = text;
318 	        this.width = width;
319 	        this.alignment = alignment;
320 	        this.isNumber = isNumber;
321 	    }
322

324 	        return title;
325 	    }
326 
327 	    public int getAlignment(){
328 	        return alignment;
329 	    }
330 
331 		public int getWidth() {
332 			return width;
333 		}
334 		
335 		public boolean isNumber() {
336 			return this.isNumber;
337 		}
