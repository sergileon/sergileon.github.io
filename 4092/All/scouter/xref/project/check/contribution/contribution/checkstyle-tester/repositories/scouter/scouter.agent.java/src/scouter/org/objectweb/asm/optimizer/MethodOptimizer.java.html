30  package scouter.org.objectweb.asm.optimizer;
31  
32  import scouter.org.objectweb.asm.AnnotationVisitor;
33  import scouter.org.objectweb.asm.Attribute;
34  import scouter.org.objectweb.asm.FieldVisitor;
35  import scouter.org.objectweb.asm.Label;
36  import scouter.org.objectweb.asm.MethodVisitor;
37  import scouter.org.objectweb.asm.Opcodes;
38  import scouter.org.objectweb.asm.Type;
39  import scouter.org.objectweb.asm.TypePath;
40  import scouter.org.objectweb.asm.commons.Remapper;
41  import scouter.org.objectweb.asm.commons.RemappingMethodAdapter;
42  
43  /**
44   * A {@link MethodVisitor} that renames fields and methods, and removes debug
45   * info.
46   * 
47   * @author Eugene Kuleshov
48   */
49  public class MethodOptimizer extends RemappingMethodAdapter implements Opcodes {
50  
51      private final ClassOptimizer classOptimizer;
52  
53      public MethodOptimizer(ClassOptimizer classOptimizer, int access,
54              String desc, MethodVisitor mv, Remapper remapper) {
55          super(Opcodes.ASM5, access, desc, mv, remapper);
56          this.classOptimizer = classOptimizer;
57      }
58  
59      // ------------------------------------------------------------------------
60      // Overridden methods
61      // ------------------------------------------------------------------------
62  
63      @Override
64      public void visitParameter(String name, int access) {
65          // remove parameter info
66      }
67  
68      @Override
69      public AnnotationVisitor visitAnnotationDefault() {
70          // remove annotations
71          return null;
72      }
73  
74      @Override
75      public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
76          // remove annotations
77          return null;
78      }
79  
80      @Override
81      public AnnotationVisitor visitTypeAnnotation(int typeRef,
82              TypePath typePath, String desc, boolean visible) {
83          return null;
84      }
85  
86      @Override
87      public AnnotationVisitor visitParameterAnnotation(final int parameter,
88              final String desc, final boolean visible) {
89          // remove annotations
90          return null;
91      }
92  
93      @Override
94      public void visitLocalVariable(final String name, final String desc,
95              final String signature, final Label start, final Label end,
96              final int index) {
97          // remove debug info
98      }
99  
100     @Override
101     public void visitLineNumber(final int line, final Label start) {
102         // remove debug info
103     }
104 
105     @Override
106     public void visitFrame(int type, int local, Object[] local2, int stack,
107             Object[] stack2) {
108         // remove frame info
109     }
110 
111     @Override
112     public void visitAttribute(Attribute attr) {
113         // remove non standard attributes
114     }
115 
116     @Override
117     public void visitLdcInsn(Object cst) {
118         if (!(cst instanceof Type)) {
119             super.visitLdcInsn(cst);
120             return;
121         }
122 
123         // transform Foo.class for 1.2 compatibility
124         String ldcName = ((Type) cst).getInternalName();
125         String fieldName = "class$" + ldcName.replace('/', '$');
126         if (!classOptimizer.syntheticClassFields.contains(ldcName)) {
127             classOptimizer.syntheticClassFields.add(ldcName);
128             FieldVisitor fv = classOptimizer.syntheticFieldVisitor(ACC_STATIC
129                     | ACC_SYNTHETIC, fieldName, "Ljava/lang/Class;");
130             fv.visitEnd();
131         }
132 
133         String clsName = classOptimizer.clsName;
134         mv.visitFieldInsn(GETSTATIC, clsName, fieldName, "Ljava/lang/Class;");
135     }
136 }
