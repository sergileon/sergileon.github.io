59      public static class Method {
60          public CtClass declaring;
61          public MethodInfo info;
62          public int notmatch;
63  
64          public Method(CtClass c, MethodInfo i, int n) {
65              declaring = c;
66              info = i;
67              notmatch = n;
68          }
69  
70          /**
71           * Returns true if the invoked method is static.
72           */
73          public boolean isStatic() {
74              int acc = info.getAccessFlags();
75              return (acc & AccessFlag.STATIC) != 0;
76          }
77      }

79      public Method lookupMethod(CtClass clazz, CtClass currentClass, MethodInfo current,
80                                  String methodName,
81                                  int[] argTypes, int[] argDims,
82                                  String[] argClassNames)
83          throws CompileError
84      {
85          Method maybe = null;
86          // to enable the creation of a recursively called method
87          if (current != null && clazz == currentClass)
88              if (current.getName().equals(methodName)) {
89                  int res = compareSignature(current.getDescriptor(),
90                                             argTypes, argDims, argClassNames);
91                  if (res != NO) {
92                      Method r = new Method(clazz, current, res);
93                      if (res == YES)
94                          return r;
95                      else
96                          maybe = r;
97                  }
98              }
99  
100         Method m = lookupMethod(clazz, methodName, argTypes, argDims,
101                                 argClassNames, maybe != null);
102         if (m != null)
103             return m;
104         else
105             return maybe;
106     }

108     private Method lookupMethod(CtClass clazz, String methodName,
109                                int[] argTypes, int[] argDims,
110                                String[] argClassNames, boolean onlyExact)
111         throws CompileError
112     {
113         Method maybe = null;
114         ClassFile cf = clazz.getClassFile2();
115         // If the class is an array type, the class file is null.
116         // If so, search the super class java.lang.Object for clone() etc.
117         if (cf != null) {
118             List list = cf.getMethods();
119             int n = list.size();
120             for (int i = 0; i < n; ++i) {
121                 MethodInfo minfo = (MethodInfo)list.get(i);
122                 if (minfo.getName().equals(methodName)) {
123                     int res = compareSignature(minfo.getDescriptor(),
124                                            argTypes, argDims, argClassNames);
125                     if (res != NO) {
126                         Method r = new Method(clazz, minfo, res);
127                         if (res == YES)
128                             return r;
129                         else if (maybe == null || maybe.notmatch > res)
130                             maybe = r;
131                     }
132                 }
133             }
134         }
135 
136         if (onlyExact)
137             maybe = null;
138         else
139             onlyExact = maybe != null;
140 
141         int mod = clazz.getModifiers();
142         boolean isIntf = Modifier.isInterface(mod);
143         try {
144             // skip searching java.lang.Object if clazz is an interface type.
145             if (!isIntf) {
146                 CtClass pclazz = clazz.getSuperclass();
147                 if (pclazz != null) {
148                     Method r = lookupMethod(pclazz, methodName, argTypes,
149                                             argDims, argClassNames, onlyExact);
150                     if (r != null)
151                         return r;
152                 }
153             }
154         }
155         catch (NotFoundException e) {}
156 
157         try {
158             CtClass[] ifs = clazz.getInterfaces();
159             int size = ifs.length;
160             for (int i = 0; i < size; ++i) {
161                 Method r = lookupMethod(ifs[i], methodName,
162                         argTypes, argDims, argClassNames,
163                         onlyExact);
164                 if (r != null)
165                     return r;
166             }
167 
168             if (isIntf) {
169                 // finally search java.lang.Object.
170                 CtClass pclazz = clazz.getSuperclass();
171                 if (pclazz != null) {
172                     Method r = lookupMethod(pclazz, methodName, argTypes,
173                                             argDims, argClassNames, onlyExact);
174                     if (r != null)
175                         return r;
176                 }
177             }
178         }
179         catch (NotFoundException e) {}
180 
181         return maybe;
182     }

184     private static final int YES = 0;
185     private static final int NO = -1;
186 
187     /*
188      * Returns YES if actual parameter types matches the given signature.
189      *
190      * argTypes, argDims, and argClassNames represent actual parameters.
191      *
192      * This method does not correctly implement the Java method dispatch
193      * algorithm.
194      *
195      * If some of the parameter types exactly match but others are subtypes of
196      * the corresponding type in the signature, this method returns the number
197      * of parameter types that do not exactly match.
198      */
199     private int compareSignature(String desc, int[] argTypes,
200                                  int[] argDims, String[] argClassNames)
201         throws CompileError
202     {
203         int result = YES;
204         int i = 1;
205         int nArgs = argTypes.length;
206         if (nArgs != Descriptor.numOfParameters(desc))
207             return NO;
208 
209         int len = desc.length();
210         for (int n = 0; i < len; ++n) {
211             char c = desc.charAt(i++);
212             if (c == ')')
213                 return (n == nArgs ? result : NO);
214             else if (n >= nArgs)
215                 return NO;
216 
217             int dim = 0;
218             while (c == '[') {
219                 ++dim;
220                 c = desc.charAt(i++);
221             }
222 
223             if (argTypes[n] == NULL) {
224                 if (dim == 0 && c != 'L')
225                     return NO;
226 
227                 if (c == 'L')
228                     i = desc.indexOf(';', i) + 1;
229             }
230             else if (argDims[n] != dim) {
231                 if (!(dim == 0 && c == 'L'
232                       && desc.startsWith("java/lang/Object;", i)))
233                     return NO;
234 
235                 // if the thread reaches here, c must be 'L'.
236                 i = desc.indexOf(';', i) + 1;
237                 result++;
238                 if (i <= 0)
239                     return NO;  // invalid descriptor?
240             }
241             else if (c == 'L') {        // not compare
242                 int j = desc.indexOf(';', i);
243                 if (j < 0 || argTypes[n] != CLASS)
244                     return NO;
245 
246                 String cname = desc.substring(i, j);
247                 if (!cname.equals(argClassNames[n])) {
248                     CtClass clazz = lookupClassByJvmName(argClassNames[n]);
249                     try {
250                         if (clazz.subtypeOf(lookupClassByJvmName(cname)))
251                             result++;
252                         else
253                             return NO;
254                     }
255                     catch (NotFoundException e) {
256                         result++; // should be NO?
257                     }
258                 }
259 
260                 i = j + 1;
261             }
262             else {
263                 int t = descToType(c);
264                 int at = argTypes[n];
265                 if (t != at)
266                     if (t == INT
267                         && (at == SHORT || at == BYTE || at == CHAR))
268                         result++;
269                     else
270                         return NO;
271             }
272         }
273 
274         return NO;
275     }

277     /**
278      * Only used by fieldAccess() in MemberCodeGen and TypeChecker.
279      *
280      * @param jvmClassName  a JVM class name.  e.g. java/lang/String
281      * @see #lookupClass(String, boolean)
282      */
283     public CtField lookupFieldByJvmName2(String jvmClassName, Symbol fieldSym,
284                                          ASTree expr) throws NoFieldException
285     {
286         String field = fieldSym.get();
287         CtClass cc = null;
288         try {
289             cc = lookupClass(jvmToJavaName(jvmClassName), true);
290         }
291         catch (CompileError e) {
292             // EXPR might be part of a qualified class name.
293             throw new NoFieldException(jvmClassName + "/" + field, expr);
294         }
295 
296         try {
297             return cc.getField(field);
298         }
299         catch (NotFoundException e) {
300             // maybe an inner class.
301             jvmClassName = javaToJvmName(cc.getName());
302             throw new NoFieldException(jvmClassName + "$" + field, expr);
303         }
304     }

342     /**
343      * @param classname         jvm class name.
344      */
345     public CtClass lookupClass(int type, int dim, String classname)
346         throws CompileError
347     {
348         String cname = "";
349         CtClass clazz;
350         if (type == CLASS) {
351             clazz = lookupClassByJvmName(classname);
352             if (dim > 0)
353                 cname = clazz.getName();
354             else
355                 return clazz;
356         }
357         else
358             cname = getTypeName(type);
359 
360         while (dim-- > 0)
361             cname += "[]";
362 
363         return lookupClass(cname, false);
364     }

366     /*
367      * type cannot be CLASS
368      */
369     static String getTypeName(int type) throws CompileError {
370         String cname = "";
371         switch (type) {
372         case BOOLEAN :
373             cname = "boolean";
374             break;
375         case CHAR :
376             cname = "char";
377             break;
378         case BYTE :
379             cname = "byte";
380             break;
381         case SHORT :
382             cname = "short";
383             break;
384         case INT :
385             cname = "int";
386             break;
387         case LONG :
388             cname = "long";
389             break;
390         case FLOAT :
391             cname = "float";
392             break;
393         case DOUBLE :
394             cname = "double";
395             break;
396         case VOID :
397             cname = "void";
398             break;
399         default :
400             fatal();
401         }
402 
403         return cname;
404     }

406     /**
407      * @param name      a qualified class name. e.g. java.lang.String
408      */
409     public CtClass lookupClass(String name, boolean notCheckInner)
410         throws CompileError
411     {
412         Hashtable cache = getInvalidNames();
413         Object found = cache.get(name);
414         if (found == INVALID)
415             throw new CompileError("no such class: " + name);
416         else if (found != null)
417             try {
418                 return classPool.get((String)found);
419             }
420             catch (NotFoundException e) {}
421 
422         CtClass cc = null;
423         try {
424             cc = lookupClass0(name, notCheckInner);
425         }
426         catch (NotFoundException e) {
427             cc = searchImports(name);
428         }
429 
430         cache.put(name, cc.getName());
431         return cc;
432     }

434     private static final String INVALID = "<invalid>";
435     private static WeakHashMap invalidNamesMap = new WeakHashMap();
436     private Hashtable invalidNames = null;
437 
438     // for unit tests
439     public static int getInvalidMapSize() { return invalidNamesMap.size(); }
440 
441     private Hashtable getInvalidNames() {
442         Hashtable ht = invalidNames;
443         if (ht == null) {
444             synchronized (MemberResolver.class) {
445                 WeakReference ref = (WeakReference)invalidNamesMap.get(classPool);
446                 if (ref != null)
447                     ht = (Hashtable)ref.get();
448 
449                 if (ht == null) {
450                     ht = new Hashtable();
451                     invalidNamesMap.put(classPool, new WeakReference(ht));
452                 }
453             }
454 
455             invalidNames = ht;
456         }
457 
458         return ht;
459     }

461     private CtClass searchImports(String orgName)
462         throws CompileError
463     {
464         if (orgName.indexOf('.') < 0) {
465             Iterator it = classPool.getImportedPackages();
466             while (it.hasNext()) {
467                 String pac = (String)it.next();
468                 String fqName = pac + '.' + orgName;
469                 try {
470                     return classPool.get(fqName);
471                 }
472                 catch (NotFoundException e) {
473                     try {
474                         if (pac.endsWith("." + orgName))
475                             return classPool.get(pac);
476                     }
477                     catch (NotFoundException e2) {}
478                 }
479             }
480         }
481 
482         getInvalidNames().put(orgName, INVALID);
483         throw new CompileError("no such class: " + orgName);
484     }
485 
486     private CtClass lookupClass0(String classname, boolean notCheckInner)
487         throws NotFoundException
488     {
489         CtClass cc = null;
490         do {
491             try {
492                 cc = classPool.get(classname);
493             }
494             catch (NotFoundException e) {
495                 int i = classname.lastIndexOf('.');
496                 if (notCheckInner || i < 0)
497                     throw e;
498                 else {
499                     StringBuffer sbuf = new StringBuffer(classname);
500                     sbuf.setCharAt(i, '$');
501                     classname = sbuf.toString();
502                 }
503             }
504         } while (cc == null);
505         return cc;
506     }

508     /* Converts a class name into a JVM-internal representation.
509      *
510      * It may also expand a simple class name to java.lang.*.
511      * For example, this converts Object into java/lang/Object.
512      */
513     public String resolveClassName(ASTList name) throws CompileError {
514         if (name == null)
515             return null;
516         else
517             return javaToJvmName(lookupClassByName(name).getName());
518     }

520     /* Expands a simple class name to java.lang.*.
521      * For example, this converts Object into java/lang/Object.
522      */
523     public String resolveJvmClassName(String jvmName) throws CompileError {
524         if (jvmName == null)
525             return null;
526         else
527             return javaToJvmName(lookupClassByJvmName(jvmName).getName());
528     }

530     public static CtClass getSuperclass(CtClass c) throws CompileError {
531         try {
532             CtClass sc = c.getSuperclass();
533             if (sc != null)
534                 return sc;
535         }
536         catch (NotFoundException e) {}
537         throw new CompileError("cannot find the super class of "
538                                + c.getName());
539     }

541     public static CtClass getSuperInterface(CtClass c, String interfaceName)
542         throws CompileError
543     {
544         try {
545             CtClass[] intfs = c.getInterfaces();
546             for (int i = 0; i < intfs.length; i++)
547                 if (intfs[i].getName().equals(interfaceName))
548                     return intfs[i];
549         } catch (NotFoundException e) {}
550         throw new CompileError("cannot find the super inetrface " + interfaceName
551                                + " of " + c.getName());
552     }

562     public static int descToType(char c) throws CompileError {
563         switch (c) {
564         case 'Z' :
565             return BOOLEAN;
566         case 'C' :
567             return CHAR;
568         case 'B' :
569             return  BYTE;
570         case 'S' :
571             return SHORT;
572         case 'I' :
573             return INT;
574         case 'J' :
575             return LONG;
576         case 'F' :
577             return FLOAT;
578         case 'D' :
579             return DOUBLE;
580         case 'V' :
581             return VOID;
582         case 'L' :
583         case '[' :
584             return CLASS;
585         default :
586             fatal();
587             return VOID;    // never reach here
588         }
589     }

591     public static int getModifiers(ASTList mods) {
592         int m = 0;
593         while (mods != null) {
594             Keyword k = (Keyword)mods.head();
595             mods = mods.tail();
596             switch (k.get()) {
597             case STATIC :
598                 m |= Modifier.STATIC;
599                 break;
600             case FINAL :
601                 m |= Modifier.FINAL;
602                 break;
603             case SYNCHRONIZED :
604                 m |= Modifier.SYNCHRONIZED;
605                 break;
606             case ABSTRACT :
607                 m |= Modifier.ABSTRACT;
608                 break;
609             case PUBLIC :
610                 m |= Modifier.PUBLIC;
611                 break;
612             case PROTECTED :
613                 m |= Modifier.PROTECTED;
614                 break;
615             case PRIVATE :
616                 m |= Modifier.PRIVATE;
617                 break;
618             case VOLATILE :
619                 m |= Modifier.VOLATILE;
620                 break;
621             case TRANSIENT :
622                 m |= Modifier.TRANSIENT;
623                 break;
624             case STRICT :
625                 m |= Modifier.STRICT;
626                 break;
627             }
628         }
629 
630         return m;
631     }
