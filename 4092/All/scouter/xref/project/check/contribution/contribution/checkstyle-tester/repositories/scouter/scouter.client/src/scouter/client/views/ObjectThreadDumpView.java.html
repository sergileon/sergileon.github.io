107 	public void createUpperMenu(Composite composite){
108 		headerComp = new Composite(composite, SWT.NONE);
109 		headerComp.setLayoutData(new GridData(SWT.FILL, SWT.FILL, true, false));
110 		headerComp.setLayout(UIUtil.formLayout(0, 0));
111 		
112 		analyzeBtn = new Button(headerComp, SWT.PUSH);
113 		analyzeBtn.setLayoutData(UIUtil.formData(null, -1, 0, 2, 100, -5, null, -1));
114 		analyzeBtn.setText("SFA");
115 		analyzeBtn.addListener(SWT.Selection, new Listener() {
116 			public void handleEvent(Event event) {
117 				switch (event.type) {
118 				case SWT.Selection:
119 					MainProcessor.instance().processStackContents(text.getText());
120 					break;
121 				}
122 			}
123 		});
124 		
125 		findButton = new Button(headerComp, SWT.PUSH);
126 		findButton.setLayoutData(UIUtil.formData(null, -1, 0, 2, analyzeBtn, -3, null, -1));
127 		findButton.setText("Find");
128 		findButton.addListener(SWT.Selection, new Listener() {
129 			public void handleEvent(Event event) {
130 				switch (event.type) {
131 				case SWT.Selection:
132 					listener.setSearchString(searchText.getText());
133 			        text.redraw();
134 				}
135 			}
136 		});
137 		
138 		searchText = new Text(headerComp, SWT.BORDER);
139 		searchText.setLayoutData(UIUtil.formData(null, -1, 0, 5, findButton, -5, null, -1, 300));
140 		searchText.addKeyListener(new KeyListener() {
141 			public void keyPressed(KeyEvent e) {
142 				if(e.keyCode == 13){ // enter key
143 					listener.setSearchString(searchText.getText());
144 			        text.redraw();
145 				}
146 			}
147 			public void keyReleased(KeyEvent e) {}
148 		});
149 		
150 	}

159 		shell = parent.getShell();

162 			public void keyReleased(KeyEvent e) {

172 		gLayout.horizontalSpacing = 0;

179 		textComposite.setLayoutData(new GridData(GridData.FILL, GridData.FILL, true, true));
180 		textComposite.setLayout(new FillLayout());

183 		text.addKeyListener(new KeyListener() {
184 			public void keyReleased(KeyEvent e) {

187 				if(e.stateMask == SWT.CTRL){
188 					if(e.keyCode == 'a'){
189 						text.selectAll();
190 					}else if(e.keyCode == 'f'){
191 						searchText.setFocus();

193 					}
194 				}
195 				
196 			}
197 		});

199 		text.addLineStyleListener(listener);
200 		
201 		IToolBarManager man = getViewSite().getActionBars().getToolBarManager();
202 		man.add(new Action("Copy", ImageUtil.getImageDescriptor(Images.copy)) {
203 			public void run() {
204 				Clipboard clipboard = new Clipboard(Display.getDefault());
205 	            clipboard.setContents(new Object[] { text.getText() }, new Transfer[] { TextTransfer.getInstance() });
206 	            clipboard.dispose();
207 	            MessageDialog.openInformation(parent.getShell(), "Copied", "Copy all contents to clipboard");
208 			}
209 		});
210 	}

212 	public void setInput(int serverId){
213 		this.serverId = serverId;
214 		this.setPartName("ThreadDump[" + TextProxy.object.getLoadText(DateUtil.yyyymmdd(TimeUtil.getCurrentTime(serverId)), objHash, serverId) + "]");
215 		load();
216 	}
217 	
218 	public void load() {
219 		ExUtil.asyncRun(new Runnable() {
220 			public void run() {
221 				MapPack mpack = null;
222 				TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
223 				try {
224 					MapPack param = new MapPack();
225 					param.put("objHash", objHash);
226 					mpack = (MapPack) tcp.getSingle(RequestCmd.OBJECT_THREAD_DUMP, param);
227 				} catch(Exception e){
228 					ConsoleProxy.errorSafe(e.toString());
229 				} finally {
230 					TcpProxy.putTcpProxy(tcp);
231 				}
232 				
233 				if (mpack != null) {
234 					String error = mpack.getText("error");
235 					if (error != null) {
236 						ConsoleProxy.errorSafe(error);
237 					}
238 					final ListValue lv = mpack.getList("threadDump");
239 					
240 					ExUtil.exec(text, new Runnable() {
241 						public void run() {
242 							StringBuilder sb = new StringBuilder();
243 							for (int i = 0; i < lv.size(); i++) {
244 								sb.append(lv.getString(i) + "\n");
245 							}
246 							text.setText(sb.toString());
247 						}
248 					});
249 				}
250 			}
251 		});
252 	}
