17  package scouter.javassist.convert;
18  
19  import scouter.javassist.CtClass;
20  import scouter.javassist.CtMethod;
21  import scouter.javassist.NotFoundException;
22  import scouter.javassist.bytecode.BadBytecode;
23  import scouter.javassist.bytecode.Bytecode;
24  import scouter.javassist.bytecode.CodeAttribute;
25  import scouter.javassist.bytecode.CodeIterator;
26  import scouter.javassist.bytecode.ConstPool;
27  import scouter.javassist.bytecode.Descriptor;
28  import scouter.javassist.convert.TransformCall;
29  import scouter.javassist.convert.Transformer;
30  
31  public class TransformBefore extends TransformCall {
32      protected CtClass[] parameterTypes;
33      protected int locals;
34      protected int maxLocals;
35      protected byte[] saveCode, loadCode;
36  
37      public TransformBefore(Transformer next,
38                             CtMethod origMethod, CtMethod beforeMethod)
39          throws NotFoundException
40      {
41          super(next, origMethod, beforeMethod);
42  
43          // override
44          methodDescriptor = origMethod.getMethodInfo2().getDescriptor();
45  
46          parameterTypes = origMethod.getParameterTypes();
47          locals = 0;
48          maxLocals = 0;
49          saveCode = loadCode = null;
50      }
51  
52      public void initialize(ConstPool cp, CodeAttribute attr) {
53          super.initialize(cp, attr);
54          locals = 0;
55          maxLocals = attr.getMaxLocals();
56          saveCode = loadCode = null;
57      }
58  
59      protected int match(int c, int pos, CodeIterator iterator,
60                          int typedesc, ConstPool cp) throws BadBytecode
61      {
62          if (newIndex == 0) {
63              String desc = Descriptor.ofParameters(parameterTypes) + 'V';
64              desc = Descriptor.insertParameter(classname, desc);
65              int nt = cp.addNameAndTypeInfo(newMethodname, desc);
66              int ci = cp.addClassInfo(newClassname);
67              newIndex = cp.addMethodrefInfo(ci, nt);
68              constPool = cp;
69          }
70  
71          if (saveCode == null)
72              makeCode(parameterTypes, cp);
73  
74          return match2(pos, iterator);
75      }
76  
77      protected int match2(int pos, CodeIterator iterator) throws BadBytecode {
78          iterator.move(pos);
79          iterator.insert(saveCode);
80          iterator.insert(loadCode);
81          int p = iterator.insertGap(3);
82          iterator.writeByte(INVOKESTATIC, p);
83          iterator.write16bit(newIndex, p + 1);
84          iterator.insert(loadCode);
85          return iterator.next();
86      }
87  
88      public int extraLocals() { return locals; }
89  
90      protected void makeCode(CtClass[] paramTypes, ConstPool cp) {
91          Bytecode save = new Bytecode(cp, 0, 0);
92          Bytecode load = new Bytecode(cp, 0, 0);
93  
94          int var = maxLocals;
95          int len = (paramTypes == null) ? 0 : paramTypes.length;
96          load.addAload(var);
97          makeCode2(save, load, 0, len, paramTypes, var + 1);
98          save.addAstore(var);
99  
100         saveCode = save.get();
101         loadCode = load.get();
102     }
103 
104     private void makeCode2(Bytecode save, Bytecode load,
105                            int i, int n, CtClass[] paramTypes, int var)
106     {
107         if (i < n) {
108             int size = load.addLoad(var, paramTypes[i]);
109             makeCode2(save, load, i + 1, n, paramTypes, var + size);
110             save.addStore(var, paramTypes[i]);
111         }
112         else
113             locals = var - maxLocals;
114     }
115 }
