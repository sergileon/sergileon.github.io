89  	public void init(IViewSite site) throws PartInitException {
90  		super.init(site);
91  		String secId = site.getSecondaryId();
92  		String[] ids = StringUtil.split(secId, "&");
93  		this.serverId = CastUtil.cint(ids[0]);

95  		this.counter = ids[2];
96  		this.mode = CounterUtil.getTotalMode(objType, counter);

99  	private String yesterday;
100 	boolean isActive = false;
101 	public void refresh() {
102 		String date =  DateUtil.yyyymmdd(TimeUtil.getCurrentTime(serverId) - DateUtil.MILLIS_PER_DAY);
103 		if (date.equals(yesterday) == false) {
104 			yesterday = date;
105 			getYesterdayData(date);
106 		}
107 		final ArrayList<Pack> packs = new ArrayList<Pack>();
108 		TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
109 		try {
110 			MapPack param = new MapPack();
111 			param.put("objType", objType);
112 			param.put("counter", counter);
113 			param.put("mode", CounterUtil.getTotalMode(objType, counter));
114 			isActive = false;
115 			tcp.process(RequestCmd.COUNTER_TODAY_ALL, param, new INetReader() {
116 				public void process(DataInputX in) throws IOException {
117 					packs.add(in.readPack());
118 				}
119 			});
120 		} catch (Throwable t) {
121 			ConsoleProxy.errorSafe(t.toString());
122 		} finally {
123 			TcpProxy.putTcpProxy(tcp);
124 		}
125 		
126 		final Map<Long, Double> valueMap = ScouterUtil.getLoadTotalMap(counter, packs, mode, TimeTypeEnum.FIVE_MIN);
127 		if (valueMap.size() > 0) {
128 			isActive = true;
129 		}
130 		
131 		ExUtil.exec(this.canvas, new Runnable() {
132 			public void run() {
133 				if(isActive){
134 					setActive();
135 				}else{
136 					setInactive();
137 				}
138 				String date = DateUtil.yyyymmdd(TimeUtil.getCurrentTime(serverId));
139 				try {
140 					long stime = DateUtil.getTime(date, "yyyyMMdd");
141 					long etime = stime + DateUtil.MILLIS_PER_DAY;
142 					xyGraph.primaryXAxis.setRange(stime, etime);
143 					traceDataProvider.clearTrace();
144 					Set<Long> timeSet = valueMap.keySet();
145 					for (long time : timeSet) {
146 						traceDataProvider.addSample(new Sample(CastUtil.cdouble(time), CastUtil.cdouble(valueMap.get(time))));
147 					}
148 					if (CounterUtil.isPercentValue(objType, counter)) {
149 						xyGraph.primaryYAxis.setRange(0, 100);
150 					} else {
151 						double max = ChartUtil.getMax(traceDataProvider.iterator());
152 						max = Math.max(max, yesterdayMax);
153 						xyGraph.primaryYAxis.setRange(0, max);
154 					}
155 				} catch (Exception e) { e.printStackTrace(); }
156 				canvas.redraw();
157 				xyGraph.repaint();
158 			}
159 		});
160 	}

162 	double yesterdayMax;
163 
164 	private void getYesterdayData(String date) {
165 		final ArrayList<Pack> values = new ArrayList<Pack>();
166 		TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
167 		try {
168 			MapPack param = new MapPack();
169 			param.put("date", date);
170 			param.put("objType", objType);
171 			param.put("counter", counter);
172 
173 			tcp.process(RequestCmd.COUNTER_PAST_DATE_ALL, param, new INetReader() {
174 				public void process(DataInputX in) throws IOException {
175 					Pack p = in.readPack();
176 					values.add(p);
177 				};
178 			});
179 		} catch (Throwable t) {
180 			t.printStackTrace();
181 		} finally {
182 			TcpProxy.putTcpProxy(tcp);
183 		}
184 		
185 		final Map<Long, Double> valueMap = ScouterUtil.getLoadTotalMap(counter, values, mode, TimeTypeEnum.FIVE_MIN);
186 		
187 		ExUtil.exec(this.canvas, new Runnable() {
188 			public void run() {
189 				yesterdayDataProvider.clearTrace();
190 				Set<Long> timeSet = valueMap.keySet();
191 				for (long time : timeSet) {
192 					yesterdayDataProvider.addSample(new Sample(CastUtil.cdouble(time + DateUtil.MILLIS_PER_DAY), CastUtil.cdouble(valueMap.get(time))));
193 				}
194 				yesterdayMax = ChartUtil.getMax(yesterdayDataProvider.iterator());
195 			}
196 		});
197 	}

205 	public void createPartControl(Composite parent) {
206 		GridLayout layout = new GridLayout(1, true);
207 		layout.marginHeight = 5;
208 		layout.marginWidth = 5;
209 		parent.setLayout(layout);
210 		parent.setBackground(ColorUtil.getInstance().getColor(SWT.COLOR_WHITE));
211 		parent.setBackgroundMode(SWT.INHERIT_FORCE);
212 		
213 		window = PlatformUI.getWorkbench().getActiveWorkbenchWindow();
214 		man = getViewSite().getActionBars().getToolBarManager();
215 		
216 		man.add(new Action("Reload", ImageUtil.getImageDescriptor(Images.refresh)) {
217 			public void run() {
218 				yesterday = null;
219 				yesterdayDataProvider.clearTrace();
220 				traceDataProvider.clearTrace();
221 				thread.interrupt();
222 			}
223 		});
224 		
225 		canvas = new FigureCanvas(parent);
226 		canvas.setScrollBarVisibility(FigureCanvas.NEVER);
227 		canvas.setLayoutData(new GridData(GridData.FILL_BOTH));
228 		
229 		canvas.addControlListener(new ControlListener() {
230 			boolean lock = false;
231 			public void controlResized(ControlEvent e) {
232 				org.eclipse.swt.graphics.Rectangle r = canvas.getClientArea();
233 				if (!lock) { 
234 					lock = true;
235 					if (ChartUtil.isShowDescriptionAllowSize(r.height)) {
236 						CounterTodayTotalView.this.setContentDescription(desc);
237 					} else {
238 						CounterTodayTotalView.this.setContentDescription("");
239 					}
240 					r = canvas.getClientArea();
241 					lock = false;
242 				}
243 				xyGraph.setSize(r.width, r.height);
244 			}
245 
246 			public void controlMoved(ControlEvent e) {
247 			}
248 		});
249 		
250 		xyGraph = new XYGraph();
251 		xyGraph.setShowLegend(false);
252 		//xyGraph.setShowTitle(false);
253 
254 		canvas.setContents(xyGraph);
255 
256 		xyGraph.primaryXAxis.setDateEnabled(true);
257 		xyGraph.primaryXAxis.setShowMajorGrid(true);
258 
259 		xyGraph.primaryYAxis.setAutoScale(true);
260 		xyGraph.primaryYAxis.setShowMajorGrid(true);
261 		
262 		xyGraph.primaryXAxis.setTitle("");
263 		xyGraph.primaryYAxis.setTitle("");
264 		
265 		ScouterUtil.addShowTotalValueListener(canvas, xyGraph);
266 		
267 		createDataProvider();
268 		createYesterdayProvider();
269 		
270 		canvas.addKeyListener(new KeyListener() {
271 			public void keyReleased(KeyEvent e) {
272 			}
273 			public void keyPressed(KeyEvent e) {
274 				if(e.keyCode == SWT.F5){
275 					yesterday = null;
276 					yesterdayDataProvider.clearTrace();
277 					traceDataProvider.clearTrace();
278 					thread.interrupt();
279 				}
280 			}
281 		});
282 		
283 		String date = DateUtil.yyyymmdd(TimeUtil.getCurrentTime(serverId));
284 		
285 		Server server = ServerManager.getInstance().getServer(serverId);
286 		String svrName = "";
287 		String counterDisplay = "";
288 		String counterUnit = "";
289 		if(server != null){
290 			svrName = server.getName();
291 			counterDisplay = server.getCounterEngine().getCounterDisplayName(objType, counter);
292 			counterUnit = server.getCounterEngine().getCounterUnit(objType, counter);
293 		}
294 		desc = "ⓢ"+svrName+" | (Today) [" + date.substring(0, 4) + "-" + date.substring(4, 6) + "-" + date.substring(6, 8) + "] Total " + counterDisplay+(!"".equals(counterUnit)?" ("+counterUnit+")":"");
295 		try {
296 			setViewTab(objType, counter, serverId);
297 		} catch (Exception e1) {
298 			e1.printStackTrace();
299 		}
300 		MenuUtil.createCounterContextMenu(ID, canvas, serverId, objType, counter);
301 		thread = new RefreshThread(this, 10000);
302 		thread.setName(this.toString() + " - " + "objType:"+objType + ", counter:"+counter + ", serverId:"+serverId);
303 		thread.start();
304 	}

306 	private void createDataProvider() {
307 		traceDataProvider = new CircularBufferDataProvider(true);
308 		traceDataProvider.setBufferSize(288);
309 		traceDataProvider.setCurrentXDataArray(new double[] {});
310 		traceDataProvider.setCurrentYDataArray(new double[] {});
311 
312 		// create the trace
313 		trace = new Trace("TOTAL", xyGraph.primaryXAxis, xyGraph.primaryYAxis, traceDataProvider);
314 
315 		// set trace property
316 		trace.setPointStyle(PointStyle.NONE);
317 		trace.getXAxis().setFormatPattern("HH:mm:ss");
318 		trace.getYAxis().setFormatPattern("#,##0");
319 
320 		trace.setLineWidth(PManager.getInstance().getInt(PreferenceConstants.P_CHART_LINE_WIDTH));
321 		trace.setTraceType(TraceType.AREA);
322 		trace.setTraceColor(ColorUtil.getInstance().TOTAL_CHART_COLOR);
323 
324 		// add the trace to xyGraph
325 		xyGraph.addTrace(trace);
326 		ChartUtil.addSolidLine(xyGraph, traceDataProvider, ColorUtil.getInstance().TOTAL_CHART_COLOR);
327 	}
