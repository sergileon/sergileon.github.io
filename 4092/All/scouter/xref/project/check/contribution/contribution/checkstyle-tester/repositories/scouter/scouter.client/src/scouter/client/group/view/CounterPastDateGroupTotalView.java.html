294 	private IDataProvider getDataProvider(int size) {
295 		CircularBufferDataProvider provider = new CircularBufferDataProvider(true);
296 		provider.setBufferSize(size);
297 		Trace trace = new Trace(grpName, xyGraph.primaryXAxis, xyGraph.primaryYAxis, provider);
298 		trace.setPointStyle(PointStyle.NONE);
299 		trace.getXAxis().setFormatPattern("yyyy-MM-dd\n  HH:mm:ss");
300 		trace.getYAxis().setFormatPattern("#,##0");
301 		trace.setLineWidth(PManager.getInstance().getInt(PreferenceConstants.P_CHART_LINE_WIDTH));
302 		trace.setTraceType(TraceType.AREA);
303 		trace.setTraceColor(ColorUtil.getInstance().TOTAL_CHART_COLOR);
304 		xyGraph.addTrace(trace);
305 		Trace lineTrace = ChartUtil.addSolidLine(xyGraph, provider, ColorUtil.getInstance().TOTAL_CHART_COLOR);
306 		traces.add(trace);
307 		traces.add(lineTrace);
308 		return provider;
309 	}

311 	private void forceRefresh() {
312 		for (Trace trace : traces) {
313 			xyGraph.removeTrace(trace);
314 		}
315 		traces.clear();
316 		ExUtil.asyncRun(new Runnable() {
317 			public void run() {
318 				load();
319 			}
320 		});
321 	}

323 	private GroupManager manager = GroupManager.getInstance();
324 	
325 	private void collectObj() {
326 		serverObjMap.clear();
327 		Set<Integer> objHashs = manager.getObjectsByGroup(grpName);
328 		for (int objHash : objHashs) {
329 			AgentObject agentObj = AgentModelThread.getInstance().getAgentObject(objHash);
330 			if (agentObj == null || agentObj.isAlive() == false) {
331 				continue;
332 			}
333 			int serverId = agentObj.getServerId();
334 			ListValue lv = serverObjMap.get(serverId);
335 			if (lv == null) {
336 				lv = new ListValue();
337 				serverObjMap.put(serverId, lv);
338 			}
339 			lv.add(objHash);
340 		}
341 	}

343 	private void load() {
344 		final long stime = DateUtil.getTime(sDate, "yyyyMMdd");
345 		final long etime = DateUtil.getTime(eDate, "yyyyMMdd") + (DateUtil.MILLIS_PER_DAY - 1);
346 		collectObj();
347 		Iterator<Integer> serverIds = serverObjMap.keySet().iterator();
348 		final List<Pack> result = new ArrayList<Pack>();
349 		while (serverIds.hasNext()) {
350 			int serverId = serverIds.next();
351 			TcpProxy tcp = TcpProxy.getTcpProxy(serverId);
352 			try {
353 				MapPack param = new MapPack();
354 				param.put("stime", stime);
355 				param.put("etime", etime);
356 				param.put("counter", counter);
357 				param.put("objHash", serverObjMap.get(serverId));
358 				tcp.process(RequestCmd.COUNTER_PAST_LONGDATE_GROUP, param, new INetReader() {
359 					public void process(DataInputX in) throws IOException {
360 						Pack p = in.readPack();
361 						result.add(p);
362 					}
363 				});
364 			} catch (Exception e) {
365 				e.printStackTrace();
366 			} finally {
367 				TcpProxy.putTcpProxy(tcp);
368 			}
369 		}
370 		final Map<Long, Double> valueMap = ScouterUtil.getLoadTotalMap(counter, result, mode, TimeTypeEnum.FIVE_MIN);
371 		ExUtil.exec(canvas, new Runnable() {
372 			public void run() {
373 				xyGraph.primaryXAxis.setRange(stime, etime);
374 				CircularBufferDataProvider provider = (CircularBufferDataProvider) getDataProvider((int)((etime - stime) / DateUtil.MILLIS_PER_FIVE_MINUTE));
375 				provider.clearTrace();
376 				Set<Long> timeSet = valueMap.keySet();
377 				for (long time : timeSet) {
378 					provider.addSample(new Sample(time, CastUtil.cdouble(valueMap.get(time))));
379 				}
380 				if (CounterUtil.isPercentValue(objType, counter)) {
381 					xyGraph.primaryYAxis.setRange(0, 100);
382 				} else {
383 					double max = ChartUtil.getMax(provider.iterator());
384 					xyGraph.primaryYAxis.setRange(0, max);
385 				}
386 				applyBtn.setEnabled(true);
387 				canvas.redraw();
388 			}
389 		});
390 	}
