63      public PerformanceWindow(Shell shell, StackFileInfo stackFileInfo, String filter, boolean isExcludeStack, boolean isAscending, boolean isRemoveLine, boolean isInerPercent) {
64      	m_parentShell = shell;
65      	m_stackFileInfo = stackFileInfo;
66          m_filter = filter;
67          m_isExcludeStack = isExcludeStack;
68          m_isAscending = isAscending;
69          m_isRemoveLine = isRemoveLine;
70          m_isInerPercent = isInerPercent;
71  
72          ArrayList<String> list = stackFileInfo.getParserConfig().getExcludeStack();
73          if ( m_isExcludeStack && list != null && list.size() > 0 ) {
74              m_excludeStack = list;
75          }
76  
77          list = stackFileInfo.getParserConfig().getSingleStack();
78          if ( list != null && list.size() > 0 ) {
79              m_singleStack = list;
80              m_singleStackCount = list.size();
81          }
82          
83          StringBuilder buffer = new StringBuilder(200);
84          buffer.append('[');
85          if ( filter == null )
86              buffer.append("All");
87          else
88              buffer.append(filter);
89          buffer.append("] ").append(stackFileInfo.getFilename());
90          buffer.append(" [EXC:").append(isExcludeStack).append("] [ASC:");
91          buffer.append(isAscending).append(']');
92          
93          init(shell, buffer.toString());
94      }

111     private void constructTree() {
112         String filename = StackParser.getWorkingThreadFilename(m_stackFileInfo.getFilename());
113         if ( filename == null )
114             return;
115 
116         HashMap<String, Counter> tree = new HashMap<String, Counter>();
117         BufferedReader reader = null;
118         int startStackLine = m_stackFileInfo.getUsedParser().getConfig().getStackStartLine();
119         try {
120             File file = new File(filename);
121             reader = new BufferedReader(new FileReader(file));
122 
123             String line = null;
124             int lineCount = 0;
125             boolean isFiltered = false;
126             int filterIndex = 0;
127 
128             ArrayList<String> list = new ArrayList<String>(300);
129 
130             while ( (line = reader.readLine()) != null ) {
131                 if ( line.length() == 0 ) {
132                     if ( isFiltered && list.size() > 0 ) {
133                         createPerformanceTree(tree, list, filterIndex);
134                     }
135 
136                     list = new ArrayList<String>(300);
137                     lineCount = 0;
138                     isFiltered = false;
139                 } else {
140                     lineCount++;
141                     if ( lineCount > startStackLine ) {
142                         if ( !isFiltered ) {
143                             if ( m_filter == null ) {
144                                 isFiltered = true;
145                             } else if ( line.indexOf(m_filter) >= 0 ) {
146                                 isFiltered = true;
147                                 filterIndex = lineCount - (startStackLine + 1);
148                             }
149                         }
150 
151                         list.add(line);
152                     }
153                 }
154             }
155             if ( isFiltered && list.size() > 0 ) {
156                 createPerformanceTree(tree, list, filterIndex);
157             }
158         } catch ( Exception ex ) {
159             throw new RuntimeException(ex);
160         } finally {
161             if ( reader != null ) {
162                 try {
163                     reader.close();
164                 } catch ( Exception e ) {
165                 }
166             }
167         }
168 
169         if ( m_isInerPercent && tree != null ) {
170             Counter cntr;
171             Iterator<Counter> itor = tree.values().iterator();
172             while ( itor.hasNext() ) {
173                 cntr = itor.next();
174                 cntr.caculInerCount();
175             }
176         }
177 
178         TreeItem treeItem = new TreeItem(m_performanceTree, SWT.NONE);
179         treeItem.setText("Stack File Nodes");
180         treeItem.setData(null);
181         treeItem.setExpanded(true);
182         makePerformanceTreeUI(treeItem, tree, 0);
183     }

185     private void makePerformanceTreeUI( TreeItem parent, HashMap<String, Counter> tree, int depth ) {
186         if ( tree == null || tree.size() == 0 )
187             return;
188 
189         depth++;
190         
191         String key = null;
192         Counter counter = null;
193         StringBuilder buffer = null;
194         int index = 0;
195 
196         Iterator<String> itor = tree.keySet().iterator();
197         ArrayList<Counter> sortList = new ArrayList<Counter>();
198         while ( itor.hasNext() ) {
199             key = itor.next();
200             counter = tree.get(key);
201             counter.setValue(key);
202             sortList.add(counter);
203         }
204 
205         Collections.sort(sortList, new ValueComp());
206         
207         int itemCount = sortList.size();
208         if(itemCount > 1 && m_expandStartInx == 0){
209         	m_expandStartInx = depth;
210         }
211         
212         TreeItem treeItem;
213         for ( index = 0; index < itemCount; index++ ) {
214             counter = sortList.get(index);
215             buffer = new StringBuilder(100);
216             buffer.append(counter.getCount()).append(" (").append(NumberUtils.intToPercent((counter.getCount() * 10000) / m_totalCount)).append("%) ");
217             if ( m_isInerPercent ) {
218                 buffer.append(NumberUtils.intToPercent((counter.getInerCount() * 10000) / m_totalCount)).append("% ");
219             }
220             buffer.append(counter.getValue());
221 
222             treeItem = new TreeItem(parent,  SWT.NONE );
223             treeItem.setText(buffer.toString());
224             treeItem.setData(counter.getValue());
225         	treeItem.setExpanded(true);
226 
227             if(m_expandStartInx == 0){
228             	m_performanceTree.showItem(treeItem);            	
229             }else if((depth - m_expandStartInx) <= 3){
230             	m_performanceTree.showItem(treeItem);            	
231             }
232             
233             if ( counter.getMap() != null ) {
234                 makePerformanceTreeUI(treeItem, counter.getMap(), depth);
235             }
236         }
237 
238     }

240     private void createPerformanceTree( HashMap<String, Counter> tree, ArrayList<String> list, int filterIndex ) {
241         int startIndex = 0;
242         int size = list.size() - 1;
243         if ( m_filter != null ) {
244             startIndex = filterIndex;
245         } else {
246             startIndex = list.size() - 1;
247         }
248 
249         m_totalCount++;
250         String line = null;
251         HashMap<String, Counter> currentMap = tree;
252         Counter currentCounter = null;
253         int index;
254         int baseIndex = startIndex;
255 
256         while ( true ) {
257             line = list.get(startIndex);
258 
259             if ( m_singleStackCount > 0 ) {
260                 for ( index = 0; index < m_singleStackCount; index++ ) {
261                     if ( line.indexOf(m_singleStack.get(index)) >= 0 ) {
262                         line = m_singleStack.get(index);
263                         break;
264                     }
265                 }
266             }
267 
268             if ( m_excludeStack != null && currentMap != null && currentMap != tree ) {
269                 if ( StringUtils.checkExist(line, m_excludeStack) ) {
270                     if ( m_isAscending ) {
271                         if ( startIndex == 0 )
272                             break;
273 
274                         startIndex--;
275                     } else {
276                         if ( startIndex == size )
277                             break;
278 
279                         startIndex++;
280                     }
281                     continue;
282                 }
283             }
284 
285             if ( m_filter != null && baseIndex == startIndex )
286                 line = StringUtils.makeSimpleLine(line, false);
287             else
288                 line = StringUtils.makeSimpleLine(line, m_isRemoveLine);
289 
290             if ( currentCounter != null ) {
291                 currentMap = currentCounter.getMap();
292                 if ( currentMap == null )
293                     currentMap = currentCounter.addMap();
294             }
295 
296             currentCounter = currentMap.get(line);
297             if ( currentCounter == null ) {
298                 currentCounter = new Counter();
299                 currentMap.put(line, currentCounter);
300             }
301             currentCounter.addCount();
302             if ( m_isAscending ) {
303                 if ( startIndex == 0 )
304                     break;
305 
306                 startIndex--;
307             } else {
308                 if ( startIndex == size )
309                     break;
310 
311                 startIndex++;
312             }
313         }
314     }

316     private class Counter {
317         private int m_count = 0;
318         private int m_inerCount = 0;
319         private String m_value = null;
320         private HashMap<String, Counter> m_map = null;
321 
322         public void addCount() {
323             m_count++;
324         }
325 
326         public int getCount() {
327             return m_count;
328         }
329 
330         public int getInerCount() {
331             return m_inerCount;
332         }
333 
334         public void setValue( String value ) {
335             m_value = value;
336         }
337 
338         public String getValue() {
339             return m_value;
340         }
341 
342         public HashMap<String, Counter> addMap() {
343             if ( m_map == null ) {
344                 m_map = new HashMap<String, Counter>();
345             }
346             return m_map;
347         }
348 
349         public HashMap<String, Counter> getMap() {
350             return m_map;
351         }
352 
353         public void caculInerCount() {
354             if ( m_map == null ) {
355                 m_inerCount = m_count;
356                 return;
357             }
358             int subCount = 0;
359             Iterator<Counter> itor = m_map.values().iterator();
360             Counter cntr;
361             while ( itor.hasNext() ) {
362                 cntr = itor.next();
363                 subCount += cntr.getCount();
364                 cntr.caculInerCount();
365             }
366             m_inerCount = m_count - subCount;
367         }
368     }

370     private class ValueComp implements Comparator<Counter> {
371 
372         public int compare( Counter o1, Counter o2 ) {
373             if ( o1.getCount() > o2.getCount() )
374                 return -1;
375             else if ( o1.getCount() < o2.getCount() )
376                 return 1;
377 
378             return 0;
379         }
380     }

382     private void createTreePopupMenu() {
383 		Menu popupMenu = new Menu(m_performanceTree);
384 		
385 		MenuItem menuItem = new MenuItem(popupMenu, SWT.NONE);
386 		menuItem.setText("Performance tree(Ascending)");
387 		menuItem.addListener(SWT.Selection, this);

389 		menuItem.setText("Performance tree(Descending)");
390 		menuItem.addListener(SWT.Selection, this);
391 		
392 		menuItem = new MenuItem(popupMenu, SWT.SEPARATOR);
393 		
394 		menuItem = new MenuItem(popupMenu, SWT.NONE);
395 		menuItem.setText("Copy function");
396 		menuItem.addListener(SWT.Selection, this);
397 
398 		m_performanceTree.setMenu(popupMenu);       
399     }

401 	public void handleEvent(Event event) {
402 		try {
403 			MenuItem item = (MenuItem)event.widget;
404 			String menuText = item.getText();
405 			if("Performance tree(Ascending)".endsWith(menuText)){
406                 createAnalyzedPerformance(true);
407 			}else if("Performance tree(Descending)".endsWith(menuText)){
408                 createAnalyzedPerformance(false);
409 			}else if("Copy function".endsWith(menuText)){
410                 CopyFunctionName();
411 			}
412 		}catch(Exception ex){
413 			ex.printStackTrace();
414 		}		
415 	}    
416     private void createAnalyzedPerformance( boolean isAscending ) {
417         String filter = getSelectedFunctionName();
418 
419         if ( filter != null && m_stackFileInfo != null ){
420         	new PerformanceWindow(m_parentShell, m_stackFileInfo, filter, m_isExcludeStack, isAscending, m_isRemoveLine, m_isInerPercent);
421         }	
422     }

424     private void CopyFunctionName() {
425         String functionName = getSelectedFunctionName();
426         if ( functionName != null )
427             StringUtils.setClipboard(functionName.trim());
428     }
429 
430     private String getSelectedFunctionName() {
431         TreeItem [] items = m_performanceTree.getSelection();
432         if(items == null || items.length == 0){
433         	return null;
434         }
435         
436         String function = (String)items[0].getData();
437 
438         int endIndex = function.indexOf((int)'(');
439         if ( endIndex >= 0 )
440             function = function.substring(0, endIndex);
441 
442         return function;
443     }
