98      /**
99       * Returns the constructor or method containing the expression.
100      */
101     public CtBehavior where() {
102         MethodInfo mi = thisMethod;
103         CtBehavior[] cb = thisClass.getDeclaredBehaviors();
104         for (int i = cb.length - 1; i >= 0; --i)
105             if (cb[i].getMethodInfo2() == mi)
106                 return cb[i];
107 
108         CtConstructor init = thisClass.getClassInitializer();
109         if (init != null && init.getMethodInfo2() == mi)
110             return init;
111 
112         /* getDeclaredBehaviors() returns a list of methods/constructors.
113          * Although the list is cached in a CtClass object, it might be
114          * recreated for some reason.  Thus, the member name and the signature
115          * must be also checked.
116          */
117         for (int i = cb.length - 1; i >= 0; --i) {
118             if (thisMethod.getName().equals(cb[i].getMethodInfo2().getName())
119                 && thisMethod.getDescriptor()
120                              .equals(cb[i].getMethodInfo2().getDescriptor())) {
121                 return cb[i];
122             }
123         }
124 
125         throw new RuntimeException("fatal: not found");
126     }

128     /**
129      * Returns the list of exceptions that the expression may throw. This list
130      * includes both the exceptions that the try-catch statements including the
131      * expression can catch and the exceptions that the throws declaration
132      * allows the method to throw.
133      */
134     public CtClass[] mayThrow() {
135         ClassPool pool = thisClass.getClassPool();
136         ConstPool cp = thisMethod.getConstPool();
137         LinkedList list = new LinkedList();
138         try {
139             CodeAttribute ca = thisMethod.getCodeAttribute();
140             ExceptionTable et = ca.getExceptionTable();
141             int pos = currentPos;
142             int n = et.size();
143             for (int i = 0; i < n; ++i)
144                 if (et.startPc(i) <= pos && pos < et.endPc(i)) {
145                     int t = et.catchType(i);
146                     if (t > 0)
147                         try {
148                             addClass(list, pool.get(cp.getClassInfo(t)));
149                         }
150                         catch (NotFoundException e) {
151                         }
152                 }
153         }
154         catch (NullPointerException e) {
155         }
156 
157         ExceptionsAttribute ea = thisMethod.getExceptionsAttribute();
158         if (ea != null) {
159             String[] exceptions = ea.getExceptions();
160             if (exceptions != null) {
161                 int n = exceptions.length;
162                 for (int i = 0; i < n; ++i)
163                     try {
164                         addClass(list, pool.get(exceptions[i]));
165                     }
166                     catch (NotFoundException e) {
167                     }
168             }
169         }
170 
171         return (CtClass[])list.toArray(new CtClass[list.size()]);
172     }
173 
174     private static void addClass(LinkedList list, CtClass c) {
175         Iterator it = list.iterator();
176         while (it.hasNext())
177             if (it.next() == c)
178                 return;
179 
180         list.add(c);
181     }

214     static final boolean checkResultValue(CtClass retType, String prog)
215             throws CannotCompileException {
216         /*
217          * Is $_ included in the source code?
218          */
219         boolean hasIt = (prog.indexOf(Javac.resultVarName) >= 0);
220         if (!hasIt && retType != CtClass.voidType)
221             throw new CannotCompileException(
222                     "the resulting value is not stored in "
223                             + Javac.resultVarName);
224 
225         return hasIt;
226     }

228     /*
229      * If isStaticCall is true, null is assigned to $0. So $0 must be declared
230      * by calling Javac.recordParams().
231      * 
232      * After executing this method, the current stack depth might be less than
233      * 0.
234      */
235     static final void storeStack(CtClass[] params, boolean isStaticCall,
236             int regno, Bytecode bytecode) {
237         storeStack0(0, params.length, params, regno + 1, bytecode);
238         if (isStaticCall)
239             bytecode.addOpcode(ACONST_NULL);
240 
241         bytecode.addAstore(regno);
242     }
243 
244     private static void storeStack0(int i, int n, CtClass[] params, int regno,
245             Bytecode bytecode) {
246         if (i >= n)
247             return;
248         else {
249             CtClass c = params[i];
250             int size;
251             if (c instanceof CtPrimitiveType)
252                 size = ((CtPrimitiveType)c).getDataSize();
253             else
254                 size = 1;
255 
256             storeStack0(i + 1, n, params, regno + size, bytecode);
257             bytecode.addStore(regno, c);
258         }
259     }

266     /**
267      * Replaces this expression with the bytecode derived from
268      * the given source text.
269      *
270      * @param statement         a Java statement except try-catch.
271      */
272     public abstract void replace(String statement) throws CannotCompileException;
273 
274     /**
275      * Replaces this expression with the bytecode derived from
276      * the given source text and <code>ExprEditor</code>.
277      *
278      * @param statement         a Java statement except try-catch.
279      * @param recursive         if not null, the substituted bytecode
280      *                          is recursively processed by the given
281      *                          <code>ExprEditor</code>.
282      * @since 3.1
283      */
284     public void replace(String statement, ExprEditor recursive)
285         throws CannotCompileException
286     {
287         replace(statement);
288         if (recursive != null)
289             runEditor(recursive, iterator);
290     }

292     protected void replace0(int pos, Bytecode bytecode, int size)
293             throws BadBytecode {
294         byte[] code = bytecode.get();
295         edited = true;
296         int gap = code.length - size;
297         for (int i = 0; i < size; ++i)
298             iterator.writeByte(NOP, pos + i);
299 
300         if (gap > 0)
301             pos = iterator.insertGapAt(pos, gap, false).position;
302 
303         iterator.write(code, pos);
304         iterator.insert(bytecode.getExceptionTable(), pos);
305         maxLocals = bytecode.getMaxLocals();
306         maxStack = bytecode.getMaxStack();
307     }
308 
309     protected void runEditor(ExprEditor ed, CodeIterator oldIterator)
310         throws CannotCompileException
311     {
312         CodeAttribute codeAttr = oldIterator.get();
313         int orgLocals = codeAttr.getMaxLocals();
314         int orgStack = codeAttr.getMaxStack();
315         int newLocals = locals();
316         codeAttr.setMaxStack(stack());
317         codeAttr.setMaxLocals(newLocals);
318         ExprEditor.LoopContext context
319             = new ExprEditor.LoopContext(newLocals);
320         int size = oldIterator.getCodeLength();
321         int endPos = oldIterator.lookAhead();
322         oldIterator.move(currentPos);
323         if (ed.doit(thisClass, thisMethod, context, oldIterator, endPos))
324             edited = true;
325 
326         oldIterator.move(endPos + oldIterator.getCodeLength() - size);
327         codeAttr.setMaxLocals(orgLocals);
328         codeAttr.setMaxStack(orgStack);
329         maxLocals = context.maxLocals;
330         maxStack += context.maxStack;
331     }
