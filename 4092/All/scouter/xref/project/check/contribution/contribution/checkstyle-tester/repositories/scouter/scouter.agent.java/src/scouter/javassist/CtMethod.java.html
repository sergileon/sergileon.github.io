38  /**
39   * An instance of <code>CtMethod</code> represents a method.
40   *
41   * <p>See the super class <code>CtBehavior</code> since
42   * a number of useful methods are in <code>CtBehavior</code>.
43   * A number of useful factory methods are in <code>CtNewMethod</code>.
44   *
45   * @see CtClass#getDeclaredMethods()
46   * @see CtNewMethod
47   */
48  public final class CtMethod extends CtBehavior {
49      protected String cachedStringRep;
50  
51      /**
52       * @see #make(MethodInfo minfo, CtClass declaring)
53       */
54      CtMethod(MethodInfo minfo, CtClass declaring) {
55          super(declaring, minfo);
56          cachedStringRep = null;
57      }
58  
59      /**
60       * Creates a public abstract method.  The created method must be
61       * added to a class with <code>CtClass.addMethod()</code>.
62       *
63       * @param declaring         the class to which the created method is added.
64       * @param returnType        the type of the returned value
65       * @param mname             the method name
66       * @param parameters        a list of the parameter types
67       *
68       * @see CtClass#addMethod(CtMethod)
69       */
70      public CtMethod(CtClass returnType, String mname,
71                      CtClass[] parameters, CtClass declaring) {
72          this(null, declaring);
73          ConstPool cp = declaring.getClassFile2().getConstPool();
74          String desc = Descriptor.ofMethod(returnType, parameters);
75          methodInfo = new MethodInfo(cp, mname, desc);
76          setModifiers(Modifier.PUBLIC | Modifier.ABSTRACT);
77      }
78  
79      /**
80       * Creates a copy of a <code>CtMethod</code> object.
81       * The created method must be
82       * added to a class with <code>CtClass.addMethod()</code>.
83       *
84       * <p>All occurrences of class names in the created method
85       * are replaced with names specified by
86       * <code>map</code> if <code>map</code> is not <code>null</code>.
87       *
88       * <p>For example, suppose that a method <code>at()</code> is as
89       * follows:
90       *
91       * <pre>
92       * public X at(int i) {
93       *     return (X)super.elementAt(i);
94       * }</pre>
95       *
96       * <p>(<code>X</code> is a class name.)  If <code>map</code> substitutes
97       * <code>String</code> for <code>X</code>, then the created method is:
98       *
99       * <pre>
100      * public String at(int i) {
101      *     return (String)super.elementAt(i);
102      * }</pre>
103      *
104      * <p>By default, all the occurrences of the names of the class
105      * declaring <code>at()</code> and the superclass are replaced
106      * with the name of the class and the superclass that the
107      * created method is added to.
108      * This is done whichever <code>map</code> is null or not.
109      * To prevent this replacement, call <code>ClassMap.fix()</code>
110      * or <code>put()</code> to explicitly specify replacement.
111      *
112      * <p><b>Note:</b> if the <code>.class</code> notation (for example,
113      * <code>String.class</code>) is included in an expression, the
114      * Javac compiler may produce a helper method.
115      * Since this constructor never
116      * copies this helper method, the programmers have the responsiblity of
117      * copying it.  Otherwise, use <code>Class.forName()</code> in the
118      * expression.
119      *
120      * @param src       the source method.
121      * @param declaring    the class to which the created method is added.
122      * @param map       the hashtable associating original class names
123      *                  with substituted names.
124      *                  It can be <code>null</code>.
125      *
126      * @see CtClass#addMethod(CtMethod)
127      * @see ClassMap#fix(String)
128      */
129     public CtMethod(CtMethod src, CtClass declaring, ClassMap map)
130         throws CannotCompileException
131     {
132         this(null, declaring);
133         copy(src, false, map);
134     }

136     /**
137      * Compiles the given source code and creates a method.
138      * This method simply delegates to <code>make()</code> in
139      * <code>CtNewMethod</code>.  See it for more details.
140      * <code>CtNewMethod</code> has a number of useful factory methods.
141      *
142      * @param src               the source text. 
143      * @param declaring    the class to which the created method is added.
144      * @see CtNewMethod#make(String, CtClass)
145      */
146     public static CtMethod make(String src, CtClass declaring)
147         throws CannotCompileException
148     {
149         return CtNewMethod.make(src, declaring);
150     }

152     /**
153      * Creates a method from a <code>MethodInfo</code> object.
154      *
155      * @param declaring     the class declaring the method.
156      * @throws CannotCompileException       if the the <code>MethodInfo</code>
157      *          object and the declaring class have different
158      *          <code>ConstPool</code> objects
159      * @since 3.6
160      */
161     public static CtMethod make(MethodInfo minfo, CtClass declaring)
162         throws CannotCompileException
163     {
164         if (declaring.getClassFile2().getConstPool() != minfo.getConstPool())
165             throw new CannotCompileException("bad declaring class");
166 
167         return new CtMethod(minfo, declaring);
168     }

187     /* This method is also called by CtClassType.getMethods0(). 
188      */
189     final String getStringRep() {
190         if (cachedStringRep == null)
191             cachedStringRep = methodInfo.getName()
192                 + Descriptor.getParamDescriptor(methodInfo.getDescriptor());
193 
194         return cachedStringRep;
195     }

197     /**
198      * Indicates whether <code>obj</code> has the same name and the
199      * same signature as this method.

201     public boolean equals(Object obj) {
202         return obj != null && obj instanceof CtMethod
203                && ((CtMethod)obj).getStringRep().equals(getStringRep());

239     /**
240      * Returns true if the method body is empty, that is, <code>{}</code>.
241      * It also returns true if the method is an abstract method.
242      */
243     public boolean isEmpty() {
244         CodeAttribute ca = getMethodInfo2().getCodeAttribute();
245         if (ca == null)         // abstract or native
246             return (getModifiers() & Modifier.ABSTRACT) != 0;
247 
248         CodeIterator it = ca.iterator();
249         try {
250             return it.hasNext() && it.byteAt(it.next()) == Opcode.RETURN
251                 && !it.hasNext();
252         }
253         catch (BadBytecode e) {}
254         return false;
255     }

257     /**
258      * Copies a method body from another method.
259      * If this method is abstract, the abstract modifier is removed
260      * after the method body is copied.
261      *
262      * <p>All occurrences of the class names in the copied method body
263      * are replaced with the names specified by
264      * <code>map</code> if <code>map</code> is not <code>null</code>.
265      *
266      * @param src       the method that the body is copied from.
267      * @param map       the hashtable associating original class names
268      *                  with substituted names.
269      *                  It can be <code>null</code>.
270      */
271     public void setBody(CtMethod src, ClassMap map)
272         throws CannotCompileException
273     {
274         setBody0(src.declaringClass, src.methodInfo,
275                  declaringClass, methodInfo, map);
276     }

278     /**
279      * Replace a method body with a new method body wrapping the
280      * given method.
281      *
282      * @param mbody             the wrapped method
283      * @param constParam        the constant parameter given to
284      *                          the wrapped method
285      *                          (maybe <code>null</code>).
286      *
287      * @see CtNewMethod#wrapped(CtClass,String,CtClass[],CtClass[],CtMethod,CtMethod.ConstParameter,CtClass)
288      */
289     public void setWrappedBody(CtMethod mbody, ConstParameter constParam)
290         throws CannotCompileException
291     {
292         declaringClass.checkModify();
293 
294         CtClass clazz = getDeclaringClass();
295         CtClass[] params;
296         CtClass retType;
297         try {
298             params = getParameterTypes();
299             retType = getReturnType();
300         }
301         catch (NotFoundException e) {
302             throw new CannotCompileException(e);
303         }
304 
305         Bytecode code = CtNewWrappedMethod.makeBody(clazz,
306                                                     clazz.getClassFile2(),
307                                                     mbody,
308                                                     params, retType,
309                                                     constParam);
310         CodeAttribute cattr = code.toCodeAttribute();
311         methodInfo.setCodeAttribute(cattr);
312         methodInfo.setAccessFlags(methodInfo.getAccessFlags()
313                                   & ~AccessFlag.ABSTRACT);
314         // rebuilding a stack map table is not needed.
315     }

321      * They are used to specify the parameter given to the methods

326      * @see CtNewConstructor#make(CtClass[],CtClass[],int,CtMethod,CtMethod.ConstParameter,CtClass)
