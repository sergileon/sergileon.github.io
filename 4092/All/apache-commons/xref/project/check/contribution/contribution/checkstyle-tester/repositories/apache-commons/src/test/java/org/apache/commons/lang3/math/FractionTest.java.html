344     public void testFactory_double() {
345         Fraction f = null;
346         
347         try {
348             f = Fraction.getFraction(Double.NaN);
349             fail("expecting ArithmeticException");
350         } catch (final ArithmeticException ex) {}
351         
352         try {
353             f = Fraction.getFraction(Double.POSITIVE_INFINITY);
354             fail("expecting ArithmeticException");
355         } catch (final ArithmeticException ex) {}
356         
357         try {
358             f = Fraction.getFraction(Double.NEGATIVE_INFINITY);
359             fail("expecting ArithmeticException");
360         } catch (final ArithmeticException ex) {}
361         
362         try {
363             f = Fraction.getFraction((double) Integer.MAX_VALUE + 1);
364             fail("expecting ArithmeticException");
365         } catch (final ArithmeticException ex) {}
366         
367         // zero
368         f = Fraction.getFraction(0.0d);
369         assertEquals(0, f.getNumerator());
370         assertEquals(1, f.getDenominator());
371         
372         // one
373         f = Fraction.getFraction(1.0d);
374         assertEquals(1, f.getNumerator());
375         assertEquals(1, f.getDenominator());
376         
377         // one half
378         f = Fraction.getFraction(0.5d);
379         assertEquals(1, f.getNumerator());
380         assertEquals(2, f.getDenominator());
381         
382         // negative
383         f = Fraction.getFraction(-0.875d);
384         assertEquals(-7, f.getNumerator());
385         assertEquals(8, f.getDenominator());
386         
387         // over 1
388         f = Fraction.getFraction(1.25d);
389         assertEquals(5, f.getNumerator());
390         assertEquals(4, f.getDenominator());
391         
392         // two thirds
393         f = Fraction.getFraction(0.66666d);
394         assertEquals(2, f.getNumerator());
395         assertEquals(3, f.getDenominator());
396         
397         // small
398         f = Fraction.getFraction(1.0d/10001d);
399         assertEquals(0, f.getNumerator());
400         assertEquals(1, f.getDenominator());
401         
402         // normal
403         Fraction f2 = null;
404         for (int i = 1; i <= 100; i++) {  // denominator
405             for (int j = 1; j <= i; j++) {  // numerator
406                 try {
407                     f = Fraction.getFraction((double) j / (double) i);
408                 } catch (final ArithmeticException ex) {
409                     System.err.println(j + " " + i);
410                     throw ex;
411                 }
412                 f2 = Fraction.getReducedFraction(j, i);
413                 assertEquals(f2.getNumerator(), f.getNumerator());
414                 assertEquals(f2.getDenominator(), f.getDenominator());
415             }
416         }
417         // save time by skipping some tests!  (
418         for (int i = 1001; i <= 10000; i+=SKIP) {  // denominator
419             for (int j = 1; j <= i; j++) {  // numerator
420                 try {
421                     f = Fraction.getFraction((double) j / (double) i);
422                 } catch (final ArithmeticException ex) {
423                     System.err.println(j + " " + i);
424                     throw ex;
425                 }
426                 f2 = Fraction.getReducedFraction(j, i);
427                 assertEquals(f2.getNumerator(), f.getNumerator());
428                 assertEquals(f2.getDenominator(), f.getDenominator());
429             }
430         }
431     }
