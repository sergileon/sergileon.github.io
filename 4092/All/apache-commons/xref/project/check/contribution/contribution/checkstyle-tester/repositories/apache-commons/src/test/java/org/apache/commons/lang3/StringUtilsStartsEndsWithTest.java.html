28  public class StringUtilsStartsEndsWithTest {
29      private static final String foo    = "foo";
30      private static final String bar    = "bar";
31      private static final String foobar = "foobar";

33      private static final String BAR    = "BAR";

38      /**
39       * Test StringUtils.startsWith()
40       */
41      @Test
42      public void testStartsWith() {
43          assertTrue("startsWith(null, null)", StringUtils.startsWith(null, (String)null));
44          assertFalse("startsWith(FOOBAR, null)", StringUtils.startsWith(FOOBAR, (String)null));
45          assertFalse("startsWith(null, FOO)",    StringUtils.startsWith(null, FOO));
46          assertTrue("startsWith(FOOBAR, \"\")",  StringUtils.startsWith(FOOBAR, ""));
47  
48          assertTrue("startsWith(foobar, foo)",  StringUtils.startsWith(foobar, foo));
49          assertTrue("startsWith(FOOBAR, FOO)",  StringUtils.startsWith(FOOBAR, FOO));
50          assertFalse("startsWith(foobar, FOO)", StringUtils.startsWith(foobar, FOO));
51          assertFalse("startsWith(FOOBAR, foo)", StringUtils.startsWith(FOOBAR, foo));
52  
53          assertFalse("startsWith(foo, foobar)", StringUtils.startsWith(foo, foobar));
54          assertFalse("startsWith(foo, foobar)", StringUtils.startsWith(bar, foobar));
55  
56          assertFalse("startsWith(foobar, bar)", StringUtils.startsWith(foobar, bar));
57          assertFalse("startsWith(FOOBAR, BAR)", StringUtils.startsWith(FOOBAR, BAR));
58          assertFalse("startsWith(foobar, BAR)", StringUtils.startsWith(foobar, BAR));
59          assertFalse("startsWith(FOOBAR, bar)", StringUtils.startsWith(FOOBAR, bar));
60      }

62      /**
63       * Test StringUtils.testStartsWithIgnoreCase()
64       */
65      @Test
66      public void testStartsWithIgnoreCase() {

68          assertFalse("startsWithIgnoreCase(FOOBAR, null)", StringUtils.startsWithIgnoreCase(FOOBAR, (String)null));
69          assertFalse("startsWithIgnoreCase(null, FOO)",    StringUtils.startsWithIgnoreCase(null, FOO));

72          assertTrue("startsWithIgnoreCase(foobar, foo)", StringUtils.startsWithIgnoreCase(foobar, foo));
73          assertTrue("startsWithIgnoreCase(FOOBAR, FOO)", StringUtils.startsWithIgnoreCase(FOOBAR, FOO));
74          assertTrue("startsWithIgnoreCase(foobar, FOO)", StringUtils.startsWithIgnoreCase(foobar, FOO));
75          assertTrue("startsWithIgnoreCase(FOOBAR, foo)", StringUtils.startsWithIgnoreCase(FOOBAR, foo));

86      @Test
87      public void testStartsWithAny() {
88          assertFalse(StringUtils.startsWithAny(null, (String[])null));
89          assertFalse(StringUtils.startsWithAny(null, "abc"));
90          assertFalse(StringUtils.startsWithAny("abcxyz", (String[])null));
91          assertFalse(StringUtils.startsWithAny("abcxyz"));
92          assertTrue(StringUtils.startsWithAny("abcxyz", "abc"));
93          assertTrue(StringUtils.startsWithAny("abcxyz", null, "xyz", "abc"));
94          assertFalse(StringUtils.startsWithAny("abcxyz", null, "xyz", "abcd"));
95  
96          assertTrue("StringUtils.startsWithAny(abcxyz, StringBuilder(xyz), StringBuffer(abc))", StringUtils.startsWithAny("abcxyz", new StringBuilder("xyz"), new StringBuffer("abc")));
97          assertTrue("StringUtils.startsWithAny( StrBuilder(abcxyz), StringBuilder(xyz), StringBuffer(abc))", StringUtils.startsWithAny( new StrBuilder("abcxyz"), new StringBuilder("xyz"), new StringBuffer("abc")));
98      }

101     /**
102      * Test StringUtils.endsWith()
103      */
104     @Test
105     public void testEndsWith() {
106         assertTrue("endsWith(null, null)",    StringUtils.endsWith(null, (String)null));
107         assertFalse("endsWith(FOOBAR, null)", StringUtils.endsWith(FOOBAR, (String)null));
108         assertFalse("endsWith(null, FOO)",    StringUtils.endsWith(null, FOO));
109         assertTrue("endsWith(FOOBAR, \"\")",  StringUtils.endsWith(FOOBAR, ""));
110 
111         assertFalse("endsWith(foobar, foo)", StringUtils.endsWith(foobar, foo));
112         assertFalse("endsWith(FOOBAR, FOO)", StringUtils.endsWith(FOOBAR, FOO));
113         assertFalse("endsWith(foobar, FOO)", StringUtils.endsWith(foobar, FOO));
114         assertFalse("endsWith(FOOBAR, foo)", StringUtils.endsWith(FOOBAR, foo));
115 
116         assertFalse("endsWith(foo, foobar)", StringUtils.endsWith(foo, foobar));
117         assertFalse("endsWith(foo, foobar)", StringUtils.endsWith(bar, foobar));
118 
119         assertTrue("endsWith(foobar, bar)",  StringUtils.endsWith(foobar, bar));
120         assertTrue("endsWith(FOOBAR, BAR)",  StringUtils.endsWith(FOOBAR, BAR));
121         assertFalse("endsWith(foobar, BAR)", StringUtils.endsWith(foobar, BAR));
122         assertFalse("endsWith(FOOBAR, bar)", StringUtils.endsWith(FOOBAR, bar));
123 
124         // "alpha,beta,gamma,delta".endsWith("delta")
125         assertTrue("endsWith(\u03B1\u03B2\u03B3\u03B4, \u03B4)",
126                 StringUtils.endsWith("\u03B1\u03B2\u03B3\u03B4", "\u03B4"));
127         // "alpha,beta,gamma,delta".endsWith("gamma,DELTA")
128         assertFalse("endsWith(\u03B1\u03B2\u03B3\u03B4, \u03B3\u0394)",
129                 StringUtils.endsWith("\u03B1\u03B2\u03B3\u03B4", "\u03B3\u0394"));
130     }

132     /**
133      * Test StringUtils.endsWithIgnoreCase()
134      */
135     @Test
136     public void testEndsWithIgnoreCase() {
137         assertTrue("endsWithIgnoreCase(null, null)",    StringUtils.endsWithIgnoreCase(null, (String)null));
138         assertFalse("endsWithIgnoreCase(FOOBAR, null)", StringUtils.endsWithIgnoreCase(FOOBAR, (String)null));
139         assertFalse("endsWithIgnoreCase(null, FOO)",    StringUtils.endsWithIgnoreCase(null, FOO));
140         assertTrue("endsWithIgnoreCase(FOOBAR, \"\")",  StringUtils.endsWithIgnoreCase(FOOBAR, ""));
141 
142         assertFalse("endsWithIgnoreCase(foobar, foo)", StringUtils.endsWithIgnoreCase(foobar, foo));
143         assertFalse("endsWithIgnoreCase(FOOBAR, FOO)", StringUtils.endsWithIgnoreCase(FOOBAR, FOO));
144         assertFalse("endsWithIgnoreCase(foobar, FOO)", StringUtils.endsWithIgnoreCase(foobar, FOO));
145         assertFalse("endsWithIgnoreCase(FOOBAR, foo)", StringUtils.endsWithIgnoreCase(FOOBAR, foo));
146 
147         assertFalse("endsWithIgnoreCase(foo, foobar)", StringUtils.endsWithIgnoreCase(foo, foobar));
148         assertFalse("endsWithIgnoreCase(foo, foobar)", StringUtils.endsWithIgnoreCase(bar, foobar));
149 
150         assertTrue("endsWithIgnoreCase(foobar, bar)", StringUtils.endsWithIgnoreCase(foobar, bar));
151         assertTrue("endsWithIgnoreCase(FOOBAR, BAR)", StringUtils.endsWithIgnoreCase(FOOBAR, BAR));
152         assertTrue("endsWithIgnoreCase(foobar, BAR)", StringUtils.endsWithIgnoreCase(foobar, BAR));
153         assertTrue("endsWithIgnoreCase(FOOBAR, bar)", StringUtils.endsWithIgnoreCase(FOOBAR, bar));
154 
155         // javadoc
156         assertTrue(StringUtils.endsWithIgnoreCase("abcdef", "def"));
157         assertTrue(StringUtils.endsWithIgnoreCase("ABCDEF", "def"));
158         assertFalse(StringUtils.endsWithIgnoreCase("ABCDEF", "cde"));
159 
160         // "alpha,beta,gamma,delta".endsWith("DELTA")
161         assertTrue("endsWith(\u03B1\u03B2\u03B3\u03B4, \u0394)",
162                 StringUtils.endsWithIgnoreCase("\u03B1\u03B2\u03B3\u03B4", "\u0394"));
163         // "alpha,beta,gamma,delta".endsWith("GAMMA")
164         assertFalse("endsWith(\u03B1\u03B2\u03B3\u03B4, \u0393)",
165                 StringUtils.endsWithIgnoreCase("\u03B1\u03B2\u03B3\u03B4", "\u0393"));
166     }

168     @Test
169     public void testEndsWithAny() {
170         assertFalse("StringUtils.endsWithAny(null, null)", StringUtils.endsWithAny(null, (String)null));
171         assertFalse("StringUtils.endsWithAny(null, new String[] {abc})", StringUtils.endsWithAny(null, new String[] {"abc"}));
172         assertFalse("StringUtils.endsWithAny(abcxyz, null)", StringUtils.endsWithAny("abcxyz", (String)null));
173         assertTrue("StringUtils.endsWithAny(abcxyz, new String[] {\"\"})", StringUtils.endsWithAny("abcxyz", new String[] {""}));
174         assertTrue("StringUtils.endsWithAny(abcxyz, new String[] {xyz})", StringUtils.endsWithAny("abcxyz", new String[] {"xyz"}));
175         assertTrue("StringUtils.endsWithAny(abcxyz, new String[] {null, xyz, abc})", StringUtils.endsWithAny("abcxyz", new String[] {null, "xyz", "abc"}));
176         assertFalse("StringUtils.endsWithAny(defg, new String[] {null, xyz, abc})", StringUtils.endsWithAny("defg", new String[] {null, "xyz", "abc"}));
177 
178         assertTrue("StringUtils.endsWithAny(abcxyz, StringBuilder(abc), StringBuffer(xyz))", StringUtils.endsWithAny("abcxyz", new StringBuilder("abc"), new StringBuffer("xyz")));
179         assertTrue("StringUtils.endsWithAny( StrBuilder(abcxyz), StringBuilder(abc), StringBuffer(xyz))", StringUtils.endsWithAny( new StrBuilder("abcxyz"), new StringBuilder("abc"), new StringBuffer("xyz")));
180     }
