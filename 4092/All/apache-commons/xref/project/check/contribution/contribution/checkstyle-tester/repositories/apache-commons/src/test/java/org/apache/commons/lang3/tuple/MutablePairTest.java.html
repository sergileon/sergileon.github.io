17  package org.apache.commons.lang3.tuple;
18  
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertFalse;
21  import static org.junit.Assert.assertNull;
22  import static org.junit.Assert.assertTrue;
23  
24  import java.io.ByteArrayInputStream;
25  import java.io.ByteArrayOutputStream;
26  import java.io.ObjectInputStream;
27  import java.io.ObjectOutputStream;
28  
29  import org.junit.Test;
30  
31  /**
32   * Test the MutablePair class.
33   * @version $Id$
34   */
35  public class MutablePairTest {
36  
37      @Test
38      public void testBasic() throws Exception {
39          final MutablePair<Integer, String> pair = new MutablePair<Integer, String>(0, "foo");
40          assertEquals(0, pair.getLeft().intValue());
41          assertEquals("foo", pair.getRight());
42          final MutablePair<Object, String> pair2 = new MutablePair<Object, String>(null, "bar");
43          assertNull(pair2.getLeft());
44          assertEquals("bar", pair2.getRight());
45      }
46  
47      @Test
48      public void testDefault() throws Exception {
49          final MutablePair<Integer, String> pair = new MutablePair<Integer, String>();
50          assertNull(pair.getLeft());
51          assertNull(pair.getRight());
52      }
53      
54      @Test
55      public void testMutate() throws Exception {
56          final MutablePair<Integer, String> pair = new MutablePair<Integer, String>(0, "foo");
57          pair.setLeft(42);
58          pair.setRight("bar");
59          assertEquals(42, pair.getLeft().intValue());
60          assertEquals("bar", pair.getRight());
61      }
62  
63      @Test
64      public void testPairOf() throws Exception {
65          final MutablePair<Integer, String> pair = MutablePair.of(0, "foo");
66          assertEquals(0, pair.getLeft().intValue());
67          assertEquals("foo", pair.getRight());
68          final MutablePair<Object, String> pair2 = MutablePair.of(null, "bar");
69          assertNull(pair2.getLeft());
70          assertEquals("bar", pair2.getRight());
71      }
72  
73      @Test
74      public void testEquals() throws Exception {
75          assertEquals(MutablePair.of(null, "foo"), MutablePair.of(null, "foo"));
76          assertFalse(MutablePair.of("foo", 0).equals(MutablePair.of("foo", null)));
77          assertFalse(MutablePair.of("foo", "bar").equals(MutablePair.of("xyz", "bar")));
78  
79          final MutablePair<String, String> p = MutablePair.of("foo", "bar");
80          assertTrue(p.equals(p));
81          assertFalse(p.equals(new Object()));
82      }
83  
84      @Test
85      public void testHashCode() throws Exception {
86          assertEquals(MutablePair.of(null, "foo").hashCode(), MutablePair.of(null, "foo").hashCode());
87      }
88  
89      @Test
90      public void testToString() throws Exception {
91          assertEquals("(null,null)", MutablePair.of(null, null).toString());
92          assertEquals("(null,two)", MutablePair.of(null, "two").toString());
93          assertEquals("(one,null)", MutablePair.of("one", null).toString());
94          assertEquals("(one,two)", MutablePair.of("one", "two").toString());
95      }
96  
97      @Test
98      @SuppressWarnings("unchecked")
99      public void testSerialization() throws Exception {
100         final MutablePair<Integer, String> origPair = MutablePair.of(0, "foo");
101         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
102         final ObjectOutputStream out = new ObjectOutputStream(baos);
103         out.writeObject(origPair);
104         final MutablePair<Integer, String> deserializedPair = (MutablePair<Integer, String>) new ObjectInputStream(
105                 new ByteArrayInputStream(baos.toByteArray())).readObject();
106         assertEquals(origPair, deserializedPair);
107         assertEquals(origPair.hashCode(), deserializedPair.hashCode());
108     }
109 }
