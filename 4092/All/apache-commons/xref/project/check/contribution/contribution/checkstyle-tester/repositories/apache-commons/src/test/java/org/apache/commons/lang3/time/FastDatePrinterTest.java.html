74      @Test
75      public void testFormat() {
76          final Locale realDefaultLocale = Locale.getDefault();
77          final TimeZone realDefaultZone = TimeZone.getDefault();
78          try {
79              Locale.setDefault(Locale.US);
80              TimeZone.setDefault(NEW_YORK);
81  
82              final GregorianCalendar cal1 = new GregorianCalendar(2003, 0, 10, 15, 33, 20);
83              final GregorianCalendar cal2 = new GregorianCalendar(2003, 6, 10, 9, 00, 00);
84              final Date date1 = cal1.getTime();
85              final Date date2 = cal2.getTime();
86              final long millis1 = date1.getTime();
87              final long millis2 = date2.getTime();
88  
89              DatePrinter fdf = getInstance("yyyy-MM-dd'T'HH:mm:ss");
90              SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
91              assertEquals(sdf.format(date1), fdf.format(date1));
92              assertEquals("2003-01-10T15:33:20", fdf.format(date1));
93              assertEquals("2003-01-10T15:33:20", fdf.format(cal1));
94              assertEquals("2003-01-10T15:33:20", fdf.format(millis1));
95              assertEquals("2003-07-10T09:00:00", fdf.format(date2));
96              assertEquals("2003-07-10T09:00:00", fdf.format(cal2));
97              assertEquals("2003-07-10T09:00:00", fdf.format(millis2));
98  
99              fdf = getInstance("Z");
100             assertEquals("-0500", fdf.format(date1));
101             assertEquals("-0500", fdf.format(cal1));
102             assertEquals("-0500", fdf.format(millis1));
103 
104             assertEquals("-0400", fdf.format(date2));
105             assertEquals("-0400", fdf.format(cal2));
106             assertEquals("-0400", fdf.format(millis2));
107 
108             fdf = getInstance("ZZ");
109             assertEquals("-05:00", fdf.format(date1));
110             assertEquals("-05:00", fdf.format(cal1));
111             assertEquals("-05:00", fdf.format(millis1));
112 
113             assertEquals("-04:00", fdf.format(date2));
114             assertEquals("-04:00", fdf.format(cal2));
115             assertEquals("-04:00", fdf.format(millis2));
116 
117             final String pattern = "GGGG GGG GG G yyyy yyy yy y MMMM MMM MM M" +
118                 " dddd ddd dd d DDDD DDD DD D EEEE EEE EE E aaaa aaa aa a zzzz zzz zz z";
119             fdf = getInstance(pattern);
120             sdf = new SimpleDateFormat(pattern);
121             // SDF bug fix starting with Java 7
122             assertEquals(sdf.format(date1).replaceAll("2003 03 03 03", "2003 2003 03 2003"), fdf.format(date1));
123             assertEquals(sdf.format(date2).replaceAll("2003 03 03 03", "2003 2003 03 2003"), fdf.format(date2));
124         } finally {
125             Locale.setDefault(realDefaultLocale);
126             TimeZone.setDefault(realDefaultZone);
127         }
128     }

279     private static Calendar initializeCalendar(TimeZone tz) {
280         Calendar cal = Calendar.getInstance(tz);
281         cal.set(Calendar.YEAR, 2001);
282         cal.set(Calendar.MONTH, 1); // not daylight savings
283         cal.set(Calendar.DAY_OF_MONTH, 4);
284         cal.set(Calendar.HOUR_OF_DAY, 12);
285         cal.set(Calendar.MINUTE, 8);
286         cal.set(Calendar.SECOND, 56);
287         cal.set(Calendar.MILLISECOND, 235);
288         return cal;
289     }

296     private static enum Expected1806 {
297         India(INDIA, "+05", "+0530", "+05:30"), Greenwich(GMT, "Z", "Z", "Z"), NewYork(
298                 NEW_YORK, "-05", "-0500", "-05:00");

300         private Expected1806(TimeZone zone, String one, String two, String three) {
301             this.zone = zone;
302             this.one = one;
303             this.two = two;
304             this.three = three;
305         }
306 
307         final TimeZone zone;
308         final String one;
309         final String two;
310         final String three;
311     }

313     @Test
314     public void test1806() throws ParseException {
315         for (Expected1806 trial : Expected1806.values()) {
316             Calendar cal = initializeCalendar(trial.zone);
317 
318             DatePrinter printer = getInstance("X", trial.zone);
319             assertEquals(trial.one, printer.format(cal));
320 
321             printer = getInstance("XX", trial.zone);
322             assertEquals(trial.two, printer.format(cal));
323 
324             printer = getInstance("XXX", trial.zone);
325             assertEquals(trial.three, printer.format(cal));
326         }
327     }
