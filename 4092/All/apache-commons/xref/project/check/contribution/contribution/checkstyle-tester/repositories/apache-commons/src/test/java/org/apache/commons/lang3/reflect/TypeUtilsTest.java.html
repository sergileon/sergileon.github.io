448         final Type type1 = types[i1];
449         final Type type2 = types[i2];
450         final boolean isAssignable = TypeUtils.isAssignable(type2, type1);
451 
452         if (expected) {
453             Assert.assertTrue("[" + i1 + ", " + i2 + "]: From "
454                     + StringEscapeUtils.escapeHtml4(String.valueOf(type2)) + " to "
455                     + StringEscapeUtils.escapeHtml4(String.valueOf(type1)), isAssignable);
456         } else {
457             Assert.assertFalse("[" + i1 + ", " + i2 + "]: From "
458                     + StringEscapeUtils.escapeHtml4(String.valueOf(type2)) + " to "
459                     + StringEscapeUtils.escapeHtml4(String.valueOf(type1)), isAssignable);
