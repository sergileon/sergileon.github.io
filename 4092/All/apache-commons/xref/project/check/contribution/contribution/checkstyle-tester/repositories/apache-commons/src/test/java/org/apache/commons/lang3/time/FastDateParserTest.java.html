198     private Calendar getEraStart(int year, final TimeZone zone, final Locale locale) {
199         final Calendar cal = Calendar.getInstance(zone, locale);
200         cal.clear();
201 
202         // http://docs.oracle.com/javase/6/docs/technotes/guides/intl/calendar.doc.html
203         if (locale.equals(FastDateParser.JAPANESE_IMPERIAL)) {
204             if(year < 1868) {
205                 cal.set(Calendar.ERA, 0);
206                 cal.set(Calendar.YEAR, 1868-year);
207             }
208         }
209         else {
210             if (year < 0) {
211                 cal.set(Calendar.ERA, GregorianCalendar.BC);
212                 year= -year;
213             }
214             cal.set(Calendar.YEAR, year/100 * 100);
215         }
216         return cal;
217     }

219     private void validateSdfFormatFdpParseEquality(final String format, final Locale locale, final TimeZone tz, final DateParser fdp, final Date in, final int year, final Date cs) throws ParseException {
220         final SimpleDateFormat sdf = new SimpleDateFormat(format, locale);
221         sdf.setTimeZone(tz);
222         if (format.equals(SHORT_FORMAT)) {
223             sdf.set2DigitYearStart( cs );
224         }
225         final String fmt = sdf.format(in);
226         try {
227             final Date out = fdp.parse(fmt);
228             assertEquals(locale.toString()+" "+in+" "+ format+ " "+tz.getID(), in, out);
229         } catch (final ParseException pe) {
230             if (year >= 1868 || !locale.getCountry().equals("JP")) {// LANG-978
231                 throw pe;
232             }
233         }
234     }

237     // Check that all Locales can parse the formats we use
238     public void testParses() throws Exception {
239         for(final String format : new String[]{LONG_FORMAT, SHORT_FORMAT}) {
240             for(final Locale locale : Locale.getAvailableLocales()) {
241                 for(final TimeZone tz :  new TimeZone[]{NEW_YORK, REYKJAVIK, GMT}) {
242                      for(final int year : new int[]{2003, 1940, 1868, 1867, 1, -1, -1940}) {
243                         final Calendar cal= getEraStart(year, tz, locale);
244                         final Date centuryStart= cal.getTime();
245 
246                         cal.set(Calendar.MONTH, 1);
247                         cal.set(Calendar.DAY_OF_MONTH, 10);
248                         final Date in= cal.getTime();
249 
250                         final FastDateParser fdp= new FastDateParser(format, tz, locale, centuryStart);
251                         validateSdfFormatFdpParseEquality(format, locale, tz, fdp, in, year, centuryStart);
252                     }
253                 }
254             }
255         }

321     private void testLocales(final String format, final boolean eraBC) throws Exception {
322 
323         final Calendar cal= Calendar.getInstance(GMT);
324         cal.clear();
325         cal.set(2003, 1, 10);
326         if (eraBC) {
327             cal.set(Calendar.ERA, GregorianCalendar.BC);
328         }
329         for(final Locale locale : Locale.getAvailableLocales()) {
330             // ja_JP_JP cannot handle dates before 1868 properly
331             if (eraBC && locale.equals(FastDateParser.JAPANESE_IMPERIAL)) {
332                 continue;
333             }
334             final SimpleDateFormat sdf = new SimpleDateFormat(format, locale);
335             final DateParser fdf = getInstance(format, locale);
336 
337             try {
338                 checkParse(locale, cal, sdf, fdf);
339             } catch(final ParseException ex) {
340                 Assert.fail("Locale "+locale+ " failed with "+format+" era "+(eraBC?"BC":"AD")+"\n" + trimMessage(ex.toString()));
341             }
342         }
343     }

345     private String trimMessage(final String msg) {
346         if (msg.length() < 100) {
347             return msg;
348         }
349         final int gmt = msg.indexOf("(GMT");
350         if (gmt > 0) {
351             return msg.substring(0, gmt+4)+"...)";
352         }
353         return msg.substring(0, 100)+"...";
354     }

356     private void checkParse(final Locale locale, final Calendar cal, final SimpleDateFormat sdf, final DateParser fdf) throws ParseException {
357         final String formattedDate= sdf.format(cal.getTime());
358         checkParse(locale, sdf, fdf, formattedDate);
359         checkParse(locale, sdf, fdf, formattedDate.toLowerCase(locale));
360         checkParse(locale, sdf, fdf, formattedDate.toUpperCase(locale));
361     }

366         assertEquals(locale.toString()+" "+formattedDate +"\n",expectedTime, actualTime);
367     }

420     private void testSdfAndFdp(final String format, final String date, final boolean shouldFail)
421             throws Exception {
422         Date dfdp = null;
423         Date dsdf = null;
424         Throwable f = null;
425         Throwable s = null;
426 
427         try {
428             final SimpleDateFormat sdf = new SimpleDateFormat(format, Locale.US);
429             sdf.setTimeZone(NEW_YORK);
430             dsdf = sdf.parse(date);
431             if (shouldFail) {
432                 Assert.fail("Expected SDF failure, but got " + dsdf + " for ["+format+","+date+"]");
433             }
434         } catch (final Exception e) {
435             s = e;
436             if (!shouldFail) {
437                 throw e;
438             }
439         }
440 
441         try {
442             final DateParser fdp = getInstance(format, NEW_YORK, Locale.US);
443             dfdp = fdp.parse(date);
444             if (shouldFail) {
445                 Assert.fail("Expected FDF failure, but got " + dfdp + " for ["+format+","+date+"] using "+((FastDateParser)fdp).getParsePattern());
446             }
447         } catch (final Exception e) {
448             f = e;
449             if (!shouldFail) {
450                 throw e;
451             }
452         }
453         // SDF and FDF should produce equivalent results
454         assertTrue("Should both or neither throw Exceptions", (f==null)==(s==null));
455         assertEquals("Parsed dates should be equal", dsdf, dfdp);
456     }

591         Calendar cal = Calendar.getInstance(tz);
592         cal.set(Calendar.YEAR, 2001);
593         cal.set(Calendar.MONTH, 1); // not daylight savings
594         cal.set(Calendar.DAY_OF_MONTH, 4);
595         cal.set(Calendar.HOUR_OF_DAY, 12);
596         cal.set(Calendar.MINUTE, 8);
597         cal.set(Calendar.SECOND, 56);
598         cal.set(Calendar.MILLISECOND, 235);
599         return cal;
600     }
