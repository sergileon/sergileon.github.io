1671      * (in addition to the proper start and stop dates)
1672      */
1673     private static void assertWeekIterator(final Iterator<?> it, final Calendar start, final Calendar end) {
1674         Calendar cal = (Calendar) it.next();
1675         assertCalendarsEquals("", start, cal, 0);
1676         Calendar last = null;
1677         int count = 1;
1678         while (it.hasNext()) {
1679             //Check this is just a date (no time component)
1680             assertCalendarsEquals("", cal, DateUtils.truncate(cal, Calendar.DATE), 0);
1681 
1682             last = cal;
1683             cal = (Calendar) it.next();
1684             count++;
1685 
1686             //Check that this is one day more than the last date
1687             last.add(Calendar.DATE, 1);
1688             assertCalendarsEquals("", last, cal, 0);
1689         }
1690         if (count % 7 != 0) {
1691             throw new AssertionFailedError("There were " + count + " days in this iterator");
1692         }
1693         assertCalendarsEquals("", end, cal, 0);
1694     }
